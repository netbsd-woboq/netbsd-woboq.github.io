34#ifndef _CHAR_H_
35#define _CHAR_H_
36
37#include <ctype.h>
38
39extern unsigned short _cmap[];
40
41#ifndef NLS
42extern unsigned char _cmap_lower[], _cmap_upper[];
43
44#endif
45
46#define _QF 0x0001 /* '" (Forward quotes) */
47#define _QB 0x0002 /* ` (Backquote) */
48#define _SP 0x0004 /* space and tab */
49#define _NL 0x0008 /* \n */
50#define _META 0x0010 /* lex meta characters, sp #'`";&<>()|\t\n */
51#define _GLOB 0x0020 /* glob characters, *?{[` */
52#define _ESC 0x0040 /* \ */
53#define _DOL 0x0080 /* $ */
54#define _DIG 0x0100 /* 0-9 */
55#define _LET 0x0200 /* a-z, A-Z, _ */
56#define _UP 0x0400 /* A-Z */
57#define _LOW 0x0800 /* a-z */
58#define _XD 0x1000 /* 0-9, a-f, A-F */
59#define _CMD 0x2000 /* lex end of command chars, ;&(|` */
60#define _CTR 0x4000 /* control */
61#define _PUN 0x8000 /* punctuation */
62
63#define cmap(c, bits) \
64 (((c) & QUOTE) ? 0 : (_cmap[(unsigned char)(c)] & (bits)))
65
66#define isglob(c) cmap(c, _GLOB)
67#define isspc(c) cmap(c, _SP)
68#define ismeta(c) cmap(c, _META)
69#define iscmdmeta(c) cmap(c, _CMD)
70#define letter(c) (((c) & QUOTE) ? 0 : \
71 (isalpha((unsigned char) (c)) || (c) == '_'))
72#define alnum(c) (((c) & QUOTE) ? 0 : \
73 (isalnum((unsigned char) (c)) || (c) == '_'))
74#ifdef NLS
75#define Isspace(c) (((c) & QUOTE) ? 0 : isspace((unsigned char) (c)))
76#define Isdigit(c) (((c) & QUOTE) ? 0 : isdigit((unsigned char) (c)))
77#define Isalpha(c) (((c) & QUOTE) ? 0 : isalpha((unsigned char) (c)))
78#define Islower(c) (((c) & QUOTE) ? 0 : islower((unsigned char) (c)))
79#define Isupper(c) (((c) & QUOTE) ? 0 : isupper((unsigned char) (c)))
80#define Tolower(c) (((c) & QUOTE) ? 0 : tolower((unsigned char) (c)))
81#define Toupper(c) (((c) & QUOTE) ? 0 : toupper((unsigned char) (c)))
82#define Isxdigit(c) (((c) & QUOTE) ? 0 : isxdigit((unsigned char) (c)))
83#define Isalnum(c) (((c) & QUOTE) ? 0 : isalnum((unsigned char) (c)))
84#define Iscntrl(c) (((c) & QUOTE) ? 0 : iscntrl((unsigned char) (c)))
85#define Isprint(c) (((c) & QUOTE) ? 0 : isprint((unsigned char) (c)))
86#else
87#define Isspace(c) cmap(c, _SP|_NL)
88#define Isdigit(c) cmap(c, _DIG)
89#define Isalpha(c) (cmap(c,_LET) && !(((c) & META) && AsciiOnly))
90#define Islower(c) (cmap(c,_LOW) && !(((c) & META) && AsciiOnly))
91#define Isupper(c) (cmap(c, _UP) && !(((c) & META) && AsciiOnly))
92#define Tolower(c) (_cmap_lower[(unsigned char)(c)])
93#define Toupper(c) (_cmap_upper[(unsigned char)(c)])
94#define Isxdigit(c) cmap(c, _XD)
95#define Isalnum(c) (cmap(c, _DIG|_LET) && !(((c) & META) && AsciiOnly))
96#define Iscntrl(c) (cmap(c,_CTR) && !(((c) & META) && AsciiOnly))
97#define Isprint(c) (!cmap(c,_CTR) && !(((c) & META) && AsciiOnly))
98#endif
99
100#endif /* !_CHAR_H_ */
