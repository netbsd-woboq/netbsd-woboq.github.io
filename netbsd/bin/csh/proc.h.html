34#ifndef _PROC_H_
35#define _PROC_H_
36
37/*
38 * Structure for each process the shell knows about:
39 * allocated and filled by pcreate.
40 * flushed by pflush; freeing always happens at top level
41 * so the interrupt level has less to worry about.
42 * processes are related to "friends" when in a pipeline;
43 * p_friends links makes a circular list of such jobs
44 */
45struct process {
46 struct process *p_next; /* next in global "proclist" */
47 struct process *p_friends; /* next in job list (or self) */
48 struct directory *p_cwd; /* cwd of the job (only in head) */
49 int p_flags; /* various job status flags */
50 int p_reason; /* reason for entering this state */
51 int p_index; /* shorthand job index */
52 pid_t p_pid;
53 pid_t p_jobid; /* pid of job leader */
54 /* if a job is stopped/background p_jobid gives its pgrp */
55 struct timespec p_btime; /* begin time */
56 struct timespec p_etime; /* end time */
57 struct rusage p_rusage;
58 Char *p_command; /* first PMAXLEN chars of command */
59};
60
61/* flag values for p_flags */
62#define PRUNNING (1<<0) /* running */
63#define PSTOPPED (1<<1) /* stopped */
64#define PNEXITED (1<<2) /* normally exited */
65#define PAEXITED (1<<3) /* abnormally exited */
66#define PSIGNALED (1<<4) /* terminated by a signal != SIGINT */
67#define PNOTIFY (1<<5) /* notify async when done */
68#define PTIME (1<<6) /* job times should be printed */
69#define PAWAITED (1<<7) /* top level is waiting for it */
70#define PFOREGND (1<<8) /* started in shells pgrp */
71#define PDUMPED (1<<9) /* process dumped core */
72#define PERR (1<<10) /* diagnostic output also piped out */
73#define PPOU (1<<11) /* piped output */
74#define PREPORTED (1<<12) /* status has been reported */
75#define PINTERRUPTED (1<<13) /* job stopped via interrupt signal */
76#define PPTIME (1<<14) /* time individual process */
77#define PNEEDNOTE (1<<15) /* notify as soon as practical */
78
79#define PALLSTATES (PRUNNING|PSTOPPED|PNEXITED|PAEXITED|PSIGNALED|PINTERRUPTED)
80
81#define PMAXLEN 80
82
83/* defines for arguments to pprint */
84#define NUMBER 01
85#define NAME 02
86#define REASON 04
87#define AMPERSAND 010
88#define FANCY 020
89#define SHELLDIR 040 /* print shell's dir if not the same */
90#define JOBDIR 0100 /* print job's dir if not the same */
91#define AREASON 0200
92
93struct process proclist; /* list head of all processes */
94int pnoprocesses; /* pchild found nothing to wait for */
95
96struct process *pholdjob; /* one level stack of current jobs */
97
98struct process *pcurrjob; /* current job */
99struct process *pcurrent; /* current job in table */
100struct process *pprevious; /* previous job in table */
101
102int pmaxindex; /* current maximum job index */
103
104#endif /* !_PROC_H_ */
