46/* build_active_list: add line matching a pattern to the global-active list */
47int
48build_active_list(int isgcmd)
49{
50 pattern_t *pat;
51 line_t *lp;
52 long n;
53 char *s;
54 char delimiter;
55
56 if ((delimiter = *ibufp) == ' ' || delimiter == '\n') {
57 seterrmsg("invalid pattern delimiter");
58 return ERR;
59 } else if ((pat = get_compiled_pattern()) == NULL)
60 return ERR;
61 else if (*ibufp == delimiter)
62 ibufp++;
63 clear_active_list();
64 lp = get_addressed_line_node(first_addr);
65 for (n = first_addr; n <= second_addr; n++, lp = lp->q_forw) {
66 if ((s = get_sbuf_line(lp)) == NULL)
67 return ERR;
68 if (isbinary)
69 NUL_TO_NEWLINE(s, lp->len);
70 if ((!regexec(pat, s, 0, NULL, 0)) == isgcmd &&
71 set_active_node(lp) < 0)
72 return ERR;
73 }
74 return 0;
75}

78/* exec_global: apply command list in the command buffer to the active
79 lines in a range; return command status */
80long
81exec_global(int interact, int gflag)
82{
83 static char *ocmd = NULL;
84 static int ocmdsz = 0;
85
86 line_t *lp = NULL;
87 int status;
88 int n;
89 char *cmd = NULL;
90#ifdef BACKWARDS
91 char cmdp[] = "p\n";
92
93 if (!interact) {
94 if (!strcmp(ibufp, "\n"))
95 cmd = cmdp; /* null cmd-list == `p' */
96 else if ((cmd = get_extended_line(&n, 0)) == NULL)
97 return ERR;
98 }
99#else
100 if (!interact && (cmd = get_extended_line(&n, 0)) == NULL)
101 return ERR;
102#endif
103 clear_undo_stack();
104 while ((lp = next_active_node()) != NULL) {
105 if ((current_addr = get_line_node_addr(lp)) < 0)
106 return ERR;
107 if (interact) {
108 /* print current_addr; get a command in global syntax */
109 if (display_lines(current_addr, current_addr, gflag) < 0)
110 return ERR;
111 while ((n = get_tty_line()) > 0 &&
112 ibuf[n - 1] != '\n')
113 clearerr(stdin);
114 if (n < 0)
115 return ERR;
116 else if (n == 0) {
117 seterrmsg("unexpected end-of-file");
118 return ERR;
119 } else if (n == 1 && !strcmp(ibuf, "\n"))
120 continue;
121 else if (n == 2 && !strcmp(ibuf, "&\n")) {
122 if (cmd == NULL) {
123 seterrmsg("no previous command");
124 return ERR;
125 } else cmd = ocmd;
126 } else if ((cmd = get_extended_line(&n, 0)) == NULL)
127 return ERR;
128 else {
129 REALLOC(ocmd, ocmdsz, n + 1, ERR);
130 memcpy(ocmd, cmd, n + 1);
131 cmd = ocmd;
132 }
133
134 }
135 ibufp = cmd;
136 for (; *ibufp;)
137 if ((status = extract_addr_range()) < 0 ||
138 (status = exec_command()) < 0 ||
139 (status > 0 && (status = display_lines(
140 current_addr, current_addr, status)) < 0))
141 return status;
142 }
143 return 0;
144}

147line_t **active_list; /* list of lines active in a global command */
148long active_last; /* index of last active line in active_list */
149long active_size; /* size of active_list */
150long active_ptr; /* active_list index (non-decreasing) */
151long active_ndx; /* active_list index (modulo active_last) */
152
153/* set_active_node: add a line node to the global-active list */
154int
155set_active_node(line_t *lp)
156{
157 if (active_last + 1 > active_size) {
158 int ti = active_size;
159 line_t **ts;
160 SPL1();
161 if ((ts = (line_t **) realloc(active_list,
162 (ti += MINBUFSZ) * sizeof(line_t **))) == NULL) {
163 fprintf(stderr, "%s\n", strerror(errno));
164 seterrmsg("out of memory");
165 SPL0();
166 return ERR;
167 }
168 active_size = ti;
169 active_list = ts;
170 SPL0();
171 }
172 active_list[active_last++] = lp;
173 return 0;
174}

177/* unset_active_nodes: remove a range of lines from the global-active list */
178void
179unset_active_nodes(line_t *np, line_t *mp)
180{
181 line_t *lp;
182 long i;
183
184 for (lp = np; lp != mp; lp = lp->q_forw)
185 for (i = 0; i < active_last; i++)
186 if (active_list[active_ndx] == lp) {
187 active_list[active_ndx] = NULL;
188 active_ndx = INC_MOD(active_ndx, active_last - 1);
189 break;
190 } else active_ndx = INC_MOD(active_ndx, active_last - 1);
191}
192
193
194/* next_active_node: return the next global-active line node */
195line_t *
196next_active_node(void)
197{
198 while (active_ptr < active_last && active_list[active_ptr] == NULL)
199 active_ptr++;
200 return (active_ptr < active_last) ? active_list[active_ptr++] : NULL;
201}
202
203
204/* clear_active_list: clear the global-active list */
205void
206clear_active_list(void)
207{
208 SPL1();
209 active_size = active_last = active_ptr = active_ndx = 0;
210 free(active_list);
211 active_list = NULL;
212 SPL0();
213}
