6/*
7 * Shell file I/O routines
8 */
9/* $Id: shf.h,v 1.3 1999/10/20 15:10:00 hubertf Exp $ */
10
11#define SHF_BSIZE 512
12
13#define shf_fileno(shf) ((shf)->fd)
14#define shf_setfileno(shf,nfd) ((shf)->fd = (nfd))
15#define shf_getc(shf) ((shf)->rnleft > 0 ? (shf)->rnleft--, *(shf)->rp++ : \
16 shf_getchar(shf))
17#define shf_putc(c, shf) ((shf)->wnleft == 0 ? shf_putchar((c), (shf)) \
18 : ((shf)->wnleft--, *(shf)->wp++ = (c)))
19#define shf_eof(shf) ((shf)->flags & SHF_EOF)
20#define shf_error(shf) ((shf)->flags & SHF_ERROR)
21#define shf_errno(shf) ((shf)->errno_)
22#define shf_clearerr(shf) ((shf)->flags &= ~(SHF_EOF | SHF_ERROR))
23
24/* Flags passed to shf_*open() */
25#define SHF_RD 0x0001
26#define SHF_WR 0x0002
27#define SHF_RDWR (SHF_RD|SHF_WR)
28#define SHF_ACCMODE 0x0003 /* mask */
29#define SHF_GETFL 0x0004 /* use fcntl() to figure RD/WR flags */
30#define SHF_UNBUF 0x0008 /* unbuffered I/O */
31#define SHF_CLEXEC 0x0010 /* set close on exec flag */
32#define SHF_MAPHI 0x0020 /* make fd > FDBASE (and close orig)
33 * (shf_open() only) */
34#define SHF_DYNAMIC 0x0040 /* string: increase buffer as needed */
35#define SHF_INTERRUPT 0x0080 /* EINTR in read/write causes error */
36/* Flags used internally */
37#define SHF_STRING 0x0100 /* a string, not a file */
38#define SHF_ALLOCS 0x0200 /* shf and shf->buf were alloc()ed */
39#define SHF_ALLOCB 0x0400 /* shf->buf was alloc()ed */
40#define SHF_ERROR 0x0800 /* read()/write() error */
41#define SHF_EOF 0x1000 /* read eof (sticky) */
42#define SHF_READING 0x2000 /* currently reading: rnleft,rp valid */
43#define SHF_WRITING 0x4000 /* currently writing: wnleft,wp valid */
44
45
46struct shf {
47 int flags; /* see SHF_* */
48 unsigned char *rp; /* read: current position in buffer */
49 int rbsize; /* size of buffer (1 if SHF_UNBUF) */
50 int rnleft; /* read: how much data left in buffer */
51 unsigned char *wp; /* write: current position in buffer */
52 int wbsize; /* size of buffer (0 if SHF_UNBUF) */
53 int wnleft; /* write: how much space left in buffer */
54 unsigned char *buf; /* buffer */
55 int fd; /* file descriptor */
56 int errno_; /* saved value of errno after error */
57 int bsize; /* actual size of buf */
58 Area *areap; /* area shf/buf were allocated in */
59};
