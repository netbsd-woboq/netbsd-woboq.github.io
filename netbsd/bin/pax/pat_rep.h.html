38#include <regex.h>
39/*
40 * data structure for storing user supplied replacement strings (-s)
41 */
42typedef struct replace {
43 char *nstr; /* the new string we will substitute with */
44 regex_t rcmp; /* compiled regular expression used to match */
45 int flgs; /* print conversions? global in operation? */
46#define PRNT 0x1
47#define GLOB 0x2
48#define RENM 0x4
49#define SYML 0x8
50 struct replace *fow; /* pointer to next pattern */
51} REPLACE;
