36#define UNLIMITED 0 /* unlimited terminal width */
37
38enum mode {
39 PRINTMODE = 0, /* print values */
40 WIDTHMODE = 1 /* determine width of column */
41};
42
43enum type {
44 UNSPECIFIED,
45 CHAR, UCHAR, SHORT, USHORT, INT, UINT, LONG, ULONG,
46 KPTR, KPTR24, INT32, UINT32, SIGLIST, INT64, UINT64,
47 TIMEVAL, CPUTIME, PCPU, VSIZE
48};
49
50/* Variables. */
51typedef SIMPLEQ_HEAD(varlist, varent) VARLIST;
52
53typedef struct varent {
54 SIMPLEQ_ENTRY(varent) next;
55 struct var *var;
56} VARENT;
57
58struct pinfo {
59 struct kinfo_proc2 *ki;
60 struct kinfo_lwp *li;
61 char *prefix;
62 int level;
63 double pcpu;
64};
65
66typedef struct var {
67 const char *name; /* name(s) of variable */
68 const char *header; /* header, possibly changed from default */
69#define COMM 0x01 /* needs exec arguments and environment (XXX) */
70#define ARGV0 0x02 /* only print argv[0] */
71#define LJUST 0x04 /* left adjust on output (trailing blanks) */
72#define INF127 0x08 /* 127 = infinity: if > 127, print 127. */
73#define LWP 0x10 /* dispatch to kinfo_lwp routine */
74#define UAREA 0x20 /* need to check p_uvalid */
75#define ALIAS 0x40 /* entry is alias for 'header' */
76 u_int flag;
77 /* output routine */
78 void (*oproc)(struct pinfo *pi, struct varent *, enum mode);
79 /*
80 * The following (optional) elements are hooks for passing information
81 * to the generic output routine: pvar (that which prints simple
82 * elements from struct kinfo_proc2).
83 */
84 int off; /* offset in structure */
85 enum type type; /* type of element */
86 const char *fmt; /* printf format */
87
88 /* current longest element */
89 int width; /* printing width */
90 int64_t longestp; /* longest positive signed value */
91 int64_t longestn; /* longest negative signed value */
92 u_int64_t longestu; /* longest unsigned value */
93 double longestpd; /* longest positive double */
94 double longestnd; /* longest negative double */
95} VAR;
96
97
98#define OUTPUT(vent, kl, pi, ki, mode) do { \
99 if ((vent)->var->flag & LWP) \
100 pi->li = kl; \
101 ((vent)->var->oproc)(pi, (vent), (mode)); \
102 } while (/*CONSTCOND*/ 0)
