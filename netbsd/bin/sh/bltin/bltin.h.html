37/*
38 * This file is included by programs which are optionally built into the
39 * shell.
40 *
41 * We always define SHELL_BUILTIN, to allow other included headers to
42 * hide some of their symbols if appropriate.
43 *
44 * If SHELL is defined, we try to map the standard UNIX library routines
45 * to ash routines using defines.
46 */
47
48#define SHELL_BUILTIN
49#include "../shell.h"
50#include "../mystring.h"
51#ifdef SHELL
52#include "../output.h"
53#include "../error.h"
54#include "../var.h"
55#undef stdout
56#undef stderr
57#undef putc
58#undef putchar
59#undef fileno
60#undef ferror
61#define FILE struct output
62#define stdout out1
63#define stderr out2
64#ifdef __GNUC__
65#define _RETURN_INT(x) ({(x); 0;}) /* map from void foo() to int bar() */
66#else
67#define _RETURN_INT(x) ((x), 0) /* map from void foo() to int bar() */
68#endif
69#define fprintf(...) _RETURN_INT(outfmt(__VA_ARGS__))
70#define printf(...) _RETURN_INT(out1fmt(__VA_ARGS__))
71#define putc(c, file) _RETURN_INT(outc(c, file))
72#define putchar(c) _RETURN_INT(out1c(c))
73#define fputs(...) _RETURN_INT(outstr(__VA_ARGS__))
74#define fflush(f) _RETURN_INT(flushout(f))
75#define fileno(f) ((f)->fd)
76#define ferror(f) ((f)->flags & OUTPUT_ERR)
77#define INITARGS(argv)
78#define err sh_err
79#define verr sh_verr
80#define errx sh_errx
81#define verrx sh_verrx
82#define warn sh_warn
83#define vwarn sh_vwarn
84#define warnx sh_warnx
85#define vwarnx sh_vwarnx
86#define exit sh_exit
87#define setprogname(s)
88#define getprogname() commandname
89#define setlocate(l,s) 0
90
91#define getenv(p) bltinlookup((p),0)
92
93#else /* ! SHELL */
94#undef NULL
95#include <stdio.h>
96#undef main
97#define INITARGS(argv) if ((commandname = argv[0]) == NULL) {fputs("Argc is zero\n", stderr); exit(2);} else
98#endif /* ! SHELL */
