44 int fd; /* file descriptor to read from */

60/* reasons for skipping commands (see comment on breakcmd routine) */
61enum skipstate {
62 SKIPNONE = 0, /* not skipping */
63 SKIPBREAK, /* break */
64 SKIPCONT, /* continue */
65 SKIPFUNC, /* return in a function */
66 SKIPFILE /* return in a dot command */
67};
68
69struct skipsave {
70 enum skipstate state; /* skipping or not */
71 int count; /* when break or continue, how many */
72};
73
74enum skipstate current_skipstate(void);
75void save_skipstate(struct skipsave *);
76void restore_skipstate(const struct skipsave *);
77void stop_skipping(void); /* reset internal skipping state to SKIPNONE */
78int set_dot_funcnest(int);

83void reset_eval(void);
