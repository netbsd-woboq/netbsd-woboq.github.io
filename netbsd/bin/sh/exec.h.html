37/* values of cmdtype */
38#define CMDUNKNOWN -1 /* no entry in table for command */
39#define CMDNORMAL 0 /* command is an executable program */
40#define CMDFUNCTION 1 /* command is a shell function */
41#define CMDBUILTIN 2 /* command is a shell builtin */
42#define CMDSPLBLTIN 3 /* command is a special shell builtin */
43
44
45struct cmdentry {
46 short cmdtype;
47 short lno_frel; /* for functions: Line numbers count from 1 */
48 int lineno; /* for functions: Abs line number of defn */
49 union param {
50 int index;
51 int (*bltin)(int, char**);
52 struct funcdef *func;
53 } u;
54};
