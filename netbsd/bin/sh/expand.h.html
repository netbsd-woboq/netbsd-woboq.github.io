50/*
51 * expandarg() flags
52 */
53#define EXP_SPLIT 0x1 /* perform word splitting */
54#define EXP_TILDE 0x2 /* do normal tilde expansion */
55#define EXP_VARTILDE 0x4 /* expand tildes in an assignment */
56#define EXP_REDIR 0x8 /* file glob for a redirection (1 match only) */
57#define EXP_CASE 0x10 /* keeps quotes around for CASE pattern */
58#define EXP_IFS_SPLIT 0x20 /* need to record arguments for ifs breakup */
59#define EXP_IN_QUOTES 0x40 /* don't set EXP_IFS_SPLIT again */
60#define EXP_GLOB 0x80 /* perform filename globbing */
61#define EXP_NL 0x100 /* keep CRTNONL in output */
62
63#define EXP_FULL (EXP_SPLIT | EXP_GLOB)
64#define EXP_QNEEDED (EXP_GLOB | EXP_CASE | EXP_REDIR)
