54/*
55 * A job structure contains information about a job. A job is either a
56 * single process or a set of processes contained in a pipeline. In the
57 * latter case, pidlist will be non-NULL, and will point to a -1 terminated
58 * array of pids.
59 */
60#define MAXCMDTEXT 200
61
62struct procstat {
63 pid_t pid; /* process id */
64 int status; /* last process status from wait() */
65 char cmd[MAXCMDTEXT];/* text of command being run */
66};
67
68struct job {
69 struct procstat ps0; /* status of process */
70 struct procstat *ps; /* status or processes when more than one */
71 void *ref; /* temporary reference, used variously */
72 int nprocs; /* number of processes */
73 pid_t pgrp; /* process group of this job */
74 char state;
75#define JOBRUNNING 0 /* at least one proc running */
76#define JOBSTOPPED 1 /* all procs are stopped */
77#define JOBDONE 2 /* all procs are completed */
78 char used; /* true if this entry is in used */
79 char flags;
80#define JOBCHANGED 1 /* set if status has changed */
81#define JOBWANTED 2 /* set if this is a job being sought */
82#define JPIPEFAIL 4 /* set if -o pipefail when job created */
83#if JOBS
84 char jobctl; /* job running under job control */
85 int prev_job; /* previous job index */
86#endif
87};
88
89extern pid_t backgndpid; /* pid of last background process */
90extern int job_warning; /* user was warned about stopped jobs */
