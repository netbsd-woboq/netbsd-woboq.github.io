37#ifndef OUTPUT_INCL
38
39#include <stdarg.h>
40
41struct output {
42 char *nextc;
43 int nleft;
44 int bufsize;
45 char *buf;
46 short fd;
47 short flags;
48#ifndef SMALL
49 struct output *chain;
50#endif
51};
52
53/* flags for ->flags */
54#define OUTPUT_ERR 0x01 /* error occurred on output */
55#define OUTPUT_CLONE 0x02 /* this is a clone of another */
56
57extern struct output output;
58extern struct output errout;
59extern struct output memout;
60extern struct output *out1;
61extern struct output *out2;
62#ifdef SMALL
63#define outx out2
64#else
65extern struct output *outx;
66#endif

68void open_mem(char *, int, struct output *);
69void out1str(const char *);
70void out2str(const char *);
71void outstr(const char *, struct output *);
72void out2shstr(const char *);
73#ifdef SMALL
74#define outxstr out2str
75#define outxshstr out2shstr

77void outxstr(const char *);
78void outxshstr(const char *);
79#endif
80void outshstr(const char *, struct output *);
81void emptyoutbuf(struct output *);
82void flushall(void);
83void flushout(struct output *);
84void freestdout(void);
85void outfmt(struct output *, const char *, ...) __printflike(2, 3);
86void out1fmt(const char *, ...) __printflike(1, 2);
87#ifdef DEBUG
88void debugprintf(const char *, ...) __printflike(1, 2);

90void fmtstr(char *, size_t, const char *, ...) __printflike(3, 4);

92int xwrite(int, char *, int);
93#ifdef SMALL
94#define xtracefdsetup(x) do { break; } while (0)
95#define xtrace_clone(x) do { break; } while (0)
96#define xtrace_pop() do { break; } while (0)
97#else
98void xtracefdsetup(int);
99void xtrace_clone(int);
100void xtrace_pop(void);
101#endif

103#define outc(c, file) (--(file)->nleft < 0? (emptyoutbuf(file), *(file)->nextc++ = (c)) : (*(file)->nextc++ = (c)))
104#define out1c(c) outc(c, out1)
105#define out2c(c) outc(c, out2)
106#define outxc(c) outc(c, outx)
