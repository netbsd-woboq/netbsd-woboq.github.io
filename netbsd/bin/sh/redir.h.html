37/* flags passed to redirect */
38#define REDIR_PUSH 0x01 /* save previous values of file descriptors */
39#define REDIR_BACKQ 0x02 /* save the command output in memory */
40#define REDIR_VFORK 0x04 /* running under vfork(2), be careful */
41#define REDIR_KEEP 0x08 /* don't close-on-exec */

53int outredir(struct output *, union node *, int);
