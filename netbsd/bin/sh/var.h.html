38/*
39 * Shell variables.
40 */

42/* flags */
43#define VUNSET 0x0001 /* the variable is not set */
44#define VEXPORT 0x0002 /* variable is exported */
45#define VREADONLY 0x0004 /* variable cannot be modified */
46#define VNOEXPORT 0x0008 /* variable may not be exported */
47
48#define VSTRFIXED 0x0010 /* variable struct is statically allocated */
49#define VTEXTFIXED 0x0020 /* text is statically allocated */
50#define VSTACK 0x0040 /* text is allocated on the stack */
51#define VNOFUNC 0x0100 /* don't call the callback function */
52#define VFUNCREF 0x0200 /* the function is called on ref, not set */
53
54#define VSPECIAL 0x1000 /* magic properties not lost when set */
55#define VDOEXPORT 0x2000 /* obey VEXPORT even if VNOEXPORT */
56#define VNOSET 0x4000 /* do not set variable - just readonly test */
57#define VNOERROR 0x8000 /* be quiet if set fails (no error msg) */
58
59struct var;
60
61union var_func_union { /* function to be called when: */
62 void (*set_func)(const char *); /* variable gets set/unset */
63 char*(*ref_func)(struct var *); /* variable is referenced */
64};
65
66struct var {
67 struct var *next; /* next entry in hash list */
68 int flags; /* flags are defined above */
69 char *text; /* name=value */
70 int name_len; /* length of name */
71 union var_func_union v_u; /* function to apply (sometimes) */
72};
73
74
75struct localvar {
76 struct localvar *next; /* next local variable in list */
77 struct var *vp; /* the variable that was made local */
78 int flags; /* saved flags */
79 char *text; /* saved text */
80 union var_func_union v_u; /* saved function */
81};

110/*
111 * The following macros access the values of the above variables.
112 * They have to skip over the name. They return the null string
113 * for unset variables.
114 */

116#define ifsset() ((vifs.flags & VUNSET) == 0)
117#define ifsval() (ifsset() ? (vifs.text + 4) : ifs_default)
118#define mailval() (vmail.text + 5)
119#define mpathval() (vmpath.text + 9)
120#define pathval() (vpath.text + 5)
121#define ps1val() (vps1.text + 4)
122#define ps2val() (vps2.text + 4)
123#define ps4val() (vps4.text + 4)
124#define optindval() (voptind.text + 7)
125#ifndef SMALL
126#define histsizeval() (vhistsize.text + 9)
127#define termval() (vterm.text + 5)
128#endif
