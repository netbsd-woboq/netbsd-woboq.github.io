50__dead static void gerr(char *);
51
52static void
53gerr(char *s)
54{
55 if (s)
56 errx(1, "illegal gfmt1 option -- %s", s);
57 else
58 errx(1, "illegal gfmt1 option");
59}

61void
62gprint(struct termios *tp)
63{
64 const struct cchar *cp;
65
66 (void)printf("gfmt1:cflag=%x:iflag=%x:lflag=%x:oflag=%x:",
67 tp->c_cflag, tp->c_iflag, tp->c_lflag, tp->c_oflag);
68 for (cp = cchars1; cp->name; ++cp)
69 (void)printf("%s=%x:", cp->name, tp->c_cc[cp->sub]);
70 (void)printf("ispeed=%d:ospeed=%d\n", cfgetispeed(tp), cfgetospeed(tp));
71}

73void
74gread(struct termios *tp, char *s)
75{
76 const struct cchar *cp;
77 char *ep, *p;
78 long tmp;
79
80 if ((s = strchr(s, ':')) == NULL)
81 gerr(NULL);
82 for (++s; s != NULL;) {
83 p = strsep(&s, ":\0");
84 if (!p || !*p)
85 break;
86 if (!(ep = strchr(p, '=')))
87 gerr(p);
88 *ep++ = '\0';
89 (void)sscanf(ep, "%lx", &tmp);
90
91#define CHK(s) (*p == s[0] && !strcmp(p, s))
92 if (CHK("cflag")) {
93 tp->c_cflag = tmp;
94 continue;
95 }
96 if (CHK("iflag")) {
97 tp->c_iflag = tmp;
98 continue;
99 }
100#ifdef BSD4_4
101 if (CHK("ispeed")) {
102 (void)sscanf(ep, "%ld", &tmp);
103 tp->c_ispeed = tmp;
104 continue;
105 }
106#endif
107 if (CHK("lflag")) {
108 tp->c_lflag = tmp;
109 continue;
110 }
111 if (CHK("oflag")) {
112 tp->c_oflag = tmp;
113 continue;
114 }
115#ifdef BSD4_4
116 if (CHK("ospeed")) {
117 (void)sscanf(ep, "%ld", &tmp);
118 tp->c_ospeed = tmp;
119 continue;
120 }
121#endif
122 for (cp = cchars1; cp->name != NULL; ++cp)
123 if (CHK(cp->name)) {
124 if (cp->sub == VMIN || cp->sub == VTIME)
125 (void)sscanf(ep, "%ld", &tmp);
126 tp->c_cc[cp->sub] = tmp;
127 break;
128 }
129 if (cp->name == NULL)
130 gerr(p);
131 }
132}
