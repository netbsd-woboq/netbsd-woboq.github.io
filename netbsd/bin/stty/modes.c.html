173#define CHK(s) (!strcmp(name, s))
174
175static int
176modeset(const char *name, const struct modes *mp,
177 const struct specialmodes *smp, tcflag_t *f)
178{
179 bool neg;
180
181 for (; smp->name; ++smp)
182 if (CHK(smp->name)) {
183 *f &= ~smp->unset;
184 *f |= smp->set;
185 return 1;
186 }
187
188 if ((neg = (*name == '-')))
189 name++;
190
191 for (; mp->name; ++mp)
192 if (CHK(mp->name)) {
193 if (neg)
194 *f &= ~mp->flag;
195 else
196 *f |= mp->flag;
197 return 1;
198 }
199
200 return 0;
201}
202
203int
204msearch(char ***argvp, struct info *ip)
205{
206 const char *name = **argvp;
207
208 if (modeset(name, cmodes, cspecialmodes, &ip->t.c_cflag))
209 goto out;
210
211 if (modeset(name, imodes, ispecialmodes, &ip->t.c_iflag))
212 goto out;
213
214 if (modeset(name, lmodes, lspecialmodes, &ip->t.c_lflag))
215 goto out;
216
217 if (modeset(name, omodes, ospecialmodes, &ip->t.c_oflag))
218 goto out;
219
220 return 0;
221out:
222 ip->set = 1;
223 return 1;
224}
