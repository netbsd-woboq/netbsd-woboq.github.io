61int
62main(int argc, char *argv[])
63{
64 struct info i;
65 enum FMT fmt;
66 int ch;
67
68 setprogname(argv[0]);
69 (void)setlocale(LC_ALL, "");
70
71 fmt = STTY_NOTSET;
72 i.fd = STDIN_FILENO;
73
74 opterr = 0;
75 while (optind < argc &&
76 strspn(argv[optind], "-aefg") == strlen(argv[optind]) &&
77 (ch = getopt(argc, argv, "aef:g")) != -1)
78 switch(ch) {
79 case 'a': /* undocumented: POSIX compatibility */
80 fmt = STTY_POSIX;
81 break;
82 case 'e':
83 fmt = STTY_BSD;
84 break;
85 case 'f':
86 if ((i.fd = open(optarg, O_RDONLY | O_NONBLOCK)) < 0)
87 err(1, "%s", optarg);
88 break;
89 case 'g':
90 fmt = STTY_GFLAG;
91 break;
92 case '?':
93 default:
94 goto args;
95 }
96
97args: argc -= optind;
98 argv += optind;
99
100 if (ioctl(i.fd, TIOCGLINED, i.ldisc) < 0)
101 err(1, "TIOCGLINED");
102 if (tcgetattr(i.fd, &i.t) < 0)
103 err(1, "tcgetattr");
104 if (ioctl(i.fd, TIOCGWINSZ, &i.win) < 0)
105 warn("TIOCGWINSZ");
106 if (ioctl(i.fd, TIOCGQSIZE, &i.queue) < 0)
107 warn("TIOCGQSIZE");
108
109 switch(fmt) {
110 case STTY_NOTSET:
111 if (*argv)
112 break;
113 /* FALLTHROUGH */
114 case STTY_BSD:
115 case STTY_POSIX:
116 print(&i.t, &i.win, i.queue, i.ldisc, fmt);
117 break;
118 case STTY_GFLAG:
119 gprint(&i.t);
120 break;
121 }
122
123 for (i.set = i.wset = 0; *argv; ++argv) {
124 if (ksearch(&argv, &i))
125 continue;
126
127 if (csearch(&argv, &i))
128 continue;
129
130 if (msearch(&argv, &i))
131 continue;
132
133 if (isdigit((unsigned char)**argv)) {
134 int speed;
135
136 speed = atoi(*argv);
137 cfsetospeed(&i.t, speed);
138 cfsetispeed(&i.t, speed);
139 i.set = 1;
140 continue;
141 }
142
143 if (!strncmp(*argv, "gfmt1", sizeof("gfmt1") - 1)) {
144 gread(&i.t, *argv + sizeof("gfmt1") - 1);
145 i.set = 1;
146 continue;
147 }
148
149 warnx("illegal option -- %s", *argv);
150 usage();
151 }
152
153 if (i.set && tcsetattr(i.fd, 0, &i.t) < 0)
154 err(1, "tcsetattr");
155 if (i.wset && ioctl(i.fd, TIOCSWINSZ, &i.win) < 0)
156 warn("TIOCSWINSZ");
157 exit(0);
158 /* NOTREACHED */
159}

161void
162usage(void)
163{
164
165 (void)fprintf(stderr, "usage: %s [-a|-e|-g] [-f file] [operand ...]\n", getprogname());
166 exit(1);
167 /* NOTREACHED */
168}
