34#ifndef _STTY_H_
35#define _STTY_H_
36
37#include <sys/ioctl.h>
38#include <termios.h>
39
40struct info {
41 int fd; /* file descriptor */
42 linedn_t ldisc; /* line discipline */
43 int queue; /* queue size */
44 int off; /* turn off */
45 int set; /* need set */
46 int wset; /* need window set */
47 char *arg; /* argument */
48 struct termios t; /* terminal info */
49 struct winsize win; /* window info */
50};
51
52struct cchar {
53 const char *name;
54 int sub;
55 u_char def;
56};
57
58enum FMT { STTY_NOTSET, STTY_GFLAG, STTY_BSD, STTY_POSIX };
59
60#define LINELENGTH 72
61
62#endif /* !_STTY_H_ */
