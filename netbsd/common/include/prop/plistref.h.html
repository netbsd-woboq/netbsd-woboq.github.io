32#ifndef _PROPLIB_PLISTREF_H_
33#define _PROPLIB_PLISTREF_H_
34
35/* for size_t */
36#include <sys/types.h>
37
38/*
39 * Property List Reference --
40 * Used to pass externalized property lists across protection
41 * boundaries (ioctls, syscalls, etc.).
42 */
43struct plistref {
44 void *pref_plist; /* plist data */
45 size_t pref_len; /* total length of plist data */
46};
47
48#endif /* _PROPLIB_PLISTREF_H_ */
