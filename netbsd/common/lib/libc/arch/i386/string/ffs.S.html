12ENTRY(ffs)
13 bsfl 4(%esp),%eax
14 jz L1 /* ZF is set if all bits are 0 */
15 incl %eax /* bits numbered from 1, not 0 */
16 ret
17
18 _ALIGN_TEXT
19L1: xorl %eax,%eax /* clear result */
20 ret
21END(ffs)
