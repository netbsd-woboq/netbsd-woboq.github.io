32#include <machine/asm.h>
33 RCSID("$NetBSD: memset.S,v 1.2 2014/03/22 19:38:46 jakllsch Exp $")
34
35ENTRY(memset)
36 pushl %edi
37 movl 8(%esp), %edi
38 movl 12(%esp), %eax
39 movl 16(%esp), %ecx
40 pushl %edi
41 rep
42 stosb
43 popl %eax
44 popl %edi
45 ret
46END(memset)
