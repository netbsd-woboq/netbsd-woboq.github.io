31#include <machine/asm.h>
32 RCSID("$NetBSD: strcpy.S,v 1.2 2014/03/22 19:38:46 jakllsch Exp $")
33
34ENTRY(strcpy)
35 pushl %esi
36 pushl %edi
37 movl 12(%esp), %edi
38 movl 16(%esp), %esi
39 pushl %edi
401:
41 lodsb
42 stosb
43 testb %al, %al
44 jne 1b
45 popl %eax
46 popl %edi
47 popl %esi
48 ret
49END(strcpy)
