38uint64_t fetch_and_and_8(volatile uint64_t *, uint64_t, ...)
39 asm("__sync_fetch_and_and_8");
40
41uint64_t
42fetch_and_and_8(volatile uint64_t *addr, uint64_t val, ...)
43{
44 uint64_t old, new;
45
46 do {
47 old = *addr;
48 new = old & val;
49 } while (atomic_cas_64(addr, old, new) != old);
50 return old;
51}
52
53void
54atomic_and_64(volatile uint64_t *addr, uint64_t val)
55{
56 (void) fetch_and_and_8(addr, val);
57}
58
59__strong_alias(__atomic_fetch_and_8,__sync_fetch_and_and_8)
