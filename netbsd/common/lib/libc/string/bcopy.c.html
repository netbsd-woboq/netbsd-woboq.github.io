60#ifndef __OPTIMIZE_SIZE__
61/*
62 * sizeof(word) MUST BE A POWER OF TWO
63 * SO THAT wmask BELOW IS ALL ONES
64 */
65typedef long word; /* "word" used for optimal copy speed */
66
67#define wsize sizeof(word)
68#define wmask (wsize - 1)
69
70/*
71 * Copy a block of memory, handling overlap.
72 * This is the routine that actually implements
73 * (the portable versions of) bcopy, memcpy, and memmove.
74 */
75#if defined(MEMCOPY)
76void *
77memcpy(void *dst0, const void *src0, size_t length)
78#elif defined(MEMMOVE)
79void *
80memmove(void *dst0, const void *src0, size_t length)
81#else
82void
83bcopy(const void *src0, void *dst0, size_t length)
84#endif
85{
86 char *dst = dst0;
87 const char *src = src0;
88 size_t t;
89 unsigned long u;
90
91 if (length == 0 || dst == src) /* nothing to do */
92 goto done;
93
94 /*
95 * Macros: loop-t-times; and loop-t-times, t>0
96 */
97#define TLOOP(s) if (t) TLOOP1(s)
98#define TLOOP1(s) do { s; } while (--t)
99
100 if ((unsigned long)dst < (unsigned long)src) {
101 /*
102 * Copy forward.
103 */
104 u = (unsigned long)src; /* only need low bits */
105 if ((u | (unsigned long)dst) & wmask) {
106 /*
107 * Try to align operands. This cannot be done
108 * unless the low bits match.
109 */
110 if ((u ^ (unsigned long)dst) & wmask || length < wsize)
111 t = length;
112 else
113 t = wsize - (size_t)(u & wmask);
114 length -= t;
115 TLOOP1(*dst++ = *src++);
116 }
117 /*
118 * Copy whole words, then mop up any trailing bytes.
119 */
120 t = length / wsize;
121 TLOOP(*(word *)(void *)dst = *(const word *)(const void *)src; src += wsize; dst += wsize);
122 t = length & wmask;
123 TLOOP(*dst++ = *src++);
124 } else {
125 /*
126 * Copy backwards. Otherwise essentially the same.
127 * Alignment works as before, except that it takes
128 * (t&wmask) bytes to align, not wsize-(t&wmask).
129 */
130 src += length;
131 dst += length;
132 _DIAGASSERT((unsigned long)dst >= (unsigned long)dst0);
133 _DIAGASSERT((unsigned long)src >= (unsigned long)src0);
134 u = (unsigned long)src;
135 if ((u | (unsigned long)dst) & wmask) {
136 if ((u ^ (unsigned long)dst) & wmask || length <= wsize)
137 t = length;
138 else
139 t = (size_t)(u & wmask);
140 length -= t;
141 TLOOP1(*--dst = *--src);
142 }
143 t = length / wsize;
144 TLOOP(src -= wsize; dst -= wsize; *(word *)(void *)dst = *(const word *)(const void *)src);
145 t = length & wmask;
146 TLOOP(*--dst = *--src);
147 }
148done:
149#if defined(MEMCOPY) || defined(MEMMOVE)
150 return (dst0);
151#else
152 return;
153#endif
154}
155#else /* __OPTIMIZE_SIZE__ */
156#if defined(MEMCOPY)
157/*
158 * This is designed to be small, not fast.
159 */
160void *
161memcpy(void *s1, const void *s2, size_t n)
162{
163 const char *f = s2;
164 char *t = s1;
165
166 while (n-- > 0)
167 *t++ = *f++;
168 return s1;
169}
170#elif defined(MEMMOVE)
171/*
172 * This is designed to be small, not fast.
173 */
174void *
175memmove(void *s1, const void *s2, size_t n)
176{
177 const char *f = s2;
178 char *t = s1;
179
180 if (f < t) {
181 f += n;
182 t += n;
183 while (n-- > 0)
184 *--t = *--f;
185 } else {
186 while (n-- > 0)
187 *t++ = *f++;
188 }
189 return s1;
190}
191#else
192/*
193 * This is designed to be small, not fast.
194 */
195void
196bcopy(const void *s2, void *s1, size_t n)
197{
198 const char *f = s2;
199 char *t = s1;
200
201 while (n-- > 0)
202 *t++ = *f++;
203}
204#endif
205#endif /* __OPTIMIZE_SIZE__ */
