46#ifndef popcount64 // might be defined to use a __builtin
47
48/*
49 * If uint64_t is larger than size_t, the follow assumes that
50 * splitting into 32bit halves is faster.
51 *
52 * The native pocount64 version is based on the same ideas as popcount32(3),
53 * see popcount32.c for comments.
54 */
55
56#if SIZE_MAX < 0xffffffffffffffffULL
57unsigned int
58popcount64(uint64_t v)
59{
60 return popcount32((uint32_t)(v >> 32)) +
61 popcount32((uint32_t)(v & 0xffffffffULL));
62}
63#else
64unsigned int
65popcount64(uint64_t v)
66{
67 unsigned int c;
68
69 v = v - ((v >> 1) & (uint64_t)0x5555555555555555ULL);
70 v = (v & (uint64_t)0x3333333333333333ULL) +
71 ((v >> 2) & (uint64_t)0x3333333333333333ULL);
72 v = ((v + (v >> 4)) & (uint64_t)0x0f0f0f0f0f0f0f0fULL) *
73 (uint64_t)0x0101010101010101ULL;
74 c = (unsigned int)(v >> 56);
75
76 return c;
77}
78#endif
79
80#if ULONG_MAX == 0xffffffffffffffffULL
81__strong_alias(popcountl, popcount64)
82#endif
83
84#if ULLONG_MAX == 0xffffffffffffffffULL
85__strong_alias(popcountll, popcount64)
86#endif
87
88#endif /* !popcount64 */
