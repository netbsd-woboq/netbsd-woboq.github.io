40/* 64bit version is in strspn.c */
41#if ULONG_MAX != 0xffffffffffffffffull
42
43size_t
44strcspn(const char *s, const char *charset)
45{
46 static const uint8_t idx[8] = { 1, 2, 4, 8, 16, 32, 64, 128 };
47 const char *t;
48 uint8_t set[32];
49#define UC(a) ((unsigned int)(unsigned char)(a))
50
51 if (charset[0] == '\0')
52 return strlen(s);
53 if (charset[1] == '\0') {
54 for (t = s; *t != '\0'; ++t)
55 if (*t == *charset)
56 break;
57 return t - s;
58 }
59
60 (void)memset(set, 0, sizeof(set));
61
62 for (; *charset != '\0'; ++charset)
63 set[UC(*charset) >> 3] |= idx[UC(*charset) & 7];
64
65 for (t = s; *t != '\0'; ++t)
66 if (set[UC(*t) >> 3] & idx[UC(*t) & 7])
67 break;
68 return t - s;
69}
