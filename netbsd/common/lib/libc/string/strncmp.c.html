48int
49strncmp(const char *s1, const char *s2, size_t n)
50{
51
52 if (n == 0)
53 return (0);
54 do {
55 if (*s1 != *s2++)
56 return (*(const unsigned char *)s1 -
57 *(const unsigned char *)--s2);
58 if (*s1++ == 0)
59 break;
60 } while (--n != 0);
61 return (0);
62}
