29/*
30 * snprintb: print an interpreted bitmask to a buffer
31 *
32 * => returns the length of the buffer that would be required to print the
33 * string minus the terminating NUL.
34 */

61# ifndef HAVE_SNPRINTB_M
62int
63snprintb_m(char *buf, size_t buflen, const char *bitfmt, uint64_t val,
64 size_t l_max)
65{
66 char *bp = buf, *s_bp = NULL;
67 const char *c_fmt, *s_fmt = NULL, *cur_fmt;
68 const char *sbase;
69 int bit, ch, t_len, s_len = 0, l_len, f_len, v_len, sep;
70 int restart = 0, matched = 1;
71 uint64_t field;
72
73#ifdef _KERNEL
74 /*
75 * For safety; no other *s*printf() do this, but in the kernel
76 * we don't usually check the return value
77 */
78 (void)memset(buf, 0, buflen);
79#endif /* _KERNEL */
80
81 ch = *bitfmt++;
82 switch (ch != '\177' ? ch : *bitfmt++) {
83 case 8:
84 sbase = "0%" PRIo64;
85 break;
86 case 10:
87 sbase = "%" PRId64;
88 break;
89 case 16:
90 sbase = "0x%" PRIx64;
91 break;
92 default:
93 goto internal;
94 }
95
96 /* Reserve space for trailing blank line if needed */
97 if (l_max > 0)
98 buflen--;
99
100 t_len = snprintf(bp, buflen, sbase, val);
101 if (t_len < 0)
102 goto internal;
103
104 v_len = l_len = t_len;
105
106 if ((size_t)t_len < buflen)
107 bp += t_len;
108 else
109 bp += buflen - 1;
110
111 /*
112 * If the value we printed was 0 and we're using the old-style format,
113 * we're done.
114 */
115 if ((val == 0) && (ch != '\177'))
116 goto terminate;
117
118#define STORE(c) do { l_len++; \
119 if ((size_t)(++t_len) < buflen) \
120 *bp++ = (c); \
121 } while ( /* CONSTCOND */ 0)
122
123#define BACKUP do { if (s_bp != NULL) { \
124 bp = s_bp; s_bp = NULL; \
125 t_len -= l_len - s_len; \
126 restart = 1; \
127 bitfmt = s_fmt; \
128 } \
129 STORE('>'); STORE('\0'); \
130 if ((size_t)t_len < buflen) \
131 snprintf(bp, buflen - t_len, sbase, val); \
132 t_len += v_len; l_len = v_len; bp += v_len; \
133 } while ( /* CONSTCOND */ 0)
134
135#define PUTSEP do { \
136 if (l_max > 0 && (size_t)l_len >= l_max) { \
137 BACKUP; \
138 STORE('<'); \
139 } else { \
140 /* Remember separator location */ \
141 if (l_max > 0 && sep != '<') { \
142 s_len = l_len; \
143 s_bp = bp; \
144 s_fmt = cur_fmt; \
145 } \
146 STORE(sep); \
147 restart = 0; \
148 } \
149 } while ( /* CONSTCOND */ 0)
150
151#define PUTCHR(c) do { \
152 if (l_max > 0 && (size_t)l_len >= (l_max - 1)) {\
153 BACKUP; \
154 if (restart == 0) \
155 STORE(c); \
156 else \
157 sep = '<'; \
158 } else { \
159 STORE(c); \
160 restart = 0; \
161 } \
162 } while ( /* CONSTCOND */ 0)
163
164#define PUTS(s) while ((ch = *(s)++) != 0) { \
165 PUTCHR(ch); \
166 if (restart) \
167 break; \
168 }
169#define FMTSTR(sb, f) \
170 do { \
171 f_len = snprintf(bp, buflen - t_len, sb, f); \
172 if (f_len < 0) \
173 goto internal; \
174 t_len += f_len; \
175 l_len += f_len; \
176 if ((size_t)t_len < buflen) \
177 bp += f_len; \
178 } while (/*CONSTCOND*/0)
179
180 /*
181 * Chris Torek's new bitmask format is identified by a leading \177
182 */
183 sep = '<';
184 if (ch != '\177') {
185 /* old (standard) format. */
186 for (;(bit = *bitfmt) != 0;) {
187 cur_fmt = bitfmt++;
188 if (val & (1U << (bit - 1))) {
189 PUTSEP;
190 if (restart)
191 continue;
192 sep = ',';
193 for (; (ch = *bitfmt) > ' '; ++bitfmt) {
194 PUTCHR(ch);
195 if (restart)
196 break;
197 }
198 } else
199 for (; *bitfmt > ' '; ++bitfmt)
200 continue;
201 }
202 } else {
203 /* new quad-capable format; also does fields. */
204 field = val;
205 while (c_fmt = bitfmt, (ch = *bitfmt++) != '\0') {
206 bit = *bitfmt++; /* now 0-origin */
207 switch (ch) {
208 case 'b':
209 if (((unsigned int)(val >> bit) & 1) == 0)
210 goto skip;
211 cur_fmt = c_fmt;
212 PUTSEP;
213 if (restart)
214 break;
215 PUTS(bitfmt);
216 if (restart == 0)
217 sep = ',';
218 break;
219 case 'f':
220 case 'F':
221 matched = 0;
222 cur_fmt = c_fmt;
223 f_len = *bitfmt++; /* field length */
224 field = (val >> bit) &
225 (((uint64_t)1 << f_len) - 1);
226 PUTSEP;
227 if (restart == 0)
228 sep = ',';
229 if (ch == 'F') { /* just extract */
230 /* duplicate PUTS() effect on bitfmt */
231 while (*bitfmt++ != '\0')
232 continue;
233 break;
234 }
235 if (restart == 0)
236 PUTS(bitfmt);
237 if (restart == 0)
238 PUTCHR('=');
239 if (restart == 0) {
240 FMTSTR(sbase, field);
241 if (l_max > 0 && (size_t)l_len > l_max)
242 PUTCHR('#');
243 }
244 break;
245 case '=':
246 case ':':
247 /*
248 * Here "bit" is actually a value instead,
249 * to be compared against the last field.
250 * This only works for values in [0..255],
251 * of course.
252 */
253 if ((int)field != bit)
254 goto skip;
255 matched = 1;
256 if (ch == '=')
257 PUTCHR('=');
258 PUTS(bitfmt);
259 break;
260 case '*':
261 bitfmt--;
262 if (!matched) {
263 matched = 1;
264 FMTSTR(bitfmt, field);
265 }
266 /*FALLTHROUGH*/
267 default:
268 skip:
269 while (*bitfmt++ != '\0')
270 continue;
271 break;
272 }
273 }
274 }
275 l_len++;
276 if (sep != '<' && (size_t)(++t_len) < buflen)
277 *bp++ = '>';
278terminate:
279 *bp++ = '\0';
280 if (l_max != 0) {
281 t_len++;
282 *bp = '\0';
283 }
284 return t_len;
285internal:
286#ifndef _KERNEL
287 errno = EINVAL;
288#endif
289 return -1;
290}
291
292int
293snprintb(char *buf, size_t buflen, const char *bitfmt, uint64_t val)
294{
295 return snprintb_m(buf, buflen, bitfmt, val, 0);
296}
