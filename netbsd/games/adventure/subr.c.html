89static int
90liq2(int pbotl)
91{
92 return ((1 - pbotl) * water + (pbotl / 2) * (water + oil));
93}

95int
96liq(void)
97{
98 int i;
99 i = prop[bottle];
100 if (i > -1 - i)
101 return (liq2(i));
102 else
103 return (liq2(-1 - i));
104}
