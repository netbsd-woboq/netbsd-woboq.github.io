57void
58datime(int *d, int *t)
59{
60 time_t tvec;
61 struct tm *tptr;
62
63 time(&tvec);
64 tptr = localtime(&tvec);
65 /* day since 1977 (mod leap) */
66 *d = (tptr->tm_yday + 365 * (tptr->tm_year - 77)
67 + (tptr->tm_year - 77) / 4 - (tptr->tm_year - 1) / 100
68 + (tptr->tm_year + 299) / 400);
69 /* bug: this will overflow in the year 2066 AD (with 16 bit int) */
70 /* it will be attributed to Wm the C's millenial celebration */
71 /* and minutes since midnite */
72 *t = tptr->tm_hour * 60 + tptr->tm_min;
73} /* pretty painless */

78void
79poof(void)
80{
81 strcpy(magic, DECR('d', 'w', 'a', 'r', 'f'));
82 latency = 45;
83}

85int
86Start(void)
87{
88 int d, t, delay;
89
90 datime(&d, &t);
91 delay = (d - saveday) * 1440 + (t - savet); /* good for about a
92 * month */
93
94 if (delay >= latency) {
95 saved = -1;
96 return (FALSE);
97 }
98 printf("This adventure was suspended a mere %d minute%s ago.",
99 delay, delay == 1 ? "" : "s");
100 if (delay <= latency / 3) {
101 mspeak(2);
102 exit(0);
103 }
104 mspeak(8);
105 if (!wizard()) {
106 mspeak(9);
107 exit(0);
108 }
109 saved = -1;
110 return (FALSE);
111}

113/* not as complex as advent/10 (for now) */
114static int
115wizard(void)
116{
117 char *word, *x;
118 if (!yesm(16, 0, 7))
119 return (FALSE);
120 mspeak(17);
121 getin(&word, &x);
122 if (!weq(word, magic)) {
123 mspeak(20);
124 return (FALSE);
125 }
126 mspeak(19);
127 return (TRUE);
128}

130void
131ciao(void)
132{
133 char fname[80];
134 size_t pos;
135
136 printf("What would you like to call the saved version?\n");
137 /* XXX - should use fgetln to avoid arbitrary limit */
138 for (pos = 0; pos < sizeof(fname) - 1; pos++) {
139 int ch;
140 ch = getchar();
141 if (ch == '\n' || ch == EOF)
142 break;
143 fname[pos] = ch;
144 }
145 fname[pos] = '\0';
146 if (save(fname) != 0)
147 return; /* Save failed */
148 printf("To resume, say \"adventure %s\".\n", fname);
149 printf("\"With these rooms I might now have been familiarly ");
150 printf("acquainted.\"\n");
151 exit(0);
152}
