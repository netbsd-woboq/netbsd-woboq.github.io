109int
110wrtext(const char *const *t)
111{
112 int i;
113 const char *s, *a;
114
115 fixtty(&noech);
116 while (*t != 0) {
117 s = a = *t;
118 for (i = 0; *a != '\0'; i--)
119 a++;
120 if (i) {
121 writel(s);
122 writec('\n');
123 } else {
124 writel("-->");
125 fixtty(&raw);
126 while ((i = readc()) != ' ' && i != '\n');
127 fixtty(&noech);
128 clear();
129 }
130 t++;
131 }
132 fixtty(&raw);
133 return (0);
134}
