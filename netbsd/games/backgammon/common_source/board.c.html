43static int i, j, k;
44static char ln[60];
45
46static void wrbsub(void);
47
48void
49wrboard(void)
50{
51 int l;
52 static const char bl[] =
53 "| | | |\n";
54 static const char sv[] =
55 "| | | | \n";
56
57 fixtty(&noech);
58 clear();
59
60 if (tflag) {
61 fboard();
62 goto lastline;
63 }
64 writel("_____________________________________________________\n");
65 writel(bl);
66 strcpy(ln, bl);
67 for (j = 1; j < 50; j += 4) {
68 k = j / 4 + (j > 24 ? 12 : 13);
69 ln[j + 1] = k % 10 + '0';
70 ln[j] = k / 10 + '0';
71 if (j == 21)
72 j += 4;
73 }
74 writel(ln);
75 for (i = 0; i < 5; i++) {
76 strcpy(ln, sv);
77 for (j = 1; j < 50; j += 4) {
78 k = j / 4 + (j > 24 ? 12 : 13);
79 wrbsub();
80 if (j == 21)
81 j += 4;
82 }
83 if (-board[25] > i)
84 ln[26] = 'w';
85 if (-board[25] > i + 5)
86 ln[25] = 'w';
87 if (-board[25] > i + 10)
88 ln[27] = 'w';
89 l = 53;
90 if (off[1] > i || (off[1] < 0 && off[1] + 15 > i)) {
91 ln[54] = 'r';
92 l = 55;
93 }
94 if (off[1] > i + 5 || (off[1] < 0 && off[1] + 15 > i + 5)) {
95 ln[55] = 'r';
96 l = 56;
97 }
98 if (off[1] > i + 10 || (off[1] < 0 && off[1] + 15 > i + 10)) {
99 ln[56] = 'r';
100 l = 57;
101 }
102 ln[l++] = '\n';
103 ln[l] = '\0';
104 writel(ln);
105 }
106 strcpy(ln, bl);
107 ln[25] = 'B';
108 ln[26] = 'A';
109 ln[27] = 'R';
110 writel(ln);
111 strcpy(ln, sv);
112 for (i = 4; i > -1; i--) {
113 for (j = 1; j < 50; j += 4) {
114 k = ((j > 24 ? 53 : 49) - j) / 4;
115 wrbsub();
116 if (j == 21)
117 j += 4;
118 }
119 if (board[0] > i)
120 ln[26] = 'r';
121 if (board[0] > i + 5)
122 ln[25] = 'r';
123 if (board[0] > i + 10)
124 ln[27] = 'r';
125 l = 53;
126 if (off[0] > i || (off[0] < 0 && off[0] + 15 > i)) {
127 ln[54] = 'w';
128 l = 55;
129 }
130 if (off[0] > i + 5 || (off[0] < 0 && off[0] + 15 > i + 5)) {
131 ln[55] = 'w';
132 l = 56;
133 }
134 if (off[0] > i + 10 || (off[0] < 0 && off[0] + 15 > i + 10)) {
135 ln[56] = 'w';
136 l = 57;
137 }
138 ln[l++] = '\n';
139 ln[l] = '\0';
140 writel(ln);
141 }
142 strcpy(ln, bl);
143 for (j = 1; j < 50; j += 4) {
144 k = ((j > 24 ? 53 : 49) - j) / 4;
145 ln[j + 1] = k % 10 + '0';
146 if (k > 9)
147 ln[j] = k / 10 + '0';
148 if (j == 21)
149 j += 4;
150 }
151 writel(ln);
152 writel("|_______________________|___|_______________________|\n");
153
154lastline:
155 gwrite();
156 if (tflag)
157 curmove(18, 0);
158 else {
159 writec('\n');
160 writec('\n');
161 }
162 fixtty(&raw);
163}
164
165static void
166wrbsub(void)
167{
168 int m;
169 char d;
170
171 if (board[k] > 0) {
172 m = board[k];
173 d = 'r';
174 } else {
175 m = -board[k];
176 d = 'w';
177 }
178 if (m > i)
179 ln[j + 1] = d;
180 if (m > i + 5)
181 ln[j] = d;
182 if (m > i + 10)
183 ln[j + 2] = d;
184}
