97void
98move_init(struct move *mm)
99{
100 mm->D0 = 0;
101 mm->D1 = 0;
102 mm->mvlim = 0;
103 mm->p[0] = mm->p[1] = mm->p[2] = mm->p[3] = mm->p[4] = 0;
104 mm->g[0] = mm->g[1] = mm->g[2] = mm->g[3] = mm->g[4] = 0;
105 mm->h[0] = mm->h[1] = mm->h[2] = mm->h[3] = 0;
106 mm->d0 = 0;
107}
