43void
44odds(int r1, int r2, int val)
45{
46 int i, j;
47
48 if (r1 == 0) {
49 for (i = 0; i < 6; i++)
50 for (j = 0; j < 6; j++)
51 table[i][j] = 0;
52 return;
53 } else {
54 r1--;
55 if (r2-- == 0) {
56 for (i = 0; i < 6; i++) {
57 table[i][r1] += val;
58 table[r1][i] += val;
59 }
60 } else {
61 table[r2][r1] += val;
62 table[r1][r2] += val;
63 }
64 }
65}

67int
68count(void)
69{
70 int i;
71 int j;
72 int total;
73
74 total = 0;
75 for (i = 0; i < 6; i++)
76 for (j = 0; j < 6; j++)
77 total += table[i][j];
78 return (total);
79}

81int
82canhit(int i, int c)
83{
84 int j, k, b;
85 int a, diff, place, addon, menstuck;
86
87 if (c == 0)
88 odds(0, 0, 0);
89 if (board[i] > 0) {
90 a = -1;
91 b = 25;
92 } else {
93 a = 1;
94 b = 0;
95 }
96 place = abs(25 - b - i);
97 menstuck = abs(board[b]);
98 for (j = b; j != i; j += a) {
99 if (board[j] * a > 0) {
100 diff = abs(j - i);
101 addon = place + ((board[j] * a > 2 || j == b) ? 5 : 0);
102 if ((j == b && menstuck == 1) ||
103 (j != b && menstuck == 0))
104 for (k = 1; k < diff; k++)
105 if (k < 7 && diff - k < 7 &&
106 (board[i + a * k] * a >= 0 ||
107 board[i + a * (diff - k)] >= 0))
108 odds(k, diff - k, addon);
109 if ((j == b || menstuck < 2) && diff < 7)
110 odds(diff, 0, addon);
111 }
112 if (j == b && menstuck > 1)
113 break;
114 }
115 return (count());
116}
