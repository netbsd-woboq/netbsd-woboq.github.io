46int
47makmove(struct move *mm, int i)
48{
49 int n, d;
50 int max;
51
52 d = mm->d0;
53 n = abs(mm->g[i] - mm->p[i]);
54 max = (*offptr < 0 ? 7 : last());
55 if (board[mm->p[i]] * cturn <= 0)
56 return (checkd(mm, d) + 2);
57 if (mm->g[i] != home && board[mm->g[i]] * cturn < -1)
58 return (checkd(mm, d) + 3);
59 if (i || mm->D0 == mm->D1) {
60 if (n == max ? mm->D1 < n : mm->D1 != n)
61 return (checkd(mm, d) + 1);
62 } else {
63 if (n == max ? mm->D0 < n && mm->D1 < n : mm->D0 != n && mm->D1 != n)
64 return (checkd(mm, d) + 1);
65 if (n == max ? mm->D0 < n : mm->D0 != n) {
66 if (mm->d0)
67 return (checkd(mm, d) + 1);
68 mswap(mm);
69 }
70 }
71 if (mm->g[i] == home && *offptr < 0)
72 return (checkd(mm, d) + 4);
73 mm->h[i] = 0;
74 board[mm->p[i]] -= cturn;
75 if (mm->g[i] != home) {
76 if (board[mm->g[i]] == -cturn) {
77 board[home] -= cturn;
78 board[mm->g[i]] = 0;
79 mm->h[i] = 1;
80 if (abs(bar - mm->g[i]) < 7) {
81 (*inopp)--;
82 if (*offopp >= 0)
83 *offopp -= 15;
84 }
85 }
86 board[mm->g[i]] += cturn;
87 if (abs(home - mm->g[i]) < 7 && abs(home - mm->p[i]) > 6) {
88 (*inptr)++;
89 if (*inptr + *offptr == 0)
90 *offptr += 15;
91 }
92 } else {
93 (*offptr)++;
94 (*inptr)--;
95 }
96 return (0);
97}

99void
100moverr(struct move *mm, int i)
101{
102 int j;
103
104 if (tflag)
105 curmove(20, 0);
106 else
107 writec('\n');
108 writel("Error: ");
109 for (j = 0; j <= i; j++) {
110 wrint(mm->p[j]);
111 writec('-');
112 wrint(mm->g[j]);
113 if (j < i)
114 writec(',');
115 }
116 writel("... ");
117 movback(mm, i);
118}

121static int
122checkd(struct move *mm, int d)
123{
124 if (mm->d0 != d)
125 mswap(mm);
126 return (0);
127}
128
129static int
130last(void)
131{
132 int i;
133
134 for (i = home - 6 * cturn; i != home; i += cturn)
135 if (board[i] * cturn > 0)
136 return (abs(home - i));
137 return (-1);
138}
139
140void
141movback(struct move *mm, int i)
142{
143 int j;
144
145 for (j = i - 1; j >= 0; j--)
146 backone(mm, j);
147}
148
149void
150backone(struct move *mm, int i)
151{
152 board[mm->p[i]] += cturn;
153 if (mm->g[i] != home) {
154 board[mm->g[i]] -= cturn;
155 if (abs(mm->g[i] - home) < 7 && abs(mm->p[i] - home) > 6) {
156 (*inptr)--;
157 if (*inptr + *offptr < 15 && *offptr >= 0)
158 *offptr -= 15;
159 }
160 } else {
161 (*offptr)--;
162 (*inptr)++;
163 }
164 if (mm->h[i]) {
165 board[home] += cturn;
166 board[mm->g[i]] = -cturn;
167 if (abs(bar - mm->g[i]) < 7) {
168 (*inopp)++;
169 if (*inopp + *offopp == 0)
170 *offopp += 15;
171 }
172 }
