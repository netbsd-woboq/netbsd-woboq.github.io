49static const char *const contin[] = {
50 "",
51 0
52};
53
54int
55main(int argc __unused, char *argv[])
56{
57 int i;
58 struct move mmstore, *mm;
59
60 /* revoke setgid privileges */
61 setgid(getgid());
62
63 signal(SIGINT, getout);
64 if (tcgetattr(0, &old) == -1) /* get old tty mode */
65 errexit("teachgammon(gtty)");
66 noech = old;
67 noech.c_lflag &= ~ECHO;
68 raw = noech;
69 raw.c_lflag &= ~ICANON; /* set up modes */
70 ospeed = cfgetospeed(&old); /* for termlib */
71 tflag = getcaps(getenv("TERM"));
72
73 /* need this now beceause getarg() may try to load a game */
74 mm = &mmstore;
75 move_init(mm);
76 while (*++argv != 0)
77 getarg(mm, &argv);
78 if (tflag) {
79 noech.c_oflag &= ~(ONLCR | OXTABS);
80 raw.c_oflag &= ~(ONLCR | OXTABS);
81 clear();
82 }
83 wrtext(hello);
84 wrtext(list);
85 i = wrtext(contin);
86 if (i == 0)
87 i = 2;
88 init();
89 while (i)
90 switch (i) {
91 case 1:
92 leave();
93
94 case 2:
95 if ((i = wrtext(intro1)) != 0)
96 break;
97 wrboard();
98 if ((i = wrtext(intro2)) != 0)
99 break;
100
101 /* FALLTHROUGH */
102 case 3:
103 if ((i = wrtext(moves)) != 0)
104 break;
105
106 /* FALLTHROUGH */
107 case 4:
108 if ((i = wrtext(removepiece)) != 0)
109 break;
110
111 /* FALLTHROUGH */
112 case 5:
113 if ((i = wrtext(hits)) != 0)
114 break;
115
116 /* FALLTHROUGH */
117 case 6:
118 if ((i = wrtext(endgame)) != 0)
119 break;
120
121 /* FALLTHROUGH */
122 case 7:
123 if ((i = wrtext(doubl)) != 0)
124 break;
125
126 /* FALLTHROUGH */
127 case 8:
128 if ((i = wrtext(stragy)) != 0)
129 break;
130
131 /* FALLTHROUGH */
132 case 9:
133 if ((i = wrtext(prog)) != 0)
134 break;
135
136 /* FALLTHROUGH */
137 case 10:
138 if ((i = wrtext(lastch)) != 0)
139 break;
140 }
141 tutor(mm);
142 /* NOTREACHED */
143 return (0);
144}
145
146void
147leave(void)
148{
149 if (tflag)
150 clear();
151 else
152 writec('\n');
153 fixtty(&old);
154 execl(EXEC, "backgammon", "-n", args[0]?args:0, (char *) 0);
155 writel("Help! Backgammon program is missing\007!!\n");
156 exit(1);
157}
