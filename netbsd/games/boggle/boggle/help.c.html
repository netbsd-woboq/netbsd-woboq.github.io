50extern int nlines;
51
52int
53help(void)
54{
55 int eof, i;
56 FILE *fp;
57 WINDOW *win;
58 char buf[BUFSIZ];
59
60 if ((fp = fopen(HELPFILE, "r")) == NULL)
61 return(-1);
62 win = newwin(0, 0, 0, 0);
63 clearok(win, 1);
64
65 eof = 0;
66 if (ungetc(getc(fp), fp) == EOF) {
67 wprintw(win, "There doesn't seem to be any help.");
68 eof = 1; /* Nothing there... */
69 }
70
71 while (!eof) {
72 for (i = 0; i < nlines - 3; i++) {
73 if (fgets(buf, sizeof(buf), fp) == NULL) {
74 eof = 1;
75 break;
76 }
77 if (buf[0] == '.' && buf[1] == '\n')
78 break;
79 wprintw(win, "%s", buf);
80 }
81 if (eof || ungetc(getc(fp), fp) == EOF) {
82 eof = 1;
83 break;
84 }
85 wmove(win, nlines - 1, 0);
86 wprintw(win,
87 "Type <space> to continue, anything else to quit...");
88 wrefresh(win);
89 if ((inputch() & 0177) != ' ')
90 break;
91 wclear(win);
92 }
93
94 fclose(fp);
95 if (eof) {
96 wmove(win, nlines - 1, 0);
97 wprintw(win, "Hit any key to continue...");
98 wrefresh(win);
99 inputch();
100 }
101 delwin(win);
102 clearok(stdscr, 1);
103 refresh();
104 return(0);
105}
