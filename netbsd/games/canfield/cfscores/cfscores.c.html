57static int dbfd;
58
59static void printuser(const struct passwd *, int);
60
61int
62main(int argc, char *argv[])
63{
64 struct passwd *pw;
65 uid_t uid;
66
67 /* Revoke setgid privileges */
68 setgid(getgid());
69
70 if (argc > 2) {
71 fprintf(stderr, "Usage: %s -a | %s [user]\n",
72 getprogname(), getprogname());
73 exit(1);
74 }
75 dbfd = open(_PATH_SCORE, O_RDONLY);
76 if (dbfd < 0)
77 err(2, "open %s", _PATH_SCORE);
78 setpwent();
79 if (argc == 1) {
80 uid = getuid();
81 pw = getpwuid(uid);
82 if (pw == NULL) {
83 errx(2, "You are not listed in the password file?!?");
84 }
85 printuser(pw, 1);
86 exit(0);
87 }
88 if (strcmp(argv[1], "-a") == 0) {
89 while ((pw = getpwent()) != NULL)
90 printuser(pw, 0);
91 exit(0);
92 }
93 pw = getpwnam(argv[1]);
94 if (pw == NULL) {
95 errx(3, "User %s unknown", argv[1]);
96 }
97 printuser(pw, 1);
98 exit(0);
99}

101/*
102 * print out info for specified password entry
103 */
104static void
105printuser(const struct passwd *pw, int printfail)
106{
107 struct betinfo total;
108 off_t pos;
109 ssize_t i;
110
111 pos = pw->pw_uid * (off_t)sizeof(struct betinfo);
112 /* test pos, not pw_uid; uid_t can be unsigned, which makes gcc warn */
113 if (pos < 0) {
114 warnx("Bad uid %d", (int)pw->pw_uid);
115 return;
116 }
117 if (lseek(dbfd, pos, SEEK_SET) < 0) {
118 warn("%s: lseek", _PATH_SCORE);
119 }
120 i = read(dbfd, &total, sizeof(total));
121 if (i < 0)
122 warn("read %s", _PATH_SCORE);
123 if (i == 0 || total.hand == 0) {
124 if (printfail)
125 printf("%s has never played canfield.\n", pw->pw_name);
126 return;
127 }
128 printf("*----------------------*\n");
129 if (total.worth >= 0)
130 printf("* Winnings for %-8s*\n", pw->pw_name);
131 else
132 printf("* Losses for %-10s*\n", pw->pw_name);
133 printf("*======================*\n");
134 printf("|Costs Total |\n");
135 printf("| Hands %8ld |\n", total.hand);
136 printf("| Inspections %8ld |\n", total.inspection);
137 printf("| Games %8ld |\n", total.game);
138 printf("| Runs %8ld |\n", total.runs);
139 printf("| Information %8ld |\n", total.information);
140 printf("| Think time %8ld |\n", total.thinktime);
141 printf("|Total Costs %8ld |\n", total.wins - total.worth);
142 printf("|Winnings %8ld |\n", total.wins);
143 printf("|Net Worth %8ld |\n", total.worth);
144 printf("*----------------------*\n\n");
145}
