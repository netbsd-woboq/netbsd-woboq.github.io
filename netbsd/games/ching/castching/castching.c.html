53/*
54 * castching - Read a question, cast a change, and output the line
55 * values to the standard output for processing by "printching".
56 */
57#include <stdio.h>
58#include <stdlib.h>
59#include <time.h>
60#include <unistd.h>
61#include "ching.h"
62
63static time_t now; /* current time */
64
65static unsigned seed; /* seed for random number generator */
66
67static int getquest(void);
68static unsigned getrand(void);
69static unsigned getrnum(void);
70static char *change(void);
71
72static char string[6+1]; /* where the actual change string is put */
73
74static int table[2][2][2] = {
75 { { OYIN, YYANG,}, { YYANG, YYIN,} },
76 { { YYANG, YYIN,}, { YYIN, OYANG,} },
77};
78
79/*ARGSUSED*/
80int
81main(int argc, char **argv)
82{
83 time(&now);
84 /* randomize */
85 seed = (int)now + getquest() + getgid() + getuid() + getpid();
86 printf("%s\n", change());
87 exit(0);
88}
89
90/*
91 * Hash the question by adding all the characters together.
92 */
93static int
94getquest(void)
95{
96 int result;
97 int c;
98
99 result = 0;
100 while ((c = getchar()) != EOF)
101 result += c;
102 return(result);
103}
104
105/*
106 * Get a set of six lines making up a change.
107 */
108static char *
109change(void)
110{
111 int i;
112
113 for (i = 0; i < 6; i++)
114 string[i] = table[getrnum()&01][getrnum()&01][getrnum()&01] + '0';
115 string[i] = '\0';
116 return(string);
117}
118
119/*
120 * Get a number more random than what getrand() gives.
121 */
122static unsigned
123getrnum(void)
124{
125 return((getrand())>>(getrand()%17));
126}
127
128/*
129 * Get a random number.
130 */
131static unsigned
132getrand(void)
133{
134 return(seed = (seed*13077) + 6925);
135}
