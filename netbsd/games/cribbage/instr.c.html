58void
59instructions(void)
60{
61 int pstat;
62 int fd;
63 pid_t pid;
64 const char *path;
65
66 switch (pid = vfork()) {
67 case -1:
68 err(1, "vfork");
69 case 0:
70 /* Follow the same behaviour for pagers as defined in POSIX.2
71 * for mailx and man. We only use a pager if stdout is
72 * a terminal, and we pass the file on stdin to sh -c pager.
73 */
74 if (!isatty(1))
75 path = "cat";
76 else {
77 if (!(path = getenv("PAGER")) || (*path == 0))
78 path = _PATH_MORE;
79 }
80 if ((fd = open(_PATH_INSTR, O_RDONLY)) == -1) {
81 warn("open %s", _PATH_INSTR);
82 _exit(1);
83 }
84 if (dup2(fd, 0) == -1) {
85 warn("dup2");
86 _exit(1);
87 }
88 execl("/bin/sh", "sh", "-c", path, (char *) NULL);
89 warn(NULL);
90 _exit(1);
91 default:
92 do {
93 pid = waitpid(pid, &pstat, 0);
94 } while (pid == -1 && errno == EINTR);
95 if (pid == -1 || WEXITSTATUS(pstat))
96 exit(1);
97 }
98}
