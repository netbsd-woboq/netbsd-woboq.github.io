83int
84main(int argc __unused, char *argv[])
85{
86 char *cp;
87
88 nogamefile();
89 game = (cp = strrchr(*argv, '/')) ? ++cp : *argv;
90
91 if (!strcmp(game, "dm"))
92 exit(0);
93
94 gametty = ttyname(0);
95 unsetenv("TZ");
96 (void)time(&now);
97 read_config();
98#ifdef LOG
99 logfile();
100#endif
101 play(argv);
102 /*NOTREACHED*/
103 return (0);
104}

106/*
107 * play --
108 * play the game
109 */
110static void
111play(char **args)
112{
113 char pbuf[MAXPATHLEN];
114
115 snprintf(pbuf, sizeof(pbuf), "%s%s", _PATH_HIDE, game);
116 if (priority > 0) /* < 0 requires root */
117 (void)setpriority(PRIO_PROCESS, 0, priority);
118 execv(pbuf, args);
119 err(1, "%s", pbuf);
120}

122/*
123 * read_config --
124 * read through config file, looking for key words.
125 */
126static void
127read_config(void)
128{
129 FILE *cfp;
130 char lbuf[BUFSIZ], f1[40], f2[40], f3[40], f4[40], f5[40];
131
132 if (!(cfp = fopen(_PATH_CONFIG, "r")))
133 return;
134 while (fgets(lbuf, sizeof(lbuf), cfp))
135 switch (*lbuf) {
136 case 'b': /* badtty */
137 if (sscanf(lbuf, "%39s%39s", f1, f2) != 2 ||
138 strcasecmp(f1, "badtty"))
139 break;
140 c_tty(f2);
141 break;
142 case 'g': /* game */
143 if (sscanf(lbuf, "%39s%39s%39s%39s%39s",
144 f1, f2, f3, f4, f5) != 5 || strcasecmp(f1, "game"))
145 break;
146 c_game(f2, f3, f4, f5);
147 break;
148 case 't': /* time */
149 if (sscanf(lbuf, "%39s%39s%39s%39s", f1, f2, f3, f4) != 4 ||
150 strcasecmp(f1, "time"))
151 break;
152 c_day(f2, f3, f4);
153 }
154 (void)fclose(cfp);
155}

157/*
158 * c_day --
159 * if day is today, see if okay to play
160 */
161static void
162c_day(const char *s_day, const char *s_start, const char *s_stop)
163{
164 static const char *const days[] = {
165 "sunday", "monday", "tuesday", "wednesday",
166 "thursday", "friday", "saturday",
167 };
168 static struct tm *ct;
169 int start, stop;
170
171 if (!ct)
172 ct = localtime(&now);
173 if (strcasecmp(s_day, days[ct->tm_wday]))
174 return;
175 if (!isdigit((unsigned char)*s_start) ||
176 !isdigit((unsigned char)*s_stop))
177 return;
178 start = atoi(s_start);
179 stop = atoi(s_stop);
180 if (ct->tm_hour >= start && ct->tm_hour < stop) {
181 if (start == 0 && stop == 24)
182 errx(0, "Sorry, games are not available today.");
183 else
184 errx(0, "Sorry, games are not available from %s to %s today.",
185 hour(start), hour(stop));
186 }
187}

189/*
190 * c_tty --
191 * decide if this tty can be used for games.
192 */
193static void
194c_tty(const char *tty)
195{
196 static int first = 1;
197 static char *p_tty;
198
199 if (first) {
200 p_tty = strrchr(gametty, '/');
201 first = 0;
202 }
203
204 if (!strcmp(gametty, tty) || (p_tty && !strcmp(p_tty, tty)))
205 errx(1, "Sorry, you may not play games on %s.", gametty);
206}

208/*
209 * c_game --
210 * see if game can be played now.
211 */
212static void
213c_game(const char *s_game, const char *s_load, const char *s_users,
214 const char *s_priority)
215{
216 static int found;
217
218 if (found)
219 return;
220 if (strcmp(game, s_game) && strcasecmp("default", s_game))
221 return;
222 ++found;
223 if (isdigit((unsigned char)*s_load) && atoi(s_load) < load())
224 errx(0, "Sorry, the load average is too high right now.");
225 if (isdigit((unsigned char)*s_users) && atoi(s_users) <= users())
226 errx(0, "Sorry, there are too many users logged on right now.");
227 if (isdigit((unsigned char)*s_priority))
228 priority = atoi(s_priority);
229}

231/*
232 * load --
233 * return 15 minute load average
234 */
235static double
236load(void)
237{
238 double avenrun[3];
239
240 if (getloadavg(avenrun, sizeof(avenrun)/sizeof(avenrun[0])) < 0)
241 err(1, "getloadavg() failed");
242 return (avenrun[2]);
243}

245/*
246 * users --
247 * return current number of users
248 * todo: check idle time; if idle more than X minutes, don't
249 * count them.
250 */
251static int
252users(void)
253{
254 struct utmpentry *ep;
255 int nusers;
256
257 nusers = getutentries(NULL, &ep);
258 return nusers;
259}

261static void
262nogamefile(void)
263{
264 int fd, n;
265 char buf[BUFSIZ];
266
267 if ((fd = open(_PATH_NOGAMES, O_RDONLY, 0)) >= 0) {
268#define MESG "Sorry, no games right now.\n\n"
269 (void)write(2, MESG, sizeof(MESG) - 1);
270 while ((n = read(fd, buf, sizeof(buf))) > 0)
271 (void)write(2, buf, n);
272 exit(1);
273 }
274}

276/*
277 * hour --
278 * print out the hour in human form
279 */
280static const char *
281hour(int h)
282{
283 static const char *const hours[] = {
284 "midnight", "1am", "2am", "3am", "4am", "5am",
285 "6am", "7am", "8am", "9am", "10am", "11am",
286 "noon", "1pm", "2pm", "3pm", "4pm", "5pm",
287 "6pm", "7pm", "8pm", "9pm", "10pm", "11pm", "midnight" };
288
289 if (h < 0 || h > 24)
290 return ("BAD TIME");
291 else
292 return (hours[h]);
293}

295#ifdef LOG
296/*
297 * logfile --
298 * log play of game
299 */
300static void
301logfile(void)
302{
303 struct passwd *pw;
304 FILE *lp;
305 uid_t uid;
306 int lock_cnt;
307
308 if (lp = fopen(_PATH_LOG, "a")) {
309 for (lock_cnt = 0;; ++lock_cnt) {
310 if (!flock(fileno(lp), LOCK_EX))
311 break;
312 if (lock_cnt == 4) {
313 warnx("log lock");
314 (void)fclose(lp);
315 return;
316 }
317 sleep(1);
318 }
319 if (pw = getpwuid(uid = getuid()))
320 fputs(pw->pw_name, lp);
321 else
322 fprintf(lp, "%u", uid);
323 fprintf(lp, "\t%s\t%s\t%s", game, gametty, ctime(&now));
324 (void)flock(fileno(lp), LOCK_UN);
325 (void)fclose(lp);
326 }
327}
