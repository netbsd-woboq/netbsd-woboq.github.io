64#ifndef _DEF_FLAG_H_
65#define _DEF_FLAG_H_
66struct flag {
67 unsigned ident; /* social security number for each monster */
68 unsigned debug:1; /* in debugging mode */
69#define wizard flags.debug
70 unsigned toplin:2; /* a top line (message) has been printed */
71 /* 0: top line empty; 2: no --More-- reqd. */
72 unsigned cbreak:1; /* in cbreak mode, rogue format */
73 unsigned standout:1; /* use standout for --More-- */
74 unsigned nonull:1; /* avoid sending nulls to the terminal */
75 unsigned time:1; /* display elapsed 'time' */
76 unsigned nonews:1; /* suppress news printing */
77 unsigned notombstone:1;
78 unsigned end_top, end_around; /* describe desired score list */
79 unsigned end_own:1; /* idem (list all own scores) */
80 unsigned no_rest_on_space:1; /* spaces are ignored */
81 unsigned beginner:1;
82 unsigned female:1;
83 unsigned invlet_constant:1; /* let objects keep their
84 inventory symbol */
85 unsigned move:1;
86 unsigned mv:1;
87 unsigned run:3; /* 0: h (etc), 1: H (etc), 2: fh (etc) */
88 /* 3: FH, 4: ff+, 5: ff-, 6: FF+, 7: FF- */
89 unsigned nopick:1; /* do not pickup objects */
90 unsigned echo:1; /* 1 to echo characters */
91 unsigned botl:1; /* partially redo status line */
92 unsigned botlx:1; /* print an entirely new bottom line */
93 unsigned nscrinh:1; /* inhibit nscr() in pline(); */
94 unsigned made_amulet:1;
95 unsigned no_of_wizards:2;/* 0, 1 or 2 (wizard and his shadow) */
96 /* reset from 2 to 1, but never to 0 */
97 unsigned moonphase:3;
98#define NEW_MOON 0
99#define FULL_MOON 4
100
101};
102
103extern struct flag flags;
104
105#endif /* _DEF_FLAG_H_ */
