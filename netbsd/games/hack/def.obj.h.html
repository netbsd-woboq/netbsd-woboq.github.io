64#ifndef _DEF_OBJ_H_
65#define _DEF_OBJ_H_
66struct obj {
67 struct obj *nobj;
68 unsigned o_id;
69 unsigned o_cnt_id; /* id of container object is in */
70 xchar ox,oy;
71 xchar odx,ody;
72 uchar otyp;
73 uchar owt;
74 uchar quan; /* use oextra for tmp gold objects */
75 schar spe; /* quality of weapon, armor or ring (+ or -)
76 number of charges for wand ( >= -1 )
77 special for uball and amulet %% BAH */
78 char olet;
79 char invlet;
80 Bitfield(oinvis,1); /* not yet implemented */
81 Bitfield(odispl,1);
82 Bitfield(known,1); /* exact nature known */
83 Bitfield(dknown,1); /* color or text known */
84 Bitfield(cursed,1);
85 Bitfield(unpaid,1); /* on some bill */
86 Bitfield(rustfree,1);
87 Bitfield(onamelth,6);
88 long age; /* creation date */
89 long owornmask;
90#define W_ARM 01L
91#define W_ARM2 02L
92#define W_ARMH 04L
93#define W_ARMS 010L
94#define W_ARMG 020L
95#define W_ARMOR (W_ARM | W_ARM2 | W_ARMH | W_ARMS | W_ARMG)
96#define W_RINGL 010000L /* make W_RINGL = RING_LEFT (see uprop) */
97#define W_RINGR 020000L
98#define W_RING (W_RINGL | W_RINGR)
99#define W_WEP 01000L
100#define W_BALL 02000L
101#define W_CHAIN 04000L
102 long oextra[1]; /* used for name of ordinary objects - length
103 is flexible; amount for tmp gold objects */
104};
105
106extern struct obj *fobj;
107
108#define newobj(xl) ((struct obj *) alloc((xl) + sizeof(struct obj)))
109#define ONAME(otmp) ((char *) otmp->oextra)
110#define OGOLD(otmp) (otmp->oextra[0])
111#endif /* _DEF_OBJ_H_ */
