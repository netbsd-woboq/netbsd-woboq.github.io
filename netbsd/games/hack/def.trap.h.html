64#ifndef _DEF_TRAP_H_
65#define _DEF_TRAP_H_
66struct trap {
67 struct trap *ntrap;
68 xchar tx,ty;
69 unsigned ttyp:5;
70 unsigned tseen:1;
71 unsigned once:1;
72};
73
74extern struct trap *ftrap;
75#define newtrap() ((struct trap *) alloc(sizeof(struct trap)))
76
77/* various kinds of traps */
78#define BEAR_TRAP 0
79#define ARROW_TRAP 1
80#define DART_TRAP 2
81#define TRAPDOOR 3
82#define TELEP_TRAP 4
83#define PIT 5
84#define SLP_GAS_TRAP 6
85#define PIERC 7
86#define MIMIC 8 /* used only in mklev.c */
87#define TRAPNUM 9 /* if not less than 32, change sizeof(ttyp) */
88 /* see also mtrapseen (bit map) */
89#endif /* _DEF_TRAP_H_ */
