69/*
70 * This cannot be part of hack.tty.c (as it was earlier) since on some
71 * systems (e.g. MUNIX) the include files <termio.h> and <sgtty.h> define the
72 * same constants, and the C preprocessor complains.
73 */
74#include <termios.h>
75#include "hack.h"
76#include "extern.h"
77static struct termios termios;
78
79void
80getioctls(void)
81{
82 (void) tcgetattr(fileno(stdin), &termios);
83}
84
85void
86setioctls(void)
87{
88 (void) tcsetattr(fileno(stdin), TCSADRAIN, &termios);
89}

91#ifdef SUSPEND /* implies BSD */
92#include <signal.h>
93int
94dosuspend(void)
95{
96#ifdef SIGTSTP
97 if (signal(SIGTSTP, SIG_IGN) == SIG_DFL) {
98 settty(NULL);
99 (void) signal(SIGTSTP, SIG_DFL);
100 (void) kill(0, SIGTSTP);
101 gettty();
102 setftty();
103 docrt();
104 } else {
105 pline("I don't think your shell has job control.");
106 }
107#else /* SIGTSTP */
108 pline("Sorry, it seems we have no SIGTSTP here. Try ! or S.");
109#endif /* SIGTSTP */
110 return (0);
111}
112#endif /* SUSPEND */
