72void
73setuwep(struct obj *obj)
74{
75 setworn(obj, W_WEP);
76}

78int
79dowield(void)
80{
81 struct obj *wep;
82 int res = 0;
83
84 multi = 0;
85 if (!(wep = getobj("#-)", "wield"))) /* nothing */
86 ;
87 else if (uwep == wep)
88 pline("You are already wielding that!");
89 else if (uwep && uwep->cursed)
90 pline("The %s welded to your hand!",
91 aobjnam(uwep, "are"));
92 else if (wep == &zeroobj) {
93 if (uwep == 0) {
94 pline("You are already empty handed.");
95 } else {
96 setuwep((struct obj *) 0);
97 res++;
98 pline("You are empty handed.");
99 }
100 } else if (uarms && wep->otyp == TWO_HANDED_SWORD)
101 pline("You cannot wield a two-handed sword and wear a shield.");
102 else if (wep->owornmask & (W_ARMOR | W_RING))
103 pline("You cannot wield that!");
104 else {
105 setuwep(wep);
106 res++;
107 if (uwep->cursed)
108 pline("The %s %s to your hand!",
109 aobjnam(uwep, "weld"),
110 (uwep->quan == 1) ? "itself" : "themselves"); /* a3 */
111 else
112 prinv(uwep);
113 }
114 return (res);
115}

117void
118corrode_weapon(void)
119{
120 if (!uwep || uwep->olet != WEAPON_SYM)
121 return; /* %% */
122 if (uwep->rustfree)
123 pline("Your %s not affected.", aobjnam(uwep, "are"));
124 else {
125 pline("Your %s!", aobjnam(uwep, "corrode"));
126 uwep->spe--;
127 }
128}

130int
131chwepon(struct obj *otmp, int amount)
132{
133 const char *color = (amount < 0) ? "black" : "green";
134 const char *stime;
135 if (!uwep || uwep->olet != WEAPON_SYM) {
136 strange_feeling(otmp,
137 (amount > 0) ? "Your hands twitch."
138 : "Your hands itch.");
139 return (0);
140 }
141 if (uwep->otyp == WORM_TOOTH && amount > 0) {
142 uwep->otyp = CRYSKNIFE;
143 pline("Your weapon seems sharper now.");
144 uwep->cursed = 0;
145 return (1);
146 }
147 if (uwep->otyp == CRYSKNIFE && amount < 0) {
148 uwep->otyp = WORM_TOOTH;
149 pline("Your weapon looks duller now.");
150 return (1);
151 }
152 /* there is a (soft) upper limit to uwep->spe */
153 if (amount > 0 && uwep->spe > 5 && rn2(3)) {
154 pline("Your %s violently green for a while and then evaporate%s.",
155 aobjnam(uwep, "glow"), plur(uwep->quan));
156 while (uwep) /* let all of them disappear */
157 /* note: uwep->quan = 1 is nogood if unpaid */
158 useup(uwep);
159 return (1);
160 }
161 if (!rn2(6))
162 amount *= 2;
163 stime = (amount * amount == 1) ? "moment" : "while";
164 pline("Your %s %s for a %s.",
165 aobjnam(uwep, "glow"), color, stime);
166 uwep->spe += amount;
167 if (amount > 0)
168 uwep->cursed = 0;
169 return (1);
170}
