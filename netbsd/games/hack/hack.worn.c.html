72static struct worn {
73 long w_mask;
74 struct obj **w_obj;
75} worn[] = {
76 {
77 W_ARM, &uarm
78 },
79 {
80 W_ARM2, &uarm2
81 },
82 {
83 W_ARMH, &uarmh
84 },
85 {
86 W_ARMS, &uarms
87 },
88 {
89 W_ARMG, &uarmg
90 },
91 {
92 W_RINGL, &uleft
93 },
94 {
95 W_RINGR, &uright
96 },
97 {
98 W_WEP, &uwep
99 },
100 {
101 W_BALL, &uball
102 },
103 {
104 W_CHAIN, &uchain
105 },
106 {
107 0, 0
108 }
109};
110
111void
112setworn(struct obj *obj, long mask)
113{
114 struct worn *wp;
115 struct obj *oobj;
116
117 for (wp = worn; wp->w_mask; wp++)
118 if (wp->w_mask & mask) {
119 oobj = *(wp->w_obj);
120 if (oobj && !(oobj->owornmask & wp->w_mask))
121 impossible("Setworn: mask = %ld.", wp->w_mask);
122 if (oobj)
123 oobj->owornmask &= ~wp->w_mask;
124 if (obj && oobj && wp->w_mask == W_ARM) {
125 if (uarm2) {
126 impossible("Setworn: uarm2 set?");
127 } else
128 setworn(uarm, W_ARM2);
129 }
130 *(wp->w_obj) = obj;
131 if (obj)
132 obj->owornmask |= wp->w_mask;
133 }
134 if (uarm2 && !uarm) {
135 uarm = uarm2;
136 uarm2 = 0;
137 uarm->owornmask ^= (W_ARM | W_ARM2);
138 }
139}
140
141/* called e.g. when obj is destroyed */
142void
143setnotworn(struct obj *obj)
144{
145 struct worn *wp;
146
147 for (wp = worn; wp->w_mask; wp++)
148 if (obj == *(wp->w_obj)) {
149 *(wp->w_obj) = 0;
150 obj->owornmask &= ~wp->w_mask;
151 }
152 if (uarm2 && !uarm) {
153 uarm = uarm2;
154 uarm2 = 0;
155 uarm->owornmask ^= (W_ARM | W_ARM2);
156 }
157}
