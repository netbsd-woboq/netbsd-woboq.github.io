69#include <stdlib.h>
70#include "hack.h"
71#include "extern.h"
72
73#define RND(x) ((random()>>3) % x)
74
75int
76rn1(int x, int y)
77{
78 return (RND(x) + y);
79}
80
81int
82rn2(int x)
83{
84 return (RND(x));
85}
86
87int
88rnd(int x)
89{
90 return (RND(x) + 1);
91}
92
93int
94d(int n, int x)
95{
96 int tmp = n;
97
98 while (n--)
99 tmp += RND(x);
100 return (tmp);
101}
