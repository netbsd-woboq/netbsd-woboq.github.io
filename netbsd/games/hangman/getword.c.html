43/*
44 * getword:
45 * Get a valid word out of the dictionary file
46 */
47void
48getword(void)
49{
50 FILE *inf;
51 char *wp, *gp;
52 long pos;
53
54 inf = Dict;
55 for (;;) {
56 pos = (double) rand() / (RAND_MAX + 1.0) * (double) Dict_size;
57 fseek(inf, pos, SEEK_SET);
58 if (fgets(Word, BUFSIZ, inf) == NULL)
59 continue;
60 if (fgets(Word, BUFSIZ, inf) == NULL)
61 continue;
62 Word[strlen(Word) - 1] = '\0';
63 if (strlen(Word) < Minlen)
64 continue;
65 for (wp = Word; *wp; wp++)
66 if (!islower((unsigned char)*wp))
67 goto cont;
68 break;
69cont: ;
70 }
71 gp = Known;
72 wp = Word;
73 while (*wp) {
74 *gp++ = '-';
75 wp++;
76 }
77 *gp = '\0';
78}
