49/*
50 * This game written by Ken Arnold.
51 */
52int
53main(int argc, char *argv[])
54{
55 int ch;
56
57 /* Revoke setgid privileges */
58 setgid(getgid());
59
60 while ((ch = getopt(argc, argv, "d:m:")) != -1) {
61 switch (ch) {
62 case 'd':
63 Dict_name = optarg;
64 break;
65 case 'm':
66 Minlen = atoi(optarg);
67 if (Minlen < 2)
68 errx(1, "minimum word length too short");
69 break;
70 case '?':
71 default:
72 (void)fprintf(stderr, "usage: hangman [-d wordlist] [-m minlen]\n");
73 exit(1);
74 }
75 }
76
77 if (!initscr())
78 errx(0, "couldn't initialize screen");
79 signal(SIGINT, die);
80 setup();
81 for (;;) {
82 Wordnum++;
83 playgame();
84 Average = (Average * (Wordnum - 1) + Errors) / Wordnum;
85 }
86 /* NOTREACHED */
87}
88/*
89 * die:
90 * Die properly.
91 */
92void
93die(int dummy __unused)
94{
95 mvcur(0, COLS - 1, LINES - 1, 0);
96 endwin();
97 putchar('\n');
98 exit(0);
99}
