45/*
46 * setup:
47 * Set up the strings on the screen.
48 */
49void
50setup(void)
51{
52 const char *const *sp;
53 static struct stat sbuf;
54
55 noecho();
56 cbreak();
57
58 mvaddstr(PROMPTY, PROMPTX, "Guess:");
59 mvaddstr(GUESSY, GUESSX, "Guessed:");
60 mvaddstr(NUMBERY, NUMBERX, "Word #:");
61 mvaddstr(AVGY, AVGX, "Current Average:");
62 mvaddstr(AVGY + 1, AVGX, "Overall Average:");
63 mvaddstr(KNOWNY, KNOWNX, "Word: ");
64
65 for (sp = Noose_pict; *sp != NULL; sp++) {
66 move(sp - Noose_pict, 0);
67 addstr(*sp);
68 }
69
70 srand(time(NULL) + getpid());
71 if ((Dict = fopen(Dict_name, "r")) == NULL) {
72 endwin();
73 err(1, "fopen %s", Dict_name);
74 }
75 fstat(fileno(Dict), &sbuf);
76 Dict_size = sbuf.st_size;
77}
