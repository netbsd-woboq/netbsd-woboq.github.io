60static struct sockaddr_in *daemons;
61static unsigned int numdaemons, maxdaemons;
62
63static struct sockaddr_in *broadcastaddrs;
64static int numbroadcastaddrs;
65
66static bool initial = true;
67static struct in_addr local_address;
68static const char *explicit_host;
69static uint16_t port;
70
71void
72serverlist_setup(const char *explicit_host_arg, uint16_t port_arg)
73{
74 char local_name[MAXHOSTNAMELEN + 1];
75 struct hostent *hp;
76
77 if (gethostname(local_name, sizeof(local_name)) < 0) {
78 leavex(1, "Sorry, I have no hostname.");
79 }
80 local_name[sizeof(local_name) - 1] = '\0';
81 if ((hp = gethostbyname(local_name)) == NULL) {
82 leavex(1, "Can't find myself.");
83 }
84 memcpy(&local_address, hp->h_addr, sizeof(local_address));
85
86 numdaemons = 0;
87 maxdaemons = 20;
88 daemons = malloc(maxdaemons * sizeof(daemons[0]));
89 if (daemons == NULL) {
90 leavex(1, "Out of memory.");
91 }
92
93 if (explicit_host_arg) {
94 explicit_host = explicit_host_arg;
95 }
96 port = port_arg;
97}

99static void
100add_daemon_addr(const struct sockaddr_storage *addr, socklen_t addrlen,
101 uint16_t port_num)
102{
103 const struct sockaddr_in *sin;
104
105 if (addr->ss_family != AF_INET) {
106 return;
107 }
108 assert(addrlen == sizeof(struct sockaddr_in));
109 sin = (const struct sockaddr_in *)addr;
110
111 assert(numdaemons <= maxdaemons);
112 if (numdaemons == maxdaemons) {
113 maxdaemons += 20;
114 daemons = realloc(daemons, maxdaemons * sizeof(daemons[0]));
115 if (daemons == NULL) {
116 leave(1, "realloc");
117 }
118 }
119
120 /*
121 * Note that we do *not* convert from network to host
122 * order since the port number we were sent *should*
123 * already be in network order.
124 *
125 * The result may be either the port number for the hunt
126 * socket or the port number for the stats socket... or the
127 * number of players connected and not a port number at all,
128 * depending on the packet type.
129 *
130 * For now at least it is ok to stuff it in here, because the
131 * usage of the various different packet types and the
132 * persistence of the results vs. the program exiting does not
133 * cause us to get confused. If the game is made more
134 * self-contained in the future we'll need to be more careful
135 * about this, especially if we make the caching of results
136 * less scattershot.
137 */
138 daemons[numdaemons] = *sin;
139 daemons[numdaemons].sin_port = port_num;
140 numdaemons++;
141}

143static bool
144have_daemon_addr(const struct sockaddr_storage *addr, socklen_t addrlen)
145{
146 unsigned j;
147 const struct sockaddr_in *sin;
148
149 if (addr->ss_family != AF_INET) {
150 return false;
151 }
152 assert(addrlen == sizeof(struct sockaddr_in));
153 sin = (const struct sockaddr_in *)addr;
154
155 for (j = 0; j < numdaemons; j++) {
156 if (sin->sin_addr.s_addr == daemons[j].sin_addr.s_addr) {
157 return true;
158 }
159 }
160 return false;
161}

163static void
164getbroadcastaddrs(void)
165{
166 unsigned num, i;
167 struct ifaddrs *ifp, *ip;
168
169 broadcastaddrs = NULL;
170 numbroadcastaddrs = 0;
171
172 if (getifaddrs(&ifp) < 0) {
173 return;
174 }
175
176 num = 0;
177 for (ip = ifp; ip; ip = ip->ifa_next) {
178 if ((ip->ifa_addr->sa_family == AF_INET) &&
179 (ip->ifa_flags & IFF_BROADCAST)) {
180 num++;
181 }
182 }
183
184 broadcastaddrs = malloc(num * sizeof(broadcastaddrs[0]));
185 if (broadcastaddrs == NULL) {
186 leavex(1, "Out of memory");
187 }
188
189 i = 0;
190 for (ip = ifp; ip; ip = ip->ifa_next) {
191 if ((ip->ifa_addr->sa_family == AF_INET) &&
192 (ip->ifa_flags & IFF_BROADCAST)) {
193 memcpy(&broadcastaddrs[i], ip->ifa_broadaddr,
194 sizeof(broadcastaddrs[i]));
195 i++;
196 }
197 }
198 assert(i == num);
199 numbroadcastaddrs = num;
200
201 freeifaddrs(ifp);
202}

204static void
205send_messages(int contactsock, unsigned short msg)
206{
207 struct sockaddr_in contactaddr;
208 struct hostent *hp;
209 uint16_t wiremsg;
210 int option;
211 int i;
212
213 memset(&contactaddr, 0, sizeof(contactaddr));
214 contactaddr.sin_family = SOCK_FAMILY;
215 contactaddr.sin_port = htons(port);
216
217 if (explicit_host != NULL) { /* explicit host given */
218 hp = gethostbyname(explicit_host);
219 if (hp == NULL) {
220 leavex(1, "%s: Unknown host", explicit_host);
221 }
222 memcpy(&contactaddr.sin_addr, hp->h_addr,
223 sizeof(contactaddr.sin_addr));
224 wiremsg = htons(msg);
225 (void) sendto(contactsock, &wiremsg, sizeof(wiremsg), 0,
226 (struct sockaddr *)&contactaddr,
227 sizeof(contactaddr));
228 return;
229 }
230
231 if (!initial) {
232 /* favor host of previous session by contacting it first */
233 contactaddr.sin_addr = Daemon.sin_addr;
234
235 /* Must be playing! */
236 assert(msg == C_PLAYER);
237 wiremsg = htons(msg);
238
239 (void) sendto(contactsock, &wiremsg, sizeof(wiremsg), 0,
240 (struct sockaddr *)&contactaddr, sizeof(contactaddr));
241 }
242
243 if (initial) {
244 getbroadcastaddrs();
245 }
246
247#ifdef SO_BROADCAST
248 /* Sun's will broadcast even though this option can't be set */
249 option = 1;
250 if (setsockopt(contactsock, SOL_SOCKET, SO_BROADCAST,
251 &option, sizeof option) < 0) {
252 leave(1, "setsockopt broadcast");
253 /* NOTREACHED */
254 }
255#endif
256
257 /* send broadcast packets on all interfaces */
258 wiremsg = htons(msg);
259 for (i = 0; i < numbroadcastaddrs; i++) {
260 contactaddr.sin_addr = broadcastaddrs[i].sin_addr;
261 if (sendto(contactsock, &wiremsg, sizeof(wiremsg), 0,
262 (struct sockaddr *)&contactaddr,
263 sizeof(contactaddr)) < 0) {
264 leave(1, "sendto");
265 }
266 }
267 contactaddr.sin_addr = local_address;
268 if (sendto(contactsock, &wiremsg, sizeof(wiremsg), 0,
269 (struct sockaddr *)&contactaddr, sizeof(contactaddr)) < 0) {
270 leave(1, "sendto");
271 }
272}

274static void
275get_responses(int contactsock)
276{
277 struct pollfd set[1];
278 struct sockaddr_storage addr;
279 socklen_t addrlen;
280 int r;
281 uint16_t port_num;
282 ssize_t portlen;
283
284 /* forget all old responses */
285 numdaemons = 0;
286
287 errno = 0;
288 set[0].fd = contactsock;
289 set[0].events = POLLIN;
290 for (;;) {
291 r = poll(set, 1, 1000);
292 if (r < 0) {
293 if (errno == EINTR) {
294 continue;
295 }
296 leave(1, "poll");
297 }
298 if (r == 0) {
299 break;
300 }
301
302 addrlen = sizeof(addr);
303 portlen = recvfrom(contactsock, &port_num, sizeof(port_num), 0,
304 (struct sockaddr *)&addr, &addrlen);
305 if (portlen < 0) {
306 if (errno == EINTR) {
307 continue;
308 }
309 leave(1, "recvfrom");
310 }
311 if (portlen == 0) {
312 leavex(1, "recvfrom: Unexpected EOF");
313 }
314 if ((size_t)portlen != sizeof(port_num)) {
315 /* trash, ignore it */
316 continue;
317 }
318 if (have_daemon_addr(&addr, addrlen)) {
319 /* this shouldn't happen */
320 continue;
321 }
322
323 add_daemon_addr(&addr, addrlen, port_num);
324 }
325
326 initial = false;
327}

329void
330serverlist_query(unsigned short msg)
331{
332 int contactsock;
333
334 if (!initial && explicit_host != NULL) {
335 /* already did the work, no point doing it again */
336 return;
337 }
338
339 contactsock = socket(SOCK_FAMILY, SOCK_DGRAM, 0);
340 if (contactsock < 0) {
341 leave(1, "socket system call failed");
342 }
343
344 send_messages(contactsock, msg);
345 get_responses(contactsock);
346
347 (void) close(contactsock);
348}

356const struct sockaddr_storage *
357serverlist_gethost(unsigned i, socklen_t *len_ret)
358{
359 struct sockaddr_in *ret;
360
361 assert(i < numdaemons);
362 ret = &daemons[i];
363 *len_ret = sizeof(*ret);
364 return (struct sockaddr_storage *)ret;
365}
