39__RCSID("$NetBSD: support.c,v 1.2 2014/03/29 22:30:27 dholland Exp $");
40
41void
42complain(int level, const char *fmt, ...)
43{
44 va_list ap;
45#ifdef LOG
46 char buf[4096];
47 int serrno = errno;
48#endif
49
50 va_start(ap, fmt);
51#ifdef LOG
52 vsnprintf(buf, sizeof(buf), fmt, ap);
53 snprintf(buf, sizeof(buf) - strlen(buf), ": %s", strerror(serrno));
54 syslog(level, "%s", buf);
55#else
56 (void)level;
57 vwarn(fmt, ap);
58#endif
59 va_end(ap);
60}
