47#define COMP_STRT 20
48#define CARD_STRT 2
49
50static void show_card(int, int, CARD, CARD *);
51static void show_score(int, int, int, int *);
52
53void
54prboard(void)
55{
56 PLAY *pp;
57 int i, j, k, temp;
58
59 for (k = 0; k < 2; k++) {
60 pp = &Player[k];
61 temp = k * COMP_STRT + CARD_STRT;
62 for (i = 0; i < NUM_SAFE; i++)
63 if (pp->safety[i] == S_PLAYED && !pp->sh_safety[i]) {
64 mvaddstr(i, temp, C_name[i + S_CONV]);
65 if (pp->coups[i])
66 mvaddch(i, temp - CARD_STRT, '*');
67 pp->sh_safety[i] = TRUE;
68 }
69 show_card(14, temp, pp->battle, &pp->sh_battle);
70 show_card(16, temp, pp->speed, &pp->sh_speed);
71 for (i = C_25; i <= C_200; i++) {
72 const char *name;
73 int end;
74
75 if (pp->nummiles[i] == pp->sh_nummiles[i])
76 continue;
77
78 name = C_name[i];
79 temp = k * 40;
80 end = pp->nummiles[i];
81 for (j = pp->sh_nummiles[i]; j < end; j++)
82 mvwaddstr(Miles, i + 1, (j << 2) + temp, name);
83 pp->sh_nummiles[i] = end;
84 }
85 }
86 prscore(TRUE);
87 temp = CARD_STRT;
88 pp = &Player[PLAYER];
89 for (i = 0; i < HAND_SZ; i++)
90 show_card(i + 6, temp, pp->hand[i], &pp->sh_hand[i]);
91 mvprintw(6, COMP_STRT + CARD_STRT, "%2ld", (long)(Topcard - Deck));
92 show_card(8, COMP_STRT + CARD_STRT, Discard, &Sh_discard);
93 if (End == 1000) {
94 move(EXT_Y, EXT_X);
95 standout();
96 addstr("Extension");
97 standend();
98 }
99 wrefresh(Board);
100 wrefresh(Miles);
101 wrefresh(Score);
102}

104/*
105 * show_card:
106 * Show the given card if it is different from the last one shown
107 */
108static void
109show_card(int y, int x, CARD c, CARD *lc)
110{
111 if (c == *lc)
112 return;
113
114 mvprintw(y, x, C_fmt, C_name[c]);
115 *lc = c;
116}

118static const char Score_fmt[] = "%4d";
119
120void
121prscore(bool for_real)
122{
123 PLAY *pp;
124 int x;
125
126 stdscr = Score;
127 for (pp = Player; pp < &Player[2]; pp++) {
128 x = (pp - Player) * 6 + 21;
129 show_score(1, x, pp->mileage, &pp->sh_mileage);
130 if (pp->safescore != pp->sh_safescore) {
131 mvprintw(2, x, Score_fmt, pp->safescore);
132 if (pp->safescore == 400)
133 mvaddstr(3, x + 1, "300");
134 else
135 mvaddstr(3, x + 1, " 0");
136 mvprintw(4, x, Score_fmt, pp->coupscore);
137 pp->sh_safescore = pp->safescore;
138 }
139 if (Window == W_FULL || Finished) {
140#ifdef EXTRAP
141 if (for_real)
142 finalscore(pp);
143 else
144 extrapolate(pp);
145#else
146 finalscore(pp);
147#endif
148 show_score(11, x, pp->hand_tot, &pp->sh_hand_tot);
149 show_score(13, x, pp->total, &pp->sh_total);
150 show_score(14, x, pp->games, &pp->sh_games);
151 }
152 else {
153 show_score(6, x, pp->hand_tot, &pp->sh_hand_tot);
154 show_score(8, x, pp->total, &pp->sh_total);
155 show_score(9, x, pp->games, &pp->sh_games);
156 }
157 }
158 stdscr = Board;
159}

161/*
162 * show_score:
163 * Show a score value if it is different from the last time we
164 * showed it.
165 */
166static void
167show_score(int y, int x, int s, int *ls)
168{
169 if (s == *ls)
170 return;
171
172 mvprintw(y, x, Score_fmt, s);
173 *ls = s;
174}
