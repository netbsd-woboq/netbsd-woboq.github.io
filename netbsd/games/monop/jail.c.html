43/*
44 * This routine uses a get-out-of-jail-free card to get the
45 * player out of jail.
46 */
47void
48card(void)
49{
50 if (cur_p->loc != JAIL) {
51 printf("But you're not IN Jail\n");
52 return;
53 }
54 if (cur_p->num_gojf == 0) {
55 printf("But you don't HAVE a get out of jail free card\n");
56 return;
57 }
58 ret_card(cur_p);
59 cur_p->loc = 10; /* just visiting */
60 cur_p->in_jail = 0;
61}
62
63/*
64 * This routine deals with paying your way out of jail.
65 */
66void
67pay(void)
68{
69 if (cur_p->loc != JAIL) {
70 printf("But you're not IN Jail\n");
71 return;
72 }
73 cur_p->loc = 10;
74 cur_p->money -= 50;
75 cur_p->in_jail = 0;
76 printf("That cost you $50\n");
77}
78
79/*
80 * This routine deals with a move in jail
81 */
82int
83move_jail(int r1, int r2)
84{
85 if (r1 != r2) {
86 printf("Sorry, that doesn't get you out\n");
87 if (++(cur_p->in_jail) == 3) {
88 printf("It's your third turn and you didn't roll "
89 "doubles. You have to pay $50\n");
90 cur_p->money -= 50;
91moveit:
92 cur_p->loc = 10;
93 cur_p->in_jail = 0;
94 move(r1+r2);
95 r1 = r2 - 1; /* kludge: stop new roll w/doub */
96 return TRUE;
97 }
98 return FALSE;
99 } else {
100 printf("Double roll gets you out.\n");
101 goto moveit;
102 }
103}

105void
106printturn(void)
107{
108 if (cur_p->loc != JAIL)
109 return;
110 printf("(This is your ");
111 switch (cur_p->in_jail) {
112 case 0:
113 printf("1st");
114 break;
115 case 1:
116 printf("2nd");
117 break;
118 case 2:
119 printf("3rd (and final)");
120 break;
121 }
122 printf(" turn in JAIL)\n");
123}
