55int main(int, char *[]);
56static void pigout(char *, int);
57static void usage(void) __dead;
58
59int
60main(int argc, char *argv[])
61{
62 int len;
63 int ch;
64 char buf[1024];
65
66 while ((ch = getopt(argc, argv, "")) != -1)
67 switch(ch) {
68 case '?':
69 default:
70 usage();
71 }
72 argc -= optind;
73 argv += optind;
74
75 for (len = 0; (ch = getchar()) != EOF;) {
76 if (isalpha(ch)) {
77 if ((size_t)len >= sizeof(buf))
78 errx(1, "ate too much!");
79 buf[len++] = ch;
80 continue;
81 }
82 if (len != 0) {
83 pigout(buf, len);
84 len = 0;
85 }
86 (void)putchar(ch);
87 }
88 exit(0);
89}
90
91static void
92pigout(char *buf, int len)
93{
94 int ch, start, i;
95 int olen, allupper, firstupper;
96
97 /* See if the word is all upper case */
98 allupper = firstupper = isupper((unsigned char)buf[0]);
99 for (i = 1; i < len && allupper; i++)
100 allupper = allupper && isupper((unsigned char)buf[i]);
101
102 /*
103 * If the word starts with a vowel, append "way". Don't treat 'y'
104 * as a vowel if it appears first.
105 */
106 if (strchr("aeiouAEIOU", buf[0]) != NULL) {
107 (void)printf("%.*s%s", len, buf,
108 allupper ? "WAY" : "way");
109 return;
110 }
111
112 /*
113 * Copy leading consonants to the end of the word. The unit "qu"
114 * isn't treated as a vowel.
115 */
116 if (!allupper)
117 buf[0] = tolower((unsigned char)buf[0]);
118 for (start = 0, olen = len;
119 !strchr("aeiouyAEIOUY", buf[start]) && start < olen;) {
120 ch = buf[len++] = buf[start++];
121 if ((ch == 'q' || ch == 'Q') && start < olen &&
122 (buf[start] == 'u' || buf[start] == 'U'))
123 buf[len++] = buf[start++];
124 }
125 if (firstupper)
126 buf[start] = toupper((unsigned char)buf[start]);
127 (void)printf("%.*s%s", olen, buf + start, allupper ? "AY" : "ay");
128}
