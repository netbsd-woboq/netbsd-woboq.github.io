57__dead static void
58usage(void)
59{
60 extern char *__progname;
61 fprintf(stderr, "usage: %s [-d] [string ...]\n", __progname);
62 exit(1);
63}

65int
66main(int argc, char **argv)
67{
68 char *p, buf[132];
69 int c, start, neednl, dflag;
70
71 /* Revoke setgid privileges */
72 setgid(getgid());
73
74 dflag = 0;
75 while ((c = getopt(argc, argv, "dh")) != -1)
76 switch(c) {
77 case 'd':
78 dflag = 1;
79 break;
80 case 'h':
81 case '?':
82 default:
83 usage();
84 }
85 argc -= optind;
86 argv += optind;
87
88 if (dflag) {
89 if (argc > 0)
90 usage();
91
92 start = 0;
93 neednl = 0;
94 while (fgets(buf, sizeof(buf), stdin) != NULL) {
95 c = getppt(buf);
96 if (c < 0) {
97 if (start) {
98 /* lost sync? */
99 if (neednl)
100 putchar('\n');
101 exit(0);
102 } else
103 continue;
104 }
105 start = 1;
106 putchar(c);
107 neednl = (c != '\n');
108 }
109 if (!feof(stdin))
110 err(1, "fgets");
111 if (neednl)
112 putchar('\n');
113 } else {
114 (void) puts(EDGE);
115 if (argc > 0)
116 while ((p = *argv++)) {
117 for (; *p; ++p)
118 putppt((int)*p);
119 if ((*(argv)))
120 putppt((int)' ');
121 }
122 else while ((c = getchar()) != EOF)
123 putppt(c);
124 (void) puts(EDGE);
125 }
126 exit(0);
127}

129static void
130putppt(int c)
131{
132 int i;
133
134 (void) putchar('|');
135 for (i = 7; i >= 0; i--) {
136 if (i == 2)
137 (void) putchar('.'); /* feed hole */
138 if ((c&(1<<i)) != 0)
139 (void) putchar('o');
140 else
141 (void) putchar(' ');
142 }
143 (void) putchar('|');
144 (void) putchar('\n');
145}
146
147static int
148getppt(const char *buf)
149{
150 const char *p = strchr(buf, '.');
151 int c;
152
153 if (p == NULL)
154 return (-1);
155
156 c = 0;
157 if (p[ 3] != ' ')
158 c |= 0001;
159 if (p[ 2] != ' ')
160 c |= 0002;
161 if (p[ 1] != ' ')
162 c |= 0004;
163 if (p[-1] != ' ')
164 c |= 0010;
165 if (p[-2] != ' ')
166 c |= 0020;
167 if (p[-3] != ' ')
168 c |= 0040;
169 if (p[-4] != ' ')
170 c |= 0100;
171 if (p[-5] != ' ')
172 c |= 0200;
173
174 return (c);
175}
