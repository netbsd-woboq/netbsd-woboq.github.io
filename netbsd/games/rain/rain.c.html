61static volatile sig_atomic_t sig_caught = 0;
62
63int main(int, char **);
64static void onsig(int);
65
66
67int
68main(int argc, char **argv)
69{
70 int x, y, j;
71 long cols, lines;
72 unsigned int delay = 0;
73 unsigned long val = 0;
74 int ch;
75 char *ep;
76 int xpos[5], ypos[5];
77
78 while ((ch = getopt(argc, argv, "d:")) != -1)
79 switch (ch) {
80 case 'd':
81 val = strtoul(optarg, &ep, 0);
82 if (ep == optarg || *ep)
83 errx(1, "Invalid delay `%s'", optarg);
84 if (errno == ERANGE && val == ULONG_MAX)
85 err(1, "Invalid delay `%s'", optarg);
86 if (val >= 1000)
87 errx(1, "Invalid delay `%s' (1-999)", optarg);
88 delay = (unsigned int)val * 1000; /* ms -> us */
89 break;
90 default:
91 (void)fprintf(stderr, "Usage: %s [-d delay]\n",
92 getprogname());
93 return 1;
94 }
95
96 if (!initscr())
97 errx(0, "couldn't initialize screen");
98 cols = COLS - 4;
99 lines = LINES - 4;
100
101 (void)signal(SIGHUP, onsig);
102 (void)signal(SIGINT, onsig);
103 (void)signal(SIGTERM, onsig);
104
105 (void)curs_set(0);
106 for (j = 4; j >= 0; --j) {
107 xpos[j] = random() % cols + 2;
108 ypos[j] = random() % lines + 2;
109 }
110 for (j = 0;;) {
111 if (sig_caught) {
112 (void)endwin();
113 exit(0);
114 }
115 x = random() % cols + 2;
116 y = random() % lines + 2;
117 (void)mvaddch(y, x, '.');
118 (void)mvaddch(ypos[j], xpos[j], 'o');
119 if (!j--)
120 j = 4;
121 (void)mvaddch(ypos[j], xpos[j], 'O');
122 if (!j--)
123 j = 4;
124 (void)mvaddch(ypos[j] - 1, xpos[j], '-');
125 (void)mvaddstr(ypos[j], xpos[j] - 1, "|.|");
126 (void)mvaddch(ypos[j] + 1, xpos[j], '-');
127 if (!j--)
128 j = 4;
129 (void)mvaddch(ypos[j] - 2, xpos[j], '-');
130 (void)mvaddstr(ypos[j] - 1, xpos[j] - 1, "/ \\");
131 (void)mvaddstr(ypos[j], xpos[j] - 2, "| O |");
132 (void)mvaddstr(ypos[j] + 1, xpos[j] - 1, "\\ /");
133 (void)mvaddch(ypos[j] + 2, xpos[j], '-');
134 if (!j--)
135 j = 4;
136 (void)mvaddch(ypos[j] - 2, xpos[j], ' ');
137 (void)mvaddstr(ypos[j] - 1, xpos[j] - 1, " ");
138 (void)mvaddstr(ypos[j], xpos[j] - 2, " ");
139 (void)mvaddstr(ypos[j] + 1, xpos[j] - 1, " ");
140 (void)mvaddch(ypos[j] + 2, xpos[j], ' ');
141 xpos[j] = x;
142 ypos[j] = y;
143 (void)refresh();
144 if (delay) (void)usleep(delay);
145 }
146}
147
148/* ARGSUSED */
149static void
150onsig(int dummy __unused)
151{
152 sig_caught = 1;
153}
