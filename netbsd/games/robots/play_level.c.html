45/*
46 * play_level:
47 * Let the player play the current level
48 */
49void
50play_level(void)
51{
52 COORD *cp;
53
54 move(My_pos.y, My_pos.x);
55 addch(PLAYER);
56 refresh();
57 for (cp = Robots; cp < &Robots[MAXROBOTS]; cp++) {
58 if (cp->y < 0)
59 continue;
60 move(cp->y, cp->x);
61 addch(ROBOT);
62 }
63 refresh();
64#ifdef DEBUG
65 standout();
66 move(Min.y, Min.x);
67 addch(inch());
68 move(Max.y, Max.x);
69 addch(inch());
70 standend();
71#endif /* DEBUG */
72 setjmp(End_move);
73 flush_in();
74 while (!Dead && Num_robots > 0) {
75 move(My_pos.y, My_pos.x);
76 if (!jumping())
77 refresh();
78 get_move();
79 if (Real_time)
80 alarm(0);
81 if (Field[My_pos.y][My_pos.x] != 0)
82 Dead = true;
83 if (!Dead)
84 move_robots(false);
85 if (Was_bonus) {
86 move(Y_PROMPT, X_PROMPT);
87 clrtoeol();
88 move(Y_PROMPT + 1, X_PROMPT);
89 clrtoeol();
90 Was_bonus = false;
91 }
92 }
93
94 /*
95 * if the player didn't die, add on the possible bonuses
96 */
97
98 if (!Dead) {
99 Was_bonus = false;
100
101 if (Level == Start_level && Start_level > 1) {
102 move(Y_PROMPT, X_PROMPT);
103 printw("Advance bonus: %d", S_BONUS);
104 refresh();
105 add_score(S_BONUS);
106 Was_bonus = true;
107 }
108
109 if (Wait_bonus != 0) {
110 if (!Was_bonus)
111 move(Y_PROMPT, X_PROMPT);
112 else
113 move(Y_PROMPT + 1, X_PROMPT);
114 printw("Wait bonus: %d", Wait_bonus);
115 refresh();
116 add_score(Wait_bonus);
117 Was_bonus = true;
118 }
119 }
120}
