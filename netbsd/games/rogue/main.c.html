63int
64main(int argc, char *argv[])
65{
66 if (init(argc, argv)) { /* restored game */
67 goto PL;
68 }
69
70 for (;;) {
71 clear_level();
72 make_level();
73 put_objects();
74 put_stairs();
75 add_traps();
76 put_mons();
77 put_player(party_room);
78 print_stats(STAT_ALL);
79PL:
80 play_level();
81 free_stuff(&level_objects);
82 free_stuff(&level_monsters);
83 }
84}
