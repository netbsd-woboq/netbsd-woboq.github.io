68static long *fptr = &rntb[4];
69static long *rptr = &rntb[1];
70static long *state = &rntb[1];
71static int rand_type = 3;
72static int rand_deg = 31;
73static int rand_sep = 3;
74static long *end_ptr = &rntb[32];
75
76static long rrandom(void);
77
78void
79srrandom(int x)
80{
81 int i;
82
83 state[0] = x;
84 if (rand_type != 0) {
85 for (i = 1; i < rand_deg; i++) {
86 state[i] = 1103515245 * state[i - 1] + 12345;
87 }
88 fptr = &state[rand_sep];
89 rptr = &state[0];
90 for (i = 0; i < 10 * rand_deg; i++) {
91 (void)rrandom();
92 }
93 }
94}
95
96static long
97rrandom(void)
98{
99 long i;
100
101 if (rand_type == 0) {
102 i = state[0] = (state[0]*1103515245 + 12345) & 0x7fffffff;
103 } else {
104 *fptr += *rptr;
105 i = (*fptr >> 1) & 0x7fffffff;
106 if (++fptr >= end_ptr) {
107 fptr = state;
108 ++rptr;
109 } else {
110 if (++rptr >= end_ptr) {
111 rptr = state;
112 }
113 }
114 }
115 return(i);
116}

118int
119get_rand(int x, int y)
120{
121 int r, t;
122 long lr;
123
124 if (x > y) {
125 t = y;
126 y = x;
127 x = t;
128 }
129 lr = rrandom();
130 lr &= 0x00003fffL;
131 r = (int)lr;
132 r = (r % ((y - x) + 1)) + x;
133 return(r);
134}
135
136int
137rand_percent(int percentage)
138{
139 return(get_rand(1, 100) <= percentage);
140}

142int
143coin_toss(void)
144{
145 return(((rrandom() & 01) ? 1 : 0));
146}
