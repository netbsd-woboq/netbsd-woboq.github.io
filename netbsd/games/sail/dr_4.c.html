44void
45ungrap(struct ship *from, struct ship *to)
46{
47 int k;
48 char friend;
49
50 if ((k = grappled2(from, to)) == 0)
51 return;
52 friend = capship(from)->nationality == capship(to)->nationality;
53 while (--k >= 0) {
54 if (friend || dieroll() < 3) {
55 cleangrapple(from, to, 0);
56 makesignal(from, "ungrappling $$", to);
57 }
58 }
59}
60
61void
62grap(struct ship *from, struct ship *to)
63{
64 if (capship(from)->nationality != capship(to)->nationality &&
65 dieroll() > 2)
66 return;
67 send_grap(from, to->file->index);
68 send_grap(to, from->file->index);
69 makesignal(from, "grappled with $$", to);
70}
