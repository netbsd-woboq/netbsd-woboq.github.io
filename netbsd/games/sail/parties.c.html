44int
45meleeing(struct ship *from, struct ship *to)
46{
47 struct BP *p = from->file->OBP;
48 struct BP *q = p + NBP;
49
50 for (; p < q; p++)
51 if (p->turnsent && p->toship == to)
52 return 1;
53 return 0;
54}
55
56int
57boarding(struct ship *from, int isdefense)
58{
59 struct BP *p = isdefense ? from->file->DBP : from->file->OBP;
60 struct BP *q = p + NBP;
61
62 for (; p < q; p++)
63 if (p->turnsent)
64 return 1;
65 return 0;
66}
67
68void
69unboard(struct ship *ship, struct ship *to, int isdefense)
70{
71 struct BP *p = isdefense ? ship->file->DBP : ship->file->OBP;
72 int n;
73
74 for (n = 0; n < NBP; p++, n++) {
75 if (p->turnsent &&
76 (p->toship == to || isdefense || ship == to)) {
77 if (isdefense) {
78 send_dbp(ship, n, 0, 0, 0);
79 } else {
80 send_obp(ship, n, 0, 0, 0);
81 }
82 }
83 }
