50/*
51 * If we get here before a ship is chosen, then ms == 0 and
52 * we don't want to update the score file, or do any Write's either.
53 * We can assume the sync file is already created and may need
54 * to be removed.
55 * Of course, we don't do any more Sync()'s if we got here
56 * because of a Sync() failure.
57 */
58void
59leave(int conditions)
60{
61 signal(SIGHUP, SIG_IGN);
62 signal(SIGINT, SIG_IGN);
63 signal(SIGQUIT, SIG_IGN);
64 signal(SIGALRM, SIG_IGN);
65 signal(SIGCHLD, SIG_IGN);
66
67 if (done_curses) {
68 Msg("It looks like you've had it!");
69 switch (conditions) {
70 case LEAVE_QUIT:
71 break;
72 case LEAVE_CAPTURED:
73 Msg("Your ship was captured.");
74 break;
75 case LEAVE_HURRICAN:
76 Msg("Hurricane! All ships destroyed.");
77 break;
78 case LEAVE_DRIVER:
79 Msg("The driver died.");
80 break;
81 case LEAVE_SYNC:
82 Msg("Synchronization error.");
83 break;
84 default:
85 Msg("A funny thing happened (%d).", conditions);
86 }
87 } else {
88 switch (conditions) {
89 case LEAVE_QUIT:
90 break;
91 case LEAVE_DRIVER:
92 printf("The driver died.\n");
93 break;
94 case LEAVE_FORK:
95 break;
96 case LEAVE_SYNC:
97 printf("Synchronization error\n.");
98 break;
99 default:
100 printf("A funny thing happened (%d).\n",
101 conditions);
102 }
103 }
104
105 if (ms != 0) {
106 logger(ms);
107 if (conditions != LEAVE_SYNC) {
108 makemsg(ms, "Captain %s relinquishing.",
109 mf->captain);
110 send_end(ms);
111 Sync();
112 }
113 }
114 sync_close(!hasdriver);
115 sleep(5);
116 cleanupscreen();
117 exit(0);
118}
119
120/*ARGSUSED*/
121void
122choke(int n __unused)
123{
124 leave(LEAVE_QUIT);
125}
126
127/*ARGSUSED*/
128void
129child(int n __unused)
130{
131 int status;
132 int pid;
133
134 signal(SIGCHLD, SIG_IGN);
135 do {
136 pid = wait3(&status, WNOHANG, (struct rusage *)0);
137 if (pid < 0 || (pid > 0 && !WIFSTOPPED(status)))
138 hasdriver = 0;
139 } while (pid > 0);
140 signal(SIGCHLD, child);
141}
