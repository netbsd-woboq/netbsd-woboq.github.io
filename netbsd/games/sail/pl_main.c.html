55void
56pl_main_init(void)
57{
58 int nat[NNATION];
59 int n;
60 struct ship *sp;
61
62 if (game < 0 || game >= NSCENE) {
63 errx(1, "Very funny.");
64 }
65 cc = &scene[game];
66 ls = SHIP(cc->vessels);
67
68 for (n = 0; n < NNATION; n++)
69 nat[n] = 0;
70 foreachship(sp) {
71 if (sp->file == NULL &&
72 (sp->file = calloc(1, sizeof (struct File))) == NULL) {
73 err(1, "calloc");
74 }
75 sp->file->index = sp - SHIP(0);
76 sp->file->stern = nat[sp->nationality]++;
77 sp->file->dir = sp->shipdir;
78 sp->file->row = sp->shiprow;
79 sp->file->col = sp->shipcol;
80 }
81 windspeed = cc->windspeed;
82 winddir = cc->winddir;
83
84 signal(SIGHUP, choke);
85 signal(SIGINT, choke);
86 signal(SIGCHLD, child);
87}
88
89void
90pl_main_uninit(void)
91{
92 struct ship *sp;
93
94 foreachship(sp) {
95 free(sp->file);
96 sp->file = NULL;
97 }
98}

100static void
101initialize(void)
102{
103 char captain[MAXNAMESIZE];
104 char message[60];
105
106 if (game < 0 || game >= NSCENE) {
107 errx(1, "Very funny.");
108 }
109
110 send_begin(ms);
111 if (Sync() < 0)
112 leave(LEAVE_SYNC);
113
114#if 0
115 printf("Your ship is the %s, a %d gun %s (%s crew).\n",
116 ms->shipname, mc->guns, classname[mc->class],
117 qualname[mc->qual]);
118#endif
119
120 strlcpy(captain, myname, sizeof(captain));
121 send_captain(ms, captain);
122
123 display_redraw();
124 snprintf(message, sizeof message, "Captain %s assuming command",
125 captain);
126 send_signal(ms, message);
127 newturn(0);
128}
