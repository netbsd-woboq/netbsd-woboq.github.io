55static const char *recfile = _PATH_RAWSCORES;
56#define MAXPLAYERS 256
57
58struct player {
59 short uids;
60 short scores;
61 char *name;
62};
63
64static struct player players[MAXPLAYERS], temp;
65
66int main(void);
67
68int
69main(void)
70{
71 short uid, score;
72 FILE *fd;
73 int noplayers;
74 int i, j, notsorted;
75 short whoallbest, allbest;
76 const char *q;
77 struct passwd *p;
78
79 /* Revoke setgid privileges */
80 setgid(getgid());
81
82 fd = fopen(recfile, "r");
83 if (fd == NULL)
84 err(1, "opening `%s'", recfile);
85 printf("Snake players scores to date\n");
86 if (fread(&whoallbest, sizeof(short), 1, fd) == 0) {
87 printf("No scores recorded yet!\n");
88 exit(0);
89 }
90 fread(&allbest, sizeof(short), 1, fd);
91 noplayers = 0;
92 for (uid = 2; ;uid++) {
93 if(fread(&score, sizeof(short), 1, fd) == 0)
94 break;
95 if (score > 0) {
96 if (noplayers >= MAXPLAYERS) {
97 printf("too many players\n");
98 exit(2);
99 }
100 players[noplayers].uids = uid;
101 players[noplayers].scores = score;
102 p = getpwuid(uid);
103 if (p == NULL)
104 continue;
105 q = p -> pw_name;
106 players[noplayers].name = strdup(q);
107 if (players[noplayers].name == NULL)
108 err(1, NULL);
109 noplayers++;
110 }
111 }
112
113 /* bubble sort scores */
114 for (notsorted = 1; notsorted; ) {
115 notsorted = 0;
116 for (i = 0; i < noplayers - 1; i++)
117 if (players[i].scores < players[i + 1].scores) {
118 temp = players[i];
119 players[i] = players[i + 1];
120 players[i + 1] = temp;
121 notsorted++;
122 }
123 }
124
125 j = 1;
126 for (i = 0; i < noplayers; i++) {
127 printf("%d:\t$%d\t%s\n", j, players[i].scores, players[i].name);
128 if (players[i].scores > players[i + 1].scores)
129 j = i + 2;
130 }
131 exit(0);
132}
