37/*
38 * Tetris scores.
39 */
40
41/* Header for high score file. */
42#define HSH_MAGIC_SIZE 8
43struct highscore_header {
44 char hsh_magic[HSH_MAGIC_SIZE];
45 uint32_t hsh_endiantag;
46 uint32_t hsh_version;
47};
48
49/* Current on-disk high score record. */
50struct highscore_ondisk {
51 char hso_name[20];
52 int32_t hso_score;
53 int32_t hso_level;
54 int32_t hso_pad;
55 int64_t hso_time;
56};
57
58/* 5.99.x after time_t change, on 32-bit machines */
59struct highscore_ondisk_599 {
60 char hso599_name[20];
61 int32_t hso599_score;
62 int32_t hso599_level;
63 int32_t hso599_time[2];
64};
65
66/* 5.0 and earlier on-disk high score record. */
67struct highscore_ondisk_50 {
68 char hso50_name[20];
69 int32_t hso50_score;
70 int32_t hso50_level;
71 int32_t hso50_time;
72};
73
74/* In-memory high score record. */
75struct highscore {
76 char hs_name[20]; /* login name */
77 int hs_score; /* raw score */
78 int hs_level; /* play level */
79 time_t hs_time; /* time at game end */
80};

83#define MAXSCORES 9 /* maximum high score entries per person */
84#define EXPIRATION (5L * 365 * 24 * 60 * 60)
