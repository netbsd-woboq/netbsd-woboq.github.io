45/*
46** Klingon Attack Routine
47**
48** This routine performs the Klingon attack provided that
49** (1) Something happened this move (i.e., not free), and
50** (2) You are not cloaked. Note that if you issue the
51** cloak command, you are not considered cloaked until you
52** expend some time.
53**
54** Klingons are permitted to move both before and after the
55** attack. They will tend to move toward you before the
56** attack and away from you after the attack.
57**
58** Under certain conditions you can get a critical hit. This
59** sort of hit damages devices. The probability that a given
60** device is damaged depends on the device. Well protected
61** devices (such as the computer, which is in the core of the
62** ship and has considerable redundancy) almost never get
63** damaged, whereas devices which are exposed (such as the
64** warp engines) or which are particularly delicate (such as
65** the transporter) have a much higher probability of being
66** damaged.
67**
68** The actual amount of damage (i.e., how long it takes to fix
69** it) depends on the amount of the hit and the "damfac[]"
70** entry for the particular device.
71**
72** Casualties can also occur.
73**
74** resting -- set if attack while resting
75*/
76
77void
78attack(int resting)
79{
80 int hit, i, l;
81 int maxhit, tothit, shldabsb;
82 double chgfac, propor, extradm;
83 double dustfac, tothe;
84 int cas;
85 int hitflag;
86
87 if (Move.free)
88 return;
89 if (Etc.nkling <= 0 || Quad[Ship.quadx][Ship.quady].stars < 0)
90 return;
91 if (Ship.cloaked && Ship.cloakgood)
92 return;
93 /* move before attack */
94 klmove(0);
95 if (Ship.cond == DOCKED) {
96 if (!resting)
97 printf("Starbase shields protect the %s\n",
98 Ship.shipname);
99 return;
100 }
101 /* setup shield effectiveness */
102 chgfac = 1.0;
103 if (Move.shldchg)
104 chgfac = 0.25 + 0.50 * franf();
105 maxhit = tothit = 0;
106 hitflag = 0;
107
108 /* let each Klingon do his damndest */
109 for (i = 0; i < Etc.nkling; i++) {
110 /* if he's low on power he won't attack */
111 if (Etc.klingon[i].power < 20)
112 continue;
113 if (!hitflag) {
114 printf("\nStardate %.2f: Klingon attack:\n",
115 Now.date);
116 hitflag++;
117 }
118 /* complete the hit */
119 dustfac = 0.90 + 0.01 * franf();
120 tothe = Etc.klingon[i].avgdist;
121 hit = Etc.klingon[i].power * pow(dustfac, tothe) * Param.hitfac;
122 /* deplete his energy */
123 dustfac = Etc.klingon[i].power;
124 Etc.klingon[i].power = dustfac * Param.phasfac *
125 (1.0 + (franf() - 0.5) * 0.2);
126 /* see how much of hit shields will absorb */
127 shldabsb = 0;
128 if (Ship.shldup || Move.shldchg) {
129 propor = Ship.shield;
130 propor /= Param.shield;
131 shldabsb = propor * chgfac * hit;
132 if (shldabsb > Ship.shield)
133 shldabsb = Ship.shield;
134 Ship.shield -= shldabsb;
135 }
136 /* actually do the hit */
137 printf("\aHIT: %d units", hit);
138 if (!damaged(SRSCAN))
139 printf(" from %d,%d",
140 Etc.klingon[i].x, Etc.klingon[i].y);
141 cas = (shldabsb * 100) / hit;
142 hit -= shldabsb;
143 if (shldabsb > 0)
144 printf(", shields absorb %d%%, effective hit %d\n",
145 cas, hit);
146 else
147 printf("\n");
148 tothit += hit;
149 if (hit > maxhit)
150 maxhit = hit;
151 Ship.energy -= hit;
152 /* see if damages occurred */
153 if (hit >= (15 - Game.skill) * (25 - ranf(12))) {
154 printf("\aCRITICAL HIT!!!\a\n");
155 /* select a device from probability vector */
156 cas = ranf(1000);
157 for (l = 0; cas >= 0; l++)
158 cas -= Param.damprob[l];
159 l -= 1;
160 /* compute amount of damage */
161 extradm = (hit * Param.damfac[l]) /
162 (75 + ranf(25)) + 0.5;
163 /* damage the device */
164 damage(l, extradm);
165 if (damaged(SHIELD)) {
166 if (Ship.shldup)
167 printf("Sulu: Shields knocked down, "
168 "captain.\n");
169 Ship.shldup = 0;
170 Move.shldchg = 0;
171 }
172 }
173 if (Ship.energy <= 0)
174 lose(L_DSTRYD);
175 }
176
177 /* see what our casualities are like */
178 if (maxhit >= 200 || tothit >= 500) {
179 cas = tothit * 0.015 * franf();
180 if (cas >= 2) {
181 printf("McCoy: we suffered %d casualties in that "
182 "attack.\n",
183 cas);
184 Game.deaths += cas;
185 Ship.crew -= cas;
186 }
187 }
188
189 /* allow Klingons to move after attacking */
190 klmove(1);
191
192 return;
193}
