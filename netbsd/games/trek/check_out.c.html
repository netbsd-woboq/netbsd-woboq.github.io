44/*
45** CHECK IF A DEVICE IS OUT
46**
47** The indicated device is checked to see if it is disabled. If
48** it is, an attempt is made to use the starbase device. If both
49** of these fails, it returns non-zero (device is REALLY out),
50** otherwise it returns zero (I can get to it somehow).
51**
52** It prints appropriate messages too.
53*/
54
55int
56check_out(int device)
57{
58 int dev;
59
60 dev = device;
61
62 /* check for device ok */
63 if (!damaged(dev))
64 return (0);
65
66 /* report it as being dead */
67 out(dev);
68
69 /* but if we are docked, we can go ahead anyhow */
70 if (Ship.cond != DOCKED)
71 return (1);
72 printf(" Using starbase %s\n", Device[dev].name);
73 return (0);
74}
