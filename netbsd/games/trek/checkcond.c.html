43/*
44** Check for Condition After a Move
45**
46** Various ship conditions are checked. First we check
47** to see if we have already lost the game, due to running
48** out of life support reserves, running out of energy,
49** or running out of crew members. The check for running
50** out of time is in events().
51**
52** If we are in automatic override mode (Etc.nkling < 0), we
53** don't want to do anything else, lest we call autover
54** recursively.
55**
56** In the normal case, if there is a supernova, we call
57** autover() to help us escape. If after calling autover()
58** we are still in the grips of a supernova, we get burnt
59** up.
60**
61** If there are no Klingons in this quadrant, we nullify any
62** distress calls which might exist.
63**
64** We then set the condition code, based on the energy level
65** and battle conditions.
66*/
67
68void
69checkcond(void)
70{
71 /* see if we are still alive and well */
72 if (Ship.reserves < 0.0)
73 lose(L_NOLIFE);
74 if (Ship.energy <= 0)
75 lose(L_NOENGY);
76 if (Ship.crew <= 0)
77 lose(L_NOCREW);
78 /* if in auto override mode, ignore the rest */
79 if (Etc.nkling < 0)
80 return;
81 /* call in automatic override if appropriate */
82 if (Quad[Ship.quadx][Ship.quady].stars < 0)
83 autover();
84 if (Quad[Ship.quadx][Ship.quady].stars < 0)
85 lose(L_SNOVA);
86 /* nullify distress call if appropriate */
87 if (Etc.nkling <= 0)
88 killd(Ship.quadx, Ship.quady, 1);
89
90 /* set condition code */
91 if (Ship.cond == DOCKED)
92 return;
93
94 if (Etc.nkling > 0) {
95 Ship.cond = RED;
96 return;
97 }
98 if (Ship.energy < Param.energylow) {
99 Ship.cond = YELLOW;
100 return;
101 }
102 Ship.cond = GREEN;
103 return;
104}
