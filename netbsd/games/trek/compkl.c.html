44/*
45** compute klingon distances
46**
47** The klingon list has the distances for all klingons recomputed
48** and sorted. The parameter is a Boolean flag which is set if
49** we have just entered a new quadrant.
50**
51** This routine is used every time the Enterprise or the Klingons
52** move.
53**
54** f -- set if new quadrant
55*/
56
57static void sortkl(void);
58
59void
60compkldist(int f)
61{
62 int i, dx, dy;
63 double d;
64 double temp;
65
66 if (Etc.nkling == 0)
67 return;
68 for (i = 0; i < Etc.nkling; i++) {
69 /* compute distance to the Klingon */
70 dx = Ship.sectx - Etc.klingon[i].x;
71 dy = Ship.secty - Etc.klingon[i].y;
72 d = dx * dx + dy * dy;
73 d = sqrt(d);
74
75 /* compute average of new and old distances to Klingon */
76 if (!f) {
77 temp = Etc.klingon[i].dist;
78 Etc.klingon[i].avgdist = 0.5 * (temp + d);
79 } else {
80 /* new quadrant: average is current */
81 Etc.klingon[i].avgdist = d;
82 }
83 Etc.klingon[i].dist = d;
84 }
85
86 /* leave them sorted */
87 sortkl();
88}

91/*
92** sort klingons
93**
94** bubble sort on ascending distance
95*/
96
97static void
98sortkl(void)
99{
100 struct kling t;
101 int f, i, m;
102
103 m = Etc.nkling - 1;
104 f = 1;
105 while (f) {
106 f = 0;
107 for (i = 0; i < m; i++) {
108 if (Etc.klingon[i].dist > Etc.klingon[i+1].dist) {
109 t = Etc.klingon[i];
110 Etc.klingon[i] = Etc.klingon[i+1];
111 Etc.klingon[i+1] = t;
112 f = 1;
113 }
114 }
115 }
