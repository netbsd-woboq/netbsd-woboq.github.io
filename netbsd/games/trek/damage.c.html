45/*
46** Schedule Ship.damages to a Device
47**
48** Device `dev1' is damaged in an amount `dam'. Dam is measured
49** in stardates, and is an additional amount of damage. It should
50** be the amount to occur in non-docked mode. The adjustment
51** to docked mode occurs automatically if we are docked.
52**
53** Note that the repair of the device occurs on a DATE, meaning
54** that the dock() and undock() have to reschedule the event.
55**
56** dev1 -- device index
57** dam -- time to repair
58*/
59
60void
61damage(int dev1, double dam)
62{
63 int i;
64 struct event *e;
65 int f;
66 int dev;
67
68 /* ignore zero damages */
69 if (dam <= 0.0)
70 return;
71 dev = dev1;
72
73 printf("\t%s damaged\n", Device[dev].name);
74
75 /* find actual length till it will be fixed */
76 if (Ship.cond == DOCKED)
77 dam *= Param.dockfac;
78 /* set the damage flag */
79 f = damaged(dev);
80 if (!f) {
81 /* new damages -- schedule a fix */
82 schedule(E_FIXDV, dam, 0, 0, dev);
83 return;
84 }
85 /* device already damaged -- add to existing damages */
86 /* scan for old damages */
87 for (i = 0; i < MAXEVENTS; i++) {
88 e = &Event[i];
89 if (e->evcode != E_FIXDV || e->systemname != dev)
90 continue;
91 /* got the right one; add on the new damages */
92 reschedule(e, e->date - Now.date + dam);
93 return;
94 }
95 errx(1, "Cannot find old damages %d", dev);
96}
