43/* DAMAGED -- check for device damaged
44**
45** This is a boolean function which returns non-zero if the
46** specified device is broken. It does this by checking the
47** event list for a "device fix" action on that device.
48*/
49
50int
51damaged(int dev)
52{
53 int d;
54 struct event *e;
55 int i;
56
57 d = dev;
58
59 for (i = 0; i < MAXEVENTS; i++) {
60 e = &Event[i];
61 if (e->evcode != E_FIXDV)
62 continue;
63 if (e->systemname == d)
64 return (1);
65 }
66
67 /* device fix not in event list -- device must not be broken */
68 return (0);
69}
