44/*
45** Dump the starship somewhere in the galaxy
46**
47** Parameter is zero if bounce off of negative energy barrier,
48** one if through a black hole
49**
50** Note that the quadrant is NOT initialized here. This must
51** be done from the calling routine.
52**
53** Repair of devices must be deferred.
54*/
55
56void
57dumpme(int flag)
58{
59 int f;
60 double x;
61 struct event *e;
62 int i;
63
64 f = flag;
65 Ship.quadx = ranf(NQUADS);
66 Ship.quady = ranf(NQUADS);
67 Ship.sectx = ranf(NSECTS);
68 Ship.secty = ranf(NSECTS);
69 x = 1.5 * franf();
70 Move.time += x;
71 if (f) {
72 printf("%s falls into a black hole.\n", Ship.shipname);
73 } else {
74 printf("Computer applies full reverse power to avoid hitting "
75 "the\n");
76 printf(" negative energy barrier. A space warp was "
77 "entered.\n");
78 }
79 /* bump repair dates forward */
80 for (i = 0; i < MAXEVENTS; i++) {
81 e = &Event[i];
82 if (e->evcode != E_FIXDV)
83 continue;
84 reschedule(e, (e->date - Now.date) + x);
85 }
86 events(1);
87 printf("You are now in quadrant %d,%d. It is stardate %.2f\n",
88 Ship.quadx, Ship.quady, Now.date);
89 Move.time = 0;
90}
