44/**
45 ** output hidden distress calls
46 **/
47
48int
49dumpssradio(void)
50{
51 struct event *e;
52 int j;
53 int chkrest;
54
55 chkrest = 0;
56 for (j = 0; j < MAXEVENTS; j++) {
57 e = &Event[j];
58 /* if it is not hidden, then just ignore it */
59 if ((e->evcode & E_HIDDEN) == 0)
60 continue;
61 if (e->evcode & E_GHOST) {
62 unschedule(e);
63 printf("Starsystem %s in quadrant %d,%d is no longer "
64 "distressed\n",
65 systemname(&Quad[e->x][e->y]), e->x, e->y);
66 continue;
67 }
68
69 switch (e->evcode) {
70
71 case E_KDESB:
72 printf("Starbase in quadrant %d,%d is under attack\n",
73 e->x, e->y);
74 chkrest++;
75 break;
76
77 case E_ENSLV:
78 case E_REPRO:
79 printf("Starsystem %s in quadrant %d,%d is "
80 "distressed\n",
81 systemname(&Quad[e->x][e->y]), e->x, e->y);
82 chkrest++;
83 break;
84
85 }
86 }
87
88 return (chkrest);
89}
