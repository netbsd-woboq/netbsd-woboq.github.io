45/**
46 ** move under impulse power
47 **/
48
49/*ARGSUSED*/
50void
51impulse(int v __unused)
52{
53 int course;
54 int power;
55 double dist, time;
56 int percent;
57
58 if (Ship.cond == DOCKED) {
59 printf("Scotty: Sorry captain, but we are still docked.\n");
60 return;
61 }
62 if (damaged(IMPULSE)) {
63 out(IMPULSE);
64 return;
65 }
66 if (getcodi(&course, &dist))
67 return;
68 power = 20 + 100 * dist;
69 percent = 100 * power / Ship.energy + 0.5;
70 if (percent >= 85) {
71 printf("Scotty: That would consume %d%% of our remaining "
72 "energy.\n",
73 percent);
74 if (!getynpar("Are you sure that is wise"))
75 return;
76 printf("Aye aye, sir\n");
77 }
78 time = dist / 0.095;
79 percent = 100 * time / Now.time + 0.5;
80 if (percent >= 85) {
81 printf("Spock: That would take %d%% of our remaining time.\n",
82 percent);
83 if (!getynpar("Are you sure that is wise"))
84 return;
85 printf("(He's finally gone mad)\n");
86 }
87 Move.time = move(0, course, time, 0.095);
88 Ship.energy -= 20 + 100 * Move.time * 0.095;
89}
