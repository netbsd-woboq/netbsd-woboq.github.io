47/*
48** PRINT OUT LOSER MESSAGES
49**
50** The messages are printed out, the score is computed and
51** printed, and the game is restarted. Oh yeh, any special
52** actions which need be taken are taken.
53*/
54
55extern jmp_buf env;
56
57static const char *const Losemsg[] = {
58 "You ran out of time",
59 "You ran out of energy",
60 "You have been destroyed",
61 "You ran into the negative energy barrier",
62 "You destroyed yourself by nova'ing that star",
63 "You have been caught in a supernova",
64 "You just suffocated in outer space",
65 "You could not be rematerialized",
66 "\n\032\014 ***\07 Ship's hull has imploded\07 ***",
67 "You have burned up in a star",
68 "Well, you destroyed yourself, but it didn't do any good",
69 "You have been captured by Klingons and mercilessly tortured",
70 "Your last crew member died",
71};
72
73void
74lose(int why)
75{
76 Game.killed = 1;
77 sleep(1);
78 printf("\n%s\n", Losemsg[why - 1]);
79 switch (why) {
80 case L_NOTIME:
81 Game.killed = 0;
82 break;
83 }
84 Move.endgame = -1;
85 score();
86 skiptonl(0);
87 longjmp(env, 1);
88}
