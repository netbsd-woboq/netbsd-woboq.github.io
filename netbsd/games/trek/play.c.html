48/*
49** INSTRUCTION READ AND MAIN PLAY LOOP
50**
51** Well folks, this is it. Here we have the guts of the game.
52** This routine executes moves. It sets up per-move variables,
53** gets the command, and executes the command. After the command,
54** it calls events() to use up time, attack() to have Klingons
55** attack if the move was not free, and checkcond() to check up
56** on how we are doing after the move.
57*/

88/*ARGSUSED*/
89static void
90myreset(int v __unused)
91{
92
93 longjmp(env, 1);
94}
95
96void
97play(void)
98{
99 const struct cvntab *r;
100
101 while (1) {
102 Move.free = 1;
103 Move.time = 0.0;
104 Move.shldchg = 0;
105 Move.newquad = 0;
106 Move.resting = 0;
107 skiptonl(0);
108 r = getcodpar("\nCommand", Comtab);
109 (*r->value)(r->value2);
110 events(0);
111 attack(0);
112 checkcond();
113 }
114}
