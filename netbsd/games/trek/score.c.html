46/*
47** PRINT OUT THE CURRENT SCORE
48*/
49
50static void scoreitem(long amount, const char *descfmt, ...)
51 __printflike(2, 3);
52
53static void
54scoreitem(long amount, const char *descfmt, ...)
55{
56 va_list ap;
57 char buf[128];
58
59 if (amount == 0)
60 return;
61
62 va_start(ap, descfmt);
63 vsnprintf(buf, sizeof(buf), descfmt, ap);
64 va_end(ap);
65
66 printf("%-40s %10ld\n", buf, amount);
67}
68
69long
70score(void)
71{
72 int u;
73 int t;
74 long s;
75 double r;
76
77 printf("\n*** Your score:\n");
78
79 s = t = Param.klingpwr / 4 * (u = Game.killk);
80 scoreitem(t, "%d Klingons killed", u);
81
82 r = Now.date - Param.date;
83 if (r < 1.0)
84 r = 1.0;
85 r = Game.killk / r;
86 s += (t = 400 * r);
87 scoreitem(t, "Kill rate %.2f Klingons/stardate", r);
88
89 r = Now.klings;
90 r /= Game.killk + 1;
91 s += (t = -400 * r);
92 scoreitem(t, "Penalty for %d klingons remaining", Now.klings);
93
94 if (Move.endgame > 0) {
95 s += (t = 100 * (u = Game.skill));
96 scoreitem(t, "Bonus for winning a %s%s game",
97 Skitab[u - 1].abbrev, Skitab[u - 1].full);
98 }
99
100 if (Game.killed) {
101 s -= 500;
102 scoreitem(-500, "Penalty for getting killed");
103 }
104
105 s += (t = -100 * (u = Game.killb));
106 scoreitem(t, "%d starbases killed", u);
107
108 s += (t = -100 * (u = Game.helps));
109 scoreitem(t, "%d calls for help", u);
110
111 s += (t = -5 * (u = Game.kills));
112 scoreitem(t, "%d stars destroyed", u);
113
114 s += (t = -150 * (u = Game.killinhab));
115 scoreitem(t, "%d inhabited starsystems destroyed", u);
116
117 if (Ship.ship != ENTERPRISE) {
118 s -= 200;
119 scoreitem(-200, "penalty for abandoning ship");
120 }
121
122 s += (t = 3 * (u = Game.captives));
123 scoreitem(t, "%d Klingons captured", u);
124
125 s += (t = -(u = Game.deaths));
126 scoreitem(t, "%d casualties", u);
127
128 printf("\n");
129 scoreitem(s, "*** TOTAL");
130 return (s);
131}
