45/*
46** VISUAL SCAN
47**
48** A visual scan is made in a particular direction of three sectors
49** in the general direction specified. This takes time, and
50** Klingons can attack you, so it should be done only when sensors
51** are out.
52*/
53
54/* This struct[] has the delta x, delta y for particular directions */
55static struct xy Visdelta[11] = {
56 { -1, -1 },
57 { -1, 0 },
58 { -1, 1 },
59 { 0, 1 },
60 { 1, 1 },
61 { 1, 0 },
62 { 1, -1 },
63 { 0, -1 },
64 { -1, -1 },
65 { -1, 0 },
66 { -1, 1 }
67};
68
69/*ARGSUSED*/
70void
71visual(int z __unused)
72{
73 int ix, iy;
74 int co;
75 struct xy *v;
76
77 co = getintpar("direction");
78 if (co < 0 || co > 360)
79 return;
80 co = (co + 22) / 45;
81 v = &Visdelta[co];
82 ix = Ship.sectx + v->x;
83 iy = Ship.secty + v->y;
84 if (ix < 0 || ix >= NSECTS || iy < 0 || iy >= NSECTS)
85 co = '?';
86 else
87 co = Sect[ix][iy];
88 printf("%d,%d %c ", ix, iy, co);
89 v++;
90 ix = Ship.sectx + v->x;
91 iy = Ship.secty + v->y;
92 if (ix < 0 || ix >= NSECTS || iy < 0 || iy >= NSECTS)
93 co = '?';
94 else
95 co = Sect[ix][iy];
96 printf("%c ", co);
97 v++;
98 ix = Ship.sectx + v->x;
99 iy = Ship.secty + v->y;
100 if (ix < 0 || ix >= NSECTS || iy < 0 || iy >= NSECTS)
101 co = '?';
102 else
103 co = Sect[ix][iy];
104 printf("%c %d,%d\n", co, ix, iy);
105 Move.time = 0.05;
106 Move.free = 0;
107}
