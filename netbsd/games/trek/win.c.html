47/*
48** Signal game won
49**
50** This routine prints out the win message, arranges to print out
51** your score, tells you if you have a promotion coming to you,
52** cleans up the current input line, and arranges to have you
53** asked whether or not you want another game (via the longjmp()
54** call).
55**
56** Pretty straightforward, although the promotion algorithm is
57** pretty off the wall.
58*/
59
60extern jmp_buf env;
61
62void
63win(void)
64{
65 long s;
66 const struct cvntab *p;
67
68 sleep(1);
69 printf("\nCongratulations, you have saved the Federation\n");
70 Move.endgame = 1;
71
72 /* print and return the score */
73 s = score();
74
75 /* decide if she gets a promotion */
76 if (Game.helps == 0 && Game.killb == 0 && Game.killinhab == 0 &&
77 5 * Game.kills + Game.deaths < 100 &&
78 s >= 1000 && Ship.ship == ENTERPRISE) {
79 printf("In fact, you are promoted one step in rank,\n");
80 if (Game.skill >= 6) {
81 printf("to the exalted rank of Commodore Emeritus\n");
82 } else {
83 p = &Skitab[Game.skill - 1];
84 printf("from %s%s ", p->abbrev, p->full);
85 p++;
86 printf("to %s%s\n", p->abbrev, p->full);
87 }
88 }
89
90 /* clean out input, and request new game */
91 skiptonl(0);
92 longjmp(env, 1);
93}
