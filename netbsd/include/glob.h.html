37#ifndef _GLOB_H_
38#define _GLOB_H_
39
40#include <sys/cdefs.h>
41#include <sys/featuretest.h>
42#include <sys/types.h>
43#include <sys/stat.h>
44
45#ifndef __gl_size_t
46#define __gl_size_t size_t
47#endif
48#ifndef __gl_stat_t
49#define __gl_stat_t struct stat
50#endif
51
52typedef struct {
53 __gl_size_t gl_pathc; /* Count of total paths so far. */
54 __gl_size_t gl_matchc; /* Count of paths matching pattern. */
55 __gl_size_t gl_offs; /* Reserved at beginning of gl_pathv. */
56 int gl_flags; /* Copy of flags parameter to glob. */
57 char **gl_pathv; /* List of paths matching pattern. */
58 /* Copy of errfunc parameter to glob. */
59 int (*gl_errfunc)(const char *, int);
60
61 /*
62 * Alternate filesystem access methods for glob; replacement
63 * versions of closedir(3), readdir(3), opendir(3), stat(2)
64 * and lstat(2).
65 */
66 void (*gl_closedir)(void *);
67 struct dirent *(*gl_readdir)(void *);
68 void *(*gl_opendir)(const char *);
69 int (*gl_lstat)(const char *, __gl_stat_t *);
70 int (*gl_stat)(const char *, __gl_stat_t *);
71} glob_t;
72
73#define GLOB_APPEND 0x00001 /* Append to output from previous call. */
74#define GLOB_DOOFFS 0x00002 /* Use gl_offs. */
75#define GLOB_ERR 0x00004 /* Return on error. */
76#define GLOB_MARK 0x00008 /* Append / to matching directories. */
77#define GLOB_NOCHECK 0x00010 /* Return pattern itself if nothing matches. */
78#define GLOB_NOSORT 0x00020 /* Don't sort. */
79#define GLOB_NOESCAPE 0x01000 /* Disable backslash escaping. */
80
81#define GLOB_NOSPACE (-1) /* Malloc call failed. */
82#define GLOB_ABORTED (-2) /* Unignored error. */
83#define GLOB_NOMATCH (-3) /* No match, and GLOB_NOCHECK was not set. */
84#define GLOB_NOSYS (-4) /* Implementation does not support function. */
85
86#if defined(_NETBSD_SOURCE) || defined(HAVE_NBTOOL_CONFIG_H)
87#define GLOB_ALTDIRFUNC 0x00040 /* Use alternately specified directory funcs. */
88#define GLOB_BRACE 0x00080 /* Expand braces ala csh. */
89#define GLOB_MAGCHAR 0x00100 /* Pattern had globbing characters. */
90#define GLOB_NOMAGIC 0x00200 /* GLOB_NOCHECK without magic chars (csh). */
91#define GLOB_LIMIT 0x00400 /* Limit memory used by matches to ARG_MAX */
92#define GLOB_TILDE 0x00800 /* Expand tilde names from the passwd file. */
93/* GLOB_NOESCAPE 0x01000 above */
94#define GLOB_PERIOD 0x02000 /* Allow metachars to match leading periods. */
95#define GLOB_NO_DOTDIRS 0x04000 /* Make . and .. vanish from wildcards. */
96#define GLOB_STAR 0x08000 /* Use glob ** to recurse directories */
97#define GLOB_TILDE_CHECK 0x10000 /* Expand tilde names from the passwd file. */
98#define GLOB_QUOTE 0 /* source compatibility */
99
100#define GLOB_ABEND GLOB_ABORTED /* source compatibility */
101#endif

108#endif

110int glob_pattern_p(const char *, int);
111#endif
112__END_DECLS
113
114#endif /* !_GLOB_H_ */
