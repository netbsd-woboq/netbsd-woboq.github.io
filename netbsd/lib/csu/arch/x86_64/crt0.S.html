40STRONG_ALIAS(_start,__start)
41 .hidden ___start
42
43_ENTRY(__start)
44 andq $~15,%rsp
45 subq $8,%rsp
46 movq %rdx, %rdi
47 movq %rbx, %rsi
48 jmp ___start
