32#include <sys/cdefs.h>
33#include <sys/types.h>
34
35#include <machine/segments.h>
36#include <machine/sysarch.h>
37
38int
39i386_get_ldt(int start, union descriptor *desc, int num)
40{
41 struct i386_get_ldt_args p;
42
43 p.start = start;
44 p.desc = desc;
45 p.num = num;
46
47 return sysarch(I386_GET_LDT, (void *)&p);
48}
