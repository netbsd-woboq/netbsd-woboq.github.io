29#ifndef _SDP_INT_H_
30#define _SDP_INT_H_
31
32struct sdp_session {
33 uint16_t tid; /* current session transaction ID */
34 uint16_t imtu; /* incoming MTU */
35 uint8_t *ibuf; /* incoming buffer */
36 uint8_t *rbuf; /* response buffer */
37 uint8_t cs[17];/* continuation state */
38 int32_t s; /* L2CAP socket */
39};
40
41/* sdp_session.c */
42sdp_session_t _sdp_open(const bdaddr_t *, const bdaddr_t *);
43sdp_session_t _sdp_open_local(const char *);
44void _sdp_close(sdp_session_t);
45bool _sdp_send_pdu(struct sdp_session *, uint8_t, struct iovec *, int);
46ssize_t _sdp_recv_pdu(struct sdp_session *, uint8_t);
47int _sdp_errno(uint16_t);
48
49/* sdp_data.c */
50bool _sdp_data_print(const uint8_t *, const uint8_t *, int);
51
52/* sdp_service.c */
53bool sdp_service_search_attribute(sdp_session_t, const sdp_data_t *,
54 const sdp_data_t *, sdp_data_t *);
55
56#endif /* _SDP_INT_H_ */
