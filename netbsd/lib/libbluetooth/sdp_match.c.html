38/******************************************************************************
39 * sdp_match_xxxx(data, value)
40 *
41 * examine SDP data element. Advance pointer and return true if it
42 * matches the supplied type value
43 */
44
45bool
46sdp_match_uuid16(sdp_data_t *data, uint16_t uuid)
47{
48 sdp_data_t d = *data;
49 uuid_t u1, u2;
50
51 u1 = BLUETOOTH_BASE_UUID;
52 u1.time_low = uuid;
53
54 if (!sdp_get_uuid(&d, &u2)
55 || !uuid_equal(&u1, &u2, NULL))
56 return false;
57
58 *data = d;
59 return true;
60}
