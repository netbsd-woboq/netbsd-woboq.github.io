37#include "namespace.h"
38#include <sys/types.h>
39#include <inttypes.h>
40#include <ucontext.h>
41#include <lwp.h>
42#include <stdlib.h>
43
44void
45_lwp_makecontext(ucontext_t *u, void (*start)(void *),
46 void *arg, void *private, caddr_t stack_base, size_t stack_size)
47{
48 void **sp;
49
50 getcontext(u);
51 u->uc_link = NULL;
52
53 u->uc_stack.ss_sp = stack_base;
54 u->uc_stack.ss_size = stack_size;
55
56 /* LINTED uintptr_t is safe */
57 u->uc_mcontext.__gregs[_REG_EIP] = (uintptr_t)start;
58
59 /* Align to a 16-byte boundary for SSE */
60 /* LINTED uintptr_t is safe */
61 sp = (void **) (((uintptr_t)(stack_base + stack_size - 4) & ~0xf) + 4);
62
63 *--sp = arg;
64 *--sp = (void *) _lwp_exit;
65
66 /* LINTED uintptr_t is safe */
67 u->uc_mcontext.__gregs[_REG_UESP] = (uintptr_t) sp;
68
69 u->uc_mcontext._mc_tlsbase = (uintptr_t)private;
70 u->uc_flags |= _UC_TLSBASE;
71}
