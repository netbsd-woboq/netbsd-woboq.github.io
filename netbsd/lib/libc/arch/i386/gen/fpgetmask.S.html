10#ifdef WEAK_ALIAS

12ENTRY(_fpgetmask)
13#else
14ENTRY(fpgetmask)
15#endif
16 subl $4,%esp
17 fnstcw (%esp)
18 movl (%esp),%eax
19 notl %eax
20 andl $63,%eax
21 addl $4,%esp
22 ret
23#ifdef WEAK_ALIAS
24END(_fpgetmask)
