10#ifdef WEAK_ALIAS
11WEAK_ALIAS(fpgetprec, _fpgetprec)
12ENTRY(_fpgetprec)
13#else
14ENTRY(fpgetprec)
15#endif
16 subl $4,%esp
17 fnstcw (%esp)
18 movl (%esp),%eax
19 rorl $8,%eax
20 andl $3,%eax
21 addl $4,%esp
22 ret
23#ifdef WEAK_ALIAS
24END(_fpgetprec)
25#else
26END(fpgetprec)
