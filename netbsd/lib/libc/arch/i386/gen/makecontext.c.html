32#include <sys/cdefs.h>
33#if defined(LIBC_SCCS) && !defined(lint)
34__RCSID("$NetBSD: makecontext.c,v 1.4 2008/04/28 20:22:56 martin Exp $");
35#endif
36
37#include <inttypes.h>
38#include <stddef.h>
39#include <ucontext.h>
40#include "extern.h"
41
42#if __STDC__
43#include <stdarg.h>
44#else
45#include <varargs.h>
46#endif
47
48
49void
50#if __STDC__
51makecontext(ucontext_t *ucp, void (*func)(void), int argc, ...)
52#else
53makecontext(ucp, func, argc, va_alist)
54 ucontext_t *ucp;
55 void (*func)();
56 int argc;
57 va_dcl
58#endif
59{
60 __greg_t *gr = ucp->uc_mcontext.__gregs;
61 unsigned int *sp;
62 va_list ap;
63
64 /* LINTED __greg_t is safe */
65 gr[_REG_EIP] = (__greg_t)func;
66
67 /* LINTED uintptr_t is safe */
68 sp = (unsigned int *)((uintptr_t)ucp->uc_stack.ss_sp +
69 ucp->uc_stack.ss_size);
70 /* Align on word boundary. */
71 /* LINTED uintptr_t is safe */
72 sp = (unsigned int *)((uintptr_t)sp & ~0x3);
73 sp -= argc + 1; /* Make room for ret and args. */
74 /* LINTED __greg_t is safe */
75 gr[_REG_UESP] = (__greg_t)sp;
76 gr[_REG_EBP] = (__greg_t)0; /* Wipe out frame pointer. */
77
78 /* Put return address on top of stack. */
79 /* LINTED uintptr_t is safe */
80 *sp++ = (uintptr_t)_resumecontext;
81
82 /* Construct argument list. */
83#if __STDC__
84 va_start(ap, argc);
85#else
86 va_start(ap);
87#endif
88 while (argc-- > 0) {
89 /* LINTED uintptr_t is safe */
90 *sp++ = va_arg(ap, uintptr_t);
91 }
92 va_end(ap);
93}
