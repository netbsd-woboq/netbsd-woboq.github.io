39/*
40 * This assembly-language implementation differs from the (obvious)
41 * C-language implementation only in not clobbering the previous
42 * function's return address (us), which is the point of the exercise.
43 */
44
45NENTRY(_resumecontext) /* profiling prologue would clobber TOS */
46 leal -(4 + 776)(%esp),%ecx /* retaddr + sizeof (ucontext_t) */
47 andl $~(0xf),%ecx /* align on _UC_UCONTEXT_ALIGN */
48 movl %ecx,%esp
49 PIC_PROLOGUE
50 pushl %ecx
51#ifdef __PIC__
52 call PIC_PLT(_C_LABEL(_getcontext))
53#else
54 call _C_LABEL(_getcontext)
55#endif
56 addl $4,%esp
57 PIC_EPILOGUE
58 movl 4(%esp),%ecx /* uc_link */
59 PIC_PROLOGUE
60 pushl %ecx /* argument for exit() or setcontext() */
61 testl %ecx,%ecx /* link end? */
62 jnz 9f /* watch out for local label use in PIC_* */
63
64 /* normal exit */
65#ifdef __PIC__
66 call PIC_PLT(_C_LABEL(exit))
67#else
68 call _C_LABEL(exit)
69#endif
70 /* NOTREACHED */
71
729:
73#ifdef __PIC__
74 call PIC_PLT(_C_LABEL(setcontext))
75#else
76 call _C_LABEL(setcontext)
77#endif
78 /* NOTREACHED */
79
80 /* something is wrong, pull the brake */
81 pushl $-1
82 pushl %eax /* unused return address */
83 SYSTRAP(exit)
84END(_resumecontext)
