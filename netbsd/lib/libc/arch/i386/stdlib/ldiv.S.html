13ENTRY(ldiv)
14 pushl %ebx
15 movl 8(%esp),%ebx
16 movl 12(%esp),%eax
17 movl 16(%esp),%ecx
18 cdq
19 idiv %ecx
20 movl %eax,(%ebx)
21 movl %edx,4(%ebx)
22 movl %ebx,%eax
23 popl %ebx
24 ret $4
25END(ldiv)
