38#include <machine/asm.h>
39#include <sys/errno.h>
40#include "SYS.h"
41
42#ifdef WEAK_ALIAS
43WEAK_ALIAS(clone, __clone)
44#endif
45 .text
46
47/*
48 * int clone(int (*fn)(void *), void *stack, int flags, void *arg);
49 */
50ENTRY(__clone)
51 pushl %ebp
52
53 /*
54 * Sanity checks: func and stack may not be NULL.
55 */
56 movl 8(%esp), %ebp
57 cmpl $0,%ebp /* function */
58 je 3f
59 movl 12(%esp),%eax /* stack */
60 cmpl $0,%eax
61 je 3f
62
63 /*
64 * Set up the stack for the clone.
65 */
66 movl 20(%esp),%ecx
67 movl %ecx,-4(%eax) /* argument */
68 leal -4(%eax),%eax /* sp points to arg */
69
70 pushl %eax /* stack */
71 pushl 20(%esp) /* flags */
72 pushl $0 /* dummy return address */
73
74 SYSTRAP(__clone)
75 jc 4f
76 cmpl $0,%eax
77 jne 2f /* we're the parent */
78 call *%ebp /* this is the clone, call the function */
79
80#ifdef __PIC__
81 PIC_PROLOGUE
82 pushl %eax /* clone does _exit(func(arg)); */
83 call PIC_PLT(_C_LABEL(_exit))
84 addl $4,%esp
85 PIC_EPILOGUE
86#else
87 pushl %eax
88 call _C_LABEL(_exit)
89 addl $4,%esp
90#endif
91
922:
93 addl $12,%esp
94 popl %ebp
95 ret
963:
97 movl $EINVAL,%eax
98 jmp 5f
994:
100 addl $12,%esp
1015:
102 popl %ebp
103 jmp CERROR
104END(__clone)
