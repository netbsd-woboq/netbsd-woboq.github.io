44#ifdef WEAK_ALIAS
45WEAK_ALIAS(pipe, _pipe)
46#endif
47
48_SYSCALL(_pipe,pipe)
49 movl 4(%esp),%ecx
50 movl %eax,(%ecx)
51 movl %edx,4(%ecx)
52 xorl %eax,%eax
53 ret
54END(_pipe)
