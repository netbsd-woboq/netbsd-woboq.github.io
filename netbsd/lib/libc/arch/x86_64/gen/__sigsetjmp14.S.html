46/*
47 * C library -- _setjmp, _longjmp
48 *
49 * longjmp(a,v)
50 * will generate a "return(v)" from the last call to
51 * setjmp(a)
52 * by restoring registers from the stack.
53 * The previous signal state is restored.
54 */
55
56ENTRY(__sigsetjmp14)
57 movq (%rsp),%r11
58 movq %rbx,(_JB_RBX * 8)(%rdi)
59 movq %rbp,(_JB_RBP * 8)(%rdi)
60 movq %r12,(_JB_R12 * 8)(%rdi)
61 movq %r13,(_JB_R13 * 8)(%rdi)
62 movq %r14,(_JB_R14 * 8)(%rdi)
63 movq %r15,(_JB_R15 * 8)(%rdi)
64 movq %rsp,(_JB_RSP * 8)(%rdi)
65 movq %r11,(_JB_PC * 8)(%rdi)
66
67 movq %rsi,(_JB_SIGFLAG * 8)(%rdi)
68 testl %esi,%esi
69 jz 2f
70
71 leaq (_JB_SIGMASK * 8)(%rdi),%rdx
72 xorl %edi,%edi
73 xorq %rsi,%rsi
74
75#ifdef __PIC__
76 call PIC_PLT(_C_LABEL(__sigprocmask14))
77#else
78 call _C_LABEL(__sigprocmask14)
79#endif
802: xorl %eax,%eax
81 ret
82END(__sigsetjmp14)
83
84ENTRY(__siglongjmp14)
85 movq %rdi,%r12
86 pushq %rsi
87 cmpl $0, (_JB_SIGFLAG * 8)(%rdi)
88
89 jz 2f
90 leaq (_JB_SIGMASK * 8)(%rdi),%rsi
91 movl $3,%edi /* SIG_SETMASK */
92 xorq %rdx,%rdx
93
94#ifdef __PIC__
95 call PIC_PLT(_C_LABEL(__sigprocmask14))
96#else
97 call _C_LABEL(__sigprocmask14)
98#endif
992: popq %rax
100 movq (_JB_RBX * 8)(%r12),%rbx
101 movq (_JB_RBP * 8)(%r12),%rbp
102 movq (_JB_R13 * 8)(%r12),%r13
103 movq (_JB_R14 * 8)(%r12),%r14
104 movq (_JB_R15 * 8)(%r12),%r15
105 movq (_JB_RSP * 8)(%r12),%rsp
106 movq (_JB_PC * 8)(%r12),%r11
107 movq (_JB_R12 * 8)(%r12),%r12
108
109 testl %eax,%eax
110 jnz 1f
111 incl %eax
1121: movq %r11,0(%rsp)
113 ret
114END(__siglongjmp14)
