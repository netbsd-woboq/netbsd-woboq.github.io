45/*
46 * C library -- _setjmp, _longjmp
47 *
48 * _longjmp(a,v)
49 * will generate a "return(v)" from the last call to
50 * _setjmp(a)
51 * by restoring registers from the stack.
52 * The previous signal state is NOT restored.
53 */
54
55ENTRY(_setjmp)
56 movq (%rsp),%r11
57 movq %rbx,(_JB_RBX * 8)(%rdi)
58 movq %rbp,(_JB_RBP * 8)(%rdi)
59 movq %r12,(_JB_R12 * 8)(%rdi)
60 movq %r13,(_JB_R13 * 8)(%rdi)
61 movq %r14,(_JB_R14 * 8)(%rdi)
62 movq %r15,(_JB_R15 * 8)(%rdi)
63 movq %rsp,(_JB_RSP * 8)(%rdi)
64 movq %r11,(_JB_PC * 8)(%rdi)
65
66 xorl %eax,%eax
67 ret
68END(_setjmp)
69
70ENTRY(_longjmp)
71 movq (_JB_RBX * 8)(%rdi),%rbx
72 movq (_JB_RBP * 8)(%rdi),%rbp
73 movq (_JB_R12 * 8)(%rdi),%r12
74 movq (_JB_R13 * 8)(%rdi),%r13
75 movq (_JB_R14 * 8)(%rdi),%r14
76 movq (_JB_R15 * 8)(%rdi),%r15
77 movq (_JB_RSP * 8)(%rdi),%rsp
78 movq (_JB_PC * 8)(%rdi),%r11
79
80 movl %esi,%eax
81 testl %eax,%eax
82 jnz 1f
83 incl %eax
841: movq %r11,0(%rsp)
85 ret
86END(_longjmp)
