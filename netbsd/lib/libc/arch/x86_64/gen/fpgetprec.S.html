8#include <machine/asm.h>

10/*
11 * XXX store only x87 state. If an application only uses the fp*
12 * interface, this should be in sync with the SSE mxcsr register.
13 */
14
15#ifdef WEAK_ALIAS
16WEAK_ALIAS(fpgetprec, _fpgetprec)
17ENTRY(_fpgetprec)
18#else
19ENTRY(fpgetprec)
20#endif
21 fnstcw -4(%rsp)
22 movl -4(%rsp),%eax
23 rorl $8,%eax
24 andl $3,%eax
25 ret
26#ifdef WEAK_ALIAS
27END(_fpgetprec)
28#else
29END(fpgetprec)
30#endif
