11/*
12 * XXX read both the x87 and SSE status words here, and OR
13 * them to get a complete picture of exceptions.
14 */
15
16#ifdef WEAK_ALIAS
17WEAK_ALIAS(fpgetsticky, _fpgetsticky)
18ENTRY(_fpgetsticky)
19#else
20ENTRY(fpgetsticky)
21#endif
22 fnstsw -4(%rsp)
23 stmxcsr -8(%rsp)
24 movl -4(%rsp),%eax
25 orl -8(%rsp),%eax
26 andl $63,%eax
27 ret
28#ifdef WEAK_ALIAS
29END(_fpgetsticky)
30#else
31END(fpgetsticky)
32#endif
