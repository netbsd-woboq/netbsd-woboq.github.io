40#if defined(LIBC_SCCS) && !defined(lint)

44/*
45 * 56 == offsetof(ucontext_t, uc_mcontext)
46 * 21 == _REG_RIP
47 * 24 == _REG_URSP
48 */
49
50ENTRY(swapcontext)
51 pushq %rdi /* preserve oucp */
52 pushq %rsi /* preserve ucp */
53#ifdef __PIC__
54 call PIC_PLT(_C_LABEL(_getcontext))
55#else
56 call _C_LABEL(_getcontext)
57#endif
58 popq %rsi
59 popq %rdi
60 testl %eax,%eax
61 jnz 2f
62 movq 0(%rsp),%r11
63 movq %r11,(56 + 21 * 8)(%rdi)
64 leaq 8(%rsp),%r11
65 movq %r11,(56 + 24 * 8)(%rdi)
66 movq %rsi,%rdi
67#ifdef __PIC__
68 call PIC_PLT(_C_LABEL(setcontext))
69#else
70 call _C_LABEL(setcontext)
71#endif
722:
73 ret
74END(swapcontext)
