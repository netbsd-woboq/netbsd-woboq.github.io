29#ifndef _CITRUS_ALIASNAME_LOCAL_H_
30#define _CITRUS_ALIASNAME_LOCAL_H_
31
32static __inline const char *
33__unaliasname(const char *dbname, const char *alias,
34 void *buf, size_t bufsize)
35{
36 return _lookup_simple(dbname, alias,
37 buf, bufsize, _LOOKUP_CASE_SENSITIVE);
38}
39
40static __inline int
41__isforcemapping(const char *name)
42{
43 _DIAGASSERT(name != NULL);
44
45 return _bcs_strcasecmp("/force", name);
46}
47
48#endif /*_CITRUS_ALIASNAME_LOCAL_H_*/
