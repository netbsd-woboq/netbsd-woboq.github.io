29#ifndef _CITRUS_DB_FILE_H_
30#define _CITRUS_DB_FILE_H_
31
32/*
33 * db format:
34 * +---
35 * | header
36 * | - magic
37 * | - num entries
38 * +---
39 * | entry directory
40 * | +------------
41 * | | entry0
42 * | | - hash value
43 * | | - next entry
44 * | | - key offset
45 * | | - key len
46 * | | - data offset
47 * | | - data size
48 * | |---
49 * | | entry1
50 * | | ..
51 * | | entryN
52 * | +---
53 * +---
54 * | key table
55 * | - key0
56 * | ...
57 * | - keyN
58 * +---
59 * | data table
60 * | - data0
61 * | ...
62 * | - dataN
63 * +---
64 */
65
66#define _CITRUS_DB_MAGIC_SIZE 8
67#define _CITRUS_DB_HEADER_SIZE 16
68struct _citrus_db_header_x {
69 char dhx_magic[_CITRUS_DB_MAGIC_SIZE];
70 uint32_t dhx_num_entries;
71 uint32_t dhx_entry_offset;
72} __packed;
73
74struct _citrus_db_entry_x {
75 uint32_t dex_hash_value;
76 uint32_t dex_next_offset;
77 uint32_t dex_key_offset;
78 uint32_t dex_key_size;
79 uint32_t dex_data_offset;
80 uint32_t dex_data_size;
81} __packed;
82#define _CITRUS_DB_ENTRY_SIZE 24
83
84#endif
