46#define _CITRUS_ICONV_F_HIDE_INVALID 0x0001

48/*
49 * _citrus_iconv_convert:
50 * convert a string.
51 */
52static __inline int
53_citrus_iconv_convert(struct _citrus_iconv * __restrict cv,
54 const char * __restrict * __restrict in,
55 size_t * __restrict inbytes,
56 char * __restrict * __restrict out,
57 size_t * __restrict outbytes, uint32_t flags,
58 size_t * __restrict nresults)
59{
60
61 _DIAGASSERT(cv && cv->cv_shared && cv->cv_shared->ci_ops &&
62 cv->cv_shared->ci_ops->io_convert);
63 _DIAGASSERT(out || outbytes == 0);
64
65 return (*cv->cv_shared->ci_ops->io_convert)(cv, in, inbytes, out,
66 outbytes, flags, nresults);
67}
