29#ifndef _CITRUS_LOOKUP_H_
30#define _CITRUS_LOOKUP_H_
31
32#define _CITRUS_LOOKUP_CASE_SENSITIVE 0
33#define _CITRUS_LOOKUP_CASE_IGNORE 1
34
35struct _citrus_lookup;
36__BEGIN_DECLS
37char *_citrus_lookup_simple(const char *, const char *, char *, size_t, int);
38int _citrus_lookup_seq_open(struct _citrus_lookup **, const char *, int);
39void _citrus_lookup_seq_rewind(struct _citrus_lookup *);
40int _citrus_lookup_seq_next(struct _citrus_lookup *,
41 struct _region *, struct _region *);
42int _citrus_lookup_seq_lookup(struct _citrus_lookup *, const char *,
43 struct _region *);
44int _citrus_lookup_get_number_of_entries(struct _citrus_lookup *);
45void _citrus_lookup_seq_close(struct _citrus_lookup *);
46__END_DECLS
47
48static __inline const char *
49_citrus_lookup_alias(const char *path, const char *key, char *buf, size_t n,
50 int ignore_case)
51{
52 const char *ret;
53
54 ret = _citrus_lookup_simple(path, key, buf, n, ignore_case);
55 if (ret == NULL)
56 ret = key;
57
58 return ret;
59}
