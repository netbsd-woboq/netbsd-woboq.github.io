54#define T_COMM '#'
55static int
56convert_line(struct _citrus_db_factory *df, const char *line, size_t len)
57{
58 const char *p;
59 char key[LINE_MAX], data[LINE_MAX];
60
61 /* cut off trailing comment */
62 p = memchr(line, T_COMM, len);
63 if (p)
64 len = p - line;
65
66 /* key */
67 line = _bcs_skip_ws_len(line, &len);
68 if (len == 0)
69 return 0;
70 p = _bcs_skip_nonws_len(line, &len);
71 if (p==line)
72 return 0;
73 snprintf(key, sizeof(key), "%.*s", (int)(p-line), line);
74 _bcs_convert_to_lower(key);
75
76 /* data */
77 line = _bcs_skip_ws_len(p, &len);
78 _bcs_trunc_rws_len(line, &len);
79 snprintf(data, sizeof(data), "%.*s", (int)len, line);
80
81 return _db_factory_addstr_by_s(df, key, data);
82}
83
84static int
85dump_db(struct _citrus_db_factory *df, struct _region *r)
86{
87 size_t size;
88 void *ptr;
89
90 size = _db_factory_calc_size(df);
91 ptr = malloc(size);
92 if (ptr == NULL)
93 return errno;
94 _region_init(r, ptr, size);
95
96 return _db_factory_serialize(df, _CITRUS_LOOKUP_MAGIC, r);
97}
98
99int
100_citrus_lookup_factory_convert(FILE *out, FILE *in)
101{
102 struct _citrus_db_factory *df;
103 struct _region r;
104 char *line;
105 size_t size;
106 int ret;
107
108 ret = _db_factory_create(&df, &_db_hash_std, NULL);
109 if (ret)
110 return ret;
111
112 while ((line = fgetln(in, &size)) != NULL)
113 if ((ret = convert_line(df, line, size))) {
114 _db_factory_free(df);
115 return ret;
116 }
117
118 ret = dump_db(df, &r);
119 _db_factory_free(df);
120 if (ret)
121 return ret;
122
123 if (fwrite(_region_head(&r), _region_size(&r), 1, out) != 1)
124 return errno;
125
126 return 0;
127}
