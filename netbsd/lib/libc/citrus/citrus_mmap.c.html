52int
53_citrus_map_file(struct _citrus_region * __restrict r,
54 const char * __restrict path)
55{
56 int fd, ret = 0;
57 struct stat st;
58 void *head;
59
60 _DIAGASSERT(r != NULL);
61
62 _region_init(r, NULL, 0);
63
64 if ((fd = open(path, O_RDONLY | O_CLOEXEC)) == -1)
65 return errno;
66
67 if (fstat(fd, &st) == -1) {
68 ret = errno;
69 goto error;
70 }
71 if (!S_ISREG(st.st_mode)) {
72 ret = EOPNOTSUPP;
73 goto error;
74 }
75
76 head = mmap(NULL, (size_t)st.st_size, PROT_READ, MAP_FILE|MAP_PRIVATE,
77 fd, (off_t)0);
78 if (head == MAP_FAILED) {
79 ret = errno;
80 goto error;
81 }
82 _region_init(r, head, (size_t)st.st_size);
83
84error:
85 (void)close(fd);
86 return ret;
87}
88
89void
90_citrus_unmap_file(struct _citrus_region *r)
91{
92
93 _DIAGASSERT(r != NULL);
94
95 if (_region_head(r) != NULL) {
96 (void)munmap(_region_head(r), _region_size(r));
97 _region_init(r, NULL, 0);
98 }
99}
