30#ifndef _CITRUS_PROP_H_
31#define _CITRUS_PROP_H_
32
33typedef enum {
34 _CITRUS_PROP_BOOL = 0,
35 _CITRUS_PROP_STR = 1,
36 _CITRUS_PROP_CHR = 2,
37 _CITRUS_PROP_NUM = 3,
38} _citrus_prop_type_t;
39
40typedef struct _citrus_prop_hint_t _citrus_prop_hint_t;
41
42#define _CITRUS_PROP_CB0_T(_func_, _type_) \
43typedef int (*_citrus_prop_##_func_##_cb_func_t) \
44 (void * __restrict, const char *, _type_); \
45typedef struct { \
46 _citrus_prop_##_func_##_cb_func_t func; \
47} _citrus_prop_##_func_##_cb_t;
48_CITRUS_PROP_CB0_T(boolean, int)
49_CITRUS_PROP_CB0_T(str, const char *)
50#undef _CITRUS_PROP_CB0_T
51
52#define _CITRUS_PROP_CB1_T(_func_, _type_) \
53typedef int (*_citrus_prop_##_func_##_cb_func_t) \
54 (void * __restrict, const char *, _type_, _type_); \
55typedef struct { \
56 _citrus_prop_##_func_##_cb_func_t func; \
57} _citrus_prop_##_func_##_cb_t;
58_CITRUS_PROP_CB1_T(chr, int)
59_CITRUS_PROP_CB1_T(num, uint64_t)
60#undef _CITRUS_PROP_CB1_T
61
62struct _citrus_prop_hint_t {
63 const char *name;
64 _citrus_prop_type_t type;
65#define _CITRUS_PROP_CB_T_OPS(_name_) \
66 _citrus_prop_##_name_##_cb_t _name_
67 union {
68 _CITRUS_PROP_CB_T_OPS(boolean);
69 _CITRUS_PROP_CB_T_OPS(str);
70 _CITRUS_PROP_CB_T_OPS(chr);
71 _CITRUS_PROP_CB_T_OPS(num);
72 } cb;
73};
74
75#define _CITRUS_PROP_HINT_BOOL(name, cb) \
76 { name, _CITRUS_PROP_BOOL, { .boolean = { cb } } }
77#define _CITRUS_PROP_HINT_STR(name, cb) \
78 { name, _CITRUS_PROP_STR, { .str = { cb } } }
79#define _CITRUS_PROP_HINT_CHR(name, cb) \
80 { name, _CITRUS_PROP_CHR, { .chr = { cb } } }
81#define _CITRUS_PROP_HINT_NUM(name, cb) \
82 { name, _CITRUS_PROP_NUM, { .num = { cb } } }
83#define _CITRUS_PROP_HINT_END \
84 { .name = NULL }
85
86__BEGIN_DECLS
87int _citrus_prop_parse_variable(const _citrus_prop_hint_t * __restrict,
88 void * __restrict, const void *, size_t);
89__END_DECLS
90
91#endif /* !_CITRUS_PROP_H_ */
