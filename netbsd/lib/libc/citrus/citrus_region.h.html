30#ifndef _CITRUS_REGION_H_
31#define _CITRUS_REGION_H_
32
33#if HAVE_NBTOOL_CONFIG_H
34#include "nbtool_config.h"
35#endif
36
37struct _citrus_region {
38/* private: */
39 void *r_head;
40 size_t r_size;
41};
42
43static __inline void
44_citrus_region_init(struct _citrus_region *r, void *h, size_t sz)
45{
46 _DIAGASSERT(r);
47 r->r_head = h;
48 r->r_size = sz;
49}
50
51static __inline void *
52_citrus_region_head(const struct _citrus_region *r)
53{
54 return r->r_head;
55}
56
57static __inline size_t
58_citrus_region_size(const struct _citrus_region *r)
59{
60 return r->r_size;
61}
62
63static __inline int
64_citrus_region_check(const struct _citrus_region *r, size_t ofs, size_t sz)
65{
66 return r->r_size >= ofs + sz ? 0 : -1;
67}
68
69static __inline void *
70_citrus_region_offset(const struct _citrus_region *r, size_t pos)
71{
72 return (void *)((uint8_t *)r->r_head + pos);
73}
74
75static __inline uint8_t
76_citrus_region_peek8(const struct _citrus_region *r, size_t pos)
77{
78 return *(uint8_t *)_citrus_region_offset(r, pos);
79}
80
81static __inline uint16_t
82_citrus_region_peek16(const struct _citrus_region *r, size_t pos)
83{
84 uint16_t val;
85 memcpy(&val, _citrus_region_offset(r, pos), (size_t)2);
86 return val;
87}
88
89static __inline uint32_t
90_citrus_region_peek32(const struct _citrus_region *r, size_t pos)
91{
92 uint32_t val;
93 memcpy(&val, _citrus_region_offset(r, pos), (size_t)4);
94 return val;
95}
96
97static __inline int
98_citrus_region_get_subregion(struct _citrus_region *subr,
99 const struct _citrus_region *r,
100 size_t ofs, size_t sz)
101{
102 if (_citrus_region_check(r, ofs, sz))
103 return -1;
104 _citrus_region_init(subr, _citrus_region_offset(r, ofs), sz);
105 return 0;
106}
