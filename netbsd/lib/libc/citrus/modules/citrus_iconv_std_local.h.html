29#ifndef _CITRUS_ICONV_STD_LOCAL_H_
30#define _CITRUS_ICONV_STD_LOCAL_H_
31
32/*
33 * encoding
34 */
35struct _citrus_iconv_std_encoding {
36 struct _citrus_stdenc *se_handle;
37 void *se_ps;
38 void *se_pssaved;
39};
40
41/*
42 * dst
43 */
44struct _citrus_iconv_std_dst {
45 TAILQ_ENTRY(_citrus_iconv_std_dst) sd_entry;
46 _citrus_csid_t sd_csid;
47 unsigned long sd_norm;
48 struct _citrus_csmapper *sd_mapper;
49};
50TAILQ_HEAD(_citrus_iconv_std_dst_list, _citrus_iconv_std_dst);
51
52/*
53 * src
54 */
55struct _citrus_iconv_std_src {
56 TAILQ_ENTRY(_citrus_iconv_std_src) ss_entry;
57 _citrus_csid_t ss_csid;
58 struct _citrus_iconv_std_dst_list ss_dsts;
59};
60TAILQ_HEAD(_citrus_iconv_std_src_list, _citrus_iconv_std_src);
61
62/*
63 * iconv_std handle
64 */
65struct _citrus_iconv_std_shared {
66 struct _citrus_stdenc *is_src_encoding;
67 struct _citrus_stdenc *is_dst_encoding;
68 struct _citrus_iconv_std_src_list is_srcs;
69 int is_use_invalid;
70 _citrus_wc_t is_invalid;
71};
72
73/*
74 * iconv_std context
75 */
76struct _citrus_iconv_std_context {
77 struct _citrus_iconv_std_encoding sc_src_encoding;
78 struct _citrus_iconv_std_encoding sc_dst_encoding;
79};

81#endif
