55int
56_citrus_mapper_none_mapper_getops(struct _citrus_mapper_ops *ops,
57 size_t lenops, uint32_t expected_version)
58{
59 if (expected_version<_CITRUS_MAPPER_ABI_VERSION || lenops<sizeof(*ops))
60 return (EINVAL);
61
62 memcpy(ops, &_citrus_mapper_none_mapper_ops,
63 sizeof(_citrus_mapper_none_mapper_ops));
64
65 return 0;
66}

68static int
69/*ARGSUSED*/
70_citrus_mapper_none_mapper_init(struct _citrus_mapper_area *__restrict ma,
71 struct _citrus_mapper * __restrict cm,
72 const char * __restrict dir,
73 const void * __restrict var, size_t lenvar,
74 struct _citrus_mapper_traits * __restrict mt,
75 size_t lenmt)
76{
77
78 _DIAGASSERT(cm && dir && mt);
79
80 if (lenmt<sizeof(*mt))
81 return (EINVAL);
82
83 cm->cm_closure = NULL;
84 mt->mt_src_max = mt->mt_dst_max = 1; /* 1:1 converter */
85 mt->mt_state_size = 0; /* stateless */
86
87 return 0;
88}
