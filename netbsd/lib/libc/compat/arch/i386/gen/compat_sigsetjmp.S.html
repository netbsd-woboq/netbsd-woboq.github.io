42ENTRY(sigsetjmp)
43 movl 4(%esp),%ecx
44 movl 8(%esp),%eax
45 movl %eax,28(%ecx)
46 testl %eax,%eax
47 jz 2f
48
49 PIC_PROLOGUE
50 pushl $0
51#ifdef __PIC__
52 call PIC_PLT(_C_LABEL(sigblock))
53#else
54 call _C_LABEL(sigblock)
55#endif
56 addl $4,%esp
57 PIC_EPILOGUE
58
59 movl 4(%esp),%ecx
60 movl %eax,24(%ecx)
612: movl 0(%esp),%edx
62 movl %edx, 0(%ecx)
63 movl %ebx, 4(%ecx)
64 movl %esp, 8(%ecx)
65 movl %ebp,12(%ecx)
66 movl %esi,16(%ecx)
67 movl %edi,20(%ecx)
68 xorl %eax,%eax
69 ret
70END(sigsetjmp)
71
72ENTRY(siglongjmp)
73 movl 4(%esp),%edx
74 cmpl $0,28(%edx)
75 jz 2f
76
77 PIC_PROLOGUE
78 pushl 24(%edx)
79#ifdef __PIC__
80 call PIC_PLT(_C_LABEL(sigsetmask))
81#else
82 call _C_LABEL(sigsetmask)
83#endif
84 addl $4,%esp
85 PIC_EPILOGUE
86
872: movl 4(%esp),%edx
88 movl 8(%esp),%eax
89 movl 0(%edx),%ecx
90 movl 4(%edx),%ebx
91 movl 8(%edx),%esp
92 movl 12(%edx),%ebp
93 movl 16(%edx),%esi
94 movl 20(%edx),%edi
95 testl %eax,%eax
96 jnz 3f
97 incl %eax
983: movl %ecx,0(%esp)
99 ret
100END(siglongjmp)
