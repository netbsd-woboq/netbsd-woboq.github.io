44WARN_REFERENCES(vfork, \
45 "warning: reference to compatibility vfork(); include <unistd.h> for correct reference")
46
47/*
48 * pid = vfork();
49 *
50 * %edx == 0 in parent process, %edx == 1 in child process.
51 * %eax == pid of child in parent, %eax == pid of parent in child.
52 *
53 */
54ENTRY(vfork)
55 popl %ecx /* my rta into ecx */
56 SYSTRAP(vfork)
57 jc err
58 decl %edx
59 andl %edx,%eax
60 jmp *%ecx
61err:
62 pushl %ecx
63#ifdef __PIC__
64 PIC_PROLOGUE
65 mov PIC_GOT(CERROR), %ecx
66 PIC_EPILOGUE
67 jmp *%ecx
68#else
69 jmp CERROR
70#endif
71END(vfork)
