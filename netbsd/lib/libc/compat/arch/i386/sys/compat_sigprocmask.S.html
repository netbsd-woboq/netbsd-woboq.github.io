42#include "SYS.h"
43
44WARN_REFERENCES(sigprocmask, \
45 "warning: reference to compatibility sigprocmask(); include <signal.h> for correct reference")
46
47ENTRY(sigprocmask)
48 movl 8(%esp),%ecx # fetch new sigset pointer
49 testl %ecx,%ecx # check new sigset pointer
50 jnz 1f # if not null, indirect
51/* movl $0,8(%esp) # null mask pointer: block empty set */
52 movl $1,4(%esp) # SIG_BLOCK
53 jmp 2f
541: movl (%ecx),%ecx # fetch indirect ...
55 movl %ecx,8(%esp) # to new mask arg
562: SYSTRAP(compat_13_sigprocmask13)
57 jc err
58 movl 12(%esp),%ecx # fetch old mask requested
59 testl %ecx,%ecx # test if old mask requested
60 jz out
61 movl %eax,(%ecx) # store old mask
62out:
63 xorl %eax,%eax
64 ret
65err:
66#ifdef __PIC__
67 PIC_PROLOGUE
68 mov PIC_GOT(CERROR), %ecx
69 PIC_EPILOGUE
70 jmp *%ecx
71#else
72 jmp CERROR
73#endif
74END(sigprocmask)
