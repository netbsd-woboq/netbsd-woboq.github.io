38#if defined(SYSLIBC_SCCS) && !defined(lint)

40#endif /* SYSLIBC_SCCS and not lint */
41
42#include "SYS.h"
43
44WARN_REFERENCES(sigsuspend, \
45 "warning: reference to compatibility sigsuspend(); include <signal.h> for correct reference")
46
47ENTRY(sigsuspend)
48 movl (%rdi),%edi # indirect to mask arg
49 SYSTRAP(compat_13_sigsuspend13)
50 jc err
51 xorl %eax,%eax # shouldn t happen
52 ret
53err:
54#ifdef __PIC__
55 movq PIC_GOT(CERROR), %rcx
56 jmp *%rcx
57#else
58 jmp CERROR
59#endif
60END(sigsuspend)
