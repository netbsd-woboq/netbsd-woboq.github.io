61int
62utime(const char *path, const struct utimbuf50 *times50)
63{
64 struct timeval tv[2], *tvp;
65
66 _DIAGASSERT(path != NULL);
67
68 if (times50 == NULL)
69 tvp = NULL;
70 else {
71 tv[0].tv_sec = times50->actime;
72 tv[1].tv_sec = times50->modtime;
73 tv[0].tv_usec = tv[1].tv_usec = 0;
74 tvp = tv;
75 }
76 return __utimes50(path, tvp);
77}
