32#ifndef _COMPAT_FSTYPES_H_
33#define _COMPAT_FSTYPES_H_
34
35#include <stddef.h>
36
37/* Old, fixed size filehandle structures (used upto (including) 3.x) */
38struct compat_30_fid {
39 unsigned short fid30_len;
40 unsigned short fid30_reserved;
41 char fid30_data[16];
42};
43struct compat_30_fhandle {
44 fsid_t fh_fsid;
45 struct compat_30_fid fh_fid;
46};
47
48#define FHANDLE30_SIZE sizeof(struct compat_30_fhandle) /* 28 */
49
50#endif /* _COMPAT_FSTYPES_H_ */
