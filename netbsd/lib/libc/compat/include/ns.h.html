34#ifndef _NETNS_NS_H_
35#define _NETNS_NS_H_
36
37/*
38 * NS addressing
39 */
40union ns_host {
41 u_int8_t c_host[6];
42 u_int16_t s_host[3];
43};
44
45union ns_net {
46 u_int8_t c_net[4];
47 u_int16_t s_net[2];
48};
49
50union ns_net_u {
51 union ns_net net_e;
52 u_int32_t long_e;
53};
54
55struct ns_addr {
56 union ns_net x_net;
57 union ns_host x_host;
58 u_int16_t x_port;
59};
60
61__BEGIN_DECLS
62extern struct ns_addr ns_addr __P((const char *));
63extern char *ns_ntoa __P((struct ns_addr));
64__END_DECLS
65
66#endif /* !_NETNS_NS_H_ */
