39#ifndef _COMPAT_UTMP_H_
40#define _COMPAT_UTMP_H_
41
42struct utmp50 {
43 char ut_line[UT_LINESIZE];
44 char ut_name[UT_NAMESIZE];
45 char ut_host[UT_HOSTSIZE];
46 int32_t ut_time;
47};
48
49__BEGIN_DECLS
50static __inline void
51utmp_to_utmp50(const struct utmp *ut, struct utmp50 *ut50)
52{
53 (void)memcpy(ut50, ut, sizeof(*ut50));
54 ut50->ut_time = (int32_t)ut->ut_time;
55}
56
57static __inline void
58utmp50_to_utmp(const struct utmp50 *ut50, struct utmp *ut)
59{
60 (void)memcpy(ut, ut50, sizeof(*ut50));
61 ut->ut_time = ut50->ut_time;
62}
63
64struct utmp50 *getutent(void);
65struct utmp *__getutent50(void);
66__END_DECLS
67
68#endif /* !_COMPAT_UTMP_H_ */
