43__warn_references(setlocale,
44 "warning: reference to compatibility setlocale();"
45 "include <locale.h> for correct reference")

47/*
48 * Preparation for the future import of multibyte locale.
49 * This function will ensure binary compatibility for old executables.
50 */
51char *
52compat_setlocale(int category, const char *locale) __RENAME(setlocale);
53
54char *
55compat_setlocale(int category, const char *locale)
56{
57
58 /* locale may be NULL */
59
60 __mb_len_max_runtime = 1;
61 return __setlocale(category, locale);
62}
