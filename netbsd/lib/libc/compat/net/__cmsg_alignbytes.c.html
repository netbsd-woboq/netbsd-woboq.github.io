35#include "namespace.h"
36#include <sys/types.h>
37#include <sys/param.h>
38#include <sys/sysctl.h>
39#include <sys/socket.h>
40
41int __cmsg_alignbytes(void);
42
43int
44__cmsg_alignbytes(void)
45{
46 static int alignbytes = -1;
47#ifdef HW_ALIGNBYTES
48 int mib[2];
49 size_t len;
50 int ret;
51#endif
52
53 if (alignbytes > 0)
54 return alignbytes;
55
56#ifdef HW_ALIGNBYTES
57 mib[0] = CTL_HW;
58 mib[1] = HW_ALIGNBYTES;
59 len = sizeof(alignbytes);
60 ret = sysctl(mib, (u_int) (sizeof(mib) / sizeof(mib[0])),
61 (void *)&alignbytes, &len, NULL, (size_t)0);
62 if (ret >= 0 && alignbytes >= 0)
63 return alignbytes;
64#endif
65 /* last resort */
66 alignbytes = ALIGNBYTES;
67 return alignbytes;
68}
