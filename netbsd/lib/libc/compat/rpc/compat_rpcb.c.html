63enum clnt_stat
64rpcb_rmtcall(const struct netconfig *nc,
65 const char *name, const rpcprog_t prog, const rpcvers_t vers,
66 const rpcproc_t proc, const xdrproc_t inproc, const char *inbuf,
67 const xdrproc_t outproc, caddr_t outbuf,
68 const struct timeval50 tout50, const struct netbuf *nb)
69{
70 struct timeval tout;
71 timeval50_to_timeval(&tout50, &tout);
72 return __rpcb_rmtcall50(nc, name, prog, vers, proc, inproc, inbuf,
73 outproc, outbuf, tout, nb);
74}
75
76bool_t
77rpcb_gettime(const char *name, int32_t *t50)
78{
79 time_t t;
80 bool_t rv = __rpcb_gettime50(name, &t);
81 *t50 = (int32_t)t;
82 return rv;
83}
