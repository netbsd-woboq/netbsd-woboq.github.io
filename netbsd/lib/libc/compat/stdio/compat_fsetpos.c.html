58/*
59 * fsetpos: like fseek.
60 */
61int
62fsetpos(FILE * __restrict iop, const off_t * __restrict pos)
63{
64 _DIAGASSERT(iop != NULL);
65 _DIAGASSERT(pos != NULL);
66
67 return fseeko(iop, *pos, SEEK_SET) == (off_t)-1;
68}
