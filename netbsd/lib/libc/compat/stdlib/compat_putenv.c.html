56/*
57 * putenv(name) --
58 * This version implicitly copies the string for compatibility.
59 */
60int
61putenv(char *name)
62{
63 size_t l_name;
64 char *copy;
65 int rv;
66
67 _DIAGASSERT(name != NULL);
68
69 l_name = __envvarnamelen(name, true);
70 if (l_name == 0) {
71 errno = EINVAL;
72 return -1;
73 }
74
75 if ((copy = strdup(name)) == NULL)
76 return -1;
77 copy[l_name++] = '\0';
78
79 rv = setenv(copy, copy + l_name, 1);
80
81 free(copy);
82 return rv;
83}
