46#ifdef __weak_alias
47__weak_alias(initstate,_initstate)
48__weak_alias(srandom,_srandom)

51__warn_references(initstate,
52 "warning: reference to compatibility initstate();"
53 " include <stdlib.h> for correct reference")
54__warn_references(srandom,
55 "warning: reference to compatibility srandom();"
56 " include <stdlib.h> for correct reference")
57
58char *
59initstate(unsigned long seed, char * buf, size_t len) {
60 return __initstate60((unsigned int)seed, buf, len);
61}
62
63void
64srandom(unsigned long seed) {
65 __srandom60((unsigned int)seed);
66}
