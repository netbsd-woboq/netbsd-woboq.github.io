49/*
50 * Convert old __lwp_park() call to new
51 */
52int
53___lwp_park50(const struct timespec *ts, lwpid_t unpark, const void *hint,
54 const void *unparkhint)
55{
56 struct timespec ts1, *tsp = &ts1;
57 if (ts)
58 ts1 = *ts;
59 else
60 tsp = NULL;
61 return ___lwp_park60(CLOCK_REALTIME, TIMER_ABSTIME, tsp, unpark,
62 hint, unparkhint);
63}
