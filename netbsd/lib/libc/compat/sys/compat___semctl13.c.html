57/*
58 * Copy timeout to local variable and call the syscall.
59 */
60int
61__semctl13(int semid, int semnum, int cmd, ...)
62{
63 va_list ap;
64 union __semun semun;
65 struct semid_ds13 *ds13;
66 struct semid_ds ds;
67 int error;
68
69 va_start(ap, cmd);
70 switch (cmd) {
71 case IPC_SET:
72 case IPC_STAT:
73 case GETALL:
74 case SETVAL:
75 case SETALL:
76#ifdef __lint__
77 memcpy(&semun, &ap, sizeof(semun));
78#else
79 semun = va_arg(ap, union __semun);
80#endif
81 break;
82 default:
83 break;
84 }
85 va_end(ap);
86
87 switch (cmd) {
88 case IPC_SET:
89 case IPC_STAT:
90 ds13 = (void *)semun.buf;
91 semun.buf = &ds;
92 if (cmd == IPC_SET)
93 __semid_ds13_to_native(ds13, &ds);
94 break;
95 default:
96 ds13 = NULL;
97 break;
98 }
99
100
101 error = ____semctl50(semid, semnum, cmd, &semun);
102 if (error)
103 return error;
104
105 if (cmd == IPC_STAT)
106 __native_to_semid_ds13(&ds, ds13);
107 return 0;
108}
