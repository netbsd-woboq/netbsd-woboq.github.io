56/*
57 * Copy timeout to local variable and call the syscall.
58 */
59int
60aio_suspend(const struct aiocb * const list[], int nent,
61 const struct timespec50 * ts50)
62{
63 struct timespec ts, *tsp;
64
65 if (ts50)
66 timespec50_to_timespec(ts50, tsp = &ts);
67 else
68 tsp = NULL;
69 return __aio_suspend50(list, nent, tsp);
70}
