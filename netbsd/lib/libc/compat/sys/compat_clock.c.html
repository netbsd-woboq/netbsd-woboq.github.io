60int
61clock_settime(clockid_t clockid, const struct timespec50 * tim50)
62{
63 struct timespec tim, *timp;
64 int error;
65
66 if (tim50)
67 timespec50_to_timespec(tim50, timp = &tim);
68 else
69 timp = NULL;
70 error = __clock_settime50(clockid, timp);
71 if (error)
72 return error;
73 return 0;
74}
75
76int
77clock_gettime(clockid_t clockid, struct timespec50 *tim50)
78{
79 struct timespec tim, *timp = tim50 ? &tim : NULL;
80 int error = __clock_gettime50(clockid, timp);
81 if (error)
82 return error;
83 if (tim50)
84 timespec_to_timespec50(timp, tim50);
85 return 0;
86}
87
88int
89clock_getres(clockid_t clockid, struct timespec50 *tim50)
90{
91 struct timespec tim, *timp = tim50 ? &tim : NULL;
92 int error = __clock_getres50(clockid, timp);
93 if (error)
94 return error;
95 if (tim50)
96 timespec_to_timespec50(timp, tim50);
97 return 0;
98}
