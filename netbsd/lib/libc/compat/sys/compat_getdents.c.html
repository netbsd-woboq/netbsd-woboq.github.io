47/*
48 * libc12 compatible getdents routine.
49 */
50int
51getdents(int fd, char *buf, size_t nbytes)
52{
53 struct dirent *ndp, *nndp, *endp;
54 struct dirent12 *odp;
55 ino_t ino;
56 int rv;
57 size_t len;
58
59 if ((rv = __getdents30(fd, buf, nbytes)) == -1)
60 return rv;
61
62 odp = (struct dirent12 *)(void *)buf;
63 ndp = (struct dirent *)(void *)buf;
64 endp = (struct dirent *)(void *)&buf[rv];
65
66 /*
67 * In-place conversion. This works because odp
68 * is smaller than ndp, but it has to be done
69 * in the right sequence.
70 */
71 for (; ndp < endp; ndp = nndp) {
72 nndp = _DIRENT_NEXT(ndp);
73 /* XXX: avoid unaligned 64-bit access on sparc64 */
74 /* XXX: does this work? */
75 memcpy(&ino, &ndp->d_ino, sizeof(ino_t));
76 odp->d_ino = (uint32_t)ino;
77 if (ndp->d_namlen >= sizeof(odp->d_name))
78 odp->d_namlen = sizeof(odp->d_name) - 1;
79 else
80 odp->d_namlen = (u_int8_t)ndp->d_namlen;
81 odp->d_type = ndp->d_type;
82 (void)memcpy(odp->d_name, ndp->d_name, (size_t)odp->d_namlen);
83 odp->d_name[odp->d_namlen] = '\0';
84 len = _DIRENT_SIZE(odp);
85 _DIAGASSERT(__type_fit(uint16_t, len));
86 odp->d_reclen = (uint16_t)len;
87 odp = _DIRENT_NEXT(odp);
88 }
89 ptrdiff_t td = (((char *)(void *)odp) - buf);
90 _DIAGASSERT(__type_fit(int, td));
91 return (int)td;
92}
