40#define __LIBC12_SOURCE__
41#include "namespace.h"
42#include <sys/types.h>
43#include <dirent.h>
44#include <compat/include/dirent.h>
45#include <unistd.h>
46
47__warn_references(getdirentries,
48 "reference to compatibility-only getdirentries(); this will break; use getdents() or readdir() instead")
49
50int
51getdirentries(int fd, char *buf, int nbytes, long *basep)
52{
53 *basep = (long)lseek(fd, (off_t)0, SEEK_CUR);
54 return getdents(fd, buf, (size_t)nbytes);
55}
