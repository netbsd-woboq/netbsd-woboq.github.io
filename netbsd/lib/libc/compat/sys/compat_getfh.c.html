51int getfh(const char *path, struct compat_30_fhandle *fhp);

53/*
54 * Convert old getfh() call to new calling convention
55 */
56int
57getfh(const char *path, struct compat_30_fhandle *fhp)
58{
59 size_t fh_size = sizeof(*fhp);
60 int ret;
61
62 ret = __getfh30(path, (void*)fhp, &fh_size);
63 if (ret != 0) {
64 return ret;
65 }
66 if (fh_size != FHANDLE30_SIZE) {
67 errno = EINVAL;
68 return -1;
69 }
70 return 0;
71}
