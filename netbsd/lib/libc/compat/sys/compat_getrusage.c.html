45__warn_references(getrusage,
46 "warning: reference to compatibility getrusage(); include <sys/resource.h> to generate correct reference")
47
48__strong_alias(getrusage, __compat_getrusage)
49
50void __rusage_to_rusage50(const struct rusage *, struct rusage50 *);
51
52void
53__rusage_to_rusage50(const struct rusage *ru, struct rusage50 *ru50)
54{
55 (void)memcpy(&ru50->ru_first, &ru->ru_first,
56 (size_t)((char *)(void *)&ru50->ru_last -
57 (char *)(void *)&ru50->ru_first));
58 ru50->ru_maxrss = ru->ru_maxrss;
59 timeval_to_timeval50(&ru->ru_utime, &ru50->ru_utime);
60 timeval_to_timeval50(&ru->ru_stime, &ru50->ru_stime);
61}
62
63/*
64 * libc12 compatible getrusage routine.
65 */
66int
67__compat_getrusage(int who, struct rusage50 *ru50)
68{
69 struct rusage ru;
70 int rv;
71
72 if ((rv = __getrusage50(who, &ru)) == -1)
73 return rv;
74 __rusage_to_rusage50(&ru, ru50);
75 return rv;
76}
