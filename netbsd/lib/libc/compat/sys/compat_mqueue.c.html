51 "warning: reference to compatibility mq_timedreceive(); include <mqueue.h> to generate correct reference")

57__weak_alias(mq_timedsend, _mq_timedsend)

61
62ssize_t
63mq_timedreceive(mqd_t mq, char * __restrict name, size_t len,
64 unsigned * __restrict buf, const struct timespec50 * __restrict ts50)
65{
66 struct timespec ts, *tsp;
67
68 if (ts50)
69 timespec50_to_timespec(ts50, tsp = &ts);
70 else
71 tsp = NULL;
72 return __mq_timedreceive50(mq, name, len, buf, tsp);
73}
74
75int
76mq_timedsend(mqd_t mq, const char *name, size_t len,
77 unsigned buf, const struct timespec50 *ts50)
78{
79 struct timespec ts, *tsp;
80
81 if (ts50)
82 timespec50_to_timespec(ts50, tsp = &ts);
83 else
84 tsp = NULL;
85 return __mq_timedsend50(mq, name, len, buf, tsp);
86}
