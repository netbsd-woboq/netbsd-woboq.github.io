56/*
57 * Copy timeout to local variable and call the syscall.
58 */
59int
60nanosleep(const struct timespec50 *ts50, struct timespec50 *rts50)
61{
62 struct timespec ts, *tsp;
63 struct timespec rts, *rtsp;
64 int error;
65
66 rtsp = rts50 ? &rts : NULL;
67 if (ts50)
68 timespec50_to_timespec(ts50, tsp = &ts);
69 else
70 tsp = NULL;
71 error = __nanosleep50(tsp, rtsp);
72 if (error)
73 return error;
74 if (rts50)
75 timespec_to_timespec50(rtsp, rts50);
76 return 0;
77}
