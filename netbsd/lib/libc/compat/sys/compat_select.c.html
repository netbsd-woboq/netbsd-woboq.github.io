63int
64pollts(struct pollfd * __restrict fds, nfds_t nfds,
65 const struct timespec50 * __restrict tim50,
66 const sigset_t * __restrict sigmask)
67{
68 struct timespec tim, *timp;
69
70 if (tim50)
71 timespec50_to_timespec(tim50, timp = &tim);
72 else
73 timp = NULL;
74 return __pollts50(fds, nfds, timp, sigmask);
75}
76
77int
78select(int nfds, fd_set * __restrict readfds, fd_set * __restrict writefds,
79 fd_set * __restrict exceptfds, struct timeval50 * __restrict tim50)
80{
81 struct timeval tim, *timp;
82
83 if (tim50)
84 timeval50_to_timeval(tim50, timp = &tim);
85 else
86 timp = NULL;
87 return __select50(nfds, readfds, writefds, exceptfds, timp);
88}

90int
91pselect(int nfds, fd_set * __restrict readfds, fd_set * __restrict writefds,
92 fd_set * __restrict exceptfds, const struct timespec50 *__restrict tim50,
93 const sigset_t * __restrict sigmask)
94{
95 struct timespec tim, *timp;
96
97 if (tim50)
98 timespec50_to_timespec(tim50, timp = &tim);
99 else
100 timp = NULL;
101 return __pselect50(nfds, readfds, writefds, exceptfds, timp, sigmask);
102}
