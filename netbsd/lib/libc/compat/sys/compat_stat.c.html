55/*
56 * Convert from a new to an old stat structure.
57 */
58
59static void cvtstat(const struct stat *, struct stat12 *);
60
61__strong_alias(stat, __compat_stat)
62__strong_alias(lstat, __compat_lstat)
63__strong_alias(fstat, __compat_fstat)
64
65static void
66cvtstat(const struct stat *st, struct stat12 *ost)
67{
68
69 ost->st_dev = (uint32_t)st->st_dev;
70 ost->st_ino = (uint32_t)st->st_ino;
71 ost->st_mode = st->st_mode;
72 if (st->st_nlink >= (1 << 15))
73 ost->st_nlink = (1 << 15) - 1;
74 else
75 ost->st_nlink = st->st_nlink;
76 ost->st_uid = st->st_uid;
77 ost->st_gid = st->st_gid;
78 ost->st_rdev = (uint32_t)st->st_rdev;
79 ost->st_atimespec.tv_sec = (int32_t)st->st_atimespec.tv_sec;
80 ost->st_atimespec.tv_nsec = st->st_atimespec.tv_nsec;
81 ost->st_mtimespec.tv_sec = (int32_t)st->st_mtimespec.tv_sec;
82 ost->st_mtimespec.tv_nsec = st->st_mtimespec.tv_nsec;
83 ost->st_ctimespec.tv_sec = (int32_t)st->st_ctimespec.tv_sec;
84 ost->st_ctimespec.tv_nsec = st->st_ctimespec.tv_nsec;
85 ost->st_size = st->st_size;
86 ost->st_blocks = st->st_blocks;
87 ost->st_blksize = st->st_blksize;
88 ost->st_flags = st->st_flags;
89 ost->st_gen = st->st_gen;
90}
91
92int
93__compat_stat(const char *file, struct stat12 *ost)
94{
95 struct stat nst;
96 int ret;
97
98 if ((ret = __stat50(file, &nst)) < 0)
99 return ret;
100 cvtstat(&nst, ost);
101 return ret;
102}
