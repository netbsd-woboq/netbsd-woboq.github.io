59/*
60 * libc12 compatible wait4 routine.
61 */
62pid_t
63wait3(int *status, int options, struct rusage50 *ru50)
64{
65 struct rusage ru;
66 pid_t rv;
67
68 if ((rv = __wait350(status, options, ru50 ? &ru : NULL)) == -1)
69 return rv;
70 if (ru50)
71 __rusage_to_rusage50(&ru, ru50);
72 return rv;
73}
74
75pid_t
76wait4(pid_t wpid, int *status, int options, struct rusage50 *ru50)
77{
78 struct rusage ru;
79 pid_t rv;
80
81 if ((rv = __wait450(wpid, status, options, ru50 ? &ru : NULL)) == -1)
82 return rv;
83 if (ru50)
84 __rusage_to_rusage50(&ru, ru50);
85 return rv;
86}
