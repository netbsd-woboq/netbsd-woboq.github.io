58DB *
59dbopen(const char *fname, int flags, mode_t mode, DBTYPE type,
60 const void *openinfo)
61{
62
63#define DB_FLAGS (DB_LOCK | DB_SHMEM | DB_TXN)
64#define USE_OPEN_FLAGS \
65 (O_CREAT | O_EXCL | O_EXLOCK | O_NONBLOCK | O_RDONLY | \
66 O_RDWR | O_SHLOCK | O_TRUNC | O_CLOEXEC)
67
68 if ((flags & ~(USE_OPEN_FLAGS | DB_FLAGS)) == 0)
69 switch (type) {
70 case DB_BTREE:
71 return (__bt_open(fname, flags & USE_OPEN_FLAGS,
72 mode, openinfo, (int)(flags & DB_FLAGS)));
73 case DB_HASH:
74 return (__hash_open(fname, flags & USE_OPEN_FLAGS,
75 mode, openinfo, (int)(flags & DB_FLAGS)));
76 case DB_RECNO:
77 return (__rec_open(fname, flags & USE_OPEN_FLAGS,
78 mode, openinfo, (int)(flags & DB_FLAGS)));
79 }
80 errno = EINVAL;
81 return (NULL);
82}

84static int
85__dberr(void)
86{
87 return (RET_ERROR);
88}

90/*
91 * __DBPANIC -- Stop.
92 *
93 * Parameters:
94 * dbp: pointer to the DB structure.
95 */
96void
97__dbpanic(DB *dbp)
98{
99 /* The only thing that can succeed is a close. */
100 dbp->del = (int (*)(const struct __db *, const DBT*, u_int))__dberr;
101 dbp->fd = (int (*)(const struct __db *))__dberr;
102 dbp->get = (int (*)(const struct __db *, const DBT*, DBT *, u_int))__dberr;
103 dbp->put = (int (*)(const struct __db *, DBT *, const DBT *, u_int))__dberr;
104 dbp->seq = (int (*)(const struct __db *, DBT *, DBT *, u_int))__dberr;
105 dbp->sync = (int (*)(const struct __db *, u_int))__dberr;
106}
