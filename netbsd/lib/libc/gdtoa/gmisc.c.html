36 void
37#ifdef KR_headers
38rshift(b, k) Bigint *b; int k;
39#else
40rshift(Bigint *b, int k)
41#endif
42{
43 ULong *x, *x1, *xe, y;
44 int n;
45
46 x = x1 = b->x;
47 n = (unsigned int)k >> kshift;
48 if (n < b->wds) {
49 xe = x + b->wds;
50 x += n;
51 if (k &= kmask) {
52 n = ULbits - k;
53 y = *x++ >> k;
54 while(x < xe) {
55 *x1++ = (y | (*x << n)) & ALL_ON;
56 y = *x++ >> k;
57 }
58 if ((*x1 = y) !=0)
59 x1++;
60 }
61 else
62 while(x < xe)
63 *x1++ = *x++;
64 }
65
66 ptrdiff_t td = x1 - b->x;
67 _DIAGASSERT(__type_fit(int, td));
68 if ((b->wds = (int)td) == 0)
69 b->x[0] = 0;
70 }
71
72 int
73#ifdef KR_headers
74trailz(b) CONST Bigint *b;
75#else
76trailz(CONST Bigint *b)
77#endif
78{
79 ULong L;
80 CONST ULong *x, *xe;
81 int n = 0;
82
83 x = b->x;
84 xe = x + b->wds;
85 for(n = 0; x < xe && !*x; x++)
86 n += ULbits;
87 if (x < xe) {
88 L = *x;
89 n += lo0bits(&L);
90 }
91 return n;
92 }
