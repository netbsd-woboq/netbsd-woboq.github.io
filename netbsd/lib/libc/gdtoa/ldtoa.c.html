46/*
47 * ldtoa() is a wrapper for gdtoa() that makes it smell like dtoa(),
48 * except that the floating point argument is passed by reference.
49 * When dtoa() is passed a NaN or infinity, it sets expt to 9999.
50 * However, a long double could have a valid exponent of 9999, so we
51 * use INT_MAX in ldtoa() instead.
52 */
53char *
54ldtoa(long double *ld, int mode, int ndigits, int *decpt, int *sign, char **rve)
55{
56#ifdef EXT_EXPBITS
57 static FPI fpi = {
58 LDBL_MANT_DIG, /* nbits */
59 LDBL_MIN_EXP - LDBL_MANT_DIG, /* emin */
60 LDBL_MAX_EXP - LDBL_MANT_DIG, /* emax */
61 FPI_Round_near, /* rounding */
62#ifdef Sudden_Underflow /* unused, but correct anyway */
63 1
64#else
65 0
66#endif
67 };
68 int be, kind;
69 char *ret;
70 union ieee_ext_u u;
71 uint32_t bits[(LDBL_MANT_DIG + 31) / 32];
72
73 u.extu_ld = *ld;
74 *sign = u.extu_ext.ext_sign;
75 be = u.extu_ext.ext_exp - (LDBL_MAX_EXP - 1) - (LDBL_MANT_DIG - 1);
76 EXT_TO_ARRAY32(u, bits);
77
78 switch (fpclassify(u.extu_ld)) {
79 case FP_NORMAL:
80 kind = STRTOG_Normal;
81#ifdef LDBL_IMPLICIT_NBIT
82 bits[LDBL_MANT_DIG / 32] |= 1 << ((LDBL_MANT_DIG - 1) % 32);
83#endif /* LDBL_IMPLICIT_NBIT */
84 break;
85 case FP_ZERO:
86 kind = STRTOG_Zero;
87 break;
88 case FP_SUBNORMAL:
89 kind = STRTOG_Denormal;
90 be++;
91 break;
92 case FP_INFINITE:
93 kind = STRTOG_Infinite;
94 break;
95 case FP_NAN:
96 kind = STRTOG_NaN;
97 break;
98 default:
99 abort();
100 }
101
102 ret = gdtoa(&fpi, be, (ULong *)bits, &kind, mode, ndigits, decpt, rve);
103 if (*decpt == -32768)
104 *decpt = INT_MAX;
105 return ret;
106#else
107 return dtoa((double)*ld, mode, ndigits, decpt, sign, rve);
108#endif
109}
