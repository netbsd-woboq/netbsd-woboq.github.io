34#include "gdtoaimp.h"
35
36 Bigint *
37#ifdef KR_headers
38sum(a, b) Bigint *a; Bigint *b;
39#else
40sum(Bigint *a, Bigint *b)
41#endif
42{
43 Bigint *c;
44 ULong carry, *xc, *xa, *xb, *xe, y;
45#ifdef Pack_32
46 ULong z;
47#endif
48
49 if (a->wds < b->wds) {
50 c = b; b = a; a = c;
51 }
52 c = Balloc(a->k);
53 if (c == NULL)
54 return NULL;
55 c->wds = a->wds;
56 carry = 0;
57 xa = a->x;
58 xb = b->x;
59 xc = c->x;
60 xe = xc + b->wds;
61#ifdef Pack_32
62 do {
63 y = (*xa & 0xffff) + (*xb & 0xffff) + carry;
64 carry = (y & 0x10000) >> 16;
65 z = (*xa++ >> 16) + (*xb++ >> 16) + carry;
66 carry = (z & 0x10000) >> 16;
67 Storeinc(xc, z, y);
68 }
69 while(xc < xe);
70 xe += a->wds - b->wds;
71 while(xc < xe) {
72 y = (*xa & 0xffff) + carry;
73 carry = (y & 0x10000) >> 16;
74 z = (*xa++ >> 16) + carry;
75 carry = (z & 0x10000) >> 16;
76 Storeinc(xc, z, y);
77 }
78#else
79 do {
80 y = *xa++ + *xb++ + carry;
81 carry = (y & 0x10000) >> 16;
82 *xc++ = y & 0xffff;
83 }
84 while(xc < xe);
85 xe += a->wds - b->wds;
86 while(xc < xe) {
87 y = *xa++ + carry;
88 carry = (y & 0x10000) >> 16;
89 *xc++ = y & 0xffff;
90 }
91#endif
92 if (carry) {
93 if (c->wds == c->maxwds) {
94 b = Balloc(c->k + 1);
95 if (b == NULL)
96 return NULL;
97 Bcopy(b, c);
98 Bfree(c);
99 c = b;
100 }
101 c->x[c->wds++] = 1;
102 }
103 return c;
104 }
