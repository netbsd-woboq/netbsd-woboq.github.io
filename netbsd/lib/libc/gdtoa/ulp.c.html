34#include "gdtoaimp.h"
35
36 double
37ulp
38#ifdef KR_headers
39 (x) U *x;
40#else
41 (U *x)
42#endif
43{
44 Long L;
45 U a;
46
47 L = (word0(x) & Exp_mask) - (P-1)*Exp_msk1;
48#ifndef Sudden_Underflow
49 if (L > 0) {
50#endif
51#ifdef IBM
52 L |= Exp_msk1 >> 4;
53#endif
54 word0(&a) = L;
55 word1(&a) = 0;
56#ifndef Sudden_Underflow
57 }
58 else {
59 L = (unsigned int)-L >> Exp_shift;
60 if (L < Exp_shift) {
61 word0(&a) = 0x80000 >> L;
62 word1(&a) = 0;
63 }
64 else {
65 word0(&a) = 0;
66 L -= Exp_shift;
67 word1(&a) = L >= 31 ? 1 : 1 << (31 - L);
68 }
69 }
70#endif
71 return dval(&a);
72 }
