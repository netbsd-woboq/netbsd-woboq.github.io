48#define ALPHASORTARG struct dirent **
49#endif

51/*
52 * Alphabetic order comparison routine for those who want it.
53 */
54int
55alphasort(const ALPHASORTARG d1, const ALPHASORTARG d2)
56{
57
58 _DIAGASSERT(d1 != NULL);
59 _DIAGASSERT(d2 != NULL);
60
61 return (strcmp((*(const struct dirent *const *)d1)->d_name,
62 (*(const struct dirent *const *)d2)->d_name));
63}
