48static size_t
49xbasename_r(const char *path, char *buf, size_t buflen)
50{
51 const char *startp, *endp;
52 size_t len;
53
54 /*
55 * If `path' is a null pointer or points to an empty string,
56 * return a pointer to the string ".".
57 */
58 if (path == NULL || *path == '\0') {
59 startp = ".";
60 len = 1;
61 goto out;
62 }
63
64 /* Strip trailing slashes, if any. */
65 endp = path + strlen(path) - 1;
66 while (endp != path && *endp == '/')
67 endp--;
68
69 /* Only slashes -> "/" */
70 if (endp == path && *endp == '/') {
71 startp = "/";
72 len = 1;
73 goto out;
74 }
75
76 /* Now find the beginning of this (final) component. */
77 for (startp = endp; startp > path && *(startp - 1) != '/'; startp--)
78 continue;
79
80 /* ...and copy the result into the result buffer. */
81 len = (endp - startp) + 1 /* last char */;
82out:
83 if (buf != NULL && buflen != 0) {
84 buflen = MIN(len, buflen - 1);
85 memcpy(buf, startp, buflen);
86 buf[buflen] = '\0';
87 }
88 return len;
89}

91#if !HAVE_BASENAME
92
93char *
94basename(char *path) {
95 static char result[PATH_MAX];
96
97 (void)xbasename_r(path, result, sizeof(result));
98 return result;
99}
