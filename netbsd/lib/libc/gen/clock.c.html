47/*
48 * This code is all rather silly because the kernel counts actual
49 * execution time (to usec accuracy) then splits it into user, system and
50 * interrupt based on when clock ticks happen. getrusage apportions the
51 * time based on the number of ticks, and here we are trying to generate
52 * a number which was, traditionally, the number of ticks!
53 *
54 * Due to the way the time is apportioned, this code (and indeed getrusage
55 * itself) are not guaranteed monotonic.
56 */
57
58clock_t
59clock(void)
60{
61 struct rusage ru;
62 clock_t hz = CLOCKS_PER_SEC;
63
64 if (getrusage(RUSAGE_SELF, &ru))
65 return ((clock_t) -1);
66 return (clock_t)((ru.ru_utime.tv_sec + ru.ru_stime.tv_sec) * hz +
67 (ru.ru_utime.tv_usec + ru.ru_stime.tv_usec + 50)
68 / 100 * hz / 10000);
69}
