46#ifdef __weak_alias
47__weak_alias(execle,_execle)
48#endif
49
50int
51execle(const char *name, const char *arg, ...)
52{
53 va_list ap;
54 char **argv, **envp;
55 int i;
56
57 va_start(ap, arg);
58 for (i = 2; va_arg(ap, char *) != NULL; i++)
59 continue;
60 va_end(ap);
61
62 argv = alloca(i * sizeof (char *));
63
64 va_start(ap, arg);
65 argv[0] = __UNCONST(arg);
66 for (i = 1; (argv[i] = va_arg(ap, char *)) != NULL; i++)
67 continue;
68 envp = va_arg(ap, char **);
69 va_end(ap);
70
71 return execve(name, argv, envp);
72}
