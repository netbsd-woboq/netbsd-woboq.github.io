41#ifdef __weak_alias
42__weak_alias(ftok,_ftok)
43#endif
44
45key_t
46ftok(const char *path, int id)
47{
48 struct stat st;
49
50 _DIAGASSERT(path != NULL);
51
52 if (stat(path, &st) < 0)
53 return (key_t)-1;
54
55 return (key_t)
56 ((unsigned int)id << 24 | (st.st_dev & 0xff) << 16 |
57 (st.st_ino & 0xffff));
58}
