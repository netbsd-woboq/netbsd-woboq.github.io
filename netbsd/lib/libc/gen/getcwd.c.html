63/*
64 * char *realpath(const char *path, char *resolved);
65 *
66 * Find the real name of path, by removing all ".", ".." and symlink
67 * components. Returns (resolved) on success, or (NULL) on failure,
68 * in which case the path which caused trouble is left in (resolved).
69 */
70char *
71realpath(const char * __restrict path, char * __restrict resolved)
72{
73 struct stat sb;
74 int idx = 0, nlnk = 0;
75 const char *q;
76 char *p, wbuf[2][MAXPATHLEN], *fres;
77 size_t len;
78 ssize_t n;
79
80 /* POSIX sez we must test for this */
81 if (path == NULL) {
82 errno = EINVAL;
83 return NULL;
84 }
85
86 if (resolved == NULL) {
87 fres = resolved = malloc(MAXPATHLEN);
88 if (resolved == NULL)
89 return NULL;
90 } else
91 fres = NULL;
92
93
94 /*
95 * Build real path one by one with paying an attention to .,
96 * .. and symbolic link.
97 */
98
99 /*
100 * `p' is where we'll put a new component with prepending
101 * a delimiter.
102 */
103 p = resolved;
104
105 if (*path == '\0') {
106 *p = '\0';
107 errno = ENOENT;
108 goto out;
109 }
110
111 /* If relative path, start from current working directory. */
112 if (*path != '/') {
113 /* check for resolved pointer to appease coverity */
114 if (resolved && getcwd(resolved, MAXPATHLEN) == NULL) {
115 p[0] = '.';
116 p[1] = '\0';
117 goto out;
118 }
119 len = strlen(resolved);
120 if (len > 1)
121 p += len;
122 }
123
124loop:
125 /* Skip any slash. */
126 while (*path == '/')
127 path++;
128
129 if (*path == '\0') {
130 if (p == resolved)
131 *p++ = '/';
132 *p = '\0';
133 return resolved;
134 }
135
136 /* Find the end of this component. */
137 q = path;
138 do
139 q++;
140 while (*q != '/' && *q != '\0');
141
142 /* Test . or .. */
143 if (path[0] == '.') {
144 if (q - path == 1) {
145 path = q;
146 goto loop;
147 }
148 if (path[1] == '.' && q - path == 2) {
149 /* Trim the last component. */
150 if (p != resolved)
151 while (*--p != '/')
152 continue;
153 path = q;
154 goto loop;
155 }
156 }
157
158 /* Append this component. */
159 if (p - resolved + 1 + q - path + 1 > MAXPATHLEN) {
160 errno = ENAMETOOLONG;
161 if (p == resolved)
162 *p++ = '/';
163 *p = '\0';
164 goto out;
165 }
166 p[0] = '/';
167 memcpy(&p[1], path,
168 /* LINTED We know q > path. */
169 q - path);
170 p[1 + q - path] = '\0';
171
172 /*
173 * If this component is a symlink, toss it and prepend link
174 * target to unresolved path.
175 */
176 if (lstat(resolved, &sb) == -1)
177 goto out;
178
179 if (S_ISLNK(sb.st_mode)) {
180 if (nlnk++ >= MAXSYMLINKS) {
181 errno = ELOOP;
182 goto out;
183 }
184 n = readlink(resolved, wbuf[idx], sizeof(wbuf[0]) - 1);
185 if (n < 0)
186 goto out;
187 if (n == 0) {
188 errno = ENOENT;
189 goto out;
190 }
191
192 /* Append unresolved path to link target and switch to it. */
193 if (n + (len = strlen(q)) + 1 > sizeof(wbuf[0])) {
194 errno = ENAMETOOLONG;
195 goto out;
196 }
197 memcpy(&wbuf[idx][n], q, len + 1);
198 path = wbuf[idx];
199 idx ^= 1;
200
201 /* If absolute symlink, start from root. */
202 if (*path == '/')
203 p = resolved;
204 goto loop;
205 }
206 if (*q == '/' && !S_ISDIR(sb.st_mode)) {
207 errno = ENOTDIR;
208 goto out;
209 }
210
211 /* Advance both resolved and unresolved path. */
212 p += 1 + q - path;
213 path = q;
214 goto loop;
215out:
216 free(fres);
217 return NULL;
218}

220char *
221__ssp_real(getcwd)(char *pt, size_t size)
222{
223 char *npt;
224
225 /*
226 * If a buffer is specified, the size has to be non-zero.
227 */
228 if (pt != NULL) {
229 if (size == 0) {
230 /* __getcwd(pt, 0) results ERANGE. */
231 errno = EINVAL;
232 return (NULL);
233 }
234 if (__getcwd(pt, size) >= 0)
235 return (pt);
236 return (NULL);
237 }
238
239 /*
240 * If no buffer specified by the user, allocate one as necessary.
241 */
242 size = 1024 >> 1;
243 do {
244 if ((npt = realloc(pt, size <<= 1)) == NULL)
245 break;
246 pt = npt;
247 if (__getcwd(pt, size) >= 0)
248 return (pt);
249 } while (size <= MAXPATHLEN * 4 && errno == ERANGE);
250
251 free(pt);
252 return (NULL);
253}
