41/*
42 * calculate group access list
43 */
44
45#include "namespace.h"
46#include <sys/param.h>
47
48#include <assert.h>
49#include <nsswitch.h>
50#include <stdarg.h>
51#include <unistd.h>
52
53#ifdef __weak_alias
54__weak_alias(getgrouplist,_getgrouplist)
55#endif
56
57int
58getgrouplist(const char *uname, gid_t agroup, gid_t *groups, int *grpcnt)
59{
60 int rv, groupc;
61
62 _DIAGASSERT(uname != NULL);
63 /* groups may be NULL if just sizing when invoked with *grpcnt = 0 */
64 _DIAGASSERT(grpcnt != NULL);
65
66 groupc = 0;
67 rv = getgroupmembership(uname, agroup, groups, *grpcnt, &groupc);
68 *grpcnt = groupc; /* set groupc to the actual # of groups */
69 return rv;
70}
