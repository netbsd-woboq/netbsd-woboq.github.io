56/*
57 * getloadavg() -- Get system load averages.
58 *
59 * Put `nelem' samples into `loadavg' array.
60 * Return number of samples retrieved, or -1 on error.
61 */
62int
63getloadavg(double loadavg[], int nelem)
64{
65 struct loadavg loadinfo;
66 static const int mib[] = { CTL_VM, VM_LOADAVG };
67 size_t size, i;
68
69 _DIAGASSERT(loadavg != NULL);
70 _DIAGASSERT(nelem >= 0);
71
72 size = sizeof(loadinfo);
73 if (sysctl(mib, (u_int)__arraycount(mib), &loadinfo, &size, NULL, 0)
74 == -1)
75 return -1;
76
77 size = MIN((size_t)nelem, __arraycount(loadinfo.ldavg));
78 for (i = 0; i < size; i++)
79 loadavg[i] = (double) loadinfo.ldavg[i] / loadinfo.fscale;
80 return nelem;
81}
