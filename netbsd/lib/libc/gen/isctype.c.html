50#define _RUNE_LOCALE(loc) \
51 ((_RuneLocale *)((loc)->part_impl[(size_t)LC_CTYPE]))
52
53#define _ISCTYPE_FUNC(name, bit) \
54int \
55is##name(int c) \
56{ \
57 return (int)_ctype_tab_[c + 1] & (bit); \
58} \
59int \
60is##name ## _l(int c, locale_t loc) \
61{ \
62 return (int)((_RUNE_LOCALE(loc)->rl_ctype_tab[c + 1]) & (bit)); \
63}
64

78int
79toupper(int c)
80{
81 return (int)_toupper_tab_[c + 1];
82}
83
84int
85toupper_l(int c, locale_t loc)
86{

88}

90int
91tolower(int c)
92{
93 return (int)_tolower_tab_[c + 1];
94}
95
96int
97tolower_l(int c, locale_t loc)
98{
99 return (int)(_RUNE_LOCALE(loc)->rl_tolower_tab[c + 1]);
100}
101
102int
103_toupper(int c)
104{
105 return (c - 'a' + 'A');
106}
107
108int
109_tolower(int c)
110{
111 return (c - 'A' + 'a');
112}
