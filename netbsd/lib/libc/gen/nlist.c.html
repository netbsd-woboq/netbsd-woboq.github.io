94static const struct {
95 int (*fdnlist)(int, struct nlist *);
96} fdnlist_fmts[] = {
97#ifdef NLIST_AOUT
98 { __fdnlist_aout },
99#endif
100#ifdef NLIST_COFF
101 { __fdnlist_coff },
102#endif
103#ifdef NLIST_ECOFF
104 { __fdnlist_ecoff },
105#endif
106#ifdef NLIST_ELF32
107 { __fdnlist_elf32 },
108#endif
109#ifdef NLIST_ELF64
110 { __fdnlist_elf64 },
111#endif
112};
113
114int
115nlist(const char *name, struct nlist *list)
116{
117 int fd, n;
118
119 _DIAGASSERT(name != NULL);
120 _DIAGASSERT(list != NULL);
121
122 fd = open(name, O_RDONLY | O_CLOEXEC, 0);
123 if (fd < 0)
124 return (-1);
125 n = __fdnlist(fd, list);
126 (void)close(fd);
127 return (n);
128}
129
130int
131__fdnlist(int fd, struct nlist *list)
132{
133 size_t i;
134 int rv;
135
136 _DIAGASSERT(fd != -1);
137 _DIAGASSERT(list != NULL);
138
139 for (i = 0; i < sizeof(fdnlist_fmts) / sizeof(fdnlist_fmts[0]); i++)
140 if ((rv = (*fdnlist_fmts[i].fdnlist)(fd, list)) != -1)
141 return (rv);
142 return (-1);
143}
