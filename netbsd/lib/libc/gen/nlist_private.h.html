51# define NLIST_AOUT
52# define NLIST_ELF32

54# define NLIST_AOUT
55# define NLIST_ELF32
56# define NLIST_ELF64
57#elif defined(__SH5__)
58# define NLIST_ELF32
59# define NLIST_ELF64
60#elif defined(__sh__)
61# define NLIST_COFF
62# define NLIST_ELF32
63#elif defined(__hppa__)
64# define NLIST_ELF32
65#elif defined(__riscv__)
66# define NLIST_ELF32
67# ifdef __riscv64
68# define NLIST_ELF64
69# endif
70#elif defined(__aarch64__)
71# define NLIST_ELF64
72#elif defined(__ia64__)
73# define NLIST_ELF64
74#elif defined(__or1k__)
75# define NLIST_ELF32
76#else

79/* #define NLIST_ELF32 */
