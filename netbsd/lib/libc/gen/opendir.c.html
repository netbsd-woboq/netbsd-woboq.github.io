63/*
64 * Open a directory.
65 */
66DIR *
67opendir(const char *name)
68{
69
70 _DIAGASSERT(name != NULL);
71
72 return (__opendir2(name, DTF_HIDEW|DTF_NODUP));
73}
74
75DIR *
76__opendir2(const char *name, int flags)
77{
78 DIR *dirp;
79 int fd;
80
81 if ((fd = open(name, O_RDONLY|O_DIRECTORY|O_NONBLOCK|O_CLOEXEC)) == -1)
82 return NULL;
83
84 dirp = __opendir_common(fd, name, flags);
85 if (dirp == NULL) {
86 int serrno = errno;
87 (void)close(fd);
88 errno = serrno;
89 }
90 return dirp;
91}

93#ifndef __LIBC12_SOURCE__
94DIR *
95_fdopendir(int fd)
96{
97 struct stat sb;
98
99 if (fstat(fd, &sb) == -1)
100 return NULL;
101
102 if (!S_ISDIR(sb.st_mode)) {
103 errno = ENOTDIR;
104 return NULL;
105 }
106
107 /* This is optional according to POSIX, but a good measure */
108 if (fcntl(fd, F_SETFD, FD_CLOEXEC) == -1)
109 return NULL;
110
111 return __opendir_common(fd, NULL, DTF_HIDEW|DTF_NODUP);
112}
113#endif
114
115static DIR *
116__opendir_common(int fd, const char *name, int flags)
117{
118 DIR *dirp;
119 int serrno;
120 struct statvfs sfb;
121 int error;
122
123 if ((dirp = malloc(sizeof(*dirp))) == NULL)
124 goto error;
125 dirp->dd_buf = NULL;
126 dirp->dd_internal = NULL;
127#ifdef _REENTRANT
128 if (__isthreaded) {
129 if ((dirp->dd_lock = malloc(sizeof(mutex_t))) == NULL)
130 goto error;
131 mutex_init((mutex_t *)dirp->dd_lock, NULL);
132 }
133#endif
134
135 /*
136 * Tweak flags for the underlying filesystem.
137 */
138
139 if (fstatvfs1(fd, &sfb, ST_NOWAIT) < 0)
140 goto error;
141 if ((flags & DTF_NODUP) != 0) {
142 if (!strncmp(sfb.f_fstypename, MOUNT_UNION,
143 sizeof(sfb.f_fstypename)) ||
144 (sfb.f_flag & MNT_UNION) != 0) {
145 flags |= __DTF_READALL;
146 } else {
147 flags &= ~DTF_NODUP;
148 }
149 }
150 if (!strncmp(sfb.f_fstypename, MOUNT_NFS, sizeof(sfb.f_fstypename))) {
151 flags |= __DTF_READALL | __DTF_RETRY_ON_BADCOOKIE;
152 }
153
154 dirp->dd_flags = flags;
155 error = _initdir(dirp, fd, name);
156 if (error) {
157 errno = error;
158 goto error;
159 }
160
161 return (dirp);
162error:
163 serrno = errno;
164 if (dirp != NULL) {
165#ifdef _REENTRANT
166 if (__isthreaded) {
167 mutex_destroy((mutex_t *)dirp->dd_lock);
168 free(dirp->dd_lock);
169 }
170#endif
171 free(dirp->dd_buf);
172 }
173 free(dirp);
174 errno = serrno;
175 return NULL;
176}
