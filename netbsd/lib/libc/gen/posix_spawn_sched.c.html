41/*
42 * Spawn attributes
43 */
44
45int
46posix_spawnattr_init(posix_spawnattr_t *ret)
47{
48 if (ret == NULL)
49 return -1;
50
51 memset(ret, 0, sizeof(posix_spawnattr_t));
52 return 0;
53}
54
55int
56posix_spawnattr_destroy(posix_spawnattr_t *sa)
57{
58 if (sa == NULL)
59 return -1;
60
61 return 0;
62}
