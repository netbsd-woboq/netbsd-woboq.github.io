57void
58psignal(int sig, const char *s)
59{
60 struct iovec *v;
61 struct iovec iov[4];
62 char buf[NL_TEXTMAX];
63
64 v = iov;
65 if (s && *s) {
66 v->iov_base = __UNCONST(s);
67 v->iov_len = strlen(s);
68 v++;
69 v->iov_base = __UNCONST(": ");
70 v->iov_len = 2;
71 v++;
72 }
73 v->iov_base = __UNCONST(__strsignal((int)sig, buf, sizeof(buf)));
74 v->iov_len = strlen(v->iov_base);
75 v++;
76 v->iov_base = __UNCONST("\n");
77 v->iov_len = 1;
78 (void)writev(STDERR_FILENO, iov, (int)((v - iov) + 1));
79}
80
81void
82psiginfo(const siginfo_t *si, const char *s)
83{
84 psignal(si->si_signo, s);
85}
