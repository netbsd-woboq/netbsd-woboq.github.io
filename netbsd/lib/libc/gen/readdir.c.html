53/*
54 * get next entry in a directory.
55 */
56struct dirent *
57_readdir_unlocked(DIR *dirp, int skipdeleted)
58{
59 struct dirent *dp;
60
61 for (;;) {
62 if (dirp->dd_loc >= dirp->dd_size) {
63 if (dirp->dd_flags & __DTF_READALL)
64 return (NULL);
65 dirp->dd_loc = 0;
66 }
67 if (dirp->dd_loc == 0 && !(dirp->dd_flags & __DTF_READALL)) {
68 dirp->dd_seek = lseek(dirp->dd_fd, (off_t)0, SEEK_CUR);
69 dirp->dd_size = getdents(dirp->dd_fd,
70 dirp->dd_buf, (size_t)dirp->dd_len);
71 if (dirp->dd_size <= 0)
72 return (NULL);
73 }
74 dp = (struct dirent *)
75 (void *)(dirp->dd_buf + (size_t)dirp->dd_loc);
76 if ((intptr_t)dp & _DIRENT_ALIGN(dp))/* bogus pointer check */
77 return (NULL);
78 /* d_reclen is unsigned; no need to compare it <= 0 */
79 if (dp->d_reclen > dirp->dd_len + 1 - dirp->dd_loc)
80 return (NULL);
81 dirp->dd_loc += dp->d_reclen;
82 if (dp->d_ino == 0 && skipdeleted)
83 continue;
84 if (dp->d_type == DT_WHT && (dirp->dd_flags & DTF_HIDEW))
85 continue;
86 return (dp);
87 }
88}

90struct dirent *
91readdir(DIR *dirp)
92{
93 struct dirent *dp;
94
95#ifdef _REENTRANT
96 if (__isthreaded) {
97 mutex_lock((mutex_t *)dirp->dd_lock);
98 dp = _readdir_unlocked(dirp, 1);
99 mutex_unlock((mutex_t *)dirp->dd_lock);
100 }
101 else
102#endif
103 dp = _readdir_unlocked(dirp, 1);
104 return (dp);
105}
106
107int
108readdir_r(DIR *dirp, struct dirent *entry, struct dirent **result)
109{
110 struct dirent *dp;
111 int saved_errno;
112
113 saved_errno = errno;
114 errno = 0;
115#ifdef _REENTRANT
116 if (__isthreaded) {
117 mutex_lock((mutex_t *)dirp->dd_lock);
118 if ((dp = _readdir_unlocked(dirp, 1)) != NULL)
119 memcpy(entry, dp, (size_t)_DIRENT_SIZE(dp));
120 mutex_unlock((mutex_t *)dirp->dd_lock);
121 }
122 else
123#endif
124 if ((dp = _readdir_unlocked(dirp, 1)) != NULL)
125 memcpy(entry, dp, (size_t)_DIRENT_SIZE(dp));
126
127 if (errno != 0) {
128 if (dp == NULL)
129 return (errno);
130 } else
131 errno = saved_errno;
132
133 if (dp != NULL)
134 *result = entry;
135 else
136 *result = NULL;
137
138 return (0);
139}
