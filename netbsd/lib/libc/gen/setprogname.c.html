37#include <sys/cdefs.h>
38#if defined(LIBC_SCCS) && !defined(lint)
39__RCSID("$NetBSD: setprogname.c,v 1.3 2003/07/26 19:24:44 salo Exp $");
40#endif /* LIBC_SCCS and not lint */
41
42/* In NetBSD, the program name is set by crt0. It can't be overridden. */
43#undef REALLY_SET_PROGNAME
44
45#include <stdlib.h>
46
47#ifdef REALLY_SET_PROGNAME
48#include <string.h>
49
50extern const char *__progname;
51#endif
52
53/*ARGSUSED*/
54void
55setprogname(const char *progname)
56{
57
58#ifdef REALLY_SET_PROGNAME
59 __progname = strrchr(progname, '/');
60 if (__progname == NULL)
61 __progname = progname;
62 else
63 __progname++;
64#endif
65}
