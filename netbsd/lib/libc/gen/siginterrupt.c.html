44#ifdef __weak_alias
45__weak_alias(siginterrupt,_siginterrupt)
46#endif
47
48extern sigset_t __sigintr;
49/*
50 * Set signal state to prevent restart of system calls
51 * after an instance of the indicated signal.
52 */
53int
54siginterrupt(int sig, int flag)
55{
56 struct sigaction sa;
57 int ret;
58
59 if ((ret = sigaction(sig, (struct sigaction *)0, &sa)) < 0)
60 return (ret);
61 if (flag) {
62 sigaddset(&__sigintr, sig);
63 sa.sa_flags &= ~SA_RESTART;
64 } else {
65 sigdelset(&__sigintr, sig);
66 sa.sa_flags |= SA_RESTART;
67 }
68 return (sigaction(sig, &sa, (struct sigaction *)0));
69}
