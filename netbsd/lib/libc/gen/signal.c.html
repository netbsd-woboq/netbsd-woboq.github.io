42 * Almost backwards compatible signal.
43 */
44#include "namespace.h"
45#include <signal.h>

47#ifdef __weak_alias
48__weak_alias(signal,_signal)
49#endif
50
51sigset_t __sigintr; /* shared with siginterrupt */
52
53sig_t
54signal(int s, sig_t a)
55{
56 struct sigaction sa, osa;
57
58 sa.sa_handler = a;
59 sigemptyset(&sa.sa_mask);
60 sa.sa_flags = 0;
61 if (!sigismember(&__sigintr, s))
62 sa.sa_flags |= SA_RESTART;
63 if (sigaction(s, &sa, &osa) < 0)
64 return (SIG_ERR);
65 return (osa.sa_handler);
66}
67
68void
69(*bsd_signal(int sig, void (*func)(int)))(int)
70{
71 struct sigaction act, oact;
72
73 act.sa_handler = func;
74 act.sa_flags = SA_RESTART;
75
76 sigemptyset(&act.sa_mask);
77 sigaddset(&act.sa_mask, sig);
78 if (sigaction(sig, &act, &oact) < 0)
79 return (SIG_ERR);
80
81 return (oact.sa_handler);
82}
