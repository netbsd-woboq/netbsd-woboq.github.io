34__RCSID("$NetBSD");
35#endif
36
37#include <machine/ieee.h>
38#include <math.h>
39
40#ifdef __HAVE_LONG_DOUBLE
41
42/*
43 * 7.12.3.6 signbit - determine whether the sign of an argument is negative
44 * IEEE 754 compatible 80-bit or 128-bit extended precision.
45 *
46 * Differences between 128-bit and 80-bit are handled by having a
47 * different struct ieee_ext.
48 */
49int
50__signbitl(long double x)
51{
52 union ieee_ext_u u;
53
54 u.extu_ld = x;
55
56 return (u.extu_ext.ext_sign == 1);
57}
58
59#endif /* __HAVE_LONG_DOUBLE */
