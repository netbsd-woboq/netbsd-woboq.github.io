37#include "namespace.h"
38#include <errno.h>
39#include <signal.h>
40#include <stddef.h>
41
42sig_t
43sigset(int sig, void (*disp)(int))
44{
45 sigset_t set, oset;
46 struct sigaction sa, osa;
47
48 osa.sa_handler = SIG_ERR;
49
50 if (disp == SIG_HOLD) {
51 /* Add sig to current signal mask. */
52 if (sigemptyset(&set) != 0)
53 goto out;
54 if (sigaddset(&set, sig) != 0)
55 goto out;
56 if (sigprocmask(SIG_BLOCK, &set, &oset) != 0)
57 goto out;
58
59 if (sigismember(&oset, sig)) {
60 /* Had been masked before, return SIG_HOLD. */
61 osa.sa_handler = SIG_HOLD;
62 } else {
63 /* Return previous disposition. */
64 (void)sigaction(sig, NULL, &osa);
65 }
66 } else if (disp == SIG_ERR) {
67 errno = EINVAL;
68 } else {
69 /* Set up and install new disposition. */
70 sa.sa_handler = disp;
71 if (sigemptyset(&sa.sa_mask) != 0)
72 goto out;
73 sa.sa_flags = 0;
74
75 if (sigaction(sig, &sa, &osa) != 0) {
76 osa.sa_handler = SIG_ERR;
77 goto out;
78 }
79
80 /* Delete sig from current signal mask. */
81 if (sigemptyset(&set) != 0)
82 return (SIG_ERR);
83 if (sigaddset(&set, sig) != 0)
84 return (SIG_ERR);
85 if (sigprocmask(SIG_UNBLOCK, &set, &oset) != 0)
86 return (SIG_ERR);
87
88 /* If had been masked before, return SIG_HOLD. */
89 if (sigismember(&oset, sig))
90 osa.sa_handler = SIG_HOLD;
91 }
92
93 out:
94 return (osa.sa_handler);
95}
