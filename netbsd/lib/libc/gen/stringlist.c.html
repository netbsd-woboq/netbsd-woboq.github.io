55#define _SL_CHUNKSIZE 20
56
57/*
58 * sl_init(): Initialize a string list
59 */
60StringList *
61sl_init(void)
62{
63 StringList *sl;
64
65 sl = malloc(sizeof(StringList));
66 if (sl == NULL)
67 return NULL;
68
69 sl->sl_cur = 0;
70 sl->sl_max = _SL_CHUNKSIZE;
71 sl->sl_str = NULL;
72 errno = reallocarr(&sl->sl_str, sl->sl_max, sizeof(char *));
73 if (errno) {
74 free(sl);
75 sl = NULL;
76 }
77 return sl;
78}

81/*
82 * sl_add(): Add an item to the string list
83 */
84int
85sl_add(StringList *sl, char *name)
86{
87
88 _DIAGASSERT(sl != NULL);
89
90 if (sl->sl_cur == sl->sl_max - 1) {
91 char **new = sl->sl_str;
92
93 errno = reallocarr(&new, (sl->sl_max + _SL_CHUNKSIZE),
94 sizeof(char *));
95 if (errno)
96 return -1;
97 sl->sl_max += _SL_CHUNKSIZE;
98 sl->sl_str = new;
99 }
100 sl->sl_str[sl->sl_cur++] = name;
101 return 0;
102}

105/*
106 * sl_free(): Free a stringlist
107 */
108void
109sl_free(StringList *sl, int all)
110{
111 size_t i;
112
113 if (sl == NULL)
114 return;
115 if (sl->sl_str) {
116 if (all)
117 for (i = 0; i < sl->sl_cur; i++)
118 free(sl->sl_str[i]);
119 free(sl->sl_str);
120 }
121 free(sl);
122}
123
124
125/*
126 * sl_find(): Find a name in the string list
127 */
128char *
129sl_find(StringList *sl, const char *name)
130{
131 size_t i;
132
133 _DIAGASSERT(sl != NULL);
134
135 for (i = 0; i < sl->sl_cur; i++)
136 if (strcmp(sl->sl_str[i], name) == 0)
137 return sl->sl_str[i];
138
139 return NULL;
140}
141
142int
143sl_delete(StringList *sl, const char *name, int all)
144{
145 size_t i, j;
146
147 for (i = 0; i < sl->sl_cur; i++)
148 if (strcmp(sl->sl_str[i], name) == 0) {
149 if (all)
150 free(sl->sl_str[i]);
151 for (j = i + 1; j < sl->sl_cur; j++)
152 sl->sl_str[j - 1] = sl->sl_str[j];
153 sl->sl_str[--sl->sl_cur] = NULL;
154 return 0;
155 }
156 return -1;
157}
