53/*
54 * freebsd compatible sysctlnametomib() function, implemented as an
55 * extremely thin wrapper around sysctlgetmibinfo(). i think the use
56 * of size_t as the third argument is erroneous, but what can we do
57 * about that?
58 */
59int
60sysctlnametomib(const char *gname, int *iname, size_t *namelenp)
61{
62 u_int unamelen;
63 int rc;
64
65 _DIAGASSERT(__type_fit(u_int, *namelenp));
66 unamelen = (u_int)*namelenp;
67 rc = sysctlgetmibinfo(gname, iname, &unamelen, NULL, NULL, NULL,
68 SYSCTL_VERSION);
69 *namelenp = unamelen;
70
71 return (rc);
72}
