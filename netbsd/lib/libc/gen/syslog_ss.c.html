42static size_t
43timefun_ss(char *p, size_t tbuf_left)
44{
45 return snprintf_ss(p, tbuf_left, "-");
46#if 0
47 /*
48 * if gmtime_r() was signal-safe we could output
49 * the UTC-time:
50 */
51 gmtime_r(&now, &tmnow);
52 prlen = strftime(p, tbuf_left, "%FT%TZ", &tmnow);
53 return prlen;
54#endif
55}

57static int
58lock_ss(const struct syslog_data *data __unused)
59{
60 return 0;
61}

63static int
64unlock_ss(const struct syslog_data *data __unused)
65{
66 return 0;

68
69struct syslog_fun _syslog_ss_fun = {
70 timefun_ss,
71 strerror_r_ss,
72 vsnprintf_ss,
73 lock_ss,
74 unlock_ss,
75};
76
77void
78syslog_ss(int pri, struct syslog_data *data, const char *fmt, ...)
79{
80 va_list ap;
81
82 va_start(ap, fmt);
83 _vxsyslogp_r(pri, &_syslog_ss_fun, data, NULL, NULL, fmt, ap);
84 va_end(ap);
85}
86
87void
88syslogp_ss(int pri, struct syslog_data *data, const char *msgid,
89 const char *sdfmt, const char *msgfmt, ...)
90{
91 va_list ap;
92
93 va_start(ap, msgfmt);
94 _vxsyslogp_r(pri, &_syslog_ss_fun, data, msgid, sdfmt, msgfmt, ap);
95 va_end(ap);
96}

98void
99vsyslog_ss(int pri, struct syslog_data *data, const char *fmt, va_list ap)
100{

102}
