57long
58telldir(DIR *dirp)
59{
60 long rv;
61#ifdef _REENTRANT
62 if (__isthreaded) {
63 mutex_lock((mutex_t *)dirp->dd_lock);
64 rv = _telldir_unlocked(dirp);
65 mutex_unlock((mutex_t *)dirp->dd_lock);
66 } else
67#endif
68 rv = _telldir_unlocked(dirp);
69 return rv;
70}
71
72/*
73 * return a pointer into a directory
74 */
75long
76_telldir_unlocked(DIR *dirp)
77{
78 struct dirpos *lp;
79
80 for (lp = dirp->dd_internal; lp; lp = lp->dp_next)
81 if (lp->dp_seek == dirp->dd_seek &&
82 lp->dp_loc == dirp->dd_loc)
83 return (intptr_t)lp;
84
85 if ((lp = malloc(sizeof(*lp))) == NULL)
86 return (-1);
87
88 lp->dp_seek = dirp->dd_seek;
89 lp->dp_loc = dirp->dd_loc;
90 lp->dp_next = dirp->dd_internal;
91 dirp->dd_internal = lp;
92
93 return (intptr_t)lp;
94}

96/*
97 * seek to an entry in a directory.
98 * Only values returned by "telldir" should be passed to seekdir.
99 */
100void
101_seekdir_unlocked(DIR *dirp, long loc)
102{
103 struct dirpos *lp;
104
105 _DIAGASSERT(dirp != NULL);
106
107 for (lp = dirp->dd_internal; lp; lp = lp->dp_next)
108 if ((intptr_t)lp == loc)
109 break;
110
111 if (lp == NULL)
112 return;
113
114 if (lp->dp_loc == dirp->dd_loc && lp->dp_seek == dirp->dd_seek)
115 return;
116
117 dirp->dd_seek = lseek(dirp->dd_fd, lp->dp_seek, SEEK_SET);
118 dirp->dd_loc = 0;
119 while (dirp->dd_loc < lp->dp_loc)
120 if (_readdir_unlocked(dirp, 0) == NULL)
121 break;
122}
