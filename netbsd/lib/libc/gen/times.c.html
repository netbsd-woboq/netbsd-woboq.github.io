51/*
52 * Convert usec to clock ticks; could do (usec * CLK_TCK) / 1000000,
53 * but this would overflow if we switch to nanosec.
54 */
55#define CONVTCK(r) \
56 (clock_t)(r.tv_sec * clk_tck + r.tv_usec / (1000000 / (uint)clk_tck))
57
58#ifndef __times_rusage
59#define __times_rusage struct rusage
60#endif
61#ifndef __times_timeval
62#define __times_timeval struct timeval
63#endif
64
65clock_t
66times(struct tms *tp)
67{
68 __times_rusage ru;
69 __times_timeval t;
70 static clock_t clk_tck;
71
72 _DIAGASSERT(tp != NULL);
73
74 /*
75 * we use a local copy of CLK_TCK because it expands to a
76 * moderately expensive function call.
77 */
78 if (clk_tck == 0)
79 clk_tck = (clock_t)CLK_TCK;
80
81 if (getrusage(RUSAGE_SELF, &ru) < 0)
82 return ((clock_t)-1);
83 tp->tms_utime = CONVTCK(ru.ru_utime);
84 tp->tms_stime = CONVTCK(ru.ru_stime);
85 if (getrusage(RUSAGE_CHILDREN, &ru) < 0)
86 return ((clock_t)-1);
87 tp->tms_cutime = CONVTCK(ru.ru_utime);
88 tp->tms_cstime = CONVTCK(ru.ru_stime);
89 if (gettimeofday(&t, (struct timezone *)0))
90 return ((clock_t)-1);
91 return ((clock_t)(CONVTCK(t)));
92}
