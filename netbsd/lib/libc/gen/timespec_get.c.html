37#include <assert.h>
38#include <time.h>
39
40/* ISO/IEC 9899:201x 7.27.2.5 The timespec_get function */
41
42int
43timespec_get(struct timespec *ts, int base)
44{
45
46 _DIAGASSERT(ts != NULL);
47
48 switch (base) {
49 case TIME_UTC:
50 if (clock_gettime(CLOCK_REALTIME, ts) == -1)
51 return 0;
52 break;
53 default:
54 return 0;
55 }
56
57 return base;
58}
