59int
60ttyname_r(int fd, char *buf, size_t len)
61{
62 struct stat sb;
63 struct termios ttyb;
64 struct ptmget ptm;
65
66 _DIAGASSERT(fd != -1);
67
68 /* If it is a pty, deal with it quickly */
69 if (ioctl(fd, TIOCPTSNAME, &ptm) != -1) {
70 if (strlcpy(buf, ptm.sn, len) >= len) {
71 return ERANGE;
72 }
73 return 0;
74 }
75
76 /* Must be a terminal. */
77 if (tcgetattr(fd, &ttyb) == -1)
78 return errno;
79
80 if (fstat(fd, &sb))
81 return errno;
82
83 if (strlcpy(buf, _PATH_DEV, len) >= len)
84 return ERANGE;
85 buf += strlen(_PATH_DEV);
86 len -= strlen(_PATH_DEV);
87 return devname_r(sb.st_rdev, sb.st_mode & S_IFMT, buf, len);
88}
89
90char *
91ttyname(int fd)
92{
93 static char buf[PATH_MAX];
94 int rv;
95
96 rv = ttyname_r(fd, buf, sizeof(buf));
97 if (rv != 0) {
98 errno = rv;
99 return NULL;
100 }
101 return buf;
102}
