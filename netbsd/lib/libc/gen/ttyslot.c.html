53
54int
55ttyslot(void)
56{
57 struct ttyent *ttyp;
58 int slot = 0, ispty = 0;
59 char *p;
60 int cnt;
61 char *name;
62 struct ptmget ptm;
63
64 setttyent();
65 for (cnt = 0; cnt < 3; ++cnt) {
66 if (ioctl(cnt, TIOCPTSNAME, &ptm) != -1) {
67 ispty = 1;
68 name = ptm.sn;
69 } else if ((name = ttyname(cnt)) != NULL) {
70 ispty = 0;
71 } else
72 continue;
73
74 if ((p = strstr(name, "/pts/")) != NULL)
75 ++p;
76 else if ((p = strrchr(name, '/')) != NULL)
77 ++p;
78 else
79 p = name;
80
81 for (slot = 1; (ttyp = getttyent()) != NULL; ++slot)
82 if (!strcmp(ttyp->ty_name, p)) {
83 endttyent();
84 return slot;
85 }
86 break;
87 }
88 endttyent();
89 if (ispty) {
90 struct stat st;
91 if (fstat(cnt, &st) == -1)
92 return 0;
93 return slot + (int)minor(st.st_rdev) + 1;
94 }
95 return 0;
96}
