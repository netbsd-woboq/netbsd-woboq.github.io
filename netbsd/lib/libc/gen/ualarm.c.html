49#define USPS 1000000 /* # of microseconds in a second */
50
51/*
52 * Generate a SIGALRM signal in ``usecs'' microseconds.
53 * If ``reload'' is non-zero, keep generating SIGALRM
54 * every ``reload'' microseconds after the first signal.
55 */
56useconds_t
57ualarm(useconds_t usecs, useconds_t reload)
58{
59 struct itimerval new, old;
60
61 new.it_interval.tv_usec = reload % USPS;
62 new.it_interval.tv_sec = reload / USPS;
63
64 new.it_value.tv_usec = usecs % USPS;
65 new.it_value.tv_sec = usecs / USPS;
66
67 if (setitimer(ITIMER_REAL, &new, &old) == 0)
68 return (useconds_t)(old.it_value.tv_sec * USPS +
69 old.it_value.tv_usec);
70
71 return (useconds_t)-1;
72}
