32#include <sys/cdefs.h>
33#if defined(LIBC_SCCS) && !defined(lint)

36
37#include <sys/resource.h>
38#include <errno.h>
39#include <stdarg.h>
40#include <ulimit.h>
41
42long int
43ulimit(int cmd, ...)
44{
45 va_list ap;
46 struct rlimit rlimit;
47 long int new_limit, result;
48
49 va_start(ap, cmd);
50
51 result = -1L;
52 switch (cmd) {
53 case UL_GETFSIZE:
54 if (getrlimit(RLIMIT_FSIZE, &rlimit) == 0)
55 result = (long int)(rlimit.rlim_cur / 512);
56 break;
57 case UL_SETFSIZE:
58 new_limit = va_arg(ap, long int);
59 rlimit.rlim_cur = rlimit.rlim_max = (rlim_t)new_limit * 512;
60 if (setrlimit(RLIMIT_FSIZE, &rlimit) == 0)
61 result = new_limit;
62 break;
63 default:
64 errno = EINVAL;
65 }
66
67 va_end(ap);
68
69 return (result);
70}
