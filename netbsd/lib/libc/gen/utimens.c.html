37#include "namespace.h"
38
39#define _INCOMPLETE_XOPEN_C063
40#include <fcntl.h>
41#include <sys/stat.h>
42
43int
44utimens(const char *path, const struct timespec *times)
45{
46 return utimensat(AT_FDCWD, path, times, 0);
47}
48
49int
50lutimens(const char *path, const struct timespec *times)
51{
52 return utimensat(AT_FDCWD, path, times, AT_SYMLINK_NOFOLLOW);
53}
