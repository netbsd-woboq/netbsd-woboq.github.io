46
47int
48waitid(idtype_t idtype, id_t id, siginfo_t *info, int flags)
49{
50 int status;
51
52 switch (wait6(idtype, id, &status, flags, NULL, info)) {
53 case -1:
54 return -1;
55 case 0:
56 if (info != NULL)
57 memset(info, 0, sizeof(*info));
58 /*FALLTHROUGH*/
59 default:
60 return 0;
61 }
62}
