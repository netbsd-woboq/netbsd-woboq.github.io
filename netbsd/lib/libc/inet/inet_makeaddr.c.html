50/*
51 * Formulate an Internet address from network + host. Used in
52 * building addresses stored in the ifnet structure.
53 */
54struct in_addr
55inet_makeaddr(in_addr_t net, in_addr_t host)
56{
57 in_addr_t addr;
58 struct in_addr ret;
59
60 if (net < 128)
61 addr = (net << IN_CLASSA_NSHIFT) | (host & IN_CLASSA_HOST);
62 else if (net < 65536)
63 addr = (net << IN_CLASSB_NSHIFT) | (host & IN_CLASSB_HOST);
64 else if (net < 16777216L)
65 addr = (net << IN_CLASSC_NSHIFT) | (host & IN_CLASSC_HOST);
66 else
67 addr = net | host;
68 ret.s_addr = htonl(addr);
69 return ret;
70}
