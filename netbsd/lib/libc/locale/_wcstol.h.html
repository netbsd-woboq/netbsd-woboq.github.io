37/*
38 * function template for wcstol, wcstoll and wcstoimax.
39 *
40 * parameters:
41 * _FUNCNAME : function name
42 * __INT : return type
43 * __INT_MIN : lower limit of the return type
44 * __INT_MAX : upper limit of the return type
45 */
46
47#include <locale.h>
48#include "setlocale_local.h"
49#define INT_FUNCNAME_(pre, name, post) pre ## name ## post
50#define INT_FUNCNAME(pre, name, post) INT_FUNCNAME_(pre, name, post)
51
52static __INT
53INT_FUNCNAME(_int_, _FUNCNAME, _l)(const wchar_t *nptr, wchar_t **endptr,
54 int base, locale_t loc)
55{
56 const wchar_t *s;
57 __INT acc, cutoff;
58 wint_t wc;
59 int i;
60 int neg, any, cutlim;
61
62 _DIAGASSERT(nptr != NULL);
63 /* endptr may be NULL */
64
65#ifdef __GNUC__
66 (void)&acc; (void)&cutoff;
67#endif
68
69 /* check base value */
70 if (base && (base < 2 || base > 36)) {
71 errno = EINVAL;
72 return 0;
73 }
74
75 /*
76 * Skip white space and pick up leading +/- sign if any.
77 * If base is 0, allow 0x for hex and 0 for octal, else
78 * assume decimal; if base is already 16, allow 0x.
79 */
80 s = nptr;
81 do {
82 wc = (wchar_t) *s++;
83 } while (iswspace_l(wc, loc));
84 if (wc == L'-') {
85 neg = 1;
86 wc = *s++;
87 } else {
88 neg = 0;
89 if (wc == L'+')
90 wc = *s++;
91 }
92 if ((base == 0 || base == 16) &&
93 wc == L'0' && (*s == L'x' || *s == L'X')) {
94 wc = s[1];
95 s += 2;
96 base = 16;
97 }
98 if (base == 0)
99 base = wc == L'0' ? 8 : 10;
100
101 /*
102 * See strtol for comments as to the logic used.
103 */
104 cutoff = neg ? __INT_MIN : __INT_MAX;
105 cutlim = (int)(cutoff % base);
106 cutoff /= base;
107 if (neg) {
108 if (cutlim > 0) {
109 cutlim -= base;
110 cutoff += 1;
111 }
112 cutlim = -cutlim;
113 }
114 for (acc = 0, any = 0;; wc = (wchar_t) *s++) {
115 i = __wctoint(wc);
116 if (i == -1)
117 break;
118 if (i >= base)
119 break;
120 if (any < 0)
121 continue;
122 if (neg) {
123 if (acc < cutoff || (acc == cutoff && i > cutlim)) {
124 any = -1;
125 acc = __INT_MIN;
126 errno = ERANGE;
127 } else {
128 any = 1;
129 acc *= base;
130 acc -= i;
131 }
132 } else {
133 if (acc > cutoff || (acc == cutoff && i > cutlim)) {
134 any = -1;
135 acc = __INT_MAX;
136 errno = ERANGE;
137 } else {
138 any = 1;
139 acc *= base;
140 acc += i;
141 }
142 }
143 }
144 if (endptr != 0)
145 *endptr = __UNCONST(any ? s - 1 : nptr);
146 return (acc);
147}
148
149__INT
150_FUNCNAME(const wchar_t *nptr, wchar_t **endptr, int base)
151{
152 return INT_FUNCNAME(_int_, _FUNCNAME, _l)(nptr, endptr, base,
153 _current_locale());
154}
155
156__INT
157INT_FUNCNAME(, _FUNCNAME, _l)(const wchar_t *nptr, wchar_t **endptr,
158 int base, locale_t loc)
159{
160 return INT_FUNCNAME(_int_, _FUNCNAME, _l)(nptr, endptr, base, loc);
161}
