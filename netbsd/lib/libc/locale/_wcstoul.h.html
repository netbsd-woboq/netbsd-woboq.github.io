37/*
38 * function template for wcstoul, wcstoull and wcstoumax.
39 *
40 * parameters:
41 * _FUNCNAME : function name
42 * __UINT : return type
43 * __UINT_MAX : upper limit of the return type
44 */
45
46#include <locale.h>
47#include "setlocale_local.h"
48#define INT_FUNCNAME_(pre, name, post) pre ## name ## post
49#define INT_FUNCNAME(pre, name, post) INT_FUNCNAME_(pre, name, post)
50
51static __UINT
52INT_FUNCNAME(_int_, _FUNCNAME, _l)(const wchar_t *nptr, wchar_t **endptr,
53 int base, locale_t loc)
54{
55 const wchar_t *s;
56 __UINT acc, cutoff;
57 wint_t wc;
58 int i;
59 int neg, any, cutlim;
60
61 _DIAGASSERT(nptr != NULL);
62 /* endptr may be NULL */
63
64 if (base && (base < 2 || base > 36)) {
65 errno = EINVAL;
66 return 0;
67 }
68
69 /*
70 * Skip white space and pick up leading +/- sign if any.
71 * If base is 0, allow 0x for hex and 0 for octal, else
72 * assume decimal; if base is already 16, allow 0x.
73 */
74 s = nptr;
75 do {
76 wc = (wchar_t) *s++;
77 } while (iswspace_l(wc, loc));
78 if (wc == L'-') {
79 neg = 1;
80 wc = *s++;
81 } else {
82 neg = 0;
83 if (wc == L'+')
84 wc = *s++;
85 }
86 if ((base == 0 || base == 16) &&
87 wc == L'0' && (*s == L'x' || *s == L'X')) {
88 wc = s[1];
89 s += 2;
90 base = 16;
91 }
92 if (base == 0)
93 base = wc == L'0' ? 8 : 10;
94
95 /*
96 * See strtoul for comments as to the logic used.
97 */
98 cutoff = __UINT_MAX / (__UINT)base;
99 cutlim = (int)(__UINT_MAX % (__UINT)base);
100 for (acc = 0, any = 0;; wc = (wchar_t) *s++) {
101 i = __wctoint(wc);
102 if (i == (wint_t)-1)
103 break;
104 if (i >= base)
105 break;
106 if (any < 0)
107 continue;
108 if (acc > cutoff || (acc == cutoff && i > cutlim)) {
109 any = -1;
110 acc = __UINT_MAX;
111 errno = ERANGE;
112 } else {
113 any = 1;
114 acc *= (__UINT)base;
115 acc += i;
116 }
117 }
118 if (neg && any > 0)
119 acc = -acc;
120 if (endptr != 0)
121 *endptr = __UNCONST(any ? s - 1 : nptr);
122 return (acc);
123}
124
125__UINT
126_FUNCNAME(const wchar_t *nptr, wchar_t **endptr, int base)
127{
128 return INT_FUNCNAME(_int_, _FUNCNAME, _l)(nptr, endptr, base,
129 _current_locale());
130}
131
132__UINT
133INT_FUNCNAME(, _FUNCNAME, _l)(const wchar_t *nptr, wchar_t **endptr,
134 int base, locale_t loc)
135{
136 return INT_FUNCNAME(_int_, _FUNCNAME, _l)(nptr, endptr, base, loc);
137}
