75/*
76 * _towctrans_ext:
77 * translate a character (extended part)
78 */
79wint_t
80_towctrans_ext(wint_t c, struct _WCTransEntry const *te)
81{
82 __nbrune_t c0;
83 uint32_t x;
84 _RuneRange *rr;
85 _RuneEntry *base, *re;
86
87 _DIAGASSERT(te != NULL);
88
89 if (c == WEOF)
90 return (c);
91
92 c0 = (__nbrune_t)c; /* XXX assumes wchar_t = int */
93 rr = te->te_extmap;
94 base = rr->rr_rune_ranges;
95 for (x = rr->rr_nranges; x != 0; x >>= 1) {
96 re = base + (x >> 1);
97 if (re->re_min <= c0 && re->re_max >= c0)
98 return (re->re_map + c0 - re->re_min);
99 else if (c0 >= re->re_max) {
100 base = re + 1;
101 x--;
102 }
103 }
104
105 return (c);
106}
