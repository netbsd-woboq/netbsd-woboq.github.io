74_RuneType
75_runetype_priv(_RuneLocale const *rl, wint_t wc)
76{
77 __nbrune_t wc0;
78 _RuneRange const *rr;
79 _RuneEntry *base, *re;
80 uint32_t x;
81
82 _DIAGASSERT(rl != NULL);
83
84 if (wc == WEOF)
85 return 0U;
86 if (_RUNE_ISCACHED(wc))
87 return rl->rl_runetype[(size_t)wc];
88 wc0 = (__nbrune_t)wc;
89 rr = &rl->rl_runetype_ext;
90 _DIAGASSERT(rr != NULL);
91 base = rr->rr_rune_ranges;
92 for (x = rr->rr_nranges; x != (uint32_t)0; x >>= 1) {
93 re = base + (x >> 1);
94 if (re->re_min <= wc0 && re->re_max >= wc0) {
95 if (re->re_rune_types)
96 return re->re_rune_types[wc0 - re->re_min];
97 else
98 return re->re_map;
99 } else if (wc0 > re->re_max) {
100 base = re + 1;
101 --x;
102 }
103 }
104 return (_RuneType)0U;
105}
106
107int
108_iswctype_priv(_RuneLocale const *rl,
109 wint_t wc, _WCTypeEntry const *te)
110{
111 return !!(_runetype_priv(rl, wc) & te->te_mask);
112}
