29#ifndef _MULTIBYTE_H_
30#define _MULTIBYTE_H_
31
32/* mbstate_t private */
33
34#ifdef _BSD_MBSTATE_T_
35typedef _BSD_MBSTATE_T_ mbstate_t;
36#undef _BSD_MBSTATE_T_
37#endif
38
39typedef struct _RuneStatePriv {
40 _RuneLocale *__runelocale;
41 char __private __attribute__((__aligned__));
42} _RuneStatePriv;
43
44typedef union _RuneState {
45 mbstate_t __pad;
46 struct _RuneStatePriv __priv;
47#define rs_runelocale __priv.__runelocale
48#define rs_private __priv.__private
49} _RuneState;
50#define _PRIVSIZE (sizeof(mbstate_t)-offsetof(_RuneStatePriv, __private))
51
52#define _RUNE_LOCALE(loc) \
53 ((_RuneLocale *)((loc)->part_impl[(size_t)LC_CTYPE]))
54
55#define _CITRUS_CTYPE(loc) \
56 (((_RuneLocale *)((loc)->part_impl[(size_t)LC_CTYPE]))->rl_citrus_ctype)
57
58/* */
59
60static __inline _RuneState *
61_ps_to_runestate(mbstate_t *ps)
62{
63 return (_RuneState *)(void *)ps;
64}
65
66static __inline _RuneState const *
67_ps_to_runestate_const(mbstate_t const *ps)
68{
69 return (_RuneState const *)(void const *)ps;
70}
71
72static __inline _RuneLocale *
73_ps_to_runelocale(mbstate_t const *ps)
74{
75 return _ps_to_runestate_const(ps)->rs_runelocale;
76}
77
78static __inline _citrus_ctype_t
79_ps_to_ctype(mbstate_t const *ps, locale_t loc)
80{
81 if (!ps)
82 return _CITRUS_CTYPE(loc);
83
84 _DIAGASSERT(_ps_to_runelocale(ps) != NULL);
85
86 return _ps_to_runelocale(ps)->rl_citrus_ctype;
87}
88
89static __inline void *
90_ps_to_private(mbstate_t *ps)
91{
92 if (ps == NULL)
93 return NULL;
94 return (void *)&_ps_to_runestate(ps)->rs_private;
95}
96
97static __inline void const *
98_ps_to_private_const(mbstate_t const *ps)
99{
100 if (ps == NULL)
101 return NULL;
102 return (void const *)&_ps_to_runestate_const(ps)->rs_private;
103}
104
105static __inline void
106_init_ps(_RuneLocale *rl, mbstate_t *ps)
107{
108 size_t dum;
109 _ps_to_runestate(ps)->rs_runelocale = rl;
110 _citrus_ctype_mbrtowc(rl->rl_citrus_ctype, NULL, NULL, 0,
111 _ps_to_private(ps), &dum);
112}
113
114static __inline void
115_fixup_ps(_RuneLocale *rl, mbstate_t *ps, int forceinit)
116{
117 /* for future multi-locale facility */
118 _DIAGASSERT(rl != NULL);
119
120 if (ps != NULL && (_ps_to_runelocale(ps) == NULL || forceinit)) {
121 _init_ps(rl, ps);
122 }
123}
