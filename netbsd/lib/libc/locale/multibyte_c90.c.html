50int
51mblen_l(const char *s, size_t n, locale_t loc)
52{
53 int ret;
54 int err0;
55
56 err0 = _citrus_ctype_mblen(_CITRUS_CTYPE(loc), s, n, &ret);
57 if (err0)
58 errno = err0;
59
60 return ret;
61}

127size_t
128wcsnrtombs_l(char *s, const wchar_t **ppwcs, size_t in, size_t n, mbstate_t *ps,
129 locale_t loc)
130{
131 size_t ret;
132 int err0;
133
134 _fixup_ps(_RUNE_LOCALE(loc), ps, s == NULL);
135
136 err0 = _citrus_ctype_wcsnrtombs(_ps_to_ctype(ps, loc), s, ppwcs, in, n,
137 _ps_to_private(ps), &ret);
138 if (err0)
139 errno = err0;
140
141 return ret;
142}
