37#define _LOCALENAME_LEN_MAX 33
38
39#define _C_LOCALE "C"
40#define _POSIX_LOCALE "POSIX"

42extern const char *_PathLocale;
43#define _LOCALE_ALIAS_NAME "locale.alias"
44
45typedef void *_locale_part_t;
46
47struct _locale_cache_t {
48 SLIST_ENTRY(_locale_cache_t) cache_link;
49 const char *monetary_name;
50 const char *numeric_name;
51 struct lconv ldata;
52};
53
54struct _locale {
55 const struct _locale_cache_t *cache;
56 char query[_LOCALENAME_LEN_MAX * (_LC_LAST - 1)];
57 const char *part_name[_LC_LAST];
58 _locale_part_t part_impl[_LC_LAST];
59};
60
61typedef const char *(*_locale_set_t)(const char * __restrict,
62 struct _locale * __restrict);
