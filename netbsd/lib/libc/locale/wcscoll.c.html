41/*
42 * Compare strings with using collating information.
43 */
44int
45wcscoll_l(const wchar_t *s1, const wchar_t *s2, locale_t loc)
46{
47 /* XXX: LC_COLLATE should be implemented. */
48 /* LINTED */ (void)loc;
49 return (wcscmp(s1, s2));
50}
51
52int
53wcscoll(const wchar_t *s1, const wchar_t *s2)
54{
55 return wcscoll_l(s1, s2, _current_locale());
56}
