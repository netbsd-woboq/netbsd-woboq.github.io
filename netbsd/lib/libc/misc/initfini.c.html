70static bool libc_initialised;

72void _libc_init(void);

74/*
75 * Declare as common symbol to allow new libc with older binaries to
76 * not trigger an undefined reference.
77 */
78struct ps_strings *__ps_strings;
79
80/*
81 * _libc_init is called twice. One call comes explicitly from crt0.o
82 * (for newer versions) and the other is via global constructor handling.
83 *
84 * In static binaries the explicit call is first; in dynamically linked
85 * binaries the global constructors of libc are called from ld.elf_so
86 * before crt0.o is reached.
87 *
88 * Note that __ps_strings is set by crt0.o. So in the dynamic case, it
89 * hasn't been set yet when we get here, and __libc_dlauxinfo is not
90 * (ever) assigned. But this is ok because __libc_dlauxinfo is only
91 * used in static binaries, because it's there to substitute for the
92 * dynamic linker. In static binaries __ps_strings will have been set
93 * up when we get here and we get a valid __libc_dlauxinfo.
94 *
95 * This code causes problems for Emacs because Emacs's undump
96 * mechanism saves the __ps_strings value from the startup execution;
97 * then running the resulting binary it gets here before crt0 has
98 * assigned the current execution's value to __ps_strings, and in an
99 * environment with ASLR this can cause the assignment of
100 * __libc_dlauxinfo to receive SIGSEGV.
101 */
102void __section(".text.startup")
103_libc_init(void)
104{
105
106 if (libc_initialised)
107 return;
108
109 libc_initialised = 1;
110
111 /* Only initialize _dlauxinfo for static binaries. */
112 if (__ps_strings != NULL && real_dlauxinfo == ___dlauxinfo)
113 __libc_dlauxinfo = __ps_strings->ps_argvstr +
114 __ps_strings->ps_nargvstr + __ps_strings->ps_nenvstr + 2;
115
116 /* For -fstack-protector */
117 __guard_setup();
118
119 /* Atomic operations */
120 __libc_atomic_init();
121
122#if defined(__HAVE_TLS_VARIANT_I) || defined(__HAVE_TLS_VARIANT_II)
123 /* Initialize TLS for statically linked programs. */
124 __libc_static_tls_setup();
125#endif
126
127 /* Threads */
128 __libc_thr_init();
129
130 /* Initialize the atexit mutexes */
131 __libc_atexit_init();
132
133 /* Initialize environment memory RB tree. */
134 __libc_env_init();
135}
