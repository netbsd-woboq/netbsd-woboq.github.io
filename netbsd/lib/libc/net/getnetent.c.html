66#define MAXALIASES 35
67
68static FILE *netf;
69static char line[BUFSIZ+1];
70static struct netent net;
71static char *net_aliases[MAXALIASES];
72int _net_stayopen;
73
74static void __setnetent(int);
75static void __endnetent(void);
76
77void
78setnetent(int stayopen)
79{
80
81 sethostent(stayopen);
82 __setnetent(stayopen);
83}
84
85void
86endnetent(void)
87{
88
89 endhostent();
90 __endnetent();
91}
92
93static void
94__setnetent(int f)
95{
96
97 if (netf == NULL)
98 netf = fopen(_PATH_NETWORKS, "re");
99 else
100 rewind(netf);
101 _net_stayopen |= f;
102}
103
104static void
105__endnetent(void)
106{
107
108 if (netf) {
109 fclose(netf);
110 netf = NULL;
111 }
112 _net_stayopen = 0;
113}
114
115struct netent *
116getnetent(void)
117{
118 char *p;
119 register char *cp, **q;
120
121 if (netf == NULL && (netf = fopen(_PATH_NETWORKS, "re")) == NULL)
122 return (NULL);
123#if (defined(__sparc__) && defined(_LP64)) || \
124 defined(__alpha__) || \
125 (defined(__i386__) && defined(_LP64)) || \
126 (defined(__sh__) && defined(_LP64))
127 net.__n_pad0 = 0;
128#endif
129again:
130 p = fgets(line, (int)sizeof line, netf);
131 if (p == NULL)
132 return (NULL);
133 if (*p == '#')
134 goto again;
135 cp = strpbrk(p, "#\n");
136 if (cp == NULL)
137 goto again;
138 *cp = '\0';
139 net.n_name = p;
140 cp = strpbrk(p, " \t");
141 if (cp == NULL)
142 goto again;
143 *cp++ = '\0';
144 while (*cp == ' ' || *cp == '\t')
145 cp++;
146 p = strpbrk(cp, " \t");
147 if (p != NULL)
148 *p++ = '\0';
149 net.n_net = inet_network(cp);
150 net.n_addrtype = AF_INET;
151 q = net.n_aliases = net_aliases;
152 if (p != NULL) {
153 cp = p;
154 while (cp && *cp) {
155 if (*cp == ' ' || *cp == '\t') {
156 cp++;
157 continue;
158 }
159 if (q < &net_aliases[MAXALIASES - 1])
160 *q++ = cp;
161 cp = strpbrk(cp, " \t");
162 if (cp != NULL)
163 *cp++ = '\0';
164 }
165 }
166 *q = NULL;
167 return (&net);
168}
