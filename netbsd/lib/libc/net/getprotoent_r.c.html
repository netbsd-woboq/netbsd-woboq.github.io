56void
57setprotoent_r(int f, struct protoent_data *pd)
58{
59 if (pd->fp == NULL)
60 pd->fp = fopen(_PATH_PROTOCOLS, "re");
61 else
62 rewind(pd->fp);
63 pd->stayopen |= f;
64}
65
66void
67endprotoent_r(struct protoent_data *pd)
68{
69 if (pd->fp) {
70 (void)fclose(pd->fp);
71 pd->fp = NULL;
72 }
73 if (pd->aliases) {
74 free(pd->aliases);
75 pd->aliases = NULL;
76 pd->maxaliases = 0;
77 }
78 if (pd->line) {
79 free(pd->line);
80 pd->line = NULL;
81 }
82 pd->stayopen = 0;
83}
84
85struct protoent *
86getprotoent_r(struct protoent *pr, struct protoent_data *pd)
87{
88 char *p, *cp, **q;
89 size_t i = 0;
90 int oerrno;
91
92 if (pd->fp == NULL && (pd->fp = fopen(_PATH_PROTOCOLS, "re")) == NULL)
93 return NULL;
94
95 for (;;) {
96 if (pd->line)
97 free(pd->line);
98 pd->line = fparseln(pd->fp, NULL, NULL, NULL,
99 FPARSELN_UNESCALL);
100 if (pd->line == NULL)
101 return NULL;
102 pr->p_name = p = pd->line;
103 cp = strpbrk(p, " \t");
104 if (cp == NULL)
105 continue;
106 *cp++ = '\0';
107 while (*cp == ' ' || *cp == '\t')
108 cp++;
109 p = strpbrk(cp, " \t");
110 if (p != NULL)
111 *p++ = '\0';
112 pr->p_proto = atoi(cp);
113 if (pd->aliases == NULL) {
114 pd->maxaliases = 10;
115 pd->aliases = malloc(pd->maxaliases * sizeof(char *));
116 if (pd->aliases == NULL) {
117 oerrno = errno;
118 endprotoent_r(pd);
119 errno = oerrno;
120 return NULL;
121 }
122 }
123 q = pr->p_aliases = pd->aliases;
124 if (p != NULL) {
125 cp = p;
126 while (cp && *cp) {
127 if (*cp == ' ' || *cp == '\t') {
128 cp++;
129 continue;
130 }
131 if (i == pd->maxaliases - 2) {
132 pd->maxaliases *= 2;
133 q = realloc(q,
134 pd->maxaliases * sizeof(char *));
135 if (q == NULL) {
136 oerrno = errno;
137 endprotoent_r(pd);
138 errno = oerrno;
139 return NULL;
140 }
141 pr->p_aliases = pd->aliases = q;
142 }
143 q[i++] = cp;
144
145 cp = strpbrk(cp, " \t");
146 if (cp != NULL)
147 *cp++ = '\0';
148 }
149 }
150 q[i] = NULL;
151 return pr;
152 }
