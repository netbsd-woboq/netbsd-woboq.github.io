54static struct servent *
55_servent_getbyname(struct servent_data *sd, struct servent *sp,
56 const char *name, const char *proto)
57{
58
59 if ((sd->flags & (_SV_CDB | _SV_PLAINFILE)) == 0)
60 return NULL;
61
62 if (sd->flags & _SV_CDB) {
63 uint8_t buf[255 * 2 + 2];
64 size_t namelen, protolen;
65 const uint8_t *data, *data_end;
66 const void *data_ptr;
67 size_t datalen;
68
69 namelen = strlen(name);
70 if (namelen == 0 || namelen > 255)
71 return NULL;
72 if (proto != NULL) {
73 protolen = strlen(proto);
74 if (protolen == 0 || protolen > 255)
75 return NULL;
76 } else
77 protolen = 0;
78 if (namelen + protolen > 255)
79 return NULL;
80
81 buf[0] = (uint8_t)namelen;
82 buf[1] = (uint8_t)protolen;
83 memcpy(buf + 2, name, namelen);
84 memcpy(buf + 2 + namelen, proto, protolen);
85
86 if (cdbr_find(sd->cdb, buf, 2 + namelen + protolen,
87 &data_ptr, &datalen))
88 return NULL;
89
90 if (datalen < namelen + protolen + 6)
91 return NULL;
92
93 data = data_ptr;
94 data_end = data + datalen;
95 if (protolen) {
96 if (data[2] != protolen)
97 return NULL;
98 if (memcmp(data + 3, proto, protolen + 1))
99 return NULL;
100 }
101 data += 3 + data[2] + 1;
102 if (data > data_end)
103 return NULL;
104 while (data != data_end) {
105 if (*data == '\0')
106 return NULL;
107 if (data + data[0] + 2 > data_end)
108 return NULL;
109 if (data[0] == namelen &&
110 memcmp(data + 1, name, namelen + 1) == 0)
111 return _servent_parsedb(sd, sp, data_ptr,
112 datalen);
113 data += data[0] + 2;
114 }
115 return NULL;
116 } else {
117 while (_servent_getline(sd) != -1) {
118 char **cp;
119 if (_servent_parseline(sd, sp) == NULL)
120 continue;
121
122 if (strcmp(name, sp->s_name) == 0)
123 goto gotname;
124
125 for (cp = sp->s_aliases; *cp; cp++)
126 if (strcmp(name, *cp) == 0)
127 goto gotname;
128 continue;
129gotname:
130 if (proto == NULL || strcmp(sp->s_proto, proto) == 0)
131 return sp;
132 }
133 return NULL;
134 }
135}
136
137struct servent *
138getservbyname_r(const char *name, const char *proto, struct servent *sp,
139 struct servent_data *sd)
140{
141 _DIAGASSERT(name != NULL);
142 /* proto may be NULL */
143
144 setservent_r(sd->flags & _SV_STAYOPEN, sd);
145 sp = _servent_getbyname(sd, sp, name, proto);
146 if (!(sd->flags & _SV_STAYOPEN))
147 _servent_close(sd);
148 return sp;
149}
