53static struct servent *
54_servent_getbyport(struct servent_data *sd, struct servent *sp, int port,
55 const char *proto)
56{
57
58 if ((sd->flags & (_SV_CDB | _SV_PLAINFILE)) == 0)
59 return NULL;
60
61 if (sd->flags & _SV_CDB) {
62 uint8_t buf[255 + 4];
63 size_t protolen;
64 const uint8_t *data;
65 const void *data_ptr;
66 size_t datalen;
67
68 port = be16toh(port);
69
70 if (proto != NULL) {
71 protolen = strlen(proto);
72 if (protolen == 0 || protolen > 255)
73 return NULL;
74 } else
75 protolen = 0;
76 if (port < 0 || port > 65536)
77 return NULL;
78
79 buf[0] = 0;
80 buf[1] = (uint8_t)protolen;
81 be16enc(buf + 2, port);
82 memcpy(buf + 4, proto, protolen);
83
84 if (cdbr_find(sd->cdb, buf, 4 + protolen,
85 &data_ptr, &datalen))
86 return NULL;
87
88 if (datalen < protolen + 4)
89 return NULL;
90
91 data = data_ptr;
92 if (be16dec(data) != port)
93 return NULL;
94 if (protolen) {
95 if (data[2] != protolen)
96 return NULL;
97 if (memcmp(data + 3, proto, protolen + 1))
98 return NULL;
99 }
100 return _servent_parsedb(sd, sp, data, datalen);
101 } else {
102 while (_servent_getline(sd) != -1) {
103 if (_servent_parseline(sd, sp) == NULL)
104 continue;
105 if (sp->s_port != port)
106 continue;
107 if (proto == NULL || strcmp(sp->s_proto, proto) == 0)
108 return sp;
109 }
110 return NULL;
111 }
112}
113
114struct servent *
115getservbyport_r(int port, const char *proto, struct servent *sp,
116 struct servent_data *sd)
117{
118 setservent_r(sd->flags & _SV_STAYOPEN, sd);
119 sp = _servent_getbyport(sd, sp, port, proto);
120 if (!(sd->flags & _SV_STAYOPEN))
121 _servent_close(sd);
122 return sp;
123}
