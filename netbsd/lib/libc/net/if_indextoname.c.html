52/*
53 * From RFC 2533:
54 *
55 * The second function maps an interface index into its corresponding
56 * name.
57 *
58 * #include <net/if.h>
59 *
60 * char *if_indextoname(unsigned int ifindex, char *ifname);
61 *
62 * The ifname argument must point to a buffer of at least IF_NAMESIZE
63 * bytes into which the interface name corresponding to the specified
64 * index is returned. (IF_NAMESIZE is also defined in <net/if.h> and
65 * its value includes a terminating null byte at the end of the
66 * interface name.) This pointer is also the return value of the
67 * function. If there is no interface corresponding to the specified
68 * index, NULL is returned, and errno is set to ENXIO, if there was a
69 * system error (such as running out of memory), if_indextoname returns
70 * NULL and errno would be set to the proper value (e.g., ENOMEM).
71 */
72
73char *
74if_indextoname(unsigned int ifindex, char *ifname)
75{
76 struct ifaddrs *ifaddrs, *ifa;
77 int error = 0;
78
79 if (getifaddrs(&ifaddrs) < 0)
80 return(NULL); /* getifaddrs properly set errno */
81
82 for (ifa = ifaddrs; ifa != NULL; ifa = ifa->ifa_next) {
83 if (ifa->ifa_addr &&
84 ifa->ifa_addr->sa_family == AF_LINK &&
85 ifindex == ((struct sockaddr_dl*)
86 (void *)ifa->ifa_addr)->sdl_index)
87 break;
88 }
89
90 if (ifa == NULL) {
91 error = ENXIO;
92 ifname = NULL;
93 }
94 else
95 strlcpy(ifname, ifa->ifa_name, IFNAMSIZ);
96
97 freeifaddrs(ifaddrs);
98
99 errno = error;
100 return(ifname);
101}
