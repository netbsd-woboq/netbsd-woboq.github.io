54/*
55 * From RFC 2553:
56 *
57 * 4.1 Name-to-Index
58 *
59 *
60 * The first function maps an interface name into its corresponding
61 * index.
62 *
63 * #include <net/if.h>
64 *
65 * unsigned int if_nametoindex(const char *ifname);
66 *
67 * If the specified interface name does not exist, the return value is
68 * 0, and errno is set to ENXIO. If there was a system error (such as
69 * running out of memory), the return value is 0 and errno is set to the
70 * proper value (e.g., ENOMEM).
71 */
72
73unsigned int
74if_nametoindex(const char *ifname)
75{
76 int s;
77 struct ifreq ifr;
78 struct ifaddrs *ifaddrs, *ifa;
79 unsigned int ni;
80
81 s = socket(AF_INET, SOCK_DGRAM | SOCK_CLOEXEC, 0);
82 if (s != -1) {
83 memset(&ifr, 0, sizeof(ifr));
84 strlcpy(ifr.ifr_name, ifname, sizeof(ifr.ifr_name));
85 if (ioctl(s, SIOCGIFINDEX, &ifr) != -1) {
86 close(s);
87 return (ifr.ifr_index);
88 }
89 close(s);
90 }
91
92 if (getifaddrs(&ifaddrs) < 0)
93 return 0;
94
95 ni = 0;
96
97 for (ifa = ifaddrs; ifa != NULL; ifa = ifa->ifa_next) {
98 if (ifa->ifa_addr &&
99 ifa->ifa_addr->sa_family == AF_LINK &&
100 strcmp(ifa->ifa_name, ifname) == 0) {
101 ni = ((struct sockaddr_dl*)
102 (void *)ifa->ifa_addr)->sdl_index;
103 break;
104 }
105 }
106
107 freeifaddrs(ifaddrs);
108 if (!ni)
109 errno = ENXIO;
110 return ni;
111}
