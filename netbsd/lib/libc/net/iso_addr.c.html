46struct iso_addr {
47 uint8_t isoa_len; /* length (in bytes) */
48 char isoa_genaddr[20]; /* general opaque address */
49};
50
51__BEGIN_DECLS
52struct iso_addr *iso_addr(const char *);
53char *iso_ntoa(const struct iso_addr *);
54__END_DECLS
55
56/* States*/
57#define VIRGIN 0
58#define GOTONE 1
59#define GOTTWO 2
60/* Inputs */
61#define DIGIT (4*0)
62#define END (4*1)
63#define DELIM (4*2)
64
65struct iso_addr *
66iso_addr(const char *addr)
67{
68 static struct iso_addr out_addr;
69 register char *cp = out_addr.isoa_genaddr;
70 char *cplim = cp + sizeof(out_addr.isoa_genaddr);
71 register int byte = 0, state = VIRGIN;
72 register int newaddr = 0; /* pacify gcc */
73
74 _DIAGASSERT(addr != NULL);
75
76 (void)memset(&out_addr, 0, sizeof (out_addr));
77 do {
78 if ((*addr >= '0') && (*addr <= '9')) {
79 newaddr = *addr - '0';
80 } else if ((*addr >= 'a') && (*addr <= 'f')) {
81 newaddr = *addr - 'a' + 10;
82 } else if ((*addr >= 'A') && (*addr <= 'F')) {
83 newaddr = *addr - 'A' + 10;
84 } else if (*addr == 0)
85 state |= END;
86 else
87 state |= DELIM;
88 addr++;
89 switch (state /* | INPUT */) {
90 case GOTTWO | DIGIT:
91 *cp++ = byte; /*FALLTHROUGH*/
92 case VIRGIN | DIGIT:
93 state = GOTONE; byte = newaddr; continue;
94 case GOTONE | DIGIT:
95 state = GOTTWO; byte = newaddr + (byte << 4); continue;
96 default: /* | DELIM */
97 state = VIRGIN; *cp++ = byte; byte = 0; continue;
98 case GOTONE | END:
99 case GOTTWO | END:
100 *cp++ = byte; /* FALLTHROUGH */
101 case VIRGIN | END:
102 break;
103 }
104 break;
105 } while (cp < cplim);
106 _DIAGASSERT(__type_fit(uint8_t, cp - out_addr.isoa_genaddr));
107 out_addr.isoa_len = (uint8_t)(cp - out_addr.isoa_genaddr);
108 return (&out_addr);
109}

111static const char hexlist[16] = "0123456789abcdef";
112
113char *
114iso_ntoa(const struct iso_addr *isoa)
115{
116 static char obuf[64];
117 char *out = obuf;
118 size_t i;
119 const u_char *in = (const u_char *)isoa->isoa_genaddr;
120 const u_char *inlim = in + isoa->isoa_len;
121
122 _DIAGASSERT(isoa != NULL);
123
124 out[1] = 0;
125 while (in < inlim) {
126 i = *in++;
127 *out++ = '.';
128 if (i > 0xf) {
129 out[1] = hexlist[i & 0xf];
130 i >>= 4;
131 out[0] = hexlist[i];
132 out += 2;
133 } else
134 *out++ = hexlist[i];
135 }
136 *out = 0;
137 return(obuf + 1);
138}
