49char *
50_catgets(nl_catd catd, int set_id, int msg_id, const char *s)
51{
52 struct _nls_cat_hdr *cat_hdr;
53 struct _nls_set_hdr *set_hdr;
54 struct _nls_msg_hdr *msg_hdr;
55 int l, u, i, r;
56
57 if (catd == (nl_catd) -1) {
58 errno = EBADF;
59 return __UNCONST(s);
60 }
61
62 cat_hdr = (struct _nls_cat_hdr *)catd->__data;
63 set_hdr = (struct _nls_set_hdr *)(void *)((char *)catd->__data
64 + sizeof(struct _nls_cat_hdr));
65
66 /* binary search, see knuth algorithm b */
67 l = 0;
68 u = ntohl((u_int32_t)cat_hdr->__nsets) - 1;
69 while (l <= u) {
70 i = (l + u) / 2;
71 r = set_id - ntohl((u_int32_t)set_hdr[i].__setno);
72
73 if (r == 0) {
74 msg_hdr = (struct _nls_msg_hdr *)
75 (void *)((char *)catd->__data +
76 sizeof(struct _nls_cat_hdr) +
77 ntohl((u_int32_t)cat_hdr->__msg_hdr_offset));
78
79 l = ntohl((u_int32_t)set_hdr[i].__index);
80 u = l + ntohl((u_int32_t)set_hdr[i].__nmsgs) - 1;
81 while (l <= u) {
82 i = (l + u) / 2;
83 r = msg_id -
84 ntohl((u_int32_t)msg_hdr[i].__msgno);
85 if (r == 0) {
86 return ((char *) catd->__data +
87 sizeof(struct _nls_cat_hdr) +
88 ntohl((u_int32_t)
89 cat_hdr->__msg_txt_offset) +
90 ntohl((u_int32_t)
91 msg_hdr[i].__offset));
92 } else if (r < 0) {
93 u = i - 1;
94 } else {
95 l = i + 1;
96 }
97 }
98
99 /* not found */
100 goto notfound;
101
102 } else if (r < 0) {
103 u = i - 1;
104 } else {
105 l = i + 1;
106 }
107 }
108
109notfound:
110 /* not found */
111 errno = ENOMSG;
112 return __UNCONST(s);
113}
