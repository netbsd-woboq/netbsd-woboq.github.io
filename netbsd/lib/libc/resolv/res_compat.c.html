45#undef _res
46
47/*
48 * Binary Compatibility; this symbol does not appear in a header file
49 * Most userland programs use this to set res_options before res_init()
50 * is called. There are hooks to res_init() to consult the data in this
51 * structure. The hooks are provided indirectly by the two functions below.
52 * We depend on the fact the first 440 [32 bit machines] bytes are
53 * shared between the two structures.
54 */
55#ifndef __BIND_NOSTATIC
56struct __res_state _res
57#if defined(__BIND_RES_TEXT)
58 = { RES_TIMEOUT, } /* Motorola, et al. */
59# endif
60;
61
62void *__res_get_old_state(void);
63void __res_put_old_state(void *);
64
65void *
66__res_get_old_state(void)
67{
68 return &_res;
69}
70
71void
72__res_put_old_state(void *res)
73{
74 (void)memcpy(&_res, res, sizeof(_res));
75}
76#endif
