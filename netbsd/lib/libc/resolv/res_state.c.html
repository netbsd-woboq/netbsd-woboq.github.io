59res_state
60__res_get_state_nothread(void)
61{
62 if ((_nres.options & RES_INIT) == 0 && res_ninit(&_nres) == -1) {
63 h_errno = NETDB_INTERNAL;
64 return NULL;
65 }
66 return &_nres;
67}
