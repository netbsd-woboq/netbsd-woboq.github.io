38#include <sys/types.h>
39
40#include "namespace.h"
41
42#include <stdlib.h>
43#include <randomid.h>
44#include <rpc/rpc.h>
45#include "rpc_internal.h"
46
47u_int32_t
48__rpc_getxid(void)
49{
50 static randomid_t ctx = NULL;
51
52 if (!ctx) {
53 ctx = randomid_new(32, RANDOMID_TIMEO_DEFAULT);
54 if (!ctx)
55 abort();
56 }
57
58 return randomid(ctx);
59}
