65int
66getrpcport(char *host, int prognum, int versnum, int proto)
67{
68 struct sockaddr_in addr;
69 struct hostent *hp;
70
71 _DIAGASSERT(host != NULL);
72
73 if ((hp = gethostbyname(host)) == NULL)
74 return (0);
75 memset(&addr, 0, sizeof(addr));
76 addr.sin_len = sizeof(struct sockaddr_in);
77 addr.sin_family = AF_INET;
78 addr.sin_port = 0;
79 if (hp->h_length > addr.sin_len)
80 hp->h_length = addr.sin_len;
81 memcpy(&addr.sin_addr.s_addr, hp->h_addr, (size_t)hp->h_length);
82 /* Inconsistent interfaces need casts! :-( */
83 return (pmap_getport(&addr, (u_long)prognum, (u_long)versnum,
84 (u_int)proto));
85}
