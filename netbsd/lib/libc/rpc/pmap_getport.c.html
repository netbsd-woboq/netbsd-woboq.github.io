69static const struct timeval timeout = { 5, 0 };
70static const struct timeval tottimeout = { 60, 0 };
71
72/*
73 * Find the mapped port for program,version.
74 * Calls the pmap service remotely to do the lookup.
75 * Returns 0 if no map exists.
76 */
77
78static void
79remote_pmap_getport(CLIENT *client, struct pmap *parms, u_short *port)
80{
81 if (CLNT_CALL(client, (rpcproc_t)PMAPPROC_GETPORT, (xdrproc_t)xdr_pmap,
82 parms, (xdrproc_t)xdr_u_short, port, tottimeout) != RPC_SUCCESS) {
83 rpc_createerr.cf_stat = RPC_PMAPFAILURE;
84 clnt_geterr(client, &rpc_createerr.cf_error);
85 } else if (*port == 0) {
86 rpc_createerr.cf_stat = RPC_PROGNOTREGISTERED;
87 clnt_geterr(client, &rpc_createerr.cf_error);
88 }
89 CLNT_DESTROY(client);
90}
91
92static CLIENT *
93get_client(struct sockaddr_in *address, int tcp)
94{
95 int sock = -1;
96 if (tcp)
97 return clnttcp_create(address, PMAPPROG, PMAPVERS, &sock, 0, 0);
98 else
99 return clntudp_bufcreate(address, PMAPPROG, PMAPVERS, timeout,
100 &sock, RPCSMALLMSGSIZE, RPCSMALLMSGSIZE);
101}
102
103u_short
104pmap_getport(struct sockaddr_in *address, u_long program, u_long version,
105 u_int protocol)
106{
107 u_short port = 0;
108 CLIENT *client;
109 struct pmap parms;
110
111 _DIAGASSERT(address != NULL);
112
113 parms.pm_prog = program;
114 parms.pm_vers = version;
115 parms.pm_prot = protocol;
116 parms.pm_port = 0; /* not needed or used */
117
118 address->sin_port = htons(PMAPPORT);
119
120 client = get_client(address, protocol == IPPROTO_TCP);
121 if (client != NULL)
122 remote_pmap_getport(client, &parms, &port);
123
124 if (port == 0) {
125 client = get_client(address, protocol != IPPROTO_TCP);
126 if (client != NULL)
127 remote_pmap_getport(client, &parms, &port);
128 }
129
130 address->sin_port = 0;
131 return port;
132}
