50/*
51 * Cache the result of sysconf(_SC_OPEN_MAX), so we don't have to do an
52 * expensive system call every time.
53 */
54int
55_rpc_dtablesize(void)
56{
57 static int size;
58 if (size == 0)
59 size = (int)sysconf(_SC_OPEN_MAX);
60 return (size);
61}
