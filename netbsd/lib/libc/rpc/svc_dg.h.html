40 * kept in xprt->xp_p2

42struct svc_dg_data {
43 /* XXX: optbuf should be the first field, used by ti_opts.c code */
44 size_t su_iosz; /* size of send.recv buffer */
45 u_int32_t su_xid; /* transaction id */
46 XDR su_xdrs; /* XDR handle */
47 char su_verfbody[MAX_AUTH_BYTES]; /* verifier body */
48 void *su_cache; /* cached data, NULL if none */
49};
