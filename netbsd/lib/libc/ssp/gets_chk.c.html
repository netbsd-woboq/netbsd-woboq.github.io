36#include <ssp/ssp.h>
37#include <stdio.h>
38#include <string.h>
39#include <limits.h>
40#include <stdlib.h>
41#include <ssp/stdio.h>
42
43extern char *__gets(char *);
44#undef gets
45
46char *
47__gets_chk(char * __restrict buf, size_t slen)
48{
49 char *abuf;
50 size_t len;
51
52 if (slen >= (size_t)INT_MAX)
53 return __gets(buf);
54
55 if ((abuf = malloc(slen + 1)) == NULL)
56 return __gets(buf);
57
58 if (fgets(abuf, (int)(slen + 1), stdin) == NULL) {
59 free(abuf);
60 return NULL;
61 }
62
63 len = strlen(abuf);
64 if (len > 0 && abuf[len - 1] == '\n')
65 --len;
66
67 if (len >= slen)
68 __chk_fail();
69
70 (void)memcpy(buf, abuf, len);
71
72 buf[len] = '\0';
73 free(abuf);
74 return buf;
75}
