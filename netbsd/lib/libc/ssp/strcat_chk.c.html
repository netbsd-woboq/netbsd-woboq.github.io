36#include <ssp/ssp.h>
37#include <string.h>
38
39char *__strcat_chk(char * __restrict, const char * __restrict, size_t);
40
41char *
42__strcat_chk(char * __restrict dst, const char * __restrict src, size_t slen)
43{
44 char *d;
45
46 for (d = dst; *d; d++) {
47 if (slen-- == 0)
48 __chk_fail();
49 }
50
51 while (*src) {
52 if (slen-- == 0)
53 __chk_fail();
54 *d++ = *src++;
55 }
56
57 if (slen-- == 0)
58 __chk_fail();
59
60 *d = '\0';
61 return dst;
62}
