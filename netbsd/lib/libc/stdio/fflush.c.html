55/* Flush a single file, or (if fp is NULL) all files. */
56int
57fflush(FILE *fp)
58{
59 int r;
60
61 if (fp == NULL) {
62 rwlock_rdlock(&__sfp_lock);
63 r = _fwalk(__sflush);
64 rwlock_unlock(&__sfp_lock);
65 return r;
66 }
67
68 FLOCKFILE(fp);
69 if ((fp->_flags & (__SWR | __SRW)) == 0) {
70 errno = EBADF;
71 r = EOF;
72 } else {
73 r = __sflush(fp);
74 }
75 FUNLOCKFILE(fp);
76 return r;
77}
78
79int
80__sflush(FILE *fp)
81{
82 unsigned char *p;
83 size_t n;
84 ssize_t t;
85
86 _DIAGASSERT(fp != NULL);
87
88 t = fp->_flags;
89 if ((t & __SWR) == 0)
90 return 0;
91
92 if ((p = fp->_bf._base) == NULL)
93 return 0;
94
95 ptrdiff_t tp = fp->_p - p;
96 _DIAGASSERT(__type_fit(ssize_t, tp));
97 n = (ssize_t)tp; /* write this much */
98
99 /*
100 * Set these immediately to avoid problems with longjmp and to allow
101 * exchange buffering (via setvbuf) in user write function.
102 */
103 fp->_p = p;
104 fp->_w = t & (__SLBF|__SNBF) ? 0 : fp->_bf._size;
105
106 for (; n > 0; n -= t, p += t) {
107 t = (*fp->_write)(fp->_cookie, (char *)p, n);
108 if (t <= 0) {
109 fp->_flags |= __SERR;
110 return EOF;
111 }
112 }
113 if (fp->_flush)
114 return (*fp->_flush)(fp->_cookie);
115 return 0;
116}
