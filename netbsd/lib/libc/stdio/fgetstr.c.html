43/*
44 * Get an input line.
45 * This now uses getdelim(3) for a code reduction.
46 * The upside is that strings are now always null-terminated, but relying
47 * on this is non portable - better to use the POSIX getdelim(3) function.
48 */
49char *
50__fgetstr(FILE *__restrict fp, size_t *__restrict lenp, int sep)
51{
52 ssize_t n;
53
54 _DIAGASSERT(fp != NULL);
55 _DIAGASSERT(lenp != NULL);
56
57 n = __getdelim(&_EXT(fp)->_fgetstr_buf, &_EXT(fp)->_fgetstr_len, sep, fp);
58 if (n == -1) {
59 *lenp = 0;
60 if (__sferror(fp) && errno == EOVERFLOW) /* fixup errno */
61 errno = EINVAL;
62 return NULL;
63 }
64 *lenp = n;
65 return _EXT(fp)->_fgetstr_buf;
66}
