52/*
53 * Expand the line buffer. Return -1 on error.
54#ifdef notdef
55 * The `new size' does not account for a terminating '\0',
56 * so we add 1 here.
57#endif
58 */
59static int
60__slbexpand(FILE *fp, size_t newsize)
61{
62 void *p;
63
64#ifdef notdef
65 ++newsize;
66#endif
67 _DIAGASSERT(fp != NULL);
68
69 if (_EXT(fp)->_fgetstr_len >= newsize)
70 return 0;
71 if ((p = realloc(_EXT(fp)->_fgetstr_buf, newsize)) == NULL)
72 return -1;
73 _EXT(fp)->_fgetstr_buf = p;
74 _EXT(fp)->_fgetstr_len = newsize;
75 return 0;
76}

78wchar_t *
79fgetwln(FILE * __restrict fp, size_t *lenp)
80{
81 wint_t wc;
82 size_t len;
83
84 FLOCKFILE(fp);
85 _SET_ORIENTATION(fp, 1);
86
87 len = 0;
88 while ((wc = __fgetwc_unlock(fp)) != WEOF) {
89#define GROW 512
90 if (len * sizeof(wchar_t) >= _EXT(fp)->_fgetstr_len &&
91 __slbexpand(fp, (len + GROW) * sizeof(wchar_t))) {
92 fp->_flags |= __SERR;
93 goto error;
94 }
95 *((wchar_t *)(void *)_EXT(fp)->_fgetstr_buf + len++) = wc;
96 if (wc == L'\n')
97 break;
98 }
99
100 /*
101 * The following test assumes that fgetwc() fails when
102 * feof() is already set, and that fgetwc() will never
103 * set feof() in the same call where it also sets ferror()
104 * or returns non-WEOF.
105 * Testing ferror() would not be better because fgetwc()
106 * may succeed even when ferror() is already set.
107 */
108
109 if (len == 0 || (wc == WEOF && !__sfeof(fp)))
110 goto error;
111
112 FUNLOCKFILE(fp);
113 *lenp = len;
114 return (wchar_t *)(void *)_EXT(fp)->_fgetstr_buf;
115
116error:
117 FUNLOCKFILE(fp);
118 *lenp = 0;
119 return NULL;
120}
