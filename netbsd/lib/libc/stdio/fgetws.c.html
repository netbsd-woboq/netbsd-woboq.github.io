33#include <sys/cdefs.h>
34#if defined(LIB_SCCS) && !defined(lint)

36#endif

38#include <assert.h>
39#include <errno.h>
40#include <stdio.h>
41#include <wchar.h>
42#include "reentrant.h"
43#include "local.h"
44
45wchar_t *
46fgetws(wchar_t * __restrict ws, int n, FILE * __restrict fp)
47{
48 wchar_t *wsp;
49 wint_t wc;
50
51 _DIAGASSERT(fp != NULL);
52 _DIAGASSERT(ws != NULL);
53
54 FLOCKFILE(fp);
55 _SET_ORIENTATION(fp, 1);
56
57 if (n <= 0) {
58 errno = EINVAL;
59 goto error;
60 }
61
62 wsp = ws;
63 while (n-- > 1) {
64 wc = __fgetwc_unlock(fp);
65 if (__sferror(fp) != 0)
66 goto error;
67 if (__sfeof(fp) != 0) {
68 if (wsp == ws) {
69 /* EOF/error, no characters read yet. */
70 goto error;
71 }
72 break;
73 }
74 *wsp++ = (wchar_t)wc;
75 if (wc == L'\n') {
76 break;
77 }
78 }
79
80 *wsp++ = L'\0';
81 FUNLOCKFILE(fp);
82
83 return ws;
84
85error:
86 FUNLOCKFILE(fp);
87 return NULL;
88}
