55FILE *
56fopen(const char *file, const char *mode)
57{
58 FILE *fp;
59 int f;
60 int flags, oflags;
61
62 _DIAGASSERT(file != NULL);
63 if ((flags = __sflags(mode, &oflags)) == 0)
64 return NULL;
65 if ((fp = __sfp()) == NULL)
66 return NULL;
67 if ((f = open(file, oflags, DEFFILEMODE)) < 0)
68 goto release;
69 /*
70 * File descriptors are a full int, but _file is only a short.
71 * If we get a valid file descriptor that is greater or equal to
72 * USHRT_MAX, then the fd will get sign-extended into an
73 * invalid file descriptor. Handle this case by failing the
74 * open. (We treat the short as unsigned, and special-case -1).
75 */
76 if (f >= USHRT_MAX) {
77 (void)close(f);
78 errno = EMFILE;
79 goto release;
80 }
81
82 fp->_file = f;
83 fp->_flags = flags;
84 fp->_cookie = fp;
85 fp->_read = __sread;
86 fp->_write = __swrite;
87 fp->_seek = __sseek;
88 fp->_close = __sclose;
89
90 /*
91 * When opening in append mode, even though we use O_APPEND,
92 * we need to seek to the end so that ftell() gets the right
93 * answer. If the user then alters the seek pointer, or
94 * the file extends, this will fail, but there is not much
95 * we can do about this. (We could set __SAPP and check in
96 * fseek and ftell.)
97 */
98 if (oflags & O_APPEND)
99 (void) __sseek((void *)fp, (off_t)0, SEEK_END);
100 return fp;
101release:
102 fp->_flags = 0; /* release */
103 return NULL;
104}
