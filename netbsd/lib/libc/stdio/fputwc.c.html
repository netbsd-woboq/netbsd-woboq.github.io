45wint_t
46__fputwc_unlock(wchar_t wc, FILE *fp)
47{
48 struct wchar_io_data *wcio;
49 mbstate_t *st;
50 size_t size;
51 char buf[MB_LEN_MAX];
52 struct __suio uio;
53 struct __siov iov;
54
55 _DIAGASSERT(fp != NULL);
56
57 /* LINTED we don't play with buf */
58 iov.iov_base = (void *)buf;
59 uio.uio_iov = &iov;
60 uio.uio_iovcnt = 1;
61
62 _SET_ORIENTATION(fp, 1);
63 wcio = WCIO_GET(fp);
64 if (wcio == 0) {
65 errno = ENOMEM;
66 return WEOF;
67 }
68
69 wcio->wcio_ungetwc_inbuf = 0;
70 st = &wcio->wcio_mbstate_out;
71
72 size = wcrtomb(buf, wc, st);
73 if (size == (size_t)-1) {
74 errno = EILSEQ;
75 return WEOF;
76 }
77
78 _DIAGASSERT(size != 0);
79
80 uio.uio_resid = iov.iov_len = size;
81 if (__sfvwrite(fp, &uio)) {
82 return WEOF;
83 }
84
85 return (wint_t)wc;
86}
87
88wint_t
89fputwc(wchar_t wc, FILE *fp)
90{
91 wint_t r;
92
93 _DIAGASSERT(fp != NULL);
94
95 FLOCKFILE(fp);
96 r = __fputwc_unlock(wc, fp);
97 FUNLOCKFILE(fp);
98
99 return r;
100}
