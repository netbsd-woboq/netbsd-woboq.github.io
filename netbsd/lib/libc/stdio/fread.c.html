51size_t
52fread(void *buf, size_t size, size_t count, FILE *fp)
53{
54 size_t resid;
55 char *p;
56 int r;
57 size_t total;
58
59 _DIAGASSERT(fp != NULL);
60 /*
61 * The ANSI standard requires a return value of 0 for a count
62 * or a size of 0. Whilst ANSI imposes no such requirements on
63 * fwrite, the SUSv2 does.
64 */
65 if ((resid = count * size) == 0)
66 return 0;
67
68 _DIAGASSERT(buf != NULL);
69
70 FLOCKFILE(fp);
71 if (fp->_r < 0)
72 fp->_r = 0;
73 total = resid;
74 p = buf;
75 while (resid > (size_t)(r = fp->_r)) {
76 (void)memcpy(p, fp->_p, (size_t)r);
77 fp->_p += r;
78 /* fp->_r = 0 ... done in __srefill */
79 p += r;
80 resid -= r;
81 if (__srefill(fp)) {
82 /* no more input: return partial result */
83 FUNLOCKFILE(fp);
84 return (total - resid) / size;
85 }
86 }
87 (void)memcpy(p, fp->_p, resid);
88
89 _DIAGASSERT(__type_fit(int, fp->_r - resid));
90 fp->_r -= (int)resid;
91 fp->_p += resid;
92 FUNLOCKFILE(fp);
93 return count;
94}
