50/*
51 * ftell: return current offset.
52 */
53long
54ftell(FILE *fp)
55{
56 off_t pos;
57
58
59 FLOCKFILE(fp);
60
61 if (fp->_seek == NULL) {
62 FUNLOCKFILE(fp);
63 errno = ESPIPE; /* historic practice */
64 return -1L;
65 }
66
67 /*
68 * Find offset of underlying I/O object, then
69 * adjust for buffered bytes.
70 */
71 (void)__sflush(fp); /* may adjust seek offset on append stream */
72 if (fp->_flags & __SOFF)
73 pos = fp->_offset;
74 else {
75 pos = (*fp->_seek)(fp->_cookie, (off_t)0, SEEK_CUR);
76 if (pos == -1L) {
77 FUNLOCKFILE(fp);
78 return (long)pos;
79 }
80 }
81 if (fp->_flags & __SRD) {
82 /*
83 * Reading. Any unread characters (including
84 * those from ungetc) cause the position to be
85 * smaller than that in the underlying object.
86 */
87 pos -= fp->_r;
88 if (HASUB(fp))
89 pos -= fp->_ur;
90 } else if (fp->_flags & __SWR && fp->_p != NULL) {
91 /*
92 * Writing. Any buffered characters cause the
93 * position to be greater than that in the
94 * underlying object.
95 */
96 pos += fp->_p - fp->_bf._base;
97 }
98 FUNLOCKFILE(fp);
99
100 if (__long_overflow(pos)) {
101 errno = EOVERFLOW;
102 return -1L;
103 }
104
105 return (long)pos;
106}
