51/*
52 * ftell: return current offset.
53 */
54off_t
55ftello(FILE *fp)
56{
57 off_t pos;
58
59
60 FLOCKFILE(fp);
61
62 if (fp->_seek == NULL) {
63 FUNLOCKFILE(fp);
64 errno = ESPIPE; /* historic practice */
65 return (off_t)-1;
66 }
67
68 /*
69 * Find offset of underlying I/O object, then
70 * adjust for buffered bytes.
71 */
72 (void)__sflush(fp); /* may adjust seek offset on append stream */
73 if (fp->_flags & __SOFF)
74 pos = fp->_offset;
75 else {
76 pos = (*fp->_seek)(fp->_cookie, (off_t)0, SEEK_CUR);
77 if (pos == (off_t)-1) {
78 FUNLOCKFILE(fp);
79 return pos;
80 }
81 }
82 if (fp->_flags & __SRD) {
83 /*
84 * Reading. Any unread characters (including
85 * those from ungetc) cause the position to be
86 * smaller than that in the underlying object.
87 */
88 pos -= fp->_r;
89 if (HASUB(fp))
90 pos -= fp->_ur;
91 } else if (fp->_flags & __SWR && fp->_p != NULL) {
92 /*
93 * Writing. Any buffered characters cause the
94 * position to be greater than that in the
95 * underlying object.
96 */
97 pos += fp->_p - fp->_bf._base;
98 }
99 FUNLOCKFILE(fp);
100 return pos;
101}
