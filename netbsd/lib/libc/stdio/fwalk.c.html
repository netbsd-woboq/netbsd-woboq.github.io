51int
52_fwalk(int (*function)(FILE *))
53{
54 FILE *fp;
55 int n, ret;
56 struct glue *g;
57
58 _DIAGASSERT(function != NULL);
59
60 ret = 0;
61 for (g = &__sglue; g != NULL; g = g->next)
62 for (fp = g->iobs, n = g->niobs; --n >= 0; fp++)
63 if (fp->_flags != 0)
64 ret |= (*function)(fp);
65 return ret;
66}
