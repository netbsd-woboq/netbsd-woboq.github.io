36#include <assert.h>
37#include <stdio.h>
38#include <wchar.h>
39#include "reentrant.h"
40#include "local.h"
41
42int
43fwide(FILE *fp, int mode)
44{
45 struct wchar_io_data *wcio;
46
47 _DIAGASSERT(fp != NULL);
48
49 /*
50 * this implementation use only -1, 0, 1
51 * for mode value.
52 * (we don't need to do this, but
53 * this can make things simpler.)
54 */
55 if (mode > 0)
56 mode = 1;
57 else if (mode < 0)
58 mode = -1;
59
60 FLOCKFILE(fp);
61 wcio = WCIO_GET(fp);
62 if (!wcio)
63 return 0; /* XXX */
64
65 if (wcio->wcio_mode == 0 && mode != 0)
66 wcio->wcio_mode = mode;
67 else
68 mode = wcio->wcio_mode;
69 FUNLOCKFILE(fp);
70
71 return mode;
72}
