57/*
58 * Override the file buffering based on the environment setting STDBUF%d
59 * (for the specific file descriptor) and STDBUF (for all descriptors).
60 * the setting is ULF<num> standing for "Unbuffered", "Linebuffered",
61 * and Fullybuffered", and <num> is a value from 0 to 1M.
62 */
63static int
64__senvbuf(FILE *fp, size_t *size, int *couldbetty)
65{
66 char evb[64], *evp;
67 int flags, e;
68 intmax_t s;
69
70 flags = 0;
71 if (snprintf(evb, sizeof(evb), "STDBUF%d", fp->_file) < 0)
72 return flags;
73
74 if ((evp = getenv(evb)) == NULL && (evp = getenv("STDBUF")) == NULL)
75 return flags;
76
77 switch (*evp) {
78 case 'u':
79 case 'U':
80 evp++;
81 flags |= __SNBF;
82 break;
83 case 'l':
84 case 'L':
85 evp++;
86 flags |= __SLBF;
87 break;
88 case 'f':
89 case 'F':
90 evp++;
91 *couldbetty = 0;
92 break;
93 }
94
95 if (!isdigit((unsigned char)*evp))
96 return flags;
97
98 s = strtoi(evp, NULL, 0, 0, 1024 * 1024, &e);
99 if (e != 0)
100 return flags;
101
102 *size = (size_t)s;
103 if (*size == 0)
104 return __SNBF;
105
106 return flags;
107}

109/*
110 * Allocate a file buffer, or switch to unbuffered I/O.
111 * Per the ANSI C standard, ALL tty devices default to line buffered.
112 *
113 * As a side effect, we set __SOPT or __SNPT (en/dis-able fseek
114 * optimisation) right after the fstat() that finds the buffer size.
115 */
116void
117__smakebuf(FILE *fp)
118{
119 void *p;
120 int flags;
121 size_t size;
122 int couldbetty;
123
124 _DIAGASSERT(fp != NULL);
125
126 if (fp->_flags & __SNBF)
127 goto unbuf;
128
129 flags = __swhatbuf(fp, &size, &couldbetty);
130
131 if ((fp->_flags & (__SLBF|__SNBF|__SMBF)) == 0
132 && fp->_cookie == fp && fp->_file >= 0) {
133 flags |= __senvbuf(fp, &size, &couldbetty);
134 if (flags & __SNBF)
135 goto unbuf;
136 }
137
138 if ((p = malloc(size)) == NULL)
139 goto unbuf;
140
141 __cleanup = _cleanup;
142 flags |= __SMBF;
143 fp->_bf._base = fp->_p = p;
144 _DIAGASSERT(__type_fit(int, size));
145 fp->_bf._size = (int)size;
146 if (couldbetty && isatty(__sfileno(fp)))
147 flags |= __SLBF;
148 fp->_flags |= flags;
149 return;
150unbuf:
151 fp->_flags |= __SNBF;
152 fp->_bf._base = fp->_p = fp->_nbuf;
153 fp->_bf._size = 1;
154}

156/*
157 * Internal routine to determine `proper' buffering for a file.
158 */
159int
160__swhatbuf(FILE *fp, size_t *bufsize, int *couldbetty)
161{
162 struct stat st;
163
164 _DIAGASSERT(fp != NULL);
165 _DIAGASSERT(bufsize != NULL);
166 _DIAGASSERT(couldbetty != NULL);
167
168 if (__sfileno(fp) == -1 || fstat(__sfileno(fp), &st) < 0) {
169 *couldbetty = 0;
170 *bufsize = BUFSIZ;
171 return __SNPT;
172 }
173
174 /* could be a tty iff it is a character device */
175 *couldbetty = S_ISCHR(st.st_mode);
176 if (st.st_blksize == 0) {
177 *bufsize = BUFSIZ;
178 return __SNPT;
179 }
180
181 /*
182 * Optimise fseek() only if it is a regular file. (The test for
183 * __sseek is mainly paranoia.) It is safe to set _blksize
184 * unconditionally; it will only be used if __SOPT is also set.
185 */
186 *bufsize = st.st_blksize;
187 fp->_blksize = st.st_blksize;
188 return (st.st_mode & S_IFMT) == S_IFREG && fp->_seek == __sseek ?
189 __SOPT : __SNPT;
190}
