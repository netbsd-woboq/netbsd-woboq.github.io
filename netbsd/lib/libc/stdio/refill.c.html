55static int
56lflush(FILE *fp)
57{
58
59 _DIAGASSERT(fp != NULL);
60
61 if ((fp->_flags & (__SLBF|__SWR)) == (__SLBF|__SWR))
62 return __sflush(fp);
63 return 0;
64}

66/*
67 * Refill a stdio buffer.
68 * Return EOF on eof or error, 0 otherwise.
69 */
70int
71__srefill(FILE *fp)
72{
73
74 _DIAGASSERT(fp != NULL);
75
76 /* make sure stdio is set up */
77 if (!__sdidinit)
78 __sinit();
79
80 fp->_r = 0; /* largely a convenience for callers */
81
82 /* SysV does not make this test; take it out for compatibility */
83 if (fp->_flags & __SEOF)
84 return EOF;
85
86 /* if not already reading, have to be reading and writing */
87 if ((fp->_flags & __SRD) == 0) {
88 if ((fp->_flags & __SRW) == 0) {
89 errno = EBADF;
90 fp->_flags |= __SERR;
91 return EOF;
92 }
93 /* switch to reading */
94 if (fp->_flags & __SWR) {
95 if (__sflush(fp))
96 return EOF;
97 fp->_flags &= ~__SWR;
98 fp->_w = 0;
99 fp->_lbfsize = 0;
100 }
101 fp->_flags |= __SRD;
102 } else {
103 /*
104 * We were reading. If there is an ungetc buffer,
105 * we must have been reading from that. Drop it,
106 * restoring the previous buffer (if any). If there
107 * is anything in that buffer, return.
108 */
109 if (HASUB(fp)) {
110 FREEUB(fp);
111 if ((fp->_r = fp->_ur) != 0) {
112 fp->_p = fp->_up;
113 return 0;
114 }
115 }
116 }
117
118 if (fp->_bf._base == NULL)
119 __smakebuf(fp);
120
121 /*
122 * Before reading from a line buffered or unbuffered file,
123 * flush all line buffered output files, per the ANSI C
124 * standard.
125 */
126 if (fp->_flags & (__SLBF|__SNBF)) {
127 rwlock_rdlock(&__sfp_lock);
128 (void) _fwalk(lflush);
129 rwlock_unlock(&__sfp_lock);
130 }
131 fp->_p = fp->_bf._base;
132 fp->_r = (int)(*fp->_read)(fp->_cookie, (char *)fp->_p,
133 (size_t)fp->_bf._size);
134 fp->_flags &= ~__SMOD; /* buffer contents are again pristine */
135 if (fp->_r <= 0) {
136 if (fp->_r == 0)
137 fp->_flags |= __SEOF;
138 else {
139 fp->_r = 0;
140 fp->_flags |= __SERR;
141 }
142 return EOF;
143 }
144 return 0;
145}
