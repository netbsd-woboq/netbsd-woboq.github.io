55/*
56 * Small standard I/O/seek/close functions.
57 * These maintain the `known seek offset' for seek optimisation.
58 */
59ssize_t
60__sread(void *cookie, void *buf, size_t n)
61{
62 FILE *fp = cookie;
63 ssize_t ret;
64
65 _DIAGASSERT(cookie != NULL);
66 _DIAGASSERT(cookie == fp->_cookie);
67 _DIAGASSERT(buf != NULL);
68
69 ret = read(__sfileno(fp), buf, n);
70 /* if the read succeeded, update the current offset */
71 if (ret >= 0)
72 fp->_offset += ret;
73 else
74 fp->_flags &= ~__SOFF; /* paranoia */
75
76 return ret;
77}
78
79ssize_t
80__swrite(void *cookie, const void *buf, size_t n)
81{
82 FILE *fp = cookie;
83
84 _DIAGASSERT(cookie != NULL);
85 _DIAGASSERT(cookie == fp->_cookie);
86 _DIAGASSERT(buf != NULL);
87
88 if (fp->_flags & __SAPP)
89 if (lseek(__sfileno(fp), (off_t)0, SEEK_END) == (off_t)-1) {
90 if (errno == ESPIPE) /* if unseekable, OK, */
91 fp->_flags &= ~__SAPP; /* all writes append. */
92 else
93 return -1;
94 }
95 fp->_flags &= ~__SOFF; /* in case FAPPEND mode is set */
96 return write(__sfileno(fp), buf, n);
97}
98
99off_t
100__sseek(void *cookie, off_t offset, int whence)
101{
102 FILE *fp = cookie;
103 off_t ret;
104
105 _DIAGASSERT(cookie != NULL);
106 _DIAGASSERT(cookie == fp->_cookie);
107
108 ret = lseek(__sfileno(fp), offset, whence);
109 if (ret == (off_t)-1L)
110 fp->_flags &= ~__SOFF;
111 else {
112 fp->_flags |= __SOFF;
113 fp->_offset = ret;
114 }
115 return ret;
116}
117
118int
119__sclose(void *cookie)
120{
121 FILE *fp = cookie;
122
123 _DIAGASSERT(cookie != NULL);
124 _DIAGASSERT(cookie == fp->_cookie);
125
126 return close(__sfileno(fp));
127}
