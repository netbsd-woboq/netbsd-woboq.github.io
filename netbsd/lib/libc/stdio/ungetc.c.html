52/*
53 * Expand the ungetc buffer `in place'. That is, adjust fp->_p when
54 * the buffer moves, so that it points the same distance from the end,
55 * and move the bytes in the buffer around as necessary so that they
56 * are all at the end (stack-style).
57 */
58static int
59__submore(FILE *fp)
60{
61 int i;
62 unsigned char *p;
63
64 _DIAGASSERT(fp != NULL);
65
66 if (_UB(fp)._base == fp->_ubuf) {
67 /*
68 * Get a new buffer (rather than expanding the old one).
69 */
70 if ((p = malloc((size_t)BUFSIZ)) == NULL)
71 return EOF;
72 _UB(fp)._base = p;
73 _UB(fp)._size = BUFSIZ;
74 p += BUFSIZ - sizeof(fp->_ubuf);
75 for (i = sizeof(fp->_ubuf); --i >= 0;)
76 p[i] = fp->_ubuf[i];
77 fp->_p = p;
78 return 0;
79 }
80 i = _UB(fp)._size;
81 p = realloc(_UB(fp)._base, (size_t)(i << 1));
82 if (p == NULL)
83 return EOF;
84 /* no overlap (hence can use memcpy) because we doubled the size */
85 (void)memcpy((p + i), p, (size_t)i);
86 fp->_p = p + i;
87 _UB(fp)._base = p;
88 _UB(fp)._size = i << 1;
89 return 0;
90}

92int
93ungetc(int c, FILE *fp)
94{
95
96 _DIAGASSERT(fp != NULL);
97
98 if (c == EOF)
99 return EOF;
100 if (!__sdidinit)
101 __sinit();
102 FLOCKFILE(fp);
103 _SET_ORIENTATION(fp, -1);
104 if ((fp->_flags & __SRD) == 0) {
105 /*
106 * Not already reading: no good unless reading-and-writing.
107 * Otherwise, flush any current write stuff.
108 */
109 if ((fp->_flags & __SRW) == 0) {
110 FUNLOCKFILE(fp);
111 return EOF;
112 }
113 if (fp->_flags & __SWR) {
114 if (__sflush(fp)) {
115 FUNLOCKFILE(fp);
116 return EOF;
117 }
118 fp->_flags &= ~__SWR;
119 fp->_w = 0;
120 fp->_lbfsize = 0;
121 }
122 fp->_flags |= __SRD;
123 }
124 c = (unsigned char)c;
125
126 /*
127 * If we are in the middle of ungetc'ing, just continue.
128 * This may require expanding the current ungetc buffer.
129 */
130 if (HASUB(fp)) {
131 if (fp->_r >= _UB(fp)._size && __submore(fp)) {
132 FUNLOCKFILE(fp);
133 return EOF;
134 }
135 *--fp->_p = c;
136 fp->_r++;
137 FUNLOCKFILE(fp);
138 return c;
139 }
140 fp->_flags &= ~__SEOF;
141
142 /*
143 * If we can handle this by simply backing up, do so,
144 * but never replace the original character.
145 * (This makes sscanf() work when scanning `const' data.)
146 */
147 if (fp->_bf._base != NULL && fp->_p > fp->_bf._base &&
148 fp->_p[-1] == c) {
149 fp->_p--;
150 fp->_r++;
151 FUNLOCKFILE(fp);
152 return c;
153 }
154
155 /*
156 * Create an ungetc buffer.
157 * Initially, we will use the `reserve' buffer.
158 */
159 fp->_ur = fp->_r;
160 fp->_up = fp->_p;
161 _UB(fp)._base = fp->_ubuf;
162 _UB(fp)._size = sizeof(fp->_ubuf);
163 fp->_ubuf[sizeof(fp->_ubuf) - 1] = c;
164 fp->_p = &fp->_ubuf[sizeof(fp->_ubuf) - 1];
165 fp->_r = 1;
166 FUNLOCKFILE(fp);
167 return c;
168}
