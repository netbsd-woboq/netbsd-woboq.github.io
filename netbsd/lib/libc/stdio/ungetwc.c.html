36#include <assert.h>
37#include <errno.h>
38#include <stdio.h>
39#include <wchar.h>
40#include "reentrant.h"
41#include "local.h"
42
43wint_t
44ungetwc(wint_t wc, FILE *fp)
45{
46 struct wchar_io_data *wcio;
47
48 _DIAGASSERT(fp);
49
50 if (wc == WEOF)
51 return WEOF;
52
53 FLOCKFILE(fp);
54 _SET_ORIENTATION(fp, 1);
55 /*
56 * XXX since we have no way to transform a wchar string to
57 * a char string in reverse order, we can't use ungetc.
58 */
59 /* XXX should we flush ungetc buffer? */
60
61 wcio = WCIO_GET(fp);
62 if (wcio == 0) {
63 FUNLOCKFILE(fp);
64 errno = ENOMEM; /* XXX */
65 return WEOF;
66 }
67
68 if (wcio->wcio_ungetwc_inbuf >= WCIO_UNGETWC_BUFSIZE) {
69 FUNLOCKFILE(fp);
70 return WEOF;
71 }
72
73 wcio->wcio_ungetwc_buf[wcio->wcio_ungetwc_inbuf++] = wc;
74 __sclearerr(fp);
75 FUNLOCKFILE(fp);
76
77 return wc;
78}
