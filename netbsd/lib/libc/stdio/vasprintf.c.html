52int
53vasprintf_l(char **str, locale_t loc, const char *fmt, va_list ap)
54{
55 int ret;
56 FILE f;
57 struct __sfileext fext;
58 unsigned char *_base;
59
60 _DIAGASSERT(str != NULL);
61 _DIAGASSERT(fmt != NULL);
62
63 _FILEEXT_SETUP(&f, &fext);
64 f._file = -1;
65 f._flags = __SWR | __SSTR | __SALC;
66 f._bf._base = f._p = malloc(128);
67 if (f._bf._base == NULL)
68 goto err;
69 f._bf._size = f._w = 127; /* Leave room for the NUL */
70 ret = __vfprintf_unlocked_l(&f, loc, fmt, ap);
71 if (ret == -1)
72 goto err;
73 *f._p = '\0';
74 _base = realloc(f._bf._base, (size_t)(ret + 1));
75 if (_base == NULL)
76 goto err;
77 *str = (char *)_base;
78 return ret;
79
80err:
81 if (f._bf._base)
82 free(f._bf._base);
83 *str = NULL;
84 errno = ENOMEM;
85 return -1;
86}
87
88int
89vasprintf(char **str, const char *fmt, va_list ap)
90{
91 return vasprintf_l(str, _current_locale(), fmt, ap);
92}
93
94int
95asprintf_l(char **str, locale_t loc, char const *fmt, ...)
96{
97 va_list ap;
98 int ret;
99
100 va_start(ap, fmt);
101 ret = vasprintf_l(str, loc, fmt, ap);
102 va_end(ap);
103 return ret;
104}
105
106int
107asprintf(char **str, char const *fmt, ...)
108{
109 va_list ap;
110 int ret;
111
112 va_start(ap, fmt);
113 ret = vasprintf(str, fmt, ap);
114 va_end(ap);
115 return ret;
116}
