61int
62vdprintf_l(int fd, locale_t loc, const char * __restrict fmt, va_list ap)
63{
64 FILE f;
65 struct __sfileext fext;
66 unsigned char buf[BUFSIZ];
67 int ret, fdflags, tmp;
68
69 _DIAGASSERT(fd != -1);
70
71 /*
72 * File descriptors are a full int, but _file is only a short.
73 * If we get a valid file descriptor that is greater or equal to
74 * USHRT_MAX, then the fd will get sign-extended into an
75 * invalid file descriptor. Handle this case by failing the
76 * open. (We treat the short as unsigned, and special-case -1).
77 */
78 if (fd >= USHRT_MAX) {
79 errno = EMFILE;
80 return EOF;
81 }
82
83 if ((fdflags = fcntl(fd, F_GETFL, 0)) == -1)
84 return EOF;
85
86 tmp = fdflags & O_ACCMODE;
87 if (tmp != O_RDWR && tmp != O_WRONLY) {
88 errno = EINVAL;
89 return EOF;
90 }
91
92 if (fdflags & O_NONBLOCK) {
93 struct stat st;
94 if (fstat(fd, &st) == -1)
95 return -1;
96 if (!S_ISREG(st.st_mode)) {
97 errno = EFTYPE;
98 return EOF;
99 }
100 }
101
102 _FILEEXT_SETUP(&f, &fext);
103 __sfpinit(&f);
104 f._p = buf;
105 f._w = sizeof(buf);
106 f._flags = __SWR;
107 f._file = fd;
108 f._bf._base = buf;
109 f._bf._size = sizeof(buf);
110 f._cookie = &f;
111 f._read = NULL;
112 f._write = __swrite;
113 f._seek = NULL;
114 f._close = NULL;
115
116 if ((ret = vfprintf_l(&f, loc, fmt, ap)) < 0)
117 return ret;
118
119 return fflush(&f) ? EOF : ret;
120}
121
122int
123vdprintf(int fd, const char * __restrict fmt, va_list ap)
124{
125 return vdprintf_l(fd, _current_locale(), fmt, ap);
126}
