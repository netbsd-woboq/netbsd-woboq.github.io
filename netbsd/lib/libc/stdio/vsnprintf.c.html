72int
73vsnprintf_l(char *str, size_t n, locale_t loc, const char *fmt, va_list ap)
74{
75 int ret;
76 FILE f;
77 struct __sfileext fext;
78 unsigned char dummy[1];
79
80 _DIAGASSERT(n == 0 || str != NULL);
81 _DIAGASSERT(fmt != NULL);
82
83 if (n > INT_MAX) {
84 errno = EOVERFLOW;
85 return -1;
86 }
87
88 _FILEEXT_SETUP(&f, &fext);
89 f._file = -1;
90 f._flags = __SWR | __SSTR;
91 if (n == 0) {
92 f._bf._base = f._p = dummy;
93 f._bf._size = f._w = 0;
94 } else {
95 f._bf._base = f._p = (unsigned char *)str;
96 _DIAGASSERT(__type_fit(int, n - 1));
97 f._bf._size = f._w = (int)(n - 1);
98 }
99 ret = __vfprintf_unlocked_l(&f, loc, fmt, ap);
100 *f._p = 0;
101 return ret;
102}
103
104int
105vsnprintf(char *str, size_t n, const char *fmt, va_list ap)
106{
107 return vsnprintf_l(str, n, _current_locale(), fmt, ap);
108}
109
110int
111snprintf(char *str, size_t n, const char *fmt, ...)
112{
113 va_list ap;
114 int ret;
115
116 va_start(ap, fmt);
117 ret = vsnprintf(str, n, fmt, ap);
118 va_end(ap);
119 return ret;
120}
121
122int
123snprintf_l(char *str, size_t n, locale_t loc, const char *fmt, ...)
124{
125 va_list ap;
126 int ret;
127
128 va_start(ap, fmt);
129 ret = vsnprintf_l(str, n, loc, fmt, ap);
130 va_end(ap);
131 return ret;
132}
