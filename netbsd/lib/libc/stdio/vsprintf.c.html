60__weak_alias(sprintf_l, _sprintf_l)
61__weak_alias(vsprintf_l, _vsprintf_l)
62
63int
64vsprintf_l(char *str, locale_t loc, const char *fmt, va_list ap)
65{
66 int ret;
67 FILE f;
68 struct __sfileext fext;
69
70 _DIAGASSERT(str != NULL);
71 _DIAGASSERT(fmt != NULL);
72
73 _FILEEXT_SETUP(&f, &fext);
74 f._file = -1;
75 f._flags = __SWR | __SSTR;
76 f._bf._base = f._p = (unsigned char *)str;
77 f._bf._size = f._w = INT_MAX;
78 ret = __vfprintf_unlocked_l(&f, loc, fmt, ap);
79 *f._p = 0;
80 return ret;
81}
82
83int
84vsprintf(char *str, const char *fmt, va_list ap)
85{
86 return vsprintf_l(str, _current_locale(), fmt, ap);
87}
88
89int
90sprintf_l(char *str, locale_t loc, char const *fmt, ...)
91{
92 int ret;
93 va_list ap;
94
95 va_start(ap, fmt);
96 ret = vsprintf_l(str, loc, fmt, ap);
97 va_end(ap);
98
99 return ret;
100}
101
102int
103sprintf(char *str, char const *fmt, ...)
104{
105 int ret;
106 va_list ap;
107
108 va_start(ap, fmt);
109 ret = vsprintf(str, fmt, ap);
110 va_end(ap);
111
112 return ret;
113}
