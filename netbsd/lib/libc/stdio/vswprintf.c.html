51__weak_alias(vswprintf_l, _vswprintf_l)
52
53int
54vswprintf_l(wchar_t * __restrict s, size_t n, locale_t loc,
55 const wchar_t * __restrict fmt, va_list ap)
56{
57 static const mbstate_t initial;
58 mbstate_t mbs;
59 FILE f;
60 char *mbp;
61 int ret, sverrno;
62 size_t nwc;
63 struct __sfileext fext;
64
65 if (n == 0) {
66 errno = EINVAL;
67 return -1;
68 }
69
70 _FILEEXT_SETUP(&f, &fext);
71 f._file = -1;
72 f._flags = __SWR | __SSTR | __SALC;
73 f._bf._base = f._p = malloc(128);
74 if (f._bf._base == NULL) {
75 errno = ENOMEM;
76 return -1;
77 }
78 f._bf._size = f._w = 127; /* Leave room for the NUL */
79 ret = __vfwprintf_unlocked_l(&f, loc, fmt, ap);
80 if (ret < 0) {
81 sverrno = errno;
82 free(f._bf._base);
83 errno = sverrno;
84 return -1;
85 }
86 *f._p = '\0';
87 mbp = (char *)f._bf._base;
88 /*
89 * XXX Undo the conversion from wide characters to multibyte that
90 * fputwc() did in __vfwprintf().
91 */
92 mbs = initial;
93 nwc = mbsrtowcs_l(s, (void *)&mbp, n, &mbs, loc);
94 free(f._bf._base);
95 if (nwc == (size_t)-1) {
96 errno = EILSEQ;
97 return -1;
98 }
99 if (nwc == n) {
100 s[n - 1] = L'\0';
101 errno = EOVERFLOW;
102 return -1;
103 }
104
105 return ret;
106}
107
108int
109vswprintf(wchar_t * __restrict s, size_t n, const wchar_t * __restrict fmt,
110 va_list ap)
111{
112 return vswprintf_l(s, n, _current_locale(), fmt, ap);
113}
