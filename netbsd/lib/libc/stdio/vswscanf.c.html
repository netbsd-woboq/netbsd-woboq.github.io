73int
74vswscanf_l(const wchar_t * __restrict str, locale_t loc,
75 const wchar_t * __restrict fmt, va_list ap)
76{
77 static const mbstate_t initial;
78 mbstate_t mbs;
79 FILE f;
80 char *mbstr;
81 size_t mlen;
82 int r;
83 const wchar_t *rstr = str;
84 struct __sfileext fext;
85
86 /*
87 * XXX Convert the wide character string to multibyte, which
88 * __vfwscanf() will convert back to wide characters.
89 */
90 mbstr = malloc(wcslen(str) * MB_CUR_MAX_L(loc) + 1);
91 if (mbstr == NULL)
92 return EOF;
93 mbs = initial;
94 if ((mlen = wcsrtombs_l(mbstr, &rstr, SIZE_T_MAX, &mbs, loc))
95 == (size_t)-1) {
96 free(mbstr);
97 return EOF;
98 }
99 _FILEEXT_SETUP(&f, &fext);
100 (void)memset(WCIO_GET(&f), 0, sizeof(struct wchar_io_data));
101 f._file = -1;
102 f._flags = __SRD;
103 f._bf._base = f._p = (unsigned char *)mbstr;
104 _DIAGASSERT(__type_fit(int, mlen));
105 f._bf._size = f._r = (int)mlen;
106 f._read = eofread;
107 _UB(&f)._base = NULL;
108 r = __vfwscanf_unlocked_l(&f, loc, fmt, ap);
109 free(mbstr);
110
111 return r;
112}
113
114int
115vswscanf(const wchar_t * __restrict str, const wchar_t * __restrict fmt,
116 va_list ap)
117{
118 return vswscanf_l(str, _current_locale(), fmt, ap);
119}
