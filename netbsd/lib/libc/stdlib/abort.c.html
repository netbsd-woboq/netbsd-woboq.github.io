41#include "namespace.h"
42#include <signal.h>
43#include <stdlib.h>
44#include <unistd.h>
45
46extern void (*__cleanup)(void);
47static int aborting = 0;
48
49void
50abort(void)
51{
52 sigset_t mask;
53
54 sigfillset(&mask);
55 /*
56 * don't block SIGABRT to give any handler a chance; we ignore
57 * any errors -- X311J doesn't allow abort to return anyway.
58 */
59 sigdelset(&mask, SIGABRT);
60 (void)sigprocmask(SIG_SETMASK, &mask, (sigset_t *)NULL);
61
62 /*
63 * POSIX.1 requires that stdio buffers be flushed on abort.
64 * We ensure the cleanup routines are only called once in
65 * case the user calls abort() in a SIGABRT handler.
66 */
67 if (!aborting) {
68 aborting = 1;
69
70 if (__cleanup)
71 (*__cleanup)();
72 }
73
74 (void)raise(SIGABRT);
75
76 /*
77 * if SIGABRT ignored, or caught and the handler returns, do
78 * it again, only harder.
79 */
80 (void)signal(SIGABRT, SIG_DFL);
81 (void)sigprocmask(SIG_SETMASK, &mask, (sigset_t *)NULL);
82 (void)raise(SIGABRT);
83 _exit(1);
84}
