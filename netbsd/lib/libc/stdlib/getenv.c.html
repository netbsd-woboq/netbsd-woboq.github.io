52/*
53 * getenv --
54 * Returns ptr to value associated with name, if any, else NULL.
55 * XXX: we cannot use getenv_r to implement this, because getenv()
56 * cannot use a shared buffer, because if it did, subsequent calls
57 * to getenv would trash previous results.
58 */
59char *
60getenv(const char *name)
61{
62 size_t l_name;
63 char *result;
64
65 _DIAGASSERT(name != NULL);
66
67 l_name = __envvarnamelen(name, false);
68 if (l_name == 0)
69 return NULL;
70
71 result = NULL;
72 if (__readlockenv()) {
73 result = __findenvvar(name, l_name);
74 (void)__unlockenv();
75 }
76
77 return result;
78}
79
80int
81getenv_r(const char *name, char *buf, size_t len)
82{
83 size_t l_name;
84 int rv;
85
86 _DIAGASSERT(name != NULL);
87
88 l_name = __envvarnamelen(name, false);
89 if (l_name == 0) {
90 errno = ENOENT;
91 return -1;
92 }
93
94 rv = -1;
95 if (__readlockenv()) {
96 const char *value;
97
98 value = __findenvvar(name, l_name);
99 if (value != NULL) {
100 if (strlcpy(buf, value, len) < len) {
101 rv = 0;
102 } else {
103 errno = ERANGE;
104 }
105 } else {
106 errno = ENOENT;
107 }
108 (void)__unlockenv();
109 }
110
111 return rv;
112}
