44int
45grantpt(int fildes)
46{
47
48 return ioctl(fildes, TIOCGRANTPT, 0);
49}
50
51int
52/*ARGSUSED*/
53unlockpt(int fildes)
54{
55
56 return 0;
57}
58
59char *
60ptsname(int fildes)
61{
62 static struct ptmget pm;
63
64 if (ioctl(fildes, TIOCPTSNAME, &pm) == -1)
65 return NULL;
66
67 return pm.sn;
68}
69
70int
71ptsname_r(int fildes, char *buf, size_t buflen) {
72 struct ptmget pm;
73
74 if (buf == NULL)
75 return EINVAL;
76 if (ioctl(fildes, TIOCPTSNAME, &pm) == -1)
77 return errno;
78 if (strlcpy(buf, pm.sn, buflen) > buflen)
79 return ERANGE;
80 return 0;
81}
