44static u_long next = 1;
45
46int
47rand(void)
48{
49 /* LINTED integer overflow */
50 return (int)((next = next * 1103515245 + 12345) % ((u_long)RAND_MAX + 1));
51}
52
53void
54srand(u_int seed)
55{
56 next = seed;
57}
