34#include <assert.h>
35#include <search.h>
36
37struct qelem {
38 struct qelem *q_forw;
39 struct qelem *q_back;
40};
41
42void
43remque(void *element)
44{
45 struct qelem *e = (struct qelem *) element;
46
47 _DIAGASSERT(e != 0);
48
49 if (e->q_forw)
50 e->q_forw->q_back = e->q_back;
51 if (e->q_back)
52 e->q_back->q_forw = e->q_forw;
53}
