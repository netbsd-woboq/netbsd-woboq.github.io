55/*
56 * setenv --
57 * Set the value of the environmental variable "name" to be
58 * "value". If rewrite is set, replace any current value.
59 */
60int
61setenv(const char *name, const char *value, int rewrite)
62{
63 size_t l_name, l_value, length;
64 ssize_t offset;
65 char *envvar;
66
67 _DIAGASSERT(name != NULL);
68 _DIAGASSERT(value != NULL);
69
70 l_name = __envvarnamelen(name, false);
71 if (l_name == 0 || value == NULL) {
72 errno = EINVAL;
73 return -1;
74 }
75
76 if (!__writelockenv())
77 return -1;
78
79 /* Find slot in the enviroment. */
80 offset = __getenvslot(name, l_name, true);
81 if (offset == -1)
82 goto bad;
83
84 l_value = strlen(value);
85 length = l_name + l_value + 2;
86
87 /* Handle overwriting a current environt variable. */
88 envvar = environ[offset];
89 if (envvar != NULL) {
90 if (!rewrite)
91 goto good;
92 /*
93 * Check whether the buffer was allocated via setenv(3) and
94 * whether there is enough space. If so simply overwrite the
95 * existing value.
96 */
97 if (__canoverwriteenvvar(envvar, length)) {
98 envvar += l_name + 1;
99 goto copy;
100 }
101 }
102
103 /* Allocate memory for name + `=' + value + NUL. */
104 if ((envvar = __allocenvvar(length)) == NULL)
105 goto bad;
106
107 if (environ[offset] != NULL)
108 __freeenvvar(environ[offset]);
109
110 environ[offset] = envvar;
111
112 (void)memcpy(envvar, name, l_name);
113
114 envvar += l_name;
115 *envvar++ = '=';
116
117copy:
118 (void)memcpy(envvar, value, l_value + 1);
119
120good:
121 (void)__unlockenv();
122 return 0;
123
124bad:
125 (void)__unlockenv();
126 return -1;
127}
