24#ifdef __weak_alias

26#endif
27
28void
29srand48(long seed)
30{
31 __rand48_seed[0] = RAND48_SEED_0;
32 __rand48_seed[1] = (unsigned short) seed;
33 __rand48_seed[2] = (unsigned short) ((unsigned long)seed >> 16);
34 __rand48_mult[0] = RAND48_MULT_0;
35 __rand48_mult[1] = RAND48_MULT_1;
36 __rand48_mult[2] = RAND48_MULT_2;
37 __rand48_add = RAND48_ADD;
38}
