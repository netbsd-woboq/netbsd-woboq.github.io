32__RCSID("$NetBSD: strtonum.c,v 1.6 2018/12/06 06:29:56 kamil Exp $");
33
34#include "namespace.h"
35
36#define _OPENBSD_SOURCE
37#include <stdio.h>
38#include <stdlib.h>
39#include <errno.h>
40#include <inttypes.h>
41
42long long
43strtonum(const char *nptr, long long minval, long long maxval,
44 const char **errstr)
45{
46 int e;
47 long long rv;
48 const char *resp;
49
50 if (errstr == NULL)
51 errstr = &resp;
52
53 if (minval > maxval) {
54 *errstr = "invalid";
55 return 0;
56 }
57
58 rv = (long long)strtoi(nptr, NULL, 10, minval, maxval, &e);
59
60 if (e == 0) {
61 *errstr = NULL;
62 return rv;
63 }
64
65 if (e == ERANGE)
66 *errstr = (rv == maxval ? "too large" : "too small");
67 else
68 *errstr = "invalid";
69
70 return 0;
71}
