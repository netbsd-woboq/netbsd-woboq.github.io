57/* ARGSUSED */
58const char *
59__strsignal(int num, char *buf, size_t buflen)
60{
61#define UPREFIX "Unknown signal: %u"
62#define RPREFIX "Real time signal %u"
63 unsigned int signum;
64
65#ifdef NLS
66 nl_catd catd ;
67 catd = catopen("libc", NL_CAT_LOCALE);
68#endif
69
70 _DIAGASSERT(buf != NULL);
71
72 signum = num; /* convert to unsigned */
73 if (signum < (unsigned int) sys_nsig) {
74#ifdef NLS
75 (void)strlcpy(buf, catgets(catd, 2, (int)signum,
76 sys_siglist[signum]), buflen);
77#else
78 return((char *)sys_siglist[signum]);
79#endif
80 } else if (signum >= SIGRTMIN && signum <= SIGRTMAX) {
81#ifdef NLS
82 (void)snprintf(buf, buflen,
83 catgets(catd, 2, SIGRTMIN, RPREFIX), signum);
84#else
85 (void)snprintf(buf, buflen, RPREFIX, signum);
86#endif
87 } else {
88#ifdef NLS
89 (void)snprintf(buf, buflen,
90 catgets(catd, 1, 0xffff, UPREFIX), signum);
91#else
92 (void)snprintf(buf, buflen, UPREFIX, signum);
93#endif
94 }
95
96#ifdef NLS
97 catclose(catd);
98#endif
99
100 return buf;
101}
