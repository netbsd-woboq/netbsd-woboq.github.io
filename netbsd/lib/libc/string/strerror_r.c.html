51__weak_alias(strerror_r, _strerror_r)
52
53int
54_strerror_lr(int num, char *buf, size_t buflen, locale_t loc)
55{
56#define UPREFIX "Unknown error: %u"
57 unsigned int errnum = num;
58 int retval = 0;
59 size_t slen;
60#ifdef NLS
61 int saved_errno = errno;
62 nl_catd catd;
63 catd = catopen_l("libc", NL_CAT_LOCALE, loc);
64#endif
65 _DIAGASSERT(buf != NULL);
66
67 if (errnum < (unsigned int) sys_nerr) {
68#ifdef NLS
69 slen = strlcpy(buf, catgets(catd, 1, (int)errnum,
70 sys_errlist[errnum]), buflen);
71#else
72 slen = strlcpy(buf, sys_errlist[errnum], buflen);
73#endif
74 } else {
75#ifdef NLS
76 slen = snprintf_l(buf, buflen, loc,
77 catgets(catd, 1, 0xffff, UPREFIX), errnum);
78#else
79 slen = snprintf(buf, buflen, UPREFIX, errnum);
80#endif
81 retval = EINVAL;
82 }
83
84 if (slen >= buflen)
85 retval = ERANGE;
86
87#ifdef NLS
88 catclose(catd);
89 errno = saved_errno;
90#endif
91
92 return retval;
93}
94
95int
96strerror_r(int num, char *buf, size_t buflen)
97{
98#ifdef NLS
99 return _strerror_lr(num, buf, buflen, _current_locale());
100#else
101 return _strerror_lr(num, buf, buflen, NULL);
102#endif
103}
