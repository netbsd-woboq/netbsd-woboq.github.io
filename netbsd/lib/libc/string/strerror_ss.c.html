42#ifdef __weak_alias
43__weak_alias(strerror_r_ss,_strerror_r_ss)
44#endif
45
46int
47strerror_r_ss(int num, char *buf, size_t len)
48{
49 if (num >= 0 || num < sys_nerr)
50 strlcpy(buf, sys_errlist[num], len);
51 else
52 snprintf_ss(buf, len, "Unknown error %d", num);
53 return 0;
54}
55
56__aconst char *
57strerror_ss(int num)
58{
59 static char buf[64];
60
61 strerror_r_ss(num, buf, sizeof(buf));
62 return buf;
63}
