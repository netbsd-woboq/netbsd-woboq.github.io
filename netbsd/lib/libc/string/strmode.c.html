52#if !HAVE_STRMODE
53void
54strmode(mode_t mode, char *p)
55{
56
57 _DIAGASSERT(p != NULL);
58
59 /* print type */
60 switch (mode & S_IFMT) {
61 case S_IFDIR: /* directory */
62 *p++ = 'd';
63 break;
64 case S_IFCHR: /* character special */
65 *p++ = 'c';
66 break;
67 case S_IFBLK: /* block special */
68 *p++ = 'b';
69 break;
70 case S_IFREG: /* regular */
71#ifdef S_ARCH2
72 if ((mode & S_ARCH2) != 0) {
73 *p++ = 'A';
74 } else if ((mode & S_ARCH1) != 0) {
75 *p++ = 'a';
76 } else {
77#endif
78 *p++ = '-';
79#ifdef S_ARCH2
80 }
81#endif
82 break;
83 case S_IFLNK: /* symbolic link */
84 *p++ = 'l';
85 break;
86#ifdef S_IFSOCK
87 case S_IFSOCK: /* socket */
88 *p++ = 's';
89 break;
90#endif
91#ifdef S_IFIFO
92 case S_IFIFO: /* fifo */
93 *p++ = 'p';
94 break;
95#endif
96#ifdef S_IFWHT
97 case S_IFWHT: /* whiteout */
98 *p++ = 'w';
99 break;
100#endif
101#ifdef S_IFDOOR
102 case S_IFDOOR: /* door */
103 *p++ = 'D';
104 break;
105#endif
106 default: /* unknown */
107 *p++ = '?';
108 break;
109 }
110 /* usr */
111 if (mode & S_IRUSR)
112 *p++ = 'r';
113 else
114 *p++ = '-';
115 if (mode & S_IWUSR)
116 *p++ = 'w';
117 else
118 *p++ = '-';
119 switch (mode & (S_IXUSR | S_ISUID)) {
120 case 0:
121 *p++ = '-';
122 break;
123 case S_IXUSR:
124 *p++ = 'x';
125 break;
126 case S_ISUID:
127 *p++ = 'S';
128 break;
129 case S_IXUSR | S_ISUID:
130 *p++ = 's';
131 break;
132 }
133 /* group */
134 if (mode & S_IRGRP)
135 *p++ = 'r';
136 else
137 *p++ = '-';
138 if (mode & S_IWGRP)
139 *p++ = 'w';
140 else
141 *p++ = '-';
142 switch (mode & (S_IXGRP | S_ISGID)) {
143 case 0:
144 *p++ = '-';
145 break;
146 case S_IXGRP:
147 *p++ = 'x';
148 break;
149 case S_ISGID:
150 *p++ = 'S';
151 break;
152 case S_IXGRP | S_ISGID:
153 *p++ = 's';
154 break;
155 }
156 /* other */
157 if (mode & S_IROTH)
158 *p++ = 'r';
159 else
160 *p++ = '-';
161 if (mode & S_IWOTH)
162 *p++ = 'w';
163 else
164 *p++ = '-';
165 switch (mode & (S_IXOTH | S_ISVTX)) {
166 case 0:
167 *p++ = '-';
168 break;
169 case S_IXOTH:
170 *p++ = 'x';
171 break;
172 case S_ISVTX:
173 *p++ = 'T';
174 break;
175 case S_IXOTH | S_ISVTX:
176 *p++ = 't';
177 break;
178 }
179 *p++ = ' '; /* will be a '+' if ACL's implemented */
180 *p = '\0';
181}
182#endif /* !HAVE_STRMODE */
