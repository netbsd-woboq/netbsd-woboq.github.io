32#include <sys/cdefs.h>
33__RCSID("$NetBSD: wcscspn.c,v 1.4 2011/11/24 18:44:25 joerg Exp $");
34
35#include <assert.h>
36#include <inttypes.h>
37#include <string.h>
38#include <wchar.h>
39
40#include "wcscspn_bloom.h"
41
42size_t
43wcscspn(const wchar_t *s, const wchar_t *set)
44{
45 size_t bloom[BLOOM_ARRAY_SIZE];
46 const wchar_t *p;
47 const wchar_t *q;
48
49 _DIAGASSERT(s != NULL);
50 _DIAGASSERT(set != NULL);
51
52 if (set[0] == '\0')
53 return wcslen(s);
54 if (set[1] == '\0') {
55 for (p = s; *p; ++p)
56 if (*p == set[0])
57 break;
58 return p - s;
59 }
60
61 wcsspn_bloom_init(bloom, set);
62
63 for (p = s; *p; ++p) {
64 if (!wcsspn_in_bloom(bloom, *p))
65 continue;
66
67 q = set;
68 do {
69 if (*p == *q)
70 goto done;
71 } while (*++q);
72 }
73
74done:
75 return (p - s);
76}
