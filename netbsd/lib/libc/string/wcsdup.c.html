20#include "namespace.h"
21#include <stdlib.h>
22#include <assert.h>
23#include <wchar.h>
24
25__weak_alias(wcsdup,_wcsdup)
26
27wchar_t *
28wcsdup(const wchar_t *str)
29{
30 wchar_t *copy;
31 size_t len;
32
33 _DIAGASSERT(str != NULL);
34
35 len = wcslen(str) + 1;
36 copy = malloc(len * sizeof (wchar_t));
37
38 if (!copy)
39 return NULL;
40
41 return wmemcpy(copy, str, len);
42}
