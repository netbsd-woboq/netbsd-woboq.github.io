36#include <sys/types.h>
37#include <assert.h>
38#include <wchar.h>

40/*
41 * Appends src to string dst of size siz (unlike wcsncat, siz is the
42 * full size of dst, not space left). At most siz-1 characters
43 * will be copied. Always NUL terminates (unless siz == 0).
44 * Returns wcslen(initial dst) + wcslen(src); if retval >= siz,
45 * truncation occurred.
46 */
47size_t
48wcslcat(wchar_t *dst, const wchar_t *src, size_t siz)
49{
50 wchar_t *d = dst;
51 const wchar_t *s = src;
52 size_t n = siz;
53 size_t dlen;
54
55 _DIAGASSERT(dst != NULL);
56 _DIAGASSERT(src != NULL);
57
58 /* Find the end of dst and adjust bytes left but don't go past end */
59 while (*d != '\0' && n-- != 0)
60 d++;
61 dlen = d - dst;
62 n = siz - dlen;
63
64 if (n == 0)
65 return(dlen + wcslen(s));
66 while (*s != '\0') {
67 if (n != 1) {
68 *d++ = *s;
69 n--;
70 }
71 s++;
72 }
73 *d = '\0';
74
75 return(dlen + (s - src)); /* count does not include NUL */
76}
