49#include <assert.h>
50#include <wchar.h>
51
52wchar_t *
53wcstok(wchar_t * __restrict s, const wchar_t * __restrict delim,
54 wchar_t ** __restrict last)
55{
56 const wchar_t *spanp;
57 wchar_t c, sc;
58 wchar_t *tok;
59
60 /* s may be NULL */
61 _DIAGASSERT(delim != NULL);
62 _DIAGASSERT(last != NULL);
63
64 if (s == NULL && (s = *last) == NULL)
65 return (NULL);
66
67 /*
68 * Skip (span) leading delimiters (s += wcsspn(s, delim), sort of).
69 */
70cont:
71 c = *s++;
72 for (spanp = delim; (sc = *spanp++) != L'\0';) {
73 if (c == sc)
74 goto cont;
75 }
76
77 if (c == L'\0') { /* no non-delimiter characters */
78 *last = NULL;
79 return (NULL);
80 }
81 tok = s - 1;
82
83 /*
84 * Scan token (scan for delimiters: s += wcscspn(s, delim), sort of).
85 * Note that delim must have one NUL; we stop if we see that, too.
86 */
87 for (;;) {
88 c = *s++;
89 spanp = delim;
90 do {
91 if ((sc = *spanp++) == c) {
92 if (c == L'\0')
93 s = NULL;
94 else
95 s[-1] = L'\0';
96 *last = s;
97 return (tok);
98 }
99 } while (sc != L'\0');
100 }
101 /* NOTREACHED */
102}
