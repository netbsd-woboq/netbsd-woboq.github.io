36#include <assert.h>
37#include <wchar.h>
38#include "runetype_local.h"
39
40int
41wmemcmp(const wchar_t *s1, const wchar_t *s2, size_t n)
42{
43 size_t i;
44
45 _DIAGASSERT(s1 != NULL);
46 _DIAGASSERT(s2 != NULL);
47
48 for (i = 0; i < n; i++) {
49 if (*s1 != *s2) {
50 /* wchar might be unsigned */
51 return *(const __nbrune_t *)s1 >
52 *(const __nbrune_t *)s2 ? 1 : -1;
53 }
54 s1++;
55 s2++;
56 }
57 return 0;
58}
