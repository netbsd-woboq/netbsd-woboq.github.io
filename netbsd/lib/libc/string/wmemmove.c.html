36#include <assert.h>
37#include <string.h>
38#include <wchar.h>
39
40wchar_t *
41wmemmove(wchar_t *d, const wchar_t *s, size_t n)
42{
43
44 _DIAGASSERT(d != NULL);
45 _DIAGASSERT(s != NULL);
46
47 return (wchar_t *)memmove(d, s, n * sizeof(wchar_t));
48}
