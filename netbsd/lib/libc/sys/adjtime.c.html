52extern int __clockctl_fd;
53
54int ____adjtime50(const struct timeval *, struct timeval *);
55
56int
57adjtime(const struct timeval *delta, struct timeval *olddelta)
58{
59 struct clockctl_adjtime args;
60 int rv;
61
62 /*
63 * we always attempt the syscall first and switch to
64 * clockctl if that fails with EPERM
65 */
66 if (__clockctl_fd == -1) {
67 rv = ____adjtime50(delta, olddelta);
68
69 /*
70 * try via clockctl if the call fails with EPERM
71 */
72 if (rv != -1 || errno != EPERM)
73 return rv;
74
75 /*
76 * If this fails, it means that we are not root
77 * and we cannot open clockctl. This is a failure.
78 */
79 __clockctl_fd = open(_PATH_CLOCKCTL, O_WRONLY | O_CLOEXEC, 0);
80 if (__clockctl_fd == -1) {
81 /* original error was EPERM - don't leak open errors */
82 errno = EPERM;
83 return -1;
84 }
85 }
86
87 /*
88 * If __clockctl_fd >=0, clockctl has already been open
89 * and used, so we carry on using it.
90 */
91 args.delta = delta;
92 args.olddelta = olddelta;
93 return ioctl(__clockctl_fd, CLOCKCTL_ADJTIME, &args);
94}
