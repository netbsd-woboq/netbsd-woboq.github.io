54extern int __clockctl_fd;
55
56int ____clock_settime50(clockid_t, const struct timespec *);
57
58int
59clock_settime(clockid_t clock_id, const struct timespec *tp)
60{
61 struct clockctl_clock_settime args;
62 int rv;
63
64 /*
65 * always try the syscall first and attempt to switch to
66 * clockctl if that fails.
67 */
68 if (__clockctl_fd == -1) {
69 rv = ____clock_settime50(clock_id, tp);
70
71 /*
72 * return unless we failed with EPERM
73 */
74 if (rv != -1 || errno != EPERM)
75 return rv;
76
77 /*
78 * If this fails, it means that we are not root
79 * and we cannot open clockctl. This is a failure.
80 */
81 __clockctl_fd = open(_PATH_CLOCKCTL, O_WRONLY | O_CLOEXEC, 0);
82 if (__clockctl_fd == -1) {
83 /* original error was EPERM - don't leak open errors */
84 errno = EPERM;
85 return -1;
86 }
87 }
88
89 /*
90 * If __clockctl_fd >=0, clockctl has already been open
91 * and used, so we carry on using it.
92 */
93 args.clock_id = clock_id;
94 args.tp = tp;
95 return ioctl(__clockctl_fd, CLOCKCTL_CLOCK_SETTIME, &args);
96}
