40#ifdef __weak_alias
41__weak_alias(statvfs,_statvfs)
42__weak_alias(fstatvfs,_fstatvfs)
43#endif
44
45int
46statvfs(const char *file, struct statvfs *st)
47{
48 return statvfs1(file, st, ST_WAIT);
49}
50
51int
52fstatvfs(int f, struct statvfs *st)
53{
54 return fstatvfs1(f, st, ST_WAIT);
55}

61int
62__fhstatvfs40(const void *fh, size_t fh_size, struct statvfs *st)
63{
64 return __fhstatvfs140(fh, fh_size, st, ST_WAIT);
65}
