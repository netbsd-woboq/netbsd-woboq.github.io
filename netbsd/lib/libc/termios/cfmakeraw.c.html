51/*
52 * Make a pre-existing termios structure into "raw" mode: character-at-a-time
53 * mode with no characters interpreted, 8-bit data path.
54 */
55void
56cfmakeraw(struct termios *t)
57{
58
59 _DIAGASSERT(t != NULL);
60
61 t->c_iflag &= ~(IMAXBEL|IGNBRK|BRKINT|PARMRK|ISTRIP|INLCR|IGNCR|ICRNL|IXON);
62 t->c_oflag &= ~OPOST;
63 t->c_lflag &= ~(ECHO|ECHONL|ICANON|ISIG|IEXTEN);
64 t->c_cflag &= ~(CSIZE|PARENB);
65 t->c_cflag |= CS8;
66 /* XXX set MIN/TIME */
67}
