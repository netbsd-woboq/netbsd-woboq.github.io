42/* libpthread init */
43
44__weak_alias(__libc_thr_init,__libc_thr_init_stub)
45
46void __section(".text.startup")
47__libc_thr_init_stub(void)
48{
49

53#define DIE() (void)raise(SIGABRT)
54
55__weak_alias(__libc_thr_errno,__libc_thr_errno_stub)
56int *
57__libc_thr_errno_stub(void)
58{
59 DIE();
60
61 return (NULL);
62}
