46/*
47 * uuid_create_nil() - create a nil UUID.
48 * See also:
49 * http://www.opengroup.org/onlinepubs/009629399/uuid_create_nil.htm
50 */
51void
52uuid_create_nil(uuid_t *u, uint32_t *status)
53{
54
55 memset(u, 0, sizeof(*u));
56
57 if (status)
58 *status = uuid_s_ok;
59}
