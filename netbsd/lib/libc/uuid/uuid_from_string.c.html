43/*
44 * uuid_from_string() - convert a string representation of an UUID into
45 * a binary representation.
46 * See also:
47 * http://www.opengroup.org/onlinepubs/009629399/uuid_from_string.htm
48 *
49 * NOTE: The sequence field is in big-endian, while the time fields are in
50 * native byte order.
51 */
52void
53uuid_from_string(const char *s, uuid_t *u, uint32_t *status)
54{
55 int n;
56
57 /* Short-circuit 2 special cases: NULL pointer and empty string. */
58 if (s == NULL || *s == '\0') {
59 uuid_create_nil(u, status);
60 return;
61 }
62
63 /* Assume the worst. */
64 if (status != NULL)
65 *status = uuid_s_invalid_string_uuid;
66
67 /* The UUID string representation has a fixed length. */
68 if (strlen(s) != 36)
69 return;
70
71 /*
72 * We only work with "new" UUIDs. New UUIDs have the form:
73 * 01234567-89ab-cdef-0123-456789abcdef
74 * The so called "old" UUIDs, which we don't support, have the form:
75 * 0123456789ab.cd.ef.01.23.45.67.89.ab
76 */
77 if (s[8] != '-')
78 return;
79
80 n = sscanf(s,
81 "%8x-%4hx-%4hx-%2hhx%2hhx-%2hhx%2hhx%2hhx%2hhx%2hhx%2hhx",
82 &u->time_low, &u->time_mid, &u->time_hi_and_version,
83 &u->clock_seq_hi_and_reserved, &u->clock_seq_low, &u->node[0],
84 &u->node[1], &u->node[2], &u->node[3], &u->node[4], &u->node[5]);
85
86 /* Make sure we have all conversions. */
87 if (n != 11)
88 return;
89
90 /* We have a successful scan. Check semantics... */
91 n = u->clock_seq_hi_and_reserved;
92 if ((n & 0x80) != 0x00 && /* variant 0? */
93 (n & 0xc0) != 0x80 && /* variant 1? */
94 (n & 0xe0) != 0xc0) { /* variant 2? */
95 if (status != NULL)
96 *status = uuid_s_bad_version;
97 } else {
98 if (status != NULL)
99 *status = uuid_s_ok;
100 }
101}
