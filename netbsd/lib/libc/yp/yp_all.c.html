48int
49yp_all(const char *indomain, const char *inmap,
50 struct ypall_callback *incallback)
51{
52 struct ypreq_nokey yprnk;
53 struct dom_binding *ysd;
54 struct sockaddr_in clnt_sin;
55 enum clnt_stat status;
56 CLIENT *clnt;
57 int clnt_sock;
58
59 if (_yp_invalid_domain(indomain))
60 return YPERR_BADARGS;
61 if (inmap == NULL || *inmap == '\0'
62 || strlen(inmap) > YPMAXMAP)
63 return YPERR_BADARGS;
64 if (incallback == NULL)
65 return YPERR_BADARGS;
66
67 if (_yp_dobind(indomain, &ysd) != 0)
68 return YPERR_DOMAIN;
69
70 clnt_sock = RPC_ANYSOCK;
71 clnt_sin = ysd->dom_server_addr;
72 clnt_sin.sin_port = 0;
73 clnt = clnttcp_create(&clnt_sin, YPPROG, YPVERS, &clnt_sock, 0, 0);
74 if (clnt == NULL) {
75 warnx("clnttcp_create failed");
76 return YPERR_PMAP;
77 }
78 yprnk.domain = indomain;
79 yprnk.map = inmap;
80
81 status = clnt_call(clnt, (rpcproc_t)YPPROC_ALL,
82 (xdrproc_t)xdr_ypreq_nokey, &yprnk,
83 (xdrproc_t)xdr_ypall, (char *)(void *)incallback, _yplib_timeout);
84 clnt_destroy(clnt);
85
86 /* not really needed... */
87 __yp_unbind(ysd);
88
89 if (status != RPC_SUCCESS)
90 return YPERR_RPC;
91
92 return 0;
93}
