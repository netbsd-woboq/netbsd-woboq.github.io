41extern struct timeval _yplib_timeout;
42extern int _yplib_nerrs;
43extern int _yplib_bindtries;
44
45#ifdef __weak_alias
46__weak_alias(yp_maplist,_yp_maplist)
47#endif
48
49int
50yp_maplist( const char *indomain, struct ypmaplist **outmaplist)
51{
52 struct dom_binding *ysd;
53 struct ypresp_maplist ypml;
54 int r, nerrs = 0;
55
56 if (_yp_invalid_domain(indomain))
57 return YPERR_BADARGS;
58 if (outmaplist == NULL)
59 return YPERR_BADARGS;
60again:
61 if (_yp_dobind(indomain, &ysd) != 0)
62 return YPERR_DOMAIN;
63
64 memset(&ypml, 0, sizeof ypml);
65
66 r = clnt_call(ysd->dom_client, (rpcproc_t)YPPROC_MAPLIST,
67 (xdrproc_t)xdr_ypdomain_wrap_string, &indomain,
68 (xdrproc_t)xdr_ypresp_maplist, &ypml, _yplib_timeout);
69 if (r != RPC_SUCCESS) {
70 if (_yplib_bindtries <= 0 && ++nerrs == _yplib_nerrs) {
71 clnt_perror(ysd->dom_client, "yp_maplist: clnt_call");
72 nerrs = 0;
73 } else if (_yplib_bindtries > 0 && ++nerrs == _yplib_bindtries)
74 return YPERR_YPSERV;
75 ysd->dom_vers = -1;
76 goto again;
77 }
78 *outmaplist = ypml.list;
79 /* NO: xdr_free(xdr_ypresp_maplist, &ypml); */
80 __yp_unbind(ysd);
81 return ypprot_err(ypml.status);
82}
