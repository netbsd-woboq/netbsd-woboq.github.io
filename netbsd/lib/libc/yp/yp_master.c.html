42extern struct timeval _yplib_timeout;
43extern int _yplib_nerrs;
44extern int _yplib_bindtries;
45
46#ifdef __weak_alias
47__weak_alias(yp_master,_yp_master)
48#endif
49
50int
51yp_master(const char *indomain, const char *inmap, char **outname)
52{
53 struct dom_binding *ysd;
54 struct ypresp_master yprm;
55 struct ypreq_nokey yprnk;
56 int r, nerrs = 0;
57
58 if (outname == NULL)
59 return YPERR_BADARGS;
60 *outname = NULL;
61
62 if (_yp_invalid_domain(indomain))
63 return YPERR_BADARGS;
64 if (inmap == NULL || *inmap == '\0'
65 || strlen(inmap) > YPMAXMAP)
66 return YPERR_BADARGS;
67
68again:
69 if (_yp_dobind(indomain, &ysd) != 0)
70 return YPERR_DOMAIN;
71
72 yprnk.domain = indomain;
73 yprnk.map = inmap;
74
75 (void)memset(&yprm, 0, sizeof yprm);
76
77 r = clnt_call(ysd->dom_client, (rpcproc_t)YPPROC_MASTER,
78 (xdrproc_t)xdr_ypreq_nokey, &yprnk,
79 (xdrproc_t)xdr_ypresp_master, &yprm, _yplib_timeout);
80 if (r != RPC_SUCCESS) {
81 if (_yplib_bindtries <= 0 && ++nerrs == _yplib_nerrs) {
82 clnt_perror(ysd->dom_client, "yp_master: clnt_call");
83 nerrs = 0;
84 } else if (_yplib_bindtries > 0 && ++nerrs == _yplib_bindtries)
85 return YPERR_YPSERV;
86 ysd->dom_vers = -1;
87 goto again;
88 }
89 if (!(r = ypprot_err(yprm.status))) {
90 if ((*outname = strdup(yprm.master)) == NULL)
91 r = YPERR_RESRC;
92 }
93 xdr_free((xdrproc_t)xdr_ypresp_master, (char *)(void *)&yprm);
94 __yp_unbind(ysd);
95 if (r != 0) {
96 if (*outname) {
97 free(*outname);
98 *outname = NULL;
99 }
100 }
101 return r;
102}
