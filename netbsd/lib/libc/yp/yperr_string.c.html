44char *
45yperr_string(int incode)
46{
47 static char err[80];
48
49 switch (incode) {
50 case 0:
51 return __UNCONST("Success");
52 case YPERR_BADARGS:
53 return __UNCONST("Request arguments bad");
54 case YPERR_RPC:
55 return __UNCONST("RPC failure");
56 case YPERR_DOMAIN:
57 return __UNCONST(
58 "Can't bind to server which serves this domain");
59 case YPERR_MAP:
60 return __UNCONST("No such map in server's domain");
61 case YPERR_KEY:
62 return __UNCONST("No such key in map");
63 case YPERR_YPERR:
64 return __UNCONST("YP server error");
65 case YPERR_RESRC:
66 return __UNCONST("Local resource allocation failure");
67 case YPERR_NOMORE:
68 return __UNCONST("No more records in map database");
69 case YPERR_PMAP:
70 return __UNCONST("Can't communicate with portmapper");
71 case YPERR_YPBIND:
72 return __UNCONST("Can't communicate with ypbind");
73 case YPERR_YPSERV:
74 return __UNCONST("Can't communicate with ypserv");
75 case YPERR_NODOM:
76 return __UNCONST("Local domain name not set");
77 case YPERR_BADDB:
78 return __UNCONST("Server data base is bad");
79 case YPERR_VERS:
80 return __UNCONST(
81 "YP server version mismatch - server can't supply service."
82 );
83 case YPERR_ACCESS:
84 return __UNCONST("Access violation");
85 case YPERR_BUSY:
86 return __UNCONST("Database is busy");
87 }
88 (void) snprintf(err, sizeof(err), "YP unknown error %d", incode);
89 return err;
90}
