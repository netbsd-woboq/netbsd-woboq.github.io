37#include <rpcsvc/ypclnt.h>

39#ifdef __weak_alias
40__weak_alias(ypprot_err,_ypprot_err)
41#endif
42
43int
44ypprot_err(unsigned int incode)
45{
46 switch (incode) {
47 case YP_TRUE:
48 return 0;
49 case YP_FALSE:
50 return YPERR_YPBIND;
51 case YP_NOMORE:
52 return YPERR_NOMORE;
53 case YP_NOMAP:
54 return YPERR_MAP;
55 case YP_NODOM:
56 return YPERR_NODOM;
57 case YP_NOKEY:
58 return YPERR_KEY;
59 case YP_BADOP:
60 return YPERR_YPERR;
61 case YP_BADDB:
62 return YPERR_BADDB;
63 case YP_YPERR:
64 return YPERR_YPERR;
65 case YP_BADARGS:
66 return YPERR_BADARGS;
67 case YP_VERS:
68 return YPERR_VERS;
69 }
70 return YPERR_YPERR;
71}
