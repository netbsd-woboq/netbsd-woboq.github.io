44#include <sgtty.h>

46/*
47 * Set tty modes.
48 * This was defined in ioctl_compat.h as:
49 * #define stty(fd, tty) ioctl(fd, TIOCSETP, tty)
50 */
51int
52stty(fd, tty)
53 int fd;
54 struct sgttyb *tty;
55{
56
57 _DIAGASSERT(fd != -1);
58 _DIAGASSERT(tty != 0);
59
60 return (ioctl(fd, TIOCSETP, tty));
61}
