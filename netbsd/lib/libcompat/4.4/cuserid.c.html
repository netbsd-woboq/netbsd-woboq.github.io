39#include <pwd.h>
40#include <stdio.h>
41#include <string.h>
42#include <unistd.h>
43
44char *
45cuserid(char *s)
46{
47 struct passwd *pw, pwres;
48 static char buf[L_cuserid];
49 char pwbuf[1024];
50
51 /* s may be NULL */
52
53 if (getpwuid_r(geteuid(), &pwres, pwbuf, sizeof(pwbuf), &pw) != 0
54 || pw == NULL) {
55 if (s != NULL)
56 *s = '\0';
57 return s;
58 }
59 if (s == NULL)
60 s = buf;
61 (void)strncpy(s, pw->pw_name, L_cuserid);
62 return s;
63}
