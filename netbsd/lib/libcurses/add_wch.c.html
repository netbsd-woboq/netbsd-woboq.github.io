49/*
50 * add_wch --
51 * Add the wide character to the current position in stdscr.
52 *
53 */
54int
55add_wch(const cchar_t *wch)
56{
57 return wadd_wch(stdscr, wch);
58}
59
60
61/*
62 * mvadd_wch --
63 * Add the wide character to stdscr at the given location.
64 */
65int
66mvadd_wch(int y, int x, const cchar_t *wch)
67{
68 return mvwadd_wch(stdscr, y, x, wch);
69}
70
71
72/*
73 * mvwadd_wch --
74 * Add the character to the given window at the given location.
75 */
76int
77mvwadd_wch(WINDOW *win, int y, int x, const cchar_t *wch)
78{
79 if (wmove(win, y, x) == ERR)
80 return ERR;
81
82 return wadd_wch(win, wch);
83}
84
85
86/*
87 * wadd_wch --
88 * Add the wide character to the current position in the
89 * given window.
90 *
91 */
92int
93wadd_wch(WINDOW *win, const cchar_t *wch)
94{
95 int y = win->cury;
96 __LINE *lnp = NULL;
97
98#ifdef DEBUG
99 int i;
100 int x = win->curx;
101
102 for (i = 0; i < win->maxy; i++) {
103 assert(win->alines[i]->sentinel == SENTINEL_VALUE);
104 }
105 __CTRACE(__CTRACE_INPUT, "wadd_wch: win(%p), x: %d, y: %d\n",
106 win, x, y);
107#endif
108 lnp = win->alines[y];
109 return _cursesi_addwchar(win, &lnp, &(win->cury), &(win->curx), wch, 1);
110}
