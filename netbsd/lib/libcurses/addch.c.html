44#ifndef _CURSES_USE_MACROS
45
46/*
47 * addch --
48 * Add the character to the current position in stdscr.
49 *
50 */
51int
52addch(chtype ch)
53{
54
55 return waddch(stdscr, ch);
56}
57
58/*
59 * mvaddch --
60 * Add the character to stdscr at the given location.
61 */
62int
63mvaddch(int y, int x, chtype ch)
64{
65
66 return mvwaddch(stdscr, y, x, ch);
67}
68
69/*
70 * mvwaddch --
71 * Add the character to the given window at the given location.
72 */
73int
74mvwaddch(WINDOW *win, int y, int x, chtype ch)
75{
76
77 if (wmove(win, y, x) == ERR)
78 return ERR;
79
80 return waddch(win, ch);
81}

83#endif

85#ifdef HAVE_WCHAR
86/*
87 * waddch --
88 * Add the character to the current position in the given window.
89 */
90int
91waddch(WINDOW *win, chtype ch)
92{
93 cchar_t cc;
94
95 __cursesi_chtype_to_cchar(ch, &cc);
96#ifdef DEBUG
97 __CTRACE(__CTRACE_INPUT,
98 "addch: %d : 0x%x (adding char as wide char)\n",
99 cc.vals[0], cc.attributes);
100#endif
101
102 return wadd_wch(win, &cc);
103}
104
105#else /* !HAVE_WCHAR */
106
107/*
108 * waddch --
109 * Add the character to the current position in the given window.
110 */
111int
112waddch(WINDOW *win, chtype ch)
113{
114 __LDATA buf;
115
116 buf.ch = (wchar_t)ch & __CHARTEXT;
117 buf.attr = (attr_t)ch & __ATTRIBUTES;
118#ifdef DEBUG
119 __CTRACE(__CTRACE_INPUT, "addch: %d : 0x%x\n", buf.ch, buf.attr);
120#endif
121 return __waddch(win, &buf);
122}
123#endif /* !HAVE_WCHAR */
124
125int
126__waddch(WINDOW *win, __LDATA *dp)
127{
128 char buf[2];
129
130 buf[0] = dp->ch;
131 buf[1] = '\0';
132 return _cursesi_waddbytes(win, buf, 1, dp->attr, 1);
133}
