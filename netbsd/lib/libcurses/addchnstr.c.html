44/*
45 * addchstr --
46 * Add a string to stdscr starting at (_cury, _curx).
47 */
48int
49addchstr(const chtype *chstr)
50{
51 return waddchnstr(stdscr, chstr, -1);
52}
53
54/*
55 * waddchstr --
56 * Add a string to the given window starting at (_cury, _curx).
57 */
58int
59waddchstr(WINDOW *win, const chtype *chstr)
60{
61 return waddchnstr(win, chstr, -1);
62}
63
64/*
65 * addchnstr --
66 * Add a string (at most n characters) to stdscr starting
67 * at (_cury, _curx). If n is negative, add the entire string.
68 */
69int
70addchnstr(const chtype *chstr, int n)
71{
72 return waddchnstr(stdscr, chstr, n);
73}

75/*
76 * mvaddchstr --

79int
80mvaddchstr(int y, int x, const chtype *chstr)

82 return mvwaddchnstr(stdscr, y, x, chstr, -1);

85/*
86 * mvwaddchstr --

90mvwaddchstr(WINDOW *win, int y, int x, const chtype *chstr)

92 return mvwaddchnstr(win, y, x, chstr, -1);

106/*
107 * mvwaddchnstr --
108 * Add a string of at most n characters to the given window
109 * starting at (y, x).

112mvwaddchnstr(WINDOW *win, int y, int x, const chtype *chstr, int n)
113{
114 if (wmove(win, y, x) == ERR)
115 return ERR;

117 return waddchnstr(win, chstr, n);

122/*
123 * waddchnstr --
124 * Add a string (at most n characters) to the given window
125 * starting at (_cury, _curx) until the end of line is reached or
126 * n characters have been added. If n is negative, add as much
127 * of the string that will fit on the current line. SUSv2 says
128 * that the addchnstr family does not wrap and strings are truncated
129 * to the RHS of the window.
130 */
131int
132waddchnstr(WINDOW *win, const chtype *chstr, int n)
133{
134 size_t len;
135 const chtype *chp;
136 attr_t attr;
137 char *ocp, *cp, *start;
138 int i, ret, ox, oy;
139
140#ifdef DEBUG
141 __CTRACE(__CTRACE_INPUT, "waddchnstr: win = %p, chstr = %p, n = %d\n",
142 win, chstr, n);
143#endif
144
145 if (n >= 0)
146 for (chp = chstr, len = 0; n-- && *chp++; ++len);
147 else
148 for (chp = chstr, len = 0; *chp++; ++len);
149
150 /* check if string is too long for current location */
151 if (len > (win->maxx - win->curx))
152 len = win->maxx - win->curx;
153
154 if ((ocp = malloc(len + 1)) == NULL)
155 return ERR;
156 chp = chstr;
157 cp = ocp;
158 start = ocp;
159 i = 0;
160 attr = (*chp) & __ATTRIBUTES;
161 ox = win->curx;
162 oy = win->cury;
163 while (len) {
164 *cp = (*chp) & __CHARTEXT;
165 cp++;
166 chp++;
167 i++;
168 len--;
169 if (((*chp) & __ATTRIBUTES) != attr) {
170 *cp = '\0';
171 if (_cursesi_waddbytes(win, start, i, attr, 0) == ERR) {
172 free(ocp);
173 return ERR;
174 }
175 attr = (*chp) & __ATTRIBUTES;
176 start = cp;
177 i = 0;
178 }
179 }
180 *cp = '\0';
181 ret = _cursesi_waddbytes(win, start, i, attr, 0);
182 free(ocp);
183 wmove(win, oy, ox);
184 return ret;
185}
