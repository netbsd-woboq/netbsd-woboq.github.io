48/*
49 * addstr --
50 * Add a string to stdscr starting at (_cury, _curx).
51 */
52int
53addstr(const char *s)
54{
55 return waddnstr(stdscr, s, -1);
56}
57
58/*
59 * waddstr --
60 * Add a string to the given window starting at (_cury, _curx).
61 */
62int
63waddstr(WINDOW *win, const char *s)
64{
65 return waddnstr(win, s, -1);
66}
67
68/*
69 * addnstr --
70 * Add a string (at most n characters) to stdscr starting
71 * at (_cury, _curx). If n is negative, add the entire string.
72 */
73int
74addnstr(const char *str, int n)
75{
76 return waddnstr(stdscr, str, n);
77}

111 * mvwaddnstr --

126/*
127 * waddnstr --
128 * Add a string (at most n characters) to the given window
129 * starting at (_cury, _curx). If n is negative, add the
130 * entire string.
131 */
132int
133waddnstr(WINDOW *win, const char *s, int n)
134{
135 size_t len;
136 const char *p;
137
138#ifdef DEBUG
139 __CTRACE(__CTRACE_INPUT, "ADDNSTR: win %p, length %d\n",
140 win, n);
141#endif
142 /*
143 * behavior changed from traditional BSD curses, for better XCURSES
144 * conformance.
145 *
146 * BSD curses: if (n > 0) then "at most n", else "len = strlen(s)"
147 * ncurses: if (n >= 0) then "at most n", else "len = strlen(s)"
148 * XCURSES: if (n != -1) then "at most n", else "len = strlen(s)"
149 *
150 */
151 if (n >= 0)
152 for (p = s, len = 0; n-- && *p++; ++len);
153 else
154 len = strlen(s);
155
156 return waddbytes(win, s, (int) len);
157}
