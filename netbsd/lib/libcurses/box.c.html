43/*
44 * box --
45 * Draw a box around the given window with "vert" as the vertical
46 * delimiting char, and "hor", as the horizontal one. Uses wborder().
47 */
48int
49box(WINDOW *win, chtype vert, chtype hor)
50{
51 return (wborder(win, vert, vert, hor, hor, 0, 0, 0, 0));
52}
53
54int
55box_set(WINDOW *win, const cchar_t *verch, const cchar_t *horch)
56{
57#ifndef HAVE_WCHAR
58 return ERR;
59#else
60 return wborder_set(win, verch, verch, horch, horch,
61 NULL, NULL, NULL, NULL);
62#endif /* HAVE_WCHAR */
63}
