57int
58mvwchgat(WINDOW *win , int y, int x, int count, attr_t attr, short color,
59 const void *opts)
60{
61 __LINE *lp;
62 __LDATA *lc;
63
64 if (x < 0 || y < 0)
65 return (ERR);
66 if (x >= win->maxx || y >= win->maxy)
67 return (ERR);
68
69 attr = (attr & ~__COLOR) | COLOR_PAIR(color);
70
71 if (count < 0 || count > win->maxx - x)
72 count = win->maxx - x;
73
74#ifdef DEBUG
75 __CTRACE(__CTRACE_ATTR, "mvwchgat: x: %d y: %d count: %d attr: 0x%x "
76 "color pair %d\n", x, y, count, (attr & ~__COLOR),
77 PAIR_NUMBER(color));
78#endif
79 lp = win->alines[y];
80 lc = &lp->line[x];
81
82 if (x + win->ch_off < *lp->firstchp)
83 *lp->firstchp = x + win->ch_off;
84 if (x + win->ch_off + count > *lp->lastchp)
85 *lp->lastchp = x + win->ch_off + count;
86
87 while (count-- > 0) {
88 lp->flags |= __ISDIRTY;
89#ifdef HAVE_WCHAR
90 lc->attr = (lc->attr & ~WA_ATTRIBUTES) | attr;
91#else
92 lc->attr = attr;
93#endif
94 ++lc;
95 }
96
97 return OK;
98}
