47/*
48 * clrtoeol --
49 * Clear up to the end of line.
50 */
51int
52clrtoeol(void)
53{
54 return wclrtoeol(stdscr);
55}

59/*
60 * wclrtoeol --
61 * Clear up to the end of line.
62 */
63int
64wclrtoeol(WINDOW *win)
65{
66 int minx, x, y;
67 __LDATA *end, *maxx, *sp;
68 attr_t attr;
69
70 y = win->cury;
71 x = win->curx;
72 if (win->alines[y]->flags & __ISPASTEOL) {
73 if (y < win->maxy - 1) {
74 win->alines[y]->flags &= ~__ISPASTEOL;
75 y++;
76 x = 0;
77 win->cury = y;
78 win->curx = x;
79 } else
80 return OK;
81 }
82 end = &win->alines[y]->line[win->maxx];
83 minx = -1;
84 maxx = &win->alines[y]->line[x];
85 if (win != curscr)
86 attr = win->battr & __ATTRIBUTES;
87 else
88 attr = 0;
89 for (sp = maxx; sp < end; sp++)
90#ifndef HAVE_WCHAR
91 if (sp->ch != win->bch || sp->attr != attr) {
92#else
93 if (sp->ch != (wchar_t)btowc((int) win->bch ) ||
94 (sp->attr & WA_ATTRIBUTES) != attr || sp->nsp
95 || (WCOL(*sp) < 0)) {
96#endif /* HAVE_WCHAR */
97 maxx = sp;
98 if (minx == -1)
99 minx = (int) (sp - win->alines[y]->line);
100 sp->attr = attr | (sp->attr & __ALTCHARSET);
101#ifdef HAVE_WCHAR
102 sp->ch = (wchar_t)btowc((int) win->bch);
103 if (_cursesi_copy_nsp(win->bnsp, sp) == ERR)
104 return ERR;
105 SET_WCOL(*sp, 1);
106#else
107 sp->ch = win->bch;
108#endif /* HAVE_WCHAR */
109 }
110#ifdef DEBUG
111 __CTRACE(__CTRACE_ERASE, "CLRTOEOL: y = %d, minx = %d, maxx = %d, "
112 "firstch = %d, lastch = %d\n",
113 y, minx, (int)(maxx - win->alines[y]->line),
114 *win->alines[y]->firstchp, *win->alines[y]->lastchp);
115#endif
116 /* Update firstch and lastch for the line. */
117 __touchline(win, y, x, (int)win->maxx - 1);
118 __sync(win);
119 return OK;
120}
