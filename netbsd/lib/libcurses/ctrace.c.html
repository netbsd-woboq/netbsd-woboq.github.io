52static FILE *tracefp = NULL; /* Curses debugging file descriptor. */
53
54static int tracemask; /* Areas of trace output we want. */
55
56static int init_done = 0;
57
58static void
59__CTRACE_init(void)
60{
61 char *tf, *tm;
62
63 tm = getenv("CURSES_TRACE_MASK");
64 if (tm == NULL)
65 tracemask = __CTRACE_ALL;
66 else
67 tracemask = (int)strtol(tm, NULL, 0);
68 if (tracemask < 0)
69 tracemask = (0 - tracemask) ^ __CTRACE_ALL;
70 if (tracemask == 0)
71 return;
72
73 tf = getenv("CURSES_TRACE_FILE");
74
75 if ((tf != NULL) && !strcmp( tf, "<none>"))
76 tf = NULL;
77
78 if (tf != NULL)
79 tracefp = fopen(tf, "w");
80
81 init_done = 1;
82 __CTRACE(__CTRACE_ALL, "Trace mask: 0x%08x\n", tracemask);
83}
84
85void
86__CTRACE(int area, const char *fmt,...)
87{
88 static int seencr = 1;
89 va_list ap;
90
91 if (!init_done)
92 __CTRACE_init();
93 if (tracefp == NULL || !(tracemask & area))
94 return;
95
96 if (seencr && (tracemask & __CTRACE_TSTAMP)) {
97 struct timeval tv;
98 gettimeofday(&tv, NULL);
99 (void)fprintf(tracefp, "%llu.%06lu: ",
100 (long long)tv.tv_sec, (long)tv.tv_usec);
101 }
102 va_start(ap, fmt);
103 (void)vfprintf(tracefp, fmt, ap);
104 seencr = (strchr(fmt, '\n') != NULL);
105 va_end(ap);
106 (void)fflush(tracefp);
107}
