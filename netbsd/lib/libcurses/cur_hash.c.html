46/*
47 * __hash_more() is "hashpjw" from the Dragon Book, Aho, Sethi & Ullman, p.436.
48 */
49unsigned int
50__hash_more(const void *v_s, size_t len, unsigned int h)
51{
52 unsigned int g;
53 size_t i = 0;
54 const char *s = v_s;
55
56 while (i < len) {
57 h = (h << 4) + s[i];
58 if ((g = h & 0xf0000000) != 0) {
59 h = h ^ (g >> 24);
60 h = h ^ g;
61 }
62 i++;
63 }
64 return h;
65}
