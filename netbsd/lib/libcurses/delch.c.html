47#ifndef _CURSES_USE_MACROS
48/*
49 * delch --
50 * Do a delete-char on the line, leaving (cury, curx) unchanged.
51 */
52int
53delch(void)
54{
55 return wdelch(stdscr);
56}
57
58/*
59 * mvdelch --
60 * Do a delete-char on the line at (y, x) on stdscr.
61 */
62int
63mvdelch(int y, int x)
64{
65 return mvwdelch(stdscr, y, x);
66}
67
68/*
69 * mvwdelch --
70 * Do a delete-char on the line at (y, x) of the given window.
71 */
72int
73mvwdelch(WINDOW *win, int y, int x)
74{
75 if (wmove(win, y, x) == ERR)
76 return ERR;
77
78 return wdelch(win);
79}
80
81#endif

83/*
84 * wdelch --
85 * Do a delete-char on the line, leaving (cury, curx) unchanged.
86 */
87int
88wdelch(WINDOW *win)
89{
90 __LDATA *end, *temp1, *temp2;
91
92#ifndef HAVE_WCHAR
93 end = &win->alines[win->cury]->line[win->maxx - 1];
94 temp1 = &win->alines[win->cury]->line[win->curx];
95 temp2 = temp1 + 1;
96 while (temp1 < end) {
97 (void)memcpy(temp1, temp2, sizeof(__LDATA));
98 temp1++, temp2++;
99 }
100 temp1->ch = win->bch;
101 if (__using_color && win != curscr)
102 temp1->attr = win->battr & __COLOR;
103 else
104 temp1->attr = 0;
105 __touchline(win, (int)win->cury, (int)win->curx, (int)win->maxx - 1);
106 return OK;
107#else
108 int cw, sx;
109 nschar_t *np, *tnp;
110
111 end = &win->alines[win->cury]->line[win->maxx - 1];
112 sx = win->curx;
113 temp1 = &win->alines[win->cury]->line[win->curx];
114 cw = WCOL(*temp1);
115 if (cw < 0) {
116 temp1 += cw;
117 sx += cw;
118 cw = WCOL(*temp1);
119 }
120 np = temp1->nsp;
121 if (np) {
122 while (np) {
123 tnp = np->next;
124 free(np);
125 np = tnp;
126 }
127 temp1->nsp = NULL;
128 }
129 if (sx + cw < win->maxx) {
130 temp2 = temp1 + cw;
131 while (temp1 < end - (cw - 1)) {
132 (void)memcpy(temp1, temp2, sizeof(__LDATA));
133 temp1++, temp2++;
134 }
135 }
136 while (temp1 <= end) {
137 temp1->ch = ( wchar_t )btowc((int) win->bch);
138 temp1->attr = 0;
139 if (_cursesi_copy_nsp(win->bnsp, temp1) == ERR)
140 return ERR;
141 SET_WCOL(*temp1, 1);
142 temp1++;
143 }
144 __touchline(win, (int)win->cury, sx, (int)win->maxx - 1);
145 __sync(win);
146 return OK;
147#endif /* HAVE_WCHAR */
148}
