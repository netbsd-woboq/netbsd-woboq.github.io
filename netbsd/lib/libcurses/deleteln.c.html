42#include "curses_private.h"

44#ifndef _CURSES_USE_MACROS
45/*
46 * deleteln --
47 * Delete a line from stdscr. It leaves (cury, curx) unchanged.
48 */
49int
50deleteln(void)
51{
52 return winsdelln(stdscr, -1);
53}
54
55#endif
56
57/*
58 * wdeleteln --
59 * Delete a line from the screen. It leaves (cury, curx) unchanged.
60 */
61int
62wdeleteln(WINDOW *win)
63{
64 return winsdelln(win, -1);
65}
