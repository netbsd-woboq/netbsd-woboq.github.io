46/*
47 * delwin --
48 * Delete a window and release it back to the system.
49 */
50int
51delwin(WINDOW *win)
52{
53 WINDOW *wp, *np;
54 struct __winlist *wl, *pwl;
55 SCREEN *screen;
56
57#ifdef DEBUG
58 __CTRACE(__CTRACE_WINDOW, "delwin(%p)\n", win);
59#endif
60
61 if (win == NULL)
62 return OK;
63
64 /*
65 * Free any storage used by non-spacing characters in the window.
66 */
67#ifdef HAVE_WCHAR
68 __cursesi_win_free_nsp(win);
69#endif
70
71 if (win->orig == NULL) {
72 /*
73 * If we are the original window, delete the space for all
74 * the subwindows and the window space.
75 */
76 free(win->wspace);
77 wp = win->nextp;
78 while (wp != win) {
79 np = wp->nextp;
80 delwin(wp);
81 wp = np;
82 }
83 /* Remove ourselves from the list of windows on the screen. */
84 pwl = NULL;
85 screen = win->screen;
86 for (wl = screen->winlistp; wl; pwl = wl, wl = wl->nextp) {
87 if (wl->winp != win)
88 continue;
89 if (pwl != NULL)
90 pwl->nextp = wl->nextp;
91 else
92 screen->winlistp = wl->nextp;
93 free(wl);
94 break;
95 }
96 } else {
97 /*
98 * If we are a subwindow, take ourselves out of the list.
99 * NOTE: if we are a subwindow, the minimum list is orig
100 * followed by this subwindow, so there are always at least
101 * two windows in the list.
102 */
103 for (wp = win->nextp; wp->nextp != win; wp = wp->nextp)
104 continue;
105 wp->nextp = win->nextp;
106 }
107 free(win->lspace);
108 free(win->alines);
109 if (win == _cursesi_screen->curscr)
110 _cursesi_screen->curscr = NULL;
111 if (win == _cursesi_screen->stdscr)
112 _cursesi_screen->stdscr = NULL;
113 if (win == _cursesi_screen->__virtscr)
114 _cursesi_screen->__virtscr = NULL;
115 if (win->fp)
116 fclose(win->fp);
117 free(win->buf);
118 free(win);
119 return OK;
120}
