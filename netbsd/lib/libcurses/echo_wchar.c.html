42#include "curses.h"
43#include "curses_private.h"
44
45/*
46 * echo_wchar --
47 * Echo wide character and attributes on stdscr and refresh stdscr.
48 */
49int
50echo_wchar(const cchar_t *wch)
51{
52 return wecho_wchar(stdscr, wch);
53}
54
55/*
56 * echo_wchar --
57 * Echo wide character and attributes on "win" and refresh "win".
58 */
59int
60wecho_wchar(WINDOW *win, const cchar_t *wch)
61{
62 int retval;
63
64 retval = wadd_wch(win, wch);
65 if (retval == OK)
66 retval = wrefresh(win);
67 return retval;
68}
69
70/*
71 * pecho_wchar --
72 * Echo character and attributes on "pad" and refresh "pad" at
73 * its previous position on the screen.
74 */
75int
76pecho_wchar(WINDOW *pad, const cchar_t *wch)
77{
78 int retval;
79
80 retval = wadd_wch(pad, wch);
81 if (retval == OK)
82 retval = prefresh(pad, pad->pbegy, pad->pbegx,
83 pad->sbegy, pad->sbegx, pad->smaxy, pad->smaxx);
84 return retval;
85}
