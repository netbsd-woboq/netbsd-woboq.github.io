37#include "curses.h"
38#include "curses_private.h"

40#ifndef _CURSES_USE_MACROS
41/*
42 * echochar --
43 * Echo character and attributes on stdscr and refresh stdscr.
44 */
45int
46echochar(const chtype ch)
47{
48
49 return wechochar(stdscr, ch);
50}
51#endif /* _CURSES_USE_MACROS */
52
53/*
54 * echochar --
55 * Echo character and attributes on "win" and refresh "win".
56 */
57int
58wechochar(WINDOW *win, const chtype ch)
59{
60 int retval;
61
62 retval = waddch(win, ch);
63 if (retval == OK)
64 retval = wrefresh(win);
65 return retval;
66}
67
68/*
69 * pechochar --
70 * Echo character and attributes on "pad" and refresh "pad" at
71 * its previous position on the screen.
72 */
73int
74pechochar(WINDOW *pad, const chtype ch)
75{
76 int retval;
77
78 retval = waddch(pad, ch);
79 if (retval == OK)
80 retval = prefresh(pad, pad->pbegy, pad->pbegx,
81 pad->sbegy, pad->sbegx, pad->smaxy, pad->smaxx);
82 return retval;
83}
