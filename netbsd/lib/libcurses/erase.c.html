60/*
61 * werase --
62 * Erases everything on the window.
63 */
64int
65werase(WINDOW *win)
66{
67
68 int y;
69 __LDATA *sp, *end, *start;
70 attr_t attr;
71
72#ifdef DEBUG
73 __CTRACE(__CTRACE_ERASE, "werase: (%p)\n", win);
74#endif
75 if (win != curscr)
76 attr = win->battr & __ATTRIBUTES;
77 else
78 attr = 0;
79 for (y = 0; y < win->maxy; y++) {
80 start = win->alines[y]->line;
81 end = &start[win->maxx];
82 for (sp = start; sp < end; sp++)
83#ifndef HAVE_WCHAR
84 if (sp->ch != win->bch || sp->attr != 0) {
85#else
86 if (sp->ch != ( wchar_t )btowc(( int ) win->bch ) ||
87 (sp->attr & WA_ATTRIBUTES) != 0 || sp->nsp) {
88#endif /* HAVE_WCHAR */
89 if (sp->attr & __ALTCHARSET)
90 sp->attr = attr | __ALTCHARSET;
91 else
92 sp->attr = attr;
93#ifdef HAVE_WCHAR
94 sp->ch = ( wchar_t )btowc(( int ) win->bch);
95 if (_cursesi_copy_nsp(win->bnsp, sp) == ERR)
96 return ERR;
97 SET_WCOL( *sp, 1 );
98#else
99 sp->ch = win->bch;
100#endif /* HAVE_WCHAR */
101 }
102 }
103 /*
104 * Mark the whole window as changed in case we have overlapping
105 * windows - this will result in the (intended) clearing of the
106 * screen over the area covered by the window. */
107 __touchwin(win);
108 wmove(win, 0, 0);
109 return OK;
110}
