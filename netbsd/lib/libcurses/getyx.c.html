42/*
43 * getpary --
44 * Get the y postion of the window relative to the parent window
45 * return -1 if not a subwindow.
46 */
47int
48getpary(WINDOW *win)
49{
50
51 if (win == NULL)
52 return -1;
53
54 if (win->orig == NULL)
55 return -1;
56
57 return win->begy - win->orig->begy;
58}
59
60/*
61 * getparx --
