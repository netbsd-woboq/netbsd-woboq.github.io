44/*
45 * idcok --
46 * Turn on and off using insert/delete char sequences for the
47 * given window. Note, currently, the ic/dc capabilities are not
48 * used, this function is only provided for compatibility.
49 */
50int
51idcok(WINDOW *win, bool bf)
52{
53 if (bf)
54 win->flags |= __IDCHAR;
55 else
56 win->flags &= ~__IDCHAR;
57 return OK;
58}
