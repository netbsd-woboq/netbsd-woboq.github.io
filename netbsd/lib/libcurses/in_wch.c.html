42#include "curses.h"
43#include "curses_private.h"

45/*
46 * in_wch --
47 * Return wide character at cursor position from stdscr.
48 */
49int
50in_wch(cchar_t *wcval)
51{
52 return win_wch(stdscr, wcval);
53}
54
55/*
56 * mvin_wch --
57 * Return wide character at position (y, x) from stdscr.
58 */
59int
60mvin_wch(int y, int x, cchar_t *wcval)
61{
62 return mvwin_wch(stdscr, y, x, wcval);
63}
64
65/*
66 * mvwin_wch --
67 * Return wide character at position (y, x) from the given window.
68 */
69int
70mvwin_wch(WINDOW *win, int y, int x, cchar_t *wcval)
71{
72 if (wmove(win, y, x) == ERR)
73 return ERR;
74
75 return win_wch(win, wcval);
76}
77
78/*
79 * win_wch --
80 * Return wide character at cursor position.
81 */
82int
83win_wch(WINDOW *win, cchar_t *wcval)
84{
85 nschar_t *np;
86 __LDATA *lp = &win->alines[win->cury]->line[win->curx];
87 int cw = WCOL(*lp);
88
89 if (cw < 0)
90 lp += cw;
91 wcval->vals[0] = lp->ch;
92 wcval->attributes = lp->attr;
93 wcval->elements = 1;
94 np = lp->nsp;
95 if (np) {
96 do {
97 wcval->vals[wcval->elements++] = np->ch;
98 np = np-> next;
99 } while (np);
100 }
101
102 return OK;
103}
