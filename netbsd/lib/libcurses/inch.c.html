40#ifndef _CURSES_USE_MACROS
41
42/*
43 * inch --
44 * Return character at cursor position from stdscr.
45 */
46chtype
47inch(void)
48{
49
50 return winch(stdscr);
51}
52
53/*
54 * mvinch --
55 * Return character at position (y, x) from stdscr.
56 */
57chtype
58mvinch(int y, int x)
59{
60
61 return mvwinch(stdscr, y, x);
62}
63
64/*
65 * mvwinch --
66 * Return character at position (y, x) from the given window.
67 */
68chtype
69mvwinch(WINDOW *win, int y, int x)
70{
71
72 if (wmove(win, y, x) == ERR)
73 return ERR;
74
75 return winch(win);
76}
77
78#endif
79
80/*
81 * winch --
82 * Return character at cursor position.
83 */
84chtype
85winch(WINDOW *win)
86{
87 chtype ch;
88 attr_t attr;
89
90 ch = (chtype) ((win)->alines[(win)->cury]->line[(win)->curx].ch &
91 __CHARTEXT);
92 attr = (attr_t) ((win)->alines[(win)->cury]->line[(win)->curx].attr &
93 __ATTRIBUTES);
94 if (__using_color && ((attr & __COLOR) == __default_color))
95 attr &= ~__COLOR;
96 return (ch | attr);
97}
