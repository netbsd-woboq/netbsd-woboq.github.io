49/*
50 * insch --
51 * Do an insert-char on the line, leaving (cury, curx) unchanged.
52 */
53int
54insch(chtype ch)
55{
56
57 return winsch(stdscr, ch);
58}
59
60/*
61 * mvinsch --
62 * Do an insert-char on the line at (y, x).
63 */
64int
65mvinsch(int y, int x, chtype ch)
66{
67
68 return mvwinsch(stdscr, y, x, ch);
69}
70
71/*
72 * mvwinsch --
73 * Do an insert-char on the line at (y, x) in the given window.
74 */
75int
76mvwinsch(WINDOW *win, int y, int x, chtype ch)
77{
78
79 if (wmove(win, y, x) == ERR)
80 return ERR;
81
82 return winsch(stdscr, ch);
83}

87/*
88 * winsch --
89 * Do an insert-char on the line, leaving (cury, curx) unchanged.
90 */
91int
92winsch(WINDOW *win, chtype ch)
93{
94 __LDATA *end, *temp1, *temp2;
95 attr_t attr;
96
97 if (__using_color)
98 attr = win->battr & __COLOR;
99 else
100 attr = 0;
101 end = &win->alines[win->cury]->line[win->curx];
102 temp1 = &win->alines[win->cury]->line[win->maxx - 1];
103 temp2 = temp1 - 1;
104 while (temp1 > end) {
105 (void)memcpy(temp1, temp2, sizeof(__LDATA));
106 temp1--, temp2--;
107 }
108 temp1->ch = (wchar_t)ch & __CHARTEXT;
109 if (temp1->ch == ' ')
110 temp1->ch = win->bch;
111 temp1->attr = (attr_t) ch & __ATTRIBUTES;
112 if (temp1->attr & __COLOR)
113 temp1->attr |= (win->battr & ~__COLOR);
114 else
115 temp1->attr |= win->battr;
116#ifdef HAVE_WCHAR
117 if (_cursesi_copy_nsp(win->bnsp, temp1) == ERR)
118 return ERR;
119 SET_WCOL(*temp1, 1);
120#endif /* HAVE_WCHAR */
121 __touchline(win, (int)win->cury, (int)win->curx, (int)win->maxx - 1);
122 if (win->cury == LINES - 1 &&
123 (win->alines[LINES - 1]->line[COLS - 1].ch != ' ' ||
124 win->alines[LINES - 1]->line[COLS - 1].attr != attr)) {
125 if (win->flags & __SCROLLOK) {
126 wrefresh(win);
127 scroll(win);
128 win->cury--;
129 } else
130 return ERR;
131 }
132 return OK;
133}
