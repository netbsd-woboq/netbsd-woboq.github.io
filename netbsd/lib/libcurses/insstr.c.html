50/*
51 * insstr --
52 * insert a multi-byte character string into the current window
53 */
54int
55insstr(const char *str)
56{
57
58 return winsstr(stdscr, str);
59}
60
61/*
62 * insnstr --
63 * insert a multi-byte character string into the current window
64 * with at most n characters
65 */
66int
67insnstr(const char *str, int n)
68{
69
70 return winsnstr(stdscr, str, n);
71}

73/*
74 * mvinsstr --
75 * Do an insert-string on the line at (y, x).
76 */
77int
78mvinsstr(int y, int x, const char *str)
79{
80
81 return mvwinsstr(stdscr, y, x, str);
82}

84/*
85 * mvinsnstr --
86 * Do an insert-n-string on the line at (y, x).
87 */
88int
89mvinsnstr(int y, int x, const char *str, int n)
90{
91
92 return mvwinsnstr(stdscr, y, x, str, n);
93}

95/*
96 * mvwinsstr --
97 * Do an insert-string on the line at (y, x) in the given window.
98 */
99int
100mvwinsstr(WINDOW *win, int y, int x, const char *str)
101{
102
103 if (wmove(win, y, x) == ERR)
104 return ERR;
105
106 return winsstr(stdscr, str);
107}

110 * mvwinsnstr --

113int
114mvwinsnstr(WINDOW *win, int y, int x, const char *str, int n)
115{
116
117 if (wmove(win, y, x) == ERR)
118 return ERR;
119
120 return winsnstr(stdscr, str, n);
121}

125/*
126 * winsstr --
127 * Do an insert-string on the line, leaving (cury, curx) unchanged.
128 * No wrapping.
129 */
130int
131winsstr(WINDOW *win, const char *str)
132{
133
134 return winsnstr(win, str, -1);
135}
136
137/*
138 * winsnstr --
139 * Do an insert-n-string on the line, leaving (cury, curx) unchanged.
140 * Performs wrapping.
141 */
142int
143winsnstr(WINDOW *win, const char *str, int n)
144{
145 __LDATA *end, *temp1, *temp2;
146 const char *scp;
147 int len, x;
148 __LINE *lnp;
149#ifdef HAVE_WCHAR
150 nschar_t *np, *tnp;
151#endif /* HAVE_WCHAR */
152
153 /* find string length */
154 if (n > 0)
155 for (scp = str, len = 0; n-- && *scp++; ++len);
156 else
157 for (scp = str, len = 0; *scp++; ++len);
158#ifdef DEBUG
159 __CTRACE(__CTRACE_INPUT, "winsnstr: len = %d\n", len);
160#endif /* DEBUG */
161
162 /* move string */
163 end = &win->alines[win->cury]->line[win->curx];
164 if (len < win->maxx - win->curx) {
165#ifdef DEBUG
166 __CTRACE(__CTRACE_INPUT, "winsnstr: shift %d cells\n", len);
167#endif /* DEBUG */
168 temp1 = &win->alines[win->cury]->line[win->maxx - 1];
169 temp2 = temp1 - len;
170 while (temp2 >= end) {
171#ifdef HAVE_WCHAR
172 np = temp1->nsp;
173 if (np){
174 while (np) {
175 tnp = np->next;
176 free(np);
177 np = tnp;
178 }
179 temp1->nsp = NULL;
180 }
181#endif /* HAVE_WCHAR */
182 (void)memcpy(temp1, temp2, sizeof(__LDATA));
183 temp1--, temp2--;
184 }
185 }
186
187 for (scp = str, temp1 = end, x = win->curx;
188 *scp && x < len + win->curx && x < win->maxx;
189 scp++, temp1++, x++)
190 {
191 temp1->ch = (wchar_t)*scp & __CHARTEXT;
192 temp1->attr = win->wattr;
193#ifdef HAVE_WCHAR
194 SET_WCOL(*temp1, 1);
195#endif /* HAVE_WCHAR */
196 }
197#ifdef DEBUG
198 {
199 int i;
200
201 for (i = win->curx; i < win->curx + len; i++) {
202 __CTRACE(__CTRACE_INPUT,
203 "winsnstr: (%d,%d)=('%c',%x)\n", win->cury, i,
204 win->alines[win->cury]->line[i].ch,
205 win->alines[win->cury]->line[i].attr);
206 }
207 }
208#endif /* DEBUG */
209 lnp = win->alines[win->cury];
210 lnp->flags |= __ISDIRTY;
211 if (win->ch_off < *lnp->firstchp)
212 *lnp->firstchp = win->ch_off;
213 if (win->ch_off + win->maxx - 1 > *lnp->lastchp)
214 *lnp->lastchp = win->ch_off + win->maxx - 1;
215 __touchline(win, (int)win->cury, (int)win->curx, (int)win->maxx - 1);
216 __sync(win);
217 return OK;
218}
