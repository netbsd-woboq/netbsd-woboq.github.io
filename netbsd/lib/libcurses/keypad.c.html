37/*
38 * keypad --
39 * Turn on and off interpretation of function/keypad keys in the
40 * given window.
41 */
42int
43keypad(WINDOW *win, bool bf)
44{
45#ifdef DEBUG
46 __CTRACE(__CTRACE_MISC,
47 "keypad: win %p, %s\n", win, bf ? "TRUE" : "FALSE");
48#endif
49 if (bf) {
50 win->flags |= __KEYPAD;
51 if (!(curscr->flags & __KEYPAD)) {
52 tputs(keypad_xmit, 0, __cputchar);
53 curscr->flags |= __KEYPAD;
54 }
55 } else
56 win->flags &= ~__KEYPAD;
57
58 return OK;
59}
60
61/*
62 * is_keypad --
63 * Return true if window is set for keypad.
64 */
65bool
66is_keypad(const WINDOW *win)
67{
68
69 return win->flags & __KEYPAD ? true : false;
70}
