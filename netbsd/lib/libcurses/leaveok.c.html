37#include "curses.h"
38#include "curses_private.h"
39
40/*
41 * leaveok --
42 * Turn on and off leave cursor after refresh for the given window.
43 */
44int
45leaveok(WINDOW *win, bool bf)
46{
47 if (bf)
48 win->flags |= __LEAVEOK;
49 else
50 win->flags &= ~__LEAVEOK;
51 return OK;
52}
53
54/*
55 * is_leaveok --
56 * Return true if window is set for leaveok.
57 */
58bool
59is_leaveok(const WINDOW *win)
60{
61
62 return win->flags & __LEAVEOK ? true : false;
63}
