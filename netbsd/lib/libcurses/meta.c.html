40/*
41 * meta --
42 * Turn on or off the terminal meta mode.
43 */
44int
45meta(/*ARGSUSED*/ WINDOW *win, bool bf)
46{
47
48 if (bf == TRUE) {
49 if (meta_on != NULL) {
50#ifdef DEBUG
51 __CTRACE(__CTRACE_MISC, "meta: TRUE\n");
52#endif
53 tputs(meta_on, 0, __cputchar);
54 _cursesi_screen->meta_state = TRUE;
55 fflush(_cursesi_screen->outfd);
56 }
57 } else {
58 if (meta_off != NULL) {
59#ifdef DEBUG
60 __CTRACE(__CTRACE_MISC, "meta: FALSE\n");
61#endif
62 tputs(meta_off, 0, __cputchar);
63 _cursesi_screen->meta_state = FALSE;
64 fflush(_cursesi_screen->outfd);
65 }
66 }
67
68 return OK;
69}
70
71/*
72 * __restore_meta_state --
73 * Restore old meta state.
74 */
75void
76__restore_meta_state(void)
77{
78
79 meta(NULL, _cursesi_screen->meta_state);
80}
