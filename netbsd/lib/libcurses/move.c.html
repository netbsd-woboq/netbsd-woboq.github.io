44#ifndef _CURSES_USE_MACROS
45/*
46 * move --
47 * Moves the cursor to the given point on stdscr.
48 */
49int
50move(int y, int x)
51{
52
53 return wmove(stdscr, y, x);
54}
55
56#endif
57
58/*
59 * wmove --
60 * Moves the cursor to the given point.
61 */
62int
63wmove(WINDOW *win, int y, int x)
64{
65#ifdef DEBUG
66 __CTRACE(__CTRACE_MISC, "wmove: (%d, %d)\n", y, x);
67#endif
68 if (x < 0 || y < 0)
69 return ERR;
70 if (x >= win->maxx || y >= win->maxy)
71 return ERR;
72
73 win->curx = x;
74 win->cury = y;
75
76 return OK;
77}
78
79void
80wcursyncup(WINDOW *win)
81{
82
83 while (win->orig) {
84 wmove(win->orig, win->cury + win->begy - win->orig->begy,
85 win->curx + win->begx - win->orig->begx);
86 win = win->orig;
87 }
88}
