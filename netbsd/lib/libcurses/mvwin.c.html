44/*
45 * mvderwin --
46 * Move a derived window. This does not change the physical screen
47 * coordinates of the subwin, rather maps the characters in the subwin
48 * sized part of the parent window starting at dy, dx into the subwin.
49 *
50 */
51int
52mvderwin(WINDOW *win, int dy, int dx)
53{
54 WINDOW *parent;
55 int x, i;
56 __LINE *plp;
57
58 if (win == NULL)
59 return ERR;
60
61 parent = win->orig;
62
63 if (parent == NULL)
64 return ERR;
65
66 if (((win->maxx + dx) > parent->maxx) ||
67 ((win->maxy + dy) > parent->maxy))
68 return ERR;
69
70 win->flags |= __ISDERWIN;
71 win->derx = dx;
72 win->dery = dy;
73
74 x = parent->begx + dx;
75
76 /*
77 * Mark the source area for the derwin as changed so it will be
78 * copied to the destination window on refresh.
79 */
80 for (i = 0; i < win->maxy; i++) {
81 plp = parent->alines[i + dy];
82 plp->flags = __ISDIRTY;
83 if (*plp->firstchp > x)
84 *plp->firstchp = x;
85 if (*plp->lastchp < x + win->maxx)
86 *plp->lastchp = x + win->maxx;
87#ifdef DEBUG
88 __CTRACE(__CTRACE_REFRESH, "mvderwin: firstchp = %d, lastchp = %d\n", *plp->firstchp, *plp->lastchp);
89#endif
90 }
91
92 return OK;
93}

95/*
96 * mvwin --
97 * Relocate the starting position of a window.
98 */
99int
100mvwin(WINDOW *win, int by, int bx)
101{
102 WINDOW *orig;
103 int dy, dx;
104
105 if (by < 0 || by + win->maxy > win->screen->LINES ||
106 bx < 0 || bx + win->maxx > win->screen->COLS)
107 return ERR;
108 dy = by - win->begy;
109 dx = bx - win->begx;
110 orig = win->orig;
111 if (orig == NULL) {
112 orig = win;
113 do {
114 win->begy += dy;
115 win->begx += dx;
116 __swflags(win);
117 win = win->nextp;
118 } while (win != orig);
119 } else {
120 if (by < orig->begy || win->maxy + dy > orig->maxy)
121 return ERR;
122 if (bx < orig->begx || win->maxx + dx > orig->maxx)
123 return ERR;
124 win->begy = by;
125 win->begx = bx;
126 __swflags(win);
127 __set_subwin(orig, win);
128 }
129 __touchwin(win);
130 return OK;
131}
