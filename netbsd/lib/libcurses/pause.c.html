44/*
45 * napms --
46 * Sleep for ms milliseconds.
47 */
48int
49napms(int ms)
50{
51 struct timespec ts;
52
53#ifdef DEBUG
54 __CTRACE(__CTRACE_MISC, "napms: %d\n", ms);
55#endif
56 ts.tv_sec = ms / 1000;
57 ts.tv_nsec = (ms % 1000) * 1000000;
58 (void)nanosleep(&ts, NULL);
59 return OK;
60}
61
62/*
63 * delay_output --
64 * Pause output using terminal pad character.
65 */
66int
67delay_output(int ms)
68{
69 char *delstr;
70
71#ifdef DEBUG
72 __CTRACE(__CTRACE_MISC, "delay_output: %d\n", ms);
73#endif
74 if (!_cursesi_screen->padchar)
75 return napms(ms);
76
77 if (asprintf(&delstr, "%d", ms) == -1)
78 return ERR;
79 tputs(delstr, 0, __cputchar);
80 free(delstr);
81 return OK;
82}
