44/*
45 * scroll --
46 * Scroll the window up a line.
47 */
48int
49scroll(WINDOW *win)
50{
51
52 return wscrl(win, 1);
53}

81/*
82 * wscrl --
83 * Scroll a window n lines - up if n is positive, down if n is negative.
84 */
85int
86wscrl(WINDOW *win, int nlines)
87{
88 int oy, ox;
89
90#ifdef DEBUG
91 __CTRACE(__CTRACE_WINDOW, "wscrl: (%p) lines=%d\n", win, nlines);
92#endif
93
94 if (!(win->flags & __SCROLLOK))
95 return ERR;
96 if (!nlines)
97 return OK;
98
99 getyx(win, oy, ox);
100#ifdef DEBUG
101 __CTRACE(__CTRACE_WINDOW, "wscrl: y=%d\n", oy);
102#endif
103 wmove(win, win->scr_t, 1);
104 winsdelln(win, 0 - nlines);
105 wmove(win, oy, ox);
106
107 if (win == curscr) {
108 __cputchar('\n');
109 if (!__NONL)
110 win->curx = 0;
111#ifdef DEBUG
112 __CTRACE(__CTRACE_WINDOW, "scroll: win == curscr\n");
113#endif
114 }
115 return OK;
116}
117
118/*
119 * wsetscrreg --
120 * Set the top and bottom of the scrolling region for win.
121 */
122int
123wsetscrreg(WINDOW *win, int top, int bottom)
124{
125 if (top < 0 || bottom >= win->maxy || bottom - top < 1)
126 return ERR;
127 win->scr_t = top;
128 win->scr_b = bottom;
129 return OK;
130}
