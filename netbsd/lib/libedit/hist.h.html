37/*
38 * el.hist.c: History functions
39 */
40#ifndef _h_el_hist
41#define _h_el_hist
42
43typedef int (*hist_fun_t)(void *, HistEventW *, int, ...);
44
45typedef struct el_history_t {
46 wchar_t *buf; /* The history buffer */
47 size_t sz; /* Size of history buffer */
48 wchar_t *last; /* The last character */
49 int eventno; /* Event we are looking for */
50 void *ref; /* Argument for history fcns */
51 hist_fun_t fun; /* Event access */
52 HistEventW ev; /* Event cookie */
53} el_history_t;
54
55#define HIST_FUN_INTERNAL(el, fn, arg) \
56 ((((*(el)->el_history.fun) ((el)->el_history.ref, &(el)->el_history.ev, \
57 fn, arg)) == -1) ? NULL : (el)->el_history.ev.str)
58#define HIST_FUN(el, fn, arg) \
59 (((el)->el_flags & NARROW_HISTORY) ? hist_convert(el, fn, arg) : \
60 HIST_FUN_INTERNAL(el, fn, arg))
61
62#define HIST_NEXT(el) HIST_FUN(el, H_NEXT, NULL)
63#define HIST_FIRST(el) HIST_FUN(el, H_FIRST, NULL)
64#define HIST_LAST(el) HIST_FUN(el, H_LAST, NULL)
65#define HIST_PREV(el) HIST_FUN(el, H_PREV, NULL)
66#define HIST_SET(el, num) HIST_FUN(el, H_SET, num)
67#define HIST_LOAD(el, fname) HIST_FUN(el, H_LOAD fname)
68#define HIST_SAVE(el, fname) HIST_FUN(el, H_SAVE fname)
69#define HIST_SAVE_FP(el, fp) HIST_FUN(el, H_SAVE_FP, fp)
70#define HIST_NSAVE_FP(el, n, fp) HIST_FUN(el, H_NSAVE_FP, n, fp)
71
72libedit_private int hist_init(EditLine *);
73libedit_private void hist_end(EditLine *);
74libedit_private el_action_t hist_get(EditLine *);
75libedit_private int hist_set(EditLine *, hist_fun_t, void *);
76libedit_private int hist_command(EditLine *, int, const wchar_t **);
77libedit_private int hist_enlargebuf(EditLine *, size_t, size_t);
78libedit_private wchar_t *hist_convert(EditLine *, int, void *);
79
80#endif /* _h_el_hist */
