37/*
38 * el.keymacro.h: Key macro header
39 */
40#ifndef _h_el_keymacro
41#define _h_el_keymacro
42
43typedef union keymacro_value_t {
44 el_action_t cmd; /* If it is a command the # */
45 wchar_t *str; /* If it is a string... */
46} keymacro_value_t;
47
48typedef struct keymacro_node_t keymacro_node_t;
49
50typedef struct el_keymacro_t {
51 wchar_t *buf; /* Key print buffer */
52 keymacro_node_t *map; /* Key map */
53 keymacro_value_t val; /* Local conversion buffer */
54} el_keymacro_t;
55
56#define XK_CMD 0
57#define XK_STR 1
58#define XK_NOD 2
59
60libedit_private int keymacro_init(EditLine *);
61libedit_private void keymacro_end(EditLine *);
62libedit_private keymacro_value_t *keymacro_map_cmd(EditLine *, int);
63libedit_private keymacro_value_t *keymacro_map_str(EditLine *, wchar_t *);
64libedit_private void keymacro_reset(EditLine *);
65libedit_private int keymacro_get(EditLine *, wchar_t *, keymacro_value_t *);
66libedit_private void keymacro_add(EditLine *, const wchar_t *,
67 keymacro_value_t *, int);
68libedit_private void keymacro_clear(EditLine *, el_action_t *, const wchar_t *);
69libedit_private int keymacro_delete(EditLine *, const wchar_t *);
70libedit_private void keymacro_print(EditLine *, const wchar_t *);
71libedit_private void keymacro_kprint(EditLine *, const wchar_t *,
72 keymacro_value_t *, int);
73libedit_private size_t keymacro__decode_str(const wchar_t *, char *, size_t,
74 const char *);

76#endif /* _h_el_keymacro */
