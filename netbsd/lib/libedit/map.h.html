37/*
38 * el.map.h: Editor maps
39 */
40#ifndef _h_el_map
41#define _h_el_map
42
43typedef el_action_t (*el_func_t)(EditLine *, wint_t);
44
45typedef struct el_bindings_t { /* for the "bind" shell command */
46 const wchar_t *name; /* function name for bind command */
47 int func; /* function numeric value */
48 const wchar_t *description; /* description of function */
49} el_bindings_t;
50
51typedef struct el_map_t {
52 el_action_t *alt; /* The current alternate key map */
53 el_action_t *key; /* The current normal key map */
54 el_action_t *current; /* The keymap we are using */
55 const el_action_t *emacs; /* The default emacs key map */
56 const el_action_t *vic; /* The vi command mode key map */
57 const el_action_t *vii; /* The vi insert mode key map */
58 int type; /* Emacs or vi */
59 el_bindings_t *help; /* The help for the editor functions */
60 el_func_t *func; /* List of available functions */
61 size_t nfunc; /* The number of functions/help items */
62} el_map_t;
63
64#define MAP_EMACS 0
65#define MAP_VI 1
66
67#define N_KEYS 256
68
69libedit_private int map_bind(EditLine *, int, const wchar_t **);
70libedit_private int map_init(EditLine *);
71libedit_private void map_end(EditLine *);
72libedit_private void map_init_vi(EditLine *);
73libedit_private void map_init_emacs(EditLine *);
74libedit_private int map_set_editor(EditLine *, wchar_t *);
75libedit_private int map_get_editor(EditLine *, const wchar_t **);
76libedit_private int map_addfunc(EditLine *, const wchar_t *, const wchar_t *,
77 el_func_t);
78
79#endif /* _h_el_map */
