31#include <sys/cdefs.h>
32#include <sys/types.h>
33#include <stdio.h>
34
35#include "unwind.h"
36#include "execinfo.h"
37
38struct tracer_context {
39 void **arr;
40 size_t len;
41 size_t n;
42};
43
44static _Unwind_Reason_Code
45tracer(struct _Unwind_Context *ctx, void *arg)
46{
47 struct tracer_context *t = arg;
48 if (t->n == (size_t)~0) {
49 /* Skip backtrace frame */
50 t->n = 0;
51 return 0;
52 }
53 if (t->n < t->len)
54 t->arr[t->n++] = (void *)_Unwind_GetIP(ctx);
55 else
56 return _URC_FOREIGN_EXCEPTION_CAUGHT;
57 return 0;
58}
59
60size_t
61backtrace(void **arr, size_t len)
62{
63 struct tracer_context ctx;
64
65 ctx.arr = arr;
66 ctx.len = len;
67 ctx.n = (size_t)~0;
68
69 _Unwind_Backtrace(tracer, &ctx);
70 if (ctx.n != (size_t)~0 && ctx.n > 0)
71 ctx.arr[--ctx.n] = NULL; /* Skip frame below __start */
72
73 return ctx.n;
74}
