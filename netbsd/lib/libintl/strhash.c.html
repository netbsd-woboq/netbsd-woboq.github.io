39/*
40 * string hash function by P.J.Weinberger.
41 * this implementation is derived from src/lib/libc/citrus/citrus_db_hash.c.
42 */
43uint32_t
44/*ARGSUSED*/
45__intl_string_hash(const char *str)
46{
47 const uint8_t *p;
48 uint32_t hash = 0, tmp;
49
50 for (p = (const uint8_t *)str; *p; p++) {
51 hash <<= 4;
52 hash += *p;
53 tmp = hash & 0xF0000000;
54 if (tmp != 0) {
55 hash ^= tmp;
56 hash ^= tmp >> 24;
57 }
58 }
59 return hash;
60}
