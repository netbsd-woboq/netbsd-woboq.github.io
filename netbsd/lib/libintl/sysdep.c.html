211static int
212cmp_sysdep_tag(const void *tag, const void *elem)
213{
214 const struct sysdep_pair *pair = elem;
215
216 return strcmp(tag, pair->tag);
217}
218
219const char *
220__intl_sysdep_get_string_by_tag(const char *tag, size_t *rlen)
221{
222 const struct sysdep_pair *found;
223
224 found = bsearch(tag, sysdep_pair_table, numof(sysdep_pair_table),
225 sizeof(sysdep_pair_table[0]), &cmp_sysdep_tag);
226
227 if (found) {
228 if (rlen)
229 *rlen = found->len;
230 return found->string;
231 }
232
233 if (rlen)
234 *rlen = 0;
235 return "";
236}
