51enum {
52 _ISNS_MUTEXATTR_TYPEMASK = 0x0000000f,
53 ISNS_MUTEX_TYPE_NORMAL = 0,
54 ISNS_MUTEX_TYPE_ERRORCHECK = 1,
55 ISNS_MUTEX_TYPE_RECURSIVE = 2,
56 ISNS_MUTEX_TYPE_ERRORABORT = 3
57};
58
59struct isns_config_s {
60 int kq;
61 int pipe_fds[2];
62
63 pthread_t *control_thread_p;
64
65 isns_socket_t sd;
66 int sd_connected;
67 struct addrinfo *ai_p;
68 struct isns_pdu_s *pdu_in_p;
69
70 pthread_mutex_t taskq_mutex;
71 struct isns_task_s *curtask_p;
72 SIMPLEQ_HEAD(isns_taskq_head_s, isns_task_s) taskq_head;
73
74 pthread_mutex_t trans_mutex;
75
76 int is_server;
77
78 struct isns_refresh_s *refresh_p;
79};
80
81#define isns_is_socket_init_done(_cfg_p) \
82 ((_cfg_p) == NULL ? 0 \
83 : (((struct isns_config_s *)(_cfg_p))->ai_p != NULL))
84
