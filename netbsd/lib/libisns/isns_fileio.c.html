38/*
39 * isns_file_writev()
40 */
41ssize_t
42isns_file_writev(int fd, const struct iovec *iov, int iovcnt)
43{
44 ssize_t rval;
45#if HAVE_WEPE
46 int error;
47
48 error = wepe_sys_writev(fd, iov, iovcnt, &rval);
49 if (error)
50 rval = -1;
51#else
52 rval = writev(fd, iov, iovcnt);
53#endif
54
55 return rval;
56}
57
58/*
59 * isns_file_readv()
60 */
61ssize_t
62isns_file_readv(int fd, const struct iovec *iov, int iovcnt)
63{
64 ssize_t rval;
65#if HAVE_WEPE
66 int error;
67
68 error = wepe_sys_readv(fd, iov, iovcnt, &rval);
69 if (error)
70 rval = -1;
71#else
72 rval = readv(fd, iov, iovcnt);
73#endif
74
75 return rval;
76}
