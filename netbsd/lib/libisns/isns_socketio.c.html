46/*
47 * isns_socket_create()
48 */
49int
50isns_socket_create(isns_socket_t *s, int domain, int type)
51{
52#if HAVE_WEPE
53 return (wepe_sys_socket(domain, type, 0, s));
54#else
55 return *s = socket(domain, type, 0);
56#endif
57}
58
59/*
60 * isns_socket_connect()
61 */
62int
63isns_socket_connect(isns_socket_t s, const struct sockaddr *name, socklen_t
64 namelen)
65{
66#if HAVE_WEPE
67 return (wepe_sys_connect(s, name, namelen));
68#else
69 return connect(s, name, namelen);
70#endif
71}
72
73/*
74 * isns_socket_close()
75 */
76int
77isns_socket_close(isns_socket_t s)
78{
79#if HAVE_WEPE
80 return wepe_sys_close(s);
81#else
82 return close(s);
83#endif
84}
85
86/*
87 * isns_socket_writev()
88 */
89ssize_t
90isns_socket_writev(isns_socket_t s, const struct iovec *iov, int iovcnt)
91{
92 return isns_file_writev(s, iov, iovcnt);
93}
94
95/*
96 * isns_socket_readv()
97 */
98ssize_t
99isns_socket_readv(isns_socket_t s, const struct iovec *iov, int iovcnt)
100{
101 return isns_file_readv(s, iov, iovcnt);
102}
