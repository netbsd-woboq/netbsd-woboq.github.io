12/* asin = atan (x / sqrt(1 - x^2)) */
13ENTRY(__ieee754_asin)
14 XMM_ONE_ARG_DOUBLE_PROLOGUE
15 fldl ARG_DOUBLE_ONE /* x */
16 fld %st(0)
17 fmul %st(0) /* x^2 */
18 fld1
19 fsubp /* 1 - x^2 */
20 fsqrt /* sqrt (1 - x^2) */
21 fpatan
22 XMM_DOUBLE_EPILOGUE
23 ret
