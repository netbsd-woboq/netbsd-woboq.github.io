16struct complex {
17 double x;
18 double y;
19};
20
21double cabs(struct complex);
22__warn_references(cabs, "warning: reference to compatibility cabs()")
23
24double
25cabs(struct complex z)
26{
27
28 return hypot(z.x, z.y);
29}
