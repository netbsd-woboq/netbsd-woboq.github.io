42{

48 y = cimagl(z);
49

56#endif

58 w = asinl(x) + 0.0L * I;

104
105 ca = x + y * I;
106 ct = ca * I;
107 /* sqrtl( 1 - z*z) */
108 /* cmull( &ca, &ca, &zz ) */
109 /*x * x - y * y */
110 zz = (x - y) * (x + y) + (2.0L * x * y) * I;
111
112 zz = 1.0L - creall(zz) - cimagl(zz) * I;
113 z2 = csqrtl(zz);
114
115 zz = ct + z2;
116 zz = clogl(zz);
117 /* multiply by 1/i = -i */
118 w = zz * (-1.0L * I);
119 return w;
120}
