32#include "../src/namespace.h"
33#include <complex.h>
34#include <math.h>
35#include <float.h>
36#include "cephes_subrl.h"
37
38#ifdef __weak_alias
39__weak_alias(catanl, _catanl)
40#endif
41
42#define MAXNUM LDBL_MAX
43
44long double complex
45catanl(long double complex z)
46{
47 long double complex w;
48 long double a, t, x, x2, y;
49
50 x = creall(z);
51 y = cimagl(z);
52
53 if ((x == 0.0L) && (y > 1.0L))
54 goto ovrf;
55
56 x2 = x * x;
57 a = 1.0L - x2 - (y * y);
58 if (a == 0.0)
59 goto ovrf;
60
61 t = 0.5L * atan2l(2.0L * x, a);
62 w = _redupil(t);
63
64 t = y - 1.0L;
65 a = x2 + (t * t);
66 if (a == 0.0L)
67 goto ovrf;
68
69 t = y + 1.0L;
70 a = (x2 + (t * t))/a;
71 w = w + (0.25L * logl(a)) * I;
72 return w;
73
74ovrf:
75#if 0
76 mtherr ("catanl", OVERFLOW);
77#endif
78 w = MAXNUM + MAXNUM * I;
79 return w;
80}
