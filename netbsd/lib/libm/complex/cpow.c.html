32#include "../src/namespace.h"
33#include <complex.h>
34#include <math.h>
35
36double complex
37cpow(double complex a, double complex z)
38{
39 double complex w;
40 double x, y, r, theta, absa, arga;
41
42 x = creal(z);
43 y = cimag(z);
44 absa = cabs(a);
45 if (absa == 0.0) {
46 return (0.0 + 0.0 * I);
47 }
48 arga = carg(a);
49 r = pow(absa, x);
50 theta = x * arga;
51 if (y != 0.0) {
52 r = r * exp(-y * arga);
53 theta = theta + y * log(absa);
54 }
55 w = r * cos(theta) + (r * sin(theta)) * I;
56 return w;
57}
