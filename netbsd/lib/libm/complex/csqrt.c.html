32#include <complex.h>
33#include <math.h>
34
35double complex
36csqrt(double complex z)
37{
38 double complex w;
39 double x, y, r, t, scale;
40
41 x = creal (z);
42 y = cimag (z);
43
44 /*
45 * input is a real number and imaginary part isn't -0.0.
46 * negative zero is on the branch cut.
47 */
48 if ((y == 0.0) && !signbit(y)) {
49 if (x == 0.0) {
50 w = 0.0 + y * I;
51 } else {
52 if (x < 0.0) {
53 r = sqrt(-x);
54 w = 0.0 + r * I;
55 } else {
56 r = sqrt(x);
57 w = r;
58 }
59 }
60 return w;
61 }
62 if (x == 0.0) {
63 if (y > 0) {
64 r = sqrt(0.5 * y);
65 w = r + r * I;
66 } else {
67 r = sqrt(-0.5 * y);
68 w = r - r * I;
69 }
70 return w;
71 }
72 /* Rescale to avoid internal overflow or underflow. */
73 if ((fabs(x) > 4.0) || (fabs(y) > 4.0)) {
74 x *= 0.25;
75 y *= 0.25;
76 scale = 2.0;
77 } else {
78#if 1
79 x *= 1.8014398509481984e16; /* 2^54 */
80 y *= 1.8014398509481984e16;
81 scale = 7.450580596923828125e-9; /* 2^-27 */
82#else
83 x *= 4.0;
84 y *= 4.0;
85 scale = 0.5;
86#endif
87 }
88 w = x + y * I;
89 r = cabs(w);
90 if (x > 0) {
91 t = sqrt(0.5 * r + 0.5 * x);
92 r = scale * fabs((0.5 * y) / t );
93 t *= scale;
94 } else {
95 r = sqrt(0.5 * r - 0.5 * x);
96 t = scale * fabs((0.5 * y) / r);
97 r *= scale;
98 }
99 if (y > 0)
100 w = t + r * I;
101 else
102 w = t - r * I;
103 return w;
104}
