32#include "../src/namespace.h"
33#include <complex.h>
34#include <math.h>
35
36double complex
37ctanh(double complex z)
38{
39 double complex w;
40 double x, y, d;
41
42 x = creal(z);
43 y = cimag(z);
44 d = cosh(2.0 * x) + cos(2.0 * y);
45 w = sinh(2.0 * x) / d + (sin(2.0 * y) / d) * I;
46
47 return w;
48}
