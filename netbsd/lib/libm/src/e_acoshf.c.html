21#include "math.h"
22#include "math_private.h"
23
24static const float
25one = 1.0,
26ln2 = 6.9314718246e-01; /* 0x3f317218 */
27
28float
29__ieee754_acoshf(float x)
30{
31 float t;
32 int32_t hx;
33 GET_FLOAT_WORD(hx,x);
34 if(hx<0x3f800000) { /* x < 1 */
35 return (x-x)/(x-x);
36 } else if(hx >=0x4d800000) { /* x > 2**28 */
37 if(hx >=0x7f800000) { /* x is inf of NaN */
38 return x+x;
39 } else
40 return __ieee754_logf(x)+ln2; /* acosh(huge)=log(2x) */
41 } else if (hx==0x3f800000) {
42 return 0.0; /* acosh(1) = 0 */
43 } else if (hx > 0x40000000) { /* 2**28 > x > 2 */
44 t=x*x;
45 return __ieee754_logf((float)2.0*x-one/(x+__ieee754_sqrtf(t-one)));
46 } else { /* 1<x<2 */
47 t = x-one;
48 return log1pf(t+sqrtf((float)2.0*t+t*t));
49 }
50}
