22#include "math_private.h"

24static const float one = 1.0, huge = 1e30;
25
26static const float zero = 0.0;
27
28float
29__ieee754_atanhf(float x)
30{
31 float t;
32 int32_t hx,ix;
33 GET_FLOAT_WORD(hx,x);
34 ix = hx&0x7fffffff;
35 if (ix>0x3f800000) /* |x|>1 */
36 return (x-x)/(x-x);
37 if(ix==0x3f800000)
38 return x/zero;
39 if(ix<0x31800000&&(huge+x)>zero) return x; /* x<2**-28 */
40 SET_FLOAT_WORD(x,ix);
41 if(ix<0x3f000000) { /* x < 0.5 */
42 t = x+x;
43 t = (float)0.5*log1pf(t+t*x/(one-x));
44 } else
45 t = (float)0.5*log1pf((x+x)/(one-x));
46 if(hx>=0) return t; else return -t;
47}
