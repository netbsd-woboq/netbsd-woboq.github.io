18/* __ieee754_rem_pio2(x,y)
19 *
20 * return the remainder of x rem pi/2 in y[0]+y[1]
21 * use __kernel_rem_pio2()
22 */
23
24#include "math.h"
25#include "math_private.h"

53/*
54 * invpio2: 53 bits of 2/pi
55 * pio2_1: first 33 bit of pi/2
56 * pio2_1t: pi/2 - pio2_1
57 * pio2_2: second 33 bit of pi/2
58 * pio2_2t: pi/2 - (pio2_1+pio2_2)
59 * pio2_3: third 33 bit of pi/2
60 * pio2_3t: pi/2 - (pio2_1+pio2_2+pio2_3)
61 */
62
63static const double
64zero = 0.00000000000000000000e+00, /* 0x00000000, 0x00000000 */
65half = 5.00000000000000000000e-01, /* 0x3FE00000, 0x00000000 */
66two24 = 1.67772160000000000000e+07, /* 0x41700000, 0x00000000 */
67invpio2 = 6.36619772367581382433e-01, /* 0x3FE45F30, 0x6DC9C883 */
68pio2_1 = 1.57079632673412561417e+00, /* 0x3FF921FB, 0x54400000 */
69pio2_1t = 6.07710050650619224932e-11, /* 0x3DD0B461, 0x1A626331 */
70pio2_2 = 6.07710050630396597660e-11, /* 0x3DD0B461, 0x1A600000 */
71pio2_2t = 2.02226624879595063154e-21, /* 0x3BA3198A, 0x2E037073 */
72pio2_3 = 2.02226624871116645580e-21, /* 0x3BA3198A, 0x2E000000 */
73pio2_3t = 8.47842766036889956997e-32; /* 0x397B839A, 0x252049C1 */
74
75int32_t
76__ieee754_rem_pio2(double x, double *y)
77{
78 double z,w,t,r,fn;
79 double tx[3];
80 int32_t e0,i,j,nx,n,ix,hx;
81 u_int32_t low;
82
83 z = 0;
84 GET_HIGH_WORD(hx,x); /* high word of x */
85 ix = hx&0x7fffffff;
86 if(ix<=0x3fe921fb) /* |x| ~<= pi/4 , no need for reduction */
87 {y[0] = x; y[1] = 0; return 0;}
88 if(ix<0x4002d97c) { /* |x| < 3pi/4, special case with n=+-1 */
89 if(hx>0) {
90 z = x - pio2_1;
91 if(ix!=0x3ff921fb) { /* 33+53 bit pi is good enough */
92 y[0] = z - pio2_1t;
93 y[1] = (z-y[0])-pio2_1t;
94 } else { /* near pi/2, use 33+33+53 bit pi */
95 z -= pio2_2;
96 y[0] = z - pio2_2t;
97 y[1] = (z-y[0])-pio2_2t;
98 }
99 return 1;
100 } else { /* negative x */
101 z = x + pio2_1;
102 if(ix!=0x3ff921fb) { /* 33+53 bit pi is good enough */
103 y[0] = z + pio2_1t;
104 y[1] = (z-y[0])+pio2_1t;
105 } else { /* near pi/2, use 33+33+53 bit pi */
106 z += pio2_2;
107 y[0] = z + pio2_2t;
108 y[1] = (z-y[0])+pio2_2t;
109 }
110 return -1;
111 }
112 }
113 if(ix<=0x413921fb) { /* |x| ~<= 2^19*(pi/2), medium size */
114 t = fabs(x);
115 n = (int32_t) (t*invpio2+half);
116 fn = (double)n;
117 r = t-fn*pio2_1;
118 w = fn*pio2_1t; /* 1st round good to 85 bit */
119 if(n<32&&ix!=npio2_hw[n-1]) {
120 y[0] = r-w; /* quick check no cancellation */
121 } else {
122 u_int32_t high;
123 j = ix>>20;
124 y[0] = r-w;
125 GET_HIGH_WORD(high,y[0]);
126 i = j-((high>>20)&0x7ff);
127 if(i>16) { /* 2nd iteration needed, good to 118 */
128 t = r;
129 w = fn*pio2_2;
130 r = t-w;
131 w = fn*pio2_2t-((t-r)-w);
132 y[0] = r-w;
133 GET_HIGH_WORD(high,y[0]);
134 i = j-((high>>20)&0x7ff);
135 if(i>49) { /* 3rd iteration need, 151 bits acc */
136 t = r; /* will cover all possible cases */
137 w = fn*pio2_3;
138 r = t-w;
139 w = fn*pio2_3t-((t-r)-w);
140 y[0] = r-w;
141 }
142 }
143 }
144 y[1] = (r-y[0])-w;
145 if(hx<0) {y[0] = -y[0]; y[1] = -y[1]; return -n;}
146 else return n;
147 }
148 /*
149 * all other (large) arguments
150 */
151 if(ix>=0x7ff00000) { /* x is inf or NaN */
152 y[0]=y[1]=x-x; return 0;
153 }
154 /* set z = scalbn(|x|,ilogb(x)-23) */
155 GET_LOW_WORD(low,x);
156 SET_LOW_WORD(z,low);
157 e0 = (ix>>20)-1046; /* e0 = ilogb(z)-23; */
158 SET_HIGH_WORD(z, ix - ((int32_t)(e0<<20)));
159 for(i=0;i<2;i++) {
160 tx[i] = (double)((int32_t)(z));
161 z = (z-tx[i])*two24;
162 }
163 tx[2] = z;
164 nx = 3;
165 while(tx[nx-1]==zero) nx--; /* skip zero term */
166 n = __kernel_rem_pio2(tx,y,e0,nx,2,two_over_pi);
167 if(hx<0) {y[0] = -y[0]; y[1] = -y[1]; return -n;}
168 return n;
169}
