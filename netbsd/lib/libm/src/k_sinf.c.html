24static const float
25half = 5.0000000000e-01,/* 0x3f000000 */
26S1 = -1.6666667163e-01, /* 0xbe2aaaab */
27S2 = 8.3333337680e-03, /* 0x3c088889 */
28S3 = -1.9841270114e-04, /* 0xb9500d01 */
29S4 = 2.7557314297e-06, /* 0x3638ef1b */
30S5 = -2.5050759689e-08, /* 0xb2d72f34 */
31S6 = 1.5896910177e-10; /* 0x2f2ec9d3 */
32
33float
34__kernel_sinf(float x, float y, int iy)
35{
36 float z,r,v;
37 int32_t ix;
38 GET_FLOAT_WORD(ix,x);
39 ix &= 0x7fffffff; /* high word of x */
40 if(ix<0x32000000) /* |x| < 2**-27 */
41 {if((int)x==0) return x;} /* generate inexact */
42 z = x*x;
43 v = z*x;
44 r = S2+z*(S3+z*(S4+z*(S5+z*S6)));
45 if(iy==0) return x+v*(S1+z*r);
46 else return x-((z*(half*y-v*r)-y)-v*S1);
47}
