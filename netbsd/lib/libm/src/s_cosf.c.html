35float
36cosf(float x)
37{
38 float y[2],z=0.0;
39 int32_t n,ix;
40
41 GET_FLOAT_WORD(ix,x);
42
43 /* |x| ~< pi/4 */
44 ix &= 0x7fffffff;
45 if(ix <= 0x3f490fd8) return __kernel_cosf(x,z);
46
47 /* cos(Inf or NaN) is NaN */
48 else if (ix>=0x7f800000) return x-x;
49
50 /* argument reduction needed */
51 else {
52 n = __ieee754_rem_pio2f(x,y);
53 switch(n&3) {
54 case 0: return __kernel_cosf(y[0],y[1]);
55 case 1: return -__kernel_sinf(y[0],y[1],1);
56 case 2: return -__kernel_cosf(y[0],y[1]);
57 default:
58 return __kernel_sinf(y[0],y[1],1);
59 }
60 }
61}
