99float
100erff(float x)
101{
102 int32_t hx,ix,i;
103 float R,S,P,Q,s,y,z,r;
104 GET_FLOAT_WORD(hx,x);
105 ix = hx&0x7fffffff;
106 if(ix>=0x7f800000) { /* erf(nan)=nan */
107 i = ((u_int32_t)hx>>31)<<1;
108 return (float)(1-i)+one/x; /* erf(+-inf)=+-1 */
109 }
110
111 if(ix < 0x3f580000) { /* |x|<0.84375 */
112 if(ix < 0x31800000) { /* |x|<2**-28 */
113 if (ix < 0x04000000)
114 /*avoid underflow */
115 return (float)0.125*((float)8.0*x+efx8*x);
116 return x + efx*x;
117 }
118 z = x*x;
119 r = pp0+z*(pp1+z*(pp2+z*(pp3+z*pp4)));
120 s = one+z*(qq1+z*(qq2+z*(qq3+z*(qq4+z*qq5))));
121 y = r/s;
122 return x + x*y;
123 }
124 if(ix < 0x3fa00000) { /* 0.84375 <= |x| < 1.25 */
125 s = fabsf(x)-one;
126 P = pa0+s*(pa1+s*(pa2+s*(pa3+s*(pa4+s*(pa5+s*pa6)))));
127 Q = one+s*(qa1+s*(qa2+s*(qa3+s*(qa4+s*(qa5+s*qa6)))));
128 if(hx>=0) return erx + P/Q; else return -erx - P/Q;
129 }
130 if (ix >= 0x40c00000) { /* inf>|x|>=6 */
131 if(hx>=0) return one-tiny; else return tiny-one;
132 }
133 x = fabsf(x);
134 s = one/(x*x);
135 if(ix< 0x4036DB6E) { /* |x| < 1/0.35 */
136 R=ra0+s*(ra1+s*(ra2+s*(ra3+s*(ra4+s*(
137 ra5+s*(ra6+s*ra7))))));
138 S=one+s*(sa1+s*(sa2+s*(sa3+s*(sa4+s*(
139 sa5+s*(sa6+s*(sa7+s*sa8)))))));
140 } else { /* |x| >= 1/0.35 */
141 R=rb0+s*(rb1+s*(rb2+s*(rb3+s*(rb4+s*(
142 rb5+s*rb6)))));
143 S=one+s*(sb1+s*(sb2+s*(sb3+s*(sb4+s*(
144 sb5+s*(sb6+s*sb7))))));
145 }
146 GET_FLOAT_WORD(ix,x);
147 SET_FLOAT_WORD(z,ix&0xfffff000);
148 r = __ieee754_expf(-z*z-(float)0.5625)*__ieee754_expf((z-x)*(z+x)+R/S);
149 if(hx>=0) return one-r/x; else return r/x-one;
150}

152float
153erfcf(float x)
154{
155 int32_t hx,ix;
156 float R,S,P,Q,s,y,z,r;
157 GET_FLOAT_WORD(hx,x);
158 ix = hx&0x7fffffff;
159 if(ix>=0x7f800000) { /* erfc(nan)=nan */
160 /* erfc(+-inf)=0,2 */
161 return (float)(((u_int32_t)hx>>31)<<1)+one/x;
162 }
163
164 if(ix < 0x3f580000) { /* |x|<0.84375 */
165 if(ix < 0x23800000) /* |x|<2**-56 */
166 return one-x;
167 z = x*x;
168 r = pp0+z*(pp1+z*(pp2+z*(pp3+z*pp4)));
169 s = one+z*(qq1+z*(qq2+z*(qq3+z*(qq4+z*qq5))));
170 y = r/s;
171 if(hx < 0x3e800000) { /* x<1/4 */
172 return one-(x+x*y);
173 } else {
174 r = x*y;
175 r += (x-half);
176 return half - r ;
177 }
178 }
179 if(ix < 0x3fa00000) { /* 0.84375 <= |x| < 1.25 */
180 s = fabsf(x)-one;
181 P = pa0+s*(pa1+s*(pa2+s*(pa3+s*(pa4+s*(pa5+s*pa6)))));
182 Q = one+s*(qa1+s*(qa2+s*(qa3+s*(qa4+s*(qa5+s*qa6)))));
183 if(hx>=0) {
184 z = one-erx; return z - P/Q;
185 } else {
186 z = erx+P/Q; return one+z;
187 }
188 }
189 if (ix < 0x41e00000) { /* |x|<28 */
190 x = fabsf(x);
191 s = one/(x*x);
192 if(ix< 0x4036DB6D) { /* |x| < 1/.35 ~ 2.857143*/
193 R=ra0+s*(ra1+s*(ra2+s*(ra3+s*(ra4+s*(
194 ra5+s*(ra6+s*ra7))))));
195 S=one+s*(sa1+s*(sa2+s*(sa3+s*(sa4+s*(
196 sa5+s*(sa6+s*(sa7+s*sa8)))))));
197 } else { /* |x| >= 1/.35 ~ 2.857143 */
198 if(hx<0&&ix>=0x40c00000) return two-tiny;/* x < -6 */
199 R=rb0+s*(rb1+s*(rb2+s*(rb3+s*(rb4+s*(
200 rb5+s*rb6)))));
201 S=one+s*(sb1+s*(sb2+s*(sb3+s*(sb4+s*(
202 sb5+s*(sb6+s*sb7))))));
203 }
204 GET_FLOAT_WORD(ix,x);
205 SET_FLOAT_WORD(z,ix&0xfffff000);
206 r = __ieee754_expf(-z*z-(float)0.5625)*
207 __ieee754_expf((z-x)*(z+x)+R/S);
208 if(hx>0) return r/x; else return two-r/x;
209 } else {
210 if(hx>0) return tiny*tiny; else return two-tiny;
211 }
