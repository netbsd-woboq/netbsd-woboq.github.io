321/*
322 * exp2(x): compute the base 2 exponential of x
323 *
324 * Accuracy: Peak error < 0.503 ulp for normalized results.
325 *
326 * Method: (accurate tables)
327 *
328 * Reduce x:
329 * x = 2**k + y, for integer k and |y| <= 1/2.
330 * Thus we have exp2(x) = 2**k * exp2(y).
331 *
332 * Reduce y:
333 * y = i/TBLSIZE + z - eps[i] for integer i near y * TBLSIZE.
334 * Thus we have exp2(y) = exp2(i/TBLSIZE) * exp2(z - eps[i]),
335 * with |z - eps[i]| <= 2**-9 + 2**-39 for the table used.
336 *
337 * We compute exp2(i/TBLSIZE) via table lookup and exp2(z - eps[i]) via
338 * a degree-5 minimax polynomial with maximum error under 1.3 * 2**-61.
339 * The values in exp2t[] and eps[] are chosen such that
340 * exp2t[i] = exp2(i/TBLSIZE + eps[i]), and eps[i] is a small offset such
341 * that exp2t[i] is accurate to 2**-64.
342 *
343 * Note that the range of i is +-TBLSIZE/2, so we actually index the tables
344 * by i0 = i + TBLSIZE/2.
345 *
346 * This method is due to Gal, with many details due to Gal and Bachelis:
347 *
348 * Gal, S. and Bachelis, B. An Accurate Elementary Mathematical Library
349 * for the IEEE Floating Point Standard. TOMS 17(1), 26-46 (1991).
350 */
351double
352exp2(double x)
353{
354 volatile ieee_double_shape_type x_p_redux;
355 double r, t, twopk, z;
356 uint32_t hx, ix, lx;
357 int32_t i0;
358 int k;
359
360 /* Filter out exceptional cases. */
361 GET_HIGH_WORD(hx,x);
362 ix = hx & 0x7fffffff; /* high word of |x| */
363 if(ix >= 0x40900000) { /* |x| >= 1024 */
364 if(ix >= 0x7ff00000) {
365 /* x is infinity or NaN */
366 GET_LOW_WORD(lx,x);
367 if(((ix & 0xfffff) | lx) != 0 || (hx & 0x80000000) == 0)
368 return (x + x); /* x is NaN or +Inf */
369 else
370 return (0.0); /* x is -Inf */
371 }
372 if(x >= 0x1.0p10)
373 return (huge * huge); /* overflow */
374 if(x <= -0x1.0ccp10)
375 return (tiny * tiny); /* underflow */
376 } else if (ix < 0x3c900000) { /* |x| < 0x1p-54 */
377 return (1.0 + x);
378 }
379
380 /* Reduce x, computing z, i0, and k. */
381 x_p_redux.value = x + redux;
382 i0 = x_p_redux.parts.lsw;
383
384 i0 += TBLSIZE / 2;
385 k = i0 >> TBLBITS;
386 i0 &= TBLSIZE - 1;
387 z = x - (x_p_redux.value - redux);
388
389 /* Compute r = exp2(y) = exp2t[i0] * p(z - eps[i]). */
390 t = tbl[i0].exp2;
391 z -= tbl[i0].eps;
392 r = t + t * z * (P1 + z * (P2 + z * (P3 + z * (P4 + z * P5))));
393
394 /*
395 * Scale by 2**k.
396 * For 'normal' values of 'k' we just multply be the correct value,
397 * but at the extremes this doesn't work and the multiply must
398 * be split.
399 */
400 if (k >= -1021) {
401 if (k == 1024)
402 /* Overflow */
403 return (r * 2.0 * 0x1p1023);
404 } else {
405 /* result might be subnormal */
406 r *= 0x1p-1000;
407 k += 1000;
408 }
409 INSERT_WORDS(twopk, 0x3ff00000 + (k << 20), 0);
410 return (r * twopk);
411}
