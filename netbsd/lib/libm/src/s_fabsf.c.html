21/*
22 * fabsf(x) returns the absolute value of x.
23 */
24
25#include "math.h"
26#include "math_private.h"
27
28float
29fabsf(float x)
30{
31 u_int32_t ix;
32 GET_FLOAT_WORD(ix,x);
33 SET_FLOAT_WORD(x,ix&0x7fffffff);
34 return x;
35}
