33#include <math.h>
34
35#define DECL(type, fn) \
36type \
37fn(type x, type y) \
38{ \
39 \
40 if (isnan(x)) \
41 return (x); \
42 if (isnan(y)) \
43 return (y); \
44 return (x > y ? x - y : 0.0); \
45}

47DECL(double, fdim)
48DECL(float, fdimf)
49DECL(long double, fdiml)
