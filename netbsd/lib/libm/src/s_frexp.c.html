18/*
19 * for non-zero x
20 * x = frexp(arg,&exp);
21 * return a double fp quantity x such that 0.5 <= |x| <1.0
22 * and the corresponding binary exponent "exp". That is
23 * arg = x*2^exp.
24 * If arg is inf, 0.0, or NaN, then frexp(arg,&exp) returns arg
25 * with *exp=0.
26 */
27
28#include "math.h"
29#include "math_private.h"
30
31static const double
32two54 = 1.80143985094819840000e+16; /* 0x43500000, 0x00000000 */
33
34double
35frexp(double x, int *eptr)
36{
37 int32_t hx, ix, lx;
38 EXTRACT_WORDS(hx,lx,x);
39 ix = 0x7fffffff&hx;
40 *eptr = 0;
41 if(ix>=0x7ff00000||((ix|lx)==0)) return x; /* 0,inf,nan */
42 if (ix<0x00100000) { /* subnormal */
43 x *= two54;
44 GET_HIGH_WORD(hx,x);
45 ix = hx&0x7fffffff;
46 *eptr = -54;
47 }
48 *eptr += ((uint32_t)ix>>20)-1022;
49 hx = (hx&0x800fffff)|0x3fe00000;
50 SET_HIGH_WORD(x,hx);
51 return x;
52}
