34#include <machine/ieee.h>
35#include <float.h>
36#include <math.h>

38#include "math_private.h"

40#ifdef __HAVE_LONG_DOUBLE
41#if LDBL_MAX_EXP != 0x4000
42#error "Unsupported long double format"
43#endif
44
45long double
46frexpl(long double x, int *ex)
47{
48 union ieee_ext_u u;
49
50 u.extu_ld = x;
51 switch (u.extu_ext.ext_exp) {
52 case 0: /* 0 or subnormal */
53 if ((u.extu_ext.ext_fracl | u.extu_ext.ext_frach) == 0) {
54 *ex = 0;
55 } else {
56 u.extu_ld *= 0x1.0p514;
57 *ex = u.extu_ext.ext_exp - 0x4200;
58 u.extu_ext.ext_exp = 0x3ffe;
59 }
60 break;
61 case 0x7fff: /* infinity or NaN; value of *ex is unspecified */
62 break;
63 default: /* normal */
64 *ex = u.extu_ext.ext_exp - 0x3ffe;
65 u.extu_ext.ext_exp = 0x3ffe;
66 break;
67 }
68 return (u.extu_ld);
69}
