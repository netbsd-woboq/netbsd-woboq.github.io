35#include "namespace.h"

37#include <float.h>
38#include <math.h>
39#define __TEST_FENV
40#include <fenv.h>
41#ifndef __HAVE_FENV
42#define feraiseexcept(a)
43#endif
44#include <machine/ieee.h>
45
46#ifdef __HAVE_LONG_DOUBLE
47
48#if LDBL_MANT_DIG == 64
49#define FROM_UNDERFLOW 0x1p65L
50#elif LDBL_MANT_DIG == 113
51#define FROM_UNDERFLOW 0x1p114L
52#else
53#error Unsupported long double format
54#endif
55
56int
57ilogbl(long double x)
58{
59 union ieee_ext_u u;
60
61 if (x == 0.0L) {
62 feraiseexcept(FE_INVALID);
63 return FP_ILOGB0; /* ilogbl(0) = 0x80000001 */
64 }
65
66 u.extu_ld = x;
67
68 if (u.extu_ext.ext_exp == EXT_EXP_INFNAN) {
69 feraiseexcept(FE_INVALID);
70 return isnan(x) ? FP_ILOGBNAN : INT_MAX;
71 }
72
73 if (u.extu_ext.ext_exp == 0) {
74 /*
75 * Scale denormalized numbers slightly,
76 * so that they are normal.
77 */
78 u.extu_ld *= FROM_UNDERFLOW;
79 return u.extu_ext.ext_exp - EXT_EXP_BIAS - LDBL_MANT_DIG - 1;
80 }
81 return u.extu_ext.ext_exp - EXT_EXP_BIAS;
82
83}
84
85#endif
