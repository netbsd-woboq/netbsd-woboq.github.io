11/*
12 * isinff(x) returns 1 is x is inf, else 0;
13 * no branching!
14 */
15
16#include "math.h"
17#include "math_private.h"
18
19int
20isinff(float x)
21{
22 int32_t ix;
23 GET_FLOAT_WORD(ix,x);
24 ix &= 0x7fffffff;
25 ix ^= 0x7f800000;
26 return (ix == 0);
27}
