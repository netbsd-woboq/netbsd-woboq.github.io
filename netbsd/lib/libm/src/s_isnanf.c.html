21/*
22 * isnanf(x) returns 1 is x is nan, else 0;
23 * no branching!
24 */
25
26#include "math.h"
27#include "math_private.h"
28
29int
30isnanf(float x)
31{
32 int32_t ix;
33 GET_FLOAT_WORD(ix,x);
34 ix &= 0x7fffffff;
35 ix = 0x7f800000 - ix;
36 return (int)(((u_int32_t)(ix))>>31);
37}
