33#ifndef stype

39#define stype double
40#define roundit rint
41#define dtype long
42#define fn lrint
43#endif
44
45/*
46 * C99 says we should not raise a spurious inexact exception when an
47 * invalid exception is raised. Unfortunately, the set of inputs
48 * that overflows depends on the rounding mode when 'dtype' has more
49 * significant bits than 'stype'. Hence, we bend over backwards for the
50 * sake of correctness; an MD implementation could be more efficient.
51 */
52dtype
53fn(stype x)
54{
55 fenv_t env;
56 dtype d;
57
58 feholdexcept(&env);
59 d = (dtype)roundit(x);
60 if (fetestexcept(FE_INVALID))
61 feclearexcept(FE_INEXACT);
62 feupdateenv(&env);
63 return (d);
64}
