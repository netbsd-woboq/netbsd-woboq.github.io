29#include <sys/cdefs.h>

32#include <math.h>

34#ifndef stype
35#ifdef __FreeBSD__
36__FBSDID("$FreeBSD: head/lib/msun/src/s_lround.c 144770 2005-04-08 00:52:16Z das $");
37#else
38__RCSID("$NetBSD: s_lround.c,v 1.1 2017/05/06 18:03:24 christos Exp $");
39#endif
40#define stype double
41#define roundit round
42#define dtype long
43#define DTYPE_MIN LONG_MIN
44#define DTYPE_MAX LONG_MAX
45#define fn lround
46#endif
47
48/*
49 * If stype has more precision than dtype, the endpoints dtype_(min|max) are
50 * of the form xxx.5; they are "out of range" because lround() rounds away
51 * from 0. On the other hand, if stype has less precision than dtype, then
52 * all values that are out of range are integral, so we might as well assume
53 * that everything is in range. At compile time, INRANGE(x) should reduce to
54 * two floating-point comparisons in the former case, or TRUE otherwise.
55 */
56static const stype dtype_min = DTYPE_MIN - 0.5;
57static const stype dtype_max = DTYPE_MAX + 0.5;
58#define INRANGE(x) (dtype_max - DTYPE_MAX != 0.5 || \
59 ((x) > dtype_min && (x) < dtype_max))
60
61dtype
62fn(stype x)
63{
64
65 if (INRANGE(x)) {
66 x = roundit(x);
67 return ((dtype)x);
68 } else {
69 feraiseexcept(FE_INVALID);
70 return (DTYPE_MAX);
71 }
72}
