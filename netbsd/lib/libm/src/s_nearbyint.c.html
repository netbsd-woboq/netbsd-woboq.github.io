42/*
43 * We save and restore the floating-point environment to avoid raising
44 * an inexact exception. We can get away with using fesetenv()
45 * instead of feclearexcept()/feupdateenv() to restore the environment
46 * because the only exception defined for rint() is overflow, and
47 * rounding can't overflow as long as emax >= p.
48 */
49#define DECL(type, fn, rint) \
50type \
51fn(type x) \
52{ \
53 type ret; \
54 fenv_t env; \
55 \
56 fegetenv(&env); \
57 ret = rint(x); \
58 fesetenv(&env); \
59 return (ret); \
60}
