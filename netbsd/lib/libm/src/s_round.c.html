35#include <math.h>

37#ifndef __HAVE_LONG_DOUBLE
38__strong_alias(_roundl, round)
39__weak_alias(roundl, round)
40#endif
41
42double
43round(double x)
44{
45 double t;
46 int i;
47
48 i = fpclassify(x);
49 if (i == FP_INFINITE || i == FP_NAN)
50 return (x);
51
52 if (x >= 0.0) {
53 t = floor(x);
54 if (x - t >= 0.5)
55 t += 1.0;
56 return (t);
57 } else {
58 x = -x;
59 t = floor(x);
60 if (x - t >= 0.5)
61 t += 1.0;
62 return (-t);
63 }
64}
