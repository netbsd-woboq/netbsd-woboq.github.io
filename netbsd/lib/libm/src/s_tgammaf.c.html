36/*
37 * We simply call tgamma() rather than bloating the math library with
38 * a float-optimized version of it. The reason is that tgammaf() is
39 * essentially useless, since the function is superexponential and
40 * floats have very limited range.
41 */
42float
43tgammaf(float x)
44{
45
46 return (tgamma(x));
47}
