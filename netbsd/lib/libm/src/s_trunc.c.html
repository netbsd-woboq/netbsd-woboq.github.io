21/*
22 * trunc(x)
23 * Return x rounded toward 0 to integral value
24 * Method:
25 * Bit twiddling.
26 * Exception:
27 * Inexact flag raised if x not equal to trunc(x).
28 */
29
30#include "math.h"
31#include "math_private.h"
32
33#ifndef __HAVE_LONG_DOUBLE
34__strong_alias(_truncl, trunc)
35__weak_alias(truncl, trunc)
36#endif
37
38static const double huge = 1.0e300;
39
40double
41trunc(double x)
42{
43 int32_t i0,i1,jj0;
44 uint32_t i;
45 EXTRACT_WORDS(i0,i1,x);
46 jj0 = ((i0>>20)&0x7ff)-0x3ff;
47 if(jj0<20) {
48 if(jj0<0) { /* raise inexact if x != 0 */
49 if(huge+x>0.0) {/* |x|<1, so return 0*sign(x) */
50 i0 &= 0x80000000U;
51 i1 = 0;
52 }
53 } else {
54 i = (0x000fffff)>>jj0;
55 if(((i0&i)|i1)==0) return x; /* x is integral */
56 if(huge+x>0.0) { /* raise inexact flag */
57 i0 &= (~i); i1=0;
58 }
59 }
60 } else if (jj0>51) {
61 if(jj0==0x400) return x+x; /* inf or NaN */
62 else return x; /* x is integral */
63 } else {
64 i = ((u_int32_t)(0xffffffff))>>(jj0-20);
65 if((i1&i)==0) return x; /* x is integral */
66 if(huge+x>0.0) /* raise inexact flag */
67 i1 &= (~i);
68 }
69 INSERT_WORDS(x,i0,i1);
70 return x;
71}
