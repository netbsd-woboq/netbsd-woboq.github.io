21/*
22 * truncf(x)
23 * Return x rounded toward 0 to integral value
24 * Method:
25 * Bit twiddling.
26 * Exception:
27 * Inexact flag raised if x not equal to truncf(x).
28 */
29
30#include "math.h"
31#include "math_private.h"
32
33static const float huge = 1.0e30F;
34
35float
36truncf(float x)
37{
38 int32_t i0,jj0;
39 uint32_t i;
40 GET_FLOAT_WORD(i0,x);
41 jj0 = ((i0>>23)&0xff)-0x7f;
42 if(jj0<23) {
43 if(jj0<0) { /* raise inexact if x != 0 */
44 if(huge+x>0.0F) /* |x|<1, so return 0*sign(x) */
45 i0 &= 0x80000000;
46 } else {
47 i = (0x007fffff)>>jj0;
48 if((i0&i)==0) return x; /* x is integral */
49 if(huge+x>0.0F) /* raise inexact flag */
50 i0 &= (~i);
51 }
52 } else {
53 if(jj0==0x80) return x+x; /* inf or NaN */
54 else return x; /* x is integral */
55 }
56 SET_FLOAT_WORD(x,i0);
57 return x;
58}
