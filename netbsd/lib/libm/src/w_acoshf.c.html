23 * wrapper acoshf(x)
24 */
25
26#include "math.h"
27#include "math_private.h"
28
29float
30acoshf(float x) /* wrapper acoshf */
31{
32#ifdef _IEEE_LIBM
33 return __ieee754_acoshf(x);
34#else
35 float z;
36 z = __ieee754_acoshf(x);
37 if(_LIB_VERSION == _IEEE_ || isnanf(x)) return z;
38 if(x<(float)1.0) {
39 /* acosh(x<1) */
40 return (float)__kernel_standard((double)x,(double)x,129);
41 } else
42 return z;
43#endif
44}
