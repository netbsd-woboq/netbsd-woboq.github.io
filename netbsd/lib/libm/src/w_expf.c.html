21/*
22 * wrapper expf(x)
23 */
24
25#include "namespace.h"
26#include "math.h"
27#include "math_private.h"
28
29#ifdef __weak_alias
30__weak_alias(expf, _expf)
31#endif
32
33static const float
34o_threshold= 8.8721679688e+01, /* 0x42b17180 */
35u_threshold= -1.0397208405e+02; /* 0xc2cff1b5 */
36
37float
38expf(float x) /* wrapper expf */
39{
40#ifdef _IEEE_LIBM
41 return __ieee754_expf(x);
42#else
43 float z;
44 z = __ieee754_expf(x);
45 if(_LIB_VERSION == _IEEE_) return z;
46 if(finitef(x)) {
47 if(x>o_threshold)
48 /* exp overflow */
49 return (float)__kernel_standard((double)x,(double)x,106);
50 else if(x<u_threshold)
51 /* exp underflow */
52 return (float)__kernel_standard((double)x,(double)x,107);
53 }
54 return z;
55#endif
56}
