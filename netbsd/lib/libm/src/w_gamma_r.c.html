18/*
19 * wrapper double gamma_r(double x, int *signgamp)
20 */
21
22#include "math.h"
23#include "math_private.h"
24
25double
26gamma_r(double x, int *signgamp) /* wrapper lgamma_r */
27{
28#ifdef _IEEE_LIBM
29 return __ieee754_lgamma_r(x,signgamp);
30#else
31 double y;
32 y = __ieee754_lgamma_r(x,signgamp);
33 if(_LIB_VERSION == _IEEE_) return y;
34 if(!finite(y)&&finite(x)) {
35 if(floor(x)==x&&x<=0.0)
36 return __kernel_standard(x,x,41); /* gamma pole */
37 else
38 return __kernel_standard(x,x,40); /* gamma overflow */
39 } else
40 return y;
41#endif
42}
