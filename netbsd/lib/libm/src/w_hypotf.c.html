25#include "namespace.h"
26#include "math.h"
27#include "math_private.h"
28
29#ifdef __weak_alias
30__weak_alias(hypotf, _hypotf)
31#endif
32
33float
34hypotf(float x, float y) /* wrapper hypotf */
35{
36#ifdef _IEEE_LIBM
37 return __ieee754_hypotf(x,y);
38#else
39 float z;
40 z = __ieee754_hypotf(x,y);
41 if(_LIB_VERSION == _IEEE_) return z;
42 if((!finitef(z))&&finitef(x)&&finitef(y))
43 /* hypot overflow */
44 return (float)__kernel_standard((double)x,(double)y,104);
45 else
46 return z;
47#endif
48}
