21/*
22 * wrapper float lgammaf_r(float x, int *signgamp)
23 */
24
25#include "math.h"
26#include "math_private.h"
27
28float
29lgammaf_r(float x, int *signgamp) /* wrapper lgammaf_r */
30{
31#ifdef _IEEE_LIBM
32 return __ieee754_lgammaf_r(x,signgamp);
33#else
34 float y;
35 y = __ieee754_lgammaf_r(x,signgamp);
36 if(_LIB_VERSION == _IEEE_) return y;
37 if(!finitef(y)&&finitef(x)) {
38 if(floorf(x)==x&&x<=(float)0.0)
39 /* lgamma pole */
40 return (float)__kernel_standard((double)x,(double)x,115);
41 else
42 /* lgamma overflow */
43 return (float)__kernel_standard((double)x,(double)x,114);
44 } else
45 return y;
46#endif
47}
