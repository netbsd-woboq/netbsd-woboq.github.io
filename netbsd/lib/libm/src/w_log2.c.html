18/*
19 * wrapper log2(X)
20 */
21
22#include "math.h"
23#include "math_private.h"
24
25
26double
27log2(double x) /* wrapper log10 */
28{
29#ifdef _IEEE_LIBM
30 return __ieee754_log2(x);
31#else
32 double z;
33 z = __ieee754_log2(x);
34 if(_LIB_VERSION == _IEEE_ || isnan(x)) return z;
35 if(x<=0.0) {
36 if(x==0.0)
37 return __kernel_standard(x,x,48); /* log2(0) */
38 else
39 return __kernel_standard(x,x,49); /* log2(x<0) */
40 } else
41 return z;
42#endif
43}
