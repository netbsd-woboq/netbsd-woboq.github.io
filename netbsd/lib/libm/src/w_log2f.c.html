22 * wrapper log2f(X)
23 */
24
25#include "math.h"
26#include "math_private.h"
27
28
29float
30log2f(float x) /* wrapper log2f */
31{
32#ifdef _IEEE_LIBM
33 return __ieee754_log2f(x);
34#else
35 float z;
36 z = __ieee754_log2f(x);
37 if(_LIB_VERSION == _IEEE_ || isnanf(x)) return z;
38 if(x<=(float)0.0) {
39 if(x==(float)0.0)
40 /* log2(0) */
41 return (float)__kernel_standard((double)x,(double)x,148);
42 else
43 /* log2(x<0) */
44 return (float)__kernel_standard((double)x,(double)x,149);
45 } else
46 return z;
47#endif
48}
