18/*
19 * wrapper sinh(x)
20 */
21
22#include "namespace.h"
23#include "math.h"
24#include "math_private.h"
25
26#ifdef __weak_alias
27__weak_alias(sinh, _sinh)
28#endif
29
30double
31sinh(double x) /* wrapper sinh */
32{
33#ifdef _IEEE_LIBM
34 return __ieee754_sinh(x);
35#else
36 double z;
37 z = __ieee754_sinh(x);
38 if(_LIB_VERSION == _IEEE_) return z;
39 if(!finite(z)&&finite(x)) {
40 return __kernel_standard(x,x,25); /* sinh overflow */
41 } else
42 return z;
43#endif
44}
