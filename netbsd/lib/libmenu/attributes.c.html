32#include <menu.h>
33
34/* defined in menu.c - the default menu struct */
35extern MENU _menui_default_menu;
36
37/*
38 * Set the menu foreground attribute
39 */
40int
41set_menu_fore(MENU *menu, attr_t attr)
42{
43 if (menu == NULL)
44 _menui_default_menu.fore = attr;
45 else
46 menu->fore = attr;
47 return E_OK;
48}
49
50/*
51 * Return the menu foreground attribute
52 */
53char
54menu_fore(MENU *menu)
55{
56 if (menu == NULL)
57 return _menui_default_menu.fore;
58 else
59 return menu->fore;
60}
