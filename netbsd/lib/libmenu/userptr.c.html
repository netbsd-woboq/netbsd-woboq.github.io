29#include <sys/cdefs.h>
30__RCSID("$NetBSD: userptr.c,v 1.9 2003/03/09 01:08:48 lukem Exp $");
31
32#include <menu.h>
33#include <stdlib.h>
34#include <string.h>
35
36/* the following is defined in menu.c */
37extern MENU _menui_default_menu;
38
39/* the following is defined in item.c */
40extern ITEM _menui_default_item;
41
42/*
43 * Set the item user pointer data
44 */
45int
46set_item_userptr(ITEM *param_item, char *userptr)
47{
48 ITEM *item = (param_item != NULL) ? param_item : &_menui_default_item;
49
50 item->userptr = userptr;
51 return E_OK;
52}
53
54
55/*
56 * Return the item user pointer
57 */
58char *
59item_userptr(ITEM *item)
60{
61 if (item == NULL)
62 return _menui_default_item.userptr;
63 else
64 return item->userptr;
65}
66
67/*
68 * Return the user pointer for the given menu
69 */
70char *
71menu_userptr(MENU *menu)
72{
73 if (menu == NULL)
74 return _menui_default_menu.userptr;
75 else
76 return menu->userptr;
77}
78
79/*
80 * Set the user pointer for the given menu
81 */
82int
83set_menu_userptr(MENU *param_menu, char *userptr)
84{
85 MENU *menu = (param_menu != NULL) ? param_menu : &_menui_default_menu;
86
87 menu->userptr = userptr;
88
89 return E_OK;
90}
