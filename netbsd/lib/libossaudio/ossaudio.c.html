72int
73_oss_ioctl(int fd, unsigned long com, ...)
74{
75 va_list ap;
76 void *argp;
77
78 va_start(ap, com);
79 argp = va_arg(ap, void *);
80 va_end(ap);
81
82 if (IOCGROUP(com) == 'P')
83 return audio_ioctl(fd, com, argp);
84 else if (IOCGROUP(com) == 'M')
85 return mixer_ioctl(fd, com, argp);
86 else
87 return ioctl(fd, com, argp);
88}

90static int
91audio_ioctl(int fd, unsigned long com, void *argp)
92{
93
94 struct audio_info tmpinfo;
95 struct audio_offset tmpoffs;
96 struct audio_buf_info bufinfo;
97 struct count_info cntinfo;
98 struct audio_encoding tmpenc;
99 struct oss_sysinfo tmpsysinfo;
100 struct oss_audioinfo *tmpaudioinfo;
101 audio_device_t tmpaudiodev;
102 struct stat tmpstat;
103 dev_t devno;
104 char version[32] = "4.01";
105 char license[16] = "NetBSD";
106 u_int u;
107 int idat, idata;
108 int retval;
109 int i;
110
111 idat = 0;
112
113 switch (com) {
114 case SNDCTL_DSP_RESET:
115 retval = ioctl(fd, AUDIO_FLUSH, 0);
116 if (retval < 0)
117 return retval;
118 break;
119 case SNDCTL_DSP_SYNC:
120 retval = ioctl(fd, AUDIO_DRAIN, 0);
121 if (retval < 0)
122 return retval;
123 break;
124 case SNDCTL_DSP_POST:
125 /* This call is merely advisory, and may be a nop. */
126 break;
127 case SNDCTL_DSP_SPEED:
128 AUDIO_INITINFO(&tmpinfo);
129 tmpinfo.play.sample_rate =
130 tmpinfo.record.sample_rate = INTARG;
131 (void) ioctl(fd, AUDIO_SETINFO, &tmpinfo);
132 /* FALLTHRU */
133 case SOUND_PCM_READ_RATE:
134 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
135 if (retval < 0)
136 return retval;
137 INTARG = tmpinfo.play.sample_rate;
138 break;
139 case SNDCTL_DSP_STEREO:
140 AUDIO_INITINFO(&tmpinfo);
141 tmpinfo.play.channels =
142 tmpinfo.record.channels = INTARG ? 2 : 1;
143 (void) ioctl(fd, AUDIO_SETINFO, &tmpinfo);
144 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
145 if (retval < 0)
146 return retval;
147 INTARG = tmpinfo.play.channels - 1;
148 break;
149 case SNDCTL_DSP_GETBLKSIZE:
150 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
151 if (retval < 0)
152 return retval;
153 setblocksize(fd, &tmpinfo);
154 INTARG = tmpinfo.blocksize;
155 break;
156 case SNDCTL_DSP_SETFMT:
157 AUDIO_INITINFO(&tmpinfo);
158 switch (INTARG) {
159 case AFMT_MU_LAW:
160 tmpinfo.play.precision =
161 tmpinfo.record.precision = 8;
162 tmpinfo.play.encoding =
163 tmpinfo.record.encoding = AUDIO_ENCODING_ULAW;
164 break;
165 case AFMT_A_LAW:
166 tmpinfo.play.precision =
167 tmpinfo.record.precision = 8;
168 tmpinfo.play.encoding =
169 tmpinfo.record.encoding = AUDIO_ENCODING_ALAW;
170 break;
171 case AFMT_U8:
172 tmpinfo.play.precision =
173 tmpinfo.record.precision = 8;
174 tmpinfo.play.encoding =
175 tmpinfo.record.encoding = AUDIO_ENCODING_ULINEAR;
176 break;
177 case AFMT_S8:
178 tmpinfo.play.precision =
179 tmpinfo.record.precision = 8;
180 tmpinfo.play.encoding =
181 tmpinfo.record.encoding = AUDIO_ENCODING_SLINEAR;
182 break;
183 case AFMT_S16_LE:
184 tmpinfo.play.precision =
185 tmpinfo.record.precision = 16;
186 tmpinfo.play.encoding =
187 tmpinfo.record.encoding = AUDIO_ENCODING_SLINEAR_LE;
188 break;
189 case AFMT_S16_BE:
190 tmpinfo.play.precision =
191 tmpinfo.record.precision = 16;
192 tmpinfo.play.encoding =
193 tmpinfo.record.encoding = AUDIO_ENCODING_SLINEAR_BE;
194 break;
195 case AFMT_U16_LE:
196 tmpinfo.play.precision =
197 tmpinfo.record.precision = 16;
198 tmpinfo.play.encoding =
199 tmpinfo.record.encoding = AUDIO_ENCODING_ULINEAR_LE;
200 break;
201 case AFMT_U16_BE:
202 tmpinfo.play.precision =
203 tmpinfo.record.precision = 16;
204 tmpinfo.play.encoding =
205 tmpinfo.record.encoding = AUDIO_ENCODING_ULINEAR_BE;
206 break;
207 case AFMT_S24_LE:
208 tmpinfo.play.precision =
209 tmpinfo.record.precision = 24;
210 tmpinfo.play.encoding =
211 tmpinfo.record.encoding = AUDIO_ENCODING_SLINEAR_LE;
212 break;
213 case AFMT_S24_BE:
214 tmpinfo.play.precision =
215 tmpinfo.record.precision = 24;
216 tmpinfo.play.encoding =
217 tmpinfo.record.encoding = AUDIO_ENCODING_SLINEAR_BE;
218 break;
219 case AFMT_S32_LE:
220 tmpinfo.play.precision =
221 tmpinfo.record.precision = 32;
222 tmpinfo.play.encoding =
223 tmpinfo.record.encoding = AUDIO_ENCODING_SLINEAR_LE;
224 break;
225 case AFMT_S32_BE:
226 tmpinfo.play.precision =
227 tmpinfo.record.precision = 32;
228 tmpinfo.play.encoding =
229 tmpinfo.record.encoding = AUDIO_ENCODING_SLINEAR_BE;
230 break;
231 case AFMT_AC3:
232 tmpinfo.play.precision =
233 tmpinfo.record.precision = 16;
234 tmpinfo.play.encoding =
235 tmpinfo.record.encoding = AUDIO_ENCODING_AC3;
236 break;
237 default:
238 return EINVAL;
239 }
240 (void) ioctl(fd, AUDIO_SETINFO, &tmpinfo);
241 /* FALLTHRU */
242 case SOUND_PCM_READ_BITS:
243 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
244 if (retval < 0)
245 return retval;
246 switch (tmpinfo.play.encoding) {
247 case AUDIO_ENCODING_ULAW:
248 idat = AFMT_MU_LAW;
249 break;
250 case AUDIO_ENCODING_ALAW:
251 idat = AFMT_A_LAW;
252 break;
253 case AUDIO_ENCODING_SLINEAR_LE:
254 if (tmpinfo.play.precision == 32)
255 idat = AFMT_S32_LE;
256 else if (tmpinfo.play.precision == 24)
257 idat = AFMT_S24_LE;
258 else if (tmpinfo.play.precision == 16)
259 idat = AFMT_S16_LE;
260 else
261 idat = AFMT_S8;
262 break;
263 case AUDIO_ENCODING_SLINEAR_BE:
264 if (tmpinfo.play.precision == 32)
265 idat = AFMT_S32_BE;
266 else if (tmpinfo.play.precision == 24)
267 idat = AFMT_S24_BE;
268 else if (tmpinfo.play.precision == 16)
269 idat = AFMT_S16_BE;
270 else
271 idat = AFMT_S8;
272 break;
273 case AUDIO_ENCODING_ULINEAR_LE:
274 if (tmpinfo.play.precision == 16)
275 idat = AFMT_U16_LE;
276 else
277 idat = AFMT_U8;
278 break;
279 case AUDIO_ENCODING_ULINEAR_BE:
280 if (tmpinfo.play.precision == 16)
281 idat = AFMT_U16_BE;
282 else
283 idat = AFMT_U8;
284 break;
285 case AUDIO_ENCODING_ADPCM:
286 idat = AFMT_IMA_ADPCM;
287 break;
288 case AUDIO_ENCODING_AC3:
289 idat = AFMT_AC3;
290 break;
291 }
292 INTARG = idat;
293 break;
294 case SNDCTL_DSP_CHANNELS:
295 AUDIO_INITINFO(&tmpinfo);
296 tmpinfo.play.channels = INTARG;
297 (void) ioctl(fd, AUDIO_SETINFO, &tmpinfo);
298 AUDIO_INITINFO(&tmpinfo);
299 tmpinfo.record.channels = INTARG;
300 (void) ioctl(fd, AUDIO_SETINFO, &tmpinfo);
301 /* FALLTHRU */
302 case SOUND_PCM_READ_CHANNELS:
303 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
304 if (retval < 0)
305 return retval;
306 INTARG = tmpinfo.play.channels;
307 break;
308 case SOUND_PCM_WRITE_FILTER:
309 case SOUND_PCM_READ_FILTER:
310 errno = EINVAL;
311 return -1; /* XXX unimplemented */
312 case SNDCTL_DSP_SUBDIVIDE:
313 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
314 if (retval < 0)
315 return retval;
316 setblocksize(fd, &tmpinfo);
317 idat = INTARG;
318 if (idat == 0)
319 idat = tmpinfo.play.buffer_size / tmpinfo.blocksize;
320 idat = (tmpinfo.play.buffer_size / idat) & -4;
321 AUDIO_INITINFO(&tmpinfo);
322 tmpinfo.blocksize = idat;
323 retval = ioctl(fd, AUDIO_SETINFO, &tmpinfo);
324 if (retval < 0)
325 return retval;
326 INTARG = tmpinfo.play.buffer_size / tmpinfo.blocksize;
327 break;
328 case SNDCTL_DSP_SETFRAGMENT:
329 AUDIO_INITINFO(&tmpinfo);
330 idat = INTARG;
331 if ((idat & 0xffff) < 4 || (idat & 0xffff) > 17)
332 return EINVAL;
333 tmpinfo.blocksize = 1 << (idat & 0xffff);
334 tmpinfo.hiwat = ((unsigned)idat >> 16) & 0x7fff;
335 if (tmpinfo.hiwat == 0) /* 0 means set to max */
336 tmpinfo.hiwat = 65536;
337 (void) ioctl(fd, AUDIO_SETINFO, &tmpinfo);
338 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
339 if (retval < 0)
340 return retval;
341 u = tmpinfo.blocksize;
342 for(idat = 0; u > 1; idat++, u >>= 1)
343 ;
344 idat |= (tmpinfo.hiwat & 0x7fff) << 16;
345 INTARG = idat;
346 break;
347 case SNDCTL_DSP_GETFMTS:
348 for(idat = 0, tmpenc.index = 0;
349 ioctl(fd, AUDIO_GETENC, &tmpenc) == 0;
350 tmpenc.index++) {
351 switch(tmpenc.encoding) {
352 case AUDIO_ENCODING_ULAW:
353 idat |= AFMT_MU_LAW;
354 break;
355 case AUDIO_ENCODING_ALAW:
356 idat |= AFMT_A_LAW;
357 break;
358 case AUDIO_ENCODING_SLINEAR:
359 idat |= AFMT_S8;
360 break;
361 case AUDIO_ENCODING_SLINEAR_LE:
362 if (tmpenc.precision == 32)
363 idat |= AFMT_S32_LE;
364 else if (tmpenc.precision == 24)
365 idat |= AFMT_S24_LE;
366 else if (tmpenc.precision == 16)
367 idat |= AFMT_S16_LE;
368 else
369 idat |= AFMT_S8;
370 break;
371 case AUDIO_ENCODING_SLINEAR_BE:
372 if (tmpenc.precision == 32)
373 idat |= AFMT_S32_BE;
374 else if (tmpenc.precision == 24)
375 idat |= AFMT_S24_BE;
376 else if (tmpenc.precision == 16)
377 idat |= AFMT_S16_BE;
378 else
379 idat |= AFMT_S8;
380 break;
381 case AUDIO_ENCODING_ULINEAR:
382 idat |= AFMT_U8;
383 break;
384 case AUDIO_ENCODING_ULINEAR_LE:
385 if (tmpenc.precision == 16)
386 idat |= AFMT_U16_LE;
387 else
388 idat |= AFMT_U8;
389 break;
390 case AUDIO_ENCODING_ULINEAR_BE:
391 if (tmpenc.precision == 16)
392 idat |= AFMT_U16_BE;
393 else
394 idat |= AFMT_U8;
395 break;
396 case AUDIO_ENCODING_ADPCM:
397 idat |= AFMT_IMA_ADPCM;
398 break;
399 case AUDIO_ENCODING_AC3:
400 idat |= AFMT_AC3;
401 break;
402 default:
403 break;
404 }
405 }
406 INTARG = idat;
407 break;
408 case SNDCTL_DSP_GETOSPACE:
409 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
410 if (retval < 0)
411 return retval;
412 setblocksize(fd, &tmpinfo);
413 bufinfo.fragsize = tmpinfo.blocksize;
414 bufinfo.fragments = tmpinfo.hiwat - (tmpinfo.play.seek
415 + tmpinfo.blocksize - 1) / tmpinfo.blocksize;
416 bufinfo.fragstotal = tmpinfo.hiwat;
417 bufinfo.bytes = tmpinfo.hiwat * tmpinfo.blocksize
418 - tmpinfo.play.seek;
419 *(struct audio_buf_info *)argp = bufinfo;
420 break;
421 case SNDCTL_DSP_GETISPACE:
422 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
423 if (retval < 0)
424 return retval;
425 setblocksize(fd, &tmpinfo);
426 bufinfo.fragsize = tmpinfo.blocksize;
427 bufinfo.fragments = tmpinfo.record.seek / tmpinfo.blocksize;
428 bufinfo.fragstotal =
429 tmpinfo.record.buffer_size / tmpinfo.blocksize;
430 bufinfo.bytes = tmpinfo.record.seek;
431 *(struct audio_buf_info *)argp = bufinfo;
432 break;
433 case SNDCTL_DSP_NONBLOCK:
434 idat = 1;
435 retval = ioctl(fd, FIONBIO, &idat);
436 if (retval < 0)
437 return retval;
438 break;
439 case SNDCTL_DSP_GETCAPS:
440 retval = ioctl(fd, AUDIO_GETPROPS, &idata);
441 if (retval < 0)
442 return retval;
443 idat = DSP_CAP_TRIGGER; /* pretend we have trigger */
444 if (idata & AUDIO_PROP_FULLDUPLEX)
445 idat |= DSP_CAP_DUPLEX;
446 if (idata & AUDIO_PROP_MMAP)
447 idat |= DSP_CAP_MMAP;
448 INTARG = idat;
449 break;
450#if 0
451 case SNDCTL_DSP_GETTRIGGER:
452 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
453 if (retval < 0)
454 return retval;
455 idat = (tmpinfo.play.pause ? 0 : PCM_ENABLE_OUTPUT) |
456 (tmpinfo.record.pause ? 0 : PCM_ENABLE_INPUT);
457 retval = copyout(&idat, SCARG(uap, data), sizeof idat);
458 if (retval < 0)
459 return retval;
460 break;
461 case SNDCTL_DSP_SETTRIGGER:
462 AUDIO_INITINFO(&tmpinfo);
463 retval = copyin(SCARG(uap, data), &idat, sizeof idat);
464 if (retval < 0)
465 return retval;
466 tmpinfo.play.pause = (idat & PCM_ENABLE_OUTPUT) == 0;
467 tmpinfo.record.pause = (idat & PCM_ENABLE_INPUT) == 0;
468 (void) ioctl(fd, AUDIO_SETINFO, &tmpinfo);
469 retval = copyout(&idat, SCARG(uap, data), sizeof idat);
470 if (retval < 0)
471 return retval;
472 break;
473#else
474 case SNDCTL_DSP_GETTRIGGER:
475 case SNDCTL_DSP_SETTRIGGER:
476 /* XXX Do nothing for now. */
477 INTARG = PCM_ENABLE_OUTPUT;
478 break;
479#endif
480 case SNDCTL_DSP_GETIPTR:
481 retval = ioctl(fd, AUDIO_GETIOFFS, &tmpoffs);
482 if (retval < 0)
483 return retval;
484 cntinfo.bytes = tmpoffs.samples;
485 cntinfo.blocks = tmpoffs.deltablks;
486 cntinfo.ptr = tmpoffs.offset;
487 *(struct count_info *)argp = cntinfo;
488 break;
489 case SNDCTL_DSP_GETOPTR:
490 retval = ioctl(fd, AUDIO_GETOOFFS, &tmpoffs);
491 if (retval < 0)
492 return retval;
493 cntinfo.bytes = tmpoffs.samples;
494 cntinfo.blocks = tmpoffs.deltablks;
495 cntinfo.ptr = tmpoffs.offset;
496 *(struct count_info *)argp = cntinfo;
497 break;
498 case SNDCTL_SYSINFO:
499 strlcpy(tmpsysinfo.product, "OSS/NetBSD",
500 sizeof tmpsysinfo.product);
501 strlcpy(tmpsysinfo.version, version, sizeof tmpsysinfo.version);
502 strlcpy(tmpsysinfo.license, license, sizeof tmpsysinfo.license);
503 tmpsysinfo.versionnum = SOUND_VERSION;
504 memset(tmpsysinfo.options, 0, 8);
505 tmpsysinfo.numaudios = OSS_MAX_AUDIO_DEVS;
506 tmpsysinfo.numaudioengines = 1;
507 memset(tmpsysinfo.openedaudio, 0, sizeof(tmpsysinfo.openedaudio));
508 tmpsysinfo.numsynths = 1;
509 tmpsysinfo.nummidis = -1;
510 tmpsysinfo.numtimers = -1;
511 tmpsysinfo.nummixers = 1;
512 tmpsysinfo.numcards = 1;
513 memset(tmpsysinfo.openedmidi, 0, sizeof(tmpsysinfo.openedmidi));
514 *(struct oss_sysinfo *)argp = tmpsysinfo;
515 break;
516 case SNDCTL_ENGINEINFO:
517 case SNDCTL_AUDIOINFO:
518 devno = 0;
519 tmpaudioinfo = (struct oss_audioinfo*)argp;
520 if (tmpaudioinfo == NULL)
521 return EINVAL;
522 if (tmpaudioinfo->dev < 0) {
523 fstat(fd, &tmpstat);
524 if ((tmpstat.st_rdev & 0xff00) == 0x2a00)
525 devno = tmpstat.st_rdev & 0xff;
526 if (devno >= 0x80)
527 tmpaudioinfo->dev = devno & 0x7f;
528 }
529 if (tmpaudioinfo->dev < 0)
530 tmpaudioinfo->dev = 0;
531
532 snprintf(tmpaudioinfo->devnode, OSS_DEVNODE_SIZE,
533 "/dev/audio%d", tmpaudioinfo->dev);
534
535 retval = ioctl(fd, AUDIO_GETDEV, &tmpaudiodev);
536 if (retval < 0)
537 return retval;
538 retval = ioctl(fd, AUDIO_GETINFO, &tmpinfo);
539 if (retval < 0)
540 return retval;
541 retval = ioctl(fd, AUDIO_GETPROPS, &idata);
542 if (retval < 0)
543 return retval;
544 idat = DSP_CAP_TRIGGER; /* pretend we have trigger */
545 if (idata & AUDIO_PROP_FULLDUPLEX)
546 idat |= DSP_CAP_DUPLEX;
547 if (idata & AUDIO_PROP_MMAP)
548 idat |= DSP_CAP_MMAP;
549 idat = PCM_CAP_INPUT | PCM_CAP_OUTPUT;
550 strlcpy(tmpaudioinfo->name, tmpaudiodev.name,
551 sizeof tmpaudioinfo->name);
552 tmpaudioinfo->busy = tmpinfo.play.open;
553 tmpaudioinfo->pid = -1;
554 tmpaudioinfo->caps = idat;
555 ioctl(fd, SNDCTL_DSP_GETFMTS, &tmpaudioinfo->iformats);
556 tmpaudioinfo->oformats = tmpaudioinfo->iformats;
557 tmpaudioinfo->magic = -1;
558 memset(tmpaudioinfo->cmd, 0, 64);
559 tmpaudioinfo->card_number = -1;
560 memset(tmpaudioinfo->song_name, 0, 64);
561 memset(tmpaudioinfo->label, 0, 16);
562 tmpaudioinfo->port_number = tmpinfo.play.port;
563 tmpaudioinfo->mixer_dev = tmpaudioinfo->dev;
564 tmpaudioinfo->legacy_device = -1;
565 tmpaudioinfo->enabled = 1;
566 tmpaudioinfo->flags = -1;
567 tmpaudioinfo->min_rate = tmpinfo.play.sample_rate;
568 tmpaudioinfo->max_rate = tmpinfo.play.sample_rate;
569 tmpaudioinfo->nrates = 2;
570 for (i = 0; i < tmpaudioinfo->nrates; i++)
571 tmpaudioinfo->rates[i] = tmpinfo.play.sample_rate;
572 tmpaudioinfo->min_channels = tmpinfo.play.channels;
573 tmpaudioinfo->max_channels = tmpinfo.play.channels;
574 tmpaudioinfo->binding = -1;
575 tmpaudioinfo->rate_source = -1;
576 memset(tmpaudioinfo->handle, 0, 16);
577 tmpaudioinfo->next_play_engine = 0;
578 tmpaudioinfo->next_rec_engine = 0;
579 argp = tmpaudioinfo;
580 break;
581 case SNDCTL_DSP_GETPLAYVOL:
582 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
583 if (retval < 0)
584 return retval;
585 *(uint *)argp = tmpinfo.play.gain;
586 break;
587 case SNDCTL_DSP_SETPLAYVOL:
588 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
589 if (retval < 0)
590 return retval;
591 if (*(uint *)argp > 255)
592 tmpinfo.play.gain = 255;
593 else
594 tmpinfo.play.gain = *(uint *)argp;
595 retval = ioctl(fd, AUDIO_SETINFO, &tmpinfo);
596 if (retval < 0)
597 return retval;
598 break;
599 case SNDCTL_DSP_GETRECVOL:
600 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
601 if (retval < 0)
602 return retval;
603 *(uint *)argp = tmpinfo.record.gain;
604 break;
605 case SNDCTL_DSP_SETRECVOL:
606 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
607 if (retval < 0)
608 return retval;
609 if (*(uint *)argp > 255)
610 tmpinfo.record.gain = 255;
611 else
612 tmpinfo.record.gain = *(uint *)argp;
613 retval = ioctl(fd, AUDIO_SETINFO, &tmpinfo);
614 if (retval < 0)
615 return retval;
616 break;
617 case SNDCTL_DSP_SKIP:
618 case SNDCTL_DSP_SILENCE:
619 return EINVAL;
620 case SNDCTL_DSP_SETDUPLEX:
621 idat = 1;
622 retval = ioctl(fd, AUDIO_SETFD, &idat);
623 if (retval < 0)
624 return retval;
625 break;
626 case SNDCTL_DSP_GETODELAY:
627 retval = ioctl(fd, AUDIO_GETBUFINFO, &tmpinfo);
628 if (retval < 0)
629 return retval;
630 idat = tmpinfo.play.seek + tmpinfo.blocksize / 2;
631 INTARG = idat;
632 break;
633 case SNDCTL_DSP_PROFILE:
634 /* This gives just a hint to the driver,
635 * implementing it as a NOP is ok
636 */
637 break;
638 case SNDCTL_DSP_MAPINBUF:
639 case SNDCTL_DSP_MAPOUTBUF:
640 case SNDCTL_DSP_SETSYNCRO:
641 errno = EINVAL;
642 return -1; /* XXX unimplemented */
643 default:
644 errno = EINVAL;
645 return -1;
646 }
647
648 return 0;
649}

652/* If the NetBSD mixer device should have more than NETBSD_MAXDEVS devices
653 * some will not be available to Linux */
654#define NETBSD_MAXDEVS 64
655struct audiodevinfo {
656 int done;
657 dev_t dev;
658 int16_t devmap[SOUND_MIXER_NRDEVICES],
659 rdevmap[NETBSD_MAXDEVS];
660 char names[NETBSD_MAXDEVS][MAX_AUDIO_DEV_LEN];
661 int enum2opaque[NETBSD_MAXDEVS];
662 u_long devmask, recmask, stereomask;
663 u_long caps;
664 int source;
665};

667static int
668opaque_to_enum(struct audiodevinfo *di, audio_mixer_name_t *label, int opq)
669{
670 int i, o;
671
672 for (i = 0; i < NETBSD_MAXDEVS; i++) {
673 o = di->enum2opaque[i];
674 if (o == opq)
675 break;
676 if (o == -1 && label != NULL &&
677 !strncmp(di->names[i], label->name, sizeof di->names[i])) {
678 di->enum2opaque[i] = opq;
679 break;
680 }
681 }
682 if (i >= NETBSD_MAXDEVS)
683 i = -1;
684 /*printf("opq_to_enum %s %d -> %d\n", label->name, opq, i);*/
685 return (i);
686}

688static int
689enum_to_ord(struct audiodevinfo *di, int enm)
690{
691 if (enm >= NETBSD_MAXDEVS)
692 return (-1);
693
694 /*printf("enum_to_ord %d -> %d\n", enm, di->enum2opaque[enm]);*/
695 return (di->enum2opaque[enm]);
696}

698static int
699enum_to_mask(struct audiodevinfo *di, int enm)
700{
701 int m;
702 if (enm >= NETBSD_MAXDEVS)
703 return (0);
704
705 m = di->enum2opaque[enm];
706 if (m == -1)
707 m = 0;
708 /*printf("enum_to_mask %d -> %d\n", enm, di->enum2opaque[enm]);*/
709 return (m);
710}

712/*
713 * Collect the audio device information to allow faster
714 * emulation of the Linux mixer ioctls. Cache the information
715 * to eliminate the overhead of repeating all the ioctls needed
716 * to collect the information.
717 */
718static struct audiodevinfo *
719getdevinfo(int fd)
720{
721 mixer_devinfo_t mi;
722 int i, j, e;
723 static struct {
724 const char *name;
725 int code;
726 } *dp, devs[] = {
727 { AudioNmicrophone, SOUND_MIXER_MIC },
728 { AudioNline, SOUND_MIXER_LINE },
729 { AudioNcd, SOUND_MIXER_CD },
730 { AudioNdac, SOUND_MIXER_PCM },
731 { AudioNaux, SOUND_MIXER_LINE1 },
732 { AudioNrecord, SOUND_MIXER_IMIX },
733 { AudioNmaster, SOUND_MIXER_VOLUME },
734 { AudioNtreble, SOUND_MIXER_TREBLE },
735 { AudioNbass, SOUND_MIXER_BASS },
736 { AudioNspeaker, SOUND_MIXER_SPEAKER },
737/* { AudioNheadphone, ?? },*/
738 { AudioNoutput, SOUND_MIXER_OGAIN },
739 { AudioNinput, SOUND_MIXER_IGAIN },
740/* { AudioNmaster, SOUND_MIXER_SPEAKER },*/
741/* { AudioNstereo, ?? },*/
742/* { AudioNmono, ?? },*/
743 { AudioNfmsynth, SOUND_MIXER_SYNTH },
744/* { AudioNwave, SOUND_MIXER_PCM },*/
745 { AudioNmidi, SOUND_MIXER_SYNTH },
746/* { AudioNmixerout, ?? },*/
747 { 0, -1 }
748 };
749 static struct audiodevinfo devcache = { .done = 0 };
750 struct audiodevinfo *di = &devcache;
751 struct stat sb;
752 size_t mlen, dlen;
753
754 /* Figure out what device it is so we can check if the
755 * cached data is valid.
756 */
757 if (fstat(fd, &sb) < 0)
758 return 0;
759 if (di->done && di->dev == sb.st_dev)
760 return di;
761
762 di->done = 1;
763 di->dev = sb.st_dev;
764 di->devmask = 0;
765 di->recmask = 0;
766 di->stereomask = 0;
767 di->source = ~0;
768 di->caps = 0;
769 for(i = 0; i < SOUND_MIXER_NRDEVICES; i++)
770 di->devmap[i] = -1;
771 for(i = 0; i < NETBSD_MAXDEVS; i++) {
772 di->rdevmap[i] = -1;
773 di->names[i][0] = '\0';
774 di->enum2opaque[i] = -1;
775 }
776 for(i = 0; i < NETBSD_MAXDEVS; i++) {
777 mi.index = i;
778 if (ioctl(fd, AUDIO_MIXER_DEVINFO, &mi) < 0)
779 break;
780 switch(mi.type) {
781 case AUDIO_MIXER_VALUE:
782 for(dp = devs; dp->name; dp++) {
783 if (strcmp(dp->name, mi.label.name) == 0)
784 break;
785 dlen = strlen(dp->name);
786 mlen = strlen(mi.label.name);
787 if (dlen < mlen
788 && mi.label.name[mlen-dlen-1] == '.'
789 && strcmp(dp->name,
790 mi.label.name + mlen - dlen) == 0)
791 break;
792 }
793 if (dp->code >= 0) {
794 di->devmap[dp->code] = i;
795 di->rdevmap[i] = dp->code;
796 di->devmask |= 1 << dp->code;
797 if (mi.un.v.num_channels == 2)
798 di->stereomask |= 1 << dp->code;
799 strlcpy(di->names[i], mi.label.name,
800 sizeof di->names[i]);
801 }
802 break;
803 }
804 }
805 for(i = 0; i < NETBSD_MAXDEVS; i++) {
806 mi.index = i;
807 if (ioctl(fd, AUDIO_MIXER_DEVINFO, &mi) < 0)
808 break;
809 if (strcmp(mi.label.name, AudioNsource) != 0)
810 continue;
811 di->source = i;
812 switch(mi.type) {
813 case AUDIO_MIXER_ENUM:
814 for(j = 0; j < mi.un.e.num_mem; j++) {
815 e = opaque_to_enum(di,
816 &mi.un.e.member[j].label,
817 mi.un.e.member[j].ord);
818 if (e >= 0)
819 di->recmask |= 1 << di->rdevmap[e];
820 }
821 di->caps = SOUND_CAP_EXCL_INPUT;
822 break;
823 case AUDIO_MIXER_SET:
824 for(j = 0; j < mi.un.s.num_mem; j++) {
825 e = opaque_to_enum(di,
826 &mi.un.s.member[j].label,
827 mi.un.s.member[j].mask);
828 if (e >= 0)
829 di->recmask |= 1 << di->rdevmap[e];
830 }
831 break;
832 }
833 }
834 return di;
835}

837int
838mixer_ioctl(int fd, unsigned long com, void *argp)
839{
840 struct audiodevinfo *di;
841 struct mixer_info *omi;
842 struct audio_device adev;
843 mixer_ctrl_t mc;
844 u_long idat, n;
845 int i;
846 int retval;
847 int l, r, error, e;
848
849 idat = 0;
850 di = getdevinfo(fd);
851 if (di == 0)
852 return -1;
853
854 switch (com) {
855 case OSS_GETVERSION:
856 idat = SOUND_VERSION;
857 break;
858 case SOUND_MIXER_INFO:
859 case SOUND_OLD_MIXER_INFO:
860 error = ioctl(fd, AUDIO_GETDEV, &adev);
861 if (error)
862 return (error);
863 omi = argp;
864 if (com == SOUND_MIXER_INFO)
865 omi->modify_counter = 1;
866 strlcpy(omi->id, adev.name, sizeof omi->id);
867 strlcpy(omi->name, adev.name, sizeof omi->name);
868 return 0;
869 case SOUND_MIXER_READ_RECSRC:
870 if (di->source == -1)
871 return EINVAL;
872 mc.dev = di->source;
873 if (di->caps & SOUND_CAP_EXCL_INPUT) {
874 mc.type = AUDIO_MIXER_ENUM;
875 retval = ioctl(fd, AUDIO_MIXER_READ, &mc);
876 if (retval < 0)
877 return retval;
878 e = opaque_to_enum(di, NULL, mc.un.ord);
879 if (e >= 0)
880 idat = 1 << di->rdevmap[e];
881 } else {
882 mc.type = AUDIO_MIXER_SET;
883 retval = ioctl(fd, AUDIO_MIXER_READ, &mc);
884 if (retval < 0)
885 return retval;
886 e = opaque_to_enum(di, NULL, mc.un.mask);
887 if (e >= 0)
888 idat = 1 << di->rdevmap[e];
889 }
890 break;
891 case SOUND_MIXER_READ_DEVMASK:
892 idat = di->devmask;
893 break;
894 case SOUND_MIXER_READ_RECMASK:
895 idat = di->recmask;
896 break;
897 case SOUND_MIXER_READ_STEREODEVS:
898 idat = di->stereomask;
899 break;
900 case SOUND_MIXER_READ_CAPS:
901 idat = di->caps;
902 break;
903 case SOUND_MIXER_WRITE_RECSRC:
904 case SOUND_MIXER_WRITE_R_RECSRC:
905 if (di->source == -1)
906 return EINVAL;
907 mc.dev = di->source;
908 idat = INTARG;
909 if (di->caps & SOUND_CAP_EXCL_INPUT) {
910 mc.type = AUDIO_MIXER_ENUM;
911 for(i = 0; i < SOUND_MIXER_NRDEVICES; i++)
912 if (idat & (1 << i))
913 break;
914 if (i >= SOUND_MIXER_NRDEVICES ||
915 di->devmap[i] == -1)
916 return EINVAL;
917 mc.un.ord = enum_to_ord(di, di->devmap[i]);
918 } else {
919 mc.type = AUDIO_MIXER_SET;
920 mc.un.mask = 0;
921 for(i = 0; i < SOUND_MIXER_NRDEVICES; i++) {
922 if (idat & (1 << i)) {
923 if (di->devmap[i] == -1)
924 return EINVAL;
925 mc.un.mask |=
926 enum_to_mask(di, di->devmap[i]);
927 }
928 }
929 }
930 return ioctl(fd, AUDIO_MIXER_WRITE, &mc);
931 default:
932 if (MIXER_READ(SOUND_MIXER_FIRST) <= com &&
933 com < MIXER_READ(SOUND_MIXER_NRDEVICES)) {
934 n = GET_DEV(com);
935 if (di->devmap[n] == -1)
936 return EINVAL;
937 mc.dev = di->devmap[n];
938 mc.type = AUDIO_MIXER_VALUE;
939 doread:
940 mc.un.value.num_channels =
941 di->stereomask & (1 << (u_int)n) ? 2 : 1;
942 retval = ioctl(fd, AUDIO_MIXER_READ, &mc);
943 if (retval < 0)
944 return retval;
945 if (mc.type != AUDIO_MIXER_VALUE)
946 return EINVAL;
947 if (mc.un.value.num_channels != 2) {
948 l = r =
949 mc.un.value.level[AUDIO_MIXER_LEVEL_MONO];
950 } else {
951 l = mc.un.value.level[AUDIO_MIXER_LEVEL_LEFT];
952 r = mc.un.value.level[AUDIO_MIXER_LEVEL_RIGHT];
953 }
954 idat = TO_OSSVOL(l) | (TO_OSSVOL(r) << 8);
955 break;
956 } else if ((MIXER_WRITE_R(SOUND_MIXER_FIRST) <= com &&
957 com < MIXER_WRITE_R(SOUND_MIXER_NRDEVICES)) ||
958 (MIXER_WRITE(SOUND_MIXER_FIRST) <= com &&
959 com < MIXER_WRITE(SOUND_MIXER_NRDEVICES))) {
960 n = GET_DEV(com);
961 if (di->devmap[n] == -1)
962 return EINVAL;
963 idat = INTARG;
964 l = FROM_OSSVOL((u_int)idat & 0xff);
965 r = FROM_OSSVOL(((u_int)idat >> 8) & 0xff);
966 mc.dev = di->devmap[n];
967 mc.type = AUDIO_MIXER_VALUE;
968 if (di->stereomask & (1 << (u_int)n)) {
969 mc.un.value.num_channels = 2;
970 mc.un.value.level[AUDIO_MIXER_LEVEL_LEFT] = l;
971 mc.un.value.level[AUDIO_MIXER_LEVEL_RIGHT] = r;
972 } else {
973 mc.un.value.num_channels = 1;
974 mc.un.value.level[AUDIO_MIXER_LEVEL_MONO] =
975 (l + r) / 2;
976 }
977 retval = ioctl(fd, AUDIO_MIXER_WRITE, &mc);
978 if (retval < 0)
979 return retval;
980 if (MIXER_WRITE(SOUND_MIXER_FIRST) <= com &&
981 com < MIXER_WRITE(SOUND_MIXER_NRDEVICES))
982 return 0;
983 goto doread;
984 } else {
985 errno = EINVAL;
986 return -1;
987 }
988 }
989 INTARG = (int)idat;
990 return 0;
991}
