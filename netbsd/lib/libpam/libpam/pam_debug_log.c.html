45/* Print a verbose error, including the function name and a
46 * cleaned up filename.
47 */
48void
49_pam_verbose_error(pam_handle_t *pamh, int flags,
50 const char *file, const char *function, const char *format, ...)
51{
52 va_list ap;
53 char *msg;
54 int rv;
55 const char *modname, *period;
56
57 if ((flags & PAM_SILENT) || openpam_get_option(pamh, "no_warn"))
58 return;
59 modname = strrchr(file, '/');
60 if (modname == NULL)
61 modname = file;
62 period = strchr(modname, '.');
63 if (period == NULL)
64 period = strchr(modname, '\0');
65 va_start(ap, format);
66 rv = vasprintf(&msg, format, ap);
67 va_end(ap);
68 if (rv < 0)
69 return;
70 pam_error(pamh, "%.*s: %s: %s\n", (int)(ssize_t)(period - modname),
71 modname, function, msg);
72 free(msg);
73}
