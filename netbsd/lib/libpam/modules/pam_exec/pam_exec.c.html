57#define ENV_ITEM(n) { (n), #n }
58static struct {
59 int item;
60 const char *name;
61} env_items[] = {
62 ENV_ITEM(PAM_SERVICE),
63 ENV_ITEM(PAM_USER),
64 ENV_ITEM(PAM_TTY),
65 ENV_ITEM(PAM_RHOST),
66 ENV_ITEM(PAM_RUSER),
67};
68
69static int
70_pam_exec(pam_handle_t *pamh __unused, int flags __unused,
71 int argc, const char *argv[])
72{
73 size_t envlen, i, nitems;
74 int pam_err, status;
75 char **envlist, **tmp;
76 volatile int childerr;
77 pid_t pid;
78
79 if (argc < 1)
80 return (PAM_SERVICE_ERR);
81
82 /*
83 * XXX For additional credit, divert child's stdin/stdout/stderr
84 * to the conversation function.
85 */
86
87 /*
88 * Set up the child's environment list. It consists of the PAM
89 * environment, plus a few hand-picked PAM items.
90 */
91 envlist = pam_getenvlist(pamh);
92 for (envlen = 0; envlist[envlen] != NULL; ++envlen)
93 /* nothing */ ;
94 nitems = sizeof(env_items) / sizeof(*env_items);
95 tmp = realloc(envlist, (envlen + nitems + 1) * sizeof(*envlist));
96 if (tmp == NULL) {
97 openpam_free_envlist(envlist);
98 return (PAM_BUF_ERR);
99 }
100 envlist = tmp;
101 for (i = 0; i < nitems; ++i) {
102 const void *item;
103 char *envstr;
104
105 pam_err = pam_get_item(pamh, env_items[i].item, &item);
106 if (pam_err != PAM_SUCCESS || item == NULL)
107 continue;
108 asprintf(&envstr, "%s=%s", env_items[i].name,
109 (const char *)item);
110 if (envstr == NULL) {
111 openpam_free_envlist(envlist);
112 return (PAM_BUF_ERR);
113 }
114 envlist[envlen++] = envstr;
115 envlist[envlen] = NULL;
116 }
117
118 /*
119 * Fork and run the command. By using vfork() instead of fork(),
120 * we can distinguish between an execve() failure and a non-zero
121 * exit code from the command.
122 */
123 childerr = 0;
124 if ((pid = vfork()) == 0) {
125 /*LINTED const cast*/
126 execve(argv[0], (char **)__UNCONST(argv), envlist);
127 childerr = errno;
128 _exit(1);
129 }
130 openpam_free_envlist(envlist);
131 if (pid == -1) {
132 openpam_log(PAM_LOG_ERROR, "vfork(): %s", strerror(errno));
133 return (PAM_SYSTEM_ERR);
134 }
135 if (waitpid(pid, &status, 0) == -1) {
136 openpam_log(PAM_LOG_ERROR, "waitpid(): %s", strerror(errno));
137 return (PAM_SYSTEM_ERR);
138 }
139 if (childerr != 0) {
140 openpam_log(PAM_LOG_ERROR, "execve(): %s", strerror(errno));
141 return (PAM_SYSTEM_ERR);
142 }
143 if (WIFSIGNALED(status)) {
144 openpam_log(PAM_LOG_ERROR, "%s caught signal %d%s",
145 argv[0], WTERMSIG(status),
146 WCOREDUMP(status) ? " (core dumped)" : "");
147 return (PAM_SYSTEM_ERR);
148 }
149 if (!WIFEXITED(status)) {
150 openpam_log(PAM_LOG_ERROR, "unknown status 0x%x", status);
151 return (PAM_SYSTEM_ERR);
152 }
153 if (WEXITSTATUS(status) != 0) {
154 openpam_log(PAM_LOG_ERROR, "%s returned code %d",
155 argv[0], WEXITSTATUS(status));
156 return (PAM_SYSTEM_ERR);
157 }
158 return (PAM_SUCCESS);
159}
