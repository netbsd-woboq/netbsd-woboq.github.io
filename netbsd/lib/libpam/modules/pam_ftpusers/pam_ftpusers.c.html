60PAM_EXTERN int
61pam_sm_acct_mgmt(pam_handle_t *pamh, int flags __unused,
62 int argc __unused, const char *argv[] __unused)
63{
64 struct passwd *pwd, pwres;
65 struct group *grp, grres;
66 const char *user;
67 int pam_err, found, allow;
68 char *line, *name, **mem;
69 size_t len, ulen;
70 FILE *f;
71 char pwbuf[1024], grbuf[1024];
72
73 pam_err = pam_get_user(pamh, &user, NULL);
74 if (pam_err != PAM_SUCCESS)
75 return (pam_err);
76 if (user == NULL ||
77 getpwnam_r(user, &pwres, pwbuf, sizeof(pwbuf), &pwd) != 0 ||
78 pwd == NULL)
79 return (PAM_SERVICE_ERR);
80
81 found = 0;
82 ulen = strlen(user);
83 if ((f = fopen(_PATH_FTPUSERS, "r")) == NULL) {
84 PAM_LOG("%s: %s", _PATH_FTPUSERS, strerror(errno));
85 goto done;
86 }
87 while (!found && (line = fgetln(f, &len)) != NULL) {
88 if (*line == '#')
89 continue;
90 while (len > 0 && isspace((unsigned char)line[len - 1]))
91 --len;
92 if (len == 0)
93 continue;
94 /* simple case first */
95 if (*line != '@') {
96 if (len == ulen && strncmp(user, line, len) == 0)
97 found = 1;
98 continue;
99 }
100 /* member of specified group? */
101 asprintf(&name, "%.*s", (int)len - 1, line + 1);
102 if (name == NULL) {
103 fclose(f);
104 return (PAM_BUF_ERR);
105 }
106 (void)getgrnam_r(name, &grres, grbuf, sizeof(grbuf), &grp);
107 free(name);
108 if (grp == NULL)
109 continue;
110 for (mem = grp->gr_mem; mem && *mem && !found; ++mem)
111 if (strcmp(user, *mem) == 0)
112 found = 1;
113 }
114 done:
115 allow = (openpam_get_option(pamh, "disallow") == NULL);
116 if (found)
117 pam_err = allow ? PAM_SUCCESS : PAM_AUTH_ERR;
118 else
119 pam_err = allow ? PAM_AUTH_ERR : PAM_SUCCESS;
120 if (f != NULL)
121 fclose(f);
122 return (pam_err);
123}
