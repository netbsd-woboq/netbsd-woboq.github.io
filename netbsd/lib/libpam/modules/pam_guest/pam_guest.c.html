54static int
55lookup(const char *str, const char *list)
56{
57 const char *next;
58 size_t len;
59
60 len = strlen(str);
61 while (*list != '\0') {
62 while (*list == ',')
63 ++list;
64 if ((next = strchr(list, ',')) == NULL)
65 next = strchr(list, '\0');
66 if (next - list == (ptrdiff_t)len &&
67 strncmp(list, str, len) == 0)
68 return (1);
69 list = next;
70 }
71 return (0);
72}

74PAM_EXTERN int
75pam_sm_authenticate(pam_handle_t *pamh, int flags __unused,
76 int argc __unused, const char *argv[] __unused)
77{
78 const char *authtok, *guests, *user;
79 int err, is_guest;
80
81 /* get target account */
82 if (pam_get_user(pamh, &user, NULL) != PAM_SUCCESS || user == NULL)
83 return (PAM_AUTH_ERR);
84
85 /* get list of guest logins */
86 if ((guests = openpam_get_option(pamh, "guests")) == NULL)
87 guests = DEFAULT_GUESTS;
88
89 /* check if the target account is on the list */
90 is_guest = lookup(user, guests);
91
92 /* check password */
93 if (!openpam_get_option(pamh, "nopass")) {
94 err = pam_get_authtok(pamh, PAM_AUTHTOK, &authtok, NULL);
95 if (err != PAM_SUCCESS)
96 return (err);
97 if (openpam_get_option(pamh, "pass_is_user") &&
98 strcmp(user, authtok) != 0)
99 return (PAM_AUTH_ERR);
100 if (openpam_get_option(pamh, "pass_as_ruser"))
101 pam_set_item(pamh, PAM_RUSER, authtok);
102 }
103
104 /* done */
105 if (is_guest) {
106 pam_setenv(pamh, "GUEST", user, 1);
107 return (PAM_SUCCESS);
108 }
109 return (PAM_AUTH_ERR);
110}
