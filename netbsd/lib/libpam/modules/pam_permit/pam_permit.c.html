46PAM_EXTERN int
47pam_sm_authenticate(pam_handle_t *pamh, int flags __unused,
48 int argc __unused, const char *argv[] __unused)
49{
50 const char *user;
51 int r;
52
53 if ((r = pam_get_user(pamh, &user, NULL)) != PAM_SUCCESS)
54 return (r);
55
56 return (PAM_SUCCESS);
57}
