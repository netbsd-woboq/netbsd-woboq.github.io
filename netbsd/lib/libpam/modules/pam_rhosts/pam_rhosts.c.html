58PAM_EXTERN int
59pam_sm_authenticate(pam_handle_t *pamh, int flags __unused,
60 int argc __unused, const char *argv[] __unused)
61{
62 struct passwd *pwd, pwres;
63 const char *user;
64 const void *ruser, *rhost;
65 int err, superuser;
66 char pwbuf[1024];
67
68 err = pam_get_user(pamh, &user, NULL);
69 if (err != PAM_SUCCESS)
70 return (err);
71
72 if (getpwnam_r(user, &pwres, pwbuf, sizeof(pwbuf), &pwd) != 0 ||
73 pwd == NULL)
74 return (PAM_USER_UNKNOWN);
75 if (pwd->pw_uid == 0 &&
76 openpam_get_option(pamh, OPT_ALLOW_ROOT) == NULL)
77 return (PAM_AUTH_ERR);
78
79 err = pam_get_item(pamh, PAM_RUSER, &ruser);
80 if (err != PAM_SUCCESS)
81 return (PAM_AUTH_ERR);
82
83 err = pam_get_item(pamh, PAM_RHOST, &rhost);
84 if (err != PAM_SUCCESS)
85 return (PAM_AUTH_ERR);
86
87 superuser = (strcmp(user, "root") == 0);
88 err = ruserok(rhost, superuser, ruser, user);
89 if (err != 0)
90 return (PAM_AUTH_ERR);
91
92 return (PAM_SUCCESS);
93}
