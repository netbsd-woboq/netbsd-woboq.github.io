59PAM_EXTERN int
60pam_sm_authenticate(pam_handle_t *pamh, int flags __unused,
61 int argc __unused, const char *argv[] __unused)
62{
63 struct passwd *pwd, pwres;
64 const char *luser;
65 int pam_err;
66 uid_t uid;
67 char pwbuf[1024];
68
69 pam_err = pam_get_user(pamh, &luser, NULL);
70 if (pam_err != PAM_SUCCESS)
71 return (pam_err);
72 if (luser == NULL ||
73 getpwnam_r(luser, &pwres, pwbuf, sizeof(pwbuf), &pwd) != 0 ||
74 pwd == NULL)
75 return (PAM_AUTH_ERR);
76
77 uid = getuid();
78 if (uid == 0 && !openpam_get_option(pamh, OPT_ALLOW_ROOT))
79 return (PAM_AUTH_ERR);
80
81 if (uid == (uid_t)pwd->pw_uid)
82 return (PAM_SUCCESS);
83
84 PAM_VERBOSE_ERROR("Refused; source and target users differ");
85
86 return (PAM_AUTH_ERR);
87}
