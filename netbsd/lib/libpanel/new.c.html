37static PANEL *
38_new_panel(WINDOW *w)
39{
40 PANEL *p;
41
42 p = (PANEL *)malloc(sizeof(PANEL));
43 if (__predict_false(p == NULL))
44 return NULL;
45
46 p->win = w;
47 p->user = NULL;
48
49 DECK_INSERT_TOP(p);
50 return p;
51}
52
53
54PANEL *
55new_panel(WINDOW *w)
56{
57
58 if (__predict_false(w == NULL))
59 return NULL;
60
61 if (__predict_false(w == stdscr))
62 return NULL;
63
64 /*
65 * Ensure there's phantom panel for stdscr at (below) the
66 * bottom. We explicitly re-assign stdscr in case it changed.
67 */
68 if (TAILQ_EMPTY(&_deck)) {
69 assert(PANEL_HIDDEN(&_stdscr_panel));
70
71 _stdscr_panel.win = stdscr;
72 DECK_INSERT_TOP(&_stdscr_panel);
73 }
74
75 return _new_panel(w);
76}
