31#include "panel_impl.h"

33
34int
35show_panel(PANEL *p)
36{
37
38 if (__predict_false(p == NULL))
39 return ERR;
40
41 /* unlike top_panel() */
42 if (__predict_false(!PANEL_HIDDEN(p)))
43 return ERR;
44
45 DECK_INSERT_TOP(p);
46 return OK;
47}
