38/*
39 * Interface to PCI config space registers for a specific PCI device.
40 */
41
42#include <sys/cdefs.h>
43__RCSID("$NetBSD: pci_device.c,v 1.2 2003/03/08 09:53:45 lukem Exp $");
44
45#include <sys/types.h>
46#include <sys/ioctl.h>
47
48#include <pci.h>
49
50#include <dev/pci/pciio.h>
51
52/*
53 * pcidev_conf_read:
54 *
55 * Read a config word for a given bus/device/function.
56 */
57int
58pcidev_conf_read(int fd, u_int reg, uint32_t *valp)
59{
60 struct pciio_cfgreg r;
61 int rv;
62
63 r.reg = reg;
64
65 rv = ioctl(fd, PCI_IOC_CFGREAD, &r);
66 if (rv == -1) {
67 /* errno already set. */
68 return (-1);
69 }
70
71 *valp = r.val;
72 return (0);
73}
74
75/*
76 * pcidev_conf_write:
77 *
78 * Write a config word for a given bus/device/function.
79 */
80int
81pcidev_conf_write(int fd, u_int reg, uint32_t val)
82{
83 struct pciio_cfgreg r;
84
85 r.reg = reg;
86 r.val = val;
87
88 return (ioctl(fd, PCI_IOC_CFGWRITE, &r));
89}
