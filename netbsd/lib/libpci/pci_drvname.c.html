42/*
43 * pci_drvname:
44 *
45 * What's the driver name for a PCI device?
46 */
47int
48pci_drvname(int fd, u_int device, u_int func, char *name, size_t len)
49{
50 struct pciio_drvname drvname;
51 int rv;
52
53 drvname.device = device;
54 drvname.function = func;
55
56 rv = ioctl(fd, PCI_IOC_DRVNAME, &drvname);
57 if (rv == -1)
58 return -1;
59
60 strlcpy(name, drvname.name, len);
61 return 0;
62}
63

65 * pci_drvnameonbus:
66 *
67 * What's the driver name for a PCI device on any PCI bus?

86}
