82 uint64_t ps_unique;
83 perfuse_new_msg_fn ps_new_msg;
84 perfuse_xchg_msg_fn ps_xchg_msg;
85 perfuse_destroy_msg_fn ps_destroy_msg;
86 perfuse_get_inhdr_fn ps_get_inhdr;
87 perfuse_get_inpayload_fn ps_get_inpayload;
88 perfuse_get_outhdr_fn ps_get_outhdr;

92 uint64_t ps_tracecount;
93 struct perfuse_node_hashlist *ps_nidhash;

95 int ps_nodecount;
96 int ps_xchgcount;
97};

123struct perfuse_node_data {
124 uint64_t pnd_rfh;
125 uint64_t pnd_wfh;
126 uint64_t pnd_nodeid; /* nodeid, this is not inode */
127 uint64_t pnd_parent_nodeid; /* parent's nodeid */
128 uint32_t pnd_hashidx; /* node hash index */
129 uint64_t pnd_fuse_nlookup; /* vnode refcount */
130 int pnd_puffs_nlookup; /* vnode refcount */
131 uint64_t pnd_lock_owner;
132 struct dirent *pnd_dirent; /* native buffer for readdir */
133 off_t pnd_dirent_len;
134 struct fuse_dirent *pnd_all_fd; /* FUSE buffer for readdir */
135 size_t pnd_all_fd_len;
136 uint64_t pnd_fd_cookie; /* opaque readdir ref from fs */
137 TAILQ_HEAD(,perfuse_cc_queue) pnd_pcq; /* queued requests */
138 int pnd_flags;
139#define PND_RECLAIMED 0x001 /* reclaim pending */
140#define PND_INREADDIR 0x002 /* readdir in progress */
141#define PND_DIRTY 0x004 /* There is some data to sync */
142#define PND_RFH 0x008 /* Read FH allocated */
143#define PND_WFH 0x010 /* Write FH allocated */
144#define PND_REMOVED 0x020 /* Node was removed */
145#define PND_INWRITE 0x040 /* write in progress */
146#define PND_INOPEN 0x100 /* open in progress */
147#define PND_INVALID 0x400 /* node freed, usage is a bug */
148#define PND_INRESIZE 0x800 /* resize in progress */
149
150#define PND_OPEN (PND_RFH|PND_WFH) /* File is open */
151#define PND_BUSY (PND_INREADDIR|PND_INWRITE|PND_INOPEN)
152
153 LIST_ENTRY(perfuse_node_data) pnd_nident;
154 puffs_cookie_t pnd_pn;
155 char pnd_name[MAXPATHLEN]; /* node name */
156 struct timespec pnd_cn_expire;
157 int pnd_inxchg;
158 int pnd_ref;
159};
