33__RCSID("$NetBSD: call_once.c,v 1.2 2019/04/24 21:41:15 kamil Exp $");
34
35#include <assert.h>
36#include <pthread.h>
37#include <threads.h>
38
39void
40call_once(once_flag *flag, void (*func)(void))
41{
42
43 _DIAGASSERT(flag != NULL);
44 _DIAGASSERT(func != NULL);
45
46 /*
47 * The call_once(3) function that conforms to C11 returns no value.
48 */
49 (void)pthread_once(flag, func);
50}
