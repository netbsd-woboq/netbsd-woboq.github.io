54int
55pthread_setspecific(pthread_key_t key, const void *value)
56{
57 pthread_t self;
58
59 if (__predict_false(__uselibcstub))
60 return __libc_thr_setspecific_stub(key, value);
61
62 self = pthread__self();
63 /*
64 * We can't win here on constness. Having been given a
65 * "const void *", we can only assign it to other const void *,
66 * and return it from functions that are const void *, without
67 * generating a warning.
68 */
69 return pthread__add_specific(self, key, value);
70}
71
72void *
73pthread_getspecific(pthread_key_t key)
74{
75 if (__predict_false(__uselibcstub))
76 return __libc_thr_getspecific_stub(key);
77
78 return pthread__self()->pt_specific[key].pts_value;
79}
80
81unsigned int
82pthread_curcpu_np(void)
83{
84 if (__predict_false(__uselibcstub))
85 return __libc_thr_curcpu_stub();
86
87 {
88 const int curcpu = pthread__self()->pt_lwpctl->lc_curcpu;
89
90 pthread__assert(curcpu != LWPCTL_CPU_NONE);
91 pthread__assert(curcpu != LWPCTL_CPU_EXITED);
92 pthread__assert(curcpu >= 0);
93 return curcpu;
94 }
95}
96
97/*
98 * Override setcontext so that pthread private pointer is preserved
99 */
100int
101pthread_setcontext(const ucontext_t *ucp)
102{
103#ifdef _UC_TLSBASE
104 ucontext_t uc;
105 /*
106 * Only copy and clear _UC_TLSBASE if it is set.
107 */
108 if (ucp->uc_flags & _UC_TLSBASE) {
109 uc = *ucp;
110 uc.uc_flags &= ~_UC_TLSBASE;
111 ucp = &uc;
112 }
113#endif /* _UC_TLSBASE */
114 return _sys_setcontext(ucp);
115}
