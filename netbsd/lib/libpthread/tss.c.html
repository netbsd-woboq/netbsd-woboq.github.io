32#include <sys/cdefs.h>
33__RCSID("$NetBSD: tss.c,v 1.1 2019/04/24 11:43:19 kamil Exp $");
34
35#include <assert.h>
36#include <errno.h>
37#include <pthread.h>
38#include <threads.h>
39
40int
41tss_create(tss_t *key, tss_dtor_t dtor)
42{
43
44 _DIAGASSERT(key != NULL);
45
46 if (pthread_key_create(key, dtor) == 0)
47 return thrd_success;
48
49 return thrd_error;
50}
51
52void
53tss_delete(tss_t key)
54{
55
56 _DIAGASSERT(key != NULL);
57
58 /*
59 * The tss_delete(3) function that conforms to C11 returns no value.
60 */
61 (void)pthread_key_delete(key);
62}
63
64void *
65tss_get(tss_t key)
66{
67
68 _DIAGASSERT(key != NULL);
69
70 return pthread_getspecific(key);
71}
72
73int
74tss_set(tss_t key, void *val)
75{
76
77 _DIAGASSERT(key != NULL);
78
79 if (pthread_setspecific(key, val) == 0)
80 return thrd_success;
81
82 return thrd_error;
83}
