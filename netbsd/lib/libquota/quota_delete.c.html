31#include <sys/cdefs.h>
32__RCSID("$NetBSD: quota_delete.c,v 1.4 2012/02/01 05:34:40 dholland Exp $");
33
34#include <errno.h>
35
36#include <quota.h>
37#include "quotapvt.h"
38
39int
40quota_delete(struct quotahandle *qh, const struct quotakey *qk)
41{
42 switch (qh->qh_mode) {
43 case QUOTA_MODE_NFS:
44 errno = EOPNOTSUPP;
45 return -1;
46
47 case QUOTA_MODE_OLDFILES:
48 return __quota_oldfiles_delete(qh, qk);
49
50 case QUOTA_MODE_KERNEL:
51 return __quota_kernel_delete(qh, qk);
52
53 default:
54 break;
55 }
56 errno = EINVAL;
57 return -1;
58}
