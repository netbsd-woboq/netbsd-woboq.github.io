31#include <sys/cdefs.h>
32__RCSID("$NetBSD: quota_get.c,v 1.5 2012/02/01 05:34:40 dholland Exp $");
33
34#include <errno.h>
35
36#include <quota.h>
37#include "quotapvt.h"
38
39void
40quotaval_clear(struct quotaval *qv)
41{
42 qv->qv_hardlimit = QUOTA_NOLIMIT;
43 qv->qv_softlimit = QUOTA_NOLIMIT;
44 qv->qv_usage = 0;
45 qv->qv_expiretime = QUOTA_NOTIME;
46 qv->qv_grace = QUOTA_NOTIME;
47}
48
49int
50quota_get(struct quotahandle *qh, const struct quotakey *qk, struct quotaval *qv)
51{
52 switch (qh->qh_mode) {
53 case QUOTA_MODE_NFS:
54 return __quota_nfs_get(qh, qk, qv);
55
56 case QUOTA_MODE_OLDFILES:
57 return __quota_oldfiles_get(qh, qk, qv);
58
59 case QUOTA_MODE_KERNEL:
60 return __quota_kernel_get(qh, qk, qv);
61
62 default:
63 break;
64 }
65 errno = EINVAL;
66 return -1;
67}
