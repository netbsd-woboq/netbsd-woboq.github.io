51/*
52 * Shared memory objects are supported using tmpfs.
53 */
54#define SHMFS_DIR_PATH "/var/shm"
55#define SHMFS_DIR_MODE (S_ISVTX | S_IRWXU | S_IRWXG | S_IRWXO)
56#define SHMFS_OBJ_PREFIX ".shmobj_"
57
58#define MOUNT_SHMFS MOUNT_TMPFS
59
60static bool shm_ok = false;
61
62static bool
63_shm_check_fs(void)
64{
65 int fd;
66 struct statvfs sv;
67 struct stat st;
68
69 fd = open(SHMFS_DIR_PATH, O_DIRECTORY|O_RDONLY);
70 if (fd == -1)
71 return false;
72
73 if (fstatvfs1(fd, &sv, ST_NOWAIT) == -1)
74 goto out;
75
76 if (strncmp(sv.f_fstypename, MOUNT_SHMFS, sizeof(sv.f_fstypename)))
77 goto out;
78
79 if (fstat(fd, &st) == -1)
80 goto out;
81
82 if ((st.st_mode & SHMFS_DIR_MODE) != SHMFS_DIR_MODE)
83 goto out;
84
85 shm_ok = true;
86
87out:
88 close(fd);
89 return shm_ok;
90}
91
92static bool
93_shm_get_path(char *buf, size_t len, const char *name)
94{
95 int ret;
96
97 if (__predict_false(!shm_ok) && !_shm_check_fs()) {
98 errno = ENOTSUP;
99 return false;
100 }
101
102 /*
103 * As per POSIX: the name should begin with a slash character.
104 * We may disallow other slashes (implementation-defined behaviour).
105 */
106 if (*name++ != '/' || strchr(name, '/') != NULL) {
107 errno = EINVAL;
108 return false;
109 }
110
111 ret = snprintf(buf, len, SHMFS_DIR_PATH "/" SHMFS_OBJ_PREFIX "%s",
112 name);
113
114 if ((size_t)ret >= len) {
115 errno = ENAMETOOLONG;
116 return false;
117 }
118 return ret != -1;
119}
120
121int
122shm_open(const char *name, int oflag, mode_t mode)
123{
124 char path[PATH_MAX];
125
126 if (!_shm_get_path(path, sizeof(path), name)) {
127 return -1;
128 }
129 return open(path, oflag | O_CLOEXEC | O_NOFOLLOW, mode);
130}
131
132int
133shm_unlink(const char *name)
134{
135 char path[PATH_MAX];
136
137 if (!_shm_get_path(path, sizeof(path), name)) {
138 return -1;
139 }
140 return unlink(path);
141}
