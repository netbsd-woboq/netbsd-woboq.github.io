44static char *area;
45
46int getent(char *, char *);
47char *getstr(char *, char **);
48
49/*ARGSUSED*/
50int
51getent(char *cp, char *name)
52{
53#ifdef HAS_CGETENT
54 const char *dba[2];
55
56 dba[0] = "/etc/gettytab";
57 dba[1] = 0;
58 return((cgetent(&area, dba, name) == 0) ? 1 : 0);
59#else
60 return(0);
61#endif
62}
63
64#ifndef SOLARIS
65/*ARGSUSED*/
66char *
67getstr(char *id, char **cpp)
68{
69# ifdef HAS_CGETENT
70 char *answer;
71 return((cgetstr(area, id, &answer) > 0) ? answer : 0);
72# else
73 return(0);
74# endif
75}
76#endif
