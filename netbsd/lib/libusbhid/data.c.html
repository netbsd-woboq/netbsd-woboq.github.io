36int
37hid_get_data(const void *p, const hid_item_t *h)
38{
39 const unsigned char *buf;
40 unsigned int hpos;
41 unsigned int hsize;
42 uint32_t data;
43 int i, end, offs;
44
45 _DIAGASSERT(p != NULL);
46 _DIAGASSERT(h != NULL);
47
48 buf = p;
49 hpos = h->pos; /* bit position of data */
50 hsize = h->report_size; /* bit length of data */
51
52 if (hsize == 0)
53 return (0);
54 offs = hpos / 8;
55 end = (hpos + hsize + 7) / 8 - offs;
56 data = 0;
57 for (i = 0; i < end; i++)
58 data |= buf[offs + i] << (i*8);
59 data >>= hpos % 8;
60 if (hsize < 32) {
61 data &= (1 << hsize) - 1;
62 if (h->logical_minimum < 0 && (data & (1<<(hsize-1)))) {
63 /* Need to sign extend */
64 data |= 0xffffffff & ~((1<<hsize)-1);
65 }
66 }
67 return (int)(data);
68}

70void
71hid_set_data(void *p, const hid_item_t *h, int data)
72{
73 unsigned char *buf;
74 unsigned int hpos;
75 unsigned int hsize;
76 int i, end, offs, mask;
77
78 _DIAGASSERT(p != NULL);
79 _DIAGASSERT(h != NULL);
80
81 buf = p;
82 hpos = h->pos; /* bit position of data */
83 hsize = h->report_size; /* bit length of data */
84
85 if (hsize != 32) {
86 mask = (1 << hsize) - 1;
87 data &= mask;
88 } else
89 mask = ~0;
90
91 data <<= (hpos % 8);
92 mask <<= (hpos % 8);
93 mask = ~mask;
94
95 offs = hpos / 8;
96 end = (hpos + hsize) / 8 - offs;
97
98 for (i = 0; i < end; i++)
99 buf[offs + i] = (buf[offs + i] & ((uint32_t)mask >> (i*8))) |
100 (((uint32_t)data >> (i*8)) & 0xff);
101}
