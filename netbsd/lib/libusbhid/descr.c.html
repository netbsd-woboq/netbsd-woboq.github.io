46report_desc_t
47hid_get_report_desc(int fd)
48{
49 struct usb_ctl_report_desc rep;
50
51 _DIAGASSERT(fd != -1);
52
53 rep.ucrd_size = 0;
54 if (ioctl(fd, USB_GET_REPORT_DESC, &rep) < 0)
55 return (NULL);
56
57 return hid_use_report_desc(rep.ucrd_data, (unsigned int)rep.ucrd_size);
58}
59
60report_desc_t
61hid_use_report_desc(const uint8_t *data, unsigned int size)
62{
63 report_desc_t r;
64
65 r = malloc(sizeof(*r) + size);
66 if (r == NULL) {
67 errno = ENOMEM;
68 return (NULL);
69 }
70 r->size = size;
71 memcpy(r->data, data, size);
72 return (r);
73}
74
75void
76hid_dispose_report_desc(report_desc_t r)
77{
78
79 free(r);
80}
