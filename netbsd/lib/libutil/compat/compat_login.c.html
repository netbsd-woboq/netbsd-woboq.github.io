57 "warning: reference to compatibility login(); include <util.h> to generate correct reference")
58
59void
60login(const struct utmp50 *ut50)
61{
62 int fd;
63 int tty;
64 struct utmp ut;
65
66 utmp50_to_utmp(ut50, &ut);
67
68 tty = ttyslot();
69 if (tty > 0 && (fd = open(_PATH_UTMP, O_WRONLY|O_CREAT, 0644)) >= 0) {
70 (void)lseek(fd, (off_t)(tty * sizeof(struct utmp)), SEEK_SET);
71 (void)write(fd, &ut, sizeof(ut));
72 (void)close(fd);
73 }
74 if ((fd = open(_PATH_WTMP, O_WRONLY|O_APPEND, 0)) >= 0) {
75 (void)write(fd, &ut, sizeof(ut));
76 (void)close(fd);
77 }
78}
