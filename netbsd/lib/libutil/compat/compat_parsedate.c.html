40#define __LIBC12_SOURCE__
41#include <string.h>
42#include <time.h>
43#include <sys/time.h>
44#include <compat/include/time.h>
45#include <compat/sys/time.h>
46#include <util.h>
47#include <compat/util.h>
48
49__warn_references(parsedate,
50 "warning: reference to compatibility parsedate();"
51 " include <util.h> for correct reference")
52
53int32_t
54parsedate(const char *str, const int32_t *t50, const int *tzoff)
55{
56 time_t t;
57 if (t50)
58 t = *t50;
59 return (int32_t)__parsedate50(str, t50 ? &t : NULL, tzoff);
60}
