44#include <sys/disklabel.h>

46#define SCAN_INCR 4
47
48int
49disklabel_scan(struct disklabel *lp, char *buf, size_t buflen)
50{
51 size_t i;
52
53 /* scan for the correct magic numbers. */
54
55 for (i=0; i <= buflen - sizeof(*lp); i += SCAN_INCR) {
56 memcpy(lp, buf + i, sizeof(*lp));
57 if (lp->d_magic == DISKMAGIC && lp->d_magic2 == DISKMAGIC)
58 goto sanity;
59 }
60
61 return 1;
62
63sanity:
64 /* we've found something, let's sanity check it */
65 if (lp->d_npartitions > MAXPARTITIONS || disklabel_dkcksum(lp))
66 return 1;
67
68 return 0;
69}
