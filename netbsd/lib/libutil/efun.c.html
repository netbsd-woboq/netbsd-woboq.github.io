50static void (*efunc)(int, const char *, ...) = err;
51
52void (*
53esetfunc(void (*ef)(int, const char *, ...)))(int, const char *, ...)
54{
55 void (*of)(int, const char *, ...) = efunc;
56 efunc = ef == NULL ? (void (*)(int, const char *, ...))exit : ef;
57 return of;
58}

60size_t
61estrlcpy(char *dst, const char *src, size_t len)
62{
63 size_t rv;
64 if ((rv = strlcpy(dst, src, len)) >= len) {
65 errno = ENAMETOOLONG;
66 (*efunc)(1,
67 "Cannot copy string; %zu chars needed %zu provided",
68 rv, len);
69 }
70 return rv;
71}
72
73size_t
74estrlcat(char *dst, const char *src, size_t len)
75{
76 size_t rv;
77 if ((rv = strlcat(dst, src, len)) >= len) {
78 errno = ENAMETOOLONG;
79 (*efunc)(1,
80 "Cannot append to string; %zu chars needed %zu provided",
81 rv, len);
82 }
83 return rv;
84}

86char *
87estrdup(const char *s)
88{
89 char *d = strdup(s);
90 if (d == NULL)
91 (*efunc)(1, "Cannot copy string");
92 return d;
93}
94
95char *
96estrndup(const char *s, size_t len)
97{
98 char *d = strndup(s, len);
99 if (d == NULL)
100 (*efunc)(1, "Cannot copy string");
101 return d;
102}

104void *
105emalloc(size_t n)
106{
107 void *p = malloc(n);
108 if (p == NULL && n != 0)
109 (*efunc)(1, "Cannot allocate %zu bytes", n);
110 return p;
111}
112
113void *
114ecalloc(size_t n, size_t s)
115{
116 void *p = calloc(n, s);
117 if (p == NULL && n != 0 && s != 0)
118 (*efunc)(1, "Cannot allocate %zu blocks of size %zu", n, s);
119 return p;
120}
121
122void *
123erealloc(void *p, size_t n)
124{
125 void *q = realloc(p, n);
126 if (q == NULL && n != 0)
127 (*efunc)(1, "Cannot re-allocate %zu bytes", n);
128 return q;
129}
130
131void
132ereallocarr(void *p, size_t n, size_t s)
133{
134 int rv = reallocarr(p, n, s);
135 if (rv != 0) {
136 errno = rv;
137 (*efunc)(1, "Cannot re-allocate %zu * %zu bytes", n, s);
138 }
139}

141FILE *
142efopen(const char *p, const char *m)
143{
144 FILE *fp = fopen(p, m);
145 if (fp == NULL)
146 (*efunc)(1, "Cannot open `%s'", p);
147 return fp;
148}

150int
151easprintf(char ** __restrict ret, const char * __restrict format, ...)
152{
153 int rv;
154 va_list ap;
155 va_start(ap, format);
156 if ((rv = vasprintf(ret, format, ap)) == -1)
157 (*efunc)(1, "Cannot format string");
158 va_end(ap);
159 return rv;
160}
161
162int
163evasprintf(char ** __restrict ret, const char * __restrict format, va_list ap)
164{
165 int rv;
166 if ((rv = vasprintf(ret, format, ap)) == -1)
167 (*efunc)(1, "Cannot format string");
168 return rv;
169}

171intmax_t
172estrtoi(const char * nptr, int base, intmax_t lo, intmax_t hi)
173{
174 int e;
175 intmax_t rv = strtoi(nptr, NULL, base, lo, hi, &e);
176 if (e != 0) {
177 errno = e;
178 (*efunc)(1,
179 "Cannot convert string value '%s' with base %d to a number in range [%jd .. %jd]",
180 nptr, base, lo, hi);
181 }
182 return rv;
183}
184
185uintmax_t
186estrtou(const char * nptr, int base, uintmax_t lo, uintmax_t hi)
187{
188 int e;
189 uintmax_t rv = strtou(nptr, NULL, base, lo, hi, &e);
190 if (e != 0) {
191 errno = e;
192 (*efunc)(1,
193 "Cannot convert string value '%s' with base %d to a number in range [%ju .. %ju]",
194 nptr, base, lo, hi);
195 }
196 return rv;
197}
