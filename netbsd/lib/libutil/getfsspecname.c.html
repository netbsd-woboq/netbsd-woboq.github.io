49#define COMPAT_DKWEDGE /* To be removed */
50
51const char *
52getfsspecname(char *buf, size_t bufsiz, const char *name)
53{
54 static const int mib[] = { CTL_HW, HW_DISKNAMES };
55 static const unsigned int miblen = __arraycount(mib);
56 char *drives, *dk, *p;
57 size_t len;
58 int fd, savee = errno;
59 char *vname;
60
61 p = drives = vname = NULL;
62
63 /*
64 * If the name starts with "ROOT.", replace it with "/dev/<root_device>",
65 * where <root_device> is the value of the kern.root_device sysctl. Any
66 * characters after the special "ROOT." token are appended to the end
67 * of this path.
68 */
69 if (strncasecmp(name, "ROOT.", 5) == 0 && strchr(name, ':') == NULL) {
70 static const int mib_root[] = { CTL_KERN, KERN_ROOT_DEVICE };
71 static const unsigned int mib_rootlen = __arraycount(mib_root);
72
73 strlcpy(buf, "/dev/", bufsiz);
74 len = bufsiz - 5;
75 if (sysctl(mib_root, mib_rootlen, buf + 5, &len, NULL, 0) == -1) {
76 savee = errno;
77 strlcpy(buf, "sysctl kern.root_device failed", bufsiz);
78 goto out;
79 }
80 strlcat(buf, name + 5, bufsiz);
81 return buf;
82 }
83
84 if (strncasecmp(name, "NAME=", 5) != 0) {
85#ifdef COMPAT_DKWEDGE
86 /*
87 * We try to open the disk name, and if we fail with EBUSY
88 * we use the name as the label to find the wedge.
89 */
90 char rbuf[MAXPATHLEN];
91 if (name[0] == '/') {
92 if (getdiskrawname(rbuf, sizeof(rbuf), name) != NULL) {
93 if ((fd = open(rbuf, O_RDONLY)) == -1) {
94 if (errno == EBUSY) {
95 name = strrchr(name, '/') + 1;
96 goto search;
97 }
98 } else
99 close(fd);
100 }
101 }
102#endif
103 strlcpy(buf, name, bufsiz);
104 return buf;
105 } else
106 name += 5;
107
108#ifdef COMPAT_DKWEDGE
109search:
110#endif
111 vname = malloc(strlen(name) + 1);
112 if (vname == NULL) {
113 savee = errno;
114 strlcpy(buf, "malloc failed", bufsiz);
115 goto out;
116 }
117
118 strunvis(vname, name);
119
120 if (sysctl(mib, miblen, NULL, &len, NULL, 0) == -1) {
121 savee = errno;
122 strlcpy(buf, "sysctl hw.disknames failed", bufsiz);
123 goto out;
124 }
125
126 drives = malloc(len);
127 if (drives == NULL) {
128 savee = errno;
129 strlcpy(buf, "malloc failed", bufsiz);
130 goto out;
131 }
132 if (sysctl(mib, miblen, drives, &len, NULL, 0) == -1) {
133 savee = errno;
134 strlcpy(buf, "sysctl hw.disknames failed", bufsiz);
135 goto out;
136 }
137
138 for (dk = strtok(drives, " "); dk != NULL; dk = strtok(NULL, " ")) {
139 struct dkwedge_info dkw;
140 if (strncmp(dk, "dk", 2) != 0)
141 continue;
142 fd = opendisk(dk, O_RDONLY, buf, bufsiz, 0);
143 if (fd == -1)
144 continue;
145 if (ioctl(fd, DIOCGWEDGEINFO, &dkw) == -1) {
146 savee = errno;
147 snprintf(buf, bufsiz, "%s: getwedgeinfo", dk);
148 (void)close(fd);
149 goto out;
150 }
151 (void)close(fd);
152 if (strcmp(vname, (char *)dkw.dkw_wname) == 0) {
153 p = strstr(buf, "/rdk");
154 goto good;
155 }
156 }
157#ifdef COMPAT_DKWEDGE
158 /* Last ditch effort assuming NAME=label, and label is a disk name */
159 fd = opendisk(name, O_RDONLY, buf, bufsiz, 0);
160 if (fd != -1) {
161 close(fd);
162 p = strstr(buf, "/r");
163 goto good;
164 }
165#endif
166 savee = ESRCH;
167 snprintf(buf, bufsiz, "no match for `%s'", vname);
168out:
169 buf = NULL;
170good:
171 if (p++ != NULL)
172 strcpy(p, p + 1);
173 free(drives);
174 free(vname);
175 errno = savee;
176 return buf;
177}
