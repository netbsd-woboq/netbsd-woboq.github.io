57 */
58#define TTY_LETTERS "pqrstuwxyzPQRST"
59#define TTY_OLD_SUFFIX "0123456789abcdef"
60#define TTY_NEW_SUFFIX "ghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ"
61
62int
63openpty(int *amaster, int *aslave, char *name, struct termios *term,
64 struct winsize *winp)
65{
66 char line[] = "/dev/XtyXX";
67 const char *cp1, *cp2, *cp, *linep;
68 int master, slave;
69 gid_t ttygid;
70 mode_t mode;
71 struct group grs, *grp;
72 char grbuf[1024];
73 struct ptmget pt;
74
75 _DIAGASSERT(amaster != NULL);
76 _DIAGASSERT(aslave != NULL);
77 /* name may be NULL */
78 /* term may be NULL */
79 /* winp may be NULL */
80
81 if ((master = open("/dev/ptm", O_RDWR)) != -1) {
82 if (ioctl(master, TIOCPTMGET, &pt) != -1) {
83 (void)close(master);
84 master = pt.cfd;
85 slave = pt.sfd;
86 linep = pt.sn;
87 goto gotit;
88 }
89 (void)close(master);
90 }
91
92 (void)getgrnam_r("tty", &grs, grbuf, sizeof(grbuf), &grp);
93 if (grp != NULL) {
94 ttygid = grp->gr_gid;
95 mode = S_IRUSR|S_IWUSR|S_IWGRP;
96 } else {
97 ttygid = getgid();
98 mode = S_IRUSR|S_IWUSR;
99 }
100
101 for (cp1 = TTY_LETTERS; *cp1; cp1++) {
102 line[8] = *cp1;
103 for (cp = cp2 = TTY_OLD_SUFFIX TTY_NEW_SUFFIX; *cp2; cp2++) {
104 line[5] = 'p';
105 line[9] = *cp2;
106 if ((master = open(line, O_RDWR, 0)) == -1) {
107 if (errno != ENOENT)
108 continue; /* busy */
109 if ((size_t)(cp2 - cp + 1) < sizeof(TTY_OLD_SUFFIX))
110 return -1; /* out of ptys */
111 else
112 break; /* out of ptys in this group */
113 }
114 line[5] = 't';
115 linep = line;
116 if (chown(line, getuid(), ttygid) == 0 &&
117 chmod(line, mode) == 0 &&
118 revoke(line) == 0 &&
119 (slave = open(line, O_RDWR, 0)) != -1) {
120gotit:
121 *amaster = master;
122 *aslave = slave;
123 if (name)
124 (void)strcpy(name, linep);
125 if (term)
126 (void)tcsetattr(slave, TCSAFLUSH, term);
127 if (winp)
128 (void)ioctl(slave, TIOCSWINSZ, winp);
129 return 0;
130 }
131 (void)close(master);
132 }
133 }
134 errno = ENOENT; /* out of ptys */
135 return -1;
136}

138pid_t
139forkpty(int *amaster, char *name, struct termios *term, struct winsize *winp)
140{
141 int master, slave;
142 pid_t pid;
143
144 _DIAGASSERT(amaster != NULL);
145 /* name may be NULL */
146 /* term may be NULL */
147 /* winp may be NULL */
148
149 if (openpty(&master, &slave, name, term, winp) == -1)
150 return -1;
151 switch (pid = fork()) {
152 case -1:
153 return -1;
154 case 0:
155 /*
156 * child
157 */
158 (void)close(master);
159 login_tty(slave);
160 return 0;
161 }
162 /*
163 * parent
164 */
165 *amaster = master;
166 (void)close(slave);
167 return pid;
168}
