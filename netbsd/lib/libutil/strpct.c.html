32/*
33 * Calculate a percentage without resorting to floating point
34 * and return a pointer to a string
35 *
36 * "digits" is the number of digits past the decimal place you want
37 * (zero being the straight percentage with no decimals)
38 *
39 * Erik E. Fair <fair@clock.org>, May 8, 1997
40 */
41
42#include <sys/cdefs.h>
43__RCSID("$NetBSD: strpct.c,v 1.3 2012/01/07 18:40:56 christos Exp $");
44
45#include <stdint.h>
46#include <locale.h>
47#include <limits.h>
48#include <stdio.h>
49#include <errno.h>
50#include <util.h>
51
52char *
53strspct(char *buf, size_t bufsiz, intmax_t numerator, intmax_t denominator,
54 size_t digits)
55{
56 int sign;
57
58 switch (bufsiz) {
59 case 1:
60 *buf = '\0';
61 /*FALLTHROUGH*/
62 case 0:
63 return buf;
64 default:
65 break;
66 }
67
68 if (denominator < 0) {
69 denominator = -denominator;
70 sign = 1;
71 } else
72 sign = 0;
73
74 if (numerator < 0) {
75 numerator = -numerator;
76 sign++;
77 }
78
79 sign &= 1;
80 (void)strpct(buf + sign, bufsiz - sign, (uintmax_t)numerator,
81 (uintmax_t)denominator, digits);
82 if (sign)
83 *buf = '-';
84 return buf;
85}
86
87char *
88strpct(char *buf, size_t bufsiz, uintmax_t numerator, uintmax_t denominator,
89 size_t digits)
90{
91 uintmax_t factor, result;
92 size_t u;
93
94 factor = 100;
95 for (u = 0; u < digits; u++) {
96 /* watch out for overflow! */
97 if (factor < (UINTMAX_MAX / 10))
98 factor *= 10;
99 else
100 break;
101 }
102
103 /* watch out for overflow! */
104 if (numerator < (UINTMAX_MAX / factor))
105 numerator *= factor;
106 else {
107 /* toss some of the bits of lesser significance */
108 denominator /= factor;
109 }
110
111 if (denominator == 0)
112 denominator = 1;
113
114 result = numerator / denominator;
115
116 if (digits == 0)
117 (void)snprintf(buf, bufsiz, "%ju", result);
118 else {
119 factor /= 100; /* undo initialization */
120
121 (void)snprintf(buf, bufsiz, "%ju%s%0*ju",
122 result / factor, localeconv()->decimal_point, (int)u,
123 result % factor);
124 }
125
126 return buf;
127}
