181static bozo_content_map_t *
182search_map(bozo_content_map_t *map, const char *name, size_t len)
183{
184
185 for ( ; map && map->name; map++) {
186 const size_t namelen = strlen(map->name);
187
188 if (namelen < len &&
189 strcasecmp(map->name, name + (len - namelen)) == 0)
190 return map;
191 }
192 return NULL;
193}
194
195/* match a suffix on a file - dynamiconly means no static content search */
196bozo_content_map_t *
197bozo_match_content_map(bozohttpd_t *httpd, const char *name,
198 const int dynamiconly)
199{
200 bozo_content_map_t *map;
201 size_t len;
202
203 len = strlen(name);
204 map = search_map(httpd->dynamic_content_map, name, len);
205 if (map == NULL && !dynamiconly)
206 map = search_map(static_content_map, name, len);
207
208 return map;
209}

211/*
212 * given the file name, return a valid Content-Type: value.
213 */
214/* ARGSUSED */
215const char *
216bozo_content_type(bozo_httpreq_t *request, const char *file)
217{
218 bozohttpd_t *httpd = request->hr_httpd;
219 bozo_content_map_t *map;
220
221 map = bozo_match_content_map(httpd, file, 0);
222 if (map)
223 return map->type;
224 return httpd->consts.text_plain;
225}
226
227/*
228 * given the file name, return a valid Content-Encoding: value.
229 */
230const char *
231bozo_content_encoding(bozo_httpreq_t *request, const char *file)
232{
233 bozohttpd_t *httpd = request->hr_httpd;
234 bozo_content_map_t *map;
235
236 map = bozo_match_content_map(httpd, file, 0);
237 if (map)
238 return (request->hr_proto == httpd->consts.http_11) ?
239 map->encoding11 : map->encoding;
240 return NULL;
241}

243#ifndef NO_DYNAMIC_CONTENT
244
245bozo_content_map_t *
246bozo_get_content_map(bozohttpd_t *httpd, const char *name)
247{
248 bozo_content_map_t *map;
249
250 if ((map = bozo_match_content_map(httpd, name, 1)) != NULL)
251 return map;
252
253 httpd->dynamic_content_map_size++;
254 httpd->dynamic_content_map = bozorealloc(httpd,
255 httpd->dynamic_content_map,
256 (httpd->dynamic_content_map_size + 1) * sizeof *map);
257 if (httpd->dynamic_content_map == NULL)
258 bozoerr(httpd, 1, "out of memory allocating content map");
259 map = &httpd->dynamic_content_map[httpd->dynamic_content_map_size];
260 map->name = map->type = map->encoding = map->encoding11 =
261 map->cgihandler = NULL;
262 map--;
263
264 return map;
265}

267/*
268 * mime content maps look like:
269 * ".name type encoding encoding11"
270 * where any of type, encoding or encoding11 a dash "-" means "".
271 * eg the .gtar, .tar.Z from above could be written like:
272 * ".gtar multipart/x-gtar - -"
273 * ".tar.Z multipart/x-tar x-compress compress"
274 * or
275 * ".gtar multipart/x-gtar"
276 * ".tar.Z multipart/x-tar x-compress compress"
277 * NOTE: we destroy 'arg'
278 */
279void
280bozo_add_content_map_mime(bozohttpd_t *httpd, const char *cmap0,
281 const char *cmap1, const char *cmap2, const char *cmap3)
282{
283 bozo_content_map_t *map;
284
285 debug((httpd, DEBUG_FAT,
286 "add_content_map: name %s type %s enc %s enc11 %s ",
287 cmap0, cmap1, cmap2, cmap3));
288
289 map = bozo_get_content_map(httpd, cmap0);
290#define CHECKMAP(s) (!s || ((s)[0] == '-' && (s)[1] == '\0') ? "" : (s))
291 map->name = CHECKMAP(cmap0);
292 map->type = CHECKMAP(cmap1);
293 map->encoding = CHECKMAP(cmap2);
294 map->encoding11 = CHECKMAP(cmap3);
295#undef CHECKMAP
296 map->cgihandler = NULL;
297}
298#endif /* NO_DYNAMIC_CONTENT */
