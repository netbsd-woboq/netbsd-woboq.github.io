72typedef struct sslinfo_t {
73 SSL_CTX *ssl_context;
74 const SSL_METHOD *ssl_method;
75 SSL *bozossl;
76 char *certificate_file;
77 char *privatekey_file;
78 char *ciphers;
79} sslinfo_t;

81/*
82 * bozo_clear_ssl_queue: print the contents of the SSL error queue
83 */
84static void
85bozo_clear_ssl_queue(bozohttpd_t *httpd)
86{
87 unsigned long sslcode = ERR_get_error();
88
89 do {
90 static const char sslfmt[] = "SSL Error: %s:%s:%s";
91
92 if (httpd->logstderr || isatty(STDERR_FILENO)) {
93 fprintf(stderr, sslfmt,
94 ERR_lib_error_string(sslcode),
95 ERR_func_error_string(sslcode),
96 ERR_reason_error_string(sslcode));
97 } else {
98 syslog(LOG_ERR, sslfmt,
99 ERR_lib_error_string(sslcode),
100 ERR_func_error_string(sslcode),
101 ERR_reason_error_string(sslcode));
102 }
103 } while (0 != (sslcode = ERR_get_error()));
104}

106/*
107 * bozo_ssl_warn works just like bozowarn, plus the SSL error queue
108 */
109BOZO_PRINTFLIKE(2, 3) static void
110bozo_ssl_warn(bozohttpd_t *httpd, const char *fmt, ...)
111{
112 va_list ap;
113
114 va_start(ap, fmt);
115 if (httpd->logstderr || isatty(STDERR_FILENO)) {
116 vfprintf(stderr, fmt, ap);
117 fputs("\n", stderr);
118 } else
119 vsyslog(LOG_ERR, fmt, ap);
120 va_end(ap);
121
122 bozo_clear_ssl_queue(httpd);
123}

126/*
127 * bozo_ssl_err works just like bozoerr, plus the SSL error queue
128 */
129BOZO_PRINTFLIKE(3, 4) BOZO_DEAD static void
130bozo_ssl_err(bozohttpd_t *httpd, int code, const char *fmt, ...)
131{
132 va_list ap;
133
134 va_start(ap, fmt);
135 if (httpd->logstderr || isatty(STDERR_FILENO)) {
136 vfprintf(stderr, fmt, ap);
137 fputs("\n", stderr);
138 } else
139 vsyslog(LOG_ERR, fmt, ap);
140 va_end(ap);
141
142 bozo_clear_ssl_queue(httpd);
143 exit(code);
144}

146/*
147 * bozo_check_error_queue: print warnings if the error isn't expected
148 */
149static void
150bozo_check_error_queue(bozohttpd_t *httpd, const char *tag, int ret)
151{
152 if (ret > 0)
153 return;
154
155 const sslinfo_t *sslinfo = httpd->sslinfo;
156 const int sslerr = SSL_get_error(sslinfo->bozossl, ret);
157
158 if (sslerr != SSL_ERROR_ZERO_RETURN &&
159 sslerr != SSL_ERROR_SYSCALL &&
160 sslerr != SSL_ERROR_NONE)
161 bozo_ssl_warn(httpd, "%s: SSL_ERROR %d", tag, sslerr);
162}
163
164static BOZO_PRINTFLIKE(2, 0) int
165bozo_ssl_printf(bozohttpd_t *httpd, const char * fmt, va_list ap)
166{
167 char *buf;
168 int nbytes;
169
170 if ((nbytes = vasprintf(&buf, fmt, ap)) != -1) {
171 const sslinfo_t *sslinfo = httpd->sslinfo;
172 int ret = SSL_write(sslinfo->bozossl, buf, nbytes);
173 bozo_check_error_queue(httpd, "write", ret);
174 }
175
176 free(buf);
177
178 return nbytes;
179}

181static ssize_t
182bozo_ssl_read(bozohttpd_t *httpd, int fd, void *buf, size_t nbytes)
183{
184 const sslinfo_t *sslinfo = httpd->sslinfo;
185 int ret;
186
187 USE_ARG(fd);
188 ret = SSL_read(sslinfo->bozossl, buf, (int)nbytes);
189 bozo_check_error_queue(httpd, "read", ret);

191 return (ssize_t)ret;
192}

207void
208bozo_ssl_init(bozohttpd_t *httpd)
209{
210 sslinfo_t *sslinfo = httpd->sslinfo;
211 long options;
212
213 if (sslinfo == NULL || !sslinfo->certificate_file)
214 return;
215 SSL_library_init();
216 SSL_load_error_strings();
217
218 sslinfo->ssl_method = SSLv23_server_method();
219 sslinfo->ssl_context = SSL_CTX_new(sslinfo->ssl_method);
220
221 if (NULL == sslinfo->ssl_context)
222 bozo_ssl_err(httpd, EXIT_FAILURE,
223 "SSL context creation failed");
224
225 options = SSL_CTX_set_options(sslinfo->ssl_context,
226 BOZO_SSL_OPTIONS);
227 if ((options & BOZO_SSL_OPTIONS) != BOZO_SSL_OPTIONS)
228 bozo_ssl_err(httpd, EXIT_FAILURE,
229 "Error setting ssl options requested %#lx, got %#lx",
230 BOZO_SSL_OPTIONS, options);
231
232 if (!SSL_CTX_set_cipher_list(sslinfo->ssl_context,
233 sslinfo->ciphers ? sslinfo->ciphers : BOZO_SSL_CIPHERS))
234 bozo_ssl_err(httpd, EXIT_FAILURE,
235 "Error setting cipher list '%s'", sslinfo->ciphers);
236
237 if (1 != SSL_CTX_use_certificate_chain_file(sslinfo->ssl_context,
238 sslinfo->certificate_file))
239 bozo_ssl_err(httpd, EXIT_FAILURE,
240 "Unable to use certificate file '%s'",
241 sslinfo->certificate_file);
242
243 if (1 != SSL_CTX_use_PrivateKey_file(sslinfo->ssl_context,
244 sslinfo->privatekey_file, SSL_FILETYPE_PEM))
245 bozo_ssl_err(httpd, EXIT_FAILURE,
246 "Unable to use private key file '%s'",
247 sslinfo->privatekey_file);
248
249 /* check consistency of key vs certificate */
250 if (!SSL_CTX_check_private_key(sslinfo->ssl_context))
251 bozo_ssl_err(httpd, EXIT_FAILURE,
252 "Check private key failed");
253}

255/*
256 * returns non-zero for failure
257 */
258int
259bozo_ssl_accept(bozohttpd_t *httpd)
260{
261 sslinfo_t *sslinfo = httpd->sslinfo;
262
263 if (sslinfo == NULL || !sslinfo->ssl_context)
264 return 0;
265
266 alarm(httpd->ssl_timeout);
267
268 sslinfo->bozossl = SSL_new(sslinfo->ssl_context);
269 if (sslinfo->bozossl == NULL)
270 bozoerr(httpd, 1, "SSL_new failed");
271
272 SSL_set_rfd(sslinfo->bozossl, 0);
273 SSL_set_wfd(sslinfo->bozossl, 1);
274
275 const int ret = SSL_accept(sslinfo->bozossl);
276 bozo_check_error_queue(httpd, "accept", ret);
277
278 alarm(0);
279
280 if (bozo_timeout_hit) {
281 SSL_free(sslinfo->bozossl);
282 sslinfo->bozossl = NULL;
283 return 1;
284 }
285
286 return ret != 1;
287}

289void
290bozo_ssl_destroy(bozohttpd_t *httpd)
291{
292 const sslinfo_t *sslinfo = httpd->sslinfo;
293
294 if (sslinfo && sslinfo->bozossl)
295 SSL_free(sslinfo->bozossl);
296}

298static sslinfo_t *
299bozo_get_sslinfo(bozohttpd_t *httpd)
300{
301 sslinfo_t *sslinfo;
302 if (httpd->sslinfo)
303 return httpd->sslinfo;
304 sslinfo = bozomalloc(httpd, sizeof(*sslinfo));
305 if (sslinfo == NULL)
306 bozoerr(httpd, 1, "sslinfo allocation failed");
307 memset(sslinfo, 0, sizeof(*sslinfo));
308 return httpd->sslinfo = sslinfo;
309}

311void
312bozo_ssl_set_opts(bozohttpd_t *httpd, const char *cert, const char *priv)
313{
314 sslinfo_t *sslinfo = bozo_get_sslinfo(httpd);
315
316 sslinfo->certificate_file = bozostrdup(httpd, NULL, cert);
317 sslinfo->privatekey_file = bozostrdup(httpd, NULL, priv);
318 debug((httpd, DEBUG_NORMAL, "using cert/priv files: %s & %s",
319 sslinfo->certificate_file,
320 sslinfo->privatekey_file));
321 if (!httpd->bindport)
322 httpd->bindport = bozostrdup(httpd, NULL, "https");
323}

325void
326bozo_ssl_set_ciphers(bozohttpd_t *httpd, const char *ciphers)

328 sslinfo_t *sslinfo = bozo_get_sslinfo(httpd);
329
330 sslinfo->ciphers = bozostrdup(httpd, NULL, ciphers);
331 debug((httpd, DEBUG_NORMAL, "using ciphers: %s", sslinfo->ciphers));
332}

336/*
337 * These functions are always present, so that caller code can simply
338 * use bozo_*() for IO, regardless of SSL.
339 */
340int
341bozo_printf(bozohttpd_t *httpd, const char *fmt, ...)
342{
343 va_list args;
344 int cc;
345
346 va_start(args, fmt);
347#ifndef NO_SSL_SUPPORT
348 if (httpd->sslinfo)
349 cc = bozo_ssl_printf(httpd, fmt, args);
350 else
351#endif
352 cc = vprintf(fmt, args);
353 va_end(args);
354 return cc;
355}
