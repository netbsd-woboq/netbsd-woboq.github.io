32#include <machine/asm.h>
33
34 .text
35 .align 4
36 .globl .rtld_start
37 .hidden .rtld_start
38 .type .rtld_start,@function
39.rtld_start:
40 subl $8,%esp # make room of obj_main and exit proc
41 pushl %ebx # save ps_strings
42
43 call 1f
441:
45 popl %edx
46 leal _DYNAMIC-1b(%edx),%ecx # &_DYNAMIC
47 movl %ecx,%ebx
48 subl _GLOBAL_OFFSET_TABLE_-1b(%edx),%ebx
49
50 pushl %ebx # relocbase
51 pushl %ecx # &_DYNAMIC
52 call _rtld_relocate_nonplt_self
53
54 leal 12(%esp),%eax # &cleanup
55 pushl %ebx # relocbase
56 pushl %eax # sp
57 call _rtld # _rtld(sp, relocbase)
58
59 addl $16,%esp # pop args
60
61 popl %ebx # %ebx = ps_strings
62 popl %edx # %edx = cleanup
63 popl %ecx # %ecx = obj_main
64 jmp *%eax
65
66 .align 4
67 .globl _rtld_bind_start
68 .hidden _rtld_bind_start
69 .type _rtld_bind_start,@function
70_rtld_bind_start: # (obj, reloff)
71 pushf # save registers
72 pushl %eax
73 pushl %ecx
74 pushl %edx
75
76 pushl 20(%esp) # Copy of reloff
77 pushl 20(%esp) # Copy of obj
78 call _rtld_bind # Call the binder
79 addl $8,%esp # pop binder args
80 movl %eax,20(%esp) # Store function to be called in obj
81
82 popl %edx
83 popl %ecx
84 popl %eax
85 popf
86
87 leal 4(%esp),%esp # Discard reloff, do not change eflags
88 ret
