37/*
38 * Here we define the NetBSD OS Version in an ELF .note section, structured
39 * like:
40 *
41 * [NOTE HEADER]
42 * long name size
43 * long description size
44 * long note type
45 *
46 * [NOTE DATUM]
47 * string OS name
48 *
49 * OSVERSION notes also have:
50 * long OS version (__NetBSD_Version__ constant from param.h)
51 *
52 * The DATUM fields should be padded out such that their actual (not
53 * declared) sizes % 4 == 0.
54 *
55 * These are used by the kernel to determine if this binary is really a
56 * NetBSD binary, or some other OS's.
57 */
58
59/* XXX: NetBSD 1.5 compatibility only! */
60#if __NetBSD_Version__ < 105010000
61#define ELF_NOTE_TYPE_NETBSD_TAG 1
62#endif
63
64#define __S(x) __STRING(x)
65__asm(
66 ".pushsection\t\".note.netbsd.ident\", \"a\"\n"
67 "\t.p2align\t2\n\n"
68
69 "\t.long\t" __S(ELF_NOTE_NETBSD_NAMESZ) "\n"
70 "\t.long\t" __S(ELF_NOTE_NETBSD_DESCSZ) "\n"
71 "\t.long\t" __S(ELF_NOTE_TYPE_NETBSD_TAG) "\n"
72 "\t.ascii\t" __S(ELF_NOTE_NETBSD_NAME) "\n"
73 "\t.long\t" __S(__NetBSD_Version__) "\n\n"
74
75 "\t.popsection\n"
76);
77
78__asm(
79 ".pushsection\t\".note.netbsd.pax\", \"a\"\n"
80 "\t.p2align\t2\n\n"
81
82 "\t.long\t" __S(ELF_NOTE_PAX_NAMESZ) "\n"
83 "\t.long\t" __S(ELF_NOTE_PAX_DESCSZ) "\n"
84 "\t.long\t" __S(ELF_NOTE_TYPE_PAX_TAG) "\n"
85 "\t.ascii\t" __S(ELF_NOTE_PAX_NAME) "\n"
86 "\t.long\t" __S(0) "\n\n"
87
88 "\t.popsection\n"
89);
90
91#ifdef ELF_NOTE_MARCH_DESC
92__asm(
93 ".pushsection\t\".note.netbsd.march\", \"a\"\n"
94 "\t.p2align\t2\n\n"
95
96 "\t.long\t" __S(ELF_NOTE_MARCH_NAMESZ) "\n"
97 "\t.long\t2f-1f\n"
98 "\t.long\t" __S(ELF_NOTE_TYPE_MARCH_TAG) "\n"
99 "\t.ascii\t" __S(ELF_NOTE_MARCH_NAME) "\n"
100 "1:\t.asciz\t" __S(ELF_NOTE_MARCH_DESC) "\n"
101 "2:\n"
102 "\t.p2align\t2\n\n"
103
104 "\t.popsection\n"
105);
106#endif
