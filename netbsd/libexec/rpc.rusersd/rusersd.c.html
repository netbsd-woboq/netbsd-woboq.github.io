36#include <stdio.h>
37#include <rpc/rpc.h>
38#include <sys/socket.h>
39#include <signal.h>
40#include <syslog.h>
41#include <stdlib.h>
42#include <rpcsvc/rusers.h> /* New version */
43#include <rpcsvc/rnusers.h> /* Old version */
44
45#include "rusers_proc.h"
46
47
48int from_inetd = 1;
49
50static void cleanup(int) __dead;
51
52static void
53/*ARGSUSED*/
54cleanup(int n)
55{
56
57 (void) rpcb_unset(RUSERSPROG, RUSERSVERS_3, NULL);
58 (void) rpcb_unset(RUSERSPROG, RUSERSVERS_IDLE, NULL);
59 exit(0);
60}
61
62int
63/*ARGSUSED*/
64main(int argc, char *argv[])
65{
66 SVCXPRT *transp;
67 struct sockaddr_storage from;
68 socklen_t fromlen;
69
70 /*
71 * See if inetd started us
72 */
73 fromlen = sizeof(from);
74 if (getsockname(0, (struct sockaddr *)(void *)&from, &fromlen) == -1)
75 from_inetd = 0;
76
77 if (!from_inetd) {
78 (void)daemon(0, 0);
79
80 (void)rpcb_unset(RUSERSPROG, RUSERSVERS_3, NULL);
81 (void)rpcb_unset(RUSERSPROG, RUSERSVERS_IDLE, NULL);
82
83 (void)signal(SIGINT, cleanup);
84 (void)signal(SIGTERM, cleanup);
85 (void)signal(SIGHUP, cleanup);
86 }
87
88 openlog("rpc.rusersd", LOG_PID, LOG_DAEMON);
89
90 if (from_inetd) {
91 transp = svc_dg_create(0, 0, 0);
92 if (transp == NULL) {
93 syslog(LOG_ERR, "cannot create udp service.");
94 return 1;
95 }
96 if (!svc_reg(transp, RUSERSPROG, RUSERSVERS_3, rusers_service,
97 NULL)) {
98 syslog(LOG_ERR, "unable to register "
99 "(RUSERSPROG, RUSERSVERS_3).");
100 return 1;
101 }
102 if (!svc_reg(transp, RUSERSPROG, RUSERSVERS_IDLE,
103 rusers_service, NULL)) {
104 syslog(LOG_ERR, "unable to register "
105 "(RUSERSPROG, RUSERSVERS_IDLE).");
106 return 1;
107 }
108 } else {
109 if (!svc_create(rusers_service, RUSERSPROG, RUSERSVERS_3,
110 "udp")) {
111 syslog(LOG_ERR, "unable to create "
112 "(RUSERSPROG, RUSERSVERS_3).");
113 return 1;
114 }
115 if (!svc_create(rusers_service, RUSERSPROG, RUSERSVERS_IDLE,
116 "udp")) {
117 syslog(LOG_ERR, "unable to create "
118 "(RUSERSPROG, RUSERSVERS_IDLE).");
119 return 1;
120 }
121 }
122
123 svc_run();
124 syslog(LOG_ERR, "svc_run returned");
125 return 1;
126}
