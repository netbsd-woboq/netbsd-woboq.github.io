41/* debug/logging print routines */
42
43#include <sys/types.h>
44#include <sys/socket.h>
45#include <protocols/talkd.h>
46#include <inttypes.h>
47#include <syslog.h>
48#include <stdio.h>
49#include "extern.h"
50
51static const char *types[] = {
52 "leave_invite",
53 "look_up",
54 "delete",
55 "announce"
56};
57#define NTYPES (sizeof (types) / sizeof (types[0]))
58
59static const char *answers[] = {
60 "success",
61 "not_here",
62 "failed",
63 "machine_unknown",
64 "permission_denied",
65 "unknown_request",
66 "badversion",
67 "badaddr",
68 "badctladdr"
69};
70#define NANSWERS (sizeof (answers) / sizeof (answers[0]))
71
72void
73print_request(const char *cp, CTL_MSG *mp)
74{
75 char tbuf[80];
76 const char *tp;
77
78 if (mp->type >= NTYPES) {
79 (void)snprintf(tbuf, sizeof tbuf, "type %d", mp->type);
80 tp = tbuf;
81 } else
82 tp = types[mp->type];
83 syslog(debug ? LOG_DEBUG : LOG_INFO,
84 "%s: %s: id %"PRIu32", l_user %s, r_user %s, r_tty %s",
85 cp, tp, mp->id_num, mp->l_name, mp->r_name, mp->r_tty);
86}
87
88void
89print_response(const char *cp, CTL_RESPONSE *rp)
90{
91 char tbuf[80], abuf[80];
92 const char *tp, *ap;
93
94 if (rp->type >= NTYPES) {
95 (void)snprintf(tbuf, sizeof tbuf, "type %d", rp->type);
96 tp = tbuf;
97 } else
98 tp = types[rp->type];
99 if (rp->answer >= NANSWERS) {
100 (void)snprintf(abuf, sizeof abuf, "answer %d", rp->answer);
101 ap = abuf;
102 } else
103 ap = answers[rp->answer];
104 syslog(debug ? LOG_DEBUG : LOG_INFO, "%s: %s: %s, id %"PRIu32,
105 cp, tp, ap, ntohl(rp->id_num));
106}
