66/*
67 * init_termbuf()
68 * copy_termbuf(cp)
69 * set_termbuf()
70 *
71 * These three routines are used to get and set the "termbuf" structure
72 * to and from the kernel. init_termbuf() gets the current settings.
73 * copy_termbuf() hands in a new "termbuf" to write to the kernel, and
74 * set_termbuf() writes the structure into the kernel.
75 */
76
77void
78init_termbuf(void)
79{
80 (void) tcgetattr(pty, &termbuf);
81 termbuf2 = termbuf;
82}
83
84#if defined(LINEMODE) && defined(TIOCPKT_IOCTL)
85void
86copy_termbuf(char *cp, int len)
87{
88 if ((size_t)len > sizeof(termbuf))
89 len = sizeof(termbuf);
90 memmove((char *)&termbuf, cp, len);
91 termbuf2 = termbuf;
92}
93#endif /* defined(LINEMODE) && defined(TIOCPKT_IOCTL) */
94
95void
96set_termbuf(void)
97{
98 /*
99 * Only make the necessary changes.
100 */
101 if (memcmp((char *)&termbuf, (char *)&termbuf2, sizeof(termbuf)))
102 (void) tcsetattr(pty, TCSANOW, &termbuf);
103}

106/*
107 * spcset(func, valp, valpp)
108 *
109 * This function takes various special characters (func), and
110 * sets *valp to the current value of that character, and
111 * *valpp to point to where in the "termbuf" structure that
112 * value is kept.
113 *
114 * It returns the SLC_ level of support for this function.
115 */
116
117
118int
119spcset(int func, cc_t *valp, cc_t **valpp)
120{
121
122#define setval(a, b) *valp = termbuf.c_cc[a]; \
123 *valpp = &termbuf.c_cc[a]; \
124 return(b);
125#define defval(a) *valp = ((cc_t)a); *valpp = (cc_t *)0; return(SLC_DEFAULT);
126
127 switch(func) {
128 case SLC_EOF:
129 setval(VEOF, SLC_VARIABLE);
130 case SLC_EC:
131 setval(VERASE, SLC_VARIABLE);
132 case SLC_EL:
133 setval(VKILL, SLC_VARIABLE);
134 case SLC_IP:
135 setval(VINTR, SLC_VARIABLE|SLC_FLUSHIN|SLC_FLUSHOUT);
136 case SLC_ABORT:
137 setval(VQUIT, SLC_VARIABLE|SLC_FLUSHIN|SLC_FLUSHOUT);
138 case SLC_XON:
139 setval(VSTART, SLC_VARIABLE);
140 case SLC_XOFF:
141 setval(VSTOP, SLC_VARIABLE);
142 case SLC_EW:
143 setval(VWERASE, SLC_VARIABLE);
144 case SLC_RP:
145 setval(VREPRINT, SLC_VARIABLE);
146 case SLC_LNEXT:
147 setval(VLNEXT, SLC_VARIABLE);
148 case SLC_AO:
149 setval(VDISCARD, SLC_VARIABLE|SLC_FLUSHOUT);
150 case SLC_SUSP:
151 setval(VSUSP, SLC_VARIABLE|SLC_FLUSHIN);
152 case SLC_FORW1:
153 setval(VEOL, SLC_VARIABLE);
154 case SLC_FORW2:
155 setval(VEOL2, SLC_VARIABLE);
156 case SLC_AYT:
157 setval(VSTATUS, SLC_VARIABLE);
158
159 case SLC_BRK:
160 case SLC_SYNCH:
161 case SLC_EOR:
162 defval(0);
163
164 default:
165 *valp = 0;
166 *valpp = 0;
167 return(SLC_NOSUPPORT);
168 }
169}

352void
353tty_setsofttab(int on)
354{
355 if (on) {
356# ifdef OXTABS
357 termbuf.c_oflag |= OXTABS;
358# endif
359# ifdef TABDLY
360 termbuf.c_oflag &= ~TABDLY;

362# endif

364# ifdef OXTABS
365 termbuf.c_oflag &= ~OXTABS;

369 termbuf.c_oflag |= TAB0;

426/*
427 * getptyslave()
428 *
429 * Open the slave side of the pty, and do any initialization
430 * that is necessary. The return value is a file descriptor
431 * for the slave side.
432 */
433extern int def_tspeed, def_rspeed;
434 extern int def_row, def_col;
435
436void
437getptyslave(void)
438{
439 int t = -1;
440
441#ifdef LINEMODE
442 int waslm;
443#endif
444 struct winsize ws;
445 /*
446 * Opening the slave side may cause initilization of the
447 * kernel tty structure. We need remember the state of
448 * if linemode was turned on
449 * terminal window size
450 * terminal speed
451 * so that we can re-set them if we need to.
452 */
453#ifdef LINEMODE
454 waslm = tty_linemode();
455#endif
456
457 /*
458 * Make sure that we don't have a controlling tty, and
459 * that we are the session (process group) leader.
460 */
461 t = open(_PATH_TTY, O_RDWR);
462 if (t >= 0) {
463 (void) ioctl(t, TIOCNOTTY, (char *)0);
464 (void) close(t);
465 }
466
467
468
469 t = cleanopen(line);
470 if (t < 0)
471 fatalperror(net, line);
472
473
474 /*
475 * set up the tty modes as we like them to be.
476 */
477 init_termbuf();
478 if (def_row || def_col) {
479 memset((char *)&ws, 0, sizeof(ws));
480 ws.ws_col = def_col;
481 ws.ws_row = def_row;
482 (void)ioctl(t, TIOCSWINSZ, (char *)&ws);
483 }
484
485 /*
486 * Settings for sgtty based systems
487 */
488
489 /*
490 * Settings for all other termios/termio based
491 * systems, other than 4.4BSD. In 4.4BSD the
492 * kernel does the initial terminal setup.
493 */
494 tty_rspeed((def_rspeed > 0) ? def_rspeed : 9600);
495 tty_tspeed((def_tspeed > 0) ? def_tspeed : 9600);
496#ifdef LINEMODE
497 if (waslm)
498 tty_setlinemode(1);
499#endif /* LINEMODE */
500
501 /*
502 * Set the tty modes, and make this our controlling tty.
503 */
504 set_termbuf();
505 if (login_tty(t) == -1)
506 fatalperror(net, "login_tty");
507 if (net > 2)
508 (void) close(net);
509 if (pty > 2) {
510 (void) close(pty);
511 pty = -1;
512 }
513}

574/*
575 * start_login(host)
576 *
577 * Assuming that we are now running as a child processes, this
578 * function will turn us into the login process.
579 */
580extern char *gettyname;
581
582void
583start_login(char *host, int autologin, char *name)
584{
585 char **argv;
586#define TABBUFSIZ 512
587 char defent[TABBUFSIZ];
588 char defstrs[TABBUFSIZ];
589#undef TABBUFSIZ
590 const char *loginprog = NULL;
591 extern struct sockaddr_storage from;
592 char buf[sizeof(from) * 4 + 1];
593
594 scrub_env();
595
596 /*
597 * -a : pass on the address of the host.
598 * -h : pass on name of host.
599 * WARNING: -h and -a are accepted by login
600 * if and only if getuid() == 0.
601 * -p : don't clobber the environment (so terminal type stays set).
602 *
603 * -f : force this login, he has already been authenticated
604 */
605 argv = addarg(0, "login");
606
607 argv = addarg(argv, "-a");
608 (void)strvisx(buf, (const char *)(const void *)&from, sizeof(from),
609 VIS_WHITE);
610 argv = addarg(argv, buf);
611
612 argv = addarg(argv, "-h");
613 argv = addarg(argv, host);
614
615 argv = addarg(argv, "-p");
616#ifdef LINEMODE
617 /*
618 * Set the environment variable "LINEMODE" to either
619 * "real" or "kludge" if we are operating in either
620 * real or kludge linemode.
621 */
622 if (lmodetype == REAL_LINEMODE)
623 setenv("LINEMODE", "real", 1);
624# ifdef KLUDGELINEMODE
625 else if (lmodetype == KLUDGE_LINEMODE || lmodetype == KLUDGE_OK)
626 setenv("LINEMODE", "kludge", 1);
627# endif
628#endif
629#ifdef SECURELOGIN
630 /*
631 * don't worry about the -f that might get sent.
632 * A -s is supposed to override it anyhow.
633 */
634 if (require_secure_login)
635 argv = addarg(argv, "-s");
636#endif
637#ifdef AUTHENTICATION
638 if (auth_level >= 0 && autologin == AUTH_VALID) {
639 argv = addarg(argv, "-f");
640 argv = addarg(argv, "--");
641 argv = addarg(argv, name);
642 } else
643#endif
644 if (getenv("USER")) {
645 argv = addarg(argv, "--");
646 argv = addarg(argv, getenv("USER"));
647 /*
648 * Assume that login will set the USER variable
649 * correctly. For SysV systems, this means that
650 * USER will no longer be set, just LOGNAME by
651 * login. (The problem is that if the auto-login
652 * fails, and the user then specifies a different
653 * account name, he can get logged in with both
654 * LOGNAME and USER in his environment, but the
655 * USER value will be wrong.
656 */
657 unsetenv("USER");
658 }
659 if (getent(defent, gettyname) == 1) {
660 char *cp = defstrs;
661
662 loginprog = getstr("lo", &cp);
663 }
664 if (loginprog == NULL)
665 loginprog = _PATH_LOGIN;
666 closelog();
667 /*
668 * This sleep(1) is in here so that telnetd can
669 * finish up with the tty. There's a race condition
670 * the login banner message gets lost...
671 */
672 sleep(1);
673 execv(loginprog, argv);
674
675 syslog(LOG_ERR, "%s: %m", loginprog);
676 fatalperror(net, loginprog);
677 /*NOTREACHED*/
678}

680char **
681addarg(char **argv, const char *val)
682{
683 char **cpp;
684 char **nargv;
685
686 if (argv == NULL) {
687 /*
688 * 10 entries, a leading length, and a null
689 */
690 argv = malloc(sizeof(*argv) * 12);
691 if (argv == NULL)
692 return(NULL);
693 *argv++ = (char *)10;
694 *argv = (char *)0;
695 }
696 for (cpp = argv; *cpp; cpp++)
697 ;
698 if (cpp == &argv[(long)argv[-1]]) {
699 --argv;
700 nargv = realloc(argv, sizeof(*argv) * ((long)(*argv) + 10 + 2));
701 if (nargv == NULL) {
702 fatal(net, "not enough memory");
703 /*NOTREACHED*/
704 }
705 argv = nargv;
706 *argv = (char *)((long)(*argv) + 10);
707 argv++;
708 cpp = &argv[(long)argv[-1] - 10];
709 }
710 *cpp++ = __UNCONST(val);
711 *cpp = 0;
712 return(argv);
713}

715/*
716 * scrub_env()
717 *
718 * We only accept the environment variables listed below.
719 */
720
721void
722scrub_env(void)
723{
724 static const char *reject[] = {
725 "TERMCAP=/",
726 NULL
727 };
728
729 static const char *acceptstr[] = {
730 "XAUTH=", "XAUTHORITY=", "DISPLAY=",
731 "TERM=",
732 "EDITOR=",
733 "PAGER=",
734 "LOGNAME=",
735 "POSIXLY_CORRECT=",
736 "TERMCAP=",
737 "PRINTER=",
738 NULL
739 };
740
741 char **cpp, **cpp2;
742 const char **p;
743
744 for (cpp2 = cpp = environ; *cpp; cpp++) {
745 int reject_it = 0;
746
747 for(p = reject; *p; p++)
748 if(strncmp(*cpp, *p, strlen(*p)) == 0) {
749 reject_it = 1;
750 break;
751 }
752 if (reject_it)
753 continue;
754
755 for(p = acceptstr; *p; p++)
756 if(strncmp(*cpp, *p, strlen(*p)) == 0)
757 break;
758 if(*p != NULL)
759 *cpp2++ = *cpp;
760 }
761 *cpp2 = NULL;
762}

764/*
765 * cleanup()
766 *
767 * This is the routine to call when we are all through, to
768 * clean up anything that needs to be cleaned up.
769 */
770/* ARGSUSED */
771void
772cleanup(int sig)
773{
774 char *p, c;
775
776 p = line + sizeof(_PATH_DEV) - 1;
777#ifdef SUPPORT_UTMP
778 if (logout(p))
779 logwtmp(p, "", "");
780#endif
781#ifdef SUPPORT_UTMPX
782 if (logoutx(p, 0, DEAD_PROCESS))
783 logwtmpx(p, "", "", 0, DEAD_PROCESS);
784#endif
785 (void)chmod(line, 0666);
786 (void)chown(line, 0, 0);
787 c = *p; *p = 'p';
788 (void)chmod(line, 0666);
789 (void)chown(line, 0, 0);
790 *p = c;
791 if (ttyaction(line, "telnetd", "root"))
792 syslog(LOG_ERR, "%s: ttyaction failed", line);
793 (void) shutdown(net, 2);
794 exit(1);
795}
