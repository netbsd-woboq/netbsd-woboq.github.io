601#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
602 newps = (YYLTYPE *)realloc(data->p_base, newsize * sizeof(*newps));
603 if (newps == 0)
604 return YYENOMEM;
605
606 data->p_base = newps;
607 data->p_mark = newps + i;
608#endif
609
610 data->stacksize = newsize;
611 data->s_last = data->s_base + newsize - 1;
612
613#if YYDEBUG
614 if (yydebug)
615 fprintf(stderr, "%sdebug: stack size increased to %d\n", YYPREFIX, newsize);
616#endif
617 return 0;
618}

674#define YYREJECT goto yyabort
675#define YYACCEPT goto yyaccept
676#define YYERROR goto yyerrlab

682
683int
684YYPARSE_DECL()
685{
686 int yym, yyn, yystate, yyresult;
687#if YYBTYACC
688 int yynewerrflag;
689 YYParseState *yyerrctx = NULL;
690#endif /* YYBTYACC */
691#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
692 YYLTYPE yyerror_loc_range[3]; /* position of error start/end (0 unused) */
693#endif
694#if YYDEBUG
695 const char *yys;
696
697 if ((yys = getenv("YYDEBUG")) != 0)
698 {
699 yyn = *yys;
700 if (yyn >= '0' && yyn <= '9')
701 yydebug = yyn - '0';
702 }
703 if (yydebug)
704 fprintf(stderr, "%sdebug[<# of symbols on state stack>]\n", YYPREFIX);
705#endif
706#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
707 memset(yyerror_loc_range, 0, sizeof(yyerror_loc_range));
708#endif
709
710#if YYBTYACC
711 yyps = yyNewState(0); if (yyps == 0) goto yyenomem;
712 yyps->save = 0;
713#endif /* YYBTYACC */
714 yym = 0;
715 yyn = 0;
716 yynerrs = 0;
717 yyerrflag = 0;
718 yychar = YYEMPTY;
719 yystate = 0;
720
721#if YYPURE
722 memset(&yystack, 0, sizeof(yystack));
723#endif
724
725 if (yystack.s_base == NULL && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
726 yystack.s_mark = yystack.s_base;
727 yystack.l_mark = yystack.l_base;
728#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
729 yystack.p_mark = yystack.p_base;
730#endif
731 yystate = 0;
732 *yystack.s_mark = 0;
733
734yyloop:
735 if ((yyn = yydefred[yystate]) != 0) goto yyreduce;
736 if (yychar < 0)
737 {
738#if YYBTYACC
739 do {
740 if (yylvp < yylve)
741 {
742 /* we're currently re-reading tokens */
743 yylval = *yylvp++;
744#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
745 yylloc = *yylpp++;
746#endif
747 yychar = *yylexp++;
748 break;
749 }
750 if (yyps->save)
751 {
752 /* in trial mode; save scanner results for future parse attempts */
753 if (yylvp == yylvlim)
754 { /* Enlarge lexical value queue */
755 size_t p = (size_t) (yylvp - yylvals);
756 size_t s = (size_t) (yylvlim - yylvals);
757
758 s += YYLVQUEUEGROWTH;
759 if ((yylexemes = (YYINT *)realloc(yylexemes, s * sizeof(YYINT))) == NULL) goto yyenomem;
760 if ((yylvals = (YYSTYPE *)realloc(yylvals, s * sizeof(YYSTYPE))) == NULL) goto yyenomem;
761#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
762 if ((yylpsns = (YYLTYPE *)realloc(yylpsns, s * sizeof(YYLTYPE))) == NULL) goto yyenomem;
763#endif
764 yylvp = yylve = yylvals + p;
765 yylvlim = yylvals + s;
766#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
767 yylpp = yylpe = yylpsns + p;
768 yylplim = yylpsns + s;
769#endif
770 yylexp = yylexemes + p;
771 }
772 *yylexp = (YYINT) YYLEX;
773 *yylvp++ = yylval;
774 yylve++;
775#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
776 *yylpp++ = yylloc;
777 yylpe++;
778#endif
779 yychar = *yylexp++;
780 break;
781 }
782 /* normal operation, no conflict encountered */
783#endif /* YYBTYACC */
784 yychar = YYLEX;
785#if YYBTYACC
786 } while (0);
787#endif /* YYBTYACC */
788 if (yychar < 0) yychar = YYEOF;
789#if YYDEBUG
790 if (yydebug)
791 {
792 if ((yys = yyname[YYTRANSLATE(yychar)]) == NULL) yys = yyname[YYUNDFTOKEN];
793 fprintf(stderr, "%s[%d]: state %d, reading token %d (%s)",
794 YYDEBUGSTR, yydepth, yystate, yychar, yys);
795#ifdef YYSTYPE_TOSTRING
796#if YYBTYACC
797 if (!yytrial)
798#endif /* YYBTYACC */
799 fprintf(stderr, " <%s>", YYSTYPE_TOSTRING(yychar, yylval));
800#endif
801 fputc('\n', stderr);
802 }
803#endif
804 }
805#if YYBTYACC
806
807 /* Do we have a conflict? */
808 if (((yyn = yycindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
809 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
810 {
811 YYINT ctry;
812
813 if (yypath)
814 {
815 YYParseState *save;
816#if YYDEBUG
817 if (yydebug)
818 fprintf(stderr, "%s[%d]: CONFLICT in state %d: following successful trial parse\n",
819 YYDEBUGSTR, yydepth, yystate);
820#endif
821 /* Switch to the next conflict context */
822 save = yypath;
823 yypath = save->save;
824 save->save = NULL;
825 ctry = save->ctry;
826 if (save->state != yystate) YYABORT;
827 yyFreeState(save);
828
829 }
830 else
831 {
832
833 /* Unresolved conflict - start/continue trial parse */
834 YYParseState *save;
835#if YYDEBUG
836 if (yydebug)
837 {
838 fprintf(stderr, "%s[%d]: CONFLICT in state %d. ", YYDEBUGSTR, yydepth, yystate);
839 if (yyps->save)
840 fputs("ALREADY in conflict, continuing trial parse.\n", stderr);
841 else
842 fputs("Starting trial parse.\n", stderr);
843 }
844#endif
845 save = yyNewState((unsigned)(yystack.s_mark - yystack.s_base + 1));
846 if (save == NULL) goto yyenomem;
847 save->save = yyps->save;
848 save->state = yystate;
849 save->errflag = yyerrflag;
850 save->yystack.s_mark = save->yystack.s_base + (yystack.s_mark - yystack.s_base);
851 memcpy (save->yystack.s_base, yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
852 save->yystack.l_mark = save->yystack.l_base + (yystack.l_mark - yystack.l_base);
853 memcpy (save->yystack.l_base, yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
854#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
855 save->yystack.p_mark = save->yystack.p_base + (yystack.p_mark - yystack.p_base);
856 memcpy (save->yystack.p_base, yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
857#endif
858 ctry = yytable[yyn];
859 if (yyctable[ctry] == -1)
860 {
861#if YYDEBUG
862 if (yydebug && yychar >= YYEOF)
863 fprintf(stderr, "%s[%d]: backtracking 1 token\n", YYDEBUGSTR, yydepth);
864#endif
865 ctry++;
866 }
867 save->ctry = ctry;
868 if (yyps->save == NULL)
869 {
870 /* If this is a first conflict in the stack, start saving lexemes */
871 if (!yylexemes)
872 {
873 yylexemes = (YYINT *) malloc((YYLVQUEUEGROWTH) * sizeof(YYINT));
874 if (yylexemes == NULL) goto yyenomem;
875 yylvals = (YYSTYPE *) malloc((YYLVQUEUEGROWTH) * sizeof(YYSTYPE));
876 if (yylvals == NULL) goto yyenomem;
877 yylvlim = yylvals + YYLVQUEUEGROWTH;
878#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
879 yylpsns = (YYLTYPE *) malloc((YYLVQUEUEGROWTH) * sizeof(YYLTYPE));
880 if (yylpsns == NULL) goto yyenomem;
881 yylplim = yylpsns + YYLVQUEUEGROWTH;
882#endif
883 }
884 if (yylvp == yylve)
885 {
886 yylvp = yylve = yylvals;
887#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
888 yylpp = yylpe = yylpsns;
889#endif
890 yylexp = yylexemes;
891 if (yychar >= YYEOF)
892 {
893 *yylve++ = yylval;
894#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
895 *yylpe++ = yylloc;
896#endif
897 *yylexp = (YYINT) yychar;
898 yychar = YYEMPTY;
899 }
900 }
901 }
902 if (yychar >= YYEOF)
903 {
904 yylvp--;
905#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
906 yylpp--;
907#endif
908 yylexp--;
909 yychar = YYEMPTY;
910 }
911 save->lexeme = (int) (yylvp - yylvals);
912 yyps->save = save;
913 }
914 if (yytable[yyn] == ctry)
915 {
916#if YYDEBUG
917 if (yydebug)
918 fprintf(stderr, "%s[%d]: state %d, shifting to state %d\n",
919 YYDEBUGSTR, yydepth, yystate, yyctable[ctry]);
920#endif
921 if (yychar < 0)
922 {
923 yylvp++;
924#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
925 yylpp++;
926#endif
927 yylexp++;
928 }
929 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM)
930 goto yyoverflow;
931 yystate = yyctable[ctry];
932 *++yystack.s_mark = (YYINT) yystate;
933 *++yystack.l_mark = yylval;
934#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
935 *++yystack.p_mark = yylloc;
936#endif
937 yychar = YYEMPTY;
938 if (yyerrflag > 0) --yyerrflag;
939 goto yyloop;
940 }
941 else
942 {
943 yyn = yyctable[ctry];
944 goto yyreduce;
945 }
946 } /* End of code dealing with conflicts */
947#endif /* YYBTYACC */
948 if (((yyn = yysindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
949 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
950 {
951#if YYDEBUG
952 if (yydebug)
953 fprintf(stderr, "%s[%d]: state %d, shifting to state %d\n",
954 YYDEBUGSTR, yydepth, yystate, yytable[yyn]);
955#endif
956 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
957 yystate = yytable[yyn];
958 *++yystack.s_mark = yytable[yyn];
959 *++yystack.l_mark = yylval;
960#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
961 *++yystack.p_mark = yylloc;
962#endif
963 yychar = YYEMPTY;
964 if (yyerrflag > 0) --yyerrflag;
965 goto yyloop;
966 }
967 if (((yyn = yyrindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
968 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
969 {
970 yyn = yytable[yyn];
971 goto yyreduce;
972 }
973 if (yyerrflag != 0) goto yyinrecovery;
974#if YYBTYACC
975
976 yynewerrflag = 1;
977 goto yyerrhandler;
978 goto yyerrlab; /* redundant goto avoids 'unused label' warning */
979
980yyerrlab:
981 /* explicit YYERROR from an action -- pop the rhs of the rule reduced
982 * before looking for error recovery */
983 yystack.s_mark -= yym;
984 yystate = *yystack.s_mark;
985 yystack.l_mark -= yym;
986#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
987 yystack.p_mark -= yym;
988#endif
989
990 yynewerrflag = 0;
991yyerrhandler:
992 while (yyps->save)
993 {
994 int ctry;
995 YYParseState *save = yyps->save;
996#if YYDEBUG
997 if (yydebug)
998 fprintf(stderr, "%s[%d]: ERROR in state %d, CONFLICT BACKTRACKING to state %d, %d tokens\n",
999 YYDEBUGSTR, yydepth, yystate, yyps->save->state,
1000 (int)(yylvp - yylvals - yyps->save->lexeme));
1001#endif
1002 /* Memorize most forward-looking error state in case it's really an error. */
1003 if (yyerrctx == NULL || yyerrctx->lexeme < yylvp - yylvals)
1004 {
1005 /* Free old saved error context state */
1006 if (yyerrctx) yyFreeState(yyerrctx);
1007 /* Create and fill out new saved error context state */
1008 yyerrctx = yyNewState((unsigned)(yystack.s_mark - yystack.s_base + 1));
1009 if (yyerrctx == NULL) goto yyenomem;
1010 yyerrctx->save = yyps->save;
1011 yyerrctx->state = yystate;
1012 yyerrctx->errflag = yyerrflag;
1013 yyerrctx->yystack.s_mark = yyerrctx->yystack.s_base + (yystack.s_mark - yystack.s_base);
1014 memcpy (yyerrctx->yystack.s_base, yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1015 yyerrctx->yystack.l_mark = yyerrctx->yystack.l_base + (yystack.l_mark - yystack.l_base);
1016 memcpy (yyerrctx->yystack.l_base, yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1017#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1018 yyerrctx->yystack.p_mark = yyerrctx->yystack.p_base + (yystack.p_mark - yystack.p_base);
1019 memcpy (yyerrctx->yystack.p_base, yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1020#endif
1021 yyerrctx->lexeme = (int) (yylvp - yylvals);
1022 }
1023 yylvp = yylvals + save->lexeme;
1024#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1025 yylpp = yylpsns + save->lexeme;
1026#endif
1027 yylexp = yylexemes + save->lexeme;
1028 yychar = YYEMPTY;
1029 yystack.s_mark = yystack.s_base + (save->yystack.s_mark - save->yystack.s_base);
1030 memcpy (yystack.s_base, save->yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1031 yystack.l_mark = yystack.l_base + (save->yystack.l_mark - save->yystack.l_base);
1032 memcpy (yystack.l_base, save->yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1033#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1034 yystack.p_mark = yystack.p_base + (save->yystack.p_mark - save->yystack.p_base);
1035 memcpy (yystack.p_base, save->yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1036#endif
1037 ctry = ++save->ctry;
1038 yystate = save->state;
1039 /* We tried shift, try reduce now */
1040 if ((yyn = yyctable[ctry]) >= 0) goto yyreduce;
1041 yyps->save = save->save;
1042 save->save = NULL;
1043 yyFreeState(save);
1044
1045 /* Nothing left on the stack -- error */
1046 if (!yyps->save)
1047 {
1048#if YYDEBUG
1049 if (yydebug)
1050 fprintf(stderr, "%sdebug[%d,trial]: trial parse FAILED, entering ERROR mode\n",
1051 YYPREFIX, yydepth);
1052#endif
1053 /* Restore state as it was in the most forward-advanced error */
1054 yylvp = yylvals + yyerrctx->lexeme;
1055#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1056 yylpp = yylpsns + yyerrctx->lexeme;
1057#endif
1058 yylexp = yylexemes + yyerrctx->lexeme;
1059 yychar = yylexp[-1];
1060 yylval = yylvp[-1];
1061#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1062 yylloc = yylpp[-1];
1063#endif
1064 yystack.s_mark = yystack.s_base + (yyerrctx->yystack.s_mark - yyerrctx->yystack.s_base);
1065 memcpy (yystack.s_base, yyerrctx->yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1066 yystack.l_mark = yystack.l_base + (yyerrctx->yystack.l_mark - yyerrctx->yystack.l_base);
1067 memcpy (yystack.l_base, yyerrctx->yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1068#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1069 yystack.p_mark = yystack.p_base + (yyerrctx->yystack.p_mark - yyerrctx->yystack.p_base);
1070 memcpy (yystack.p_base, yyerrctx->yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1071#endif
1072 yystate = yyerrctx->state;
1073 yyFreeState(yyerrctx);
1074 yyerrctx = NULL;
1075 }
1076 yynewerrflag = 1;
1077 }
1078 if (yynewerrflag == 0) goto yyinrecovery;
1079#endif /* YYBTYACC */
1080
1081 YYERROR_CALL("syntax error");
1082#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1083 yyerror_loc_range[1] = yylloc; /* lookahead position is error start position */
1084#endif
1085
1086#if !YYBTYACC
1087 goto yyerrlab; /* redundant goto avoids 'unused label' warning */
1088yyerrlab:
1089#endif
1090 ++yynerrs;
1091
1092yyinrecovery:
1093 if (yyerrflag < 3)
1094 {
1095 yyerrflag = 3;
1096 for (;;)
1097 {
1098 if (((yyn = yysindex[*yystack.s_mark]) != 0) && (yyn += YYERRCODE) >= 0 &&
1099 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) YYERRCODE)
1100 {
1101#if YYDEBUG
1102 if (yydebug)
1103 fprintf(stderr, "%s[%d]: state %d, error recovery shifting to state %d\n",
1104 YYDEBUGSTR, yydepth, *yystack.s_mark, yytable[yyn]);
1105#endif
1106 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
1107 yystate = yytable[yyn];
1108 *++yystack.s_mark = yytable[yyn];
1109 *++yystack.l_mark = yylval;
1110#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1111 /* lookahead position is error end position */
1112 yyerror_loc_range[2] = yylloc;
1113 YYLLOC_DEFAULT(yyloc, yyerror_loc_range, 2); /* position of error span */
1114 *++yystack.p_mark = yyloc;
1115#endif
1116 goto yyloop;
1117 }
1118 else
1119 {
1120#if YYDEBUG
1121 if (yydebug)
1122 fprintf(stderr, "%s[%d]: error recovery discarding state %d\n",
1123 YYDEBUGSTR, yydepth, *yystack.s_mark);
1124#endif
1125 if (yystack.s_mark <= yystack.s_base) goto yyabort;
1126#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1127 /* the current TOS position is the error start position */
1128 yyerror_loc_range[1] = *yystack.p_mark;
1129#endif
1130#if defined(YYDESTRUCT_CALL)
1131#if YYBTYACC
1132 if (!yytrial)
1133#endif /* YYBTYACC */
1134#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1135 YYDESTRUCT_CALL("error: discarding state",
1136 yystos[*yystack.s_mark], yystack.l_mark, yystack.p_mark);
1137#else
1138 YYDESTRUCT_CALL("error: discarding state",
1139 yystos[*yystack.s_mark], yystack.l_mark);
1140#endif /* defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED) */
1141#endif /* defined(YYDESTRUCT_CALL) */
