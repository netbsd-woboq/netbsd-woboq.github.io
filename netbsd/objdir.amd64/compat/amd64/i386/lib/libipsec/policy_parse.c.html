1756 yystate = *yystack.s_mark;
1757 yystack.l_mark -= yym;
1758#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1759 yystack.p_mark -= yym;
1760#endif
1761 yym = yylhs[yyn];
1762 if (yystate == 0 && yym == 0)
1763 {
1764#if YYDEBUG
1765 if (yydebug)
1766 {
1767 fprintf(stderr, "%s[%d]: after reduction, ", YYDEBUGSTR, yydepth);
1768#ifdef YYSTYPE_TOSTRING
1769#if YYBTYACC
1770 if (!yytrial)
1771#endif /* YYBTYACC */
1772 fprintf(stderr, "result is <%s>, ", YYSTYPE_TOSTRING(yystos[YYFINAL], yyval));
1773#endif
1774 fprintf(stderr, "shifting from state 0 to final state %d\n", YYFINAL);
1775 }
1776#endif
1777 yystate = YYFINAL;
1778 *++yystack.s_mark = YYFINAL;
1779 *++yystack.l_mark = yyval;
1780#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1781 *++yystack.p_mark = yyloc;
1782#endif
1783 if (yychar < 0)
1784 {
1785#if YYBTYACC
1786 do {
1787 if (yylvp < yylve)
1788 {
1789 /* we're currently re-reading tokens */
1790 yylval = *yylvp++;
1791#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1792 yylloc = *yylpp++;
1793#endif
1794 yychar = *yylexp++;
1795 break;
1796 }
1797 if (yyps->save)
1798 {
1799 /* in trial mode; save scanner results for future parse attempts */
1800 if (yylvp == yylvlim)
1801 { /* Enlarge lexical value queue */
1802 size_t p = (size_t) (yylvp - yylvals);
1803 size_t s = (size_t) (yylvlim - yylvals);
1804
1805 s += YYLVQUEUEGROWTH;
1806 if ((yylexemes = (YYINT *)realloc(yylexemes, s * sizeof(YYINT))) == NULL)
1807 goto yyenomem;
1808 if ((yylvals = (YYSTYPE *)realloc(yylvals, s * sizeof(YYSTYPE))) == NULL)
1809 goto yyenomem;
1810#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1811 if ((yylpsns = (YYLTYPE *)realloc(yylpsns, s * sizeof(YYLTYPE))) == NULL)
1812 goto yyenomem;
1813#endif
1814 yylvp = yylve = yylvals + p;
1815 yylvlim = yylvals + s;
1816#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1817 yylpp = yylpe = yylpsns + p;
1818 yylplim = yylpsns + s;
1819#endif
1820 yylexp = yylexemes + p;
1821 }
1822 *yylexp = (YYINT) YYLEX;
1823 *yylvp++ = yylval;
1824 yylve++;
1825#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1826 *yylpp++ = yylloc;
1827 yylpe++;
1828#endif
1829 yychar = *yylexp++;
1830 break;
1831 }
1832 /* normal operation, no conflict encountered */
1833#endif /* YYBTYACC */
1834 yychar = YYLEX;
1835#if YYBTYACC
1836 } while (0);
1837#endif /* YYBTYACC */
1838 if (yychar < 0) yychar = YYEOF;
1839#if YYDEBUG
1840 if (yydebug)
1841 {
1842 if ((yys = yyname[YYTRANSLATE(yychar)]) == NULL) yys = yyname[YYUNDFTOKEN];
1843 fprintf(stderr, "%s[%d]: state %d, reading token %d (%s)\n",
1844 YYDEBUGSTR, yydepth, YYFINAL, yychar, yys);
1845 }
1846#endif
1847 }
1848 if (yychar == YYEOF) goto yyaccept;
1849 goto yyloop;
1850 }
1851 if (((yyn = yygindex[yym]) != 0) && (yyn += yystate) >= 0 &&
1852 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yystate)
1853 yystate = yytable[yyn];
1854 else
1855 yystate = yydgoto[yym];
1856#if YYDEBUG
1857 if (yydebug)
1858 {
1859 fprintf(stderr, "%s[%d]: after reduction, ", YYDEBUGSTR, yydepth);
1860#ifdef YYSTYPE_TOSTRING
1861#if YYBTYACC
1862 if (!yytrial)
1863#endif /* YYBTYACC */
1864 fprintf(stderr, "result is <%s>, ", YYSTYPE_TOSTRING(yystos[yystate], yyval));
1865#endif
1866 fprintf(stderr, "shifting from state %d to state %d\n", *yystack.s_mark, yystate);
1867 }
1868#endif
1869 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
1870 *++yystack.s_mark = (YYINT) yystate;
1871 *++yystack.l_mark = yyval;
1872#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1873 *++yystack.p_mark = yyloc;
1874#endif
1875 goto yyloop;
1876#if YYBTYACC
1877
1878 /* Reduction declares that this path is valid. Set yypath and do a full parse */
1879yyvalid:
1880 if (yypath) YYABORT;
1881 while (yyps->save)
1882 {
1883 YYParseState *save = yyps->save;
1884 yyps->save = save->save;
1885 save->save = yypath;
1886 yypath = save;
1887 }
1888#if YYDEBUG
1889 if (yydebug)
1890 fprintf(stderr, "%s[%d]: state %d, CONFLICT trial successful, backtracking to state %d, %d tokens\n",
1891 YYDEBUGSTR, yydepth, yystate, yypath->state, (int)(yylvp - yylvals - yypath->lexeme));
1892#endif
1893 if (yyerrctx)
1894 {
1895 yyFreeState(yyerrctx);
1896 yyerrctx = NULL;
1897 }
1898 yylvp = yylvals + yypath->lexeme;
1899#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1900 yylpp = yylpsns + yypath->lexeme;
1901#endif
1902 yylexp = yylexemes + yypath->lexeme;
1903 yychar = YYEMPTY;
1904 yystack.s_mark = yystack.s_base + (yypath->yystack.s_mark - yypath->yystack.s_base);
1905 memcpy (yystack.s_base, yypath->yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1906 yystack.l_mark = yystack.l_base + (yypath->yystack.l_mark - yypath->yystack.l_base);
1907 memcpy (yystack.l_base, yypath->yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1908#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1909 yystack.p_mark = yystack.p_base + (yypath->yystack.p_mark - yypath->yystack.p_base);
1910 memcpy (yystack.p_base, yypath->yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1911#endif
1912 yystate = yypath->state;
1913 goto yyloop;
1914#endif /* YYBTYACC */
1915
1916yyoverflow:
1917 YYERROR_CALL("yacc stack overflow");
1918#if YYBTYACC
1919 goto yyabort_nomem;
1920yyenomem:
1921 YYERROR_CALL("memory exhausted");
1922yyabort_nomem:
1923#endif /* YYBTYACC */
1924 yyresult = 2;
1925 goto yyreturn;
1926
1927yyabort:
1928 yyresult = 1;
1929 goto yyreturn;
1930
1931yyaccept:
1932#if YYBTYACC
1933 if (yyps->save) goto yyvalid;
1934#endif /* YYBTYACC */
1935 yyresult = 0;
1936
1937yyreturn:
1938#if defined(YYDESTRUCT_CALL)
1939 if (yychar != YYEOF && yychar != YYEMPTY)
1940#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1941 YYDESTRUCT_CALL("cleanup: discarding token", yychar, &yylval, &yylloc);
1942#else
1943 YYDESTRUCT_CALL("cleanup: discarding token", yychar, &yylval);
1944#endif /* defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED) */
1945
1946 {
1947 YYSTYPE *pv;
1948#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1949 YYLTYPE *pp;
1950
1951 for (pv = yystack.l_base, pp = yystack.p_base; pv <= yystack.l_mark; ++pv, ++pp)
1952 YYDESTRUCT_CALL("cleanup: discarding state",
1953 yystos[*(yystack.s_base + (pv - yystack.l_base))], pv, pp);
1954#else
1955 for (pv = yystack.l_base; pv <= yystack.l_mark; ++pv)
1956 YYDESTRUCT_CALL("cleanup: discarding state",
1957 yystos[*(yystack.s_base + (pv - yystack.l_base))], pv);
1958#endif /* defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED) */
1959 }
1960#endif /* defined(YYDESTRUCT_CALL) */
1961
1962#if YYBTYACC
1963 if (yyerrctx)
1964 {
1965 yyFreeState(yyerrctx);
1966 yyerrctx = NULL;
1967 }
1968 while (yyps)
1969 {
1970 YYParseState *save = yyps;
1971 yyps = save->save;
1972 save->save = NULL;
1973 yyFreeState(save);
1974 }
1975 while (yypath)
1976 {
1977 YYParseState *save = yypath;
1978 yypath = save->save;
1979 save->save = NULL;
1980 yyFreeState(save);
1981 }
1982#endif /* YYBTYACC */
1983 yyfreestack(&yystack);
1984 return (yyresult);
