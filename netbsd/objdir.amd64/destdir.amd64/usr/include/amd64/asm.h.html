37#ifndef _AMD64_ASM_H_
38#define _AMD64_ASM_H_
39
40#ifdef __x86_64__
41
42#ifdef __PIC__
43#define PIC_PLT(x) x@PLT
44#define PIC_GOT(x) x@GOTPCREL(%rip)
45#else
46#define PIC_PLT(x) x
47#define PIC_GOT(x) x
48#endif
49
50#define _C_LABEL(x) x
51#define _ASM_LABEL(x) x
52
53#define CVAROFF(x,y) (_C_LABEL(x)+y)(%rip)
54
55#ifdef __STDC__
56# define __CONCAT(x,y) x ## y
57# define __STRING(x) #x
58#else
59# define __CONCAT(x,y) x/**/y
60# define __STRING(x) "x"
61#endif
62
63/* let kernels and others override entrypoint alignment */
64#ifndef _ALIGN_TEXT
65# ifdef _STANDALONE
66# define _ALIGN_TEXT .align 4
67# else
68# define _ALIGN_TEXT .align 16
69# endif
70#endif
71
72#define _ENTRY(x) \
73 .text; _ALIGN_TEXT; .globl x; .type x,@function; x:
74#define _LABEL(x) \
75 .globl x; x:
76
77#ifdef _KERNEL
78/* XXX Can't use __CONCAT() here, as it would be evaluated incorrectly. */
79#ifdef __STDC__
80#define IDTVEC(name) \
81 ALIGN_TEXT; .globl X ## name; .type X ## name,@function; X ## name:
82#define IDTVEC_END(name) \
83 .size X ## name, . - X ## name
84#else
85#define IDTVEC(name) \
86 ALIGN_TEXT; .globl X/**/name; .type X/**/name,@function; X/**/name:
87#define IDTVEC_END(name) \
88 .size X/**/name, . - X/**/name
89#endif /* __STDC__ */
90#endif /* _KERNEL */
91
92#ifdef __STDC__
93#define CPUVAR(off) %gs:CPU_INFO_ ## off
94#else
95#define CPUVAR(off) %gs:CPU_INFO_/**/off
96#endif
97
98
99#ifdef GPROF
100# define _PROF_PROLOGUE \
101 pushq %rbp; leaq (%rsp),%rbp; call PIC_PLT(__mcount); popq %rbp
102#else
103# define _PROF_PROLOGUE
104#endif
105
106#define ENTRY(y) _ENTRY(_C_LABEL(y)); _PROF_PROLOGUE
107#define NENTRY(y) _ENTRY(_C_LABEL(y))
108#define ALTENTRY(x) NENTRY(x)
109#define ASENTRY(y) _ENTRY(_ASM_LABEL(y)); _PROF_PROLOGUE
110#define LABEL(y) _LABEL(_C_LABEL(y))
111#define END(y) .size y, . - y
112
113#define ASMSTR .asciz
114
115#define RCSID(x) .pushsection ".ident"; .asciz x; .popsection
116
117#define WEAK_ALIAS(alias,sym) \
118 .weak alias; \
119 alias = sym
120
121/*
122 * STRONG_ALIAS: create a strong alias.
123 */
124#define STRONG_ALIAS(alias,sym) \
125 .globl alias; \
126 alias = sym
127
128#ifdef __STDC__
129#define WARN_REFERENCES(sym,msg) \
130 .pushsection .gnu.warning. ## sym; \
131 .ascii msg; \
132 .popsection
133#else
134#define WARN_REFERENCES(sym,msg) \
135 .pushsection .gnu.warning./**/sym; \
136 .ascii msg; \
137 .popsection
138#endif /* __STDC__ */
