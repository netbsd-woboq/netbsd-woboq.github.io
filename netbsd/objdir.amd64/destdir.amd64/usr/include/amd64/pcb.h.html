70#ifndef _AMD64_PCB_H_
71#define _AMD64_PCB_H_
72
73#ifdef __x86_64__
74
75#include <x86/cpu_extended_state.h>
76#include <amd64/reg.h>
77
78#define NIOPORTS 1024 /* # of ports we allow to be mapped */
79
80struct pcb {
81 int pcb_flags;
82#define PCB_COMPAT32 0x01
83#define PCB_DBREGS 0x02
84 u_int pcb_cr0; /* saved image of CR0 */
85 uint64_t pcb_rsp0;
86 uint64_t pcb_cr2; /* page fault address (CR2) */
87 uint64_t pcb_cr3;
88 uint64_t pcb_rsp;
89 uint64_t pcb_rbp;
90 void *pcb_onfault; /* copyin/out fault recovery */
91 uint64_t pcb_fs;
92 uint64_t pcb_gs;
93 struct dbreg *pcb_dbregs;
94 uint16_t pcb_fpu_dflt_cw;
95 int pcb_iopl;
96
97 uint32_t pcb_unused[8]; /* unused */
98
99 struct cpu_info *pcb_fpcpu; /* cpu holding our fp state. */
100 union savefpu pcb_savefpu __aligned(64); /* floating point state */
101 /* **** DO NOT ADD ANYTHING HERE **** */
102};
103__CTASSERT(sizeof(struct pcb) - sizeof (union savefpu) == 128);
104
105#else /* __x86_64__ */
106
107#include <i386/pcb.h>
108
109#endif /* __x86_64__ */
110
111#endif /* _AMD64_PCB_H_ */
