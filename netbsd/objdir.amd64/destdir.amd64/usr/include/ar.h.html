42#ifndef _AR_H_
43#define _AR_H_
44
45/* Pre-4BSD archives had these magic numbers in them. */
46#define OARMAG1 0177555
47#define OARMAG2 0177545
48
49#define ARMAG "!<arch>\n" /* ar "magic number" */
50#define SARMAG 8 /* strlen(ARMAG); */
51
52#define AR_EFMT1 "#1/" /* extended format #1 */
53
54struct ar_hdr {
55 char ar_name[16]; /* name */
56 char ar_date[12]; /* modification time */
57 char ar_uid[6]; /* user id */
58 char ar_gid[6]; /* group id */
59 char ar_mode[8]; /* octal file permissions */
60 char ar_size[10]; /* size in bytes */
61#define ARFMAG "`\n"
62 char ar_fmag[2]; /* consistency check */
63};
64
65#endif /* !_AR_H_ */
