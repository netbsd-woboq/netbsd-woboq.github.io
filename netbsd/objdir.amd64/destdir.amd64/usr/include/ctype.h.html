71int isgraph_l(int, locale_t);
72int islower_l(int, locale_t);
73int isprint_l(int, locale_t);
74int ispunct_l(int, locale_t);
75int isspace_l(int, locale_t);
76int isupper_l(int, locale_t);
77int isxdigit_l(int, locale_t);
78int tolower_l(int, locale_t);
79int toupper_l(int, locale_t);
