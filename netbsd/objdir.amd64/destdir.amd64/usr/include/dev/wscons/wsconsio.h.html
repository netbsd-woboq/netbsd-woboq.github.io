36/*
37 * WSCONS (wsdisplay, wskbd, wsmouse) exported interfaces.
38 *
39 * Ioctls are all in group 'W'. Ioctl number space is partitioned like:
40 * 0-31 keyboard ioctls (WSKBDIO)
41 * 32-63 mouse ioctls (WSMOUSEIO)
42 * 64-95 display ioctls (WSDISPLAYIO)
43 * 96-127 mux ioctls (WSMUXIO)
44 * 128-255 reserved for future use
45 */
46
47#include <sys/types.h>
48#include <sys/ioccom.h>
49#include <sys/time.h>
50#include <dev/wscons/wsksymvar.h>

53/*
54 * Common event structure (used by keyboard and mouse)
55 */
56struct wscons_event {
57 u_int type;
58 int value;
59 struct timespec time;
60};
61#define WSEVENT_VERSION 1
62
63/* Event type definitions. Comment for each is information in value. */
64#define WSCONS_EVENT_KEY_UP 1 /* key code */
65#define WSCONS_EVENT_KEY_DOWN 2 /* key code */
66#define WSCONS_EVENT_ALL_KEYS_UP 3 /* void */
67#define WSCONS_EVENT_MOUSE_UP 4 /* button # (leftmost = 0) */
68#define WSCONS_EVENT_MOUSE_DOWN 5 /* button # (leftmost = 0) */
69#define WSCONS_EVENT_MOUSE_DELTA_X 6 /* X delta amount */
70#define WSCONS_EVENT_MOUSE_DELTA_Y 7 /* Y delta amount */
71#define WSCONS_EVENT_MOUSE_ABSOLUTE_X 8 /* X location */
72#define WSCONS_EVENT_MOUSE_ABSOLUTE_Y 9 /* Y location */
73#define WSCONS_EVENT_MOUSE_DELTA_Z 10 /* Z delta amount */
74#define WSCONS_EVENT_MOUSE_ABSOLUTE_Z 11 /* Z location */
75#define WSCONS_EVENT_SCREEN_SWITCH 12 /* New screen number */
76#define WSCONS_EVENT_ASCII 13 /* key code is already ascii */
77#define WSCONS_EVENT_MOUSE_DELTA_W 14 /* W delta amount */
78#define WSCONS_EVENT_MOUSE_ABSOLUTE_W 15 /* W location */

114/* Manipulate the keyboard bell. */
115struct wskbd_bell_data {
116 u_int which; /* values to get/set */
117#define WSKBD_BELL_DOPITCH 0x1 /* get/set pitch */
118#define WSKBD_BELL_DOPERIOD 0x2 /* get/set period */
119#define WSKBD_BELL_DOVOLUME 0x4 /* get/set volume */
120#define WSKBD_BELL_DOALL 0x7 /* all of the above */
121 u_int pitch; /* pitch, in Hz */
122 u_int period; /* period, in milliseconds */
123 u_int volume; /* percentage of max volume */
124};
