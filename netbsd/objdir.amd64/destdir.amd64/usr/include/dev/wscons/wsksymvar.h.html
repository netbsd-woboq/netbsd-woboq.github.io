32#ifndef _DEV_WSCONS_WSKSYMVAR_H_
33#define _DEV_WSCONS_WSKSYMVAR_H_
34
35#ifndef _KERNEL
36#include <sys/types.h>
37#endif
38
39typedef u_int16_t keysym_t;
40typedef u_int32_t kbd_t;
41
42struct wscons_keymap {
43 keysym_t command;
44 keysym_t group1[2];
45 keysym_t group2[2];
46};
47
48#ifdef _KERNEL
49struct wscons_keydesc {
50 kbd_t name; /* name of this map */
51 kbd_t base; /* map this one is based on */
52 int map_size; /* size of map */
53 const keysym_t *map; /* the map itself */
54};
55
56struct wskbd_mapdata {
57 const struct wscons_keydesc *keydesc;
58 kbd_t layout;
59};
60
61/* layout variant bits ignored by mapping code */
62#define KB_HANDLEDBYWSKBD KB_METAESC
63
64/*
65 * Utility functions.
66 */
67void wskbd_get_mapentry(const struct wskbd_mapdata *, int,
68 struct wscons_keymap *);
69void wskbd_init_keymap(int, struct wscons_keymap **, int *);
70int wskbd_load_keymap(const struct wskbd_mapdata *,
71 struct wscons_keymap **, int *);
72keysym_t wskbd_compose_value(keysym_t *);
73
74#endif
75
76#endif /* !_DEV_WSCONS_WSKSYMVAR_H_ */
