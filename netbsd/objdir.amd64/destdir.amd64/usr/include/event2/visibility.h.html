33#if defined(event_EXPORTS) || defined(event_extra_EXPORTS) || defined(event_core_EXPORTS)
34# if defined (__SUNPRO_C) && (__SUNPRO_C >= 0x550)
35# define EVENT2_EXPORT_SYMBOL __global
36# elif defined __GNUC__
37# define EVENT2_EXPORT_SYMBOL __attribute__ ((visibility("default")))
38# elif defined(_MSC_VER)
39# define EVENT2_EXPORT_SYMBOL extern __declspec(dllexport)
40# else
41# define EVENT2_EXPORT_SYMBOL /* unknown compiler */
42# endif
43#else
44# if defined(EVENT__NEED_DLLIMPORT) && defined(_MSC_VER) && !defined(EVENT_BUILDING_REGRESS_TEST)
45# define EVENT2_EXPORT_SYMBOL extern __declspec(dllimport)
46# else
47# define EVENT2_EXPORT_SYMBOL
48# endif
49#endif
50
51#endif /* EVENT2_VISIBILITY_H_INCLUDED_ */
