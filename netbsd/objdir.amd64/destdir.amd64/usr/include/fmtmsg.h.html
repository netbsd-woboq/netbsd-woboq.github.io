32#ifndef _FMTMSG_H_
33#define _FMTMSG_H_
34
35/* Major Classifications: identifies the source of the condition. */
36#define MM_HARD 0x01L /* Hardware */
37#define MM_SOFT 0x02L /* Software */
38#define MM_FIRM 0x03L /* Firmware */
39
40/* Message Source Subclassifications: type of software. */
41#define MM_APPL 0x04L /* Application */
42#define MM_UTIL 0x08L /* Utility */
43#define MM_OPSYS 0x0cL /* Operating system */
44
45/* Display Subclassifications: where to display the message. */
46#define MM_PRINT 0x10L /* Display on standard error */
47#define MM_CONSOLE 0x20L /* Display on system console */
48
49/* Status subclassifications: whether the application will recover. */
50#define MM_RECOVER 0x40L /* Recoverable */
51#define MM_NRECOV 0x80L /* Non-recoverable */
52
53/* Severity: seriousness of the condition. */
54#define MM_NOSEV 0 /* No severity level provided */
55#define MM_HALT 1 /* Error causing application to halt */
56#define MM_ERROR 2 /* Encountered a non-fatal fault */
57#define MM_WARNING 3 /* Unusual non-error condition */
58#define MM_INFO 4 /* Informative message */
59
60/* `Null' values for message components. */
61#define MM_NULLMC 0L /* `Null' classsification component */
62#define MM_NULLLBL (char *)0 /* `Null' label component */
63#define MM_NULLSEV 0 /* `Null' severity component */
64#define MM_NULLTXT (char *)0 /* `Null' text component */
65#define MM_NULLACT (char *)0 /* `Null' action component */
66#define MM_NULLTAG (char *)0 /* `Null' tag component */
67
68/* Return values for fmtmsg(). */
69#define MM_OK 0 /* Function succeeded */
70#define MM_NOTOK (-1) /* Function failed completely */
71#define MM_NOMSG 0x01 /* Unable to perform MM_PRINT */
72#define MM_NOCON 0x02 /* Unable to perform MM_CONSOLE */
73
74#include <sys/cdefs.h>
75
76__BEGIN_DECLS
77int fmtmsg(long, const char *, int, const char *, const char *,
78 const char *);
79__END_DECLS
80
81#endif /* !_FMTMSG_H_ */
