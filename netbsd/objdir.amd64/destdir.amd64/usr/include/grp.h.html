39#ifndef _GRP_H_
40#define _GRP_H_
41
42#include <sys/cdefs.h>
43#include <sys/featuretest.h>
44#include <sys/types.h>
45
46#if defined(_NETBSD_SOURCE)
47#define _PATH_GROUP "/etc/group"
48#endif
49
50struct group {
51 __aconst char *gr_name; /* group name */
52 __aconst char *gr_passwd; /* group password */
53 gid_t gr_gid; /* group id */
54 __aconst char *__aconst *gr_mem; /* group members */
55};
56

58struct group *getgrgid(gid_t);
59struct group *getgrnam(const char *);

61 defined(_REENTRANT) || defined(_NETBSD_SOURCE)
62int getgrgid_r(gid_t, struct group *, char *, size_t,
63 struct group **);
64int getgrnam_r(const char *, struct group *, char *, size_t,
65 struct group **);
66#endif
67#if defined(_XOPEN_SOURCE) || defined(_NETBSD_SOURCE)
68struct group *getgrent(void);
69void setgrent(void);
70void endgrent(void);
71#endif
72#if defined(_NETBSD_SOURCE)
73void setgrfile(const char *);
74int setgroupent(int);
75int getgrent_r(struct group *, char *, size_t, struct group **);
76const char *group_from_gid(gid_t, int);
77int gid_from_group(const char *, gid_t *);
78int pwcache_groupdb(int (*)(int), void (*)(void),
79 struct group * (*)(const char *),
80 struct group * (*)(gid_t));
81#endif
82__END_DECLS

84#endif /* !_GRP_H_ */
