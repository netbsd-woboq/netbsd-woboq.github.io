32#ifndef _I386_BYTE_SWAP_H_
33#define _I386_BYTE_SWAP_H_
34
35#include <sys/types.h>
36
37#ifdef __GNUC__
38__BEGIN_DECLS
39
40#define __BYTE_SWAP_U32_VARIABLE __byte_swap_u32_variable
41static __inline uint32_t __byte_swap_u32_variable(uint32_t);
42static __inline uint32_t
43__byte_swap_u32_variable(uint32_t x)
44{
45 __asm volatile (
46 "bswap %1"
47 : "=r" (x) : "0" (x));
48 return (x);
49}
50
51#define __BYTE_SWAP_U16_VARIABLE __byte_swap_u16_variable
52static __inline uint16_t __byte_swap_u16_variable(uint16_t);
53static __inline uint16_t
54__byte_swap_u16_variable(uint16_t x)
55{
56 __asm volatile ("rorw $8, %w1" : "=r" (x) : "0" (x));
57 return (x);
58}
59
60__END_DECLS
61#elif defined(_KERNEL) || defined(_LKM)
62#define __BYTE_SWAP_U32_VARIABLE __byte_swap_u32_variable
63#define __BYTE_SWAP_U16_VARIABLE __byte_swap_u16_variable
64uint32_t __byte_swap_u32_variable(uint32_t);
65uint16_t __byte_swap_u16_variable(uint16_t);
66#endif
67
68#endif /* !_I386_BYTE_SWAP_H_ */
