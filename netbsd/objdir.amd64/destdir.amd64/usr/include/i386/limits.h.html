34#ifndef _I386_LIMITS_H_
35#define _I386_LIMITS_H_
36
37#include <sys/featuretest.h>
38
39#define CHAR_BIT 8 /* number of bits in a char */
40
41#define UCHAR_MAX 0xff /* max value for an unsigned char */
42#define SCHAR_MAX 0x7f /* max value for a signed char */
43#define SCHAR_MIN (-0x7f-1) /* min value for a signed char */
44
45#define USHRT_MAX 0xffff /* max value for an unsigned short */
46#define SHRT_MAX 0x7fff /* max value for a short */
47#define SHRT_MIN (-0x7fff-1) /* min value for a short */
48
49#define UINT_MAX 0xffffffffU /* max value for an unsigned int */
50#define INT_MAX 0x7fffffff /* max value for an int */
51#define INT_MIN (-0x7fffffff-1) /* min value for an int */
52
53#define ULONG_MAX 0xffffffffUL /* max value for an unsigned long */
54#define LONG_MAX 0x7fffffffL /* max value for a long */
55#define LONG_MIN (-0x7fffffffL-1) /* min value for a long */
56
57#if defined(_ISOC99_SOURCE) || (__STDC_VERSION__ - 0) >= 199901L || \
58 defined(_NETBSD_SOURCE)
59#define ULLONG_MAX 0xffffffffffffffffULL /* max unsigned long long */
60#define LLONG_MAX 0x7fffffffffffffffLL /* max signed long long */
61#define LLONG_MIN (-0x7fffffffffffffffLL-1) /* min signed long long */
62#endif
63
64#if defined(_POSIX_C_SOURCE) || defined(_XOPEN_SOURCE) || \
65 defined(_NETBSD_SOURCE)
66#define SSIZE_MAX INT_MAX /* max value for a ssize_t */
67
68#if defined(_NETBSD_SOURCE)
69#define SSIZE_MIN INT_MIN /* min value for a ssize_t */
70#define SIZE_T_MAX UINT_MAX /* max value for a size_t */
71
72#define UQUAD_MAX 0xffffffffffffffffULL /* max unsigned quad */
73#define QUAD_MAX 0x7fffffffffffffffLL /* max signed quad */
74#define QUAD_MIN (-0x7fffffffffffffffLL-1) /* min signed quad */
75
76#endif /* _NETBSD_SOURCE */
77#endif /* _POSIX_C_SOURCE || _XOPEN_SOURCE || _NETBSD_SOURCE */
78
79#if defined(_XOPEN_SOURCE) || defined(_NETBSD_SOURCE)
80#define LONG_BIT 32
81#define WORD_BIT 32
82
83#define DBL_DIG __DBL_DIG__
84#define DBL_MAX __DBL_MAX__
85#define DBL_MIN __DBL_MIN__
86
87#define FLT_DIG __FLT_DIG__
88#define FLT_MAX __FLT_MAX__
89#define FLT_MIN __FLT_MIN__
90#endif
91
92#endif /* _I386_LIMITS_H_ */
