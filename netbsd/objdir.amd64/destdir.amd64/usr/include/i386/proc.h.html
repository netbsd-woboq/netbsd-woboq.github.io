34#ifndef _I386_PROC_H_
35#define _I386_PROC_H_
36
37#include <machine/frame.h>
38#include <machine/pcb.h>
39
40/*
41 * Machine-dependent part of the lwp structure for i386.
42 */
43struct pmap;
44struct vm_page;
45
46struct mdlwp {
47 struct trapframe *md_regs; /* registers on current frame */
48 int md_flags; /* machine-dependent flags */
49 volatile int md_astpending; /* AST pending for this process */
50 struct pmap *md_gc_pmap; /* pmap being garbage collected */
51 struct vm_page *md_gc_ptp; /* pages from pmap g/c */
52};
53
54/* md_flags */
55#define MDL_IOPL 0x0002 /* XEN: i/o privilege */
56
57struct mdproc {
58 int md_flags;
59 void (*md_syscall)(struct trapframe *);
60 /* Syscall handling function */
61};
62
63/* md_flags */
64#define MDP_USEDMTRR 0x0002 /* has set volatile MTRRs */
65
66#endif /* _I386_PROC_H_ */
