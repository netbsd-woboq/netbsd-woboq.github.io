34#ifndef _I386_SIGNAL_H_
35#define _I386_SIGNAL_H_
36
37#include <sys/featuretest.h>
38
39typedef int sig_atomic_t;
40
41#if defined(_NETBSD_SOURCE)
42/*
43 * Get the "code" values
44 */
45#include <machine/trap.h>
46
47#if defined(_KERNEL)
48/*
49 * Information pushed on stack when a signal is delivered.
50 * This is used by the kernel to restore state following
51 * execution of the signal handler. It is also made available
52 * to the handler to allow it to restore state properly if
53 * a non-standard exit is performed.
54 */
55struct sigcontext13 {
56 int sc_gs;
57 int sc_fs;
58 int sc_es;
59 int sc_ds;
60 int sc_edi;
61 int sc_esi;
62 int sc_ebp;
63 int sc_ebx;
64 int sc_edx;
65 int sc_ecx;
66 int sc_eax;
67 /* XXX */
68 int sc_eip;
69 int sc_cs;
70 int sc_eflags;
71 int sc_esp;
72 int sc_ss;
73
74 int sc_onstack; /* sigstack state to restore */
75 int sc_mask; /* signal mask to restore (old style) */
76
77 int sc_trapno; /* XXX should be above */
78 int sc_err;
79};
80
81struct sigcontext {
82 int sc_gs;
83 int sc_fs;
84 int sc_es;
85 int sc_ds;
86 int sc_edi;
87 int sc_esi;
88 int sc_ebp;
89 int sc_ebx;
90 int sc_edx;
91 int sc_ecx;
92 int sc_eax;
93 /* XXX */
94 int sc_eip;
95 int sc_cs;
96 int sc_eflags;
97 int sc_esp;
98 int sc_ss;
99
100 int sc_onstack; /* sigstack state to restore */
101 int __sc_mask13; /* signal mask to restore (old style) */
102
103 int sc_trapno; /* XXX should be above */
104 int sc_err;
105
106 sigset_t sc_mask; /* signal mask to restore (new style) */
107};
108#endif /* _KERNEL */
109
110#endif /* _NETBSD_SOURCE */
111#endif /* !_I386_SIGNAL_H_ */
