38#ifndef _ISOFS_CD9660_CD9660_MOUNT_H_
39#define _ISOFS_CD9660_CD9660_MOUNT_H_
40
41/*
42 * Arguments to mount ISO 9660 filesystems.
43 */
44struct iso_args {
45 const char *fspec; /* block special device to mount */
46 struct export_args30 _pad1; /* compat with old userland tools */
47 int flags; /* mounting flags, see below */
48};
49#define ISOFSMNT_NORRIP 0x00000001 /* disable Rock Ridge Ext.*/
50#define ISOFSMNT_GENS 0x00000002 /* enable generation numbers */
51#define ISOFSMNT_EXTATT 0x00000004 /* enable extended attributes */
52#define ISOFSMNT_NOJOLIET 0x00000008 /* disable Joliet extensions */
53#define ISOFSMNT_NOCASETRANS 0x00000010 /* do not make names lower case */
54#define ISOFSMNT_RRCASEINS 0x00000020 /* case insensitive Rock Ridge */
55
56#define ISOFSMNT_BITS "\177\20" \
57 "b\00norrip\0b\01gens\0b\02extatt\0b\03nojoliet\0" \
58 "b\04nocasetrans\0b\05rrcaseins\0"
59#endif /* _ISOFS_CD9660_CD9660_MOUNT_H_ */
