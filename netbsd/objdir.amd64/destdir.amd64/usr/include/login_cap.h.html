37#ifndef _LOGIN_CAP_H_
38#define _LOGIN_CAP_H_
39
40#define LOGIN_DEFCLASS "default"
41#define LOGIN_DEFSERVICE "login"
42#define LOGIN_DEFUMASK 022
43#define _PATH_LOGIN_CONF "/etc/login.conf"
44
45#define LOGIN_OSETGROUP 0x0001 /* Obsolete setgroup */
46#define LOGIN_SETLOGIN 0x0002 /* Set login */
47#define LOGIN_SETPATH 0x0004 /* Set path */
48#define LOGIN_SETPRIORITY 0x0008 /* Set priority */
49#define LOGIN_SETRESOURCES 0x0010 /* Set resource limits */
50#define LOGIN_SETUMASK 0x0020 /* Set umask */
51#define LOGIN_SETUSER 0x0040 /* Set user */
52#define LOGIN_SETENV 0x0080 /* Set user environment */
53#define LOGIN_SETGID 0x0100 /* Set group id */
54#define LOGIN_SETGROUPS 0x0200 /* Set group membership (initgroups) */
55#define LOGIN_SETALL 0x03fe /* Set all. 0x0001 is obsolete! */
56#define LOGIN_SETGROUP (LOGIN_SETGID|LOGIN_SETGROUPS) /* Set group */
57
58typedef struct {
59 char *lc_class;
60 char *lc_cap;
61 char *lc_style;
62} login_cap_t;
63
64typedef int (*envfunc_t)(void *, const char *, const char *, int);

82int setclasscontext(const char *, unsigned int);

86#endif

90__END_DECLS
