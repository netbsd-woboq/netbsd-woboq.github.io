33#ifndef _MISCFS_GENFS_GENFS_NODE_H_
34#define _MISCFS_GENFS_GENFS_NODE_H_
35
36#include <sys/rwlock.h>
37
38struct vm_page;
39struct kauth_cred;
40struct uio;
41struct vnode;
42
43struct genfs_ops {
44 void (*gop_size)(struct vnode *, off_t, off_t *, int);
45 int (*gop_alloc)(struct vnode *, off_t, off_t, int,
46 struct kauth_cred *);
47 int (*gop_write)(struct vnode *, struct vm_page **, int, int);
48 void (*gop_markupdate)(struct vnode *, int);
49 void (*gop_putrange)(struct vnode *, off_t, off_t *, off_t *);
50};
51
52#define GOP_SIZE(vp, size, eobp, flags) \
53 (*VTOG(vp)->g_op->gop_size)((vp), (size), (eobp), (flags))
54#define GOP_ALLOC(vp, off, len, flags, cred) \
55 (*VTOG(vp)->g_op->gop_alloc)((vp), (off), (len), (flags), (cred))
56#define GOP_WRITE(vp, pgs, npages, flags) \
57 (*VTOG(vp)->g_op->gop_write)((vp), (pgs), (npages), (flags))
58#define GOP_PUTRANGE(vp, off, lop, hip) \
59 (*VTOG(vp)->g_op->gop_putrange)((vp), (off), (lop), (hip))
60
61/*
62 * GOP_MARKUPDATE: mark vnode's timestamps for update.
63 *
64 * => called with v_interlock (and possibly other locks) held.
65 * => used for accesses via mmap.
66 */
67
68#define GOP_MARKUPDATE(vp, flags) \
69 (VTOG(vp)->g_op->gop_markupdate) ? \
70 (*VTOG(vp)->g_op->gop_markupdate)((vp), (flags)) : \
71 (void)0;
72
73/* Flags to GOP_SIZE */
74#define GOP_SIZE_MEM 0x4 /* in-memory size */
75
76/* Flags to GOP_MARKUPDATE */
77#define GOP_UPDATE_ACCESSED 1
78#define GOP_UPDATE_MODIFIED 2
79
80struct genfs_node {
81 const struct genfs_ops *g_op; /* ops vector */
82 krwlock_t g_glock; /* getpages lock */
83 int g_dirtygen;
84};
85
86#define VTOG(vp) ((struct genfs_node *)(vp)->v_data)
87
88void genfs_size(struct vnode *, off_t, off_t *, int);
89void genfs_node_init(struct vnode *, const struct genfs_ops *);
90void genfs_node_destroy(struct vnode *);
91void genfs_gop_putrange(struct vnode *, off_t, off_t *, off_t *);
92int genfs_gop_write(struct vnode *, struct vm_page **, int, int);
93int genfs_gop_write_rwmap(struct vnode *, struct vm_page **, int, int);
94int genfs_compat_gop_write(struct vnode *, struct vm_page **, int, int);
95void genfs_directio(struct vnode *, struct uio *, int);
96
97void genfs_node_wrlock(struct vnode *);
98void genfs_node_rdlock(struct vnode *);
99int genfs_node_rdtrylock(struct vnode *);
100void genfs_node_unlock(struct vnode *);
101int genfs_node_wrlocked(struct vnode *);
102
103#endif /* _MISCFS_GENFS_GENFS_NODE_H_ */
