66typedef struct specnode {
67 vnode_t *sn_next;
68 struct specdev *sn_dev;
69 u_int sn_opencnt;
70 dev_t sn_rdev;
71 bool sn_gone;
72} specnode_t;

74typedef struct specdev {
75 struct mount *sd_mountpoint;
76 struct lockf *sd_lockf;
77 vnode_t *sd_bdevvp;
78 u_int sd_opencnt;
79 u_int sd_refcnt;
80 dev_t sd_rdev;
81} specdev_t;
