33#ifndef _MNTOPTS_H_
34#define _MNTOPTS_H_
35
36#include <sys/cdefs.h>
37
38struct mntopt {
39 const char *m_option; /* option name */
40 int m_inverse; /* if a negative option, eg "dev" */
41 int m_flag; /* bit to set, eg. MNT_RDONLY */
42 int m_altloc; /* 1 => set bit in altflags */
43};
44
45/* User-visible MNT_ flags. */
46#define MOPT_ASYNC { "async", 0, MNT_ASYNC, 0 }
47#define MOPT_NOCOREDUMP { "coredump", 1, MNT_NOCOREDUMP, 0 }
48#define MOPT_NODEV { "dev", 1, MNT_NODEV, 0 }
49#define MOPT_NODEVMTIME { "devmtime", 1, MNT_NODEVMTIME, 0 }
50#define MOPT_NOEXEC { "exec", 1, MNT_NOEXEC, 0 }
51#define MOPT_NOSUID { "suid", 1, MNT_NOSUID, 0 }
52#define MOPT_RDONLY { "rdonly", 0, MNT_RDONLY, 0 }
53#define MOPT_SYNC { "sync", 0, MNT_SYNCHRONOUS, 0 }
54#define MOPT_UNION { "union", 0, MNT_UNION, 0 }
55#define MOPT_USERQUOTA { "userquota", 0, 0, 0 }
56#define MOPT_GROUPQUOTA { "groupquota", 0, 0, 0 }
57#define MOPT_NOATIME { "atime", 1, MNT_NOATIME, 0 }
58#define MOPT_RELATIME { "relatime", 0, MNT_RELATIME, 0 }
59#define MOPT_SYMPERM { "symperm", 0, MNT_SYMPERM, 0 }
60#define MOPT_SOFTDEP { "softdep", 0, MNT_SOFTDEP, 0 }
61#define MOPT_LOG { "log", 0, MNT_LOG, 0 }
62#define MOPT_IGNORE { "hidden", 0, MNT_IGNORE, 0 }
63#define MOPT_EXTATTR { "extattr", 0, MNT_EXTATTR, 0 }
64#define MOPT_DISCARD { "discard", 0, MNT_DISCARD, 0 }
65#define MOPT_AUTOMOUNTED { "automounted",0, MNT_AUTOMOUNTED, 0 }
66
67/* Control flags. */
68#define MOPT_FORCE { "force", 0, MNT_FORCE, 0 }
69#define MOPT_UPDATE { "update", 0, MNT_UPDATE, 0 }
70#define MOPT_RELOAD { "reload", 0, MNT_RELOAD, 0 }
71#define MOPT_GETARGS { "getargs", 0, MNT_GETARGS, 0 }
72
73/* Support for old-style "ro", "rw" flags. */
74#define MOPT_RO { "ro", 0, MNT_RDONLY, 0 }
75#define MOPT_RW { "rw", 1, MNT_RDONLY, 0 }
76
77/* This is parsed by mount(8), but is ignored by specific mount_*(8)s. */
78#define MOPT_AUTO { "auto", 0, 0, 0 }
79#define MOPT_RUMP { "rump", 0, 0, 0 }
80#define MOPT_NULL { NULL, 0, 0, 0 }
81
82#define MOPT_FSTAB_COMPAT \
83 MOPT_RO, \
84 MOPT_RW, \
85 MOPT_AUTO
86
87/* Standard options which all mounts can understand. */
88#define MOPT_STDOPTS \
89 MOPT_USERQUOTA, \
90 MOPT_GROUPQUOTA, \
91 MOPT_FSTAB_COMPAT, \
92 MOPT_NOCOREDUMP, \
93 MOPT_NODEV, \
94 MOPT_NOEXEC, \
95 MOPT_NOSUID, \
96 MOPT_RDONLY, \
97 MOPT_UNION, \
98 MOPT_IGNORE, \
99 MOPT_SYMPERM, \
100 MOPT_RUMP, \
101 MOPT_AUTOMOUNTED

113#endif /* _MNTOPTS_H_ */
