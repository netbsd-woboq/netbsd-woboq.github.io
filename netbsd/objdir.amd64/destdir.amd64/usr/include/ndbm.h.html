37#ifndef _NDBM_H_
38#define _NDBM_H_
39
40#include <sys/cdefs.h>
41#include <sys/featuretest.h>
42#include <stdint.h>
43#include <db.h>
44
45#if defined(_NETBSD_SOURCE)
46/* Map dbm interface onto db(3). */
47#define DBM_RDONLY O_RDONLY
48#endif
49
50/* Flags to dbm_store(). */
51#define DBM_INSERT 0
52#define DBM_REPLACE 1
53
54#if defined(_NETBSD_SOURCE)
55/*
56 * The db(3) support for ndbm(3) always appends this suffix to the
57 * file name to avoid overwriting the user's original database.
58 */
59#define DBM_SUFFIX ".db"
60#endif
61
62typedef struct {
63 void *dptr;
64 size_t dsize; /* XPG4.2 */
65} datum;
66
67typedef DB DBM;
68#if defined(_NETBSD_SOURCE)
69#define dbm_pagfno(a) DBM_PAGFNO_NOT_AVAILABLE
70#endif
71
72__BEGIN_DECLS
73void dbm_close(DBM *);
74DBM *dbm_open(const char *, int, mode_t);
75int dbm_error(DBM *);
76int dbm_clearerr(DBM *);
77#if defined(_NETBSD_SOURCE)
78int dbm_dirfno(DBM *);
79#endif
80#ifndef __LIBC12_SOURCE__
81int dbm_delete(DBM *, datum) __RENAME(__dbm_delete13);
82datum dbm_fetch(DBM *, datum) __RENAME(__dbm_fetch13);
83datum dbm_firstkey(DBM *) __RENAME(__dbm_firstkey13);
84datum dbm_nextkey(DBM *) __RENAME(__dbm_nextkey13);
85int dbm_store(DBM *, datum, datum, int) __RENAME(__dbm_store13);
86#endif
87__END_DECLS
88
89#endif /* !_NDBM_H_ */
