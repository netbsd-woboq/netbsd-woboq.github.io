29#ifndef _NET_AGR_IF_AGRIOCTL_H_
30#define _NET_AGR_IF_AGRIOCTL_H_
31
32/*
33 * kernel-userland interface for agr(4) driver.
34 *
35 * it's only file exported to userland in this driver.
36 */
37
38struct agrreq {
39 int ar_version; /* AGRREQ_VERSION */
40 int ar_cmd;
41 void *ar_buf;
42 size_t ar_buflen;
43};
44
45#define AGRREQ_VERSION 2
46
47/* ar_cmd (SIOCSETAGR) */
48#define AGRCMD_ADDPORT 1
49#define AGRCMD_REMPORT 2
50
51#define SIOCSETAGR SIOCSIFGENERIC
52
53/* ar_cmd (SIOCGETAGR) */
54#define AGRCMD_PORTLIST 3 /* ar_buf points agrportlist */
55
56#define SIOCGETAGR SIOCGIFGENERIC
57
58struct agrportinfo {
59 char api_ifname[IFNAMSIZ];
60 int api_flags; /* AGRPORTINFO_ */
61};
62#define AGRPORTINFO_COLLECTING 1
63#define AGRPORTINFO_DISTRIBUTING 2
64#define AGRPORTINFO_BITS \
65 "\177\020" \
66 "b\0COLLECTING\0" \
67 "b\0DISTRIBUTING\0"
68
69struct agrportlist {
70 int apl_nports;
71 /* struct agrportinfo apl_ports[]; */
72};
73
74#endif /* !_NET_AGR_IF_AGRIOCTL_H_ */
