33#define _NET_BPFJIT_H_
34
35#ifndef _KERNEL
36#include <stddef.h>
37#include <stdint.h>
38#endif
39
40#include <sys/types.h>
41
42#ifdef __linux
43#include <pcap-bpf.h>
44#else
45#include <net/bpf.h>
46#endif
47
48/*
49 * Return value of a function generated by sljit have sljit_uw type
50 * which can have a greater width. In such cases, we rely on the fact
51 * that calling conventions use same registers for smaller types.
52 * SLJIT_MOV_UI is passed to sljit_emit_return() to make sure that the
53 * return value is truncated to unsigned int.
54 */
55typedef unsigned int (*bpfjit_func_t)(const bpf_ctx_t *, bpf_args_t *);
56
57bpfjit_func_t bpfjit_generate_code(const bpf_ctx_t *,
58 const struct bpf_insn *, size_t);
59void bpfjit_free_code(bpfjit_func_t);
60
61#ifdef _KERNEL
62struct bpfjit_ops {
63 bpfjit_func_t (*bj_generate_code)(const bpf_ctx_t *,
64 const struct bpf_insn *, size_t);
65 void (*bj_free_code)(bpfjit_func_t);
66};
67
68extern struct bpfjit_ops bpfjit_module_ops;
69#endif
70
71#endif /* !_NET_BPFJIT_H_ */
