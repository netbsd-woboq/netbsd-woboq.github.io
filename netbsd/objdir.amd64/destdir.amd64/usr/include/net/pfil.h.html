29#ifndef _NET_PFIL_H_
30#define _NET_PFIL_H_
31
32#include <sys/queue.h>
33
34struct mbuf;
35struct ifnet;
36struct ifaddr;
37
38/*
39 * The packet filter hooks are designed for anything to call them to
40 * possibly intercept the packet.
41 */
42typedef int (*pfil_func_t)(void *, struct mbuf **, struct ifnet *, int);
43typedef void (*pfil_ifunc_t)(void *, unsigned long, void *);
44
45#define PFIL_IN 0x00000001
46#define PFIL_OUT 0x00000002
47#define PFIL_ALL (PFIL_IN|PFIL_OUT)
48#define PFIL_IFADDR 0x00000008
49#define PFIL_IFNET 0x00000010
50
51/* events notified by PFIL_IFNET */
52#define PFIL_IFNET_ATTACH 0
53#define PFIL_IFNET_DETACH 1
54
55#define PFIL_TYPE_AF 1 /* key is AF_* type */
56#define PFIL_TYPE_IFNET 2 /* key is ifnet or ifaddr pointer */
57
58typedef struct pfil_head pfil_head_t;
59
60#ifdef _KERNEL
61
62void pfil_init(void);
63int pfil_run_hooks(pfil_head_t *, struct mbuf **, struct ifnet *, int);
64void pfil_run_addrhooks(pfil_head_t *, unsigned long, struct ifaddr *);
65void pfil_run_ifhooks(pfil_head_t *, unsigned long, struct ifnet *);
66
67int pfil_add_hook(pfil_func_t, void *, int, pfil_head_t *);
68int pfil_remove_hook(pfil_func_t, void *, int, pfil_head_t *);
69
70int pfil_add_ihook(pfil_ifunc_t, void *, int, pfil_head_t *);
71int pfil_remove_ihook(pfil_ifunc_t, void *, int, pfil_head_t *);
72
73pfil_head_t * pfil_head_create(int, void *);
74void pfil_head_destroy(pfil_head_t *);
75pfil_head_t * pfil_head_get(int, void *);
76
77/* Packet filtering hook for interfaces (in sys/net/if.c module). */
78extern pfil_head_t *if_pfil;
79
80#endif /* _KERNEL */
81
82#endif /* !_NET_PFIL_H_ */
