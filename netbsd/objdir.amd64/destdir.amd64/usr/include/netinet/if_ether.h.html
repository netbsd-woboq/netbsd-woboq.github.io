63
64/*
65 * Ethernet Address Resolution Protocol.
66 *
67 * See RFC 826 for protocol description. Structure below is not
68 * used by our kernel!!! Only for userland programs which are externally
69 * maintained and need it.
70 */
71
72struct ether_arp {
73 struct arphdr ea_hdr; /* fixed-size header */
74 u_int8_t arp_sha[ETHER_ADDR_LEN]; /* sender hardware address */
75 u_int8_t arp_spa[4]; /* sender protocol address */
76 u_int8_t arp_tha[ETHER_ADDR_LEN]; /* target hardware address */
77 u_int8_t arp_tpa[4]; /* target protocol address */
78} __packed;
79#define arp_hrd ea_hdr.ar_hrd
80#define arp_pro ea_hdr.ar_pro
81#define arp_hln ea_hdr.ar_hln
82#define arp_pln ea_hdr.ar_pln
83#define arp_op ea_hdr.ar_op
84
85#endif /* _netinet_if_ether_compat_h_ */
