40struct llinfo_arp {
41 LIST_ENTRY(llinfo_arp) la_list;
42 struct rtentry *la_rt;
43 struct mbuf *la_hold; /* last packet until resolved/timeout */
44 long la_asked; /* last time we QUERIED for this addr */
45};
46
47struct sockaddr_inarp {
48 u_int8_t sin_len;
49 u_int8_t sin_family;
50 u_int16_t sin_port;
51 struct in_addr sin_addr;
52 struct in_addr sin_srcaddr;
53 u_int16_t sin_tos;
54 u_int16_t sin_other;
55#define SIN_PROXY 1
56};
