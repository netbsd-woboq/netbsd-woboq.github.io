121/*
122 * Local port number conventions:
123 *
124 * Ports < IPPORT_RESERVED are reserved for privileged processes (e.g. root),
125 * unless a kernel is compiled with IPNOPRIVPORTS defined.
126 *
127 * When a user does a bind(2) or connect(2) with a port number of zero,
128 * a non-conflicting local port address is chosen.
129 *
130 * The default range is IPPORT_ANONMIN to IPPORT_ANONMAX, although
131 * that is settable by sysctl(3); net.inet.ip.anonportmin and
132 * net.inet.ip.anonportmax respectively.
133 *
134 * A user may set the IPPROTO_IP option IP_PORTRANGE to change this
135 * default assignment range.
136 *
137 * The value IP_PORTRANGE_DEFAULT causes the default behavior.
138 *
139 * The value IP_PORTRANGE_HIGH is the same as IP_PORTRANGE_DEFAULT,
140 * and exists only for FreeBSD compatibility purposes.
141 *
142 * The value IP_PORTRANGE_LOW changes the range to the "low" are
143 * that is (by convention) restricted to privileged processes.
144 * This convention is based on "vouchsafe" principles only.
145 * It is only secure if you trust the remote host to restrict these ports.
146 * The range is IPPORT_RESERVEDMIN to IPPORT_RESERVEDMAX.
147 */
148
149#define IPPORT_RESERVED 1024
150#define IPPORT_ANONMIN 49152
151#define IPPORT_ANONMAX 65535
152#define IPPORT_RESERVEDMIN 600
153#define IPPORT_RESERVEDMAX (IPPORT_RESERVED-1)

162/*
163 * Definitions of bits in internet address integers.
164 * On subnets, the decomposition of addresses to host and net parts
165 * is done according to subnet mask, not the masks here.
166 *
167 * By byte-swapping the constants, we avoid ever having to byte-swap IP
168 * addresses inside the kernel. Unfortunately, user-level programs rely
169 * on these macros not doing byte-swapping.
170 */
171#ifdef _KERNEL
172#define __IPADDR(x) ((uint32_t) htonl((uint32_t)(x)))
173#else
174#define __IPADDR(x) ((uint32_t)(x))
175#endif

177#define IN_CLASSA(i) (((uint32_t)(i) & __IPADDR(0x80000000)) == \
178 __IPADDR(0x00000000))
179#define IN_CLASSA_NET __IPADDR(0xff000000)
180#define IN_CLASSA_NSHIFT 24
181#define IN_CLASSA_HOST __IPADDR(0x00ffffff)
182#define IN_CLASSA_MAX 128
183
184#define IN_CLASSB(i) (((uint32_t)(i) & __IPADDR(0xc0000000)) == \
185 __IPADDR(0x80000000))
186#define IN_CLASSB_NET __IPADDR(0xffff0000)
187#define IN_CLASSB_NSHIFT 16
188#define IN_CLASSB_HOST __IPADDR(0x0000ffff)
189#define IN_CLASSB_MAX 65536
190
191#define IN_CLASSC(i) (((uint32_t)(i) & __IPADDR(0xe0000000)) == \
192 __IPADDR(0xc0000000))
193#define IN_CLASSC_NET __IPADDR(0xffffff00)
194#define IN_CLASSC_NSHIFT 8
195#define IN_CLASSC_HOST __IPADDR(0x000000ff)
196
197#define IN_CLASSD(i) (((uint32_t)(i) & __IPADDR(0xf0000000)) == \
198 __IPADDR(0xe0000000))
199/* These ones aren't really net and host fields, but routing needn't know. */
200#define IN_CLASSD_NET __IPADDR(0xf0000000)
201#define IN_CLASSD_NSHIFT 28
202#define IN_CLASSD_HOST __IPADDR(0x0fffffff)
203#define IN_MULTICAST(i) IN_CLASSD(i)
204
205#define IN_EXPERIMENTAL(i) (((uint32_t)(i) & __IPADDR(0xf0000000)) == \
206 __IPADDR(0xf0000000))
207#define IN_BADCLASS(i) (((uint32_t)(i) & __IPADDR(0xf0000000)) == \
208 __IPADDR(0xf0000000))
209
210#define IN_LINKLOCAL(i) (((uint32_t)(i) & __IPADDR(0xffff0000)) == \
211 __IPADDR(0xa9fe0000))
212
213#define IN_PRIVATE(i) ((((uint32_t)(i) & __IPADDR(0xff000000)) == \
214 __IPADDR(0x0a000000)) || \
215 (((uint32_t)(i) & __IPADDR(0xfff00000)) == \
216 __IPADDR(0xac100000)) || \
217 (((uint32_t)(i) & __IPADDR(0xffff0000)) == \
218 __IPADDR(0xc0a80000)))
219
220#define IN_LOCAL_GROUP(i) (((uint32_t)(i) & __IPADDR(0xffffff00)) == \
221 __IPADDR(0xe0000000))
222
223#define IN_ANY_LOCAL(i) (IN_LINKLOCAL(i) || IN_LOCAL_GROUP(i))

378/*
379 * in_cksum_phdr:
380 *
381 * Compute significant parts of the IPv4 checksum pseudo-header
382 * for use in a delayed TCP/UDP checksum calculation.
383 *
384 * Args:
385 *
386 * src Source IP address
387 * dst Destination IP address
388 * lenproto htons(proto-hdr-len + proto-number)
389 */
390static __inline u_int16_t __unused
391in_cksum_phdr(u_int32_t src, u_int32_t dst, u_int32_t lenproto)
392{
393 u_int32_t sum;
394
395 sum = lenproto +
396 (u_int16_t)(src >> 16) +
397 (u_int16_t)(src /*& 0xffff*/) +
398 (u_int16_t)(dst >> 16) +
399 (u_int16_t)(dst /*& 0xffff*/);
400
401 sum = (u_int16_t)(sum >> 16) + (u_int16_t)(sum /*& 0xffff*/);
402
403 if (sum > 0xffff)
404 sum -= 0xffff;
405
406 return (sum);
407}

409/*
410 * in_cksum_addword:
411 *
412 * Add the two 16-bit network-order values, carry, and return.
413 */
414static __inline u_int16_t __unused
415in_cksum_addword(u_int16_t a, u_int16_t b)
416{
417 u_int32_t sum = a + b;
418
419 if (sum > 0xffff)
420 sum -= 0xffff;
421
422 return (sum);
423}

466static __inline void
467sockaddr_in_init1(struct sockaddr_in *sin, const struct in_addr *addr,
468 in_port_t port)
469{
470 sin->sin_port = port;
471 sin->sin_addr = *addr;
472 memset(sin->sin_zero, 0, sizeof(sin->sin_zero));
473}
474
475static __inline void
476sockaddr_in_init(struct sockaddr_in *sin, const struct in_addr *addr,
477 in_port_t port)
478{
479 sin->sin_family = AF_INET;
480 sin->sin_len = sizeof(*sin);
481 sockaddr_in_init1(sin, addr, port);
482}

484static __inline struct sockaddr *
485sockaddr_in_alloc(const struct in_addr *addr, in_port_t port, int flags)
486{
487 struct sockaddr *sa;
488
489 sa = sockaddr_alloc(AF_INET, sizeof(struct sockaddr_in), flags);
490
491 if (sa == NULL)
492 return NULL;
493
494 sockaddr_in_init1(satosin(sa), addr, port);
495
496 return sa;
497}
