34#ifndef _NETINET_IN_SYSTM_H_
35#define _NETINET_IN_SYSTM_H_
36
37/*
38 * Miscellaneous internetwork
39 * definitions for kernel.
40 */
41
42/*
43 * Network types.
44 *
45 * Internally the system keeps counters in the headers with the bytes
46 * swapped so that VAX instructions will work on them. It reverses
47 * the bytes before transmission at each protocol level. The n_ types
48 * represent the types with the bytes in ``high-ender'' order.
49 */
50typedef u_int16_t n_short; /* short as received from the net */
51typedef u_int32_t n_long; /* long as received from the net */
52
53typedef u_int32_t n_time; /* ms since 00:00 GMT, byte rev */
54
55#ifdef _KERNEL
56n_time iptime (void);
57#endif
58
59#endif /* !_NETINET_IN_SYSTM_H_ */
