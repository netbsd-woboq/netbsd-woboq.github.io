36#define _NETINET_PIM_VAR_H_

38/*
39 * Protocol Independent Multicast (PIM),
40 * kernel variables and implementation-specific definitions.
41 *
42 * Written by George Edmond Eddy (Rusty), ISI, February 1998.
43 * Modified by Pavlin Radoslavov, USC/ISI, May 1998, Aug 1999, October 2000.
44 * Modified by Hitoshi Asaeda, WIDE, August 1998.

46
47/*
48 * PIM statistics kept in the kernel
49 */
50struct pimstat {
51 u_quad_t pims_rcv_total_msgs; /* total PIM messages received */
52 u_quad_t pims_rcv_total_bytes; /* total PIM bytes received */
53 u_quad_t pims_rcv_tooshort; /* rcvd with too few bytes */
54 u_quad_t pims_rcv_badsum; /* rcvd with bad checksum */
55 u_quad_t pims_rcv_badversion; /* rcvd bad PIM version */
56 u_quad_t pims_rcv_registers_msgs; /* rcvd regs. msgs (data only) */
57 u_quad_t pims_rcv_registers_bytes; /* rcvd regs. bytes (data only) */
58 u_quad_t pims_rcv_registers_wrongiif; /* rcvd regs. on wrong iif */
59 u_quad_t pims_rcv_badregisters; /* rcvd invalid registers */
60 u_quad_t pims_snd_registers_msgs; /* sent regs. msgs (data only) */
61 u_quad_t pims_snd_registers_bytes; /* sent regs. bytes (data only) */
62};
63
64/*
65 * Names for PIM sysctl objects
66 */
67#define PIMCTL_STATS 1 /* statistics (read-only) */
68
69#ifdef _KERNEL
70extern struct pimstat pimstat;
71
72void pim_input(struct mbuf *, int, int);
73#endif
74
75#endif /* !_NETINET_PIM_VAR_H_ */
