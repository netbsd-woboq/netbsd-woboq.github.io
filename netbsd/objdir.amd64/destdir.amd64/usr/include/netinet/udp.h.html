34#ifndef _NETINET_UDP_H_
35#define _NETINET_UDP_H_
36
37#include <sys/types.h>
38/*
39 * Udp protocol header.
40 * Per RFC 768, September, 1981.
41 */
42struct udphdr {
43 uint16_t uh_sport; /* source port */
44 uint16_t uh_dport; /* destination port */
45 uint16_t uh_ulen; /* udp length */
46 uint16_t uh_sum; /* udp checksum */
47} __packed;
48
49/* socket options for UDP */
50#define UDP_ENCAP 100
51
52/* Encapsulation types */
53#define UDP_ENCAP_ESPINUDP_NON_IKE 1 /* draft-ietf-ipsec-nat-t-ike-00/01 */
54#define UDP_ENCAP_ESPINUDP 2 /* draft-ietf-ipsec-udp-encaps-06 */
55
56/* Default encapsulation port */
57#define UDP_ENCAP_ESPINUDP_PORT 500
58
59/* Maximum UDP fragment size for ESP over UDP */
60#define UDP_ENCAP_ESPINUDP_MAXFRAGLEN 552
61
62#endif /* !_NETINET_UDP_H_ */
