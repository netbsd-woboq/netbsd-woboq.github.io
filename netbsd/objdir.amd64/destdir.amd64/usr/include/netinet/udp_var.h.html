37/*
38 * UDP kernel structures and variables.
39 */
40struct udpiphdr {
41 struct ipovly ui_i; /* overlaid ip structure */
42 struct udphdr ui_u; /* udp header */
43} __packed;
44#define ui_x1 ui_i.ih_x1
45#define ui_pr ui_i.ih_pr
46#define ui_len ui_i.ih_len
47#define ui_src ui_i.ih_src
48#define ui_dst ui_i.ih_dst
49#define ui_sport ui_u.uh_sport
50#define ui_dport ui_u.uh_dport
51#define ui_ulen ui_u.uh_ulen
52#define ui_sum ui_u.uh_sum
53
54/*
55 * UDP statistics.
56 * Each counter is an unsigned 64-bit value.
57 */
58#define UDP_STAT_IPACKETS 0 /* total input packets */
59#define UDP_STAT_HDROPS 1 /* packet shorter than header */
60#define UDP_STAT_BADSUM 2 /* checksum error */
61#define UDP_STAT_BADLEN 3 /* data length larger than packet */
62#define UDP_STAT_NOPORT 4 /* no socket on port */
63#define UDP_STAT_NOPORTBCAST 5 /* of above, arrived as broadcast */
64#define UDP_STAT_FULLSOCK 6 /* not delivered, input socket full */
65#define UDP_STAT_PCBHASHMISS 7 /* input packets missing PCB hash */
66#define UDP_STAT_OPACKETS 8 /* total output packets */
67
68#define UDP_NSTATS 9
69
70/*
71 * Names for UDP sysctl objects
72 */
73#define UDPCTL_CHECKSUM 1 /* checksum UDP packets */
74#define UDPCTL_SENDSPACE 2 /* default send buffer */
75#define UDPCTL_RECVSPACE 3 /* default recv buffer */
76#define UDPCTL_LOOPBACKCKSUM 4 /* do UDP checksum on loopback */
77#define UDPCTL_STATS 5 /* UDP statistics */
