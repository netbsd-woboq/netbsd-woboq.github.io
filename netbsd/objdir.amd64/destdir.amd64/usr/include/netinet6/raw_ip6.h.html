36/*
37 * ICMPv6 stat is counted separately. see netinet/icmp6.h
38 */
39#define RIP6_STAT_IPACKETS 0 /* total input packets */
40#define RIP6_STAT_ISUM 1 /* input checksum computations */
41#define RIP6_STAT_BADSUM 2 /* of above, checksum error */
42#define RIP6_STAT_NOSOCK 3 /* no matching socket */
43#define RIP6_STAT_NOSOCKMCAST 4 /* of above, arrived as multicast */
44#define RIP6_STAT_FULLSOCK 5 /* not delivered, input socket full */
45#define RIP6_STAT_OPACKETS 6 /* total output packets */
46
47#define RIP6_NSTATS 7
