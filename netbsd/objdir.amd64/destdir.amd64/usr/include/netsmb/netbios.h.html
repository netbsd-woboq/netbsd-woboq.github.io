36#ifndef _NETSMB_NETBIOS_H_
37#define _NETSMB_NETBIOS_H_
38
39/*
40 * make this file dirty...
41 */
42#ifndef _NETINET_IN_H_
43#include <netinet/in.h>
44#endif
45
46#define AF_NETBIOS AF_NS /* XXX: should go to socket.h */
47#define PF_NETBIOS AF_NETBIOS
48
49#define NBPROTO_TCPSSN 1 /* NETBIOS session over TCP */
50#define NBPROTO_IPXSSN 11 /* NETBIOS over IPX */
51
52#define NB_NAMELEN 16
53#define NB_ENCNAMELEN (NB_NAMELEN * 2)
54#define NB_MAXLABLEN 63
55
56#define NB_MINSALEN (sizeof(struct sockaddr_nb))
57
58/*
59 * name types
60 */
61#define NBT_WKSTA 0x00
62#define NBT_SERVER 0x20
63
64/*
65 * Session packet types
66 */
67#define NB_SSN_MESSAGE 0x0
68#define NB_SSN_REQUEST 0x81
69#define NB_SSN_POSRESP 0x82
70#define NB_SSN_NEGRESP 0x83
71#define NB_SSN_RTGRESP 0x84
72#define NB_SSN_KEEPALIVE 0x85
73
74/*
75 * resolver: Opcodes
76 */
77#define NBNS_OPCODE_QUERY 0x00
78#define NBNS_OPCODE_REGISTER 0x05
79#define NBNS_OPCODE_RELEASE 0x06
80#define NBNS_OPCODE_WACK 0x07
81#define NBNS_OPCODE_REFRESH 0x08
82#define NBNS_OPCODE_RESPONSE 0x10 /* or'ed with other opcodes */
83
84/*
85 * resolver: NM_FLAGS
86 */
87#define NBNS_NMFLAG_BCAST 0x01
88#define NBNS_NMFLAG_RA 0x08 /* recursion available */
89#define NBNS_NMFLAG_RD 0x10 /* recursion desired */
90#define NBNS_NMFLAG_TC 0x20 /* truncation occurred */
91#define NBNS_NMFLAG_AA 0x40 /* authoritative answer */
92
93/*
94 * resolver: Question types
95 */
96#define NBNS_QUESTION_TYPE_NB 0x0020
97#define NBNS_QUESTION_TYPE_NBSTAT 0x0021
98
99/*
100 * resolver: Question class
101 */
102#define NBNS_QUESTION_CLASS_IN 0x0001
103
104/*
105 * resolver: Limits
106 */
107#define NBNS_MAXREDIRECTS 3 /* maximum number of accepted redirects */
108#define NBDG_MAXSIZE 576 /* maximum nbns datagram size */
109
110/*
111 * NETBIOS addressing
112 */
113union nb_tran {
114 struct sockaddr_in x_in;
115};
116
117struct nb_name {
118 u_int nn_type;
119 u_char nn_name[NB_NAMELEN + 1];
120 u_char * nn_scope;
121};
122
123/*
124 * Socket address
125 */
126struct sockaddr_nb {
127 u_char snb_len;
128 u_char snb_family;
129 union nb_tran snb_tran; /* transport */
130 u_char snb_name[1 + NB_ENCNAMELEN + 1]; /* encoded */
131};
132
133#define snb_addrin snb_tran.x_in
134
135#endif /* !_NETSMB_NETBIOS_H_ */
