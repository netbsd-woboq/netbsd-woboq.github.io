39#ifndef _NLIST_H_
40#define _NLIST_H_
41
42#include <sys/cdefs.h>
43
44/*
45 * Symbol table entry format. The #ifdef's are so that programs including
46 * nlist.h can initialize nlist structures statically.
47 */
48struct nlist {
49#ifdef _AOUT_INCLUDE_
50 union {
51 __aconst char *n_name; /* symbol name (in memory) */
52 long n_strx; /* file string table offset (on disk) */
53 } n_un;
54# define N_NAME(nlp) ((nlp)->n_un.n_name)
55#else
56 const char *n_name; /* symbol name (in memory) */
57# define N_NAME(nlp) ((nlp)->n_name)
58#endif
59
60#define N_UNDF 0x00 /* undefined */
61#define N_ABS 0x02 /* absolute address */
62#define N_TEXT 0x04 /* text segment */
63#define N_DATA 0x06 /* data segment */
64#define N_BSS 0x08 /* bss segment */
65#define N_INDR 0x0a /* alias definition */
66#define N_SIZE 0x0c /* pseudo type, defines a symbol's size */
67#define N_COMM 0x12 /* common reference */
68#define N_SETA 0x14 /* absolute set element symbol */
69#define N_SETT 0x16 /* text set element symbol */
70#define N_SETD 0x18 /* data set element symbol */
71#define N_SETB 0x1a /* bss set element symbol */
72#define N_SETV 0x1c /* set vector symbol */
73#define N_FN 0x1e /* file name (N_EXT on) */
74#define N_WARN 0x1e /* warning message (N_EXT off) */
75
76#define N_EXT 0x01 /* external (global) bit, OR'ed in */
77#define N_TYPE 0x1e /* mask for all the type bits */
78 unsigned char n_type; /* type defines */
79
80 char n_other; /* spare */
81#define n_hash n_desc /* used internally by ld(1); XXX */
82 short n_desc; /* used by stab entries */
83 unsigned long n_value; /* address/value of the symbol */
84};
85
86#define N_FORMAT "%08x" /* namelist value format; XXX */
87#define N_STAB 0x0e0 /* mask for debugger symbols -- stab(5) */
88
89__BEGIN_DECLS
90int nlist(const char *, struct nlist *);
91int __fdnlist(int, struct nlist *); /* XXX for libkvm */
92__END_DECLS
93
94#endif /* !_NLIST_H_ */
