32#ifndef _LIBNVMM_H_
33#define _LIBNVMM_H_
34
35#include <stdint.h>
36#include <stdbool.h>
37
38#include <dev/nvmm/nvmm.h>
39#include <dev/nvmm/nvmm_ioctl.h>
40
41struct nvmm_io {
42 uint64_t port;
43 bool in;
44 size_t size;
45 uint8_t *data;
46};
47
48struct nvmm_mem {
49 gpaddr_t gpa;
50 bool write;
51 size_t size;
52 uint8_t *data;
53};
54
55struct nvmm_callbacks {
56 void (*io)(struct nvmm_io *);
57 void (*mem)(struct nvmm_mem *);
58};
59
60struct nvmm_machine {
61 nvmm_machid_t machid;
62 struct nvmm_comm_page **pages;
63 void *areas; /* opaque */
64 struct nvmm_callbacks cbs;
65};
66
67struct nvmm_vcpu {
68 nvmm_cpuid_t cpuid;
69 struct nvmm_vcpu_state *state;
70 struct nvmm_event *event;
71 struct nvmm_exit *exit;
72};
73
74#define NVMM_MACH_CONF_CALLBACKS NVMM_MACH_CONF_LIBNVMM_BEGIN
75
76#define NVMM_PROT_READ 0x01
77#define NVMM_PROT_WRITE 0x02
78#define NVMM_PROT_EXEC 0x04
79#define NVMM_PROT_USER 0x08
80#define NVMM_PROT_ALL 0x0F
81typedef uint64_t nvmm_prot_t;
82
83int nvmm_capability(struct nvmm_capability *);

89int nvmm_vcpu_create(struct nvmm_machine *, nvmm_cpuid_t, struct nvmm_vcpu *);

92int nvmm_vcpu_getstate(struct nvmm_machine *, struct nvmm_vcpu *, uint64_t);
93int nvmm_vcpu_inject(struct nvmm_machine *, struct nvmm_vcpu *);
94int nvmm_vcpu_run(struct nvmm_machine *, struct nvmm_vcpu *);
95
96int nvmm_gpa_map(struct nvmm_machine *, uintptr_t, gpaddr_t, size_t, int);
97int nvmm_gpa_unmap(struct nvmm_machine *, uintptr_t, gpaddr_t, size_t);

99int nvmm_hva_unmap(struct nvmm_machine *, uintptr_t, size_t);
100
101int nvmm_gva_to_gpa(struct nvmm_machine *, struct nvmm_vcpu *, gvaddr_t, gpaddr_t *,

103int nvmm_gpa_to_hva(struct nvmm_machine *, gpaddr_t, uintptr_t *,

106int nvmm_assist_io(struct nvmm_machine *, struct nvmm_vcpu *);
