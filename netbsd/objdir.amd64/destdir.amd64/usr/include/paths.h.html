34#ifndef _PATHS_H_
35#define _PATHS_H_

37/*
38 * Default user search path.
39 * Set by login(1), rshd(8), rexecd(8)
40 * Used by execvp(3) absent PATH from the environ(7)
41 */
42#ifdef RESCUEDIR
43#define _PATH_DEFPATH RESCUEDIR ":/usr/bin:/bin:/usr/pkg/bin:/usr/local/bin"
44#else
45#define _PATH_DEFPATH "/usr/bin:/bin:/usr/pkg/bin:/usr/local/bin"
46#endif
47
48/*
49 * All standard utilities path.
50 * Set by init(8) for system programs & scripts (e.g. /etc/rc)
51 * Used by ttyaction(3), whereis(1)
52 */
53#define _PATH_STDPATH \
54 "/usr/bin:/bin:/usr/sbin:/sbin:/usr/pkg/bin:/usr/pkg/sbin:/usr/local/bin:/usr/local/sbin"
