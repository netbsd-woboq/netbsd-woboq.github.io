32#ifndef _PROPLIB_PROP_INGEST_H_
33#define _PROPLIB_PROP_INGEST_H_
34
35#include <prop/prop_dictionary.h>
36
37typedef enum {
38 PROP_INGEST_ERROR_NO_ERROR = 0,
39 PROP_INGEST_ERROR_NO_KEY = 1,
40 PROP_INGEST_ERROR_WRONG_TYPE = 2,
41 PROP_INGEST_ERROR_HANDLER_FAILED = 3
42} prop_ingest_error_t;
43
44typedef enum {
45 PROP_INGEST_FLAG_OPTIONAL = 0x01
46} prop_ingest_flag_t;
47
48typedef struct _prop_ingest_context *prop_ingest_context_t;
49
50typedef bool (*prop_ingest_handler_t)(prop_ingest_context_t, prop_object_t);
51
52typedef struct {
53 const char *pite_key;
54 prop_type_t pite_type;
55 unsigned int pite_flags;
56 prop_ingest_handler_t pite_handler;
57} prop_ingest_table_entry;
58
59#define PROP_INGEST(key_, type_, handler_) \
60 { .pite_key = key_ , \
61 .pite_type = type_ , \
62 .pite_flags = 0 , \
63 .pite_handler = handler_ }
64
65#define PROP_INGEST_OPTIONAL(key_, type_, handler_) \
66 { .pite_key = key_ , \
67 .pite_type = type_ , \
68 .pite_flags = PROP_INGEST_FLAG_OPTIONAL , \
69 .pite_handler = handler_ }
70
71#define PROP_INGEST_END \
72 { .pite_key = NULL }
73
74__BEGIN_DECLS
75prop_ingest_context_t
76 prop_ingest_context_alloc(void *);
77void prop_ingest_context_free(prop_ingest_context_t);
78
79prop_ingest_error_t
80 prop_ingest_context_error(prop_ingest_context_t);
81prop_type_t prop_ingest_context_type(prop_ingest_context_t);
82const char * prop_ingest_context_key(prop_ingest_context_t);
83void * prop_ingest_context_private(prop_ingest_context_t);
84
85bool prop_dictionary_ingest(prop_dictionary_t,
86 const prop_ingest_table_entry[],
87 prop_ingest_context_t);
88__END_DECLS
89
90#endif /* _PROPLIB_PROP_INGEST_H_ */
