32#ifndef _PROPLIB_PROP_OBJECT_H_
33#define _PROPLIB_PROP_OBJECT_H_
34
35#include <sys/types.h>
36
37#if !defined(_KERNEL) && !defined(_STANDALONE)
38#include <stdbool.h>
39#endif /* ! _KERNEL && ! _STANDALONE */
40
41typedef void *prop_object_t;
42
43typedef enum {
44 PROP_TYPE_UNKNOWN = 0x00000000,
45#ifndef _PROPLIB_ZFS_CONFLICT
46 PROP_TYPE_BOOL = 0x626f6f6c, /* 'bool' */
47 PROP_TYPE_NUMBER = 0x6e6d6272, /* 'nmbr' */
48 PROP_TYPE_STRING = 0x73746e67, /* 'stng' */
49 PROP_TYPE_DATA = 0x64617461, /* 'data' */
50 PROP_TYPE_ARRAY = 0x61726179, /* 'aray' */
51 PROP_TYPE_DICTIONARY = 0x64696374, /* 'dict' */
52 PROP_TYPE_DICT_KEYSYM = 0x646b6579 /* 'dkey' */
53#endif /* !_PROPLIB_ZFS_CONFLICT */
54} prop_type_t;
55
56__BEGIN_DECLS
57void prop_object_retain(prop_object_t);
58void prop_object_release(prop_object_t);
59
60prop_type_t prop_object_type(prop_object_t);
61
62bool prop_object_equals(prop_object_t, prop_object_t);
63bool prop_object_equals_with_error(prop_object_t, prop_object_t, bool *);
64
65typedef struct _prop_object_iterator *prop_object_iterator_t;
66
67prop_object_t prop_object_iterator_next(prop_object_iterator_t);
68void prop_object_iterator_reset(prop_object_iterator_t);
69void prop_object_iterator_release(prop_object_iterator_t);
70__END_DECLS
71
72#endif /* _PROPLIB_PROP_OBJECT_H_ */
