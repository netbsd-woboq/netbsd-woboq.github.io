85#define _PW_KEYBYNAME '1' /* stored by name */
86#define _PW_KEYBYNUM '2' /* stored by entry in the "file" */
87#define _PW_KEYBYUID '3' /* stored by uid */
88
89#define _PASSWORD_EFMT1 '_' /* extended DES encryption format */
90#define _PASSWORD_NONDES '$' /* non-DES encryption formats */
91
92#define _PASSWORD_LEN 128 /* max length, not counting NUL */
93
94#define _PASSWORD_NOUID 0x01 /* flag for no specified uid. */
95#define _PASSWORD_NOGID 0x02 /* flag for no specified gid. */
96#define _PASSWORD_NOCHG 0x04 /* flag for no specified change. */
97#define _PASSWORD_NOEXP 0x08 /* flag for no specified expire. */
98
99#define _PASSWORD_OLDFMT 0x10 /* flag to expect an old style entry */
100#define _PASSWORD_NOWARN 0x20 /* no warnings for bad entries */
101
102#define _PASSWORD_WARNDAYS 14 /* days to warn about expiry */
103#define _PASSWORD_CHGNOW -1 /* special day to force password
104 * change at next login */
105#endif
106
107struct passwd {
108 __aconst char *pw_name; /* user name */
109 __aconst char *pw_passwd; /* encrypted password */
110 uid_t pw_uid; /* user uid */
111 gid_t pw_gid; /* user gid */
112 time_t pw_change; /* password change time */
113 __aconst char *pw_class; /* user login class */
114 __aconst char *pw_gecos; /* general information */
115 __aconst char *pw_dir; /* home directory */
116 __aconst char *pw_shell; /* default shell */
117 time_t pw_expire; /* account expiration */
118};
