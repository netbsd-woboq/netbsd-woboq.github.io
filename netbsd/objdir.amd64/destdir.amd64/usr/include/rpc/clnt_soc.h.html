53/*
54 * TCP based rpc
55 * CLIENT *
56 * clnttcp_create(raddr, prog, vers, sockp, sendsz, recvsz)
57 * struct sockaddr_in *raddr;
58 * unsigned long prog;
59 * unsigned long version;
60 * int *sockp;
61 * unsigned sendsz;
62 * unsigned recvsz;
63 */
64__BEGIN_DECLS
65extern CLIENT *clnttcp_create(struct sockaddr_in *,
66 unsigned long,
67 unsigned long,
68 int *,
69 unsigned int,
70 unsigned int);
71__END_DECLS

74 * Raw (memory) rpc.
75 */
76__BEGIN_DECLS
77extern CLIENT *clntraw_create (unsigned long, unsigned long);

80
81/*
82 * UDP based rpc.
83 * CLIENT *
84 * clntudp_create(raddr, program, version, wait, sockp)
85 * struct sockaddr_in *raddr;
86 * unsigned long program;
87 * unsigned long version;
88 * struct timeval wait;
89 * int *sockp;
90 *
91 * Same as above, but you specify max packet sizes.
92 * CLIENT *
93 * clntudp_bufcreate(raddr, program, version, wait, sockp, sendsz, recvsz)
94 * struct sockaddr_in *raddr;
95 * unsigned long program;
96 * unsigned long version;
97 * struct timeval wait;
98 * int *sockp;
99 * unsigned sendsz;
100 * unsigned recvsz;
101 */
102__BEGIN_DECLS
103extern CLIENT *clntudp_create(struct sockaddr_in *,
104 unsigned long,
105 unsigned long,
106 struct timeval,
107 int *);
108extern CLIENT *clntudp_bufcreate(struct sockaddr_in *,
109 unsigned long,
110 unsigned long,
111 struct timeval,
112 int *,
113 unsigned int,
114 unsigned int);
115__END_DECLS
