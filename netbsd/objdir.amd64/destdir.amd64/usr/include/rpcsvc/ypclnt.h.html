34#define YPERR_BADARGS 1 /* args to function are bad */
35#define YPERR_RPC 2 /* RPC failure */

37#define YPERR_MAP 4 /* no such map in server's domain */
38#define YPERR_KEY 5 /* no such key in map */
39#define YPERR_YPERR 6 /* some internal YP server or client error */
40#define YPERR_RESRC 7 /* local resource allocation failure */
41#define YPERR_NOMORE 8 /* no more records in map database */
42#define YPERR_PMAP 9 /* can't communicate with portmapper */
43#define YPERR_YPBIND 10 /* can't communicate with ypbind */
44#define YPERR_YPSERV 11 /* can't communicate with ypserv */
45#define YPERR_NODOM 12 /* local domain name not set */
46#define YPERR_BADDB 13 /* YP data base is bad */

48#define YPERR_ACCESS 15 /* access violation */
49#define YPERR_BUSY 16 /* database is busy */
