84/*
85 * vfork needs to be implemented as an inline to make everything
86 * run in the caller's stackframe.
87 */
88static __attribute__((__always_inline__)) __returns_twice inline pid_t
89rumpclient__dofork(pid_t (*forkfn)(void))
90{
91 struct rumpclient_fork *rf;
92 pid_t pid;
93 int childran = 0;
94
95 if (!(rf = rumpclient_prefork()))
96 return -1;
97
98 switch ((pid = forkfn())) {
99 case -1:
100 rumpclient_fork_cancel(rf);
101 break;
102 case 0:
103 childran = 1;
104 if (rumpclient_fork_init(rf) == -1)
105 pid = -1;
106 break;
107 default:
108 /* XXX: multithreaded vforker? do they exist? */
109 if (childran)
110 rumpclient_fork_vparent(rf);
111 break;
112 }
113
114 return pid;
115}
