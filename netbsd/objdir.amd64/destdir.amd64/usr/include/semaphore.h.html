32#ifndef _SEMAPHORE_H_
33#define _SEMAPHORE_H_
34
35/* POSIX 1003.1b semaphores */
36
37struct _sem_st;
38typedef struct _sem_st *sem_t;
39
40#define SEM_FAILED ((sem_t *)0)
41
42#include <sys/semaphore.h> /* some kernel-only bits */
43#include <sys/time.h>
44
45__BEGIN_DECLS
46int sem_close(sem_t *);
47int sem_destroy(sem_t *);
48int sem_getvalue(sem_t * __restrict, int * __restrict);
49int sem_init(sem_t *, int, unsigned int);
50int sem_post(sem_t *);
51int sem_timedwait(sem_t *, const struct timespec * __restrict);
52int sem_trywait(sem_t *);
53int sem_unlink(const char *);
54int sem_wait(sem_t *);
55sem_t *sem_open(const char *, int, ...);
56__END_DECLS
57
58#endif /* !_SEMAPHORE_H_ */
