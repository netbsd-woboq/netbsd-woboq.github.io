31#ifndef _SSP_SSP_H_
32#define _SSP_SSP_H_
33
34#include <sys/cdefs.h>
35
36#if !defined(__cplusplus)
37# if _FORTIFY_SOURCE > 0 && !defined(__lint__) && \
38 (__OPTIMIZE__ > 0 || defined(__clang__)) && __GNUC_PREREQ__(4, 1)
39# if _FORTIFY_SOURCE > 1
40# define __SSP_FORTIFY_LEVEL 2
41# else
42# define __SSP_FORTIFY_LEVEL 1
43# endif
44# else
45# define __SSP_FORTIFY_LEVEL 0
46# endif
47#else
48# define __SSP_FORTIFY_LEVEL 0
49#endif
50
51/* __ssp_real is used by the implementation in libc */
52#if __SSP_FORTIFY_LEVEL == 0
53#define __ssp_real_(fun) fun
54#else
55#define __ssp_real_(fun) __ssp_real_ ## fun
56#endif
57#define __ssp_real(fun) __ssp_real_(fun)
58
59#define __ssp_inline static __inline __attribute__((__always_inline__))
60
61#define __ssp_bos(ptr) __builtin_object_size(ptr, __SSP_FORTIFY_LEVEL > 1)
62#define __ssp_bos0(ptr) __builtin_object_size(ptr, 0)
63
64#define __ssp_check(buf, len, bos) \
65 if (bos(buf) != (size_t)-1 && len > bos(buf)) \
66 __chk_fail()
67#define __ssp_redirect_raw(rtype, fun, symbol, args, call, cond, bos) \
68rtype __ssp_real_(fun) args __RENAME(symbol); \
69__ssp_inline rtype fun args __RENAME(__ssp_protected_ ## fun); \
70__ssp_inline rtype fun args { \
71 if (cond) \
72 __ssp_check(__buf, __len, bos); \
73 return __ssp_real_(fun) call; \
74}
75
76#define __ssp_redirect(rtype, fun, args, call) \
77 __ssp_redirect_raw(rtype, fun, fun, args, call, 1, __ssp_bos)
78#define __ssp_redirect0(rtype, fun, args, call) \
79 __ssp_redirect_raw(rtype, fun, fun, args, call, 1, __ssp_bos0)
80
81#define __ssp_overlap(a, b, l) \
82 (((a) <= (b) && (b) < (a) + (l)) || ((b) <= (a) && (a) < (b) + (l)))
83
84__BEGIN_DECLS
85void __stack_chk_fail(void) __dead;
86void __chk_fail(void) __dead;
87__END_DECLS
88
89#endif /* _SSP_SSP_H_ */
