31#ifndef _SSP_STDIO_H_
32#define _SSP_STDIO_H_
33
34#include <ssp/ssp.h>
35
36__BEGIN_DECLS
37int __sprintf_chk(char *__restrict, int, size_t, const char *__restrict, ...)
38 __printflike(4, 5);
39int __vsprintf_chk(char *__restrict, int, size_t, const char *__restrict,
40 __va_list)
41 __printflike(4, 0);
42int __snprintf_chk(char *__restrict, size_t, int, size_t,
43 const char *__restrict, ...)
44 __printflike(5, 6);
45int __vsnprintf_chk(char *__restrict, size_t, int, size_t,
46 const char *__restrict, __va_list)
47 __printflike(5, 0);
48char *__gets_chk(char *, size_t);
49char *__fgets_chk(char *, int, size_t, FILE *);
50__END_DECLS
51
52#if __SSP_FORTIFY_LEVEL > 0
53
54
55#define sprintf(str, ...) \
56 __builtin___sprintf_chk(str, 0, __ssp_bos(str), __VA_ARGS__)
57
58#define vsprintf(str, fmt, ap) \
59 __builtin___vsprintf_chk(str, 0, __ssp_bos(str), fmt, ap)
60
61#define snprintf(str, len, ...) \
62 __builtin___snprintf_chk(str, len, 0, __ssp_bos(str), __VA_ARGS__)
63
64#define vsnprintf(str, len, fmt, ap) \
65 __builtin___vsnprintf_chk(str, len, 0, __ssp_bos(str), fmt, ap)
66
67#define gets(str) \
68 __gets_chk(str, __ssp_bos(str))
69
70#define fgets(str, len, fp) \
71 __fgets_chk(str, len, __ssp_bos(str), fp)
72#endif /* __SSP_FORTIFY_LEVEL > 0 */
73
74#endif /* _SSP_STDIO_H_ */
