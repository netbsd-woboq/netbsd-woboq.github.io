38void *__memcpy_chk(void *, const void *, size_t, size_t);
39void *__memmove_chk(void *, void *, size_t, size_t);
40void *__memset_chk(void *, int, size_t, size_t);
41char *__stpcpy_chk(char *, const char *, size_t);
42char *__strcat_chk(char *, const char *, size_t);
43char *__strcpy_chk(char *, const char *, size_t);

48#if __SSP_FORTIFY_LEVEL > 0
49
50#define __ssp_bos_check3(fun, dst, src, len) \
51 ((__ssp_bos0(dst) != (size_t)-1) ? \
52 __builtin___ ## fun ## _chk(dst, src, len, __ssp_bos0(dst)) : \
53 __ ## fun ## _ichk(dst, src, len))
54
55#define __ssp_bos_check2(fun, dst, src) \
56 ((__ssp_bos0(dst) != (size_t)-1) ? \
57 __builtin___ ## fun ## _chk(dst, src, __ssp_bos0(dst)) : \
58 __ ## fun ## _ichk(dst, src))
59
60#define __ssp_bos_icheck3_restrict(fun, type1, type2) \
61static __inline type1 __ ## fun ## _ichk(type1 __restrict, type2 __restrict, size_t); \
62static __inline __attribute__((__always_inline__)) type1 \
63__ ## fun ## _ichk(type1 __restrict dst, type2 __restrict src, size_t len) { \
64 return __builtin___ ## fun ## _chk(dst, src, len, __ssp_bos0(dst)); \
65}
66
67#define __ssp_bos_icheck3(fun, type1, type2) \
68static __inline type1 __ ## fun ## _ichk(type1, type2, size_t); \
69static __inline __attribute__((__always_inline__)) type1 \
70__ ## fun ## _ichk(type1 dst, type2 src, size_t len) { \
71 return __builtin___ ## fun ## _chk(dst, src, len, __ssp_bos0(dst)); \
72}
73
74#define __ssp_bos_icheck2_restrict(fun, type1, type2) \
75static __inline type1 __ ## fun ## _ichk(type1, type2); \
76static __inline __attribute__((__always_inline__)) type1 \
77__ ## fun ## _ichk(type1 __restrict dst, type2 __restrict src) { \
78 return __builtin___ ## fun ## _chk(dst, src, __ssp_bos0(dst)); \
79}
80
81__BEGIN_DECLS
82__ssp_bos_icheck3_restrict(memcpy, void *, const void *)
83__ssp_bos_icheck3(memmove, void *, const void *)
84__ssp_bos_icheck3(memset, void *, int)
85__ssp_bos_icheck2_restrict(stpcpy, char *, const char *)
86#if __GNUC_PREREQ__(4,8) || defined(__clang__)
87__ssp_bos_icheck3_restrict(stpncpy, char *, const char *)
88#endif
89__ssp_bos_icheck2_restrict(strcpy, char *, const char *)
90__ssp_bos_icheck2_restrict(strcat, char *, const char *)
91__ssp_bos_icheck3_restrict(strncpy, char *, const char *)
92__ssp_bos_icheck3_restrict(strncat, char *, const char *)
93__END_DECLS
94
95#define memcpy(dst, src, len) __ssp_bos_check3(memcpy, dst, src, len)
96#define memmove(dst, src, len) __ssp_bos_check3(memmove, dst, src, len)
97#define memset(dst, val, len) __ssp_bos_check3(memset, dst, val, len)
98#define stpcpy(dst, src) __ssp_bos_check2(stpcpy, dst, src)
99#if __GNUC_PREREQ__(4,8) || defined(__clang__)
100#define stpncpy(dst, src, len) __ssp_bos_check3(stpncpy, dst, src, len)
101#endif
102#define strcpy(dst, src) __ssp_bos_check2(strcpy, dst, src)
103#define strcat(dst, src) __ssp_bos_check2(strcat, dst, src)
104#define strncpy(dst, src, len) __ssp_bos_check3(strncpy, dst, src, len)
105#define strncat(dst, src, len) __ssp_bos_check3(strncat, dst, src, len)
106
107#endif /* __SSP_FORTIFY_LEVEL > 0 */
