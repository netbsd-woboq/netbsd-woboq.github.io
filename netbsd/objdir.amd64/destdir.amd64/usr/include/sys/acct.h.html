39#ifndef _SYS_ACCT_H_
40#define _SYS_ACCT_H_
41
42/*
43 * Accounting structures; these use a comp_t type which is a 3 bits base 8
44 * exponent, 13 bit fraction ``floating point'' number. Units are 1/AHZ
45 * seconds.
46 */
47typedef uint16_t comp_t;
48
49struct acct {
50 char ac_comm[16]; /* command name */
51 comp_t ac_utime; /* user time */
52 comp_t ac_stime; /* system time */
53 comp_t ac_etime; /* elapsed time */
54 time_t ac_btime; /* starting time */
55 uid_t ac_uid; /* user id */
56 gid_t ac_gid; /* group id */
57 uint16_t ac_mem; /* average memory usage */
58 comp_t ac_io; /* count of IO blocks */
59 dev_t ac_tty; /* controlling tty */
60
61#define AFORK 0x01 /* fork'd but not exec'd */
62#define ASU 0x02 /* used super-user permissions */
63#define ACOMPAT 0x04 /* used compatibility mode */
64#define ACORE 0x08 /* dumped core */
65#define AXSIG 0x10 /* killed by a signal */
66 uint8_t ac_flag; /* accounting flags */
67};
68
69/*
70 * 1/AHZ is the granularity of the data encoded in the comp_t fields.
71 * This is not necessarily equal to hz.
72 */
73#define AHZ 64
74
75#ifdef _KERNEL
76void acct_init(void);
77int acct_process(struct lwp *);
78#endif
79
80#endif /* !_SYS_ACCT_H_ */
