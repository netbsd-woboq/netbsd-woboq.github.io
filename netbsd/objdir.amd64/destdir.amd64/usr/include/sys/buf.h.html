95/*
96 * The buffer header describes an I/O operation in the kernel.
97 *
98 * Field markings and the corresponding locks:
99 *
100 * b thread of execution that holds BC_BUSY, does not correspond
101 * directly to any particular LWP
102 * c bufcache_lock
103 * o b_objlock
104 *
105 * For buffers associated with a vnode, b_objlock points to vp->v_interlock.
106 * If not associated with a vnode, it points to the generic buffer_lock.
107 */
108
109/* required for the conditional union member below to be ~safe */
110#if defined(_KERNEL)
111__CTASSERT(sizeof(struct work) <= sizeof(TAILQ_ENTRY(buf)));
112#endif
113
114struct buf {
115 union {
116 TAILQ_ENTRY(buf) u_actq;
117 rb_node_t u_rbnode;
118#if defined(_KERNEL)
119 /* u_work is smaller than u_actq */
120 struct work u_work;
121#endif
122 } b_u; /* b: device driver queue */
123#define b_actq b_u.u_actq
124#define b_work b_u.u_work
125 void (*b_iodone)(struct buf *);/* b: call when done */
126 int b_error; /* b: errno value. */
127 int b_resid; /* b: remaining I/O. */
128 u_int b_flags; /* b: B_* flags */
129 int b_prio; /* b: priority for queue */
130 int b_bufsize; /* b: allocated size */
131 int b_bcount; /* b: valid bytes in buffer */
132 dev_t b_dev; /* b: associated device */
133 void *b_data; /* b: fs private data */
134 daddr_t b_blkno; /* b: physical block number
135 (partition relative) */
136 daddr_t b_rawblkno; /* b: raw physical block number
137 (volume relative) */
138 struct proc *b_proc; /* b: proc if BB_PHYS */
139 void *b_saveaddr; /* b: saved b_data for physio */
140 struct cpu_info *b_ci; /* b: originating CPU */
141
142 /*
143 * b: private data for owner.
144 * - buffer cache buffers are owned by corresponding filesystem.
145 * - non-buffer cache buffers are owned by subsystem which
146 * allocated them. (filesystem, disk driver, etc)
147 */
148 void *b_private;
149 off_t b_dcookie; /* NFS: Offset cookie if dir block */
150
151 kcondvar_t b_busy; /* c: threads waiting on buf */
152 u_int b_refcnt; /* c: refcount for b_busy */
153 void *b_unused; /* : unused */
154 LIST_ENTRY(buf) b_hash; /* c: hash chain */
155 LIST_ENTRY(buf) b_vnbufs; /* c: associated vnode */
156 TAILQ_ENTRY(buf) b_freelist; /* c: position if not active */
157 TAILQ_ENTRY(buf) b_wapbllist; /* c: transaction buffer list */
158 daddr_t b_lblkno; /* c: logical block number */
159 int b_freelistindex;/* c: free list index (BQ_) */
160 u_int b_cflags; /* c: BC_* flags */
161 struct vnode *b_vp; /* c: file vnode */
162
163 kcondvar_t b_done; /* o: waiting on completion */
164 u_int b_oflags; /* o: BO_* flags */
165 kmutex_t *b_objlock; /* o: completion lock */
166};

168/*
169 * For portability with historic industry practice, the cylinder number has
170 * to be maintained in the `b_resid' field.
171 */
172#define b_cylinder b_resid /* Cylinder number for disksort(). */
173
174/*
175 * These flags are kept in b_cflags (owned by buffer cache).
176 */
177#define BC_AGE 0x00000001 /* Move to age queue when I/O done. */
178#define BC_BUSY 0x00000010 /* I/O in progress. */
179#define BC_INVAL 0x00002000 /* Does not contain valid info. */
180#define BC_NOCACHE 0x00008000 /* Do not cache block after use. */
181#define BC_WANTED 0x00800000 /* Process wants this buffer. */
182#define BC_VFLUSH 0x04000000 /* Buffer is being synced. */
183
184/*
185 * These flags are kept in b_oflags (owned by associated object).
186 */
187#define BO_DELWRI 0x00000080 /* Delay I/O until buffer reused. */
188#define BO_DONE 0x00000200 /* I/O completed. */
189
190/*
191 * These flags are kept in b_flags (owned by buffer holder).
192 */
193#define B_WRITE 0x00000000 /* Write buffer (pseudo flag). */
194#define B_ASYNC 0x00000004 /* Start I/O, do not wait. */
195#define B_COWDONE 0x00000400 /* Copy-on-write already done. */
196#define B_GATHERED 0x00001000 /* LFS: already in a segment. */
197#define B_LOCKED 0x00004000 /* Locked in core (not reusable). */
198#define B_PHYS 0x00040000 /* I/O to user memory. */
199#define B_RAW 0x00080000 /* Set by physio for raw transfers. */
200#define B_READ 0x00100000 /* Read buffer. */
201#define B_DEVPRIVATE 0x02000000 /* Device driver private flag. */
202#define B_MEDIA_FUA 0x08000000 /* Set Force Unit Access for media. */
203#define B_MEDIA_DPO 0x10000000 /* Set Disable Page Out for media. */
204
205#define BUF_FLAGBITS \
206 "\20\1AGE\3ASYNC\4BAD\5BUSY\10DELWRI" \
207 "\12DONE\13COWDONE\15GATHERED\16INVAL\17LOCKED\20NOCACHE" \
208 "\23PHYS\24RAW\25READ\32DEVPRIVATE\33VFLUSH\34MEDIA_FUA\35MEDIA_DPO"
209
210/* Avoid weird code due to B_WRITE being a "pseudo flag" */
211#define BUF_ISREAD(bp) (((bp)->b_flags & B_READ) == B_READ)
212#define BUF_ISWRITE(bp) (((bp)->b_flags & B_READ) == B_WRITE)
213
214/* Media flags, to be passed for nested I/O */
215#define B_MEDIA_FLAGS (B_MEDIA_FUA|B_MEDIA_DPO)
216
217/*
218 * This structure describes a clustered I/O. It is stored in the b_saveaddr
219 * field of the buffer on which I/O is done. At I/O completion, cluster
220 * callback uses the structure to parcel I/O's to individual buffers, and
221 * then free's this structure.
222 */
223struct cluster_save {
224 long bs_bcount; /* Saved b_bcount. */
225 long bs_bufsize; /* Saved b_bufsize. */
226 void *bs_saveaddr; /* Saved b_addr. */
227 int bs_nchildren; /* Number of associated buffers. */
228 struct buf *bs_children; /* List of associated buffers. */
229};

231/*
232 * Zero out the buffer's data area.
233 */
234#define clrbuf(bp) \
235do { \
236 memset((bp)->b_data, 0, (u_int)(bp)->b_bcount); \
237 (bp)->b_resid = 0; \
238} while (/* CONSTCOND */ 0)

240/* Flags to low-level allocation routines. */
241#define B_CLRBUF 0x01 /* Request allocated buffer be cleared. */
242#define B_SYNC 0x02 /* Do all allocations synchronously. */
243#define B_METAONLY 0x04 /* Return indirect block buffer. */
244#define B_CONTIG 0x08 /* Allocate file contiguously. */
245
246/* Flags to bread() and breadn(). */
247#define B_MODIFY 0x01 /* Hint: caller might modify buffer */
