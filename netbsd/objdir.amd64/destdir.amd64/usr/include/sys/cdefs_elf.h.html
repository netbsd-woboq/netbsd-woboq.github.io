197/*
198 * On multiprocessor systems we can gain an improvement in performance
199 * by being mindful of which cachelines data is placed in.
200 *
201 * __read_mostly:
202 *
203 * It makes sense to ensure that rarely modified data is not
204 * placed in the same cacheline as frequently modified data.
205 * To mitigate the phenomenon known as "false-sharing" we
206 * can annotate rarely modified variables with __read_mostly.
207 * All such variables are placed into the .data.read_mostly
208 * section in the kernel ELF.
209 *
210 * Prime candidates for __read_mostly annotation are variables
211 * which are hardly ever modified and which are used in code
212 * hot-paths, e.g. pmap_initialized.
213 *
214 * __cacheline_aligned:
215 *
216 * Some data structures (mainly locks) benefit from being aligned
217 * on a cacheline boundary, and having a cacheline to themselves.
218 * This way, the modification of other data items cannot adversely
219 * affect the lock and vice versa.
220 *
221 * Any variables annotated with __cacheline_aligned will be
222 * placed into the .data.cacheline_aligned ELF section.
223 */
224#define __read_mostly \
225 __attribute__((__section__(".data.read_mostly")))
226
227#define __cacheline_aligned \
228 __attribute__((__aligned__(COHERENCY_UNIT), \
229 __section__(".data.cacheline_aligned")))
