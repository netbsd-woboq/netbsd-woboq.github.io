43/* 7.18.1.1 Exact-width integer types */
44
45typedef __INT8_TYPE__ __int8_t;
46typedef __UINT8_TYPE__ __uint8_t;
47typedef __INT16_TYPE__ __int16_t;
48typedef __UINT16_TYPE__ __uint16_t;
49typedef __INT32_TYPE__ __int32_t;
50typedef __UINT32_TYPE__ __uint32_t;
51typedef __INT64_TYPE__ __int64_t;
52typedef __UINT64_TYPE__ __uint64_t;

54#define __BIT_TYPES_DEFINED__
55
56/* 7.18.1.4 Integer types capable of holding object pointers */
57
58typedef __INTPTR_TYPE__ __intptr_t;
59typedef __UINTPTR_TYPE__ __uintptr_t;
