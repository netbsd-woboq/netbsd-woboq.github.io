305/*
306 * `configuration' attachment and driver (what the machine-independent
307 * autoconf uses). As devices are found, they are applied against all
308 * the potential matches. The one with the best match is taken, and a
309 * device structure (plus any other data desired) is allocated. Pointers
310 * to these are placed into an array of pointers. The array itself must
311 * be dynamic since devices can be found long after the machine is up
312 * and running.
313 *
314 * Devices can have multiple configuration attachments if they attach
315 * to different attributes (busses, or whatever), to allow specification
316 * of multiple match and attach functions. There is only one configuration
317 * driver per driver, so that things like unit numbers and the device
318 * structure array will be shared.
319 */
320struct cfattach {
321 const char *ca_name; /* name of attachment */
322 LIST_ENTRY(cfattach) ca_list; /* link on cfdriver's list */
323 size_t ca_devsize; /* size of dev data (for alloc) */
324 int ca_flags; /* flags for driver allocation etc */
325 int (*ca_match)(device_t, cfdata_t, void *);
326 void (*ca_attach)(device_t, device_t, void *);
327 int (*ca_detach)(device_t, int);
328 int (*ca_activate)(device_t, devact_t);
329 /* technically, the next 2 belong into "struct cfdriver" */
330 int (*ca_rescan)(device_t, const char *,
331 const int *); /* scan for new children */
332 void (*ca_childdetached)(device_t, device_t);
333};
334LIST_HEAD(cfattachlist, cfattach);
335
336#define CFATTACH_DECL3_NEW(name, ddsize, matfn, attfn, detfn, actfn, \
337 rescanfn, chdetfn, __flags) \
338struct cfattach __CONCAT(name,_ca) = { \
339 .ca_name = ___STRING(name), \
340 .ca_devsize = ddsize, \
341 .ca_flags = (__flags) | DVF_PRIV_ALLOC, \
342 .ca_match = matfn, \
343 .ca_attach = attfn, \
344 .ca_detach = detfn, \
345 .ca_activate = actfn, \
346 .ca_rescan = rescanfn, \
347 .ca_childdetached = chdetfn, \
348}
349
350#define CFATTACH_DECL2_NEW(name, ddsize, matfn, attfn, detfn, actfn, \
351 rescanfn, chdetfn) \
352 CFATTACH_DECL3_NEW(name, ddsize, matfn, attfn, detfn, actfn, \
353 rescanfn, chdetfn, 0)
354
355#define CFATTACH_DECL_NEW(name, ddsize, matfn, attfn, detfn, actfn) \
356 CFATTACH_DECL2_NEW(name, ddsize, matfn, attfn, detfn, actfn, NULL, NULL)

399/*
400 * Configuration printing functions, and their return codes. The second
401 * argument is NULL if the device was configured; otherwise it is the name
402 * of the parent device. The return value is ignored if the device was
403 * configured, so most functions can return UNCONF unconditionally.
404 */
405typedef int (*cfprint_t)(void *, const char *); /* XXX const char * */
406#define QUIET 0 /* print nothing */
407#define UNCONF 1 /* print " not configured\n" */
408#define UNSUPP 2 /* print " not supported\n" */
