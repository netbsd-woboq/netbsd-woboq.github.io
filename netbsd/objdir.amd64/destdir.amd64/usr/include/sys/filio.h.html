39#ifndef _SYS_FILIO_H_
40#define _SYS_FILIO_H_
41
42#include <sys/ioccom.h>
43
44/* Generic file-descriptor ioctl's. */
45#define FIOCLEX _IO('f', 1) /* set close on exec on fd */
46#define FIONCLEX _IO('f', 2) /* remove close on exec */
47/* Handle lseek SEEK_DATA and SEEK_HOLE for holey file knowledge. */
48#define FIOSEEKDATA _IOWR('f', 97, off_t) /* SEEK_DATA */
49#define FIOSEEKHOLE _IOWR('f', 98, off_t) /* SEEK_HOLE */
50#define FIONREAD _IOR('f', 127, int) /* get # bytes to read */
51#define FIONBIO _IOW('f', 126, int) /* set/clear non-blocking i/o */
52#define FIOASYNC _IOW('f', 125, int) /* set/clear async i/o */
53#define FIOSETOWN _IOW('f', 124, int) /* set owner */
54#define FIOGETOWN _IOR('f', 123, int) /* get owner */
55#define OFIOGETBMAP _IOWR('f', 122, uint32_t) /* get underlying block no. */
56#define FIOGETBMAP _IOWR('f', 122, daddr_t) /* get underlying block no. */
57#define FIONWRITE _IOR('f', 121, int) /* get # bytes outstanding
58 * in send queue. */
59#define FIONSPACE _IOR('f', 120, int) /* get space in send queue. */
60
61/* Ugly symbol for compatibility with other operating systems */
62#define FIBMAP FIOGETBMAP
63
64#endif /* !_SYS_FILIO_H_ */
