29#ifndef _SYS_KSEM_H_
30#define _SYS_KSEM_H_
31
32#include <sys/cdefs.h>
33
34struct timespec;
35
36#ifdef _KERNEL
37#define KSEM_MAX 128
38
39typedef struct ksem {
40 LIST_ENTRY(ksem) ks_entry; /* global list entry */
41 struct proc * ks_pshared_proc;/* owner of pshared sem */
42 intptr_t ks_pshared_id; /* global id for pshared sem */
43 kmutex_t ks_lock; /* lock on this ksem */
44 kcondvar_t ks_cv; /* condition variable */
45 u_int ks_pshared_fd; /* fd in owning proc */
46 u_int ks_ref; /* number of references */
47 u_int ks_value; /* current value */
48 u_int ks_waiters; /* number of waiters */
49 char * ks_name; /* name, if named */
50 size_t ks_namelen; /* length of name */
51 int ks_flags; /* for KS_UNLINKED */
52 mode_t ks_mode; /* protection bits */
53 uid_t ks_uid; /* creator uid */
54 gid_t ks_gid; /* creator gid */
55} ksem_t;
