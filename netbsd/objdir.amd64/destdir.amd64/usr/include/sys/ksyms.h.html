42struct ksyms_symtab {
43 TAILQ_ENTRY(ksyms_symtab) sd_queue; /* All active tables */
44 const char *sd_name; /* Name of this table */
45 Elf_Sym *sd_symstart; /* Address of symbol table */
46 uintptr_t sd_minsym; /* symbol with minimum value */
47 uintptr_t sd_maxsym; /* symbol with maximum value */
48 char *sd_strstart; /* Address of corresponding string table */
49 int sd_usroffset; /* Real address for userspace */
50 int sd_symsize; /* Size in bytes of symbol table */
51 int sd_strsize; /* Size of string table */
52 int sd_nglob; /* Number of global symbols */
53 bool sd_gone; /* dead but around for open() */
54 void *sd_ctfstart; /* Address of CTF contents */
55 int sd_ctfsize; /* Size in bytes of CTF contents */
56 uint32_t *sd_nmap; /* Name map for sorted symbols */
57 int sd_nmapsize; /* Total span of map */
58};
59
60/*
61 * Static allocated ELF header.
62 * Basic info is filled in at attach, sizes at open.
63 */
64#define SHNOTE 1
65#define SYMTAB 2
66#define STRTAB 3
67#define SHSTRTAB 4
68#define SHBSS 5
69#define SHCTF 6
70#define NSECHDR 7
71
72#define NPRGHDR 1
73#define SHSTRSIZ 64
74
75struct ksyms_hdr {
76 Elf_Ehdr kh_ehdr;
77 Elf_Phdr kh_phdr[NPRGHDR];
78 Elf_Shdr kh_shdr[NSECHDR];
79 char kh_strtab[SHSTRSIZ];
80 /* 0=NameSize, 1=DescSize, 2=Tag, 3="NetB", 4="SD\0\0", 5=Version */
81 int32_t kh_note[6];
82};
83#endif /* _KSYMS_PRIVATE */
84
85/*
86 * Do a lookup of a symbol using the in-kernel lookup algorithm.
87 */
88struct ksyms_ogsymbol {
89 const char *kg_name;
90 union {
91 void *ku_sym; /* Normally Elf_Sym */
92 unsigned long *ku_value;
93 } _un;
94#define kg_sym _un.ku_sym
95#define kg_value _un.ku_value
96};
97
98#ifdef ELFSIZE
99struct ksyms_gsymbol {
100 const char *kg_name;
101 union {
102 Elf_Sym ku_sym;
103 } _un;
104};
105#endif
106
107struct ksyms_gvalue {
108 const char *kv_name;
109 uint64_t kv_value;
110};
111
112#define OKIOCGSYMBOL _IOW('l', 1, struct ksyms_ogsymbol)
113#define OKIOCGVALUE _IOW('l', 2, struct ksyms_ogsymbol)
114#define KIOCGSIZE _IOR('l', 3, int)
115#define KIOCGVALUE _IOWR('l', 4, struct ksyms_gvalue)
116#define KIOCGSYMBOL _IOWR('l', 5, struct ksyms_gsymbol)
