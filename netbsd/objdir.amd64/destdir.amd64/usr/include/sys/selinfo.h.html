64#ifndef _SYS_SELINFO_H_
65#define _SYS_SELINFO_H_
66
67#include <sys/event.h> /* for struct klist */
68
69/*
70 * Used to maintain information about processes that wish to be
71 * notified when I/O becomes possible.
72 */
73struct selinfo {
74 struct klist sel_klist; /* knotes attached to this selinfo */
75 void *sel_cluster; /* current cluster association */
76 struct lwp *sel_lwp; /* first LWP to be notified */
77 uintptr_t sel_fdinfo; /* selected descriptor by first LWP */
78 SLIST_ENTRY(selinfo) sel_chain; /* entry on LWP's list of selinfo */
79 uint32_t sel_collision; /* mask of colliding cpus */
80 uint32_t sel_reserved[3];/* reserved for future expansion */
81};
82
83#endif /* !_SYS_SELINFO_H_ */
