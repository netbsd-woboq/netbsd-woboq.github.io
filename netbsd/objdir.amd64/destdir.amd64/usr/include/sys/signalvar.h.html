52/*
53 * Process signal actions, possibly shared between processes.
54 */
55struct sigacts {
56 struct sigact_sigdesc {
57 struct sigaction sd_sigact;
58 const void *sd_tramp;
59 int sd_vers;
60 } sa_sigdesc[NSIG]; /* disposition of signals */
61
62 int sa_refcnt; /* reference count */
63 kmutex_t sa_mutex; /* lock on sa_refcnt */
64};

74/*
75 * Process signal state.
76 */
77struct sigctx {
78 struct _ksiginfo ps_info; /* for core dump/debugger XXX */
79 int ps_lwp; /* for core dump/debugger XXX */
80 bool ps_faked; /* for core dump/debugger XXX */
81 void *ps_sigcode; /* address of signal trampoline */
82 sigset_t ps_sigignore; /* Signals being ignored. */
83 sigset_t ps_sigcatch; /* Signals being caught by user. */
84};

89/*
90 * get signal action for process and signal; currently only for current process
91 */
92#define SIGACTION(p, sig) (p->p_sigacts->sa_sigdesc[(sig)].sd_sigact)
93#define SIGACTION_PS(ps, sig) (ps->sa_sigdesc[(sig)].sd_sigact)

95/*
96 * Signal properties and actions.
97 * The array below categorizes the signals and their default actions
98 * according to the following properties:
99 */
100#define SA_KILL 0x0001 /* terminates process by default */
101#define SA_CORE 0x0002 /* ditto and coredumps */
102#define SA_STOP 0x0004 /* suspend process */
103#define SA_TTYSTOP 0x0008 /* ditto, from tty */
104#define SA_IGNORE 0x0010 /* ignore by default */
105#define SA_CONT 0x0020 /* continue if suspended */
106#define SA_CANTMASK 0x0040 /* non-maskable, catchable */
107#define SA_NORESET 0x0080 /* not reset when caught */
108#define SA_TOLWP 0x0100 /* to LWP that generated, if local */
109#define SA_TOALL 0x0200 /* always to all LWPs */

196/*
197 * firstsig:
198 *
199 * Return the first signal in a signal set.
200 */
201static __inline int
202firstsig(const sigset_t *ss)
203{
204 int sig;
205
206 sig = ffs(ss->__bits[0]);
207 if (sig != 0)
208 return (sig);
209#if NSIG > 33
210 sig = ffs(ss->__bits[1]);
211 if (sig != 0)
212 return (sig + 32);
213#endif
214#if NSIG > 65
215 sig = ffs(ss->__bits[2]);
216 if (sig != 0)
217 return (sig + 64);
218#endif
219#if NSIG > 97
220 sig = ffs(ss->__bits[3]);
221 if (sig != 0)
222 return (sig + 96);
223#endif
224 return (0);
225}
