31#ifndef _SYS_TLS_H_
32#define _SYS_TLS_H_
33
34#include <sys/cdefs.h>
35#include <machine/types.h>
36
37#if defined(__HAVE_TLS_VARIANT_I) || defined(__HAVE_TLS_VARIANT_II)
38
39#if defined(__HAVE_TLS_VARIANT_I) && defined(__HAVE_TLS_VARIANT_II)
40#error Only one TLS variant can be supported at a time
41#endif
42
43struct tls_tcb {
44#ifdef __HAVE_TLS_VARIANT_I
45 void **tcb_dtv;
46 void *tcb_pthread;
47#else
48 void *tcb_self;
49 void **tcb_dtv;
50 void *tcb_pthread;
51#endif
52};
53
54__BEGIN_PUBLIC_DECLS
55struct tls_tcb *_rtld_tls_allocate(void);
56void _rtld_tls_free(struct tls_tcb *);
57__END_PUBLIC_DECLS
58#endif /* __HAVE_TLS_VARIANT_I || __HAVE_TLS_VARIANT_II */
59
60#endif /* _SYS_TLS_H_ */
