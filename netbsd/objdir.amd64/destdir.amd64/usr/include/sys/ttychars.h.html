34/*
35 * 4.3 COMPATIBILITY FILE
36 *
37 * User visible structures and constants related to terminal handling.
38 */
39#ifndef _SYS_TTYCHARS_H_
40#define _SYS_TTYCHARS_H_
41
42struct ttychars {
43 char tc_erase; /* erase last character */
44 char tc_kill; /* erase entire line */
45 char tc_intrc; /* interrupt */
46 char tc_quitc; /* quit */
47 char tc_startc; /* start output */
48 char tc_stopc; /* stop output */
49 char tc_eofc; /* end-of-file */
50 char tc_brkc; /* input delimiter (like nl) */
51 char tc_suspc; /* stop process signal */
52 char tc_dsuspc; /* delayed stop process signal */
53 char tc_rprntc; /* reprint line */
54 char tc_flushc; /* flush output (toggles) */
55 char tc_werasc; /* word erase */
56 char tc_lnextc; /* literal next character */
57};
58#endif /* !_SYS_TTYCHARS_H_ */
