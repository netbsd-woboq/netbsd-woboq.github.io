39/*
40 * System wide defaults for terminal state.
41 */
42#ifndef _SYS_TTYDEFAULTS_H_
43#define _SYS_TTYDEFAULTS_H_
44
45/*
46 * Defaults on "first" open.
47 */
48#define TTYDEF_IFLAG (BRKINT | ICRNL | IMAXBEL | IXON | IXANY)
49#define TTYDEF_OFLAG (OPOST | ONLCR | OXTABS)
50#define TTYDEF_LFLAG (ECHO | ICANON | ISIG | IEXTEN | ECHOE|ECHOKE|ECHOCTL)
51#define TTYDEF_CFLAG (CREAD | CS8 | HUPCL)
52#define TTYDEF_SPEED (B9600)
53
54/*
55 * Control Character Defaults
56 */
57#define CTRL(x) (x&037)
58#define CEOF CTRL('d')
59#define CEOL ((unsigned char)'\377') /* XXX avoid _POSIX_VDISABLE */
60#define CERASE 0177
61#define CINTR CTRL('c')
62#define CSTATUS CTRL('t')
63#define CKILL CTRL('u')
64#define CMIN 1
65#define CQUIT 034 /* FS, ^\ */
66#define CSUSP CTRL('z')
67#define CTIME 0
68#define CDSUSP CTRL('y')
69#define CSTART CTRL('q')
70#define CSTOP CTRL('s')
71#define CLNEXT CTRL('v')
72#define CDISCARD CTRL('o')
73#define CWERASE CTRL('w')
74#define CREPRINT CTRL('r')
75#define CEOT CEOF
76/* compat */
77#define CBRK CEOL
78#define CRPRNT CREPRINT
79#define CFLUSH CDISCARD
80
81/* PROTECTED INCLUSION ENDS HERE */
82#endif /* !_SYS_TTYDEFAULTS_H_ */
83
84/*
85 * #define TTYDEFCHARS to include an array of default control characters.
86 */
87#ifdef _KERNEL
88#ifdef TTYDEFCHARS
89const cc_t ttydefchars[NCCS] = {
90 [VEOF] = CEOF,
91 [VEOL] = CEOL,
92 [VEOL2] = CEOL,
93 [VERASE] = CERASE,
94 [VWERASE] = CWERASE,
95 [VKILL] = CKILL,
96 [VREPRINT] = CREPRINT,
97 [7] = _POSIX_VDISABLE, /* spare */
98 [VINTR] = CINTR,
99 [VQUIT] = CQUIT,
100 [VSUSP] = CSUSP,
101 [VDSUSP] = CDSUSP,
102 [VSTART] = CSTART,
103 [VSTOP] = CSTOP,
104 [VLNEXT] = CLNEXT,
105 [VDISCARD] = CDISCARD,
106 [VMIN] = CMIN,
107 [VTIME] = CTIME,
108 [VSTATUS] = CSTATUS,
109 [19] = _POSIX_VDISABLE, /* spare */
110};
111#undef TTYDEFCHARS
112#else
113extern const cc_t ttydefchars[NCCS];
114#endif
115#endif /* _KERNEL */
