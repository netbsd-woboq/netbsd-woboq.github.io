34#ifndef _SYS_UCRED_H_
35#define _SYS_UCRED_H_
36
37#ifdef _KERNEL
38#include <sys/syslimits.h>
39#else
40#include <limits.h>
41#endif
42
43/*
44 * Credentials.
45 */
46
47/* Userland's view of credentials. This should not change */
48struct uucred {
49 unsigned short cr_unused; /* not used, compat */
50 uid_t cr_uid; /* effective user id */
51 gid_t cr_gid; /* effective group id */
52 short cr_ngroups; /* number of groups */
53 gid_t cr_groups[NGROUPS_MAX]; /* groups */
54};
55
56#endif /* !_SYS_UCRED_H_ */
