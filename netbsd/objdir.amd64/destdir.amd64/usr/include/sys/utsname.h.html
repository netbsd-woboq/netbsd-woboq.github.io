37#ifndef _SYS_UTSNAME_H_
38#define _SYS_UTSNAME_H_
39
40#include <sys/featuretest.h>
41
42#define _SYS_NMLN 256
43
44#if defined(_NETBSD_SOURCE)
45#define SYS_NMLN _SYS_NMLN
46#endif
47
48struct utsname {
49 char sysname[_SYS_NMLN]; /* Name of this OS. */
50 char nodename[_SYS_NMLN]; /* Name of this network node. */
51 char release[_SYS_NMLN]; /* Release level. */
52 char version[_SYS_NMLN]; /* Version level. */
53 char machine[_SYS_NMLN]; /* Hardware type. */
54};
55
56#include <sys/cdefs.h>
57
58__BEGIN_DECLS
59int uname(struct utsname *);
60__END_DECLS
61
62#endif /* !_SYS_UTSNAME_H_ */
