31#ifndef _SYS_UUID_H_
32#define _SYS_UUID_H_
33
34#include <sys/cdefs.h>
35
36/* Length of a node address (an IEEE 802 address). */
37#define _UUID_NODE_LEN 6
38
39/* Length of a printed UUID. */
40#define _UUID_STR_LEN 38
41
42/*
43 * See also:
44 * http://www.opengroup.org/dce/info/draft-leach-uuids-guids-01.txt
45 * http://www.opengroup.org/onlinepubs/009629399/apdxa.htm
46 *
47 * A DCE 1.1 compatible source representation of UUIDs.
48 */
49struct uuid {
50 uint32_t time_low;
51 uint16_t time_mid;
52 uint16_t time_hi_and_version;
53 uint8_t clock_seq_hi_and_reserved;
54 uint8_t clock_seq_low;
55 uint8_t node[_UUID_NODE_LEN];
56};
57
58#ifdef _KERNEL
59
60#define UUID_NODE_LEN _UUID_NODE_LEN
61#define UUID_STR_LEN _UUID_STR_LEN
62
63int uuid_snprintf(char *, size_t, const struct uuid *);
64int uuid_printf(const struct uuid *);
65void uuid_dec_be(const void *, struct uuid *);
66void uuid_dec_le(const void *, struct uuid *);
67void uuid_enc_be(void *, const struct uuid *);
68void uuid_enc_le(void *, const struct uuid *);
69int uuidgen(struct uuid *, int);
70
71#else /* _KERNEL */
72
73typedef struct uuid uuid_t;
74
75__BEGIN_DECLS
76int uuidgen(struct uuid *, int);
77__END_DECLS
78
79#endif /* _KERNEL */
80
81#endif /* _SYS_UUID_H_ */
