76/*
77 * The root inode is the root of the file system. Inode 0 can't be used for
78 * normal purposes and bad blocks are normally linked to inode 1, thus
79 * the root inode is 2.
80 * Inode 3 to 10 are reserved in ext2fs.
81 */

83#define EXT2_ROOTINO ((ino_t)2)

86#define EXT2_BOOTLOADERINO ((ino_t)5)
87#define EXT2_UNDELDIRINO ((ino_t)6)
88#define EXT2_RESIZEINO ((ino_t)7)
89#define EXT2_JOURNALINO ((ino_t)8)

92/*
93 * A dinode contains all the meta-data associated with a UFS file.
94 * This structure defines the on-disk format of a dinode. Since
95 * this structure describes an on-disk structure, all its fields
96 * are defined by types with precise widths.
97 */
98
99/*
100 * XXX these are the same values as UFS_NDADDR/UFS_NIADDR and it is
101 * far from clear that there isn't code that relies on them being the
102 * same.
103 */
104#define EXT2FS_NDADDR 12 /* Direct addresses in inode. */
105#define EXT2FS_NIADDR 3 /* Indirect addresses in inode. */
106
107#define EXT2_MAXSYMLINKLEN ((EXT2FS_NDADDR+EXT2FS_NIADDR) * sizeof (uint32_t))
108#define E2MAXSYMLINKLEN EXT2_MAXSYMLINKLEN
109
110struct ext2fs_dinode {
111 uint16_t e2di_mode; /* 0: IFMT, permissions; see below. */
112 uint16_t e2di_uid; /* 2: Owner UID */
113 uint32_t e2di_size; /* 4: Size (in bytes) */
114 uint32_t e2di_atime; /* 8: Access time */
115 uint32_t e2di_ctime; /* 12: Create time */
116 uint32_t e2di_mtime; /* 16: Modification time */
117 uint32_t e2di_dtime; /* 20: Deletion time */
118 uint16_t e2di_gid; /* 24: Owner GID */
119 uint16_t e2di_nlink; /* 26: File link count */
120 uint32_t e2di_nblock; /* 28: Blocks count */
121 uint32_t e2di_flags; /* 32: Status flags (chflags) */
122 uint32_t e2di_version; /* 36: was reserved1 */
123 uint32_t e2di_blocks[EXT2FS_NDADDR+EXT2FS_NIADDR];
124 /* 40: disk blocks */
125 uint32_t e2di_gen; /* 100: generation number */
126 uint32_t e2di_facl; /* 104: file ACL (ext3) */
127 uint32_t e2di_size_high; /* 108: Size (in bytes) high */
128 uint32_t e2di_obso_faddr;/* 112: obsolete fragment address (ext2) */
129 uint16_t e2di_nblock_high; /* 116: Blocks count bits 47:32 (ext4) */
130 uint16_t e2di_facl_high; /* 118: file ACL bits 47:32 (ext4/64bit) */
131 uint16_t e2di_uid_high; /* 120: Owner UID top 16 bits (ext4) */
132 uint16_t e2di_gid_high; /* 122: Owner GID top 16 bits (ext4) */
133 uint16_t e2di_checksum_low; /* 124: crc LE (not implemented) (ext4) */
134 uint16_t e2di_reserved; /* 126: reserved */
135 uint16_t e2di_extra_isize; /* 128: inode extra size (over 128) actually used (ext4) */
136 uint16_t e2di_checksum_high; /* 130: crc BE (not implemented) (ext4) */
137 uint32_t e2di_ctime_extra; /* 132: ctime (nsec << 2 | high epoch) (ext4) */
138 uint32_t e2di_mtime_extra; /* 136: mtime (nsec << 2 | high epoch) (ext4) */
139 uint32_t e2di_atime_extra; /* 140: atime (nsec << 2 | high epoch) (ext4) */
140 uint32_t e2di_crtime; /* 144: creation time (epoch) (ext4) */
141 uint32_t e2di_crtime_extra; /* 148: creation time (nsec << 2 | high epoch) (ext4) */
142 uint32_t e2di_version_high; /* 152: version high (ext4) */
143 uint32_t e2di_projid; /* 156: project id (not implemented) (ext4) */
144};

213 (isize > EXT2_REV0_DINODE_SIZE) \

217/*
218 * Time encoding
219 * Lower two bits of extra field are extra high bits for epoch; unfortunately still, Linux kernels treat 11 there as 00 for compatibility
220 * Rest of extra fields are nanoseconds
221 */
222static __inline void
223ext2fs_dinode_time_get(struct timespec *ts, uint32_t epoch, uint32_t extra)
224{
225 ts->tv_sec = (signed) epoch;
226
227 if (extra) {
228 uint64_t epoch_bits = extra & 0x3;
229 /* XXX compatibility with linux kernel < 4.20 */
230 if (epoch_bits == 3 && ts->tv_sec < 0)
231 epoch_bits = 0;
232
233 ts->tv_sec |= epoch_bits << 32;
234
235 ts->tv_nsec = extra >> 2;
236 } else {
237 ts->tv_nsec = 0;
238 }
239}
240#define EXT2_DINODE_TIME_GET(ts, dinode, field, isize) \
241 ext2fs_dinode_time_get(ts, (dinode)->field, \
242 EXT2_DINODE_FITS(dinode, field ## _extra, isize) \
243 ? (dinode)->field ## _extra : 0 \
244 )
245
246static __inline void
247ext2fs_dinode_time_set(const struct timespec *ts, uint32_t *epoch, uint32_t *extra)
248{
249 *epoch = (int32_t) ts->tv_sec;
250
251 if (extra) {
252 uint32_t epoch_bits = (ts->tv_sec >> 32) & 0x3;
253
254 *extra = (ts->tv_nsec << 2) | epoch_bits;
255 }
256}
257#define EXT2_DINODE_TIME_SET(ts, dinode, field, isize) \
258 ext2fs_dinode_time_set(ts, &(dinode)->field, \
259 EXT2_DINODE_FITS(dinode, field ## _extra, isize) \
260 ? &(dinode)->field ## _extra : NULL \
261 )
