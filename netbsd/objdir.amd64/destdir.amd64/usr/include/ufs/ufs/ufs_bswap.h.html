28#ifndef _UFS_UFS_BSWAP_H_
29#define _UFS_UFS_BSWAP_H_
30
31#if defined(_KERNEL_OPT)
32#include "opt_ffs.h"
33#endif
34
35#include <sys/bswap.h>
36
37/* Macros to access UFS flags */
38#ifdef FFS_EI
39#define UFS_MPNEEDSWAP(ump) ((ump)->um_flags & UFS_NEEDSWAP)
40#define UFS_FSNEEDSWAP(fs) ((fs)->fs_flags & FS_SWAPPED)
41#define UFS_IPNEEDSWAP(ip) UFS_MPNEEDSWAP((ip)->i_ump)
42#else
43#define UFS_MPNEEDSWAP(ump) ((void)(ump), 0)
44#define UFS_FSNEEDSWAP(fs) ((void)(fs), 0)
45#define UFS_IPNEEDSWAP(ip) ((void)(ip), 0)
46#endif
47
48#if (!defined(_KERNEL) && !defined(NO_FFS_EI)) || defined(FFS_EI)
49/* inlines for access to swapped data */
50static __inline u_int16_t
51ufs_rw16(uint16_t a, int ns)
52{
53 return ((ns) ? bswap16(a) : (a));
54}
55
56static __inline u_int32_t
57ufs_rw32(uint32_t a, int ns)
58{
59 return ((ns) ? bswap32(a) : (a));
60}
61
62static __inline u_int64_t
63ufs_rw64(uint64_t a, int ns)
64{
65 return ((ns) ? bswap64(a) : (a));
66}
67#else
68static __inline u_int16_t
69ufs_rw16(uint16_t a, int ns)
70{
71 return a;
72}
73
74static __inline u_int32_t
75ufs_rw32(uint32_t a, int ns)
76{
77 return a;
78}
79
80static __inline u_int64_t
81ufs_rw64(uint64_t a, int ns)
82{
83 return a;
84}
85#endif
86
87#define ufs_add16(a, b, ns) \
88 (a) = ufs_rw16(ufs_rw16((a), (ns)) + (b), (ns))
89#define ufs_add32(a, b, ns) \
90 (a) = ufs_rw32(ufs_rw32((a), (ns)) + (b), (ns))
91#define ufs_add64(a, b, ns) \
92 (a) = ufs_rw64(ufs_rw64((a), (ns)) + (b), (ns))
93
94#endif /* !_UFS_UFS_BSWAP_H_ */
