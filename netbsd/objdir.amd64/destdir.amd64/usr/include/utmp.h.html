39#ifndef _UTMP_H_
40#define _UTMP_H_
41
42#include <sys/cdefs.h>
43#include <time.h> /* for time_t */
44
45#define _PATH_UTMP "/var/run/utmp"
46#define _PATH_WTMP "/var/log/wtmp"
47#define _PATH_LASTLOG "/var/log/lastlog"
48
49#define UT_NAMESIZE 8
50#define UT_LINESIZE 8
51#define UT_HOSTSIZE 16
52
53struct lastlog {
54 time_t ll_time;
55 char ll_line[UT_LINESIZE];
56 char ll_host[UT_HOSTSIZE];
57};
58
59struct utmp {
60 char ut_line[UT_LINESIZE];
61 char ut_name[UT_NAMESIZE];
62 char ut_host[UT_HOSTSIZE];
63 time_t ut_time;
64};
65
66__BEGIN_DECLS
67int utmpname(const char *);
68void setutent(void);
69#ifndef __LIBC12_SOURCE__
70struct utmp *getutent(void) __RENAME(__getutent50);
71#endif
72void endutent(void);
73__END_DECLS
74
75#endif /* !_UTMP_H_ */
