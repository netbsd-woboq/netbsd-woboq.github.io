30#ifndef _UVM_UVM_DEVICE_H_
31#define _UVM_UVM_DEVICE_H_
32
33/*
34 * uvm_device.h
35 *
36 * device handle into the VM system.
37 */
38
39/*
40 * the uvm_device structure. object is put at the top of the data structure.
41 * this allows:
42 * (struct uvm_device *) == (struct uvm_object *)
43 */
44
45struct uvm_device {
46 struct uvm_object u_obj; /* the actual VM object */
47 int u_flags; /* flags [LOCKED BY UDV_LOCK!] */
48 dev_t u_device; /* our device */
49 LIST_ENTRY(uvm_device) u_list; /* list of device objects */
50};
51
52/*
53 * u_flags values
54 */
55
56#define UVM_DEVICE_HOLD 0x1 /* someone has a "hold" on it */
57#define UVM_DEVICE_WANTED 0x2 /* someone wants to put a "hold" on */

67#endif /* _KERNEL */
68
69#endif /* _UVM_UVM_DEVICE_H_ */
