34#ifndef _VIS_H_
35#define _VIS_H_
36
37#include <sys/types.h>
38
39/*
40 * to select alternate encoding format
41 */
42#define VIS_OCTAL 0x0001 /* use octal \ddd format */
43#define VIS_CSTYLE 0x0002 /* use \[nrft0..] where appropiate */
44
45/*
46 * to alter set of characters encoded (default is to encode all
47 * non-graphic except space, tab, and newline).
48 */
49#define VIS_SP 0x0004 /* also encode space */
50#define VIS_TAB 0x0008 /* also encode tab */
51#define VIS_NL 0x0010 /* also encode newline */
52#define VIS_WHITE (VIS_SP | VIS_TAB | VIS_NL)
53#define VIS_SAFE 0x0020 /* only encode "unsafe" characters */
54#define VIS_DQ 0x8000 /* also encode double quotes */
55
56/*
57 * other
58 */
59#define VIS_NOSLASH 0x0040 /* inhibit printing '\' */
60#define VIS_HTTP1808 0x0080 /* http-style escape % hex hex */
61#define VIS_HTTPSTYLE 0x0080 /* http-style escape % hex hex */
62#define VIS_MIMESTYLE 0x0100 /* mime-style escape = HEX HEX */
63#define VIS_HTTP1866 0x0200 /* http-style &#num; or &string; */
64#define VIS_NOESCAPE 0x0400 /* don't decode `\' */
65#define _VIS_END 0x0800 /* for unvis */
66#define VIS_GLOB 0x1000 /* encode glob(3) magic characters */
67#define VIS_SHELL 0x2000 /* encode shell special characters [not glob] */
68#define VIS_META (VIS_WHITE | VIS_GLOB | VIS_SHELL)
69#define VIS_NOLOCALE 0x4000 /* encode using the C locale */
70
71/*
72 * unvis return codes
73 */
74#define UNVIS_VALID 1 /* character valid */
75#define UNVIS_VALIDPUSH 2 /* character valid, push back passed char */
76#define UNVIS_NOCHAR 3 /* valid sequence, no character produced */
77#define UNVIS_SYNBAD -1 /* unrecognized escape sequence */
78#define UNVIS_ERROR -2 /* decoder in unknown state (unrecoverable) */
79
80/*
81 * unvis flags
82 */
83#define UNVIS_END _VIS_END /* no more characters */
