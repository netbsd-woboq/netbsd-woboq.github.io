56int
57main(int argc, char **argv, char **envp)
58{
59 const char *basename;
60 const struct stub *ep;
61
62 if (argv[0] == NULL || *argv[0] == '\0')
63 crunched_usage();
64
65 basename = strrchr(argv[0], '/');
66 basename = basename ? basename + 1 : argv[0];
67
68 for (ep = entry_points; ep->name != NULL; ep++)
69 if (strcmp(basename, ep->name) == 0)
70 return ep->f(argc, argv, envp);
71
72 fprintf(stderr, "%s: %s not compiled in\n", EXECNAME, basename);
73 crunched_usage();
74}

87static int
88crunched_usage(void)
89{
90 size_t columns, len;
91 const struct stub *ep;
92
93 fprintf(stderr, "Usage: %s <prog> <args> ..., "
94 "where <prog> is one of:\n", EXECNAME);
95 columns = 0;
96 for (ep = entry_points; ep->name != NULL; ep++) {
97 len = strlen(ep->name) + 1;
98 if (columns + len < 80)
99 columns += len;
100 else {
101 fprintf(stderr, "\n");
102 columns = len;
103 }
104 fprintf(stderr, " %s", ep->name);
105 }
106 fprintf(stderr, "\n");
107 exit(1);
108}
