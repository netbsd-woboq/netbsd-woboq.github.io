335copynodelist(struct nodelist *lp, struct nodecopystate *st)
336{
337 struct nodelist *start;
338 struct nodelist **lpp;
339
340 lpp = &start;
341 while (lp) {

355STATIC char *
356nodesavestr(char *s, struct nodecopystate *st)
357{
358 register char *p = s;
359 register char *q = st->string;
360 char *rtn = st->string;
361
362 while ((*q++ = *p++) != 0)
363 continue;
364 st->string = q;
365 return rtn;
366}
