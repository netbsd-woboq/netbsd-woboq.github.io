14static void
15expand_option_text(menudesc *menu, void *arg, int sel)
16{
17 arg_replace *ar = arg;
18 struct menu_ent *opt = &menu->opts[sel];
19
20 if (menu->opts[sel].opt_exp_name)
21 return; /* has already been expanded */
22
23 opt->opt_exp_name =
24 str_arg_subst(MSG_XLAT(opt->opt_name), ar->argc, ar->argv);
25}
26
27void
28expand_all_option_texts(menudesc *menu, void *arg)
29{
30 arg_replace *ar = arg;
31 const char *title;
32 int i;
33
34 if (menu->title != NULL && menu->exp_title == NULL) {
35 title = MSG_XLAT(menu->title);
36 if (needs_expanding(title, ar->argc)) {
37 menu->exp_title = str_arg_subst(title,
38 ar->argc, ar->argv);
39 }
40 }
41 for (i = 0; i < menu->numopts; i++) {
42 const char *t = MSG_XLAT(menu->opts[i].opt_name);
43
44 if (t == NULL) continue;
45
46 if (needs_expanding(t, ar->argc))
47 expand_option_text(menu, arg, i);
48 }
49}

84static void
85remove_sub_menu(int menuID)
86{
87
88 for (size_t i = 0; i < DYN_MENU_START; i++) {
89 for (int j = 0; j < menu_def[i].numopts; j++) {
90 if ((menu_def[i].opts[j].opt_flags & OPT_SUB)
91 && menu_def[i].opts[j].opt_menu == menuID) {
92
93 for (int k = j + 1; k < menu_def[i].numopts;
94 k++) {
95 menu_def[i].opts[k-1] =
96 menu_def[i].opts[k];
97 }
98 menu_def[i].numopts--;
99 return;
100
101 }
102 }
103 }
104}

2534{

2537 while (TRUE) {
2538 ch = wgetch(w);
2539
2540 switch (ch) {
2541 case '\n':
2542 return REQ_EXECUTE;
2543 case '\016': /* Control-P */
2544 case KEY_DOWN:
2545 return REQ_NEXT_ITEM;
2546 case '\020': /* Control-N */
2547 case KEY_UP:
2548 return REQ_PREV_ITEM;
2549 case '\014': /* Control-L */
2550 return REQ_REDISPLAY;
2551 case '<':
2552 case '\010': /* Control-H (backspace) */
2553 case KEY_PPAGE:
2554 case KEY_LEFT:
2555 return REQ_SCROLLUP;
2556 case '\026': /* Control-V */
2557 case '>':
2558 case ' ':
2559 case KEY_NPAGE:
2560 case KEY_RIGHT:
2561 return REQ_SCROLLDOWN;
2562 case '?':
2563 return REQ_HELP;
2564 case '\033': /* esc-v is scroll down */
2565 ch = wgetch(w);
2566 if (ch == 'v')
2567 return REQ_SCROLLUP;
2568 else
2569 ch = 0; /* zap char so we beep */
2570 }
2571
2572 if (isalpha(ch))
2573 return ch;

2575 mbeep();
2576 wrefresh(w);

2641 /* move menu up screen */

2650 /* We need to add the scroll text...

2782static void
2783draw_menu(menudesc *m, void *arg)
2784{
2785 int opt;
2786 int hasbox, cury, maxy;
2787 int tadd;
2788 int hasexit = (m->mopt & MC_NOEXITOPT ? 0 : 1);
2789 const char *tp, *ep;
2790 const char *title;
2791
2792 hasbox = (m->mopt & MC_NOBOX ? 0 : 1);
2793
2794 /* Clear the window */
2795 wclear(m->mw);
2796
2797 tadd = hasbox;
2798 title = m->title;
2799#ifdef MENU_EXPANDS
2800 if (m->exp_title)
2801 title = m->exp_title;
2802#endif
2803 if (title) {
2804 for (tp = MSG_XLAT(title); ; tp = ep + 1) {
2805 ep = strchr(tp , '\n');
2806 mvwaddnstr(m->mw, tadd++, hasbox + 1, tp,
2807 ep ? ep - tp : -1);
2808 if (ep == NULL || *ep == 0)
2809 break;
2810 }
2811 if ((m->mopt & MC_CONTINUOUS) == 0)
2812 tadd++;
2813 }
2814
2815 cury = tadd;
2816 maxy = getmaxy(m->mw) - hasbox;
2817 if (m->numopts + hasexit > m->h)
2818 /* allow for scroll text */
2819 maxy--;
2820
2821 if (m->cursel == -1) {
2822 m->cursel = m->numopts;
2823 if (m->h <= m->numopts)
2824 m->topline = m->numopts + 1 - m->h;
2825 }
2826
2827 while (m->cursel >= m->topline + m->h)
2828 m->topline = MIN(m->topline + m->h,
2829 m->numopts + hasexit - m->h);
2830 while (m->cursel < m->topline)
2831 m->topline = MAX(0, m->topline - m->h);
2832
2833 for (opt = m->topline; opt < m->numopts; opt++) {
2834 if (cury >= maxy)
2835 break;
2836 draw_menu_line(m, opt, cury++, arg,
2837#ifdef MENU_EXPANDS
2838 m->opts[opt].opt_exp_name ?
2839 m->opts[opt].opt_exp_name : m->opts[opt].opt_name
2840#else
2841 m->opts[opt].opt_name
2842#endif
2843 );
2844 }
2845
2846 /* Add the exit option. */
2847 if (!(m->mopt & MC_NOEXITOPT)) {
2848 if (cury < maxy)
2849 draw_menu_line(m, m->numopts, cury++, arg, m->exitstr);
2850 else
2851 opt = 0;
2852 }
2853
2854 /* Add the scroll line */
2855 if (opt != m->numopts || m->topline != 0)
2856 mvwaddstr(m->mw, cury, hasbox, scrolltext);
2857
2858 /* Add the box. */
2859 if (!(m->mopt & MC_NOBOX))
2860 box(m->mw, 0, 0);
2861
2862 wmove(m->mw, tadd + m->cursel - m->topline, hasbox);
2863 wrefresh(m->mw);
2864}

2867static void
2868/*ARGSUSED*/
2869process_help(menudesc *m)
2870{
2871 const char *help = m->helpstr;
2872 WINDOW *sv_curscr;
2873 int lineoff = 0;
2874 int curoff = 0;
2875 int again;
2876 int winin;
2877
2878 /* Is there help? */
2879 if (!help) {
2880 mbeep();
2881 return;
2882 }
2883 sv_curscr = newwin(getmaxy(curscr), getmaxx(curscr), 0, 0);
2884 if (!sv_curscr) {
2885 mbeep();
2886 return;
2887 }
2888 /*
2889 * Save screen contents so we can restore before returning.
2890 * cursrc contains post-doupdate() data, not post-refresh()
2891 * data so we must call doupdate to ensure we save the
2892 * correct data. Avoids PR 26660.
2893 */
2894 doupdate();
2895 overwrite(curscr, sv_curscr);
2896 touchwin(stdscr);
2897
2898 help = MSG_XLAT(help);
2899 /* Display the help information. */
2900 do {
2901 if (lineoff < curoff) {
2902 help = MSG_XLAT(m->helpstr);
2903 curoff = 0;
2904 }
2905 while (*help && curoff < lineoff) {
2906 if (*help == '\n')
2907 curoff++;
2908 help++;
2909 }
2910
2911 wclear(stdscr);
2912 mvwaddstr(stdscr, 0, 0,
2913 "Help: exit: x, page up: u <, page down: d >");
2914 mvwaddstr(stdscr, 2, 0, help);
2915 wmove(stdscr, 1, 0);
2916 wrefresh(stdscr);
2917
2918 do {
2919 winin = wgetch(stdscr);
2920 if (winin < KEY_MIN)
2921 winin = tolower(winin);
2922 again = 0;
2923 switch (winin) {
2924 case '<':
2925 case 'u':
2926 case KEY_UP:
2927 case KEY_LEFT:
2928 case KEY_PPAGE:
2929 if (lineoff)
2930 lineoff -= max_lines - 2;
2931 else
2932 again = 1;
2933 break;
2934 case '>':
2935 case 'd':
2936 case KEY_DOWN:
2937 case KEY_RIGHT:
2938 case KEY_NPAGE:
2939 if (*help)
2940 lineoff += max_lines - 2;
2941 else
2942 again = 1;
2943 break;
2944 case 'q':
2945 break;
2946 case 'x':
2947 winin = 'q';
2948 break;
2949 default:
2950 again = 1;
2951 }
2952 if (again)
2953 mbeep();
2954 } while (again);
2955 } while (winin != 'q');
2956
2957 /* Restore the original screen contents */
2958 touchwin(sv_curscr);
2959 wnoutrefresh(sv_curscr);
2960 delwin(sv_curscr);
2961
2962 /* Some code thinks that wrefresh(stdout) is a good idea... */
2963 wclear(stdscr);
2964}

2985static void
2986process_req(menudesc *m, void *arg, int req)
2987{
2988 int ch;
2989 int hasexit = (m->mopt & MC_NOEXITOPT ? 0 : 1);
2990
2991 switch(req) {
2992
2993 case REQ_EXECUTE:
2994 return;
2995
2996 case REQ_NEXT_ITEM:
2997 ch = m->cursel;
2998 for (;;) {
2999 ch++;
3000 if (ch >= m->numopts + hasexit) {
3001 mbeep();
3002 return;
3003 }
3004 if (hasexit && ch == m->numopts)
3005 break;
3006 if (!(m->opts[ch].opt_flags & OPT_IGNORE))
3007 break;
3008 }
3009 m->cursel = ch;
3010 if (m->cursel >= m->topline + m->h)
3011 m->topline = m->cursel - m->h + 1;
3012 break;
3013
3014 case REQ_PREV_ITEM:
3015 ch = m->cursel;
3016 for (;;) {
3017 if (ch <= 0) {
3018 mbeep();
3019 return;
3020 }
3021 ch--;
3022 if (!(m->opts[ch].opt_flags & OPT_IGNORE))
3023 break;
3024 }
3025 m->cursel = ch;
3026 if (m->cursel < m->topline)
3027 m->topline = m->cursel;
3028 break;
3029
3030 case REQ_HELP:
3031 process_help(m);
3032 break;
3033
3034 case REQ_REDISPLAY:
3035 endwin();
3036 doupdate();
3037 break;
3038
3039 case REQ_SCROLLUP:
3040 if (m->cursel == 0) {
3041 mbeep();
3042 return;
3043 }
3044 m->topline = MAX(0, m->topline - m->h);
3045 m->cursel = MAX(0, m->cursel - m->h);
3046 wclear(m->mw);
3047 break;
3048
3049 case REQ_SCROLLDOWN:
3050 if (m->cursel >= m->numopts + hasexit - 1) {
3051 mbeep();
3052 return;
3053 }
3054 m->topline = MIN(m->topline + m->h,
3055 MAX(m->numopts + hasexit - m->h, 0));
3056 m->cursel = MIN(m->numopts + hasexit - 1, m->cursel + m->h);
3057 wclear(m->mw);
3058 break;
3059
3060 default:
3061 ch = req;
3062 if (ch == 'x' && hasexit) {
3063 m->cursel = m->numopts;
3064 break;
3065 }
3066 if (m->mopt & MC_NOSHORTCUT) {
3067 mbeep();
3068 return;
3069 }
3070 if (ch > 'z')
3071 ch = 255;
3072 if (ch >= 'a') {
3073 if (ch > 'x')
3074 ch--;
3075 ch = ch - 'a';
3076 } else
3077 ch = 25 + ch - 'A';
3078 if (ch < 0 || ch >= m->numopts) {
3079 mbeep();
3080 return;
3081 }
3082 if (m->opts[ch].opt_flags & OPT_IGNORE) {
3083 mbeep();
3084 return;
3085 }
3086 m->cursel = ch;
3087 }
3088
3089 draw_menu(m, arg);
3090}
