974/*
975 * get_line - a hacked up version of fgets to ignore TELNET escape codes.
976 * `s' is the buffer to read into.
977 * `n' is the 1 less than the size of the buffer, to allow trailing NUL
978 * `iop' is the FILE to read from.
979 * Returns 0 on success, -1 on EOF, -2 if the command was too long.
980 */
981int
982get_line(char *s, int n, FILE *iop)
983{
984 int c;
985 char *cs;
986
987 cs = s;
988/* tmpline may contain saved command from urgent mode interruption */
989 for (c = 0; tmpline[c] != '\0' && --n > 0; ++c) {
990 *cs++ = tmpline[c];
991 if (tmpline[c] == '\n') {
992 *cs++ = '\0';
993 if (ftpd_debug)
994 syslog(LOG_DEBUG, "command: %s", s);
995 tmpline[0] = '\0';
996 return(0);
997 }
998 if (c == 0)
999 tmpline[0] = '\0';
1000 }
1001 while ((c = getc(iop)) != EOF) {
1002 total_bytes++;
1003 total_bytes_in++;
1004 c &= 0377;
1005 if (c == IAC) {
1006 if ((c = getc(iop)) != EOF) {
1007 total_bytes++;
1008 total_bytes_in++;
1009 c &= 0377;
1010 switch (c) {
1011 case WILL:
1012 case WONT:
1013 c = getc(iop);
1014 total_bytes++;
1015 total_bytes_in++;
1016 cprintf(stdout, "%c%c%c", IAC, DONT, 0377&c);
1017 (void) fflush(stdout);
1018 continue;
1019 case DO:
1020 case DONT:
1021 c = getc(iop);
1022 total_bytes++;
1023 total_bytes_in++;
1024 cprintf(stdout, "%c%c%c", IAC, WONT, 0377&c);
1025 (void) fflush(stdout);
1026 continue;
1027 case IAC:
1028 break;
1029 default:
1030 continue; /* ignore command */
1031 }
1032 }
1033 }
1034 *cs++ = c;
1035 if (--n <= 0) {
1036 /*
1037 * If command doesn't fit into buffer, discard the
1038 * rest of the command and indicate truncation.
1039 * This prevents the command to be split up into
1040 * multiple commands.
1041 */
1042 if (ftpd_debug)
1043 syslog(LOG_DEBUG,
1044 "command too long, last char: %d", c);
1045 while (c != '\n' && (c = getc(iop)) != EOF)
1046 continue;
1047 return (-2);
1048 }
1049 if (c == '\n')
1050 break;
1051 }
1052 if (c == EOF && cs == s)
1053 return (-1);
1054 *cs++ = '\0';
1055 if (ftpd_debug) {
1056 if ((curclass.type != CLASS_GUEST &&
1057 strncasecmp(s, "PASS ", 5) == 0) ||
1058 strncasecmp(s, "ACCT ", 5) == 0) {
1059 /* Don't syslog passwords */
1060 syslog(LOG_DEBUG, "command: %.4s ???", s);
1061 } else {
1062 char *cp;
1063 int len;
1064
1065 /* Don't syslog trailing CR-LF */
1066 len = strlen(s);
1067 cp = s + len - 1;
1068 while (cp >= s && (*cp == '\n' || *cp == '\r')) {
1069 --cp;
1070 --len;
1071 }
1072 syslog(LOG_DEBUG, "command: %.*s", len, s);
1073 }
1074 }
1075 return (0);
1076}
