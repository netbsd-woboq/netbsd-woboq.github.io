528#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
529 p->yystack.p_base = NULL;
530#endif
531 return p;
532 }
533 p->yystack.s_base = (YYINT *) malloc(size * sizeof(YYINT));
534 if (p->yystack.s_base == NULL) return NULL;
535 p->yystack.l_base = (YYSTYPE *) malloc(size * sizeof(YYSTYPE));
536 if (p->yystack.l_base == NULL) return NULL;
537 memset(p->yystack.l_base, 0, size * sizeof(YYSTYPE));
538#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
539 p->yystack.p_base = (YYLTYPE *) malloc(size * sizeof(YYLTYPE));
540 if (p->yystack.p_base == NULL) return NULL;
541 memset(p->yystack.p_base, 0, size * sizeof(YYLTYPE));
542#endif
543
544 return p;
545}

549{
550 yyfreestack(&p->yystack);
551 free(p);

553#endif /* YYBTYACC */

555#define YYABORT goto yyabort
556#define YYREJECT goto yyabort
557#define YYACCEPT goto yyaccept
558#define YYERROR goto yyerrlab
559#if YYBTYACC
560#define YYVALID do { if (yyps->save) goto yyvalid; } while(0)
561#define YYVALID_NESTED do { if (yyps->save && \
562 yyps->save->save == 0) goto yyvalid; } while(0)
563#endif /* YYBTYACC */
564
565int
566YYPARSE_DECL()
567{
568 int yym, yyn, yystate, yyresult;
569#if YYBTYACC
570 int yynewerrflag;
571 YYParseState *yyerrctx = NULL;
572#endif /* YYBTYACC */
573#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
574 YYLTYPE yyerror_loc_range[3]; /* position of error start/end (0 unused) */
575#endif
576#if YYDEBUG
577 const char *yys;
578
579 if ((yys = getenv("YYDEBUG")) != 0)
580 {
581 yyn = *yys;
582 if (yyn >= '0' && yyn <= '9')
583 yydebug = yyn - '0';
584 }
585 if (yydebug)
586 fprintf(stderr, "%sdebug[<# of symbols on state stack>]\n", YYPREFIX);
587#endif
588#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
589 memset(yyerror_loc_range, 0, sizeof(yyerror_loc_range));
590#endif
591
592#if YYBTYACC
593 yyps = yyNewState(0); if (yyps == 0) goto yyenomem;
594 yyps->save = 0;
595#endif /* YYBTYACC */
596 yym = 0;
597 yyn = 0;
598 yynerrs = 0;
599 yyerrflag = 0;
600 yychar = YYEMPTY;
601 yystate = 0;
602
603#if YYPURE
604 memset(&yystack, 0, sizeof(yystack));
605#endif
606
607 if (yystack.s_base == NULL && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
608 yystack.s_mark = yystack.s_base;
609 yystack.l_mark = yystack.l_base;
610#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
611 yystack.p_mark = yystack.p_base;
612#endif
613 yystate = 0;
614 *yystack.s_mark = 0;
615
616yyloop:
617 if ((yyn = yydefred[yystate]) != 0) goto yyreduce;
618 if (yychar < 0)
619 {
620#if YYBTYACC
621 do {
622 if (yylvp < yylve)
623 {
624 /* we're currently re-reading tokens */
625 yylval = *yylvp++;
626#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
627 yylloc = *yylpp++;
628#endif
629 yychar = *yylexp++;
630 break;
631 }
632 if (yyps->save)
633 {
634 /* in trial mode; save scanner results for future parse attempts */
635 if (yylvp == yylvlim)
636 { /* Enlarge lexical value queue */
637 size_t p = (size_t) (yylvp - yylvals);
638 size_t s = (size_t) (yylvlim - yylvals);
639
640 s += YYLVQUEUEGROWTH;
641 if ((yylexemes = (YYINT *)realloc(yylexemes, s * sizeof(YYINT))) == NULL) goto yyenomem;
642 if ((yylvals = (YYSTYPE *)realloc(yylvals, s * sizeof(YYSTYPE))) == NULL) goto yyenomem;
643#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
644 if ((yylpsns = (YYLTYPE *)realloc(yylpsns, s * sizeof(YYLTYPE))) == NULL) goto yyenomem;
645#endif
646 yylvp = yylve = yylvals + p;
647 yylvlim = yylvals + s;
648#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
649 yylpp = yylpe = yylpsns + p;
650 yylplim = yylpsns + s;
651#endif
652 yylexp = yylexemes + p;
653 }
654 *yylexp = (YYINT) YYLEX;
655 *yylvp++ = yylval;
656 yylve++;
657#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
658 *yylpp++ = yylloc;
659 yylpe++;
660#endif
661 yychar = *yylexp++;
662 break;
663 }
664 /* normal operation, no conflict encountered */
665#endif /* YYBTYACC */
666 yychar = YYLEX;
667#if YYBTYACC
668 } while (0);
669#endif /* YYBTYACC */
670 if (yychar < 0) yychar = YYEOF;
671#if YYDEBUG
672 if (yydebug)
673 {
674 if ((yys = yyname[YYTRANSLATE(yychar)]) == NULL) yys = yyname[YYUNDFTOKEN];
675 fprintf(stderr, "%s[%d]: state %d, reading token %d (%s)",
676 YYDEBUGSTR, yydepth, yystate, yychar, yys);
677#ifdef YYSTYPE_TOSTRING
678#if YYBTYACC
679 if (!yytrial)
680#endif /* YYBTYACC */
681 fprintf(stderr, " <%s>", YYSTYPE_TOSTRING(yychar, yylval));
682#endif
683 fputc('\n', stderr);
684 }
685#endif
686 }
687#if YYBTYACC
688
689 /* Do we have a conflict? */
690 if (((yyn = yycindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
691 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
692 {
693 YYINT ctry;
694
695 if (yypath)
696 {
697 YYParseState *save;
698#if YYDEBUG
699 if (yydebug)
700 fprintf(stderr, "%s[%d]: CONFLICT in state %d: following successful trial parse\n",
701 YYDEBUGSTR, yydepth, yystate);
702#endif
703 /* Switch to the next conflict context */
704 save = yypath;
705 yypath = save->save;
706 save->save = NULL;
707 ctry = save->ctry;
708 if (save->state != yystate) YYABORT;
709 yyFreeState(save);
710
711 }
712 else
713 {
714
715 /* Unresolved conflict - start/continue trial parse */
716 YYParseState *save;
717#if YYDEBUG
718 if (yydebug)
719 {
720 fprintf(stderr, "%s[%d]: CONFLICT in state %d. ", YYDEBUGSTR, yydepth, yystate);
721 if (yyps->save)
722 fputs("ALREADY in conflict, continuing trial parse.\n", stderr);
723 else
724 fputs("Starting trial parse.\n", stderr);
725 }
726#endif
727 save = yyNewState((unsigned)(yystack.s_mark - yystack.s_base + 1));
728 if (save == NULL) goto yyenomem;
729 save->save = yyps->save;
730 save->state = yystate;
731 save->errflag = yyerrflag;
732 save->yystack.s_mark = save->yystack.s_base + (yystack.s_mark - yystack.s_base);
733 memcpy (save->yystack.s_base, yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
734 save->yystack.l_mark = save->yystack.l_base + (yystack.l_mark - yystack.l_base);
735 memcpy (save->yystack.l_base, yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
736#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
737 save->yystack.p_mark = save->yystack.p_base + (yystack.p_mark - yystack.p_base);
738 memcpy (save->yystack.p_base, yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
739#endif
740 ctry = yytable[yyn];
741 if (yyctable[ctry] == -1)
742 {
743#if YYDEBUG
744 if (yydebug && yychar >= YYEOF)
745 fprintf(stderr, "%s[%d]: backtracking 1 token\n", YYDEBUGSTR, yydepth);
746#endif
747 ctry++;
748 }
749 save->ctry = ctry;
750 if (yyps->save == NULL)
751 {
752 /* If this is a first conflict in the stack, start saving lexemes */
753 if (!yylexemes)
754 {
755 yylexemes = (YYINT *) malloc((YYLVQUEUEGROWTH) * sizeof(YYINT));
756 if (yylexemes == NULL) goto yyenomem;
757 yylvals = (YYSTYPE *) malloc((YYLVQUEUEGROWTH) * sizeof(YYSTYPE));
758 if (yylvals == NULL) goto yyenomem;
759 yylvlim = yylvals + YYLVQUEUEGROWTH;
760#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
761 yylpsns = (YYLTYPE *) malloc((YYLVQUEUEGROWTH) * sizeof(YYLTYPE));
762 if (yylpsns == NULL) goto yyenomem;
763 yylplim = yylpsns + YYLVQUEUEGROWTH;
764#endif
765 }
766 if (yylvp == yylve)
767 {
768 yylvp = yylve = yylvals;
769#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
770 yylpp = yylpe = yylpsns;
771#endif
772 yylexp = yylexemes;
773 if (yychar >= YYEOF)
774 {
775 *yylve++ = yylval;
776#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
777 *yylpe++ = yylloc;
778#endif
779 *yylexp = (YYINT) yychar;
780 yychar = YYEMPTY;
781 }
782 }
783 }
784 if (yychar >= YYEOF)
785 {
786 yylvp--;
787#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
788 yylpp--;
789#endif
790 yylexp--;
791 yychar = YYEMPTY;
792 }
793 save->lexeme = (int) (yylvp - yylvals);
794 yyps->save = save;
795 }
796 if (yytable[yyn] == ctry)
797 {
798#if YYDEBUG
799 if (yydebug)
800 fprintf(stderr, "%s[%d]: state %d, shifting to state %d\n",
801 YYDEBUGSTR, yydepth, yystate, yyctable[ctry]);
802#endif
803 if (yychar < 0)
804 {
805 yylvp++;
806#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
807 yylpp++;
808#endif
809 yylexp++;
810 }
811 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM)
812 goto yyoverflow;
813 yystate = yyctable[ctry];
814 *++yystack.s_mark = (YYINT) yystate;
815 *++yystack.l_mark = yylval;
816#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
817 *++yystack.p_mark = yylloc;
818#endif
819 yychar = YYEMPTY;
820 if (yyerrflag > 0) --yyerrflag;
821 goto yyloop;
822 }
823 else
824 {
825 yyn = yyctable[ctry];
826 goto yyreduce;
827 }
828 } /* End of code dealing with conflicts */
829#endif /* YYBTYACC */
830 if (((yyn = yysindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
831 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
832 {
833#if YYDEBUG
834 if (yydebug)
835 fprintf(stderr, "%s[%d]: state %d, shifting to state %d\n",
836 YYDEBUGSTR, yydepth, yystate, yytable[yyn]);
837#endif
838 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
839 yystate = yytable[yyn];
840 *++yystack.s_mark = yytable[yyn];
841 *++yystack.l_mark = yylval;
842#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
843 *++yystack.p_mark = yylloc;
844#endif
845 yychar = YYEMPTY;
846 if (yyerrflag > 0) --yyerrflag;
847 goto yyloop;
848 }
849 if (((yyn = yyrindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
850 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
851 {
852 yyn = yytable[yyn];
853 goto yyreduce;
854 }
855 if (yyerrflag != 0) goto yyinrecovery;
856#if YYBTYACC
857
858 yynewerrflag = 1;
859 goto yyerrhandler;
860 goto yyerrlab; /* redundant goto avoids 'unused label' warning */
861
862yyerrlab:
863 /* explicit YYERROR from an action -- pop the rhs of the rule reduced
864 * before looking for error recovery */
865 yystack.s_mark -= yym;
866 yystate = *yystack.s_mark;
867 yystack.l_mark -= yym;
868#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
869 yystack.p_mark -= yym;
870#endif
871
872 yynewerrflag = 0;
873yyerrhandler:
874 while (yyps->save)
875 {
876 int ctry;
877 YYParseState *save = yyps->save;
878#if YYDEBUG
879 if (yydebug)
880 fprintf(stderr, "%s[%d]: ERROR in state %d, CONFLICT BACKTRACKING to state %d, %d tokens\n",
881 YYDEBUGSTR, yydepth, yystate, yyps->save->state,
882 (int)(yylvp - yylvals - yyps->save->lexeme));
883#endif
884 /* Memorize most forward-looking error state in case it's really an error. */
885 if (yyerrctx == NULL || yyerrctx->lexeme < yylvp - yylvals)
886 {
887 /* Free old saved error context state */
888 if (yyerrctx) yyFreeState(yyerrctx);
889 /* Create and fill out new saved error context state */
890 yyerrctx = yyNewState((unsigned)(yystack.s_mark - yystack.s_base + 1));
891 if (yyerrctx == NULL) goto yyenomem;
892 yyerrctx->save = yyps->save;
893 yyerrctx->state = yystate;
894 yyerrctx->errflag = yyerrflag;
895 yyerrctx->yystack.s_mark = yyerrctx->yystack.s_base + (yystack.s_mark - yystack.s_base);
896 memcpy (yyerrctx->yystack.s_base, yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
897 yyerrctx->yystack.l_mark = yyerrctx->yystack.l_base + (yystack.l_mark - yystack.l_base);
898 memcpy (yyerrctx->yystack.l_base, yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
899#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
900 yyerrctx->yystack.p_mark = yyerrctx->yystack.p_base + (yystack.p_mark - yystack.p_base);
901 memcpy (yyerrctx->yystack.p_base, yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
902#endif
903 yyerrctx->lexeme = (int) (yylvp - yylvals);
904 }
905 yylvp = yylvals + save->lexeme;
906#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
907 yylpp = yylpsns + save->lexeme;
908#endif
909 yylexp = yylexemes + save->lexeme;
910 yychar = YYEMPTY;
911 yystack.s_mark = yystack.s_base + (save->yystack.s_mark - save->yystack.s_base);
912 memcpy (yystack.s_base, save->yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
913 yystack.l_mark = yystack.l_base + (save->yystack.l_mark - save->yystack.l_base);
914 memcpy (yystack.l_base, save->yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
915#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
916 yystack.p_mark = yystack.p_base + (save->yystack.p_mark - save->yystack.p_base);
917 memcpy (yystack.p_base, save->yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
918#endif
919 ctry = ++save->ctry;
920 yystate = save->state;
921 /* We tried shift, try reduce now */
922 if ((yyn = yyctable[ctry]) >= 0) goto yyreduce;
923 yyps->save = save->save;
924 save->save = NULL;
925 yyFreeState(save);
926
927 /* Nothing left on the stack -- error */
928 if (!yyps->save)
929 {
930#if YYDEBUG
931 if (yydebug)
932 fprintf(stderr, "%sdebug[%d,trial]: trial parse FAILED, entering ERROR mode\n",
933 YYPREFIX, yydepth);
934#endif
935 /* Restore state as it was in the most forward-advanced error */
936 yylvp = yylvals + yyerrctx->lexeme;
937#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
938 yylpp = yylpsns + yyerrctx->lexeme;
939#endif
940 yylexp = yylexemes + yyerrctx->lexeme;
941 yychar = yylexp[-1];
942 yylval = yylvp[-1];
943#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
944 yylloc = yylpp[-1];
945#endif
946 yystack.s_mark = yystack.s_base + (yyerrctx->yystack.s_mark - yyerrctx->yystack.s_base);
947 memcpy (yystack.s_base, yyerrctx->yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
948 yystack.l_mark = yystack.l_base + (yyerrctx->yystack.l_mark - yyerrctx->yystack.l_base);
949 memcpy (yystack.l_base, yyerrctx->yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
950#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
951 yystack.p_mark = yystack.p_base + (yyerrctx->yystack.p_mark - yyerrctx->yystack.p_base);
952 memcpy (yystack.p_base, yyerrctx->yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
953#endif
954 yystate = yyerrctx->state;
955 yyFreeState(yyerrctx);
956 yyerrctx = NULL;
957 }
958 yynewerrflag = 1;
959 }
960 if (yynewerrflag == 0) goto yyinrecovery;
961#endif /* YYBTYACC */
962
963 YYERROR_CALL("syntax error");
964#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
965 yyerror_loc_range[1] = yylloc; /* lookahead position is error start position */
966#endif
967
968#if !YYBTYACC
969 goto yyerrlab; /* redundant goto avoids 'unused label' warning */
970yyerrlab:
971#endif
972 ++yynerrs;
973
974yyinrecovery:
975 if (yyerrflag < 3)
976 {
977 yyerrflag = 3;
978 for (;;)
979 {
980 if (((yyn = yysindex[*yystack.s_mark]) != 0) && (yyn += YYERRCODE) >= 0 &&
981 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) YYERRCODE)
982 {
983#if YYDEBUG
984 if (yydebug)
985 fprintf(stderr, "%s[%d]: state %d, error recovery shifting to state %d\n",
986 YYDEBUGSTR, yydepth, *yystack.s_mark, yytable[yyn]);
987#endif
988 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
989 yystate = yytable[yyn];
990 *++yystack.s_mark = yytable[yyn];
991 *++yystack.l_mark = yylval;
992#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
993 /* lookahead position is error end position */
994 yyerror_loc_range[2] = yylloc;
995 YYLLOC_DEFAULT(yyloc, yyerror_loc_range, 2); /* position of error span */
996 *++yystack.p_mark = yyloc;
997#endif
998 goto yyloop;
999 }
1000 else
1001 {
1002#if YYDEBUG
1003 if (yydebug)
1004 fprintf(stderr, "%s[%d]: error recovery discarding state %d\n",
1005 YYDEBUGSTR, yydepth, *yystack.s_mark);
1006#endif
1007 if (yystack.s_mark <= yystack.s_base) goto yyabort;
1008#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1009 /* the current TOS position is the error start position */
1010 yyerror_loc_range[1] = *yystack.p_mark;
1011#endif
1012#if defined(YYDESTRUCT_CALL)
1013#if YYBTYACC
1014 if (!yytrial)
1015#endif /* YYBTYACC */
1016#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1017 YYDESTRUCT_CALL("error: discarding state",
1018 yystos[*yystack.s_mark], yystack.l_mark, yystack.p_mark);
1019#else
1020 YYDESTRUCT_CALL("error: discarding state",
1021 yystos[*yystack.s_mark], yystack.l_mark);
1022#endif /* defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED) */
1023#endif /* defined(YYDESTRUCT_CALL) */
1024 --yystack.s_mark;
1025 --yystack.l_mark;
1026#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1027 --yystack.p_mark;
1028#endif
1029 }
1030 }
1031 }
1032 else
1033 {
1034 if (yychar == YYEOF) goto yyabort;
1035#if YYDEBUG
1036 if (yydebug)
1037 {
1038 if ((yys = yyname[YYTRANSLATE(yychar)]) == NULL) yys = yyname[YYUNDFTOKEN];
1039 fprintf(stderr, "%s[%d]: state %d, error recovery discarding token %d (%s)\n",
1040 YYDEBUGSTR, yydepth, yystate, yychar, yys);
1041 }
1042#endif
1043#if defined(YYDESTRUCT_CALL)
1044#if YYBTYACC
1045 if (!yytrial)
1046#endif /* YYBTYACC */
1047#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1048 YYDESTRUCT_CALL("error: discarding token", yychar, &yylval, &yylloc);
1049#else
1050 YYDESTRUCT_CALL("error: discarding token", yychar, &yylval);
1051#endif /* defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED) */
1052#endif /* defined(YYDESTRUCT_CALL) */
1053 yychar = YYEMPTY;
1054 goto yyloop;
1055 }
1056
1057yyreduce:
1058 yym = yylen[yyn];
1059#if YYDEBUG
1060 if (yydebug)
1061 {
1062 fprintf(stderr, "%s[%d]: state %d, reducing by rule %d (%s)",
1063 YYDEBUGSTR, yydepth, yystate, yyn, yyrule[yyn]);
1064#ifdef YYSTYPE_TOSTRING
1065#if YYBTYACC
1066 if (!yytrial)
1067#endif /* YYBTYACC */
1068 if (yym > 0)
1069 {
