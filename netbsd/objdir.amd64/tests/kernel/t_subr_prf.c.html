170/*
171 * vsnprintf: print a message to a buffer [already have va_list]
172 */
173int
174vsnprintf(char *bf, size_t size, const char *fmt, va_list ap)
175{
176 int retval;
177 char *p;
178
179 p = bf + size;
180 retval = kprintf(fmt, TOBUFONLY, &p, bf, ap);
181 if (bf && size > 0) {
182 /* nul terminate */
183 if (size <= (size_t)retval)
184 bf[size - 1] = '\0';
185 else
186 bf[retval] = '\0';
187 }
188 return retval;
189}

191int
192vasprintf(char **bf, const char *fmt, va_list ap)
193{
194 int retval;
195 va_list cap;
196
197 va_copy(cap, ap);
198 retval = kprintf(fmt, TOBUFONLY, NULL, NULL, cap) + 1;
199 va_end(cap);
200 *bf = kmem_alloc(retval, KM_SLEEP);
201 return vsnprintf(*bf, retval, fmt, ap);
202}

204/*
205 * kprintf: scaled down version of printf(3).
206 *
207 * this version based on vfprintf() from libc which was derived from
208 * software contributed to Berkeley by Chris Torek.
209 *
210 * NOTE: The kprintf mutex must be held if we're going TOBUF or TOCONS!
211 */
212
213/*
214 * macros for converting digits to letters and vice versa
215 */
216#define to_digit(c) ((c) - '0')
217#define is_digit(c) ((unsigned)to_digit(c) <= 9)
218#define to_char(n) ((n) + '0')
219
220/*
221 * flags used during conversion.
222 */
223#define ALT 0x001 /* alternate form */
224#define HEXPREFIX 0x002 /* add 0x or 0X prefix */
225#define LADJUST 0x004 /* left adjustment */
226#define LONGDBL 0x008 /* long double; unimplemented */
227#define LONGINT 0x010 /* long integer */
228#define QUADINT 0x020 /* quad integer */
229#define SHORTINT 0x040 /* short integer */
230#define MAXINT 0x080 /* intmax_t */
231#define PTRINT 0x100 /* intptr_t */
232#define SIZEINT 0x200 /* size_t */
233#define ZEROPAD 0x400 /* zero (as opposed to blank) pad */
234#define FPT 0x800 /* Floating point number */
235
236 /*
237 * To extend shorts properly, we need both signed and unsigned
238 * argument extraction methods.
239 */
240#define SARG() \
241 (flags&MAXINT ? va_arg(ap, intmax_t) : \
242 flags&PTRINT ? va_arg(ap, intptr_t) : \
243 flags&SIZEINT ? va_arg(ap, ssize_t) : /* XXX */ \
244 flags&QUADINT ? va_arg(ap, quad_t) : \
245 flags&LONGINT ? va_arg(ap, long) : \
246 flags&SHORTINT ? (long)(short)va_arg(ap, int) : \
247 (long)va_arg(ap, int))
248#define UARG() \
249 (flags&MAXINT ? va_arg(ap, uintmax_t) : \
250 flags&PTRINT ? va_arg(ap, uintptr_t) : \
251 flags&SIZEINT ? va_arg(ap, size_t) : \
252 flags&QUADINT ? va_arg(ap, u_quad_t) : \
253 flags&LONGINT ? va_arg(ap, u_long) : \
254 flags&SHORTINT ? (u_long)(u_short)va_arg(ap, int) : \
255 (u_long)va_arg(ap, u_int))
256
257#define KPRINTF_PUTCHAR(C) { \
258 if (oflags == TOBUFONLY) { \
259 if (sbuf && ((vp == NULL) || (sbuf < tailp))) \
260 *sbuf++ = (C); \
261 } else { \
262 putchar((C), oflags, vp); \
263 } \
264}

278/*
279 * Guts of kernel printf. Note, we already expect to be in a mutex!
280 */
281int
282kprintf(const char *fmt0, int oflags, void *vp, char *sbuf, va_list ap)
283{
284 const char *fmt; /* format string */
285 int ch; /* character from fmt */
286 int n; /* handy integer (short term usage) */
287 char *cp; /* handy char pointer (short term usage) */
288 int flags; /* flags as above */
289 int ret; /* return value accumulator */
290 int width; /* width from format (%8d), or 0 */
291 int prec; /* precision from format (%.3d), or -1 */
292 char sign; /* sign prefix (' ', '+', '-', or \0) */
293
294 u_quad_t _uquad; /* integer arguments %[diouxX] */
295 enum { OCT, DEC, HEX } base;/* base for [diouxX] conversion */
296 int dprec; /* a copy of prec if [diouxX], 0 otherwise */
297 int realsz; /* field size expanded by dprec */
298 int size; /* size of converted field or string */
299 const char *xdigs; /* digits for [xX] conversion */
300 char bf[KPRINTF_BUFSIZE]; /* space for %c, %[diouxX] */
301 char *tailp; /* tail pointer for snprintf */
302
303 if (oflags == TOBUFONLY && (vp != NULL))
304 tailp = *(char **)vp;
305 else
306 tailp = NULL;
307
308 cp = NULL; /* XXX: shutup gcc */
309 size = 0; /* XXX: shutup gcc */
310
311 fmt = fmt0;
312 ret = 0;
313
314 xdigs = NULL; /* XXX: shut up gcc warning */
315
316 /*
317 * Scan the format for conversions (`%' character).
318 */
319 for (;;) {
320 for (; *fmt != '%' && *fmt; fmt++) {
321 ret++;
322 KPRINTF_PUTCHAR(*fmt);
323 }
324 if (*fmt == 0)
325 goto done;
326
327 fmt++; /* skip over '%' */
328
329 flags = 0;
330 dprec = 0;
331 width = 0;
332 prec = -1;
333 sign = '\0';
334
335rflag: ch = *fmt++;
336reswitch: switch (ch) {
337 case ' ':
338 /*
339 * ``If the space and + flags both appear, the space
340 * flag will be ignored.''
341 * -- ANSI X3J11
342 */
343 if (!sign)
344 sign = ' ';
345 goto rflag;
346 case '#':
347 flags |= ALT;
348 goto rflag;
349 case '*':
350 /*
351 * ``A negative field width argument is taken as a
352 * - flag followed by a positive field width.''
353 * -- ANSI X3J11
354 * They don't exclude field widths read from args.
355 */
356 if ((width = va_arg(ap, int)) >= 0)
357 goto rflag;
358 width = -width;
359 /* FALLTHROUGH */
360 case '-':
361 flags |= LADJUST;
362 goto rflag;
363 case '+':
364 sign = '+';
365 goto rflag;
366 case '.':
367 if ((ch = *fmt++) == '*') {
368 n = va_arg(ap, int);
369 prec = n < 0 ? -1 : n;
370 goto rflag;
371 }
372 n = 0;
373 while (is_digit(ch)) {
374 n = 10 * n + to_digit(ch);
375 ch = *fmt++;
376 }
377 prec = n < 0 ? -1 : n;
378 goto reswitch;
379 case '0':
380 /*
381 * ``Note that 0 is taken as a flag, not as the
382 * beginning of a field width.''
383 * -- ANSI X3J11
384 */
385 flags |= ZEROPAD;
386 goto rflag;
387 case '1': case '2': case '3': case '4':
388 case '5': case '6': case '7': case '8': case '9':
389 n = 0;
390 do {
391 n = 10 * n + to_digit(ch);
392 ch = *fmt++;
393 } while (is_digit(ch));
394 width = n;
395 goto reswitch;
396 case 'h':
397 flags |= SHORTINT;
398 goto rflag;
399 case 'j':
400 flags |= MAXINT;
401 goto rflag;
402 case 'l':
403 if (*fmt == 'l') {
404 fmt++;
405 flags |= QUADINT;
406 } else {
407 flags |= LONGINT;
408 }
409 goto rflag;
410 case 'q':
411 flags |= QUADINT;
412 goto rflag;
413 case 't':
414 flags |= PTRINT;
415 goto rflag;
416 case 'z':
417 flags |= SIZEINT;
418 goto rflag;
419 case 'c':
420 *(cp = bf) = va_arg(ap, int);
421 size = 1;
422 sign = '\0';
423 break;
424 case 'D':
425 flags |= LONGINT;
426 /*FALLTHROUGH*/
427 case 'd':
428 case 'i':
429 _uquad = SARG();
430 if ((quad_t)_uquad < 0) {
431 _uquad = -_uquad;
432 sign = '-';
433 }
434 base = DEC;
435 goto number;
436 case 'n':
437 if (flags & MAXINT)
438 *va_arg(ap, intmax_t *) = ret;
439 else if (flags & PTRINT)
440 *va_arg(ap, intptr_t *) = ret;
441 else if (flags & SIZEINT)
442 *va_arg(ap, ssize_t *) = ret;
443 else if (flags & QUADINT)
444 *va_arg(ap, quad_t *) = ret;
445 else if (flags & LONGINT)
446 *va_arg(ap, long *) = ret;
447 else if (flags & SHORTINT)
448 *va_arg(ap, short *) = ret;
449 else
450 *va_arg(ap, int *) = ret;
451 continue; /* no output */
452 case 'O':
453 flags |= LONGINT;
454 /*FALLTHROUGH*/
455 case 'o':
456 _uquad = UARG();
457 base = OCT;
458 goto nosign;
459 case 'p':
460 /*
461 * ``The argument shall be a pointer to void. The
462 * value of the pointer is converted to a sequence
463 * of printable characters, in an implementation-
464 * defined manner.''
465 * -- ANSI X3J11
466 */
467 /* NOSTRICT */
468 _uquad = (u_long)va_arg(ap, void *);
469 base = HEX;
470 xdigs = hexdigits;
471 flags |= HEXPREFIX;
472 ch = 'x';
473 goto nosign;
474 case 's':
475 if ((cp = va_arg(ap, char *)) == NULL)
476 /*XXXUNCONST*/
477 cp = __UNCONST("(null)");
478 if (prec >= 0) {
479 /*
480 * can't use strlen; can only look for the
481 * NUL in the first `prec' characters, and
482 * strlen() will go further.
483 */
484 char *p = memchr(cp, 0, prec);
485
486 if (p != NULL) {
487 size = p - cp;
488 if (size > prec)
489 size = prec;
490 } else
491 size = prec;
492 } else
493 size = strlen(cp);
494 sign = '\0';
495 break;
496 case 'U':
497 flags |= LONGINT;
498 /*FALLTHROUGH*/
499 case 'u':
500 _uquad = UARG();
501 base = DEC;
502 goto nosign;
503 case 'X':
504 xdigs = HEXDIGITS;
505 goto hex;
506 case 'x':
507 xdigs = hexdigits;
508hex: _uquad = UARG();
509 base = HEX;
510 /* leading 0x/X only if non-zero */
511 if (flags & ALT && _uquad != 0)
512 flags |= HEXPREFIX;
513
514 /* unsigned conversions */
515nosign: sign = '\0';
516 /*
517 * ``... diouXx conversions ... if a precision is
518 * specified, the 0 flag will be ignored.''
519 * -- ANSI X3J11
520 */
521number: if ((dprec = prec) >= 0)
522 flags &= ~ZEROPAD;
523
524 /*
525 * ``The result of converting a zero value with an
526 * explicit precision of zero is no characters.''
527 * -- ANSI X3J11
528 */
529 cp = bf + KPRINTF_BUFSIZE;
530 if (_uquad != 0 || prec != 0) {
531 /*
532 * Unsigned mod is hard, and unsigned mod
533 * by a constant is easier than that by
534 * a variable; hence this switch.
535 */
536 switch (base) {
537 case OCT:
538 do {
539 *--cp = to_char(_uquad & 7);
540 _uquad >>= 3;
541 } while (_uquad);
542 /* handle octal leading 0 */
543 if (flags & ALT && *cp != '0')
544 *--cp = '0';
545 break;
546
547 case DEC:
548 /* many numbers are 1 digit */
549 while (_uquad >= 10) {
550 *--cp = to_char(_uquad % 10);
551 _uquad /= 10;
552 }
553 *--cp = to_char(_uquad);
554 break;
555
556 case HEX:
557 do {
558 *--cp = xdigs[_uquad & 15];
559 _uquad >>= 4;
560 } while (_uquad);
561 break;
562
563 default:
564 /*XXXUNCONST*/
565 cp = __UNCONST("bug in kprintf: bad base");
566 size = strlen(cp);
567 goto skipsize;
568 }
569 }
570 size = bf + KPRINTF_BUFSIZE - cp;
571 skipsize:
572 break;
573 default: /* "%?" prints ?, unless ? is NUL */
574 if (ch == '\0')
575 goto done;
576 /* pretend it was %c with argument ch */
577 cp = bf;
578 *cp = ch;
579 size = 1;
580 sign = '\0';
581 break;
582 }
583
584 /*
585 * All reasonable formats wind up here. At this point, `cp'
586 * points to a string which (if not flags&LADJUST) should be
587 * padded out to `width' places. If flags&ZEROPAD, it should
588 * first be prefixed by any sign or other prefix; otherwise,
589 * it should be blank padded before the prefix is emitted.
590 * After any left-hand padding and prefixing, emit zeroes
591 * required by a decimal [diouxX] precision, then print the
592 * string proper, then emit zeroes required by any leftover
593 * floating precision; finally, if LADJUST, pad with blanks.
594 *
595 * Compute actual size, so we know how much to pad.
596 * size excludes decimal prec; realsz includes it.
597 */
598 realsz = dprec > size ? dprec : size;
599 if (sign)
600 realsz++;
601 else if (flags & HEXPREFIX)
602 realsz+= 2;
603
604 /* adjust ret */
605 ret += width > realsz ? width : realsz;
606
607 /* right-adjusting blank padding */
608 if ((flags & (LADJUST|ZEROPAD)) == 0) {
609 n = width - realsz;
610 while (n-- > 0)
611 KPRINTF_PUTCHAR(' ');
612 }
613
614 /* prefix */
615 if (sign) {
616 KPRINTF_PUTCHAR(sign);
617 } else if (flags & HEXPREFIX) {
618 KPRINTF_PUTCHAR('0');
619 KPRINTF_PUTCHAR(ch);
620 }
621
622 /* right-adjusting zero padding */
623 if ((flags & (LADJUST|ZEROPAD)) == ZEROPAD) {
624 n = width - realsz;
625 while (n-- > 0)
626 KPRINTF_PUTCHAR('0');
627 }
628
629 /* leading zeroes from decimal precision */
630 n = dprec - size;
631 while (n-- > 0)
632 KPRINTF_PUTCHAR('0');
633
634 /* the string or number proper */
635 for (; size--; cp++)
636 KPRINTF_PUTCHAR(*cp);
637 /* left-adjusting padding (always blank) */
638 if (flags & LADJUST) {
639 n = width - realsz;
640 while (n-- > 0)
641 KPRINTF_PUTCHAR(' ');
642 }
643 }
644
645done:
646 if ((oflags == TOBUFONLY) && (vp != NULL))
647 *(char **)vp = sbuf;
648 (*v_flush)();
649
650#ifdef RND_PRINTF
651 if (!cold) {
652 struct timespec ts;
653 (void)nanotime(&ts);
654 SHA512_Update(&kprnd_sha, (char *)&ts, sizeof(ts));
655 }
656#endif
657 return ret;
658}
