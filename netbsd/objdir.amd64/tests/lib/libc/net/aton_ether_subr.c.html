10int
11ether_aton_r(u_char *dest, size_t len, const char *str)
12{
13 const u_char *cp = (const void *)str;
14 u_char *ep;
15
16#define atox(c) (((c) <= '9') ? ((c) - '0') : ((toupper(c) - 'A') + 10))
17
18 if (len < ETHER_ADDR_LEN)
19 return ENOSPC;
20
21 ep = dest + ETHER_ADDR_LEN;
22
23 while (*cp) {
24 if (!isxdigit(*cp))
25 return EINVAL;
26
27 *dest = atox(*cp);
28 cp++;
29 if (isxdigit(*cp)) {
30 *dest = (*dest << 4) | atox(*cp);
31 cp++;
32 }
33 dest++;
34
35 if (dest == ep)
36 return (*cp == '\0') ? 0 : ENAMETOOLONG;
37
38 switch (*cp) {
39 case ':':
40 case '-':
41 case '.':
42 cp++;
43 break;
44 }
45 }
46 return ENOBUFS;
47}
