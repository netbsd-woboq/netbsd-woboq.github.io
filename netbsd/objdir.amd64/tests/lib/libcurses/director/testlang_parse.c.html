1683/*
1684 * Read a response from the command pipe, first we will receive the
1685 * length of the response then the actual data.
1686 */
1687static void
1688read_cmd_pipe(returns_t *response)
1689{
1690 int len, type;
1691 struct pollfd rfd[2];
1692 char *str;
1693
1694 /*
1695 * Check if there is data to read - just in case slave has died, we
1696 * don't want to block on the read and just hang. We also check
1697 * output from the slave because the slave may be blocked waiting
1698 * for a flush on its stdout.
1699 */
1700 rfd[0].fd = slvpipe[READ_PIPE];
1701 rfd[0].events = POLLIN;
1702 rfd[1].fd = master;
1703 rfd[1].events = POLLIN;
1704
1705 do {
1706 if (poll(rfd, 2, 4000) == 0)
1707 errx(2, "%s, %zu: Command pipe read timeout",
1708 cur_file, line);
1709
1710 if ((rfd[1].revents & POLLIN) == POLLIN) {
1711 if (verbose) {
1712 fprintf(stderr,
1713 "draining output from slave\n");
1714 }
1715 save_slave_output(false);
1716 }
1717 }
1718 while((rfd[1].revents & POLLIN) == POLLIN);
1719
1720 if (read(slvpipe[READ_PIPE], &type, sizeof(int)) < 0)
1721 err(1, "command pipe read for type failed");
1722 response->return_type = type;
1723
1724 if ((type != ret_ok) && (type != ret_err) && (type != ret_count)) {
1725 if (read(slvpipe[READ_PIPE], &len, sizeof(int)) < 0)
1726 err(1, "command pipe read for length failed");
1727 response->return_len = len;
1728
1729 if (verbose) {
1730 fprintf(stderr,
1731 "Reading %d bytes from command pipe\n", len);
1732 }
1733
1734 if ((response->return_value = malloc(len + 1)) == NULL)
1735 err(1, "Failed to alloc memory for cmd pipe read");
1736
1737 if (read(slvpipe[READ_PIPE], response->return_value, len) < 0)
1738 err(1, "command pipe read of data failed");
1739
1740 if (response->return_type != ret_byte) {
1741 str = response->return_value;
1742 str[len] = '\0';
1743
1744 if (verbose) {
1745 fprintf(stderr, "Read data >%s< from pipe\n",
1746 (const char *)response->return_value);
1747 }
1748 }
1749 } else {
1750 response->return_value = NULL;
1751 if (type == ret_count) {
1752 if (read(slvpipe[READ_PIPE], &len, sizeof(int)) < 0)
1753 err(1, "command pipe read for number of "
1754 "returns failed");
1755 response->return_len = len;
1756 }
1757
1758 if (verbose) {
1759 fprintf(stderr, "Read type %s from pipe\n",
1760 returns_enum_names[type]);
1761 }
1762 }

1767 * buffer for later checking if discard is false.
1768 */
1769#define MAX_DRAIN 256

1771static void
1772save_slave_output(bool discard)
1773{
1774 char *new_data, drain[MAX_DRAIN];
1775 size_t to_allocate;
1776 ssize_t result;
1777 size_t i;
1778
1779 result = 0;

1781 if (result == -1)
1782 err(2, "poll of slave pty failed");
1783 result = MAX_DRAIN;
1784 if ((result = read(master, drain, result)) < 0) {
1785 if (errno == EAGAIN)
1786 break;
1787 else
1788 err(2, "draining slave pty failed");
1789 }
1790 if (result == 0)

1799 saved_output.allocated + to_allocate))

1802 saved_output.data = new_data;
1803 saved_output.allocated += to_allocate;
1804 }
1805
1806 if (verbose) {
1807 fprintf(stderr, "count = %zu, "
1808 "allocated = %zu\n", saved_output.count,
1809 saved_output.allocated);

1813 saved_output.count + i, drain[i],
1814 (drain[i] >= ' ')? drain[i] : '-');
1815 }

1817

1823 fprintf(stderr, "count = %zu, "
