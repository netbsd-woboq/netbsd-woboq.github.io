59static void
60gldelf_i386_map_segments (bfd_boolean need_layout)
61{
62 int tries = 10;
63
64 do
65 {
66 lang_relax_sections (need_layout);
67 need_layout = FALSE;
68
69 if (link_info.output_bfd->xvec->flavour == bfd_target_elf_flavour
70 && !bfd_link_relocatable (&link_info))
71 {
72 bfd_size_type phdr_size;
73
74 phdr_size = elf_program_header_size (link_info.output_bfd);
75 /* If we don't have user supplied phdrs, throw away any
76 previous linker generated program headers. */
77 if (lang_phdr_list == NULL)
78 elf_seg_map (link_info.output_bfd) = NULL;
79 if (!_bfd_elf_map_sections_to_segments (link_info.output_bfd,
80 &link_info))
81 einfo (_("%F%P: map sections to segments failed: %E\n"));
82
83 if (phdr_size != elf_program_header_size (link_info.output_bfd))
84 {
85 if (tries > 6)
86 /* The first few times we allow any change to
87 phdr_size . */
88 need_layout = TRUE;
89 else if (phdr_size
90 < elf_program_header_size (link_info.output_bfd))
91 /* After that we only allow the size to grow. */
92 need_layout = TRUE;
93 else
94 elf_program_header_size (link_info.output_bfd) = phdr_size;
95 }
96 }
97 }
98 while (need_layout && --tries);
99
100 if (tries == 0)
101 einfo (_("%F%P: looping in map_segments"));
102}

187/* On Linux, it's possible to have different versions of the same
188 shared library linked against different versions of libc. The
189 dynamic linker somehow tags which libc version to use in
190 /etc/ld.so.cache, and, based on the libc that it sees in the
191 executable, chooses which version of the shared library to use.
192
193 We try to do a similar check here by checking whether this shared
194 library needs any other shared libraries which may conflict with
195 libraries we have already included in the link. If it does, we
196 skip it, and try to find another shared library farther on down the
197 link path.
198
199 This is called via lang_for_each_input_file.
200 GLOBAL_VERCHECK_NEEDED is the list of objects needed by the object
201 which we are checking. This sets GLOBAL_VERCHECK_FAILED if we find
202 a conflicting version. */
203
204static void
205gldelf_i386_vercheck (lang_input_statement_type *s)
206{
207 const char *soname;
208 struct bfd_link_needed_list *l;
209
210 if (global_vercheck_failed)
211 return;
212 if (s->the_bfd == NULL
213 || (bfd_get_file_flags (s->the_bfd) & DYNAMIC) == 0)
214 return;
215
216 soname = bfd_elf_get_dt_soname (s->the_bfd);
217 if (soname == NULL)
218 soname = lbasename (bfd_get_filename (s->the_bfd));
219
220 for (l = global_vercheck_needed; l != NULL; l = l->next)
221 {
222 const char *suffix;
223
224 if (filename_cmp (soname, l->name) == 0)
225 {
226 /* Probably can't happen, but it's an easy check. */
227 continue;
228 }
229
230 if (strchr (l->name, '/') != NULL)
231 continue;
232
233 suffix = strstr (l->name, ".so.");
234 if (suffix == NULL)
235 continue;
236
237 suffix += sizeof ".so." - 1;
238
239 if (filename_ncmp (soname, l->name, suffix - l->name) == 0)
240 {
241 /* Here we know that S is a dynamic object FOO.SO.VER1, and
242 the object we are considering needs a dynamic object
243 FOO.SO.VER2, and VER1 and VER2 are different. This
244 appears to be a version mismatch, so we tell the caller
245 to try a different version of this library. */
246 global_vercheck_failed = TRUE;
247 return;
248 }
249 }

256static void
257gldelf_i386_stat_needed (lang_input_statement_type *s)
258{
259 struct stat st;
260 const char *suffix;
261 const char *soname;
262
263 if (global_found != NULL)
264 return;
265 if (s->the_bfd == NULL)
266 return;
267
268 /* If this input file was an as-needed entry, and wasn't found to be
269 needed at the stage it was linked, then don't say we have loaded it. */
270 if ((bfd_elf_get_dyn_lib_class (s->the_bfd) & DYN_AS_NEEDED) != 0)
271 return;
272
273 if (bfd_stat (s->the_bfd, &st) != 0)
274 {
275 einfo (_("%P: %pB: bfd_stat failed: %E\n"), s->the_bfd);
276 return;
277 }
278
279 /* Some operating systems, e.g. Windows, do not provide a meaningful
280 st_ino; they always set it to zero. (Windows does provide a
281 meaningful st_dev.) Do not indicate a duplicate library in that
282 case. While there is no guarantee that a system that provides
283 meaningful inode numbers will never set st_ino to zero, this is
284 merely an optimization, so we do not need to worry about false
285 negatives. */
286 if (st.st_dev == global_stat.st_dev
287 && st.st_ino == global_stat.st_ino
288 && st.st_ino != 0)
289 {
290 global_found = s;
291 return;
292 }
293
294 /* We issue a warning if it looks like we are including two
295 different versions of the same shared library. For example,
296 there may be a problem if -lc picks up libc.so.6 but some other
297 shared library has a DT_NEEDED entry of libc.so.5. This is a
298 heuristic test, and it will only work if the name looks like
299 NAME.so.VERSION. FIXME: Depending on file names is error-prone.
300 If we really want to issue warnings about mixing version numbers
301 of shared libraries, we need to find a better way. */
302
303 if (strchr (global_needed->name, '/') != NULL)
304 return;
305 suffix = strstr (global_needed->name, ".so.");
306 if (suffix == NULL)
307 return;
308 suffix += sizeof ".so." - 1;
309
310 soname = bfd_elf_get_dt_soname (s->the_bfd);
311 if (soname == NULL)
312 soname = lbasename (s->filename);
313
314 if (filename_ncmp (soname, global_needed->name, suffix - global_needed->name) == 0)
315 einfo (_("%P: warning: %s, needed by %pB, may conflict with %s\n"),
316 global_needed->name, global_needed->by, soname);
317}

325/* This function is called for each possible name for a dynamic object
326 named by a DT_NEEDED entry. The FORCE parameter indicates whether
327 to skip the check for a conflicting version. */
328
329static bfd_boolean
330gldelf_i386_try_needed (struct dt_needed *needed,
331 int force)
332{
333 bfd *abfd;
334 const char *name = needed->name;
335 const char *soname;
336 int link_class;
337
338 abfd = bfd_openr (name, bfd_get_target (link_info.output_bfd));
339 if (abfd == NULL)
340 {
341 if (verbose)
342 info_msg (_("attempt to open %s failed\n"), name);
343 return FALSE;
344 }
345
346 /* Linker needs to decompress sections. */
347 abfd->flags |= BFD_DECOMPRESS;
348
349 if (! bfd_check_format (abfd, bfd_object))
350 {
351 bfd_close (abfd);
352 return FALSE;
353 }
354 if ((bfd_get_file_flags (abfd) & DYNAMIC) == 0)
355 {
356 bfd_close (abfd);
357 return FALSE;
358 }
359
360 /* For DT_NEEDED, they have to match. */
361 if (abfd->xvec != link_info.output_bfd->xvec)
362 {
363 bfd_close (abfd);
364 return FALSE;
365 }
366
367 /* Check whether this object would include any conflicting library
368 versions. If FORCE is set, then we skip this check; we use this
369 the second time around, if we couldn't find any compatible
370 instance of the shared library. */
371
372 if (! force)
373 {
374 struct bfd_link_needed_list *needs;
375
376 if (! bfd_elf_get_bfd_needed_list (abfd, &needs))
377 einfo (_("%F%P: %pB: bfd_elf_get_bfd_needed_list failed: %E\n"), abfd);
378
379 if (needs != NULL)
380 {
381 global_vercheck_needed = needs;
382 global_vercheck_failed = FALSE;
383 lang_for_each_input_file (gldelf_i386_vercheck);
384 if (global_vercheck_failed)
385 {
386 bfd_close (abfd);
387 /* Return FALSE to force the caller to move on to try
388 another file on the search path. */
389 return FALSE;
390 }
391
392 /* But wait! It gets much worse. On Linux, if a shared
393 library does not use libc at all, we are supposed to skip
394 it the first time around in case we encounter a shared
395 library later on with the same name which does use the
396 version of libc that we want. This is much too horrible
397 to use on any system other than Linux. */
398
399 }
400 }
401
402 /* We've found a dynamic object matching the DT_NEEDED entry. */
403
404 /* We have already checked that there is no other input file of the
405 same name. We must now check again that we are not including the
406 same file twice. We need to do this because on many systems
407 libc.so is a symlink to, e.g., libc.so.1. The SONAME entry will
408 reference libc.so.1. If we have already included libc.so, we
409 don't want to include libc.so.1 if they are the same file, and we
410 can only check that using stat. */
411
412 if (bfd_stat (abfd, &global_stat) != 0)
413 einfo (_("%F%P: %pB: bfd_stat failed: %E\n"), abfd);
414
415 /* First strip off everything before the last '/'. */
416 soname = lbasename (abfd->filename);
417
418 if (verbose)
419 info_msg (_("found %s at %s\n"), soname, name);
420
421 global_found = NULL;
422 lang_for_each_input_file (gldelf_i386_stat_needed);
423 if (global_found != NULL)
424 {
425 /* Return TRUE to indicate that we found the file, even though
426 we aren't going to do anything with it. */
427 return TRUE;
428 }
429
430 /* Specify the soname to use. */
431 bfd_elf_set_dt_needed_name (abfd, soname);
432
433 /* Tell the ELF linker that we don't want the output file to have a
434 DT_NEEDED entry for this file, unless it is used to resolve
435 references in a regular object. */
436 link_class = DYN_DT_NEEDED;
437
438 /* Tell the ELF linker that we don't want the output file to have a
439 DT_NEEDED entry for this file at all if the entry is from a file
440 with DYN_NO_ADD_NEEDED. */
441 if (needed->by != NULL
442 && (bfd_elf_get_dyn_lib_class (needed->by) & DYN_NO_ADD_NEEDED) != 0)
443 link_class |= DYN_NO_NEEDED | DYN_NO_ADD_NEEDED;
444
445 bfd_elf_set_dyn_lib_class (abfd, (enum dynamic_lib_link_class) link_class);
446
447 /* Add this file into the symbol table. */
448 if (! bfd_link_add_symbols (abfd, &link_info))
449 einfo (_("%F%P: %pB: error adding symbols: %E\n"), abfd);
450
451 return TRUE;
452}

456static bfd_boolean
457gldelf_i386_search_needed (const char *path,
458 struct dt_needed *n, int force, int prepend_sysroot)
459{
460 const char *s;
461 const char *name = n->name;
462 size_t len;
463 struct dt_needed needed;
464
465 if (name[0] == '/') {
466 if (prepend_sysroot && ld_sysroot) {
467 bfd_boolean rv;
468 needed.by = n->by;
469 char *filename= concat(ld_sysroot, n->name, (const char *)NULL);
470 needed.name = filename;
471 rv = gldelf_i386_try_needed (&needed, force);
472 free(filename);
473 return rv;
474 }
475 return gldelf_i386_try_needed (n, force);
476 }
477
478 if (path == NULL || *path == '\0')
479 return FALSE;
480
481 needed.by = n->by;
482 needed.name = n->name;
483
484 len = strlen (name);
485 while (1)
486 {
487 unsigned offset = 0;
488 char * var;
489 char *filename, *sset;
490
491 s = strchr (path, config.rpath_separator);
492 if (s == NULL)
493 s = path + strlen (path);
494
495#if HAVE_DOS_BASED_FILE_SYSTEM
496 /* Assume a match on the second char is part of drive specifier. */
497 else if (config.rpath_separator == ':'
498 && s == path + 1
499 && ISALPHA (*path))
500 {
501 s = strchr (s + 1, config.rpath_separator);
502 if (s == NULL)
503 s = path + strlen (path);
504 }
505#endif
506 filename = (char *) xmalloc (s - path + len + 2);
507 if (s == path)
508 sset = filename;
509 else
510 {
511 memcpy (filename, path, s - path);
512 filename[s - path] = '/';
513 sset = filename + (s - path) + 1;
514 }
515 strcpy (sset, name);
516
517 if (prepend_sysroot && filename[0] == '=')
518 abort();
519 if (filename[0] == '/' && prepend_sysroot && ld_sysroot) {
520 char *filename2 = concat(ld_sysroot, filename, (const char *)NULL);
521 free(filename);
522 filename = filename2;
523 }
524 /* PR 20535: Support the same pseudo-environment variables that
525 are supported by ld.so. Namely, , and .
526 Since there can be more than one occurrence of these tokens in
527 the path we loop until no more are found. Since we might not
528 be able to substitute some of the tokens we maintain an offset
529 into the filename for where we should begin our scan. */
530 while ((var = strchr (filename + offset, '$')) != NULL)
531 {
532 /* The ld.so manual page does not say, but I am going to assume that
533 these tokens are terminated by a directory separator character
534 (/) or the end of the string. There is also an implication that
535 should only be used at the start of a path, but that is
536 not enforced here.
537
538 The ld.so manual page also states that it allows ,
539 and , so these are supported as well.
540
541 FIXME: The code could be a lot cleverer about allocating space
542 for the processed string. */
543 char * end = strchr (var, '/');
544 char * replacement = NULL;
545 char * v = var + 1;
546 char * freeme = NULL;
547 unsigned flen = strlen (filename);
548
549 if (end != NULL)
550 /* Temporarily terminate the filename at the end of the token. */
551 * end = 0;
552
553 if (*v == '{')
554 ++ v;
555 switch (*v++)
556 {
557 case 'O':
558 if (strcmp (v, "RIGIN") == 0 || strcmp (v, "RIGIN}") == 0)
559 {
560 /* ORIGIN - replace with the full path to the directory
561 containing the program or shared object. */
562 if (needed.by == NULL)
563 {
564 if (link_info.output_bfd == NULL)
565 {
566 break;
567 }
568 else
569 replacement = bfd_get_filename (link_info.output_bfd);
570 }
571 else
572 replacement = bfd_get_filename (needed.by);
573
574 if (replacement)
575 {
576 char * slash;
577
578 if (replacement[0] == '/')
579 freeme = xstrdup (replacement);
580 else
581 {
582 char * current_dir = getpwd ();
583
584 freeme = xmalloc (strlen (replacement) + strlen (current_dir) + 2);
585 sprintf (freeme, "%s/%s", current_dir, replacement);
586 }
587
588 replacement = freeme;
589 if ((slash = strrchr (replacement, '/')) != NULL)
590 * slash = 0;
591 }
592 }
593 break;
594
595 case 'L':
596 if (strcmp (v, "IB") == 0 || strcmp (v, "IB}") == 0)
597 {
598 /* LIB - replace with "lib" in 32-bit environments
599 and "lib64" in 64-bit environments. */
600
601 /* Note - we could replace this switch statement by
602 conditional fragments of shell script, but that is messy.
603 Any compiler worth its salt is going to optimize away
604 all but one of these case statements anyway. */
605 switch (32)
606 {
607 case 32: replacement = "lib"; break;
608 case 64: replacement = "lib64"; break;
609 default:
610 /* 32 is not 32 or 64 ... */
611 abort ();
612 }
613 }
614 break;
615
616 case 'P':
617 /* Supporting in a cross-hosted environment is not
618 possible. Supporting it in a native environment involves
619 loading the <sys/auxv.h> header file which loads the
620 system <elf.h> header file, which conflicts with the
621 "include/elf/mips.h" header file. */
622 /* Fall through. */
623 default:
624 break;
625 }
626
627 if (replacement)
628 {
629 char * filename2 = xmalloc (flen + strlen (replacement));
630
631 if (end)
632 {
633 sprintf (filename2, "%.*s%s/%s",
634 (int)(var - filename), filename,
635 replacement, end + 1);
636 offset = (var - filename) + 1 + strlen (replacement);
637 }
638 else
639 {
640 sprintf (filename2, "%.*s%s",
641 (int)(var - filename), filename,
642 replacement);
643 offset = var - filename + strlen (replacement);
644 }
645
646 free (filename);
647 filename = filename2;
648 /* There is no need to restore the path separator (when
649 end != NULL) as we have replaced the entire string. */
650 }
651 else
652 {
653 if (verbose)
654 /* We only issue an "unrecognised" message in verbose mode
655 as the $<foo> token might be a legitimate component of
656 a path name in the target's file system. */
657 info_msg (_("unrecognised or unsupported token '%s' in search path\n"), var);
658
659 if (end)
660 /* Restore the path separator. */
661 * end = '/';
662
663 /* PR 20784: Make sure that we resume the scan *after*
664 the token that we could not replace. */
665 offset = (var + 1) - filename;
666 }
667
668 free (freeme);
669 }
670
671 needed.name = filename;
672
673 if (gldelf_i386_try_needed (&needed, force))
674 return TRUE;
675
676 free (filename);
677
678 if (*s == '\0')
679 break;
680 path = s + 1;
681 }
682
683 return FALSE;
684}

687/* See if an input file matches a DT_NEEDED entry by name. */
688
689static void
690gldelf_i386_check_needed (lang_input_statement_type *s)
691{
692 const char *soname;
693
694 /* Stop looking if we've found a loaded lib. */
695 if (global_found != NULL
696 && (bfd_elf_get_dyn_lib_class (global_found->the_bfd)
697 & DYN_AS_NEEDED) == 0)
698 return;
699
700 if (s->filename == NULL || s->the_bfd == NULL)
701 return;
702
703 /* Don't look for a second non-loaded as-needed lib. */
704 if (global_found != NULL
705 && (bfd_elf_get_dyn_lib_class (s->the_bfd) & DYN_AS_NEEDED) != 0)
706 return;
707
708 if (filename_cmp (s->filename, global_needed->name) == 0)
709 {
710 global_found = s;
711 return;
712 }
713
714 if (s->flags.search_dirs)
715 {
716 const char *f = strrchr (s->filename, '/');
717 if (f != NULL
718 && filename_cmp (f + 1, global_needed->name) == 0)
719 {
720 global_found = s;
721 return;
722 }
723 }
724
725 soname = bfd_elf_get_dt_soname (s->the_bfd);
726 if (soname != NULL
727 && filename_cmp (soname, global_needed->name) == 0)
728 {
729 global_found = s;
730 return;
731 }
732}

735static bfd_size_type
736id_note_section_size (bfd *abfd ATTRIBUTE_UNUSED)
737{
738 const char *style = emit_note_gnu_build_id;
739 bfd_size_type size;
740 bfd_size_type build_id_size;
741
742 size = offsetof (Elf_External_Note, name[sizeof "GNU"]);
743 size = (size + 3) & -(bfd_size_type) 4;
744
745 build_id_size = compute_build_id_size (style);
746 if (build_id_size)
747 size += build_id_size;
748 else
749 size = 0;
750
751 return size;
752}

754static bfd_boolean
755write_build_id (bfd *abfd)
756{
757 const struct elf_backend_data *bed = get_elf_backend_data (abfd);
758 struct elf_obj_tdata *t = elf_tdata (abfd);
759 const char *style;
760 asection *asec;
761 Elf_Internal_Shdr *i_shdr;
762 unsigned char *contents, *id_bits;
763 bfd_size_type size;
764 file_ptr position;
765 Elf_External_Note *e_note;
766
767 style = t->o->build_id.style;
768 asec = t->o->build_id.sec;
769 if (bfd_is_abs_section (asec->output_section))
770 {
771 einfo (_("%P: warning: .note.gnu.build-id section discarded,"
772 " --build-id ignored\n"));
773 return TRUE;
774 }
775 i_shdr = &elf_section_data (asec->output_section)->this_hdr;
776
777 if (i_shdr->contents == NULL)
778 {
779 if (asec->contents == NULL)
780 asec->contents = (unsigned char *) xmalloc (asec->size);
781 contents = asec->contents;
782 }
783 else
784 contents = i_shdr->contents + asec->output_offset;
785
786 e_note = (Elf_External_Note *) contents;
787 size = offsetof (Elf_External_Note, name[sizeof "GNU"]);
788 size = (size + 3) & -(bfd_size_type) 4;
789 id_bits = contents + size;
790 size = asec->size - size;
791
792 bfd_h_put_32 (abfd, sizeof "GNU", &e_note->namesz);
793 bfd_h_put_32 (abfd, size, &e_note->descsz);
794 bfd_h_put_32 (abfd, NT_GNU_BUILD_ID, &e_note->type);
795 memcpy (e_note->name, "GNU", sizeof "GNU");
796
797 generate_build_id (abfd, style, bed->s->checksum_contents, id_bits, size);
798
799 position = i_shdr->sh_offset + asec->output_offset;
800 size = asec->size;
801 return (bfd_seek (abfd, position, SEEK_SET) == 0
802 && bfd_bwrite (contents, size, abfd) == size);
803}

840/* This is called after all the input files have been opened. */
841
842static void
843gldelf_i386_after_open (void)
844{
845 struct bfd_link_needed_list *needed, *l;
846 struct elf_link_hash_table *htab;
847 asection *s;
848 bfd *abfd;
849
850 after_open_default ();
851
852 htab = elf_hash_table (&link_info);
853 if (!is_elf_hash_table (htab))
854 return;
855
856 if (command_line.out_implib_filename)
857 {
858 unlink_if_ordinary (command_line.out_implib_filename);
859 link_info.out_implib_bfd
860 = bfd_openw (command_line.out_implib_filename,
861 bfd_get_target (link_info.output_bfd));
862
863 if (link_info.out_implib_bfd == NULL)
864 {
865 einfo (_("%F%P: %s: can't open for writing: %E\n"),
866 command_line.out_implib_filename);
867 }
868 }
869
870 if (emit_note_gnu_build_id != NULL)
871 {
872 /* Find an ELF input. */
873 for (abfd = link_info.input_bfds;
874 abfd != (bfd *) NULL; abfd = abfd->link.next)
875 if (bfd_get_flavour (abfd) == bfd_target_elf_flavour
876 && bfd_count_sections (abfd) != 0
877 && !((lang_input_statement_type *) abfd->usrdata)->flags.just_syms)
878 break;
879
880 /* PR 10555: If there are no ELF input files do not try to
881 create a .note.gnu-build-id section. */
882 if (abfd == NULL
883 || !setup_build_id (abfd))
884 {
885 free ((char *) emit_note_gnu_build_id);
886 emit_note_gnu_build_id = NULL;
887 }
888 }
889
890 get_elf_backend_data (link_info.output_bfd)->setup_gnu_properties (&link_info);
891
892 if (bfd_link_relocatable (&link_info))
893 {
894 if (link_info.execstack == ! link_info.noexecstack)
895 /* PR ld/16744: If "-z [no]execstack" has been specified on the
896 command line and we are perfoming a relocatable link then no
897 PT_GNU_STACK segment will be created and so the
898 linkinfo.[no]execstack values set in _handle_option() will have no
899 effect. Instead we create a .note.GNU-stack section in much the
900 same way as the assembler does with its --[no]execstack option. */
901 (void) bfd_make_section_with_flags (link_info.input_bfds,
902 ".note.GNU-stack",
903 SEC_READONLY | (link_info.execstack ? SEC_CODE : 0));
904
905 return;
906 }
907
908 if (!link_info.traditional_format)
909 {
910 bfd *elfbfd = NULL;
911 bfd_boolean warn_eh_frame = FALSE;
912 int seen_type = 0;
913
914 for (abfd = link_info.input_bfds; abfd; abfd = abfd->link.next)
915 {
916 int type = 0;
917
918 if (((lang_input_statement_type *) abfd->usrdata)->flags.just_syms)
919 continue;
920
921 for (s = abfd->sections; s && type < COMPACT_EH_HDR; s = s->next)
922 {
923 const char *name = bfd_get_section_name (abfd, s);
924
925 if (bfd_is_abs_section (s->output_section))
926 continue;
927 if (CONST_STRNEQ (name, ".eh_frame_entry"))
928 type = COMPACT_EH_HDR;
929 else if (strcmp (name, ".eh_frame") == 0 && s->size > 8)
930 type = DWARF2_EH_HDR;
931 }
932
933 if (type != 0)
934 {
935 if (seen_type == 0)
936 {
937 seen_type = type;
938 }
939 else if (seen_type != type)
940 {
941 einfo (_("%F%P: compact frame descriptions incompatible with"
942 " DWARF2 .eh_frame from %pB\n"),
943 type == DWARF2_EH_HDR ? abfd : elfbfd);
944 break;
945 }
946
947 if (!elfbfd
948 && (type == COMPACT_EH_HDR || link_info.eh_frame_hdr_type != 0))
949 {
950 if (bfd_get_flavour (abfd) == bfd_target_elf_flavour)
951 elfbfd = abfd;
952
953 warn_eh_frame = TRUE;
954 }
955 }
956
957 if (seen_type == COMPACT_EH_HDR)
958 link_info.eh_frame_hdr_type = COMPACT_EH_HDR;
959 }
960 if (elfbfd)
961 {
962 const struct elf_backend_data *bed;
963
964 bed = get_elf_backend_data (elfbfd);
965 s = bfd_make_section_with_flags (elfbfd, ".eh_frame_hdr",
966 bed->dynamic_sec_flags
967 | SEC_READONLY);
968 if (s != NULL
969 && bfd_set_section_alignment (elfbfd, s, 2))
970 {
971 htab->eh_info.hdr_sec = s;
972 warn_eh_frame = FALSE;
973 }
974 }
975 if (warn_eh_frame)
976 einfo (_("%P: warning: cannot create .eh_frame_hdr section,"
977 " --eh-frame-hdr ignored\n"));
978 }
979
980 /* Get the list of files which appear in DT_NEEDED entries in
981 dynamic objects included in the link (often there will be none).
982 For each such file, we want to track down the corresponding
983 library, and include the symbol table in the link. This is what
984 the runtime dynamic linker will do. Tracking the files down here
985 permits one dynamic object to include another without requiring
986 special action by the person doing the link. Note that the
987 needed list can actually grow while we are stepping through this
988 loop. */
989 needed = bfd_elf_get_needed_list (link_info.output_bfd, &link_info);
990 for (l = needed; l != NULL; l = l->next)
991 {
992 struct bfd_link_needed_list *ll;
993 struct dt_needed n, nn;
994 int force;
995
996 /* If the lib that needs this one was --as-needed and wasn't
997 found to be needed, then this lib isn't needed either. */
998 if (l->by != NULL
999 && (bfd_elf_get_dyn_lib_class (l->by) & DYN_AS_NEEDED) != 0)
1000 continue;
1001
1002 /* Skip the lib if --no-copy-dt-needed-entries and
1003 --allow-shlib-undefined is in effect. */
1004 if (l->by != NULL
1005 && link_info.unresolved_syms_in_shared_libs == RM_IGNORE
1006 && (bfd_elf_get_dyn_lib_class (l->by) & DYN_NO_ADD_NEEDED) != 0)
1007 continue;
1008
1009 /* If we've already seen this file, skip it. */
1010 for (ll = needed; ll != l; ll = ll->next)
1011 if ((ll->by == NULL
1012 || (bfd_elf_get_dyn_lib_class (ll->by) & DYN_AS_NEEDED) == 0)
1013 && strcmp (ll->name, l->name) == 0)
1014 break;
1015 if (ll != l)
1016 continue;
1017
1018 /* See if this file was included in the link explicitly. */
1019 global_needed = l;
1020 global_found = NULL;
1021 lang_for_each_input_file (gldelf_i386_check_needed);
1022 if (global_found != NULL
1023 && (bfd_elf_get_dyn_lib_class (global_found->the_bfd)
1024 & DYN_AS_NEEDED) == 0)
1025 continue;
1026
1027 n.by = l->by;
1028 n.name = l->name;
1029 nn.by = l->by;
1030 if (verbose)
1031 info_msg (_("%s needed by %pB\n"), l->name, l->by);
1032
1033 /* As-needed libs specified on the command line (or linker script)
1034 take priority over libs found in search dirs. */
1035 if (global_found != NULL)
1036 {
1037 nn.name = global_found->filename;
1038 if (gldelf_i386_try_needed (&nn, TRUE))
1039 continue;
1040 }
1041
1042 /* We need to find this file and include the symbol table. We
1043 want to search for the file in the same way that the dynamic
1044 linker will search. That means that we want to use
1045 rpath_link, rpath, then the environment variable
1046 LD_LIBRARY_PATH (native only), then the DT_RPATH/DT_RUNPATH
1047 entries (native only), then the linker script LIB_SEARCH_DIRS.
1048 We do not search using the -L arguments.
1049
1050 We search twice. The first time, we skip objects which may
1051 introduce version mismatches. The second time, we force
1052 their use. See gldelf_i386_vercheck comment. */
1053 for (force = 0; force < 2; force++)
1054 {
1055 size_t len;
1056 search_dirs_type *search;
1057
1058 if (gldelf_i386_search_needed (command_line.rpath_link,
1059 &n, force, 0))
1060 break;
1061 len = strlen (l->name);
1062 for (search = search_head; search != NULL; search = search->next)
1063 {
1064 char *filename;
1065
1066 if (search->cmdline)
1067 continue;
1068 filename = (char *) xmalloc (strlen (search->name) + len + 2);
1069 sprintf (filename, "%s/%s", search->name, l->name);
1070 nn.name = filename;
1071 if (gldelf_i386_try_needed (&nn, force))
1072 break;
1073 free (filename);
1074 }
1075 if (search != NULL)
1076 break;
1077 }
1078
1079 if (force < 2)
1080 continue;
1081
1082 einfo (_("%P: warning: %s, needed by %pB, not found "
1083 "(try using -rpath or -rpath-link)\n"),
1084 l->name, l->by);
1085 }
1086
1087 if (link_info.eh_frame_hdr_type == COMPACT_EH_HDR)
1088 if (!bfd_elf_parse_eh_frame_entries (NULL, &link_info))
1089 einfo (_("%F%P: failed to parse EH frame entries\n"));
1090}

1093/* Look through an expression for an assignment statement. */
1094
1095static void
1096gldelf_i386_find_exp_assignment (etree_type *exp)
1097{
1098 bfd_boolean provide = FALSE;
1099
1100 switch (exp->type.node_class)
1101 {
1102 case etree_provide:
1103 case etree_provided:
1104 provide = TRUE;
1105 /* Fallthru */
1106 case etree_assign:
1107 /* We call record_link_assignment even if the symbol is defined.
1108 This is because if it is defined by a dynamic object, we
1109 actually want to use the value defined by the linker script,
1110 not the value from the dynamic object (because we are setting
1111 symbols like etext). If the symbol is defined by a regular
1112 object, then, as it happens, calling record_link_assignment
1113 will do no harm. */
1114 if (strcmp (exp->assign.dst, ".") != 0)
1115 {
1116 if (!bfd_elf_record_link_assignment (link_info.output_bfd,
1117 &link_info,
1118 exp->assign.dst, provide,
1119 exp->assign.hidden))
1120 einfo (_("%F%P: failed to record assignment to %s: %E\n"),
1121 exp->assign.dst);
1122 }
1123 gldelf_i386_find_exp_assignment (exp->assign.src);
1124 break;
1125
1126 case etree_binary:
1127 gldelf_i386_find_exp_assignment (exp->binary.lhs);
1128 gldelf_i386_find_exp_assignment (exp->binary.rhs);
1129 break;
1130
1131 case etree_trinary:
1132 gldelf_i386_find_exp_assignment (exp->trinary.cond);
1133 gldelf_i386_find_exp_assignment (exp->trinary.lhs);
1134 gldelf_i386_find_exp_assignment (exp->trinary.rhs);
1135 break;
1136
1137 case etree_unary:
1138 gldelf_i386_find_exp_assignment (exp->unary.child);
1139 break;
1140
1141 default:
1142 break;
1143 }
1144}

1205/* This is called after the sections have been attached to output
1206 sections, but before any sizes or addresses have been set. */
1207
1208static void
1209gldelf_i386_before_allocation (void)
1210{
1211 const char *rpath;
1212 asection *sinterp;
1213 bfd *abfd;
1214 struct elf_link_hash_entry *ehdr_start = NULL;
1215#if defined(__GNUC__) && GCC_VERSION < 4006
1216 /* Work around a GCC uninitialized warning bug fixed in GCC 4.6. */
1217 struct bfd_link_hash_entry ehdr_start_save = ehdr_start_empty;
1218#else
1219 struct bfd_link_hash_entry ehdr_start_save;
1220#endif
1221
1222 if (is_elf_hash_table (link_info.hash))
1223 {
1224 _bfd_elf_tls_setup (link_info.output_bfd, &link_info);
1225
1226 /* Make __ehdr_start hidden if it has been referenced, to
1227 prevent the symbol from being dynamic. */
1228 if (!bfd_link_relocatable (&link_info))
1229 {
1230 struct elf_link_hash_table *htab = elf_hash_table (&link_info);
1231 struct elf_link_hash_entry *h
1232 = elf_link_hash_lookup (htab, "__ehdr_start", FALSE, FALSE, TRUE);
1233
1234 /* Only adjust the export class if the symbol was referenced
1235 and not defined, otherwise leave it alone. */
1236 if (h != NULL
1237 && (h->root.type == bfd_link_hash_new
1238 || h->root.type == bfd_link_hash_undefined
1239 || h->root.type == bfd_link_hash_undefweak
1240 || h->root.type == bfd_link_hash_common))
1241 {
1242 const struct elf_backend_data *bed;
1243 bed = get_elf_backend_data (link_info.output_bfd);
1244 (*bed->elf_backend_hide_symbol) (&link_info, h, TRUE);
1245 if (ELF_ST_VISIBILITY (h->other) != STV_INTERNAL)
1246 h->other = (h->other & ~ELF_ST_VISIBILITY (-1)) | STV_HIDDEN;
1247 /* Don't leave the symbol undefined. Undefined hidden
1248 symbols typically won't have dynamic relocations, but
1249 we most likely will need dynamic relocations for
1250 __ehdr_start if we are building a PIE or shared
1251 library. */
1252 ehdr_start = h;
1253 ehdr_start_save = h->root;
1254 h->root.type = bfd_link_hash_defined;
1255 h->root.u.def.section = bfd_abs_section_ptr;
1256 h->root.u.def.value = 0;
1257 }
1258 }
1259
1260 /* If we are going to make any variable assignments, we need to
1261 let the ELF backend know about them in case the variables are
1262 referred to by dynamic objects. */
1263 lang_for_each_statement (gldelf_i386_find_statement_assignment);
1264 }
1265
1266 /* Let the ELF backend work out the sizes of any sections required
1267 by dynamic linking. */
1268 rpath = command_line.rpath;
1269 if (rpath == NULL)
1270 rpath = (const char *) getenv ("LD_RUN_PATH");
1271
1272 for (abfd = link_info.input_bfds; abfd; abfd = abfd->link.next)
1273 if (bfd_get_flavour (abfd) == bfd_target_elf_flavour)
1274 {
1275 const char *audit_libs = elf_dt_audit (abfd);
1276
1277 /* If the input bfd contains an audit entry, we need to add it as
1278 a dep audit entry. */
1279 if (audit_libs && *audit_libs != '\0')
1280 {
1281 char *cp, *fcp;
1282 fcp = cp = xstrdup (audit_libs);
1283 do
1284 {
1285 int more = 0;
1286 char *cp2 = strchr (cp, config.rpath_separator);
1287
1288 if (cp2)
1289 {
1290 *cp2 = '\0';
1291 more = 1;
1292 }
1293
1294 if (cp != NULL && *cp != '\0')
1295 gldelf_i386_append_to_separated_string (&depaudit, cp);
1296
1297 cp = more ? ++cp2 : NULL;
1298 }
1299 while (cp != NULL);
1300 free (fcp);
1301 }
1302 }
1303
1304 if (! (bfd_elf_size_dynamic_sections
1305 (link_info.output_bfd, command_line.soname, rpath,
1306 command_line.filter_shlib, audit, depaudit,
1307 (const char * const *) command_line.auxiliary_filters,
1308 &link_info, &sinterp)))
1309 einfo (_("%F%P: failed to set dynamic section sizes: %E\n"));
1310
1311
1312 /* Let the user override the dynamic linker we are using. */
1313 if (command_line.interpreter != NULL
1314 && sinterp != NULL)
1315 {
1316 sinterp->contents = (bfd_byte *) command_line.interpreter;
1317 sinterp->size = strlen (command_line.interpreter) + 1;
1318 }
1319
1320 /* Look for any sections named .gnu.warning. As a GNU extensions,
1321 we treat such sections as containing warning messages. We print
1322 out the warning message, and then zero out the section size so
1323 that it does not get copied into the output file. */
1324
1325 {
1326 LANG_FOR_EACH_INPUT_STATEMENT (is)
1327 {
1328 asection *s;
1329 bfd_size_type sz;
1330 char *msg;
1331
1332 if (is->flags.just_syms)
1333 continue;
1334
1335 s = bfd_get_section_by_name (is->the_bfd, ".gnu.warning");
1336 if (s == NULL)
1337 continue;
1338
1339 sz = s->size;
1340 msg = (char *) xmalloc ((size_t) (sz + 1));
1341 if (! bfd_get_section_contents (is->the_bfd, s, msg,
1342 (file_ptr) 0, sz))
1343 einfo (_("%F%P: %pB: can't read contents of section .gnu.warning: %E\n"),
1344 is->the_bfd);
1345 msg[sz] = '\0';
1346 (*link_info.callbacks->warning) (&link_info, msg,
1347 (const char *) NULL, is->the_bfd,
1348 (asection *) NULL, (bfd_vma) 0);
1349 free (msg);
1350
1351 /* Clobber the section size, so that we don't waste space
1352 copying the warning into the output file. If we've already
1353 sized the output section, adjust its size. The adjustment
1354 is on rawsize because targets that size sections early will
1355 have called lang_reset_memory_regions after sizing. */
1356 if (s->output_section != NULL
1357 && s->output_section->rawsize >= s->size)
1358 s->output_section->rawsize -= s->size;
1359
1360 s->size = 0;
1361
1362 /* Also set SEC_EXCLUDE, so that local symbols defined in the
1363 warning section don't get copied to the output. */
1364 s->flags |= SEC_EXCLUDE | SEC_KEEP;
1365 }
1366 }
1367
1368 before_allocation_default ();
1369
1370 if (!bfd_elf_size_dynsym_hash_dynstr (link_info.output_bfd, &link_info))
1371 einfo (_("%F%P: failed to set dynamic section sizes: %E\n"));
1372
1373 if (ehdr_start != NULL)
1374 {
1375 /* If we twiddled __ehdr_start to defined earlier, put it back
1376 as it was. */
1377 ehdr_start->root.type = ehdr_start_save.type;
1378 ehdr_start->root.u = ehdr_start_save.u;
1379 }
1380}

1383/* Try to open a dynamic archive. This is where we know that ELF
1384 dynamic libraries have an extension of .so (or .sl on oddball systems
1385 like hpux). */
1386
1387static bfd_boolean
1388gldelf_i386_open_dynamic_archive
1389 (const char *arch, search_dirs_type *search, lang_input_statement_type *entry)
1390{
1391 const char *filename;
1392 char *string;
1393 size_t len;
1394 bfd_boolean opened = FALSE;
1395
1396 if (! entry->flags.maybe_archive)
1397 return FALSE;
1398
1399 filename = entry->filename;
1400 len = strlen (search->name) + strlen (filename);
1401 if (entry->flags.full_name_provided)
1402 {
1403 len += sizeof "/";
1404 string = (char *) xmalloc (len);
1405 sprintf (string, "%s/%s", search->name, filename);
1406 }
1407 else
1408 {
1409 size_t xlen = 0;
1410
1411 len += strlen (arch) + sizeof "/lib.so";
1412#ifdef EXTRA_SHLIB_EXTENSION
1413 xlen = (strlen (EXTRA_SHLIB_EXTENSION) > 3
1414 ? strlen (EXTRA_SHLIB_EXTENSION) - 3
1415 : 0);
1416#endif
1417 string = (char *) xmalloc (len + xlen);
1418 sprintf (string, "%s/lib%s%s.so", search->name, filename, arch);
1419#ifdef EXTRA_SHLIB_EXTENSION
1420 /* Try the .so extension first. If that fails build a new filename
1421 using EXTRA_SHLIB_EXTENSION. */
1422 opened = ldfile_try_open_bfd (string, entry);
1423 if (!opened)
1424 strcpy (string + len - 4, EXTRA_SHLIB_EXTENSION);
1425#endif
1426 }
1427
1428 if (!opened && !ldfile_try_open_bfd (string, entry))
1429 {
1430 free (string);
1431 return FALSE;
1432 }
1433
1434 entry->filename = string;
1435
1436 /* We have found a dynamic object to include in the link. The ELF
1437 backend linker will create a DT_NEEDED entry in the .dynamic
1438 section naming this file. If this file includes a DT_SONAME
1439 entry, it will be used. Otherwise, the ELF linker will just use
1440 the name of the file. For an archive found by searching, like
1441 this one, the DT_NEEDED entry should consist of just the name of
1442 the file, without the path information used to find it. Note
1443 that we only need to do this if we have a dynamic object; an
1444 archive will never be referenced by a DT_NEEDED entry.
1445
1446 FIXME: This approach--using bfd_elf_set_dt_needed_name--is not
1447 very pretty. I haven't been able to think of anything that is
1448 pretty, though. */
1449 if (bfd_check_format (entry->the_bfd, bfd_object)
1450 && (entry->the_bfd->flags & DYNAMIC) != 0)
1451 {
1452 ASSERT (entry->flags.maybe_archive && entry->flags.search_dirs);
1453
1454 /* Rather than duplicating the logic above. Just use the
1455 filename we recorded earlier. */
1456
1457 if (!entry->flags.full_name_provided)
1458 filename = lbasename (entry->filename);
1459 bfd_elf_set_dt_needed_name (entry->the_bfd, filename);
1460 }
1461
1462 return TRUE;
1463}

1466/* A variant of lang_output_section_find used by place_orphan. */
1467
1468static lang_output_section_statement_type *
1469output_rel_find (asection *sec, int isdyn)
1470{
1471 lang_output_section_statement_type *lookup;
1472 lang_output_section_statement_type *last = NULL;
1473 lang_output_section_statement_type *last_alloc = NULL;
1474 lang_output_section_statement_type *last_ro_alloc = NULL;
1475 lang_output_section_statement_type *last_rel = NULL;
1476 lang_output_section_statement_type *last_rel_alloc = NULL;
1477 int rela = sec->name[4] == 'a';
1478
1479 for (lookup = &lang_output_section_statement.head->output_section_statement;
1480 lookup != NULL;
1481 lookup = lookup->next)
1482 {
1483 if (lookup->constraint >= 0
1484 && CONST_STRNEQ (lookup->name, ".rel"))
1485 {
1486 int lookrela = lookup->name[4] == 'a';
1487
1488 /* .rel.dyn must come before all other reloc sections, to suit
1489 GNU ld.so. */
1490 if (isdyn)
1491 break;
1492
1493 /* Don't place after .rel.plt as doing so results in wrong
1494 dynamic tags. */
1495 if (strcmp (".plt", lookup->name + 4 + lookrela) == 0)
1496 break;
1497
1498 if (rela == lookrela || last_rel == NULL)
1499 last_rel = lookup;
1500 if ((rela == lookrela || last_rel_alloc == NULL)
1501 && lookup->bfd_section != NULL
1502 && (lookup->bfd_section->flags & SEC_ALLOC) != 0)
1503 last_rel_alloc = lookup;
1504 }
1505
1506 last = lookup;
1507 if (lookup->bfd_section != NULL
1508 && (lookup->bfd_section->flags & SEC_ALLOC) != 0)
1509 {
1510 last_alloc = lookup;
1511 if ((lookup->bfd_section->flags & SEC_READONLY) != 0)
1512 last_ro_alloc = lookup;
1513 }
1514 }
1515
1516 if (last_rel_alloc)
1517 return last_rel_alloc;
1518
1519 if (last_rel)
1520 return last_rel;
1521
1522 if (last_ro_alloc)
1523 return last_ro_alloc;
1524
1525 if (last_alloc)
1526 return last_alloc;
1527
1528 return last;
1529}

1531/* Return whether IN is suitable to be part of OUT. */
1532
1533static bfd_boolean
1534elf_orphan_compatible (asection *in, asection *out)
1535{
1536 /* Non-zero sh_info implies a section with SHF_INFO_LINK with
1537 unknown semantics for the generic linker, or a SHT_REL/SHT_RELA
1538 section where sh_info specifies a symbol table. (We won't see
1539 SHT_GROUP, SHT_SYMTAB or SHT_DYNSYM sections here.) We clearly
1540 can't merge SHT_REL/SHT_RELA using differing symbol tables, and
1541 shouldn't merge sections with differing unknown semantics. */
1542 if (elf_section_data (out)->this_hdr.sh_info
1543 != elf_section_data (in)->this_hdr.sh_info)
1544 return FALSE;
1545 /* We can't merge two sections with differing SHF_EXCLUDE when doing
1546 a relocatable link. */
1547 if (bfd_link_relocatable (&link_info)
1548 && ((elf_section_flags (out) ^ elf_section_flags (in)) & SHF_EXCLUDE) != 0)
1549 return FALSE;
1550 return _bfd_elf_match_sections_by_type (link_info.output_bfd, out,
1551 in->owner, in);
1552}

1554/* Place an orphan section. We use this to put random SHF_ALLOC
1555 sections in the right segment. */
1556
1557static lang_output_section_statement_type *
1558gldelf_i386_place_orphan (asection *s,
1559 const char *secname,
1560 int constraint)
1561{
1562 static struct orphan_save hold[] =
1563 {
1564 { ".text",
1565 SEC_HAS_CONTENTS | SEC_ALLOC | SEC_LOAD | SEC_READONLY | SEC_CODE,
1566 0, 0, 0, 0 },
1567 { ".rodata",
1568 SEC_HAS_CONTENTS | SEC_ALLOC | SEC_LOAD | SEC_READONLY | SEC_DATA,
1569 0, 0, 0, 0 },
1570 { ".tdata",
1571 SEC_HAS_CONTENTS | SEC_ALLOC | SEC_LOAD | SEC_DATA | SEC_THREAD_LOCAL,
1572 0, 0, 0, 0 },
1573 { ".data",
1574 SEC_HAS_CONTENTS | SEC_ALLOC | SEC_LOAD | SEC_DATA,
1575 0, 0, 0, 0 },
1576 { ".bss",
1577 SEC_ALLOC,
1578 0, 0, 0, 0 },
1579 { 0,
1580 SEC_HAS_CONTENTS | SEC_ALLOC | SEC_LOAD | SEC_READONLY | SEC_DATA,
1581 0, 0, 0, 0 },
1582 { ".interp",
1583 SEC_HAS_CONTENTS | SEC_ALLOC | SEC_LOAD | SEC_READONLY | SEC_DATA,
1584 0, 0, 0, 0 },
1585 { ".sdata",
1586 SEC_HAS_CONTENTS | SEC_ALLOC | SEC_LOAD | SEC_DATA | SEC_SMALL_DATA,
1587 0, 0, 0, 0 },
1588 { ".comment",
1589 SEC_HAS_CONTENTS,
1590 0, 0, 0, 0 },
1591 };
1592 enum orphan_save_index
1593 {
1594 orphan_text = 0,
1595 orphan_rodata,
1596 orphan_tdata,
1597 orphan_data,
1598 orphan_bss,
1599 orphan_rel,
1600 orphan_interp,
1601 orphan_sdata,
1602 orphan_nonalloc
1603 };
1604 static int orphan_init_done = 0;
1605 struct orphan_save *place;
1606 lang_output_section_statement_type *after;
1607 lang_output_section_statement_type *os;
1608 lang_output_section_statement_type *match_by_name = NULL;
1609 int isdyn = 0;
1610 int elfinput = s->owner->xvec->flavour == bfd_target_elf_flavour;
1611 int elfoutput = link_info.output_bfd->xvec->flavour == bfd_target_elf_flavour;
1612 unsigned int sh_type = elfinput ? elf_section_type (s) : SHT_NULL;
1613 flagword flags;
1614 asection *nexts;
1615
1616 if (!bfd_link_relocatable (&link_info)
1617 && link_info.combreloc
1618 && (s->flags & SEC_ALLOC))
1619 {
1620 if (elfinput)
1621 switch (sh_type)
1622 {
1623 case SHT_RELA:
1624 secname = ".rela.dyn";
1625 isdyn = 1;
1626 break;
1627 case SHT_REL:
1628 secname = ".rel.dyn";
1629 isdyn = 1;
1630 break;
1631 default:
1632 break;
1633 }
1634 else if (CONST_STRNEQ (secname, ".rel"))
1635 {
1636 secname = secname[4] == 'a' ? ".rela.dyn" : ".rel.dyn";
1637 isdyn = 1;
1638 }
1639 }
1640
1641 if (!bfd_link_relocatable (&link_info)
1642 && elfinput
1643 && elfoutput
1644 && (s->flags & SEC_ALLOC) != 0
1645 && (elf_section_flags (s) & SHF_GNU_MBIND) != 0)
1646 {
1647 /* Find the output mbind section with the same type, attributes
1648 and sh_info field. */
1649 for (os = &lang_output_section_statement.head->output_section_statement;
1650 os != NULL;
1651 os = os->next)
1652 if (os->bfd_section != NULL
1653 && !bfd_is_abs_section (os->bfd_section)
1654 && (elf_section_flags (os->bfd_section) & SHF_GNU_MBIND) != 0
1655 && ((s->flags & (SEC_ALLOC
1656 | SEC_LOAD
1657 | SEC_HAS_CONTENTS
1658 | SEC_READONLY
1659 | SEC_CODE))
1660 == (os->bfd_section->flags & (SEC_ALLOC
1661 | SEC_LOAD
1662 | SEC_HAS_CONTENTS
1663 | SEC_READONLY
1664 | SEC_CODE)))
1665 && (elf_section_data (os->bfd_section)->this_hdr.sh_info
1666 == elf_section_data (s)->this_hdr.sh_info))
1667 {
1668 lang_add_section (&os->children, s, NULL, os);
1669 return os;
1670 }
1671
1672 /* Create the output mbind section with the ".mbind." prefix
1673 in section name. */
1674 if ((s->flags & (SEC_LOAD | SEC_HAS_CONTENTS)) == 0)
1675 secname = ".mbind.bss";
1676 else if ((s->flags & SEC_READONLY) == 0)
1677 secname = ".mbind.data";
1678 else if ((s->flags & SEC_CODE) == 0)
1679 secname = ".mbind.rodata";
1680 else
1681 secname = ".mbind.text";
1682 }
1683
1684 /* Look through the script to see where to place this section. The
1685 script includes entries added by previous lang_insert_orphan
1686 calls, so this loop puts multiple compatible orphans of the same
1687 name into a single output section. */
1688 if (constraint == 0)
1689 for (os = lang_output_section_find (secname);
1690 os != NULL;
1691 os = next_matching_output_section_statement (os, 0))
1692 {
1693 /* If we don't match an existing output section, tell
1694 lang_insert_orphan to create a new output section. */
1695 constraint = SPECIAL;
1696
1697 /* Check to see if we already have an output section statement
1698 with this name, and its bfd section has compatible flags.
1699 If the section already exists but does not have any flags
1700 set, then it has been created by the linker, possibly as a
1701 result of a --section-start command line switch. */
1702 if (os->bfd_section != NULL
1703 && (os->bfd_section->flags == 0
1704 || (((s->flags ^ os->bfd_section->flags)
1705 & (SEC_LOAD | SEC_ALLOC)) == 0
1706 && (!elfinput
1707 || !elfoutput
1708 || elf_orphan_compatible (s, os->bfd_section)))))
1709 {
1710 lang_add_section (&os->children, s, NULL, os);
1711 return os;
1712 }
1713
1714 /* Save unused output sections in case we can match them
1715 against orphans later. */
1716 if (os->bfd_section == NULL)
1717 match_by_name = os;
1718 }
1719
1720 /* If we didn't match an active output section, see if we matched an
1721 unused one and use that. */
1722 if (match_by_name)
1723 {
1724 lang_add_section (&match_by_name->children, s, NULL, match_by_name);
1725 return match_by_name;
1726 }
1727
1728 if (!orphan_init_done)
1729 {
1730 struct orphan_save *ho;
1731
1732 for (ho = hold; ho < hold + sizeof (hold) / sizeof (hold[0]); ++ho)
1733 if (ho->name != NULL)
1734 {
1735 ho->os = lang_output_section_find (ho->name);
1736 if (ho->os != NULL && ho->os->flags == 0)
1737 ho->os->flags = ho->flags;
1738 }
1739 orphan_init_done = 1;
1740 }
1741
1742 /* If this is a final link, then always put .gnu.warning.SYMBOL
1743 sections into the .text section to get them out of the way. */
1744 if (bfd_link_executable (&link_info)
1745 && CONST_STRNEQ (s->name, ".gnu.warning.")
1746 && hold[orphan_text].os != NULL)
1747 {
1748 os = hold[orphan_text].os;
1749 lang_add_section (&os->children, s, NULL, os);
1750 return os;
1751 }
1752
1753 flags = s->flags;
1754 if (!bfd_link_relocatable (&link_info))
1755 {
1756 nexts = s;
1757 while ((nexts = bfd_get_next_section_by_name (nexts->owner, nexts))
1758 != NULL)
1759 if (nexts->output_section == NULL
1760 && (nexts->flags & SEC_EXCLUDE) == 0
1761 && ((nexts->flags ^ flags) & (SEC_LOAD | SEC_ALLOC)) == 0
1762 && (nexts->owner->flags & DYNAMIC) == 0
1763 && nexts->owner->usrdata != NULL
1764 && !(((lang_input_statement_type *) nexts->owner->usrdata)
1765 ->flags.just_syms)
1766 && _bfd_elf_match_sections_by_type (nexts->owner, nexts,
1767 s->owner, s))
1768 flags = (((flags ^ SEC_READONLY)
1769 | (nexts->flags ^ SEC_READONLY))
1770 ^ SEC_READONLY);
1771 }
1772
1773 /* Decide which segment the section should go in based on the
1774 section name and section flags. We put loadable .note sections
1775 right after the .interp section, so that the PT_NOTE segment is
1776 stored right after the program headers where the OS can read it
1777 in the first page. */
1778
1779 place = NULL;
1780 if ((flags & (SEC_ALLOC | SEC_DEBUGGING)) == 0)
1781 place = &hold[orphan_nonalloc];
1782 else if ((flags & SEC_ALLOC) == 0)
1783 ;
1784 else if ((flags & SEC_LOAD) != 0
1785 && ((elfinput && sh_type == SHT_NOTE)
1786 || (!elfinput && CONST_STRNEQ (secname, ".note"))))
1787 place = &hold[orphan_interp];
1788 else if ((flags & (SEC_LOAD | SEC_HAS_CONTENTS | SEC_THREAD_LOCAL)) == 0)
1789 place = &hold[orphan_bss];
1790 else if ((flags & SEC_SMALL_DATA) != 0)
1791 place = &hold[orphan_sdata];
1792 else if ((flags & SEC_THREAD_LOCAL) != 0)
1793 place = &hold[orphan_tdata];
1794 else if ((flags & SEC_READONLY) == 0)
1795 place = &hold[orphan_data];
1796 else if (((elfinput && (sh_type == SHT_RELA || sh_type == SHT_REL))
1797 || (!elfinput && CONST_STRNEQ (secname, ".rel")))
1798 && (flags & SEC_LOAD) != 0)
1799 place = &hold[orphan_rel];
1800 else if ((flags & SEC_CODE) == 0)
1801 place = &hold[orphan_rodata];
1802 else
1803 place = &hold[orphan_text];
1804
1805 after = NULL;
1806 if (place != NULL)
1807 {
1808 if (place->os == NULL)
1809 {
1810 if (place->name != NULL)
1811 place->os = lang_output_section_find (place->name);
1812 else
1813 place->os = output_rel_find (s, isdyn);
1814 }
1815 after = place->os;
1816 if (after == NULL)
1817 after
1818 = lang_output_section_find_by_flags (s, flags, &place->os,
1819 _bfd_elf_match_sections_by_type);
1820 if (after == NULL)
1821 /* *ABS* is always the first output section statement. */
1822 after = &lang_output_section_statement.head->output_section_statement;
1823 }
1824
1825 return lang_insert_orphan (s, secname, constraint, after, place, NULL, NULL);
1826}

1980{

1984 return FALSE;

1986 case OPTION_BUILD_ID:
