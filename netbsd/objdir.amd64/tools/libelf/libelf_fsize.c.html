101size_t
102_libelf_fsize(Elf_Type t, int ec, unsigned int v, size_t c)
103{
104 size_t sz;
105
106 sz = 0;
107 if (v != EV_CURRENT)
108 LIBELF_SET_ERROR(VERSION, 0);
109 else if ((int) t < ELF_T_FIRST || t > ELF_T_LAST)
110 LIBELF_SET_ERROR(ARGUMENT, 0);
111 else {
112 sz = ec == ELFCLASS64 ? fsize[t].fsz64 : fsize[t].fsz32;
113 if (sz == 0)
114 LIBELF_SET_ERROR(UNIMPL, 0);
115 }
116
117 return (sz*c);
118}
