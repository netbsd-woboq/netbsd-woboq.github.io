105size_t
106_libelf_msize(Elf_Type t, int elfclass, unsigned int version)
107{
108 size_t sz;
109
110 assert(elfclass == ELFCLASS32 || elfclass == ELFCLASS64);
111 assert((signed) t >= ELF_T_FIRST && t <= ELF_T_LAST);
112
113 if (version != EV_CURRENT) {
114 LIBELF_SET_ERROR(VERSION, 0);
115 return (0);
116 }
117
118 sz = (elfclass == ELFCLASS32) ? msize[t].msz32 : msize[t].msz64;
119
120 return (sz);
121}
