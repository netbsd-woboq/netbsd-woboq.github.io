3247/*
3248 * Handle lint comments. Following comments are currently understood:
3249 * ARGSUSEDn
3250 * BITFIELDTYPE
3251 * CONSTCOND CONSTANTCOND CONSTANTCONDITION
3252 * FALLTHRU FALLTHROUGH
3253 * LINTLIBRARY
3254 * LINTEDn NOSTRICTn
3255 * LONGLONG
3256 * NOTREACHED
3257 * PRINTFLIKEn
3258 * PROTOLIB
3259 * SCANFLIKEn
3260 * VARARGSn
3261 * If one of this comments is recognized, the arguments, if any, are
3262 * parsed and a function which handles this comment is called.
3263 */
3264static void
3265comment(void)
3266{
3267 int c, lc;
3268 static struct {
3269 const char *keywd;
3270 int arg;
3271 void (*func)(int);
3272 } keywtab[] = {
3273 { "ARGSUSED", 1, argsused },
3274 { "BITFIELDTYPE", 0, bitfieldtype },
3275 { "CONSTCOND", 0, constcond },
3276 { "CONSTANTCOND", 0, constcond },
3277 { "CONSTANTCONDITION", 0, constcond },
3278 { "FALLTHRU", 0, fallthru },
3279 { "FALLTHROUGH", 0, fallthru },
3280 { "LINTLIBRARY", 0, lintlib },
3281 { "LINTED", 1, linted },
3282 { "LONGLONG", 0, longlong },
3283 { "NOSTRICT", 1, linted },
3284 { "NOTREACHED", 0, notreach },
3285 { "PRINTFLIKE", 1, printflike },
3286 { "PROTOLIB", 1, protolib },
3287 { "SCANFLIKE", 1, scanflike },
3288 { "VARARGS", 1, varargs },
3289 };
3290 char keywd[32];
3291 char arg[32];
3292 size_t l, i;
3293 int a;
3294 int eoc;
3295
3296 eoc = 0;
3297
3298 /* Skip white spaces after the start of the comment */
3299 while ((c = inpc()) != EOF && isspace(c))
3300 continue;
3301
3302 /* Read the potential keyword to keywd */
3303 l = 0;
3304 while (c != EOF && isupper(c) && l < sizeof (keywd) - 1) {
3305 keywd[l++] = (char)c;
3306 c = inpc();
3307 }
3308 keywd[l] = '\0';
3309
3310 /* look for the keyword */
3311 for (i = 0; i < sizeof (keywtab) / sizeof (keywtab[0]); i++) {
3312 if (strcmp(keywtab[i].keywd, keywd) == 0)
3313 break;
3314 }
3315 if (i == sizeof (keywtab) / sizeof (keywtab[0]))
3316 goto skip_rest;
3317
3318 /* skip white spaces after the keyword */
3319 while (c != EOF && isspace(c))
3320 c = inpc();
3321
3322 /* read the argument, if the keyword accepts one and there is one */
3323 l = 0;
3324 if (keywtab[i].arg) {
3325 while (c != EOF && isdigit(c) && l < sizeof (arg) - 1) {
3326 arg[l++] = (char)c;
3327 c = inpc();
3328 }
3329 }
3330 arg[l] = '\0';
3331 a = l != 0 ? atoi(arg) : -1;
3332
3333 /* skip white spaces after the argument */
3334 while (c != EOF && isspace(c))
3335 c = inpc();
3336
3337 if (c != '*' || (c = inpc()) != '/') {
3338 if (keywtab[i].func != linted)
3339 /* extra characters in lint comment */
3340 warning(257);
3341 } else {
3342 /*
3343 * remember that we have already found the end of the
3344 * comment
3345 */
3346 eoc = 1;
3347 }
3348
3349 if (keywtab[i].func != NULL)
3350 (*keywtab[i].func)(a);
3351
3352 skip_rest:
3353 while (!eoc) {
3354 lc = c;
3355 if ((c = inpc()) == EOF) {
3356 /* unterminated comment */
3357 error(256);
3358 break;
3359 }
3360 if (lc == '*' && c == '/')
3361 eoc = 1;
3362 }
3363}

3490/*
3491 * As noted above the scanner does not create new symbol table entries
3492 * for symbols it cannot find in the symbol table. This is to avoid
3493 * putting undeclared symbols into the symbol table if a syntax error
3494 * occurs.
3495 *
3496 * getsym() is called as soon as it is probably ok to put the symbol to
3497 * the symbol table. This does not mean that it is not possible that
3498 * symbols are put to the symbol table which are than not completely
3499 * declared due to syntax errors. To avoid too many problems in this
3500 * case symbols get type int in getsym().
3501 *
3502 * XXX calls to getsym() should be delayed until decl1*() is called
3503 */
3504sym_t *
3505getsym(sbuf_t *sb)
3506{
3507 dinfo_t *di;
3508 char *s;
3509 sym_t *sym;
3510
3511 sym = sb->sb_sym;
3512
3513 /*
3514 * During member declaration it is possible that name() looked
3515 * for symbols of type FVFT, although it should have looked for
3516 * symbols of type FTAG. Same can happen for labels. Both cases
3517 * are compensated here.
3518 */
3519 if (symtyp == FMOS || symtyp == FLAB) {
3520 if (sym == NULL || sym->s_kind == FVFT)
3521 sym = search(sb);
3522 }
3523
3524 if (sym != NULL) {
3525 if (sym->s_kind != symtyp)
3526 LERROR("storesym(%d, %d)", sym->s_kind, symtyp);
3527 symtyp = FVFT;
3528 freesb(sb);
3529 return (sym);
3530 }
3531
3532 /* create a new symbol table entry */
3533
3534 /* labels must always be allocated at level 1 (outhermost block) */
3535 if (symtyp == FLAB) {
3536 sym = getlblk(1, sizeof (sym_t));
3537 s = getlblk(1, sb->sb_len + 1);
3538 (void)memcpy(s, sb->sb_name, sb->sb_len + 1);
3539 sym->s_name = s;
3540 sym->s_blklev = 1;
3541 di = dcs;
3542 while (di->d_nxt != NULL && di->d_nxt->d_nxt != NULL)
3543 di = di->d_nxt;
3544 if (di->d_ctx != AUTO)
3545 LERROR("storesym()");
3546 } else {
3547 sym = getblk(sizeof (sym_t));
3548 sym->s_name = sb->sb_name;
3549 sym->s_blklev = blklev;
3550 di = dcs;
3551 }
3552
3553 UNIQUE_CURR_POS(sym->s_dpos);
3554 if ((sym->s_kind = symtyp) != FLAB)
3555 sym->s_type = gettyp(INT);
3556
3557 symtyp = FVFT;
3558
3559 if ((sym->s_link = symtab[sb->sb_hash]) != NULL)
3560 symtab[sb->sb_hash]->s_rlink = &sym->s_link;
3561 sym->s_rlink = &symtab[sb->sb_hash];
3562 symtab[sb->sb_hash] = sym;
3563
3564 *di->d_ldlsym = sym;
3565 di->d_ldlsym = &sym->s_dlnxt;
3566
3567 freesb(sb);
3568 return (sym);
3569}
