446 snprintf(buf, sizeof(buf), _CITRUS_ESDB_SYM_CSNAME_PREFIX "%d",
447 i);
448 CHKERR(ret, _db_factory_addstr_by_s,
449 (df, buf, csid->ci_symbol));

454 }
455

460 fp = fopen(output, "wb");

698 p->yystack.s_base = NULL;
699 p->yystack.l_base = NULL;
700#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
701 p->yystack.p_base = NULL;
702#endif
703 return p;
704 }
705 p->yystack.s_base = (YYINT *) malloc(size * sizeof(YYINT));
706 if (p->yystack.s_base == NULL) return NULL;
707 p->yystack.l_base = (YYSTYPE *) malloc(size * sizeof(YYSTYPE));
708 if (p->yystack.l_base == NULL) return NULL;
709 memset(p->yystack.l_base, 0, size * sizeof(YYSTYPE));
710#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
711 p->yystack.p_base = (YYLTYPE *) malloc(size * sizeof(YYLTYPE));
712 if (p->yystack.p_base == NULL) return NULL;
713 memset(p->yystack.p_base, 0, size * sizeof(YYLTYPE));
714#endif
715
716 return p;
717}

728#define YYREJECT goto yyabort
729#define YYACCEPT goto yyaccept
730#define YYERROR goto yyerrlab
731#if YYBTYACC
732#define YYVALID do { if (yyps->save) goto yyvalid; } while(0)

734 yyps->save->save == 0) goto yyvalid; } while(0)

737int
738YYPARSE_DECL()
739{
740 int yym, yyn, yystate, yyresult;
741#if YYBTYACC
742 int yynewerrflag;
743 YYParseState *yyerrctx = NULL;
744#endif /* YYBTYACC */
745#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
746 YYLTYPE yyerror_loc_range[3]; /* position of error start/end (0 unused) */
747#endif
748#if YYDEBUG
749 const char *yys;
750
751 if ((yys = getenv("YYDEBUG")) != 0)
752 {
753 yyn = *yys;
754 if (yyn >= '0' && yyn <= '9')
755 yydebug = yyn - '0';
756 }
757 if (yydebug)
758 fprintf(stderr, "%sdebug[<# of symbols on state stack>]\n", YYPREFIX);
759#endif
760#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
761 memset(yyerror_loc_range, 0, sizeof(yyerror_loc_range));
762#endif
763
764#if YYBTYACC
765 yyps = yyNewState(0); if (yyps == 0) goto yyenomem;
766 yyps->save = 0;
767#endif /* YYBTYACC */
768 yym = 0;
769 yyn = 0;
770 yynerrs = 0;
771 yyerrflag = 0;
772 yychar = YYEMPTY;
773 yystate = 0;
774
775#if YYPURE
776 memset(&yystack, 0, sizeof(yystack));
777#endif
778
779 if (yystack.s_base == NULL && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
780 yystack.s_mark = yystack.s_base;
781 yystack.l_mark = yystack.l_base;
782#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
783 yystack.p_mark = yystack.p_base;
784#endif
785 yystate = 0;
786 *yystack.s_mark = 0;
787
788yyloop:
789 if ((yyn = yydefred[yystate]) != 0) goto yyreduce;
790 if (yychar < 0)
791 {
792#if YYBTYACC
793 do {
794 if (yylvp < yylve)
795 {
796 /* we're currently re-reading tokens */
797 yylval = *yylvp++;
798#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
799 yylloc = *yylpp++;
800#endif
801 yychar = *yylexp++;
802 break;
803 }
804 if (yyps->save)
805 {
806 /* in trial mode; save scanner results for future parse attempts */
807 if (yylvp == yylvlim)
808 { /* Enlarge lexical value queue */
809 size_t p = (size_t) (yylvp - yylvals);
810 size_t s = (size_t) (yylvlim - yylvals);
811
812 s += YYLVQUEUEGROWTH;
813 if ((yylexemes = (YYINT *)realloc(yylexemes, s * sizeof(YYINT))) == NULL) goto yyenomem;
814 if ((yylvals = (YYSTYPE *)realloc(yylvals, s * sizeof(YYSTYPE))) == NULL) goto yyenomem;
815#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
816 if ((yylpsns = (YYLTYPE *)realloc(yylpsns, s * sizeof(YYLTYPE))) == NULL) goto yyenomem;
817#endif
818 yylvp = yylve = yylvals + p;
819 yylvlim = yylvals + s;
820#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
821 yylpp = yylpe = yylpsns + p;
822 yylplim = yylpsns + s;
823#endif
824 yylexp = yylexemes + p;
825 }
826 *yylexp = (YYINT) YYLEX;
827 *yylvp++ = yylval;
828 yylve++;
829#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
830 *yylpp++ = yylloc;
831 yylpe++;
832#endif
833 yychar = *yylexp++;
834 break;
835 }
836 /* normal operation, no conflict encountered */
837#endif /* YYBTYACC */
838 yychar = YYLEX;
839#if YYBTYACC
840 } while (0);
841#endif /* YYBTYACC */
842 if (yychar < 0) yychar = YYEOF;
843#if YYDEBUG
844 if (yydebug)
845 {
846 if ((yys = yyname[YYTRANSLATE(yychar)]) == NULL) yys = yyname[YYUNDFTOKEN];
847 fprintf(stderr, "%s[%d]: state %d, reading token %d (%s)",
848 YYDEBUGSTR, yydepth, yystate, yychar, yys);
849#ifdef YYSTYPE_TOSTRING
850#if YYBTYACC
851 if (!yytrial)
852#endif /* YYBTYACC */
853 fprintf(stderr, " <%s>", YYSTYPE_TOSTRING(yychar, yylval));
854#endif
855 fputc('\n', stderr);
856 }
857#endif
858 }
859#if YYBTYACC
860
861 /* Do we have a conflict? */
862 if (((yyn = yycindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
863 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
864 {
865 YYINT ctry;
866
867 if (yypath)
868 {
869 YYParseState *save;
870#if YYDEBUG
871 if (yydebug)
872 fprintf(stderr, "%s[%d]: CONFLICT in state %d: following successful trial parse\n",
873 YYDEBUGSTR, yydepth, yystate);
874#endif
875 /* Switch to the next conflict context */
876 save = yypath;
877 yypath = save->save;
878 save->save = NULL;
879 ctry = save->ctry;
880 if (save->state != yystate) YYABORT;
881 yyFreeState(save);
882
883 }
884 else
885 {
886
887 /* Unresolved conflict - start/continue trial parse */
888 YYParseState *save;
889#if YYDEBUG
890 if (yydebug)
891 {
892 fprintf(stderr, "%s[%d]: CONFLICT in state %d. ", YYDEBUGSTR, yydepth, yystate);
893 if (yyps->save)
894 fputs("ALREADY in conflict, continuing trial parse.\n", stderr);
895 else
896 fputs("Starting trial parse.\n", stderr);
897 }
898#endif
899 save = yyNewState((unsigned)(yystack.s_mark - yystack.s_base + 1));
900 if (save == NULL) goto yyenomem;
901 save->save = yyps->save;
902 save->state = yystate;
903 save->errflag = yyerrflag;
904 save->yystack.s_mark = save->yystack.s_base + (yystack.s_mark - yystack.s_base);
905 memcpy (save->yystack.s_base, yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
906 save->yystack.l_mark = save->yystack.l_base + (yystack.l_mark - yystack.l_base);
907 memcpy (save->yystack.l_base, yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
908#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
909 save->yystack.p_mark = save->yystack.p_base + (yystack.p_mark - yystack.p_base);
910 memcpy (save->yystack.p_base, yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
911#endif
912 ctry = yytable[yyn];
913 if (yyctable[ctry] == -1)
914 {
915#if YYDEBUG
916 if (yydebug && yychar >= YYEOF)
917 fprintf(stderr, "%s[%d]: backtracking 1 token\n", YYDEBUGSTR, yydepth);
918#endif
919 ctry++;
920 }
921 save->ctry = ctry;
922 if (yyps->save == NULL)
923 {
924 /* If this is a first conflict in the stack, start saving lexemes */
925 if (!yylexemes)
926 {
927 yylexemes = (YYINT *) malloc((YYLVQUEUEGROWTH) * sizeof(YYINT));
928 if (yylexemes == NULL) goto yyenomem;
929 yylvals = (YYSTYPE *) malloc((YYLVQUEUEGROWTH) * sizeof(YYSTYPE));
930 if (yylvals == NULL) goto yyenomem;
931 yylvlim = yylvals + YYLVQUEUEGROWTH;
932#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
933 yylpsns = (YYLTYPE *) malloc((YYLVQUEUEGROWTH) * sizeof(YYLTYPE));
934 if (yylpsns == NULL) goto yyenomem;
935 yylplim = yylpsns + YYLVQUEUEGROWTH;
936#endif
937 }
938 if (yylvp == yylve)
939 {
940 yylvp = yylve = yylvals;
941#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
942 yylpp = yylpe = yylpsns;
943#endif
944 yylexp = yylexemes;
945 if (yychar >= YYEOF)
946 {
947 *yylve++ = yylval;
948#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
949 *yylpe++ = yylloc;
950#endif
951 *yylexp = (YYINT) yychar;
952 yychar = YYEMPTY;
953 }
954 }
955 }
956 if (yychar >= YYEOF)
957 {
958 yylvp--;
959#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
960 yylpp--;
961#endif
962 yylexp--;
963 yychar = YYEMPTY;
964 }
965 save->lexeme = (int) (yylvp - yylvals);
966 yyps->save = save;
967 }
968 if (yytable[yyn] == ctry)
969 {
970#if YYDEBUG
971 if (yydebug)
972 fprintf(stderr, "%s[%d]: state %d, shifting to state %d\n",
973 YYDEBUGSTR, yydepth, yystate, yyctable[ctry]);
974#endif
975 if (yychar < 0)
976 {
977 yylvp++;
978#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
979 yylpp++;
980#endif
981 yylexp++;
982 }
983 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM)
984 goto yyoverflow;
985 yystate = yyctable[ctry];
986 *++yystack.s_mark = (YYINT) yystate;
987 *++yystack.l_mark = yylval;
988#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
989 *++yystack.p_mark = yylloc;
990#endif
991 yychar = YYEMPTY;
992 if (yyerrflag > 0) --yyerrflag;
993 goto yyloop;
994 }
995 else
996 {
997 yyn = yyctable[ctry];
998 goto yyreduce;
999 }
1000 } /* End of code dealing with conflicts */
1001#endif /* YYBTYACC */
1002 if (((yyn = yysindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
1003 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
1004 {
1005#if YYDEBUG
1006 if (yydebug)
1007 fprintf(stderr, "%s[%d]: state %d, shifting to state %d\n",
1008 YYDEBUGSTR, yydepth, yystate, yytable[yyn]);
1009#endif
1010 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
1011 yystate = yytable[yyn];
1012 *++yystack.s_mark = yytable[yyn];
1013 *++yystack.l_mark = yylval;
1014#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1015 *++yystack.p_mark = yylloc;
1016#endif
1017 yychar = YYEMPTY;
1018 if (yyerrflag > 0) --yyerrflag;
1019 goto yyloop;
1020 }
1021 if (((yyn = yyrindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
1022 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
1023 {
1024 yyn = yytable[yyn];
1025 goto yyreduce;
1026 }
1027 if (yyerrflag != 0) goto yyinrecovery;
1028#if YYBTYACC
1029
1030 yynewerrflag = 1;
1031 goto yyerrhandler;
1032 goto yyerrlab; /* redundant goto avoids 'unused label' warning */
1033
1034yyerrlab:
1035 /* explicit YYERROR from an action -- pop the rhs of the rule reduced
1036 * before looking for error recovery */
1037 yystack.s_mark -= yym;
1038 yystate = *yystack.s_mark;
1039 yystack.l_mark -= yym;
1040#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1041 yystack.p_mark -= yym;
1042#endif
1043
1044 yynewerrflag = 0;
1045yyerrhandler:
1046 while (yyps->save)
1047 {
1048 int ctry;
1049 YYParseState *save = yyps->save;
1050#if YYDEBUG
1051 if (yydebug)
1052 fprintf(stderr, "%s[%d]: ERROR in state %d, CONFLICT BACKTRACKING to state %d, %d tokens\n",
1053 YYDEBUGSTR, yydepth, yystate, yyps->save->state,
1054 (int)(yylvp - yylvals - yyps->save->lexeme));
1055#endif
1056 /* Memorize most forward-looking error state in case it's really an error. */
1057 if (yyerrctx == NULL || yyerrctx->lexeme < yylvp - yylvals)
1058 {
1059 /* Free old saved error context state */
1060 if (yyerrctx) yyFreeState(yyerrctx);
1061 /* Create and fill out new saved error context state */
1062 yyerrctx = yyNewState((unsigned)(yystack.s_mark - yystack.s_base + 1));
1063 if (yyerrctx == NULL) goto yyenomem;
1064 yyerrctx->save = yyps->save;
1065 yyerrctx->state = yystate;
1066 yyerrctx->errflag = yyerrflag;
1067 yyerrctx->yystack.s_mark = yyerrctx->yystack.s_base + (yystack.s_mark - yystack.s_base);
1068 memcpy (yyerrctx->yystack.s_base, yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1069 yyerrctx->yystack.l_mark = yyerrctx->yystack.l_base + (yystack.l_mark - yystack.l_base);
1070 memcpy (yyerrctx->yystack.l_base, yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1071#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1072 yyerrctx->yystack.p_mark = yyerrctx->yystack.p_base + (yystack.p_mark - yystack.p_base);
1073 memcpy (yyerrctx->yystack.p_base, yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1074#endif
1075 yyerrctx->lexeme = (int) (yylvp - yylvals);
1076 }
1077 yylvp = yylvals + save->lexeme;
1078#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1079 yylpp = yylpsns + save->lexeme;
1080#endif
1081 yylexp = yylexemes + save->lexeme;
1082 yychar = YYEMPTY;
1083 yystack.s_mark = yystack.s_base + (save->yystack.s_mark - save->yystack.s_base);
1084 memcpy (yystack.s_base, save->yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1085 yystack.l_mark = yystack.l_base + (save->yystack.l_mark - save->yystack.l_base);
1086 memcpy (yystack.l_base, save->yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1087#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1088 yystack.p_mark = yystack.p_base + (save->yystack.p_mark - save->yystack.p_base);
1089 memcpy (yystack.p_base, save->yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1090#endif
1091 ctry = ++save->ctry;
1092 yystate = save->state;
1093 /* We tried shift, try reduce now */
1094 if ((yyn = yyctable[ctry]) >= 0) goto yyreduce;
1095 yyps->save = save->save;
1096 save->save = NULL;
1097 yyFreeState(save);
1098
1099 /* Nothing left on the stack -- error */
1100 if (!yyps->save)
1101 {
1102#if YYDEBUG
1103 if (yydebug)
1104 fprintf(stderr, "%sdebug[%d,trial]: trial parse FAILED, entering ERROR mode\n",
1105 YYPREFIX, yydepth);
1106#endif
1107 /* Restore state as it was in the most forward-advanced error */
1108 yylvp = yylvals + yyerrctx->lexeme;
1109#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1110 yylpp = yylpsns + yyerrctx->lexeme;
1111#endif
1112 yylexp = yylexemes + yyerrctx->lexeme;
1113 yychar = yylexp[-1];
1114 yylval = yylvp[-1];
1115#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1116 yylloc = yylpp[-1];
1117#endif
1118 yystack.s_mark = yystack.s_base + (yyerrctx->yystack.s_mark - yyerrctx->yystack.s_base);
1119 memcpy (yystack.s_base, yyerrctx->yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1120 yystack.l_mark = yystack.l_base + (yyerrctx->yystack.l_mark - yyerrctx->yystack.l_base);
1121 memcpy (yystack.l_base, yyerrctx->yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1122#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1123 yystack.p_mark = yystack.p_base + (yyerrctx->yystack.p_mark - yyerrctx->yystack.p_base);
1124 memcpy (yystack.p_base, yyerrctx->yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1125#endif
1126 yystate = yyerrctx->state;
1127 yyFreeState(yyerrctx);
1128 yyerrctx = NULL;
1129 }
1130 yynewerrflag = 1;
1131 }
1132 if (yynewerrflag == 0) goto yyinrecovery;
1133#endif /* YYBTYACC */
1134
1135 YYERROR_CALL("syntax error");
1136#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1137 yyerror_loc_range[1] = yylloc; /* lookahead position is error start position */
1138#endif
1139
1140#if !YYBTYACC
1141 goto yyerrlab; /* redundant goto avoids 'unused label' warning */
1142yyerrlab:
1143#endif
1144 ++yynerrs;
1145
1146yyinrecovery:
1147 if (yyerrflag < 3)
1148 {
1149 yyerrflag = 3;
1150 for (;;)
1151 {
1152 if (((yyn = yysindex[*yystack.s_mark]) != 0) && (yyn += YYERRCODE) >= 0 &&
1153 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) YYERRCODE)
1154 {
1155#if YYDEBUG
1156 if (yydebug)
1157 fprintf(stderr, "%s[%d]: state %d, error recovery shifting to state %d\n",
1158 YYDEBUGSTR, yydepth, *yystack.s_mark, yytable[yyn]);
1159#endif
1160 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
1161 yystate = yytable[yyn];
1162 *++yystack.s_mark = yytable[yyn];
1163 *++yystack.l_mark = yylval;
1164#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1165 /* lookahead position is error end position */
1166 yyerror_loc_range[2] = yylloc;
1167 YYLLOC_DEFAULT(yyloc, yyerror_loc_range, 2); /* position of error span */
1168 *++yystack.p_mark = yyloc;
1169#endif
1170 goto yyloop;
1171 }
1172 else
1173 {
1174#if YYDEBUG
1175 if (yydebug)
1176 fprintf(stderr, "%s[%d]: error recovery discarding state %d\n",
1177 YYDEBUGSTR, yydepth, *yystack.s_mark);
1178#endif
1179 if (yystack.s_mark <= yystack.s_base) goto yyabort;
1180#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1181 /* the current TOS position is the error start position */
1182 yyerror_loc_range[1] = *yystack.p_mark;
1183#endif
1184#if defined(YYDESTRUCT_CALL)
1185#if YYBTYACC
1186 if (!yytrial)
1187#endif /* YYBTYACC */
1188#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1189 YYDESTRUCT_CALL("error: discarding state",
1190 yystos[*yystack.s_mark], yystack.l_mark, yystack.p_mark);
1191#else
1192 YYDESTRUCT_CALL("error: discarding state",
1193 yystos[*yystack.s_mark], yystack.l_mark);
1194#endif /* defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED) */
1195#endif /* defined(YYDESTRUCT_CALL) */
1196 --yystack.s_mark;
1197 --yystack.l_mark;
1198#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1199 --yystack.p_mark;
1200#endif
1201 }
1202 }
1203 }
1204 else
1205 {
1206 if (yychar == YYEOF) goto yyabort;
1207#if YYDEBUG
1208 if (yydebug)
1209 {
1210 if ((yys = yyname[YYTRANSLATE(yychar)]) == NULL) yys = yyname[YYUNDFTOKEN];
1211 fprintf(stderr, "%s[%d]: state %d, error recovery discarding token %d (%s)\n",
1212 YYDEBUGSTR, yydepth, yystate, yychar, yys);
1213 }
1214#endif
1215#if defined(YYDESTRUCT_CALL)
1216#if YYBTYACC
1217 if (!yytrial)
1218#endif /* YYBTYACC */
1219#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1220 YYDESTRUCT_CALL("error: discarding token", yychar, &yylval, &yylloc);
1221#else
1222 YYDESTRUCT_CALL("error: discarding token", yychar, &yylval);
1223#endif /* defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED) */
1224#endif /* defined(YYDESTRUCT_CALL) */
1225 yychar = YYEMPTY;
1226 goto yyloop;
1227 }
