26/* Threshold when testing coverage */
27#if defined(MPFR_TUNE_COVERAGE)
28#define MPFR_TUNE_CASE "src/generic/coverage/mparam.h"
29#include "generic/coverage/mparam.h"
30
31/* Note: the different macros used here are those defined by gcc,
32 for example with gcc -dM -E -xc /dev/null
33 As of gcc 4.2, you can also use: -march=native or -mtune=native */
34
35#elif defined (__tune_pentium4__) /* Threshold for Pentium 4 */
36#define MPFR_TUNE_CASE "src/x86_64/pentium4/mparam.h"
37#include "x86_64/pentium4/mparam.h"
38
39#elif (defined (__tune_core2__) || defined (__tune_nocona__)) && defined (__x86_64) /* 64-bit Core 2 or Xeon */
40#define MPFR_TUNE_CASE "src/x86_64/core2/mparam.h"
41#include "x86_64/core2/mparam.h"
42
43#elif defined (__tune_core2__) && defined (__i386) /* 32-bit Core 2,
44 for example a 64-bit machine with gmp/mpfr compiled with ABI=32 */
45#define MPFR_TUNE_CASE "src/x86/core2/mparam.h"
46#include "x86/core2/mparam.h"
47
48#elif defined (__tune_k8__) && defined (__amd64__) /* AMD K8 */
49#define MPFR_TUNE_CASE "src/amd/k8/mparam.h"
50#include "amd/k8/mparam.h"
51
52#elif defined (__tune_athlon__) /* Threshold for Athlon */
53#define MPFR_TUNE_CASE "src/amd/athlon/mparam.h"
54#include "amd/athlon/mparam.h"
55
56#elif defined (__tune_pentiumpro__) || defined (__tune_i686__) || defined (__i386) /* we consider all other 386's here */
57#define MPFR_TUNE_CASE "src/x86/mparam.h"
58#include "x86/mparam.h"
59
60#elif defined (__ia64) || defined (__itanium__) || defined (__tune_ia64__)
61/* Threshold for IA64 */
62#define MPFR_TUNE_CASE "src/ia64/mparam.h"
63#include "ia64/mparam.h"
64
65#elif defined (__arm__) /* Threshold for ARM */
66#define MPFR_TUNE_CASE "src/arm/mparam.h"
67#include "arm/mparam.h"
68
69#elif defined (__PPC64__) /* Threshold for 64-bit PowerPC, test it before
70 32-bit PowerPC since _ARCH_PPC is also defined
71 on 64-bit PowerPC */
72#define MPFR_TUNE_CASE "src/powerpc64/mparam.h"
73#include "powerpc64/mparam.h"
74
75#elif defined (_ARCH_PPC) /* Threshold for 32-bit PowerPC */
76#define MPFR_TUNE_CASE "src/powerpc32/mparam.h"
77#include "powerpc32/mparam.h"
78
79#elif defined (__sparc_v9__) /* Threshold for 64-bits Sparc */
80#define MPFR_TUNE_CASE "src/sparc64/mparam.h"
81#include "sparc64/mparam.h"
82
83#elif defined (__hppa__) /* Threshold for HPPA */
84#define MPFR_TUNE_CASE "src/hppa/mparam.h"
85#include "hppa/mparam.h"
86
87#elif defined (__mips__) /* MIPS */
88#define MPFR_TUNE_CASE "src/mips/mparam.h"
89#include "mips/mparam.h"
90
91#else
92#define MPFR_TUNE_CASE "default"
93#endif
