1846#if YYBTYACC
1847 } while (0);
1848#endif /* YYBTYACC */
1849 if (yychar < 0) yychar = YYEOF;
1850#if YYDEBUG
1851 if (yydebug)
1852 {
1853 if ((yys = yyname[YYTRANSLATE(yychar)]) == NULL) yys = yyname[YYUNDFTOKEN];
1854 fprintf(stderr, "%s[%d]: state %d, reading token %d (%s)\n",
1855 YYDEBUGSTR, yydepth, YYFINAL, yychar, yys);
1856 }
1857#endif
1858 }
1859 if (yychar == YYEOF) goto yyaccept;
1860 goto yyloop;
1861 }
1862 if (((yyn = yygindex[yym]) != 0) && (yyn += yystate) >= 0 &&
1863 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yystate)
1864 yystate = yytable[yyn];
1865 else
1866 yystate = yydgoto[yym];
1867#if YYDEBUG
1868 if (yydebug)
1869 {
1870 fprintf(stderr, "%s[%d]: after reduction, ", YYDEBUGSTR, yydepth);
1871#ifdef YYSTYPE_TOSTRING
1872#if YYBTYACC
1873 if (!yytrial)
1874#endif /* YYBTYACC */
1875 fprintf(stderr, "result is <%s>, ", YYSTYPE_TOSTRING(yystos[yystate], yyval));
1876#endif
1877 fprintf(stderr, "shifting from state %d to state %d\n", *yystack.s_mark, yystate);
1878 }
1879#endif
1880 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
1881 *++yystack.s_mark = (YYINT) yystate;
1882 *++yystack.l_mark = yyval;
1883#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1884 *++yystack.p_mark = yyloc;
1885#endif
1886 goto yyloop;
1887#if YYBTYACC
1888
1889 /* Reduction declares that this path is valid. Set yypath and do a full parse */
1890yyvalid:
1891 if (yypath) YYABORT;
1892 while (yyps->save)
1893 {
1894 YYParseState *save = yyps->save;
1895 yyps->save = save->save;
1896 save->save = yypath;
1897 yypath = save;
1898 }
1899#if YYDEBUG
1900 if (yydebug)
1901 fprintf(stderr, "%s[%d]: state %d, CONFLICT trial successful, backtracking to state %d, %d tokens\n",
1902 YYDEBUGSTR, yydepth, yystate, yypath->state, (int)(yylvp - yylvals - yypath->lexeme));
1903#endif
1904 if (yyerrctx)
1905 {
1906 yyFreeState(yyerrctx);
1907 yyerrctx = NULL;
1908 }
1909 yylvp = yylvals + yypath->lexeme;
1910#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1911 yylpp = yylpsns + yypath->lexeme;
1912#endif
1913 yylexp = yylexemes + yypath->lexeme;
1914 yychar = YYEMPTY;
1915 yystack.s_mark = yystack.s_base + (yypath->yystack.s_mark - yypath->yystack.s_base);
1916 memcpy (yystack.s_base, yypath->yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1917 yystack.l_mark = yystack.l_base + (yypath->yystack.l_mark - yypath->yystack.l_base);
1918 memcpy (yystack.l_base, yypath->yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1919#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1920 yystack.p_mark = yystack.p_base + (yypath->yystack.p_mark - yypath->yystack.p_base);
1921 memcpy (yystack.p_base, yypath->yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1922#endif
1923 yystate = yypath->state;
1924 goto yyloop;
1925#endif /* YYBTYACC */
1926
1927yyoverflow:
1928 YYERROR_CALL("yacc stack overflow");
1929#if YYBTYACC
1930 goto yyabort_nomem;
1931yyenomem:
1932 YYERROR_CALL("memory exhausted");
1933yyabort_nomem:
1934#endif /* YYBTYACC */
1935 yyresult = 2;
1936 goto yyreturn;
1937
1938yyabort:
1939 yyresult = 1;
1940 goto yyreturn;
1941
1942yyaccept:
1943#if YYBTYACC
1944 if (yyps->save) goto yyvalid;
1945#endif /* YYBTYACC */
1946 yyresult = 0;
1947
1948yyreturn:
1949#if defined(YYDESTRUCT_CALL)
1950 if (yychar != YYEOF && yychar != YYEMPTY)
1951#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1952 YYDESTRUCT_CALL("cleanup: discarding token", yychar, &yylval, &yylloc);
1953#else
1954 YYDESTRUCT_CALL("cleanup: discarding token", yychar, &yylval);
1955#endif /* defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED) */
1956
1957 {
1958 YYSTYPE *pv;
1959#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1960 YYLTYPE *pp;
1961
1962 for (pv = yystack.l_base, pp = yystack.p_base; pv <= yystack.l_mark; ++pv, ++pp)
1963 YYDESTRUCT_CALL("cleanup: discarding state",
1964 yystos[*(yystack.s_base + (pv - yystack.l_base))], pv, pp);
1965#else
1966 for (pv = yystack.l_base; pv <= yystack.l_mark; ++pv)
1967 YYDESTRUCT_CALL("cleanup: discarding state",
1968 yystos[*(yystack.s_base + (pv - yystack.l_base))], pv);
1969#endif /* defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED) */
1970 }
1971#endif /* defined(YYDESTRUCT_CALL) */
1972
1973#if YYBTYACC
1974 if (yyerrctx)
1975 {
1976 yyFreeState(yyerrctx);
1977 yyerrctx = NULL;
1978 }
1979 while (yyps)
1980 {
1981 YYParseState *save = yyps;
1982 yyps = save->save;
1983 save->save = NULL;
1984 yyFreeState(save);
1985 }
1986 while (yypath)
1987 {
1988 YYParseState *save = yypath;
1989 yypath = save->save;
1990 save->save = NULL;
1991 yyFreeState(save);
1992 }
1993#endif /* YYBTYACC */
1994 yyfreestack(&yystack);
1995 return (yyresult);
1996}
