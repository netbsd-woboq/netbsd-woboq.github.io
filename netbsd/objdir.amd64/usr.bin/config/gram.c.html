1208/*
1209 * Wrap allocations that live on the parser stack so that we can free
1210 * them again on error instead of leaking.
1211 */
1212
1213#define MAX_WRAP 1000
1214
1215struct wrap_entry {
1216 void *ptr;
1217 unsigned typecode;
1218};
1219
1220static struct wrap_entry wrapstack[MAX_WRAP];
1221static unsigned wrap_depth;
1222
1223/*
1224 * Remember pointer PTR with type-code CODE.
1225 */
1226static void
1227wrap_alloc(void *ptr, unsigned code)
1228{
1229 unsigned pos;
1230
1231 if (wrap_depth >= MAX_WRAP) {
1232 panic("allocation wrapper stack overflow");
1233 }
1234 pos = wrap_depth++;
1235 wrapstack[pos].ptr = ptr;
1236 wrapstack[pos].typecode = code;
1237}
1238
1239/*
1240 * We succeeded; commit to keeping everything that's been allocated so
1241 * far and clear the stack.
1242 */
1243static void
1244wrap_continue(void)
1245{
1246 wrap_depth = 0;
1247}
1248
1249/*
1250 * We failed; destroy all the objects allocated.
1251 */
1252static void
1253wrap_cleanup(void)
1254{
1255 unsigned i;
1256
1257 /*
1258 * Destroy each item. Note that because everything allocated
1259 * is entered on the list separately, lists and trees need to
1260 * have their links blanked before being destroyed. Also note
1261 * that strings are interned elsewhere and not handled by this
1262 * mechanism.
1263 */
1264
1265 for (i=0; i<wrap_depth; i++) {
1266 switch (wrapstack[i].typecode) {
1267 case WRAP_CODE_nvlist:
1268 nvfree(wrapstack[i].ptr);
1269 break;
1270 case WRAP_CODE_defoptlist:
1271 {
1272 struct defoptlist *dl = wrapstack[i].ptr;
1273
1274 dl->dl_next = NULL;
1275 defoptlist_destroy(dl);
1276 }
1277 break;
1278 case WRAP_CODE_loclist:
1279 {
1280 struct loclist *ll = wrapstack[i].ptr;
1281
1282 ll->ll_next = NULL;
1283 loclist_destroy(ll);
1284 }
1285 break;
1286 case WRAP_CODE_attrlist:
1287 {
1288 struct attrlist *al = wrapstack[i].ptr;
1289
1290 al->al_next = NULL;
1291 al->al_this = NULL;
1292 attrlist_destroy(al);
1293 }
1294 break;
1295 case WRAP_CODE_condexpr:
1296 {
1297 struct condexpr *cx = wrapstack[i].ptr;
1298
1299 cx->cx_type = CX_ATOM;
1300 cx->cx_atom = NULL;
1301 condexpr_destroy(cx);
1302 }
1303 break;
1304 default:
1305 panic("invalid code %u on allocation wrapper stack",
1306 wrapstack[i].typecode);
1307 }
1308 }
1309
1310 wrap_depth = 0;
1311}
