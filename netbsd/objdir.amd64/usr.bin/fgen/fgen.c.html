2907/*
2908 * ASCII -> long int converter, eats `.'s
2909 */
2910#define strtol(x, y, z) cvt(x, y, z)
2911static Cell
2912cvt(const char *s, char **e, int base)
2913{
2914 Cell v = 0;
2915 int c, n = 0;
2916
2917 c = *s;
2918 if (c == '-') { n = 1; s++; }
2919
2920 for (c = *s; (c = *s); s++) {
2921
2922 /* Ignore `.' */
2923 if (c == '.')
2924 continue;
2925 if (c >= '0' && c <= '9')
2926 c -= '0';
2927 else if (c >= 'a' && c <= 'f')
2928 c += 10 - 'a';
2929 else if (c >= 'A' && c <= 'F')
2930 c += 10 - 'A';
2931 if (c >= base)
2932 break;
2933 v *= base;
2934 v += c;
2935 }
2936 if (e)
2937 *e = __UNCONST(s);
2938 if (n)
2939 return (-v);
2940 return (v);
2941}

4268{
