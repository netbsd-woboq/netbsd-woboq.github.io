674{

677 rune_list *r;
678 __nbrune_t run;
679
680 while (list->min < _CTYPE_CACHE_SIZE && list->min <= list->max) {
681 if (flag)
682 map->map[list->min++] |= flag;
683 else
684 map->map[list->min++] = list->map++;

688 free(list);
689 return;
690 }
691
692 run = list->max - list->min + 1;
693
694 if (!(r = map->root) || (list->max < r->min - 1)
695 || (!flag && list->max == r->min - 1)) {
696 if (flag) {
697 list->types = xlalloc(run);

699 list->types[i] = flag;
700 }

702 map->root = list;
703 return;
704 }
705
706 for (r = map->root; r && r->max + 1 < list->min; r = r->next)
707 lr = r;
708
709 if (!r) {
710 /*
711 * We are off the end.
712 */
713 if (flag) {
714 list->types = xlalloc(run);
715 for (i = 0; i < run; ++i)
716 list->types[i] = flag;
717 }
718 list->next = 0;
719 lr->next = list;
720 return;
721 }
722
723 if (list->max < r->min - 1) {
724 /*
725 * We come before this range and we do not intersect it.
726 * We are not before the root node, it was checked before the loop
727 */
728 if (flag) {
729 list->types = xlalloc(run);
730 for (i = 0; i < run; ++i)
731 list->types[i] = flag;
732 }
733 list->next = lr->next;
734 lr->next = list;
735 return;
736 }
737
738 /*
739 * At this point we have found that we at least intersect with
740 * the range pointed to by `r', we might intersect with one or
741 * more ranges beyond `r' as well.
742 */
743
744 if (!flag && list->map - list->min != r->map - r->min) {
745 /*
746 * There are only two cases when we are doing case maps and
747 * our maps needn't have the same offset. When we are adjoining
748 * but not intersecting.
749 */
750 if (list->max + 1 == r->min) {
751 lr->next = list;
752 list->next = r;
753 return;
754 }
755 if (list->min - 1 == r->max) {
756 list->next = r->next;
757 r->next = list;
758 return;

765 /*

767 */
768

771 r->types[i - r->min] |= flag;
772 }
773 free(list);
774 return;
775 }
776 if (list->min <= r->min && list->max >= r->max) {
777 /*
778 * Superset case. Make him big enough to hold us.
779 * We might need to merge with the guy after him.
780 */
781 if (flag) {
782 list->types = xlalloc(list->max - list->min + 1);
783
784 for (i = list->min; i <= list->max; ++i)
785 list->types[i - list->min] = flag;
786
787 for (i = r->min; i <= r->max; ++i)
788 list->types[i - list->min] |= r->types[i - r->min];
789
790 free(r->types);
791 r->types = list->types;
792 } else {
793 r->map = list->map;
794 }
795 r->min = list->min;
796 r->max = list->max;
797 free(list);
798 } else if (list->min < r->min) {
799 /*
800 * Our tail intersects his head.
801 */
802 if (flag) {
803 list->types = xlalloc(r->max - list->min + 1);
804
805 for (i = r->min; i <= r->max; ++i)
806 list->types[i - list->min] = r->types[i - r->min];
807
808 for (i = list->min; i < r->min; ++i)
809 list->types[i - list->min] = flag;
810
811 for (i = r->min; i <= list->max; ++i)
812 list->types[i - list->min] |= flag;
813
814 free(r->types);
815 r->types = list->types;
816 } else {
817 r->map = list->map;
818 }
819 r->min = list->min;
820 free(list);
821 return;
822 } else {
823 /*
824 * Our head intersects his tail.
825 * We might need to merge with the guy after him.
826 */
827 if (flag) {
828 r->types = xrelalloc(r->types, list->max - r->min + 1);
829
830 for (i = list->min; i <= r->max; ++i)
831 r->types[i - r->min] |= flag;
832
833 for (i = r->max+1; i <= list->max; ++i)
834 r->types[i - r->min] = flag;
835 }
836 r->max = list->max;
837 free(list);
838 }
839
840 /*
841 * Okay, check to see if we grew into the next guy(s)
842 */
843 while ((lr = r->next) && r->max >= lr->min) {
844 if (flag) {
845 if (r->max >= lr->max) {
846 /*
847 * Good, we consumed all of him.
848 */
849 for (i = lr->min; i <= lr->max; ++i)
850 r->types[i - r->min] |= lr->types[i - lr->min];
851 } else {
852 /*
853 * "append" him on to the end of us.
854 */
855 r->types = xrelalloc(r->types, lr->max - r->min + 1);
856
857 for (i = lr->min; i <= r->max; ++i)
858 r->types[i - r->min] |= lr->types[i - lr->min];
859
860 for (i = r->max+1; i <= lr->max; ++i)
861 r->types[i - r->min] = lr->types[i - lr->min];
862
863 r->max = lr->max;
864 }
865 } else {
866 if (lr->max > r->max)
867 r->max = lr->max;
868 }
869
870 r->next = lr->next;
871
872 if (flag)
873 free(lr->types);
874 free(lr);
875 }
