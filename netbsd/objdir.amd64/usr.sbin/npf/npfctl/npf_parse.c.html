965 newps = (YYLTYPE *)realloc(data->p_base, newsize * sizeof(*newps));
966 if (newps == 0)
967 return YYENOMEM;
968
969 data->p_base = newps;
970 data->p_mark = newps + i;
971#endif
972
973 data->stacksize = newsize;
974 data->s_last = data->s_base + newsize - 1;
975
976#if YYDEBUG
977 if (yydebug)
978 fprintf(stderr, "%sdebug: stack size increased to %d\n", YYPREFIX, newsize);
979#endif
980 return 0;
981}

1068#endif

1070 memset(yyerror_loc_range, 0, sizeof(yyerror_loc_range));
1071#endif
1072
1073#if YYBTYACC
1074 yyps = yyNewState(0); if (yyps == 0) goto yyenomem;
1075 yyps->save = 0;
1076#endif /* YYBTYACC */

1078 yyn = 0;
1079 yynerrs = 0;
1080 yyerrflag = 0;
1081 yychar = YYEMPTY;

1083
1084#if YYPURE
1085 memset(&yystack, 0, sizeof(yystack));
1086#endif
1087
1088 if (yystack.s_base == NULL && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;

1090 yystack.l_mark = yystack.l_base;
1091#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1092 yystack.p_mark = yystack.p_base;
1093#endif
1094 yystate = 0;
1095 *yystack.s_mark = 0;
1096
1097yyloop:
1098 if ((yyn = yydefred[yystate]) != 0) goto yyreduce;
1099 if (yychar < 0)
1100 {
1101#if YYBTYACC
1102 do {
1103 if (yylvp < yylve)
1104 {
1105 /* we're currently re-reading tokens */
1106 yylval = *yylvp++;
1107#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1108 yylloc = *yylpp++;
1109#endif
1110 yychar = *yylexp++;
1111 break;
1112 }
1113 if (yyps->save)
1114 {
1115 /* in trial mode; save scanner results for future parse attempts */
1116 if (yylvp == yylvlim)
1117 { /* Enlarge lexical value queue */
1118 size_t p = (size_t) (yylvp - yylvals);
1119 size_t s = (size_t) (yylvlim - yylvals);
1120
1121 s += YYLVQUEUEGROWTH;
1122 if ((yylexemes = (YYINT *)realloc(yylexemes, s * sizeof(YYINT))) == NULL) goto yyenomem;
1123 if ((yylvals = (YYSTYPE *)realloc(yylvals, s * sizeof(YYSTYPE))) == NULL) goto yyenomem;
1124#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1125 if ((yylpsns = (YYLTYPE *)realloc(yylpsns, s * sizeof(YYLTYPE))) == NULL) goto yyenomem;
1126#endif
1127 yylvp = yylve = yylvals + p;
1128 yylvlim = yylvals + s;
1129#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1130 yylpp = yylpe = yylpsns + p;
1131 yylplim = yylpsns + s;
1132#endif
1133 yylexp = yylexemes + p;
1134 }
1135 *yylexp = (YYINT) YYLEX;
1136 *yylvp++ = yylval;
1137 yylve++;
1138#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1139 *yylpp++ = yylloc;
1140 yylpe++;
1141#endif
1142 yychar = *yylexp++;
1143 break;
1144 }
1145 /* normal operation, no conflict encountered */
1146#endif /* YYBTYACC */
1147 yychar = YYLEX;
1148#if YYBTYACC
1149 } while (0);
1150#endif /* YYBTYACC */
1151 if (yychar < 0) yychar = YYEOF;
1152#if YYDEBUG
1153 if (yydebug)
1154 {
1155 if ((yys = yyname[YYTRANSLATE(yychar)]) == NULL) yys = yyname[YYUNDFTOKEN];
1156 fprintf(stderr, "%s[%d]: state %d, reading token %d (%s)",
1157 YYDEBUGSTR, yydepth, yystate, yychar, yys);
1158#ifdef YYSTYPE_TOSTRING
1159#if YYBTYACC
1160 if (!yytrial)
1161#endif /* YYBTYACC */
1162 fprintf(stderr, " <%s>", YYSTYPE_TOSTRING(yychar, yylval));
1163#endif
1164 fputc('\n', stderr);
1165 }
1166#endif
1167 }
1168#if YYBTYACC
1169
1170 /* Do we have a conflict? */
1171 if (((yyn = yycindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
1172 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
1173 {
1174 YYINT ctry;
1175
1176 if (yypath)
1177 {
1178 YYParseState *save;
1179#if YYDEBUG
1180 if (yydebug)
1181 fprintf(stderr, "%s[%d]: CONFLICT in state %d: following successful trial parse\n",
1182 YYDEBUGSTR, yydepth, yystate);
1183#endif
1184 /* Switch to the next conflict context */
1185 save = yypath;
1186 yypath = save->save;
1187 save->save = NULL;
1188 ctry = save->ctry;
1189 if (save->state != yystate) YYABORT;
1190 yyFreeState(save);
1191
1192 }
1193 else
1194 {
1195
1196 /* Unresolved conflict - start/continue trial parse */
1197 YYParseState *save;
1198#if YYDEBUG
1199 if (yydebug)
1200 {
1201 fprintf(stderr, "%s[%d]: CONFLICT in state %d. ", YYDEBUGSTR, yydepth, yystate);
1202 if (yyps->save)
1203 fputs("ALREADY in conflict, continuing trial parse.\n", stderr);
1204 else
1205 fputs("Starting trial parse.\n", stderr);
1206 }
1207#endif
1208 save = yyNewState((unsigned)(yystack.s_mark - yystack.s_base + 1));
1209 if (save == NULL) goto yyenomem;
1210 save->save = yyps->save;
1211 save->state = yystate;
1212 save->errflag = yyerrflag;
1213 save->yystack.s_mark = save->yystack.s_base + (yystack.s_mark - yystack.s_base);
1214 memcpy (save->yystack.s_base, yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1215 save->yystack.l_mark = save->yystack.l_base + (yystack.l_mark - yystack.l_base);
1216 memcpy (save->yystack.l_base, yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1217#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1218 save->yystack.p_mark = save->yystack.p_base + (yystack.p_mark - yystack.p_base);
1219 memcpy (save->yystack.p_base, yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1220#endif
1221 ctry = yytable[yyn];
1222 if (yyctable[ctry] == -1)
1223 {
1224#if YYDEBUG
1225 if (yydebug && yychar >= YYEOF)
1226 fprintf(stderr, "%s[%d]: backtracking 1 token\n", YYDEBUGSTR, yydepth);
1227#endif
1228 ctry++;
1229 }
1230 save->ctry = ctry;
1231 if (yyps->save == NULL)
1232 {
1233 /* If this is a first conflict in the stack, start saving lexemes */
1234 if (!yylexemes)
1235 {
1236 yylexemes = (YYINT *) malloc((YYLVQUEUEGROWTH) * sizeof(YYINT));
1237 if (yylexemes == NULL) goto yyenomem;
1238 yylvals = (YYSTYPE *) malloc((YYLVQUEUEGROWTH) * sizeof(YYSTYPE));
1239 if (yylvals == NULL) goto yyenomem;
1240 yylvlim = yylvals + YYLVQUEUEGROWTH;
1241#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1242 yylpsns = (YYLTYPE *) malloc((YYLVQUEUEGROWTH) * sizeof(YYLTYPE));
1243 if (yylpsns == NULL) goto yyenomem;
1244 yylplim = yylpsns + YYLVQUEUEGROWTH;
1245#endif
1246 }
1247 if (yylvp == yylve)
1248 {
1249 yylvp = yylve = yylvals;
1250#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1251 yylpp = yylpe = yylpsns;
1252#endif
1253 yylexp = yylexemes;
1254 if (yychar >= YYEOF)
1255 {
1256 *yylve++ = yylval;
1257#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1258 *yylpe++ = yylloc;
1259#endif
1260 *yylexp = (YYINT) yychar;
1261 yychar = YYEMPTY;
1262 }
1263 }
1264 }
1265 if (yychar >= YYEOF)
1266 {
1267 yylvp--;
1268#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1269 yylpp--;
1270#endif
1271 yylexp--;
1272 yychar = YYEMPTY;
1273 }
1274 save->lexeme = (int) (yylvp - yylvals);
1275 yyps->save = save;
1276 }
1277 if (yytable[yyn] == ctry)
1278 {
1279#if YYDEBUG
1280 if (yydebug)
1281 fprintf(stderr, "%s[%d]: state %d, shifting to state %d\n",
1282 YYDEBUGSTR, yydepth, yystate, yyctable[ctry]);
1283#endif
1284 if (yychar < 0)
1285 {
1286 yylvp++;
1287#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1288 yylpp++;
1289#endif
1290 yylexp++;
1291 }
1292 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM)
1293 goto yyoverflow;
1294 yystate = yyctable[ctry];
1295 *++yystack.s_mark = (YYINT) yystate;
1296 *++yystack.l_mark = yylval;
1297#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1298 *++yystack.p_mark = yylloc;
1299#endif
1300 yychar = YYEMPTY;
1301 if (yyerrflag > 0) --yyerrflag;
1302 goto yyloop;
1303 }
1304 else
1305 {
1306 yyn = yyctable[ctry];
1307 goto yyreduce;
1308 }
1309 } /* End of code dealing with conflicts */
1310#endif /* YYBTYACC */
1311 if (((yyn = yysindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
1312 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
1313 {
1314#if YYDEBUG
1315 if (yydebug)
1316 fprintf(stderr, "%s[%d]: state %d, shifting to state %d\n",
1317 YYDEBUGSTR, yydepth, yystate, yytable[yyn]);
1318#endif
1319 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
1320 yystate = yytable[yyn];
1321 *++yystack.s_mark = yytable[yyn];
1322 *++yystack.l_mark = yylval;
1323#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1324 *++yystack.p_mark = yylloc;
1325#endif
1326 yychar = YYEMPTY;
1327 if (yyerrflag > 0) --yyerrflag;
1328 goto yyloop;
1329 }
1330 if (((yyn = yyrindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
1331 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
1332 {
1333 yyn = yytable[yyn];
1334 goto yyreduce;
1335 }
1336 if (yyerrflag != 0) goto yyinrecovery;
1337#if YYBTYACC
1338
1339 yynewerrflag = 1;
1340 goto yyerrhandler;
1341 goto yyerrlab; /* redundant goto avoids 'unused label' warning */
1342
1343yyerrlab:
1344 /* explicit YYERROR from an action -- pop the rhs of the rule reduced
1345 * before looking for error recovery */
1346 yystack.s_mark -= yym;
1347 yystate = *yystack.s_mark;
1348 yystack.l_mark -= yym;
1349#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1350 yystack.p_mark -= yym;
1351#endif
1352
1353 yynewerrflag = 0;
1354yyerrhandler:
1355 while (yyps->save)
1356 {
1357 int ctry;
1358 YYParseState *save = yyps->save;
1359#if YYDEBUG
1360 if (yydebug)
1361 fprintf(stderr, "%s[%d]: ERROR in state %d, CONFLICT BACKTRACKING to state %d, %d tokens\n",
1362 YYDEBUGSTR, yydepth, yystate, yyps->save->state,
1363 (int)(yylvp - yylvals - yyps->save->lexeme));
1364#endif
1365 /* Memorize most forward-looking error state in case it's really an error. */
1366 if (yyerrctx == NULL || yyerrctx->lexeme < yylvp - yylvals)
1367 {
1368 /* Free old saved error context state */
1369 if (yyerrctx) yyFreeState(yyerrctx);
1370 /* Create and fill out new saved error context state */
1371 yyerrctx = yyNewState((unsigned)(yystack.s_mark - yystack.s_base + 1));
1372 if (yyerrctx == NULL) goto yyenomem;
1373 yyerrctx->save = yyps->save;
1374 yyerrctx->state = yystate;
1375 yyerrctx->errflag = yyerrflag;
1376 yyerrctx->yystack.s_mark = yyerrctx->yystack.s_base + (yystack.s_mark - yystack.s_base);
1377 memcpy (yyerrctx->yystack.s_base, yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1378 yyerrctx->yystack.l_mark = yyerrctx->yystack.l_base + (yystack.l_mark - yystack.l_base);
1379 memcpy (yyerrctx->yystack.l_base, yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1380#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1381 yyerrctx->yystack.p_mark = yyerrctx->yystack.p_base + (yystack.p_mark - yystack.p_base);
1382 memcpy (yyerrctx->yystack.p_base, yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1383#endif
1384 yyerrctx->lexeme = (int) (yylvp - yylvals);
1385 }
1386 yylvp = yylvals + save->lexeme;
1387#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1388 yylpp = yylpsns + save->lexeme;
1389#endif
1390 yylexp = yylexemes + save->lexeme;
1391 yychar = YYEMPTY;
1392 yystack.s_mark = yystack.s_base + (save->yystack.s_mark - save->yystack.s_base);
1393 memcpy (yystack.s_base, save->yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1394 yystack.l_mark = yystack.l_base + (save->yystack.l_mark - save->yystack.l_base);
1395 memcpy (yystack.l_base, save->yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1396#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1397 yystack.p_mark = yystack.p_base + (save->yystack.p_mark - save->yystack.p_base);
1398 memcpy (yystack.p_base, save->yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1399#endif
1400 ctry = ++save->ctry;
1401 yystate = save->state;
1402 /* We tried shift, try reduce now */
1403 if ((yyn = yyctable[ctry]) >= 0) goto yyreduce;
1404 yyps->save = save->save;
1405 save->save = NULL;
1406 yyFreeState(save);
1407
1408 /* Nothing left on the stack -- error */
1409 if (!yyps->save)
1410 {
1411#if YYDEBUG
1412 if (yydebug)
1413 fprintf(stderr, "%sdebug[%d,trial]: trial parse FAILED, entering ERROR mode\n",
1414 YYPREFIX, yydepth);
1415#endif
1416 /* Restore state as it was in the most forward-advanced error */
1417 yylvp = yylvals + yyerrctx->lexeme;
1418#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1419 yylpp = yylpsns + yyerrctx->lexeme;
1420#endif
1421 yylexp = yylexemes + yyerrctx->lexeme;
1422 yychar = yylexp[-1];
1423 yylval = yylvp[-1];
1424#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1425 yylloc = yylpp[-1];
1426#endif
1427 yystack.s_mark = yystack.s_base + (yyerrctx->yystack.s_mark - yyerrctx->yystack.s_base);
1428 memcpy (yystack.s_base, yyerrctx->yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1429 yystack.l_mark = yystack.l_base + (yyerrctx->yystack.l_mark - yyerrctx->yystack.l_base);
1430 memcpy (yystack.l_base, yyerrctx->yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1431#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1432 yystack.p_mark = yystack.p_base + (yyerrctx->yystack.p_mark - yyerrctx->yystack.p_base);
1433 memcpy (yystack.p_base, yyerrctx->yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1434#endif
1435 yystate = yyerrctx->state;
1436 yyFreeState(yyerrctx);
1437 yyerrctx = NULL;
1438 }
1439 yynewerrflag = 1;
1440 }
1441 if (yynewerrflag == 0) goto yyinrecovery;
1442#endif /* YYBTYACC */
1443
1444 YYERROR_CALL("syntax error");
1445#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1446 yyerror_loc_range[1] = yylloc; /* lookahead position is error start position */
1447#endif
1448
1449#if !YYBTYACC
1450 goto yyerrlab; /* redundant goto avoids 'unused label' warning */
1451yyerrlab:
1452#endif
1453 ++yynerrs;
1454
1455yyinrecovery:
1456 if (yyerrflag < 3)
1457 {
1458 yyerrflag = 3;
1459 for (;;)
1460 {
1461 if (((yyn = yysindex[*yystack.s_mark]) != 0) && (yyn += YYERRCODE) >= 0 &&
1462 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) YYERRCODE)
1463 {
1464#if YYDEBUG
1465 if (yydebug)
1466 fprintf(stderr, "%s[%d]: state %d, error recovery shifting to state %d\n",
1467 YYDEBUGSTR, yydepth, *yystack.s_mark, yytable[yyn]);
1468#endif
1469 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
1470 yystate = yytable[yyn];
1471 *++yystack.s_mark = yytable[yyn];
1472 *++yystack.l_mark = yylval;
1473#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1474 /* lookahead position is error end position */
1475 yyerror_loc_range[2] = yylloc;
1476 YYLLOC_DEFAULT(yyloc, yyerror_loc_range, 2); /* position of error span */
1477 *++yystack.p_mark = yyloc;
1478#endif
1479 goto yyloop;
1480 }
1481 else
1482 {
1483#if YYDEBUG
1484 if (yydebug)
1485 fprintf(stderr, "%s[%d]: error recovery discarding state %d\n",
1486 YYDEBUGSTR, yydepth, *yystack.s_mark);
1487#endif
1488 if (yystack.s_mark <= yystack.s_base) goto yyabort;
1489#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1490 /* the current TOS position is the error start position */
1491 yyerror_loc_range[1] = *yystack.p_mark;

2491 size_t s = (size_t) (yylvlim - yylvals);

2493 s += YYLVQUEUEGROWTH;

2495 goto yyenomem;
2496 if ((yylvals = (YYSTYPE *)realloc(yylvals, s * sizeof(YYSTYPE))) == NULL)
2497 goto yyenomem;
2498#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
2499 if ((yylpsns = (YYLTYPE *)realloc(yylpsns, s * sizeof(YYLTYPE))) == NULL)
2500 goto yyenomem;
2501#endif

2503 yylvlim = yylvals + s;
2504#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
2505 yylpp = yylpe = yylpsns + p;
2506 yylplim = yylpsns + s;
2507#endif
2508 yylexp = yylexemes + p;
2509 }

2512 yylve++;
2513#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
2514 *yylpp++ = yylloc;
2515 yylpe++;
2516#endif
2517 yychar = *yylexp++;
2518 break;
2519 }
2520 /* normal operation, no conflict encountered */
2521#endif /* YYBTYACC */

2523#if YYBTYACC
2524 } while (0);
2525#endif /* YYBTYACC */

2529 {

2532 YYDEBUGSTR, yydepth, YYFINAL, yychar, yys);
2533 }
2534#endif
2535 }

2537 goto yyloop;
2538 }

2541 yystate = yytable[yyn];

2551#endif /* YYBTYACC */

2559 *++yystack.l_mark = yyval;

2570 {

2573 save->save = yypath;
2574 yypath = save;
2575 }
2576#if YYDEBUG

2579 YYDEBUGSTR, yydepth, yystate, yypath->state, (int)(yylvp - yylvals - yypath->lexeme));
2580#endif

2596#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
2597 yystack.p_mark = yystack.p_base + (yypath->yystack.p_mark - yypath->yystack.p_base);

2599#endif

2602#endif /* YYBTYACC */

2628#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)

2630#else

2632#endif /* defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED) */

2636#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)

2638

2640 YYDESTRUCT_CALL("cleanup: discarding state",
2641 yystos[*(yystack.s_base + (pv - yystack.l_base))], pv, pp);
2642#else
2643 for (pv = yystack.l_base; pv <= yystack.l_mark; ++pv)
2644 YYDESTRUCT_CALL("cleanup: discarding state",
2645 yystos[*(yystack.s_base + (pv - yystack.l_base))], pv);
2646#endif /* defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED) */
2647 }
2648#endif /* defined(YYDESTRUCT_CALL) */
2649
2650#if YYBTYACC
2651 if (yyerrctx)
2652 {
2653 yyFreeState(yyerrctx);
2654 yyerrctx = NULL;
2655 }
2656 while (yyps)
2657 {
2658 YYParseState *save = yyps;
2659 yyps = save->save;
2660 save->save = NULL;
2661 yyFreeState(save);
2662 }

2664 {

2669 }
