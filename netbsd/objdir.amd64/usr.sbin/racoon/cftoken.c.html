4773int
4774yycf_switch_buffer(path)
4775 char *path;
4776{
4777 char *filepath = NULL;
4778
4779 /* got the include file name */
4780 if (incstackp >= MAX_INCLUDE_DEPTH) {
4781 plog(LLV_ERROR, LOCATION, NULL,
4782 "Includes nested too deeply");
4783 return -1;
4784 }
4785
4786 if (glob(path, GLOB_TILDE, NULL, &incstack[incstackp].matches) != 0 ||
4787 incstack[incstackp].matches.gl_pathc == 0) {
4788 plog(LLV_ERROR, LOCATION, NULL,
4789 "glob found no matches for path \"%s\"\n", path);
4790 return -1;
4791 }
4792 incstack[incstackp].matchon = 0;
4793 incstack[incstackp].prevstate = YY_CURRENT_BUFFER;
4794
4795 nextmatch:
4796 if (incstack[incstackp].matchon >= incstack[incstackp].matches.gl_pathc)
4797 return -1;
4798 filepath =
4799 incstack[incstackp].matches.gl_pathv[incstack[incstackp].matchon];
4800 incstack[incstackp].matchon++;
4801 incstackp++;
4802
4803 if (yycf_set_buffer(filepath) != 0) {
4804 incstackp--;
4805 goto nextmatch;
4806 }
4807
4808 yy_switch_to_buffer(yy_create_buffer(yyin, YY_BUF_SIZE));
4809
4810 BEGIN(S_INI);
4811
4812 return 0;
4813}
