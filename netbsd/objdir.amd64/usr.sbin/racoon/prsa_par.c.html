655 else if ((newsize *= 2) > YYMAXDEPTH)
656 newsize = YYMAXDEPTH;
657
658 i = (int) (data->s_mark - data->s_base);
659 newss = (YYINT *)realloc(data->s_base, newsize * sizeof(*newss));
660 if (newss == 0)
661 return YYENOMEM;
662
663 data->s_base = newss;
664 data->s_mark = newss + i;
665
666 newvs = (YYSTYPE *)realloc(data->l_base, newsize * sizeof(*newvs));
667 if (newvs == 0)
668 return YYENOMEM;
669
670 data->l_base = newvs;
671 data->l_mark = newvs + i;
672
673#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
674 newps = (YYLTYPE *)realloc(data->p_base, newsize * sizeof(*newps));
675 if (newps == 0)
676 return YYENOMEM;
677
678 data->p_base = newps;
679 data->p_mark = newps + i;
680#endif
681
682 data->stacksize = newsize;
683 data->s_last = data->s_base + newsize - 1;
684
685#if YYDEBUG
686 if (yydebug)
687 fprintf(stderr, "%sdebug: stack size increased to %d\n", YYPREFIX, newsize);
688#endif
689 return 0;
690}

745#define YYABORT goto yyabort
746#define YYREJECT goto yyabort
747#define YYACCEPT goto yyaccept
748#define YYERROR goto yyerrlab
749#if YYBTYACC
750#define YYVALID do { if (yyps->save) goto yyvalid; } while(0)

752 yyps->save->save == 0) goto yyvalid; } while(0)

754
755int
756YYPARSE_DECL()
757{
758 int yym, yyn, yystate, yyresult;
759#if YYBTYACC
760 int yynewerrflag;
761 YYParseState *yyerrctx = NULL;
762#endif /* YYBTYACC */
763#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
764 YYLTYPE yyerror_loc_range[3]; /* position of error start/end (0 unused) */
765#endif
766#if YYDEBUG
767 const char *yys;
768
769 if ((yys = getenv("YYDEBUG")) != 0)
770 {
771 yyn = *yys;
772 if (yyn >= '0' && yyn <= '9')
773 yydebug = yyn - '0';
774 }
775 if (yydebug)
776 fprintf(stderr, "%sdebug[<# of symbols on state stack>]\n", YYPREFIX);
777#endif
778#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
779 memset(yyerror_loc_range, 0, sizeof(yyerror_loc_range));
780#endif
781
782#if YYBTYACC
783 yyps = yyNewState(0); if (yyps == 0) goto yyenomem;
784 yyps->save = 0;
785#endif /* YYBTYACC */
786 yym = 0;
787 yyn = 0;
788 yynerrs = 0;
789 yyerrflag = 0;
790 yychar = YYEMPTY;
791 yystate = 0;
792
793#if YYPURE
794 memset(&yystack, 0, sizeof(yystack));
795#endif
796
797 if (yystack.s_base == NULL && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
798 yystack.s_mark = yystack.s_base;
799 yystack.l_mark = yystack.l_base;
800#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
801 yystack.p_mark = yystack.p_base;
802#endif
803 yystate = 0;
804 *yystack.s_mark = 0;
805
806yyloop:
807 if ((yyn = yydefred[yystate]) != 0) goto yyreduce;
808 if (yychar < 0)
809 {
810#if YYBTYACC
811 do {
812 if (yylvp < yylve)
813 {
814 /* we're currently re-reading tokens */
815 yylval = *yylvp++;
816#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
817 yylloc = *yylpp++;
818#endif
819 yychar = *yylexp++;
820 break;
821 }
822 if (yyps->save)
823 {
824 /* in trial mode; save scanner results for future parse attempts */
825 if (yylvp == yylvlim)
826 { /* Enlarge lexical value queue */
827 size_t p = (size_t) (yylvp - yylvals);
828 size_t s = (size_t) (yylvlim - yylvals);
829
830 s += YYLVQUEUEGROWTH;
831 if ((yylexemes = (YYINT *)realloc(yylexemes, s * sizeof(YYINT))) == NULL) goto yyenomem;
832 if ((yylvals = (YYSTYPE *)realloc(yylvals, s * sizeof(YYSTYPE))) == NULL) goto yyenomem;
833#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
834 if ((yylpsns = (YYLTYPE *)realloc(yylpsns, s * sizeof(YYLTYPE))) == NULL) goto yyenomem;
835#endif
836 yylvp = yylve = yylvals + p;
837 yylvlim = yylvals + s;
838#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
839 yylpp = yylpe = yylpsns + p;
840 yylplim = yylpsns + s;
841#endif
842 yylexp = yylexemes + p;
843 }
844 *yylexp = (YYINT) YYLEX;
845 *yylvp++ = yylval;
846 yylve++;
847#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
848 *yylpp++ = yylloc;
849 yylpe++;
850#endif
851 yychar = *yylexp++;
852 break;
853 }
854 /* normal operation, no conflict encountered */
855#endif /* YYBTYACC */
856 yychar = YYLEX;
857#if YYBTYACC
858 } while (0);
859#endif /* YYBTYACC */
860 if (yychar < 0) yychar = YYEOF;
861#if YYDEBUG
862 if (yydebug)
863 {
864 if ((yys = yyname[YYTRANSLATE(yychar)]) == NULL) yys = yyname[YYUNDFTOKEN];
865 fprintf(stderr, "%s[%d]: state %d, reading token %d (%s)",
866 YYDEBUGSTR, yydepth, yystate, yychar, yys);
867#ifdef YYSTYPE_TOSTRING
868#if YYBTYACC
869 if (!yytrial)
870#endif /* YYBTYACC */
871 fprintf(stderr, " <%s>", YYSTYPE_TOSTRING(yychar, yylval));
872#endif
873 fputc('\n', stderr);
874 }
875#endif
876 }
877#if YYBTYACC
878
879 /* Do we have a conflict? */
880 if (((yyn = yycindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
881 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
882 {
883 YYINT ctry;
884
885 if (yypath)
886 {
887 YYParseState *save;
888#if YYDEBUG
889 if (yydebug)
890 fprintf(stderr, "%s[%d]: CONFLICT in state %d: following successful trial parse\n",
891 YYDEBUGSTR, yydepth, yystate);
892#endif
893 /* Switch to the next conflict context */
894 save = yypath;
895 yypath = save->save;
896 save->save = NULL;
897 ctry = save->ctry;
898 if (save->state != yystate) YYABORT;
899 yyFreeState(save);
900
901 }
902 else
903 {
904
905 /* Unresolved conflict - start/continue trial parse */
906 YYParseState *save;
907#if YYDEBUG
908 if (yydebug)
909 {
910 fprintf(stderr, "%s[%d]: CONFLICT in state %d. ", YYDEBUGSTR, yydepth, yystate);
911 if (yyps->save)
912 fputs("ALREADY in conflict, continuing trial parse.\n", stderr);
913 else
914 fputs("Starting trial parse.\n", stderr);
915 }
916#endif
917 save = yyNewState((unsigned)(yystack.s_mark - yystack.s_base + 1));
918 if (save == NULL) goto yyenomem;
919 save->save = yyps->save;
920 save->state = yystate;
921 save->errflag = yyerrflag;
922 save->yystack.s_mark = save->yystack.s_base + (yystack.s_mark - yystack.s_base);
923 memcpy (save->yystack.s_base, yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
924 save->yystack.l_mark = save->yystack.l_base + (yystack.l_mark - yystack.l_base);
925 memcpy (save->yystack.l_base, yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
926#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
927 save->yystack.p_mark = save->yystack.p_base + (yystack.p_mark - yystack.p_base);
928 memcpy (save->yystack.p_base, yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
929#endif
930 ctry = yytable[yyn];
931 if (yyctable[ctry] == -1)
932 {
933#if YYDEBUG
934 if (yydebug && yychar >= YYEOF)
935 fprintf(stderr, "%s[%d]: backtracking 1 token\n", YYDEBUGSTR, yydepth);
936#endif
937 ctry++;
938 }
939 save->ctry = ctry;
940 if (yyps->save == NULL)
941 {
942 /* If this is a first conflict in the stack, start saving lexemes */
943 if (!yylexemes)
944 {
945 yylexemes = (YYINT *) malloc((YYLVQUEUEGROWTH) * sizeof(YYINT));
946 if (yylexemes == NULL) goto yyenomem;
947 yylvals = (YYSTYPE *) malloc((YYLVQUEUEGROWTH) * sizeof(YYSTYPE));
948 if (yylvals == NULL) goto yyenomem;
949 yylvlim = yylvals + YYLVQUEUEGROWTH;
950#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
951 yylpsns = (YYLTYPE *) malloc((YYLVQUEUEGROWTH) * sizeof(YYLTYPE));
952 if (yylpsns == NULL) goto yyenomem;
953 yylplim = yylpsns + YYLVQUEUEGROWTH;
954#endif
955 }
956 if (yylvp == yylve)
957 {
958 yylvp = yylve = yylvals;
959#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
960 yylpp = yylpe = yylpsns;
961#endif
962 yylexp = yylexemes;
963 if (yychar >= YYEOF)
964 {
965 *yylve++ = yylval;
966#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
967 *yylpe++ = yylloc;
968#endif
969 *yylexp = (YYINT) yychar;
970 yychar = YYEMPTY;
971 }
972 }
973 }
974 if (yychar >= YYEOF)
975 {
976 yylvp--;
977#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
978 yylpp--;
979#endif
980 yylexp--;
981 yychar = YYEMPTY;
982 }
983 save->lexeme = (int) (yylvp - yylvals);
984 yyps->save = save;
985 }
986 if (yytable[yyn] == ctry)
987 {
988#if YYDEBUG
989 if (yydebug)
990 fprintf(stderr, "%s[%d]: state %d, shifting to state %d\n",
991 YYDEBUGSTR, yydepth, yystate, yyctable[ctry]);
992#endif
993 if (yychar < 0)
994 {
995 yylvp++;
996#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
997 yylpp++;
998#endif
999 yylexp++;
1000 }
1001 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM)
1002 goto yyoverflow;
1003 yystate = yyctable[ctry];
1004 *++yystack.s_mark = (YYINT) yystate;
1005 *++yystack.l_mark = yylval;
1006#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1007 *++yystack.p_mark = yylloc;
1008#endif
1009 yychar = YYEMPTY;
1010 if (yyerrflag > 0) --yyerrflag;
1011 goto yyloop;
1012 }
1013 else
1014 {
1015 yyn = yyctable[ctry];
1016 goto yyreduce;
1017 }
1018 } /* End of code dealing with conflicts */
1019#endif /* YYBTYACC */
1020 if (((yyn = yysindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
1021 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
1022 {
1023#if YYDEBUG
1024 if (yydebug)
1025 fprintf(stderr, "%s[%d]: state %d, shifting to state %d\n",
1026 YYDEBUGSTR, yydepth, yystate, yytable[yyn]);
1027#endif
1028 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
1029 yystate = yytable[yyn];
1030 *++yystack.s_mark = yytable[yyn];
1031 *++yystack.l_mark = yylval;
1032#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1033 *++yystack.p_mark = yylloc;
1034#endif
1035 yychar = YYEMPTY;
1036 if (yyerrflag > 0) --yyerrflag;
1037 goto yyloop;
1038 }
1039 if (((yyn = yyrindex[yystate]) != 0) && (yyn += yychar) >= 0 &&
1040 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) yychar)
1041 {
1042 yyn = yytable[yyn];
1043 goto yyreduce;
1044 }
1045 if (yyerrflag != 0) goto yyinrecovery;
1046#if YYBTYACC
1047
1048 yynewerrflag = 1;
1049 goto yyerrhandler;
1050 goto yyerrlab; /* redundant goto avoids 'unused label' warning */
1051
1052yyerrlab:
1053 /* explicit YYERROR from an action -- pop the rhs of the rule reduced
1054 * before looking for error recovery */
1055 yystack.s_mark -= yym;
1056 yystate = *yystack.s_mark;
1057 yystack.l_mark -= yym;
1058#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1059 yystack.p_mark -= yym;
1060#endif
1061
1062 yynewerrflag = 0;
1063yyerrhandler:
1064 while (yyps->save)
1065 {
1066 int ctry;
1067 YYParseState *save = yyps->save;
1068#if YYDEBUG
1069 if (yydebug)
1070 fprintf(stderr, "%s[%d]: ERROR in state %d, CONFLICT BACKTRACKING to state %d, %d tokens\n",
1071 YYDEBUGSTR, yydepth, yystate, yyps->save->state,
1072 (int)(yylvp - yylvals - yyps->save->lexeme));
1073#endif
1074 /* Memorize most forward-looking error state in case it's really an error. */
1075 if (yyerrctx == NULL || yyerrctx->lexeme < yylvp - yylvals)
1076 {
1077 /* Free old saved error context state */
1078 if (yyerrctx) yyFreeState(yyerrctx);
1079 /* Create and fill out new saved error context state */
1080 yyerrctx = yyNewState((unsigned)(yystack.s_mark - yystack.s_base + 1));
1081 if (yyerrctx == NULL) goto yyenomem;
1082 yyerrctx->save = yyps->save;
1083 yyerrctx->state = yystate;
1084 yyerrctx->errflag = yyerrflag;
1085 yyerrctx->yystack.s_mark = yyerrctx->yystack.s_base + (yystack.s_mark - yystack.s_base);
1086 memcpy (yyerrctx->yystack.s_base, yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1087 yyerrctx->yystack.l_mark = yyerrctx->yystack.l_base + (yystack.l_mark - yystack.l_base);
1088 memcpy (yyerrctx->yystack.l_base, yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1089#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1090 yyerrctx->yystack.p_mark = yyerrctx->yystack.p_base + (yystack.p_mark - yystack.p_base);
1091 memcpy (yyerrctx->yystack.p_base, yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1092#endif
1093 yyerrctx->lexeme = (int) (yylvp - yylvals);
1094 }
1095 yylvp = yylvals + save->lexeme;
1096#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1097 yylpp = yylpsns + save->lexeme;
1098#endif
1099 yylexp = yylexemes + save->lexeme;
1100 yychar = YYEMPTY;
1101 yystack.s_mark = yystack.s_base + (save->yystack.s_mark - save->yystack.s_base);
1102 memcpy (yystack.s_base, save->yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1103 yystack.l_mark = yystack.l_base + (save->yystack.l_mark - save->yystack.l_base);
1104 memcpy (yystack.l_base, save->yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1105#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1106 yystack.p_mark = yystack.p_base + (save->yystack.p_mark - save->yystack.p_base);
1107 memcpy (yystack.p_base, save->yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1108#endif
1109 ctry = ++save->ctry;
1110 yystate = save->state;
1111 /* We tried shift, try reduce now */
1112 if ((yyn = yyctable[ctry]) >= 0) goto yyreduce;
1113 yyps->save = save->save;
1114 save->save = NULL;
1115 yyFreeState(save);
1116
1117 /* Nothing left on the stack -- error */
1118 if (!yyps->save)
1119 {
1120#if YYDEBUG
1121 if (yydebug)
1122 fprintf(stderr, "%sdebug[%d,trial]: trial parse FAILED, entering ERROR mode\n",
1123 YYPREFIX, yydepth);
1124#endif
1125 /* Restore state as it was in the most forward-advanced error */
1126 yylvp = yylvals + yyerrctx->lexeme;
1127#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1128 yylpp = yylpsns + yyerrctx->lexeme;
1129#endif
1130 yylexp = yylexemes + yyerrctx->lexeme;
1131 yychar = yylexp[-1];
1132 yylval = yylvp[-1];
1133#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1134 yylloc = yylpp[-1];
1135#endif
1136 yystack.s_mark = yystack.s_base + (yyerrctx->yystack.s_mark - yyerrctx->yystack.s_base);
1137 memcpy (yystack.s_base, yyerrctx->yystack.s_base, (size_t) (yystack.s_mark - yystack.s_base + 1) * sizeof(YYINT));
1138 yystack.l_mark = yystack.l_base + (yyerrctx->yystack.l_mark - yyerrctx->yystack.l_base);
1139 memcpy (yystack.l_base, yyerrctx->yystack.l_base, (size_t) (yystack.l_mark - yystack.l_base + 1) * sizeof(YYSTYPE));
1140#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1141 yystack.p_mark = yystack.p_base + (yyerrctx->yystack.p_mark - yyerrctx->yystack.p_base);
1142 memcpy (yystack.p_base, yyerrctx->yystack.p_base, (size_t) (yystack.p_mark - yystack.p_base + 1) * sizeof(YYLTYPE));
1143#endif
1144 yystate = yyerrctx->state;
1145 yyFreeState(yyerrctx);
1146 yyerrctx = NULL;
1147 }
1148 yynewerrflag = 1;
1149 }
1150 if (yynewerrflag == 0) goto yyinrecovery;
1151#endif /* YYBTYACC */
1152
1153 YYERROR_CALL("syntax error");
1154#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1155 yyerror_loc_range[1] = yylloc; /* lookahead position is error start position */
1156#endif
1157
1158#if !YYBTYACC
1159 goto yyerrlab; /* redundant goto avoids 'unused label' warning */
1160yyerrlab:
1161#endif
1162 ++yynerrs;
1163
1164yyinrecovery:
1165 if (yyerrflag < 3)
1166 {
1167 yyerrflag = 3;
1168 for (;;)
1169 {
1170 if (((yyn = yysindex[*yystack.s_mark]) != 0) && (yyn += YYERRCODE) >= 0 &&
1171 yyn <= YYTABLESIZE && yycheck[yyn] == (YYINT) YYERRCODE)
1172 {
1173#if YYDEBUG
1174 if (yydebug)
1175 fprintf(stderr, "%s[%d]: state %d, error recovery shifting to state %d\n",
1176 YYDEBUGSTR, yydepth, *yystack.s_mark, yytable[yyn]);
1177#endif
1178 if (yystack.s_mark >= yystack.s_last && yygrowstack(&yystack) == YYENOMEM) goto yyoverflow;
1179 yystate = yytable[yyn];
1180 *++yystack.s_mark = yytable[yyn];
1181 *++yystack.l_mark = yylval;
1182#if defined(YYLTYPE) || defined(YYLTYPE_IS_DECLARED)
1183 /* lookahead position is error end position */
1184 yyerror_loc_range[2] = yylloc;
1185 YYLLOC_DEFAULT(yyloc, yyerror_loc_range, 2); /* position of error span */
1186 *++yystack.p_mark = yyloc;
1187#endif
1188 goto yyloop;
1189 }
1190 else
1191 {
1192#if YYDEBUG
1193 if (yydebug)
1194 fprintf(stderr, "%s[%d]: error recovery discarding state %d\n",
1195 YYDEBUGSTR, yydepth, *yystack.s_mark);
1196#endif
1197 if (yystack.s_mark <= yystack.s_base) goto yyabort;
