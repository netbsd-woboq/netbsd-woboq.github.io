456WINDOW *
457msg_window(WINDOW *window)
458{
459 size_t ncbuffersize;
460 char *ncbuffer;
461 WINDOW *old;
462
463 old = msg_win;
464 if (!window)
465 return old;
466 msg_win = window;
467
468 ncbuffersize = getmaxx(window) * getmaxy(window) + 1;
469 while (ncbuffersize > cbuffersize) {
470 ncbuffer = malloc(ncbuffersize);
471 if (ncbuffer == NULL) {
472 /* we might get truncated messages... */
473 ncbuffersize <<= 1;
474 continue;
475 }
476 if (cbuffer != NULL)
477 free(cbuffer);
478 cbuffer = ncbuffer;
479 cbuffersize = ncbuffersize;
480 break;
481 }
482 last_o_was_punct = 0;
483 last_o_was_space = 1;
484 return old;
485}

487int
488msg_file(const char *file)
489{
490 int fd;
491
492 if (msgmap != MAP_FAILED)
493 munmap(msgmap, msgmapsz);
494 msgmap = MAP_FAILED;
495 if (!file)
496 return 0;
497 fd = open(file, O_RDONLY, 0);
498 if (fd == -1)
499 return -1;
500 msgmapsz = lseek(fd, 0, SEEK_END);
501 msgmap = mmap(0, msgmapsz, PROT_READ, MAP_SHARED, fd, 0);
502 close(fd);
503 if (msgmap == MAP_FAILED)
504 return -1;
505 /* check_magic */
506 if (strcmp(msgmap, "MSGTXTS") != 0) {
507 msg_file(NULL);
508 return -1;
509 }
510 msgmapcount = atoi(msgmap + 8);
511 return 0;

516{
517 uintptr_t m = (uintptr_t)msg_no;

529 return msg_list[m];

555static int __printflike(2, 0)
556_msg_vprintf(int auto_fill, const char *fmt, va_list ap)
557{
558 const char *wstart, *afterw;
559 int wordlen, nspaces;
560 int ret;
561
562 ret = vsnprintf(cbuffer, cbuffersize, fmt, ap);
563
564 if (!auto_fill) {
565 waddstr(msg_win, cbuffer);
566
567 /*
568 * nothing is perfect if they flow text after a table,
569 * but this may be decent.
570 */
571 last_i_was_nl = last_i_was_space = 1;
572 last_o_was_punct = 0;
573 last_o_was_space = 1;
574 goto out;
575 }
576
577 for (wstart = afterw = cbuffer; *wstart; wstart = afterw) {
578
579 /* eat one space, or a whole word of non-spaces */
580 if (isspace((unsigned char)*afterw))
581 afterw++;
582 else
583 while (*afterw && !isspace((unsigned char)*afterw))
584 afterw++;
585
586 /* this is an nl: special formatting necessary */
587 if (*wstart == '\n') {
588 if (last_i_was_nl || last_i_was_space) {
589
590 if (getcurx(msg_win) != 0)
591 waddch(msg_win, '\n');
592 if (last_i_was_nl) {
593 /* last was an nl: paragraph break */
594 waddch(msg_win, '\n');
595 } else {
596 /* last was space: line break */
597 }
598 last_o_was_punct = 0;
599 last_o_was_space = 1;
600 } else {
601 /* last_o_was_punct unchanged */
602 /* last_o_was_space unchanged */
603 }
604 last_i_was_space = 1;
605 last_i_was_nl = 1;
606 continue;
607 }
608
609 /* this is a tab: special formatting necessary. */
610 if (*wstart == '\t') {
611 if (last_i_was_nl) {
612 /* last was an nl: list indent */
613 if (getcurx(msg_win) != 0)
614 waddch(msg_win, '\n');
615 } else {
616 /* last was not an nl: columns */
617 }
618 waddch(msg_win, '\t');
619 last_i_was_nl = 0;
620 last_i_was_space = 1;
621 last_o_was_punct = 0;
622 last_o_was_space = 1;
623 continue;
624 }
625
626 /* this is a space: ignore it but set flags */
627 last_i_was_nl = 0; /* all newlines handled above */
628 last_i_was_space = isspace((unsigned char)*wstart);
629 if (last_i_was_space)
630 continue;
631
632 /*
633 * we have a real "word," i.e. a sequence of non-space
634 * characters. wstart is now the start of the word,
635 * afterw is the next character after the end.
636 */
637 wordlen = afterw - wstart;
638 nspaces = last_o_was_space ? 0 : (last_o_was_punct ? 2 : 1);
639 if ((getcurx(msg_win) + nspaces + wordlen) >=
640 getmaxx(msg_win) &&
641 wordlen < (getmaxx(msg_win) / 3)) {
642 /* wrap the line */
643 waddch(msg_win, '\n');
644 nspaces = 0;
645 }
646
647 /* output the word, preceded by spaces if necessary */
648 while (nspaces-- > 0)
649 waddch(msg_win, ' ');
650 waddbytes(msg_win, wstart, wordlen);
651
652 /* set up the 'last' state for the next time around */
653 switch (afterw[-1]) {
654 case '.':
655 case '?':
656 case '!':
657 last_o_was_punct = 1;
658 break;
659 default:
660 last_o_was_punct = 0;
661 break;
662 }
663 last_o_was_space = 0;
664
665 /* ... and do it all again! */
666 }
667
668 /* String ended with a newline. They really want a line break. */
669 if (last_i_was_nl) {
670 if (getcurx(msg_win) != 0)
671 waddch(msg_win, '\n');
672 last_o_was_punct = 0;
673 last_o_was_space = 1;
674 }
675
676out:
677 wrefresh(msg_win);
678 return ret;
679}

728static void __printflike(1, 0)
729_msg_vprompt(const char *fmt, int flags, const char *def, char *val,
730 size_t val_buf_len, va_list ap)
731{
732 int ch;
733 int len, pos, npos, off;
734 int first;
735 int txt_y, txt_x;
736 char *ibuf;
737 int maxx;
738
739 if (val == NULL || val_buf_len == 0) {
740 /* No answer wanted */
741 val = NULL;
742 val_buf_len = 1;
743 }
744
745 ibuf = malloc(val_buf_len);
746
747 keypad(msg_win, TRUE);
748 _msg_vprintf(0, fmt, ap);
749 ibuf[0] = 0;
750 if (def != NULL && *def) {
751 if (flags & MSG_PROMPT_HIDE_DFLT)
752 strlcpy(ibuf, def, val_buf_len);
753 else {
754 waddstr(msg_win, " [");
755 waddstr(msg_win, def);
756 waddstr(msg_win, "]");
757 }
758 }
759 waddstr(msg_win, ": ");
760 len = strlen(ibuf);
761 pos = len;
762 getyx(msg_win, txt_y, txt_x);
763 maxx = getmaxx(msg_win) - txt_x - 1;
764 off = 0;
765
766 for (first = 1; ; first = 0) {
767
768 if (flags & MSG_PROMPT_ECHO) {
769 /* shift text right as we near the buffer start */
770 if (pos - off < 4)
771 off = pos - 4;
772 /* keep offset to a minimum if we are at the end */
773 if (pos == len)
774 off = pos - maxx;
775 if (off < 0 || len <= maxx)
776 off = 0;
777 /* shift text left as we near the buffer end */
778 npos = pos + 4;
779 if (npos > len)
780 npos = len;
781 if (npos - off > maxx)
782 off = npos - maxx;
783 /* calc. length to display */
784 npos = len - off;
785 if (npos > maxx)
786 npos = maxx;
787 mvwaddnstr(msg_win, txt_y, txt_x, ibuf + off, npos);
788 wclrtoeol(msg_win);
789 if (off != 0)
790 mvwaddstr(msg_win, txt_y, txt_x, "+");
791 wmove(msg_win, txt_y, txt_x + pos - off);
792 wrefresh(msg_win);
793 }
794
795 ch = wgetch(msg_win);
796 if (ch == '\n')
797 break;
798
799 switch (ch) {
800 case KEY_BACKSPACE:
801 case 'h' & 0x1f: case 0x7f: /* bs or del - delete left */
802 if (first) {
803 /* delete all of default string */
804 len = pos = 0;
805 break;
806 }
807 if (pos > 0) {
808 memmove(ibuf + pos - 1, ibuf + pos, len - pos);
809 len--;
810 pos--;
811 } else
812 _msg_beep();
813 break;
814 case 'l' & 0x1f:
815 endwin();
816 doupdate();
817 break;
818 case 'u' & 0x1f: /* ^U; line kill */
819 /* kill line */
820 len = pos = 0;
821 break;
822 case 'w' & 0x1f: /* ^W; word kill */
823 /*
824 * word kill kills the spaces and the 'word'
825 * (non-spaces) last typed. the spaces before
826 * the 'word' aren't killed.
827 */
828 npos = pos;
829 while (npos > 0 && isspace((unsigned char)ibuf[npos - 1]))
830 npos--;
831 while (npos > 0 && !isspace((unsigned char)ibuf[npos - 1]))
832 npos--;
833 memmove(ibuf + npos, ibuf + pos, len - pos);
834 len -= pos - npos;
835 pos = npos;
836 break;
837 case KEY_LEFT:
838 if (pos > 0)
839 pos--;
840 break;
841 case KEY_RIGHT:
842 if (len == 0 && pos == 0 && def != NULL) {
843 /* restore default! */
844 strlcpy(ibuf, def, val_buf_len);
845 len = pos = strlen(ibuf);
846 break;
847 }
848 if (pos < len)
849 pos++;
850 break;
851 default:
852 if (len < (int)(val_buf_len - 1) && isprint(ch)) {
853 memmove(ibuf + pos + 1, ibuf + pos, len - pos);
854 ibuf[pos++] = ch;
855 len++;
856 } else
857 _msg_beep();
858 break;
859 }
860 }
861
862 if (flags & MSG_PROMPT_ECHO) {
863 mvwaddch(msg_win, txt_y, txt_x + len - off, '\n');
864 last_o_was_punct = 0;
865 last_o_was_space = 1;
866 }
867
868 if (val != NULL) {
869 /* copy the appropriate string to the output */
870 if (len != 0 || flags & MSG_PROMPT_HIDE_DFLT) {
871 ibuf[len] = '\0';
872 strlcpy(val, ibuf, val_buf_len);
873 } else if (def != NULL && val != def) {
874 strlcpy(val, def, val_buf_len);
875 }
876 }
877 free(ibuf);
878}

908void __printflike(9, 10)
909msg_fmt_prompt_win(msg msg_no, int x, int y, int w, int h,
910 const char *def, char *val, size_t val_buf_len, const char *fmt, ...)
911{
912 va_list ap;
913 WINDOW *win;
914 WINDOW *svmsg = NULL, *sv_win = NULL; /* XXX -Wuninitialized [many] */
915 int maxx, maxy;
916 int msg_flags = MSG_PROMPT_ECHO | MSG_PROMPT_HIDE_DFLT;
917 const char *np, *ep;
918
919 maxx = getmaxx(msg_win);
920 maxy = getmaxy(msg_win);
921 if (w == 0) {
922 va_start(ap, fmt);
923 w = vsnprintf(NULL, 0, fmtcheck(msg_string(msg_no), fmt), ap);
924 va_end(ap);
925 if (def != NULL && *def != 0 && w + (int)val_buf_len * 2 < maxx) {
926 w += 2 + strlen(def) + 1;
927 msg_flags &= ~MSG_PROMPT_HIDE_DFLT;
928 }
929 w += 1 + 2 + val_buf_len + 1;
930 if (w > maxx) {
931 if (!(msg_flags & MSG_PROMPT_HIDE_DFLT)) {
932 w -= 2 + strlen(def) + 1;
933 msg_flags |= MSG_PROMPT_HIDE_DFLT;
934 }
935 w = maxx;
936 }
937 } else if (w > 0 && def != NULL && *def != 0) {
938 size_t tl = strlen(def);
939 if (tl + 1 + 2 + val_buf_len + 1 < (unsigned)w)
940 msg_flags &= ~MSG_PROMPT_HIDE_DFLT;
941 }
942
943 if (x == -1)
944 x = (maxx - w) / 2 + 1;
945 if (h < 0) {
946 h = 3;
947 for (np = msg_string(msg_no); (ep = strchr(np, '\n'));
948 np = ep + 1)
949 h++;
950 }
951 if (h < 3)
952 h = 3;
953 if (y < 3)
954 y = (maxy - h) / 2;
955 if (y + h > maxy)
956 y = maxy - h;
957
958 win = subwin(msg_win, h, w, y, x);
959 if (win == NULL)
960 wprintw(msg_win, "msg_prompt_win: "
961 "newwin(%d, %d, %d, %d) failed\n",
962 h, w, y, x);
963 else {
964 /*
965 * Save screen contents from under our window
966 * Due to a mis-feature of NetBSD curses, curscr contains
967 * the data processed by doupdate() not that by wnoutrefresh().
968 * We must call doupdate() here to ensure we save the correct
969 * data. See PR 26660
970 */
971 doupdate();
972 sv_win = dupwin(win);
973 if (sv_win)
974 overwrite(curscr, sv_win);
975 wbkgd(win, getbkgd(msg_win));
976 wattrset(win, getattrs(msg_win));
977 box(win, 0, 0);
978 wrefresh(win);
979
980 /* Change message window to be our little box */
981 svmsg = msg_window(subwin(msg_win, h - 2, w - 2, y + 1, x + 1));
982 wbkgd(msg_win, getbkgd(win));
983 wattrset(msg_win, getattrs(win));
984
985 msg_clear();
986 }
987
988 va_start(ap, fmt);
989 _msg_vprompt(fmtcheck(msg_string(msg_no), fmt), msg_flags, def, val,
990 val_buf_len, ap);
991 va_end(ap);
992
993 if (win != NULL) {
994 wclear(win);
995 if (sv_win) {
996 /* Restore original screen contents */
997 overwrite(sv_win, win);
998 delwin(sv_win);
999 }
1000 wnoutrefresh(win);
1001 /* Restore normal message window */
1002 delwin(msg_window(svmsg));
1003 delwin(win);
1004 }
1005}
