37#include <errno.h>
38#include <stdio.h>
39#include <stdlib.h>
40#include <string.h>
41#include <unistd.h>
42
43int
44main(argc, argv)
45 int argc;
46 char *argv[];
47{
48 if (argc != 2) {
49 fprintf(stderr, "usage: %s <progname>\n", argv[0]);
50 exit(2);
51 }
52
53 unsetenv("LANG"); /* we compare C error strings */
54 if (execve(argv[1], &argv[1], NULL) == -1) {
55 printf("%s\n", strerror(errno));
56 exit(1);
57 }
58}
