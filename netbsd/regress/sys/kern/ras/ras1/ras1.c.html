29#include <stdlib.h>
30#include <errno.h>
31#include <inttypes.h>
32#include <stdio.h>
33#include <signal.h>
34#include <sys/ras.h>
35#include <sys/time.h>
36
37#define COUNT 10
38
39volatile int handled = 0;
40volatile int count = 0;
41struct itimerval itv;
42
43void handler(int);
44
45void
46handler(int sig)
47{
48
49 handled++;
50}
51
52RAS_DECL(main);
53
54int
55main(void)
56{
57
58 signal(SIGVTALRM, handler);
59
60 itv.it_interval.tv_sec = 0;
61 itv.it_interval.tv_usec = 0;
62 itv.it_value.tv_sec = 10;
63 itv.it_value.tv_usec = 0;
64 setitimer(ITIMER_VIRTUAL, &itv, NULL);
65
66 if (rasctl(RAS_ADDR(main), RAS_SIZE(main), RAS_INSTALL) < 0) {
67 if (errno == EOPNOTSUPP) {
68 printf("RAS is not supported on this architecture\n");
69 return 0;
70 }
71 return (1);
72 }
73
74 RAS_START(main);
75 count++;
76 if (count > COUNT)
77 exit(handled != 0);
78
79 while (!handled) {
80 continue;
81 }
82 RAS_END(main);
83
84 return (handled != 0);
85}
