29#include <stdlib.h>
30#include <errno.h>
31#include <inttypes.h>
32#include <stdio.h>
33#include <signal.h>
34#include <unistd.h>
35#include <sys/ras.h>
36#include <sys/time.h>
37#include <sys/wait.h>
38
39#define COUNT 10
40
41volatile int handled = 0;
42volatile int count = 0;
43struct itimerval itv;
44
45void handler(int);
46
47void
48handler(int sig)
49{
50
51 handled++;
52}
53
54RAS_DECL(main);
55
56int
57main(int argc, char *argv[])
58{
59 int rv;
60 char *const args[] = { argv[0], "1", NULL };
61
62 signal(SIGVTALRM, handler);
63
64 itv.it_interval.tv_sec = 0;
65 itv.it_interval.tv_usec = 0;
66 itv.it_value.tv_sec = 10;
67 itv.it_value.tv_usec = 0;
68 setitimer(ITIMER_VIRTUAL, &itv, NULL);
69
70 if (argc != 2) {
71 if (rasctl(RAS_ADDR(main), RAS_SIZE(main), RAS_INSTALL) < 0) {
72 if (errno == EOPNOTSUPP) {
73 printf("RAS is not supported on this "
74 "architecture\n");
75 return 0;
76 }
77 return (1);
78 }
79 if (fork() != 0) {
80 wait(&rv);
81 return (WEXITSTATUS(rv) == 0);
82 }
83 if (execvp(argv[0],args) < 0) {
84 printf("exec failed\n");
85 return (0);
86 }
87 }
88
89 RAS_START(main);
90 count++;
91 if (count > COUNT)
92 exit(handled != 0);
93
94 while (!handled) {
95 continue;
96 }
97 RAS_END(main);
98
99 return (handled != 0);
100}
