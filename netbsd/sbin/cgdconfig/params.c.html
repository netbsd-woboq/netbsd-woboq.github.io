116struct params *
117params_combine(struct params *p1, struct params *p2)
118{
119 struct params *p;
120
121 if (p1)
122 p = p1;
123 else
124 p = params_new();
125
126 if (!p2)
127 return p;
128
129 if (p2->algorithm)
130 string_assign(&p->algorithm, p2->algorithm);
131 if (p2->ivmeth)
132 string_assign(&p->ivmeth, p2->ivmeth);
133 if (p2->keylen != (size_t)-1)
134 p->keylen = p2->keylen;
135 if (p2->bsize != (size_t)-1)
136 p->bsize = p2->bsize;
137 if (p2->verify_method != VERIFY_UNKNOWN)
138 p->verify_method = p2->verify_method;
139
140 p->dep_keygen = keygen_combine(p->dep_keygen, p2->dep_keygen);
141 keygen_addlist(&p->keygen, p2->keygen);
142
143 /*
144 * at this point we should have moved all allocated data
145 * in p2 into p, so we can free it.
146 */
147 free(p2);
148 return p;
149}

151int
152params_filldefaults(struct params *p)
153{
154 size_t i;
155
156 if (p->verify_method == VERIFY_UNKNOWN)
157 p->verify_method = VERIFY_NONE;
158 if (!p->ivmeth)
159 p->ivmeth = string_fromcharstar("encblkno1");
160 if (p->keylen == (size_t)-1) {
161 if (p->algorithm == NULL)
162 return -1;
163 i = crypt_defaults_lookup(string_tocharstar(p->algorithm));
164 if (i != (size_t)-1) {
165 p->keylen = crypto_defaults[i].keylen;
166 } else {
167 warnx("could not determine key length for unknown "
168 "algorithm \"%s\"",
169 string_tocharstar(p->algorithm));
170 return -1;
171 }
172 }
173 return 0;
174}

176/*
177 * params_verify traverses the parameters and all of the keygen methods
178 * looking for inconsistencies. It outputs warnings on non-fatal errors
179 * such as unknown encryption methods, but returns failure on fatal
180 * conditions such as a PKCS5_PBKDF2 keygen without a salt. It is intended
181 * to run before key generation.
182 */
183
184int
185params_verify(const struct params *p)
186{
187 static const char *encblkno[] = {
188 "encblkno", "encblkno1", "encblkno8"
189 };
190 static size_t i;
191 const char *meth;
192
193 if (!p->algorithm) {
194 warnx("unspecified algorithm");
195 return 0;
196 }
197 /*
198 * we only warn for the encryption method so that it is possible
199 * to use an older cgdconfig(8) with a new kernel that supports
200 * additional crypto algorithms.
201 */
202 if (crypt_defaults_lookup(string_tocharstar(p->algorithm)) == -1)
203 warnx("unknown algorithm \"%s\"(warning)",
204 string_tocharstar(p->algorithm));
205 /* same rationale with IV methods. */
206 if (!p->ivmeth) {
207 warnx("unspecified IV method");
208 return 0;
209 }
210
211 meth = string_tocharstar(p->ivmeth);
212 for (i = 0; i < __arraycount(encblkno); i++)
213 if (strcmp(encblkno[i], meth) == 0)
214 break;
215
216 if (i == __arraycount(encblkno))
217 warnx("unknown IV method \"%s\" (warning)", meth);
218
219 if (p->keylen == (size_t)-1) {
220 warnx("unspecified key length");
221 return 0;
222 }
223
224 return keygen_verify(p->keygen);
225}

263struct params *
264params_verify_method(string_t *in)
265{
266 struct params *p = params_new();
267 const char *vm = string_tocharstar(in);
268
269 if (!strcmp("none", vm))
270 p->verify_method = VERIFY_NONE;
271 if (!strcmp("disklabel", vm))
272 p->verify_method = VERIFY_DISKLABEL;
273 if (!strcmp("ffs", vm))
274 p->verify_method = VERIFY_FFS;
275 if (!strcmp("re-enter", vm))
276 p->verify_method = VERIFY_REENTER;
277 if (!strcmp("mbr", vm))
278 p->verify_method = VERIFY_MBR;
279 if (!strcmp("gpt", vm))
280 p->verify_method = VERIFY_GPT;
281
282 string_free(in);
283
284 if (p->verify_method == VERIFY_UNKNOWN)
285 warnx("params_setverify_method: unrecognized "
286 "verify method \"%s\"", vm);
287 return p;
288}

336/*
337 * keygen_verify traverses the keygen structures and ensures
338 * that the appropriate information is available.
339 */
340
341int
342keygen_verify(const struct keygen *kg)
343{
344
345 if (!kg)
346 return 1;
347 switch (kg->kg_method) {
348 case KEYGEN_PKCS5_PBKDF2_OLD:
349 if (kg->kg_iterations == (size_t)-1) {
350 warnx("keygen pkcs5_pbkdf2 must provide `iterations'");
351 return 0;
352 }
353 if (kg->kg_key)
354 warnx("keygen pkcs5_pbkdf2 does not need a `key'");
355 if (!kg->kg_salt) {
356 warnx("keygen pkcs5_pbkdf2 must provide a salt");
357 return 0;
358 }
359 if (kg->kg_cmd)
360 warnx("keygen pkcs5_pbkdf2 does not need a `cmd'");
361 break;
362 case KEYGEN_PKCS5_PBKDF2_SHA1:
363 if (kg->kg_iterations == (size_t)-1) {
364 warnx("keygen pkcs5_pbkdf2/sha1 must provide `iterations'");
365 return 0;
366 }
367 if (kg->kg_key)
368 warnx("keygen pkcs5_pbkdf2/sha1 does not need a `key'");
369 if (!kg->kg_salt) {
370 warnx("keygen pkcs5_pbkdf2/sha1 must provide a salt");
371 return 0;
372 }
373 if (kg->kg_cmd)
374 warnx("keygen pkcs5_pbkdf2/sha1 does not need a `cmd'");
375 break;
376 case KEYGEN_STOREDKEY:
377 if (kg->kg_iterations != (size_t)-1)
378 warnx("keygen storedkey does not need `iterations'");
379 if (!kg->kg_key) {
380 warnx("keygen storedkey must provide a key");
381 return 0;
382 }
383 if (kg->kg_salt)
384 warnx("keygen storedkey does not need `salt'");
385 if (kg->kg_cmd)
386 warnx("keygen storedkey does not need `cmd'");
387 break;
388 case KEYGEN_RANDOMKEY:
389 case KEYGEN_URANDOMKEY:
390 if (kg->kg_iterations != (size_t)-1)
391 warnx("keygen [u]randomkey does not need `iterations'");
392 if (kg->kg_key)
393 warnx("keygen [u]randomkey does not need `key'");
394 if (kg->kg_salt)
395 warnx("keygen [u]randomkey does not need `salt'");
396 if (kg->kg_cmd)
397 warnx("keygen [u]randomkey does not need `cmd'");
398 break;
399 case KEYGEN_SHELL_CMD:
400 if (kg->kg_iterations != (size_t)-1)
401 warnx("keygen shell_cmd does not need `iterations'");
402 if (kg->kg_key)
403 warnx("keygen shell_cmd does not need `key'");
404 if (kg->kg_salt)
405 warnx("keygen shell_cmd does not need `salt'");
406 if (!kg->kg_cmd) {
407 warnx("keygen shell_cmd must provide a `cmd'");
408 return 0;
409 }
410 break;
411 }
412 return keygen_verify(kg->next);
413}

428/*
429 * keygen_filldefaults walks the keygen list and fills in
430 * default values. The defaults may be either calibrated
431 * or randomly generated so this function is designed to be
432 * called when generating a new parameters file, not when
433 * reading a parameters file.
434 */
435
436int
437keygen_filldefaults(struct keygen *kg, size_t keylen)
438{
439
440 if (!kg)
441 return 0;
442 switch (kg->kg_method) {
443 case KEYGEN_RANDOMKEY:
444 case KEYGEN_URANDOMKEY:
445 case KEYGEN_SHELL_CMD:
446 break;
447 case KEYGEN_PKCS5_PBKDF2_OLD:
448 case KEYGEN_PKCS5_PBKDF2_SHA1:
449 kg->kg_salt = bits_getrandombits(DEFAULT_SALTLEN, 1);
450 kg->kg_iterations = pkcs5_pbkdf2_calibrate(BITS2BYTES(keylen),
451 DEFAULT_ITERATION_TIME);
452 if (kg->kg_iterations < 1) {
453 warnx("could not calibrate pkcs5_pbkdf2");
454 return -1;
455 }
456 break;
457 case KEYGEN_STOREDKEY:
458 /* Generate a random stored key */
459 kg->kg_key = bits_getrandombits(keylen, 1);
460 if (!kg->kg_key) {
461 warnx("can't generate random bits for storedkey");
462 return -1;
463 }
464 break;
465 default:
466 return -1;
467 }
468
469 return keygen_filldefaults(kg->next, keylen);
470}

472struct keygen *
473keygen_combine(struct keygen *kg1, struct keygen *kg2)
474{
475 if (!kg1 && !kg2)
476 return NULL;
477
478 if (!kg1)
479 kg1 = keygen_new();
480
481 if (!kg2)
482 return kg1;
483
484 if (kg2->kg_method != KEYGEN_UNKNOWN)
485 kg1->kg_method = kg2->kg_method;
486
487 if (kg2->kg_iterations != (size_t)-1 && kg2->kg_iterations > 0)
488 kg1->kg_iterations = kg2->kg_iterations;
489
490 if (kg2->kg_salt)
491 bits_assign(&kg1->kg_salt, kg2->kg_salt);
492
493 if (kg2->kg_key)
494 bits_assign(&kg1->kg_key, kg2->kg_key);
495
496 if (kg2->kg_cmd)
497 string_assign(&kg1->kg_cmd, kg2->kg_cmd);
498
499 return kg1;
500}

502struct keygen *
503keygen_method(string_t *in)
504{
505 struct keygen *kg = keygen_new();
506 const char *kgm = string_tocharstar(in);
507
508 if (!strcmp("pkcs5_pbkdf2", kgm))
509 kg->kg_method = KEYGEN_PKCS5_PBKDF2_OLD;
510 if (!strcmp("pkcs5_pbkdf2/sha1", kgm))
511 kg->kg_method = KEYGEN_PKCS5_PBKDF2_SHA1;
512 if (!strcmp("randomkey", kgm))
513 kg->kg_method = KEYGEN_RANDOMKEY;
514 if (!strcmp("storedkey", kgm))
515 kg->kg_method = KEYGEN_STOREDKEY;
516 if (!strcmp("urandomkey", kgm))
517 kg->kg_method = KEYGEN_URANDOMKEY;
518 if (!strcmp("shell_cmd", kgm))
519 kg->kg_method = KEYGEN_SHELL_CMD;
520
521 string_free(in);
522
523 if (kg->kg_method == KEYGEN_UNKNOWN)
524 warnx("unrecognized key generation method \"%s\"", kgm);
525 return kg;
526}

586struct params *
587params_fget(FILE *f)
588{
589 struct params *p;
590
591 p = cgdparsefile(f);
592
593 if (!p)
594 return NULL;
595
596 /*
597 * We deal with the deprecated keygen structure by prepending it
598 * to the list of keygens, so that the rest of the code does not
599 * have to deal with this backwards compat issue. The deprecated
600 * ``xor_key'' field may be stored in p->dep_keygen->kg_key. If
601 * it exists, we construct a storedkey keygen struct as well. Also,
602 * default the iteration count to 128 as the old code did.
603 */
604
605 if (p->dep_keygen) {
606 if (p->dep_keygen->kg_iterations == (size_t)-1)
607 p->dep_keygen->kg_iterations = 128;
608 p->dep_keygen->next = p->keygen;
609 if (p->dep_keygen->kg_key) {
610 p->keygen = keygen_generate(KEYGEN_STOREDKEY);
611 p->keygen->kg_key = p->dep_keygen->kg_key;
612 p->dep_keygen->kg_key = NULL;
613 p->keygen->next = p->dep_keygen;
614 } else {
615 p->keygen = p->dep_keygen;
616 }
617 p->dep_keygen = NULL;
618 }
619 return p;
620}

622struct params *
623params_cget(const char *fn)
624{
625 struct params *p;
626 FILE *f;
627 char filename[MAXPATHLEN];
628
629 if ((f = fopen(fn, "r")) == NULL && fn[0] != '/') {
630 snprintf(filename, sizeof(filename), "%s/%s",
631 CGDCONFIG_DIR, fn);
632 fn = filename;
633 f = fopen(fn, "r");
634 }
635
636 if (f == NULL) {
637 warn("failed to open params file \"%s\"", fn);
638 return NULL;
639 }
640 p = params_fget(f);
641 (void)fclose(f);
642 return p;
643}

645#define WRAP_COL 50
646#define TAB_COL 8

648static void
649spaces(FILE *f, int len)
650{
651
652 while (len-- > 0)
653 (void)fputc(' ', f);
654}

656static void
657print_kvpair_cstr(FILE *f, int ts, const char *key, const char *val)
658{
659
660 spaces(f, ts);
661 (void)fprintf(f, "%s %s;\n", key, val);
662}

664static void
665print_kvpair_string(FILE *f, int ts, const char *key, const string_t *val)
666{
667
668 print_kvpair_cstr(f, ts, key, string_tocharstar(val));
669}

671static void
672print_kvpair_int(FILE *f, int ts, const char *key, size_t val)
673{
674 char *tmp;
675
676 if (!key || val == (size_t)-1)
677 return;
678
679 if (asprintf(&tmp, "%zu", val) == -1)
680 err(1, NULL);
681 print_kvpair_cstr(f, ts, key, tmp);
682 free(tmp);
683}

685/*
686 * prints out a base64 encoded k-v pair to f. It encodes the length
687 * of the bitstream as a 32bit unsigned integer in network byte order
688 * up front.
689 */
690
691static void
692print_kvpair_b64(FILE *f, int curpos, int ts, const char *key, bits_t *val)
693{
694 string_t *str;
695 int i;
696 int len;
697 int pos;
698 const char *out;
699
700 if (!key || !val)
701 return;
702
703 str = bits_encode(val);
704 out = string_tocharstar(str);
705 len = strlen(out);
706
707 spaces(f, ts);
708 (void)fprintf(f, "%s ", key);
709 curpos += ts + strlen(key) + 1;
710 ts = curpos;
711
712 for (i=0, pos=curpos; i < len; i++, pos++) {
713 if (pos > WRAP_COL) {
714 (void)fprintf(f, " \\\n");
715 spaces(f, ts);
716 pos = ts;
717 }
718 (void)fputc(out[i], f);
719 }
720 (void)fprintf(f, ";\n");
721 string_free(str);
722}

724int
725keygen_fput(struct keygen *kg, int ts, FILE *f)
726{
727 int curpos = 0;
728
729 if (!kg)
730 return 0;
731 (void)fprintf(f, "keygen ");
732 curpos += strlen("keygen ");
733 switch (kg->kg_method) {
734 case KEYGEN_STOREDKEY:
735 (void)fprintf(f, "storedkey ");
736 curpos += strlen("storedkey ");
737 print_kvpair_b64(f, curpos, 0, "key", kg->kg_key);
738 break;
739 case KEYGEN_RANDOMKEY:
740 (void)fprintf(f, "randomkey;\n");
741 break;
742 case KEYGEN_URANDOMKEY:
743 (void)fprintf(f, "urandomkey;\n");
744 break;
745 case KEYGEN_PKCS5_PBKDF2_OLD:
746 (void)fprintf(f, "pkcs5_pbkdf2 {\n");
747 print_kvpair_int(f, ts, "iterations", kg->kg_iterations);
748 print_kvpair_b64(f, 0, ts, "salt", kg->kg_salt);
749 (void)fprintf(f, "};\n");
750 break;
751 case KEYGEN_PKCS5_PBKDF2_SHA1:
752 (void)fprintf(f, "pkcs5_pbkdf2/sha1 {\n");
753 print_kvpair_int(f, ts, "iterations", kg->kg_iterations);
754 print_kvpair_b64(f, 0, ts, "salt", kg->kg_salt);
755 (void)fprintf(f, "};\n");
756 break;
757 default:
758 warnx("keygen_fput: %d not a valid method", kg->kg_method);
759 break;
760 }
761 return keygen_fput(kg->next, ts, f);
762}

764int
765params_fput(struct params *p, FILE *f)
766{
767 int ts = 0; /* tabstop of 0 spaces */
768
769 print_kvpair_string(f, ts, "algorithm", p->algorithm);
770 print_kvpair_string(f, ts, "iv-method", p->ivmeth);
771 print_kvpair_int(f, ts, "keylength", p->keylen);
772 print_kvpair_int(f, ts, "blocksize", p->bsize);
773 switch (p->verify_method) {
774 case VERIFY_NONE:
775 print_kvpair_cstr(f, ts, "verify_method", "none");
776 break;
777 case VERIFY_DISKLABEL:
778 print_kvpair_cstr(f, ts, "verify_method", "disklabel");
779 break;
780 case VERIFY_FFS:
781 print_kvpair_cstr(f, ts, "verify_method", "ffs");
782 break;
783 case VERIFY_REENTER:
784 print_kvpair_cstr(f, ts, "verify_method", "re-enter");
785 break;
786 case VERIFY_MBR:
787 print_kvpair_cstr(f, ts, "verify_method", "mbr");
788 break;
789 case VERIFY_GPT:
790 print_kvpair_cstr(f, ts, "verify_method", "gpt");
791 break;
792 default:
793 warnx("unsupported verify_method (%d)", p->verify_method);
794 return -1;
795 }
796 return keygen_fput(p->keygen, TAB_COL, f);
797}
