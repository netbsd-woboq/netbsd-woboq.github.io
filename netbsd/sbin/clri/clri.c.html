65/*
66 * Possible superblock locations ordered from most to least likely.
67 */
68static off_t sblock_try[] = SBLOCKSEARCH;
69off_t sblockloc;
70
71
72int
73main(int argc, char *argv[])
74{
75 struct fs *sbp;
76 struct ufs1_dinode *ip1;
77 struct ufs2_dinode *ip2;
78 int fd;
79 void *ibuf;
80 int32_t generation;
81 off_t offset;
82 size_t bsize;
83 int inonum;
84 char *fs, sblock[SBLOCKSIZE];
85 int needswap = 0, is_ufs2 = 0;
86 int i, imax;
87 long dev_bsize;
88
89 if (argc < 3) {
90 (void)fprintf(stderr, "usage: clri filesystem inode ...\n");
91 exit(1);
92 }
93
94 fs = *++argv;
95
96
97 /* get the superblock. */
98 if ((fd = open(fs, O_RDWR, 0)) < 0)
99 err(1, "%s", fs);
100 for (i = 0;; i++) {
101 sblockloc = sblock_try[i];
102 if (sblockloc == -1)
103 errx(1, "%s: can't find superblock", fs);
104 if (pread(fd, sblock, sizeof(sblock), sblockloc) != sizeof(sblock))
105 errx(1, "%s: can't read superblock", fs);
106
107 sbp = (struct fs *)sblock;
108 switch(sbp->fs_magic) {
109 case FS_UFS2_MAGIC:
110 is_ufs2 = 1;
111 /*FALLTHROUGH*/
112 case FS_UFS1_MAGIC:
113 break;
114 case FS_UFS2_MAGIC_SWAPPED:
115 is_ufs2 = 1;
116 /*FALLTHROUGH*/
117 case FS_UFS1_MAGIC_SWAPPED:
118 needswap = 1;
119 break;
120 default:
121 continue;
122 }
123
124 /* check we haven't found an alternate */
125 if (is_ufs2 || sbp->fs_old_flags & FS_FLAGS_UPDATED) {
126 if ((uint64_t)sblockloc != ufs_rw64(sbp->fs_sblockloc, needswap))
127 continue;
128 } else {
129 if (sblockloc == SBLOCK_UFS2)
130 continue;
131 }
132
133 break;
134 }
135
136 /* check that inode numbers are valid */
137 imax = ufs_rw32(sbp->fs_ncg, needswap) *
138 ufs_rw32(sbp->fs_ipg, needswap);
139 for (i = 1; i < (argc - 1); i++)
140 if (atoi(argv[i]) <= 0 || atoi(argv[i]) >= imax)
141 errx(1, "%s is not a valid inode number", argv[i]);
142
143 /* delete clean flag in the superblok */
144 sbp->fs_clean = ufs_rw32(ufs_rw32(sbp->fs_clean, needswap) << 1,
145 needswap);
146 if (lseek(fd, sblockloc, SEEK_SET) < 0)
147 err(1, "%s", fs);
148 if (write(fd, sblock, sizeof(sblock)) != sizeof(sblock))
149 errx(1, "%s: can't rewrite superblock", fs);
150 (void)fsync(fd);
151
152 if (needswap)
153 ffs_sb_swap(sbp, sbp);
154
155 /* compute disk block size from superblock parameters */
156 dev_bsize = sbp->fs_fsize / FFS_FSBTODB(sbp, 1);
157
158 bsize = sbp->fs_bsize;
159 ibuf = malloc(bsize);
160 if (ibuf == NULL) {
161 err(1, "malloc");
162 }
163
164 /* remaining arguments are inode numbers. */
165 while (*++argv) {
166 /* get the inode number. */
167 inonum = atoi(*argv);
168 (void)printf("clearing %d\n", inonum);
169
170 /* read in the appropriate block. */
171 offset = ino_to_fsba(sbp, inonum); /* inode to fs blk */
172 offset = FFS_FSBTODB(sbp, offset); /* fs blk disk blk */
173 offset *= dev_bsize; /* disk blk to bytes */
174
175 /* seek and read the block */
176 if (lseek(fd, offset, SEEK_SET) < 0)
177 err(1, "%s", fs);
178 if ((size_t)read(fd, ibuf, bsize) != bsize)
179 err(1, "%s", fs);
180
181 /* get the inode within the block. */
182 if (is_ufs2) {
183 ip2 = &((struct ufs2_dinode *)ibuf)
184 [ino_to_fsbo(sbp, inonum)];
185 /* clear the inode, and bump the generation count. */
186 generation = ip2->di_gen + 1;
187 memset(ip2, 0, sizeof(*ip2));
188 ip2->di_gen = generation;
189 } else {
190 ip1 = &((struct ufs1_dinode *)ibuf)
191 [ino_to_fsbo(sbp, inonum)];
192 /* clear the inode, and bump the generation count. */
193 generation = ip1->di_gen + 1;
194 memset(ip1, 0, sizeof(*ip1));
195 ip1->di_gen = generation;
196 }
197
198 /* backup and write the block */
199 if (lseek(fd, offset, SEEK_SET) < 0)
200 err(1, "%s", fs);
201 if ((size_t)write(fd, ibuf, bsize) != bsize)
202 err(1, "%s", fs);
203 (void)fsync(fd);
204 }
205 free(ibuf);
206 (void)close(fd);
207 exit(0);
208}
