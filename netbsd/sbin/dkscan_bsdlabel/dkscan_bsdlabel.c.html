32#include <stdio.h>
33#include <stdlib.h>
34#include <stddef.h>
35#include <string.h>
36#include <unistd.h>
37#include <fcntl.h>
38#include <err.h>
39#include <util.h>
40#include <sys/disk.h>
41
42#include "dkscan_util.h"
43
44struct disk {
45 const char *dk_name; /* disk name */
46 int dk_blkshift; /* shift to convert DEV_BSIZE to blks */
47};
48
49#include "dkwedge_bsdlabel.c"
50
51__dead static void usage(void);
52
53int
54main(int argc, char **argv)
55{
56 struct disk d;
57 int ch;
58 char buf[PATH_MAX];
59 const char *devpart;
60
61 if (argc < 2)
62 usage();
63
64 while ((ch = getopt(argc, argv, "nv")) != -1) {
65 switch (ch) {
66 case 'n':
67 no_action = 1;
68 break;
69 case 'v':
70 verbose++;
71 break;
72 default:
73 usage();
74 }
75 }
76 if (optind >= argc)
77 usage();
78
79 disk_fd = opendisk(argv[optind], O_RDWR, buf, PATH_MAX, 0);
80 if (disk_fd == -1)
81 err(1, "%s", argv[optind]);
82
83 devpart = strrchr(argv[optind], '/');
84 if (devpart == NULL)
85 devpart = argv[optind];
86
87 memset(&d, 0, sizeof(d));
88 d.dk_name = devpart;
89 dkwedge_discover_bsdlabel(&d, NULL);
90
91 close(disk_fd);
92 return 0;
93}
94
95void
96usage(void)
97{
98 fprintf(stderr, "usage: %s [-vn] <diskname>\n"
99 " where\n"
100 " -n don't change anything, just print info\n"
101 " -v be more verbose\n"
102 " <diskname> device to scan\n",
103 getprogname());
104 exit(1);
105}
