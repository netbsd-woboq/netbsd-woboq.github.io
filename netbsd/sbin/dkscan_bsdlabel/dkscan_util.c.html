46int verbose = 0; /* are we verbose? */
47int no_action = 0; /* don't do anything, just print info */
48int disk_fd = -1; /* file descriptor for disk access */
49
50
51u_int
52dkcksum(struct disklabel *lp)
53{
54 return dkcksum_sized(lp, lp->d_npartitions);
55}
56
57u_int
58dkcksum_sized(struct disklabel *lp, size_t npartitions)
59{
60 u_short *start, *end;
61 u_short sum = 0;
62
63 start = (u_short *)lp;
64 end = (u_short *)&lp->d_partitions[npartitions];
65 while (start < end)
66 sum ^= *start++;
67 return (sum);
68}

70int
71dkwedge_read(struct disk *pdk, struct vnode *vp, daddr_t blkno,
72 void *tbuf, size_t len)
73{
74 if (pread(disk_fd, tbuf, len, blkno * BLOCK_SIZE) < 0)
75 return errno;
76 return 0;
77}
78
79int
80dkwedge_add(struct dkwedge_info *dkw)
81{
82 printf("wedge \"%s\" type \"%s\" start %" PRIu64 " size %" PRIu64,
83 dkw->dkw_wname, dkw->dkw_ptype, dkw->dkw_offset, dkw->dkw_size);
84
85 if (no_action) {
86 printf("\n");
87 return 0;
88 }
89
90 if (ioctl(disk_fd, DIOCAWEDGE, dkw) == -1)
91 err(1, "addwedge");
92
93 printf(": %s\n", dkw->dkw_devname);
94 return 0;
95}

97void
98aprint_error(const char *format, ...)
99{
100 va_list ap;
101
102 va_start(ap, format);
103 vfprintf(stderr, format, ap);
104 va_end(ap);
105}
106
107void
108aprint_verbose(const char *format, ...)
109{
110 va_list ap;
111
112 if (!verbose) return;
113
114 va_start(ap, format);
115 vfprintf(stdout, format, ap);
116 va_end(ap);
117}
