69/*
70 * Read the superblock from disk, and check its magic number.
71 * Determine whether byte-swapping needs to be done on this file system.
72 */
73int
74fs_read_sblock(char *superblock)
75{
76 int i;
77 int ns = 0;
78
79 sblock = (struct fs *)superblock;
80 for (i = 0; ; i++) {
81 if (sblock_try[i] == -1)
82 quit("can't find superblock");
83 rawread(sblock_try[i], (char *)superblock, MAXBSIZE);
84
85 switch(sblock->fs_magic) {
86 case FS_UFS2_MAGIC:
87 is_ufs2 = 1;
88 /*FALLTHROUGH*/
89 case FS_UFS1_MAGIC:
90 break;
91 case FS_UFS2_MAGIC_SWAPPED:
92 is_ufs2 = 1;
93 /*FALLTHROUGH*/
94 case FS_UFS1_MAGIC_SWAPPED:
95 ns = 1;
96 ffs_sb_swap(sblock, sblock);
97 break;
98 default:
99 continue;
100 }
101 if (!is_ufs2 && sblock_try[i] == SBLOCK_UFS2)
102 continue;
103 if ((is_ufs2 || sblock->fs_old_flags & FS_FLAGS_UPDATED)
104 && sblock_try[i] != sblock->fs_sblockloc)
105 continue;
106 break;
107 }
108 return ns;
109}

111/*
112 * Fill in the ufsi struct, as well as the maxino and dev_bsize global
113 * variables.
114 */
115struct ufsi *
116fs_parametrize(void)
117{
118 static struct ufsi ufsi;
119
120#ifdef FS_44INODEFMT
121 if (is_ufs2 || sblock->fs_old_inodefmt >= FS_44INODEFMT) {
122 spcl.c_flags = iswap32(iswap32(spcl.c_flags) | DR_NEWINODEFMT);
123 } else {
124 /*
125 * Determine parameters for older file systems. From
126 * /sys/ufs/ffs/ffs_vfsops.c::ffs_oldfscompat()
127 *
128 * XXX: not sure if other variables (fs_npsect, fs_interleave,
129 * fs_nrpos, fs_maxfilesize) need to be fudged too.
130 */
131 sblock->fs_qbmask = ~sblock->fs_bmask;
132 sblock->fs_qfmask = ~sblock->fs_fmask;
133 }
134#endif
135
136 /* Fill out ufsi struct */
137 ufsi.ufs_dsize = FFS_FSBTODB(sblock,sblock->fs_size);
138 ufsi.ufs_bsize = sblock->fs_bsize;
139 ufsi.ufs_bshift = sblock->fs_bshift;
140 ufsi.ufs_fsize = sblock->fs_fsize;
141 ufsi.ufs_frag = sblock->fs_frag;
142 ufsi.ufs_fsatoda = sblock->fs_fsbtodb;
143 ufsi.ufs_nindir = sblock->fs_nindir;
144 ufsi.ufs_inopb = sblock->fs_inopb;
145 ufsi.ufs_maxsymlinklen = sblock->fs_maxsymlinklen;
146 ufsi.ufs_bmask = sblock->fs_bmask;
147 ufsi.ufs_fmask = sblock->fs_fmask;
148 ufsi.ufs_qbmask = sblock->fs_qbmask;
149 ufsi.ufs_qfmask = sblock->fs_qfmask;
150
151 dev_bsize = sblock->fs_fsize / FFS_FSBTODB(sblock, 1);
152
153 return &ufsi;
154}

156ino_t
157fs_maxino(void)
158{
159
160 return sblock->fs_ipg * sblock->fs_ncg;

163void
164fs_mapinodes(ino_t maxino __unused, u_int64_t *tape_size, int *anydirskipped)
165{
166 int i, cg, inosused;
167 struct cg *cgp;
168 ino_t ino;
169 char *cp;
170
171 if ((cgp = malloc(sblock->fs_cgsize)) == NULL)
172 quite(errno, "fs_mapinodes: cannot allocate memory.");
173
174 for (cg = 0; cg < sblock->fs_ncg; cg++) {
175 ino = cg * sblock->fs_ipg;
176 bread(FFS_FSBTODB(sblock, cgtod(sblock, cg)), (char *)cgp,
177 sblock->fs_cgsize);
178 if (needswap)
179 ffs_cg_swap(cgp, cgp, sblock);
180 if (sblock->fs_magic == FS_UFS2_MAGIC)
181 inosused = cgp->cg_initediblk;
182 else
183 inosused = sblock->fs_ipg;
184 /*
185 * If we are using soft updates, then we can trust the
186 * cylinder group inode allocation maps to tell us which
187 * inodes are allocated. We will scan the used inode map
188 * to find the inodes that are really in use, and then
189 * read only those inodes in from disk.
190 */
191 if (sblock->fs_flags & FS_DOSOFTDEP) {
192 if (!cg_chkmagic(cgp, 0))
193 quit("%s: cg %d: bad magic number\n",
194 __func__, cg);
195 cp = &cg_inosused(cgp, 0)[(inosused - 1) / CHAR_BIT];
196 for ( ; inosused > 0; inosused -= CHAR_BIT, cp--) {
197 if (*cp == 0)
198 continue;
199 for (i = 1 << (CHAR_BIT - 1); i > 0; i >>= 1) {
200 if (*cp & i)
201 break;
202 inosused--;
203 }
204 break;
205 }
206 if (inosused <= 0)
207 continue;
208 }
209 for (i = 0; i < inosused; i++, ino++) {
210 if (ino < UFS_ROOTINO)
211 continue;
212 mapfileino(ino, tape_size, anydirskipped);
213 }
214 }
215
216 free(cgp);
217}

219union dinode *
220getino(ino_t inum)
221{
222 static ino_t minino, maxino;
223 static caddr_t inoblock;
224 int ntoswap, i;
225 struct ufs1_dinode *dp1;
226 struct ufs2_dinode *dp2;
227
228 if (inoblock == NULL && (inoblock = malloc(ufsib->ufs_bsize)) == NULL)
229 quite(errno, "cannot allocate inode memory.");
230 curino = inum;
231 if (inum >= minino && inum < maxino)
232 goto gotit;
233 bread(fsatoda(ufsib, ino_to_fsba(sblock, inum)), (char *)inoblock,
234 (int)ufsib->ufs_bsize);
235 minino = inum - (inum % FFS_INOPB(sblock));
236 maxino = minino + FFS_INOPB(sblock);
237 if (needswap) {
238 if (is_ufs2) {
239 dp2 = (struct ufs2_dinode *)inoblock;
240 ntoswap = ufsib->ufs_bsize / sizeof(struct ufs2_dinode);
241 for (i = 0; i < ntoswap; i++)
242 ffs_dinode2_swap(&dp2[i], &dp2[i]);
243 } else {
244 dp1 = (struct ufs1_dinode *)inoblock;
245 ntoswap = ufsib->ufs_bsize / sizeof(struct ufs1_dinode);
246 for (i = 0; i < ntoswap; i++)
247 ffs_dinode1_swap(&dp1[i], &dp1[i]);
248 }
249 }
250gotit:
251 if (is_ufs2) {
252 dp2 = &((struct ufs2_dinode *)inoblock)[inum - minino];
253 return (union dinode *)dp2;
254 }
255 dp1 = &((struct ufs1_dinode *)inoblock)[inum - minino];
256 return ((union dinode *)dp1);
257}
