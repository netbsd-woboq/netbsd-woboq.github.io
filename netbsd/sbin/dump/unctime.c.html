45/*
46 * Convert a ctime(3) format string into a system format date.
47 * Return the date thus calculated.
48 *
49 * Return -1 if the string is not in ctime format.
50 */
51
52time_t
53unctime(const char *str)
54{
55 struct tm then;
56
57 str = strptime(str, "%a %b %e %H:%M:%S %Y", &then);
58 then.tm_isdst = -1;
59 if (str == NULL || (*str != '\n' && *str != '\0'))
60 return (time_t) -1;
61 return mktime(&then);
62}
