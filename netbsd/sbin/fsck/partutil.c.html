56/*
57 * Convert disklabel geometry info to disk_geom.
58 */
59static void
60label2geom(struct disk_geom *geo, const struct disklabel *lp)
61{
62 geo->dg_secperunit = lp->d_secperunit;
63 geo->dg_secsize = lp->d_secsize;
64 geo->dg_nsectors = lp->d_nsectors;
65 geo->dg_ntracks = lp->d_ntracks;
66 geo->dg_ncylinders = lp->d_ncylinders;
67 geo->dg_secpercyl = lp->d_secpercyl;
68 geo->dg_pcylinders = lp->d_ncylinders;
69 geo->dg_sparespertrack = lp->d_sparespertrack;
70 geo->dg_sparespercyl = lp->d_sparespercyl;
71 geo->dg_acylinders = lp->d_acylinders;
72}

74/*
75 * Set what we need to know about disk geometry.
76 */
77static void
78dict2geom(struct disk_geom *geo, prop_dictionary_t dict)
79{
80 (void)memset(geo, 0, sizeof(struct disk_geom));
81 prop_dictionary_get_int64(dict, "sectors-per-unit",
82 &geo->dg_secperunit);
83 prop_dictionary_get_uint32(dict, "sector-size", &geo->dg_secsize);
84 prop_dictionary_get_uint32(dict, "sectors-per-track",
85 &geo->dg_nsectors);
86 prop_dictionary_get_uint32(dict, "tracks-per-cylinder",
87 &geo->dg_ntracks);
88 prop_dictionary_get_uint32(dict, "cylinders-per-unit",
89 &geo->dg_ncylinders);
90}

93int
94getdiskinfo(const char *s, int fd, const char *dt, struct disk_geom *geo,
95 struct dkwedge_info *dkw)
96{
97 struct disklabel lab;
98 struct disklabel *lp = &lab;
99 prop_dictionary_t disk_dict, geom_dict;
100 struct stat sb;
101 const struct partition *pp;
102 int ptn, error;
103
104 if (dt) {
105 lp = getdiskbyname(dt);
106 if (lp == NULL)
107 errx(1, "unknown disk type `%s'", dt);
108 }
109
110 /* Get disk description dictionary */
111 error = prop_dictionary_recv_ioctl(fd, DIOCGDISKINFO, &disk_dict);
112
113 /* fail quickly if the device does not exist at all */
114 if (error == ENXIO)
115 return -1;
116
117 if (error) {
118 /*
119 * Ask for disklabel if DIOCGDISKINFO failed. This is
120 * compatibility call and can be removed when all devices
121 * will support DIOCGDISKINFO.
122 * cgd, ccd pseudo disk drives doesn't support DIOCGDDISKINFO
123 */
124 if (ioctl(fd, DIOCGDINFO, lp) == -1) {
125 if (errno != ENXIO)
126 warn("DIOCGDINFO on %s failed", s);
127 return -1;
128 }
129 label2geom(geo, lp);
130 } else {
131 geom_dict = prop_dictionary_get(disk_dict, "geometry");
132 dict2geom(geo, geom_dict);
133 }
134
135 if (dkw == NULL)
136 return 0;
137
138 /* Get info about partition/wedge */
139 if (ioctl(fd, DIOCGWEDGEINFO, dkw) != -1) {
140 /* DIOCGWEDGEINFO didn't fail, we're done */
141 return 0;
142 }
143
144 if (ioctl(fd, DIOCGDINFO, lp) == -1) {
145 err(1, "Please implement DIOCGWEDGEINFO or "
146 "DIOCGDINFO for disk device %s", s);
147 }
148
149 /* DIOCGDINFO didn't fail */
150 (void)memset(dkw, 0, sizeof(*dkw));
151
152 if (stat(s, &sb) == -1)
153 return 0;
154
155 ptn = strchr(s, '\0')[-1] - 'a';
156 if ((unsigned)ptn >= lp->d_npartitions ||
157 (devminor_t)ptn != DISKPART(sb.st_rdev))
158 return 0;
159
160 pp = &lp->d_partitions[ptn];
161 if (ptn != getrawpartition()) {
162 dkw->dkw_offset = pp->p_offset;
163 dkw->dkw_size = pp->p_size;
164 } else {
165 dkw->dkw_offset = 0;
166 dkw->dkw_size = geo->dg_secperunit;
167 }
168 dkw->dkw_parent[0] = '*';
169 strlcpy(dkw->dkw_ptype, getfstypename(pp->p_fstype),
170 sizeof(dkw->dkw_ptype));
171
172 return 0;
173}

175int
176getdisksize(const char *name, u_int *secsize, off_t *mediasize)
177{
178 char buf[MAXPATHLEN];
179 struct disk_geom geo;
180 int fd, error;
181
182 if ((fd = opendisk(name, O_RDONLY, buf, sizeof(buf), 0)) == -1)
183 return -1;
184
185 error = getdiskinfo(name, fd, NULL, &geo, NULL);
186 close(fd);
187 if (error)
188 return error;
189
190 *secsize = geo.dg_secsize;
191 *mediasize = geo.dg_secsize * geo.dg_secperunit;
192 return 0;
193}
