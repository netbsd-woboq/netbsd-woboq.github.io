54void
55pass3(void)
56{
57 struct inoinfo *inp, **inpp;
58 int loopcnt, state;
59 ino_t orphan;
60 struct inodesc idesc;
61 char namebuf[MAXNAMLEN+1];
62 union dinode *dp;
63
64 for (inpp = &inpsort[inplast - 1]; inpp >= inpsort; inpp--) {
65 int inpindex = inpp - inpsort;
66 if (got_siginfo) {
67 fprintf(stderr,
68 "%s: phase 3: dir %d of %d (%d%%)\n", cdevname(),
69 (int)(inplast - inpindex - 1), (int)inplast,
70 (int)((inplast - inpindex - 1) * 100 / inplast));
71 got_siginfo = 0;
72 }
73#ifdef PROGRESS
74 progress_bar(cdevname(), preen ? NULL : "phase 3",
75 inplast - inpindex - 1, inplast);
76#endif /* PROGRESS */
77 inp = *inpp;
78 state = inoinfo(inp->i_number)->ino_state;
79 if (inp->i_number == UFS_ROOTINO ||
80 (inp->i_parent != 0 && state != DSTATE))
81 continue;
82 if (state == DCLEAR)
83 continue;
84 /*
85 * If we are running with soft updates and we come
86 * across unreferenced directories, we just leave
87 * them in DSTATE which will cause them to be pitched
88 * in pass 4.
89 */
90 if (preen &&
91 resolved && usedsoftdep && state == DSTATE) {
92 if (inp->i_dotdot >= UFS_ROOTINO)
93 inoinfo(inp->i_dotdot)->ino_linkcnt++;
94 continue;
95 }
96 for (loopcnt = 0; ; loopcnt++) {
97 orphan = inp->i_number;
98 if (inp->i_parent == 0 ||
99 inoinfo(inp->i_parent)->ino_state != DSTATE ||
100 loopcnt > countdirs)
101 break;
102 inp = getinoinfo(inp->i_parent);
103 }
104 if (loopcnt <= countdirs) {
105 if (linkup(orphan, inp->i_dotdot, NULL))
106 inoinfo(lfdir)->ino_linkcnt--;
107 continue;
108 }
109 pfatal("ORPHANED DIRECTORY LOOP DETECTED I=%lu",
110 (u_long)orphan);
111 if (reply("RECONNECT") == 0)
112 continue;
113 dp = ginode(inp->i_parent);
114 memset(&idesc, 0, sizeof(struct inodesc));
115 idesc.id_type = DATA;
116 idesc.id_number = inp->i_parent;
117 idesc.id_parent = orphan;
118 idesc.id_func = findname;
119 idesc.id_name = namebuf;
120 idesc.id_uid = iswap32(DIP(dp, uid));
121 idesc.id_gid = iswap32(DIP(dp, gid));
122 if ((ckinode(dp, &idesc) & FOUND) == 0)
123 pfatal("COULD NOT FIND NAME IN PARENT DIRECTORY");
124 if (linkup(orphan, inp->i_parent, namebuf)) {
125 idesc.id_func = clearentry;
126 if (ckinode(ginode(inp->i_parent), &idesc) & FOUND)
127 inoinfo(orphan)->ino_linkcnt++;
128 inoinfo(lfdir)->ino_linkcnt--;
129 }
130 }
131#ifdef PROGRESS
132 if (!preen)
133 progress_done();
134#endif /* PROGRESS */
135}
