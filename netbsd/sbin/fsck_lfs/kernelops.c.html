34/*
35 * Select operations vector for kernel communication -- either regular
36 * system calls or rump system calls. The latter are use when the
37 * cleaner is run as part of rump_lfs.
38 *
39 * The selection is controlled by an ifdef for now, since I see no
40 * value in making it dynamic.
41 */
42
43#include <sys/types.h>
44#include <sys/mount.h>
45#include <sys/statvfs.h>
46
47#include <fcntl.h>
48#include <unistd.h>
49
50#include "kernelops.h"
51
52#ifdef USE_RUMP
53
54#include <rump/rump.h>
55#include <rump/rump_syscalls.h>
56
57const struct kernelops kops = {
58 .ko_open = rump_sys_open,
59 .ko_fcntl = rump_sys_fcntl,
60 .ko_statvfs = rump_sys_statvfs1,
61 .ko_fhopen = rump_sys_fhopen,
62 .ko_close = rump_sys_close,
63
64 .ko_pread = rump_sys_pread,
65 .ko_pwrite = rump_sys_pwrite,
66};
67
68#else
69
70const struct kernelops kops = {
71 .ko_open = open,
72 .ko_fcntl = fcntl,
73 .ko_statvfs = statvfs1,
74 .ko_fhopen = fhopen,
75 .ko_close = close,
76
77 .ko_pread = pread,
78 .ko_pwrite = pwrite,
79};
80#endif
