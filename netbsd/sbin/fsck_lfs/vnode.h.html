40struct uvnode {
41 int v_fd;
42 int (*v_strategy_op) (struct ubuf *);
43 int (*v_bwrite_op) (struct ubuf *);
44 int (*v_bmap_op) (struct uvnode *, daddr_t, daddr_t *);
45 u_int32_t v_uflag;
46 void *v_fs;
47 void *v_data;
48#undef v_usecount
49 int v_usecount;

51 struct ubuflists v_dirtyblkhd; /* dirty blocklist head */
52 LIST_ENTRY(uvnode) v_mntvnodes;
53 LIST_ENTRY(uvnode) v_getvnodes;
54};
