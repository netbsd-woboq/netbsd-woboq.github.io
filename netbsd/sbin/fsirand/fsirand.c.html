79static const off_t sblock_try[] = SBLOCKSEARCH;
80
81/*
82 * getsblock():
83 * Return the superblock
84 */
85static void
86getsblock(int fd, const char *name, struct fs *fs)
87{
88 int i;
89
90 for (i = 0; ; i++) {
91 if (sblock_try[i] == -1)
92 errx(1, "%s: can't find superblock", name);
93 if (pread(fd, fs, SBLOCKSIZE, sblock_try[i]) != SBLOCKSIZE)
94 continue;
95
96 switch(fs->fs_magic) {
97 case FS_UFS2_MAGIC:
98 is_ufs2 = 1;
99 /* FALLTHROUGH */
100 case FS_UFS1_MAGIC:
101 break;
102 case FS_UFS2_MAGIC_SWAPPED:
103 is_ufs2 = 1;
104 /* FALLTHROUGH */
105 case FS_UFS1_MAGIC_SWAPPED:
106 needswap = 1;
107 ffs_sb_swap(fs, fs);
108 break;
109 default:
110 continue;
111 }
112
113 if (!is_ufs2 && sblock_try[i] == SBLOCK_UFS2)
114 continue;
115 break;
116 }
117
118 if (fs->fs_ncg < 1)
119 errx(1, "%s: bad ncg in superblock", name);
120
121 if (fs->fs_sbsize > SBLOCKSIZE)
122 errx(1, "%s: superblock too large", name);
123}

126/*
127 * fixinodes():
128 * Randomize the inode generation numbers
129 */
130static void
131fixinodes(int fd, struct fs *fs, struct disklabel *lab, int pflag, long xorval)
132{
133 int inopb = FFS_INOPB(fs);
134 int size;
135 caddr_t buf;
136 struct ufs1_dinode *dp1 = NULL;
137 struct ufs2_dinode *dp2 = NULL;
138 int i;
139
140 size = is_ufs2 ? inopb * sizeof (struct ufs2_dinode) :
141 inopb * sizeof (struct ufs1_dinode);
142
143 if ((buf = malloc(size)) == NULL)
144 err(1, "Out of memory");
145
146 if (is_ufs2)
147 dp2 = (struct ufs2_dinode *)buf;
148 else
149 dp1 = (struct ufs1_dinode *)buf;
150
151 for (ino = 0, imax = fs->fs_ipg * fs->fs_ncg; ino < imax;) {
152 off_t sp;
153 sp = (off_t) FFS_FSBTODB(fs, ino_to_fsba(fs, ino)) *
154 (off_t) lab->d_secsize;
155
156 if (lseek(fd, sp, SEEK_SET) == (off_t) -1)
157 err(1, "Seeking to inode %d failed", ino);
158
159 if (read(fd, buf, size) != size)
160 err(1, "Reading inodes %d+%d failed", ino, inopb);
161
162 for (i = 0; i < inopb; i++) {
163 if (is_ufs2) {
164 if (pflag)
165 printf("inode %10d gen 0x%08x\n",
166 ino,
167 ufs_rw32(dp2[i].di_gen, needswap));
168 else
169 dp2[i].di_gen =
170 ufs_rw32((arc4random() & INT32_MAX)^ xorval,
171 needswap);
172 } else {
173 if (pflag)
174 printf("inode %10d gen 0x%08x\n",
175 ino,
176 ufs_rw32(dp1[i].di_gen, needswap));
177 else
178 dp1[i].di_gen =
179 ufs_rw32((arc4random() & INT32_MAX) ^ xorval,
180 needswap);
181 }
182 if (++ino > imax)
183 errx(1, "Exceeded number of inodes");
184 }
185
186 if (pflag)
187 continue;
188
189 if (lseek(fd, sp, SEEK_SET) == (off_t) -1)
190 err(1, "Seeking to inode %d failed", ino);
191
192 if (write(fd, buf, size) != size)
193 err(1, "Writing inodes %d+%d failed", ino, inopb);
194 }
195 free(buf);
196}

198/*
199 * statussig():
200 * display current status
201 */
202static void
203statussig(int dummy)
204{
205 char msgbuf[256];
206 int len, deltat;
207 time_t tnow;
208
209 (void)time(&tnow);
210 len = snprintf(msgbuf, sizeof(msgbuf),
211 "fsirand: completed inode %d of %d (%3.2f%%)",
212 ino, imax, (ino * 100.0) / imax);
213 if (imax - ino) {
214 deltat = tstart - tnow + (1.0 * (tnow - tstart)) / ino * imax;
215 len += snprintf(msgbuf + len, sizeof(msgbuf) - len,
216 ", finished in %d:%02d\n", deltat / 60, deltat % 60);
217 } else {
218 len += snprintf(msgbuf + len, sizeof(msgbuf) - len, "\n");
219 }
220 write(STDERR_FILENO, msgbuf, len);
221}

223int
224main(int argc, char *argv[])
225{
226 const char *special;
227 char buf[SBLOCKSIZE], device[MAXPATHLEN];
228 struct fs *fs = (struct fs *) buf;
229 struct disklabel lab;
230 long xorval;
231 char *ep;
232 int fd, c, Fflag, pflag, openflags;
233
234 xorval = 0;
235 Fflag = pflag = 0;
236
237 while ((c = getopt(argc, argv, "Fpx:")) != -1)
238 switch (c) {
239 case 'F':
240 Fflag++;
241 break;
242 case 'p':
243 pflag++;
244 break;
245 case 'x':
246 errno = 0;
247 xorval = strtol(optarg, &ep, 0);
248 if ((xorval == LONG_MIN || xorval == LONG_MAX) &&
249 errno == ERANGE)
250 err(1, "Out of range constant");
251 if (*ep)
252 errx(1, "Bad constant");
253 break;
254 default:
255 usage();
256 }
257
258 argv += optind;
259 argc -= optind;
260
261 if (argc != 1)
262 usage();
263
264 special = argv[0];
265 openflags = pflag ? O_RDONLY : O_RDWR;
266 if (Fflag)
267 fd = open(special, openflags);
268 else {
269 fd = opendisk(special, openflags, device, sizeof(device), 0);
270 special = device;
271 }
272 if (fd == -1)
273 err(1, "Cannot open `%s'", special);
274
275 if (Fflag) {
276 memset(&lab, 0, sizeof(lab));
277 lab.d_secsize = DEV_BSIZE; /* XXX */
278 } else {
279 if (ioctl(fd, DIOCGDINFO, &lab) == -1)
280 err(1, "%s: cannot get disklabel information", special);
281 }
282
283 time(&tstart);
284 (void)signal(SIGINFO, statussig);
285 getsblock(fd, special, fs);
286 fixinodes(fd, fs, &lab, pflag, xorval);
287
288 (void) close(fd);
289 return 0;
290}
