42/*
43 * We define our own uuid type so that we don't have to mess around
44 * with different uuid implementation (linux+macosx which use an
45 * array, and {Free,Net}BSD who use a struct. We just need minimal
46 * support anyway
47 */
48
49// Must match the array in gpt_uuid.c
50typedef enum {
51 GPT_TYPE_INVALID = -1,
52 GPT_TYPE_APPLE_HFS = 0,
53 GPT_TYPE_APPLE_UFS,
54 GPT_TYPE_BIOS,
55 GPT_TYPE_EFI,
56 GPT_TYPE_FREEBSD,
57 GPT_TYPE_FREEBSD_SWAP,
58 GPT_TYPE_FREEBSD_UFS,
59 GPT_TYPE_FREEBSD_VINUM,
60 GPT_TYPE_FREEBSD_ZFS,
61 GPT_TYPE_LINUX_DATA,
62 GPT_TYPE_LINUX_SWAP,
63 GPT_TYPE_LINUX_RAID,
64 GPT_TYPE_LINUX_LVM,
65 GPT_TYPE_MS_BASIC_DATA,
66 GPT_TYPE_MS_RESERVED,
67 GPT_TYPE_NETBSD_CCD,
68 GPT_TYPE_NETBSD_CGD,
69 GPT_TYPE_NETBSD_FFS,
70 GPT_TYPE_NETBSD_LFS,
71 GPT_TYPE_NETBSD_RAIDFRAME,
72 GPT_TYPE_NETBSD_SWAP,
73 GPT_TYPE_VMWARE_VMKCORE,
74 GPT_TYPE_VMWARE_VMFS,
75 GPT_TYPE_VMWARE_RESERVED
76} gpt_type_t;
77
78typedef uint8_t gpt_uuid_t[16];
79extern const gpt_uuid_t gpt_uuid_nil;
80
81__BEGIN_DECLS
82static inline int
83gpt_uuid_is_nil(const gpt_uuid_t u) {
84 return memcmp(u, gpt_uuid_nil, sizeof(gpt_uuid_t)) == 0;
85}
86
87static inline int
88gpt_uuid_equal(const gpt_uuid_t u1, const gpt_uuid_t u2) {
89 return memcmp(u1, u2, sizeof(gpt_uuid_t)) == 0;
90}
91
92static inline void
93gpt_uuid_copy(gpt_uuid_t u1, const gpt_uuid_t u2) {
94 memcpy(u1, u2, sizeof(gpt_uuid_t));
95}
96
97int gpt_uuid_snprintf(char *, size_t, const char *, const gpt_uuid_t);
98
99void gpt_uuid_create(gpt_type_t, gpt_uuid_t, uint16_t *, size_t);
100
101int gpt_uuid_parse(const char *, gpt_uuid_t);
102
103struct gpt;
104int gpt_uuid_generate(struct gpt *, gpt_uuid_t);
105
106/* returns number of entries, callback func may be NULL */
107size_t gpt_uuid_query(
108 void (*func)(const char *uuid, const char *short_name, const char *desc));
109
110void gpt_uuid_help(const char *);
111
112__END_DECLS
