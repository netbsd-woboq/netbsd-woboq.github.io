31#ifndef _MAP_H_
32#define _MAP_H_
33
34struct map {
35 off_t map_start;
36 off_t map_size;
37 struct map *map_next;
38 struct map *map_prev;
39 int map_type;
40#define MAP_TYPE_UNUSED 0
41#define MAP_TYPE_MBR 1
42#define MAP_TYPE_MBR_PART 2
43#define MAP_TYPE_PRI_GPT_HDR 3
44#define MAP_TYPE_SEC_GPT_HDR 4
45#define MAP_TYPE_PRI_GPT_TBL 5
46#define MAP_TYPE_SEC_GPT_TBL 6
47#define MAP_TYPE_GPT_PART 7
48#define MAP_TYPE_PMBR 8
49 unsigned int map_index;
50 void *map_data;
51 int map_alloc;
52};
53
54struct gpt;
55
56struct map *map_add(struct gpt *, off_t, off_t, int, void *, int);
57struct map *map_alloc(struct gpt *, off_t, off_t, off_t);
58struct map *map_find(struct gpt *, int);
59struct map *map_first(struct gpt *);
60struct map *map_last(struct gpt *);
61off_t map_resize(struct gpt *, struct map *, off_t, off_t);
62off_t map_free(struct gpt *, off_t, off_t);
63int map_init(struct gpt *, off_t);
64
65#endif /* _MAP_H_ */
