52static int cmd_remove(gpt_t, int, char *[]);
53
54static const char *removehelp[] = {
55 "-a",
56 "[-b blocknr] [-i index] [-L label] [-s sectors] [-t type]",
57};
58
59struct gpt_cmd c_remove = {
60 "remove",
61 cmd_remove,
62 removehelp, __arraycount(removehelp),
63 GPT_SYNC,
64};
65
66#define usage() gpt_usage(NULL, &c_remove)
67
68static void
69change(struct gpt_ent *ent, void *v, int backup)
70{
71 /* Remove the primary entry by clearing the partition type. */
72 gpt_uuid_copy(ent->ent_type, gpt_uuid_nil);
73}
74
75static int
76cmd_remove(gpt_t gpt, int argc, char *argv[])
77{
78 int ch;
79 struct gpt_find find;
80
81 memset(&find, 0, sizeof(find));
82 find.msg = "removed";
83
84 /* Get the remove options */
85 while ((ch = getopt(argc, argv, GPT_FIND)) != -1)
86 if (gpt_add_find(gpt, &find, ch) == -1)
87 return usage();
88
89 if (argc != optind)
90 return usage();
91
92 return gpt_change_ent(gpt, &find, change, NULL);
93}
