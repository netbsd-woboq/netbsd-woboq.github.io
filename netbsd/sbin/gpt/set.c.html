52static int cmd_set(gpt_t, int, char *[]);
53
54static const char *sethelp[] = {
55 "-a attribute [-i index] [-b startsec]",
56 "-N [-i index] [-b startsec]",
57 "-l",
58};
59
60struct gpt_cmd c_set = {
61 "set",
62 cmd_set,
63 sethelp, __arraycount(sethelp),
64 GPT_OPTDEV,
65};
66
67#define usage() gpt_usage(NULL, &c_set)
68
69static int
70cmd_set(gpt_t gpt, int argc, char *argv[])
71{
72 int ch;
73 unsigned int entry = 0;
74 uint64_t attributes = 0, clear = 0;
75 off_t start = 0;
76 map_t m;
77
78 while ((ch = getopt(argc, argv, "a:b:i:lN")) != -1) {
79 switch(ch) {
80 case 'a':
81 if (gpt == NULL || gpt_attr_get(gpt, &attributes) == -1)
82 return usage();
83 break;
84 case 'b':
85 if (gpt == NULL || gpt_human_get(gpt, &start) == -1)
86 return usage();
87 break;
88 case 'i':
89 if (gpt == NULL || gpt_uint_get(gpt, &entry) == -1)
90 return usage();
91 break;
92 case 'l':
93 gpt_attr_help("\t");
94 return 0;
95 case 'N':
96 clear = ~clear;
97 break;
98 default:
99 return usage();
100 }
101 }
102
103 if (start > 0) {
104 for (m = map_first(gpt); m != NULL; m = m->map_next) {
105 if (m->map_type != MAP_TYPE_GPT_PART ||
106 m->map_index < 1)
107 continue;
108 if (start != m->map_start)
109 continue;
110 entry = m->map_index;
111 break;
112 }
113 }
114
115 if (gpt == NULL || argc != optind || (attributes != 0 && clear != 0))
116 return usage();
117
118 return gpt_attr_update(gpt, entry, attributes, clear);
119}
