52static int cmd_uuid(gpt_t, int, char *[]);
53
54static const char *uuidhelp[] = {
55 "-a",
56 "[-b blocknr] [-i index] [-L label] [-s sectors] [-t type]",
57};
58
59struct gpt_cmd c_uuid = {
60 "uuid",
61 cmd_uuid,
62 uuidhelp, __arraycount(uuidhelp),
63 GPT_SYNC,
64};
65
66#define usage() gpt_usage(NULL, &c_uuid)
67
68static void
69change_ent(struct gpt_ent *ent, void *v, int backup)
70{
71 static gpt_uuid_t uuidstore;
72
73 if (!backup)
74 gpt_uuid_generate(NULL, uuidstore);
75 memmove(ent->ent_guid, uuidstore, sizeof(ent->ent_guid));
76}
77
78static void
79change_hdr(struct gpt_hdr *hdr, void *v, int backup)
80{
81 static gpt_uuid_t uuidstore;
82
83 if (!backup)
84 gpt_uuid_generate(NULL, uuidstore);
85 memmove(hdr->hdr_guid, uuidstore, sizeof(hdr->hdr_guid));
86}
87
88static int
89cmd_uuid(gpt_t gpt, int argc, char *argv[])
90{
91 int ch, rc;
92 struct gpt_find find;
93
94 memset(&find, 0, sizeof(find));
95 find.msg = "UUID changed";
96
97 /* Get the uuid options */
98 while ((ch = getopt(argc, argv, GPT_FIND)) != -1) {
99 if (gpt == NULL || gpt_add_find(gpt, &find, ch) == -1)
100 return usage();
101 }
102
103 if (gpt == NULL || argc != optind)
104 return usage();
105
106 rc = gpt_change_ent(gpt, &find, change_ent, NULL);
107 if (rc != 0)
108 return rc;
109
110 if (find.all)
111 return gpt_change_hdr(gpt, &find, change_hdr, NULL);
112
113 return 0;
114}
