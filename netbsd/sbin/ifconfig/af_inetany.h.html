28#ifndef _IFCONFIG_AF_INETANY_H
29#define _IFCONFIG_AF_INETANY_H
30
31#include <sys/types.h>
32#include <prop/proplib.h>
33
34#define IFADDR_PARAM(__arg) {.cmd = (__arg), .desc = #__arg}
35#define BUFPARAM(__arg) {.buf = &(__arg), .buflen = sizeof(__arg)}
36
37struct apbuf {
38 void *buf;
39 size_t buflen;
40};
41
42struct afparam {
43 struct {
44 char *buf;
45 size_t buflen;
46 } name[2];
47 struct apbuf dgaddr, addr, brd, dst, mask, req, dgreq, defmask,
48 pre_aifaddr_arg;
49 struct {
50 unsigned long cmd;
51 const char *desc;
52 } aifaddr, difaddr, gifaddr;
53 int (*pre_aifaddr)(prop_dictionary_t, const struct afparam *);
54};
55
56void commit_address(prop_dictionary_t, prop_dictionary_t,
57 const struct afparam *);
58
59#endif /* _IFCONFIG_AF_INETANY_H */
