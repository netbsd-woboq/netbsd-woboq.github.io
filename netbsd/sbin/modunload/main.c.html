34#include <sys/module.h>
35
36#include <stdio.h>
37#include <stdlib.h>
38#include <string.h>
39#include <unistd.h>
40#include <err.h>
41
42#include "prog_ops.h"
43
44static void usage(void) __dead;
45
46int
47main(int argc, char **argv)
48{
49 int i;
50
51 if (argc < 2)
52 usage();
53
54 if (prog_init && prog_init() == -1)
55 err(1, "init failed");
56
57 for (i = 1; i < argc; i++) {
58 if (prog_modctl(MODCTL_UNLOAD, argv[i])) {
59 err(EXIT_FAILURE, "%s", argv[i]);
60 }
61 }
62
63 exit(EXIT_SUCCESS);
64}
65
66static void
67usage(void)
68{
69
70 (void)fprintf(stderr, "Usage: %s <module_name ...>\n", getprogname());
71 exit(EXIT_FAILURE);
72}
