47int
48a_num(const char *s, const char *id_type)
49{
50 int id;
51 char *ep;
52
53 id = strtol(s, &ep, 0);
54 if (*ep || s == ep || id < 0)
55 errx(1, "unknown %s id: %s", id_type, s);
56 return id;
57}
58
59gid_t
60a_gid(const char *s)
61{
62 struct group *gr;
63
64 if ((gr = getgrnam(s)) != NULL)
65 return gr->gr_gid;
66 return a_num(s, "group");
67}
68
69uid_t
70a_uid(const char *s)
71{
72 struct passwd *pw;
73
74 if ((pw = getpwnam(s)) != NULL)
75 return pw->pw_uid;
76 return a_num(s, "user");
77}
78
79mode_t
80a_mask(const char *s)
81{
82 int rv;
83 char *ep;
84
85 rv = strtol(s, &ep, 8);
86 if (s == ep || *ep || rv < 0)
87 errx(1, "invalid file mode: %s", s);
88 return rv;
89}
