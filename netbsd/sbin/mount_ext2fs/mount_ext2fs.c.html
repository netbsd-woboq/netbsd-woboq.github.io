89void
90mount_ext2fs_parseargs(int argc, char *argv[],
91 struct ufs_args *args, int *mntflags,
92 char *canon_dev, char *canon_dir)
93{
94 int ch;
95 mntoptparse_t mp;
96
97 memset(args, 0, sizeof(*args));
98 *mntflags = 0;
99 optind = optreset = 1; /* Reset for parse of new argv. */
100 while ((ch = getopt(argc, argv, "o:")) != -1)
101 switch (ch) {
102 case 'o':
103 mp = getmntopts(optarg, mopts, mntflags, 0);
104 if (mp == NULL)
105 err(1, "getmntopts");
106 freemntopts(mp);
107 break;
108 case '?':
109 default:
110 ext2fs_usage();
111 }
112 argc -= optind;
113 argv += optind;
114
115 if (argc != 2)
116 ext2fs_usage();
117
118 pathadj(argv[0], canon_dev);
119 args->fspec = canon_dev;
120
121 pathadj(argv[1], canon_dir);
122}
123
124int
125mount_ext2fs(int argc, char *argv[])
126{
127 struct ufs_args args; /* XXX ffs_args */
128 char fs_name[MAXPATHLEN], canon_dev[MAXPATHLEN];
129 const char *errcause;
130 int mntflags;
131
132 mount_ext2fs_parseargs(argc, argv, &args, &mntflags,
133 canon_dev, fs_name);
134
135 if (mount(MOUNT_EXT2FS, fs_name, mntflags, &args, sizeof args) == -1) {
136 switch (errno) {
137 case EMFILE:
138 errcause = "mount table full";
139 break;
140 case EINVAL:
141 if (mntflags & MNT_UPDATE)
142 errcause =
143 "specified device does not match mounted device";
144 else
145 errcause = "incorrect super block";
146 break;
147 default:
148 errcause = strerror(errno);
149 break;
150 }
151 errx(1, "%s on %s: %s", args.fspec, fs_name, errcause);
152 }
153 exit(0);
154}
