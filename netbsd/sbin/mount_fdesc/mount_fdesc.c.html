114void
115mount_fdesc_parseargs(int argc, char *argv[], void *dummy, int *mntflags,
116 char *canon_dev, char *canon_dir)
117{
118 int ch;
119 mntoptparse_t mp;
120
121 *mntflags = 0;
122 while ((ch = getopt(argc, argv, "o:")) != -1)
123 switch (ch) {
124 case 'o':
125 mp = getmntopts(optarg, mopts, mntflags, 0);
126 if (mp == NULL)
127 err(1, "getmntopts");
128 freemntopts(mp);
129 break;
130 case '?':
131 default:
132 usage();
133 }
134 argc -= optind;
135 argv += optind;
136
137 if (argc != 2)
138 usage();
139
140 /* getargs is a NULL op and kernel would return EINVAL */
141 if (*mntflags & MNT_GETARGS)
142 exit(0);
143
144 strlcpy(canon_dev, argv[0], MAXPATHLEN);
145 if (realpath(argv[1], canon_dir) == NULL) /* Check mounton path */
146 err(1, "realpath %s", argv[1]);
147 if (strncmp(argv[1], canon_dir, MAXPATHLEN)) {
148 warnx("\"%s\" is a relative path.", argv[1]);
149 warnx("using \"%s\" instead.", canon_dir);
150 }
151}
152
153int
154mount_fdesc(int argc, char *argv[])
155{
156 int mntflags;
157 char canon_dev[MAXPATHLEN], canon_dir[MAXPATHLEN];
158
159 mount_fdesc_parseargs(argc, argv, NULL, &mntflags,
160 canon_dev, canon_dir);
161 if (mount(MOUNT_FDESC, canon_dir, mntflags, NULL, 0) == -1)
162 err(1, "fdesc on %s", canon_dir);
163 exit(0);
164}
