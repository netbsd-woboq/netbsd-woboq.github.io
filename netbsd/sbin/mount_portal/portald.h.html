40/*
41 * Meta-chars in an RE. Paths in the config file containing
42 * any of these characters will be matched using regexec, other
43 * paths will be prefix-matched.
44 */

54struct portal_cred {
55 int pcr_flag; /* File open mode */
56 uid_t pcr_uid; /* From cred */
57 gid_t pcr_gid; /* From cred */
58 uint16_t pcr_ngroups; /* From cred */
59 gid_t pcr_groups[NGROUPS]; /* From cred */
60};
61
62typedef struct provider provider;
63struct provider {
64 const char *pr_match;
65 int (*pr_func)(struct portal_cred *,
66 char *key, char **v, int *fdp);
67};
68extern provider providers[];
