129static gid_t
130getgrp(const char *name)
131{
132 char *ep;
133 struct group *grp;
134 long l;
135
136 if (name == NULL)
137 errx(1, "Missing group name");
138
139 l = strtol(name, &ep, 0);
140
141 if (name == ep || *ep)
142 grp = getgrnam(name);
143 else
144 grp = getgrgid((gid_t)l);
145
146 if (grp == NULL)
147 errx(1, "Cannot find group `%s'", name);
148
149 return grp->gr_gid;
150}

153int
154mount_ptyfs(int argc, char *argv[])
155{
156 int ch, mntflags = 0, altflags = 0;
157 struct ptyfs_args args;
158 mntoptparse_t mp;
159 char canon_dir[MAXPATHLEN];
160
161
162 setprogname(argv[0]);
163
164 args.version = PTYFS_ARGSVERSION;
165 args.gid = getgrp("tty");
166 args.mode = S_IRUSR|S_IWUSR|S_IWGRP;
167 args.flags = 0;
168
169 while ((ch = getopt(argc, argv, "cg:m:o:")) != -1)
170 switch (ch) {
171 case 'c': /* compat */
172 compat:
173 warnx("-c and -o chroot options are obsolete");
174 break;
175 case 'o':
176 altflags = 0;
177 mp = getmntopts(optarg, mopts, &mntflags, &altflags);
178 if (mp == NULL)
179 err(1, "getmntopts");
180 if (altflags & ALTF_GROUP)
181 args.gid = getgrp(getmntoptstr(mp, "group"));
182 if (altflags & ALTF_MODE)
183 args.mode = (mode_t)getmntoptnum(mp, "mode");
184 if (altflags & ALTF_CHROOT)
185 goto compat;
186 freemntopts(mp);
187 break;
188 case 'g':
189 args.gid = getgrp(optarg);
190 break;
191 case 'm':
192 args.mode = (mode_t)strtol(optarg, NULL, 0);
193 break;
194 case '?':
195 default:
196 usage();
197 }
198
199 argc -= optind;
200 argv += optind;
201
202 if (argc != 2)
203 usage();
204
205 if (realpath(argv[1], canon_dir) == NULL) /* Check mounton path */
206 err(1, "realpath %s", argv[1]);
207 if (strncmp(argv[1], canon_dir, MAXPATHLEN)) {
208 warnx("\"%s\" is a relative path.", argv[1]);
209 warnx("using \"%s\" instead.", canon_dir);
210 }
211
212 if (mount(MOUNT_PTYFS, canon_dir, mntflags, &args, sizeof args) == -1)
213 err(1, "ptyfs on %s", canon_dir);
214 if (mntflags & MNT_GETARGS) {
215 printf("version=%d, gid=%lu, mode=0%o\n", args.version,
216 (unsigned long)args.gid, args.mode);
217 }
218 return 0;
219}
220
221static void
222usage(void)
223{
224 (void)fprintf(stderr,
225 "Usage: %s [-g <group|gid>] [-m <mode>] [-o options] "
226 "ptyfs mountpoint\n", getprogname());
227 exit(1);
228}
