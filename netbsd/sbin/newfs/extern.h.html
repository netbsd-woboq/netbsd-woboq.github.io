30/* * variables set up by front end. */
31extern int mfs; /* run as the memory based filesystem */
32extern int Nflag; /* run mkfs without writing file system */
33extern int Oflag; /* format as an 4.3BSD file system */
34extern int verbosity; /* amount of printf() output */
35extern int64_t fssize; /* file system size */
36extern int sectorsize; /* bytes/sector */
37extern int rpm; /* revolutions/minute of drive */
38extern int fsize; /* fragment size */
39extern int bsize; /* block size */
40extern int maxbsize;
41extern int cpg; /* cylinders/cylinder group */
42extern int cpgflg; /* cylinders/cylinder group flag was given */
43extern int minfree; /* free space threshold */
44extern int opt; /* optimization preference (space or time) */
45extern int density; /* number of bytes per inode */
46extern int num_inodes; /* number of inodes (overrides density) */
47extern int maxcontig; /* max contiguous blocks to allocate */
48extern int maxbpg; /* maximum blocks per file in a cyl group */
49extern int maxblkspercg;
50extern int nrpos; /* # of distinguished rotational positions */
51extern int avgfilesize; /* expected average file size */
52extern int avgfpdir; /* expected number of files per directory */
53extern u_long memleft; /* virtual memory available */
54extern caddr_t membase; /* start address of memory based filesystem */
55extern int quotas; /* filesystem quota to enable */
56
57#ifndef NO_FFS_EI
58extern int needswap; /* Filesystem not in native byte order */
59#else
60/* Disable Endian-Independent FFS support for install media */
61#define needswap (0)
62#define ffs_cg_swap(a, b, c) __nothing
63#define ffs_csum_swap(a, b, c) __nothing
64#define ffs_dinode1_swap(a, b) __nothing
65#define ffs_sb_swap(a, b) __nothing
66#endif
67
68#ifndef NO_APPLE_UFS
69extern int isappleufs; /* Filesystem is Apple UFS */
70extern char *appleufs_volname; /* Apple UFS volume name */
71#else
72/* Disable Apple UFS support for install media */
73#define isappleufs (0)
74#endif
