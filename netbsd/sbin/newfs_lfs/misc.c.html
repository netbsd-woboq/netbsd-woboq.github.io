48u_int
49lfs_log2(u_int num)
50{
51 u_int i, limit;
52
53 limit = 1;
54 for (i = 0; limit < num; limit = limit << 1, i++);
55 return (i);
56}

58void
59fatal(const char *fmt, ...)
60{
61 va_list ap;
62
63 va_start(ap, fmt);
64 (void)fprintf(stderr, "%s: ", progname);
65 (void)vfprintf(stderr, fmt, ap);
66 va_end(ap);
67 (void)fprintf(stderr, "\n");
68 exit(1);
69 /* NOTREACHED */
70}
