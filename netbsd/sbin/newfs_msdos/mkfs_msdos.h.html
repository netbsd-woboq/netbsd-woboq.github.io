32#include <sys/types.h>
33#include <stdbool.h>
34#define ALLOPTS \
35AOPT('@', off_t, offset, 0, "Offset in device") \
36AOPT('B', char *, bootstrap, -1, "Bootstrap file") \
37AOPT('C', off_t, create_size, 0, "Create file") \
38AOPT('F', uint8_t, fat_type, 12, "FAT type (12, 16, or 32)") \
39AOPT('I', uint32_t, volume_id, 0, "Volume ID") \
40AOPT('L', char *, volume_label, -1, "Volume Label") \
41AOPT('N', bool, no_create, -2, "Don't create file system, print params only") \
42AOPT('O', char *, OEM_string, -1, "OEM string") \
43AOPT('S', uint16_t, bytes_per_sector, 1, "Bytes per sector") \
44AOPT('a', uint32_t, sectors_per_fat, 1, "Sectors per FAT") \
45AOPT('b', uint32_t, block_size, 1, "Block size") \
46AOPT('c', uint8_t, sectors_per_cluster, 1, "Sectors per cluster") \
47AOPT('e', uint16_t, directory_entries, 1, "Directory entries") \
48AOPT('f', char *, floppy, -1, "Standard format floppies (160,180,320,360,640,720,1200,1232,1440,2880)") \
49AOPT('h', uint16_t, drive_heads, 1, "Drive heads") \
50AOPT('i', uint16_t, info_sector, 1, "Info sector") \
51AOPT('k', uint16_t, backup_sector, 1, "Backup sector") \
52AOPT('m', uint8_t, media_descriptor, 0, "Media descriptor") \
53AOPT('n', uint8_t, num_FAT, 1, "Number of FATs") \
54AOPT('o', uint32_t, hidden_sectors, 0, "Hidden sectors") \
55AOPT('r', uint16_t, reserved_sectors, 1, "Reserved sectors") \
56AOPT('s', uint32_t, size, 1, "File System size") \
57AOPT('u', uint16_t, sectors_per_track, 1, "Sectors per track")
58
59struct msdos_options {
60#define AOPT(_opt, _type, _name, _min, _desc) _type _name;
61ALLOPTS
62#undef AOPT
63 time_t timestamp;
64 uint32_t timestamp_set:1;
65 uint32_t volume_id_set:1;
66 uint32_t media_descriptor_set:1;
67 uint32_t hidden_sectors_set:1;
68};
69
70int mkfs_msdos(const char *, const char *, const struct msdos_options *);
