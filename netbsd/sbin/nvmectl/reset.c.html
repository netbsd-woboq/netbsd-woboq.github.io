51#ifdef RESET_USAGE
52static void
53reset_usage(void)
54{
55 fprintf(stderr, "usage:\n");
56 fprintf(stderr, "\t%s " RESET_USAGE, getprogname());
57 exit(1);
58}
59
60void
61reset(int argc, char *argv[])
62{
63 int ch, fd;
64
65 while ((ch = getopt(argc, argv, "")) != -1) {
66 switch (ch) {
67 default:
68 reset_usage();
69 }
70 }
71
72 /* Check that a controller was specified. */
73 if (optind >= argc)
74 reset_usage();
75
76 open_dev(argv[optind], &fd, 1, 1);
77 if (ioctl(fd, NVME_RESET_CONTROLLER) < 0)
78 err(1, "reset request to %s failed", argv[optind]);
79
80 exit(0);
81}
82#endif
