29#ifndef _PROG_OPS_H_
30#define _PROG_OPS_H_
31
32#include <sys/types.h>
33
34#ifndef CRUNCHOPS
35struct prog_ops {
36 int (*op_init)(void);
37
38 int (*op_socket)(int, int, int);
39 int (*op_bind)(int, const struct sockaddr *, socklen_t);
40 int (*op_setsockopt)(int, int, int, const void *, socklen_t);
41 int (*op_getsockname)(int, struct sockaddr * restrict,
42 socklen_t * restrict);
43
44 int (*op_poll)(struct pollfd *, nfds_t, int);
45
46 ssize_t (*op_recvmsg)(int, struct msghdr *, int);
47 ssize_t (*op_sendmsg)(int, const struct msghdr *, int);
48
49 int (*op_connect)(int, const struct sockaddr *, socklen_t);
50 int (*op_close)(int);
51
52 uid_t (*op_getuid)(void);
53 int (*op_setuid)(uid_t);
54 int (*op_seteuid)(uid_t);
55};
56extern const struct prog_ops prog_ops;
57
58#define prog_init prog_ops.op_init
59#define prog_socket prog_ops.op_socket
60#define prog_bind prog_ops.op_bind
61#define prog_setsockopt prog_ops.op_setsockopt
62#define prog_getsockname prog_ops.op_getsockname
63#define prog_shutdown prog_ops.op_shutdown
64#define prog_poll prog_ops.op_poll
65#define prog_recvmsg prog_ops.op_recvmsg
66#define prog_sendmsg prog_ops.op_sendmsg
67#define prog_connect prog_ops.op_connect
68#define prog_close prog_ops.op_close
69#define prog_getuid prog_ops.op_getuid
70#define prog_setuid prog_ops.op_setuid
71#define prog_seteuid prog_ops.op_seteuid
72#else
73#define prog_init ((int (*)(void))NULL)
74#define prog_socket socket
75#define prog_bind bind
76#define prog_setsockopt setsockopt
77#define prog_getsockname getsockname
78#define prog_shutdown shutdown
79#define prog_poll poll
80#define prog_recvmsg recvmsg
81#define prog_sendmsg sendmsg
82#define prog_connect connect
83#define prog_close close
84#define prog_getuid getuid
85#define prog_setuid setuid
86#define prog_seteuid seteuid
87#endif
88
89#endif /* _PROG_OPS_H_ */
