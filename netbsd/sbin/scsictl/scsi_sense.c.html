275static void asc2ascii(unsigned char, unsigned char, char *, size_t);
276
277static void
278asc2ascii(unsigned char asc, unsigned char ascq, char *result, size_t reslen)
279{
280 int i = 0;
281
282 while (adesc[i].description != NULL) {
283 if (adesc[i].asc == asc && adesc[i].ascq == ascq)
284 break;
285 i++;
286 }
287 if (adesc[i].description == NULL) {
288 if (asc == 0x40 && ascq != 0)
289 (void) snprintf(result, reslen,
290 "Diagnostic Failure on Component 0x%02x",
291 ascq & 0xff);
292 else
293 (void) snprintf(result, reslen,
294 "ASC 0x%02x ASCQ 0x%02x",
295 asc & 0xff, ascq & 0xff);
296 } else
297 (void) strlcpy(result, adesc[i].description, reslen);
298}

300void
301scsi_print_sense_data(const unsigned char *s, int slen, int verbosity)
302{
303 int32_t info;
304 int i, j, k;
305 char sbs[132], *cp;
306
307 /*
308 * Basics - print out SENSE KEY
309 */
310 printf(" SENSE KEY: %s", scsi_decode_sense(s, 0, sbs, sizeof(sbs)));
311
312 /*
313 * Print out, unqualified but aligned, FMK, EOM and ILI status.
314 */
315 if (s[2] & 0xe0) {
316 char pad;
317 printf("\n ");
318 pad = ' ';
319 if (s[2] & SSD_FILEMARK) {
320 printf("%c Filemark Detected", pad);
321 pad = ',';
322 }
323 if (s[2] & SSD_EOM) {
324 printf("%c EOM Detected", pad);
325 pad = ',';
326 }
327 if (s[2] & SSD_ILI)
328 printf("%c Incorrect Length Indicator Set", pad);
329 }
330
331 /*
332 * Now we should figure out, based upon device type, how
333 * to format the information field. Unfortunately, that's
334 * not convenient here, so we'll print it as a signed
335 * 32 bit integer.
336 */
337 info = _4btol(&s[3]);
338 if (info)
339 printf("\n INFO FIELD: %d", info);
340
341 /*
342 * Now we check additional length to see whether there is
343 * more information to extract.
344 */
345
346 /* enough for command specific information? */
347 if (((unsigned int) s[7]) < 4) {
348 printf("\n");
349 return;
350 }
351 info = _4btol(&s[8]);
352 if (info)
353 printf("\n COMMAND INFO: %d (0x%x)", info, info);
354
355 /*
356 * Decode ASC && ASCQ info, plus FRU, plus the rest...
357 */
358
359 cp = scsi_decode_sense(s, 1, sbs, sizeof(sbs));
360 if (cp)
361 printf("\n ASC/ASCQ: %s", cp);
362 if (s[14] != 0)
363 printf("\n FRU CODE: 0x%x", s[14] & 0xff);
364 cp = scsi_decode_sense(s, 3, sbs, sizeof(sbs));
365 if (cp)
366 printf("\n SKSV: %s", cp);
367 printf("\n");
368 if (verbosity == 0) {
369 printf("\n");
370 return;
371 }
372
373 /*
374 * Now figure whether we should print any additional informtion.
375 *
376 * Where should we start from? If we had SKSV data,
377 * start from offset 18, else from offset 15.
378 *
379 * From that point until the end of the buffer, check for any
380 * nonzero data. If we have some, go back and print the lot,
381 * otherwise we're done.
382 */
383 if (cp)
384 i = 18;
385 else
386 i = 15;
387 for (j = i; j < slen; j++)
388 if (s[j])
389 break;
390 if (j == slen)
391 return;
392
393 printf("\n Additional Sense Information (byte %d out...):\n", i);
394 if (i == 15) {
395 printf("\n\t%2d:", i);
396 k = 7;
397 } else {
398 printf("\n\t%2d:", i);
399 k = 2;
400 j -= 2;
401 }
402 while (j > 0) {
403 if (i >= slen)
404 break;
405 if (k == 8) {
406 k = 0;
407 printf("\n\t%2d:", i);
408 }
409 printf(" 0x%02x", s[i] & 0xff);
410 k++;
411 j--;
412 i++;
413 }
414 printf("\n\n");
415}

417char *
418scsi_decode_sense(const unsigned char *snsbuf, int flag, char *rqsbuf,
419 size_t rqsbuflen)
420{
421 unsigned char skey;
422 char localbuf[64];
423
424 skey = 0;
425
426 if (flag == 0 || flag == 2 || flag == 3)
427 skey = snsbuf[2] & 0xf;
428 if (flag == 0) { /* Sense Key Only */
429 (void) strlcpy(rqsbuf, sense_keys[skey], rqsbuflen);
430 return (rqsbuf);
431 } else if (flag == 1) { /* ASC/ASCQ Only */
432 asc2ascii(snsbuf[12], snsbuf[13], rqsbuf, rqsbuflen);
433 return (rqsbuf);
434 } else if (flag == 2) { /* Sense Key && ASC/ASCQ */
435 asc2ascii(snsbuf[12], snsbuf[13], localbuf, sizeof(localbuf));
436 (void) snprintf(rqsbuf, rqsbuflen, "%s, %s", sense_keys[skey],
437 localbuf);
438 return (rqsbuf);
439 } else if (flag == 3 && snsbuf[7] >= 9 && (snsbuf[15] & 0x80)) {
440 /*
441 * SKSV Data
442 */
443 switch (skey) {
444 case SKEY_ILLEGAL_REQUEST:
445 if (snsbuf[15] & 0x8)
446 (void) snprintf(rqsbuf, rqsbuflen,
447 "Error in %s, Offset %d, bit %d",
448 (snsbuf[15] & 0x40)? "CDB" : "Parameters",
449 (snsbuf[16] & 0xff) << 8 |
450 (snsbuf[17] & 0xff), snsbuf[15] & 0x7);
451 else
452 (void) snprintf(rqsbuf, rqsbuflen,
453 "Error in %s, Offset %d",
454 (snsbuf[15] & 0x40)? "CDB" : "Parameters",
455 (snsbuf[16] & 0xff) << 8 |
456 (snsbuf[17] & 0xff));
457 return (rqsbuf);
458
459 case SKEY_RECOVERED_ERROR:
460 case SKEY_MEDIUM_ERROR:
461 case SKEY_HARDWARE_ERROR:
462 (void) snprintf(rqsbuf, rqsbuflen,
463 "Actual Retry Count: %d",
464 (snsbuf[16] & 0xff) << 8 | (snsbuf[17] & 0xff));
465 return (rqsbuf);
466
467 case SKEY_NOT_READY:
468 (void) snprintf(rqsbuf, rqsbuflen,
469 "Progress Indicator: %d",
470 (snsbuf[16] & 0xff) << 8 | (snsbuf[17] & 0xff));
471 return (rqsbuf);
472
473 default:
474 break;
475 }
476 }
477 return (NULL);
478}

480void
481scsi_print_sense(const char *name, const scsireq_t *req, int verbosity)
482{
483 int i;
484
485 printf("%s: Check Condition on CDB:", name);
486 for (i = 0; i < req->cmdlen; i++)
487 printf(" %02x", req->cmd[i]);
488 printf("\n");
489 scsi_print_sense_data(req->sense, req->senselen_used, verbosity);
490}
