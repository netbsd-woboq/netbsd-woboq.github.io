59#define STRVIS_ISWHITE(x) ((x) == ' ' || (x) == '\0' || (x) == (u_char)'\377')

61void
62scsi_command(int fd, const void *cmd, size_t cmdlen,
63 void *data, size_t datalen, int timeout, int flags)
64{
65 scsireq_t req;
66
67 memset(&req, 0, sizeof(req));
68
69 cmdlen = MIN(cmdlen, sizeof(req.cmd));
70 memcpy(req.cmd, cmd, cmdlen);
71 req.cmdlen = cmdlen;
72 req.databuf = data;
73 req.datalen = datalen;
74 req.timeout = timeout;
75 req.flags = flags;
76 req.senselen = SENSEBUFLEN;
77
78 if (ioctl(fd, SCIOCCOMMAND, &req) == -1)
79 err(1, "SCIOCCOMMAND");
80
81 if (req.retsts == SCCMD_OK)
82 return;
83
84 /* Some problem; report it and exit. */
85 if (req.retsts == SCCMD_TIMEOUT)
86 fprintf(stderr, "%s: SCSI command timed out\n", dvname);
87 else if (req.retsts == SCCMD_BUSY)
88 fprintf(stderr, "%s: device is busy\n", dvname);
89 else if (req.retsts == SCCMD_SENSE)
90 scsi_print_sense(dvname, &req, 1);
91 else
92 fprintf(stderr, "%s: device had unknown status %x\n", dvname,
93 req.retsts);
94
95 exit(1);
96}

98void
99scsi_mode_sense(int fd, u_int8_t pgcode, u_int8_t pctl, void *buf, size_t len)
100{
101 struct scsi_mode_sense_6 cmd;
102
103 memset(&cmd, 0, sizeof(cmd));
104 memset(buf, 0, len);
105
106 cmd.opcode = SCSI_MODE_SENSE_6;
107 cmd.page = pgcode | pctl;
108 cmd.length = len;
109
110 scsi_command(fd, &cmd, sizeof(cmd), buf, len, 10000, SCCMD_READ);
111}

140void
141scsi_strvis(char *sdst, size_t dlen, const char *ssrc, size_t slen)
142{
143 u_char *dst = (u_char *)sdst;
144 const u_char *src = (const u_char *)ssrc;
145
146 /* Trim leading and trailing blanks and NULs. */
147 while (slen > 0 && STRVIS_ISWHITE(src[0]))
148 ++src, --slen;
149 while (slen > 0 && STRVIS_ISWHITE(src[slen - 1]))
150 --slen;
151
152 while (slen > 0) {
153 if (*src < 0x20 || *src >= 0x80) {
154 /* non-printable characters */
155 dlen -= 4;
156 if (dlen < 1)
157 break;
158 *dst++ = '\\';
159 *dst++ = ((*src & 0300) >> 6) + '0';
160 *dst++ = ((*src & 0070) >> 3) + '0';
161 *dst++ = ((*src & 0007) >> 0) + '0';
162 } else if (*src == '\\') {
163 /* quote characters */
164 dlen -= 2;
165 if (dlen < 1)
166 break;
167 *dst++ = '\\';
168 *dst++ = '\\';
169 } else {
170 /* normal characters */
171 if (--dlen < 1)
172 break;
173 *dst++ = *src;
174 }
175 ++src, --slen;
176 }
177
178 *dst++ = 0;
179}
