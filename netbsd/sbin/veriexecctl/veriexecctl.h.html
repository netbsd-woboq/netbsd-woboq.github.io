39#define dict_gets(d, k) \
40 prop_string_cstring_nocopy(prop_dictionary_get(d, k))

43 prop_dictionary_set(d, k, prop_data_create_data(v, n))

45#define dict_getd(d, k) \
46 prop_data_data_nocopy(prop_dictionary_get(d, k))
