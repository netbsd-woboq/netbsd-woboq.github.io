45static int border;
46static int dpytype;
47static struct wsdisplay_usefontdata font;
48static struct wsdisplay_param backlight;
49static struct wsdisplay_param brightness;
50static struct wsdisplay_param contrast;
51static struct wsdisplay_scroll_data scroll_l;
52static int msg_default_attrs, msg_default_bg, msg_default_fg;
53static int msg_kernel_attrs, msg_kernel_bg, msg_kernel_fg;
54static int splash_enable, splash_progress;

56struct field display_field_tab[] = {
57 { "border", &border, FMT_COLOR, 0 },
58 { "type", &dpytype, FMT_DPYTYPE, FLG_RDONLY },
59 { "font", &font.name, FMT_STRING, FLG_WRONLY },
60 { "backlight", &backlight.curval, FMT_UINT, 0 },
61 { "brightness", &brightness.curval, FMT_UINT, FLG_MODIFY },
62 { "contrast", &contrast.curval, FMT_UINT, FLG_MODIFY },
63 { "scroll.fastlines", &scroll_l.fastlines, FMT_UINT, FLG_MODIFY },
64 { "scroll.slowlines", &scroll_l.slowlines, FMT_UINT, FLG_MODIFY },
65 { "msg.default.attrs", &msg_default_attrs, FMT_ATTRS, 0 },
66 { "msg.default.bg", &msg_default_bg, FMT_COLOR, 0 },
67 { "msg.default.fg", &msg_default_fg, FMT_COLOR, 0 },
68 { "msg.kernel.attrs", &msg_kernel_attrs, FMT_ATTRS, 0 },
69 { "msg.kernel.bg", &msg_kernel_bg, FMT_COLOR, 0 },
70 { "msg.kernel.fg", &msg_kernel_fg, FMT_COLOR, 0 },
71 { "splash.enable", &splash_enable, FMT_UINT, FLG_WRONLY },
72 { "splash.progress", &splash_progress, FMT_UINT, FLG_WRONLY },
73};

75int display_field_tab_len = sizeof(display_field_tab) /
76 sizeof(display_field_tab[0]);

78void
79display_get_values(int fd)
80{
81
82 if (field_by_value(&dpytype)->flags & FLG_GET)
83 if (ioctl(fd, WSDISPLAYIO_GTYPE, &dpytype) < 0)
84 err(EXIT_FAILURE, "WSDISPLAYIO_GTYPE");
85
86 if (field_by_value(&border)->flags & FLG_GET)
87 if (ioctl(fd, WSDISPLAYIO_GBORDER, &border) < 0)
88 field_disable_by_value(&border);
89
90 if (field_by_value(&backlight.curval)->flags & FLG_GET) {
91 backlight.param = WSDISPLAYIO_PARAM_BACKLIGHT;
92 if (ioctl(fd, WSDISPLAYIO_GETPARAM, &backlight) < 0)
93 field_disable_by_value(&backlight.curval);
94 }
95
96 if (field_by_value(&brightness.curval)->flags & FLG_GET) {
97 brightness.param = WSDISPLAYIO_PARAM_BRIGHTNESS;
98 if (ioctl(fd, WSDISPLAYIO_GETPARAM, &brightness))
99 field_disable_by_value(&brightness.curval);
100 }
101
102 if (field_by_value(&contrast.curval)->flags & FLG_GET) {
103 contrast.param = WSDISPLAYIO_PARAM_CONTRAST;
104 if (ioctl(fd, WSDISPLAYIO_GETPARAM, &contrast))
105 field_disable_by_value(&contrast.curval);
106 }
107
108 if (field_by_value(&msg_default_attrs)->flags & FLG_GET ||
109 field_by_value(&msg_default_bg)->flags & FLG_GET ||
110 field_by_value(&msg_default_fg)->flags & FLG_GET ||
111 field_by_value(&msg_kernel_attrs)->flags & FLG_GET ||
112 field_by_value(&msg_kernel_bg)->flags & FLG_GET ||
113 field_by_value(&msg_kernel_fg)->flags & FLG_GET) {
114 struct wsdisplay_msgattrs ma;
115
116 if (ioctl(fd, WSDISPLAYIO_GMSGATTRS, &ma) < 0) {
117 field_disable_by_value(&msg_default_attrs);
118 field_disable_by_value(&msg_default_bg);
119 field_disable_by_value(&msg_default_fg);
120 field_disable_by_value(&msg_kernel_attrs);
121 field_disable_by_value(&msg_kernel_bg);
122 field_disable_by_value(&msg_kernel_fg);
123 } else {
124 msg_default_attrs = ma.default_attrs;
125 if (ma.default_attrs & WSATTR_WSCOLORS) {
126 msg_default_bg = ma.default_bg;
127 msg_default_fg = ma.default_fg;
128 } else
129 msg_default_bg = msg_default_fg = -1;
130
131 msg_kernel_attrs = ma.kernel_attrs;
132 if (ma.kernel_attrs & WSATTR_WSCOLORS) {
133 msg_kernel_bg = ma.kernel_bg;
134 msg_kernel_fg = ma.kernel_fg;
135 } else
136 msg_kernel_bg = msg_kernel_fg = -1;
137 }
138 }
139
140 if (field_by_value(&scroll_l.fastlines)->flags & FLG_GET ||
141 field_by_value(&scroll_l.slowlines)->flags & FLG_GET) {
142 if (ioctl(fd, WSDISPLAYIO_DGSCROLL, &scroll_l) < 0) {
143 field_disable_by_value(&scroll_l.fastlines);
144 field_disable_by_value(&scroll_l.slowlines);
145 }
146 }
147}

149void
150display_put_values(int fd)
151{
152
153 if (field_by_value(&font.name)->flags & FLG_SET) {
154 if (ioctl(fd, WSDISPLAYIO_SFONT, &font) < 0)
155 err(EXIT_FAILURE, "WSDISPLAYIO_SFONT");
156 pr_field(field_by_value(&font.name), " -> ");
157 }
158
159 if (field_by_value(&border)->flags & FLG_SET) {
160 if (ioctl(fd, WSDISPLAYIO_SBORDER, &border) < 0)
161 err(EXIT_FAILURE, "WSDISPLAYIO_SBORDER");
162 pr_field(field_by_value(&border), " -> ");
163 }
164
165 if (field_by_value(&backlight.curval)->flags & FLG_SET) {
166 backlight.param = WSDISPLAYIO_PARAM_BACKLIGHT;
167 if (ioctl(fd, WSDISPLAYIO_SETPARAM, &backlight) < 0)
168 err(EXIT_FAILURE, "WSDISPLAYIO_PARAM_BACKLIGHT");
169 pr_field(field_by_value(&backlight.curval), " -> ");
170 }
171
172 if (field_by_value(&brightness.curval)->flags & FLG_SET) {
173 brightness.param = WSDISPLAYIO_PARAM_BRIGHTNESS;
174 if (ioctl(fd, WSDISPLAYIO_SETPARAM, &brightness) < 0)
175 err(EXIT_FAILURE, "WSDISPLAYIO_PARAM_BRIGHTNESS");
176 pr_field(field_by_value(&brightness.curval), " -> ");
177 }
178
179 if (field_by_value(&contrast.curval)->flags & FLG_SET) {
180 contrast.param = WSDISPLAYIO_PARAM_CONTRAST;
181 if (ioctl(fd, WSDISPLAYIO_SETPARAM, &contrast) < 0)
182 err(EXIT_FAILURE, "WSDISPLAYIO_PARAM_CONTRAST");
183 pr_field(field_by_value(&contrast.curval), " -> ");
184 }
185
186 if (field_by_value(&splash_enable)->flags & FLG_SET) {
187 if (ioctl(fd, WSDISPLAYIO_SSPLASH, &splash_enable) < 0)
188 err(EXIT_FAILURE, "WSDISPLAYIO_SSPLASH");
189 pr_field(field_by_value(&splash_enable), " -> ");
190 }
191
192 if (field_by_value(&splash_progress)->flags & FLG_SET) {
193 if (ioctl(fd, WSDISPLAYIO_SPROGRESS, &splash_progress) < 0)
194 err(EXIT_FAILURE, "WSDISPLAYIO_SPROGRESS");
195 pr_field(field_by_value(&splash_progress), " -> ");
196 }
197
198 if (field_by_value(&msg_default_attrs)->flags & FLG_SET ||
199 field_by_value(&msg_default_bg)->flags & FLG_SET ||
200 field_by_value(&msg_default_fg)->flags & FLG_SET ||
201 field_by_value(&msg_kernel_attrs)->flags & FLG_SET ||
202 field_by_value(&msg_kernel_bg)->flags & FLG_SET ||
203 field_by_value(&msg_kernel_fg)->flags & FLG_SET) {
204 struct wsdisplay_msgattrs ma;
205
206 if (ioctl(fd, WSDISPLAYIO_GMSGATTRS, &ma) < 0)
207 err(EXIT_FAILURE, "WSDISPLAYIO_GMSGATTRS");
208
209 if (field_by_value(&msg_default_attrs)->flags & FLG_SET) {
210 ma.default_attrs = msg_default_attrs;
211 pr_field(field_by_value(&msg_default_attrs), " -> ");
212 }
213 if (ma.default_attrs & WSATTR_WSCOLORS) {
214 if (field_by_value(&msg_default_bg)->flags & FLG_SET) {
215 ma.default_bg = msg_default_bg;
216 pr_field(field_by_value(&msg_default_bg),
217 " -> ");
218 }
219 if (field_by_value(&msg_default_fg)->flags & FLG_SET) {
220 ma.default_fg = msg_default_fg;
221 pr_field(field_by_value(&msg_default_fg),
222 " -> ");
223 }
224 }
225
226 if (field_by_value(&msg_kernel_attrs)->flags & FLG_SET) {
227 ma.kernel_attrs = msg_kernel_attrs;
228 pr_field(field_by_value(&msg_kernel_attrs), " -> ");
229 }
230 if (ma.default_attrs & WSATTR_WSCOLORS) {
231 if (field_by_value(&msg_kernel_bg)->flags & FLG_SET) {
232 ma.kernel_bg = msg_kernel_bg;
233 pr_field(field_by_value(&msg_kernel_bg),
234 " -> ");
235 }
236 if (field_by_value(&msg_kernel_fg)->flags & FLG_SET) {
237 ma.kernel_fg = msg_kernel_fg;
238 pr_field(field_by_value(&msg_kernel_fg),
239 " -> ");
240 }
241 }
242
243 if (ioctl(fd, WSDISPLAYIO_SMSGATTRS, &ma) < 0)
244 err(EXIT_FAILURE, "WSDISPLAYIO_SMSGATTRS");
245 }
246
247 scroll_l.which = 0;
248 if (field_by_value(&scroll_l.fastlines)->flags & FLG_SET)
249 scroll_l.which |= WSDISPLAY_SCROLL_DOFASTLINES;
250 if (field_by_value(&scroll_l.slowlines)->flags & FLG_SET)
251 scroll_l.which |= WSDISPLAY_SCROLL_DOSLOWLINES;
252 if (scroll_l.which != 0 &&
253 ioctl(fd, WSDISPLAYIO_DSSCROLL, &scroll_l) < 0)
254 err(EXIT_FAILURE, "WSDISPLAYIO_DSSCROLL");
255 if (scroll_l.which & WSDISPLAY_SCROLL_DOFASTLINES)
256 pr_field(field_by_value(&scroll_l.fastlines), " -> ");
257 if (scroll_l.which & WSDISPLAY_SCROLL_DOSLOWLINES)
258 pr_field(field_by_value(&scroll_l.slowlines), " -> ");
259}
