30#ifndef _ALTQ_ALTQ_FIFOQ_H_
31#define _ALTQ_ALTQ_FIFOQ_H_
32
33#ifdef _KERNEL
34typedef struct fifoq_state {
35 struct fifoq_state *q_next; /* next fifoq_state in the list */
36 struct ifaltq *q_ifq; /* backpointer to ifaltq */
37
38 struct mbuf *q_head; /* head of queue */
39 struct mbuf *q_tail; /* tail of queue */
40 int q_len; /* queue length */
41 int q_limit; /* max queue length */
42
43 /* statistics */
44 struct {
45 struct pktcntr xmit_cnt;
46 struct pktcntr drop_cnt;
47 u_int period;
48 } q_stats;
49} fifoq_state_t;
50#endif
51
52struct fifoq_interface {
53 char fifoq_ifname[IFNAMSIZ];
54};
55
56struct fifoq_getstats {
57 struct fifoq_interface iface;
58 int q_len;
59 int q_limit;
60 struct pktcntr xmit_cnt;
61 struct pktcntr drop_cnt;
62 u_int period;
63};
64
65struct fifoq_conf {
66 struct fifoq_interface iface;
67 int fifoq_limit;
68};
69
70#define FIFOQ_LIMIT 50 /* default max queue length */
71
72/*
73 * IOCTLs for FIFOQ
74 */
75#define FIFOQ_IF_ATTACH _IOW('Q', 1, struct fifoq_interface)
76#define FIFOQ_IF_DETACH _IOW('Q', 2, struct fifoq_interface)
77#define FIFOQ_ENABLE _IOW('Q', 3, struct fifoq_interface)
78#define FIFOQ_DISABLE _IOW('Q', 4, struct fifoq_interface)
79#define FIFOQ_CONFIG _IOWR('Q', 6, struct fifoq_conf)
80#define FIFOQ_GETSTATS _IOWR('Q', 12, struct fifoq_getstats)
81
82#endif /* _ALTQ_ALTQ_FIFOQ_H_ */
