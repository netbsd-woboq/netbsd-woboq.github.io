152/*
153 * Attach the mainbus.
154 */
155void
156amd64_mainbus_attach(device_t parent, device_t self, void *aux)
157{
158#if NISA > 0 || NPCI > 0 || NACPICA > 0 || NIPMI > 0
159 union amd64_mainbus_attach_args mba;
160#endif
161
162#if NISADMA > 0 && NACPICA > 0
163 /*
164 * ACPI needs ISA DMA initialized before they start probing.
165 */
166 isa_dmainit(&x86_isa_chipset, x86_bus_space_io, &isa_bus_dma_tag,
167 self);
168#endif
169
170#if NACPICA > 0
171 if (acpi_present) {
172 mba.mba_acpi.aa_iot = x86_bus_space_io;
173 mba.mba_acpi.aa_memt = x86_bus_space_mem;
174 mba.mba_acpi.aa_pc = NULL;
175 mba.mba_acpi.aa_pciflags =
176 PCI_FLAGS_IO_OKAY | PCI_FLAGS_MEM_OKAY |
177 PCI_FLAGS_MRL_OKAY | PCI_FLAGS_MRM_OKAY |
178 PCI_FLAGS_MWI_OKAY;
179 mba.mba_acpi.aa_ic = &x86_isa_chipset;
180 mba.mba_acpi.aa_dmat = &pci_bus_dma_tag;
181 mba.mba_acpi.aa_dmat64 = &pci_bus_dma64_tag;
182 config_found_ia(self, "acpibus", &mba.mba_acpi, 0);
183 }
184#endif
185
186#if NIPMI > 0
187 memset(&mba.mba_ipmi, 0, sizeof(mba.mba_ipmi));
188 mba.mba_ipmi.iaa_iot = x86_bus_space_io;
189 mba.mba_ipmi.iaa_memt = x86_bus_space_mem;
190 if (ipmi_probe(&mba.mba_ipmi))
191 config_found_ia(self, "ipmibus", &mba.mba_ipmi, 0);
192#endif
193
194#if NPCI > 0
195 if (pci_mode_detect() != 0) {
196 int npcibus = 0;
197
198 mba.mba_pba.pba_iot = x86_bus_space_io;
199 mba.mba_pba.pba_memt = x86_bus_space_mem;
200 mba.mba_pba.pba_dmat = &pci_bus_dma_tag;
201 mba.mba_pba.pba_dmat64 = &pci_bus_dma64_tag;
202 mba.mba_pba.pba_pc = NULL;
203 mba.mba_pba.pba_flags =
204 PCI_FLAGS_IO_OKAY | PCI_FLAGS_MEM_OKAY |
205 PCI_FLAGS_MRL_OKAY | PCI_FLAGS_MRM_OKAY |
206 PCI_FLAGS_MWI_OKAY;
207 mba.mba_pba.pba_bus = 0;
208 mba.mba_pba.pba_bridgetag = NULL;
209#if NACPICA > 0 && defined(ACPI_SCANPCI)
210 if (npcibus == 0 && mpacpi_active)
211 npcibus = mp_pci_scan(self, &mba.mba_pba, pcibusprint);
212#endif
213#if defined(MPBIOS) && defined(MPBIOS_SCANPCI)
214 if (npcibus == 0 && mpbios_scanned != 0)
215 npcibus = mp_pci_scan(self, &mba.mba_pba, pcibusprint);
216#endif
217 if (npcibus == 0)
218 config_found_ia(self, "pcibus", &mba.mba_pba,
219 pcibusprint);
220
221#if NACPICA > 0
222 if (mp_verbose)
223 acpi_pci_link_state();
224#endif
225 }
226#endif
227
228#if NISA > 0
229 if (isa_has_been_seen == 0) {
230 mba.mba_iba = mba_iba;
231 mba.mba_iba.iba_iot = x86_bus_space_io;
232 mba.mba_iba.iba_memt = x86_bus_space_mem;
233 config_found_ia(self, "isabus", &mba.mba_iba, isabusprint);
234 }
235#endif
236
237 if (!pmf_device_register(self, NULL, NULL))
238 aprint_error_dev(self, "couldn't establish power handler\n");
239}
240
241int
242amd64_mainbus_print(void *aux, const char *pnp)
243{
244 union amd64_mainbus_attach_args *mba = aux;
245
246 if (pnp)
247 aprint_normal("%s at %s", mba->mba_busname, pnp);
248 return UNCONF;
249}
