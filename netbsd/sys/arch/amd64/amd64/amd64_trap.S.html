81/*
82 * Trap and fault vector routines
83 *
84 * On exit from the kernel to user mode, we always need to check for ASTs. In
85 * addition, we need to do this atomically; otherwise an interrupt may occur
86 * which causes an AST, but it won't get processed until the next kernel entry
87 * (possibly the next clock tick). Thus, we disable interrupt before checking,
88 * and only enable them again on the final `iret' or before calling the AST
89 * handler.
90 */
91

109/*
110 * ASM macro, used to leave the IST3 stack and to put ourselves on a non-IST
111 * stack. Only RDX, RCX and RAX are allowed to be used.
112 *
113 * +------------------------------+
114 * The iret frame we copy is: | rip | cs | rflags | rsp | ss |
115 * +------------------------------+
116 */
117.macro IST3_LEAVE is_user
118 .if \is_user
119 movq CPUVAR(CURLWP),%rax
120 movq L_PCB(%rax),%rax
121 movq PCB_RSP0(%rax),%rax
122 .else
123 movq TF_RSP(%rsp),%rax
124 andq $(~0xF),%rax
125 .endif
126
127 subq $(5*8),%rax
128 movq %rax,CPUVAR(SCRATCH)
129
130 /* Copy the iret frame. */
131 movq TF_SS(%rsp),%rcx
132 movq %rcx,(4*8)(%rax)
133 movq TF_RSP(%rsp),%rcx
134 movq %rcx,(3*8)(%rax)
135 movq TF_RFLAGS(%rsp),%rcx
136 movq %rcx,(2*8)(%rax)
137 movq TF_CS(%rsp),%rcx
138 movq %rcx,(1*8)(%rax)
139 movq TF_RIP(%rsp),%rcx
140 movq %rcx,(0*8)(%rax)
141
142 /* Restore. */
143 movq TF_RDX(%rsp),%rdx
144 movq TF_RCX(%rsp),%rcx
145 movq TF_RAX(%rsp),%rax
146
147 /* Zero out the stack we used, RDX+RCX+RAX+IRET. */
148 movq $0,TF_RDX(%rsp)
149 movq $0,TF_RCX(%rsp)
150 movq $0,TF_RAX(%rsp)
151 movq $0,TF_RIP(%rsp)
152 movq $0,TF_CS(%rsp)
153 movq $0,TF_RFLAGS(%rsp)
154 movq $0,TF_RSP(%rsp)
155 movq $0,TF_SS(%rsp)
156
157 movq CPUVAR(SCRATCH),%rsp
158.endm

166/*
167 * Handle the SS shadow, CVE-2018-8897.
168 *
169 * We are running on the IST3 stack. If we are under an SS shadow, ignore
170 * the exception and return immediately. Otherwise, copy the iret frame
171 * onto the non-IST stack, and ZTRAP on it as usual.
172 *
173 * IST3 is used temporarily, and is mapped in userland by SVS. It contains
174 * a few secrets, the values of the CPU context. These secrets are zeroed
175 * out when we leave.
176 *
177 * When we ignore an SS shadow, we can't zero out the iret frame. It is
178 * not a problem, because in this particular case, the frame is known not
179 * to contain secrets.
180 */
181IDTVEC(trap01)
182#ifndef XENPV
183 subq $(TF_REGSIZE+16),%rsp
184
185 /* We clobber only RDX, RCX and RAX. */
186 movq %rdx,TF_RDX(%rsp)
187 movq %rcx,TF_RCX(%rsp)
188 movq %rax,TF_RAX(%rsp)
189
190 testb $SEL_UPL,TF_CS(%rsp)
191 jnz .Luser_dbentry
192
193 movl $MSR_GSBASE,%ecx
194 rdmsr
195 cmpl $VM_SPACE_SEP_HIGH32,%edx
196 jae .Lkern_dbentry
197
198 /* SS shadow, ignore the exception. */
199 xorq %rax,%rax
200 movq %rax,%dr6
201
202 /* Restore and zero out. */
203 movq TF_RDX(%rsp),%rdx
204 movq TF_RCX(%rsp),%rcx
205 movq TF_RAX(%rsp),%rax
206 movq $0,TF_RDX(%rsp)
207 movq $0,TF_RCX(%rsp)
208 movq $0,TF_RAX(%rsp)
209
210 addq $(TF_REGSIZE+16),%rsp
211 iretq
212
213.Lkern_dbentry:
214 IST3_LEAVE 0
215 ZTRAP(T_TRCTRAP)
216
217.Luser_dbentry:
218 swapgs
219 SVS_ENTER_ALTSTACK
220 IST3_LEAVE 1
221 ZTRAP_NJ(T_TRCTRAP)
222 subq $TF_REGSIZE,%rsp
223 INTR_SAVE_GPRS
224 cld
225 SMAP_ENABLE
226 IBRS_ENTER
227 movw %gs,TF_GS(%rsp)
228 movw %fs,TF_FS(%rsp)
229 movw %es,TF_ES(%rsp)
230 movw %ds,TF_DS(%rsp)
231
232 jmp .Lalltraps_noentry
233#else
234 ZTRAP(T_TRCTRAP)
235#endif
236IDTVEC_END(trap01)

238/*
239 * Non Maskable Interrupts are a special case: they can be triggered even
240 * with interrupts disabled, and once triggered they block further NMIs
241 * until an 'iret' instruction is executed.
242 *
243 * Therefore we don't enable interrupts, because the CPU could switch to
244 * another LWP, call 'iret' and unintentionally leave the NMI mode.
245 *
246 * We need to be careful about %gs too, because it is possible that we were
247 * running in kernel mode with a userland %gs.
248 */

259 IBRS_ENTER

261

263 SMAP_ENABLE
264 movw %gs,TF_GS(%rsp)

266 movw %es,TF_ES(%rsp)

505#ifndef kernuser_reenter
506/*
507 * We need to worry about traps in kernel mode while the kernel %gs isn't
508 * loaded. When such traps happen, we have CPL=0 and %gs=userland, and we
509 * must perform an additional swapgs to get %gs=kernel.
510 */
511
512#define TF_SMALL(val, reg) (val - TF_REGSIZE)(reg)
513#define TF_SMALL_REGPUSHED(val, reg) (val - (TF_REGSIZE - 8))(reg)
514
515/*
516 * It is possible that we received a trap in kernel mode, but with the user
517 * context loaded. There are five cases where this can happen:
518 *
519 * o Execution of IRETQ.
520 * o Reload of ES.
521 * o Reload of DS.
522 * o Reload of FS.
523 * o Reload of GS.
524 *
525 * When this happens, the kernel is re-entered in kernel mode, but the
526 * previous context is in kernel mode too.
527 *
528 * We have two iret frames in the stack. In the first one, we also pushed
529 * 'trapno' and 'err'. The 'rsp' field points to the outer iret frame:
530 *
531 * +---------------------------------------------------+
532 * | trapno | err | rip | cs=ring0 | rflags | rsp | ss |
533 * +-------------------------------------------|-------+
534 * |
535 * +---------------------------------+
536 * |
537 * | +------------------------------------+
538 * +--> | rip | cs=ring3 | rflags | rsp | ss |
539 * +------------------------------------+
540 *
541 * We perform a three-step procedure:
542 *
543 * o We update RSP to point to the outer frame. This outer frame is in the
544 * same stack as the current frame, and likely just after the current
545 * frame.
546 *
547 * o We push, in this outer frame, the 'err' and 'trapno' fields of the
548 * CURRENT frame.
549 *
550 * o We do a normal INTRENTRY. Now that RSP points to the outer frame,
551 * everything behaves as if we had received a trap from the outer frame,
552 * that is to say, from userland directly.
553 *
554 * Finally, we jump to 'calltrap' and handle the trap smoothly.
555 *
556 * Two notes regarding SVS:
557 *
558 * o With SVS, we will receive the trap while the user page tables are
559 * loaded. That's not a problem, we don't touch anything unmapped here.
560 *
561 * o With SVS, when the user page tables are loaded, the stack is really
562 * small, and can contain only one trapframe structure. Therefore, in
563 * intrfastexit, we must save the GPRs and pop their part of the stack
564 * right away. If we weren't doing that, and the reload of ES faulted for
565 * example, then the CPU would try to push an iret frame on the current
566 * stack (nested), and would double-fault because it touches the redzone
567 * below the stack (see the documentation in x86/x86/svs.c). By popping
568 * the GPR part of the stack, we leave enough stack for the CPU to push
569 * an iret frame, and for us to push one 8-byte register (%rdi) too.
570 */
571 _ALIGN_TEXT
572LABEL(kernuser_reenter)
573 testb $SEL_UPL,TF_SMALL(TF_CS, %rsp)
574 jz .Lkernelmode
575
576.Lnormal_entry:
577 INTRENTRY
578 sti
579 jmp calltrap
580
581.Lkernelmode:
582 /* We will clobber %rdi */
583 pushq %rdi
584
585 /* Case 1: fault on iretq? */
586 leaq do_iret(%rip),%rdi
587 cmpq %rdi,TF_SMALL_REGPUSHED(TF_RIP, %rsp)
588 jne 5f
589 movq TF_SMALL_REGPUSHED(TF_RSP, %rsp),%rdi /* get %rsp */
590 testb $SEL_UPL,8(%rdi) /* check %cs of outer iret frame */
591 je .Lnormal_entry /* jump if iret was to kernel */
592 jmp .Lkernelmode_but_user /* to user - must restore %gs */
5935:
594
595 /* Case 2: move to %es? */
596 leaq do_mov_es(%rip),%rdi
597 cmpq %rdi,TF_SMALL_REGPUSHED(TF_RIP, %rsp)
598 je .Lkernelmode_but_user
599
600 /* Case 3: move to %ds? */
601 leaq do_mov_ds(%rip),%rdi
602 cmpq %rdi,TF_SMALL_REGPUSHED(TF_RIP, %rsp)
603 je .Lkernelmode_but_user
604
605 /* Case 4: move to %fs? */
606 leaq do_mov_fs(%rip),%rdi
607 cmpq %rdi,TF_SMALL_REGPUSHED(TF_RIP, %rsp)
608 je .Lkernelmode_but_user
609
610 /* Case 5: move to %gs? */
611 leaq do_mov_gs(%rip),%rdi
612 cmpq %rdi,TF_SMALL_REGPUSHED(TF_RIP, %rsp)
613 je .Lkernelmode_but_user
614
615 /* None of the above cases: normal kernel fault */
616 popq %rdi
617 jmp .Lnormal_entry
618
619.Lkernelmode_but_user:
620 /*
621 * Here we have %rdi pushed on the stack, hence 8+.
622 */
623 movq %rsp,%rdi
624 movq TF_SMALL_REGPUSHED(TF_RSP, %rsp),%rsp
625
626 /* Push tf_err and tf_trapno */
627 pushq 8+8(%rdi) /* 8+8(%rdi) = current TF_ERR */
628 pushq 8+0(%rdi) /* 8+0(%rdi) = current TF_TRAPNO */
629
630 /* Restore %rdi */
631 movq (%rdi),%rdi
632
633 jmp .Lnormal_entry
634END(kernuser_reenter)
635#endif

6733: CHECK_DEFERRED_SWITCH

675
6766:
