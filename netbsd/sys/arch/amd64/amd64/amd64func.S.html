29/*
30 * Functions to provide access to amd64-specific instructions.
31 *
32 * These are explicitly _not_ shared with xen/amd64.
33 */
34
35#include <machine/asm.h>
36
37#include "assym.h"
38
39ENTRY(tsc_get_timecount)
40 movq CPUVAR(CURLWP), %rcx
411:
42 movq L_NCSW(%rcx), %rdi
43 rdtsc
44 addl CPUVAR(CC_SKEW), %eax
45 cmpq %rdi, L_NCSW(%rcx)
46 jne 2f
47 ret
482:
49 jmp 1b
50END(tsc_get_timecount)
