32#include <machine/asm.h>
33#include "assym.h"
34
35ENTRY(cpu_in_cksum)
36 pushq %rbp
37 pushq %rbx
38
39 /*
40 * During most of the function the following values can
41 * be found in the registers:
42 *
43 * %rdi: The current element in the mbuf chain.
44 * %esi: Remaining bytes to check after the current mbuf.
45 * %ebp: Minimum of %esi at the start of the loop and the
46 * length of the current mbuf.
47 * %r8: Overall sum. Carry must be handled on increment.
48 * %r9 and %r10: Partial sums. This are normally modified
49 * without carry check, see comment in inner loop.
50 * %rbx: Remaining data of current mbuf.
51 * %dh: Partial sum must be byte swapped before adding up.
52 * %dl: Current mbuf started at odd position. A word was split.
53 */
54
55 movl %ecx, %eax
56 movl %edx, %ecx
57 movq %rax, %r8
58 xorl %edx,%edx
59
60 /* All requested bytes checksummed? */
61 testl %esi, %esi
62 jz .Mdone
63
64.Mmbuf_preloop:
65 /* No more data to process? */
66 testq %rdi, %rdi
67 jz .Mout_of_mbufs
68 movl M_LEN(%rdi), %ebp
69 cmpl %ebp, %ecx
70 jbe 1f
71 subl %ebp, %ecx
72 movq M_NEXT(%rdi), %rdi
73 jmp .Mmbuf_preloop
741:
75 subl %ecx, %ebp
76 movq M_DATA(%rdi), %rbx
77 movl %ecx, %eax
78 addq %rax, %rbx
79 jmp .Mmbuf_load_data
80
81.Mmbuf_loop:
82 /* All requested bytes checksummed? */
83 testl %esi, %esi
84 jz .Mdone
85
86 /* No more data to process? */
87 testq %rdi, %rdi
88 jz .Mout_of_mbufs
89
90 movl M_LEN(%rdi), %ebp
91 movq M_DATA(%rdi), %rbx
92.Mmbuf_load_data:
93
94 /* Skip empty mbufs. */
95 testl %ebp, %ebp
96 jz .Mmbuf_loop_next
97
98 /* If this mbuf is longer than necessary, just truncate it. */
99 cmpl %ebp, %esi
100 cmovb %esi, %ebp
101 subl %ebp, %esi
102
103 xorq %r9, %r9
104 xorq %r10, %r10
105
106.Mmbuf_align_word:
107 /* Already aligned on a word boundary? */
108 testb $1, %bl
109 jz .Mmbuf_align_dword
110
111 /* Invert %dl. */
112 testb %dl, %dl
113 setz %dl
114
115 movzbl (%rbx), %ecx
116 xchgb %cl, %ch
117 addq %rcx, %r9
118 incq %rbx
119 decl %ebp
120
121.Mmbuf_align_dword:
122 /*
123 * If the current position is equivalent to an odd index,
124 * byte swap the partial sums at the end to compensate.
125 */
126 movb %dl, %dh
127
128 /*
129 * If the data is not already aligned at a dword boundary,
130 * just add the first word to one of the partial sums.
131 */
132 testb $2, %bl
133 jz .Mmbuf_inner_loop
134 cmpl $2, %ebp
135 jb .Mmbuf_trailing_bytes
136 movzwl (%rbx), %ecx
137 addq %rcx, %r9
138 leaq 2(%rbx), %rbx
139 leal -2(%ebp), %ebp
140
141.Mmbuf_inner_loop:
142 .align 16
143 /*
144 * Inner loop is unrolled to handle 32 byte at a time.
145 * Dwords are summed up in %r9 and %10 without checking
146 * for overflow. This exploits two adders and the order
147 * constraint on flags.
148 *
149 * After the summing up, %r9 and %r10 are merged and
150 * the sum is test for having either of the two highest
151 * bits set. If that is the case, the partial sum is added
152 * to the overall sum and both registers are zeroed.
153 */
154 cmpl $32, %ebp
155 jb .Mmbuf_trailing_owords
156 movl 0(%rbx), %ecx
157 movl 4(%rbx), %eax
158 addq %rcx, %r9
159 addq %rax, %r10
160
161 movl 8(%rbx), %ecx
162 movl 12(%rbx), %eax
163 addq %rcx, %r9
164 addq %rax, %r10
165
166 movl 16(%rbx), %ecx
167 movl 20(%rbx), %eax
168 addq %rcx, %r9
169 addq %rax, %r10
170
171 movl 24(%rbx), %ecx
172 movl 28(%rbx), %eax
173 addq %rcx, %r9
174 addq %rax, %r10
175
176 leaq 32(%rbx), %rbx
177 leal -32(%ebp), %ebp
178
179 addq %r9, %r10
180 movq %r10, %rax
181 shrq $62, %rax
182 xorq %r9, %r9
183 testb %al, %al
184 jz .Mmbuf_inner_loop
185
186 testb %dh, %dh
187 jz 1f
188 rolq $8, %r10
1891:
190 addq %r10, %r8
191 adcq $0, %r8
192 xorq %r10, %r10
193
194 jmp .Mmbuf_inner_loop
195
196 /*
197 * One more check for 16, 8, 4, 2 and 1 remaining
198 * byte in the mbuf...
199 *
200 * No more overflow checks needed here.
201 */
202.Mmbuf_trailing_owords:
203 testw $16, %bp
204 jz .Mmbuf_trailing_qwords
205 movl 0(%rbx), %ecx
206 movl 4(%rbx), %eax
207 addq %rcx, %r9
208 addq %rax, %r10
209
210 movl 8(%rbx), %ecx
211 movl 12(%rbx), %eax
212 addq %rcx, %r9
213 addq %rax, %r10
214
215 leaq 16(%rbx), %rbx
216
217.Mmbuf_trailing_qwords:
218 testw $8, %bp
219 jz .Mmbuf_trailing_dwords
220 movl 0(%rbx), %ecx
221 movl 4(%rbx), %eax
222 addq %rcx, %r9
223 addq %rax, %r10
224
225 leaq 8(%rbx), %rbx
226
227.Mmbuf_trailing_dwords:
228 testw $4, %bp
229 jz .Mmbuf_trailing_words
230 movl (%rbx), %ecx
231 addq %rcx, %r9
232 leaq 4(%rbx), %rbx
233
234.Mmbuf_trailing_words:
235 testw $2, %bp
236 jz .Mmbuf_trailing_bytes
237 movzwl (%rbx), %ecx
238 addq %rcx, %r9
239 leaq 2(%rbx), %rbx
240
241.Mmbuf_trailing_bytes:
242 testw $1, %bp
243 jz .Mbyte_swap
244 movzbl (%rbx), %ecx
245 addq %rcx, %r9
246 /* Invert %dl as this is a split in a word. */
247 testb %dl, %dl
248 setz %dl
249
250.Mbyte_swap:
251 /* Byte swap by 8 bit rotate. */
252 testb %dh, %dh
253 jz 1f
254 rolq $8, %r9
255 rolq $8, %r10
2561:
257 addq %r10, %r8
258 adcq %r9, %r8
259 adcq $0, %r8
260
261.Mmbuf_loop_next:
262 movq M_NEXT(%rdi), %rdi
263 jmp .Mmbuf_loop
264
265.Mdone:
266 /*
267 * Reduce 64 bit overall sum into 16 bit sum and
268 * return the complement.
269 */
270 movq %r8, %rax
271 movq %r8, %rbx
272 shrq $32, %rax
273 addl %eax, %ebx
274 adcl $0, %ebx
275 movzwl %bx, %eax
276 shrl $16, %ebx
277 addw %ax, %bx
278 adcw $0, %bx
279 movw %bx, %ax
280 notw %ax
281
282.Mreturn:
283 popq %rbx
284 popq %rbp
285 ret
286
287.Mout_of_mbufs:
288#ifdef __PIC__
289 leaq .Mout_of_mbufs_msg(%rip), %rdi
290#else
291 movq $.Mout_of_mbufs_msg, %rdi
292#endif
293 movl $0, %eax
294 call PIC_PLT(_C_LABEL(printf))
295 jmp .Mreturn
296END(cpu_in_cksum)
297
298 .section .rodata
299 .type .Mout_of_mbufs_msg, @object
300.Mout_of_mbufs_msg:
301 .string "in_cksum: out of data\n"
302END(.Mout_of_mbufs_msg)
