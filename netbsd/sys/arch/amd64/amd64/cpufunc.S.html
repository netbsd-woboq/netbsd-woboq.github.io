105/*
106 * Big hammer: flush all TLB entries, including ones from PTE's
107 * with the G bit set. This should only be necessary if TLB
108 * shootdown falls far behind.
109 *
110 * Intel Architecture Software Developer's Manual, Volume 3,
111 * System Programming, section 9.10, "Invalidating the
112 * Translation Lookaside Buffers (TLBS)":
113 * "The following operations invalidate all TLB entries, irrespective
114 * of the setting of the G flag:
115 * ...
116 * "(P6 family processors only): Writing to control register CR4 to
117 * modify the PSE, PGE, or PAE flag."
118 *
119 * (the alternatives not quoted above are not an option here.)
120 *
121 * If PGE is not in use, we reload CR3.
122 */
123#ifndef XENPV
124ENTRY(tlbflushg)
125 movq %cr4, %rax
126 testq $CR4_PGE, %rax
127 jz tlbflush
128 movq %rax, %rdx
129 andq $~CR4_PGE, %rdx
130 movq %rdx, %cr4
131 movq %rax, %cr4
132 ret
133END(tlbflushg)
134
135ENTRY(tlbflush)
136#ifdef SVS
137 movb _C_LABEL(svs_pcid),%al
138 testb %al,%al
139 jz 1f
140 xorq %rax,%rax
141 pushq %rax
142 pushq %rax
143 movq $INVPCID_ALL_NONGLOBAL,%rax
144 invpcid (%rsp),%rax
145 addq $16,%rsp
146 ret
147#endif
1481: movq %cr3, %rax
149 movq %rax, %cr3
150 ret
151END(tlbflush)

190/*
191 * Support for reading MSRs in the safe manner (returns EFAULT on fault)
192 */
193/* int rdmsr_safe(u_int msr, uint64_t *data) */
194ENTRY(rdmsr_safe)
195 movq CPUVAR(CURLWP), %r8
196 movq L_PCB(%r8), %r8
197 movq $_C_LABEL(msr_onfault), PCB_ONFAULT(%r8)
198
199 movl %edi, %ecx /* u_int msr */
200 rdmsr /* Read MSR pointed by %ecx. Returns
201 hi byte in edx, lo in %eax */
202 salq $32, %rdx /* sign-shift %rdx left */
203 movl %eax, %eax /* zero-extend %eax -> %rax */
204 orq %rdx, %rax
205 movq %rax, (%rsi) /* *data */
206 xorq %rax, %rax /* "no error" */
207
208 movq %rax, PCB_ONFAULT(%r8)
209 ret
210END(rdmsr_safe)

280/*
281 * void lgdt(struct region_descriptor *rdp);
282 *
283 * Load a new GDT pointer (and do any necessary cleanup).
284 * XXX It's somewhat questionable whether reloading all the segment registers
285 * is necessary, since the actual descriptor data is not changed except by
286 * process creation and exit, both of which clean up via task switches.
287 */
288#ifndef XENPV
289ENTRY(lgdt)
290 /* Reload the descriptor table. */
291 movq %rdi,%rax
292 lgdt (%rax)
293 /* Flush the prefetch q. */
294 jmp 1f
295 nop
2961: jmp _C_LABEL(lgdt_finish)
297END(lgdt)
298#endif
299
300/*
301 * void lgdt_finish(void);
302 * Reload segments after a GDT change
303 */
304ENTRY(lgdt_finish)
305 movl $GSEL(GDATA_SEL, SEL_KPL),%eax
306 movl %eax,%ds
307 movl %eax,%es
308 movl %eax,%ss
309 jmp _C_LABEL(x86_flush)
310END(lgdt_finish)
