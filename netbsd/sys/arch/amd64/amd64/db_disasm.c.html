57/*
58 * Size attributes
59 */
60#define BYTE 0
61#define WORD 1
62#define LONG 2
63#define QUAD 3
64#define SNGL 4
65#define DBLR 5
66#define EXTR 6

70/*
71 * REX prefix and bits
72 */
73#define REX_B 1
74#define REX_X 2
75#define REX_R 4
76#define REX_W 8
77#define REX 0x40

79/*
80 * Addressing modes
81 */
82#define E 1 /* general effective address */
83#define Eind 2 /* indirect address (jump, call) */
84#define Ew 3 /* address, word size */
85#define Eb 4 /* address, byte size */
86#define R 5 /* register, in 'reg' field */
87#define Rw 6 /* word register, in 'reg' field */
88#define Ri 7 /* register in instruction */
89#define S 8 /* segment reg, in 'reg' field */
90#define Si 9 /* segment reg, in instruction */
91#define A 10 /* accumulator */
92#define BX 11 /* (bx) */
93#define CL 12 /* cl, for shifts */
94#define DX 13 /* dx, for IO */
95#define SI 14 /* si */
96#define DI 15 /* di */
97#define CR 16 /* control register */
98#define DR 17 /* debug register */
99#define TR 18 /* test register */
100#define I 19 /* immediate, unsigned */
101#define Is 20 /* immediate, signed */
102#define Ib 21 /* byte immediate, unsigned */
103#define Ibs 22 /* byte immediate, signed */
104#define Iw 23 /* word immediate, unsigned */
105#define Il 24 /* long immediate */
106#define O 25 /* direct address */
107#define Db 26 /* byte displacement from EIP */
108#define Dl 27 /* long displacement from EIP */
109#define o1 28 /* constant 1 */
110#define o3 29 /* constant 3 */
111#define OS 30 /* immediate offset/segment */
112#define ST 31 /* FP stack top */
113#define STI 32 /* FP stack */
114#define X 33 /* extended FP op */
115#define XA 34 /* for 'fstcw %ax' */
116#define Ed 35 /* address, double size */
117#define Iq 36 /* word immediate, maybe 64bits */
118#define Rv 40 /* register in 'r/m' field */

120struct inst {
121 const char *i_name; /* name */
122 short i_has_modrm; /* has regmodrm byte */
123 short i_size; /* operand size */
124 int i_mode; /* addressing modes */
125 const void *i_extra; /* pointer to extra opcode table */
126};
127
128#define op1(x) (x)
129#define op2(x,y) ((x)|((y)<<8))
130#define op3(x,y,z) ((x)|((y)<<8)|((z)<<16))
131
132struct finst {
133 const char *f_name; /* name for memory instruction */
134 int f_size; /* size for memory instruction */
135 int f_rrmode; /* mode for rr instruction */
136 const void *f_rrname; /* name for rr instruction
137 (or pointer to table) */
138};

993/*
994 * Read address at location and return updated location.
995 */
996db_addr_t
997db_read_address(db_addr_t loc, int short_addr, u_int rex, int regmodrm,
998 struct i_addr *addrp)
999 /* addrp: out */
1000{
1001 int mod, rm, sib, index, disp, size, have_sib;
1002
1003 size = (short_addr ? LONG : QUAD);
1004 mod = f_mod(rex, regmodrm);
1005 rm = f_rm(rex, regmodrm);
1006
1007 if (mod == 3) {
1008 addrp->is_reg = true;
1009 addrp->disp = rm;
1010 return (loc);
1011 }
1012 addrp->is_reg = false;
1013 addrp->index = 0;
1014
1015 if ((rm & 0x7) == 4) {
1016 get_value_inc(sib, loc, 1, false);
1017 rm = sib_base(rex, sib);
1018 index = sib_index(rex, sib);
1019 if (index != 4)
1020 addrp->index = db_reg[1][size][index];
1021 addrp->ss = sib_ss(rex, sib);
1022 have_sib = 1;
1023 } else
1024 have_sib = 0;
1025
1026 switch (mod) {
1027 case 0:
1028 if (rm == 5) {
1029 get_value_inc(addrp->disp, loc, 4, false);
1030 if (have_sib)
1031 addrp->base = 0;
1032 else if (short_addr)
1033 addrp->base = "%eip";
1034 else
1035 addrp->base = "%rip";
1036 } else {
1037 addrp->disp = 0;
1038 addrp->base = db_reg[1][size][rm];
1039 }
1040 break;
1041 case 1:
1042 get_value_inc(disp, loc, 1, true);
1043 addrp->disp = disp;
1044 addrp->base = db_reg[1][size][rm];
1045 break;
1046 case 2:
1047 get_value_inc(disp, loc, 4, false);
1048 addrp->disp = disp;
1049 addrp->base = db_reg[1][size][rm];
1050 break;
1051 }
1052 return (loc);
1053}

1079/*
1080 * Disassemble floating-point ("escape") instruction
1081 * and return updated location.
1082 */
1083db_addr_t
1084db_disasm_esc(db_addr_t loc, int inst, u_int rex, int short_addr, int size,
1085 const char * seg)
1086{
1087 int regmodrm;
1088 const struct finst *fp;
1089 int mod;
1090 struct i_addr address;
1091 const char * name;
1092
1093 get_value_inc(regmodrm, loc, 1, false);
1094 fp = &db_Esc_inst[inst - 0xd8][f_reg(rex, regmodrm)];
1095 mod = f_mod(rex, regmodrm);
1096 if (mod != 3) {
1097 if (*fp->f_name == '\0') {
1098 db_printf("<bad instruction>");
1099 return (loc);
1100 }
1101 /*
1102 * Normal address modes.
1103 */
1104 loc = db_read_address(loc, short_addr, rex, regmodrm, &address);
1105 db_printf("%s", fp->f_name);
1106 switch (fp->f_size) {
1107 case SNGL:
1108 db_printf("s");
1109 break;
1110 case DBLR:
1111 db_printf("l");
1112 break;
1113 case EXTR:
1114 db_printf("t");
1115 break;
1116 case WORD:
1117 db_printf("s");
1118 break;
1119 case LONG:
1120 db_printf("l");
1121 break;
1122 case QUAD:
1123 db_printf("q");
1124 break;
1125 default:
1126 break;
1127 }
1128 db_printf("\t");
1129 db_print_address(seg, rex, BYTE, &address);
1130 } else {
1131 /*
1132 * 'reg-reg' - special formats
1133 */
1134 switch (fp->f_rrmode) {
1135 case op2(ST,STI):
1136 name = (fp->f_rrname) ? fp->f_rrname : fp->f_name;
1137 db_printf("%s\t%%st,%%st(%d)", name, f_rm(rex, regmodrm));
1138 break;
1139 case op2(STI,ST):
1140 name = (fp->f_rrname) ? fp->f_rrname : fp->f_name;
1141 db_printf("%s\t%%st(%d),%%st", name, f_rm(rex, regmodrm));
1142 break;
1143 case op1(STI):
1144 name = (fp->f_rrname) ? fp->f_rrname : fp->f_name;
1145 db_printf("%s\t%%st(%d)", name, f_rm(rex, regmodrm));
1146 break;
1147 case op1(X):
1148 name = ((const char * const *)fp->f_rrname)[f_rm(rex, regmodrm)];
1149 if (*name == '\0')
1150 goto bad;
1151 db_printf("%s", name);
1152 break;
1153 case op1(XA):
1154 name = ((const char * const *)fp->f_rrname)[f_rm(rex, regmodrm)];
1155 if (*name == '\0')
1156 goto bad;
1157 db_printf("%s\t%%ax", name);
1158 break;
1159 default:
1160 bad:
1161 db_printf("<bad instruction>");
1162 break;
1163 }
1164 }
1165
1166 return (loc);
1167}

1169/*
1170 * Disassemble instruction at 'loc'. 'altfmt' specifies an
1171 * (optional) alternate format. Return address of start of
1172 * next instruction.
1173 */
1174db_addr_t
1175db_disasm(db_addr_t loc, bool altfmt)
1176{
1177 int inst;
1178 int size;
1179 int short_addr;
1180 const char *seg;
1181 const struct inst *ip;
1182 const char *i_name;
1183 int i_size;
1184 int i_mode;
1185 int regmodrm = 0;
1186 bool first;
1187 int displ;
1188 int prefix;
1189 int imm;
1190 int imm2;
1191 uint64_t imm64;
1192 int len;
1193 struct i_addr address;
1194#ifdef _KERNEL
1195 pt_entry_t *pte, *pde;
1196#endif
1197 u_int rex = 0;
1198
1199#ifdef _KERNEL
1200 /*
1201 * Don't try to disassemble the location if the mapping is invalid.
1202 * If we do, we'll fault, and end up debugging the debugger!
1203 * in the case of largepages, "pte" is really the pde and "pde" is
1204 * really the entry for the pdp itself.
1205 */
1206 if ((vaddr_t)loc >= VM_MIN_KERNEL_ADDRESS)
1207 pte = kvtopte((vaddr_t)loc);
1208 else
1209 pte = vtopte((vaddr_t)loc);
1210 if ((vaddr_t)pte >= VM_MIN_KERNEL_ADDRESS)
1211 pde = kvtopte((vaddr_t)pte);
1212 else
1213 pde = vtopte((vaddr_t)pte);
1214
1215 if ((*pde & PTE_P) == 0 || (*pte & PTE_P) == 0) {
1216 db_printf("invalid address\n");
1217 return (loc);
1218 }
1219#endif
1220
1221 get_value_inc(inst, loc, 1, false);
1222 short_addr = false;
1223 size = LONG;
1224 seg = 0;
1225
1226 /*
1227 * Get prefixes
1228 */
1229 prefix = true;
1230 do {
1231 switch (inst) {
1232 case 0x66: /* data16 */
1233 size = WORD;
1234 break;
1235 case 0x67:
1236 short_addr = true;
1237 break;
1238 case 0x26:
1239 seg = "%es";
1240 break;
1241 case 0x36:
1242 seg = "%ss";
1243 break;
1244 case 0x2e:
1245 seg = "%cs";
1246 break;
1247 case 0x3e:
1248 seg = "%ds";
1249 break;
1250 case 0x64:
1251 seg = "%fs";
1252 break;
1253 case 0x65:
1254 seg = "%gs";
1255 break;
1256 case 0xf0:
1257 db_printf("lock ");
1258 break;
1259 case 0xf2:
1260 db_printf("repne ");
1261 break;
1262 case 0xf3:
1263 db_printf("repe "); /* XXX repe VS rep */
1264 break;
1265 default:
1266 prefix = false;
1267 break;
1268 }
1269 if (inst >= 0x40 && inst <= 0x4f) {
1270 rex = inst;
1271 prefix = true;
1272 }
1273 if (prefix)
1274 get_value_inc(inst, loc, 1, false);
1275 } while (prefix);
1276
1277 if (rex != 0) {
1278 if (rex & REX_W)
1279 size = QUAD;
1280 if (altfmt == true) /* XXX */
1281 db_printf("%s", rex_str[rex & 0x0f]);
1282 }
1283
1284 if (inst >= 0xd8 && inst <= 0xdf) {
1285 loc = db_disasm_esc(loc, inst, rex, short_addr, size, seg);
1286 db_printf("\n");
1287 return (loc);
1288 }
1289
1290 if (inst == 0x0f) {
1291 get_value_inc(inst, loc, 1, false);
1292 ip = db_inst_0f[inst>>4];
1293 if (ip == 0)
1294 ip = &db_bad_inst;
1295 else
1296 ip = &ip[inst&0xf];
1297 } else {
1298 ip = &db_inst_table[inst];
1299 }
1300
1301 if (ip->i_has_modrm) {
1302 get_value_inc(regmodrm, loc, 1, false);
1303 loc = db_read_address(loc, short_addr, rex, regmodrm, &address);
1304 }
1305
1306 i_name = ip->i_name;
1307 i_size = ip->i_size;
1308 i_mode = ip->i_mode;
1309
1310 if (ip->i_extra == db_Grp9 && f_mod(rex, regmodrm) == 3) {
1311 ip = &db_Grp9b[f_reg(rex, regmodrm)];
1312 i_name = ip->i_name;
1313 i_size = ip->i_size;
1314 i_mode = ip->i_mode;
1315 } else if (ip->i_extra == (const char *)db_Grp1 ||
1316 ip->i_extra == (const char *)db_Grp2 ||
1317 ip->i_extra == (const char *)db_Grp6 ||
1318 ip->i_extra == (const char *)db_Grp7 ||
1319 ip->i_extra == (const char *)db_Grp8 ||
1320 ip->i_extra == (const char *)db_Grp9) {
1321 if (ip->i_extra == (const char *)db_Grp7 && regmodrm == 0xf8) {
1322 i_name = "swapgs";
1323 i_mode = 0;
1324 } else if (ip->i_extra == (const char *)db_Grp7 && regmodrm == 0xcb) {
1325 i_name = "stac";
1326 i_mode = 0;
1327 } else if (ip->i_extra == (const char *)db_Grp7 && regmodrm == 0xca) {
1328 i_name = "clac";
1329 i_mode = 0;
1330 } else {
1331 i_name = ((const char * const *)ip->i_extra)
1332 [f_reg(rex, regmodrm)];
1333 }
1334 } else if (ip->i_extra == (const char *)db_Grp3) {
1335 ip = (const struct inst *)ip->i_extra;
1336 ip = &ip[f_reg(rex, regmodrm)];
1337 i_name = ip->i_name;
1338 i_mode = ip->i_mode;
1339 } else if (ip->i_extra == (const char *)db_Grp4 ||
1340 ip->i_extra == (const char *)db_Grp5) {
1341 ip = (const struct inst *)ip->i_extra;
1342 ip = &ip[f_reg(rex, regmodrm)];
1343 i_name = ip->i_name;
1344 i_mode = ip->i_mode;
1345 i_size = ip->i_size;
1346 }
1347
1348 if (i_size == SDEP) {
1349 if (size == LONG)
1350 db_printf("%s", i_name);
1351 else
1352 db_printf("%s", (const char *)ip->i_extra);
1353 } else {
1354 db_printf("%s", i_name);
1355 if (i_size != NONE) {
1356 if (i_size == BYTE) {
1357 db_printf("b");
1358 size = BYTE;
1359 } else if (i_size == WORD) {
1360 db_printf("w");
1361 size = WORD;
1362 } else if (size == WORD) {
1363 db_printf("w");
1364 } else if (i_size == QUAD) {
1365 db_printf("q");
1366 size = QUAD;
1367 } else if (size == QUAD) {
1368 db_printf("q");
1369 } else {
1370 db_printf("l");
1371 }
1372 }
1373 }
1374 db_printf("\t");
1375 for (first = true; i_mode != 0; i_mode >>= 8, first = false) {
1376 char tbuf[24];
1377
1378 if (!first)
1379 db_printf(",");
1380
1381 switch (i_mode & 0xFF) {
1382 case E:
1383 case Eind:
1384 case Ed:
1385 case Ew:
1386 case Eb:
1387 if (!ip->i_has_modrm) {
1388 db_printf("Bad address mode %#x without modrm",
1389 i_mode);
1390 break;
1391 }
1392 switch (i_mode & 0xFF) {
1393 case E:
1394 db_print_address(seg, rex, size, &address);
1395 break;
1396 case Eind:
1397 db_printf("*");
1398 db_print_address(seg, rex, size, &address);
1399 break;
1400 case Ed:
1401 db_print_address(seg, rex, LONG, &address);
1402 break;
1403 case Ew:
1404 db_print_address(seg, rex, WORD, &address);
1405 break;
1406 case Eb:
1407 db_print_address(seg, rex, BYTE, &address);
1408 break;
1409 }
1410 break;
1411 case R: {
1412 int ext = ((rex & REX_R) != 0);
1413 db_printf("%s", db_reg[ext][size][f_reg(rex, regmodrm)]);
1414 break;
1415 }
1416 case Rw: {
1417 int ext = ((rex & REX_R) != 0);
1418 db_printf("%s", db_reg[ext][WORD][f_reg(rex, regmodrm)]);
1419 break;
1420 }
1421 case Ri: {
1422 db_printf("%s", db_reg[0][size][f_rm(rex, inst)]);
1423 break;
1424 }
1425 case S:
1426 db_printf("%s", db_seg_reg[f_reg(rex, regmodrm)]);
1427 break;
1428 case Si:
1429 db_printf("%s", db_seg_reg[f_reg(rex, inst)]);
1430 break;
1431 case A:
1432 db_printf("%s", db_reg[0][size][0]); /* acc */
1433 break;
1434 case BX:
1435 if (seg)
1436 db_printf("%s:", seg);
1437 db_printf("(%s)", short_addr ? "%ebx" : "%rbx");
1438 break;
1439 case CL:
1440 db_printf("%%cl");
1441 break;
1442 case DX:
1443 db_printf("%%dx");
1444 break;
1445 case SI:
1446 if (seg)
1447 db_printf("%s:", seg);
1448 db_printf("(%s)", short_addr ? "%esi" : "%rsi");
1449 break;
1450 case DI:
1451 db_printf("%%es:(%s)", short_addr ? "%edi" : "%rdi");
1452 break;
1453 case CR:
1454 db_printf("%%cr%d", f_reg(rex, regmodrm));
1455 break;
1456 case DR:
1457 db_printf("%%dr%d", f_reg(rex, regmodrm));
1458 break;
1459 case TR:
1460 db_printf("%%tr%d", f_reg(rex, regmodrm));
1461 break;
1462 case Iq:
1463 if (size == QUAD) {
1464 get_value_inc(imm64, loc, 8, false);
1465 db_format_radix(tbuf, 24, imm64, true);
1466 db_printf("$%s", tbuf);
1467 break;
1468 }
1469 /* FALLTHROUGH */
1470 case I:
1471 len = db_lengths[size];
1472 get_value_inc(imm, loc, len, false);/* unsigned */
1473 db_format_radix(tbuf, 24, (unsigned int)imm, true);
1474 db_printf("$%s", tbuf);
1475 break;
1476 case Is:
1477 len = db_lengths[size];
1478 get_value_inc(imm, loc, len, true); /* signed */
1479 db_format_radix(tbuf, 24, imm, true);
1480 db_printf("$%s", tbuf);
1481 break;
1482 case Ib:
1483 get_value_inc(imm, loc, 1, false); /* unsigned */
1484 db_format_radix(tbuf, 24, (unsigned int)imm, true);
1485 db_printf("$%s", tbuf);
1486 break;
1487 case Ibs:
1488 get_value_inc(imm, loc, 1, true); /* signed */
1489 db_format_radix(tbuf, 24, imm, true);
1490 db_printf("$%s", tbuf);
1491 break;
1492 case Iw:
1493 get_value_inc(imm, loc, 2, false); /* unsigned */
1494 db_format_radix(tbuf, 24, (unsigned int)imm, true);
1495 db_printf("$%s", tbuf);
1496 break;
1497 case Il:
1498 get_value_inc(imm, loc, 4, false);
1499 db_format_radix(tbuf, 24, (unsigned int)imm, true);
1500 db_printf("$%s", tbuf);
1501 break;
1502 case O: /* Only move %eax to/from absolute address */
1503 if (short_addr)
1504 get_value_inc(imm64, loc, 2, true);
1505 else
1506 get_value_inc(imm64, loc, 8, true);
1507 if (seg) {
1508 db_format_radix(tbuf, 24, imm64, true);
1509 db_printf("%s:%s", seg, tbuf);
1510 } else
1511 db_printsym((db_addr_t)imm64, DB_STGY_ANY,
1512 db_printf);
1513 break;
1514 case Db:
1515 get_value_inc(displ, loc, 1, true);
1516 db_printsym((db_addr_t)(displ + loc), DB_STGY_XTRN,
1517 db_printf);
1518 break;
1519 case Dl:
1520 get_value_inc(displ, loc, 4, true);
1521 db_printsym((db_addr_t)(displ + loc), DB_STGY_XTRN,
1522 db_printf);
1523 break;
1524 case o1:
1525 db_printf("$1");
1526 break;
1527 case o3:
1528 db_printf("$3");
1529 break;
1530 case OS:
1531 get_value_inc(imm, loc, 4, false); /* offset */
1532 db_format_radix(tbuf, 24, (unsigned int)imm, true);
1533 db_printf("$%s", tbuf);
1534 get_value_inc(imm2, loc, 2, false); /* segment */
1535 db_format_radix(tbuf, 24, (unsigned int)imm2, true);
1536 db_printf(",%s", tbuf);
1537 break;
1538 }
1539 }
1540
1541 db_printf("\n");
1542 return (loc);
1543}
