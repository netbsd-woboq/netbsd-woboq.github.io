60#define NSLOTS(sz) \
61 (((sz) - DYNSEL_START) / sizeof(struct sys_segment_descriptor))
62#define NDYNSLOTS NSLOTS(MAXGDTSIZ)
63
64typedef struct {
65 bool busy[NDYNSLOTS];
66 size_t nslots;
67} gdt_bitmap_t;
68
69/* size of GDT in bytes */
70#ifdef XENPV
71const size_t gdt_size = FIRST_RESERVED_GDT_BYTE;
72#else
73const size_t gdt_size = MAXGDTSIZ;
74#endif
75
76/* bitmap of busy slots */
77static gdt_bitmap_t gdt_bitmap;

83void
84update_descriptor(void *tp, void *ep)
85{
86 uint64_t *table, *entry;
87
88 table = tp;
89 entry = ep;
90
91#ifndef XENPV
92 *table = *entry;
93#else
94 paddr_t pa;
95
96 if (!pmap_extract_ma(pmap_kernel(), (vaddr_t)table, &pa) ||
97 HYPERVISOR_update_descriptor(pa, *entry))
98 panic("HYPERVISOR_update_descriptor failed");
99#endif
100}

102#if defined(USER_LDT) || !defined(XENPV)
103/*
104 * Called on a newly-allocated GDT slot, so no race between CPUs.
105 */
106static void
107set_sys_gdt(int slot, void *base, size_t limit, int type, int dpl, int gran)
108{
109 union {
110 struct sys_segment_descriptor sd;
111 uint64_t bits[2];
112 } d;
113 CPU_INFO_ITERATOR cii;
114 struct cpu_info *ci;
115 int idx;
116
117 set_sys_segment(&d.sd, base, limit, type, dpl, gran);
118 idx = IDXSEL(GDYNSEL(slot, SEL_KPL));
119 for (CPU_INFO_FOREACH(cii, ci)) {
120 KASSERT(ci->ci_gdt != NULL);
121 update_descriptor(&ci->ci_gdt[idx + 0], &d.bits[0]);
122 update_descriptor(&ci->ci_gdt[idx + 1], &d.bits[1]);
123 }
124}
125#endif /* USER_LDT || !XENPV */

127/*
128 * Initialize the GDT. We already have a gdtstore, which was temporarily used
129 * by the bootstrap code. Now, we allocate a new gdtstore, and put it in cpu0.
130 */
131void
132gdt_init(void)
133{
134 char *old_gdt;
135 struct cpu_info *ci = &cpu_info_primary;
136
137 /* Initialize the global values */
138 memset(&gdt_bitmap.busy, 0, sizeof(gdt_bitmap.busy));
139 gdt_bitmap.nslots = NSLOTS(gdt_size);
140
141 old_gdt = gdtstore;
142
143#ifdef __HAVE_PCPU_AREA
144 /* The GDT is part of the pcpuarea */
145 gdtstore = (char *)&pcpuarea->ent[cpu_index(ci)].gdt;
146#else
147 struct vm_page *pg;
148 vaddr_t va;
149
150 /* Allocate gdt_size bytes of memory. */
151 gdtstore = (char *)uvm_km_alloc(kernel_map, gdt_size, 0,
152 UVM_KMF_VAONLY);
153 for (va = (vaddr_t)gdtstore; va < (vaddr_t)gdtstore + gdt_size;
154 va += PAGE_SIZE) {
155 pg = uvm_pagealloc(NULL, 0, NULL, UVM_PGA_ZERO);
156 if (pg == NULL) {
157 panic("gdt_init: no pages");
158 }
159 pmap_kenter_pa(va, VM_PAGE_TO_PHYS(pg),
160 VM_PROT_READ | VM_PROT_WRITE, 0);
161 }
162 pmap_update(pmap_kernel());
163#endif
164
165 /* Copy the initial bootstrap GDT into the new area. */
166 memcpy(gdtstore, old_gdt, DYNSEL_START);
167 ci->ci_gdt = (void *)gdtstore;
168#ifndef XENPV
169 set_sys_segment(GDT_ADDR_SYS(gdtstore, GLDT_SEL), ldtstore,
170 LDT_SIZE - 1, SDT_SYSLDT, SEL_KPL, 0);
171#endif
172
173 gdt_init_cpu(ci);
174}

176/*
177 * Allocate shadow GDT for a secondary CPU. It contains the same values as the
178 * GDT present in cpu0 (gdtstore).
179 */
180void
181gdt_alloc_cpu(struct cpu_info *ci)
182{
183#ifdef __HAVE_PCPU_AREA
184 ci->ci_gdt = (union descriptor *)&pcpuarea->ent[cpu_index(ci)].gdt;
185#else
186 struct vm_page *pg;
187 vaddr_t va;
188
189 ci->ci_gdt = (union descriptor *)uvm_km_alloc(kernel_map, gdt_size,
190 0, UVM_KMF_VAONLY);
191 for (va = (vaddr_t)ci->ci_gdt; va < (vaddr_t)ci->ci_gdt + gdt_size;
192 va += PAGE_SIZE) {
193 while ((pg = uvm_pagealloc(NULL, 0, NULL, UVM_PGA_ZERO))
194 == NULL) {
195 uvm_wait("gdt_alloc_cpu");
196 }
197 pmap_kenter_pa(va, VM_PAGE_TO_PHYS(pg),
198 VM_PROT_READ | VM_PROT_WRITE, 0);
199 }
200 pmap_update(pmap_kernel());
201#endif
202
203 memcpy(ci->ci_gdt, gdtstore, gdt_size);
204}

206/*
207 * Load appropriate GDT descriptor into the currently running CPU, which must
208 * be ci.
209 */
210void
211gdt_init_cpu(struct cpu_info *ci)
212{
213 struct region_descriptor region;
214
215 KASSERT(curcpu() == ci);
216
217 setregion(&region, ci->ci_gdt, (uint16_t)(gdt_size - 1));
218 lgdt(&region);
219}

221#if !defined(XENPV) || defined(USER_LDT)
222static int
223gdt_get_slot(void)
224{
225 size_t i;
226
227 KASSERT(mutex_owned(&cpu_lock));
228
229 for (i = 0; i < gdt_bitmap.nslots; i++) {
230 if (!gdt_bitmap.busy[i]) {
231 gdt_bitmap.busy[i] = true;
232 return (int)i;
233 }
234 }
235 panic("gdt_get_slot: out of memory");
236
237 /* NOTREACHED */
238 return 0;
239}
240
241static void
242gdt_put_slot(int slot)
243{
244 KASSERT(mutex_owned(&cpu_lock));
245 KASSERT(slot < gdt_bitmap.nslots);
246 gdt_bitmap.busy[slot] = false;
247}
248#endif

250int
251tss_alloc(struct x86_64_tss *tss)
252{
253#ifndef XENPV
254 int slot;
255
256 mutex_enter(&cpu_lock);
257
258 slot = gdt_get_slot();
259 set_sys_gdt(slot, tss, sizeof(struct x86_64_tss) - 1, SDT_SYS386TSS,
260 SEL_KPL, 0);
261
262 mutex_exit(&cpu_lock);
263
264 return GDYNSEL(slot, SEL_KPL);
265#else
266 /* TSS, what for? */
267 return GSEL(GNULL_SEL, SEL_KPL);
268#endif
269}
270
271void
272tss_free(int sel)
273{
274#ifndef XENPV
275 mutex_enter(&cpu_lock);
276 gdt_put_slot(IDXDYNSEL(sel));
277 mutex_exit(&cpu_lock);
278#else
279 KASSERT(sel == GSEL(GNULL_SEL, SEL_KPL));
280#endif
281}

283#ifdef USER_LDT
284int
285ldt_alloc(void *ldtp, size_t len)
286{
287 int slot;
288
289 KASSERT(mutex_owned(&cpu_lock));
290
291 slot = gdt_get_slot();
292 set_sys_gdt(slot, ldtp, len - 1, SDT_SYSLDT, SEL_KPL, 0);
293
294 return GDYNSEL(slot, SEL_KPL);
295}

297void
298ldt_free(int sel)
299{
300 int slot;
301
302 KASSERT(mutex_owned(&cpu_lock));
303
304 slot = IDXDYNSEL(sel);
305
306 gdt_put_slot(slot);
307}

310#ifdef XENPV
311void
312lgdt(struct region_descriptor *desc)
313{
314 paddr_t frames[16];
315 size_t i;
316 vaddr_t va;
317
318 /*
319 * Xen even checks descriptors AFTER limit. Zero out last frame after
320 * limit if needed.
321 */
322 va = desc->rd_base + desc->rd_limit + 1;
323 memset((void *)va, 0, roundup(va, PAGE_SIZE) - va);
324
325 /*
326 * The lgdt instruction uses virtual addresses, do some translation for
327 * Xen. Mark pages R/O too, otherwise Xen will refuse to use them.
328 */
329 for (i = 0; i < roundup(desc->rd_limit, PAGE_SIZE) >> PAGE_SHIFT; i++) {
330 va = desc->rd_base + (i << PAGE_SHIFT);
331 frames[i] = ((paddr_t)xpmap_ptetomach((pt_entry_t *)va)) >>
332 PAGE_SHIFT;
333 pmap_pte_clearbits(kvtopte(va), PTE_W);
334 }
335
336 if (HYPERVISOR_set_gdt(frames, (desc->rd_limit + 1) >> 3))
337 panic("lgdt(): HYPERVISOR_set_gdt() failed");
338 lgdt_finish();
339}
