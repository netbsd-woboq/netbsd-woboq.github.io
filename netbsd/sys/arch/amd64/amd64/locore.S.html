218/*
219 * fillkpt - Fill in a kernel page table
220 * eax = pte (page frame | control | status)
221 * ebx = page table address
222 * ecx = number of pages to map
223 *
224 * Each entry is 8 (PDE_SIZE) bytes long: we must set the 4 upper bytes to 0.
225 */
226#define fillkpt \
227 cmpl $0,%ecx ; /* zero-sized? */ \
228 je 2f ; \
2291: movl $0,(PDE_SIZE-4)(%ebx) ; /* upper 32 bits: 0 */ \
230 movl %eax,(%ebx) ; /* store phys addr */ \
231 addl $PDE_SIZE,%ebx ; /* next PTE/PDE */ \
232 addl $PAGE_SIZE,%eax ; /* next phys page */ \
233 loop 1b ; \
2342: ;
235
236/*
237 * fillkpt_nox - Same as fillkpt, but sets the NX/XD bit.
238 */
239#define fillkpt_nox \
240 cmpl $0,%ecx ; /* zero-sized? */ \
241 je 2f ; \
242 pushl %ebp ; \
243 movl RELOC(nox_flag),%ebp ; \
2441: movl %ebp,(PDE_SIZE-4)(%ebx) ; /* upper 32 bits: NX */ \
245 movl %eax,(%ebx) ; /* store phys addr */ \
246 addl $PDE_SIZE,%ebx ; /* next PTE/PDE */ \
247 addl $PAGE_SIZE,%eax ; /* next phys page */ \
248 loop 1b ; \
249 popl %ebp ; \
2502: ;
251
252/*
253 * fillkpt_blank - Fill in a kernel page table with blank entries
254 * ebx = page table address
255 * ecx = number of pages to map
256 */
257#define fillkpt_blank \
258 cmpl $0,%ecx ; /* zero-sized? */ \
259 je 2f ; \
2601: movl $0,(PDE_SIZE-4)(%ebx) ; /* upper 32 bits: 0 */ \
261 movl $0,(%ebx) ; /* lower 32 bits: 0 */ \
262 addl $PDE_SIZE,%ebx ; /* next PTE/PDE */ \
263 loop 1b ; \
2642: ;
265
266/*
267 * killkpt - Destroy a kernel page table (long mode)
268 * rbx = page table address
269 * rcx = number of pages to destroy
270 */
271#define killkpt \
2721: movq $0,(%rbx) ; \
273 addq $PDE_SIZE,%rbx ; \
274 loop 1b ;

436ENTRY(start)
437#ifndef XENPV
438 .code32
439
440 /* Warm boot */
441 movw $0x1234,0x472
442
443 /*
444 * Load parameters from the stack (32 bits):
445 * boothowto, [bootdev], bootinfo, esym, biosextmem, biosbasemem
446 * We are not interested in 'bootdev'.
447 */
448
449 /* Load 'boothowto' */
450 movl 4(%esp),%eax
451 movl %eax,RELOC(boothowto)
452
453 /* Load 'bootinfo' */
454 movl 12(%esp),%eax
455 testl %eax,%eax /* bootinfo = NULL? */
456 jz .Lbootinfo_finished
457
458 movl (%eax),%ebx /* bootinfo::bi_nentries */
459 movl $RELOC(bootinfo),%ebp
460 movl %ebp,%edx
461 addl $BOOTINFO_MAXSIZE,%ebp
462 movl %ebx,(%edx)
463 addl $4,%edx
464
465.Lbootinfo_entryloop:
466 testl %ebx,%ebx /* no remaining entries? */
467 jz .Lbootinfo_finished
468
469 addl $4,%eax
470 movl (%eax),%ecx /* address of entry */
471 pushl %edi
472 pushl %esi
473 pushl %eax
474
475 movl (%ecx),%eax /* btinfo_common::len (size of entry) */
476 movl %edx,%edi
477 addl %eax,%edx /* update dest pointer */
478 cmpl %ebp,%edx /* beyond bootinfo+BOOTINFO_MAXSIZE? */
479 jg .Lbootinfo_overflow
480
481 movl %ecx,%esi
482 movl %eax,%ecx
483
484 /*
485 * If any modules were loaded, record where they end. 'eblob' is used
486 * later to compute the initial bootstrap tables.
487 */
488 cmpl $BTINFO_MODULELIST,4(%esi) /* btinfo_common::type */
489 jne .Lbootinfo_copy
490
491 /* Skip the modules if we won't have enough VA to map them */
492 movl 12(%esi),%eax /* btinfo_modulelist::endpa */
493 addl $PGOFSET,%eax /* roundup to a page */
494 andl $~PGOFSET,%eax
495 cmpl $BOOTMAP_VA_SIZE,%eax
496 jg .Lbootinfo_skip
497 movl %eax,RELOC(eblob)
498 addl $KERNBASE_LO,RELOC(eblob)
499 adcl $KERNBASE_HI,RELOC(eblob)+4
500
501.Lbootinfo_copy:
502 rep
503 movsb /* copy esi -> edi */
504 jmp .Lbootinfo_next
505
506.Lbootinfo_skip:
507 subl %ecx,%edx /* revert dest pointer */
508
509.Lbootinfo_next:
510 popl %eax
511 popl %esi
512 popl %edi
513 subl $1,%ebx /* decrement the # of entries */
514 jmp .Lbootinfo_entryloop
515
516.Lbootinfo_overflow:
517 /*
518 * Cleanup for overflow case. Pop the registers, and correct the number
519 * of entries.
520 */
521 popl %eax
522 popl %esi
523 popl %edi
524 movl $RELOC(bootinfo),%ebp
525 movl %ebp,%edx
526 subl %ebx,(%edx) /* correct the number of entries */
527.Lbootinfo_finished:
528
529 /* Load 'esym' */
530 movl 16(%esp),%eax
531 testl %eax,%eax /* esym = NULL? */
532 jz 1f
533
534 addl $KERNBASE_LO,%eax
535
5361:
537 movl $RELOC(esym),%ebp
538 movl %eax,(%ebp)
539 movl $KERNBASE_HI,4(%ebp)
540
541 /* Load 'biosextmem' */
542 movl $RELOC(biosextmem),%ebp
543 movl (%ebp),%eax
544 testl %eax,%eax /* already set? */
545 jnz .Lbiosextmem_finished
546
547 movl 20(%esp),%eax
548 movl %eax,(%ebp)
549
550.Lbiosextmem_finished:
551 /* Load 'biosbasemem' */
552 movl $RELOC(biosbasemem),%ebp
553 movl (%ebp),%eax
554 testl %eax,%eax /* already set? */
555 jnz .Lbiosbasemem_finished
556
557 movl 24(%esp),%eax
558 movl %eax,(%ebp)
559
560.Lbiosbasemem_finished:
561 /*
562 * Done with the parameters!
563 */
564
565 /* First, reset the PSL. */
566 pushl $PSL_MBO
567 popfl
568
569 xorl %eax,%eax
570 cpuid
571 movl %eax,RELOC(cpuid_level)
572
573 /*
574 * Finished with old stack; load new %esp now instead of later so we
575 * can trace this code without having to worry about the trace trap
576 * clobbering the memory test or the zeroing of the bss+bootstrap page
577 * tables.
578 *
579 * The boot program should check:
580 * text+data <= &stack_variable - more_space_for_stack
581 * text+data+bss+pad+space_for_page_tables <= end_of_memory
582 *
583 * XXX: the gdt is in the carcass of the boot program so clearing
584 * the rest of memory is still not possible.
585 */
586 movl $RELOC(tmpstk),%esp
587
588 /*
589 * Retrieve the NX/XD flag. We use the 32bit version of PTE_NX.
590 */
591 movl $0x80000001,%eax
592 cpuid
593 andl $CPUID_NOX,%edx
594 jz .Lno_NOX
595 movl $PTE_NX32,RELOC(nox_flag)
596.Lno_NOX:
597
598/*
599 * There are four levels of pages in amd64: PML4 -> PDP -> PD -> PT. They will
600 * be referred to as: L4 -> L3 -> L2 -> L1.
601 *
602 * Virtual address space of the kernel:
603 * +------+--------+------+-----+--------+---------------------+----------
604 * | TEXT | RODATA | DATA | BSS | [SYMS] | [PRELOADED MODULES] | L4 ->
605 * +------+--------+------+-----+--------+---------------------+----------
606 * (1) (2) (3)
607 *
608 * --------------+-----+-----+----+-------------+
609 * -> PROC0 STK -> L3 -> L2 -> L1 | ISA I/O MEM |
610 * --------------+-----+-----+----+-------------+
611 * (4)
612 *
613 * PROC0 STK is obviously not linked as a page level. It just happens to be
614 * caught between L4 and L3.
615 *
616 * (PROC0 STK + L4 + L3 + L2 + L1) is later referred to as BOOTSTRAP TABLES.
617 *
618 * ISA I/O MEM has no physical page allocated here, just virtual addresses.
619 *
620 * Important note: the kernel segments are properly 4k-aligned
621 * (see kern.ldscript), so there's no need to enforce alignment.
622 */
623
624 /* Find end of kernel image; brings us on (1). */
625 movl $RELOC(__kernel_end),%edi
626
627#if (NKSYMS || defined(DDB) || defined(MODULAR)) && !defined(makeoptions_COPY_SYMTAB)
628 /* Save the symbols (if loaded); brinds us on (2). */
629 movl RELOC(esym),%eax
630 testl %eax,%eax
631 jz 1f
632 subl $KERNBASE_LO,%eax /* XXX */
633 movl %eax,%edi
6341:
635#endif
636 /* Skip over any modules/blobs; brings us on (3). */
637 movl RELOC(eblob),%eax
638 testl %eax,%eax
639 jz 1f
640 subl $KERNBASE_LO,%eax /* XXX */
641 movl %eax,%edi
6421:
643
644 /* We are on (3). Align up for BOOTSTRAP TABLES. */
645 movl %edi,%esi
646 addl $PGOFSET,%esi
647 andl $~PGOFSET,%esi
648
649 /* We are on the BOOTSTRAP TABLES. Save L4's physical address. */
650 movl $RELOC(PDPpaddr),%ebp
651 movl %esi,(%ebp)
652 movl $0,4(%ebp)
653
654 /* Now, zero out the BOOTSTRAP TABLES (before filling them in). */
655 movl %esi,%edi
656 xorl %eax,%eax
657 cld
658 movl $TABLESIZE,%ecx
659 shrl $2,%ecx
660 rep
661 stosl /* copy eax -> edi */
662
663/*
664 * Build the page tables and levels. We go from L1 to L4, and link the levels
665 * together. Note: RELOC computes &addr - KERNBASE in 32 bits; the value can't
666 * be > 4G, or we can't deal with it anyway, since we are in 32bit mode.
667 */
668 /*
669 * Build L1.
670 */
671 leal (PROC0_PTP1_OFF)(%esi),%ebx
672
673 /* Skip the area below the kernel text. */
674 movl $(KERNTEXTOFF_LO - KERNBASE_LO),%ecx
675 shrl $PGSHIFT,%ecx
676 fillkpt_blank
677
678 /* Map the kernel text RX. */
679 movl $(KERNTEXTOFF_LO - KERNBASE_LO),%eax /* start of TEXT */
680 movl $RELOC(__rodata_start),%ecx
681 subl %eax,%ecx
682 shrl $PGSHIFT,%ecx
683 orl $(PTE_P),%eax
684 fillkpt
685
686 /* Map the kernel rodata R. */
687 movl $RELOC(__rodata_start),%eax
688 movl $RELOC(__data_start),%ecx
689 subl %eax,%ecx
690 shrl $PGSHIFT,%ecx
691 orl $(PTE_P),%eax
692 fillkpt_nox
693
694 /* Map the kernel data+bss RW. */
695 movl $RELOC(__data_start),%eax
696 movl $RELOC(__kernel_end),%ecx
697 subl %eax,%ecx
698 shrl $PGSHIFT,%ecx
699 orl $(PTE_P|PTE_W),%eax
700 fillkpt_nox
701
702 /* Map [SYMS]+[PRELOADED MODULES] RW. */
703 movl $RELOC(__kernel_end),%eax
704 movl %esi,%ecx /* start of BOOTSTRAP TABLES */
705 subl %eax,%ecx
706 shrl $PGSHIFT,%ecx
707 orl $(PTE_P|PTE_W),%eax
708 fillkpt_nox
709
710 /* Map the BOOTSTRAP TABLES RW. */
711 movl %esi,%eax /* start of BOOTSTRAP TABLES */
712 movl $TABLESIZE,%ecx /* length of BOOTSTRAP TABLES */
713 shrl $PGSHIFT,%ecx
714 orl $(PTE_P|PTE_W),%eax
715 fillkpt_nox
716
717 /* We are on (4). Map ISA I/O MEM RW. */
718 movl $IOM_BEGIN,%eax
719 movl $IOM_SIZE,%ecx /* size of ISA I/O MEM */
720 shrl $PGSHIFT,%ecx
721 orl $(PTE_P|PTE_W/*|PTE_PCD*/),%eax
722 fillkpt_nox
723
724 /*
725 * Build L2. Linked to L1.
726 */
727 leal (PROC0_PTP2_OFF)(%esi),%ebx
728 leal (PROC0_PTP1_OFF)(%esi),%eax
729 orl $(PTE_P|PTE_W),%eax
730 movl $(NKL2_KIMG_ENTRIES+1),%ecx
731 fillkpt
732
733#if L2_SLOT_KERNBASE > 0
734 /* If needed, set up level 2 entries for actual kernel mapping */
735 leal (PROC0_PTP2_OFF + L2_SLOT_KERNBASE * PDE_SIZE)(%esi),%ebx
736 leal (PROC0_PTP1_OFF)(%esi),%eax
737 orl $(PTE_P|PTE_W),%eax
738 movl $(NKL2_KIMG_ENTRIES+1),%ecx
739 fillkpt
740#endif
741
742 /*
743 * Build L3. Linked to L2.
744 */
745 leal (PROC0_PTP3_OFF)(%esi),%ebx
746 leal (PROC0_PTP2_OFF)(%esi),%eax
747 orl $(PTE_P|PTE_W),%eax
748 movl $NKL3_KIMG_ENTRIES,%ecx
749 fillkpt
750
751#if L3_SLOT_KERNBASE > 0
752 /* If needed, set up level 3 entries for actual kernel mapping */
753 leal (PROC0_PTP3_OFF + L3_SLOT_KERNBASE * PDE_SIZE)(%esi),%ebx
754 leal (PROC0_PTP2_OFF)(%esi),%eax
755 orl $(PTE_P|PTE_W),%eax
756 movl $NKL3_KIMG_ENTRIES,%ecx
757 fillkpt
758#endif
759
760 /*
761 * Build L4 for identity mapping. Linked to L3.
762 */
763 leal (PROC0_PML4_OFF)(%esi),%ebx
764 leal (PROC0_PTP3_OFF)(%esi),%eax
765 orl $(PTE_P|PTE_W),%eax
766 movl $NKL4_KIMG_ENTRIES,%ecx
767 fillkpt
768
769 /* Set up L4 entries for actual kernel mapping */
770 leal (PROC0_PML4_OFF + L4_SLOT_KERNBASE * PDE_SIZE)(%esi),%ebx
771 leal (PROC0_PTP3_OFF)(%esi),%eax
772 orl $(PTE_P|PTE_W),%eax
773 movl $NKL4_KIMG_ENTRIES,%ecx
774 fillkpt
775
776 /*
777 * Startup checklist:
778 * 1. Enable PAE (and SSE while here).
779 */
780 movl %cr4,%eax
781 orl $(CR4_PAE|CR4_OSFXSR|CR4_OSXMMEXCPT),%eax
782 movl %eax,%cr4
783
784 /*
785 * 2. Set Long Mode Enable in EFER. Also enable the syscall extensions,
786 * and NOX if available.
787 */
788 movl $MSR_EFER,%ecx
789 rdmsr
790 xorl %eax,%eax /* XXX */
791 orl $(EFER_LME|EFER_SCE),%eax
792 movl RELOC(nox_flag),%ebx
793 cmpl $0,%ebx
794 je .Lskip_NOX
795 orl $(EFER_NXE),%eax
796.Lskip_NOX:
797 wrmsr
798
799 /*
800 * 3. Load %cr3 with pointer to PML4.
801 */
802 movl %esi,%eax
803 movl %eax,%cr3
804
805 /*
806 * 4. Enable paging and the rest of it.
807 */
808 movl %cr0,%eax
809 orl $(CR0_PE|CR0_PG|CR0_NE|CR0_TS|CR0_MP|CR0_WP|CR0_AM),%eax
810 movl %eax,%cr0
811 jmp compat
812compat:
813
814 /*
815 * 5. Not quite done yet, we're now in a compatibility segment, in
816 * legacy mode. We must jump to a long mode segment. Need to set up
817 * a temporary GDT with a long mode segment in it to do that.
818 */
819 movl $RELOC(gdt64_lo),%eax
820 lgdt (%eax)
821 movl $RELOC(farjmp64),%eax
822 ljmp *(%eax)
823
824 .code64
825longmode:
826 /*
827 * 6. Finally, we're in long mode. However, we're still in the identity
828 * mapped area (could not jump out of that earlier because it would
829 * have been a > 32bit jump). We can do that now, so here we go.
830 */
831 movabsq $longmode_hi,%rax
832 jmp *%rax
833
834longmode_hi:
835
836 /*
837 * We left the identity mapped area. Base address of
838 * the temporary gdt64 should now be in high memory.
839 */
840 movq $RELOC(gdt64_hi),%rax
841 lgdt (%rax)
842
843 /*
844 * We have arrived. There's no need anymore for the identity mapping in
845 * low memory, remove it.
846 */
847 movq $KERNBASE,%r8
848
849#if L2_SLOT_KERNBASE > 0
850 movq $(NKL2_KIMG_ENTRIES+1),%rcx
851 leaq (PROC0_PTP2_OFF)(%rsi),%rbx /* old, phys address */
852 addq %r8,%rbx /* new, virt address */
853 killkpt
854#endif
855
856#if L3_SLOT_KERNBASE > 0
857 movq $NKL3_KIMG_ENTRIES,%rcx
858 leaq (PROC0_PTP3_OFF)(%rsi),%rbx /* old, phys address */
859 addq %r8,%rbx /* new, virt address */
860 killkpt
861#endif
862
863 movq $NKL4_KIMG_ENTRIES,%rcx
864 leaq (PROC0_PML4_OFF)(%rsi),%rbx /* old, phys address of PML4 */
865 addq %r8,%rbx /* new, virt address of PML4 */
866 killkpt
867
868 /* Relocate atdevbase. */
869 movq $(TABLESIZE+KERNBASE),%rdx
870 addq %rsi,%rdx
871 movq %rdx,_C_LABEL(atdevbase)(%rip)
872
873 /* Set up bootstrap stack. */
874 leaq (PROC0_STK_OFF)(%rsi),%rax
875 addq %r8,%rax
876 movq %rax,_C_LABEL(lwp0uarea)(%rip)
877 leaq (USPACE-FRAMESIZE)(%rax),%rsp
878 xorq %rbp,%rbp /* mark end of frames */
879
880 xorw %ax,%ax
881 movw %ax,%gs
882 movw %ax,%fs
883
884 /* The first physical page available. */
885 leaq (TABLESIZE)(%rsi),%rdi
886
887#else /* XENPV */
888 /* First, reset the PSL. */
889 pushq $2
890 popfq
891
892 cld
893
894 /*
895 * Xen info:
896 * - %rsi -> start_info struct
897 * - %rsp -> stack, *theoretically* the last used page by Xen bootstrap
898 */
899 movq %rsi,%rbx
900
901 /* Clear BSS. */
902 xorq %rax,%rax
903 movq $_C_LABEL(__bss_start),%rdi
904 movq $_C_LABEL(_end),%rcx
905 subq %rdi,%rcx
906 rep
907 stosb
908
909 /* Copy start_info to a safe place. */
910 movq %rbx,%rsi
911 movq $_C_LABEL(start_info_union),%rdi
912 movq $64,%rcx
913 rep
914 movsq
915
916 /*
917 * Memory layout at start of the day:
918 * - Kernel image
919 * - Page frames list
920 * - start_info struct. we copied it, so it can be recycled.
921 * - xenstore
922 * - console
923 * - Xen bootstrap page tables
924 * - kernel stack. provided by Xen
925 * - guaranteed 512kB padding
926 *
927 * As we want to rebuild our page tables and place our stack
928 * in proc0 struct, all data starting from after console can be
929 * discarded after we've done a little setup.
930 */
931
932 /*
933 * We want our own page tables, and will rebuild them. We will reclaim
934 * the Xen space later, INCLUDING the stack. So we need to switch to a
935 * temporary one now.
936 */
937 movq $tmpstk,%rax
938 subq $8,%rax
939 movq %rax,%rsp
940
941 xorl %eax,%eax
942 cpuid
943 movl %eax,_C_LABEL(cpuid_level)
944
945 movq $cpu_info_primary,%rdi
946 movq %rdi,CPU_INFO_SELF(%rdi) /* ci->ci_self = ci */
947 movq $1,%rsi
948 call cpu_init_msrs /* cpu_init_msrs(ci, true); */
949
950 call xen_locore
951
952 /*
953 * The first VA available is returned by xen_locore in %rax. We
954 * use it as the UAREA, and set up the stack here.
955 */
956 movq %rax,%rsi
957 movq %rsi,_C_LABEL(lwp0uarea)(%rip)
958 leaq (USPACE-FRAMESIZE)(%rsi),%rsp
959 xorq %rbp,%rbp
960
961 /* Clear segment registers. */
962 xorw %ax,%ax
963 movw %ax,%gs
964 movw %ax,%fs
965
966 /* Set first_avail after the DUMMY PAGE (see xen_locore). */
967 movq %rsi,%rdi
968 addq $(USPACE+PAGE_SIZE),%rdi
969 subq $KERNBASE,%rdi /* init_x86_64 wants a physical address */
970#endif /* XENPV */
971
972 pushq %rdi
973 call _C_LABEL(init_bootspace)
974 call _C_LABEL(init_slotspace)
975 popq %rdi
976 call _C_LABEL(init_x86_64)
977 call _C_LABEL(main)
978END(start)

1061/*
1062 * struct lwp *cpu_switchto(struct lwp *oldlwp, struct lwp *newlwp,
1063 * bool returning)
1064 *
1065 * 1. if (oldlwp != NULL), save its context.
1066 * 2. then, restore context of newlwp.
1067 *
1068 * Note that the stack frame layout is known to "struct switchframe" in
1069 * <machine/frame.h> and to the code in cpu_lwp_fork() which initializes
1070 * it for a new lwp.
1071 */
1072ENTRY(cpu_switchto)
1073 pushq %rbx
1074 pushq %r12
1075 pushq %r13
1076 pushq %r14
1077 pushq %r15
1078
1079 movq %rdi,%r13 /* oldlwp */
1080 movq %rsi,%r12 /* newlwp */
1081
1082 testq %r13,%r13 /* oldlwp = NULL ? */
1083 jz .Lskip_save
1084
1085 /* Save old context. */
1086 movq L_PCB(%r13),%rax
1087 movq %rsp,PCB_RSP(%rax)
1088 movq %rbp,PCB_RBP(%rax)
1089.Lskip_save:
1090
1091 /* Switch to newlwp's stack. */
1092 movq L_PCB(%r12),%r14
1093 movq PCB_RSP(%r14),%rsp
1094 movq PCB_RBP(%r14),%rbp
1095
1096 /*
1097 * Set curlwp. This must be globally visible in order to permit
1098 * non-interlocked mutex release.
1099 */
1100 movq %r12,%rcx
1101 xchgq %rcx,CPUVAR(CURLWP)
1102
1103 /* Skip the rest if returning to a pinned LWP. */
1104 testb %dl,%dl /* returning = true ? */
1105 jnz .Lswitch_return
1106
1107#ifdef SVS
1108 movb _C_LABEL(svs_enabled),%dl
1109 testb %dl,%dl
1110 jz .Lskip_svs
1111 callq _C_LABEL(svs_lwp_switch)
1112.Lskip_svs:
1113#endif
1114
1115#ifndef XEN
1116 movq %r13,%rdi
1117 movq %r12,%rsi
1118 callq _C_LABEL(speculation_barrier)
1119#endif
1120
1121 /* Switch ring0 stack */
1122#ifdef SVS
1123 movb _C_LABEL(svs_enabled),%al
1124 testb %al,%al
1125 jz .Lno_svs_switch
1126
1127 movq CPUVAR(RSP0),%rax
1128 movq CPUVAR(TSS),%rdi
1129 movq %rax,TSS_RSP0(%rdi)
1130 jmp .Lring0_switched
1131
1132.Lno_svs_switch:
1133#endif
1134
1135#if !defined(XENPV)
1136 movq PCB_RSP0(%r14),%rax
1137 movq CPUVAR(TSS),%rdi
1138 movq %rax,TSS_RSP0(%rdi)
1139#else
1140 movq %r14,%rdi
1141 callq _C_LABEL(x86_64_switch_context)
1142#endif
1143.Lring0_switched:
1144
1145 /* Switch the dbregs. */
1146 movq %r13,%rdi
1147 movq %r12,%rsi
1148 callq _C_LABEL(x86_dbregs_switch)
1149
1150 movb _C_LABEL(x86_fpu_eager),%dl
1151 testb %dl,%dl
1152 jz .Lno_eagerfpu
1153 movq %r13,%rdi
1154 movq %r12,%rsi
1155 callq _C_LABEL(fpu_eagerswitch)
1156.Lno_eagerfpu:
1157
1158 /* Don't bother with the rest if switching to a system process. */
1159 testl $LW_SYSTEM,L_FLAG(%r12)
1160 jnz .Lswitch_return
1161
1162 /* Is this process using RAS (restartable atomic sequences)? */
1163 movq L_PROC(%r12),%rdi
1164 cmpq $0,P_RASLIST(%rdi)
1165 je .Lno_RAS
1166
1167 /* Handle restartable atomic sequences (RAS). */
1168 movq L_MD_REGS(%r12),%rbx
1169 movq TF_RIP(%rbx),%rsi
1170 call _C_LABEL(ras_lookup)
1171 cmpq $-1,%rax
1172 je .Lno_RAS
1173 movq %rax,TF_RIP(%rbx)
1174.Lno_RAS:
1175
1176 /*
1177 * Restore cr0 including FPU state (may have CR0_TS set). Note that
1178 * IPL_SCHED prevents from FPU interrupt altering the LWP's saved cr0.
1179 */
1180#ifndef XENPV
1181 movl $IPL_HIGH,CPUVAR(ILEVEL)
1182 movl PCB_CR0(%r14),%ecx /* has CR0_TS clear */
1183 movq %cr0,%rdx
1184
1185 /*
1186 * If our floating point registers are on a different CPU,
1187 * set CR0_TS so we'll trap rather than reuse bogus state.
1188 */
1189 cmpq CPUVAR(FPCURLWP),%r12
1190 je .Lskip_TS
1191 orq $CR0_TS,%rcx
1192.Lskip_TS:
1193
1194 /* Reloading CR0 is very expensive - avoid if possible. */
1195 cmpq %rdx,%rcx
1196 je .Lskip_CR0
1197 movq %rcx,%cr0
1198.Lskip_CR0:
1199
1200 /* The 32bit LWPs are handled differently. */
1201 testl $PCB_COMPAT32,PCB_FLAGS(%r14)
1202 jnz .Llwp_32bit
1203
1204.Llwp_64bit:
1205 /* Set default 64bit values in %ds, %es, %fs and %gs. */
1206 movq $GSEL(GUDATA_SEL, SEL_UPL),%rax
1207 movw %ax,%ds
1208 movw %ax,%es
1209 xorq %rax,%rax
1210 movw %ax,%fs
1211 CLI(cx)
1212 SWAPGS
1213 movw %ax,%gs
1214 SWAPGS
1215 STI(cx)
1216
1217 /* Zero out GDT descriptors. */
1218 movq CPUVAR(GDT),%rcx
1219 movq %rax,(GUFS_SEL*8)(%rcx)
1220 movq %rax,(GUGS_SEL*8)(%rcx)
1221
1222 /* Reload 64-bit %fs/%gs MSRs. */
1223 movl $MSR_FSBASE,%ecx
1224 movl PCB_FS(%r14),%eax
1225 movl 4+PCB_FS(%r14),%edx
1226 wrmsr
1227 movl $MSR_KERNELGSBASE,%ecx
1228 movl PCB_GS(%r14),%eax
1229 movl 4+PCB_GS(%r14),%edx
1230 wrmsr
1231
1232 jmp .Lswitch_return
1233
1234.Llwp_32bit:
1235 /* Reload %fs/%gs GDT descriptors. */
1236 movq CPUVAR(GDT),%rcx
1237 movq PCB_FS(%r14),%rax
1238 movq %rax,(GUFS_SEL*8)(%rcx)
1239 movq PCB_GS(%r14),%rax
1240 movq %rax,(GUGS_SEL*8)(%rcx)
1241
1242 /* Set default 32bit values in %ds, %es, %fs and %gs. */
1243 movq L_MD_REGS(%r12),%rbx
1244 movq $GSEL(GUDATA32_SEL, SEL_UPL),%rax
1245 movw %ax,%ds
1246 movw %ax,%es
1247 movw %ax,%fs
1248 CLI(ax)
1249 SWAPGS
1250 movw %ax,%gs
1251 SWAPGS
1252 STI(ax)
1253#else
1254 movq %r12,%rdi
1255 callq _C_LABEL(x86_64_tls_switch)
1256#endif
1257
1258.Lswitch_return:
1259 /* Return to the new LWP, returning 'oldlwp' in %rax. */
1260 movq %r13,%rax
1261 popq %r15
1262 popq %r14
1263 popq %r13
1264 popq %r12
1265 popq %rbx
1266 ret
1267END(cpu_switchto)

1288 incq CPUVAR(NSYSCALL) /* count it atomically */
1289 movq %rsp,L_MD_REGS(%r14) /* save pointer to frame */
1290 movq L_PROC(%r14),%r15

1296 * Disable interrupts to avoid new ASTs (etc) being added and
1297 * to ensure we don't take an interrupt with some of the user
1298 * registers loaded.

1301 /* Check for ASTs on exit to user mode. */
1302 movl L_MD_ASTPENDING(%r14),%eax
1303 orl CPUVAR(WANT_PMAPLOAD),%eax
1304 jnz 9f

1308 jne .Lspl_error

1314 * we're returning to a 32bit LWP (MDL_COMPAT32 set).
1315 *
1316 * In either case, we jump into intrfastexit and return to userland
1317 * with the iret instruction.
1318 */
