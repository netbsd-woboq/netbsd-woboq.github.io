334/*
335 * Machine-dependent startup code
336 */
337void
338cpu_startup(void)
339{
340 int x, y;
341 vaddr_t minaddr, maxaddr;
342 psize_t sz;
343
344 /*
345 * For console drivers that require uvm and pmap to be initialized,
346 * we'll give them one more chance here...
347 */
348 consinit();
349
350 /*
351 * Initialize error message buffer (et end of core).
352 */
353 if (msgbuf_p_cnt == 0)
354 panic("msgbuf paddr map has not been set up");
355 for (x = 0, sz = 0; x < msgbuf_p_cnt; sz += msgbuf_p_seg[x++].sz)
356 continue;
357
358 msgbuf_vaddr = uvm_km_alloc(kernel_map, sz, 0, UVM_KMF_VAONLY);
359 if (msgbuf_vaddr == 0)
360 panic("failed to valloc msgbuf_vaddr");
361
362 for (y = 0, sz = 0; y < msgbuf_p_cnt; y++) {
363 for (x = 0; x < btoc(msgbuf_p_seg[y].sz); x++, sz += PAGE_SIZE)
364 pmap_kenter_pa((vaddr_t)msgbuf_vaddr + sz,
365 msgbuf_p_seg[y].paddr + x * PAGE_SIZE,
366 VM_PROT_READ|VM_PROT_WRITE, 0);
367 }
368
369 pmap_update(pmap_kernel());
370
371 initmsgbuf((void *)msgbuf_vaddr, round_page(sz));
372
373 minaddr = 0;
374
375 /*
376 * Allocate a submap for physio.
377 */
378 phys_map = uvm_km_suballoc(kernel_map, &minaddr, &maxaddr,
379 VM_PHYS_SIZE, 0, false, NULL);
380
381 /*
382 * Create the module map.
383 *
384 * The kernel uses RIP-relative addressing with a maximum offset of
385 * 2GB. Because of that, we can't put the kernel modules in kernel_map
386 * (like i386 does), since kernel_map is too far away in memory from
387 * the kernel sections. So we have to create a special module_map.
388 *
389 * The module map is taken as what is left of the bootstrap memory
390 * created in locore/prekern.
391 */
392 uvm_map_setup(&module_map_store, bootspace.smodule,
393 bootspace.emodule, 0);
394 module_map_store.pmap = pmap_kernel();
395 module_map = &module_map_store;
396
397 /* Say hello. */
398 banner();
399
400#if NISA > 0 || NPCI > 0
401 /* Safe for i/o port / memory space allocation to use malloc now. */
402 x86_bus_space_mallocok();
403#endif
404
405#ifdef __HAVE_PCPU_AREA
406 cpu_pcpuarea_init(&cpu_info_primary);
407#endif
408 gdt_init();
409 x86_64_proc0_pcb_ldt_init();
410
411 cpu_init_tss(&cpu_info_primary);
412#if !defined(XENPV)
413 ltr(cpu_info_primary.ci_tss_sel);
414#endif
415
416 x86_startup();
417}

436void
437x86_64_tls_switch(struct lwp *l)
438{
439 struct cpu_info *ci = curcpu();
440 struct pcb *pcb = lwp_getpcb(l);
441 struct trapframe *tf = l->l_md.md_regs;
442 uint64_t zero = 0;
443
444 /*
445 * Raise the IPL to IPL_HIGH.
446 * FPU IPIs can alter the LWP's saved cr0. Dropping the priority
447 * is deferred until mi_switch(), when cpu_switchto() returns.
448 */
449 (void)splhigh();
450 /*
451 * If our floating point registers are on a different CPU,
452 * set CR0_TS so we'll trap rather than reuse bogus state.
453 */
454 if (l != ci->ci_fpcurlwp) {
455 HYPERVISOR_fpu_taskswitch(1);
456 }
457
458 /* Update segment registers */
459 if (pcb->pcb_flags & PCB_COMPAT32) {
460 update_descriptor(&ci->ci_gdt[GUFS_SEL], &pcb->pcb_fs);
461 update_descriptor(&ci->ci_gdt[GUGS_SEL], &pcb->pcb_gs);
462 setds(GSEL(GUDATA32_SEL, SEL_UPL));
463 setes(GSEL(GUDATA32_SEL, SEL_UPL));
464 setfs(GSEL(GUDATA32_SEL, SEL_UPL));
465 HYPERVISOR_set_segment_base(SEGBASE_GS_USER_SEL, tf->tf_gs);
466 } else {
467 update_descriptor(&ci->ci_gdt[GUFS_SEL], &zero);
468 update_descriptor(&ci->ci_gdt[GUGS_SEL], &zero);
469 setds(GSEL(GUDATA_SEL, SEL_UPL));
470 setes(GSEL(GUDATA_SEL, SEL_UPL));
471 setfs(0);
472 HYPERVISOR_set_segment_base(SEGBASE_GS_USER_SEL, 0);
473 HYPERVISOR_set_segment_base(SEGBASE_FS, pcb->pcb_fs);
474 HYPERVISOR_set_segment_base(SEGBASE_GS_USER, pcb->pcb_gs);
475 }
476}

479/*
480 * Set up proc0's PCB and LDT.
481 */
482static void
483x86_64_proc0_pcb_ldt_init(void)
484{
485 struct lwp *l = &lwp0;
486 struct pcb *pcb = lwp_getpcb(l);
487
488 pcb->pcb_flags = 0;
489 pcb->pcb_fs = 0;
490 pcb->pcb_gs = 0;
491 pcb->pcb_rsp0 = (uvm_lwp_getuarea(l) + USPACE - 16) & ~0xf;
492 pcb->pcb_iopl = IOPL_KPL;
493 pcb->pcb_dbregs = NULL;
494 pcb->pcb_cr0 = rcr0() & ~CR0_TS;
495 l->l_md.md_regs = (struct trapframe *)pcb->pcb_rsp0 - 1;
496
497#if !defined(XENPV)
498 lldt(GSYSSEL(GLDT_SEL, SEL_KPL));
499#else
500 struct physdev_op physop;
501 xen_set_ldt((vaddr_t)ldtstore, LDT_SIZE >> 3);
502 /* Reset TS bit and set kernel stack for interrupt handlers */
503 HYPERVISOR_fpu_taskswitch(1);
504 HYPERVISOR_stack_switch(GSEL(GDATA_SEL, SEL_KPL), pcb->pcb_rsp0);
505 physop.cmd = PHYSDEVOP_SET_IOPL;
506 physop.u.set_iopl.iopl = pcb->pcb_iopl;
507 HYPERVISOR_physdev_op(&physop);
508#endif
509}

511/*
512 * Set up TSS and I/O bitmap.
513 */
514void
515cpu_init_tss(struct cpu_info *ci)
516{
517#ifdef __HAVE_PCPU_AREA
518 const cpuid_t cid = cpu_index(ci);
519#endif
520 struct cpu_tss *cputss;
521 struct nmistore *store;
522 uintptr_t p;
523
524#ifdef __HAVE_PCPU_AREA
525 cputss = (struct cpu_tss *)&pcpuarea->ent[cid].tss;
526#else
527 cputss = (struct cpu_tss *)uvm_km_alloc(kernel_map,
528 sizeof(struct cpu_tss), 0, UVM_KMF_WIRED|UVM_KMF_ZERO);
529#endif
530
531 cputss->tss.tss_iobase = IOMAP_INVALOFF << 16;
532
533 /* DDB stack */
534#ifdef __HAVE_PCPU_AREA
535 p = (vaddr_t)&pcpuarea->ent[cid].ist0;
536#else
537 p = uvm_km_alloc(kernel_map, PAGE_SIZE, 0, UVM_KMF_WIRED|UVM_KMF_ZERO);
538#endif
539 cputss->tss.tss_ist[0] = p + PAGE_SIZE - 16;
540
541 /* double fault */
542#ifdef __HAVE_PCPU_AREA
543 p = (vaddr_t)&pcpuarea->ent[cid].ist1;
544#else
545 p = uvm_km_alloc(kernel_map, PAGE_SIZE, 0, UVM_KMF_WIRED|UVM_KMF_ZERO);
546#endif
547 cputss->tss.tss_ist[1] = p + PAGE_SIZE - 16;
548
549 /* NMI - store a structure at the top of the stack */
550#ifdef __HAVE_PCPU_AREA
551 p = (vaddr_t)&pcpuarea->ent[cid].ist2;
552#else
553 p = uvm_km_alloc(kernel_map, PAGE_SIZE, 0, UVM_KMF_WIRED|UVM_KMF_ZERO);
554#endif
555 cputss->tss.tss_ist[2] = p + PAGE_SIZE - sizeof(struct nmistore);
556 store = (struct nmistore *)(p + PAGE_SIZE - sizeof(struct nmistore));
557 store->cr3 = pmap_pdirpa(pmap_kernel(), 0);
558
559 /* DB */
560#ifdef __HAVE_PCPU_AREA
561 p = (vaddr_t)&pcpuarea->ent[cid].ist3;
562#else
563 p = uvm_km_alloc(kernel_map, PAGE_SIZE, 0, UVM_KMF_WIRED|UVM_KMF_ZERO);
564#endif
565 cputss->tss.tss_ist[3] = p + PAGE_SIZE - 16;
566
567 ci->ci_tss = cputss;
568 ci->ci_tss_sel = tss_alloc(&cputss->tss);
569}

571void
572buildcontext(struct lwp *l, void *catcher, void *f)
573{
574 struct trapframe *tf = l->l_md.md_regs;
575
576 tf->tf_ds = GSEL(GUDATA_SEL, SEL_UPL);
577 tf->tf_es = GSEL(GUDATA_SEL, SEL_UPL);
578 tf->tf_fs = GSEL(GUDATA_SEL, SEL_UPL);
579 tf->tf_gs = GSEL(GUDATA_SEL, SEL_UPL);
580
581 tf->tf_rip = (uint64_t)catcher;
582 tf->tf_cs = GSEL(GUCODE_SEL, SEL_UPL);
583 tf->tf_rflags &= ~PSL_CLEARSIG;
584 tf->tf_rsp = (uint64_t)f;
585 tf->tf_ss = GSEL(GUDATA_SEL, SEL_UPL);
586
587 /* Ensure FP state is sane */
588 fpu_sigreset(l);
589}

599void
600sendsig_siginfo(const ksiginfo_t *ksi, const sigset_t *mask)
601{
602 struct lwp *l = curlwp;
603 struct proc *p = l->l_proc;
604 struct sigacts *ps = p->p_sigacts;
605 int onstack, error;
606 int sig = ksi->ksi_signo;
607 struct sigframe_siginfo *fp, frame;
608 sig_t catcher = SIGACTION(p, sig).sa_handler;
609 struct trapframe *tf = l->l_md.md_regs;
610 char *sp;
611
612 KASSERT(mutex_owned(p->p_lock));
613
614 /* Do we need to jump onto the signal stack? */
615 onstack =
616 (l->l_sigstk.ss_flags & (SS_DISABLE | SS_ONSTACK)) == 0 &&
617 (SIGACTION(p, sig).sa_flags & SA_ONSTACK) != 0;
618
619 /* Allocate space for the signal handler context. */
620 if (onstack)
621 sp = ((char *)l->l_sigstk.ss_sp + l->l_sigstk.ss_size);
622 else
623 /* AMD64 ABI 128-bytes "red zone". */
624 sp = (char *)tf->tf_rsp - 128;
625
626 sp -= sizeof(struct sigframe_siginfo);
627 /* Round down the stackpointer to a multiple of 16 for the ABI. */
628 fp = (struct sigframe_siginfo *)(((unsigned long)sp & ~15) - 8);
629
630 memset(&frame, 0, sizeof(frame));
631 frame.sf_ra = (uint64_t)ps->sa_sigdesc[sig].sd_tramp;
632 frame.sf_si._info = ksi->ksi_info;
633 frame.sf_uc.uc_flags = _UC_SIGMASK;
634 frame.sf_uc.uc_sigmask = *mask;
635 frame.sf_uc.uc_link = l->l_ctxlink;
636 frame.sf_uc.uc_flags |= (l->l_sigstk.ss_flags & SS_ONSTACK)
637 ? _UC_SETSTACK : _UC_CLRSTACK;
638 sendsig_reset(l, sig);
639
640 mutex_exit(p->p_lock);
641 cpu_getmcontext(l, &frame.sf_uc.uc_mcontext, &frame.sf_uc.uc_flags);
642 /* Copyout all the fp regs, the signal handler might expect them. */
643 error = copyout(&frame, fp, sizeof frame);
644 mutex_enter(p->p_lock);
645
646 if (error != 0) {
647 /*
648 * Process has trashed its stack; give it an illegal
649 * instruction to halt it in its tracks.
650 */
651 sigexit(l, SIGILL);
652 /* NOTREACHED */
653 }
654
655 buildcontext(l, catcher, fp);
656
657 tf->tf_rdi = sig;
658 tf->tf_rsi = (uint64_t)&fp->sf_si;
659 tf->tf_rdx = tf->tf_r15 = (uint64_t)&fp->sf_uc;
660
661 /* Remember that we're now on the signal stack. */
662 if (onstack)
663 l->l_sigstk.ss_flags |= SS_ONSTACK;
664
665 if ((vaddr_t)catcher >= VM_MAXUSER_ADDRESS) {
666 /*
667 * process has given an invalid address for the
668 * handler. Stop it, but do not do it before so
669 * we can return the right info to userland (or in core dump)
670 */
671 sigexit(l, SIGILL);
672 /* NOTREACHED */
673 }

829/*
830 * Include or exclude pages in a sparse dump.
831 */
832void
833sparse_dump_mark(void)
834{
835 paddr_t p, pstart, pend;
836 struct vm_page *pg;
837 int i;
838 uvm_physseg_t upm;
839
840 /*
841 * Mark all memory pages, then unmark pages that are uninteresting.
842 * Dereferenceing pg->uobject might crash again if another CPU
843 * frees the object out from under us, but we can't lock anything
844 * so it's a risk we have to take.
845 */
846
847 for (i = 0; i < mem_cluster_cnt; ++i) {
848 pstart = mem_clusters[i].start / PAGE_SIZE;
849 pend = pstart + mem_clusters[i].size / PAGE_SIZE;
850
851 for (p = pstart; p < pend; p++) {
852 setbit(sparse_dump_physmap, p);
853 }
854 }
855 for (upm = uvm_physseg_get_first();
856 uvm_physseg_valid_p(upm);
857 upm = uvm_physseg_get_next(upm)) {
858 paddr_t pfn;
859
860 /*
861 * We assume that seg->start to seg->end are
862 * uvm_page_physload()ed
863 */
864 for (pfn = uvm_physseg_get_start(upm);
865 pfn < uvm_physseg_get_end(upm);
866 pfn++) {
867 pg = PHYS_TO_VM_PAGE(ptoa(pfn));
868
869 if (pg->uanon || (pg->pqflags & PQ_FREE) ||
870 (pg->uobject && pg->uobject->pgops)) {
871 p = VM_PAGE_TO_PHYS(pg) / PAGE_SIZE;
872 clrbit(sparse_dump_physmap, p);
873 }
874 }
875 }

893/*
894 * Abstractly iterate over the collection of memory segments to be
895 * dumped; the callback lacks the customary environment-pointer
896 * argument because none of the current users really need one.
897 *
898 * To be used only after dump_seg_prep is called to set things up.
899 */
900int
901dump_seg_iter(int (*callback)(paddr_t, paddr_t))
902{
903 int error, i;
904
905#define CALLBACK(start,size) do { \
906 error = callback(start,size); \
907 if (error) \
908 return error; \
909} while(0)
910
911 for (i = 0; i < mem_cluster_cnt; ++i) {
912#ifndef NO_SPARSE_DUMP
913 /*
914 * The bitmap is scanned within each memory segment,
915 * rather than over its entire domain, in case any
916 * pages outside of the memory proper have been mapped
917 * into kva; they might be devices that wouldn't
918 * appreciate being arbitrarily read, and including
919 * them could also break the assumption that a sparse
920 * dump will always be smaller than a full one.
921 */
922 if (sparse_dump && sparse_dump_physmap) {
923 paddr_t p, start, end;
924 int lastset;
925
926 start = mem_clusters[i].start;
927 end = start + mem_clusters[i].size;
928 start = rounddown(start, PAGE_SIZE); /* unnecessary? */
929 lastset = 0;
930 for (p = start; p < end; p += PAGE_SIZE) {
931 int thisset = isset(sparse_dump_physmap,
932 p/PAGE_SIZE);
933
934 if (!lastset && thisset)
935 start = p;
936 if (lastset && !thisset)
937 CALLBACK(start, p - start);
938 lastset = thisset;
939 }
940 if (lastset)
941 CALLBACK(start, p - start);
942 } else
943#endif
944 CALLBACK(mem_clusters[i].start, mem_clusters[i].size);
945 }
946 return 0;
947#undef CALLBACK
948}

1020

1301/*
1302 * This is called by main to set dumplo and dumpsize.
1303 * Dumps always skip the first PAGE_SIZE of disk space
1304 * in case there might be a disk label stored there.
1305 * If there is extra space, put dump at the end to
1306 * reduce the chance that swapping trashes it.
1307 *
1308 * Sparse dumps can't placed as close to the end as possible, because
1309 * savecore(8) has to know where to start reading in the dump device
1310 * before it has access to any of the crashed system's state.
1311 *
1312 * Note also that a sparse dump will never be larger than a full one:
1313 * in order to add a phys_ram_seg_t to the header, at least one page
1314 * must be removed.
1315 */
1316void
1317cpu_dumpconf(void)
1318{
1319 int nblks, dumpblks; /* size of dump area */
1320
1321 if (dumpdev == NODEV)
1322 goto bad;
1323 nblks = bdev_size(dumpdev);
1324 if (nblks <= ctod(1))
1325 goto bad;
1326
1327 dumpblks = cpu_dumpsize();
1328 if (dumpblks < 0)
1329 goto bad;
1330
1331 /* dumpsize is in page units, and doesn't include headers. */
1332 dumpsize = cpu_dump_mempagecnt();
1333
1334 dumpblks += ctod(dumpsize);
1335
1336 /* If dump won't fit (incl. room for possible label), punt. */
1337 if (dumpblks > (nblks - ctod(1))) {
1338#ifndef NO_SPARSE_DUMP
1339 /* A sparse dump might (and hopefully will) fit. */
1340 dumplo = ctod(1);
1341#else
1342 /* But if we're not configured for that, punt. */
1343 goto bad;
1344#endif
1345 } else {
1346 /* Put dump at end of partition */
1347 dumplo = nblks - dumpblks;
1348 }
1349
1350
1351 /* Now that we've decided this will work, init ancillary stuff. */
1352 dump_misc_init();
1353 return;
1354
1355 bad:
1356 dumpsize = 0;
1357}

1666void __noasan
1667init_x86_64(paddr_t first_avail)
1668{
1669 extern void consinit(void);
1670 struct region_descriptor region;
1671 struct mem_segment_descriptor *ldt_segp;
1672 int x;
1673 struct pcb *pcb;
1674 extern vaddr_t lwp0uarea;
1675#ifndef XENPV
1676 extern paddr_t local_apic_pa;
1677#endif
1678
1679 KASSERT(first_avail % PAGE_SIZE == 0);
1680
1681#ifdef XENPV
1682 KASSERT(HYPERVISOR_shared_info != NULL);
1683 cpu_info_primary.ci_vcpu = &HYPERVISOR_shared_info->vcpu_info[0];
1684#endif
1685
1686 init_pte();
1687
1688#ifdef KASAN
1689 kasan_early_init((void *)lwp0uarea);
1690#endif
1691
1692 uvm_lwp_setuarea(&lwp0, lwp0uarea);
1693
1694 cpu_probe(&cpu_info_primary);
1695#ifdef SVS
1696 svs_init();
1697#endif
1698 cpu_init_msrs(&cpu_info_primary, true);
1699#ifndef XEN
1700 cpu_speculation_init(&cpu_info_primary);
1701#endif
1702
1703 use_pae = 1; /* PAE always enabled in long mode */
1704
1705 pcb = lwp_getpcb(&lwp0);
1706#ifdef XENPV
1707 mutex_init(&pte_lock, MUTEX_DEFAULT, IPL_VM);
1708 pcb->pcb_cr3 = xen_start_info.pt_base - KERNBASE;
1709#else
1710 pcb->pcb_cr3 = PDPpaddr;
1711#endif
1712
1713#if NISA > 0 || NPCI > 0
1714 x86_bus_space_init();
1715#endif
1716
1717 consinit(); /* XXX SHOULD NOT BE DONE HERE */
1718
1719 /*
1720 * Initialize PAGE_SIZE-dependent variables.
1721 */
1722 uvm_md_init();
1723
1724 uvmexp.ncolors = 2;
1725
1726 avail_start = first_avail;
1727
1728#ifndef XENPV
1729 /*
1730 * Low memory reservations:
1731 * Page 0: BIOS data
1732 * Page 1: BIOS callback (not used yet, for symmetry with i386)
1733 * Page 2: MP bootstrap code (MP_TRAMPOLINE)
1734 * Page 3: ACPI wakeup code (ACPI_WAKEUP_ADDR)
1735 * Page 4: Temporary page table for 0MB-4MB
1736 * Page 5: Temporary page directory
1737 * Page 6: Temporary page map level 3
1738 * Page 7: Temporary page map level 4
1739 */
1740 lowmem_rsvd = 8 * PAGE_SIZE;
1741
1742 /* Initialize the memory clusters (needed in pmap_boostrap). */
1743 init_x86_clusters();
1744#else
1745 /* Parse Xen command line (replace bootinfo) */
1746 xen_parse_cmdline(XEN_PARSE_BOOTFLAGS, NULL);
1747
1748 avail_end = ctob(xen_start_info.nr_pages);
1749 pmap_pa_start = (KERNTEXTOFF - KERNBASE);
1750 pmap_pa_end = avail_end;
1751#endif
1752
1753 /*
1754 * Call pmap initialization to make new kernel address space.
1755 * We must do this before loading pages into the VM system.
1756 */
1757 pmap_bootstrap(VM_MIN_KERNEL_ADDRESS);
1758
1759#ifndef XENPV
1760 /* Internalize the physical pages into the VM system. */
1761 init_x86_vm(avail_start);
1762#else
1763 physmem = xen_start_info.nr_pages;
1764 uvm_page_physload(atop(avail_start), atop(avail_end),
1765 atop(avail_start), atop(avail_end), VM_FREELIST_DEFAULT);
1766#endif
1767
1768 init_x86_msgbuf();
1769
1770#ifdef KASAN
1771 kasan_init();
1772#endif
1773
1774 pmap_growkernel(VM_MIN_KERNEL_ADDRESS + 32 * 1024 * 1024);
1775
1776 kpreempt_disable();
1777
1778#ifndef XENPV
1779 pmap_kenter_pa(local_apic_va, local_apic_pa,
1780 VM_PROT_READ|VM_PROT_WRITE, 0);
1781 pmap_update(pmap_kernel());
1782 memset((void *)local_apic_va, 0, PAGE_SIZE);
1783#endif
1784
1785 pmap_kenter_pa(idt_vaddr, idt_paddr, VM_PROT_READ|VM_PROT_WRITE, 0);
1786 pmap_kenter_pa(gdt_vaddr, gdt_paddr, VM_PROT_READ|VM_PROT_WRITE, 0);
1787 pmap_kenter_pa(ldt_vaddr, ldt_paddr, VM_PROT_READ|VM_PROT_WRITE, 0);
1788 pmap_update(pmap_kernel());
1789 memset((void *)idt_vaddr, 0, PAGE_SIZE);
1790 memset((void *)gdt_vaddr, 0, PAGE_SIZE);
1791 memset((void *)ldt_vaddr, 0, PAGE_SIZE);
1792
1793#ifndef XENPV
1794 pmap_changeprot_local(idt_vaddr, VM_PROT_READ);
1795#endif
1796
1797 pmap_update(pmap_kernel());
1798
1799 idt = (idt_descriptor_t *)idt_vaddr;
1800 gdtstore = (char *)gdt_vaddr;
1801 ldtstore = (char *)ldt_vaddr;
1802
1803 /*
1804 * Make GDT gates and memory segments.
1805 */
1806 set_mem_segment(GDT_ADDR_MEM(gdtstore, GCODE_SEL), 0,
1807 0xfffff, SDT_MEMERA, SEL_KPL, 1, 0, 1);
1808
1809 set_mem_segment(GDT_ADDR_MEM(gdtstore, GDATA_SEL), 0,
1810 0xfffff, SDT_MEMRWA, SEL_KPL, 1, 0, 1);
1811
1812 set_mem_segment(GDT_ADDR_MEM(gdtstore, GUCODE_SEL), 0,
1813 x86_btop(VM_MAXUSER_ADDRESS) - 1, SDT_MEMERA, SEL_UPL, 1, 0, 1);
1814
1815 set_mem_segment(GDT_ADDR_MEM(gdtstore, GUDATA_SEL), 0,
1816 x86_btop(VM_MAXUSER_ADDRESS) - 1, SDT_MEMRWA, SEL_UPL, 1, 0, 1);
1817
1818#ifndef XENPV
1819 set_sys_segment(GDT_ADDR_SYS(gdtstore, GLDT_SEL), ldtstore,
1820 LDT_SIZE - 1, SDT_SYSLDT, SEL_KPL, 0);
1821#endif
1822
1823 /*
1824 * Make LDT memory segments.
1825 */
1826 *(struct mem_segment_descriptor *)(ldtstore + LUCODE_SEL) =
1827 *GDT_ADDR_MEM(gdtstore, GUCODE_SEL);
1828 *(struct mem_segment_descriptor *)(ldtstore + LUDATA_SEL) =
1829 *GDT_ADDR_MEM(gdtstore, GUDATA_SEL);
1830
1831 /*
1832 * 32 bit GDT entries.
1833 */
1834 set_mem_segment(GDT_ADDR_MEM(gdtstore, GUCODE32_SEL), 0,
1835 x86_btop(VM_MAXUSER_ADDRESS32) - 1, SDT_MEMERA, SEL_UPL, 1, 1, 0);
1836
1837 set_mem_segment(GDT_ADDR_MEM(gdtstore, GUDATA32_SEL), 0,
1838 x86_btop(VM_MAXUSER_ADDRESS32) - 1, SDT_MEMRWA, SEL_UPL, 1, 1, 0);
1839
1840 set_mem_segment(GDT_ADDR_MEM(gdtstore, GUFS_SEL), 0,
1841 x86_btop(VM_MAXUSER_ADDRESS32) - 1, SDT_MEMRWA, SEL_UPL, 1, 1, 0);
1842
1843 set_mem_segment(GDT_ADDR_MEM(gdtstore, GUGS_SEL), 0,
1844 x86_btop(VM_MAXUSER_ADDRESS32) - 1, SDT_MEMRWA, SEL_UPL, 1, 1, 0);
1845
1846 /*
1847 * 32 bit LDT entries.
1848 */
1849 ldt_segp = (struct mem_segment_descriptor *)(ldtstore + LUCODE32_SEL);
1850 set_mem_segment(ldt_segp, 0, x86_btop(VM_MAXUSER_ADDRESS32) - 1,
1851 SDT_MEMERA, SEL_UPL, 1, 1, 0);
1852 ldt_segp = (struct mem_segment_descriptor *)(ldtstore + LUDATA32_SEL);
1853 set_mem_segment(ldt_segp, 0, x86_btop(VM_MAXUSER_ADDRESS32) - 1,
1854 SDT_MEMRWA, SEL_UPL, 1, 1, 0);
1855
1856 /* CPU-specific IDT exceptions. */
1857 for (x = 0; x < NCPUIDT; x++) {
1858 int sel, ist;
1859
1860 /* Reset to default. Special cases below */
1861 sel = SEL_KPL;
1862 ist = 0;
1863
1864 idt_vec_reserve(x);
1865
1866 switch (x) {
1867 case 1: /* DB */
1868 ist = 4;
1869 break;
1870 case 2: /* NMI */
1871 ist = 3;
1872 break;
1873 case 3:
1874 case 4:
1875 sel = SEL_UPL;
1876 break;
1877 case 8: /* double fault */
1878 ist = 2;
1879 break;
1880#ifdef XENPV
1881 case 18: /* MCA */
1882 sel |= 0x4; /* Auto EOI/mask */
1883 break;
1884#endif /* XENPV */
1885 default:
1886 break;
1887 }
1888
1889 set_idtgate(&idt[x], x86_exceptions[x], ist, SDT_SYS386IGT,
1890 sel, GSEL(GCODE_SEL, SEL_KPL));
1891 }
1892
1893 /* new-style interrupt gate for syscalls */
1894 idt_vec_reserve(128);
1895 set_idtgate(&idt[128], &IDTVEC(osyscall), 0, SDT_SYS386IGT, SEL_UPL,
1896 GSEL(GCODE_SEL, SEL_KPL));
1897
1898 kpreempt_enable();
1899
1900 setregion(&region, gdtstore, DYNSEL_START - 1);
1901 lgdt(&region);
1902
1903#ifdef XENPV
1904 /* Init Xen callbacks and syscall handlers */
1905 if (HYPERVISOR_set_callbacks(
1906 (unsigned long) hypervisor_callback,
1907 (unsigned long) failsafe_callback,
1908 (unsigned long) Xsyscall))
1909 panic("HYPERVISOR_set_callbacks() failed");
1910#endif /* XENPV */
1911
1912 cpu_init_idt();
1913
1914 init_x86_64_ksyms();
1915
1916#ifndef XENPV
1917 intr_default_setup();
1918#else
1919 events_default_setup();
1920#endif
1921
1922 splraise(IPL_HIGH);
1923 x86_enable_intr();
1924
1925#ifdef DDB
1926 if (boothowto & RB_KDB)
1927 Debugger();
1928#endif
1929#ifdef KGDB
1930 kgdb_port_init();
1931 if (boothowto & RB_KDB) {
1932 kgdb_debug_init = 1;
1933 kgdb_connect(1);
1934 }
1935#endif
1936
1937 pcb->pcb_dbregs = NULL;
1938 x86_dbregs_init();
1939}

2096cpu_mcontext_validate(struct lwp *l, const mcontext_t *mcp)
2097{

2099 struct trapframe *tf = l->l_md.md_regs;

2103 KASSERT((p->p_flag & PK_32) == 0);
2104 gr = mcp->__gregs;
2105
2106 if (((gr[_REG_RFLAGS] ^ tf->tf_rflags) & PSL_USERSTATIC) != 0)
2107 return EINVAL;
2108
2109 sel = gr[_REG_ES] & 0xffff;
2110 if (sel != 0 && !VALID_USER_DSEL(sel))
2111 return EINVAL;
2112
2113 sel = gr[_REG_FS] & 0xffff;
2114 if (sel != 0 && !VALID_USER_DSEL(sel))
2115 return EINVAL;
2116
2117 sel = gr[_REG_GS] & 0xffff;
2118 if (sel != 0 && !VALID_USER_DSEL(sel))
2119 return EINVAL;
2120
2121 sel = gr[_REG_DS] & 0xffff;
2122 if (!VALID_USER_DSEL(sel))
2123 return EINVAL;
2124
2125#ifndef XENPV
2126 sel = gr[_REG_SS] & 0xffff;
2127 if (!VALID_USER_DSEL(sel))
2128 return EINVAL;
2129
2130 sel = gr[_REG_CS] & 0xffff;
2131 if (!VALID_USER_CSEL(sel))
2132 return EINVAL;
2133#endif
2134
2135 if (gr[_REG_RIP] >= VM_MAXUSER_ADDRESS)
2136 return EINVAL;
