64/*
65 * MP startup ...
66 * the stuff from cpu_spinup_trampoline to mp_startup is copied into the
67 * first 640 KB.
68 *
69 * We startup the processors now when the kthreads become ready.
70 * The steps are:
71 * 1) Get the processors running kernel-code from a special
72 * page-table and stack page, do chip identification.
73 * 2) halt the processors waiting for them to be enabled
74 * by a idle-thread
75 */
76
77#include "assym.h"
78#include <machine/asm.h>
79#include <machine/specialreg.h>
80#include <machine/segments.h>
81#include <machine/mpbiosvar.h>
82#include <machine/i82489reg.h>
83#include <machine/gdt.h>
84
85#define _TRMP_LABEL(a) a = . - _C_LABEL(cpu_spinup_trampoline) + MP_TRAMPOLINE
86
87/*
88 * A smp_data structure is packed at the end of the trampoline page. The stack
89 * is right below this structure.
90 */
91#define SMP_DATA (MP_TRAMPOLINE + PAGE_SIZE - 3 * 4)
92#define SMP_DATA_STACK (SMP_DATA + 0 * 4)
93#define SMP_DATA_LARGE (SMP_DATA + 0 * 4)
94#define SMP_DATA_NOX (SMP_DATA + 1 * 4)
95#define SMP_DATA_PDIR (SMP_DATA + 2 * 4)
96
97 .global _C_LABEL(cpu_spinup_trampoline)
98 .global _C_LABEL(cpu_spinup_trampoline_end)
99 .global _C_LABEL(cpu_hatch)
100
101 .text
102 .align 4,0x0
103 .code16
104/* XXX ENTRY() */
105LABEL(cpu_spinup_trampoline)
106 cli
107 xorw %ax,%ax
108 movw %ax,%ds
109 movw %ax,%es
110 movw %ax,%ss
111
112 /* load flat descriptor table */
113#ifdef __clang__
114 lgdt (mptramp_gdt32_desc)
115#else
116 data32 addr32 lgdt (mptramp_gdt32_desc)
117#endif
118
119 /* enable protected mode */
120 movl %cr0,%eax
121 orl $CR0_PE,%eax
122 movl %eax,%cr0
123 ljmpl $0x8,$mp_startup
124
125_TRMP_LABEL(mp_startup)
126 .code32
127
128 movl $0x10,%eax /* data segment */
129 movw %ax,%ds
130 movw %ax,%ss
131 movw %ax,%es
132 movw %ax,%fs
133 movw %ax,%gs
134
135 /* bootstrap stack end */
136 movl $SMP_DATA_STACK,%esp
137
138 /* First, reset the PSL. */
139 pushl $PSL_MBO
140 popfl
141
142 /* Enable PAE, SSE, and PSE if available */
143 movl %cr4,%eax
144 orl $(CR4_PAE|CR4_OSFXSR|CR4_OSXMMEXCPT),%eax
145 movl $SMP_DATA_LARGE,%ecx
146 movl (%ecx),%ecx
147 orl %ecx,%ecx
148 jz .Lno_PSE
149 orl $CR4_PSE,%eax
150.Lno_PSE:
151 movl %eax,%cr4
152
153 /*
154 * Set Long Mode Enable in EFER. Also enable the syscall extensions,
155 * and NOX if available.
156 */
157 movl $MSR_EFER,%ecx
158 rdmsr
159 xorl %eax,%eax
160 orl $(EFER_LME|EFER_SCE),%eax
161 movl $SMP_DATA_NOX,%ebx
162 movl (%ebx),%ebx
163 cmpl $0,%ebx
164 je .Lno_NOX
165 orl $(EFER_NXE),%eax
166.Lno_NOX:
167 wrmsr
168
169 /* Load %cr3. */
170 movl $SMP_DATA_PDIR,%ecx
171 movl (%ecx),%ecx /* guaranteed < 4G */
172 movl %ecx,%cr3 /* load PTD addr into MMU */
173
174 /* Enable paging and the rest of it. */
175 movl %cr0,%eax
176 orl $(CR0_PE|CR0_PG|CR0_NE|CR0_TS|CR0_MP|CR0_WP|CR0_AM),%eax
177 movl %eax,%cr0
178
179 jmp .Lmptramp_compat
180.Lmptramp_compat:
181
182 movl $GSEL(GDATA_SEL, SEL_KPL),%eax /* switch to new segment */
183 movl %eax,%ds
184 movl %eax,%es
185 movl %eax,%ss
186
187 movl $mptramp_gdt64_desc,%eax
188 lgdt (%eax)
189 movl $mptramp_jmp64,%eax
190 ljmp *(%eax)
191
192_TRMP_LABEL(mptramp_jmp64)
193 .long mptramp_longmode
194 .word GSEL(GCODE_SEL, SEL_KPL)
195
196#define GDT_LIMIT 0x17 /* 23 = 3 * 8 - 1 */
197_TRMP_LABEL(mptramp_gdt32)
198 .quad 0x0000000000000000
199 .quad 0x00cf9f000000ffff /* CS */
200 .quad 0x00cf93000000ffff /* DS */
201_TRMP_LABEL(mptramp_gdt32_desc)
202 .word GDT_LIMIT
203 .long mptramp_gdt32
204
205_TRMP_LABEL(mptramp_gdt64)
206 .quad 0x0000000000000000
207 .quad 0x00af9a000000ffff /* CS */
208 .quad 0x00cf92000000ffff /* DS */
209_TRMP_LABEL(mptramp_gdt64_desc)
210 .word GDT_LIMIT
211 .long mptramp_gdt64
212#undef GDT_LIMIT
213
214_TRMP_LABEL(mptramp_longmode)
215 .code64
216 movabsq $_C_LABEL(cpu_spinup_trampoline_end),%rax
217 jmp *%rax
218
219
220_C_LABEL(cpu_spinup_trampoline_end): /* end of code copied to MP_TRAMPOLINE */
221
222 /* Wait until BP has done init sequence. */
2231:
224 movq _C_LABEL(cpu_starting),%rdi
225 pause
226 testq %rdi,%rdi
227 jz 1b
228
229 movq CPU_INFO_IDLELWP(%rdi),%rsi
230 movq L_PCB(%rsi),%rsi
231
232 movq PCB_RSP(%rsi),%rsp
233 movq PCB_RBP(%rsi),%rbp
234
235 movq CPU_INFO_GDT(%rdi),%rax
236 movw $(MAXGDTSIZ-1),-10(%rsp)
237 movq %rax,-8(%rsp)
238 lgdt -10(%rsp)
239
240 /* Switch address space. */
241 movq PCB_CR3(%rsi),%rax
242 movq %rax,%cr3
243
244 movl PCB_CR0(%rsi),%eax
245 movq %rax,%cr0
246
247 call _C_LABEL(cpu_hatch)
248END(cpu_spinup_trampoline)
