126void
127netbsd32_setregs(struct lwp *l, struct exec_package *pack, vaddr_t stack)
128{
129 struct pcb *pcb;
130 struct trapframe *tf;
131 struct proc *p = l->l_proc;
132
133 pcb = lwp_getpcb(l);
134
135#if defined(USER_LDT)
136 pmap_ldt_cleanup(l);
137#endif
138
139 netbsd32_adjust_limits(p);
140
141 fpu_clear(l, pack->ep_osversion >= 699002600
142 ? __NetBSD_NPXCW__ : __NetBSD_COMPAT_NPXCW__);
143 x86_dbregs_clear(l);
144
145 kpreempt_disable();
146 pcb->pcb_flags = PCB_COMPAT32;
147 p->p_flag |= PK_32;
148 l->l_md.md_flags = MDL_COMPAT32; /* force iret not sysret */
149 cpu_segregs32_zero(l);
150 cpu_fsgs_reload(l, LSEL(LUDATA32_SEL, SEL_UPL),
151 LSEL(LUDATA32_SEL, SEL_UPL));
152 kpreempt_enable();
153
154 tf = l->l_md.md_regs;
155 tf->tf_ds = LSEL(LUDATA32_SEL, SEL_UPL);
156 tf->tf_es = LSEL(LUDATA32_SEL, SEL_UPL);
157 tf->tf_rdi = 0;
158 tf->tf_rsi = 0;
159 tf->tf_rbp = 0;
160 tf->tf_rbx = (uint32_t)p->p_psstrp;
161 tf->tf_rdx = 0;
162 tf->tf_rcx = 0;
163 tf->tf_rax = 0;
164 tf->tf_rip = pack->ep_entry;
165 tf->tf_cs = LSEL(LUCODE32_SEL, SEL_UPL);
166 tf->tf_rflags = PSL_USERSET;
167 tf->tf_rsp = stack;
168 tf->tf_ss = LSEL(LUDATA32_SEL, SEL_UPL);
169}

171void
172netbsd32_buildcontext(struct lwp *l, struct trapframe *tf, void *fp,
173 sig_t catcher, int onstack)
174{
175 /*
176 * Build context to run handler in.
177 */
178 tf->tf_ds = GSEL(GUDATA32_SEL, SEL_UPL);
179 tf->tf_es = GSEL(GUDATA32_SEL, SEL_UPL);
180#if 0
181 tf->tf_fs = GSEL(GUDATA32_SEL, SEL_UPL);
182 tf->tf_gs = GSEL(GUDATA32_SEL, SEL_UPL);
183#endif
184
185 /* Ensure FP state is sane. */
186 fpu_sigreset(l);
187
188 tf->tf_rip = (uint64_t)catcher;
189 tf->tf_cs = GSEL(GUCODE32_SEL, SEL_UPL);
190 tf->tf_rflags &= ~PSL_CLEARSIG;
191 tf->tf_rsp = (uint64_t)fp;
192 tf->tf_ss = GSEL(GUDATA32_SEL, SEL_UPL);
193
194 /* Remember that we're now on the signal stack. */
195 if (onstack)
196 l->l_sigstk.ss_flags |= SS_ONSTACK;
197 if ((vaddr_t)catcher >= VM_MAXUSER_ADDRESS32) {
198 /*
199 * process has given an invalid address for the
200 * handler. Stop it, but do not do it before so
201 * we can return the right info to userland (or in core dump)
202 */
203 sigexit(l, SIGILL);
204 /* NOTREACHED */
205 }
206}

208int
209netbsd32_sendsig_siginfo(const ksiginfo_t *ksi, const sigset_t *mask)
210{
211 struct lwp *l = curlwp;
212 struct proc *p = l->l_proc;
213 struct sigacts *ps = p->p_sigacts;
214 int onstack, error;
215 int sig = ksi->ksi_signo;
216 struct netbsd32_sigframe_siginfo *fp, frame;
217 sig_t catcher = SIGACTION(p, sig).sa_handler;
218 struct trapframe *tf = l->l_md.md_regs;
219
220 /* Do we need to jump onto the signal stack? */
221 onstack =
222 (l->l_sigstk.ss_flags & (SS_DISABLE | SS_ONSTACK)) == 0 &&
223 (SIGACTION(p, sig).sa_flags & SA_ONSTACK) != 0;
224
225 /* Allocate space for the signal handler context. */
226 if (onstack)
227 fp = (struct netbsd32_sigframe_siginfo *)
228 ((char *)l->l_sigstk.ss_sp + l->l_sigstk.ss_size);
229 else
230 fp = (struct netbsd32_sigframe_siginfo *)tf->tf_rsp;
231
232 fp--;
233
234 /* Build stack frame for signal trampoline. */
235 switch (ps->sa_sigdesc[sig].sd_vers) {
236 case 0: /* handled by sendsig_sigcontext */
237 case 1: /* handled by sendsig_sigcontext */
238 default: /* unknown version */
239 printf("nsendsig: bad version %d\n",
240 ps->sa_sigdesc[sig].sd_vers);
241 sigexit(l, SIGILL);
242 case 2:
243 break;
244 }
245
246 memset(&frame, 0, sizeof(frame));
247 frame.sf_ra = (uint32_t)(uintptr_t)ps->sa_sigdesc[sig].sd_tramp;
248 frame.sf_signum = sig;
249 frame.sf_sip = (uint32_t)(uintptr_t)&fp->sf_si;
250 frame.sf_ucp = (uint32_t)(uintptr_t)&fp->sf_uc;
251 netbsd32_si_to_si32(&frame.sf_si, (const siginfo_t *)&ksi->ksi_info);
252 frame.sf_uc.uc_flags = _UC_SIGMASK;
253 frame.sf_uc.uc_sigmask = *mask;
254 frame.sf_uc.uc_link = (uint32_t)(uintptr_t)l->l_ctxlink;
255 frame.sf_uc.uc_flags |= (l->l_sigstk.ss_flags & SS_ONSTACK)
256 ? _UC_SETSTACK : _UC_CLRSTACK;
257 sendsig_reset(l, sig);
258
259 mutex_exit(p->p_lock);
260 cpu_getmcontext32(l, &frame.sf_uc.uc_mcontext, &frame.sf_uc.uc_flags);
261 error = copyout(&frame, fp, sizeof(frame));
262 mutex_enter(p->p_lock);
263
264 if (error != 0) {
265 /*
266 * Process has trashed its stack; give it an illegal
267 * instruction to halt it in its tracks.
268 */
269 sigexit(l, SIGILL);
270 /* NOTREACHED */
271 }
272
273 netbsd32_buildcontext(l, tf, fp, catcher, onstack);
274
275 return 0;
276}

288#ifdef COREDUMP
289/*
290 * Dump the machine specific segment at the start of a core dump.
291 */
292struct md_core32 {
293 struct reg32 intreg;
294 struct fpreg32 freg;
295};
296
297int
298cpu_coredump32(struct lwp *l, struct coredump_iostate *iocookie,
299 struct core32 *chdr)
300{
301 struct md_core32 md_core;
302 struct coreseg cseg;
303 int error;
304
305 if (iocookie == NULL) {
306 CORE_SETMAGIC(*chdr, COREMAGIC, MID_I386, 0);
307 chdr->c_hdrsize = ALIGN32(sizeof(*chdr));
308 chdr->c_seghdrsize = ALIGN32(sizeof(cseg));
309 chdr->c_cpusize = sizeof(md_core);
310 chdr->c_nseg++;
311 return 0;
312 }
313
314 /* Save integer registers. */
315 error = netbsd32_process_read_regs(l, &md_core.intreg);
316 if (error)
317 return error;
318
319 /* Save floating point registers. */
320 error = netbsd32_process_read_fpregs(l, &md_core.freg, NULL);
321 if (error)
322 return error;
323
324 CORE_SETMAGIC(cseg, CORESEGMAGIC, MID_I386, CORE_CPU);
325 cseg.c_addr = 0;
326 cseg.c_size = chdr->c_cpusize;
327
328 error = coredump_write(iocookie, UIO_SYSSPACE, &cseg,
329 chdr->c_seghdrsize);
330 if (error)
331 return error;
332
333 return coredump_write(iocookie, UIO_SYSSPACE, &md_core,
334 sizeof(md_core));
335}
336#endif

338int
339netbsd32_ptrace_translate_request(int req)
340{
341
342 switch (req)
343 {
344 case 0 ... PT_FIRSTMACH - 1: return req;
345 case PT32_STEP: return PT_STEP;
346 case PT32_GETREGS: return PT_GETREGS;
347 case PT32_SETREGS: return PT_SETREGS;
348 case PT32_GETFPREGS: return PT_GETFPREGS;
349 case PT32_SETFPREGS: return PT_SETFPREGS;
350 case PT32_GETXMMREGS: return -1;
351 case PT32_SETXMMREGS: return -1;
352 case PT32_GETDBREGS: return PT_GETDBREGS;
353 case PT32_SETDBREGS: return PT_SETDBREGS;
354 case PT32_SETSTEP: return PT_SETSTEP;
355 case PT32_CLEARSTEP: return PT_CLEARSTEP;
356 case PT32_GETXSTATE: return PT_GETXSTATE;
357 case PT32_SETXSTATE: return PT_SETXSTATE;
358 default: return -1;
359 }
360}

362int
363netbsd32_process_read_regs(struct lwp *l, struct reg32 *regs)
364{
365 struct trapframe *tf = l->l_md.md_regs;
366
367 /* XXX avoid sign extension problems with unknown upper bits? */
368 regs->r_gs = tf->tf_gs & 0xffff;
369 regs->r_fs = tf->tf_fs & 0xffff;
370 regs->r_es = tf->tf_es & 0xffff;
371 regs->r_ds = tf->tf_ds & 0xffff;
372 regs->r_eflags = tf->tf_rflags;
373 regs->r_edi = tf->tf_rdi & 0xffffffff;
374 regs->r_esi = tf->tf_rsi & 0xffffffff;
375 regs->r_ebp = tf->tf_rbp & 0xffffffff;
376 regs->r_ebx = tf->tf_rbx & 0xffffffff;
377 regs->r_edx = tf->tf_rdx & 0xffffffff;
378 regs->r_ecx = tf->tf_rcx & 0xffffffff;
379 regs->r_eax = tf->tf_rax & 0xffffffff;
380 regs->r_eip = tf->tf_rip & 0xffffffff;
381 regs->r_cs = tf->tf_cs & 0xffff;
382 regs->r_esp = tf->tf_rsp & 0xffffffff;
383 regs->r_ss = tf->tf_ss & 0xffff;
384
385 return (0);
386}

410 regs->dr[7] = regs64.dr[7] & 0xffffffff;
411
412 return 0;
413}

415int
416netbsd32_process_write_regs(struct lwp *l, const struct reg32 *regs)
417{
418 struct trapframe *tf;
419 struct pcb *pcb;
420
421 tf = l->l_md.md_regs;
422 pcb = lwp_getpcb(l);
423
424 /*
425 * Check for security violations.
426 */
427 if (((regs->r_eflags ^ tf->tf_rflags) & PSL_USERSTATIC) != 0)
428 return EINVAL;
429 if (!VALID_USER_CSEL32(regs->r_cs))
430 return EINVAL;
431 if (regs->r_fs != 0 && !VALID_USER_DSEL32(regs->r_fs) &&
432 !(VALID_USER_FSEL32(regs->r_fs) && pcb->pcb_fs != 0))
433 return EINVAL;
434 if (regs->r_gs != 0 && !VALID_USER_DSEL32(regs->r_gs) &&
435 !(VALID_USER_GSEL32(regs->r_gs) && pcb->pcb_gs != 0))
436 return EINVAL;
437 if (regs->r_es != 0 && !VALID_USER_DSEL32(regs->r_es))
438 return EINVAL;
439 if (!VALID_USER_DSEL32(regs->r_ds) ||
440 !VALID_USER_DSEL32(regs->r_ss))
441 return EINVAL;
442 if (regs->r_eip >= VM_MAXUSER_ADDRESS32)
443 return EINVAL;
444
445 tf->tf_rax = regs->r_eax;
446 tf->tf_rcx = regs->r_ecx;
447 tf->tf_rdx = regs->r_edx;
448 tf->tf_rbx = regs->r_ebx;
449 tf->tf_rsp = regs->r_esp;
450 tf->tf_rbp = regs->r_ebp;
451 tf->tf_rsi = regs->r_esi;
452 tf->tf_rdi = regs->r_edi;
453 tf->tf_rip = regs->r_eip;
454 tf->tf_rflags = regs->r_eflags;
455 tf->tf_cs = regs->r_cs & 0xFFFF;
456 tf->tf_ss = regs->r_ss & 0xFFFF;
457 tf->tf_ds = regs->r_ds & 0xFFFF;
458 tf->tf_es = regs->r_es & 0xFFFF;
459 tf->tf_fs = regs->r_fs & 0xFFFF;
460 tf->tf_gs = regs->r_gs & 0xFFFF;
461
462 return 0;
463}

475int
476netbsd32_process_write_dbregs(struct lwp *l, const struct dbreg32 *regs,
477 size_t sz)
478{
479 size_t i;
480 struct dbreg regs64;
481
482 /* Check that DR0-DR3 contain user-space address */
483 for (i = 0; i < X86_DBREGS; i++) {
484 if (regs->dr[i] >= VM_MAXUSER_ADDRESS32)
485 return EINVAL;
486 }
487
488 if (regs->dr[7] & X86_DR7_GENERAL_DETECT_ENABLE) {
489 return EINVAL;
490 }
491
492 regs64.dr[0] = regs->dr[0];
493 regs64.dr[1] = regs->dr[1];
494 regs64.dr[2] = regs->dr[2];
495 regs64.dr[3] = regs->dr[3];
496
497 regs64.dr[6] = regs->dr[6];
498 regs64.dr[7] = regs->dr[7];
499
500 x86_dbregs_write(l, &regs64);
501 return 0;
502}

504int
505netbsd32_sysarch(struct lwp *l, const struct netbsd32_sysarch_args *uap, register_t *retval)
506{
507 /* {
508 syscallarg(int) op;
509 syscallarg(netbsd32_voidp) parms;
510 } */
511 int error;
512
513 switch (SCARG(uap, op)) {
514 case X86_IOPL:
515 error = x86_iopl(l,
516 NETBSD32PTR64(SCARG(uap, parms)), retval);
517 break;
518 case X86_GET_LDT:
519 error = x86_64_get_ldt32(l,
520 NETBSD32PTR64(SCARG(uap, parms)), retval);
521 break;
522 case X86_SET_LDT:
523 error = x86_64_set_ldt32(l,
524 NETBSD32PTR64(SCARG(uap, parms)), retval);
525 break;
526 case X86_GET_MTRR:
527 error = x86_64_get_mtrr32(l,
528 NETBSD32PTR64(SCARG(uap, parms)), retval);
529 break;
530 case X86_SET_MTRR:
531 error = x86_64_set_mtrr32(l,
532 NETBSD32PTR64(SCARG(uap, parms)), retval);
533 break;
534 default:
535 error = EINVAL;
536 break;
537 }
538 return error;
539}

541#ifdef USER_LDT
542static int
543x86_64_set_ldt32(struct lwp *l, void *args, register_t *retval)
544{
545 struct x86_set_ldt_args32 ua32;
546 struct x86_set_ldt_args ua;
547 union descriptor *descv;
548 int error;
549
550 if ((error = copyin(args, &ua32, sizeof(ua32))) != 0)
551 return (error);
552
553 ua.start = ua32.start;
554 ua.num = ua32.num;
555
556 if (ua.num < 0 || ua.num > 8192)
557 return EINVAL;
558
559 descv = malloc(sizeof(*descv) * ua.num, M_TEMP, M_NOWAIT);
560 if (descv == NULL)
561 return ENOMEM;
562
563 error = copyin((void *)(uintptr_t)ua32.desc, descv,
564 sizeof(*descv) * ua.num);
565 if (error == 0)
566 error = x86_set_ldt1(l, &ua, descv);
567 *retval = ua.start;
568
569 free(descv, M_TEMP);
570 return error;
571}

573static int
574x86_64_get_ldt32(struct lwp *l, void *args, register_t *retval)
575{
576 struct x86_get_ldt_args32 ua32;
577 struct x86_get_ldt_args ua;
578 union descriptor *cp;
579 int error;
580
581 if ((error = copyin(args, &ua32, sizeof(ua32))) != 0)
582 return error;
583
584 ua.start = ua32.start;
585 ua.num = ua32.num;
586
587 if (ua.num < 0 || ua.num > 8192)
588 return EINVAL;
589
590 cp = malloc(ua.num * sizeof(union descriptor), M_TEMP, M_WAITOK);
591 if (cp == NULL)
592 return ENOMEM;
593
594 error = x86_get_ldt1(l, &ua, cp);
595 *retval = ua.num;
596 if (error == 0)
597 error = copyout(cp, (void *)(uintptr_t)ua32.desc,
598 ua.num * sizeof(*cp));
599
600 free(cp, M_TEMP);
601 return error;
602}

605#ifdef MTRR
606static int
607x86_64_get_mtrr32(struct lwp *l, void *args, register_t *retval)
608{
609 struct x86_64_get_mtrr_args32 args32;
610 int error, i;
611 int32_t n;
612 struct mtrr32 *m32p, m32;
613 struct mtrr *m64p, *mp;
614 size_t size;
615
616 m64p = NULL;
617
618 if (mtrr_funcs == NULL)
619 return ENOSYS;
620
621 error = kauth_authorize_machdep(l->l_cred, KAUTH_MACHDEP_MTRR_GET,
622 NULL, NULL, NULL, NULL);
623 if (error)
624 return (error);
625
626 error = copyin(args, &args32, sizeof args32);
627 if (error != 0)
628 return error;
629
630 if (args32.mtrrp == 0) {
631 n = (MTRR_I686_NFIXED_SOFT + MTRR_I686_NVAR_MAX);
632 return copyout(&n, (void *)(uintptr_t)args32.n, sizeof n);
633 }
634
635 error = copyin((void *)(uintptr_t)args32.n, &n, sizeof n);
636 if (error != 0)
637 return error;
638
639 if (n <= 0 || n > (MTRR_I686_NFIXED_SOFT + MTRR_I686_NVAR_MAX))
640 return EINVAL;
641
642 size = n * sizeof(struct mtrr);
643 m64p = kmem_zalloc(size, KM_SLEEP);
644 error = mtrr_get(m64p, &n, l->l_proc, 0);
645 if (error != 0)
646 goto fail;
647 m32p = (struct mtrr32 *)(uintptr_t)args32.mtrrp;
648 mp = m64p;
649 for (i = 0; i < n; i++) {
650 m32.base = mp->base;
651 m32.len = mp->len;
652 m32.type = mp->type;
653 m32.flags = mp->flags;
654 m32.owner = mp->owner;
655 error = copyout(&m32, m32p, sizeof m32);
656 if (error != 0)
657 break;
658 mp++;
659 m32p++;
660 }
661fail:
662 if (m64p != NULL)
663 kmem_free(m64p, size);
664 if (error != 0)
665 n = 0;
666 copyout(&n, (void *)(uintptr_t)args32.n, sizeof n);
667 return error;
668}

791int
792cpu_setmcontext32(struct lwp *l, const mcontext32_t *mcp, unsigned int flags)
793{
794 struct trapframe *tf = l->l_md.md_regs;
795 const __greg32_t *gr = mcp->__gregs;
796 struct proc *p = l->l_proc;
797 int error;
798
799 /* Restore register context, if any. */
800 if ((flags & _UC_CPU) != 0) {
801 /*
802 * Check for security violations.
803 */
804 error = cpu_mcontext32_validate(l, mcp);
805 if (error != 0)
806 return error;
807
808 cpu_fsgs_reload(l, gr[_REG32_FS], gr[_REG32_GS]);
809 tf->tf_es = gr[_REG32_ES] & 0xFFFF;
810 tf->tf_ds = gr[_REG32_DS] & 0xFFFF;
811 /* Only change the user-alterable part of eflags */
812 tf->tf_rflags &= ~PSL_USER;
813 tf->tf_rflags |= (gr[_REG32_EFL] & PSL_USER);
814 tf->tf_rdi = gr[_REG32_EDI];
815 tf->tf_rsi = gr[_REG32_ESI];
816 tf->tf_rbp = gr[_REG32_EBP];
817 tf->tf_rbx = gr[_REG32_EBX];
818 tf->tf_rdx = gr[_REG32_EDX];
819 tf->tf_rcx = gr[_REG32_ECX];
820 tf->tf_rax = gr[_REG32_EAX];
821 tf->tf_rip = gr[_REG32_EIP];
822 tf->tf_cs = gr[_REG32_CS] & 0xFFFF;
823 tf->tf_rsp = gr[_REG32_UESP];
824 tf->tf_ss = gr[_REG32_SS] & 0xFFFF;
825 }
826
827 if ((flags & _UC_TLSBASE) != 0)
828 lwp_setprivate(l, (void *)(uintptr_t)mcp->_mc_tlsbase);
829
830 /* Restore floating point register context, if any. */
831 if ((flags & _UC_FPU) != 0) {
832 /* Assume fxsave context */
833 process_write_fpregs_xmm(l, (const struct fxsave *)
834 &mcp->__fpregs.__fp_reg_set.__fp_xmm_state);
835 }
836
837 mutex_enter(p->p_lock);
838 if (flags & _UC_SETSTACK)
839 l->l_sigstk.ss_flags |= SS_ONSTACK;
840 if (flags & _UC_CLRSTACK)
841 l->l_sigstk.ss_flags &= ~SS_ONSTACK;
842 mutex_exit(p->p_lock);
843
844 return (0);
845}

847void
848cpu_getmcontext32(struct lwp *l, mcontext32_t *mcp, unsigned int *flags)
849{
850 const struct trapframe *tf = l->l_md.md_regs;
851 __greg32_t *gr = mcp->__gregs;
852 __greg32_t ras_eip;
853
854 /* Save register context. */
855 gr[_REG32_GS] = tf->tf_gs & 0xFFFF;
856 gr[_REG32_FS] = tf->tf_fs & 0xFFFF;
857 gr[_REG32_ES] = tf->tf_es & 0xFFFF;
858 gr[_REG32_DS] = tf->tf_ds & 0xFFFF;
859 gr[_REG32_EFL] = tf->tf_rflags;
860 gr[_REG32_EDI] = tf->tf_rdi;
861 gr[_REG32_ESI] = tf->tf_rsi;
862 gr[_REG32_EBP] = tf->tf_rbp;
863 gr[_REG32_EBX] = tf->tf_rbx;
864 gr[_REG32_EDX] = tf->tf_rdx;
865 gr[_REG32_ECX] = tf->tf_rcx;
866 gr[_REG32_EAX] = tf->tf_rax;
867 gr[_REG32_EIP] = tf->tf_rip;
868 gr[_REG32_CS] = tf->tf_cs & 0xFFFF;
869 gr[_REG32_ESP] = tf->tf_rsp;
870 gr[_REG32_UESP] = tf->tf_rsp;
871 gr[_REG32_SS] = tf->tf_ss & 0xFFFF;
872 gr[_REG32_TRAPNO] = tf->tf_trapno;
873 gr[_REG32_ERR] = tf->tf_err;
874
875 if ((ras_eip = (__greg32_t)(uintptr_t)ras_lookup(l->l_proc,
876 (void *) (uintptr_t)gr[_REG32_EIP])) != -1)
877 gr[_REG32_EIP] = ras_eip;
878
879 *flags |= _UC_CPU;
880
881 mcp->_mc_tlsbase = (uint32_t)(uintptr_t)l->l_private;
882 *flags |= _UC_TLSBASE;
883
884 /* Save floating point register context. */
885 process_read_fpregs_xmm(l, (struct fxsave *)
886 &mcp->__fpregs.__fp_reg_set.__fp_xmm_state);
887 memset(&mcp->__fpregs.__fp_pad, 0, sizeof mcp->__fpregs.__fp_pad);
888 *flags |= _UC_FXSAVE | _UC_FPU;
889}

906/*
907 * For various reasons, the amd64 port can't do what the i386 port does,
908 * and relies on catching invalid user contexts on exit from the kernel.
909 * These functions perform the needed checks.
910 */
911
912int
913check_sigcontext32(struct lwp *l, const struct netbsd32_sigcontext *scp)
914{
915 struct pmap *pmap = l->l_proc->p_vmspace->vm_map.pmap;
916 struct trapframe *tf;
917 struct pcb *pcb;
918
919 tf = l->l_md.md_regs;
920 pcb = lwp_getpcb(curlwp);
921
922 if (((scp->sc_eflags ^ tf->tf_rflags) & PSL_USERSTATIC) != 0)
923 return EINVAL;
924
925 if (__predict_false(pmap->pm_ldt != NULL)) {
926 /* Only when the LDT is user-set (with USER_LDT) */
927 if (!USERMODE(scp->sc_cs))
928 return EINVAL;
929 } else {
930 if (!VALID_USER_CSEL32(scp->sc_cs))
931 return EINVAL;
932 if (scp->sc_fs != 0 && !VALID_USER_DSEL32(scp->sc_fs) &&
933 !(VALID_USER_FSEL32(scp->sc_fs) && pcb->pcb_fs != 0))
934 return EINVAL;
935 if (scp->sc_gs != 0 && !VALID_USER_DSEL32(scp->sc_gs) &&
936 !(VALID_USER_GSEL32(scp->sc_gs) && pcb->pcb_gs != 0))
937 return EINVAL;
938 if (scp->sc_es != 0 && !VALID_USER_DSEL32(scp->sc_es))
939 return EINVAL;
940 if (!VALID_USER_DSEL32(scp->sc_ds) || !VALID_USER_DSEL32(scp->sc_ss))
941 return EINVAL;
942 }
943
944 if (scp->sc_eip >= VM_MAXUSER_ADDRESS32)
945 return EINVAL;
946
947 return 0;
948}

950int
951cpu_mcontext32_validate(struct lwp *l, const mcontext32_t *mcp)
952{
953 const __greg32_t *gr;
954 struct trapframe *tf;
955
956 gr = mcp->__gregs;
957 tf = l->l_md.md_regs;
958
959 if (((gr[_REG32_EFL] ^ tf->tf_rflags) & PSL_USERSTATIC) != 0)
960 return EINVAL;
961
962#ifdef USER_LDT
963 /* Userland is allowed to have unfamiliar segment register values */
964 if (!USERMODE(gr[_REG32_CS]))
965 return EINVAL;
966#else
967 struct pcb *pcb = lwp_getpcb(l);
968
969 if (!VALID_USER_CSEL32(gr[_REG32_CS]))
970 return EINVAL;
971 if (gr[_REG32_FS] != 0 && !VALID_USER_DSEL32(gr[_REG32_FS]) &&
972 !(VALID_USER_FSEL32(gr[_REG32_FS]) && pcb->pcb_fs != 0))
973 return EINVAL;
974 if (gr[_REG32_GS] != 0 && !VALID_USER_DSEL32(gr[_REG32_GS]) &&
975 !(VALID_USER_GSEL32(gr[_REG32_GS]) && pcb->pcb_gs != 0))
976 return EINVAL;
977 if (gr[_REG32_ES] != 0 && !VALID_USER_DSEL32(gr[_REG32_ES]))
978 return EINVAL;
979 if (!VALID_USER_DSEL32(gr[_REG32_DS]) ||
980 !VALID_USER_DSEL32(gr[_REG32_SS]))
981 return EINVAL;
982#endif
983
984 if (gr[_REG32_EIP] >= VM_MAXUSER_ADDRESS32)
985 return EINVAL;
986
987 return 0;
988}
