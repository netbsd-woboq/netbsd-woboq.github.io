84int check_sigcontext32(struct lwp *, const struct netbsd32_sigcontext *);
85
86int
87compat_13_netbsd32_sigreturn(struct lwp *l, const struct compat_13_netbsd32_sigreturn_args *uap, register_t *retval)
88{
89 /* {
90 syscallarg(struct netbsd32_sigcontext13 *) sigcntxp;
91 } */
92 struct proc *p = l->l_proc;
93 struct netbsd32_sigcontext13 *scp, context;
94 struct trapframe *tf;
95 sigset_t mask;
96 int error;
97
98 /*
99 * The trampoline code hands us the context.
100 * It is unsafe to keep track of it ourselves, in the event that a
101 * program jumps out of a signal handler.
102 */
103 scp = (struct netbsd32_sigcontext13 *)NETBSD32PTR64(SCARG(uap, sigcntxp));
104 if (copyin((void *)scp, &context, sizeof(*scp)) != 0)
105 return (EFAULT);
106
107 /* Restore register context. */
108 tf = l->l_md.md_regs;
109
110 /*
111 * Check for security violations.
112 */
113 error = check_sigcontext32(l, (const struct netbsd32_sigcontext *)&context);
114 if (error != 0)
115 return error;
116
117 tf->tf_gs = context.sc_gs & 0xFFFF;
118 tf->tf_fs = context.sc_fs & 0xFFFF;
119 tf->tf_es = context.sc_es & 0xFFFF;
120 tf->tf_ds = context.sc_ds & 0xFFFF;
121 tf->tf_rflags = context.sc_eflags;
122 tf->tf_rdi = context.sc_edi;
123 tf->tf_rsi = context.sc_esi;
124 tf->tf_rbp = context.sc_ebp;
125 tf->tf_rbx = context.sc_ebx;
126 tf->tf_rdx = context.sc_edx;
127 tf->tf_rcx = context.sc_ecx;
128 tf->tf_rax = context.sc_eax;
129 tf->tf_rip = context.sc_eip;
130 tf->tf_cs = context.sc_cs & 0xFFFF;
131 tf->tf_rsp = context.sc_esp;
132 tf->tf_ss = context.sc_ss & 0xFFFF;
133
134 mutex_enter(p->p_lock);
135 /* Restore signal stack. */
136 if (context.sc_onstack & SS_ONSTACK)
137 l->l_sigstk.ss_flags |= SS_ONSTACK;
138 else
139 l->l_sigstk.ss_flags &= ~SS_ONSTACK;
140 /* Restore signal mask. */
141 native_sigset13_to_sigset((sigset13_t *)&context.sc_mask, &mask);
142 (void) sigprocmask1(l, SIG_SETMASK, &mask, 0);
143 mutex_exit(p->p_lock);
144
145 return (EJUSTRETURN);
146}
