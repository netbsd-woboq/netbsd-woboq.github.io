91static void
92netbsd32_sendsig_sigcontext(const ksiginfo_t *ksi, const sigset_t *mask)
93{
94 struct lwp *l = curlwp;
95 struct proc *p = l->l_proc;
96 struct trapframe *tf;
97 int sig = ksi->ksi_signo;
98 sig_t catcher = SIGACTION(p, sig).sa_handler;
99 struct netbsd32_sigframe_sigcontext *fp, frame;
100 int onstack, error;
101 struct sigacts *ps = p->p_sigacts;
102
103 tf = l->l_md.md_regs;
104
105 /* Do we need to jump onto the signal stack? */
106 onstack =
107 (l->l_sigstk.ss_flags & (SS_DISABLE | SS_ONSTACK)) == 0 &&
108 (SIGACTION(p, sig).sa_flags & SA_ONSTACK) != 0;
109
110 /* Allocate space for the signal handler context. */
111 if (onstack)
112 fp = (struct netbsd32_sigframe_sigcontext *)
113 ((char *)l->l_sigstk.ss_sp + l->l_sigstk.ss_size);
114 else
115 fp = (struct netbsd32_sigframe_sigcontext *)tf->tf_rsp;
116 fp--;
117
118 memset(&frame, 0, sizeof(frame));
119
120 /* Build stack frame for signal trampoline. */
121 switch (ps->sa_sigdesc[sig].sd_vers) {
122 case 0:
123 frame.sf_ra = (uint32_t)(u_long)p->p_sigctx.ps_sigcode;
124 break;
125 case 1:
126 frame.sf_ra = (uint32_t)(u_long)ps->sa_sigdesc[sig].sd_tramp;
127 break;
128 default:
129 /* Don't know what trampoline version; kill it. */
130 sigexit(l, SIGILL);
131 }
132 frame.sf_signum = sig;
133 frame.sf_code = ksi->ksi_trap;
134 frame.sf_scp = (uint32_t)(u_long)&fp->sf_sc;
135
136 frame.sf_sc.sc_ds = tf->tf_ds & 0xFFFF;
137 frame.sf_sc.sc_es = tf->tf_es & 0xFFFF;
138 frame.sf_sc.sc_fs = tf->tf_fs & 0xFFFF;
139 frame.sf_sc.sc_gs = tf->tf_gs & 0xFFFF;
140
141 frame.sf_sc.sc_eflags = tf->tf_rflags;
142 frame.sf_sc.sc_edi = tf->tf_rdi;
143 frame.sf_sc.sc_esi = tf->tf_rsi;
144 frame.sf_sc.sc_ebp = tf->tf_rbp;
145 frame.sf_sc.sc_ebx = tf->tf_rbx;
146 frame.sf_sc.sc_edx = tf->tf_rdx;
147 frame.sf_sc.sc_ecx = tf->tf_rcx;
148 frame.sf_sc.sc_eax = tf->tf_rax;
149 frame.sf_sc.sc_eip = tf->tf_rip;
150 frame.sf_sc.sc_cs = tf->tf_cs & 0xFFFF;
151 frame.sf_sc.sc_esp = tf->tf_rsp;
152 frame.sf_sc.sc_ss = tf->tf_ss & 0xFFFF;
153 frame.sf_sc.sc_trapno = tf->tf_trapno;
154 frame.sf_sc.sc_err = tf->tf_err;
155
156 /* Save signal stack. */
157 frame.sf_sc.sc_onstack = l->l_sigstk.ss_flags & SS_ONSTACK;
158
159 /* Save signal mask. */
160 frame.sf_sc.sc_mask = *mask;
161
162 sendsig_reset(l, sig);
163
164 mutex_exit(p->p_lock);
165 error = copyout(&frame, fp, sizeof(frame));
166 mutex_enter(p->p_lock);
167
168 if (error != 0) {
169 /*
170 * Process has trashed its stack; give it an illegal
171 * instruction to halt it in its tracks.
172 */
173 sigexit(l, SIGILL);
174 /* NOTREACHED */
175 }
176
177 netbsd32_buildcontext(l, tf, fp, catcher, onstack);
178}

180void
181netbsd32_sendsig_16(const ksiginfo_t *ksi, const sigset_t *mask)
182{
183 if (curproc->p_sigacts->sa_sigdesc[ksi->ksi_signo].sd_vers < 2)
184 netbsd32_sendsig_sigcontext(ksi, mask);
185 else
186 netbsd32_sendsig_siginfo(ksi, mask);
187}

189int
190compat_16_netbsd32___sigreturn14(struct lwp *l,
191 const struct compat_16_netbsd32___sigreturn14_args *uap,
192 register_t *retval)
193{
194 /* {
195 syscallarg(netbsd32_sigcontextp_t) sigcntxp;
196 } */
197 struct netbsd32_sigcontext *scp, context;
198 struct proc *p = l->l_proc;
199 struct trapframe *tf;
200 int error;
201
202 /*
203 * The trampoline code hands us the context.
204 * It is unsafe to keep track of it ourselves, in the event that a
205 * program jumps out of a signal handler.
206 */
207 scp = NETBSD32PTR64(SCARG(uap, sigcntxp));
208 if (copyin(scp, &context, sizeof(*scp)) != 0)
209 return (EFAULT);
210
211 /*
212 * Check for security violations.
213 */
214 error = check_sigcontext32(l, &context);
215 if (error != 0)
216 return error;
217
218 /* Restore register context. */
219 tf = l->l_md.md_regs;
220 tf->tf_ds = context.sc_ds & 0xFFFF;
221 tf->tf_es = context.sc_es & 0xFFFF;
222 cpu_fsgs_reload(l, context.sc_fs, context.sc_gs);
223 tf->tf_rflags = context.sc_eflags;
224 tf->tf_rdi = context.sc_edi;
225 tf->tf_rsi = context.sc_esi;
226 tf->tf_rbp = context.sc_ebp;
227 tf->tf_rbx = context.sc_ebx;
228 tf->tf_rdx = context.sc_edx;
229 tf->tf_rcx = context.sc_ecx;
230 tf->tf_rax = context.sc_eax;
231
232 tf->tf_rip = context.sc_eip;
233 tf->tf_cs = context.sc_cs & 0xFFFF;
234 tf->tf_rsp = context.sc_esp;
235 tf->tf_ss = context.sc_ss & 0xFFFF;
236
237 mutex_enter(p->p_lock);
238 /* Restore signal stack. */
239 if (context.sc_onstack & SS_ONSTACK)
240 l->l_sigstk.ss_flags |= SS_ONSTACK;
241 else
242 l->l_sigstk.ss_flags &= ~SS_ONSTACK;
243 /* Restore signal mask. */
244 (void) sigprocmask1(l, SIG_SETMASK, &context.sc_mask, 0);
245 mutex_exit(p->p_lock);
246
247 return (EJUSTRETURN);
248}
