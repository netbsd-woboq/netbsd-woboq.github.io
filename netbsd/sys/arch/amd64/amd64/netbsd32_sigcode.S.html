32#include "assym.h"
33#include <machine/asm.h>
34#include <sys/syscall.h>
35
36/*
37 * Signal trampoline for 1.6 compatibility; copied to top of user stack.
38 */
39 .code32
40ENTRY(netbsd32_sigcode)
41 /*
42 * Handler has returned here as if we called it. The sigcontext
43 * is on the stack after the 3 args "we" pushed.
44 */
45 leal 12(%esp),%eax /* get pointer to sigcontext */
46 movl %eax,4(%esp) /* put it in the argument slot */
47 /* fake return address already there */
48 movl $SYS_compat_16___sigreturn14,%eax
49 int $0x80 /* enter kernel with args on stack */
50 movl $SYS_exit,%eax
51 int $0x80 /* exit if sigreturn fails */
52 .globl _C_LABEL(netbsd32_esigcode)
53_C_LABEL(netbsd32_esigcode):
54END(netbsd32_sigcode)
55
56/*
57 * There is no NetBSD-1.6 compatibility for native code.
58 * COMPAT_16 is useful for i386 emulation (COMPAT_NETBSD32) only.
59 */
60 .globl _C_LABEL(sigcode), _C_LABEL(esigcode)
61 .set _C_LABEL(sigcode),0
62 .set _C_LABEL(esigcode),0
