32/*
33 * This file may seem a bit stylized, but that so that it's easier to port.
34 * Functions to be implemented here are:
35 *
36 * process_read_regs(proc, regs)
37 * Get the current user-visible register set from the process
38 * and copy it into the regs structure (<machine/reg.h>).
39 * The process is stopped at the time read_regs is called.
40 *
41 * process_write_regs(proc, regs)
42 * Update the current register set from the passed in regs
43 * structure. Take care to avoid clobbering special CPU
44 * registers or privileged bits in the PSL.
45 * The process is stopped at the time write_regs is called.
46 *
47 * process_read_fpregs(proc, regs, sz)
48 * Get the current user-visible register set from the process
49 * and copy it into the regs structure (<machine/reg.h>).
50 * The process is stopped at the time read_fpregs is called.
51 *
52 * process_write_fpregs(proc, regs, sz)
53 * Update the current register set from the passed in regs
54 * structure. Take care to avoid clobbering special CPU
55 * registers or privileged bits in the PSL.
56 * The process is stopped at the time write_fpregs is called.
57 *
58 * process_read_dbregs(proc, regs, sz)
59 * Get the current user-visible register set from the process
60 * and copy it into the regs structure (<machine/reg.h>).
61 * The process is stopped at the time read_dbregs is called.
62 *
63 * process_write_dbregs(proc, regs, sz)
64 * Update the current register set from the passed in regs
65 * structure. Take care to avoid clobbering special CPU
66 * registers or privileged bits in the PSL.
67 * The process is stopped at the time write_dbregs is called.
68 *
69 * process_sstep(proc)
70 * Arrange for the process to trap after executing a single instruction.
71 *
72 * process_set_pc(proc)
73 * Set the process's program counter.
74 */
75

96static inline struct trapframe *process_frame(struct lwp *);
97
98static inline struct trapframe *
99process_frame(struct lwp *l)
100{
101
102 return l->l_md.md_regs;
103}
104
105int
106process_read_regs(struct lwp *l, struct reg *regs)
107{
108 struct trapframe *tf = process_frame(l);
109 struct proc *p = l->l_proc;
110
111 if (p->p_flag & PK_32) {
112 return EINVAL;
113 }
114
115 regs->regs[_REG_RDI] = tf->tf_rdi;
116 regs->regs[_REG_RSI] = tf->tf_rsi;
117 regs->regs[_REG_RDX] = tf->tf_rdx;
118 regs->regs[_REG_R10] = tf->tf_r10;
119 regs->regs[_REG_R8] = tf->tf_r8;
120 regs->regs[_REG_R9] = tf->tf_r9;
121 /* argX not touched */
122 regs->regs[_REG_RCX] = tf->tf_rcx;
123 regs->regs[_REG_R11] = tf->tf_r11;
124 regs->regs[_REG_R12] = tf->tf_r12;
125 regs->regs[_REG_R13] = tf->tf_r13;
126 regs->regs[_REG_R14] = tf->tf_r14;
127 regs->regs[_REG_R15] = tf->tf_r15;
128 regs->regs[_REG_RBP] = tf->tf_rbp;
129 regs->regs[_REG_RBX] = tf->tf_rbx;
130 regs->regs[_REG_RAX] = tf->tf_rax;
131 regs->regs[_REG_GS] = 0;
132 regs->regs[_REG_FS] = 0;
133 regs->regs[_REG_ES] = GSEL(GUDATA_SEL, SEL_UPL);
134 regs->regs[_REG_DS] = GSEL(GUDATA_SEL, SEL_UPL);
135 regs->regs[_REG_TRAPNO] = tf->tf_trapno;
136 regs->regs[_REG_ERR] = tf->tf_err;
137 regs->regs[_REG_RIP] = tf->tf_rip;
138 regs->regs[_REG_CS] = LSEL(LUCODE_SEL, SEL_UPL);
139 regs->regs[_REG_RFLAGS] = tf->tf_rflags;
140 regs->regs[_REG_RSP] = tf->tf_rsp;
141 regs->regs[_REG_SS] = LSEL(LUDATA_SEL, SEL_UPL);
142
143 return 0;
144}

174int
175process_write_regs(struct lwp *l, const struct reg *regp)
176{
177 struct trapframe *tf = process_frame(l);
178 struct proc *p = l->l_proc;
179 int error;
180 const long *regs = regp->regs;
181
182 if (p->p_flag & PK_32) {
183 return EINVAL;
184 }
185
186 /*
187 * Check for security violations. Note that struct regs is compatible
188 * with the __gregs array in mcontext_t.
189 */
190 error = cpu_mcontext_validate(l, (const mcontext_t *)regs);
191 if (error != 0)
192 return error;
193
194 tf->tf_rdi = regs[_REG_RDI];
195 tf->tf_rsi = regs[_REG_RSI];
196 tf->tf_rdx = regs[_REG_RDX];
197 tf->tf_r10 = regs[_REG_R10];
198 tf->tf_r8 = regs[_REG_R8];
199 tf->tf_r9 = regs[_REG_R9];
200 /* argX not touched */
201 tf->tf_rcx = regs[_REG_RCX];
202 tf->tf_r11 = regs[_REG_R11];
203 tf->tf_r12 = regs[_REG_R12];
204 tf->tf_r13 = regs[_REG_R13];
205 tf->tf_r14 = regs[_REG_R14];
206 tf->tf_r15 = regs[_REG_R15];
207 tf->tf_rbp = regs[_REG_RBP];
208 tf->tf_rbx = regs[_REG_RBX];
209 tf->tf_rax = regs[_REG_RAX];
210 tf->tf_gs = 0;
211 tf->tf_fs = 0;
212 tf->tf_es = GSEL(GUDATA_SEL, SEL_UPL);
213 tf->tf_ds = GSEL(GUDATA_SEL, SEL_UPL);
214 /* trapno, err not touched */
215 tf->tf_rip = regs[_REG_RIP];
216 tf->tf_cs = LSEL(LUCODE_SEL, SEL_UPL);
217 tf->tf_rflags = regs[_REG_RFLAGS];
218 tf->tf_rsp = regs[_REG_RSP];
219 tf->tf_ss = LSEL(LUDATA_SEL, SEL_UPL);
220
221#ifdef XENPV
222 /* see comment in cpu_setmcontext */
223 tf->tf_ss = GSEL(GUDATA_SEL, SEL_UPL);
224 tf->tf_cs = GSEL(GUCODE_SEL, SEL_UPL);
225#endif
226
227 return 0;
228}

265int
266process_sstep(struct lwp *l, int sstep)
267{
268 struct trapframe *tf = process_frame(l);
269
270 if (sstep)
271 tf->tf_rflags |= PSL_T;
272 else
273 tf->tf_rflags &= ~PSL_T;
274
275 return 0;
276}

278int
279process_set_pc(struct lwp *l, void *addr)
280{
281 struct trapframe *tf = process_frame(l);
282 struct proc *p = l->l_proc;
283
284 if (p->p_flag & PK_32) {
285 return EINVAL;
286 }
287
288 if ((uint64_t)addr >= VM_MAXUSER_ADDRESS)
289 return EINVAL;
290 tf->tf_rip = (uint64_t)addr;
291
292 return 0;
293}

317int
318ptrace_machdep_dorequest(
319 struct lwp *l,
320 struct lwp *lt,
321 int req,
322 void *addr,
323 int data
324)
325{
326 struct uio uio;
327 struct iovec iov;
328 struct vmspace *vm;
329 int error;
330 int write = 0;
331
332 switch (req) {
333 case PT_SETXSTATE:
334 write = 1;
335
336 /* FALLTHROUGH */
337 case PT_GETXSTATE:
338 /* write = 0 done above. */
339 if (!process_machdep_validxstate(lt->l_proc))
340 return EINVAL;
341 if (__predict_false(l->l_proc->p_flag & PK_32)) {
342 struct netbsd32_iovec *user_iov = addr;
343 iov.iov_base = NETBSD32PTR64(user_iov->iov_base);
344 iov.iov_len = user_iov->iov_len;
345 } else {
346 struct iovec *user_iov;
347 user_iov = (struct iovec*)addr;
348 iov.iov_base = user_iov->iov_base;
349 iov.iov_len = user_iov->iov_len;
350 }
351
352 error = proc_vmspace_getref(l->l_proc, &vm);
353 if (error)
354 return error;
355 if (iov.iov_len > sizeof(struct xstate))
356 iov.iov_len = sizeof(struct xstate);
357 uio.uio_iov = &iov;
358 uio.uio_iovcnt = 1;
359 uio.uio_offset = 0;
360 uio.uio_resid = iov.iov_len;
361 uio.uio_rw = write ? UIO_WRITE : UIO_READ;
362 uio.uio_vmspace = vm;
363 error = process_machdep_doxstate(l, lt, &uio);
364 uvmspace_free(vm);
365 return error;
366 }
367
368#ifdef DIAGNOSTIC
369 panic("ptrace_machdep: impossible");
370#endif
371
372 return 0;
373}

375/*
376 * The following functions are used by both ptrace(2) and procfs.
377 */
378
379int
380process_machdep_doxstate(struct lwp *curl, struct lwp *l, struct uio *uio)
381 /* curl: tracer */
382 /* l: traced */
383{
384 int error;
385 struct xstate r;
386 char *kv;
387 ssize_t kl;
388
389 memset(&r, 0, sizeof(r));
390 kl = MIN(uio->uio_iov->iov_len, sizeof(r));
391 kv = (char *) &r;
392
393 kv += uio->uio_offset;
394 kl -= uio->uio_offset;
395 if (kl > uio->uio_resid)
396 kl = uio->uio_resid;
397
398 if (kl < 0)
399 error = EINVAL;
400 else
401 error = process_machdep_read_xstate(l, &r);
402 if (error == 0)
403 error = uiomove(kv, kl, uio);
404 if (error == 0 && uio->uio_rw == UIO_WRITE)
405 error = process_machdep_write_xstate(l, &r);
406
407 uio->uio_offset = 0;
408 return error;
409}

411int
412process_machdep_validxstate(struct proc *p)
413{
414
415 if (p->p_flag & PK_SYSTEM)
416 return 0;
417
418 return 1;
