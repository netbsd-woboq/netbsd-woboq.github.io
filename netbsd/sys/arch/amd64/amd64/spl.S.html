82 * int splraise(int s);
83 */
84ENTRY(splraise)
85 movl CPUVAR(ILEVEL),%eax
86 cmpl %edi,%eax
87 cmoval %eax,%edi
88 movl %edi,CPUVAR(ILEVEL)
89 ret
90END(splraise)

93/*
94 * Xsoftintr()
95 *
96 * Switch to the LWP assigned to handle interrupts from the given
97 * source. We borrow the VM context from the interrupted LWP.
98 *
99 * On entry:
100 *
101 * %rax intrsource
102 * %r13 address to return to
103 */
104IDTVEC(softintr)
105 /* set up struct switchframe */
106 pushq $_C_LABEL(softintr_ret)
107 pushq %rbx
108 pushq %r12
109 pushq %r13
110 pushq %r14
111 pushq %r15
112
113 movl $IPL_HIGH,CPUVAR(ILEVEL)
114 movq CPUVAR(CURLWP),%r15
115 movq IS_LWP(%rax),%rdi /* switch to handler LWP */
116 movq L_PCB(%rdi),%rdx
117 movq L_PCB(%r15),%rcx
118 movq %rdi,CPUVAR(CURLWP)
119
120#ifdef KASAN
121 /* clear the new stack */
122 pushq %rax
123 pushq %rdx
124 pushq %rcx
125 callq _C_LABEL(kasan_softint)
126 popq %rcx
127 popq %rdx
128 popq %rax
129#endif
130
131 /* save old context */
132 movq %rsp,PCB_RSP(%rcx)
133 movq %rbp,PCB_RBP(%rcx)
134
135 /* switch to the new stack */
136 movq PCB_RSP0(%rdx),%rsp
137
138 /* dispatch */
139 sti
140 movq %r15,%rdi /* interrupted LWP */
141 movl IS_MAXLEVEL(%rax),%esi /* ipl to run at */
142 call _C_LABEL(softint_dispatch)/* run handlers */
143 cli
144
145 /* restore old context */
146 movq L_PCB(%r15),%rcx
147 movq PCB_RSP(%rcx),%rsp
148
149 xchgq %r15,CPUVAR(CURLWP) /* must be globally visible */
150 popq %r15 /* unwind switchframe */
151 addq $(5 * 8),%rsp
152 jmp *%r13 /* back to Xspllower/Xdoreti */
153IDTVEC_END(softintr)

155/*
156 * softintr_ret()
157 *
158 * Trampoline function that gets returned to by cpu_switchto() when
159 * an interrupt handler blocks. On entry:
160 *
161 * %rax prevlwp from cpu_switchto()
162 */
163ENTRY(softintr_ret)
164 incl CPUVAR(MTX_COUNT) /* re-adjust after mi_switch */
165 movl $0,L_CTXSWTCH(%rax) /* %rax from cpu_switchto */
166 cli
167 jmp *%r13 /* back to Xspllower/Xdoreti */
168END(softintr_ret)

217/*
218 * void spllower(int s);
219 *
220 * Must be the same size as cx8_spllower(). This must use
221 * pushf/cli/popf as it is used early in boot where interrupts
222 * are disabled via eflags/IE.
223 */
224ENTRY(spllower)
225 cmpl CPUVAR(ILEVEL),%edi
226 jae 1f
227 movl CPUVAR(IUNMASK)(,%rdi,4),%edx
228 pushf
229 cli
230 testl CPUVAR(IPENDING),%edx
231 jnz 2f
232 movl %edi,CPUVAR(ILEVEL)
233 popf
2341:
235 ret
236 ret
2372:
238 popf
239 jmp _C_LABEL(Xspllower)
2403:
241 .space 16
242 .align 16
243END(spllower)
244LABEL(spllower_end)

246/*
247 * void cx8_spllower(int s);
248 *
249 * For cmpxchg8b, edx/ecx are the high words and eax/ebx the low.
250 *
251 * edx : eax = old level / old ipending
252 * ecx : ebx = new level / old ipending
253 */
254ENTRY(cx8_spllower)
255 movl CPUVAR(ILEVEL),%edx
256 movq %rbx,%r8
257 cmpl %edx,%edi /* new level is lower? */
258 jae 1f
2590:
260 movl CPUVAR(IPENDING),%eax
261 movl %edi,%ecx
262 testl %eax,CPUVAR(IUNMASK)(,%rcx,4)/* deferred interrupts? */
263 movl %eax,%ebx
264 /*
265 * On the P4 this jump is cheaper than patching in junk
266 * using cmov. Is cmpxchg expensive if it fails?
267 */
268 jnz 2f
269 cmpxchg8b CPUVAR(ISTATE) /* swap in new ilevel */
270 jnz 0b
2711:
272 movq %r8,%rbx
273 ret
2742:
275 movq %r8,%rbx
276 .type _C_LABEL(cx8_spllower_patch), @function
277LABEL(cx8_spllower_patch)
278 jmp _C_LABEL(Xspllower)
279
280 .align 16
281END(cx8_spllower_patch)
282END(cx8_spllower)
283LABEL(cx8_spllower_end)
284#endif /* !XENPV */

286/*
287 * void Xspllower(int s);
288 *
289 * Process pending interrupts.
290 *
291 * Important registers:
292 * ebx - cpl
293 * r13 - address to resume loop at
294 *
295 * It is important that the bit scan instruction is bsr, it will get
296 * the highest 2 bits (currently the IPI and clock handlers) first,
297 * to avoid deadlocks where one CPU sends an IPI, another one is at
298 * splhigh() and defers it, lands in here via splx(), and handles
299 * a lower-prio one first, which needs to take the kernel lock -->
300 * the sending CPU will never see the that CPU accept the IPI
301 * (see pmap_tlb_shootnow).
302 */
303 nop
304 .align 4 /* Avoid confusion with cx8_spllower_end */
305
306IDTVEC(spllower)
307 pushq %rbx
308 pushq %r13
309 pushq %r12
310 movl %edi,%ebx
311 leaq 1f(%rip),%r13 /* address to resume loop at */
3121: movl %ebx,%eax /* get cpl */
313#if !defined(XENPV)
314 movl CPUVAR(IUNMASK)(,%rax,4),%eax
315 CLI(si)
316 andl CPUVAR(IPENDING),%eax /* any non-masked bits left? */
317 jz 2f
318 bsrl %eax,%eax
319 btrl %eax,CPUVAR(IPENDING)
320 movq CPUVAR(ISOURCES)(,%rax,8),%rax
321 jmp *IS_RECURSE(%rax)
322#endif
3232:
324#if defined(XEN)
325 movl CPUVAR(XUNMASK)(,%rax,4),%eax
326 CLI(si)
327 andl CPUVAR(XPENDING),%eax /* any non-masked bits left? */
328 jz 3f
329 bsrl %eax,%eax
330 btrl %eax,CPUVAR(XPENDING)
331 movq CPUVAR(XSOURCES)(,%rax,8),%rax
332 jmp *IS_RECURSE(%rax)
333#endif
3343:
335 movl %ebx,CPUVAR(ILEVEL)
336 STI(si)
337 popq %r12
338 popq %r13
339 popq %rbx
340 ret
341IDTVEC_END(spllower)

343/*
344 * void Xdoreti(void);
345 *
346 * Handle return from interrupt after device handler finishes.
347 *
348 * Important registers:
349 * ebx - cpl to restore
350 * r13 - address to resume loop at
351 */
352IDTVEC(doreti)
353 popq %rbx /* get previous priority */
354 decl CPUVAR(IDEPTH)
355 leaq 1f(%rip),%r13
3561: movl %ebx,%eax
357#if !defined(XENPV)
358 movl CPUVAR(IUNMASK)(,%rax,4),%eax
359 CLI(si)
360 andl CPUVAR(IPENDING),%eax
361 jz 2f
362 bsrl %eax,%eax /* slow, but not worth optimizing */
363 btrl %eax,CPUVAR(IPENDING)
364 movq CPUVAR(ISOURCES)(,%rax,8),%rax
365 jmp *IS_RESUME(%rax)
366#endif
3672:
368#if defined(XEN)
369 movl CPUVAR(XUNMASK)(,%rax,4),%eax
370 CLI(si)
371 andl CPUVAR(XPENDING),%eax
372 jz 3f
373 bsrl %eax,%eax /* slow, but not worth optimizing */
374 btrl %eax,CPUVAR(XPENDING)
375 movq CPUVAR(XSOURCES)(,%rax,8),%rax
376 jmp *IS_RESUME(%rax)
377#endif
3783: /* Check for ASTs on exit to user mode. */
379 movl %ebx,CPUVAR(ILEVEL)
3805:
381 testb $SEL_RPL,TF_CS(%rsp)
382 jz 6f
383
384 .type _C_LABEL(doreti_checkast), @function
385LABEL(doreti_checkast)
386 movq CPUVAR(CURLWP),%r14
387 CHECK_ASTPENDING(%r14)
388 je 3f
389 CLEAR_ASTPENDING(%r14)
390 STI(si)
391 movl $T_ASTFLT,TF_TRAPNO(%rsp) /* XXX undo later.. */
392 /* Pushed T_ASTFLT into tf_trapno on entry. */
393 movq %rsp,%rdi
394 call _C_LABEL(trap)
395 CLI(si)
396 jmp doreti_checkast
3973:
398 CHECK_DEFERRED_SWITCH
399 jnz 9f
4006:
401 INTRFASTEXIT
4029:
403 STI(si)
404 call _C_LABEL(do_pmap_load)
405 CLI(si)
406 jmp doreti_checkast /* recheck ASTs */
407END(doreti_checkast)
408IDTVEC_END(doreti)
