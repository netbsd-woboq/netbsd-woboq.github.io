163static void
164onfault_restore(struct trapframe *frame, void *onfault, int error)
165{
166 frame->tf_rip = (uintptr_t)onfault;
167 frame->tf_rax = error;
168}
169
170static void *
171onfault_handler(const struct pcb *pcb, const struct trapframe *tf)
172{
173 struct onfault_table {
174 uintptr_t start;
175 uintptr_t end;
176 void *handler;
177 };
178 extern const struct onfault_table onfault_table[];
179 const struct onfault_table *p;
180 uintptr_t pc;
181
182 if (pcb->pcb_onfault != NULL) {
183 return pcb->pcb_onfault;
184 }
185
186 pc = tf->tf_rip;
187 for (p = onfault_table; p->start; p++) {
188 if (p->start <= pc && pc < p->end) {
189 return p->handler;
190 }
191 }
192 return NULL;
193}

195static void
196trap_print(const struct trapframe *frame, const lwp_t *l)
197{
198 const int type = frame->tf_trapno;
199
200 if (frame->tf_trapno < trap_types) {
201 printf("fatal %s", trap_type[type]);
202 } else {
203 printf("unknown trap %d", type);
204 }
205 printf(" in %s mode\n", (type & T_USER) ? "user" : "supervisor");
206
207 printf("trap type %d code %#lx rip %#lx cs %#lx rflags %#lx cr2 %#lx "
208 "ilevel %#x rsp %#lx\n",
209 type, frame->tf_err, (u_long)frame->tf_rip, frame->tf_cs,
210 frame->tf_rflags, rcr2(), curcpu()->ci_ilevel, frame->tf_rsp);
211
212 printf("curlwp %p pid %d.%d lowest kstack %p\n",
213 l, l->l_proc->p_pid, l->l_lid, KSTACK_LOWEST_ADDR(l));
214}

216void
217nmitrap(struct trapframe *frame)
218{
219 const int type = T_NMI;
220
221 if (nmi_dispatch(frame))
222 return;
223 /* NMI can be hooked up to a pushbutton for debugging */
224 if (kgdb_trap(type, frame))
225 return;
226 if (kdb_trap(type, 0, frame))
227 return;
228 /* machine/parity/power fail/"kitchen sink" faults */
229
230 x86_nmi();
231}

233void
234doubletrap(struct trapframe *frame)
235{
236 const int type = T_DOUBLEFLT;
237 struct lwp *l = curlwp;
238
239 trap_print(frame, l);
240
241 if (kdb_trap(type, 0, frame))
242 return;
243 if (kgdb_trap(type, frame))
244 return;
245
246 panic("double fault");
247}

249/*
250 * trap(frame): exception, fault, and trap interface to BSD kernel.
251 *
252 * This common code is called from assembly language IDT gate entry routines
253 * that prepare a suitable stack frame, and restore this frame after the
254 * exception has been processed. Note that the effect is as if the arguments
255 * were passed call by reference.
256 *
257 * Note that the fpu traps (07 T_DNA, 10 T_ARITHTRAP and 13 T_XMM)
258 * jump directly into the code in x86/fpu.c so they get processed
259 * without interrupts being enabled.
260 */
261void
262trap(struct trapframe *frame)
263{
264 struct lwp *l = curlwp;
265 struct proc *p;
266 struct pcb *pcb;
267 extern char kcopy_fault[];
268 extern char IDTVEC(osyscall)[];
269 extern char IDTVEC(syscall32)[];
270 ksiginfo_t ksi;
271 void *onfault;
272 int type, error;
273 uint64_t cr2;
274 bool pfail;
275
276 if (__predict_true(l != NULL)) {
277 pcb = lwp_getpcb(l);
278 p = l->l_proc;
279 } else {
280 /*
281 * this can happen eg. on break points in early on boot.
282 */
283 pcb = NULL;
284 p = NULL;
285 }
286 type = frame->tf_trapno;
287
288 if (!KERNELMODE(frame->tf_cs)) {
289 type |= T_USER;
290 l->l_md.md_regs = frame;
291 LWP_CACHE_CREDS(l, p);
292 }
293
294#ifdef KDTRACE_HOOKS
295 /*
296 * A trap can occur while DTrace executes a probe. Before
297 * executing the probe, DTrace blocks re-scheduling and sets
298 * a flag in its per-cpu flags to indicate that it doesn't
299 * want to fault. On returning from the probe, the no-fault
300 * flag is cleared and finally re-scheduling is enabled.
301 *
302 * If the DTrace kernel module has registered a trap handler,
303 * call it and if it returns non-zero, assume that it has
304 * handled the trap and modified the trap frame so that this
305 * function can return normally.
306 */
307 if ((type == T_PROTFLT || type == T_PAGEFLT) &&
308 dtrace_trap_func != NULL) {
309 if ((*dtrace_trap_func)(frame, type)) {
310 return;
311 }
312 }
313#endif
314
315 switch (type) {
316
317 default:
318 we_re_toast:
319 trap_print(frame, l);
320
321 if (kdb_trap(type, 0, frame))
322 return;
323 if (kgdb_trap(type, frame))
324 return;
325 /*
326 * If this is a breakpoint, don't panic if we're not connected.
327 */
328 if (type == T_BPTFLT && kgdb_disconnected()) {
329 printf("kgdb: ignored %s\n", trap_type[type]);
330 return;
331 }
332 panic("trap");
333 /*NOTREACHED*/
334
335 case T_PROTFLT:
336 case T_SEGNPFLT:
337 case T_ALIGNFLT:
338 case T_STKFLT:
339 case T_TSSFLT:
340 if (p == NULL)
341 goto we_re_toast;
342
343 /* Check for copyin/copyout fault. */
344 onfault = onfault_handler(pcb, frame);
345 if (onfault != NULL) {
346 onfault_restore(frame, onfault, EFAULT);
347 return;
348 }
349
350 goto we_re_toast;
351
352 case T_PROTFLT|T_USER: /* protection fault */
353#if defined(COMPAT_NETBSD32) && defined(COMPAT_10)
354
355/*
356 * XXX This code currently not included in loadable module; it is
357 * only included in built-in modules.
358 */
359 {
360 static const char lcall[7] = { 0x9a, 0, 0, 0, 0, 7, 0 };
361 const size_t sz = sizeof(lcall);
362 char tmp[sz];
363
364 /* Check for the oosyscall lcall instruction. */
365 if (p->p_emul == &emul_netbsd32 &&
366 frame->tf_rip < VM_MAXUSER_ADDRESS32 - sz &&
367 copyin((void *)frame->tf_rip, tmp, sz) == 0 &&
368 memcmp(tmp, lcall, sz) == 0) {
369
370 /* Advance past the lcall. */
371 frame->tf_rip += sz;
372
373 /* Do the syscall. */
374 p->p_md.md_syscall(frame);
375 goto out;
376 }
377 }
378#endif
379 /* FALLTHROUGH */
380 case T_TSSFLT|T_USER:
381 case T_SEGNPFLT|T_USER:
382 case T_STKFLT|T_USER:
383 case T_ALIGNFLT|T_USER:
384 KSI_INIT_TRAP(&ksi);
385 ksi.ksi_trap = type & ~T_USER;
386 ksi.ksi_addr = (void *)frame->tf_rip;
387 switch (type) {
388 case T_SEGNPFLT|T_USER:
389 case T_STKFLT|T_USER:
390 ksi.ksi_signo = SIGBUS;
391 ksi.ksi_code = BUS_ADRERR;
392 break;
393 case T_TSSFLT|T_USER:
394 ksi.ksi_signo = SIGBUS;
395 ksi.ksi_code = BUS_OBJERR;
396 break;
397 case T_ALIGNFLT|T_USER:
398 ksi.ksi_signo = SIGBUS;
399 ksi.ksi_code = BUS_ADRALN;
400 break;
401 case T_PROTFLT|T_USER:
402 ksi.ksi_signo = SIGSEGV;
403 ksi.ksi_code = SEGV_ACCERR;
404 break;
405 default:
406 KASSERT(0);
407 break;
408 }
409 goto trapsignal;
410
411 case T_PRIVINFLT|T_USER: /* privileged instruction fault */
412 case T_FPOPFLT|T_USER: /* coprocessor operand fault */
413 KSI_INIT_TRAP(&ksi);
414 ksi.ksi_signo = SIGILL;
415 ksi.ksi_trap = type & ~T_USER;
416 ksi.ksi_addr = (void *) frame->tf_rip;
417 switch (type) {
418 case T_PRIVINFLT|T_USER:
419 ksi.ksi_code = ILL_PRVOPC;
420 break;
421 case T_FPOPFLT|T_USER:
422 ksi.ksi_code = ILL_COPROC;
423 break;
424 default:
425 KASSERT(0);
426 break;
427 }
428 goto trapsignal;
429
430 case T_ASTFLT|T_USER:
431 /* Allow process switch. */
432 //curcpu()->ci_data.cpu_nast++;
433 if (l->l_pflag & LP_OWEUPC) {
434 l->l_pflag &= ~LP_OWEUPC;
435 ADDUPROF(l);
436 }
437 /* Allow a forced task switch. */
438 if (curcpu()->ci_want_resched) {
439 preempt();
440 }
441 goto out;
442
443 case T_BOUND|T_USER:
444 case T_OFLOW|T_USER:
445 case T_DIVIDE|T_USER:
446 KSI_INIT_TRAP(&ksi);
447 ksi.ksi_signo = SIGFPE;
448 ksi.ksi_trap = type & ~T_USER;
449 ksi.ksi_addr = (void *)frame->tf_rip;
450 switch (type) {
451 case T_BOUND|T_USER:
452 ksi.ksi_code = FPE_FLTSUB;
453 break;
454 case T_OFLOW|T_USER:
455 ksi.ksi_code = FPE_INTOVF;
456 break;
457 case T_DIVIDE|T_USER:
458 ksi.ksi_code = FPE_INTDIV;
459 break;
460 default:
461#ifdef DIAGNOSTIC
462 panic("unhandled type %x\n", type);
463#endif
464 break;
465 }
466 goto trapsignal;
467
468 case T_PAGEFLT:
469 /* Allow page faults in kernel mode. */
470 if (__predict_false(l == NULL))
471 goto we_re_toast;
472
473 onfault = pcb->pcb_onfault;
474
475 if (cpu_intr_p() || (l->l_pflag & LP_INTR) != 0) {
476 goto we_re_toast;
477 }
478
479 cr2 = rcr2();
480
481 if (frame->tf_err & PGEX_X) {
482 /* SMEP might have brought us here */
483 if (cr2 < VM_MAXUSER_ADDRESS) {
484 printf("prevented execution of %p (SMEP)\n",
485 (void *)cr2);
486 goto we_re_toast;
487 }
488 }
489
490 if ((frame->tf_err & PGEX_P) &&
491 cr2 < VM_MAXUSER_ADDRESS) {
492 /* SMAP might have brought us here */
493 if (onfault_handler(pcb, frame) == NULL) {
494 printf("prevented access to %p (SMAP)\n",
495 (void *)cr2);
496 goto we_re_toast;
497 }
498 }
499
500 goto faultcommon;
501
502 case T_PAGEFLT|T_USER: {
503 register vaddr_t va;
504 register struct vmspace *vm;
505 register struct vm_map *map;
506 vm_prot_t ftype;
507 extern struct vm_map *kernel_map;
508
509 cr2 = rcr2();
510 if (p->p_emul->e_usertrap != NULL &&
511 (*p->p_emul->e_usertrap)(l, cr2, frame) != 0)
512 return;
513faultcommon:
514 vm = p->p_vmspace;
515 if (__predict_false(vm == NULL)) {
516 goto we_re_toast;
517 }
518 pcb->pcb_cr2 = cr2;
519 va = trunc_page((vaddr_t)cr2);
520 /*
521 * It is only a kernel address space fault iff:
522 * 1. (type & T_USER) == 0 and
523 * 2. pcb_onfault not set or
524 * 3. pcb_onfault set but supervisor space fault
525 * The last can occur during an exec() copyin where the
526 * argument space is lazy-allocated.
527 */
528 if (type == T_PAGEFLT && va >= VM_MIN_KERNEL_ADDRESS)
529 map = kernel_map;
530 else
531 map = &vm->vm_map;
532 if (frame->tf_err & PGEX_W)
533 ftype = VM_PROT_WRITE;
534 else if (frame->tf_err & PGEX_X)
535 ftype = VM_PROT_EXECUTE;
536 else
537 ftype = VM_PROT_READ;
538
539#ifdef DIAGNOSTIC
540 if (map == kernel_map && va == 0) {
541 printf("trap: bad kernel access at %lx\n", va);
542 goto we_re_toast;
543 }
544#endif
545 /* Fault the original page in. */
546 onfault = pcb->pcb_onfault;
547 pcb->pcb_onfault = NULL;
548 error = uvm_fault(map, va, ftype);
549 pcb->pcb_onfault = onfault;
550 if (error == 0) {
551 if (map != kernel_map && (void *)va >= vm->vm_maxsaddr)
552 uvm_grow(p, va);
553
554 pfail = false;
555 while (type == T_PAGEFLT) {
556 /*
557 * we need to switch pmap now if we're in
558 * the middle of copyin/out.
559 *
560 * but we don't need to do so for kcopy as
561 * it never touch userspace.
562 */
563 kpreempt_disable();
564 if (curcpu()->ci_want_pmapload) {
565 onfault = onfault_handler(pcb, frame);
566 if (onfault != kcopy_fault) {
567 pmap_load();
568 }
569 }
570 /*
571 * We need to keep the pmap loaded and
572 * so avoid being preempted until back
573 * into the copy functions. Disable
574 * interrupts at the hardware level before
575 * re-enabling preemption. Interrupts
576 * will be re-enabled by 'iret' when
577 * returning back out of the trap stub.
578 * They'll only be re-enabled when the
579 * program counter is once again in
580 * the copy functions, and so visible
581 * to cpu_kpreempt_exit().
582 */
583#ifndef XENPV
584 x86_disable_intr();
585#endif
586 l->l_nopreempt--;
587 if (l->l_nopreempt > 0 || !l->l_dopreempt ||
588 pfail) {
589 return;
590 }
591#ifndef XENPV
592 x86_enable_intr();
593#endif
594 /*
595 * If preemption fails for some reason,
596 * don't retry it. The conditions won't
597 * change under our nose.
598 */
599 pfail = kpreempt(0);
600 }
601 goto out;
602 }
603
604 if (type == T_PAGEFLT) {
605 onfault = onfault_handler(pcb, frame);
606 if (onfault != NULL) {
607 onfault_restore(frame, onfault, error);
608 return;
609 }
610
611 printf("uvm_fault(%p, 0x%lx, %d) -> %x\n",
612 map, va, ftype, error);
613 goto we_re_toast;
614 }
615
616 KSI_INIT_TRAP(&ksi);
617 ksi.ksi_trap = type & ~T_USER;
618 ksi.ksi_addr = (void *)cr2;
619 switch (error) {
620 case EINVAL:
621 ksi.ksi_signo = SIGBUS;
622 ksi.ksi_code = BUS_ADRERR;
623 break;
624 case EACCES:
625 ksi.ksi_signo = SIGSEGV;
626 ksi.ksi_code = SEGV_ACCERR;
627 error = EFAULT;
628 break;
629 case ENOMEM:
630 ksi.ksi_signo = SIGKILL;
631 printf("UVM: pid %d.%d (%s), uid %d killed: "
632 "out of swap\n", p->p_pid, l->l_lid, p->p_comm,
633 l->l_cred ? kauth_cred_geteuid(l->l_cred) : -1);
634 break;
635 default:
636 ksi.ksi_signo = SIGSEGV;
637 ksi.ksi_code = SEGV_MAPERR;
638 break;
639 }
640
641 SIGDEBUG(frame, &ksi, error);
642 (*p->p_emul->e_trapsignal)(l, &ksi);
643 break;
644 }
645
646 case T_TRCTRAP:
647 /*
648 * Ignore debug register trace traps due to
649 * accesses in the user's address space, which
650 * can happen under several conditions such as
651 * if a user sets a watchpoint on a buffer and
652 * then passes that buffer to a system call.
653 * We still want to get TRCTRAPS for addresses
654 * in kernel space because that is useful when
655 * debugging the kernel.
656 */
657 if (x86_dbregs_user_trap())
658 break;
659
660 /* Check whether they single-stepped into a lcall. */
661 if (frame->tf_rip == (uint64_t)IDTVEC(osyscall) ||
662 frame->tf_rip == (uint64_t)IDTVEC(syscall32)) {
663 frame->tf_rflags &= ~PSL_T;
664 return;
665 }
666 goto we_re_toast;
667
668 case T_BPTFLT|T_USER: /* bpt instruction fault */
669 case T_TRCTRAP|T_USER: /* trace trap */
670 /*
671 * Don't go single-stepping into a RAS.
672 */
673 if (p->p_raslist == NULL ||
674 (ras_lookup(p, (void *)frame->tf_rip) == (void *)-1)) {
675 KSI_INIT_TRAP(&ksi);
676 ksi.ksi_signo = SIGTRAP;
677 ksi.ksi_trap = type & ~T_USER;
678 if (x86_dbregs_user_trap()) {
679 x86_dbregs_store_dr6(l);
680 ksi.ksi_code = TRAP_DBREG;
681 } else if (type == (T_BPTFLT|T_USER))
682 ksi.ksi_code = TRAP_BRKPT;
683 else
684 ksi.ksi_code = TRAP_TRACE;
685 (*p->p_emul->e_trapsignal)(l, &ksi);
686 }
687 break;
688 }
689
690 if ((type & T_USER) == 0)
691 return;
692out:
693 userret(l);
694 return;
695trapsignal:
696 SIGDEBUG(frame, &ksi, 0);
697 (*p->p_emul->e_trapsignal)(l, &ksi);
698 userret(l);
699}

701/*
702 * startlwp: start of a new LWP.
703 */
704void
705startlwp(void *arg)
706{
707 ucontext_t *uc = arg;
708 lwp_t *l = curlwp;
709 int error __diagused;
710
711 error = cpu_setmcontext(l, &uc->uc_mcontext, uc->uc_flags);
712 KASSERT(error == 0);
713
714 kmem_free(uc, sizeof(ucontext_t));
715 userret(l);
716}

719static void
720frame_dump(const struct trapframe *tf, struct pcb *pcb)
721{
722
723 printf("trapframe %p\n", tf);
724 printf("rip %#018lx rsp %#018lx rfl %#018lx\n",
