93/*
94 * Macros for interrupt entry, call to handler, and exit.
95 *
96 * XXX
97 * The interrupt frame is set up to look like a trap frame. This may be a
98 * waste. The only handler which needs a frame is the clock handler, and it
99 * only needs a few bits. Xdoreti() needs a trap frame for handling ASTs, but
100 * it could easily convert the frame on demand.
101 *
102 * The direct costs of setting up a trap frame are two pushq's (error code and
103 * trap number), an addl to get rid of these, and pushing and popping the
104 * callee-saved registers %esi, %edi, %ebx, and %ebp twice.
105 *
106 * If the interrupt frame is made more flexible, INTR can push %eax first and
107 * decide the ipending case with less overhead, e.g., by avoiding loading the
108 * segment registers.
109 */
110

353 * This macro defines the generic stub code. Its arguments modify it
354 * for specific PICs.

357#define INTRSTUB(name, num, early_ack, late_ack, mask, unmask, level_mask) \
358IDTVEC(recurse_ ## name ## num) ;\
359 INTR_RECURSE_HWFRAME ;\
360 subq $8,%rsp ;\
361 pushq $T_ASTFLT /* trap # for doing ASTs */ ;\
362 INTR_RECURSE_ENTRY ;\
363 jmp 1f ;\
364IDTVEC_END(recurse_ ## name ## num) ;\
365IDTVEC(resume_ ## name ## num) \
3661: movq $IREENT_MAGIC,TF_ERR(%rsp) ;\
367 movl %ebx,%r13d ;\
368 movq CPUVAR(ISOURCES) + (num) * 8,%r14 ;\
369 movl IS_MAXLEVEL(%r14),%ebx ;\
370 jmp 1f ;\
371IDTVEC_END(resume_ ## name ## num) ;\
372IDTVEC(handle_ ## name ## num) ;\
373 movq CPUVAR(ISOURCES) + (num) * 8,%r14 ;\
374 mask(num) /* mask it in hardware */ ;\
375 early_ack(num) /* and allow other intrs */ ;\
376 testq %r14,%r14 ;\
377 jz 9f /* stray */ ;\
378 movl IS_MAXLEVEL(%r14),%ebx ;\
379 movl CPUVAR(ILEVEL),%r13d ;\
380 cmpl %ebx,%r13d ;\
381 jae 10f /* currently masked; hold it */ ;\
382 incq CPUVAR(NINTR) /* statistical info */ ;\
383 incq IS_EVCNT(%r14) ;\
3841: \
385 pushq %r13 /* save for Xdoreti */ ;\
386 movl %ebx,CPUVAR(ILEVEL) ;\
387 sti ;\
388 incl CPUVAR(IDEPTH) ;\
389 movq IS_HANDLERS(%r14),%rbx ;\
3906: \
391 movl IH_LEVEL(%rbx),%r12d ;\
392 cmpl %r13d,%r12d ;\
393 jle 7f ;\
394 movq %rsp,%rsi ;\
395 movq IH_ARG(%rbx),%rdi ;\
396 movl %r12d,CPUVAR(ILEVEL) ;\
397 call *IH_FUN(%rbx) /* call it */ ;\
398 movq IH_NEXT(%rbx),%rbx /* next handler in chain */ ;\
399 testq %rbx,%rbx ;\
400 jnz 6b ;\
4015: \
402 cli ;\
403 unmask(num) /* unmask it in hardware */ ;\
404 late_ack(num) ;\
405 sti ;\
406 jmp _C_LABEL(Xdoreti) /* lower spl and do ASTs */ ;\
4077: \
408 cli ;\
409 orl $(1 << num),CPUVAR(IPENDING) ;\
410 level_mask(num) ;\
411 late_ack(num) ;\
412 sti ;\
413 jmp _C_LABEL(Xdoreti) /* lower spl and do ASTs */ ;\
41410: \
415 cli ;\
416 orl $(1 << num),CPUVAR(IPENDING) ;\
417 level_mask(num) ;\
418 late_ack(num) ;\
419 INTRFASTEXIT ;\
4209: \
421 unmask(num) ;\
422 late_ack(num) ;\
423 INTRFASTEXIT ;\
424IDTVEC_END(handle_ ## name ## num) ;\
425 TEXT_USER_BEGIN ;\
426IDTVEC(intr_ ## name ## num) ;\
427 pushq $0 /* dummy error code */ ;\
428 pushq $T_ASTFLT /* trap # for doing ASTs */ ;\
429 INTRENTRY ;\
430 jmp _C_LABEL(Xhandle_ ## name ## num) ;\
431IDTVEC_END(intr_ ## name ## num) ;\
432 TEXT_USER_END
