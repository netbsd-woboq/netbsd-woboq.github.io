33#ifndef _X86_64_DISKLABEL_H_
34#define _X86_64_DISKLABEL_H_
35
36#if defined(__x86_64__) || defined(HAVE_NBTOOL_CONFIG_H)
37
38#define LABELUSESMBR 1 /* use MBR partitionning */
39#define LABELSECTOR 1 /* sector containing label */
40#define LABELOFFSET 0 /* offset of label in sector */
41#define MAXPARTITIONS 16 /* number of partitions */
42#define RAW_PART 3 /* raw partition: XX?d (XXX) */
43
44/*
45 * We use the highest bit of the minor number for the partition number.
46 * This maintains backward compatibility with device nodes created before
47 * MAXPARTITIONS was increased.
48 */
49/* Pull in MBR partition definitions. */
50#if HAVE_NBTOOL_CONFIG_H
51#include <nbinclude/sys/bootblock.h>
52#else
53#include <sys/bootblock.h>
54#endif /* HAVE_NBTOOL_CONFIG_H */
55
56#ifndef __ASSEMBLER__
57#if HAVE_NBTOOL_CONFIG_H
58#include <nbinclude/sys/dkbad.h>
59#else
60#include <sys/dkbad.h>
61#endif /* HAVE_NBTOOL_CONFIG_H */
62struct cpu_disklabel {
63#define __HAVE_DISKLABEL_DKBAD
64 struct dkbad bad;
65};
66#endif
67
68#else /* __x86_64__ */
69
70#include <i386/disklabel.h>
71
72#endif /* __x86_64__ */
73
74#endif /* _X86_64_DISKLABEL_H_ */
