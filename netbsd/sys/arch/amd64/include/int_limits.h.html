33#define _AMD64_INT_LIMITS_H_

39#ifdef __x86_64__
40
41/*
42 * 7.18.2 Limits of specified-width integer types
43 */
44
45/* 7.18.2.1 Limits of exact-width integer types */
46
47/* minimum values of exact-width signed integer types */
48#define INT8_MIN (-0x7f-1) /* int8_t */
49#define INT16_MIN (-0x7fff-1) /* int16_t */
50#define INT32_MIN (-0x7fffffff-1) /* int32_t */
51#define INT64_MIN (-0x7fffffffffffffffL-1) /* int64_t */
52
53/* maximum values of exact-width signed integer types */
54#define INT8_MAX 0x7f /* int8_t */
55#define INT16_MAX 0x7fff /* int16_t */
56#define INT32_MAX 0x7fffffff /* int32_t */
57#define INT64_MAX 0x7fffffffffffffffL /* int64_t */
58
59/* maximum values of exact-width unsigned integer types */
60#define UINT8_MAX 0xff /* uint8_t */
61#define UINT16_MAX 0xffff /* uint16_t */
62#define UINT32_MAX 0xffffffffU /* uint32_t */
63#define UINT64_MAX 0xffffffffffffffffUL /* uint64_t */
64
65/* 7.18.2.2 Limits of minimum-width integer types */
66
67/* minimum values of minimum-width signed integer types */
68#define INT_LEAST8_MIN (-0x7f-1) /* int_least8_t */
69#define INT_LEAST16_MIN (-0x7fff-1) /* int_least16_t */
70#define INT_LEAST32_MIN (-0x7fffffff-1) /* int_least32_t */
71#define INT_LEAST64_MIN (-0x7fffffffffffffffL-1) /* int_least64_t */
72
73/* maximum values of minimum-width signed integer types */
74#define INT_LEAST8_MAX 0x7f /* int_least8_t */
75#define INT_LEAST16_MAX 0x7fff /* int_least16_t */
76#define INT_LEAST32_MAX 0x7fffffff /* int_least32_t */
77#define INT_LEAST64_MAX 0x7fffffffffffffffL /* int_least64_t */
78
79/* maximum values of minimum-width unsigned integer types */
80#define UINT_LEAST8_MAX 0xff /* uint_least8_t */
81#define UINT_LEAST16_MAX 0xffff /* uint_least16_t */
82#define UINT_LEAST32_MAX 0xffffffffU /* uint_least32_t */
83#define UINT_LEAST64_MAX 0xffffffffffffffffUL /* uint_least64_t */
84
85/* 7.18.2.3 Limits of fastest minimum-width integer types */
86
87/* minimum values of fastest minimum-width signed integer types */
88#define INT_FAST8_MIN (-0x7fffffff-1) /* int_fast8_t */
89#define INT_FAST16_MIN (-0x7fffffff-1) /* int_fast16_t */
90#define INT_FAST32_MIN (-0x7fffffff-1) /* int_fast32_t */
91#define INT_FAST64_MIN (-0x7fffffffffffffffLL-1) /* int_fast64_t */
92
93/* maximum values of fastest minimum-width signed integer types */
94#define INT_FAST8_MAX 0x7fffffff /* int_fast8_t */
95#define INT_FAST16_MAX 0x7fffffff /* int_fast16_t */
96#define INT_FAST32_MAX 0x7fffffff /* int_fast32_t */
97#define INT_FAST64_MAX 0x7fffffffffffffffLL /* int_fast64_t */
98
99/* maximum values of fastest minimum-width unsigned integer types */
100#define UINT_FAST8_MAX 0xffffffffU /* uint_fast8_t */
101#define UINT_FAST16_MAX 0xffffffffU /* uint_fast16_t */
102#define UINT_FAST32_MAX 0xffffffffU /* uint_fast32_t */
103#define UINT_FAST64_MAX 0xffffffffffffffffULL /* uint_fast64_t */
104
105/* 7.18.2.4 Limits of integer types capable of holding object pointers */
106
107#define INTPTR_MIN (-0x7fffffffffffffffL-1) /* intptr_t */
108#define INTPTR_MAX 0x7fffffffffffffffL /* intptr_t */
109#define UINTPTR_MAX 0xffffffffffffffffUL /* uintptr_t */
110
111/* 7.18.2.5 Limits of greatest-width integer types */
112
113#define INTMAX_MIN (-0x7fffffffffffffffL-1) /* intmax_t */
114#define INTMAX_MAX 0x7fffffffffffffffL /* intmax_t */
115#define UINTMAX_MAX 0xffffffffffffffffUL /* uintmax_t */
116
117
118/*
119 * 7.18.3 Limits of other integer types
120 */
121
122/* limits of ptrdiff_t */
123#define PTRDIFF_MIN (-0x7fffffffffffffffL-1) /* ptrdiff_t */
124#define PTRDIFF_MAX 0x7fffffffffffffffL /* ptrdiff_t */
125
126/* limits of sig_atomic_t */
127#define SIG_ATOMIC_MIN (-0x7fffffff-1) /* sig_atomic_t */
128#define SIG_ATOMIC_MAX 0x7fffffff /* sig_atomic_t */
129
130/* limit of size_t */
131#define SIZE_MAX 0xffffffffffffffffUL /* size_t */
