186#ifndef XENPV
187#define pmap_pa2pte(a) (a)
188#define pmap_pte2pa(a) ((a) & PTE_FRAME)
189#define pmap_pte_set(p, n) do { *(p) = (n); } while (0)
190#define pmap_pte_cas(p, o, n) atomic_cas_64((p), (o), (n))
191#define pmap_pte_testset(p, n) \
192 atomic_swap_ulong((volatile unsigned long *)p, n)
193#define pmap_pte_setbits(p, b) \
194 atomic_or_ulong((volatile unsigned long *)p, b)
195#define pmap_pte_clearbits(p, b) \
196 atomic_and_ulong((volatile unsigned long *)p, ~(b))
197#define pmap_pte_flush() /* nothing */

199extern kmutex_t pte_lock;

201static __inline pt_entry_t
202pmap_pa2pte(paddr_t pa)
203{
204 return (pt_entry_t)xpmap_ptom_masked(pa);
205}
206
207static __inline paddr_t
208pmap_pte2pa(pt_entry_t pte)
209{
210 return xpmap_mtop_masked(pte & PTE_FRAME);
211}
212
213static __inline void
214pmap_pte_set(pt_entry_t *pte, pt_entry_t npte)
215{
216 int s = splvm();
217 xpq_queue_pte_update(xpmap_ptetomach(pte), npte);
218 splx(s);
219}
220
221static __inline pt_entry_t
222pmap_pte_cas(volatile pt_entry_t *ptep, pt_entry_t o, pt_entry_t n)
223{
224 pt_entry_t opte;
225
226 mutex_enter(&pte_lock);
227 opte = *ptep;
228 if (opte == o) {
229 xpq_queue_pte_update(xpmap_ptetomach(__UNVOLATILE(ptep)), n);
230 xpq_flush_queue();
231 }
232
233 mutex_exit(&pte_lock);
234 return opte;
235}
236
237static __inline pt_entry_t
238pmap_pte_testset(volatile pt_entry_t *pte, pt_entry_t npte)
239{
240 pt_entry_t opte;
241
242 mutex_enter(&pte_lock);
243 opte = *pte;
244 xpq_queue_pte_update(xpmap_ptetomach(__UNVOLATILE(pte)), npte);
245 xpq_flush_queue();
246 mutex_exit(&pte_lock);
247 return opte;
248}
249
250static __inline void
251pmap_pte_setbits(volatile pt_entry_t *pte, pt_entry_t bits)
252{
253 mutex_enter(&pte_lock);
254 xpq_queue_pte_update(xpmap_ptetomach(__UNVOLATILE(pte)), (*pte) | bits);
255 xpq_flush_queue();
256 mutex_exit(&pte_lock);
257}
258
259static __inline void
260pmap_pte_clearbits(volatile pt_entry_t *pte, pt_entry_t bits)
261{
262 mutex_enter(&pte_lock);
263 xpq_queue_pte_update(xpmap_ptetomach(__UNVOLATILE(pte)),
264 (*pte) & ~bits);
265 xpq_flush_queue();
266 mutex_exit(&pte_lock);
267}
268
269static __inline void
270pmap_pte_flush(void)
271{
272 int s = splvm();
273 xpq_flush_queue();
274 splx(s);
275}
