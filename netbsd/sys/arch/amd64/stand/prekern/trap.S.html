160/*
161 * Arguments pushed on the stack:
162 * sf_trapno
163 * sf_err (dummy inserted if not defined)
164 * sf_rip
165 * sf_cs
166 * sf_rflags
167 * sf_rsp
168 * sf_ss
169 */
170ENTRY(alltraps)
171 movq %rsp,%rdi
172 call _C_LABEL(trap)
173 /* NOTREACHED */
174END(alltraps)
