72int
73kobj_reloc(kobj_t ko, uintptr_t relocbase, const void *data,
74 bool isrela, bool local)
75{
76 Elf_Addr *where;
77 Elf_Addr addr;
78 Elf_Addr addend;
79 Elf_Word rtype, symidx;
80 const Elf_Rel *rel;
81 const Elf_Rela *rela;
82 int error;
83
84 if (isrela) {
85 rela = (const Elf_Rela *)data;
86 where = (Elf_Addr *) (relocbase + rela->r_offset);
87 addend = rela->r_addend;
88 rtype = ELF_R_TYPE(rela->r_info);
89 symidx = ELF_R_SYM(rela->r_info);
90 } else {
91 rel = (const Elf_Rel *)data;
92 where = (Elf_Addr *) (relocbase + rel->r_offset);
93 addend = *where;
94 rtype = ELF_R_TYPE(rel->r_info);
95 symidx = ELF_R_SYM(rel->r_info);
96 }
97
98 switch (rtype) {
99 case R_386_NONE: /* none */
100 return 0;
101
102 case R_386_32: /* S + A */
103 error = kobj_sym_lookup(ko, symidx, &addr);
104 if (error)
105 return -1;
106 addr += addend;
107 break;
108
109 case R_386_PC32: /* S + A - P */
110 error = kobj_sym_lookup(ko, symidx, &addr);
111 if (error)
112 return -1;
113 addr += addend - (Elf_Addr)where;
114 break;
115
116 case R_386_GLOB_DAT: /* S */
117 error = kobj_sym_lookup(ko, symidx, &addr);
118 if (error)
119 return -1;
120 break;
121
122 case R_386_RELATIVE:
123 addr = relocbase + addend;
124 break;
125
126 default:
127 printf("kobj_reloc: unexpected relocation type %d\n", rtype);
128 return -1;
129 }
130
131 *where = addr;
132 return 0;
133}

135int
136kobj_machdep(kobj_t ko, void *base, size_t size, bool load)
137{
138 uint64_t where;
139
140 /*
141 * Currently we want this to invalidate the Pentium 4 trace cache.
142 * Other caches are snoopably coherent.
143 */
144 if (load) {
145 if (cold) {
146 wbinvd();
147 } else {
148 where = xc_broadcast(0, (xcfunc_t)wbinvd, NULL, NULL);
149 xc_wait(where);
150 }
151 }
152
153 return 0;
154}
