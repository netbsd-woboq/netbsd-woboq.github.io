34/*
35 * Multiboot header structure.
36 */
37#define MULTIBOOT_HEADER_MAGIC 0x1BADB002
38#define MULTIBOOT_HEADER_MODS_ALIGNED 0x00000001
39#define MULTIBOOT_HEADER_WANT_MEMORY 0x00000002
40#define MULTIBOOT_HEADER_HAS_VBE 0x00000004
41#define MULTIBOOT_HEADER_HAS_ADDR 0x00010000
42
43#if !defined(_LOCORE)
44struct multiboot_header {
45 uint32_t mh_magic;
46 uint32_t mh_flags;
47 uint32_t mh_checksum;
48
49 /* Valid if mh_flags sets MULTIBOOT_HEADER_HAS_ADDR. */
50 paddr_t mh_header_addr;
51 paddr_t mh_load_addr;
52 paddr_t mh_load_end_addr;
53 paddr_t mh_bss_end_addr;
54 paddr_t mh_entry_addr;
55
56 /* Valid if mh_flags sets MULTIBOOT_HEADER_HAS_VBE. */
57 uint32_t mh_mode_type;
58 uint32_t mh_width;
59 uint32_t mh_height;
60 uint32_t mh_depth;
61};
62#endif /* !defined(_LOCORE) */
63
64/*
65 * Symbols defined in locore.S.
66 */
67#if !defined(_LOCORE) && defined(_KERNEL)
68extern struct multiboot_header *Multiboot_Header;
69#endif /* !defined(_LOCORE) && defined(_KERNEL) */

73/*
74 * Multiboot information structure.
75 */
76#define MULTIBOOT_INFO_MAGIC 0x2BADB002
77#define MULTIBOOT_INFO_HAS_MEMORY 0x00000001
78#define MULTIBOOT_INFO_HAS_BOOT_DEVICE 0x00000002
79#define MULTIBOOT_INFO_HAS_CMDLINE 0x00000004
80#define MULTIBOOT_INFO_HAS_MODS 0x00000008
81#define MULTIBOOT_INFO_HAS_AOUT_SYMS 0x00000010
82#define MULTIBOOT_INFO_HAS_ELF_SYMS 0x00000020
83#define MULTIBOOT_INFO_HAS_MMAP 0x00000040
84#define MULTIBOOT_INFO_HAS_DRIVES 0x00000080
85#define MULTIBOOT_INFO_HAS_CONFIG_TABLE 0x00000100
86#define MULTIBOOT_INFO_HAS_LOADER_NAME 0x00000200
87#define MULTIBOOT_INFO_HAS_APM_TABLE 0x00000400
88#define MULTIBOOT_INFO_HAS_VBE 0x00000800
89#define MULTIBOOT_INFO_HAS_FRAMEBUFFER 0x00001000
90
91#if !defined(_LOCORE)
92struct multiboot_info {
93 uint32_t mi_flags;
94
95 /* Valid if mi_flags sets MULTIBOOT_INFO_HAS_MEMORY. */
96 uint32_t mi_mem_lower;
97 uint32_t mi_mem_upper;
98
99 /* Valid if mi_flags sets MULTIBOOT_INFO_HAS_BOOT_DEVICE. */
100 uint8_t mi_boot_device_part3;
101 uint8_t mi_boot_device_part2;
102 uint8_t mi_boot_device_part1;
103 uint8_t mi_boot_device_drive;
104
105 /* Valid if mi_flags sets MULTIBOOT_INFO_HAS_CMDLINE. */
106 char * mi_cmdline;
107
108 /* Valid if mi_flags sets MULTIBOOT_INFO_HAS_MODS. */
109 uint32_t mi_mods_count;
110 vaddr_t mi_mods_addr;
111
112 /* Valid if mi_flags sets MULTIBOOT_INFO_HAS_{AOUT,ELF}_SYMS. */
113 uint32_t mi_elfshdr_num;
114 uint32_t mi_elfshdr_size;
115 vaddr_t mi_elfshdr_addr;
116 uint32_t mi_elfshdr_shndx;
117
118 /* Valid if mi_flags sets MULTIBOOT_INFO_HAS_MMAP. */
119 uint32_t mi_mmap_length;
120 vaddr_t mi_mmap_addr;
121
122 /* Valid if mi_flags sets MULTIBOOT_INFO_HAS_DRIVES. */
123 uint32_t mi_drives_length;
124 vaddr_t mi_drives_addr;
125
126 /* Valid if mi_flags sets MULTIBOOT_INFO_HAS_CONFIG_TABLE. */
127 void * unused_mi_config_table;
128
129 /* Valid if mi_flags sets MULTIBOOT_INFO_HAS_LOADER_NAME. */
130 char * mi_loader_name;
131
132 /* Valid if mi_flags sets MULTIBOOT_INFO_HAS_APM. */
133 void * unused_mi_apm_table;
134
135 /* Valid if mi_flags sets MULTIBOOT_INFO_HAS_VBE. */
136 void * unused_mi_vbe_control_info;
137 void * unused_mi_vbe_mode_info;
138 uint16_t unused_mi_vbe_mode;
139 uint16_t unused_mi_vbe_interface_seg;
140 uint16_t unused_mi_vbe_interface_off;
141 uint16_t unused_mi_vbe_interface_len;
142
143 /* Valid if mi_flags sets MULTIBOOT_INFO_HAS_FRAMEBUFFER. */
144 uint64_t framebuffer_addr;
145 uint32_t framebuffer_pitch;
146 uint32_t framebuffer_width;
147 uint32_t framebuffer_height;
148 uint8_t framebuffer_bpp;
149#define MULTIBOOT_FRAMEBUFFER_TYPE_INDEXED 0
150#define MULTIBOOT_FRAMEBUFFER_TYPE_RGB 1
151#define MULTIBOOT_FRAMEBUFFER_TYPE_EGA_TEXT 2
152 uint8_t framebuffer_type;
153 union {
154 struct {
155 uint32_t framebuffer_palette_addr;
156 uint16_t framebuffer_palette_num_colors;
157 };
158 struct {
159 uint8_t framebuffer_red_field_position;
160 uint8_t framebuffer_red_mask_size;
161 uint8_t framebuffer_green_field_position;
162 uint8_t framebuffer_green_mask_size;
163 uint8_t framebuffer_blue_field_position;
164 uint8_t framebuffer_blue_mask_size;
165 };
166 };
167
168};

172/*
173 * Drive information. This describes an entry in the drives table as
174 * pointed to by mi_drives_addr.
175 */
176struct multiboot_drive {
177 uint32_t md_length;
178 uint8_t md_number;
179 uint8_t md_mode;
180 uint16_t md_cylinders;
181 uint8_t md_heads;
182 uint8_t md_sectors;
183
184 /* The variable-sized 'ports' field comes here, so this structure
185 * can be longer. */
186};
187
188/* --------------------------------------------------------------------- */
189
190/*
191 * Memory mapping. This describes an entry in the memory mappings table
192 * as pointed to by mi_mmap_addr.
193 *
194 * Be aware that mm_size specifies the size of all other fields *except*
195 * for mm_size. In order to jump between two different entries, you
196 * have to count mm_size + 4 bytes.
197 */
198struct multiboot_mmap {
199 uint32_t mm_size;
200 uint64_t mm_base_addr;
201 uint64_t mm_length;
202 uint32_t mm_type;
203};
204
205/*
206 * Modules. This describes an entry in the modules table as pointed
207 * to by mi_mods_addr.
208 */
209
210struct multiboot_module {
211 uint32_t mmo_start;
212 uint32_t mmo_end;
213 char * mmo_string;
214 uint32_t mmo_reserved;
215};
