37/*
38 * The following information is found in the non-volatile RAM in the
39 * MC146818A (or DS1287A or other compatible) RTC on AT-compatible PCs.
40 */
41
42/* NVRAM byte 0: bios diagnostic */
43#define NVRAM_DIAG (MC_NVRAM_START + 0) /* RTC offset 0xe */
44
45#define NVRAM_DIAG_BITS "\020\010clock_battery\007ROM_cksum\006config_unit\005memory_size\004fixed_disk\003invalid_time"
46
47/* NVRAM byte 1: reset code */
48#define NVRAM_RESET (MC_NVRAM_START + 1) /* RTC offset 0xf */
49
50#define NVRAM_RESET_RST 0x00 /* normal reset */
51#define NVRAM_RESET_LOAD 0x04 /* load system */
52#define NVRAM_RESET_JUMP 0x0a /* jump through 40:67 */
53
54/* NVRAM byte 2: diskette drive type in upper/lower nibble */
55#define NVRAM_DISKETTE (MC_NVRAM_START + 2) /* RTC offset 0x10 */
56
57#define NVRAM_DISKETTE_NONE 0 /* none present */
58#define NVRAM_DISKETTE_360K 0x10 /* 360K */
59#define NVRAM_DISKETTE_12M 0x20 /* 1.2M */
60#define NVRAM_DISKETTE_720K 0x30 /* 720K */
61#define NVRAM_DISKETTE_144M 0x40 /* 1.44M */
62#define NVRAM_DISKETTE_TYPE5 0x50 /* 2.88M, presumably */
63#define NVRAM_DISKETTE_TYPE6 0x60 /* 2.88M */
64
65/* NVRAM byte 6: equipment type */
66#define NVRAM_EQUIPMENT (MC_NVRAM_START + 6)
67
68#define NVRAM_EQUIPMENT_EGAVGA 0x00 /* EGA or VGA */
69#define NVRAM_EQUIPMENT_COLOR40 0x10 /* 40 column color */
70#define NVRAM_EQUIPMENT_COLOR80 0x20 /* 80 column color */
71#define NVRAM_EQUIPMENT_MONO80 0x30 /* 80 column mono */
72#define NVRAM_EQUIPMENT_MONITOR 0x30 /* mask for monitor type */
73
74/* NVRAM bytes 7 & 8: base memory size */
75#define NVRAM_BASELO (MC_NVRAM_START + 7) /* low byte; RTC off. 0x15 */
76#define NVRAM_BASEHI (MC_NVRAM_START + 8) /* high byte; RTC off. 0x16 */
77
78/* NVRAM bytes 9 & 10: extended memory size */
79#define NVRAM_EXTLO (MC_NVRAM_START + 9) /* low byte; RTC off. 0x17 */
80#define NVRAM_EXTHI (MC_NVRAM_START + 10) /* high byte; RTC off. 0x18 */
81
82/* NVRAM byte 36: current century. (please increment in Dec99!) */
83#define NVRAM_CENTURY (MC_NVRAM_START + 36) /* RTC offset 0x32 */
