31#include <sys/bootblock.h>

33void startprog32_start(physaddr_t, uint32_t, uint32_t *, physaddr_t,
34 physaddr_t, physaddr_t, u_long, void *);
35extern void (*startprog32)(physaddr_t, uint32_t, uint32_t *, physaddr_t,
36 physaddr_t, physaddr_t, u_long, void *);
37extern u_int startprog32_size;
38
39void
40efi_md_init(void)
41{
42 EFI_STATUS status;
43 EFI_PHYSICAL_ADDRESS addr = EFI_ALLOCATE_MAX_ADDRESS;
44 u_int sz = EFI_SIZE_TO_PAGES(startprog32_size);
45
46 status = uefi_call_wrapper(BS->AllocatePages, 4, AllocateMaxAddress,
47 EfiLoaderData, sz, &addr);
48 if (EFI_ERROR(status))
49 panic("%s: AllocatePages() failed: %d page(s): %" PRIxMAX,
50 __func__, sz, (uintmax_t)status);
51 startprog32 = (void *)(u_long)addr;
52 CopyMem(startprog32, startprog32_start, startprog32_size);
53}
54
55/* ARGSUSED */
56void
57startprog(physaddr_t entry, uint32_t argc, uint32_t *argv, physaddr_t sp)
58{
59
60 (*startprog32)(entry, argc, argv,
61 (physaddr_t)startprog32 + startprog32_size,
62 efi_kernel_start, efi_kernel_start + efi_loadaddr,
63 efi_kernel_size, startprog32);
64}
