31#include <sys/bootblock.h>

33void startprog64_start(physaddr_t, physaddr_t, physaddr_t, u_long,
34 void *, physaddr_t);
35extern void (*startprog64)(physaddr_t, physaddr_t, physaddr_t, u_long,
36 void *, physaddr_t);
37extern u_int startprog64_size;
38
39void
40efi_md_init(void)
41{
42 EFI_STATUS status;
43 EFI_PHYSICAL_ADDRESS addr = EFI_ALLOCATE_MAX_ADDRESS;
44 u_int sz = EFI_SIZE_TO_PAGES(startprog64_size);
45
46 status = uefi_call_wrapper(BS->AllocatePages, 4, AllocateMaxAddress,
47 EfiLoaderData, sz, &addr);
48 if (EFI_ERROR(status))
49 panic("%s: AllocatePages() failed: %d page(s): %" PRIxMAX,
50 __func__, sz, (uintmax_t)status);
51 startprog64 = (void *)addr;
52 CopyMem(startprog64, startprog64_start, startprog64_size);
53}
54
55/* ARGSUSED */
56void
57startprog(physaddr_t entry, uint32_t argc, uint32_t *argv, physaddr_t sp)
58{
59 uint32_t *newsp = (void *)((char *)startprog64 + startprog64_size);
60
61 /* Copy argv to new stack pointer */
62 if (argc > 0) {
63 newsp -= argc;
64 memcpy(newsp, argv, sizeof(*argv) * argc);
65 }
66
67 (*startprog64)(efi_kernel_start, efi_kernel_start + efi_loadaddr,
68 (physaddr_t)newsp, efi_kernel_size, startprog64, entry);
69}
