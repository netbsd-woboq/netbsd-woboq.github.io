43static EFI_PHYSICAL_ADDRESS heap_start = EFI_ALLOCATE_MAX_ADDRESS;
44static UINTN heap_size = 1 * 1024 * 1024; /* 1MB */
45static struct btinfo_efi btinfo_efi;

49EFI_STATUS EFIAPI efi_main(EFI_HANDLE, EFI_SYSTEM_TABLE *);
50EFI_STATUS EFIAPI
51efi_main(EFI_HANDLE imageHandle, EFI_SYSTEM_TABLE *systemTable)
52{
53 EFI_STATUS status;
54 EFI_DEVICE_PATH *dp0, *dp;
55 extern char twiddle_toggle;
56
57 IH = imageHandle;
58 InitializeLib(IH, systemTable);
59
60 efi_main_sp = (uintptr_t)&status;
61 twiddle_toggle = 1; /* no twiddling until we're ready */
62
63 cninit();
64 efi_heap_init();
65 efi_md_init();
66
67 status = uefi_call_wrapper(BS->HandleProtocol, 3, IH,
68 &LoadedImageProtocol, (void **)&efi_li);
69 if (EFI_ERROR(status))
70 panic("HandleProtocol(LoadedImageProtocol): %" PRIxMAX,
71 (uintmax_t)status);
72 status = uefi_call_wrapper(BS->HandleProtocol, 3, efi_li->DeviceHandle,
73 &DevicePathProtocol, (void **)&dp0);
74 if (EFI_ERROR(status))
75 panic("HandleProtocol(DevicePathProtocol): %" PRIxMAX,
76 (uintmax_t)status);
77 efi_bootdp = dp0;
78 for (dp = dp0; !IsDevicePathEnd(dp); dp = NextDevicePathNode(dp)) {
79 if (DevicePathType(dp) == MEDIA_DEVICE_PATH &&
80 DevicePathSubType(dp) == MEDIA_CDROM_DP) {
81 efi_bootdp_type = BOOT_DEVICE_TYPE_CD;
82 break;
83 }
84 if (DevicePathType(dp) == MESSAGING_DEVICE_PATH &&
85 DevicePathSubType(dp) == MSG_MAC_ADDR_DP) {
86 efi_bootdp_type = BOOT_DEVICE_TYPE_NET;
87 break;
88 }
89 }
90
91 efi_disk_probe();
92 efi_pxe_probe();
93 efi_net_probe();
94
95 status = uefi_call_wrapper(BS->SetWatchdogTimer, 4, 0, 0, 0, NULL);
96 if (EFI_ERROR(status))
97 panic("SetWatchdogTimer: %" PRIxMAX, (uintmax_t)status);
98
99 boot();
100
101 return EFI_SUCCESS;
102}

104void
105efi_cleanup(void)
106{
107 EFI_STATUS status;
108 EFI_MEMORY_DESCRIPTOR *desc;
109 UINTN NoEntries, MapKey, DescriptorSize;
110 UINT32 DescriptorVersion;
111 struct btinfo_efimemmap *bim;
112 size_t allocsz;
113
114 clearit();
115
116 memset(&btinfo_efi, 0, sizeof(btinfo_efi));
117 btinfo_efi.systblpa = (intptr_t)ST;
118#ifdef __i386__ /* bootia32.efi */
119 btinfo_efi.flags |= BI_EFI_32BIT;
120#endif
121 BI_ADD(&btinfo_efi, BTINFO_EFI, sizeof(btinfo_efi));
122
123 NoEntries = 0;
124 desc = efi_memory_get_map(&NoEntries, &MapKey, &DescriptorSize,
125 &DescriptorVersion, true);
126 status = uefi_call_wrapper(BS->ExitBootServices, 2, IH, MapKey);
127 if (EFI_ERROR(status)) {
128 FreePool(desc);
129 desc = efi_memory_get_map(&NoEntries, &MapKey, &DescriptorSize,
130 &DescriptorVersion, true);
131 status = uefi_call_wrapper(BS->ExitBootServices, 2, IH, MapKey);
132 if (EFI_ERROR(status))
133 panic("ExitBootServices failed");
134 }
135 efi_cleanuped = true;
136
137 allocsz = sizeof(struct btinfo_efimemmap) - 1
138 + NoEntries * DescriptorSize;
139 bim = alloc(allocsz);
140 bim->num = NoEntries;
141 bim->version = DescriptorVersion;
142 bim->size = DescriptorSize;
143 memcpy(bim->memmap, desc, NoEntries * DescriptorSize);
144 BI_ADD(bim, BTINFO_EFIMEMMAP, allocsz);
145}

147static void
148efi_heap_init(void)
149{
150 EFI_STATUS status;
151 u_int sz = EFI_SIZE_TO_PAGES(heap_size);
152
153 status = uefi_call_wrapper(BS->AllocatePages, 4, AllocateMaxAddress,
154 EfiLoaderData, sz, &heap_start);
155 if (EFI_ERROR(status))
156 Panic(L"%a: AllocatePages() failed: %d page(s): %r",
157 __func__, sz, status);
158 setheap((void *)(UINTN)heap_start,
159 (void *)(UINTN)(heap_start + heap_size));
160}
