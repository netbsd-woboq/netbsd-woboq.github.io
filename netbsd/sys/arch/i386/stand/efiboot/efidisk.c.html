40static struct efidiskinfo_lh efi_disklist;
41static int nefidisks;
42
43void
44efi_disk_probe(void)
45{
46 EFI_STATUS status;
47 UINTN i, nhandles;
48 EFI_HANDLE *handles;
49 EFI_BLOCK_IO *bio;
50 EFI_BLOCK_IO_MEDIA *media;
51 EFI_DEVICE_PATH *dp;
52 struct efidiskinfo *edi;
53 int dev, depth = -1;
54
55 TAILQ_INIT(&efi_disklist);
56
57 status = LibLocateHandle(ByProtocol, &BlockIoProtocol, NULL,
58 &nhandles, &handles);
59 if (EFI_ERROR(status))
60 return;
61
62 if (efi_bootdp != NULL)
63 depth = efi_device_path_depth(efi_bootdp, MEDIA_DEVICE_PATH);
64
65 /*
66 * U-Boot incorrectly represents devices with a single
67 * MEDIA_DEVICE_PATH component. In that case include that
68 * component into the matching, otherwise we'll blindly select
69 * the first device.
70 */
71 if (depth == 0)
72 depth = 1;
73
74 for (i = 0; i < nhandles; i++) {
75 status = uefi_call_wrapper(BS->HandleProtocol, 3, handles[i],
76 &BlockIoProtocol, (void **)&bio);
77 if (EFI_ERROR(status))
78 continue;
79
80 media = bio->Media;
81 if (media->LogicalPartition || !media->MediaPresent)
82 continue;
83
84 edi = alloc(sizeof(struct efidiskinfo));
85 memset(edi, 0, sizeof(*edi));
86 edi->type = BIOSDISK_TYPE_HD;
87 edi->bio = bio;
88 edi->media_id = media->MediaId;
89
90 if (efi_bootdp != NULL && depth > 0) {
91 status = uefi_call_wrapper(BS->HandleProtocol, 3,
92 handles[i], &DevicePathProtocol, (void **)&dp);
93 if (EFI_ERROR(status))
94 goto next;
95 if (efi_device_path_ncmp(efi_bootdp, dp, depth) == 0) {
96 edi->bootdev = true;
97 TAILQ_INSERT_HEAD(&efi_disklist, edi,
98 list);
99 continue;
100 }
101 }
102next:
103 TAILQ_INSERT_TAIL(&efi_disklist, edi, list);
104 }
105
106 FreePool(handles);
107
108 if (efi_bootdp_type == BOOT_DEVICE_TYPE_CD) {
109 edi = TAILQ_FIRST(&efi_disklist);
110 if (edi != NULL && edi->bootdev) {
111 edi->type = BIOSDISK_TYPE_CD;
112 TAILQ_REMOVE(&efi_disklist, edi, list);
113 TAILQ_INSERT_TAIL(&efi_disklist, edi, list);
114 }
115 }
116
117 dev = 0x80;
118 TAILQ_FOREACH(edi, &efi_disklist, list) {
119 edi->dev = dev++;
120 if (edi->type == BIOSDISK_TYPE_HD)
121 nefidisks++;
122 if (edi->bootdev)
123 boot_biosdev = edi->dev;
124 }
125}

127void
128efi_disk_show(void)
129{
130 const struct efidiskinfo *edi;
131 EFI_BLOCK_IO_MEDIA *media;
132 struct biosdisk_partition *part;
133 uint64_t size;
134 int i, nparts;
135 bool first;
136
137 TAILQ_FOREACH(edi, &efi_disklist, list) {
138 media = edi->bio->Media;
139 first = true;
140 printf("disk ");
141 switch (edi->type) {
142 case BIOSDISK_TYPE_CD:
143 printf("cd0");
144 printf(" mediaId %u", media->MediaId);
145 if (edi->media_id != media->MediaId)
146 printf("(%u)", edi->media_id);
147 printf("\n");
148 printf(" cd0a\n");
149 break;
150 case BIOSDISK_TYPE_HD:
151 printf("hd%d", edi->dev & 0x7f);
152 printf(" mediaId %u", media->MediaId);
153 if (edi->media_id != media->MediaId)
154 printf("(%u)", edi->media_id);
155 printf(" size ");
156 size = (media->LastBlock + 1) * media->BlockSize;
157 if (size >= (10ULL * 1024 * 1024 * 1024))
158 printf("%"PRIu64" GB", size / (1024 * 1024 * 1024));
159 else
160 printf("%"PRIu64" MB", size / (1024 * 1024));
161 printf("\n");
162 break;
163 }
164 if (edi->type != BIOSDISK_TYPE_HD)
165 continue;
166
167 if (biosdisk_readpartition(edi->dev, &part, &nparts))
168 continue;
169
170 for (i = 0; i < nparts; i++) {
171 if (part[i].size == 0)
172 continue;
173 if (part[i].fstype == FS_UNUSED)
174 continue;
175 if (first) {
176 printf(" ");
177 first = false;
178 }
179 printf(" hd%d%c(", edi->dev & 0x7f, i + 'a');
180 if (part[i].guid != NULL)
181 printf("%s", part[i].guid->name);
182 else if (part[i].fstype < FSMAXTYPES)
183 printf("%s", fstypenames[part[i].fstype]);
184 else
185 printf("%d", part[i].fstype);
186 printf(")");
187 }
188 if (!first)
189 printf("\n");
190 dealloc(part, sizeof(*part) * nparts);
191 }
192}

194const struct efidiskinfo *
195efidisk_getinfo(int dev)
196{
197 const struct efidiskinfo *edi;
198
199 TAILQ_FOREACH(edi, &efi_disklist, list) {
200 if (dev == edi->dev)
201 return edi;
202 }
203 return NULL;
204}

215int
216efidisk_get_efi_system_partition(int dev, int *partition)
217{
218 extern const struct uuid GET_efi;
219 const struct efidiskinfo *edi;
220 struct biosdisk_partition *part;
221 int i, nparts;
222
223 edi = efidisk_getinfo(dev);
224 if (edi == NULL)
225 return ENXIO;
226
227 if (edi->type != BIOSDISK_TYPE_HD)
228 return ENOTSUP;
229
230 if (biosdisk_readpartition(edi->dev, &part, &nparts))
231 return EIO;
232
233 for (i = 0; i < nparts; i++) {
234 if (part[i].size == 0)
235 continue;
236 if (part[i].fstype == FS_UNUSED)
237 continue;
238 if (guid_is_equal(part[i].guid->guid, &GET_efi))
239 break;
240 }
241 dealloc(part, sizeof(*part) * nparts);
242 if (i == nparts)
243 return ENOENT;
244
245 *partition = i;
246 return 0;
247}
