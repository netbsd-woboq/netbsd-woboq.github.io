31#include <efistdarg.h>

33__dead VOID
34Panic(
35 IN CHAR16 *fmt,
36 ...
37 )
38{
39 va_list args;
40
41 va_start(args, fmt);
42 VPrint(fmt, args);
43 Print(L"\n");
44 va_end(args);
45 reboot();
46 /*NOTREACHED*/
47 __unreachable();
48}
49
50__dead void
51reboot(void)
52{
53
54 if (!efi_cleanuped)
55 WaitForSingleEvent(ST->ConIn->WaitForKey, 0);
56
57 uefi_call_wrapper(RT->ResetSystem, 4, EfiResetCold, EFI_SUCCESS,
58 0, NULL);
59 for (;;)
60 continue;
61}
62
63__dead void
64_rtt(void)
65{
66
67 reboot();
68}
