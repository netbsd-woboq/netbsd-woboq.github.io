34void self_reloc(Elf_Addr, Elf_Dyn *);

36#if defined(__aarch64__)
37#define USE_RELA
38#endif

40/*
41 * A simple elf relocator.
42 */
43void
44self_reloc(Elf_Addr baseaddr, Elf_Dyn *dynamic)
45{
46 Elf_Word relsz, relent;
47 Elf_Addr *newaddr;
48 Elf_Rel *rel = NULL;
49 Elf_Dyn *dynp;
50
51 /*
52 * Find the relocation address, its size and the relocation entry.
53 */
54 relsz = 0;
55 relent = 0;
56 for (dynp = dynamic; dynp->d_tag != DT_NULL; dynp++) {
57 switch (dynp->d_tag) {
58 case DT_REL:
59 case DT_RELA:
60 rel = (Elf_Rel *)(dynp->d_un.d_ptr + baseaddr);
61 break;
62 case DT_RELSZ:
63 case DT_RELASZ:
64 relsz = dynp->d_un.d_val;
65 break;
66 case DT_RELENT:
67 case DT_RELAENT:
68 relent = dynp->d_un.d_val;
69 break;
70 default:
71 break;
72 }
73 }
74
75 /*
76 * Perform the actual relocation.
77 */
78 for (; relsz > 0; relsz -= relent) {
79 switch (ELF_R_TYPE(rel->r_info)) {
80 case R_TYPE(NONE):
81 /* No relocation needs be performed. */
82 break;
83
84 case R_TYPE(RELATIVE):
85 /* Address relative to the base address. */
86 newaddr = (Elf_Addr *)(rel->r_offset + baseaddr);
87 *newaddr += baseaddr;
88#ifdef USE_RELA
89 /* Add the addend when the ABI uses them */
90 *newaddr += ((Elf_Rela *)rel)->r_addend;
91#endif
92 break;
93 default:
94 /* XXX: do we need other relocations ? */
95 break;
96 }
97 rel = (Elf_Rel *) ((char *) rel + relent);
98 }
99}
