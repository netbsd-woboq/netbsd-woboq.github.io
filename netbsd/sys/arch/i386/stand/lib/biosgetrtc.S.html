29#include <machine/asm.h>
30
31ENTRY(biosgetrtc)
32 pusha
33 pushl %eax
34
35 xorl %ebx, %ebx
36
37 call _C_LABEL(prot_to_real) # enter real mode
38 .code16
39
40 movb $2, %ah
41 int $0x1a
42 jnc ok
43 movl $-1, %ebx
44
45ok:
46 calll _C_LABEL(real_to_prot) # back to protected mode
47 .code32
48
49 popl %eax
50 movb %ch, (%eax)
51 movb %cl, 1(%eax)
52 movb %dh, 2(%eax)
53 movb $0, 3(%eax)
54
55 movl %ebx, 28(%esp)
56
57 popa
58 ret
