35#include <machine/asm.h>
36
37 .text
38
39ENTRY(biosvideomode)
40 pushl %ebp
41 movl %esp,%ebp
42 pushl %ebx
43 push %esi
44 push %edi
45
46 call _C_LABEL(prot_to_real)
47 .code16
48
49 movb $0, %ah
50 movb $2, %al
51 int $0x10
52 # zero-extend 16-bit result to 32 bits.
53 movl $0, %ebx
54 movw %ax,%bx
55
56 calll _C_LABEL(real_to_prot)
57 .code32
58
59 movl %ebx, %eax
60
61 pop %edi
62 pop %esi
63 popl %ebx
64 popl %ebp
65 ret
