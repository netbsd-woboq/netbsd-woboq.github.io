29#include <lib/libsa/stand.h>
30#include <lib/libkern/libkern.h>
31
32#include "libi386.h"
33#include "bootinfo.h"
34
35struct bootinfo *bootinfo;
36
37void
38bi_add(struct btinfo_common *what, int type, int size)
39{
40 what->len = size;
41 what->type = type;
42
43 if (bootinfo == NULL) {
44 return;
45 }
46 if (bootinfo->nentries >= BTINFO_MAX) {
47 panic("bootinfo too big");
48 }
49 bootinfo->entry[bootinfo->nentries++] = vtophys(what);
50}
