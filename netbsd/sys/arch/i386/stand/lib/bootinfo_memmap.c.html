29#include <lib/libsa/stand.h>
30#include "libi386.h"
31#include "bootinfo.h"
32
33extern int getmementry(int *, int *);
34
35void
36bi_getmemmap(void)
37{
38 int buf[5], i, nranges, n;
39 struct btinfo_memmap *bimm;
40
41 nranges = 0;
42 i = 0;
43 do {
44 if (getmementry(&i, buf))
45 break;
46 nranges++;
47 } while (i);
48
49 bimm = alloc(sizeof(struct btinfo_memmap)
50 + (nranges - 1) * sizeof(struct bi_memmap_entry));
51
52 i = 0;
53 for (n = 0; n < nranges; n++) {
54 getmementry(&i, buf);
55 memcpy(&bimm->entry[n], buf, sizeof(struct bi_memmap_entry));
56 }
57 bimm->num = nranges;
58
59 BI_ADD(bimm, BTINFO_MEMMAP, sizeof(struct btinfo_memmap)
60 + (nranges - 1) * sizeof(struct bi_memmap_entry));
61}
