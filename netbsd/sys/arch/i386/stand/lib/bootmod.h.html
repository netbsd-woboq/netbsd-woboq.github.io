29#ifndef _BOOTMOD_H
30#define _BOOTMOD_H
31
32typedef struct boot_module {
33 char *bm_path;
34 ssize_t bm_len;
35 uint8_t bm_type;
36#define BM_TYPE_KMOD 0x00
37#define BM_TYPE_IMAGE 0x01
38#define BM_TYPE_RND 0x02
39#define BM_TYPE_FS 0x03
40 struct boot_module *bm_next;
41} boot_module_t;
42
43extern boot_module_t *boot_modules;
44extern bool boot_modules_enabled;
45
46#endif /* !_BOOTMOD_H */
