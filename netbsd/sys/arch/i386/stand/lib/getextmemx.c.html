29/*
30 * Try 2 more fancy BIOS calls to get the size of extended
31 * memory besides the classical int15/88, take maximum.
32 * needs lowlevel parts from biosmemx.S and biosmem.S
33 */
34
35#include <lib/libsa/stand.h>
36#include "libi386.h"
37
38int
39getextmemx(void)
40{
41 int buf[5], i;
42 int extmem = getextmem1();
43#ifdef SUPPORT_PS2
44 struct {
45 uint16_t len;
46 uint32_t dta[8];
47 /* pad to 64 bytes - without this, machine would reset */
48 uint8_t __pad[30];
49 } __packed bufps2;
50#endif
51
52#ifdef DEBUG_MEMSIZE
53 printf("extmem1: %xk\n", extmem);
54#endif
55 if (!getextmem2(buf)) {
56#ifdef DEBUG_MEMSIZE
57 printf("extmem2: %xk + %xk\n", buf[0], buf[1] * 64);
58#endif
59 if (buf[0] <= 15 * 1024) {
60 int help = buf[0];
61 if (help == 15 * 1024)
62 help += buf[1] * 64;
63 if (extmem < help)
64 extmem = help;
65 }
66 }
67
68 i = 0;
69 do {
70 if (getmementry(&i, buf))
71 break;
72#ifdef DEBUG_MEMSIZE
73 printf("mementry: (%d) %x %x %x %x %x\n",
74 i, buf[0], buf[1], buf[2], buf[3], buf[4]);
75#endif
76 if ((buf[4] == 1 && buf[0] == 0x100000)
77 && extmem < buf[2] / 1024)
78 extmem = buf[2] / 1024;
79 } while (i);
80
81#ifdef SUPPORT_PS2
82 /* use local memory information from RETURN MEMORY-MAP INFORMATION */
83 if (!getextmemps2((void *) &bufps2)) {
84 int help = bufps2.dta[0];
85 if (help == 15 * 1024)
86 help += bufps2.dta[1];
87 if (extmem < help)
88 extmem = help;
89 }
90#endif
91
92 return extmem;
93}
