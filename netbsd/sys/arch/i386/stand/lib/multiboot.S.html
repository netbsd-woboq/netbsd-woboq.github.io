80#include <machine/asm.h>
81#define MULTIBOOT_INFO_MAGIC 0x2BADB002
82
83/*
84 * Starts program in protected mode / flat space with given stackframe.
85 * Needs global variables flatcodeseg and flatdataseg (gdt offsets).
86 * Derived from: NetBSD:sys/arch/i386/stand/lib/startprog.S
87 */
88
89/*
90 * multiboot(phyaddr, header, stack)
91 * start the program on protected mode where phyaddr is the entry point
92 */
93ENTRY(multiboot)
94 pushl %ebp
95 movl %esp,%ebp
96
97 /* Prepare a new stack */
98 movl $flatdataseg,%eax
99 movw %ax,%es /* for arg copy */
100 movl 16(%ebp),%ebx /* stack */
101 subl $4,%ebx
102 movl %ebx,%edi
103
104 movl 12(%ebp),%ebx /* header */
105 movl 8(%ebp),%ecx /* entry */
106
107 /* Set new stack pointer */
108 movw %ax,%ss
109 movl %edi,%esp
110
111 /* Push on our entry address */
112 movl $flatcodeseg,%eax
113 pushl %eax /* code segment */
114 pushl %ecx /* phyaddr */
115
116 /* Convert over the other data segs */
117 movl $flatdataseg,%eax
118 mov %ax,%ds
119 mov %ax,%es
120
121 movl $MULTIBOOT_INFO_MAGIC,%eax
122
123 /* Jump to phyaddr, with the new code segment */
124 lret
