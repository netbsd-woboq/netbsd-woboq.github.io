29/* read into destination in flat addr space */
30
31#include <lib/libsa/stand.h>
32
33#include "libi386.h"
34
35#ifdef SAVE_MEMORY
36#define BUFSIZE (1*1024)
37#else
38#define BUFSIZE (4*1024)
39#endif
40
41static char *buf;
42
43ssize_t
44pread(int fd, void *dest, size_t size)
45{
46 int rsize;
47
48 if (!buf)
49 buf = alloc(BUFSIZE);
50
51 rsize = size;
52 while (rsize > 0) {
53 int count, got;
54
55 count = (rsize < BUFSIZE ? rsize : BUFSIZE);
56
57 got = read(fd, buf, count);
58 if (got < 0)
59 return -1;
60
61 /* put to physical space */
62 vpbcopy(buf, dest, got);
63
64 dest += got;
65 rsize -= got;
66 if (got < count)
67 break; /* EOF */
68 }
69 return size - rsize;
70}
