28#include <lib/libsa/stand.h>
29#include "libi386.h"
30
31extern int getmementry(int *, int *);
32
33static char *memtypes[] = {
34 "available",
35 "reserved",
36 "ACPI reclaimable",
37 "ACPI NVS"
38};
39
40void
41printmemlist(void)
42{
43 int buf[5], i;
44 char *type;
45
46 i = 0;
47 do {
48 if (getmementry(&i, buf))
49 break;
50 if (buf[4] < 1 || buf[4] > 4)
51 type = "invalid entry";
52 else
53 type = memtypes[buf[4] - 1];
54 printf("%x:%x/%x:%x: %s\n", buf[1], buf[0], buf[3], buf[2],
55 type);
56 } while (i);
57}
