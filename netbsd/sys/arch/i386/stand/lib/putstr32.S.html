32#include <machine/asm.h>
33
34/*
35 * Diagnostic print routines callable from 32bit C code during early
36 * parts of the boot process.
37 */
38
39/*
40 * void putstr(const char *)
41 *
42 * display message on serial port
43 */
44
45 .globl _C_LABEL(putstr32)
46_C_LABEL(putstr32):
47 .code32
48 push %esi
49 movl 8(%esp), %esi
50
51 call message32
52
53 pop %esi
54 ret
55
56/*
57 * void putint(int)
58 *
59 * display value on serial port as 8 hex digits followed by a space
60 */
61
62 .globl _C_LABEL(putint32)
63_C_LABEL(putint32):
64 .code32
65 movl 4(%esp), %eax
66
67 call dump_eax32
68
69 ret
