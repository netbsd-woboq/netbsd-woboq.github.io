32#include <machine/asm.h>
33
34/*
35 * Routines to copy to/from absolute virtual addresses.
36 * Needed because the boot code runs with %ds having a 64k offset
37 * whereas Unix runs with a zero offset.
38 *
39 * These routines are optimised for code space, not execution speed.
40 */
41
42/*
43 * pbzero(void *dst, int cnt)
44 * zero absolute virtual memory
45 */
46ENTRY(pbzero)
47 .code32
48 push %edi
49 push %es
50 mov 12(%esp),%edi
51 mov 16(%esp),%ecx
52
53 mov $flatdataseg, %ax /* selector with offset == 0 */
54 mov %ax, %es
55 xor %eax,%eax
56
57 cld
58 rep
59 stosb
60
61 pop %es
62 pop %edi
63 ret
64
65/*
66 * vpbcopy(const void *src, void *dst, int cnt)
67 * Copy to absolute virtual address
68 */
69ENTRY(vpbcopy)
70 .code32
71 push %esi
72 push %edi
73 push %es
74 mov 16(%esp),%esi
75 mov 20(%esp),%edi
76 mov 24(%esp),%ecx
77
78 mov $flatdataseg, %ax /* selector with offset == 0 */
79 mov %ax, %es
80 xor %eax,%eax
81
82 cld
83 rep
84 movsb
85
86 popl %es
87 popl %edi
88 popl %esi
89 ret
90
91/*
92 * pvbcopy(const void *src, void *dst, int cnt)
93 * Copy from absolute virtual address
94 */
95ENTRY(pvbcopy)
96 .code32
97 push %esi
98 push %edi
99 push %ds
100 mov 16(%esp),%esi
101 mov 20(%esp),%edi
102 mov 24(%esp),%ecx
103
104 mov $flatdataseg, %ax /* selector with offset == 0 */
105 mov %ax, %ds
106 xor %eax,%eax
107
108 cld
109 rep
110 movsb
111
112 popl %ds
113 popl %edi
114 popl %esi
115 ret
