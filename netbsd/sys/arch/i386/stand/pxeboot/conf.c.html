38#include <sys/types.h>
39#include <netinet/in.h>
40#include <netinet/in_systm.h>
41
42#include <lib/libsa/stand.h>
43#ifdef SUPPORT_NFS
44#include <nfs.h>
45#endif
46#ifdef SUPPORT_TFTP
47#include <tftp.h>
48#endif
49#include <dev_net.h>
50
51#include "pxeboot.h"
52
53#ifdef SUPPORT_NFS
54struct fs_ops file_system_nfs = FS_OPS(nfs);
55#endif
56
57#ifdef SUPPORT_TFTP
58struct fs_ops file_system_tftp = FS_OPS(tftp);
59#endif
60
61struct pxeboot_fstab pxeboot_fstab[] = {
62#ifdef SUPPORT_NFS
63 { "nfs", &file_system_nfs },
64#endif
65#ifdef SUPPORT_TFTP
66 { "tftp", &file_system_tftp },
67#endif
68};
69int npxeboot_fstab = sizeof(pxeboot_fstab) / sizeof(pxeboot_fstab[0]);
70
71/* Filled in in devopen(). */
72struct fs_ops file_system[1];
73int nfsys = 1;
74
75struct devsw devsw[] = {
76 { "net", net_strategy, net_open, net_close, net_ioctl },
77};
78int ndevs = sizeof(devsw) / sizeof(devsw[0]);
