29#ifndef _HAVE_SYSCALLEMU_H
30#define _HAVE_SYSCALLEMU_H
31
32#include <sys/cdefs.h>
33#include <sys/syscall.h>
34#include <sys/syscallargs.h>
35
36struct proc;
37
38/*
39 * Process specific syscallemu configuration
40 */
41struct syscallemu_data {
42 vaddr_t sce_user_start;
43 vaddr_t sce_user_end;
44 void *sce_md_syscall;
45};
46
47/*
48 * Get process specific syscallemu data
49 */
50struct syscallemu_data * syscallemu_getsce(struct proc *);
51
52/*
53 * Set process specific syscallemu data
54 */
55void syscallemu_setsce(struct proc *, struct syscallemu_data *);
56
57/*
58 * MD specific syscallemu syscall setup (in syscallemu_${MACHINE_ARCH}.c)
59 */
60void * md_syscallemu(struct proc *);
61
62/* syscall: "syscallemu" ret: "int" args: "uintptr_t" "uintptr_t" */
63#define SYS_syscallemu 511
64
65struct sys_syscallemu_args {
66 syscallarg(uintptr_t) user_start;
67 syscallarg(uintptr_t) user_end;
68};
69check_syscall_args(sys_syscallemu);
70
71int sys_syscallemu(struct lwp *, const struct sys_syscallemu_args *,
72 register_t *);
73
74__CTASSERT(SYS_syscallemu >= SYS_MAXSYSCALL);
75__CTASSERT(SYS_syscallemu < SYS_NSYSENT);
76
77#endif /* !_HAVE_SYSCALLEMU_H */
