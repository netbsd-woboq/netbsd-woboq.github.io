64#ifndef _X86_BUS_H_
65#define _X86_BUS_H_
66
67#include <x86/busdefs.h>

69#ifdef BUS_SPACE_DEBUG
70#define BUS_SPACE_ALIGNED_ADDRESS(p, t) \
71 ((((u_long)(p)) & (sizeof(t)-1)) == 0)
72#define BUS_SPACE_ALIGNED_POINTER(p, t) BUS_SPACE_ALIGNED_ADDRESS(p, t)
73#else
74#define BUS_SPACE_ALIGNED_POINTER(p, t) ALIGNED_POINTER(p, t)
75#endif /* BUS_SPACE_DEBUG */

77/*
78 * Bus address and size types
79 */
80typedef paddr_t bus_addr_t;
81typedef size_t bus_size_t;
82
83struct bus_space_tag;
84typedef struct bus_space_tag *bus_space_tag_t;
85
86struct bus_space_tag {
87 int bst_type;
88 bus_space_tag_t bst_super;
89 /* bst_present: bitmap indicating overrides present (1) in *this* tag,
90 * bst_exists: bitmap indicating overrides present (1) in *this* tag
91 * or in an ancestor's tag (follow bst_super to ancestors)
92 */
93 uint64_t bst_present;
94 uint64_t bst_exists;
95 const struct bus_space_overrides *bst_ov;
96 void *bst_ctx;
97};
98
99typedef vaddr_t bus_space_handle_t;
100
101typedef struct x86_bus_dma_tag *bus_dma_tag_t;
102typedef struct x86_bus_dmamap *bus_dmamap_t;
103
104#define BUS_DMA_TAG_VALID(__t) ((__t) != NULL)
105
106/*
107 * bus_dma_segment_t
108 *
109 * Describes a single contiguous DMA transaction. Values
110 * are suitable for programming into DMA registers.
111 */
112typedef struct x86_bus_dma_segment {
113 bus_addr_t ds_addr; /* DMA address */
114 bus_size_t ds_len; /* length of transfer */
115} bus_dma_segment_t;
116
117/*
118 * bus_dmamap_t
119 *
120 * Describes a DMA mapping. XXX exposing structure to LKM.
121 */
122struct x86_bus_dmamap {
123 /*
124 * PRIVATE MEMBERS: not for use by machine-independent code.
125 */
126 bus_size_t _dm_size; /* largest DMA transfer mappable */
127 int _dm_segcnt; /* number of segs this map can map */
128 bus_size_t _dm_maxmaxsegsz; /* fixed largest possible segment */
129 bus_size_t _dm_boundary; /* don't cross this */
130 bus_addr_t _dm_bounce_thresh; /* bounce threshold; see tag */
131 int _dm_flags; /* misc. flags */
132
133 void *_dm_cookie; /* cookie for bus-specific functions */
134
135 /*
136 * PUBLIC MEMBERS: these are used by machine-independent code.
137 */
138 bus_size_t dm_maxsegsz; /* largest possible segment */
139 bus_size_t dm_mapsize; /* size of the mapping */
140 int dm_nsegs; /* # valid segments in mapping */
141 bus_dma_segment_t dm_segs[1]; /* segments; variable length */
142};
143
144#endif /* _X86_BUS_H_ */
