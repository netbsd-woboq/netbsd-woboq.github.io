18};

43/*
44 * AMD Cache Info:
45 *
46 * Barcelona, Phenom:
47 *
48 * Function 8000.0005 L1 TLB/Cache Information
49 * EAX -- L1 TLB 2/4MB pages
50 * EBX -- L1 TLB 4K pages
51 * ECX -- L1 D-cache
52 * EDX -- L1 I-cache
53 *
54 * Function 8000.0006 L2 TLB/Cache Information
55 * EAX -- L2 TLB 2/4MB pages
56 * EBX -- L2 TLB 4K pages
57 * ECX -- L2 Unified cache
58 * EDX -- L3 Unified Cache
59 *
60 * Function 8000.0019 TLB 1GB Page Information
61 * EAX -- L1 1GB pages
62 * EBX -- L2 1GB pages
63 * ECX -- reserved
64 * EDX -- reserved
65 *
66 * Athlon, Duron:
67 *
68 * Function 8000.0005 L1 TLB/Cache Information
69 * EAX -- L1 TLB 2/4MB pages
70 * EBX -- L1 TLB 4K pages
71 * ECX -- L1 D-cache
72 * EDX -- L1 I-cache
73 *
74 * Function 8000.0006 L2 TLB/Cache Information
75 * EAX -- L2 TLB 2/4MB pages
76 * EBX -- L2 TLB 4K pages
77 * ECX -- L2 Unified cache
78 * EDX -- reserved
79 *
80 * K5, K6:
81 *
82 * Function 8000.0005 L1 TLB/Cache Information
83 * EAX -- reserved
84 * EBX -- TLB 4K pages
85 * ECX -- L1 D-cache
86 * EDX -- L1 I-cache
87 *
88 * K6-III:
89 *
90 * Function 8000.0006 L2 Cache Information
91 * EAX -- reserved
92 * EBX -- reserved
93 * ECX -- L2 Unified cache
94 * EDX -- reserved
95 */
