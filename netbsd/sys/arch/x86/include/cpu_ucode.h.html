34#define CPU_UCODE_LOADER_AMD 0
35struct cpu_ucode_version_amd {
36 uint64_t version;
37};
38
39#define CPU_UCODE_LOADER_INTEL1 1
40struct cpu_ucode_version_intel1 {
41 uint32_t ucodeversion;
42 int platformid;
43};

59struct intel1_ucode_header {
60 uint32_t uh_header_ver;
61 uint32_t uh_rev;
62 uint32_t uh_date;
63 uint32_t uh_signature;
64 uint32_t uh_checksum;
65 uint32_t uh_loader_rev;
66 uint32_t uh_proc_flags;
67 uint32_t uh_data_size;
68 uint32_t uh_total_size;
69 uint32_t uh_reserved[3];
70};
71
72struct intel1_ucode_proc_signature {
73 uint32_t ups_signature;
74 uint32_t ups_proc_flags;
75 uint32_t ups_checksum;
76};
77
78struct intel1_ucode_ext_table {
79 uint32_t uet_count;
80 uint32_t uet_checksum;
81 uint32_t uet_reserved[3];
82 struct intel1_ucode_proc_signature uet_proc_sig[1];
83};
