62/*
63 * MP: intr_handle_t is bitfielded.
64 * ih&0xff -> legacy irq number.
65 * ih&0x10000000 -> if 0, old-style isa irq; if 1, routed via ioapic.
66 * (ih&0xff0000)>>16 -> ioapic id.
67 * (ih&0x00ff00)>>8 -> ioapic pin.
68 *
69 * MSI/MSI-X:
70 * (ih&0x000ff80000000000)>>43 -> MSI/MSI-X device id.
71 * (ih&0x000007ff00000000)>>32 -> MSI/MSI-X vector id in a device.
72 */
73#define MPSAFE_MASK 0x80000000ULL
74#define APIC_INT_VIA_APIC 0x10000000ULL
75#define APIC_INT_VIA_MSI 0x20000000ULL
76#define APIC_INT_APIC_MASK 0x00ff0000ULL
77#define APIC_INT_APIC_SHIFT 16
78#define APIC_INT_PIN_MASK 0x0000ff00ULL
79#define APIC_INT_PIN_SHIFT 8
80
81#define APIC_IRQ_APIC(x) (int)(((x) & APIC_INT_APIC_MASK) >> APIC_INT_APIC_SHIFT)
82#define APIC_IRQ_PIN(x) (int)(((x) & APIC_INT_PIN_MASK) >> APIC_INT_PIN_SHIFT)
83#define APIC_IRQ_ISLEGACY(x) (bool)(!((x) & APIC_INT_VIA_APIC))
84#define APIC_IRQ_LEGACY_IRQ(x) (int)((x) & 0xff)
85
86#define INT_VIA_MSI(x) (bool)(((x) & APIC_INT_VIA_MSI) != 0)
87
88#define MSI_INT_MSIX 0x1000000000000000ULL
89#define MSI_INT_DEV_MASK 0x000ff80000000000ULL
90#define MSI_INT_VEC_MASK 0x000007ff00000000ULL
91
92#define MSI_INT_IS_MSIX(x) (bool)((((x) & MSI_INT_MSIX) != 0))
93#define MSI_INT_MAKE_MSI(x) ((x) &= ~MSI_INT_MSIX)
94#define MSI_INT_MAKE_MSIX(x) ((x) |= MSI_INT_MSIX)
95#define MSI_INT_DEV(x) __SHIFTOUT((x), MSI_INT_DEV_MASK)
96#define MSI_INT_VEC(x) __SHIFTOUT((x), MSI_INT_VEC_MASK)
