32#ifndef _X86_MUTEX_H_
33#define _X86_MUTEX_H_
34
35struct kmutex {
36 union {
37 volatile uintptr_t mtxa_owner;
38#ifdef __MUTEX_PRIVATE
39 struct {
40 volatile uint8_t mtxs_dummy;
41 ipl_cookie_t mtxs_ipl;
42 __cpu_simple_lock_t mtxs_lock;
43 volatile uint8_t mtxs_unused;
44 } s;
45#endif
46 } u;
47};
48
49#ifdef __MUTEX_PRIVATE
50
51#define mtx_owner u.mtxa_owner
52#define mtx_ipl u.s.mtxs_ipl
53#define mtx_lock u.s.mtxs_lock
54
55#define __HAVE_MUTEX_STUBS 1
56#define __HAVE_SPIN_MUTEX_STUBS 1
57#define __HAVE_SIMPLE_MUTEXES 1
58
59/*
60 * MUTEX_RECEIVE: technically, no memory barrier is required
61 * as 'ret' implies a load fence. However we need this to
62 * handle a bug with some Opteron revisions. See patch.c,
63 * lock_stubs.S.
64 */
65#define MUTEX_RECEIVE(mtx) membar_consumer()
66
67/*
68 * MUTEX_GIVE: no memory barrier required, as _lock_cas() will take care of it.
69 */
70#define MUTEX_GIVE(mtx) /* nothing */
71
72#define MUTEX_CAS(p, o, n) \
73 (_atomic_cas_ulong((volatile unsigned long *)(p), (o), (n)) == (o))
74
75unsigned long _atomic_cas_ulong(volatile unsigned long *,
76 unsigned long, unsigned long);
77
78#endif /* __MUTEX_PRIVATE */
79
80#endif /* _X86_MUTEX_H_ */
