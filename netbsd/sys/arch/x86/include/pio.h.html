32#ifndef _X86_PIO_H_
33#define _X86_PIO_H_
34
35/*
36 * Functions to provide access to x86 programmed I/O instructions.
37 */
38
39uint8_t inb(unsigned);
40void insb(unsigned, void *, int);
41uint16_t inw(unsigned);
42void insw(unsigned, void *, int);
43uint32_t inl(unsigned);
44void insl(unsigned, void *, int);
45
46void outb(unsigned, uint8_t);
47void outsb(unsigned, void *, int);
48void outw(unsigned, uint16_t);
49void outsw(unsigned, void *, int);
50void outl(unsigned, uint32_t);
51void outsl(unsigned, void *, int);
52
53#endif /* _X86_PIO_H_ */
