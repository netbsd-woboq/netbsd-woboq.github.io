37#ifndef _X86_PSL_H_
38#define _X86_PSL_H_

40/*
41 * 386 processor status longword.
42 */
43#define PSL_C 0x00000001 /* carry flag */
44#define PSL_PF 0x00000004 /* parity flag */
45#define PSL_AF 0x00000010 /* auxiliary carry flag */
46#define PSL_Z 0x00000040 /* zero flag */
47#define PSL_N 0x00000080 /* sign flag */
48#define PSL_T 0x00000100 /* trap flag */
49#define PSL_I 0x00000200 /* interrupt enable flag */
50#define PSL_D 0x00000400 /* direction flag */
51#define PSL_V 0x00000800 /* overflow flag */
52#define PSL_IOPL 0x00003000 /* i/o privilege level */
53#define PSL_NT 0x00004000 /* nested task */
54#define PSL_RF 0x00010000 /* resume flag */
55#define PSL_VM 0x00020000 /* virtual 8086 mode */
56#define PSL_AC 0x00040000 /* alignment check flag */
57#define PSL_VIF 0x00080000 /* virtual interrupt enable flag */
58#define PSL_VIP 0x00100000 /* virtual interrupt pending flag */
59#define PSL_ID 0x00200000 /* identification flag */
60
61#define PSL_MBO 0x00000002 /* must be one bits */
62#define PSL_MBZ 0xffc08028 /* must be zero bits */
63
64#define PSL_USERSET (PSL_MBO | PSL_I)
65#define PSL_USERSTATIC (PSL_MBO | PSL_MBZ | PSL_I | PSL_IOPL | PSL_NT | PSL_VM | PSL_VIF | PSL_VIP)
66#define PSL_USER (PSL_C | PSL_PF | PSL_AF | PSL_Z | PSL_N | \
67 PSL_T | PSL_V | PSL_D | PSL_AC)
68#define PSL_CLEARSIG (PSL_T | PSL_VM | PSL_AC | PSL_D)
