42struct amdpcib_softc {
43 /* we are calling pcibattach(), which assumes this starts like this: */
44 struct pcib_softc sc_pcib;
45 bus_space_tag_t sc_memt;
46 bus_space_handle_t sc_memh;
47};
48
49static int amdpcib_match(device_t, cfdata_t, void *);
50static void amdpcib_attach(device_t, device_t, void *);
51static int amdpcib_search(device_t, cfdata_t, const int *, void *);
52
53CFATTACH_DECL_NEW(amdpcib, sizeof(struct amdpcib_softc), amdpcib_match,
54 amdpcib_attach, NULL, NULL);
55
56static int
57amdpcib_match(device_t parent, cfdata_t match, void *aux)
58{
59 struct pci_attach_args *pa = aux;
60
61 if ((PCI_VENDOR(pa->pa_id) == PCI_VENDOR_AMD) &&
62 (PCI_PRODUCT(pa->pa_id) == PCI_PRODUCT_AMD_PBC8111_LPC))
63 return 2;
64
65 return 0;
66}
67
68static void
69amdpcib_attach(device_t parent, device_t self, void *aux)
70{
71 struct pci_attach_args *pa = aux;
72
73 pcibattach(parent, self, aux);
74 config_search_loc(amdpcib_search, self, "amdpcib", NULL, pa);
75}
76
77static int
78amdpcib_search(device_t parent, cfdata_t cf, const int *locs, void *aux)
79{
80
81 if (config_match(parent, cf, aux))
82 config_attach_loc(parent, cf, locs, aux, NULL);
83
84 return 0;
85}
