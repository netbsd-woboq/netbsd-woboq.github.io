64#ifdef __HAVE_PCIIDE_MACHDEP_COMPAT_INTR_ESTABLISH
65void *
66pciide_machdep_compat_intr_establish(device_t dev,
67 const struct pci_attach_args *pa, int chan, int (*func)(void *),
68 void *arg)
69{
70 int irq;
71 void *cookie;
72#if NIOAPIC > 0
73 intr_handle_t mpih;
74 char buf[PCI_INTRSTR_LEN];
75#endif
76 char intr_xname[64];
77
78 snprintf(intr_xname, sizeof(intr_xname), "%s %s",
79 device_xname(dev), PCIIDE_CHANNEL_NAME(chan));
80
81 irq = PCIIDE_COMPAT_IRQ(chan);
82 cookie = isa_intr_establish_xname(NULL, irq, IST_EDGE, IPL_BIO,
83 func, arg, intr_xname);
84 if (cookie == NULL)
85 return NULL;
86#if NIOAPIC > 0
87 if (mp_busses != NULL &&
88 (intr_find_mpmapping(mp_isa_bus, irq, &mpih) == 0 ||
89 intr_find_mpmapping(mp_eisa_bus, irq, &mpih) == 0)) {
90 mpih |= irq;
91 aprint_normal_dev(dev, "%s channel interrupting at %s\n",
92 PCIIDE_CHANNEL_NAME(chan),
93 intr_string(mpih, buf, sizeof(buf)));
94 } else
95#endif
96 aprint_normal_dev(dev, "%s channel interrupting at irq %d\n",
97 PCIIDE_CHANNEL_NAME(chan), irq);
98 return cookie;
99}
100#endif /* __HAVE_PCIIDE_MACHDEP_COMPAT_INTR_ESTABLISH */
101
102#ifdef __HAVE_PCIIDE_MACHDEP_COMPAT_INTR_DISESTABLISH
103void
104pciide_machdep_compat_intr_disestablish(device_t dev, pci_chipset_tag_t pc,
105 int chan, void *cookie)
106{
107 isa_intr_disestablish(NULL, cookie);
108 return;
109}
