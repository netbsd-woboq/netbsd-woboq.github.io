108#define BIOS32_START 0xe0000
109#define BIOS32_SIZE 0x20000
110#define BIOS32_END (BIOS32_START + BIOS32_SIZE - 0x10)

120/*
121 * Initialize the BIOS32 interface.
122 */
123void
124bios32_init(void)
125{
126 uint8_t *p;
127#ifdef i386
128 paddr_t entry = 0;
129 unsigned char cksum;
130 int i;
131
132 for (p = (uint8_t *)ISA_HOLE_VADDR(BIOS32_START);
133 p < (uint8_t *)ISA_HOLE_VADDR(BIOS32_END);
134 p += 16) {
135 if (*(int *)p != BIOS32_MAKESIG('_', '3', '2', '_'))
136 continue;
137
138 cksum = 0;
139 for (i = 0; i < 16; i++)
140 cksum += *(unsigned char *)(p + i);
141 if (cksum != 0)
142 continue;
143
144 if (*(p + 9) != 1)
145 continue;
146
147 entry = *(uint32_t *)(p + 4);
148
149 aprint_debug("BIOS32 rev. %d found at 0x%lx\n",
150 *(p + 8), (u_long)entry);
151
152 if (entry < BIOS32_START ||
153 entry >= BIOS32_END) {
154 aprint_error("BIOS32 entry point outside "
155 "allowable range\n");
156 entry = 0;
157 }
158 break;
159 }
160
161 if (entry != 0) {
162 bios32_entry.offset = (void *)ISA_HOLE_VADDR(entry);
163 bios32_entry.segment = GSEL(GCODE_SEL, SEL_KPL);
164 }
165#endif
166
167 /* see if we have SMBIOS extensions */
168#ifndef XENPV
169 if (efi_probe()) {
170 p = efi_getcfgtbl(&EFI_UUID_SMBIOS3);
171 if (p != NULL && smbios3_check_header(p)) {
172 smbios3_map_kva(p);
173 return;
174 }
175 p = efi_getcfgtbl(&EFI_UUID_SMBIOS);
176 if (p != NULL && smbios2_check_header(p)) {
177 smbios2_map_kva(p);
178 return;
179 }
180 }
181#endif
182 for (p = ISA_HOLE_VADDR(SMBIOS_START);
183 p < (uint8_t *)ISA_HOLE_VADDR(SMBIOS_END); p+= 16) {
184 if (smbios3_check_header(p)) {
185 smbios3_map_kva(p);
186 return;
187 }
188 if (smbios2_check_header(p)) {
189 smbios2_map_kva(p);
190 return;
191 }
192 }
193}

195/*
196 * Call BIOS32 to locate the specified BIOS32 service, and fill
197 * in the entry point information.
198 */
199int
200bios32_service(uint32_t service, bios32_entry_t e, bios32_entry_info_t ei)
201{
202#ifdef i386
203 uint32_t eax, ebx, ecx, edx;
204 paddr_t entry;
205
206 if (bios32_entry.offset == 0)
207 return 0; /* BIOS32 not present */
208
209 __asm volatile("lcall *(%%edi)"
210 : "=a" (eax), "=b" (ebx), "=c" (ecx), "=d" (edx)
211 : "0" (service), "1" (0), "D" (&bios32_entry));
212
213 if ((eax & 0xff) != 0)
214 return 0; /* service not found */
215
216 entry = ebx + edx;
217
218 if (entry < BIOS32_START || entry >= BIOS32_END) {
219 aprint_error("BIOS32: entry point for service %c%c%c%c is "
220 "outside allowable range\n",
221 service & 0xff,
222 (service >> 8) & 0xff,
223 (service >> 16) & 0xff,
224 (service >> 24) & 0xff);
225 return 0;
226 }
227
228 e->offset = (void *)ISA_HOLE_VADDR(entry);
229 e->segment = GSEL(GCODE_SEL, SEL_KPL);
230
231 ei->bei_base = ebx;
232 ei->bei_size = ecx;
233 ei->bei_entry = entry;
234#else
235 (void)service;
236 (void)e;
237 (void)ei;
238 panic("bios32_service not implemented on amd64");
239#endif
240
241 return 1;
242}

244static int
245smbios2_check_header(const uint8_t *p)
246{
247 const struct smbhdr *sh = (const struct smbhdr *)p;
248 uint8_t chksum;
249 int i;
250
251 if (sh->sig != BIOS32_MAKESIG('_', 'S', 'M', '_'))
252 return 0;
253 i = sh->len;
254 for (chksum = 0; i--; )
255 chksum += p[i];
256 if (chksum != 0)
257 return 0;
258 p += 0x10;
259 if (p[0] != '_' || p[1] != 'D' || p[2] != 'M' ||
260 p[3] != 'I' || p[4] != '_')
261 return 0;
262 for (chksum = 0, i = 0xf; i--; )
263 chksum += p[i];
264 if (chksum != 0)
265 return 0;
266
267 return 1;
268}

270static void
271smbios2_map_kva(const uint8_t *p)
272{
273 const struct smbhdr *sh = (const struct smbhdr *)p;
274 paddr_t pa, end;
275 vaddr_t eva;
276
277 pa = trunc_page(sh->addr);
278 end = round_page(sh->addr + sh->size);
279 eva = uvm_km_alloc(kernel_map, end - pa, 0, UVM_KMF_VAONLY);
280 if (eva == 0)
281 return;
282
283 smbios_entry.addr = (uint8_t *)(eva + (sh->addr & PGOFSET));
284 smbios_entry.len = sh->size;
285 smbios_entry.rev = 0;
286 smbios_entry.mjr = sh->majrev;
287 smbios_entry.min = sh->minrev;
288 smbios_entry.doc = 0;
289 smbios_entry.count = sh->count;
290
291 for (; pa < end; pa+= NBPG, eva+= NBPG)
292#ifdef XENPV
293 pmap_kenter_ma(eva, pa, VM_PROT_READ, 0);
294#else
295 pmap_kenter_pa(eva, pa, VM_PROT_READ, 0);
296#endif
297 pmap_update(pmap_kernel());
298
299 aprint_debug("SMBIOS rev. %d.%d @ 0x%lx (%d entries)\n",
300 sh->majrev, sh->minrev, (u_long)sh->addr, sh->count);
301}

303static int
304smbios3_check_header(const uint8_t *p)
305{
306 const struct smb3hdr *sh = (const struct smb3hdr *)p;
307 uint8_t chksum;
308 int i;
309
310 if (p[0] != '_' || p[1] != 'S' || p[2] != 'M' ||
311 p[3] != '3' || p[4] != '_')
312 return 0;
313 i = sh->len;
314 for (chksum = 0; i--; )
315 chksum += p[i];
316 if (chksum != 0)
317 return 0;
318 if (sh->eprev != SMBIOS3_EPREV_3_0)
319 return 0;
320
321 return 1;
322}

324static void
325smbios3_map_kva(const uint8_t *p)
326{
327 const struct smb3hdr *sh = (const struct smb3hdr *)p;
328 paddr_t pa, end;
329 vaddr_t eva;
330
331 pa = trunc_page(sh->addr);
332 end = round_page(sh->addr + sh->size);
333 eva = uvm_km_alloc(kernel_map, end - pa, 0, UVM_KMF_VAONLY);
334 if (eva == 0)
335 return;
336
337 smbios_entry.addr = (uint8_t *)(eva + ((vaddr_t)sh->addr & PGOFSET));
338 smbios_entry.len = sh->size;
339 smbios_entry.rev = sh->eprev;
340 smbios_entry.mjr = sh->majrev;
341 smbios_entry.min = sh->minrev;
342 smbios_entry.doc = sh->docrev;
343 smbios_entry.count = UINT16_MAX;
344
345 for (; pa < end; pa += NBPG, eva += NBPG)
346#ifdef XENPV
347 pmap_kenter_ma(eva, pa, VM_PROT_READ, 0);
348#else
349 pmap_kenter_pa(eva, pa, VM_PROT_READ, 0);
350#endif
351 pmap_update(pmap_kernel());
352
353 aprint_debug("SMBIOS rev. %d.%d.%d @ 0x%lx\n", sh->majrev,
354 sh->minrev, sh->docrev, (u_long)sh->addr);
355}

357/*
358 * smbios_find_table() takes a caller supplied smbios struct type and
359 * a pointer to a handle (struct smbtable) returning one if the structure
360 * is sucessfully located and zero otherwise. Callers should take care
361 * to initilize the cookie field of the smbtable structure to zero before
362 * the first invocation of this function.
363 * Multiple tables of the same type can be located by repeadtly calling
364 * smbios_find_table with the same arguments.
365 */
366int
367smbios_find_table(uint8_t type, struct smbtable *st)
368{
369 uint8_t *va, *end;
370 struct smbtblhdr *hdr;
371 int ret = 0, tcount = 1;
372
373 va = smbios_entry.addr;
374 end = va + smbios_entry.len;
375
376 /*
377 * The cookie field of the smtable structure is used to locate
378 * multiple instances of a table of an arbitrary type. Following the
379 * sucessful location of a table, the type is encoded as bits 0:7 of
380 * the cookie value, the offset in terms of the number of structures
381 * preceding that referenced by the handle is encoded in bits 15:31.
382 */
383 if ((st->cookie & 0xfff) == type && st->cookie >> 16) {
384 if ((uint8_t *)st->hdr >= va && (uint8_t *)st->hdr < end) {
385 hdr = st->hdr;
386 if (hdr->type == type) {
387 va = (uint8_t *)hdr + hdr->size;
388 for (; va + 1 < end; va++)
389 if (*va == 0 && *(va + 1) == 0)
390 break;
391 va+= 2;
392 tcount = st->cookie >> 16;
393 }
394 }
395 }
396 for (; va + sizeof(struct smbtblhdr) < end && tcount <=
397 smbios_entry.count; tcount++) {
398 hdr = (struct smbtblhdr *)va;
399 if (hdr->type == type) {
400 ret = 1;
401 st->hdr = hdr;
402 st->tblhdr = va + sizeof(struct smbtblhdr);
403 st->cookie = (tcount + 1) << 16 | type;
404 break;
405 }
406 if (hdr->type == SMBIOS_TYPE_EOT)
407 break;
408 va+= hdr->size;
409 for (; va + 1 < end; va++)
410 if (*va == 0 && *(va + 1) == 0)
411 break;
412 va+=2;
413 }
414
415 return ret;
416}

418char *
419smbios_get_string(struct smbtable *st, uint8_t indx, char *dest, size_t len)
420{
421 uint8_t *va, *end;
422 char *ret = NULL;
423 int i;
424
425 va = (uint8_t *)st->hdr + st->hdr->size;
426 end = smbios_entry.addr + smbios_entry.len;
427 for (i = 1; va < end && i < indx && *va; i++)
428 while (*va++)
429 ;
430 if (i == indx) {
431 if (va + len < end) {
432 ret = dest;
433 memcpy(ret, va, len);
434 ret[len - 1] = '\0';
435 }
436 }
437
438 return ret;
439}
