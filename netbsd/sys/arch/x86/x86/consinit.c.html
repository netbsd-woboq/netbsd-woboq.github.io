151/*
152 * consinit:
153 * initialize the system console.
154 * XXX - shouldn't deal with this initted thing, but then,
155 * it shouldn't be called from init386 either.
156 */
157void
158consinit(void)
159{
160 const struct btinfo_console *consinfo;
161#if (NGENFB > 0)
162 const struct btinfo_framebuffer *fbinfo;
163#endif
164 static int initted;
165#if (NCOM > 0)
166 int rv;
167#endif
168
169 if (initted)
170 return;
171 initted = 1;
172
173#ifndef CONS_OVERRIDE
174 consinfo = lookup_bootinfo(BTINFO_CONSOLE);
175 if (!consinfo)
176#endif
177 consinfo = &default_consinfo;
178
179#if (NGENFB > 0)
180 fbinfo = lookup_bootinfo(BTINFO_FRAMEBUFFER);
181#endif
182
183 if (!strcmp(consinfo->devname, "pc")) {
184 int error;
185#if (NGENFB > 0)
186 if (fbinfo && fbinfo->physaddr > 0) {
187 if (x86_genfb_cnattach() == -1) {
188 initted = 0; /* defer */
189 return;
190 }
191 genfb_cnattach();
192 goto dokbd;
193 }
194 genfb_disable();
195#endif
196#if (NVGA > 0)
197 if (!vga_cnattach(x86_bus_space_io, x86_bus_space_mem,
198 -1, 1))
199 goto dokbd;
200#endif
201#if (NEGA > 0)
202 if (!ega_cnattach(x86_bus_space_io, x86_bus_space_mem))
203 goto dokbd;
204#endif
205#if (NPCDISPLAY > 0)
206 if (!pcdisplay_cnattach(x86_bus_space_io, x86_bus_space_mem))
207 goto dokbd;
208#endif
209 if (0) goto dokbd; /* XXX stupid gcc */
210dokbd:
211 error = ENODEV;
212#if (NPCKBC > 0)
213 error = pckbc_cnattach(x86_bus_space_io, IO_KBD, KBCMDP,
214 PCKBC_KBD_SLOT, 0);
215#endif
216#if (NHVKBD > 0)
217 if (error && vm_guest == VM_GUEST_HV)
218 error = hvkbd_cnattach();
219#endif
220#if (NUKBD > 0)
221 if (error)
222 error = ukbd_cnattach();
223#endif
224 if (error)
225 printf("WARNING: no console keyboard, error=%d\n",
226 error);
227 return;
228 }
229#if (NCOM > 0)
230 if (!strcmp(consinfo->devname, "com")) {
231 int addr = consinfo->addr;
232 int speed = consinfo->speed;
233
234#if (NCOM_PUC > 0) && defined(PUC_CNAUTO)
235 puc_cnprobe(NULL);
236 rv = puc_cninit(NULL);
237 if (rv == 0)
238 return;
239#endif
240
241 if (addr == 0)
242 addr = CONADDR;
243 if (speed == 0)
244 speed = CONSPEED;
245
246 rv = comcnattach(x86_bus_space_io, addr, speed,
247 COM_FREQ, COM_TYPE_NORMAL, comcnmode);
248 if (rv != 0)
249 panic("can't init serial console @%x", consinfo->addr);
250 return;
251 }
252#endif
253#if (NNULLCONS > 0)
254 if (!strcmp(consinfo->devname, "nullcons")) {
255 void nullcninit(struct consdev *cn);
256
257 nullcninit(0);
258 return;
259 }
260#endif
261 panic("invalid console device %s", consinfo->devname);
262}
