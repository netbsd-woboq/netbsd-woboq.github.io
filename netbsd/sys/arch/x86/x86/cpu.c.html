224#ifdef __HAVE_PCPU_AREA
225void
226cpu_pcpuarea_init(struct cpu_info *ci)
227{
228 struct vm_page *pg;
229 size_t i, npages;
230 vaddr_t base, va;
231 paddr_t pa;
232
233 CTASSERT(sizeof(struct pcpu_entry) % PAGE_SIZE == 0);
234
235 npages = sizeof(struct pcpu_entry) / PAGE_SIZE;
236 base = (vaddr_t)&pcpuarea->ent[cpu_index(ci)];
237
238 for (i = 0; i < npages; i++) {
239 pg = uvm_pagealloc(NULL, 0, NULL, UVM_PGA_ZERO);
240 if (pg == NULL) {
241 panic("failed to allocate pcpu PA");
242 }
243
244 va = base + i * PAGE_SIZE;
245 pa = VM_PAGE_TO_PHYS(pg);
246
247 pmap_kenter_pa(va, pa, VM_PROT_READ|VM_PROT_WRITE, 0);
248 }
249
250 pmap_update(pmap_kernel());
251}

254static void
255cpu_vm_init(struct cpu_info *ci)
256{
257 int ncolors = 2, i;
258
259 for (i = CAI_ICACHE; i <= CAI_L2CACHE; i++) {
260 struct x86_cache_info *cai;
261 int tcolors;
262
263 cai = &ci->ci_cinfo[i];
264
265 tcolors = atop(cai->cai_totalsize);
266 switch(cai->cai_associativity) {
267 case 0xff:
268 tcolors = 1; /* fully associative */
269 break;
270 case 0:
271 case 1:
272 break;
273 default:
274 tcolors /= cai->cai_associativity;
275 }
276 ncolors = uimax(ncolors, tcolors);
277 /*
278 * If the desired number of colors is not a power of
279 * two, it won't be good. Find the greatest power of
280 * two which is an even divisor of the number of colors,
281 * to preserve even coloring of pages.
282 */
283 if (ncolors & (ncolors - 1) ) {
284 int try, picked = 1;
285 for (try = 1; try < ncolors; try *= 2) {
286 if (ncolors % try == 0) picked = try;
287 }
288 if (picked == 1) {
289 panic("desired number of cache colors %d is "
290 " > 1, but not even!", ncolors);
291 }
292 ncolors = picked;
293 }
294 }
295
296 /*
297 * Knowing the size of the largest cache on this CPU, potentially
298 * re-color our pages.
299 */
300 aprint_debug_dev(ci->ci_dev, "%d page colors\n", ncolors);
301 uvm_page_recolor(ncolors);
302
303 pmap_tlb_cpu_init(ci);
304#ifndef __HAVE_DIRECT_MAP
305 pmap_vpage_cpu_init(ci);
306#endif
307}

309static void
310cpu_attach(device_t parent, device_t self, void *aux)
311{
312 struct cpu_softc *sc = device_private(self);
313 struct cpu_attach_args *caa = aux;
314 struct cpu_info *ci;
315 uintptr_t ptr;
316#if NLAPIC > 0
317 int cpunum = caa->cpu_number;
318#endif
319 static bool again;
320
321 sc->sc_dev = self;
322
323 if (ncpu > maxcpus) {
324#ifndef _LP64
325 aprint_error(": too many CPUs, please use NetBSD/amd64\n");
326#else
327 aprint_error(": too many CPUs\n");
328#endif
329 return;
330 }
331
332 /*
333 * If we're an Application Processor, allocate a cpu_info
334 * structure, otherwise use the primary's.
335 */
336 if (caa->cpu_role == CPU_ROLE_AP) {
337 if ((boothowto & RB_MD1) != 0) {
338 aprint_error(": multiprocessor boot disabled\n");
339 if (!pmf_device_register(self, NULL, NULL))
340 aprint_error_dev(self,
341 "couldn't establish power handler\n");
342 return;
343 }
344 aprint_naive(": Application Processor\n");
345 ptr = (uintptr_t)uvm_km_alloc(kernel_map,
346 sizeof(*ci) + CACHE_LINE_SIZE - 1, 0,
347 UVM_KMF_WIRED|UVM_KMF_ZERO);
348 ci = (struct cpu_info *)roundup2(ptr, CACHE_LINE_SIZE);
349 ci->ci_curldt = -1;
350 } else {
351 aprint_naive(": %s Processor\n",
352 caa->cpu_role == CPU_ROLE_SP ? "Single" : "Boot");
353 ci = &cpu_info_primary;
354#if NLAPIC > 0
355 if (cpunum != lapic_cpu_number()) {
356 /* XXX should be done earlier. */
357 uint32_t reg;
358 aprint_verbose("\n");
359 aprint_verbose_dev(self, "running CPU at apic %d"
360 " instead of at expected %d", lapic_cpu_number(),
361 cpunum);
362 reg = lapic_readreg(LAPIC_ID);
363 lapic_writereg(LAPIC_ID, (reg & ~LAPIC_ID_MASK) |
364 (cpunum << LAPIC_ID_SHIFT));
365 }
366 if (cpunum != lapic_cpu_number()) {
367 aprint_error_dev(self, "unable to reset apic id\n");
368 }
369#endif
370 }
371
372 ci->ci_self = ci;
373 sc->sc_info = ci;
374 ci->ci_dev = self;
375 ci->ci_acpiid = caa->cpu_id;
376 ci->ci_cpuid = caa->cpu_number;
377 ci->ci_func = caa->cpu_func;
378 aprint_normal("\n");
379
380 /* Must be before mi_cpu_attach(). */
381 cpu_vm_init(ci);
382
383 if (caa->cpu_role == CPU_ROLE_AP) {
384 int error;
385
386 error = mi_cpu_attach(ci);
387 if (error != 0) {
388 aprint_error_dev(self,
389 "mi_cpu_attach failed with %d\n", error);
390 return;
391 }
392#ifdef __HAVE_PCPU_AREA
393 cpu_pcpuarea_init(ci);
394#endif
395 cpu_init_tss(ci);
396 } else {
397 KASSERT(ci->ci_data.cpu_idlelwp != NULL);
398 }
399
400#ifdef SVS
401 cpu_svs_init(ci);
402#endif
403
404 pmap_reference(pmap_kernel());
405 ci->ci_pmap = pmap_kernel();
406 ci->ci_tlbstate = TLBSTATE_STALE;
407
408 /*
409 * Boot processor may not be attached first, but the below
410 * must be done to allow booting other processors.
411 */
412 if (!again) {
413 atomic_or_32(&ci->ci_flags, CPUF_PRESENT | CPUF_PRIMARY);
414 /* Basic init. */
415 cpu_intr_init(ci);
416 cpu_get_tsc_freq(ci);
417 cpu_init(ci);
418#ifdef i386
419 cpu_set_tss_gates(ci);
420#endif
421 pmap_cpu_init_late(ci);
422#if NLAPIC > 0
423 if (caa->cpu_role != CPU_ROLE_SP) {
424 /* Enable lapic. */
425 lapic_enable();
426 lapic_set_lvt();
427 lapic_calibrate_timer(ci);
428 }
429#endif
430 /* Make sure DELAY() is initialized. */
431 DELAY(1);
432 again = true;
433 }
434
435 /* further PCB init done later. */
436
437 switch (caa->cpu_role) {
438 case CPU_ROLE_SP:
439 atomic_or_32(&ci->ci_flags, CPUF_SP);
440 cpu_identify(ci);
441 x86_errata();
442 x86_cpu_idle_init();
443 break;
444
445 case CPU_ROLE_BP:
446 atomic_or_32(&ci->ci_flags, CPUF_BSP);
447 cpu_identify(ci);
448 x86_errata();
449 x86_cpu_idle_init();
450 break;
451
452#ifdef MULTIPROCESSOR
453 case CPU_ROLE_AP:
454 /*
455 * report on an AP
456 */
457 cpu_intr_init(ci);
458 gdt_alloc_cpu(ci);
459#ifdef i386
460 cpu_set_tss_gates(ci);
461#endif
462 pmap_cpu_init_late(ci);
463 cpu_start_secondary(ci);
464 if (ci->ci_flags & CPUF_PRESENT) {
465 struct cpu_info *tmp;
466
467 cpu_identify(ci);
468 tmp = cpu_info_list;
469 while (tmp->ci_next)
470 tmp = tmp->ci_next;
471
472 tmp->ci_next = ci;
473 }
474 break;
475#endif
476
477 default:
478 panic("unknown processor type??\n");
479 }
480
481 pat_init(ci);
482
483 if (!pmf_device_register1(self, cpu_suspend, cpu_resume, cpu_shutdown))
484 aprint_error_dev(self, "couldn't establish power handler\n");
485
486#ifdef MULTIPROCESSOR
487 if (mp_verbose) {
488 struct lwp *l = ci->ci_data.cpu_idlelwp;
489 struct pcb *pcb = lwp_getpcb(l);
490
491 aprint_verbose_dev(self,
492 "idle lwp at %p, idle sp at %p\n",
493 l,
494#ifdef i386
495 (void *)pcb->pcb_esp
496#else
497 (void *)pcb->pcb_rsp
498#endif
499 );
500 }
501#endif
502
503 /*
504 * Postpone the "cpufeaturebus" scan.
505 * It is safe to scan the pseudo-bus
506 * only after all CPUs have attached.
507 */
508 (void)config_defer(self, cpu_defer);
509}

517static int
518cpu_rescan(device_t self, const char *ifattr, const int *locators)
519{
520 struct cpu_softc *sc = device_private(self);
521 struct cpufeature_attach_args cfaa;
522 struct cpu_info *ci = sc->sc_info;
523
524 memset(&cfaa, 0, sizeof(cfaa));
525 cfaa.ci = ci;
526
527 if (ifattr_match(ifattr, "cpufeaturebus")) {
528 if (ci->ci_frequency == NULL) {
529 cfaa.name = "frequency";
530 ci->ci_frequency = config_found_ia(self,
531 "cpufeaturebus", &cfaa, NULL);
532 }
533
534 if (ci->ci_padlock == NULL) {
535 cfaa.name = "padlock";
536 ci->ci_padlock = config_found_ia(self,
537 "cpufeaturebus", &cfaa, NULL);
538 }
539
540 if (ci->ci_temperature == NULL) {
541 cfaa.name = "temperature";
542 ci->ci_temperature = config_found_ia(self,
543 "cpufeaturebus", &cfaa, NULL);
544 }
545
546 if (ci->ci_vm == NULL) {
547 cfaa.name = "vm";
548 ci->ci_vm = config_found_ia(self,
549 "cpufeaturebus", &cfaa, NULL);
550 }
551 }
552
553 return 0;
554}

575/*
576 * Initialize the processor appropriately.
577 */
578
579void
580cpu_init(struct cpu_info *ci)
581{
582 extern int x86_fpu_save;
583 uint32_t cr4 = 0;
584
585 lcr0(rcr0() | CR0_WP);
586
587 /* If global TLB caching is supported, enable it */
588 if (cpu_feature[0] & CPUID_PGE)
589 cr4 |= CR4_PGE;
590
591 /*
592 * If we have FXSAVE/FXRESTOR, use them.
593 */
594 if (cpu_feature[0] & CPUID_FXSR) {
595 cr4 |= CR4_OSFXSR;
596
597 /*
598 * If we have SSE/SSE2, enable XMM exceptions.
599 */
600 if (cpu_feature[0] & (CPUID_SSE|CPUID_SSE2))
601 cr4 |= CR4_OSXMMEXCPT;
602 }
603
604 /* If xsave is supported, enable it */
605 if (cpu_feature[1] & CPUID2_XSAVE)
606 cr4 |= CR4_OSXSAVE;
607
608 /* If SMEP is supported, enable it */
609 if (cpu_feature[5] & CPUID_SEF_SMEP)
610 cr4 |= CR4_SMEP;
611
612 /* If SMAP is supported, enable it */
613 if (cpu_feature[5] & CPUID_SEF_SMAP)
614 cr4 |= CR4_SMAP;
615
616#ifdef SVS
617 /* If PCID is supported, enable it */
618 if (svs_pcid)
619 cr4 |= CR4_PCIDE;
620#endif
621
622 if (cr4) {
623 cr4 |= rcr4();
624 lcr4(cr4);
625 }
626
627 /*
628 * Changing CR4 register may change cpuid values. For example, setting
629 * CR4_OSXSAVE sets CPUID2_OSXSAVE. The CPUID2_OSXSAVE is in
630 * ci_feat_val[1], so update it.
631 * XXX Other than ci_feat_val[1] might be changed.
632 */
633 if (cpuid_level >= 1) {
634 u_int descs[4];
635
636 x86_cpuid(1, descs);
637 ci->ci_feat_val[1] = descs[2];
638 }
639
640 if (x86_fpu_save >= FPU_SAVE_FXSAVE) {
641 fpuinit_mxcsr_mask();
642 }
643
644 /* If xsave is enabled, enable all fpu features */
645 if (cr4 & CR4_OSXSAVE)
646 wrxcr(0, x86_xsave_features & XCR0_FPU);
647
648#ifdef MTRR
649 /*
650 * On a P6 or above, initialize MTRR's if the hardware supports them.
651 */
652 if (cpu_feature[0] & CPUID_MTRR) {
653 if ((ci->ci_flags & CPUF_AP) == 0)
654 i686_mtrr_init_first();
655 mtrr_init_cpu(ci);
656 }
657
658#ifdef i386
659 if (strcmp((char *)(ci->ci_vendor), "AuthenticAMD") == 0) {
660 /*
661 * Must be a K6-2 Step >= 7 or a K6-III.
662 */
663 if (CPUID_TO_FAMILY(ci->ci_signature) == 5) {
664 if (CPUID_TO_MODEL(ci->ci_signature) > 8 ||
665 (CPUID_TO_MODEL(ci->ci_signature) == 8 &&
666 CPUID_TO_STEPPING(ci->ci_signature) >= 7)) {
667 mtrr_funcs = &k6_mtrr_funcs;
668 k6_mtrr_init_first();
669 mtrr_init_cpu(ci);
670 }
671 }
672 }
673#endif /* i386 */
674#endif /* MTRR */
675
676 if (ci != &cpu_info_primary) {
677 /* Synchronize TSC */
678 wbinvd();
679 atomic_or_32(&ci->ci_flags, CPUF_RUNNING);
680 tsc_sync_ap(ci);
681 } else {
682 atomic_or_32(&ci->ci_flags, CPUF_RUNNING);
683 }
684}

686#ifdef MULTIPROCESSOR
687void
688cpu_boot_secondary_processors(void)
689{
690 struct cpu_info *ci;
691 kcpuset_t *cpus;
692 u_long i;
693
694#ifndef XEN
695 /* Now that we know the number of CPUs, patch the text segment. */
696 x86_patch(false);
697#endif
698
699 kcpuset_create(&cpus, true);
700 kcpuset_set(cpus, cpu_index(curcpu()));
701 for (i = 0; i < maxcpus; i++) {
702 ci = cpu_lookup(i);
703 if (ci == NULL)
704 continue;
705 if (ci->ci_data.cpu_idlelwp == NULL)
706 continue;
707 if ((ci->ci_flags & CPUF_PRESENT) == 0)
708 continue;
709 if (ci->ci_flags & (CPUF_BSP|CPUF_SP|CPUF_PRIMARY))
710 continue;
711 cpu_boot_secondary(ci);
712 kcpuset_set(cpus, cpu_index(ci));
713 }
714 while (!kcpuset_match(cpus, kcpuset_running))
715 ;
716 kcpuset_destroy(cpus);
717
718 x86_mp_online = true;
719
720 /* Now that we know about the TSC, attach the timecounter. */
721 tsc_tc_init();
722
723 /* Enable zeroing of pages in the idle loop if we have SSE2. */
724 vm_page_zero_enable = ((cpu_feature[0] & CPUID_SSE2) != 0);
725}
726#endif

728static void

756void
757cpu_start_secondary(struct cpu_info *ci)
758{
759 u_long psl;
760 int i;
761
762#if NLAPIC > 0
763 paddr_t mp_pdirpa;
764 mp_pdirpa = pmap_init_tmp_pgtbl(mp_trampoline_paddr);
765 cpu_copy_trampoline(mp_pdirpa);
766#endif
767
768 atomic_or_32(&ci->ci_flags, CPUF_AP);
769 ci->ci_curlwp = ci->ci_data.cpu_idlelwp;
770 if (CPU_STARTUP(ci, mp_trampoline_paddr) != 0) {
771 return;
772 }
773
774 /*
775 * Wait for it to become ready. Setting cpu_starting opens the
776 * initial gate and allows the AP to start soft initialization.
777 */
778 KASSERT(cpu_starting == NULL);
779 cpu_starting = ci;
780 for (i = 100000; (!(ci->ci_flags & CPUF_PRESENT)) && i > 0; i--) {
781 x86_delay(10);
782 }
783
784 if ((ci->ci_flags & CPUF_PRESENT) == 0) {
785 aprint_error_dev(ci->ci_dev, "failed to become ready\n");
786#if defined(MPDEBUG) && defined(DDB)
787 printf("dropping into debugger; continue from here to resume boot\n");
788 Debugger();
789#endif
790 } else {
791 /*
792 * Synchronize time stamp counters. Invalidate cache and do
793 * twice (in tsc_sync_bp) to minimize possible cache effects.
794 * Disable interrupts to try and rule out any external
795 * interference.
796 */
797 psl = x86_read_psl();
798 x86_disable_intr();
799 wbinvd();
800 tsc_sync_bp(ci);
801 x86_write_psl(psl);
802 }
803
804 CPU_START_CLEANUP(ci);
805 cpu_starting = NULL;
806}

808void
809cpu_boot_secondary(struct cpu_info *ci)
810{
811 int64_t drift;
812 u_long psl;
813 int i;
814
815 atomic_or_32(&ci->ci_flags, CPUF_GO);
816 for (i = 100000; (!(ci->ci_flags & CPUF_RUNNING)) && i > 0; i--) {
817 x86_delay(10);
818 }
819 if ((ci->ci_flags & CPUF_RUNNING) == 0) {
820 aprint_error_dev(ci->ci_dev, "failed to start\n");
821#if defined(MPDEBUG) && defined(DDB)
822 printf("dropping into debugger; continue from here to resume boot\n");
823 Debugger();
824#endif
825 } else {
826 /* Synchronize TSC again, check for drift. */
827 drift = ci->ci_data.cpu_cc_skew;
828 psl = x86_read_psl();
829 x86_disable_intr();
830 wbinvd();
831 tsc_sync_bp(ci);
832 x86_write_psl(psl);
833 drift -= ci->ci_data.cpu_cc_skew;
834 aprint_debug_dev(ci->ci_dev, "TSC skew=%lld drift=%lld\n",
835 (long long)ci->ci_data.cpu_cc_skew, (long long)drift);
836 tsc_sync_drift(drift);
837 }
838}

840/*
841 * The CPU ends up here when it's ready to run.
842 * This is called from code in mptramp.s; at this point, we are running
843 * in the idle pcb/idle stack of the new CPU. When this function returns,
844 * this processor will enter the idle loop and start looking for work.
845 */
846void
847cpu_hatch(void *v)
848{
849 struct cpu_info *ci = (struct cpu_info *)v;
850 struct pcb *pcb;
851 int s, i;
852
853 /* ------------------------------------------------------------- */
854
855 /*
856 * This section of code must be compiled with SSP disabled, to
857 * prevent a race against cpu0. See sys/conf/ssp.mk.
858 */
859
860 cpu_init_msrs(ci, true);
861 cpu_probe(ci);
862 cpu_speculation_init(ci);
863
864 ci->ci_data.cpu_cc_freq = cpu_info_primary.ci_data.cpu_cc_freq;
865 /* cpu_get_tsc_freq(ci); */
866
867 KDASSERT((ci->ci_flags & CPUF_PRESENT) == 0);
868
869 /*
870 * Synchronize the TSC for the first time. Note that interrupts are
871 * off at this point.
872 */
873 wbinvd();
874 atomic_or_32(&ci->ci_flags, CPUF_PRESENT);
875 tsc_sync_ap(ci);
876
877 /* ------------------------------------------------------------- */
878
879 /*
880 * Wait to be brought online.
881 *
882 * Use MONITOR/MWAIT if available. These instructions put the CPU in
883 * a low consumption mode (C-state), and if the TSC is not invariant,
884 * this causes the TSC to drift. We want this to happen, so that we
885 * can later detect (in tsc_tc_init) any abnormal drift with invariant
886 * TSCs. That's just for safety; by definition such drifts should
887 * never occur with invariant TSCs.
888 *
889 * If not available, try PAUSE. We'd like to use HLT, but we have
890 * interrupts off.
891 */
892 while ((ci->ci_flags & CPUF_GO) == 0) {
893 if ((cpu_feature[1] & CPUID2_MONITOR) != 0) {
894 x86_monitor(&ci->ci_flags, 0, 0);
895 if ((ci->ci_flags & CPUF_GO) != 0) {
896 continue;
897 }
898 x86_mwait(0, 0);
899 } else {
900 /*
901 * XXX The loop repetition count could be a lot higher, but
902 * XXX currently qemu emulator takes a _very_long_time_ to
903 * XXX execute the pause instruction. So for now, use a low
904 * XXX value to allow the cpu to hatch before timing out.
905 */
906 for (i = 50; i != 0; i--) {
907 x86_pause();
908 }
909 }
910 }
911
912 /* Because the text may have been patched in x86_patch(). */
913 wbinvd();
914 x86_flush();
915 tlbflushg();
916
917 KASSERT((ci->ci_flags & CPUF_RUNNING) == 0);
918
919#ifdef PAE
920 pd_entry_t * l3_pd = ci->ci_pae_l3_pdir;
921 for (i = 0 ; i < PDP_SIZE; i++) {
922 l3_pd[i] = pmap_kernel()->pm_pdirpa[i] | PTE_P;
923 }
924 lcr3(ci->ci_pae_l3_pdirpa);
925#else
926 lcr3(pmap_pdirpa(pmap_kernel(), 0));
927#endif
928
929 pcb = lwp_getpcb(curlwp);
930 pcb->pcb_cr3 = rcr3();
931 pcb = lwp_getpcb(ci->ci_data.cpu_idlelwp);
932 lcr0(pcb->pcb_cr0);
933
934 cpu_init_idt();
935 gdt_init_cpu(ci);
936#if NLAPIC > 0
937 lapic_enable();
938 lapic_set_lvt();
939 lapic_initclocks();
940#endif
941
942 fpuinit(ci);
943 lldt(GSYSSEL(GLDT_SEL, SEL_KPL));
944 ltr(ci->ci_tss_sel);
945
946 /*
947 * cpu_init will re-synchronize the TSC, and will detect any abnormal
948 * drift that would have been caused by the use of MONITOR/MWAIT
949 * above.
950 */
951 cpu_init(ci);
952 cpu_get_tsc_freq(ci);
953
954 s = splhigh();
955#if NLAPIC > 0
956 lapic_write_tpri(0);
957#endif
958 x86_enable_intr();
959 splx(s);
960 x86_errata();
961
962 aprint_debug_dev(ci->ci_dev, "running\n");
963
964 idle_loop(NULL);
965 KASSERT(false);
966}
967#endif

998static void
999cpu_copy_trampoline(paddr_t pdir_pa)
1000{
1001 extern uint32_t nox_flag;
1002 extern u_char cpu_spinup_trampoline[];
1003 extern u_char cpu_spinup_trampoline_end[];
1004 vaddr_t mp_trampoline_vaddr;
1005 struct {
1006 uint32_t large;
1007 uint32_t nox;
1008 uint32_t pdir;
1009 } smp_data;
1010 CTASSERT(sizeof(smp_data) == 3 * 4);
1011
1012 smp_data.large = (pmap_largepages != 0);
1013 smp_data.nox = nox_flag;
1014 smp_data.pdir = (uint32_t)(pdir_pa & 0xFFFFFFFF);
1015
1016 /* Enter the physical address */
1017 mp_trampoline_vaddr = uvm_km_alloc(kernel_map, PAGE_SIZE, 0,
1018 UVM_KMF_VAONLY);
1019 pmap_kenter_pa(mp_trampoline_vaddr, mp_trampoline_paddr,
1020 VM_PROT_READ | VM_PROT_WRITE, 0);
1021 pmap_update(pmap_kernel());
1022
1023 /* Copy boot code */
1024 memcpy((void *)mp_trampoline_vaddr,
1025 cpu_spinup_trampoline,
1026 cpu_spinup_trampoline_end - cpu_spinup_trampoline);
1027
1028 /* Copy smp_data at the end */
1029 memcpy((void *)(mp_trampoline_vaddr + PAGE_SIZE - sizeof(smp_data)),
1030 &smp_data, sizeof(smp_data));
1031
1032 pmap_kremove(mp_trampoline_vaddr, PAGE_SIZE);
1033 pmap_update(pmap_kernel());
1034 uvm_km_free(kernel_map, mp_trampoline_vaddr, PAGE_SIZE, UVM_KMF_VAONLY);
1035}

1038int
1039mp_cpu_start(struct cpu_info *ci, paddr_t target)
1040{
1041 int error;
1042
1043 /*
1044 * Bootstrap code must be addressable in real mode
1045 * and it must be page aligned.
1046 */
1047 KASSERT(target < 0x10000 && target % PAGE_SIZE == 0);
1048
1049 /*
1050 * "The BSP must initialize CMOS shutdown code to 0Ah ..."
1051 */
1052
1053 outb(IO_RTC, NVRAM_RESET);
1054 outb(IO_RTC+1, NVRAM_RESET_JUMP);
1055
1056#if NLAPIC > 0
1057 /*
1058 * "and the warm reset vector (DWORD based at 40:67) to point
1059 * to the AP startup code ..."
1060 */
1061 unsigned short dwordptr[2];
1062 dwordptr[0] = 0;
1063 dwordptr[1] = target >> 4;
1064
1065 memcpy((uint8_t *)cmos_data_mapping + 0x467, dwordptr, 4);
1066#endif
1067
1068 if ((cpu_feature[0] & CPUID_APIC) == 0) {
1069 aprint_error("mp_cpu_start: CPU does not have APIC\n");
1070 return ENODEV;
1071 }
1072
1073 /*
1074 * ... prior to executing the following sequence:". We'll also add in
1075 * local cache flush, in case the BIOS has left the AP with its cache
1076 * disabled. It may not be able to cope with MP coherency.
1077 */
1078 wbinvd();
1079
1080 if (ci->ci_flags & CPUF_AP) {
1081 error = x86_ipi_init(ci->ci_cpuid);
1082 if (error != 0) {
1083 aprint_error_dev(ci->ci_dev, "%s: IPI not taken (1)\n",
1084 __func__);
1085 return error;
1086 }
1087 x86_delay(10000);
1088
1089 error = x86_ipi_startup(ci->ci_cpuid, target / PAGE_SIZE);
1090 if (error != 0) {
1091 aprint_error_dev(ci->ci_dev, "%s: IPI not taken (2)\n",
1092 __func__);
1093 return error;
1094 }
1095 x86_delay(200);
1096
1097 error = x86_ipi_startup(ci->ci_cpuid, target / PAGE_SIZE);
1098 if (error != 0) {
1099 aprint_error_dev(ci->ci_dev, "%s: IPI not taken (3)\n",
1100 __func__);
1101 return error;
1102 }
1103 x86_delay(200);
1104 }
1105
1106 return 0;
1107}

1126void
1127cpu_init_msrs(struct cpu_info *ci, bool full)
1128{
1129#ifdef __x86_64__
1130 wrmsr(MSR_STAR,
1131 ((uint64_t)GSEL(GCODE_SEL, SEL_KPL) << 32) |
1132 ((uint64_t)LSEL(LSYSRETBASE_SEL, SEL_UPL) << 48));
1133 wrmsr(MSR_LSTAR, (uint64_t)Xsyscall);
1134 wrmsr(MSR_CSTAR, (uint64_t)Xsyscall32);
1135 wrmsr(MSR_SFMASK, PSL_NT|PSL_T|PSL_I|PSL_C|PSL_D|PSL_AC);
1136
1137#ifdef SVS
1138 if (svs_enabled)
1139 wrmsr(MSR_LSTAR, (uint64_t)Xsyscall_svs);
1140#endif
1141
1142 if (full) {
1143 wrmsr(MSR_FSBASE, 0);
1144 wrmsr(MSR_GSBASE, (uint64_t)ci);
1145 wrmsr(MSR_KERNELGSBASE, 0);
1146 }
1147#endif /* __x86_64__ */
1148
1149 if (cpu_feature[2] & CPUID_NOX)
1150 wrmsr(MSR_EFER, rdmsr(MSR_EFER) | EFER_NXE);
1151}

1166{
1167 struct cpu_softc *sc = device_private(dv);
1168 struct cpu_info *ci = sc->sc_info;

1170

1172

1174 return true;

1177 return true;

1182 mutex_enter(&cpu_lock);

1185
1186 if (err != 0)
1187 return false;
1188 }
1189
1190 return true;

1296/*
1297 * Loads pmap for the current CPU.
1298 */
1299void
1300cpu_load_pmap(struct pmap *pmap, struct pmap *oldpmap)
1301{
1302#ifdef SVS
1303 if (svs_enabled) {
1304 svs_pdir_switch(pmap);
1305 }
1306#endif
1307
1308#ifdef PAE
1309 struct cpu_info *ci = curcpu();
1310 bool interrupts_enabled;
1311 pd_entry_t *l3_pd = ci->ci_pae_l3_pdir;
1312 int i;
1313
1314 /*
1315 * disable interrupts to block TLB shootdowns, which can reload cr3.
1316 * while this doesn't block NMIs, it's probably ok as NMIs unlikely
1317 * reload cr3.
1318 */
1319 interrupts_enabled = (x86_read_flags() & PSL_I) != 0;
1320 if (interrupts_enabled)
1321 x86_disable_intr();
1322
1323 for (i = 0 ; i < PDP_SIZE; i++) {
1324 l3_pd[i] = pmap->pm_pdirpa[i] | PTE_P;
1325 }
1326
1327 if (interrupts_enabled)
1328 x86_enable_intr();
1329 tlbflush();
1330#else
1331 lcr3(pmap_pdirpa(pmap, 0));
1332#endif
1333}
