49static enum {
50 CPU_RNG_NONE = 0,
51 CPU_RNG_RDRAND,
52 CPU_RNG_RDSEED,
53 CPU_RNG_VIA
54} cpu_rng_mode __read_mostly = CPU_RNG_NONE;
55
56bool
57cpu_rng_init(void)
58{
59
60 if (cpu_feature[5] & CPUID_SEF_RDSEED) {
61 cpu_rng_mode = CPU_RNG_RDSEED;
62 aprint_normal("cpu_rng: RDSEED\n");
63 return true;
64 } else if (cpu_feature[1] & CPUID2_RDRAND) {
65 cpu_rng_mode = CPU_RNG_RDRAND;
66 aprint_normal("cpu_rng: RDRAND\n");
67 return true;
68 } else if (cpu_feature[4] & CPUID_VIA_HAS_RNG) {
69 cpu_rng_mode = CPU_RNG_VIA;
70 aprint_normal("cpu_rng: VIA\n");
71 return true;
72 }
73 return false;
74}

76static size_t
77cpu_rng_rdrand(cpu_rng_t *out)
78{
79 uint8_t rndsts;
80
81#ifdef __i386__
82 uint32_t lo, hi;
83
84 __asm __volatile("rdrand %0; setc %1" : "=r"(lo), "=qm"(rndsts));
85 if (rndsts != 1)
86 return 0;
87 __asm __volatile("rdrand %0; setc %1" : "=r"(hi), "=qm"(rndsts));
88
89 *out = (uint64_t)lo | ((uint64_t)hi << 32);
90 explicit_memset(&lo, 0, sizeof(lo));
91 explicit_memset(&hi, 0, sizeof(hi));
92 if (rndsts != 1)
93 return sizeof(lo) * NBBY;
94#else
95 __asm __volatile("rdrand %0; setc %1" : "=r"(*out), "=qm"(rndsts));
96 if (rndsts != 1)
97 return 0;
98#endif
99 return sizeof(*out) * NBBY;
100}

102static size_t
103cpu_rng_rdseed(cpu_rng_t *out)
104{
105 uint8_t rndsts;
106
107#ifdef __i386__
108 uint32_t lo, hi;
109
110 __asm __volatile("rdseed %0; setc %1" : "=r"(lo), "=qm"(rndsts));
111 if (rndsts != 1)
112 goto exhausted;
113 __asm __volatile("rdseed %0; setc %1" : "=r"(hi), "=qm"(rndsts));
114 if (rndsts != 1)
115 goto exhausted;
116
117 *out = (uint64_t)lo | ((uint64_t)hi << 32);
118 explicit_memset(&lo, 0, sizeof(lo));
119 explicit_memset(&hi, 0, sizeof(hi));
120#else
121 __asm __volatile("rdseed %0; setc %1" : "=r"(*out), "=qm"(rndsts));
122#endif
123 if (rndsts != 1)
124 goto exhausted;
125
126 return sizeof(*out) * NBBY;
127
128 /*
129 * Userspace could have exhausted RDSEED, but the
130 * CPU-internal generator feeding RDRAND is guaranteed
131 * to be seeded even in this case.
132 */
133exhausted:
134 return cpu_rng_rdrand(out);
135}

137static size_t
138cpu_rng_via(cpu_rng_t *out)
139{
140 uint32_t creg0, rndsts;
141
142 /*
143 * Sadly, we have to monkey with the coprocessor enable and fault
144 * registers, which are really for the FPU, in order to read
145 * from the RNG.
146 *
147 * Don't remove CR0_TS from the call below -- comments in the Linux
148 * driver indicate that the xstorerng instruction can generate
149 * spurious DNA faults though no FPU or SIMD state is changed
150 * even if such a fault is generated.
151 *
152 * XXX can this really happen if we don't use "rep xstorrng"?
153 *
154 */
155 kpreempt_disable();
156 x86_disable_intr();
157 creg0 = rcr0();
158 lcr0(creg0 & ~(CR0_EM|CR0_TS)); /* Permit access to SIMD/FPU path */
159 /*
160 * The VIA RNG has an output queue of 8-byte values. Read one.
161 * This is atomic, so if the FPU were already enabled, we could skip
162 * all the preemption and interrupt frobbing. If we had bread,
163 * we could have a ham sandwich, if we had any ham.
164 */
165 __asm __volatile("xstorerng"
166 : "=a" (rndsts), "+D" (out) : "d" (0) : "memory");
167 /* Put CR0 back how it was */
168 lcr0(creg0);
169 x86_enable_intr();
170 kpreempt_enable();
171
172 /*
173 * The Cryptography Research paper on the VIA RNG estimates
174 * 0.75 bits of entropy per output bit and advises users to
175 * be "even more conservative".
176 */
177 return rndsts & 0xf ? 0 : sizeof(cpu_rng_t) * NBBY / 2;
178}
179
180size_t
181cpu_rng(cpu_rng_t *out)
182{
183
184 switch (cpu_rng_mode) {
185 case CPU_RNG_NONE:
186 return 0;
187 case CPU_RNG_RDSEED:
188 return cpu_rng_rdseed(out);
189 case CPU_RNG_RDRAND:
190 return cpu_rng_rdrand(out);
191 case CPU_RNG_VIA:
192 return cpu_rng_via(out);
193 default:
194 panic("cpu_rng: unknown mode %d", (int)cpu_rng_mode);
195 }
196}

200static uint64_t earlyrng_state;
201
202/*
203 * Small PRNG, that can be used very early. The only requirement is that
204 * cpu_probe got called before.
205 */
206void __noasan
207cpu_earlyrng(void *out, size_t sz)
208{
209 uint8_t digest[SHA512_DIGEST_LENGTH];
210 SHA512_CTX ctx;
211 cpu_rng_t buf[8];
212 uint64_t val;
213 int i;
214
215 bool has_rdseed = (cpu_feature[5] & CPUID_SEF_RDSEED) != 0;
216 bool has_rdrand = (cpu_feature[1] & CPUID2_RDRAND) != 0;
217
218 KASSERT(sz + sizeof(uint64_t) <= SHA512_DIGEST_LENGTH);
219
220 SHA512_Init(&ctx);
221
222 SHA512_Update(&ctx, (uint8_t *)&earlyrng_state, sizeof(earlyrng_state));
223 if (has_rdseed) {
224 for (i = 0; i < 8; i++) {
225 if (cpu_rng_rdseed(&buf[i]) == 0) {
226 break;
227 }
228 }
229 SHA512_Update(&ctx, (uint8_t *)buf, i * sizeof(cpu_rng_t));
230 } else if (has_rdrand) {
231 for (i = 0; i < 8; i++) {
232 if (cpu_rng_rdrand(&buf[i]) == 0) {
233 break;
234 }
235 }
236 SHA512_Update(&ctx, (uint8_t *)buf, i * sizeof(cpu_rng_t));
237 }
238 val = rdtsc();
239 SHA512_Update(&ctx, (uint8_t *)&val, sizeof(val));
240
241 SHA512_Final(digest, &ctx);
242
243 memcpy(out, digest, sz);
244 memcpy(&earlyrng_state, &digest[sz], sizeof(earlyrng_state));
245}
