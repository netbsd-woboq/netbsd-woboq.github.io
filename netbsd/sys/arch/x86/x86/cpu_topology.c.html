30/*
31 * x86 CPU topology detection.
32 *
33 * References:
34 * - 53668.pdf (7.10.2), 276613.pdf
35 * - 31116.pdf, 41256.pdf, 25481.pdf
36 */
37
38#include <sys/cdefs.h>
39__KERNEL_RCSID(0, "$NetBSD: cpu_topology.c,v 1.14 2018/11/21 10:34:53 msaitoh Exp $");
40
41#include <sys/param.h>
42#include <sys/bitops.h>
43
44#include <machine/specialreg.h>
45#include <machine/cpu.h>
46
47#include <x86/cpufunc.h>
48#include <x86/cputypes.h>
49#include <x86/cpuvar.h>
50
51void
52x86_cpu_topology(struct cpu_info *ci)
53{
54 u_int lp_max; /* Logical processors per package (node) */
55 u_int core_max; /* Core per package */
56 int n, cpu_family, apic_id, smt_bits, core_bits = 0;
57 uint32_t descs[4];
58
59 apic_id = ci->ci_initapicid;
60 cpu_family = CPUID_TO_FAMILY(ci->ci_signature);
61
62 /* Initial values. */
63 ci->ci_package_id = apic_id;
64 ci->ci_core_id = 0;
65 ci->ci_smt_id = 0;
66
67 switch (cpu_vendor) {
68 case CPUVENDOR_INTEL:
69 if (cpu_family < 6)
70 return;
71 break;
72 case CPUVENDOR_AMD:
73 if (cpu_family < 0xf)
74 return;
75 break;
76 default:
77 return;
78 }
79
80 /* Check for HTT support. See notes below regarding AMD. */
81 if ((ci->ci_feat_val[0] & CPUID_HTT) != 0) {
82 /* Maximum number of LPs sharing a cache (ebx[23:16]). */
83 x86_cpuid(1, descs);
84 lp_max = __SHIFTOUT(descs[1], CPUID_HTT_CORES);
85 } else {
86 lp_max = 1;
87 }
88
89 switch (cpu_vendor) {
90 case CPUVENDOR_INTEL:
91 /* Check for leaf 4 support. */
92 if (ci->ci_max_cpuid >= 4) {
93 /* Maximum number of Cores per package (eax[31:26]). */
94 x86_cpuid2(4, 0, descs);
95 core_max = __SHIFTOUT(descs[0], CPUID_DCP_CORE_P_PKG)
96 + 1;
97 } else {
98 core_max = 1;
99 }
100 break;
101 case CPUVENDOR_AMD:
102 /* In a case of AMD, HTT flag means CMP support. */
103 if ((ci->ci_feat_val[0] & CPUID_HTT) == 0) {
104 core_max = 1;
105 break;
106 }
107 /* Legacy Method, LPs represent Cores. */
108 if (cpu_family < 0x10 || ci->ci_max_ext_cpuid < 0x80000008) {
109 core_max = lp_max;
110 break;
111 }
112
113 /* Number of Cores (NC) per package (ecx[7:0]). */
114 x86_cpuid(0x80000008, descs);
115 core_max = (descs[2] & 0xff) + 1;
116 /* Amount of bits representing Core ID (ecx[15:12]). */
117 n = (descs[2] >> 12) & 0x0f;
118 if (n != 0) {
119 /*
120 * Extended Method.
121 * core_max = 2 ^ n (power of two)
122 */
123 core_bits = n;
124 }
125 break;
126 default:
127 core_max = 1;
128 }
129
130 KASSERT(lp_max >= core_max);
131 smt_bits = ilog2((lp_max / core_max) - 1) + 1;
132 if (core_bits == 0) {
133 core_bits = ilog2(core_max - 1) + 1;
134 }
135
136 /*
137 * Family 0xf and 0x10 processors may have different structure of
138 * APIC ID. Detect that via special MSR register and move the bits,
139 * if necessary (ref: InitApicIdCpuIdLo).
140 */
141 if (cpu_vendor == CPUVENDOR_AMD && cpu_family < 0x11) { /* XXX */
142 const uint64_t reg = rdmsr(MSR_NB_CFG);
143 if ((reg & NB_CFG_INITAPICCPUIDLO) == 0) {
144 /*
145 * 0xf: { CoreId, NodeId[2:0] }
146 * 0x10: { CoreId[1:0], 000b, NodeId[2:0] }
147 */
148 const u_int node_id = apic_id & __BITS(0, 2);
149 apic_id = (cpu_family == 0xf) ?
150 (apic_id >> core_bits) | (node_id << core_bits) :
151 (apic_id >> 5) | (node_id << 2);
152 }
153 }
154
155 /* Family 0x17 supports SMT */
156 if (cpu_vendor == CPUVENDOR_AMD && cpu_family == 0x17) { /* XXX */
157 x86_cpuid(0x8000001e, descs);
158 const u_int threads = ((descs[1] >> 8) & 0xff) + 1;
159
160 KASSERT(smt_bits == 0 && smt_bits <= core_bits);
161 smt_bits = ilog2(threads);
162 core_bits -= smt_bits;
163 }
164
165 if (smt_bits + core_bits) {
166 if (smt_bits + core_bits < sizeof(apic_id) * NBBY)
167 ci->ci_package_id = apic_id >> (smt_bits + core_bits);
168 else
169 ci->ci_package_id = 0;
170 }
171 if (core_bits) {
172 u_int core_mask = __BITS(smt_bits, smt_bits + core_bits - 1);
173 ci->ci_core_id = __SHIFTOUT(apic_id, core_mask);
174 }
175 if (smt_bits) {
176 u_int smt_mask = __BITS(0, smt_bits - 1);
177 ci->ci_smt_id = __SHIFTOUT(apic_id, smt_mask);
178 }
179}
