32/*
33 * Interface to the debugger for virtual memory read/write.
34 * This file is shared by DDB and KGDB, and must work even
35 * when only KGDB is included (thus no db_printf calls).
36 *
37 * To write in the text segment, we have to first make
38 * the page writable, do the write, then restore the PTE.
39 * For writes outside the text segment, and all reads,
40 * just do the access -- if it causes a fault, the debugger
41 * will recover with a longjmp to an appropriate place.
42 *
43 * ALERT! If you want to access device registers with a
44 * specific size, then the read/write functions have to
45 * make sure to do the correct sized pointer access.
46 *
47 * Modified for i386 from hp300 version by
48 * Jason R. Thorpe <thorpej@zembu.com>.
49 *
50 * Basic copy to amd64 by fvdl.
51 *
52 * i386 and amd64 merge by jym.
53 */
54

67static int
68db_validate_address(vaddr_t addr)
69{
70 struct proc *p = curproc;
71 struct pmap *pmap;
72
73 if (!p || !p->p_vmspace || !p->p_vmspace->vm_map.pmap ||
74 addr >= VM_MIN_KERNEL_ADDRESS)
75 pmap = pmap_kernel();
76 else
77 pmap = p->p_vmspace->vm_map.pmap;
78
79 return (pmap_extract(pmap, addr, NULL) == false);
80}

82/*
83 * Read bytes from kernel address space for debugger.
84 */
85void
86db_read_bytes(vaddr_t addr, size_t size, char *data)
87{
88 char *src;
89
90 src = (char *)addr;
91
92 if (db_validate_address((vaddr_t)src)) {
93#ifdef DDB
94 db_printf("address %p is invalid\n", src);
95#endif
96 return;
97 }
98
99 if (size == 8) {
100 *((long *)data) = *((long *)src);
101 return;
102 }
103
104 if (size == 4) {
105 *((int *)data) = *((int *)src);
106 return;
107 }
108
109 if (size == 2) {
110 *((short *)data) = *((short *)src);
111 return;
112 }
113
114 while (size-- > 0) {
115 if (db_validate_address((vaddr_t)src)) {
116#ifdef DDB
117 db_printf("address %p is invalid\n", src);
118#endif
119 return;
120 }
121
122 *data++ = *src++;
123 }
124}

126/*
127 * Write bytes somewhere in the kernel text. Make the text
128 * pages writable temporarily.
129 */
130static void
131db_write_text(vaddr_t addr, size_t size, const char *data)
132{
133 pt_entry_t *ppte, pte;
134 size_t limit;
135 char *dst;
136
137 if (size == 0)
138 return;
139
140 dst = (char *)addr;
141
142 do {
143 addr = (vaddr_t)dst;
144 /*
145 * Get the PTE for the page.
146 */
147 ppte = kvtopte(addr);
148 pte = *ppte;
149
150 if ((pte & PTE_P) == 0) {
151#ifdef DDB
152 db_printf(" address %p not a valid page\n", dst);
153#endif
154 return;
155 }
156
157 /*
158 * Compute number of bytes that can be written
159 * with this mapping and subtract it from the
160 * total size.
161 */
162 if (pte & PTE_PS)
163 limit = NBPD_L2 - (addr & (NBPD_L2 - 1));
164 else
165 limit = PAGE_SIZE - (addr & PGOFSET);
166 if (limit > size)
167 limit = size;
168 size -= limit;
169
170 /*
171 * Make the kernel text page writable.
172 */
173 pmap_pte_setbits(ppte, PTE_W);
174 pmap_update_pg(addr);
175
176 /*
177 * MULTIPROCESSOR: no shootdown required as the PTE continues to
178 * map the same page and other CPUs do not need write access.
179 */
180
181 /*
182 * Page is now writable. Do as much access as we
183 * can in this page.
184 */
185 for (; limit > 0; limit--)
186 *dst++ = *data++;
187
188 /*
189 * Turn the page back to read-only.
190 */
191 pmap_pte_clearbits(ppte, PTE_W);
192 pmap_update_pg(addr);
193
194 /*
195 * MULTIPROCESSOR: no shootdown required as all other CPUs
196 * should be in CPUF_PAUSE state and will not cache the PTE
197 * with the write access set.
198 */
199 } while (size != 0);
200}

202/*
203 * Write bytes to kernel address space for debugger.
204 */
205void
206db_write_bytes(vaddr_t addr, size_t size, const char *data)
207{
208 extern struct bootspace bootspace;
209 char *dst;
210 size_t i;
211
212 dst = (char *)addr;
213
214 /* If any part is in kernel text or rodata, use db_write_text() */
215 for (i = 0; i < BTSPACE_NSEGS; i++) {
216 if (bootspace.segs[i].type != BTSEG_TEXT &&
217 bootspace.segs[i].type != BTSEG_RODATA) {
218 continue;
219 }
220 if (addr >= bootspace.segs[i].va &&
221 addr < (bootspace.segs[i].va + bootspace.segs[i].sz)) {
222 db_write_text(addr, size, data);
223 return;
224 }
225 }
226
227 dst = (char *)addr;
228
229 if (size == 8) {
230 *((long *)dst) = *((const long *)data);
231 return;
232 }
233
234 if (size == 4) {
235 *((int *)dst) = *((const int *)data);
236 return;
237 }
238
239 if (size == 2) {
240 *((short *)dst) = *((const short *)data);
241 return;
242 }
243
244 while (size-- > 0)
245 *dst++ = *data++;
246}
