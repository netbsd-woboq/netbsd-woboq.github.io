55int
56db_x86_regop(const struct db_variable *vp, db_expr_t *val, int opcode)
57{
58 db_expr_t *regaddr =
59 (db_expr_t *)(((uint8_t *)DDB_REGS) + ((size_t)vp->valuep));
60
61 switch (opcode) {
62 case DB_VAR_GET:
63 *val = *regaddr;
64 break;
65 case DB_VAR_SET:
66 *regaddr = *val;
67 break;
68 default:
69 db_printf("db_x86_regop: unknown op %d", opcode);
70 db_error(NULL);
71 }
72 return 0;
73}

103 frame = (long *)ddb_regs.tf_bp; \
104 callpc = (db_addr_t)ddb_regs.tf_ip; \

107void
108db_stack_trace_print(db_expr_t addr, bool have_addr, db_expr_t count,
109 const char *modif, void (*pr)(const char *, ...))
110{
111 long *frame, *lastframe;
112 long *retaddr, *arg0;
113 long *argp;
114 db_addr_t callpc;
115 int is_trap;
116 bool kernel_only = true;
117 bool trace_thread = false;
118 bool lwpaddr = false;
119
120#if 0
121 if (!db_trace_symbols_found)
122 db_find_trace_symbols();
123#endif
124
125 {
126 const char *cp = modif;
127 char c;
128
129 while ((c = *cp++) != 0) {
130 if (c == 'a') {
131 lwpaddr = true;
132 trace_thread = true;
133 }
134 if (c == 't')
135 trace_thread = true;
136 if (c == 'u')
137 kernel_only = false;
138 }
139 }
140
141 if (have_addr && trace_thread) {
142 struct pcb *pcb;
143 proc_t p;
144 lwp_t l;
145
146 if (lwpaddr) {
147 db_read_bytes(addr, sizeof(l),
148 (char *)&l);
149 db_read_bytes((db_addr_t)l.l_proc,
150 sizeof(p), (char *)&p);
151 (*pr)("trace: pid %d ", p.p_pid);
152 } else {
153 proc_t *pp;
154
155 (*pr)("trace: pid %d ", (int)addr);
156 if ((pp = db_proc_find((pid_t)addr)) == 0) {
157 (*pr)("not found\n");
158 return;
159 }
160 db_read_bytes((db_addr_t)pp, sizeof(p), (char *)&p);
161 addr = (db_addr_t)p.p_lwps.lh_first;
162 db_read_bytes(addr, sizeof(l), (char *)&l);
163 }
164 (*pr)("lid %d ", l.l_lid);
165 pcb = lwp_getpcb(&l);
166#ifdef _KERNEL
167 if (l.l_proc == curproc && (lwp_t *)addr == curlwp)
168 set_frame_callpc();
169 else
170#endif
171 {
172 db_read_bytes((db_addr_t)&pcb->pcb_bp,
173 sizeof(frame), (char *)&frame);
174 db_read_bytes((db_addr_t)(frame + 1),
175 sizeof(callpc), (char *)&callpc);
176 db_read_bytes((db_addr_t)frame,
177 sizeof(frame), (char *)&frame);
178 }
179 (*pr)("at %p\n", frame);
180 } else if (have_addr) {
181 frame = (long *)addr;
182 db_read_bytes((db_addr_t)(frame + 1),
183 sizeof(callpc), (char *)&callpc);
184 db_read_bytes((db_addr_t)frame,
185 sizeof(frame), (char *)&frame);
186 } else {
187 set_frame_callpc();
188 }
189
190 retaddr = frame + 1;
191 arg0 = frame + 2;
192
193 lastframe = NULL;
194 while (count && frame != 0) {
195 int narg;
196 const char *name;
197 db_expr_t offset;
198 db_sym_t sym;
199 char *argnames[MAXNARG], **argnp = NULL;
200 db_addr_t lastcallpc;
201
202 name = "?";
203 is_trap = NONE;
204 offset = 0;
205 sym = db_frame_info(frame, callpc, &name, &offset, &is_trap,
206 &narg);
207
208 if (lastframe == NULL && sym == (db_sym_t)0 && callpc != 0) {
209 /* Symbol not found, peek at code */
210 u_long instr = db_get_value(callpc, 4, false);
211
212 offset = 1;
213 if (
214#ifdef __x86_64__
215 instr == 0xe5894855 ||
216 /* enter: pushq %rbp, movq %rsp, %rbp */
217 (instr & 0x00ffffff) == 0x0048e589
218 /* enter+1: movq %rsp, %rbp */)
219#else
220 (instr & 0x00ffffff) == 0x00e58955 ||
221 /* enter: pushl %ebp, movl %esp, %ebp */
222 (instr & 0x0000ffff) == 0x0000e589
223 /* enter+1: movl %esp, %ebp */)
224#endif
225 {
226 offset = 0;
227 }
228 }
229
230 if (is_trap == NONE) {
231 if (db_sym_numargs(sym, &narg, argnames))
232 argnp = argnames;
233 else
234 narg = db_numargs(frame);
235 }
236
237 (*pr)("%s(", name);
238
239 if (lastframe == NULL && offset == 0 && !have_addr) {
240 /*
241 * We have a breakpoint before the frame is set up
242 * Use %[er]sp instead
243 */
244 argp = (long *)&((struct x86_frame *)
245 (ddb_regs.tf_sp-sizeof(long)))->f_arg0;
246 } else {
247 argp = frame + 2;
248 }
249
250 while (narg) {
251 if (argnp)
252 (*pr)("%s=", *argnp++);
253 (*pr)("%lx", db_get_value((long)argp, sizeof(long),
254 false));
255 argp++;
256 if (--narg != 0)
257 (*pr)(",");
258 }
259 (*pr)(") at ");
260 db_printsym(callpc, DB_STGY_PROC, pr);
261 (*pr)("\n");
262
263 if (lastframe == NULL && offset == 0 && !have_addr) {
264 /* Frame really belongs to next callpc */
265 struct x86_frame *fp = (void *)
266 (ddb_regs.tf_sp-sizeof(long));
267 lastframe = (long *)fp;
268 callpc = (db_addr_t)
269 db_get_value((db_addr_t)&fp->f_retaddr,
270 sizeof(long), false);
271
272 continue;
273 }
274
275 lastframe = frame;
276 lastcallpc = callpc;
277 if (!db_nextframe(&frame, &retaddr, &arg0, &callpc,
278 frame + 2, is_trap, pr))
279 break;
280
281 if (INKERNEL((long)frame)) {
282 /* staying in kernel */
283#ifdef __i386__
284 if (!db_intrstack_p(frame) &&
285 db_intrstack_p(lastframe)) {
286 (*pr)("--- switch to interrupt stack ---\n");
287 } else
288#endif
289 if (frame < lastframe ||
290 (frame == lastframe && callpc == lastcallpc)) {
291 (*pr)("Bad frame pointer: %p\n", frame);
292 break;
293 }
294 } else if (INKERNEL((long)lastframe)) {
295 /* switch from user to kernel */
296 if (kernel_only)
297 break; /* kernel stack only */
298 } else {
299 /* in user */
300 if (frame <= lastframe) {
301 (*pr)("Bad user frame pointer: %p\n", frame);
302 break;
303 }
304 }
305 --count;
306 }
307
308 if (count && is_trap != NONE) {
309 db_printsym(callpc, DB_STGY_XTRN, pr);
310 (*pr)(":\n");
311 }
312}
