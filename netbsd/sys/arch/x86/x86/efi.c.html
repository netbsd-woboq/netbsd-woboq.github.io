68/*
69 * Map a physical address (PA) to a newly allocated virtual address (VA).
70 * The VA must be freed using efi_relva().
71 */
72static vaddr_t
73efi_getva(paddr_t pa)
74{
75 vaddr_t va;
76
77#ifdef __HAVE_MM_MD_DIRECT_MAPPED_PHYS
78 if (mm_md_direct_mapped_phys(pa, &va))
79 return va;
80#endif
81
82 /* XXX This code path is not tested. */
83 va = uvm_km_alloc(kernel_map, PAGE_SIZE, 0,
84 UVM_KMF_VAONLY | UVM_KMF_WAITVA);
85 if (va == 0) {
86 aprint_debug("efi: unable to allocate va\n");
87 return 0;
88 }
89 pmap_kenter_pa(va, pa, VM_PROT_READ, 0);
90 pmap_update(pmap_kernel());
91
92 return va;
93}
94
95/*
96 * Free a virtual address (VA) allocated using efi_getva().
97 */
98static void
99efi_relva(paddr_t pa, vaddr_t va)
100{
101
102#ifdef __HAVE_MM_MD_DIRECT_MAPPED_PHYS
103 vaddr_t va0 __diagused;
104 if (mm_md_direct_mapped_phys(pa, &va0)) {
105 KASSERT(va0 == va);
106 return;
107 }
108#endif
109
110 /* XXX This code path is not tested. */
111 pmap_kremove(va, PAGE_SIZE);
112 pmap_update(pmap_kernel());
113 uvm_km_free(kernel_map, va, PAGE_SIZE, UVM_KMF_VAONLY);
114}

116/*
117 * Test if 2 UUIDs matches.
118 */
119bool
120efi_uuideq(const struct uuid * a, const struct uuid * b)
121{
122 return !memcmp(a, b, sizeof(struct uuid));
123}

125/*
126 * Print an UUID in a human-readable manner.
127 */
128void
129efi_aprintuuid(const struct uuid * uuid)
130{
131 int i;
132
133 aprint_debug(" %08" PRIx32 "", uuid->time_low);
134 aprint_debug("-%04" PRIx16 "", uuid->time_mid);
135 aprint_debug("-%04" PRIx16 "", uuid->time_hi_and_version);
136 aprint_debug("-%02" PRIx8 "", uuid->clock_seq_hi_and_reserved);
137 aprint_debug("-%02" PRIx8 "", uuid->clock_seq_low);
138 aprint_debug("-");
139 for (i = 0; i < _UUID_NODE_LEN; i++) {
140 aprint_debug("%02" PRIx8 "", uuid->node[i]);
141 }
142 /* If known, also print the human-readable name */
143 if (efi_uuideq(uuid, &EFI_UUID_ACPI20)) {
144 aprint_debug(" ACPI 2.0");
145 } else if (efi_uuideq(uuid, &EFI_UUID_ACPI10)) {
146 aprint_debug(" ACPI 1.0");
147 } else if (efi_uuideq(uuid, &EFI_UUID_SMBIOS)) {
148 aprint_debug(" SMBIOS");
149 } else if (efi_uuideq(uuid, &EFI_UUID_SMBIOS3)) {
150 aprint_debug(" SMBIOS3");
151 }
152}

154/*
155 * Return the VA of the cfgtbl. Must be freed using efi_relva().
156 */
157struct efi_cfgtbl *
158efi_getcfgtblhead(void)
159{
160 paddr_t pa;
161 vaddr_t va;
162
163 if (efi_cfgtblhead_va != NULL)
164 return efi_cfgtblhead_va;
165
166 if (efi_is32x64) {
167#if defined(__amd64__)
168 struct efi_systbl32 *systbl32 = (void *) efi_systbl_va;
169 pa = systbl32->st_cfgtbl;
170#elif defined(__i386__)
171 struct efi_systbl64 *systbl64 = (void *) efi_systbl_va;
172 if (systbl64->st_cfgtbl & 0xffffffff00000000ULL)
173 return NULL;
174 pa = (paddr_t) systbl64->st_cfgtbl;
175#endif
176 } else
177 pa = (paddr_t)(u_long) efi_systbl_va->st_cfgtbl;
178 aprint_debug("efi: cfgtbl at pa %" PRIxPADDR "\n", pa);
179 va = efi_getva(pa);
180 aprint_debug("efi: cfgtbl mapped at va %" PRIxVADDR "\n", va);
181 efi_cfgtblhead_va = (struct efi_cfgtbl *) va;
182 efi_aprintcfgtbl();
183
184 return efi_cfgtblhead_va;
185}

187/*
188 * Print the config tables.
189 */
190void
191efi_aprintcfgtbl(void)
192{
193 struct efi_cfgtbl *ct;
194 unsigned long count;
195
196 if (efi_is32x64) {
197#if defined(__amd64__)
198 struct efi_systbl32 *systbl32 = (void *) efi_systbl_va;
199 struct efi_cfgtbl32 *ct32 = (void *) efi_cfgtblhead_va;
200
201 count = systbl32->st_entries;
202 aprint_debug("efi: %lu cfgtbl entries:\n", count);
203 for (; count; count--, ct32++) {
204 aprint_debug("efi: %08" PRIx32, ct32->ct_data);
205 efi_aprintuuid(&ct32->ct_uuid);
206 aprint_debug("\n");
207 }
208#elif defined(__i386__)
209 struct efi_systbl64 *systbl64 = (void *) efi_systbl_va;
210 struct efi_cfgtbl64 *ct64 = (void *) efi_cfgtblhead_va;
211 uint64_t count64 = systbl64->st_entries;
212
213 aprint_debug("efi: %" PRIu64 " cfgtbl entries:\n", count64);
214 for (; count64; count64--, ct64++) {
215 aprint_debug("efi: %016" PRIx64, ct64->ct_data);
216 efi_aprintuuid(&ct64->ct_uuid);
217 aprint_debug("\n");
218 }
219#endif
220 return;
221 }
222
223 ct = efi_cfgtblhead_va;
224 count = efi_systbl_va->st_entries;
225 aprint_debug("efi: %lu cfgtbl entries:\n", count);
226 for (; count; count--, ct++) {
227 aprint_debug("efi: %p", ct->ct_data);
228 efi_aprintuuid(&ct->ct_uuid);
229 aprint_debug("\n");
230 }
231}

233/*
234 * Return the VA of the config table with the given UUID if found.
235 * The VA must be freed using efi_relva().
236 */
237void *
238efi_getcfgtbl(const struct uuid * uuid)
239{
240 paddr_t pa;
241 vaddr_t va;
242
243 pa = efi_getcfgtblpa(uuid);
244 if (pa == 0)
245 return NULL;
246 va = efi_getva(pa);
247 return (void *) va;
248}
249
250/*
251 * Return the PA of the first config table.
252 */
253paddr_t
254efi_getcfgtblpa(const struct uuid * uuid)
255{
256 struct efi_cfgtbl *ct;
257 unsigned long count;
258
259 if (efi_is32x64) {
260#if defined(__amd64__)
261 struct efi_systbl32 *systbl32 = (void *) efi_systbl_va;
262 struct efi_cfgtbl32 *ct32 = (void *) efi_cfgtblhead_va;
263
264 count = systbl32->st_entries;
265 for (; count; count--, ct32++)
266 if (efi_uuideq(&ct32->ct_uuid, uuid))
267 return ct32->ct_data;
268#elif defined(__i386__)
269 struct efi_systbl64 *systbl64 = (void *) efi_systbl_va;
270 struct efi_cfgtbl64 *ct64 = (void *) efi_cfgtblhead_va;
271 uint64_t count64 = systbl64->st_entries;
272
273 for (; count64; count64--, ct64++)
274 if (efi_uuideq(&ct64->ct_uuid, uuid))
275 if (!(ct64->ct_data & 0xffffffff00000000ULL))
276 return ct64->ct_data;
277#endif
278 return 0; /* Not found. */
279 }
280
281 ct = efi_cfgtblhead_va;
282 count = efi_systbl_va->st_entries;
283 for (; count; count--, ct++)
284 if (efi_uuideq(&ct->ct_uuid, uuid))
285 return (paddr_t)(u_long) ct->ct_data;
286
287 return 0; /* Not found. */
288}

290/* Return the PA of the EFI System Table. */
291paddr_t
292efi_getsystblpa(void)
293{
294 struct btinfo_efi *bi;
295 paddr_t pa;
296
297 bi = lookup_bootinfo(BTINFO_EFI);
298 if (bi == NULL) {
299 /* Unable to locate the EFI System Table. */
300 return 0;
301 }
302 if (sizeof(paddr_t) == 4 && /* XXX i386 with PAE */
303 (bi->systblpa & 0xffffffff00000000ULL)) {
304 /* Unable to access EFI System Table. */
305 return 0;
306 }
307 if (bi->common.len > 16 && (bi->flags & BI_EFI_32BIT)) {
308 /* boot from 32bit UEFI */
309#if defined(__amd64__)
310 efi_is32x64 = true;
311#endif
312 } else {
313 /* boot from 64bit UEFI */
314#if defined(__i386__)
315 efi_is32x64 = true;
316#endif
317 }
318 pa = (paddr_t) bi->systblpa;
319 return pa;
320}

322/*
323 * Return a pointer to the EFI System Table. The pointer must be freed using
324 * efi_relva().
325 */
326struct efi_systbl *
327efi_getsystbl(void)
328{
329 paddr_t pa;
330 vaddr_t va;
331 struct efi_systbl *systbl;
332
333 if (efi_systbl_va)
334 return efi_systbl_va;
335
336 pa = efi_getsystblpa();
337 if (pa == 0)
338 return NULL;
339
340 aprint_normal("efi: systbl at pa %" PRIxPADDR "\n", pa);
341 efi_systbl_pa = pa;
342 va = efi_getva(pa);
343 aprint_debug("efi: systbl mapped at va %" PRIxVADDR "\n", va);
344
345 if (efi_is32x64) {
346#if defined(__amd64__)
347 struct efi_systbl32 *systbl32 = (struct efi_systbl32 *) va;
348
349 /* XXX Check the signature and the CRC32 */
350 aprint_debug("efi: signature %" PRIx64 " revision %" PRIx32
351 " crc32 %" PRIx32 "\n", systbl32->st_hdr.th_sig,
352 systbl32->st_hdr.th_rev, systbl32->st_hdr.th_crc32);
353 aprint_debug("efi: firmware revision %" PRIx32 "\n",
354 systbl32->st_fwrev);
355 /*
356 * XXX Also print fwvendor, which is an UCS-2 string (use
357 * some UTF-16 routine?)
358 */
359 aprint_debug("efi: runtime services at pa 0x%08" PRIx32 "\n",
360 systbl32->st_rt);
361 aprint_debug("efi: boot services at pa 0x%08" PRIx32 "\n",
362 systbl32->st_bs);
363
364 efi_systbl_va = (struct efi_systbl *) systbl32;
365#elif defined(__i386__)
366 struct efi_systbl64 *systbl64 = (struct efi_systbl64 *) va;
367
368 /* XXX Check the signature and the CRC32 */
369 aprint_debug("efi: signature %" PRIx64 " revision %" PRIx32
370 " crc32 %" PRIx32 "\n", systbl64->st_hdr.th_sig,
371 systbl64->st_hdr.th_rev, systbl64->st_hdr.th_crc32);
372 aprint_debug("efi: firmware revision %" PRIx32 "\n",
373 systbl64->st_fwrev);
374 /*
375 * XXX Also print fwvendor, which is an UCS-2 string (use
376 * some UTF-16 routine?)
377 */
378 aprint_debug("efi: runtime services at pa 0x%016" PRIx64 "\n",
379 systbl64->st_rt);
380 aprint_debug("efi: boot services at pa 0x%016" PRIx64 "\n",
381 systbl64->st_bs);
382
383 efi_systbl_va = (struct efi_systbl *) systbl64;
384#endif
385 return efi_systbl_va;
386 }
387
388 systbl = (struct efi_systbl *) va;
389 /* XXX Check the signature and the CRC32 */
390 aprint_debug("efi: signature %" PRIx64 " revision %" PRIx32
391 " crc32 %" PRIx32 "\n", systbl->st_hdr.th_sig,
392 systbl->st_hdr.th_rev, systbl->st_hdr.th_crc32);
393 aprint_debug("efi: firmware revision %" PRIx32 "\n", systbl->st_fwrev);
394 /*
395 * XXX Also print fwvendor, which is an UCS-2 string (use
396 * some UTF-16 routine?)
397 */
398 aprint_debug("efi: runtime services at pa %p\n", systbl->st_rt);
399 aprint_debug("efi: boot services at pa %p\n", systbl->st_bs);
400
401 efi_systbl_va = systbl;
402 return efi_systbl_va;
403}

405/*
406 * EFI is available if we are able to locate the EFI System Table.
407 */
408void
409efi_init(void)
410{
411
412 if (efi_getsystbl() == NULL) {
413 aprint_debug("efi: missing or invalid systbl\n");
414 bootmethod_efi = false;
415 return;
416 }
417 if (efi_getcfgtblhead() == NULL) {
418 aprint_debug("efi: missing or invalid cfgtbl\n");
419 efi_relva(efi_systbl_pa, (vaddr_t) efi_systbl_va);
420 bootmethod_efi = false;
421 return;
422 }
423 bootmethod_efi = true;
424#if NPCI > 0
425 pci_mapreg_map_enable_decode = true; /* PR port-amd64/53286 */
426#endif
427}

429bool
430efi_probe(void)

432
433 return bootmethod_efi;
434}

436int
437efi_getbiosmemtype(uint32_t type, uint64_t attr)
438{
439
440 switch (type) {
441 case EFI_MD_TYPE_CODE:
442 case EFI_MD_TYPE_DATA:
443 case EFI_MD_TYPE_BS_CODE:
444 case EFI_MD_TYPE_BS_DATA:
445 case EFI_MD_TYPE_FREE:
446 return (attr & EFI_MD_ATTR_WB) ? BIM_Memory : BIM_Reserved;
447
448 case EFI_MD_TYPE_RECLAIM:
449 return BIM_ACPI;
450
451 case EFI_MD_TYPE_FIRMWARE:
452 return BIM_NVS;
453
454 case EFI_MD_TYPE_PMEM:
455 return BIM_PMEM;
456
457 case EFI_MD_TYPE_NULL:
458 case EFI_MD_TYPE_RT_CODE:
459 case EFI_MD_TYPE_RT_DATA:
460 case EFI_MD_TYPE_BAD:
461 case EFI_MD_TYPE_IOMEM:
462 case EFI_MD_TYPE_IOPORT:
463 case EFI_MD_TYPE_PALCODE:
464 default:
465 return BIM_Reserved;
466 }
467}

469const char *
470efi_getmemtype_str(uint32_t type)
471{
472 static const char *efimemtypes[] = {
473 "Reserved",
474 "LoaderCode",
475 "LoaderData",
476 "BootServicesCode",
477 "BootServicesData",
478 "RuntimeServicesCode",
479 "RuntimeServicesData",
480 "ConventionalMemory",
481 "UnusableMemory",
482 "ACPIReclaimMemory",
483 "ACPIMemoryNVS",
484 "MemoryMappedIO",
485 "MemoryMappedIOPortSpace",
486 "PalCode",
487 "PersistentMemory",
488 };
489
490 if (type < __arraycount(efimemtypes))
491 return efimemtypes[type];
492 return "unknown";
493}

495struct btinfo_memmap *
496efi_get_e820memmap(void)
497{
498 struct btinfo_efimemmap *efimm;
499 struct bi_memmap_entry *entry;
500 struct efi_md *md;
501 uint64_t addr, size;
502 uint64_t start_addr = 0; /* XXX gcc -Os: maybe-uninitialized */
503 uint64_t end_addr = 0; /* XXX gcc -Os: maybe-uninitialized */
504 uint32_t i;
505 int n, type, seg_type = -1;
506
507 if (efi_e820memmap.bim.common.type == BTINFO_MEMMAP)
508 return &efi_e820memmap.bim;
509
510 efimm = lookup_bootinfo(BTINFO_EFIMEMMAP);
511 if (efimm == NULL)
512 return NULL;
513
514 for (n = 0, i = 0; i < efimm->num; i++) {
515 md = (struct efi_md *)(efimm->memmap + efimm->size * i);
516 addr = md->md_phys;
517 size = md->md_pages * EFI_PAGE_SIZE;
518 type = efi_getbiosmemtype(md->md_type, md->md_attr);
519
520#ifdef DEBUG_MEMLOAD
521 printf("MEMMAP: p0x%016" PRIx64 "-0x%016" PRIx64
522 ", v0x%016" PRIx64 "-0x%016" PRIx64
523 ", size=0x%016" PRIx64 ", attr=0x%016" PRIx64
524 ", type=%d(%s)\n",
525 addr, addr + size - 1,
526 md->md_virt, md->md_virt + size - 1,
527 size, md->md_attr, md->md_type,
528 efi_getmemtype_str(md->md_type));
529#endif
530
531 if (seg_type == -1) {
532 /* first entry */
533 } else if (seg_type == type && end_addr == addr) {
534 /* continuous region */
535 end_addr = addr + size;
536 continue;
537 } else {
538 entry = &efi_e820memmap.bim.entry[n];
539 entry->addr = start_addr;
540 entry->size = end_addr - start_addr;
541 entry->type = seg_type;
542 if (++n == VM_PHYSSEG_MAX)
543 break;
544 }
545
546 start_addr = addr;
547 end_addr = addr + size;
548 seg_type = type;
549 }
550 if (i > 0 && n < VM_PHYSSEG_MAX) {
551 entry = &efi_e820memmap.bim.entry[n];
552 entry->addr = start_addr;
553 entry->size = end_addr - start_addr;
554 entry->type = seg_type;
555 ++n;
556 } else if (n == VM_PHYSSEG_MAX) {
557 printf("WARNING: too many memory segments"
558 "(increase VM_PHYSSEG_MAX)\n");
559 }
560
561 efi_e820memmap.bim.num = n;
562 efi_e820memmap.bim.common.len =
563 (intptr_t)&efi_e820memmap.bim.entry[n] - (intptr_t)&efi_e820memmap;
564 efi_e820memmap.bim.common.type = BTINFO_MEMMAP;
565 return &efi_e820memmap.bim;
566}
