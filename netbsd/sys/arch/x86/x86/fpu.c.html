152void
153fpuinit_mxcsr_mask(void)
154{
155#ifndef XENPV
156 union savefpu fpusave __aligned(16);
157 u_long psl;
158
159 memset(&fpusave, 0, sizeof(fpusave));
160
161 /* Disable interrupts, and enable FPU */
162 psl = x86_read_psl();
163 x86_disable_intr();
164 clts();
165
166 /* Fill in the FPU area */
167 fxsave(&fpusave);
168
169 /* Restore previous state */
170 stts();
171 x86_write_psl(psl);
172
173 if (fpusave.sv_xmm.fx_mxcsr_mask == 0) {
174 x86_fpu_mxcsr_mask = __INITIAL_MXCSR_MASK__;
175 } else {
176 x86_fpu_mxcsr_mask = fpusave.sv_xmm.fx_mxcsr_mask;
177 }
178#else
179 /*
180 * XXX XXX XXX: On Xen the FXSAVE above faults. That's because
181 * &fpusave is not 16-byte aligned. Stack alignment problem
182 * somewhere, it seems.
183 */
184 x86_fpu_mxcsr_mask = __INITIAL_MXCSR_MASK__;
185#endif
186}

188static inline void
189fpu_errata_amd(void)
190{
191 uint16_t sw;
192
193 /*
194 * AMD FPUs do not restore FIP, FDP, and FOP on fxrstor and xrstor
195 * when FSW.ES=0, leaking other threads' execution history.
196 *
197 * Clear them manually by loading a zero (fldummy). We do this
198 * unconditionally, regardless of FSW.ES.
199 *
200 * Before that, clear the ES bit in the x87 status word if it is
201 * currently set, in order to avoid causing a fault in the
202 * upcoming load.
203 *
204 * Newer generations of AMD CPUs have CPUID_Fn80000008_EBX[2],
205 * which indicates that FIP/FDP/FOP are restored (same behavior
206 * as Intel). We're not using it though.
207 */
208 fnstsw(&sw);
209 if (sw & 0x80)
210 fnclex();
211 fldummy();
212}

214void
215fpu_area_save(void *area, uint64_t xsave_features)
216{
217 clts();
218
219 switch (x86_fpu_save) {
220 case FPU_SAVE_FSAVE:
221 fnsave(area);
222 break;
223 case FPU_SAVE_FXSAVE:
224 fxsave(area);
225 break;
226 case FPU_SAVE_XSAVE:
227 xsave(area, xsave_features);
228 break;
229 case FPU_SAVE_XSAVEOPT:
230 xsaveopt(area, xsave_features);
231 break;
232 }
233}
234
235void
236fpu_area_restore(void *area, uint64_t xsave_features)
237{
238 clts();
239
240 switch (x86_fpu_save) {
241 case FPU_SAVE_FSAVE:
242 frstor(area);
243 break;
244 case FPU_SAVE_FXSAVE:
245 if (cpu_vendor == CPUVENDOR_AMD)
246 fpu_errata_amd();
247 fxrstor(area);
248 break;
249 case FPU_SAVE_XSAVE:
250 case FPU_SAVE_XSAVEOPT:
251 if (cpu_vendor == CPUVENDOR_AMD)
252 fpu_errata_amd();
253 xrstor(area, xsave_features);
254 break;
255 }
256}

271void
272fpu_eagerswitch(struct lwp *oldlwp, struct lwp *newlwp)
273{
274 int s;
275
276 s = splhigh();
277#ifdef DIAGNOSTIC
278 if (oldlwp != NULL) {
279 struct pcb *pcb = lwp_getpcb(oldlwp);
280 struct cpu_info *ci = curcpu();
281 if (pcb->pcb_fpcpu == NULL) {
282 KASSERT(ci->ci_fpcurlwp != oldlwp);
283 } else if (pcb->pcb_fpcpu == ci) {
284 KASSERT(ci->ci_fpcurlwp == oldlwp);
285 } else {
286 panic("%s: oldlwp's state installed elsewhere",
287 __func__);
288 }
289 }
290#endif
291 fpusave_cpu(true);
292 if (!(newlwp->l_flag & LW_SYSTEM))
293 fpu_lwp_install(newlwp);
294 splx(s);
295}

299/*
300 * The following table is used to ensure that the FPE_... value
301 * that is passed as a trapcode to the signal handler of the user
302 * process does not have more than one bit set.
303 *
304 * Multiple bits may be set if SSE simd instructions generate errors
305 * on more than one value or if the user process modifies the control
306 * word while a status word bit is already set (which this is a sign
307 * of bad coding).
308 * We have no choise than to narrow them down to one bit, since we must
309 * not send a trapcode that is not exactly one of the FPE_ macros.
310 *
311 * The mechanism has a static table with 127 entries. Each combination
312 * of the 7 FPU status word exception bits directly translates to a
313 * position in this table, where a single FPE_... value is stored.
314 * This FPE_... value stored there is considered the "most important"
315 * of the exception bits and will be sent as the signal code. The
316 * precedence of the bits is based upon Intel Document "Numerical
317 * Applications", Chapter "Special Computational Situations".
318 *
319 * The code to choose one of these values does these steps:
320 * 1) Throw away status word bits that cannot be masked.
321 * 2) Throw away the bits currently masked in the control word,
322 * assuming the user isn't interested in them anymore.
323 * 3) Reinsert status word bit 7 (stack fault) if it is set, which
324 * cannot be masked but must be preserved.
325 * 'Stack fault' is a sub-class of 'invalid operation'.
326 * 4) Use the remaining bits to point into the trapcode table.
327 *
328 * The 6 maskable bits in order of their preference, as stated in the
329 * above referenced Intel manual:
330 * 1 Invalid operation (FP_X_INV)
331 * 1a Stack underflow
332 * 1b Stack overflow
333 * 1c Operand of unsupported format
334 * 1d SNaN operand.
335 * 2 QNaN operand (not an exception, irrelevant here)
336 * 3 Any other invalid-operation not mentioned above or zero divide
337 * (FP_X_INV, FP_X_DZ)
338 * 4 Denormal operand (FP_X_DNML)
339 * 5 Numeric over/underflow (FP_X_OFL, FP_X_UFL)
340 * 6 Inexact result (FP_X_IMP)
341 *
342 * NB: the above seems to mix up the mxscr error bits and the x87 ones.
343 * They are in the same order, but there is no EN_SW_STACK_FAULT in the mmx
344 * status.
345 *
346 * The table is nearly, but not quite, in bit order (ZERODIV and DENORM
347 * are swapped).
348 *
349 * This table assumes that any stack fault is cleared - so that an INVOP
350 * fault will only be reported as FLTSUB once.
351 * This might not happen if the mask is being changed.
352 */
353#define FPE_xxx1(f) (f & EN_SW_INVOP \
354 ? (f & EN_SW_STACK_FAULT ? FPE_FLTSUB : FPE_FLTINV) \
355 : f & EN_SW_ZERODIV ? FPE_FLTDIV \
356 : f & EN_SW_DENORM ? FPE_FLTUND \
357 : f & EN_SW_OVERFLOW ? FPE_FLTOVF \
358 : f & EN_SW_UNDERFLOW ? FPE_FLTUND \
359 : f & EN_SW_PRECLOSS ? FPE_FLTRES \
360 : f & EN_SW_STACK_FAULT ? FPE_FLTSUB : 0)
361#define FPE_xxx2(f) FPE_xxx1(f), FPE_xxx1((f + 1))
362#define FPE_xxx4(f) FPE_xxx2(f), FPE_xxx2((f + 2))
363#define FPE_xxx8(f) FPE_xxx4(f), FPE_xxx4((f + 4))
364#define FPE_xxx16(f) FPE_xxx8(f), FPE_xxx8((f + 8))
365#define FPE_xxx32(f) FPE_xxx16(f), FPE_xxx16((f + 16))
366static const uint8_t fpetable[128] = {
367 FPE_xxx32(0), FPE_xxx32(32), FPE_xxx32(64), FPE_xxx32(96)
368};
369#undef FPE_xxx1
370#undef FPE_xxx2
371#undef FPE_xxx4
372#undef FPE_xxx8
373#undef FPE_xxx16
374#undef FPE_xxx32
375
376/*
377 * This is a synchronous trap on either an x87 instruction (due to an unmasked
378 * error on the previous x87 instruction) or on an SSE/SSE2/etc instruction due
379 * to an error on the instruction itself.
380 *
381 * If trap actually generates a signal, then the fpu state is saved and then
382 * copied onto the lwp's user-stack, and then recovered from there when the
383 * signal returns.
384 *
385 * All this code needs to do is save the reason for the trap. For x87 traps the
386 * status word bits need clearing to stop the trap re-occurring. For SSE traps
387 * the mxcsr bits are 'sticky' and need clearing to not confuse a later trap.
388 *
389 * We come here with interrupts disabled.
390 */
391void
392fputrap(struct trapframe *frame)
393{
394 uint32_t statbits;
395 ksiginfo_t ksi;
396
397 if (__predict_false(!USERMODE(frame->tf_cs))) {
398 panic("fpu trap from kernel, trapframe %p\n", frame);
399 }
400
401 /*
402 * At this point, fpcurlwp should be curlwp. If it wasn't, the TS bit
403 * should be set, and we should have gotten a DNA exception.
404 */
405 KASSERT(curcpu()->ci_fpcurlwp == curlwp);
406
407 if (frame->tf_trapno == T_XMM) {
408 uint32_t mxcsr;
409 x86_stmxcsr(&mxcsr);
410 statbits = mxcsr;
411 /* Clear the sticky status bits */
412 mxcsr &= ~0x3f;
413 x86_ldmxcsr(&mxcsr);
414
415 /* Remove masked interrupts and non-status bits */
416 statbits &= ~(statbits >> 7) & 0x3f;
417 /* Mark this is an XMM status */
418 statbits |= 0x10000;
419 } else {
420 uint16_t cw, sw;
421 /* Get current control and status words */
422 fnstcw(&cw);
423 fnstsw(&sw);
424 /* Clear any pending exceptions from status word */
425 fnclex();
426
427 /* Remove masked interrupts */
428 statbits = sw & ~(cw & 0x3f);
429 }
430
431 /* Doesn't matter now if we get pre-empted */
432 x86_enable_intr();
433
434 KSI_INIT_TRAP(&ksi);
435 ksi.ksi_signo = SIGFPE;
436 ksi.ksi_addr = (void *)X86_TF_RIP(frame);
437 ksi.ksi_code = fpetable[statbits & 0x7f];
438 ksi.ksi_trap = statbits;
439 (*curlwp->l_proc->p_emul->e_trapsignal)(curlwp, &ksi);
440}

442/*
443 * Implement device not available (DNA) exception.
444 *
445 * If we were the last lwp to use the FPU, we can simply return.
446 * Otherwise, we save the previous state, if necessary, and restore
447 * our last saved state.
448 *
449 * Called directly from the trap 0x13 entry with interrupts still disabled.
450 */
451void
452fpudna(struct trapframe *frame)
453{
454 struct cpu_info *ci = curcpu();
455 struct lwp *l, *fl;
456 struct pcb *pcb;
457 int s;
458
459 if (!USERMODE(frame->tf_cs)) {
460 panic("fpudna from kernel, ip %p, trapframe %p\n",
461 (void *)X86_TF_RIP(frame), frame);
462 }
463
464 s = splhigh();
465
466 /* Save state on current CPU. */
467 l = ci->ci_curlwp;
468 pcb = lwp_getpcb(l);
469 fl = ci->ci_fpcurlwp;
470 if (fl != NULL) {
471 if (__predict_false(x86_fpu_eager)) {
472 panic("%s: FPU busy with EagerFPU enabled",
473 __func__);
474 }
475
476 /*
477 * It seems we can get here on Xen even if we didn't
478 * switch lwp. In this case do nothing
479 */
480 if (fl == l) {
481 KASSERT(pcb->pcb_fpcpu == ci);
482 clts();
483 splx(s);
484 return;
485 }
486 fpusave_cpu(true);
487 }
488
489 /* Save our state if on a remote CPU. */
490 if (pcb->pcb_fpcpu != NULL) {
491 if (__predict_false(x86_fpu_eager)) {
492 panic("%s: LWP busy with EagerFPU enabled",
493 __func__);
494 }
495
496 /* Explicitly disable preemption before dropping spl. */
497 kpreempt_disable();
498 splx(s);
499
500 /* Actually enable interrupts */
501 x86_enable_intr();
502
503 fpusave_lwp(l, true);
504 KASSERT(pcb->pcb_fpcpu == NULL);
505 s = splhigh();
506 kpreempt_enable();
507 }
508
509 /* Install the LWP's FPU state. */
510 fpu_lwp_install(l);
511
512 KASSERT(ci == curcpu());
513 splx(s);
514}

518/*
519 * Save current CPU's FPU state. Must be called at IPL_HIGH.
520 */
521void
522fpusave_cpu(bool save)
523{
524 struct cpu_info *ci;
525 struct pcb *pcb;
526 struct lwp *l;
527
528 KASSERT(curcpu()->ci_ilevel == IPL_HIGH);
529
530 ci = curcpu();
531 l = ci->ci_fpcurlwp;
532 if (l == NULL) {
533 return;
534 }
535 pcb = lwp_getpcb(l);
536
537 if (save) {
538 fpu_area_save(&pcb->pcb_savefpu, x86_xsave_features);
539 }
540
541 stts();
542 pcb->pcb_fpcpu = NULL;
543 ci->ci_fpcurlwp = NULL;
544}

546/*
547 * Save l's FPU state, which may be on this processor or another processor.
548 * It may take some time, so we avoid disabling preemption where possible.
549 * Caller must know that the target LWP is stopped, otherwise this routine
550 * may race against it.
551 */
552void
553fpusave_lwp(struct lwp *l, bool save)
554{
555 struct pcb *pcb = lwp_getpcb(l);
556 struct cpu_info *oci;
557 int s, spins, ticks;
558
559 spins = 0;
560 ticks = hardclock_ticks;
561 for (;;) {
562 s = splhigh();
563 oci = pcb->pcb_fpcpu;
564 if (oci == NULL) {
565 splx(s);
566 break;
567 }
568 if (oci == curcpu()) {
569 KASSERT(oci->ci_fpcurlwp == l);
570 fpusave_cpu(save);
571 splx(s);
572 break;
573 }
574 splx(s);
575#ifdef XENPV
576 if (xen_send_ipi(oci, XEN_IPI_SYNCH_FPU) != 0) {
577 panic("xen_send_ipi(%s, XEN_IPI_SYNCH_FPU) failed.",
578 cpu_name(oci));
579 }
580#else
581 x86_send_ipi(oci, X86_IPI_SYNCH_FPU);
582#endif
583 while (pcb->pcb_fpcpu == oci && ticks == hardclock_ticks) {
584 x86_pause();
585 spins++;
586 }
587 if (spins > 100000000) {
588 panic("fpusave_lwp: did not");
589 }
590 }
591}

593void
594fpu_set_default_cw(struct lwp *l, unsigned int x87_cw)
595{
596 union savefpu *fpu_save = lwp_fpuarea(l);
597 struct pcb *pcb = lwp_getpcb(l);
598
599 if (i386_use_fxsave) {
600 fpu_save->sv_xmm.fx_cw = x87_cw;
601
602 /* Force a reload of CW */
603 if ((x87_cw != __INITIAL_NPXCW__) &&
604 (x86_fpu_save == FPU_SAVE_XSAVE ||
605 x86_fpu_save == FPU_SAVE_XSAVEOPT)) {
606 fpu_save->sv_xsave_hdr.xsh_xstate_bv |=
607 XCR0_X87;
608 }
609 } else {
610 fpu_save->sv_87.s87_cw = x87_cw;
611 }
612 pcb->pcb_fpu_dflt_cw = x87_cw;
613}

615void
616fpu_clear(struct lwp *l, unsigned int x87_cw)
617{
618 union savefpu *fpu_save;
619 struct pcb *pcb;
620 int s;
621
622 KASSERT(l == curlwp);
623 KASSERT((l->l_flag & LW_SYSTEM) == 0);
624 fpu_save = lwp_fpuarea(l);
625 pcb = lwp_getpcb(l);
626
627 s = splhigh();
628 if (x86_fpu_eager) {
629 KASSERT(pcb->pcb_fpcpu == NULL ||
630 pcb->pcb_fpcpu == curcpu());
631 fpusave_cpu(false);
632 } else {
633 splx(s);
634 fpusave_lwp(l, false);
635 }
636 KASSERT(pcb->pcb_fpcpu == NULL);
637
638 switch (x86_fpu_save) {
639 case FPU_SAVE_FSAVE:
640 memset(&fpu_save->sv_87, 0, x86_fpu_save_size);
641 fpu_save->sv_87.s87_tw = 0xffff;
642 fpu_save->sv_87.s87_cw = x87_cw;
643 break;
644 case FPU_SAVE_FXSAVE:
645 memset(&fpu_save->sv_xmm, 0, x86_fpu_save_size);
646 fpu_save->sv_xmm.fx_mxcsr = __INITIAL_MXCSR__;
647 fpu_save->sv_xmm.fx_mxcsr_mask = x86_fpu_mxcsr_mask;
648 fpu_save->sv_xmm.fx_cw = x87_cw;
649 break;
650 case FPU_SAVE_XSAVE:
651 case FPU_SAVE_XSAVEOPT:
652 memset(&fpu_save->sv_xmm, 0, x86_fpu_save_size);
653 fpu_save->sv_xmm.fx_mxcsr = __INITIAL_MXCSR__;
654 fpu_save->sv_xmm.fx_mxcsr_mask = x86_fpu_mxcsr_mask;
655 fpu_save->sv_xmm.fx_cw = x87_cw;
656
657 /*
658 * Force a reload of CW if we're using the non-default
659 * value.
660 */
661 if (__predict_false(x87_cw != __INITIAL_NPXCW__)) {
662 fpu_save->sv_xsave_hdr.xsh_xstate_bv |=
663 XCR0_X87;
664 }
665 break;
666 }
667
668 pcb->pcb_fpu_dflt_cw = x87_cw;
669
670 if (x86_fpu_eager) {
671 fpu_lwp_install(l);
672 splx(s);
673 }
674}

676void
677fpu_sigreset(struct lwp *l)
678{
679 union savefpu *fpu_save = lwp_fpuarea(l);
680 struct pcb *pcb = lwp_getpcb(l);
681
682 /*
683 * For signal handlers the register values don't matter. Just reset
684 * a few fields.
685 */
686 if (i386_use_fxsave) {
687 fpu_save->sv_xmm.fx_mxcsr = __INITIAL_MXCSR__;
688 fpu_save->sv_xmm.fx_mxcsr_mask = x86_fpu_mxcsr_mask;
689 fpu_save->sv_xmm.fx_tw = 0;
690 fpu_save->sv_xmm.fx_cw = pcb->pcb_fpu_dflt_cw;
691 } else {
692 fpu_save->sv_87.s87_tw = 0xffff;
693 fpu_save->sv_87.s87_cw = pcb->pcb_fpu_dflt_cw;
694 }
695}

710static void
711process_xmm_to_s87(const struct fxsave *sxmm, struct save87 *s87)
712{
713 unsigned int tag, ab_tag;
714 const struct fpaccfx *fx_reg;
715 struct fpacc87 *s87_reg;
716 int i;
717
718 /*
719 * For historic reasons core dumps and ptrace all use the old save87
720 * layout. Convert the important parts.
721 * getucontext gets what we give it.
722 * setucontext should return something given by getucontext, but
723 * we are (at the moment) willing to change it.
724 *
725 * It really isn't worth setting the 'tag' bits to 01 (zero) or
726 * 10 (NaN etc) since the processor will set any internal bits
727 * correctly when the value is loaded (the 287 believed them).
728 *
729 * Additionally the s87_tw and s87_tw are 'indexed' by the actual
730 * register numbers, whereas the registers themselves have ST(0)
731 * first. Pairing the values and tags can only be done with
732 * reference to the 'top of stack'.
733 *
734 * If any x87 registers are used, they will typically be from
735 * r7 downwards - so the high bits of the tag register indicate
736 * used registers. The conversions are not optimised for this.
737 *
738 * The ABI we use requires the FP stack to be empty on every
739 * function call. I think this means that the stack isn't expected
740 * to overflow - overflow doesn't drop a core in my testing.
741 *
742 * Note that this code writes to all of the 's87' structure that
743 * actually gets written to userspace.
744 */
745
746 /* FPU control/status */
747 s87->s87_cw = sxmm->fx_cw;
748 s87->s87_sw = sxmm->fx_sw;
749 /* tag word handled below */
750 s87->s87_ip = sxmm->fx_ip;
751 s87->s87_opcode = sxmm->fx_opcode;
752 s87->s87_dp = sxmm->fx_dp;
753
754 /* FP registers (in stack order) */
755 fx_reg = sxmm->fx_87_ac;
756 s87_reg = s87->s87_ac;
757 for (i = 0; i < 8; fx_reg++, s87_reg++, i++)
758 *s87_reg = fx_reg->r;
759
760 /* Tag word and registers. */
761 ab_tag = sxmm->fx_tw & 0xff; /* Bits set if valid */
762 if (ab_tag == 0) {
763 /* none used */
764 s87->s87_tw = 0xffff;
765 return;
766 }
767
768 tag = 0;
769 /* Separate bits of abridged tag word with zeros */
770 for (i = 0x80; i != 0; tag <<= 1, i >>= 1)
771 tag |= ab_tag & i;
772 /* Replicate and invert so that 0 => 0b11 and 1 => 0b00 */
773 s87->s87_tw = (tag | tag >> 1) ^ 0xffff;
774}
775
776static void
777process_s87_to_xmm(const struct save87 *s87, struct fxsave *sxmm)
778{
779 unsigned int tag, ab_tag;
780 struct fpaccfx *fx_reg;
781 const struct fpacc87 *s87_reg;
782 int i;
783
784 /*
785 * ptrace gives us registers in the save87 format and
786 * we must convert them to the correct format.
787 *
788 * This code is normally used when overwriting the processes
789 * registers (in the pcb), so it musn't change any other fields.
790 *
791 * There is a lot of pad in 'struct fxsave', if the destination
792 * is written to userspace, it must be zeroed first.
793 */
794
795 /* FPU control/status */
796 sxmm->fx_cw = s87->s87_cw;
797 sxmm->fx_sw = s87->s87_sw;
798 /* tag word handled below */
799 sxmm->fx_ip = s87->s87_ip;
800 sxmm->fx_opcode = s87->s87_opcode;
801 sxmm->fx_dp = s87->s87_dp;
802
803 /* Tag word */
804 tag = s87->s87_tw; /* 0b11 => unused */
805 if (tag == 0xffff) {
806 /* All unused - values don't matter, zero for safety */
807 sxmm->fx_tw = 0;
808 memset(&sxmm->fx_87_ac, 0, sizeof sxmm->fx_87_ac);
809 return;
810 }
811
812 tag ^= 0xffff; /* So 0b00 is unused */
813 tag |= tag >> 1; /* Look at even bits */
814 ab_tag = 0;
815 i = 1;
816 do
817 ab_tag |= tag & i;
818 while ((tag >>= 1) >= (i <<= 1));
819 sxmm->fx_tw = ab_tag;
820
821 /* FP registers (in stack order) */
822 fx_reg = sxmm->fx_87_ac;
823 s87_reg = s87->s87_ac;
824 for (i = 0; i < 8; fx_reg++, s87_reg++, i++)
825 fx_reg->r = *s87_reg;
826}

828void
829process_write_fpregs_xmm(struct lwp *l, const struct fxsave *fpregs)
830{
831 union savefpu *fpu_save;
832
833 fpusave_lwp(l, true);
834 fpu_save = lwp_fpuarea(l);
835
836 if (i386_use_fxsave) {
837 memcpy(&fpu_save->sv_xmm, fpregs, sizeof(fpu_save->sv_xmm));
838
839 /*
840 * Invalid bits in mxcsr or mxcsr_mask will cause faults.
841 */
842 fpu_save->sv_xmm.fx_mxcsr_mask &= x86_fpu_mxcsr_mask;
843 fpu_save->sv_xmm.fx_mxcsr &= fpu_save->sv_xmm.fx_mxcsr_mask;
844
845 /*
846 * Make sure the x87 and SSE bits are set in xstate_bv.
847 * Otherwise xrstor will not restore them.
848 */
849 if (x86_fpu_save == FPU_SAVE_XSAVE ||
850 x86_fpu_save == FPU_SAVE_XSAVEOPT) {
851 fpu_save->sv_xsave_hdr.xsh_xstate_bv |=
852 (XCR0_X87 | XCR0_SSE);
853 }
854 } else {
855 process_xmm_to_s87(fpregs, &fpu_save->sv_87);
856 }
857}

918int
919process_read_xstate(struct lwp *l, struct xstate *xstate)
920{
921 union savefpu *fpu_save;
922
923 fpusave_lwp(l, true);
924 fpu_save = lwp_fpuarea(l);
925
926 if (x86_fpu_save == FPU_SAVE_FSAVE) {
927 /* Convert from legacy FSAVE format. */
928 memset(&xstate->xs_fxsave, 0, sizeof(xstate->xs_fxsave));
929 process_s87_to_xmm(&fpu_save->sv_87, &xstate->xs_fxsave);
930
931 /* We only got x87 data. */
932 xstate->xs_rfbm = XCR0_X87;
933 xstate->xs_xstate_bv = XCR0_X87;
934 return 0;
935 }
936
937 /* Copy the legacy area. */
938 memcpy(&xstate->xs_fxsave, fpu_save->sv_xsave_hdr.xsh_fxsave,
939 sizeof(xstate->xs_fxsave));
940
941 if (x86_fpu_save == FPU_SAVE_FXSAVE) {
942 /* FXSAVE means we've got x87 + SSE data. */
943 xstate->xs_rfbm = XCR0_X87 | XCR0_SSE;
944 xstate->xs_xstate_bv = XCR0_X87 | XCR0_SSE;
945 return 0;
946 }
947
948 /* Copy the bitmap indicating which states are available. */
949 xstate->xs_rfbm = x86_xsave_features & XCR0_FPU;
950 xstate->xs_xstate_bv = fpu_save->sv_xsave_hdr.xsh_xstate_bv;
951 KASSERT(!(xstate->xs_xstate_bv & ~xstate->xs_rfbm));
952
953#define COPY_COMPONENT(xcr0_val, xsave_val, field) \
954 if (xstate->xs_xstate_bv & xcr0_val) { \
955 KASSERT(x86_xsave_offsets[xsave_val] \
956 >= sizeof(struct xsave_header)); \
957 KASSERT(x86_xsave_sizes[xsave_val] \
958 >= sizeof(xstate -> field)); \
959 \
960 memcpy(&xstate -> field, \
961 (char*)fpu_save + x86_xsave_offsets[xsave_val], \
962 sizeof(xstate -> field)); \
963 }
964
965 COPY_COMPONENT(XCR0_YMM_Hi128, XSAVE_YMM_Hi128, xs_ymm_hi128);
966 COPY_COMPONENT(XCR0_Opmask, XSAVE_Opmask, xs_opmask);
967 COPY_COMPONENT(XCR0_ZMM_Hi256, XSAVE_ZMM_Hi256, xs_zmm_hi256);
968 COPY_COMPONENT(XCR0_Hi16_ZMM, XSAVE_Hi16_ZMM, xs_hi16_zmm);
969
970#undef COPY_COMPONENT
971
972 return 0;
973}

975int
976process_verify_xstate(const struct xstate *xstate)
977{
978 /* xstate_bv must be a subset of RFBM */
979 if (xstate->xs_xstate_bv & ~xstate->xs_rfbm)
980 return EINVAL;
981
982 switch (x86_fpu_save) {
983 case FPU_SAVE_FSAVE:
984 if ((xstate->xs_rfbm & ~XCR0_X87))
985 return EINVAL;
986 break;
987 case FPU_SAVE_FXSAVE:
988 if ((xstate->xs_rfbm & ~(XCR0_X87 | XCR0_SSE)))
989 return EINVAL;
990 break;
991 default:
992 /* Verify whether no unsupported features are enabled */
993 if ((xstate->xs_rfbm & ~(x86_xsave_features & XCR0_FPU)) != 0)
994 return EINVAL;
995 }
996
997 return 0;
998}

1000int
1001process_write_xstate(struct lwp *l, const struct xstate *xstate)
1002{
1003 union savefpu *fpu_save;
1004
1005 fpusave_lwp(l, true);
1006 fpu_save = lwp_fpuarea(l);
1007
1008 /* Convert data into legacy FSAVE format. */
1009 if (x86_fpu_save == FPU_SAVE_FSAVE) {
1010 if (xstate->xs_xstate_bv & XCR0_X87)
1011 process_xmm_to_s87(&xstate->xs_fxsave, &fpu_save->sv_87);
1012 return 0;
1013 }
1014
1015 /* If XSAVE is supported, make sure that xstate_bv is set correctly. */
1016 if (x86_fpu_save >= FPU_SAVE_XSAVE) {
1017 /*
1018 * Bit-wise xstate->xs_rfbm ? xstate->xs_xstate_bv
1019 * : fpu_save->sv_xsave_hdr.xsh_xstate_bv
1020 */
1021 fpu_save->sv_xsave_hdr.xsh_xstate_bv =
1022 (fpu_save->sv_xsave_hdr.xsh_xstate_bv & ~xstate->xs_rfbm) |
1023 xstate->xs_xstate_bv;
1024 }
1025
1026 if (xstate->xs_xstate_bv & XCR0_X87) {
1027 /*
1028 * X87 state is split into two areas, interspersed with SSE
1029 * data.
1030 */
1031 memcpy(&fpu_save->sv_xmm, &xstate->xs_fxsave, 24);
1032 memcpy(fpu_save->sv_xmm.fx_87_ac, xstate->xs_fxsave.fx_87_ac,
1033 sizeof(xstate->xs_fxsave.fx_87_ac));
1034 }
1035
1036 /*
1037 * Copy MXCSR if either SSE or AVX state is requested, to match the XSAVE
1038 * behavior for those flags.
1039 */
1040 if (xstate->xs_xstate_bv & (XCR0_SSE|XCR0_YMM_Hi128)) {
1041 /*
1042 * Invalid bits in mxcsr or mxcsr_mask will cause faults.
1043 */
1044 fpu_save->sv_xmm.fx_mxcsr_mask = xstate->xs_fxsave.fx_mxcsr_mask
1045 & x86_fpu_mxcsr_mask;
1046 fpu_save->sv_xmm.fx_mxcsr = xstate->xs_fxsave.fx_mxcsr &
1047 fpu_save->sv_xmm.fx_mxcsr_mask;
1048 }
1049
1050 if (xstate->xs_xstate_bv & XCR0_SSE) {
1051 memcpy(&fpu_save->sv_xsave_hdr.xsh_fxsave[160],
1052 xstate->xs_fxsave.fx_xmm,
1053 sizeof(xstate->xs_fxsave.fx_xmm));
1054 }
1055
1056#define COPY_COMPONENT(xcr0_val, xsave_val, field) \
1057 if (xstate->xs_xstate_bv & xcr0_val) { \
1058 KASSERT(x86_xsave_offsets[xsave_val] \
1059 >= sizeof(struct xsave_header)); \
1060 KASSERT(x86_xsave_sizes[xsave_val] \
1061 >= sizeof(xstate -> field)); \
1062 \
1063 memcpy((char*)fpu_save + x86_xsave_offsets[xsave_val], \
1064 &xstate -> field, sizeof(xstate -> field)); \
1065 }
1066
1067 COPY_COMPONENT(XCR0_YMM_Hi128, XSAVE_YMM_Hi128, xs_ymm_hi128);
1068 COPY_COMPONENT(XCR0_Opmask, XSAVE_Opmask, xs_opmask);
1069 COPY_COMPONENT(XCR0_ZMM_Hi256, XSAVE_ZMM_Hi256, xs_zmm_hi256);
1070 COPY_COMPONENT(XCR0_Hi16_ZMM, XSAVE_Hi16_ZMM, xs_hi16_zmm);
1071
1072#undef COPY_COMPONENT
1073
1074 return 0;
1075}

1079static volatile unsigned long eagerfpu_cpu_barrier1 __cacheline_aligned;
1080static volatile unsigned long eagerfpu_cpu_barrier2 __cacheline_aligned;
1081
1082static void
1083eager_change_cpu(void *arg1, void *arg2)
1084{
1085 struct cpu_info *ci = curcpu();
1086 bool enabled = (bool)arg1;
1087 int s;
1088
1089 s = splhigh();
1090
1091 /* Rendez-vous 1. */
1092 atomic_dec_ulong(&eagerfpu_cpu_barrier1);
1093 while (atomic_cas_ulong(&eagerfpu_cpu_barrier1, 0, 0) != 0) {
1094 x86_pause();
1095 }
1096
1097 fpusave_cpu(true);
1098 if (ci == &cpu_info_primary) {
1099 x86_fpu_eager = enabled;
1100 }
1101
1102 /* Rendez-vous 2. */
1103 atomic_dec_ulong(&eagerfpu_cpu_barrier2);
1104 while (atomic_cas_ulong(&eagerfpu_cpu_barrier2, 0, 0) != 0) {
1105 x86_pause();
1106 }
1107
1108 splx(s);
1109}

1111static int
1112eager_change(bool enabled)
1113{
1114 struct cpu_info *ci = NULL;
1115 CPU_INFO_ITERATOR cii;
1116 uint64_t xc;
1117
1118 mutex_enter(&cpu_lock);
1119
1120 /*
1121 * We expect all the CPUs to be online.
1122 */
1123 for (CPU_INFO_FOREACH(cii, ci)) {
1124 struct schedstate_percpu *spc = &ci->ci_schedstate;
1125 if (spc->spc_flags & SPCF_OFFLINE) {
1126 printf("[!] cpu%d offline, EagerFPU not changed\n",
1127 cpu_index(ci));
1128 mutex_exit(&cpu_lock);
1129 return EOPNOTSUPP;
1130 }
1131 }
1132
1133 /* Initialize the barriers */
1134 eagerfpu_cpu_barrier1 = ncpu;
1135 eagerfpu_cpu_barrier2 = ncpu;
1136
1137 printf("[+] %s EagerFPU...",
1138 enabled ? "Enabling" : "Disabling");
1139 xc = xc_broadcast(0, eager_change_cpu,
1140 (void *)enabled, NULL);
1141 xc_wait(xc);
1142 printf(" done!\n");
1143
1144 mutex_exit(&cpu_lock);
1145
1146 return 0;
1147}
