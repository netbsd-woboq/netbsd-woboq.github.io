89void
90x86_genfb_ddb_trap_callback(int where)
91{
92 if (x86_genfb_console_dev == NULL)
93 return;
94
95 if (where) {
96 genfb_enable_polling(x86_genfb_console_dev);
97 } else {
98 genfb_disable_polling(x86_genfb_console_dev);
99 }
100}

102void
103x86_genfb_mtrr_init(uint64_t physaddr, uint32_t size)
104{
105#if notyet
106#ifdef MTRR
107 struct mtrr mtrr;
108 int error, n;
109
110 if (mtrr_funcs == NULL) {
111 aprint_debug("%s: no mtrr funcs\n", __func__);
112 return;
113 }
114
115 mtrr.base = physaddr;
116 mtrr.len = size;
117 mtrr.type = MTRR_TYPE_WC;
118 mtrr.flags = MTRR_VALID;
119 mtrr.owner = 0;
120
121 aprint_debug("%s: 0x%" PRIx64 "-0x%" PRIx64 "\n", __func__,
122 mtrr.base, mtrr.base + mtrr.len - 1);
123
124 n = 1;
125 KERNEL_LOCK(1, NULL);
126 error = mtrr_set(&mtrr, &n, curlwp->l_proc, MTRR_GETSET_KERNEL);
127 if (n != 0)
128 mtrr_commit();
129 KERNEL_UNLOCK_ONE(NULL);
130
131 aprint_debug("%s: mtrr_set returned %d\n", __func__, error);
132#else
133 aprint_debug("%s: kernel lacks MTRR option\n", __func__);
134#endif
135#endif
136}

138int
139x86_genfb_cnattach(void)
140{
141 static int ncalls = 0;
142 struct rasops_info *ri = &x86_genfb_console_screen.scr_ri;
143 const struct btinfo_framebuffer *fbinfo;
144 bus_space_tag_t t = x86_bus_space_mem;
145 bus_space_handle_t h;
146 void *bits;
147 long defattr;
148 int err;
149
150 /* XXX jmcneill
151 * Defer console initialization until UVM is initialized
152 */
153 ++ncalls;
154 if (ncalls < 3)
155 return -1;
156
157 memset(&x86_genfb_console_screen, 0, sizeof(x86_genfb_console_screen));
158
159 fbinfo = lookup_bootinfo(BTINFO_FRAMEBUFFER);
160 if (fbinfo == NULL || fbinfo->physaddr == 0)
161 return 0;
162
163 err = _x86_memio_map(t, (bus_addr_t)fbinfo->physaddr,
164 fbinfo->height * fbinfo->stride,
165 BUS_SPACE_MAP_LINEAR | BUS_SPACE_MAP_PREFETCHABLE, &h);
166 if (err) {
167 aprint_error("x86_genfb_cnattach: couldn't map framebuffer\n");
168 return 0;
169 }
170
171 bits = bus_space_vaddr(t, h);
172 if (bits == NULL) {
173 aprint_error("x86_genfb_cnattach: couldn't get fb vaddr\n");
174 return 0;
175 }
176
177#if NACPICA > 0
178 acpi_md_vesa_modenum = fbinfo->vbemode;
179#endif
180
181 wsfont_init();
182
183 ri->ri_width = fbinfo->width;
184 ri->ri_height = fbinfo->height;
185 ri->ri_depth = fbinfo->depth;
186 ri->ri_stride = fbinfo->stride;
187 if (x86_genfb_use_shadowfb && lookup_bootinfo(BTINFO_EFI) != NULL) {
188 /* XXX The allocated memory is never released... */
189 ri->ri_bits = kmem_alloc(ri->ri_stride * ri->ri_height,
190 KM_NOSLEEP);
191 if (ri->ri_bits == NULL) {
192 aprint_error("%s: couldn't alloc shadowfb\n", __func__);
193 ri->ri_bits = bits;
194 } else
195 ri->ri_hwbits = bits;
196 } else
197 ri->ri_bits = bits;
198 ri->ri_flg = RI_CENTER | RI_FULLCLEAR | RI_CLEAR;
199 rasops_init(ri, ri->ri_height / 8, ri->ri_width / 8);
200 ri->ri_caps = WSSCREEN_WSCOLORS;
201 rasops_reconfig(ri, ri->ri_height / ri->ri_font->fontheight,
202 ri->ri_width / ri->ri_font->fontwidth);
203
204 x86_genfb_stdscreen.nrows = ri->ri_rows;
205 x86_genfb_stdscreen.ncols = ri->ri_cols;
206 x86_genfb_stdscreen.textops = &ri->ri_ops;
207 x86_genfb_stdscreen.capabilities = ri->ri_caps;
208
209 ri->ri_ops.allocattr(ri, 0, 0, 0, &defattr);
210 wsdisplay_preattach(&x86_genfb_stdscreen, ri, 0, 0, defattr);
211
212 return 1;
213}
