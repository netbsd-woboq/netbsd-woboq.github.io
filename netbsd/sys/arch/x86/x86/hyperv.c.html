183static __inline u_int
184atomic_load_acq_int(volatile u_int *p)
185{
186 u_int r = *p;
187 __insn_barrier();
188 return r;
189}
190
191#define HYPERV_TSC_TIMECOUNT(fence) \
192static uint64_t \
193hyperv_tc64_tsc_##fence(void) \
194{ \
195 struct hyperv_reftsc *tsc_ref = hyperv_ref_tsc.tsc_ref; \
196 uint32_t seq; \
197 \
198 while ((seq = atomic_load_acq_int(&tsc_ref->tsc_seq)) != 0) { \
199 uint64_t disc, ret, tsc; \
200 uint64_t scale = tsc_ref->tsc_scale; \
201 int64_t ofs = tsc_ref->tsc_ofs; \
202 \
203 x86_##fence(); \
204 tsc = cpu_counter(); \
205 \
206 /* ret = ((tsc * scale) >> 64) + ofs */ \
207 __asm__ __volatile__ ("mulq %3" : \
208 "=d" (ret), "=a" (disc) : \
209 "a" (tsc), "r" (scale)); \
210 ret += ofs; \
211 \
212 __insn_barrier(); \
213 if (tsc_ref->tsc_seq == seq) \
214 return ret; \
215 \
216 /* Sequence changed; re-sync. */ \
217 } \
218 /* Fallback to the generic timecounter, i.e. rdmsr. */ \
219 return rdmsr(MSR_HV_TIME_REF_COUNT); \
220} \
221 \
222static u_int \
223hyperv_tsc_timecount_##fence(struct timecounter *tc __unused) \
224{ \
225 \
226 return hyperv_tc64_tsc_##fence(); \
227}

232static bool
233hyperv_tsc_tcinit(void)
234{
235 hyperv_tc64_t tc64 = NULL;
236 uint64_t orig_msr, msr;
237
238 if ((hyperv_features &
239 (CPUID_HV_MSR_TIME_REFCNT | CPUID_HV_MSR_REFERENCE_TSC)) !=
240 (CPUID_HV_MSR_TIME_REFCNT | CPUID_HV_MSR_REFERENCE_TSC) ||
241 (cpu_feature[0] & CPUID_SSE2) == 0) /* SSE2 for mfence/lfence */
242 return false;
243
244 switch (cpu_vendor) {
245 case CPUVENDOR_AMD:
246 hyperv_tsc_timecounter.tc_get_timecount =
247 hyperv_tsc_timecount_mfence;
248 tc64 = hyperv_tc64_tsc_mfence;
249 break;
250
251 case CPUVENDOR_INTEL:
252 hyperv_tsc_timecounter.tc_get_timecount =
253 hyperv_tsc_timecount_lfence;
254 tc64 = hyperv_tc64_tsc_lfence;
255 break;
256
257 default:
258 /* Unsupport CPU vendors. */
259 return false;
260 }
261
262 hyperv_ref_tsc.tsc_ref = (void *)uvm_km_alloc(kernel_map,
263 PAGE_SIZE, PAGE_SIZE, UVM_KMF_WIRED | UVM_KMF_ZERO);
264 if (hyperv_ref_tsc.tsc_ref == NULL) {
265 aprint_error("Hyper-V: reference TSC page allocation failed\n");
266 return false;
267 }
268
269 if (!pmap_extract(pmap_kernel(), (vaddr_t)hyperv_ref_tsc.tsc_ref,
270 &hyperv_ref_tsc.tsc_paddr)) {
271 aprint_error("Hyper-V: reference TSC page setup failed\n");
272 uvm_km_free(kernel_map, (vaddr_t)hyperv_ref_tsc.tsc_ref,
273 PAGE_SIZE, UVM_KMF_WIRED);
274 hyperv_ref_tsc.tsc_ref = NULL;
275 return false;
276 }
277
278 orig_msr = rdmsr(MSR_HV_REFERENCE_TSC);
279 msr = MSR_HV_REFTSC_ENABLE | (orig_msr & MSR_HV_REFTSC_RSVD_MASK) |
280 (atop(hyperv_ref_tsc.tsc_paddr) << MSR_HV_REFTSC_PGSHIFT);
281 wrmsr(MSR_HV_REFERENCE_TSC, msr);
282
283 /* Install 64 bits timecounter method for other modules to use. */
284 hyperv_tc64 = tc64;
285
286 /* Register "enlightened" timecounter. */
287 tc_init(&hyperv_tsc_timecounter);
288
289 return true;
290}

293static void
294delay_tc(unsigned int n)
295{
296 struct timecounter *tc;
297 uint64_t end, now;
298 u_int last, u;
299
300 tc = timecounter;
301 if (tc->tc_quality <= 0) {
302 x86_delay(n);
303 return;
304 }
305
306 now = 0;
307 end = tc->tc_frequency * n / 1000000;
308 last = tc->tc_get_timecount(tc) & tc->tc_counter_mask;
309 do {
310 x86_pause();
311 u = tc->tc_get_timecount(tc) & tc->tc_counter_mask;
312 if (u < last)
313 now += tc->tc_counter_mask - last + u + 1;
314 else
315 now += u - last;
316 last = u;
317 } while (now < end);
318}
319
320static void
321delay_msr(unsigned int n)
322{
323 uint64_t end, now;
324 u_int last, u;
325
326 now = 0;
327 end = HYPERV_TIMER_FREQ * n / 1000000ULL;
328 last = (u_int)rdmsr(MSR_HV_TIME_REF_COUNT);
329 do {
330 x86_pause();
331 u = (u_int)rdmsr(MSR_HV_TIME_REF_COUNT);
332 if (u < last)
333 now += 0xffffffff - last + u + 1;
334 else
335 now += u - last;
336 last = u;
337 } while (now < end);
338}

340static __inline uint64_t
341hyperv_hypercall_md(volatile void *hc_addr, uint64_t in_val, uint64_t in_paddr,
342 uint64_t out_paddr)
343{
344 uint64_t status;
345
346#ifdef __amd64__
347 __asm__ __volatile__ ("mov %0, %%r8" : : "r" (out_paddr): "r8");
348 __asm__ __volatile__ ("call *%3" : "=a" (status) : "c" (in_val),
349 "d" (in_paddr), "m" (hc_addr));
350#else
351 uint32_t in_val_hi = in_val >> 32;
352 uint32_t in_val_lo = in_val & 0xFFFFFFFF;
353 uint32_t status_hi, status_lo;
354 uint32_t in_paddr_hi = in_paddr >> 32;
355 uint32_t in_paddr_lo = in_paddr & 0xFFFFFFFF;
356 uint32_t out_paddr_hi = out_paddr >> 32;
357 uint32_t out_paddr_lo = out_paddr & 0xFFFFFFFF;
358
359 __asm__ __volatile__ ("call *%8" : "=d" (status_hi), "=a" (status_lo) :
360 "d" (in_val_hi), "a" (in_val_lo),
361 "b" (in_paddr_hi), "c" (in_paddr_lo),
362 "D" (out_paddr_hi), "S" (out_paddr_lo),
363 "m" (hc_addr));
364 status = status_lo | ((uint64_t)status_hi << 32);
365#endif
366
367 return status;
368}

370uint64_t
371hyperv_hypercall(uint64_t control, paddr_t in_paddr, paddr_t out_paddr)
372{
373
374 if (hyperv_hypercall_ctx.hc_addr == NULL)
375 return ~HYPERCALL_STATUS_SUCCESS;
376
377 return hyperv_hypercall_md(hyperv_hypercall_ctx.hc_addr, control,
378 in_paddr, out_paddr);
379}

381static bool
382hyperv_probe(u_int *maxleaf, u_int *features, u_int *pm_features,
383 u_int *features3)
384{
385 u_int regs[4];
386
387 if (vm_guest != VM_GUEST_HV)
388 return false;
389
390 x86_cpuid(CPUID_LEAF_HV_MAXLEAF, regs);
391 *maxleaf = regs[0];
392 if (*maxleaf < CPUID_LEAF_HV_LIMITS)
393 return false;
394
395 x86_cpuid(CPUID_LEAF_HV_INTERFACE, regs);
396 if (regs[0] != CPUID_HV_IFACE_HYPERV)
397 return false;
398
399 x86_cpuid(CPUID_LEAF_HV_FEATURES, regs);
400 if (!(regs[0] & CPUID_HV_MSR_HYPERCALL)) {
401 /*
402 * Hyper-V w/o Hypercall is impossible; someone
403 * is faking Hyper-V.
404 */
405 return false;
406 }
407
408 *features = regs[0];
409 *pm_features = regs[2];
410 *features3 = regs[3];
411
412 return true;
413}

415static bool
416hyperv_identify(void)
417{
418 char buf[256];
419 u_int regs[4];
420 u_int maxleaf;
421
422 if (!hyperv_probe(&maxleaf, &hyperv_features, &hyperv_pm_features,
423 &hyperv_features3))
424 return false;
425
426 x86_cpuid(CPUID_LEAF_HV_IDENTITY, regs);
427 hyperv_ver_major = regs[1] >> 16;
428 snprintf(hyperv_version_str, sizeof(hyperv_version_str),
429 "%d.%d.%d [SP%d]",
430 hyperv_ver_major, regs[1] & 0xffff, regs[0], regs[2]);
431 aprint_verbose("Hyper-V Version: %s\n", hyperv_version_str);
432
433 snprintb(hyperv_features_str, sizeof(hyperv_features_str),
434 "\020"
435 "\001VPRUNTIME" /* MSR_HV_VP_RUNTIME */
436 "\002TMREFCNT" /* MSR_HV_TIME_REF_COUNT */
437 "\003SYNIC" /* MSRs for SynIC */
438 "\004SYNTM" /* MSRs for SynTimer */
439 "\005APIC" /* MSR_HV_{EOI,ICR,TPR} */
440 "\006HYPERCALL" /* MSR_HV_{GUEST_OS_ID,HYPERCALL} */
441 "\007VPINDEX" /* MSR_HV_VP_INDEX */
442 "\010RESET" /* MSR_HV_RESET */
443 "\011STATS" /* MSR_HV_STATS_ */
444 "\012REFTSC" /* MSR_HV_REFERENCE_TSC */
445 "\013IDLE" /* MSR_HV_GUEST_IDLE */
446 "\014TMFREQ" /* MSR_HV_{TSC,APIC}_FREQUENCY */
447 "\015DEBUG", /* MSR_HV_SYNTH_DEBUG_ */
448 hyperv_features);
449 aprint_verbose(" Features=%s\n", hyperv_features_str);
450 snprintb(buf, sizeof(buf),
451 "\020"
452 "\005C3HPET", /* HPET is required for C3 state */
453 (hyperv_pm_features & ~CPUPM_HV_CSTATE_MASK));
454 snprintf(hyperv_pm_features_str, sizeof(hyperv_pm_features_str),
455 "%s [C%u]", buf, CPUPM_HV_CSTATE(hyperv_pm_features));
456 aprint_verbose(" PM Features=%s\n", hyperv_pm_features_str);
457 snprintb(hyperv_features3_str, sizeof(hyperv_features3_str),
458 "\020"
459 "\001MWAIT" /* MWAIT */
460 "\002DEBUG" /* guest debug support */
461 "\003PERFMON" /* performance monitor */
462 "\004PCPUDPE" /* physical CPU dynamic partition event */
463 "\005XMMHC" /* hypercall input through XMM regs */
464 "\006IDLE" /* guest idle support */
465 "\007SLEEP" /* hypervisor sleep support */
466 "\010NUMA" /* NUMA distance query support */
467 "\011TMFREQ" /* timer frequency query (TSC, LAPIC) */
468 "\012SYNCMC" /* inject synthetic machine checks */
469 "\013CRASH" /* MSRs for guest crash */
470 "\014DEBUGMSR" /* MSRs for guest debug */
471 "\015NPIEP" /* NPIEP */
472 "\016HVDIS", /* disabling hypervisor */
473 hyperv_features3);
474 aprint_verbose(" Features3=%s\n", hyperv_features3_str);
475
476 x86_cpuid(CPUID_LEAF_HV_RECOMMENDS, regs);
477 hyperv_recommends = regs[0];
478 aprint_verbose(" Recommends: %08x %08x\n", regs[0], regs[1]);
479
480 x86_cpuid(CPUID_LEAF_HV_LIMITS, regs);
481 aprint_verbose(" Limits: Vcpu:%d Lcpu:%d Int:%d\n",
482 regs[0], regs[1], regs[2]);
483
484 if (maxleaf >= CPUID_LEAF_HV_HWFEATURES) {
485 x86_cpuid(CPUID_LEAF_HV_HWFEATURES, regs);
486 aprint_verbose(" HW Features: %08x, AMD: %08x\n",
487 regs[0], regs[3]);
488 }
489
490 return true;
491}

493void
494hyperv_early_init(void)
495{
496 u_int features, pm_features, features3;
497 u_int maxleaf;
498
499 if (!hyperv_probe(&maxleaf, &features, &pm_features, &features3))
500 return;
501
502 if (features & CPUID_HV_MSR_TIME_REFCNT)
503 x86_delay = delay_func = delay_msr;
504}

506static bool
507hyperv_init(void)
508{
509
510 if (!hyperv_identify()) {
511 /* Not Hyper-V; reset guest id to the generic one. */
512 if (vm_guest == VM_GUEST_HV)
513 vm_guest = VM_GUEST_VM;
514 return false;
515 }
516
517 /* Set guest id */
518 wrmsr(MSR_HV_GUEST_OS_ID, MSR_HV_GUESTID_OSTYPE_NETBSD |
519 (uint64_t)__NetBSD_Version__ << MSR_HV_GUESTID_VERSION_SHIFT);
520
521 if (hyperv_features & CPUID_HV_MSR_TIME_REFCNT) {
522 /* Register Hyper-V timecounter */
523 tc_init(&hyperv_timecounter);
524
525 /*
526 * Install 64 bits timecounter method for other modules to use.
527 */
528 hyperv_tc64 = hyperv_tc64_rdmsr;
529#ifdef __amd64__
530 hyperv_tsc_tcinit();
531#endif
532
533 /* delay with timecounter */
534 x86_delay = delay_func = delay_tc;
535 }
536
537#if NLAPIC > 0
538 if ((hyperv_features & CPUID_HV_MSR_TIME_FREQ) &&
539 (hyperv_features3 & CPUID3_HV_TIME_FREQ)) {
540 extern uint32_t lapic_per_second;
541
542 lapic_per_second = rdmsr(MSR_HV_APIC_FREQUENCY);
543 }
544#endif
545
546 return hyperv_init_hypercall();
547}

549static bool
550hyperv_is_initialized(void)
551{
552 uint64_t msr;
553
554 if (vm_guest != VM_GUEST_HV)
555 return false;
556 if (rdmsr_safe(MSR_HV_HYPERCALL, &msr) == EFAULT)
557 return false;
558 return (msr & MSR_HV_HYPERCALL_ENABLE) ? true : false;
559}

561static int
562hyperv_match(device_t parent, cfdata_t cf, void *aux)
563{
564 struct cpufeature_attach_args *cfaa = aux;
565 struct cpu_info *ci = cfaa->ci;
566
567 if (strcmp(cfaa->name, "vm") != 0)
568 return 0;
569 if ((ci->ci_flags & (CPUF_BSP|CPUF_SP|CPUF_PRIMARY)) == 0)
570 return 0;
571 if (vm_guest != VM_GUEST_HV)
572 return 0;
573
574 return 1;
575}

577static void
578hyperv_attach(device_t parent, device_t self, void *aux)
579{
580 struct hyperv_softc *sc = device_private(self);
581
582 sc->sc_dev = self;
583
584 aprint_naive("\n");
585 aprint_normal(": Hyper-V\n");
586
587 if (!hyperv_is_initialized()) {
588 if (rdmsr(MSR_HV_GUEST_OS_ID) == 0) {
589 if (!hyperv_init()) {
590 aprint_error_dev(self, "initialize failed\n");
591 return;
592 }
593 }
594 hyperv_init_hypercall();
595 }
596
597 (void) pmf_device_register(self, NULL, NULL);
598
599 (void) hyperv_sysctl_setup_root(sc);
600}

602static int
603hyperv_detach(device_t self, int flags)
604{
605 struct hyperv_softc *sc = device_private(self);
606 uint64_t hc;
607
608 /* Disable Hypercall */
609 hc = rdmsr(MSR_HV_HYPERCALL);
610 wrmsr(MSR_HV_HYPERCALL, hc & MSR_HV_HYPERCALL_RSVD_MASK);
611 hyperv_hypercall_memfree();
612
613 if (hyperv_features & CPUID_HV_MSR_TIME_REFCNT)
614 tc_detach(&hyperv_timecounter);
615
616 wrmsr(MSR_HV_GUEST_OS_ID, 0);
617
618 pmf_device_deregister(self);
619
620 if (sc->sc_log != NULL) {
621 sysctl_teardown(&sc->sc_log);
622 sc->sc_log = NULL;
623 }
624
625 return 0;
626}

680static bool
681hyperv_init_hypercall(void)
682{
683 uint64_t hc, hc_orig;
684
685 hyperv_hypercall_ctx.hc_addr = (void *)uvm_km_alloc(kernel_map,
686 PAGE_SIZE, PAGE_SIZE,
687 UVM_KMF_WIRED | UVM_KMF_EXEC | (cold ? UVM_KMF_NOWAIT : 0));
688 if (hyperv_hypercall_ctx.hc_addr == NULL) {
689 aprint_error("Hyper-V: Hypercall page allocation failed\n");
690 return false;
691 }
692
693 memset(hyperv_hypercall_ctx.hc_addr, 0xcc, PAGE_SIZE);
694 wbinvd();
695 x86_flush();
696
697 /* The hypercall page must be both readable and executable */
698 uvm_km_protect(kernel_map, (vaddr_t)hyperv_hypercall_ctx.hc_addr,
699 PAGE_SIZE, VM_PROT_READ | VM_PROT_EXECUTE);
700
701 if (!pmap_extract(pmap_kernel(), (vaddr_t)hyperv_hypercall_ctx.hc_addr,
702 &hyperv_hypercall_ctx.hc_paddr)) {
703 aprint_error("Hyper-V: Hypercall page setup failed\n");
704 hyperv_hypercall_memfree();
705 /* Can't perform any Hyper-V specific actions */
706 vm_guest = VM_GUEST_VM;
707 return false;
708 }
709
710 /* Get the 'reserved' bits, which requires preservation. */
711 hc_orig = rdmsr(MSR_HV_HYPERCALL);
712
713 /*
714 * Setup the Hypercall page.
715 *
716 * NOTE: 'reserved' bits MUST be preserved.
717 */
718 hc = (atop(hyperv_hypercall_ctx.hc_paddr) << MSR_HV_HYPERCALL_PGSHIFT) |
719 (hc_orig & MSR_HV_HYPERCALL_RSVD_MASK) |
720 MSR_HV_HYPERCALL_ENABLE;
721 wrmsr(MSR_HV_HYPERCALL, hc);
722
723 /*
724 * Confirm that Hypercall page did get setup.
725 */
726 hc = rdmsr(MSR_HV_HYPERCALL);
727 if (!(hc & MSR_HV_HYPERCALL_ENABLE)) {
728 aprint_error("Hyper-V: Hypercall setup failed\n");
729 hyperv_hypercall_memfree();
730 /* Can't perform any Hyper-V specific actions */
731 vm_guest = VM_GUEST_VM;
732 return false;
733 }
734
735 return true;
736}

792void
793vmbus_init_synic_md(struct vmbus_softc *sc, cpuid_t cpu)
794{
795 struct vmbus_percpu_data *pd;
796 uint64_t val, orig;
797 uint32_t sint;
798
799 pd = &sc->sc_percpu[cpu];
800
801 if (hyperv_features & CPUID_HV_MSR_VP_INDEX) {
802 /* Save virtual processor id. */
803 pd->vcpuid = rdmsr(MSR_HV_VP_INDEX);
804 } else {
805 /* Set virtual processor id to 0 for compatibility. */
806 pd->vcpuid = 0;
807 }
808
809 /*
810 * Setup the SynIC message.
811 */
812 orig = rdmsr(MSR_HV_SIMP);
813 val = MSR_HV_SIMP_ENABLE | (orig & MSR_HV_SIMP_RSVD_MASK) |
814 (atop(hyperv_dma_get_paddr(&pd->simp_dma)) << MSR_HV_SIMP_PGSHIFT);
815 wrmsr(MSR_HV_SIMP, val);
816
817 /*
818 * Setup the SynIC event flags.
819 */
820 orig = rdmsr(MSR_HV_SIEFP);
821 val = MSR_HV_SIEFP_ENABLE | (orig & MSR_HV_SIEFP_RSVD_MASK) |
822 (atop(hyperv_dma_get_paddr(&pd->siep_dma)) << MSR_HV_SIEFP_PGSHIFT);
823 wrmsr(MSR_HV_SIEFP, val);
824
825 /*
826 * Configure and unmask SINT for message and event flags.
827 */
828 sint = MSR_HV_SINT0 + VMBUS_SINT_MESSAGE;
829 orig = rdmsr(sint);
830 val = hyperv_idtvec | MSR_HV_SINT_AUTOEOI |
831 (orig & MSR_HV_SINT_RSVD_MASK);
832 wrmsr(sint, val);
833
834 /*
835 * Configure and unmask SINT for timer.
836 */
837 sint = MSR_HV_SINT0 + VMBUS_SINT_TIMER;
838 orig = rdmsr(sint);
839 val = hyperv_idtvec | MSR_HV_SINT_AUTOEOI |
840 (orig & MSR_HV_SINT_RSVD_MASK);
841 wrmsr(sint, val);
842
843 /*
844 * All done; enable SynIC.
845 */
846 orig = rdmsr(MSR_HV_SCONTROL);
847 val = MSR_HV_SCTRL_ENABLE | (orig & MSR_HV_SCTRL_RSVD_MASK);
848 wrmsr(MSR_HV_SCONTROL, val);
849}
850
851void
852vmbus_deinit_synic_md(struct vmbus_softc *sc, cpuid_t cpu)
853{
854 uint64_t orig;
855 uint32_t sint;
856
857 /*
858 * Disable SynIC.
859 */
860 orig = rdmsr(MSR_HV_SCONTROL);
861 wrmsr(MSR_HV_SCONTROL, (orig & MSR_HV_SCTRL_RSVD_MASK));
862
863 /*
864 * Mask message and event flags SINT.
865 */
866 sint = MSR_HV_SINT0 + VMBUS_SINT_MESSAGE;
867 orig = rdmsr(sint);
868 wrmsr(sint, orig | MSR_HV_SINT_MASKED);
869
870 /*
871 * Mask timer SINT.
872 */
873 sint = MSR_HV_SINT0 + VMBUS_SINT_TIMER;
874 orig = rdmsr(sint);
875 wrmsr(sint, orig | MSR_HV_SINT_MASKED);
876
877 /*
878 * Teardown SynIC message.
879 */
880 orig = rdmsr(MSR_HV_SIMP);
881 wrmsr(MSR_HV_SIMP, (orig & MSR_HV_SIMP_RSVD_MASK));
882
883 /*
884 * Teardown SynIC event flags.
885 */
886 orig = rdmsr(MSR_HV_SIEFP);
887 wrmsr(MSR_HV_SIEFP, (orig & MSR_HV_SIEFP_RSVD_MASK));
888}

890static int
891hyperv_sysctl_setup(struct hyperv_softc *sc,
892 const struct sysctlnode *hyperv_node)
893{
894 int error;
895
896 error = sysctl_createv(&sc->sc_log, 0, &hyperv_node, NULL,
897 CTLFLAG_READONLY, CTLTYPE_STRING, "version", NULL,
898 NULL, 0, hyperv_version_str,
899 0, CTL_CREATE, CTL_EOL);
900 if (error)
901 return error;
902
903 error = sysctl_createv(&sc->sc_log, 0, &hyperv_node, NULL,
904 CTLFLAG_READONLY, CTLTYPE_STRING, "features", NULL,
905 NULL, 0, hyperv_features_str,
906 0, CTL_CREATE, CTL_EOL);
907 if (error)
908 return error;
909
910 error = sysctl_createv(&sc->sc_log, 0, &hyperv_node, NULL,
911 CTLFLAG_READONLY, CTLTYPE_STRING, "pm_features", NULL,
912 NULL, 0, hyperv_pm_features_str,
913 0, CTL_CREATE, CTL_EOL);
914 if (error)
915 return error;
916
917 error = sysctl_createv(&sc->sc_log, 0, &hyperv_node, NULL,
918 CTLFLAG_READONLY, CTLTYPE_STRING, "features3", NULL,
919 NULL, 0, hyperv_features3_str,
920 0, CTL_CREATE, CTL_EOL);
921 if (error)
922 return error;
923
924 return 0;
925}

927static int
928hyperv_sysctl_setup_root(struct hyperv_softc *sc)
929{
930 const struct sysctlnode *machdep_node, *hyperv_node;
931 int error;
932
933 error = sysctl_createv(&sc->sc_log, 0, NULL, &machdep_node,
934 CTLFLAG_PERMANENT, CTLTYPE_NODE, "machdep", NULL,
935 NULL, 0, NULL, 0, CTL_MACHDEP, CTL_EOL);
936 if (error)
937 goto fail;
938
939 error = sysctl_createv(&sc->sc_log, 0, &machdep_node, &hyperv_node,
940 CTLFLAG_PERMANENT, CTLTYPE_NODE, "hyperv", NULL,
941 NULL, 0, NULL, 0, CTL_CREATE, CTL_EOL);
942 if (error)
943 goto fail;
944
945 error = hyperv_sysctl_setup(sc, hyperv_node);
946 if (error)
947 goto fail;
948
949 return 0;
950
951fail:
952 sysctl_teardown(&sc->sc_log);
953 sc->sc_log = NULL;
954 return error;
955}

1108 * without the address property

1118#endif
1119 prop_dictionary_set_bool(dict, "is_console", true);

1125 x86_genfb_console_screen.scr_ri.ri_crow);
