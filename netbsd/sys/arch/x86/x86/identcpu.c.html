99static const struct x86_cache_info *
100cache_info_lookup(const struct x86_cache_info *cai, uint8_t desc)
101{
102 int i;
103
104 for (i = 0; cai[i].cai_desc != 0; i++) {
105 if (cai[i].cai_desc == desc)
106 return (&cai[i]);
107 }
108
109 return (NULL);
110}

112static void
113cpu_probe_intel_cache(struct cpu_info *ci)
114{
115 const struct x86_cache_info *cai;
116 u_int descs[4];
117 int iterations, i, j;
118 uint8_t desc;
119
120 if (cpuid_level >= 2) {
121 /* Parse the cache info from `cpuid leaf 2', if we have it. */
122 x86_cpuid(2, descs);
123 iterations = descs[0] & 0xff;
124 while (iterations-- > 0) {
125 for (i = 0; i < 4; i++) {
126 if (descs[i] & 0x80000000)
127 continue;
128 for (j = 0; j < 4; j++) {
129 if (i == 0 && j == 0)
130 continue;
131 desc = (descs[i] >> (j * 8)) & 0xff;
132 if (desc == 0)
133 continue;
134 cai = cache_info_lookup(
135 intel_cpuid_cache_info, desc);
136 if (cai != NULL) {
137 ci->ci_cinfo[cai->cai_index] =
138 *cai;
139 }
140 }
141 }
142 }
143 }
144
145 if (cpuid_level >= 4) {
146 int type, level;
147 int ways, partitions, linesize, sets;
148 int caitype = -1;
149 int totalsize;
150
151 /* Parse the cache info from `cpuid leaf 4', if we have it. */
152 for (i = 0; ; i++) {
153 x86_cpuid2(4, i, descs);
154 type = __SHIFTOUT(descs[0], CPUID_DCP_CACHETYPE);
155 if (type == CPUID_DCP_CACHETYPE_N)
156 break;
157 level = __SHIFTOUT(descs[0], CPUID_DCP_CACHELEVEL);
158 switch (level) {
159 case 1:
160 if (type == CPUID_DCP_CACHETYPE_I)
161 caitype = CAI_ICACHE;
162 else if (type == CPUID_DCP_CACHETYPE_D)
163 caitype = CAI_DCACHE;
164 else
165 caitype = -1;
166 break;
167 case 2:
168 if (type == CPUID_DCP_CACHETYPE_U)
169 caitype = CAI_L2CACHE;
170 else
171 caitype = -1;
172 break;
173 case 3:
174 if (type == CPUID_DCP_CACHETYPE_U)
175 caitype = CAI_L3CACHE;
176 else
177 caitype = -1;
178 break;
179 default:
180 caitype = -1;
181 break;
182 }
183 if (caitype == -1)
184 continue;
185
186 ways = __SHIFTOUT(descs[1], CPUID_DCP_WAYS) + 1;
187 partitions =__SHIFTOUT(descs[1], CPUID_DCP_PARTITIONS)
188 + 1;
189 linesize = __SHIFTOUT(descs[1], CPUID_DCP_LINESIZE)
190 + 1;
191 sets = descs[2] + 1;
192 totalsize = ways * partitions * linesize * sets;
193 ci->ci_cinfo[caitype].cai_totalsize = totalsize;
194 ci->ci_cinfo[caitype].cai_associativity = ways;
195 ci->ci_cinfo[caitype].cai_linesize = linesize;
196 }
197 }
198}

200static void
201cpu_probe_intel_errata(struct cpu_info *ci)
202{
203 u_int family, model, stepping;
204
205 family = CPUID_TO_FAMILY(ci->ci_signature);
206 model = CPUID_TO_MODEL(ci->ci_signature);
207 stepping = CPUID_TO_STEPPING(ci->ci_signature);
208
209 if (family == 0x6 && model == 0x5C && stepping == 0x9) { /* Apollo Lake */
210 wrmsr(MSR_MISC_ENABLE,
211 rdmsr(MSR_MISC_ENABLE) & ~IA32_MISC_MWAIT_EN);
212
213 cpu_feature[1] &= ~CPUID2_MONITOR;
214 ci->ci_feat_val[1] &= ~CPUID2_MONITOR;
215 }
216}

229static void
230cpu_probe_amd_cache(struct cpu_info *ci)
231{
232 const struct x86_cache_info *cp;
233 struct x86_cache_info *cai;
234 int family, model;
235 u_int descs[4];
236 u_int lfunc;
237
238 family = CPUID_TO_FAMILY(ci->ci_signature);
239 model = CPUID_TO_MODEL(ci->ci_signature);
240
241 /*
242 * K5 model 0 has none of this info.
243 */
244 if (family == 5 && model == 0)
245 return;
246
247 /*
248 * Determine the largest extended function value.
249 */
250 x86_cpuid(0x80000000, descs);
251 lfunc = descs[0];
252
253 /*
254 * Determine L1 cache/TLB info.
255 */
256 if (lfunc < 0x80000005) {
257 /* No L1 cache info available. */
258 return;
259 }
260
261 x86_cpuid(0x80000005, descs);
262
263 /*
264 * K6-III and higher have large page TLBs.
265 */
266 if ((family == 5 && model >= 9) || family >= 6) {
267 cai = &ci->ci_cinfo[CAI_ITLB2];
268 cai->cai_totalsize = AMD_L1_EAX_ITLB_ENTRIES(descs[0]);
269 cai->cai_associativity = AMD_L1_EAX_ITLB_ASSOC(descs[0]);
270 cai->cai_linesize = (4 * 1024 * 1024);
271
272 cai = &ci->ci_cinfo[CAI_DTLB2];
273 cai->cai_totalsize = AMD_L1_EAX_DTLB_ENTRIES(descs[0]);
274 cai->cai_associativity = AMD_L1_EAX_DTLB_ASSOC(descs[0]);
275 cai->cai_linesize = (4 * 1024 * 1024);
276 }
277
278 cai = &ci->ci_cinfo[CAI_ITLB];
279 cai->cai_totalsize = AMD_L1_EBX_ITLB_ENTRIES(descs[1]);
280 cai->cai_associativity = AMD_L1_EBX_ITLB_ASSOC(descs[1]);
281 cai->cai_linesize = (4 * 1024);
282
283 cai = &ci->ci_cinfo[CAI_DTLB];
284 cai->cai_totalsize = AMD_L1_EBX_DTLB_ENTRIES(descs[1]);
285 cai->cai_associativity = AMD_L1_EBX_DTLB_ASSOC(descs[1]);
286 cai->cai_linesize = (4 * 1024);
287
288 cai = &ci->ci_cinfo[CAI_DCACHE];
289 cai->cai_totalsize = AMD_L1_ECX_DC_SIZE(descs[2]);
290 cai->cai_associativity = AMD_L1_ECX_DC_ASSOC(descs[2]);
291 cai->cai_linesize = AMD_L1_ECX_DC_LS(descs[2]);
292
293 cai = &ci->ci_cinfo[CAI_ICACHE];
294 cai->cai_totalsize = AMD_L1_EDX_IC_SIZE(descs[3]);
295 cai->cai_associativity = AMD_L1_EDX_IC_ASSOC(descs[3]);
296 cai->cai_linesize = AMD_L1_EDX_IC_LS(descs[3]);
297
298 /*
299 * Determine L2 cache/TLB info.
300 */
301 if (lfunc < 0x80000006) {
302 /* No L2 cache info available. */
303 return;
304 }
305
306 x86_cpuid(0x80000006, descs);
307
308 cai = &ci->ci_cinfo[CAI_L2CACHE];
309 cai->cai_totalsize = AMD_L2_ECX_C_SIZE(descs[2]);
310 cai->cai_associativity = AMD_L2_ECX_C_ASSOC(descs[2]);
311 cai->cai_linesize = AMD_L2_ECX_C_LS(descs[2]);
312
313 cp = cache_info_lookup(amd_cpuid_l2cache_assoc_info,
314 cai->cai_associativity);
315 if (cp != NULL)
316 cai->cai_associativity = cp->cai_associativity;
317 else
318 cai->cai_associativity = 0; /* XXX Unknown/reserved */
319
320 if (family < 0xf) {
321 /* No L3 cache info available. */
322 return;
323 }
324
325 cai = &ci->ci_cinfo[CAI_L3CACHE];
326 cai->cai_totalsize = AMD_L3_EDX_C_SIZE(descs[3]);
327 cai->cai_associativity = AMD_L3_EDX_C_ASSOC(descs[3]);
328 cai->cai_linesize = AMD_L3_EDX_C_LS(descs[3]);
329
330 cp = cache_info_lookup(amd_cpuid_l3cache_assoc_info,
331 cai->cai_associativity);
332 if (cp != NULL)
333 cai->cai_associativity = cp->cai_associativity;
334 else
335 cai->cai_associativity = 0; /* XXX Unknown reserved */
336
337 if (lfunc < 0x80000019) {
338 /* No 1GB Page TLB */
339 return;
340 }
341
342 x86_cpuid(0x80000019, descs);
343
344 cai = &ci->ci_cinfo[CAI_L1_1GBDTLB];
345 cai->cai_totalsize = AMD_L1_1GB_EAX_DTLB_ENTRIES(descs[1]);
346 cai->cai_associativity = AMD_L1_1GB_EAX_DTLB_ASSOC(descs[1]);
347 cai->cai_linesize = (1 * 1024);
348
349 cai = &ci->ci_cinfo[CAI_L1_1GBITLB];
350 cai->cai_totalsize = AMD_L1_1GB_EAX_IUTLB_ENTRIES(descs[0]);
351 cai->cai_associativity = AMD_L1_1GB_EAX_IUTLB_ASSOC(descs[0]);
352 cai->cai_linesize = (1 * 1024);
353
354 cai = &ci->ci_cinfo[CAI_L2_1GBDTLB];
355 cai->cai_totalsize = AMD_L2_1GB_EBX_DUTLB_ENTRIES(descs[1]);
356 cai->cai_associativity = AMD_L2_1GB_EBX_DUTLB_ASSOC(descs[1]);
357 cai->cai_linesize = (1 * 1024);
358
359 cai = &ci->ci_cinfo[CAI_L2_1GBITLB];
360 cai->cai_totalsize = AMD_L2_1GB_EBX_IUTLB_ENTRIES(descs[0]);
361 cai->cai_associativity = AMD_L2_1GB_EBX_IUTLB_ASSOC(descs[0]);
362 cai->cai_linesize = (1 * 1024);
363}

365static void
366cpu_probe_amd(struct cpu_info *ci)
367{
368 uint64_t val;
369 int flag;
370
371 if (cpu_vendor != CPUVENDOR_AMD)
372 return;
373 if (CPUID_TO_FAMILY(ci->ci_signature) < 5)
374 return;
375
376 switch (CPUID_TO_FAMILY(ci->ci_signature)) {
377 case 0x05: /* K5 */
378 if (CPUID_TO_MODEL(ci->ci_signature) == 0) {
379 /*
380 * According to the AMD Processor Recognition App Note,
381 * the AMD-K5 Model 0 uses the wrong bit to indicate
382 * support for global PTEs, instead using bit 9 (APIC)
383 * rather than bit 13 (i.e. "0x200" vs. 0x2000").
384 */
385 flag = ci->ci_feat_val[0];
386 if ((flag & CPUID_APIC) != 0)
387 flag = (flag & ~CPUID_APIC) | CPUID_PGE;
388 ci->ci_feat_val[0] = flag;
389 }
390 break;
391
392 case 0x10: /* Family 10h */
393 /*
394 * On Family 10h, certain BIOSes do not enable WC+ support.
395 * This causes WC+ to become CD, and degrades guest
396 * performance at the NPT level.
397 *
398 * Explicitly enable WC+ if we're not a guest.
399 */
400 if (!ISSET(ci->ci_feat_val[1], CPUID2_RAZ)) {
401 val = rdmsr(MSR_BU_CFG2);
402 val &= ~BU_CFG2_CWPLUS_DIS;
403 wrmsr(MSR_BU_CFG2, val);
404 }
405 break;
406 }
407
408 cpu_probe_amd_cache(ci);
409}

427static void
428cpu_probe_cyrix_cmn(struct cpu_info *ci)
429{
430 /*
431 * i8254 latch check routine:
432 * National Geode (formerly Cyrix MediaGX) has a serious bug in
433 * its built-in i8254-compatible clock module (cs5510 cs5520).
434 * Set the variable 'clock_broken_latch' to indicate it.
435 *
436 * This bug is not present in the cs5530, and the flag
437 * is disabled again in sys/arch/i386/pci/pcib.c if this later
438 * model device is detected. Ideally, this work-around should not
439 * even be in here, it should be in there. XXX
440 */
441 uint8_t c3;
442#ifndef XEN
443 extern int clock_broken_latch;
444
445 switch (ci->ci_signature) {
446 case 0x440: /* Cyrix MediaGX */
447 case 0x540: /* GXm */
448 clock_broken_latch = 1;
449 break;
450 }
451#endif
452
453 /* set up various cyrix registers */
454 /*
455 * Enable suspend on halt (powersave mode).
456 * When powersave mode is enabled, the TSC stops counting
457 * while the CPU is halted in idle() waiting for an interrupt.
458 * This means we can't use the TSC for interval time in
459 * microtime(9), and thus it is disabled here.
460 *
461 * It still makes a perfectly good cycle counter
462 * for program profiling, so long as you remember you're
463 * counting cycles, and not time. Further, if you don't
464 * mind not using powersave mode, the TSC works just fine,
465 * so this should really be optional. XXX
466 */
467 cyrix_write_reg(0xc2, cyrix_read_reg(0xc2) | 0x08);
468
469 /*
470 * Do not disable the TSC on the Geode GX, it's reported to
471 * work fine.
472 */
473 if (ci->ci_signature != 0x552)
474 ci->ci_feat_val[0] &= ~CPUID_TSC;
475
476 /* enable access to ccr4/ccr5 */
477 c3 = cyrix_read_reg(0xC3);
478 cyrix_write_reg(0xC3, c3 | 0x10);
479 /* cyrix's workaround for the "coma bug" */
480 cyrix_write_reg(0x31, cyrix_read_reg(0x31) | 0xf8);
481 cyrix_write_reg(0x32, cyrix_read_reg(0x32) | 0x7f);
482 cyrix_write_reg(0x33, cyrix_read_reg(0x33) & ~0xff);
483 cyrix_write_reg(0x3c, cyrix_read_reg(0x3c) | 0x87);
484 /* disable access to ccr4/ccr5 */
485 cyrix_write_reg(0xC3, c3);
486}

513static void
514cpu_probe_c3(struct cpu_info *ci)
515{
516 u_int family, model, stepping, descs[4], lfunc, msr;
517 struct x86_cache_info *cai;
518
519 if (cpu_vendor != CPUVENDOR_IDT ||
520 CPUID_TO_FAMILY(ci->ci_signature) < 6)
521 return;
522
523 family = CPUID_TO_FAMILY(ci->ci_signature);
524 model = CPUID_TO_MODEL(ci->ci_signature);
525 stepping = CPUID_TO_STEPPING(ci->ci_signature);
526
527 /* Determine the largest extended function value. */
528 x86_cpuid(0x80000000, descs);
529 lfunc = descs[0];
530
531 if (family == 6) {
532 /*
533 * VIA Eden ESP.
534 *
535 * Quoting from page 3-4 of: "VIA Eden ESP Processor Datasheet"
536 * http://www.via.com.tw/download/mainboards/6/14/Eden20v115.pdf
537 *
538 * 1. The CMPXCHG8B instruction is provided and always enabled,
539 * however, it appears disabled in the corresponding CPUID
540 * function bit 0 to avoid a bug in an early version of
541 * Windows NT. However, this default can be changed via a
542 * bit in the FCR MSR.
543 */
544 ci->ci_feat_val[0] |= CPUID_CX8;
545 wrmsr(MSR_VIA_FCR, rdmsr(MSR_VIA_FCR) | VIA_ACE_ECX8);
546 }
547
548 if (family > 6 || model > 0x9 || (model == 0x9 && stepping >= 3)) {
549 /* VIA Nehemiah or Esther. */
550 x86_cpuid(0xc0000000, descs);
551 lfunc = descs[0];
552 if (lfunc >= 0xc0000001) { /* has ACE, RNG */
553 int rng_enable = 0, ace_enable = 0;
554 x86_cpuid(0xc0000001, descs);
555 lfunc = descs[3];
556 ci->ci_feat_val[4] = lfunc;
557 /* Check for and enable RNG */
558 if (lfunc & CPUID_VIA_HAS_RNG) {
559 if (!(lfunc & CPUID_VIA_DO_RNG)) {
560 rng_enable++;
561 ci->ci_feat_val[4] |= CPUID_VIA_DO_RNG;
562 }
563 }
564 /* Check for and enable ACE (AES-CBC) */
565 if (lfunc & CPUID_VIA_HAS_ACE) {
566 if (!(lfunc & CPUID_VIA_DO_ACE)) {
567 ace_enable++;
568 ci->ci_feat_val[4] |= CPUID_VIA_DO_ACE;
569 }
570 }
571 /* Check for and enable SHA */
572 if (lfunc & CPUID_VIA_HAS_PHE) {
573 if (!(lfunc & CPUID_VIA_DO_PHE)) {
574 ace_enable++;
575 ci->ci_feat_val[4] |= CPUID_VIA_DO_PHE;
576 }
577 }
578 /* Check for and enable ACE2 (AES-CTR) */
579 if (lfunc & CPUID_VIA_HAS_ACE2) {
580 if (!(lfunc & CPUID_VIA_DO_ACE2)) {
581 ace_enable++;
582 ci->ci_feat_val[4] |= CPUID_VIA_DO_ACE2;
583 }
584 }
585 /* Check for and enable PMM (modmult engine) */
586 if (lfunc & CPUID_VIA_HAS_PMM) {
587 if (!(lfunc & CPUID_VIA_DO_PMM)) {
588 ace_enable++;
589 ci->ci_feat_val[4] |= CPUID_VIA_DO_PMM;
590 }
591 }
592
593 /*
594 * Actually do the enables. It's a little gross,
595 * but per the PadLock programming guide, "Enabling
596 * PadLock", condition 3, we must enable SSE too or
597 * else the first use of RNG or ACE instructions
598 * will generate a trap.
599 *
600 * We must do this early because of kernel RNG
601 * initialization but it is safe without the full
602 * FPU-detect as all these CPUs have SSE.
603 */
604 lcr4(rcr4() | CR4_OSFXSR);
605
606 if (rng_enable) {
607 msr = rdmsr(MSR_VIA_RNG);
608 msr |= MSR_VIA_RNG_ENABLE;
609 /* C7 stepping 8 and subsequent CPUs have dual RNG */
610 if (model > 0xA || (model == 0xA && stepping > 0x7)) {
611 msr |= MSR_VIA_RNG_2NOISE;
612 }
613 wrmsr(MSR_VIA_RNG, msr);
614 }
615
616 if (ace_enable) {
617 msr = rdmsr(MSR_VIA_ACE);
618 wrmsr(MSR_VIA_ACE, msr | VIA_ACE_ENABLE);
619 }
620 }
621 }
622
623 /* Explicitly disable unsafe ALTINST mode. */
624 if (ci->ci_feat_val[4] & CPUID_VIA_DO_ACE) {
625 msr = rdmsr(MSR_VIA_ACE);
626 wrmsr(MSR_VIA_ACE, msr & ~VIA_ACE_ALTINST);
627 }
628
629 /*
630 * Determine L1 cache/TLB info.
631 */
632 if (lfunc < 0x80000005) {
633 /* No L1 cache info available. */
634 return;
635 }
636
637 x86_cpuid(0x80000005, descs);
638
639 cai = &ci->ci_cinfo[CAI_ITLB];
640 cai->cai_totalsize = VIA_L1_EBX_ITLB_ENTRIES(descs[1]);
641 cai->cai_associativity = VIA_L1_EBX_ITLB_ASSOC(descs[1]);
642 cai->cai_linesize = (4 * 1024);
643
644 cai = &ci->ci_cinfo[CAI_DTLB];
645 cai->cai_totalsize = VIA_L1_EBX_DTLB_ENTRIES(descs[1]);
646 cai->cai_associativity = VIA_L1_EBX_DTLB_ASSOC(descs[1]);
647 cai->cai_linesize = (4 * 1024);
648
649 cai = &ci->ci_cinfo[CAI_DCACHE];
650 cai->cai_totalsize = VIA_L1_ECX_DC_SIZE(descs[2]);
651 cai->cai_associativity = VIA_L1_ECX_DC_ASSOC(descs[2]);
652 cai->cai_linesize = VIA_L1_EDX_IC_LS(descs[2]);
653 if (family == 6 && model == 9 && stepping == 8) {
654 /* Erratum: stepping 8 reports 4 when it should be 2 */
655 cai->cai_associativity = 2;
656 }
657
658 cai = &ci->ci_cinfo[CAI_ICACHE];
659 cai->cai_totalsize = VIA_L1_EDX_IC_SIZE(descs[3]);
660 cai->cai_associativity = VIA_L1_EDX_IC_ASSOC(descs[3]);
661 cai->cai_linesize = VIA_L1_EDX_IC_LS(descs[3]);
662 if (family == 6 && model == 9 && stepping == 8) {
663 /* Erratum: stepping 8 reports 4 when it should be 2 */
664 cai->cai_associativity = 2;
665 }
666
667 /*
668 * Determine L2 cache/TLB info.
669 */
670 if (lfunc < 0x80000006) {
671 /* No L2 cache info available. */
672 return;
673 }
674
675 x86_cpuid(0x80000006, descs);
676
677 cai = &ci->ci_cinfo[CAI_L2CACHE];
678 if (family > 6 || model >= 9) {
679 cai->cai_totalsize = VIA_L2N_ECX_C_SIZE(descs[2]);
680 cai->cai_associativity = VIA_L2N_ECX_C_ASSOC(descs[2]);
681 cai->cai_linesize = VIA_L2N_ECX_C_LS(descs[2]);
682 } else {
683 cai->cai_totalsize = VIA_L2_ECX_C_SIZE(descs[2]);
684 cai->cai_associativity = VIA_L2_ECX_C_ASSOC(descs[2]);
685 cai->cai_linesize = VIA_L2_ECX_C_LS(descs[2]);
686 }
687}

701static void
702cpu_probe_vortex86(struct cpu_info *ci)
703{
704#define PCI_MODE1_ADDRESS_REG 0x0cf8
705#define PCI_MODE1_DATA_REG 0x0cfc
706#define PCI_MODE1_ENABLE 0x80000000UL
707
708 uint32_t reg;
709
710 if (cpu_vendor != CPUVENDOR_VORTEX86)
711 return;
712 /*
713 * CPU model available from "Customer ID register" in
714 * North Bridge Function 0 PCI space
715 * we can't use pci_conf_read() because the PCI subsystem is not
716 * not initialised early enough
717 */
718
719 outl(PCI_MODE1_ADDRESS_REG, PCI_MODE1_ENABLE | 0x90);
720 reg = inl(PCI_MODE1_DATA_REG);
721
722 if ((reg & 0xf8ffffff) != 0x30504d44) {
723 reg = 0;
724 } else {
725 reg = (reg >> 24) & 7;
726 }
727
728 static const char *cpu_vortex86_flavor[] = {
729 "??", "SX", "DX", "MX", "DX2", "MX+", "DX3", "EX",
730 };
731 snprintf(cpu_brand_string, sizeof(cpu_brand_string), "Vortex86%s",
732 cpu_vortex86_flavor[reg]);
733
734#undef PCI_MODE1_ENABLE
735#undef PCI_MODE1_ADDRESS_REG
736#undef PCI_MODE1_DATA_REG
737}

756static void
757cpu_probe_fpu(struct cpu_info *ci)
758{
759 u_int descs[4];
760 int i;
761
762 x86_fpu_eager = true;
763 x86_fpu_save = FPU_SAVE_FSAVE;
764
765#ifdef i386
766 /* If we have FXSAVE/FXRESTOR, use them. */
767 if ((ci->ci_feat_val[0] & CPUID_FXSR) == 0) {
768 i386_use_fxsave = 0;
769 /* Allow for no fpu even if cpuid is supported */
770 cpu_probe_old_fpu(ci);
771 return;
772 }
773
774 i386_use_fxsave = 1;
775 /*
776 * If we have SSE/SSE2, enable XMM exceptions, and
777 * notify userland.
778 */
779 if (ci->ci_feat_val[0] & CPUID_SSE)
780 i386_has_sse = 1;
781 if (ci->ci_feat_val[0] & CPUID_SSE2)
782 i386_has_sse2 = 1;
783#else
784 /*
785 * For amd64 i386_use_fxsave, i386_has_sse and i386_has_sse2 are
786 * #defined to 1, because fxsave/sse/sse2 are always present.
787 */
788#endif
789
790 x86_fpu_save = FPU_SAVE_FXSAVE;
791 x86_fpu_save_size = sizeof(struct fxsave);
792
793 /* See if xsave (for AVX) is supported */
794 if ((ci->ci_feat_val[1] & CPUID2_XSAVE) == 0)
795 return;
796
797#ifdef XENPV
798 /*
799 * Xen kernel can disable XSAVE via "no-xsave" option, in that case
800 * XSAVE instructions like xrstor become privileged and trigger
801 * supervisor trap. OSXSAVE flag seems to be reliably set according
802 * to whether XSAVE is actually available.
803 */
804 if ((ci->ci_feat_val[1] & CPUID2_OSXSAVE) == 0)
805 return;
806#endif
807
808 x86_fpu_save = FPU_SAVE_XSAVE;
809
810#if 0 /* XXX PR 52966 */
811 x86_cpuid2(0xd, 1, descs);
812 if (descs[0] & CPUID_PES1_XSAVEOPT)
813 x86_fpu_save = FPU_SAVE_XSAVEOPT;
814#endif
815
816 /* Get features and maximum size of the save area */
817 x86_cpuid(0xd, descs);
818 if (descs[2] > sizeof(struct fxsave))
819 x86_fpu_save_size = descs[2];
820
821 x86_xsave_features = (uint64_t)descs[3] << 32 | descs[0];
822
823 /* Get component offsets and sizes for the save area */
824 for (i = XSAVE_YMM_Hi128; i < __arraycount(x86_xsave_offsets); i++) {
825 if (x86_xsave_features & __BIT(i)) {
826 x86_cpuid2(0xd, i, descs);
827 x86_xsave_offsets[i] = descs[1];
828 x86_xsave_sizes[i] = descs[0];
829 }
830 }
831}

833void
834cpu_probe(struct cpu_info *ci)
835{
836 u_int descs[4];
837 int i;
838 uint32_t miscbytes;
839 uint32_t brand[12];
840
841 cpu_vendor = i386_nocpuid_cpus[cputype << 1];
842 cpu_class = i386_nocpuid_cpus[(cputype << 1) + 1];
843
844 if (cpuid_level < 0) {
845 /* cpuid instruction not supported */
846 cpu_probe_old_fpu(ci);
847 return;
848 }
849
850 for (i = 0; i < __arraycount(ci->ci_feat_val); i++) {
851 ci->ci_feat_val[i] = 0;
852 }
853
854 x86_cpuid(0, descs);
855 cpuid_level = descs[0];
856 ci->ci_max_cpuid = descs[0];
857
858 ci->ci_vendor[0] = descs[1];
859 ci->ci_vendor[2] = descs[2];
860 ci->ci_vendor[1] = descs[3];
861 ci->ci_vendor[3] = 0;
862
863 if (memcmp(ci->ci_vendor, "GenuineIntel", 12) == 0)
864 cpu_vendor = CPUVENDOR_INTEL;
865 else if (memcmp(ci->ci_vendor, "AuthenticAMD", 12) == 0)
866 cpu_vendor = CPUVENDOR_AMD;
867 else if (memcmp(ci->ci_vendor, "CyrixInstead", 12) == 0)
868 cpu_vendor = CPUVENDOR_CYRIX;
869 else if (memcmp(ci->ci_vendor, "Geode by NSC", 12) == 0)
870 cpu_vendor = CPUVENDOR_CYRIX;
871 else if (memcmp(ci->ci_vendor, "CentaurHauls", 12) == 0)
872 cpu_vendor = CPUVENDOR_IDT;
873 else if (memcmp(ci->ci_vendor, "GenuineTMx86", 12) == 0)
874 cpu_vendor = CPUVENDOR_TRANSMETA;
875 else if (memcmp(ci->ci_vendor, "Vortex86 SoC", 12) == 0)
876 cpu_vendor = CPUVENDOR_VORTEX86;
877 else
878 cpu_vendor = CPUVENDOR_UNKNOWN;
879
880 if (cpuid_level >= 1) {
881 x86_cpuid(1, descs);
882 ci->ci_signature = descs[0];
883 miscbytes = descs[1];
884 ci->ci_feat_val[1] = descs[2];
885 ci->ci_feat_val[0] = descs[3];
886
887 /* Determine family + class. */
888 cpu_class = CPUID_TO_FAMILY(ci->ci_signature)
889 + (CPUCLASS_386 - 3);
890 if (cpu_class > CPUCLASS_686)
891 cpu_class = CPUCLASS_686;
892
893 /* CLFLUSH line size is next 8 bits */
894 if (ci->ci_feat_val[0] & CPUID_CFLUSH)
895 ci->ci_cflush_lsize
896 = __SHIFTOUT(miscbytes, CPUID_CLFLUSH_SIZE) << 3;
897 ci->ci_initapicid = __SHIFTOUT(miscbytes, CPUID_LOCAL_APIC_ID);
898 }
899
900 /*
901 * Get the basic information from the extended cpuid leafs.
902 * These were first implemented by amd, but most of the values
903 * match with those generated by modern intel cpus.
904 */
905 x86_cpuid(0x80000000, descs);
906 if (descs[0] >= 0x80000000)
907 ci->ci_max_ext_cpuid = descs[0];
908 else
909 ci->ci_max_ext_cpuid = 0;
910
911 if (ci->ci_max_ext_cpuid >= 0x80000001) {
912 /* Determine the extended feature flags. */
913 x86_cpuid(0x80000001, descs);
914 ci->ci_feat_val[3] = descs[2]; /* %ecx */
915 ci->ci_feat_val[2] = descs[3]; /* %edx */
916 }
917
918 if (ci->ci_max_ext_cpuid >= 0x80000004) {
919 x86_cpuid(0x80000002, brand);
920 x86_cpuid(0x80000003, brand + 4);
921 x86_cpuid(0x80000004, brand + 8);
922 /* Skip leading spaces on brand */
923 for (i = 0; i < 48; i++) {
924 if (((char *) brand)[i] != ' ')
925 break;
926 }
927 memcpy(cpu_brand_string, ((char *) brand) + i, 48 - i);
928 }
929
930 /*
931 * Get the structured extended features.
932 */
933 if (cpuid_level >= 7) {
934 x86_cpuid(7, descs);
935 ci->ci_feat_val[5] = descs[1]; /* %ebx */
936 ci->ci_feat_val[6] = descs[2]; /* %ecx */
937 ci->ci_feat_val[7] = descs[3]; /* %edx */
938 }
939
940 cpu_probe_intel(ci);
941 cpu_probe_amd(ci);
942 cpu_probe_cyrix(ci);
943 cpu_probe_winchip(ci);
944 cpu_probe_c3(ci);
945 cpu_probe_geode(ci);
946 cpu_probe_vortex86(ci);
947
948 cpu_probe_fpu(ci);
949
950 x86_cpu_topology(ci);
951
952 if (cpu_vendor != CPUVENDOR_AMD && (ci->ci_feat_val[0] & CPUID_TM) &&
953 (rdmsr(MSR_MISC_ENABLE) & (1 << 3)) == 0) {
954 /* Enable thermal monitor 1. */
955 wrmsr(MSR_MISC_ENABLE, rdmsr(MSR_MISC_ENABLE) | (1<<3));
956 }
957
958 ci->ci_feat_val[0] &= ~CPUID_FEAT_BLACKLIST;
959 if (ci == &cpu_info_primary) {
960 /* If first. Boot Processor is the cpu_feature reference. */
961 for (i = 0; i < __arraycount(cpu_feature); i++) {
962 cpu_feature[i] = ci->ci_feat_val[i];
963 }
964 identify_hypervisor();
965#ifndef XEN
966 /* Early patch of text segment. */
967 x86_patch(true);
968#endif
969 } else {
970 /*
971 * If not first. Warn about cpu_feature mismatch for
972 * secondary CPUs.
973 */
974 for (i = 0; i < __arraycount(cpu_feature); i++) {
975 if (cpu_feature[i] != ci->ci_feat_val[i])
976 aprint_error_dev(ci->ci_dev,
977 "feature mismatch: cpu_feature[%d] is "
978 "%#x, but CPU reported %#x\n",
979 i, cpu_feature[i], ci->ci_feat_val[i]);
980 }
981 }
982}

984/* Write what we know about the cpu to the console... */
985void
986cpu_identify(struct cpu_info *ci)
987{
988
989 cpu_setmodel("%s %d86-class",
990 cpu_vendor_names[cpu_vendor], cpu_class + 3);
991 if (cpu_brand_string[0] != '\0') {
992 aprint_normal_dev(ci->ci_dev, "%s", cpu_brand_string);
993 } else {
994 aprint_normal_dev(ci->ci_dev, "%s", cpu_getmodel());
995 if (ci->ci_data.cpu_cc_freq != 0)
996 aprint_normal(", %dMHz",
997 (int)(ci->ci_data.cpu_cc_freq / 1000000));
998 }
999 if (ci->ci_signature != 0)
1000 aprint_normal(", id 0x%x", ci->ci_signature);
1001 aprint_normal("\n");
1002 aprint_normal_dev(ci->ci_dev, "package %lu, core %lu, smt %lu\n",
1003 ci->ci_package_id, ci->ci_core_id, ci->ci_smt_id);
1004 if (cpu_brand_string[0] == '\0') {
1005 strlcpy(cpu_brand_string, cpu_getmodel(),
1006 sizeof(cpu_brand_string));
1007 }
1008 if (cpu_class == CPUCLASS_386) {
1009 panic("NetBSD requires an 80486DX or later processor");
1010 }
1011 if (cputype == CPU_486DLC) {
1012 aprint_error("WARNING: BUGGY CYRIX CACHE\n");
1013 }
1014
1015#if !defined(XENPV) || defined(DOM0OPS) /* on Xen PV rdmsr is for Dom0 only */
1016 if (cpu_vendor == CPUVENDOR_AMD /* check enablement of an */
1017 && device_unit(ci->ci_dev) == 0 /* AMD feature only once */
1018 && ((cpu_feature[3] & CPUID_SVM) == CPUID_SVM)) {
1019 uint64_t val;
1020
1021 val = rdmsr(MSR_VMCR);
1022 if (((val & VMCR_SVMED) == VMCR_SVMED)
1023 && ((val & VMCR_LOCK) == VMCR_LOCK)) {
1024 aprint_normal_dev(ci->ci_dev,
1025 "SVM disabled by the BIOS\n");
1026 }
1027 }
1028#endif
1029
1030#ifdef i386
1031 if (i386_fpu_fdivbug == 1)
1032 aprint_normal_dev(ci->ci_dev,
1033 "WARNING: Pentium FDIV bug detected!\n");
1034
1035 if (cpu_vendor == CPUVENDOR_TRANSMETA) {
1036 u_int descs[4];
1037 x86_cpuid(0x80860000, descs);
1038 if (descs[0] >= 0x80860007)
1039 /* Create longrun sysctls */
1040 tmx86_init_longrun();
1041 }
1042#endif /* i386 */
1043
1044}

1068void
1069identify_hypervisor(void)
1070{
1071 u_int regs[6];
1072 char hv_vendor[12];
1073 const char *p;
1074 int i;
1075
1076 if (vm_guest != VM_GUEST_NO)
1077 return;
1078
1079 /*
1080 * [RFC] CPUID usage for interaction between Hypervisors and Linux.
1081 * http://lkml.org/lkml/2008/10/1/246
1082 *
1083 * KB1009458: Mechanisms to determine if software is running in
1084 * a VMware virtual machine
1085 * http://kb.vmware.com/kb/1009458
1086 */
1087 if (ISSET(cpu_feature[1], CPUID2_RAZ)) {
1088 vm_guest = VM_GUEST_VM;
1089 x86_cpuid(0x40000000, regs);
1090 if (regs[0] >= 0x40000000) {
1091 memcpy(&hv_vendor[0], &regs[1], sizeof(*regs));
1092 memcpy(&hv_vendor[4], &regs[2], sizeof(*regs));
1093 memcpy(&hv_vendor[8], &regs[3], sizeof(*regs));
1094 if (memcmp(hv_vendor, "VMwareVMware", 12) == 0)
1095 vm_guest = VM_GUEST_VMWARE;
1096 else if (memcmp(hv_vendor, "Microsoft Hv", 12) == 0) {
1097 vm_guest = VM_GUEST_HV;
1098#if NHYPERV > 0
1099 hyperv_early_init();
1100#endif
1101 } else if (memcmp(hv_vendor, "KVMKVMKVM\0\0\0", 12) == 0)
1102 vm_guest = VM_GUEST_KVM;
1103 else if (memcmp(hv_vendor, "XenVMMXenVMM", 12) == 0)
1104 vm_guest = VM_GUEST_XEN;
1105 /* FreeBSD bhyve: "bhyve bhyve " */
1106 /* OpenBSD vmm: "OpenBSDVMM58" */
1107 /* NetBSD nvmm: "___ NVMM ___" */
1108 }
1109 return;
1110 }
1111
1112 /*
1113 * Examine SMBIOS strings for older hypervisors.
1114 */
1115 p = pmf_get_platform("system-serial");
1116 if (p != NULL) {
1117 if (strncmp(p, "VMware-", 7) == 0 || strncmp(p, "VMW", 3) == 0) {
1118 vmt_hvcall(VM_CMD_GET_VERSION, regs);
1119 if (regs[1] == VM_MAGIC) {
1120 vm_guest = VM_GUEST_VMWARE;
1121 return;
1122 }
1123 }
1124 }
1125 p = pmf_get_platform("bios-vendor");
1126 if (p != NULL) {
1127 for (i = 0; i < __arraycount(vm_bios_vendors); i++) {
1128 if (strcmp(p, vm_bios_vendors[i]) == 0) {
1129 vm_guest = VM_GUEST_VM;
1130 return;
1131 }
1132 }
1133 }
1134 p = pmf_get_platform("system-product");
1135 if (p != NULL) {
1136 for (i = 0; i < __arraycount(vm_system_products); i++) {
1137 if (strcmp(p, vm_system_products[i]) == 0) {
1138 vm_guest = VM_GUEST_VM;
1139 return;
1140 }
1141 }
1142 }
