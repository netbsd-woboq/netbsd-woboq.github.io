292/*
293 * Recalculate the interrupt masks from scratch.
294 * During early boot, anything goes and we are always called on the BP.
295 * When the system is up and running:
296 *
297 * => called with ci == curcpu()
298 * => cpu_lock held by the initiator
299 * => interrupts disabled on-chip (PSL_I)
300 *
301 * Do not call printf(), kmem_free() or other "heavyweight" routines
302 * from here. This routine must be quick and must not block.
303 */
304static void
305intr_calculatemasks(struct cpu_info *ci)
306{
307 int irq, level, unusedirqs, intrlevel[MAX_INTR_SOURCES];
308 struct intrhand *q;
309
310 /* First, figure out which levels each IRQ uses. */
311 unusedirqs = 0xffffffff;
312 for (irq = 0; irq < MAX_INTR_SOURCES; irq++) {
313 int levels = 0;
314
315 if (ci->ci_isources[irq] == NULL) {
316 intrlevel[irq] = 0;
317 continue;
318 }
319 for (q = ci->ci_isources[irq]->is_handlers; q; q = q->ih_next)
320 levels |= 1U << q->ih_level;
321 intrlevel[irq] = levels;
322 if (levels)
323 unusedirqs &= ~(1U << irq);
324 }
325
326 /* Then figure out which IRQs use each level. */
327 for (level = 0; level < NIPL; level++) {
328 int irqs = 0;
329 for (irq = 0; irq < MAX_INTR_SOURCES; irq++)
330 if (intrlevel[irq] & (1U << level))
331 irqs |= 1U << irq;
332 ci->ci_imask[level] = irqs | unusedirqs;
333 }
334
335 for (level = 0; level<(NIPL-1); level++)
336 ci->ci_imask[level+1] |= ci->ci_imask[level];
337
338 for (irq = 0; irq < MAX_INTR_SOURCES; irq++) {
339 int maxlevel = IPL_NONE;
340 int minlevel = IPL_HIGH;
341
342 if (ci->ci_isources[irq] == NULL)
343 continue;
344 for (q = ci->ci_isources[irq]->is_handlers; q;
345 q = q->ih_next) {
346 if (q->ih_level < minlevel)
347 minlevel = q->ih_level;
348 if (q->ih_level > maxlevel)
349 maxlevel = q->ih_level;
350 }
351 ci->ci_isources[irq]->is_maxlevel = maxlevel;
352 ci->ci_isources[irq]->is_minlevel = minlevel;
353 }
354
355 for (level = 0; level < NIPL; level++)
356 ci->ci_iunmask[level] = ~ci->ci_imask[level];
357}

359/*
360 * Create an interrupt id such as "ioapic0 pin 9". This interrupt id is used
361 * by MI code and intrctl(8).
362 */
363const char *
364intr_create_intrid(int legacy_irq, struct pic *pic, int pin, char *buf,
365 size_t len)
366{
367 int ih = 0;
368
369#if NPCI > 0
370#if defined(__HAVE_PCI_MSI_MSIX)
371 if ((pic->pic_type == PIC_MSI) || (pic->pic_type == PIC_MSIX)) {
372 uint64_t pih;
373 int dev, vec;
374
375 dev = msipic_get_devid(pic);
376 vec = pin;
377 pih = __SHIFTIN((uint64_t)dev, MSI_INT_DEV_MASK)
378 | __SHIFTIN((uint64_t)vec, MSI_INT_VEC_MASK)
379 | APIC_INT_VIA_MSI;
380 if (pic->pic_type == PIC_MSI)
381 MSI_INT_MAKE_MSI(pih);
382 else if (pic->pic_type == PIC_MSIX)
383 MSI_INT_MAKE_MSIX(pih);
384
385 return x86_pci_msi_string(NULL, pih, buf, len);
386 }
387#endif /* __HAVE_PCI_MSI_MSIX */
388#endif
389
390 if (pic->pic_type == PIC_XEN) {
391 ih = pin; /* Port == pin */
392 return xen_intr_string(pin, buf, len, pic);
393 }
394
395 /*
396 * If the device is pci, "legacy_irq" is alway -1. Least 8 bit of "ih"
397 * is only used in intr_string() to show the irq number.
398 * If the device is "legacy"(such as floppy), it should not use
399 * intr_string().
400 */
401 if (pic->pic_type == PIC_I8259) {
402 ih = legacy_irq;
403 return legacy_intr_string(ih, buf, len, pic);
404 }
405
406#if NIOAPIC > 0 || NACPICA > 0
407 ih = ((pic->pic_apicid << APIC_INT_APIC_SHIFT) & APIC_INT_APIC_MASK)
408 | ((pin << APIC_INT_PIN_SHIFT) & APIC_INT_PIN_MASK);
409 if (pic->pic_type == PIC_IOAPIC) {
410 ih |= APIC_INT_VIA_APIC;
411 }
412 ih |= pin;
413 return intr_string(ih, buf, len);
414#endif
415
416 return NULL; /* No pic found! */
417}

422static struct intrsource *
423intr_get_io_intrsource(const char *intrid)
424{
425 struct intrsource *isp;
426
427 KASSERT(mutex_owned(&cpu_lock));
428
429 SIMPLEQ_FOREACH(isp, &io_interrupt_sources, is_list) {
430 KASSERT(isp->is_intrid != NULL);
431 if (strncmp(intrid, isp->is_intrid, INTRIDBUF - 1) == 0)
432 return isp;
433 }
434 return NULL;
435}

437/*
438 * Allocate intrsource and add to io_interrupt_sources list.
439 */
440struct intrsource *
441intr_allocate_io_intrsource(const char *intrid)
442{
443 CPU_INFO_ITERATOR cii;
444 struct cpu_info *ci;
445 struct intrsource *isp;
446 struct percpu_evcnt *pep;
447
448 KASSERT(mutex_owned(&cpu_lock));
449
450 if (intrid == NULL)
451 return NULL;
452
453 isp = kmem_zalloc(sizeof(*isp), KM_SLEEP);
454 pep = kmem_zalloc(sizeof(*pep) * ncpu, KM_SLEEP);
455 isp->is_saved_evcnt = pep;
456 for (CPU_INFO_FOREACH(cii, ci)) {
457 pep->cpuid = ci->ci_cpuid;
458 pep++;
459 }
460 strlcpy(isp->is_intrid, intrid, sizeof(isp->is_intrid));
461
462 SIMPLEQ_INSERT_TAIL(&io_interrupt_sources, isp, is_list);
463
464 return isp;
465}

467/*
468 * Remove from io_interrupt_sources list and free by the intrsource pointer.
469 */
470static void
471intr_free_io_intrsource_direct(struct intrsource *isp)
472{
473 KASSERT(mutex_owned(&cpu_lock));
474
475 SIMPLEQ_REMOVE(&io_interrupt_sources, isp, intrsource, is_list);
476
477 /* Is this interrupt established? */
478 if (isp->is_evname[0] != '\0')
479 evcnt_detach(&isp->is_evcnt);
480
481 kmem_free(isp->is_saved_evcnt,
482 sizeof(*(isp->is_saved_evcnt)) * ncpu);
483
484 kmem_free(isp, sizeof(*isp));
485}

487/*
488 * Remove from io_interrupt_sources list and free by the interrupt id.
489 * This function can be used by MI code.
490 */
491void
492intr_free_io_intrsource(const char *intrid)
493{
494 struct intrsource *isp;
495
496 KASSERT(mutex_owned(&cpu_lock));
497
498 if (intrid == NULL)
499 return;
500
501 if ((isp = intr_get_io_intrsource(intrid)) == NULL) {
502 return;
503 }
504
505 /* If the interrupt uses shared IRQ, don't free yet. */
506 if (isp->is_handlers != NULL) {
507 return;
508 }
509
510 intr_free_io_intrsource_direct(isp);
511}

513static int
514intr_allocate_slot_cpu(struct cpu_info *ci, struct pic *pic, int pin,
515 int *index, struct intrsource *chained)
516{
517 int slot, i;
518 struct intrsource *isp;
519
520 KASSERT(mutex_owned(&cpu_lock));
521
522 if (pic == &i8259_pic) {
523 KASSERT(CPU_IS_PRIMARY(ci));
524 slot = pin;
525 } else {
526 int start = 0;
527 slot = -1;
528
529 /* avoid reserved slots for legacy interrupts. */
530 if (CPU_IS_PRIMARY(ci) && msipic_is_msi_pic(pic))
531 start = NUM_LEGACY_IRQS;
532 /*
533 * intr_allocate_slot has checked for an existing mapping.
534 * Now look for a free slot.
535 */
536 for (i = start; i < MAX_INTR_SOURCES ; i++) {
537 if (ci->ci_isources[i] == NULL) {
538 slot = i;
539 break;
540 }
541 }
542 if (slot == -1) {
543 return EBUSY;
544 }
545 }
546
547 isp = ci->ci_isources[slot];
548 if (isp == NULL) {
549 const char *via;
550
551 isp = chained;
552 KASSERT(isp != NULL);
553 if (pic->pic_type == PIC_MSI || pic->pic_type == PIC_MSIX)
554 via = "vec";
555 else
556 via = "pin";
557 snprintf(isp->is_evname, sizeof (isp->is_evname),
558 "%s %d", via, pin);
559 evcnt_attach_dynamic(&isp->is_evcnt, EVCNT_TYPE_INTR, NULL,
560 pic->pic_name, isp->is_evname);
561 isp->is_active_cpu = ci->ci_cpuid;
562 ci->ci_isources[slot] = isp;
563 }
564
565 *index = slot;
566 return 0;
567}

569/*
570 * A simple round-robin allocator to assign interrupts to CPUs.
571 */
572static int __noinline
573intr_allocate_slot(struct pic *pic, int pin, int level,
574 struct cpu_info **cip, int *index, int *idt_slot,
575 struct intrsource *chained)
576{
577 CPU_INFO_ITERATOR cii;
578 struct cpu_info *ci, *lci;
579 struct intrsource *isp;
580 int slot = 0, idtvec, error;
581
582 KASSERT(mutex_owned(&cpu_lock));
583
584 /* First check if this pin is already used by an interrupt vector. */
585 for (CPU_INFO_FOREACH(cii, ci)) {
586 for (slot = 0 ; slot < MAX_INTR_SOURCES ; slot++) {
587 if ((isp = ci->ci_isources[slot]) == NULL) {
588 continue;
589 }
590 if (isp->is_pic == pic &&
591 pin != -1 && isp->is_pin == pin) {
592 *idt_slot = isp->is_idtvec;
593 *index = slot;
594 *cip = ci;
595 return 0;
596 }
597 }
598 }
599
600 /*
601 * The pic/pin combination doesn't have an existing mapping.
602 * Find a slot for a new interrupt source. For the i8259 case,
603 * we always use reserved slots of the primary CPU. Otherwise,
604 * we make an attempt to balance the interrupt load.
605 *
606 * PIC and APIC usage are essentially exclusive, so the reservation
607 * of the ISA slots is ignored when assigning IOAPIC slots.
608 */
609 if (pic == &i8259_pic) {
610 /*
611 * Must be directed to BP.
612 */
613 ci = &cpu_info_primary;
614 error = intr_allocate_slot_cpu(ci, pic, pin, &slot, chained);
615 } else {
616 /*
617 * Find least loaded AP/BP and try to allocate there.
618 */
619 ci = NULL;
620 for (CPU_INFO_FOREACH(cii, lci)) {
621 if ((lci->ci_schedstate.spc_flags & SPCF_NOINTR) != 0) {
622 continue;
623 }
624#if 0
625 if (ci == NULL ||
626 ci->ci_nintrhand > lci->ci_nintrhand) {
627 ci = lci;
628 }
629#else
630 ci = &cpu_info_primary;
631#endif
632 }
633 KASSERT(ci != NULL);
634 error = intr_allocate_slot_cpu(ci, pic, pin, &slot, chained);
635
636 /*
637 * If that did not work, allocate anywhere.
638 */
639 if (error != 0) {
640 for (CPU_INFO_FOREACH(cii, ci)) {
641 if ((ci->ci_schedstate.spc_flags &
642 SPCF_NOINTR) != 0) {
643 continue;
644 }
645 error = intr_allocate_slot_cpu(ci, pic,
646 pin, &slot, chained);
647 if (error == 0) {
648 break;
649 }
650 }
651 }
652 }
653 if (error != 0) {
654 return error;
655 }
656 KASSERT(ci != NULL);
657
658 /*
659 * Now allocate an IDT vector.
660 * For the 8259 these are reserved up front.
661 */
662 if (pic == &i8259_pic) {
663 idtvec = ICU_OFFSET + pin;
664 } else {
665 /*
666 * TODO to support MSI (not MSI-X) multiple vectors
667 *
668 * PCI Local Bus Specification Revision 3.0 says the devices
669 * which use MSI multiple vectors increment the low order bits
670 * of MSI message data.
671 * On the other hand, Intel SDM "10.11.2 Message Data Register
672 * Format" says the 7:0 bits of MSI message data mean Interrupt
673 * Descriptor Table(IDT) vector.
674 * As the result of these two documents, the IDT vectors which
675 * are used by a device using MSI multiple vectors must be
676 * continuous.
677 */
678 idtvec = idt_vec_alloc(APIC_LEVEL(level), IDT_INTR_HIGH);
679 }
680 if (idtvec == 0) {
681 evcnt_detach(&ci->ci_isources[slot]->is_evcnt);
682 ci->ci_isources[slot] = NULL;
683 return EBUSY;
684 }
685 ci->ci_isources[slot]->is_idtvec = idtvec;
686 *idt_slot = idtvec;
687 *index = slot;
688 *cip = ci;
689 return 0;
690}

692static void
693intr_source_free(struct cpu_info *ci, int slot, struct pic *pic, int idtvec)
694{
695 struct intrsource *isp;
696
697 isp = ci->ci_isources[slot];

699 if (isp->is_handlers != NULL)
700 return;
701 ci->ci_isources[slot] = NULL;
702 if (pic != &i8259_pic)
703 idt_vec_free(idtvec);
704}

742/*
743 * Handle per-CPU component of interrupt establish.
744 *
745 * => caller (on initiating CPU) holds cpu_lock on our behalf
746 * => arg1: struct intrhand *ih
747 * => arg2: int idt_vec
748 */
749static void
750intr_establish_xcall(void *arg1, void *arg2)
751{
752 struct intrsource *source;
753 struct intrstub *stubp;
754 struct intrhand *ih;
755 struct cpu_info *ci;
756 int idt_vec;
757 u_long psl;
758
759 ih = arg1;
760
761 KASSERT(ih->ih_cpu == curcpu() || !mp_online);
762
763 ci = ih->ih_cpu;
764 source = ci->ci_isources[ih->ih_slot];
765 idt_vec = (int)(intptr_t)arg2;
766
767 /* Disable interrupts locally. */
768 psl = x86_read_psl();
769 x86_disable_intr();
770
771 /* Link in the handler and re-calculate masks. */
772 *(ih->ih_prevp) = ih;
773 intr_calculatemasks(ci);
774
775 /* Hook in new IDT vector and SPL state. */
776 if (source->is_resume == NULL || source->is_idtvec != idt_vec) {
777 if (source->is_idtvec != 0 && source->is_idtvec != idt_vec)
778 idt_vec_free(source->is_idtvec);
779 source->is_idtvec = idt_vec;
780 if (source->is_type == IST_LEVEL) {
781 stubp = &source->is_pic->pic_level_stubs[ih->ih_slot];
782 } else {
783 stubp = &source->is_pic->pic_edge_stubs[ih->ih_slot];
784 }
785 source->is_resume = stubp->ist_resume;
786 source->is_recurse = stubp->ist_recurse;
787 idt_vec_set(idt_vec, stubp->ist_entry);
788 }
789
790 /* Re-enable interrupts locally. */
791 x86_write_psl(psl);
792}

794void *
795intr_establish_xname(int legacy_irq, struct pic *pic, int pin, int type,
796 int level, int (*handler)(void *), void *arg,
797 bool known_mpsafe, const char *xname)
798{
799 struct intrhand **p, *q, *ih;
800 struct cpu_info *ci;
801 int slot, error, idt_vec;
802 struct intrsource *chained, *source;
803#ifdef MULTIPROCESSOR
804 bool mpsafe = (known_mpsafe || level != IPL_VM);
805#endif /* MULTIPROCESSOR */
806 uint64_t where;
807 const char *intrstr;
808 char intrstr_buf[INTRIDBUF];
809
810 KASSERTMSG((legacy_irq == -1 || (0 <= legacy_irq && legacy_irq < 16)),
811 "bad legacy IRQ value: %d", legacy_irq);
812 KASSERTMSG((legacy_irq != -1 || pic != &i8259_pic),
813 "non-legacy IRQ on i8259");
814
815 ih = kmem_alloc(sizeof(*ih), KM_SLEEP);
816 intrstr = intr_create_intrid(legacy_irq, pic, pin, intrstr_buf,
817 sizeof(intrstr_buf));
818 KASSERT(intrstr != NULL);
819
820 mutex_enter(&cpu_lock);
821
822 /* allocate intrsource pool, if not yet. */
823 chained = intr_get_io_intrsource(intrstr);
824 if (chained == NULL) {
825 if (msipic_is_msi_pic(pic)) {
826 mutex_exit(&cpu_lock);
827 kmem_free(ih, sizeof(*ih));
828 printf("%s: %s has no intrsource\n", __func__, intrstr);
829 return NULL;
830 }
831 chained = intr_allocate_io_intrsource(intrstr);
832 if (chained == NULL) {
833 mutex_exit(&cpu_lock);
834 kmem_free(ih, sizeof(*ih));
835 printf("%s: can't allocate io_intersource\n", __func__);
836 return NULL;
837 }
838 }
839
840 error = intr_allocate_slot(pic, pin, level, &ci, &slot, &idt_vec,
841 chained);
842 if (error != 0) {
843 intr_free_io_intrsource_direct(chained);
844 mutex_exit(&cpu_lock);
845 kmem_free(ih, sizeof(*ih));
846 printf("failed to allocate interrupt slot for PIC %s pin %d\n",
847 pic->pic_name, pin);
848 return NULL;
849 }
850
851 source = ci->ci_isources[slot];
852
853 if (source->is_handlers != NULL &&
854 source->is_pic->pic_type != pic->pic_type) {
855 intr_free_io_intrsource_direct(chained);
856 mutex_exit(&cpu_lock);
857 kmem_free(ih, sizeof(*ih));
858 printf("%s: can't share intr source between "
859 "different PIC types (legacy_irq %d pin %d slot %d)\n",
860 __func__, legacy_irq, pin, slot);
861 return NULL;
862 }
863
864 source->is_pin = pin;
865 source->is_pic = pic;
866 intr_append_intrsource_xname(source, xname);
867 switch (source->is_type) {
868 case IST_NONE:
869 source->is_type = type;
870 break;
871 case IST_EDGE:
872 case IST_LEVEL:
873 if (source->is_type == type)
874 break;
875 /* FALLTHROUGH */
876 case IST_PULSE:
877 if (type != IST_NONE) {
878 intr_source_free(ci, slot, pic, idt_vec);
879 intr_free_io_intrsource_direct(chained);
880 mutex_exit(&cpu_lock);
881 kmem_free(ih, sizeof(*ih));
882 printf("%s: pic %s pin %d: can't share "
883 "type %d with %d\n",
884 __func__, pic->pic_name, pin,
885 source->is_type, type);
886 return NULL;
887 }
888 break;
889 default:
890 panic("%s: bad intr type %d for pic %s pin %d\n",
891 __func__, source->is_type, pic->pic_name, pin);
892 /* NOTREACHED */
893 }
894
895 /*
896 * If the establishing interrupt uses shared IRQ, the interrupt uses
897 * "ci->ci_isources[slot]" instead of allocated by the establishing
898 * device's pci_intr_alloc() or this function.
899 */
900 if (source->is_handlers != NULL) {
901 struct intrsource *isp, *nisp;
902
903 SIMPLEQ_FOREACH_SAFE(isp, &io_interrupt_sources,
904 is_list, nisp) {
905 if (strncmp(intrstr, isp->is_intrid, INTRIDBUF - 1) == 0
906 && isp->is_handlers == NULL)
907 intr_free_io_intrsource_direct(isp);
908 }
909 }
910
911 /*
912 * We're now committed. Mask the interrupt in hardware and
913 * count it for load distribution.
914 */
915 (*pic->pic_hwmask)(pic, pin);
916 (ci->ci_nintrhand)++;
917
918 /*
919 * Figure out where to put the handler.
920 * This is O(N^2), but we want to preserve the order, and N is
921 * generally small.
922 */
923 for (p = &ci->ci_isources[slot]->is_handlers;
924 (q = *p) != NULL && q->ih_level > level;
925 p = &q->ih_next) {
926 /* nothing */;
927 }
928
929 ih->ih_fun = ih->ih_realfun = handler;
930 ih->ih_arg = ih->ih_realarg = arg;
931 ih->ih_prevp = p;
932 ih->ih_next = *p;
933 ih->ih_level = level;
934 ih->ih_pin = pin;
935 ih->ih_cpu = ci;
936 ih->ih_slot = slot;
937#ifdef MULTIPROCESSOR
938 if (!mpsafe) {
939 ih->ih_fun = intr_biglock_wrapper;
940 ih->ih_arg = ih;
941 }
942#endif /* MULTIPROCESSOR */
943
944 /*
945 * Call out to the remote CPU to update its interrupt state.
946 * Only make RPCs if the APs are up and running.
947 */
948 if (ci == curcpu() || !mp_online) {
949 intr_establish_xcall(ih, (void *)(intptr_t)idt_vec);
950 } else {
951 where = xc_unicast(0, intr_establish_xcall, ih,
952 (void *)(intptr_t)idt_vec, ci);
953 xc_wait(where);
954 }
955
956 /* All set up, so add a route for the interrupt and unmask it. */
957 (*pic->pic_addroute)(pic, ci, pin, idt_vec, type);
958 (*pic->pic_hwunmask)(pic, pin);
959 mutex_exit(&cpu_lock);
960
961 if (bootverbose || cpu_index(ci) != 0)
962 aprint_verbose("allocated pic %s type %s pin %d level %d to "
963 "%s slot %d idt entry %d\n",
964 pic->pic_name, type == IST_EDGE ? "edge" : "level", pin,
965 level, device_xname(ci->ci_dev), slot, idt_vec);
966
967 return (ih);
968}

979/*
980 * Called on bound CPU to handle intr_disestablish().
981 *
982 * => caller (on initiating CPU) holds cpu_lock on our behalf
983 * => arg1: struct intrhand *ih
984 * => arg2: unused
985 */
986static void
987intr_disestablish_xcall(void *arg1, void *arg2)
988{
989 struct intrhand **p, *q;
990 struct cpu_info *ci;
991 struct pic *pic;
992 struct intrsource *source;
993 struct intrhand *ih;
994 u_long psl;
995 int idtvec;
996
997 ih = arg1;
998 ci = ih->ih_cpu;
999
1000 KASSERT(ci == curcpu() || !mp_online);
1001
1002 /* Disable interrupts locally. */
1003 psl = x86_read_psl();
1004 x86_disable_intr();
1005
1006 pic = ci->ci_isources[ih->ih_slot]->is_pic;
1007 source = ci->ci_isources[ih->ih_slot];
1008 idtvec = source->is_idtvec;
1009
1010 (*pic->pic_hwmask)(pic, ih->ih_pin);
1011 atomic_and_32(&ci->ci_ipending, ~(1 << ih->ih_slot));
1012
1013 /*
1014 * Remove the handler from the chain.
1015 */
1016 for (p = &source->is_handlers; (q = *p) != NULL && q != ih;
1017 p = &q->ih_next)
1018 ;
1019 if (q == NULL) {
1020 x86_write_psl(psl);
1021 panic("%s: handler not registered", __func__);
1022 /* NOTREACHED */
1023 }
1024
1025 *p = q->ih_next;
1026
1027 intr_calculatemasks(ci);
1028 /*
1029 * If there is no any handler, 1) do delroute because it has no
1030 * any source and 2) dont' hwunmask to prevent spurious interrupt.
1031 *
1032 * If there is any handler, 1) don't delroute because it has source
1033 * and 2) do hwunmask to be able to get interrupt again.
1034 *
1035 */
1036 if (source->is_handlers == NULL)
1037 (*pic->pic_delroute)(pic, ci, ih->ih_pin, idtvec,
1038 source->is_type);
1039 else
1040 (*pic->pic_hwunmask)(pic, ih->ih_pin);
1041
1042 /* Re-enable interrupts. */
1043 x86_write_psl(psl);
1044
1045 /* If the source is free we can drop it now. */
1046 intr_source_free(ci, ih->ih_slot, pic, idtvec);
1047
1048 DPRINTF(("%s: remove slot %d (pic %s pin %d vec %d)\n",
1049 device_xname(ci->ci_dev), ih->ih_slot, pic->pic_name,
1050 ih->ih_pin, idtvec));
1051}

1066/*
1067 * Deregister an interrupt handler.
1068 */
1069void
1070intr_disestablish(struct intrhand *ih)
1071{
1072 struct cpu_info *ci;
1073 struct intrsource *isp;
1074 uint64_t where;
1075
1076 /*
1077 * Count the removal for load balancing.
1078 * Call out to the remote CPU to update its interrupt state.
1079 * Only make RPCs if the APs are up and running.
1080 */
1081 mutex_enter(&cpu_lock);
1082 ci = ih->ih_cpu;
1083 (ci->ci_nintrhand)--;
1084 KASSERT(ci->ci_nintrhand >= 0);
1085 isp = ci->ci_isources[ih->ih_slot];
1086 if (ci == curcpu() || !mp_online) {
1087 intr_disestablish_xcall(ih, NULL);
1088 } else {
1089 where = xc_unicast(0, intr_disestablish_xcall, ih, NULL, ci);
1090 xc_wait(where);
1091 }
1092 if (!msipic_is_msi_pic(isp->is_pic) && intr_num_handlers(isp) < 1) {
1093 intr_free_io_intrsource_direct(isp);
1094 }
1095 mutex_exit(&cpu_lock);
1096 kmem_free(ih, sizeof(*ih));
1097}

1529static bool
1530intr_redistribute(struct cpu_info *oci)
1531{
1532 struct intrsource *isp;
1533 struct intrhand *ih;
1534 CPU_INFO_ITERATOR cii;
1535 struct cpu_info *nci, *ici;
1536 int oslot, nslot;
1537 uint64_t where;
1538
1539 KASSERT(mutex_owned(&cpu_lock));
1540
1541 /* Look for an interrupt source that we can migrate. */
1542 for (oslot = 0; oslot < MAX_INTR_SOURCES; oslot++) {
1543 if ((isp = oci->ci_isources[oslot]) == NULL) {
1544 continue;
1545 }
1546 if (isp->is_pic->pic_type == PIC_IOAPIC) {
1547 break;
1548 }
1549 }
1550 if (oslot == MAX_INTR_SOURCES) {
1551 return false;
1552 }
1553
1554 /* Find least loaded CPU and try to move there. */
1555 nci = NULL;
1556 for (CPU_INFO_FOREACH(cii, ici)) {
1557 if ((ici->ci_schedstate.spc_flags & SPCF_NOINTR) != 0) {
1558 continue;
1559 }
1560 KASSERT(ici != oci);
1561 if (nci == NULL || nci->ci_nintrhand > ici->ci_nintrhand) {
1562 nci = ici;
1563 }
1564 }
1565 if (nci == NULL) {
1566 return false;
1567 }
1568 for (nslot = 0; nslot < MAX_INTR_SOURCES; nslot++) {
1569 if (nci->ci_isources[nslot] == NULL) {
1570 break;
1571 }
1572 }
1573
1574 /* If that did not work, allocate anywhere. */
1575 if (nslot == MAX_INTR_SOURCES) {
1576 for (CPU_INFO_FOREACH(cii, nci)) {
1577 if ((nci->ci_schedstate.spc_flags & SPCF_NOINTR) != 0) {
1578 continue;
1579 }
1580 KASSERT(nci != oci);
1581 for (nslot = 0; nslot < MAX_INTR_SOURCES; nslot++) {
1582 if (nci->ci_isources[nslot] == NULL) {
1583 break;
1584 }
1585 }
1586 if (nslot != MAX_INTR_SOURCES) {
1587 break;
1588 }
1589 }
1590 }
1591 if (nslot == MAX_INTR_SOURCES) {
1592 return false;
1593 }
1594
1595 /*
1596 * Now we have new CPU and new slot. Run a cross-call to set up
1597 * the new vector on the target CPU.
1598 */
1599 where = xc_unicast(0, intr_redistribute_xc_t, isp,
1600 (void *)(intptr_t)nslot, nci);
1601 xc_wait(where);
1602
1603 /*
1604 * We're ready to go on the target CPU. Run a cross call to
1605 * reroute the interrupt away from the source CPU.
1606 */
1607 where = xc_unicast(0, intr_redistribute_xc_s1, isp, nci, oci);
1608 xc_wait(where);
1609
1610 /* Sleep for (at least) 10ms to allow the change to take hold. */
1611 (void)kpause("intrdist", false, mstohz(10), NULL);
1612
1613 /* Complete removal from the source CPU. */
1614 where = xc_unicast(0, intr_redistribute_xc_s2,
1615 (void *)(uintptr_t)oslot, NULL, oci);
1616 xc_wait(where);
1617
1618 /* Finally, take care of book-keeping. */
1619 for (ih = isp->is_handlers; ih != NULL; ih = ih->ih_next) {
1620 oci->ci_nintrhand--;
1621 nci->ci_nintrhand++;
1622 ih->ih_cpu = nci;
1623 }
1624 intr_save_evcnt(isp, oci->ci_cpuid);
1625 intr_restore_evcnt(isp, nci->ci_cpuid);
1626 isp->is_active_cpu = nci->ci_cpuid;
1627
1628 return true;
1629}

1795static int
1796intr_set_affinity(struct intrsource *isp, const kcpuset_t *cpuset)
1797{
1798 struct cpu_info *oldci, *newci;
1799 struct intrhand *ih, *lih;
1800 struct pic *pic;
1801 u_int cpu_idx;
1802 int idt_vec;
1803 int oldslot, newslot;
1804 int err;
1805 int pin;
1806
1807 KASSERT(mutex_owned(&intr_distribute_lock));
1808 KASSERT(mutex_owned(&cpu_lock));
1809
1810 /* XXX
1811 * logical destination mode is not supported, use lowest index cpu.
1812 */
1813 cpu_idx = kcpuset_ffs(cpuset) - 1;
1814 newci = cpu_lookup(cpu_idx);
1815 if (newci == NULL) {
1816 DPRINTF(("invalid cpu index: %u\n", cpu_idx));
1817 return EINVAL;
1818 }
1819 if ((newci->ci_schedstate.spc_flags & SPCF_NOINTR) != 0) {
1820 DPRINTF(("the cpu is set nointr shield. index:%u\n", cpu_idx));
1821 return EINVAL;
1822 }
1823
1824 if (isp == NULL) {
1825 DPRINTF(("invalid intrctl handler\n"));
1826 return EINVAL;
1827 }
1828
1829 /* i8259_pic supports only primary cpu, see i8259.c. */
1830 pic = isp->is_pic;
1831 if (pic == &i8259_pic) {
1832 DPRINTF(("i8259 pic does not support set_affinity\n"));
1833 return ENOTSUP;
1834 }
1835
1836 ih = isp->is_handlers;
1837 KASSERTMSG(ih != NULL,
1838 "Don't set affinity for the device which is not established.");
1839
1840 oldci = ih->ih_cpu;
1841 if (newci == oldci) /* nothing to do */
1842 return 0;
1843
1844 oldslot = ih->ih_slot;
1845 idt_vec = isp->is_idtvec;
1846
1847 err = intr_find_unused_slot(newci, &newslot);
1848 if (err) {
1849 DPRINTF(("failed to allocate interrupt slot for PIC %s intrid "
1850 "%s\n", isp->is_pic->pic_name, isp->is_intrid));
1851 return err;
1852 }
1853
1854 pin = isp->is_pin;
1855 (*pic->pic_hwmask)(pic, pin); /* for ci_ipending check */
1856 while (oldci->ci_ipending & (1 << oldslot))
1857 (void)kpause("intrdist", false, 1, &cpu_lock);
1858
1859 kpreempt_disable();
1860
1861 /* deactivate old interrupt setting */
1862 if (oldci == curcpu() || !mp_online) {
1863 intr_deactivate_xcall(ih, NULL);
1864 } else {
1865 uint64_t where;
1866 where = xc_unicast(0, intr_deactivate_xcall, ih,
1867 NULL, oldci);
1868 xc_wait(where);
1869 }
1870 intr_save_evcnt(isp, oldci->ci_cpuid);
1871 (*pic->pic_delroute)(pic, oldci, pin, idt_vec, isp->is_type);
1872
1873 /* activate new interrupt setting */
1874 newci->ci_isources[newslot] = isp;
1875 for (lih = ih; lih != NULL; lih = lih->ih_next) {
1876 newci->ci_nintrhand++;
1877 lih->ih_cpu = newci;
1878 lih->ih_slot = newslot;
1879 }
1880 if (newci == curcpu() || !mp_online) {
1881 intr_activate_xcall(ih, NULL);
1882 } else {
1883 uint64_t where;
1884 where = xc_unicast(0, intr_activate_xcall, ih,
1885 NULL, newci);
1886 xc_wait(where);
1887 }
1888 intr_restore_evcnt(isp, newci->ci_cpuid);
1889 isp->is_active_cpu = newci->ci_cpuid;
1890 (*pic->pic_addroute)(pic, newci, pin, idt_vec, isp->is_type);
1891
1892 kpreempt_enable();
1893
1894 (*pic->pic_hwunmask)(pic, pin);
1895
1896 return err;
1897}
