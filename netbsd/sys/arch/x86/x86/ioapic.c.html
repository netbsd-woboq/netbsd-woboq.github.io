119static inline u_long
120ioapic_lock(struct ioapic_softc *sc)
121{
122 u_long flags;
123
124 flags = x86_read_psl();
125 x86_disable_intr();
126 __cpu_simple_lock(&sc->sc_pic.pic_lock);
127 return flags;
128}
129
130static inline void
131ioapic_unlock(struct ioapic_softc *sc, u_long flags)
132{
133 __cpu_simple_unlock(&sc->sc_pic.pic_lock);
134 x86_write_psl(flags);
135}

137#ifndef _IOAPIC_CUSTOM_RW
138/*
139 * Register read/write routines.
140 */
141static inline uint32_t
142ioapic_read_ul(struct ioapic_softc *sc, int regid)
143{
144 uint32_t val;
145
146 *(sc->sc_reg) = regid;
147 val = *sc->sc_data;
148
149 return val;
150}
151
152static inline void
153ioapic_write_ul(struct ioapic_softc *sc, int regid, uint32_t val)
154{
155 *(sc->sc_reg) = regid;
156 *(sc->sc_data) = val;
157}
158#endif /* !_IOAPIC_CUSTOM_RW */
159
160static inline uint32_t
161ioapic_read(struct ioapic_softc *sc, int regid)
162{
163 uint32_t val;
164 u_long flags;
165
166 flags = ioapic_lock(sc);
167 val = ioapic_read_ul(sc, regid);
168 ioapic_unlock(sc, flags);
169 return val;
170}
171
172static inline void
173ioapic_write(struct ioapic_softc *sc, int regid, int val)
174{
175 u_long flags;
176
177 flags = ioapic_lock(sc);
178 ioapic_write_ul(sc, regid, val);
179 ioapic_unlock(sc, flags);
180}

182struct ioapic_softc *
183ioapic_find(int apicid)
184{
185 struct ioapic_softc *sc;
186
187 if (apicid == MPS_ALL_APICS) { /* XXX mpbios-specific */
188 /*
189 * XXX kludge for all-ioapics interrupt support
190 * on single ioapic systems
191 */
192 if (nioapics <= 1)
193 return ioapics;
194 panic("unsupported: all-ioapics interrupt with >1 ioapic");
195 }
196
197 for (sc = ioapics; sc != NULL; sc = sc->sc_next)
198 if (sc->sc_pic.pic_apicid == apicid)
199 return sc;
200
201 return NULL;
202}
203
204/*
205 * For the case the I/O APICs were configured using ACPI, there must
206 * be an option to match global ACPI interrupts with APICs.
207 */
208struct ioapic_softc *
209ioapic_find_bybase(int vec)
210{
211 struct ioapic_softc *sc;
212
213 for (sc = ioapics; sc != NULL; sc = sc->sc_next) {
214 if (vec >= sc->sc_pic.pic_vecbase &&
215 vec < (sc->sc_pic.pic_vecbase + sc->sc_apic_sz))
216 return sc;
217 }
218
219 return NULL;
220}

222static inline void
223ioapic_add(struct ioapic_softc *sc)
224{
225 struct ioapic_softc **scp;
226
227 sc->sc_next = NULL;
228
229 for (scp = &ioapics; *scp != NULL; scp = &(*scp)->sc_next)
230 ;
231 *scp = sc;
232 nioapics++;
233}

235void
236ioapic_print_redir(struct ioapic_softc *sc, const char *why, int pin)
237{
238 uint32_t redirlo = ioapic_read(sc, IOAPIC_REDLO(pin));
239 uint32_t redirhi = ioapic_read(sc, IOAPIC_REDHI(pin));
240
241 apic_format_redir(device_xname(sc->sc_dev), why, pin,
242 APIC_VECTYPE_IOAPIC, redirhi, redirlo);
243}

255/*
256 * can't use bus_space_xxx as we don't have a bus handle ...
257 */
258void
259ioapic_attach(device_t parent, device_t self, void *aux)
260{
261 struct ioapic_softc *sc = device_private(self);
262 struct apic_attach_args *aaa = (struct apic_attach_args *)aux;
263 int apic_id;
264 uint32_t ver_sz;
265 int i;
266
267 sc->sc_dev = self;
268 sc->sc_flags = aaa->flags;
269 sc->sc_pic.pic_apicid = aaa->apic_id;
270 sc->sc_pic.pic_name = device_xname(self);
271 sc->sc_pic.pic_ioapic = sc;
272
273 aprint_naive("\n");
274
275 if (ioapic_find(aaa->apic_id) != NULL) {
276 aprint_error(": duplicate apic id (ignored)\n");
277 return;
278 }
279
280 aprint_verbose(": pa 0x%jx", (uintmax_t)aaa->apic_address);
281#ifndef _IOAPIC_CUSTOM_RW
282 {
283 bus_space_handle_t bh;
284
285 if (x86_mem_add_mapping(aaa->apic_address, PAGE_SIZE, 0, &bh) != 0) {
286 aprint_error(": map failed\n");
287 return;
288 }
289 sc->sc_reg = (volatile uint32_t *)(bh + IOAPIC_REG);
290 sc->sc_data = (volatile uint32_t *)(bh + IOAPIC_DATA);
291 }
292#endif
293 sc->sc_pa = aaa->apic_address;
294
295 sc->sc_pic.pic_type = PIC_IOAPIC;
296 __cpu_simple_lock_init(&sc->sc_pic.pic_lock);
297 sc->sc_pic.pic_hwmask = ioapic_hwmask;
298 sc->sc_pic.pic_hwunmask = ioapic_hwunmask;
299 sc->sc_pic.pic_addroute = ioapic_addroute;
300 sc->sc_pic.pic_delroute = ioapic_delroute;
301 sc->sc_pic.pic_trymask = ioapic_trymask;
302 sc->sc_pic.pic_edge_stubs = ioapic_edge_stubs;
303 sc->sc_pic.pic_level_stubs = ioapic_level_stubs;
304
305 apic_id = (ioapic_read(sc, IOAPIC_ID) & IOAPIC_ID_MASK)
306 >> IOAPIC_ID_SHIFT;
307 ver_sz = ioapic_read(sc, IOAPIC_VER);
308
309 if (ver_sz == 0xffffffff) {
310 aprint_error(": failed to read version/size\n");
311 goto out;
312 }
313
314 ioapic_add(sc);
315
316 sc->sc_apic_vers = (ver_sz & IOAPIC_VER_MASK) >> IOAPIC_VER_SHIFT;
317 sc->sc_apic_sz = (ver_sz & IOAPIC_MAX_MASK) >> IOAPIC_MAX_SHIFT;
318 sc->sc_apic_sz++;
319
320 if (aaa->apic_vecbase != -1)
321 sc->sc_pic.pic_vecbase = aaa->apic_vecbase;
322 else {
323 /*
324 * XXX this assumes ordering of ioapics in the table.
325 * Only needed for broken BIOS workaround (see mpbios.c)
326 */
327 sc->sc_pic.pic_vecbase = ioapic_vecbase;
328 ioapic_vecbase += sc->sc_apic_sz;
329 }
330
331 if (mp_verbose) {
332 printf(", %s mode",
333 aaa->flags & IOAPIC_PICMODE ? "PIC" : "virtual wire");
334 }
335
336 aprint_verbose(", version 0x%x, %d pins", sc->sc_apic_vers,
337 sc->sc_apic_sz);
338 aprint_normal("\n");
339
340 sc->sc_pins = malloc(sizeof(struct ioapic_pin) * sc->sc_apic_sz,
341 M_DEVBUF, M_WAITOK);
342
343 for (i = 0; i < sc->sc_apic_sz; i++) {
344 uint32_t redlo, redhi;
345
346 sc->sc_pins[i].ip_next = NULL;
347 sc->sc_pins[i].ip_map = NULL;
348 sc->sc_pins[i].ip_vector = 0;
349 sc->sc_pins[i].ip_type = IST_NONE;
350
351 /* Mask all pins by default. */
352 redlo = IOAPIC_REDLO_MASK;
353 /*
354 * ISA interrupts are connect to pin 0-15 and
355 * edge triggered on high, which is the default.
356 *
357 * Expect all other interrupts to be PCI-like
358 * level triggered on low.
359 */
360 if (i >= 16)
361 redlo |= IOAPIC_REDLO_LEVEL | IOAPIC_REDLO_ACTLO;
362 redhi = (cpu_info_primary.ci_cpuid << IOAPIC_REDHI_DEST_SHIFT);
363 ioapic_write(sc, IOAPIC_REDHI(i), redhi);
364 ioapic_write(sc, IOAPIC_REDLO(i), redlo);
365 }
366
367 /*
368 * In case the APIC is not initialized to the correct ID
369 * do it now.
370 * Maybe we should record the original ID for interrupt
371 * mapping later ...
372 */
373 if (apic_id != sc->sc_pic.pic_apicid) {
374 aprint_debug_dev(sc->sc_dev, "misconfigured as apic %d\n",
375 apic_id);
376
377 ioapic_write(sc, IOAPIC_ID,
378 (ioapic_read(sc, IOAPIC_ID) & ~IOAPIC_ID_MASK)
379 | (sc->sc_pic.pic_apicid << IOAPIC_ID_SHIFT));
380
381 apic_id = (ioapic_read(sc, IOAPIC_ID) & IOAPIC_ID_MASK)
382 >> IOAPIC_ID_SHIFT;
383
384 if (apic_id != sc->sc_pic.pic_apicid)
385 aprint_error_dev(sc->sc_dev,
386 "can't remap to apid %d\n", sc->sc_pic.pic_apicid);
387 else
388 aprint_debug_dev(sc->sc_dev, "remapped to apic %d\n",
389 sc->sc_pic.pic_apicid);
390 }
391
392 out:
393 if (!pmf_device_register(self, NULL, NULL))
394 aprint_error_dev(self, "couldn't establish power handler\n");
395
396#if 0
397 /* output of this was boring. */
398 if (mp_verbose)
399 for (i = 0; i < sc->sc_apic_sz; i++)
400 ioapic_print_redir(sc, "boot", i);
401#endif
402}

404static void
405apic_set_redir(struct ioapic_softc *sc, int pin, int idt_vec,
406 struct cpu_info *ci)
407{
408 uint32_t redlo;
409 uint32_t redhi;
410 int delmode;
411 struct ioapic_pin *pp;
412 struct mp_intr_map *map;
413
414 pp = &sc->sc_pins[pin];
415 map = pp->ip_map;
416 redlo = map == NULL ? IOAPIC_REDLO_MASK : map->redir;
417 redhi = 0;
418 delmode = (redlo & IOAPIC_REDLO_DEL_MASK) >> IOAPIC_REDLO_DEL_SHIFT;
419
420 if (delmode == IOAPIC_REDLO_DEL_FIXED ||
421 delmode == IOAPIC_REDLO_DEL_LOPRI) {
422 if (pp->ip_type == IST_NONE) {
423 redlo |= IOAPIC_REDLO_MASK;
424 } else {
425 redhi = (ci->ci_cpuid << IOAPIC_REDHI_DEST_SHIFT);
426 redlo |= (idt_vec & 0xff);
427 redlo |= IOAPIC_REDLO_DEL_FIXED
428 << IOAPIC_REDLO_DEL_SHIFT;
429 redlo &= ~IOAPIC_REDLO_DSTMOD;
430
431 /* XXX derive this bit from BIOS info */
432 if (pp->ip_type == IST_LEVEL)
433 redlo |= IOAPIC_REDLO_LEVEL;
434 else
435 redlo &= ~IOAPIC_REDLO_LEVEL;
436 if ((map != NULL)
437 && ((map->flags & 3) == MPS_INTPO_DEF)) {
438 if (pp->ip_type == IST_LEVEL)
439 redlo |= IOAPIC_REDLO_ACTLO;
440 else
441 redlo &= ~IOAPIC_REDLO_ACTLO;
442 }
443 }
444 }
445 ioapic_write(sc, IOAPIC_REDHI(pin), redhi);
446 ioapic_write(sc, IOAPIC_REDLO(pin), redlo);
447 if (mp_verbose)
448 ioapic_print_redir(sc, "int", pin);
449}

454
455void
456ioapic_enable(void)
457{
458 if (ioapics == NULL)
459 return;
460
461 i8259_setmask(0xffff);
462
463 if (ioapics->sc_flags & IOAPIC_PICMODE) {
464 aprint_debug_dev(ioapics->sc_dev,
465 "writing to IMCR to disable pics\n");
466 outb(IMCR_ADDR, IMCR_REGISTER);
467 outb(IMCR_DATA, IMCR_APIC);
468 }
469}

471void
472ioapic_reenable(void)
473{
474 int p, apic_id;
475 struct ioapic_softc *sc;
476
477 if (ioapics == NULL)
478 return;
479
480 aprint_normal("%s reenabling\n", device_xname(ioapics->sc_dev));
481
482 for (sc = ioapics; sc != NULL; sc = sc->sc_next) {
483 apic_id = (ioapic_read(sc, IOAPIC_ID) & IOAPIC_ID_MASK)
484 >> IOAPIC_ID_SHIFT;
485 if (apic_id != sc->sc_pic.pic_apicid) {
486 ioapic_write(sc, IOAPIC_ID,
487 (ioapic_read(sc, IOAPIC_ID) & ~IOAPIC_ID_MASK)
488 | (sc->sc_pic.pic_apicid << IOAPIC_ID_SHIFT));
489 }
490
491 for (p = 0; p < sc->sc_apic_sz; p++)
492 apic_set_redir(sc, p, sc->sc_pins[p].ip_vector,
493 sc->sc_pins[p].ip_cpu);
494 }
495
496 ioapic_enable();
497}

499void
500ioapic_hwmask(struct pic *pic, int pin)
501{
502 uint32_t redlo;
503 struct ioapic_softc *sc = pic->pic_ioapic;
504 u_long flags;
505
506 flags = ioapic_lock(sc);
507 redlo = ioapic_read_ul(sc, IOAPIC_REDLO(pin));
508 redlo |= IOAPIC_REDLO_MASK;
509 redlo &= ~IOAPIC_REDLO_RIRR;
510 ioapic_write_ul(sc, IOAPIC_REDLO(pin), redlo);
511 ioapic_unlock(sc, flags);
512}
513
514bool
515ioapic_trymask(struct pic *pic, int pin)
516{
517 uint32_t redlo;
518 struct ioapic_softc *sc = pic->pic_ioapic;
519 u_long flags;
520 bool rv;
521
522 /* Mask it. */
523 flags = ioapic_lock(sc);
524 redlo = ioapic_read_ul(sc, IOAPIC_REDLO(pin));
525 redlo |= IOAPIC_REDLO_MASK;
526 ioapic_write_ul(sc, IOAPIC_REDLO(pin), redlo);
527
528 /* If pending, unmask and abort. */
529 redlo = ioapic_read_ul(sc, IOAPIC_REDLO(pin));
530 if ((redlo & (IOAPIC_REDLO_RIRR | IOAPIC_REDLO_DELSTS)) != 0) {
531 redlo &= ~IOAPIC_REDLO_MASK;
532 ioapic_write_ul(sc, IOAPIC_REDLO(pin), redlo);
533 rv = false;
534 } else {
535 rv = true;
536 }
537 ioapic_unlock(sc, flags);
538 return rv;
539}

541void
542ioapic_hwunmask(struct pic *pic, int pin)
543{
544 uint32_t redlo;
545 struct ioapic_softc *sc = pic->pic_ioapic;
546 u_long flags;
547
548 flags = ioapic_lock(sc);
549 redlo = ioapic_read_ul(sc, IOAPIC_REDLO(pin));
550 redlo &= ~(IOAPIC_REDLO_MASK | IOAPIC_REDLO_RIRR);
551 ioapic_write_ul(sc, IOAPIC_REDLO(pin), redlo);
552 ioapic_unlock(sc, flags);
553}

555static void
556ioapic_addroute(struct pic *pic, struct cpu_info *ci, int pin,
557 int idtvec, int type)
558{
559 struct ioapic_softc *sc = pic->pic_ioapic;
560 struct ioapic_pin *pp;
561
562 pp = &sc->sc_pins[pin];
563 pp->ip_type = type;
564 pp->ip_vector = idtvec;
565 pp->ip_cpu = ci;
566 apic_set_redir(sc, pin, idtvec, ci);
567}

581void
582ioapic_dump(void)
583{
584 struct ioapic_softc *sc;
585 struct ioapic_pin *ip;
586 int p;
587
588 for (sc = ioapics; sc != NULL; sc = sc->sc_next) {
589 for (p = 0; p < sc->sc_apic_sz; p++) {
590 ip = &sc->sc_pins[p];
591 if (ip->ip_type != IST_NONE)
592 ioapic_print_redir(sc, "dump", p);
593 }
594 }
595}

597void
598ioapic_dump_raw(void)
599{
600 struct ioapic_softc *sc;
601 int i;
602 uint32_t reg;
603
604 for (sc = ioapics; sc != NULL; sc = sc->sc_next) {
605 printf("Register dump of %s\n", device_xname(sc->sc_dev));
606 i = 0;
607 do {
608 if (i % 0x08 == 0)
609 printf("%02x", i);
610 reg = ioapic_read(sc, i);
611 printf(" %08x", (u_int)reg);
612 if (++i % 0x08 == 0)
613 printf("\n");
614 } while (i < IOAPIC_REDTBL + (sc->sc_apic_sz * 2));
615 }
616}
