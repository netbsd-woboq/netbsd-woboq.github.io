223static void
224lapic_enable_x2apic(void)
225{
226 uint64_t apicbase;
227
228 apicbase = rdmsr(MSR_APICBASE);
229 if (!ISSET(apicbase, APICBASE_EN)) {
230 apicbase |= APICBASE_EN;
231 wrmsr(MSR_APICBASE, apicbase);
232 }
233 apicbase |= APICBASE_EXTD;
234 wrmsr(MSR_APICBASE, apicbase);
235}
236
237bool
238lapic_is_x2apic(void)
239{
240 uint64_t msr;
241
242 if (!ISSET(cpu_feature[0], CPUID_APIC) ||
243 rdmsr_safe(MSR_APICBASE, &msr) == EFAULT)
244 return false;
245 return (msr & (APICBASE_EN | APICBASE_EXTD)) ==
246 (APICBASE_EN | APICBASE_EXTD);
247}

249/*
250 * Initialize the local APIC on the BSP.
251 */
252static void
253lapic_setup_bsp(paddr_t lapic_base)
254{
255 u_int regs[6];
256 const char *reason = NULL;
257 const char *hw_vendor;
258 bool bios_x2apic;
259
260 if (ISSET(cpu_feature[1], CPUID2_X2APIC)) {
261#if NACPICA > 0
262 if (acpi_present) {
263 ACPI_TABLE_DMAR *dmar;
264 ACPI_STATUS status;
265
266 /*
267 * Automatically detect several configurations where
268 * x2APIC mode is known to cause troubles. User can
269 * override the setting with hw.x2apic_enable tunable.
270 */
271 status = AcpiGetTable(ACPI_SIG_DMAR, 1,
272 (ACPI_TABLE_HEADER **)&dmar);
273 if (ACPI_SUCCESS(status)) {
274 if (ISSET(dmar->Flags, ACPI_DMAR_X2APIC_OPT_OUT)) {
275 reason = "by DMAR table";
276 }
277 AcpiPutTable(&dmar->Header);
278 }
279 }
280#endif /* NACPICA > 0 */
281 if (vm_guest == VM_GUEST_VMWARE) {
282 vmt_hvcall(VM_CMD_GET_VCPU_INFO, regs);
283 if (ISSET(regs[0], VCPUINFO_VCPU_RESERVED) ||
284 !ISSET(regs[0], VCPUINFO_LEGACY_X2APIC))
285 reason = "inside VMWare without intr "
286 "redirection";
287 } else if (vm_guest == VM_GUEST_XEN) {
288 reason = "due to running under XEN";
289 } else if (vm_guest == VM_GUEST_NO &&
290 CPUID_TO_FAMILY(curcpu()->ci_signature) == 6 &&
291 CPUID_TO_MODEL(curcpu()->ci_signature) == 0x2a) {
292 hw_vendor = pmf_get_platform("board-vendor");
293 if (hw_vendor != NULL) {
294 /*
295 * It seems that some Lenovo and ASUS
296 * SandyBridge-based notebook BIOSes have a bug
297 * which prevents booting AP in x2APIC mode.
298 * Since the only way to detect mobile CPU is
299 * to check northbridge pci id, which cannot be
300 * done that early, disable x2APIC for all
301 * Lenovo and ASUS SandyBridge machines.
302 */
303 if (strcmp(hw_vendor, "LENOVO") == 0 ||
304 strcmp(hw_vendor, "ASUSTeK Computer Inc.") == 0) {
305 reason = "for a suspected SandyBridge "
306 "BIOS bug";
307 }
308 }
309 }
310 bios_x2apic = lapic_is_x2apic();
311 if (reason != NULL && bios_x2apic) {
312 aprint_verbose("x2APIC should be disabled %s but "
313 "already enabled by BIOS; enabling.\n", reason);
314 reason = NULL;
315 }
316 if (reason == NULL)
317 x2apic_mode = true;
318 else
319 aprint_verbose("x2APIC available but disabled %s\n",
320 reason);
321 if (x2apic_enable != x2apic_mode) {
322 if (bios_x2apic && !x2apic_enable)
323 aprint_verbose("x2APIC disabled by user and "
324 "enabled by BIOS; ignoring user setting.\n");
325 else
326 x2apic_mode = x2apic_enable;
327 }
328 }
329 if (x2apic_mode) {
330 x86_ipi = x2apic_ipi;
331#if NIOAPIC > 0
332 struct ioapic_softc *ioapic;
333 for (ioapic = ioapics; ioapic != NULL; ioapic = ioapic->sc_next) {
334 ioapic->sc_pic.pic_edge_stubs = x2apic_edge_stubs;
335 ioapic->sc_pic.pic_level_stubs = x2apic_level_stubs;
336 }
337#endif
338#if defined(DDB) && defined(MULTIPROCESSOR)
339#ifdef __x86_64__
340 set_idtgate(&idt[ddb_vec], &Xintr_x2apic_ddbipi, 1,
341 SDT_SYS386IGT, SEL_KPL, GSEL(GCODE_SEL, SEL_KPL));
342#else
343 /*
344 * Set DDB IPI handler in cpu_set_tss_gates() when cpu0 is
345 * attached.
346 */
347#endif
348#endif
349
350 x86_disable_intr();
351 lapic_enable_x2apic();
352#ifdef MULTIPROCESSOR
353 cpu_init_first(); /* Catch up to changed cpu_number() */
354#endif
355 lapic_write_tpri(0);
356 x86_enable_intr();
357 } else
358 lapic_map(lapic_base);
359}

361static void
362lapic_map(paddr_t lapic_base)
363{
364 pt_entry_t *pte;
365 vaddr_t va = local_apic_va;
366
367 /*
368 * If the CPU has an APIC MSR, use it and ignore the supplied value:
369 * some ACPI implementations have been observed to pass bad values.
370 * Additionally, ensure that the lapic is enabled as we are committed
371 * to using it at this point. Be conservative and assume that the MSR
372 * is not present on the Pentium (is it?).
373 */
374 if (CPUID_TO_FAMILY(curcpu()->ci_signature) >= 6) {
375 lapic_base = (paddr_t)rdmsr(MSR_APICBASE);
376 if ((lapic_base & APICBASE_PHYSADDR) == 0) {
377 lapic_base |= LAPIC_BASE;
378 }
379 wrmsr(MSR_APICBASE, lapic_base | APICBASE_EN);
380 lapic_base &= APICBASE_PHYSADDR;
381 }
382
383 x86_disable_intr();
384
385 /*
386 * Map local apic. If we have a local apic, it's safe to assume
387 * we're on a 486 or better and can use invlpg and non-cacheable PTE's
388 *
389 * Whap the PTE "by hand" rather than calling pmap_kenter_pa because
390 * the latter will attempt to invoke TLB shootdown code just as we
391 * might have changed the value of cpu_number()..
392 */
393
394 pte = kvtopte(va);
395 *pte = lapic_base | PTE_W | PTE_P | PTE_PCD | pmap_pg_g | pmap_pg_nx;
396 invlpg(va);
397
398#ifdef MULTIPROCESSOR
399 cpu_init_first(); /* Catch up to changed cpu_number() */
400#endif
401
402 lapic_write_tpri(0);
403 x86_enable_intr();
404}

415void
416lapic_set_lvt(void)
417{
418 struct cpu_info *ci = curcpu();
419 int i;
420 struct mp_intr_map *mpi;
421 uint32_t lint0, lint1;
422
423#ifdef MULTIPROCESSOR
424 if (mp_verbose) {
425 apic_format_redir(device_xname(ci->ci_dev), "prelint", 0,
426 APIC_VECTYPE_LAPIC_LVT, 0, lapic_readreg(LAPIC_LVT_LINT0));
427 apic_format_redir(device_xname(ci->ci_dev), "prelint", 1,
428 APIC_VECTYPE_LAPIC_LVT, 0, lapic_readreg(LAPIC_LVT_LINT1));
429 }
430#endif
431
432 /*
433 * If an I/O APIC has been attached, assume that it is used instead of
434 * the 8259A for interrupt delivery. Otherwise request the LAPIC to
435 * get external interrupts via LINT0 for the primary CPU.
436 */
437 lint0 = LAPIC_DLMODE_EXTINT;
438 if (nioapics > 0 || !CPU_IS_PRIMARY(curcpu()))
439 lint0 |= LAPIC_LVT_MASKED;
440 lapic_writereg(LAPIC_LVT_LINT0, lint0);
441
442 /*
443 * Non Maskable Interrupts are to be delivered to the primary CPU.
444 */
445 lint1 = LAPIC_DLMODE_NMI;
446 if (!CPU_IS_PRIMARY(curcpu()))
447 lint1 |= LAPIC_LVT_MASKED;
448 lapic_writereg(LAPIC_LVT_LINT1, lint1);
449
450 for (i = 0; i < mp_nintr; i++) {
451 mpi = &mp_intrs[i];
452 if (mpi->ioapic == NULL && (mpi->cpu_id == MPS_ALL_APICS ||
453 mpi->cpu_id == ci->ci_cpuid)) {
454 if (mpi->ioapic_pin > 1)
455 aprint_error_dev(ci->ci_dev,
456 "%s: WARNING: bad pin value %d\n",
457 __func__, mpi->ioapic_pin);
458 if (mpi->ioapic_pin == 0)
459 lapic_writereg(LAPIC_LVT_LINT0, mpi->redir);
460 else
461 lapic_writereg(LAPIC_LVT_LINT1, mpi->redir);
462 }
463 }
464
465#ifdef MULTIPROCESSOR
466 if (mp_verbose)
467 lapic_dump();
468#endif
469}

515static u_int
516lapic_get_timecount(struct timecounter *tc)
517{
518 struct cpu_info *ci;
519 uint32_t cur_timer;
520 int s;
521
522 s = splhigh();
523 ci = curcpu();
524
525 /*
526 * Check for a race against the clockinterrupt.
527 * The update of ci_lapic_counter is blocked by splhigh() and
528 * the check for a pending clockinterrupt compensates for that.
529 *
530 * If the current tick is almost the Initial Counter, explicitly
531 * check for the pending interrupt bit as the interrupt delivery
532 * could be asynchronious and compensate as well.
533 *
534 * This can't be done without splhigh() as the calling code might
535 * have masked the clockinterrupt already.
536 *
537 * This code assumes that clockinterrupts are not missed.
538 */
539 cur_timer = lapic_gettick();
540 if (cur_timer >= lapic_tval - 1) {
541 uint16_t reg = LAPIC_IRR + LAPIC_TIMER_VECTOR / 32 * 16;
542
543 if (lapic_readreg(reg) & (1 << (LAPIC_TIMER_VECTOR % 32))) {
544 cur_timer -= lapic_tval;
545 }
546 } else if (ci->ci_ipending & (1 << LIR_TIMER))
547 cur_timer = lapic_gettick() - lapic_tval;
548 cur_timer = ci->ci_lapic_counter - cur_timer;
549 splx(s);
550
551 return cur_timer;
552}

581void
582lapic_initclocks(void)
583{
584 /*
585 * Start local apic countdown timer running, in repeated mode.
586 *
587 * Mask the clock interrupt and set mode,
588 * then set divisor,
589 * then unmask and set the vector.
590 */
591 lapic_writereg(LAPIC_LVT_TIMER,
592 LAPIC_LVT_TMM_PERIODIC | LAPIC_LVT_MASKED);
593 lapic_writereg(LAPIC_DCR_TIMER, LAPIC_DCRT_DIV1);
594 lapic_writereg(LAPIC_ICR_TIMER, lapic_tval);
595 lapic_writereg(LAPIC_LVT_TIMER,
596 LAPIC_LVT_TMM_PERIODIC | LAPIC_TIMER_VECTOR);
597 lapic_eoi();
598}

603/*
604 * Calibrate the local apic count-down timer (which is running at
605 * bus-clock speed) vs. the i8254 counter/timer (which is running at
606 * a fixed rate).
607 *
608 * The Intel MP spec says: "An MP operating system may use the IRQ8
609 * real-time clock as a reference to determine the actual APIC timer clock
610 * speed."
611 *
612 * We're actually using the IRQ0 timer. Hmm.
613 */
614void
615lapic_calibrate_timer(struct cpu_info *ci)
616{
617 unsigned int seen, delta, initial_i8254, initial_lapic;
618 unsigned int cur_i8254, cur_lapic;
619 uint64_t tmp;
620 int i;
621 char tbuf[9];
622
623 if (lapic_per_second != 0)
624 goto calibrate_done;
625
626 aprint_debug_dev(ci->ci_dev, "calibrating local timer\n");
627
628 /*
629 * Configure timer to one-shot, interrupt masked,
630 * large positive number.
631 */
632 lapic_writereg(LAPIC_LVT_TIMER, LAPIC_LVT_MASKED);
633 lapic_writereg(LAPIC_DCR_TIMER, LAPIC_DCRT_DIV1);
634 lapic_writereg(LAPIC_ICR_TIMER, 0x80000000);
635
636 x86_disable_intr();
637
638 initial_lapic = lapic_gettick();
639 initial_i8254 = gettick();
640
641 for (seen = 0; seen < TIMER_FREQ / 100; seen += delta) {
642 cur_i8254 = gettick();
643 if (cur_i8254 > initial_i8254)
644 delta = rtclock_tval - (cur_i8254 - initial_i8254);
645 else
646 delta = initial_i8254 - cur_i8254;
647 initial_i8254 = cur_i8254;
648 }
649 cur_lapic = lapic_gettick();
650
651 x86_enable_intr();
652
653 tmp = initial_lapic - cur_lapic;
654 lapic_per_second = (tmp * TIMER_FREQ + seen / 2) / seen;
655
656calibrate_done:
657 humanize_number(tbuf, sizeof(tbuf), lapic_per_second, "Hz", 1000);
658
659 aprint_debug_dev(ci->ci_dev, "apic clock running at %s\n", tbuf);
660
661 if (lapic_per_second != 0) {
662 /*
663 * reprogram the apic timer to run in periodic mode.
664 * XXX need to program timer on other CPUs, too.
665 */
666 lapic_tval = (lapic_per_second * 2) / hz;
667 lapic_tval = (lapic_tval / 2) + (lapic_tval & 0x1);
668
669 lapic_writereg(LAPIC_LVT_TIMER, LAPIC_LVT_TMM_PERIODIC
670 | LAPIC_LVT_MASKED | LAPIC_TIMER_VECTOR);
671 lapic_writereg(LAPIC_DCR_TIMER, LAPIC_DCRT_DIV1);
672 lapic_writereg(LAPIC_ICR_TIMER, lapic_tval);
673
674 /*
675 * Compute fixed-point ratios between cycles and
676 * microseconds to avoid having to do any division
677 * in lapic_delay.
678 */
679
680 tmp = (1000000 * (uint64_t)1 << 32) / lapic_per_second;
681 lapic_frac_usec_per_cycle = tmp;
682
683 tmp = (lapic_per_second * (uint64_t)1 << 32) / 1000000;
684
685 lapic_frac_cycle_per_usec = tmp;
686
687 /*
688 * Compute delay in cycles for likely short delays in usec.
689 */
690 for (i = 0; i < 26; i++)
691 lapic_delaytab[i] = (lapic_frac_cycle_per_usec * i) >>
692 32;
693
694 /*
695 * Apply workaround for broken periodic timer under KVM
696 */
697 if (vm_guest == VM_GUEST_KVM) {
698 lapic_broken_periodic = true;
699 lapic_timecounter.tc_quality = -100;
700 aprint_debug_dev(ci->ci_dev,
701 "applying KVM timer workaround\n");
702 }
703
704 /*
705 * Now that the timer's calibrated, use the apic timer routines
706 * for all our timing needs..
707 */
708 delay_func = lapic_delay;
709 initclock_func = lapic_initclocks;
710 initrtclock(0);
711
712 if (lapic_timecounter.tc_frequency == 0) {
713 /*
714 * Hook up time counter.
715 * This assume that all LAPICs have the same frequency.
716 */
717 lapic_timecounter.tc_frequency = lapic_per_second;
718 tc_init(&lapic_timecounter);
719 }
720 }
721}
722
723/*
724 * delay for N usec.
725 */
726
727static void
728lapic_delay(unsigned int usec)
729{
730 int32_t xtick, otick;
731 int64_t deltat; /* XXX may want to be 64bit */
732
733 otick = lapic_gettick();
734
735 if (usec <= 0)
736 return;
737 if (usec <= 25)
738 deltat = lapic_delaytab[usec];
739 else
740 deltat = (lapic_frac_cycle_per_usec * usec) >> 32;
741
742 while (deltat > 0) {
743 xtick = lapic_gettick();
744 if (lapic_broken_periodic && xtick == 0 && otick == 0) {
745 lapic_initclocks();
746 xtick = lapic_gettick();
747 if (xtick == 0)
748 panic("lapic timer stopped ticking");
749 }
750 if (xtick > otick)
751 deltat -= lapic_tval - (xtick - otick);
752 else
753 deltat -= otick - xtick;
754 otick = xtick;
755
756 x86_pause();
757 }
758}

781static int
782i82489_ipi_init(int target)
783{
784 uint32_t esr;
785
786 i82489_writereg(LAPIC_ESR, 0);
787 (void)i82489_readreg(LAPIC_ESR);
788
789 i82489_writereg(LAPIC_ICRHI, target << LAPIC_ID_SHIFT);
790
791 i82489_writereg(LAPIC_ICRLO, LAPIC_DLMODE_INIT | LAPIC_LEVEL_ASSERT);
792 i82489_icr_wait();
793 x86_delay(10000);
794 i82489_writereg(LAPIC_ICRLO,
795 LAPIC_DLMODE_INIT | LAPIC_TRIGMODE_LEVEL | LAPIC_LEVEL_DEASSERT);
796 i82489_icr_wait();
797
798 if ((i82489_readreg(LAPIC_ICRLO) & LAPIC_DLSTAT_BUSY) != 0)
799 return EBUSY;
800
801 esr = i82489_readreg(LAPIC_ESR);
802 if (esr != 0)
803 aprint_debug("%s: ESR %08x\n", __func__, esr);
804
805 return 0;
806}

832static int
833i82489_ipi(int vec, int target, int dl)
834{
835 int result, s;
836
837 s = splhigh();
838
839 i82489_icr_wait();
840
841 if ((target & LAPIC_DEST_MASK) == 0)
842 i82489_writereg(LAPIC_ICRHI, target << LAPIC_ID_SHIFT);
843
844 i82489_writereg(LAPIC_ICRLO,
845 (target & LAPIC_DEST_MASK) | vec | dl | LAPIC_LEVEL_ASSERT);
846
847#ifdef DIAGNOSTIC
848 i82489_icr_wait();
849 result = (i82489_readreg(LAPIC_ICRLO) & LAPIC_DLSTAT_BUSY) ? EBUSY : 0;
850#else
851 /* Don't wait - if it doesn't go, we're in big trouble anyway. */
852 result = 0;
853#endif
854 splx(s);
855
856 return result;
857}

915 * 0 - timer
916 * 1 - thermal
917 * 2 - PCINT
918 * 3 - LVINT0
919 * 4 - LVINT1
920 * 5 - LVERR

923static void
924lapic_hwmask(struct pic *pic, int pin)
925{
926 int reg;
927 uint32_t val;
928
929 reg = LAPIC_LVT_TIMER + (pin << 4);
930 val = lapic_readreg(reg);
931 val |= LAPIC_LVT_MASKED;
932 lapic_writereg(reg, val);
933}

945}
