32/*
33 * nmi dispatcher.
34 *
35 * XXX no need to be nmi-specific.
36 * actual assumptions are:
37 * - dispatch() is called with preemption disabled.
38 * - handlers never block.
39 * - establish() and disestablish() are called within a thread context.
40 * (thus can block)
41 */
42
43#include <sys/param.h>
44#include <sys/kernel.h>
45#include <sys/systm.h>
46#include <sys/atomic.h>
47#include <sys/kmem.h>
48#include <sys/mutex.h>
49#include <sys/pserialize.h>
50
51#include <x86/nmi.h>
52
53struct nmi_handler {
54 int (*n_func)(const struct trapframe *, void *);
55 void *n_arg;
56 struct nmi_handler *n_next;
57};
58
59static kmutex_t nmi_list_lock; /* serialize establish and disestablish */
60static pserialize_t nmi_psz;
61static nmi_handler_t *nmi_handlers; /* list of handlers */
62
63/*
64 * nmi_establish: establish an nmi handler
65 *
66 * => can block.
67 * => returns an opaque handle.
68 */
69
70nmi_handler_t *
71nmi_establish(int (*func)(const struct trapframe *, void *), void *arg)
72{
73 struct nmi_handler *n;
74
75 n = kmem_alloc(sizeof(*n), KM_SLEEP);
76 n->n_func = func;
77 n->n_arg = arg;
78
79 /*
80 * put it into the list.
81 */
82
83 mutex_enter(&nmi_list_lock);
84 n->n_next = nmi_handlers;
85 membar_producer(); /* n->n_next should be visible before nmi_handlers */
86 nmi_handlers = n; /* atomic store */
87 mutex_exit(&nmi_list_lock);
88
89 return n;
90}
91
92/*
93 * nmi_disestablish: disestablish an nmi handler.
94 *
95 * => can block.
96 * => take an opaque handle. it must be one returned by nmi_establish.
97 */
98
99void
100nmi_disestablish(nmi_handler_t *handle)
101{
102 nmi_handler_t *n;
103 nmi_handler_t **pp;
104
105 KASSERT(handle != NULL);
106
107 /*
108 * remove the handler from the list.
109 */
110
111 mutex_enter(&nmi_list_lock);
112 for (pp = &nmi_handlers, n = *pp; n != NULL; n = *pp) {
113 if (n == handle) {
114 break;
115 }
116 pp = &n->n_next;
117 }
118#if defined(DIAGNOSTIC)
119 if (n == NULL) {
120 mutex_exit(&nmi_list_lock);
121 panic("%s: invalid handle %p", __func__, handle);
122 }
123#endif /* defined(DIAGNOSTIC) */
124 *pp = n->n_next; /* atomic store */
125 mutex_exit(&nmi_list_lock); /* mutex_exit implies a store fence */
126
127 /*
128 * before freeing 'n', ensure that no other cpus are
129 * in the middle of nmi_dispatch.
130 */
131
132 pserialize_perform(nmi_psz);
133 kmem_free(n, sizeof(*n));
134}
135
136/*
137 * nmi_dispatch: dispatch an nmi.
138 *
139 * => called by interrupts, thus preempt disabled.
140 */
141
142int
143nmi_dispatch(const struct trapframe *tf)
144{
145 const struct nmi_handler *n;
146 int handled = 0;
147
148 /*
149 * XXX abstraction violation
150 *
151 * we don't bother to call pserialize_read_enter/pserialize_read_exit
152 * because they are not necessary here as we are sure our IPL is
153 * higher than IPL_SOFTSERIAL. better to avoid unnecessary calls as
154 * we are in a dangerous context. (NMI)
155 */
156
157 for (n = nmi_handlers; /* atomic load */
158 n != NULL;
159 membar_consumer(), n = n->n_next) { /* atomic load */
160 handled |= (*n->n_func)(tf, n->n_arg);
161 }
162 return handled;
163}
164
165void
166nmi_init(void)
167{
168
169 mutex_init(&nmi_list_lock, MUTEX_DEFAULT, IPL_NONE);
170 nmi_psz = pserialize_create();
171}
