68void
69platform_init(void)
70{
71 struct smbtable smbios;
72 struct smbios_sys *psys;
73 struct smbios_struct_bios *pbios;
74 struct smbios_board *pboard;
75 struct smbios_chassis *pchassis;
76 struct smbios_processor *pproc;
77 struct smbios_slot *pslot;
78 int nisa, nother;
79
80 smbios.cookie = 0;
81 if (smbios_find_table(SMBIOS_TYPE_SYSTEM, &smbios)) {
82 psys = smbios.tblhdr;
83
84 platform_add(&smbios, "system-vendor", psys->vendor);
85 platform_add(&smbios, "system-product", psys->product);
86 platform_add(&smbios, "system-version", psys->version);
87 platform_add(&smbios, "system-serial", psys->serial);
88 platform_add_uuid(&smbios, "system-uuid", psys->uuid);
89 }
90
91 smbios.cookie = 0;
92 if (smbios_find_table(SMBIOS_TYPE_BIOS, &smbios)) {
93 pbios = smbios.tblhdr;
94
95 platform_add(&smbios, "bios-vendor", pbios->vendor);
96 platform_add(&smbios, "bios-version", pbios->version);
97 platform_add_date(&smbios, "bios-date", pbios->release);
98 }
99
100 smbios.cookie = 0;
101 if (smbios_find_table(SMBIOS_TYPE_BASEBOARD, &smbios)) {
102 pboard = smbios.tblhdr;
103
104 platform_add(&smbios, "board-vendor", pboard->vendor);
105 platform_add(&smbios, "board-product", pboard->product);
106 platform_add(&smbios, "board-version", pboard->version);
107 platform_add(&smbios, "board-serial", pboard->serial);
108 platform_add(&smbios, "board-asset-tag", pboard->asset);
109 }
110
111 smbios.cookie = 0;
112 if (smbios_find_table(SMBIOS_TYPE_ENCLOSURE, &smbios)) {
113 pchassis = smbios.tblhdr;
114
115 platform_add(&smbios, "chassis-vendor", pchassis->vendor);
116 platform_add(&smbios, "chassis-type", pchassis->shape);
117 platform_add(&smbios, "chassis-version", pchassis->version);
118 platform_add(&smbios, "chassis-serial", pchassis->serial);
119 platform_add(&smbios, "chassis-asset-tag", pchassis->asset);
120 }
121
122 smbios.cookie = 0;
123 if (smbios_find_table(SMBIOS_TYPE_PROCESSOR, &smbios)) {
124 pproc = smbios.tblhdr;
125
126 platform_add(&smbios, "processor-vendor", pproc->vendor);
127 platform_add(&smbios, "processor-version", pproc->version);
128 platform_add_word(&smbios, "processor-frequency",
129 pproc->curspeed, " MHz");
130 }
131
132 smbios.cookie = 0;
133 nisa = 0;
134 nother = 0;
135 while (smbios_find_table(SMBIOS_TYPE_SLOTS, &smbios)) {
136 pslot = smbios.tblhdr;
137 switch (pslot->type) {
138 case SMBIOS_SLOT_ISA:
139 case SMBIOS_SLOT_EISA:
140 nisa++;
141 break;
142 default:
143 nother++;
144 break;
145 }
146 }
147
148#if NISA > 0
149 if ((nother | nisa) != 0) {
150 /* Only if there seems to be good expansion slot info. */
151 isa_set_slotcount(nisa);
152 }
153#endif
154
155 platform_print();
156}

158static void
159platform_print(void)
160{
161 const char *vend, *prod, *ver;
162
163 vend = pmf_get_platform("system-vendor");
164 prod = pmf_get_platform("system-product");
165 ver = pmf_get_platform("system-version");
166
167 if (vend == NULL)
168 aprint_verbose("Generic");
169 else
170 aprint_verbose("%s", vend);
171 if (prod == NULL)
172 aprint_verbose(" PC");
173 else
174 aprint_verbose(" %s", prod);
175 if (ver != NULL)
176 aprint_verbose(" (%s)", ver);
177 aprint_verbose("\n");
178}

180static bool
181platform_sysctl_is_private(const char *key)
182{
183 unsigned int n;
184
185 for (n = 0; platform_private_nodes[n] != NULL; n++) {
186 if (strcmp(key, platform_private_nodes[n]) == 0) {
187 return true;
188 }
189 }
190
191 return false;
192}

194static void
195platform_create_sysctl(const char *key)
196{
197 int flags = 0, err;
198
199 if (pmf_get_platform(key) == NULL)
200 return;
201
202 /* If the key is marked private, set CTLFLAG_PRIVATE flag */
203 if (platform_sysctl_is_private(key))
204 flags |= CTLFLAG_PRIVATE;
205
206 err = sysctl_createv(NULL, 0, NULL, NULL,
207 CTLFLAG_READONLY | flags, CTLTYPE_STRING,
208 key, NULL, platform_dmi_sysctl, 0, NULL, 0,
209 CTL_MACHDEP, platform_dminode, CTL_CREATE, CTL_EOL);
210 if (err != 0)
211 printf("platform: sysctl_createv "
212 "(machdep.dmi.%s) failed, err = %d\n",
213 key, err);
214}

216static void
217platform_add(struct smbtable *tbl, const char *key, int idx)
218{
219 char tmpbuf[128]; /* XXX is this long enough? */
220
221 if (smbios_get_string(tbl, idx, tmpbuf, 128) != NULL) {
222 /* add to platform dictionary */
223 pmf_set_platform(key, tmpbuf);
224
225 /* create sysctl node */
226 platform_create_sysctl(key);
227 }
228}

230static void
231platform_add_word(struct smbtable *tbl, const char *key, uint16_t val,
232 const char *suf)
233{
234 char tmpbuf[128]; /* XXX is this long enough? */
235
236 if (snprintf(tmpbuf, sizeof(tmpbuf), "%u%s", val, suf)) {
237 /* add to platform dictionary */
238 pmf_set_platform(key, tmpbuf);
239
240 /* create sysctl node */
241 platform_create_sysctl(key);
242 }
243}

245static int
246platform_scan_date(char *buf, unsigned int *month, unsigned int *day,
247 unsigned int *year)
248{
249 char *p, *s;
250
251 s = buf;
252 p = strchr(s, '/');
253 if (p) *p = '\0';
254 *month = strtoul(s, NULL, 10);
255 if (!p) return 1;
256
257 s = p + 1;
258 p = strchr(s, '/');
259 if (p) *p = '\0';
260 *day = strtoul(s, NULL, 10);
261 if (!p) return 2;
262
263 s = p + 1;
264 *year = strtoul(s, NULL, 10);
265 return 3;
266}

268static void
269platform_add_date(struct smbtable *tbl, const char *key, int idx)
270{
271 unsigned int month, day, year;
272 char tmpbuf[128], datestr[9];
273
274 if (smbios_get_string(tbl, idx, tmpbuf, 128) == NULL)
275 return;
276 if (platform_scan_date(tmpbuf, &month, &day, &year) != 3)
277 return;
278 if (month == 0 || month > 12 || day == 0 || day > 31)
279 return;
280 if (year > 9999)
281 return;
282 if (year < 70)
283 year += 2000;
284 else if (year < 100)
285 year += 1900;
286 snprintf(datestr, sizeof(datestr), "%04u%02u%02u", year, month, day);
287 pmf_set_platform(key, datestr);
288 platform_create_sysctl(key);
289}
290
291static void
292platform_add_uuid(struct smbtable *tbl, const char *key, const uint8_t *buf)
293{
294 struct uuid uuid;
295 char tmpbuf[UUID_STR_LEN];
296
297 uuid_dec_le(buf, &uuid);
298 uuid_snprintf(tmpbuf, sizeof(tmpbuf), &uuid);
299
300 pmf_set_platform(key, tmpbuf);
301 platform_create_sysctl(key);
302}

304static int
305platform_dmi_sysctl(SYSCTLFN_ARGS)
306{
307 struct sysctlnode node;
308 const char *v;
309 int err = 0;
310
311 node = *rnode;
312
313 v = pmf_get_platform(node.sysctl_name);
314 if (v == NULL)
315 return ENOENT;
316
317 node.sysctl_data = __UNCONST(v);
318 err = sysctl_lookup(SYSCTLFN_CALL(&node));
319 if (err || newp == NULL)
320 return err;
321
322 return 0;
323}
324
325SYSCTL_SETUP(sysctl_dmi_setup, "sysctl machdep.dmi subtree setup")
326{
327 const struct sysctlnode *rnode;
328 int err;
329
330 err = sysctl_createv(clog, 0, NULL, &rnode,
331 CTLFLAG_PERMANENT, CTLTYPE_NODE, "machdep",
332 NULL, NULL, 0, NULL, 0,
333 CTL_MACHDEP, CTL_EOL);
334 if (err)
335 return;
336
337 err = sysctl_createv(clog, 0, &rnode, &rnode,
338 CTLFLAG_PERMANENT, CTLTYPE_NODE, "dmi",
339 SYSCTL_DESCR("DMI table information"),
340 NULL, 0, NULL, 0,
341 CTL_CREATE, CTL_EOL);
342 if (err)
343 return;
344
345 platform_dminode = rnode->sysctl_num;
346}
