199/*
200 * Linux-style /proc/cpuinfo.
201 * Only used when procfs is mounted with -o linux.
202 *
203 * In the multiprocessor case, this should be a loop over all CPUs.
204 */
205int
206procfs_getcpuinfstr(char *bf, size_t *len)
207{
208 struct cpu_info *ci;
209 CPU_INFO_ITERATOR cii;
210 size_t i, total, size, used;
211
212 i = total = 0;
213 used = size = *len;
214
215 for (CPU_INFO_FOREACH(cii, ci)) {
216 procfs_getonecpu(i++, ci, bf, &used);
217 total += used + 1;
218 if (used + 1 <= size) {
219 bf += used;
220 *bf++ = '\n';
221 size -= used + 1;
222 used = size;
223 } else
224 used = 0;
225 }
226 size = *len;
227 *len = total;
228 return size < *len ? -1 : 0;
229}

231static int
232procfs_getonefeatreg(uint32_t reg, const char * const *table, char *p,
233 size_t *left)
234{
235 size_t l;
236
237 for (size_t i = 0; i < 32; i++) {
238 if ((reg & (1U << i)) && table[i]) {
239 l = snprintf(p, *left, "%s ", table[i]);
240 if (l < *left) {
241 *left -= l;
242 p += l;
243 } else
244 break;
245 }
246 }
247
248 return 0; /* XXX */
249}

251/*
252 * Print feature bits. The code assume that unused entry of x86_features[]
253 * is zero-cleared.
254 *
255 * XXX This function will be rewritten when all of linux entries are
256 * decoded.
257 */
258static int
259procfs_getonecpufeatures(struct cpu_info *ci, char *p, size_t *left)
260{
261 size_t last = *left;
262 size_t diff;
263 u_int descs[4];
264
265 procfs_getonefeatreg(ci->ci_feat_val[0], x86_features[0], p, left);
266 diff = last - *left;
267
268 procfs_getonefeatreg(ci->ci_feat_val[2], x86_features[1], p + diff,
269 left);
270 diff = last - *left;
271
272 /* x86_features[2] is for Transmeta */
273 /* x86_features[3] is Linux defined mapping */
274
275 procfs_getonefeatreg(ci->ci_feat_val[1], x86_features[4], p + diff,
276 left);
277 diff = last - *left;
278
279 procfs_getonefeatreg(ci->ci_feat_val[4], x86_features[5], p + diff,
280 left);
281 diff = last - *left;
282
283 procfs_getonefeatreg(ci->ci_feat_val[3], x86_features[6], p + diff,
284 left);
285 diff = last - *left;
286
287 /* x86_features[7] is Linux defined mapping */
288 /* x86_features[8] is Linux defined mapping */
289
290 procfs_getonefeatreg(ci->ci_feat_val[5], x86_features[9], p + diff,
291 left);
292 diff = last - *left;
293
294 if (ci->ci_max_cpuid >= 0x0d) {
295 x86_cpuid2(0x0d, 1, descs);
296 procfs_getonefeatreg(descs[0], x86_features[10], p + diff,
297 left);
298 diff = last - *left;
299 }
300
301 if (ci->ci_max_cpuid >= 0x0f) {
302 x86_cpuid2(0x0f, 0, descs);
303 procfs_getonefeatreg(descs[3], x86_features[11], p + diff,
304 left);
305 diff = last - *left;
306
307 x86_cpuid2(0x0f, 1, descs);
308 procfs_getonefeatreg(descs[3], x86_features[12], p + diff,
309 left);
310 diff = last - *left;
311 }
312
313 if ((cpu_vendor == CPUVENDOR_AMD)
314 && (ci->ci_max_ext_cpuid >= 0x80000008)) {
315 x86_cpuid(0x80000008, descs);
316 procfs_getonefeatreg(descs[1], x86_features[13], p + diff,
317 left);
318 diff = last - *left;
319 }
320
321 if (ci->ci_max_cpuid >= 0x06) {
322 x86_cpuid(0x06, descs);
323 procfs_getonefeatreg(descs[0], x86_features[14], p + diff,
324 left);
325 diff = last - *left;
326 }
327
328 if ((cpu_vendor == CPUVENDOR_AMD)
329 && (ci->ci_max_ext_cpuid >= 0x8000000a)) {
330 x86_cpuid(0x8000000a, descs);
331 procfs_getonefeatreg(descs[3], x86_features[15], p + diff,
332 left);
333 diff = last - *left;
334 }
335
336 procfs_getonefeatreg(ci->ci_feat_val[6], x86_features[16], p + diff,
337 left);
338 diff = last - *left;
339
340 if ((cpu_vendor == CPUVENDOR_AMD)
341 && (ci->ci_max_ext_cpuid >= 0x80000007)) {
342 x86_cpuid(0x80000007, descs);
343 procfs_getonefeatreg(descs[1], x86_features[17], p + diff,
344 left);
345 diff = last - *left;
346 }
347
348 if ((cpu_vendor == CPUVENDOR_INTEL)
349 && (ci->ci_max_cpuid >= 0x00000007)) {
350 x86_cpuid(0x00000007, descs);
351 procfs_getonefeatreg(descs[3], x86_features[18], p + diff,
352 left);
353 diff = last - *left;
354 }
355
356 return 0; /* XXX */
357}

359static int
360procfs_getonecpu(int xcpu, struct cpu_info *ci, char *bf, size_t *len)
361{
362 size_t left, l, size;
363 char featurebuf[1024], *p;
364
365 p = featurebuf;
366 left = sizeof(featurebuf);
367 size = *len;
368 procfs_getonecpufeatures(ci, p, &left);
369
370 p = bf;
371 left = *len;
372 size = 0;
373 l = snprintf(p, left,
374 "processor\t: %d\n"
375 "vendor_id\t: %s\n"
376 "cpu family\t: %d\n"
377 "model\t\t: %d\n"
378 "model name\t: %s\n"
379 "stepping\t: ",
380 xcpu,
381 (char *)ci->ci_vendor,
382 CPUID_TO_FAMILY(ci->ci_signature),
383 CPUID_TO_MODEL(ci->ci_signature),
384 cpu_brand_string
385 );
386 size += l;
387 if (l < left) {
388 left -= l;
389 p += l;
390 } else
391 left = 0;
392
393 if (cpuid_level >= 0)
394 l = snprintf(p, left, "%d\n",
395 CPUID_TO_STEPPING(ci->ci_signature));
396 else
397 l = snprintf(p, left, "unknown\n");
398
399 size += l;
400 if (l < left) {
401 left -= l;
402 p += l;
403 } else
404 left = 0;
405
406 if (ci->ci_data.cpu_cc_freq != 0) {
407 uint64_t freq, fraq;
408
409 freq = (ci->ci_data.cpu_cc_freq + 4999) / 1000000;
410 fraq = ((ci->ci_data.cpu_cc_freq + 4999) / 10000) % 100;
411 l = snprintf(p, left, "cpu MHz\t\t: %" PRIu64 ".%02" PRIu64
412 "\n", freq, fraq);
413 } else
414 l = snprintf(p, left, "cpu MHz\t\t: unknown\n");
415
416 size += l;
417 if (l < left) {
418 left -= l;
419 p += l;
420 } else
421 left = 0;
422
423 l = snprintf(p, left,
424 "apicid\t\t: %lu\n"
425 "initial apicid\t: %u\n",
426 ci->ci_cpuid,
427 ci->ci_initapicid
428 );
429 size += l;
430 if (l < left) {
431 left -= l;
432 p += l;
433 } else
434 left = 0;
435
436 l = snprintf(p, left,
437#ifdef __i386__
438 "fdiv_bug\t: %s\n"
439#endif
440 "fpu\t\t: yes\n"
441 "fpu_exception\t: yes\n"
442 "cpuid level\t: %d\n"
443 "wp\t\t: %s\n"
444 "flags\t\t: %s\n"
445 "clflush size\t: %d\n",
446#ifdef __i386__
447 i386_fpu_fdivbug ? "yes" : "no", /* an old pentium */
448#endif
449 ci->ci_max_cpuid,
450 (rcr0() & CR0_WP) ? "yes" : "no",
451 featurebuf,
452 ci->ci_cflush_lsize
453 );
454 size += l;
455
456 left = *len;
457 *len = size;
458 return left < *len ? -1 : 0;
459}

463void
464procfs_machdep_allocvp(struct vnode *vp)
465{
466 struct pfsnode *pfs = vp->v_data;
467
468 switch (pfs->pfs_type) {
469 case Pmachdep_xmmregs:
470 /* /proc/N/xmmregs = -rw------- */
471 pfs->pfs_mode = S_IRUSR|S_IWUSR;
472 vp->v_type = VREG;
473 break;
474 default:
475 KASSERT(false);
476 }
477}
478
479int
480procfs_machdep_rw(struct lwp *curl, struct lwp *l, struct pfsnode *pfs,
481 struct uio *uio)
482{
483
484 switch (pfs->pfs_type) {
485 case Pmachdep_xmmregs:
486 return (procfs_machdep_doxmmregs(curl, l, pfs, uio));
487 default:
488 KASSERT(false);
489 }
490 return EINVAL;
491}
492
493int
494procfs_machdep_getattr(struct vnode *vp, struct vattr *vap, struct proc *procp)
495{
496 struct pfsnode *pfs = VTOPFS(vp);
497
498 switch (pfs->pfs_type) {
499 case Pmachdep_xmmregs:
500 vap->va_bytes = vap->va_size = sizeof(struct xmmregs);
501 break;
502 default:
503 KASSERT(false);
504 }
505 return 0;
506}
