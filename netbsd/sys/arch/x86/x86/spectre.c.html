82static void
83v2_set_name(void)
84{
85 char name[64] = "";
86 size_t nmitig = 0;
87
88#if defined(SPECTRE_V2_GCC_MITIGATION)
89 strlcat(name, "[GCC retpoline]", sizeof(name));
90 nmitig++;
91#endif
92
93 if (!v2_mitigation_enabled) {
94 if (nmitig == 0)
95 strlcat(name, "(none)", sizeof(name));
96 } else {
97 if (nmitig)
98 strlcat(name, " + ", sizeof(name));
99 switch (v2_mitigation_method) {
100 case V2_MITIGATION_AMD_DIS_IND:
101 strlcat(name, "[AMD DIS_IND]", sizeof(name));
102 break;
103 case V2_MITIGATION_INTEL_IBRS:
104 strlcat(name, "[Intel IBRS]", sizeof(name));
105 break;
106 case V2_MITIGATION_INTEL_ENHANCED_IBRS:
107 strlcat(name, "[Intel Enhanced IBRS]", sizeof(name));
108 break;
109 default:
110 panic("%s: impossible", __func__);
111 }
112 }
113
114 strlcpy(v2_mitigation_name, name,
115 sizeof(v2_mitigation_name));
116}

118static void
119v2_detect_method(void)
120{
121 struct cpu_info *ci = curcpu();
122 u_int descs[4];
123 uint64_t msr;
124
125 if (cpu_vendor == CPUVENDOR_INTEL) {
126 if (cpuid_level >= 7) {
127 x86_cpuid(7, descs);
128
129 if (descs[3] & CPUID_SEF_ARCH_CAP) {
130 msr = rdmsr(MSR_IA32_ARCH_CAPABILITIES);
131 if (msr & IA32_ARCH_IBRS_ALL) {
132 v2_mitigation_method =
133 V2_MITIGATION_INTEL_ENHANCED_IBRS;
134 return;
135 }
136 }
137#ifdef __x86_64__
138 if (descs[3] & CPUID_SEF_IBRS) {
139 v2_mitigation_method = V2_MITIGATION_INTEL_IBRS;
140 return;
141 }
142#endif
143 }
144 v2_mitigation_method = V2_MITIGATION_NONE;
145 } else if (cpu_vendor == CPUVENDOR_AMD) {
146 /*
147 * The AMD Family 10h manual documents the IC_CFG.DIS_IND bit.
148 * This bit disables the Indirect Branch Predictor.
149 *
150 * Families 12h and 16h are believed to have this bit too, but
151 * their manuals don't document it.
152 */
153 switch (CPUID_TO_FAMILY(ci->ci_signature)) {
154 case 0x10:
155 v2_mitigation_method = V2_MITIGATION_AMD_DIS_IND;
156 break;
157 default:
158 v2_mitigation_method = V2_MITIGATION_NONE;
159 break;
160 }
161 } else {
162 v2_mitigation_method = V2_MITIGATION_NONE;
163 }
164}

171#ifdef __x86_64__
172static void
173ibrs_disable_hotpatch(void)
174{
175 extern uint8_t noibrs_enter, noibrs_enter_end;
176 extern uint8_t noibrs_leave, noibrs_leave_end;
177 u_long psl, cr0;
178 uint8_t *bytes;
179 size_t size;
180
181 x86_patch_window_open(&psl, &cr0);
182
183 bytes = &noibrs_enter;
184 size = (size_t)&noibrs_enter_end - (size_t)&noibrs_enter;
185 x86_hotpatch(HP_NAME_IBRS_ENTER, bytes, size);
186
187 bytes = &noibrs_leave;
188 size = (size_t)&noibrs_leave_end - (size_t)&noibrs_leave;
189 x86_hotpatch(HP_NAME_IBRS_LEAVE, bytes, size);
190
191 x86_patch_window_close(psl, cr0);
192}

231static void
232mitigation_v2_apply_cpu(struct cpu_info *ci, bool enabled)
233{
234 uint64_t msr;
235
236 switch (v2_mitigation_method) {
237 case V2_MITIGATION_NONE:
238 panic("impossible");
239 case V2_MITIGATION_INTEL_IBRS:
240 /* cpu0 is the one that does the hotpatch job */
241 if (ci == &cpu_info_primary) {
242 if (enabled) {
243 ibrs_enable_hotpatch();
244 } else {
245 ibrs_disable_hotpatch();
246 }
247 }
248 if (!enabled) {
249 wrmsr(MSR_IA32_SPEC_CTRL, 0);
250 }
251 break;
252 case V2_MITIGATION_INTEL_ENHANCED_IBRS:
253 msr = rdmsr(MSR_IA32_SPEC_CTRL);
254 if (enabled) {
255 msr |= IA32_SPEC_CTRL_IBRS;
256 } else {
257 msr &= ~IA32_SPEC_CTRL_IBRS;
258 }
259 wrmsr(MSR_IA32_SPEC_CTRL, msr);
260 break;
261 case V2_MITIGATION_AMD_DIS_IND:
262 msr = rdmsr(MSR_IC_CFG);
263 if (enabled) {
264 msr |= IC_CFG_DIS_IND;
265 } else {
266 msr &= ~IC_CFG_DIS_IND;
267 }
268 wrmsr(MSR_IC_CFG, msr);
269 break;
270 }
271}

273/*
274 * Note: IBRS requires hotpatching, so we need barriers.
275 */
276static void
277mitigation_v2_change_cpu(void *arg1, void *arg2)
278{
279 struct cpu_info *ci = curcpu();
280 bool enabled = (bool)arg1;
281 u_long psl = 0;
282
283 /* Rendez-vous 1 (IBRS only). */
284 if (v2_mitigation_method == V2_MITIGATION_INTEL_IBRS) {
285 psl = x86_read_psl();
286 x86_disable_intr();
287
288 atomic_dec_ulong(&ibrs_cpu_barrier1);
289 while (atomic_cas_ulong(&ibrs_cpu_barrier1, 0, 0) != 0) {
290 x86_pause();
291 }
292 }
293
294 mitigation_v2_apply_cpu(ci, enabled);
295
296 /* Rendez-vous 2 (IBRS only). */
297 if (v2_mitigation_method == V2_MITIGATION_INTEL_IBRS) {
298 atomic_dec_ulong(&ibrs_cpu_barrier2);
299 while (atomic_cas_ulong(&ibrs_cpu_barrier2, 0, 0) != 0) {
300 x86_pause();
301 }
302
303 /* Write back and invalidate cache, flush pipelines. */
304 wbinvd();
305 x86_flush();
306
307 x86_write_psl(psl);
308 }
309}

311static int
312mitigation_v2_change(bool enabled)
313{
314 uint64_t xc;
315
316 v2_detect_method();
317
318 switch (v2_mitigation_method) {
319 case V2_MITIGATION_NONE:
320 printf("[!] No mitigation available\n");
321 return EOPNOTSUPP;
322 case V2_MITIGATION_AMD_DIS_IND:
323 case V2_MITIGATION_INTEL_IBRS:
324 case V2_MITIGATION_INTEL_ENHANCED_IBRS:
325 /* Initialize the barriers */
326 ibrs_cpu_barrier1 = ncpu;
327 ibrs_cpu_barrier2 = ncpu;
328
329 printf("[+] %s SpectreV2 Mitigation...",
330 enabled ? "Enabling" : "Disabling");
331 xc = xc_broadcast(XC_HIGHPRI, mitigation_v2_change_cpu,
332 (void *)enabled, NULL);
333 xc_wait(xc);
334 printf(" done!\n");
335 v2_mitigation_enabled = enabled;
336 v2_set_name();
337 return 0;
338 default:
339 panic("impossible");
340 }
341}

398static void
399v4_detect_method(void)
400{
401 struct cpu_info *ci = curcpu();
402 u_int descs[4];
403 uint64_t msr;
404
405 if (cpu_vendor == CPUVENDOR_INTEL) {
406 if (cpu_info_primary.ci_feat_val[7] & CPUID_SEF_ARCH_CAP) {
407 msr = rdmsr(MSR_IA32_ARCH_CAPABILITIES);
408 if (msr & IA32_ARCH_SSB_NO) {
409 /* Not vulnerable to SpectreV4. */
410 v4_mitigation_method = V4_MITIGATION_INTEL_SSB_NO;
411 return;
412 }
413 }
414 if (cpuid_level >= 7) {
415 x86_cpuid(7, descs);
416 if (descs[3] & CPUID_SEF_SSBD) {
417 /* descs[3] = %edx */
418 v4_mitigation_method = V4_MITIGATION_INTEL_SSBD;
419 return;
420 }
421 }
422 } else if (cpu_vendor == CPUVENDOR_AMD) {
423 switch (CPUID_TO_FAMILY(ci->ci_signature)) {
424 case 0x15:
425 v4_mitigation_method = V4_MITIGATION_AMD_NONARCH_F15H;
426 return;
427 case 0x16:
428 v4_mitigation_method = V4_MITIGATION_AMD_NONARCH_F16H;
429 return;
430 case 0x17:
431 v4_mitigation_method = V4_MITIGATION_AMD_NONARCH_F17H;
432 return;
433 default:
434 if (cpu_info_primary.ci_max_ext_cpuid < 0x80000008) {
435 break;
436 }
437 x86_cpuid(0x80000008, descs);
438 if (descs[1] & __BIT(26)) {
439 /* Not vulnerable to SpectreV4. */
440 v4_mitigation_method = V4_MITIGATION_AMD_SSB_NO;
441 return;
442 }
443
444 break;
445 }
446 }
447
448 v4_mitigation_method = V4_MITIGATION_NONE;
449}

451static void
452mitigation_v4_apply_cpu(bool enabled)
453{
454 uint64_t msr, msrval = 0, msrbit = 0;
455
456 switch (v4_mitigation_method) {
457 case V4_MITIGATION_NONE:
458 case V4_MITIGATION_INTEL_SSB_NO:
459 case V4_MITIGATION_AMD_SSB_NO:
460 panic("impossible");
461 case V4_MITIGATION_INTEL_SSBD:
462 msrval = MSR_IA32_SPEC_CTRL;
463 msrbit = IA32_SPEC_CTRL_SSBD;
464 break;
465 case V4_MITIGATION_AMD_NONARCH_F15H:
466 msrval = MSR_LS_CFG;
467 msrbit = LS_CFG_DIS_SSB_F15H;
468 break;
469 case V4_MITIGATION_AMD_NONARCH_F16H:
470 msrval = MSR_LS_CFG;
471 msrbit = LS_CFG_DIS_SSB_F16H;
472 break;
473 case V4_MITIGATION_AMD_NONARCH_F17H:
474 msrval = MSR_LS_CFG;
475 msrbit = LS_CFG_DIS_SSB_F17H;
476 break;
477 }
478
479 msr = rdmsr(msrval);
480 if (enabled) {
481 msr |= msrbit;
482 } else {
483 msr &= ~msrbit;
484 }
485 wrmsr(msrval, msr);
486}

488static void
489mitigation_v4_change_cpu(void *arg1, void *arg2)
490{
491 bool enabled = (bool)arg1;
492
493 mitigation_v4_apply_cpu(enabled);
494}

496static int
497mitigation_v4_change(bool enabled)
498{
499 uint64_t xc;
500
501 v4_detect_method();
502
503 switch (v4_mitigation_method) {
504 case V4_MITIGATION_NONE:
505 printf("[!] No mitigation available\n");
506 return EOPNOTSUPP;
507 case V4_MITIGATION_INTEL_SSBD:
508 case V4_MITIGATION_AMD_NONARCH_F15H:
509 case V4_MITIGATION_AMD_NONARCH_F16H:
510 case V4_MITIGATION_AMD_NONARCH_F17H:
511 printf("[+] %s SpectreV4 Mitigation...",
512 enabled ? "Enabling" : "Disabling");
513 xc = xc_broadcast(0, mitigation_v4_change_cpu,
514 (void *)enabled, NULL);
515 xc_wait(xc);
516 printf(" done!\n");
517 v4_mitigation_enabled = enabled;
518 v4_set_name();
519 return 0;
520 case V4_MITIGATION_INTEL_SSB_NO:
521 case V4_MITIGATION_AMD_SSB_NO:
522 printf("[+] The CPU is not affected by SpectreV4\n");
523 return 0;
524 default:
525 panic("impossible");
526 }
527}

665static void
666mds_detect_method(void)
667{
668 u_int descs[4];
669 uint64_t msr;
670
671 if (cpu_vendor != CPUVENDOR_INTEL) {
672 mds_mitigation_method = MDS_MITIGATION_MDS_NO;
673 return;
674 }
675
676 if (cpuid_level < 7) {
677 return;
678 }
679
680 x86_cpuid(0x7, descs);
681 if (descs[3] & CPUID_SEF_ARCH_CAP) {
682 msr = rdmsr(MSR_IA32_ARCH_CAPABILITIES);
683 if (msr & IA32_ARCH_MDS_NO) {
684 mds_mitigation_method = MDS_MITIGATION_MDS_NO;
685 return;
686 }
687 }
688
689#ifdef __x86_64__
690 if (descs[3] & CPUID_SEF_MD_CLEAR) {
691 mds_mitigation_method = MDS_MITIGATION_VERW;
692 }
693#endif
694}

741 printf(" done!\n");
742 mds_mitigation_enabled = enabled;
743 mds_set_name();
744 return 0;
745 case MDS_MITIGATION_MDS_NO:
746 printf("[+] The CPU is not affected by MDS\n");
747 return 0;
748 default:
749 panic("impossible");
750 }
751}

753static int
754sysctl_machdep_mds_mitigated(SYSCTLFN_ARGS)
755{
756 struct sysctlnode node;
757 int error;
758 bool val;
759
760 val = *(bool *)rnode->sysctl_data;
761
762 node = *rnode;
763 node.sysctl_data = &val;
764
765 error = sysctl_lookup(SYSCTLFN_CALL(&node));
766 if (error != 0 || newp == NULL)
767 return error;
768
769 if (val == mds_mitigation_enabled)
770 return 0;
771 return mitigation_mds_change(val);
772}

778void
779speculation_barrier(struct lwp *oldlwp, struct lwp *newlwp)
780{
781 /*
782 * Speculation barriers are applicable only to Spectre V2.
783 */
784 if (!v2_mitigation_enabled)
785 return;
786
787 /*
788 * From kernel thread to kernel thread, no need for a barrier.
789 */
790 if ((oldlwp != NULL && (oldlwp->l_flag & LW_SYSTEM)) &&
791 (newlwp->l_flag & LW_SYSTEM))
792 return;
793
794 switch (v2_mitigation_method) {
795 case V2_MITIGATION_INTEL_IBRS:
796 wrmsr(MSR_IA32_PRED_CMD, IA32_PRED_CMD_IBPB);
797 break;
798 default:
799 /* nothing */
800 break;
801 }
802}
