53/*
54 * Separate Virtual Space
55 *
56 * A per-cpu L4 page is maintained in ci_svs_updirpa. During each context
57 * switch to a user pmap, the lower half of updirpa is populated with the
58 * entries containing the userland pages.
59 *
60 * ~~~~~~~~~~ The UTLS Page ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
61 *
62 * We use a special per-cpu page that we call UTLS, for User Thread Local
63 * Storage. Each CPU has one UTLS page. This page has two VAs:
64 *
65 * o When the user page tables are loaded in CR3, the VA to access this
66 * page is &pcpuarea->utls, defined as SVS_UTLS in assembly. This VA is
67 * _constant_ across CPUs, but in the user page tables this VA points to
68 * the physical page of the UTLS that is _local_ to the CPU.
69 *
70 * o When the kernel page tables are loaded in CR3, the VA to access this
71 * page is ci->ci_svs_utls.
72 *
73 * +----------------------------------------------------------------------+
74 * | CPU0 Local Data (Physical Page) |
75 * | +------------------+ +-------------+ |
76 * | | User Page Tables | SVS_UTLS ---------------------> | cpu0's UTLS | |
77 * | +------------------+ +-------------+ |
78 * +-------------------------------------------------------------^--------+
79 * |
80 * +----------+
81 * |
82 * +----------------------------------------------------------------------+ |
83 * | CPU1 Local Data (Physical Page) | |
84 * | +------------------+ +-------------+ | |
85 * | | User Page Tables | SVS_UTLS ---------------------> | cpu1's UTLS | | |
86 * | +------------------+ +-------------+ | |
87 * +-------------------------------------------------------------^--------+ |
88 * | |
89 * +------------------+ /----------------------+ |
90 * | Kern Page Tables | ci->ci_svs_utls |
91 * +------------------+ \---------------------------------+
92 *
93 * The goal of the UTLS page is to provide an area where we can store whatever
94 * we want, in a way that it is accessible both when the Kernel and when the
95 * User page tables are loaded in CR3.
96 *
97 * We store in the UTLS page three 64bit values:
98 *
99 * o UTLS_KPDIRPA: the value we must put in CR3 in order to load the kernel
100 * page tables.
101 *
102 * o UTLS_SCRATCH: a dummy place where we temporarily store a value during
103 * the syscall entry procedure.
104 *
105 * o UTLS_RSP0: the value we must put in RSP in order to have a stack where
106 * we can push the register states. This is used only during the syscall
107 * entry procedure, because there the CPU does not automatically switch
108 * RSP (it does not use the TSS.rsp0 mechanism described below).
109 *
110 * ~~~~~~~~~~ The Stack Switching Mechanism Without SVS ~~~~~~~~~~~~~~~~~~~~~~
111 *
112 * The kernel stack is per-lwp (pcb_rsp0). When doing a context switch between
113 * two user LWPs, the kernel updates TSS.rsp0 (which is per-cpu) to point to
114 * the stack of the new LWP. Then the execution continues. At some point, the
115 * user LWP we context-switched to will perform a syscall or will receive an
116 * interrupt. There, the CPU will automatically read TSS.rsp0 and use it as a
117 * stack. The kernel then pushes the register states on this stack, and
118 * executes in kernel mode normally.
119 *
120 * TSS.rsp0 is used by the CPU only during ring3->ring0 transitions. Therefore,
121 * when an interrupt is received while we were in kernel mode, the CPU does not
122 * read TSS.rsp0. Instead, it just uses the current stack.
123 *
124 * ~~~~~~~~~~ The Stack Switching Mechanism With SVS ~~~~~~~~~~~~~~~~~~~~~~~~~
125 *
126 * In the pcpu_area structure, pointed to by the "pcpuarea" variable, each CPU
127 * has a two-page rsp0 entry (pcpuarea->ent[cid].rsp0). These two pages do
128 * _not_ have associated physical addresses. They are only two VAs.
129 *
130 * The first page is unmapped and acts as a redzone. The second page is
131 * dynamically kentered into the highest page of the real per-lwp kernel stack;
132 * but pay close attention, it is kentered _only_ in the user page tables.
133 * That is to say, the VA of this second page is mapped when the user page
134 * tables are loaded, but not mapped when the kernel page tables are loaded.
135 *
136 * During a context switch, svs_lwp_switch() gets called first. This function
137 * does the kenter job described above, not in the kernel page tables (that
138 * are currently loaded), but in the user page tables (that are not loaded).
139 *
140 * VIRTUAL ADDRESSES PHYSICAL ADDRESSES
141 *
142 * +-----------------------------+
143 * | KERNEL PAGE TABLES |
144 * | +-------------------+ | +-------------------+
145 * | | pcb_rsp0 (page 0) | ------------------> | pcb_rsp0 (page 0) |
146 * | +-------------------+ | +-------------------+
147 * | | pcb_rsp0 (page 1) | ------------------> | pcb_rsp0 (page 1) |
148 * | +-------------------+ | +-------------------+
149 * | | pcb_rsp0 (page 2) | ------------------> | pcb_rsp0 (page 2) |
150 * | +-------------------+ | +-------------------+
151 * | | pcb_rsp0 (page 3) | ------------------> | pcb_rsp0 (page 3) |
152 * | +-------------------+ | +-> +-------------------+
153 * +-----------------------------+ |
154 * |
155 * +---------------------------------------+ |
156 * | USER PAGE TABLES | |
157 * | +----------------------------------+ | |
158 * | | pcpuarea->ent[cid].rsp0 (page 0) | | |
159 * | +----------------------------------+ | |
160 * | | pcpuarea->ent[cid].rsp0 (page 1) | ----+
161 * | +----------------------------------+ |
162 * +---------------------------------------+
163 *
164 * After svs_lwp_switch() gets called, we set pcpuarea->ent[cid].rsp0 (page 1)
165 * in TSS.rsp0. Later, when returning to userland on the lwp we context-
166 * switched to, we will load the user page tables and execute in userland
167 * normally.
168 *
169 * Next time an interrupt or syscall is received, the CPU will automatically
170 * use TSS.rsp0 as a stack. Here it is executing with the user page tables
171 * loaded, and therefore TSS.rsp0 is _mapped_.
172 *
173 * As part of the kernel entry procedure, we now switch CR3 to load the kernel
174 * page tables. Here, we are still using the stack pointer we set in TSS.rsp0.
175 *
176 * Remember that it was only one page of stack which was mapped only in the
177 * user page tables. We just switched to the kernel page tables, so we must
178 * update RSP to be the real per-lwp kernel stack (pcb_rsp0). And we do so,
179 * without touching the stack (since it is now unmapped, touching it would
180 * fault).
181 *
182 * After we updated RSP, we can continue execution exactly as in the non-SVS
183 * case. We don't need to copy the values the CPU pushed on TSS.rsp0: even if
184 * we updated RSP to a totally different VA, this VA points to the same
185 * physical page as TSS.rsp0. So in the end, the values the CPU pushed are
186 * still here even with the new RSP.
187 *
188 * Thanks to this double-kenter optimization, we don't need to copy the
189 * trapframe during each user<->kernel transition.
190 *
191 * ~~~~~~~~~~ Notes On Locking And Synchronization ~~~~~~~~~~~~~~~~~~~~~~~~~~~
192 *
193 * o Touching ci_svs_updir without holding ci_svs_mtx first is *not*
194 * allowed.
195 *
196 * o pm_kernel_cpus contains the set of CPUs that have the pmap loaded
197 * in their CR3 register. It must *not* be replaced by pm_cpus.
198 *
199 * o When a context switch on the current CPU is made from a user LWP
200 * towards a kernel LWP, CR3 is not updated. Therefore, the pmap's
201 * pm_kernel_cpus still contains the current CPU. It implies that the
202 * remote CPUs that execute other threads of the user process we just
203 * left will keep synchronizing us against their changes.
204 *
205 * ~~~~~~~~~~ List Of Areas That Are Removed From Userland ~~~~~~~~~~~~~~~~~~~
206 *
207 * o PTE Space
208 * o Direct Map
209 * o Remote PCPU Areas
210 * o Kernel Heap
211 * o Kernel Image
212 *
213 * ~~~~~~~~~~ Todo List ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
214 *
215 * Ordered from highest priority to lowest:
216 *
217 * o The NMI stack is not double-entered. Therefore if we ever receive an NMI
218 * and leave it, the content of the stack will be visible to userland (via
219 * Meltdown). Normally we never leave NMIs, unless a privileged user
220 * launched PMCs. That's unlikely to happen, our PMC support is pretty
221 * minimal, and privileged only.
222 *
223 * o Narrow down the entry points: hide the 'jmp handler' instructions. This
224 * makes sense on GENERIC_KASLR kernels.
225 *
226 * o Right now there is only one global LDT, and that's not compatible with
227 * USER_LDT.
228 */

242static pd_entry_t *
243svs_tree_add(struct cpu_info *ci, vaddr_t va)
244{
245 extern const vaddr_t ptp_masks[];
246 extern const int ptp_shifts[];
247 pd_entry_t *dstpde;
248 struct vm_page *pg;
249 size_t i, pidx;
250 paddr_t pa;
251
252 dstpde = ci->ci_svs_updir;
253
254 for (i = PTP_LEVELS; i > 1; i--) {
255 pidx = pl_pi(va, i);
256
257 if (!pmap_valid_entry(dstpde[pidx])) {
258 pg = uvm_pagealloc(NULL, 0, NULL, UVM_PGA_ZERO);
259 if (pg == 0)
260 panic("%s: failed to allocate PA for CPU %d\n",
261 __func__, cpu_index(ci));
262 pa = VM_PAGE_TO_PHYS(pg);
263
264 dstpde[pidx] = PTE_P | PTE_W | pa;
265 }
266
267 pa = (paddr_t)(dstpde[pidx] & PTE_FRAME);
268 dstpde = (pd_entry_t *)PMAP_DIRECT_MAP(pa);
269 }
270
271 return dstpde;
272}

274static void
275svs_page_add(struct cpu_info *ci, vaddr_t va, bool global)
276{
277 pd_entry_t *srcpde, *dstpde, pde;
278 size_t idx, pidx;
279 paddr_t pa;
280
281 /* Create levels L4, L3 and L2. */
282 dstpde = svs_tree_add(ci, va);
283
284 pidx = pl1_pi(va);
285
286 /*
287 * If 'va' is in a large page, we need to compute its physical
288 * address manually.
289 */
290 idx = pl2_i(va);
291 srcpde = L2_BASE;
292 if (!pmap_valid_entry(srcpde[idx])) {
293 panic("%s: L2 page not mapped", __func__);
294 }
295 if (srcpde[idx] & PTE_PS) {
296 KASSERT(!global);
297 pa = srcpde[idx] & PTE_2MFRAME;
298 pa += (paddr_t)(va % NBPD_L2);
299 pde = (srcpde[idx] & ~(PTE_PS|PTE_2MFRAME)) | pa;
300
301 if (pmap_valid_entry(dstpde[pidx])) {
302 panic("%s: L1 page already mapped", __func__);
303 }
304 dstpde[pidx] = pde;
305 return;
306 }
307
308 /*
309 * Normal page, just copy the PDE.
310 */
311 idx = pl1_i(va);
312 srcpde = L1_BASE;
313 if (!pmap_valid_entry(srcpde[idx])) {
314 panic("%s: L1 page not mapped", __func__);
315 }
316 if (pmap_valid_entry(dstpde[pidx])) {
317 panic("%s: L1 page already mapped", __func__);
318 }
319 dstpde[pidx] = srcpde[idx];
320
321 /*
322 * If we want a global translation, mark both the src and dst with
323 * PTE_G.
324 */
325 if (global) {
326 srcpde[idx] |= PTE_G;
327 dstpde[pidx] |= PTE_G;
328 tlbflushg();
329 }
330}

332static void
333svs_rsp0_init(struct cpu_info *ci)
334{
335 const cpuid_t cid = cpu_index(ci);
336 vaddr_t va, rsp0;
337 pd_entry_t *pd;
338 size_t pidx;
339
340 rsp0 = (vaddr_t)&pcpuarea->ent[cid].rsp0;
341
342 /* The first page is a redzone. */
343 va = rsp0 + PAGE_SIZE;
344
345 /* Create levels L4, L3 and L2. */
346 pd = svs_tree_add(ci, va);
347
348 /* Get the info for L1. */
349 pidx = pl1_i(va % NBPD_L2);
350 if (pmap_valid_entry(pd[pidx])) {
351 panic("%s: rsp0 page already mapped", __func__);
352 }
353
354 ci->ci_svs_rsp0_pte = (pt_entry_t *)&pd[pidx];
355 ci->ci_svs_rsp0 = rsp0 + PAGE_SIZE + sizeof(struct trapframe);
356 ci->ci_svs_ursp0 = ci->ci_svs_rsp0 - sizeof(struct trapframe);
357 ci->ci_svs_krsp0 = 0;
358}

360static void
361svs_utls_init(struct cpu_info *ci)
362{
363 const vaddr_t utlsva = (vaddr_t)&pcpuarea->utls;
364 struct svs_utls *utls;
365 struct vm_page *pg;
366 pd_entry_t *pd;
367 size_t pidx;
368 paddr_t pa;
369 vaddr_t va;
370
371 /* Create levels L4, L3 and L2 of the UTLS page. */
372 pd = svs_tree_add(ci, utlsva);
373
374 /* Allocate L1. */
375 pg = uvm_pagealloc(NULL, 0, NULL, UVM_PGA_ZERO);
376 if (pg == 0)
377 panic("%s: failed to allocate PA for CPU %d\n", __func__,
378 cpu_index(ci));
379 pa = VM_PAGE_TO_PHYS(pg);
380
381 /* Enter L1. */
382 if (pmap_valid_entry(L1_BASE[pl1_i(utlsva)])) {
383 panic("%s: local page already mapped", __func__);
384 }
385 pidx = pl1_pi(utlsva);
386 if (pmap_valid_entry(pd[pidx])) {
387 panic("%s: L1 page already mapped", __func__);
388 }
389 pd[pidx] = PTE_P | PTE_W | pmap_pg_nx | pa;
390
391 /*
392 * Now, allocate a VA in the kernel map, that points to the UTLS
393 * page. After that, the UTLS page will be accessible in kernel
394 * mode via ci_svs_utls.
395 */
396 va = uvm_km_alloc(kernel_map, PAGE_SIZE, 0,
397 UVM_KMF_VAONLY|UVM_KMF_NOWAIT);
398 if (va == 0) {
399 panic("%s: unable to allocate VA\n", __func__);
400 }
401 pmap_kenter_pa(va, pa, VM_PROT_READ|VM_PROT_WRITE, 0);
402 pmap_update(pmap_kernel());
403
404 ci->ci_svs_utls = va;
405
406 /* Initialize the constant fields of the UTLS page */
407 utls = (struct svs_utls *)ci->ci_svs_utls;
408 utls->rsp0 = ci->ci_svs_rsp0;
409}

411static void
412svs_pcid_init(struct cpu_info *ci)
413{
414 if (!svs_pcid) {
415 return;
416 }
417
418 svs_pcid_ucr3 = __SHIFTIN(PMAP_PCID_USER, CR3_PCID) | CR3_NO_TLB_FLUSH;
419 svs_pcid_kcr3 = __SHIFTIN(PMAP_PCID_KERN, CR3_PCID) | CR3_NO_TLB_FLUSH;
420
421 ci->ci_svs_updirpa |= svs_pcid_ucr3;
422}

424static void
425svs_range_add(struct cpu_info *ci, vaddr_t va, size_t size, bool global)
426{
427 size_t i, n;
428
429 KASSERT(size % PAGE_SIZE == 0);
430 n = size / PAGE_SIZE;
431 for (i = 0; i < n; i++) {
432 svs_page_add(ci, va + i * PAGE_SIZE, global);
433 }
434}

436void
437cpu_svs_init(struct cpu_info *ci)
438{
439 extern char __text_user_start;
440 extern char __text_user_end;
441 const cpuid_t cid = cpu_index(ci);
442 struct vm_page *pg;
443
444 KASSERT(ci != NULL);
445
446 pg = uvm_pagealloc(NULL, 0, NULL, UVM_PGA_ZERO);
447 if (pg == 0)
448 panic("%s: failed to allocate L4 PA for CPU %d\n",
449 __func__, cpu_index(ci));
450 ci->ci_svs_updirpa = VM_PAGE_TO_PHYS(pg);
451
452 ci->ci_svs_updir = (pt_entry_t *)uvm_km_alloc(kernel_map, PAGE_SIZE, 0,
453 UVM_KMF_VAONLY | UVM_KMF_NOWAIT);
454 if (ci->ci_svs_updir == NULL)
455 panic("%s: failed to allocate L4 VA for CPU %d\n",
456 __func__, cpu_index(ci));
457
458 pmap_kenter_pa((vaddr_t)ci->ci_svs_updir, ci->ci_svs_updirpa,
459 VM_PROT_READ | VM_PROT_WRITE, 0);
460
461 pmap_update(pmap_kernel());
462
463 mutex_init(&ci->ci_svs_mtx, MUTEX_DEFAULT, IPL_VM);
464
465 svs_page_add(ci, (vaddr_t)&pcpuarea->idt, true);
466 svs_page_add(ci, (vaddr_t)&pcpuarea->ldt, true);
467 svs_range_add(ci, (vaddr_t)&pcpuarea->ent[cid],
468 offsetof(struct pcpu_entry, rsp0), true);
469 svs_range_add(ci, (vaddr_t)&__text_user_start,
470 (vaddr_t)&__text_user_end - (vaddr_t)&__text_user_start, false);
471
472 svs_rsp0_init(ci);
473 svs_utls_init(ci);
474
475 svs_pcid_init(ci);
476}

478void
479svs_pmap_sync(struct pmap *pmap, int index)
480{
481 CPU_INFO_ITERATOR cii;
482 struct cpu_info *ci;
483 cpuid_t cid;
484
485 KASSERT(pmap != NULL);
486 KASSERT(pmap != pmap_kernel());
487 KASSERT(mutex_owned(pmap->pm_lock));
488 KASSERT(kpreempt_disabled());
489 KASSERT(index < PDIR_SLOT_USERLIM);
490
491 for (CPU_INFO_FOREACH(cii, ci)) {
492 cid = cpu_index(ci);
493
494 if (!kcpuset_isset(pmap->pm_kernel_cpus, cid)) {
495 continue;
496 }
497
498 /* take the lock and check again */
499 mutex_enter(&ci->ci_svs_mtx);
500 if (kcpuset_isset(pmap->pm_kernel_cpus, cid)) {
501 ci->ci_svs_updir[index] = pmap->pm_pdir[index];
502 }
503 mutex_exit(&ci->ci_svs_mtx);
504 }
505}

507void
508svs_lwp_switch(struct lwp *oldlwp, struct lwp *newlwp)
509{
510 struct cpu_info *ci = curcpu();
511 struct svs_utls *utls;
512 struct pcb *pcb;
513 pt_entry_t *pte;
514 uintptr_t rsp0;
515 vaddr_t va;
516
517 if (newlwp->l_flag & LW_SYSTEM) {
518 return;
519 }
520
521#ifdef DIAGNOSTIC
522 if (oldlwp != NULL && !(oldlwp->l_flag & LW_SYSTEM)) {
523 pcb = lwp_getpcb(oldlwp);
524 rsp0 = pcb->pcb_rsp0;
525 va = rounddown(rsp0, PAGE_SIZE);
526 KASSERT(ci->ci_svs_krsp0 == rsp0 - sizeof(struct trapframe));
527 pte = ci->ci_svs_rsp0_pte;
528 KASSERT(*pte == L1_BASE[pl1_i(va)]);
529 }
530#endif
531
532 pcb = lwp_getpcb(newlwp);
533 rsp0 = pcb->pcb_rsp0;
534 va = rounddown(rsp0, PAGE_SIZE);
535
536 /* Update the kernel rsp0 in cpu_info */
537 ci->ci_svs_krsp0 = rsp0 - sizeof(struct trapframe);
538 KASSERT((ci->ci_svs_krsp0 % PAGE_SIZE) ==
539 (ci->ci_svs_ursp0 % PAGE_SIZE));
540
541 utls = (struct svs_utls *)ci->ci_svs_utls;
542 utls->scratch = 0;
543
544 /*
545 * Enter the user rsp0. If we're using PCID we must flush the user VA,
546 * if we aren't it will be flushed during the next CR3 reload.
547 */
548 pte = ci->ci_svs_rsp0_pte;
549 *pte = L1_BASE[pl1_i(va)];
550 if (svs_pcid) {
551 invpcid(INVPCID_ADDRESS, PMAP_PCID_USER, ci->ci_svs_rsp0);
552 }
553}

555static inline pt_entry_t
556svs_pte_atomic_read(struct pmap *pmap, size_t idx)
557{
558 /*
559 * XXX: We don't have a basic atomic_fetch_64 function?
560 */
561 return atomic_cas_64(&pmap->pm_pdir[idx], 666, 666);
562}
563
564/*
565 * We may come here with the pmap unlocked. So read its PTEs atomically. If
566 * a remote CPU is updating them at the same time, it's not a problem: the
567 * remote CPU will call svs_pmap_sync afterwards, and our updirpa will be
568 * synchronized properly.
569 */
570void
571svs_pdir_switch(struct pmap *pmap)
572{
573 struct cpu_info *ci = curcpu();
574 struct svs_utls *utls;
575 pt_entry_t pte;
576 size_t i;
577
578 KASSERT(kpreempt_disabled());
579 KASSERT(pmap != pmap_kernel());
580
581 /* Update the info in the UTLS page */
582 utls = (struct svs_utls *)ci->ci_svs_utls;
583 utls->kpdirpa = pmap_pdirpa(pmap, 0) | svs_pcid_kcr3;
584
585 mutex_enter(&ci->ci_svs_mtx);
586
587 /* User slots. */
588 for (i = 0; i < PDIR_SLOT_USERLIM; i++) {
589 pte = svs_pte_atomic_read(pmap, i);
590 ci->ci_svs_updir[i] = pte;
591 }
592
593 mutex_exit(&ci->ci_svs_mtx);
594
595 if (svs_pcid) {
596 invpcid(INVPCID_CONTEXT, PMAP_PCID_USER, 0);
597 }
598}

600static void
601svs_enable(void)
602{
603 extern uint8_t svs_enter, svs_enter_end;
604 extern uint8_t svs_enter_altstack, svs_enter_altstack_end;
605 extern uint8_t svs_enter_nmi, svs_enter_nmi_end;
606 extern uint8_t svs_leave, svs_leave_end;
607 extern uint8_t svs_leave_altstack, svs_leave_altstack_end;
608 extern uint8_t svs_leave_nmi, svs_leave_nmi_end;
609 u_long psl, cr0;
610 uint8_t *bytes;
611 size_t size;
612
613 svs_enabled = true;
614
615 x86_patch_window_open(&psl, &cr0);
616
617 bytes = &svs_enter;
618 size = (size_t)&svs_enter_end - (size_t)&svs_enter;
619 x86_hotpatch(HP_NAME_SVS_ENTER, bytes, size);
620
621 bytes = &svs_enter_altstack;
622 size = (size_t)&svs_enter_altstack_end -
623 (size_t)&svs_enter_altstack;
624 x86_hotpatch(HP_NAME_SVS_ENTER_ALT, bytes, size);
625
626 bytes = &svs_enter_nmi;
627 size = (size_t)&svs_enter_nmi_end -
628 (size_t)&svs_enter_nmi;
629 x86_hotpatch(HP_NAME_SVS_ENTER_NMI, bytes, size);
630
631 bytes = &svs_leave;
632 size = (size_t)&svs_leave_end - (size_t)&svs_leave;
633 x86_hotpatch(HP_NAME_SVS_LEAVE, bytes, size);
634
635 bytes = &svs_leave_altstack;
636 size = (size_t)&svs_leave_altstack_end -
637 (size_t)&svs_leave_altstack;
638 x86_hotpatch(HP_NAME_SVS_LEAVE_ALT, bytes, size);
639
640 bytes = &svs_leave_nmi;
641 size = (size_t)&svs_leave_nmi_end -
642 (size_t)&svs_leave_nmi;
643 x86_hotpatch(HP_NAME_SVS_LEAVE_NMI, bytes, size);
644
645 x86_patch_window_close(psl, cr0);
646}

648void
649svs_init(void)
650{
651 uint64_t msr;
652
653 if (cpu_vendor != CPUVENDOR_INTEL) {
654 return;
655 }
656 if (boothowto & RB_MD3) {
657 return;
658 }
659 if (cpu_info_primary.ci_feat_val[7] & CPUID_SEF_ARCH_CAP) {
660 msr = rdmsr(MSR_IA32_ARCH_CAPABILITIES);
661 if (msr & IA32_ARCH_RDCL_NO) {
662 /*
663 * The processor indicates it is not vulnerable to the
664 * Rogue Data Cache Load (Meltdown) flaw.
665 */
666 return;
667 }
668 }
669
670 if ((cpu_info_primary.ci_feat_val[1] & CPUID2_PCID) &&
671 (cpu_info_primary.ci_feat_val[5] & CPUID_SEF_INVPCID)) {
672 svs_pcid = true;
673 lcr4(rcr4() | CR4_PCIDE);
674 }
675
676 svs_enable();
677}
