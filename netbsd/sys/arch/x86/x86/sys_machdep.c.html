87int
88x86_get_ldt(struct lwp *l, void *args, register_t *retval)
89{
90#ifndef USER_LDT
91 return EINVAL;
92#else
93 struct x86_get_ldt_args ua;
94 union descriptor *cp;
95 int error;
96
97 if ((error = copyin(args, &ua, sizeof(ua))) != 0)
98 return error;
99
100 if (ua.num < 0 || ua.num > 8192)
101 return EINVAL;
102
103 cp = malloc(ua.num * sizeof(union descriptor), M_TEMP, M_WAITOK);
104 if (cp == NULL)
105 return ENOMEM;
106
107 error = x86_get_ldt1(l, &ua, cp);
108 *retval = ua.num;
109 if (error == 0)
110 error = copyout(cp, ua.desc, ua.num * sizeof(*cp));
111
112 free(cp, M_TEMP);
113 return error;
114#endif
115}

117int
118x86_get_ldt1(struct lwp *l, struct x86_get_ldt_args *ua, union descriptor *cp)
119{
120#ifndef USER_LDT
121 return EINVAL;
122#else
123 int error;
124 struct proc *p = l->l_proc;
125 pmap_t pmap = p->p_vmspace->vm_map.pmap;
126 int nldt, num;
127 union descriptor *lp;
128
129#ifdef __x86_64__
130 const size_t min_ldt_size = LDT_SIZE;
131#else
132 const size_t min_ldt_size = NLDT * sizeof(union descriptor);
133#endif
134
135 error = kauth_authorize_machdep(l->l_cred, KAUTH_MACHDEP_LDT_GET,
136 NULL, NULL, NULL, NULL);
137 if (error)
138 return error;
139
140 if (ua->start < 0 || ua->num < 0 || ua->start > 8192 || ua->num > 8192 ||
141 ua->start + ua->num > 8192)
142 return EINVAL;
143
144 if (ua->start * sizeof(union descriptor) < min_ldt_size)
145 return EINVAL;
146
147 mutex_enter(&cpu_lock);
148
149 if (pmap->pm_ldt != NULL) {
150 nldt = pmap->pm_ldt_len / sizeof(*lp);
151 lp = pmap->pm_ldt;
152 } else {
153#ifdef __x86_64__
154 nldt = LDT_SIZE / sizeof(*lp);
155#else
156 nldt = NLDT;
157#endif
158 lp = (union descriptor *)ldtstore;
159 }
160
161 if (ua->start > nldt) {
162 mutex_exit(&cpu_lock);
163 return EINVAL;
164 }
165
166 lp += ua->start;
167 num = uimin(ua->num, nldt - ua->start);
168 ua->num = num;
169
170 memcpy(cp, lp, num * sizeof(union descriptor));
171 mutex_exit(&cpu_lock);
172
173 return 0;
174#endif
175}

177int
178x86_set_ldt(struct lwp *l, void *args, register_t *retval)
179{
180#ifndef USER_LDT
181 return EINVAL;
182#else
183 struct x86_set_ldt_args ua;
184 union descriptor *descv;
185 int error;
186
187 if ((error = copyin(args, &ua, sizeof(ua))) != 0)
188 return error;
189
190 if (ua.num < 0 || ua.num > 8192)
191 return EINVAL;
192
193 descv = malloc(sizeof (*descv) * ua.num, M_TEMP, M_NOWAIT);
194 if (descv == NULL)
195 return ENOMEM;
196
197 error = copyin(ua.desc, descv, sizeof (*descv) * ua.num);
198 if (error == 0)
199 error = x86_set_ldt1(l, &ua, descv);
200 *retval = ua.start;
201
202 free(descv, M_TEMP);
203 return error;
204#endif
205}

207int
208x86_set_ldt1(struct lwp *l, struct x86_set_ldt_args *ua,
209 union descriptor *descv)
210{
211#ifndef USER_LDT
212 return EINVAL;
213#else
214 int error, i, n, old_sel, new_sel;
215 struct proc *p = l->l_proc;
216 pmap_t pmap = p->p_vmspace->vm_map.pmap;
217 size_t old_len, new_len;
218 union descriptor *old_ldt, *new_ldt;
219
220#ifdef __x86_64__
221 const size_t min_ldt_size = LDT_SIZE;
222#else
223 const size_t min_ldt_size = NLDT * sizeof(union descriptor);
224#endif
225
226 error = kauth_authorize_machdep(l->l_cred, KAUTH_MACHDEP_LDT_SET,
227 NULL, NULL, NULL, NULL);
228 if (error)
229 return error;
230
231 if (ua->start < 0 || ua->num < 0 || ua->start > 8192 || ua->num > 8192 ||
232 ua->start + ua->num > 8192)
233 return EINVAL;
234
235 if (ua->start * sizeof(union descriptor) < min_ldt_size)
236 return EINVAL;
237
238 /* Check descriptors for access violations. */
239 for (i = 0; i < ua->num; i++) {
240 union descriptor *desc = &descv[i];
241
242#ifdef __x86_64__
243 if (desc->sd.sd_long != 0)
244 return EACCES;
245#endif
246
247 switch (desc->sd.sd_type) {
248 case SDT_SYSNULL:
249 desc->sd.sd_p = 0;
250 break;
251 case SDT_MEMEC:
252 case SDT_MEMEAC:
253 case SDT_MEMERC:
254 case SDT_MEMERAC:
255 /* Must be "present" if executable and conforming. */
256 if (desc->sd.sd_p == 0)
257 return EACCES;
258 break;
259 case SDT_MEMRO:
260 case SDT_MEMROA:
261 case SDT_MEMRW:
262 case SDT_MEMRWA:
263 case SDT_MEMROD:
264 case SDT_MEMRODA:
265 case SDT_MEMRWD:
266 case SDT_MEMRWDA:
267 case SDT_MEME:
268 case SDT_MEMEA:
269 case SDT_MEMER:
270 case SDT_MEMERA:
271 break;
272 default:
273 return EACCES;
274 }
275
276 if (desc->sd.sd_p != 0) {
277 /* Only user (ring-3) descriptors may be present. */
278 if (desc->sd.sd_dpl != SEL_UPL)
279 return EACCES;
280 }
281 }
282
283 /*
284 * Install selected changes. We perform a copy, write, swap dance
285 * here to ensure that all updates happen atomically.
286 */
287
288 /* Allocate a new LDT. */
289 for (;;) {
290 new_len = (ua->start + ua->num) * sizeof(union descriptor);
291 new_len = uimax(new_len, pmap->pm_ldt_len);
292 new_len = uimax(new_len, min_ldt_size);
293 new_len = round_page(new_len);
294 new_ldt = (union descriptor *)uvm_km_alloc(kernel_map,
295 new_len, 0, UVM_KMF_WIRED | UVM_KMF_ZERO | UVM_KMF_WAITVA);
296 mutex_enter(&cpu_lock);
297 if (pmap->pm_ldt_len <= new_len) {
298 break;
299 }
300 mutex_exit(&cpu_lock);
301 uvm_km_free(kernel_map, (vaddr_t)new_ldt, new_len,
302 UVM_KMF_WIRED);
303 }
304
305 /* Copy existing entries, if any. */
306 if (pmap->pm_ldt != NULL) {
307 old_ldt = pmap->pm_ldt;
308 old_len = pmap->pm_ldt_len;
309 old_sel = pmap->pm_ldt_sel;
310 memcpy(new_ldt, old_ldt, old_len);
311 } else {
312 old_ldt = NULL;
313 old_len = 0;
314 old_sel = -1;
315 memcpy(new_ldt, ldtstore, min_ldt_size);
316 }
317
318 /* Apply requested changes. */
319 for (i = 0, n = ua->start; i < ua->num; i++, n++) {
320 new_ldt[n] = descv[i];
321 }
322
323 /* Allocate LDT selector. */
324 new_sel = ldt_alloc(new_ldt, new_len);
325 if (new_sel == -1) {
326 mutex_exit(&cpu_lock);
327 uvm_km_free(kernel_map, (vaddr_t)new_ldt, new_len,
328 UVM_KMF_WIRED);
329 return ENOMEM;
330 }
331
332 /* All changes are now globally visible. Swap in the new LDT. */
333 pmap->pm_ldt_len = new_len;
334 pmap->pm_ldt_sel = new_sel;
335 /* membar_store_store for pmap_fork() to read these unlocked safely */
336 membar_producer();
337 pmap->pm_ldt = new_ldt;
338
339 /* Switch existing users onto new LDT. */
340 pmap_ldt_sync(pmap);
341
342 /* Free existing LDT (if any). */
343 if (old_ldt != NULL) {
344 ldt_free(old_sel);
345 /* exit the mutex before free */
346 mutex_exit(&cpu_lock);
347 uvm_km_free(kernel_map, (vaddr_t)old_ldt, old_len,
348 UVM_KMF_WIRED);
349 } else {
350 mutex_exit(&cpu_lock);
351 }
352
353 return error;
354#endif
355}

357int
358x86_iopl(struct lwp *l, void *args, register_t *retval)
359{
360 int error;
361 struct x86_iopl_args ua;
362#ifdef XENPV
363 int iopl;
364#else
365 struct trapframe *tf = l->l_md.md_regs;
366#endif
367
368 error = kauth_authorize_machdep(l->l_cred, KAUTH_MACHDEP_IOPL,
369 NULL, NULL, NULL, NULL);
370 if (error)
371 return error;
372
373 if ((error = copyin(args, &ua, sizeof(ua))) != 0)
374 return error;
375
376#ifdef XENPV
377 if (ua.iopl)
378 iopl = SEL_UPL;
379 else
380 iopl = SEL_KPL;
381
382 {
383 struct physdev_op physop;
384 struct pcb *pcb;
385
386 pcb = lwp_getpcb(l);
387 pcb->pcb_iopl = iopl;
388
389 /* Force the change at ring 0. */
390 physop.cmd = PHYSDEVOP_SET_IOPL;
391 physop.u.set_iopl.iopl = iopl;
392 HYPERVISOR_physdev_op(&physop);
393 }
394#elif defined(__x86_64__)
395 if (ua.iopl)
396 tf->tf_rflags |= PSL_IOPL;
397 else
398 tf->tf_rflags &= ~PSL_IOPL;
399#else
400 if (ua.iopl)
401 tf->tf_eflags |= PSL_IOPL;
402 else
403 tf->tf_eflags &= ~PSL_IOPL;
404#endif
405
406 return 0;
407}

410x86_get_ioperm(struct lwp *l, void *args, register_t *retval)
411{
412#ifdef IOPERM
413 int error;
414 struct pcb *pcb = lwp_getpcb(l);
415 struct x86_get_ioperm_args ua;
416 void *dummymap = NULL;
417 void *iomap;
418
419 error = kauth_authorize_machdep(l->l_cred, KAUTH_MACHDEP_IOPERM_GET,
420 NULL, NULL, NULL, NULL);
421 if (error)
422 return error;
423
424 if ((error = copyin(args, &ua, sizeof(ua))) != 0)
425 return error;
426
427 iomap = pcb->pcb_iomap;
428 if (iomap == NULL) {
429 iomap = dummymap = kmem_alloc(IOMAPSIZE, KM_SLEEP);
430 memset(dummymap, 0xff, IOMAPSIZE);
431 }
432 error = copyout(iomap, ua.iomap, IOMAPSIZE);
433 if (dummymap != NULL) {
434 kmem_free(dummymap, IOMAPSIZE);
435 }
436 return error;
437#else
438 return EINVAL;
439#endif
440}

442int
443x86_set_ioperm(struct lwp *l, void *args, register_t *retval)
444{
445#ifdef IOPERM
446 struct cpu_info *ci;
447 int error;
448 struct pcb *pcb = lwp_getpcb(l);
449 struct x86_set_ioperm_args ua;
450 void *new;
451 void *old;
452
453 error = kauth_authorize_machdep(l->l_cred, KAUTH_MACHDEP_IOPERM_SET,
454 NULL, NULL, NULL, NULL);
455 if (error)
456 return error;
457
458 if ((error = copyin(args, &ua, sizeof(ua))) != 0)
459 return error;
460
461 new = kmem_alloc(IOMAPSIZE, KM_SLEEP);
462 error = copyin(ua.iomap, new, IOMAPSIZE);
463 if (error) {
464 kmem_free(new, IOMAPSIZE);
465 return error;
466 }
467 old = pcb->pcb_iomap;
468 pcb->pcb_iomap = new;
469 if (old != NULL) {
470 kmem_free(old, IOMAPSIZE);
471 }
472
473 CTASSERT(offsetof(struct cpu_tss, iomap) -
474 offsetof(struct cpu_tss, tss) == IOMAP_VALIDOFF);
475
476 kpreempt_disable();
477 ci = curcpu();
478 memcpy(ci->ci_tss->iomap, pcb->pcb_iomap, IOMAPSIZE);
479 ci->ci_tss->tss.tss_iobase = IOMAP_VALIDOFF << 16;
480 kpreempt_enable();
481
482 return error;
483#else
484 return EINVAL;
485#endif
486}

566int
567x86_set_sdbase32(void *arg, char which, lwp_t *l, bool direct)
568{
569 struct trapframe *tf = l->l_md.md_regs;
570 union descriptor usd;
571 struct pcb *pcb;
572 uint32_t base;
573 int error;
574
575 if (direct) {
576 base = (vaddr_t)arg;
577 } else {
578 error = copyin(arg, &base, sizeof(base));
579 if (error != 0)
580 return error;
581 }
582
583 memset(&usd, 0, sizeof(usd));
584 usd.sd.sd_lobase = base & 0xffffff;
585 usd.sd.sd_hibase = (base >> 24) & 0xff;
586 usd.sd.sd_lolimit = 0xffff;
587 usd.sd.sd_hilimit = 0xf;
588 usd.sd.sd_type = SDT_MEMRWA;
589 usd.sd.sd_dpl = SEL_UPL;
590 usd.sd.sd_p = 1;
591 usd.sd.sd_def32 = 1;
592 usd.sd.sd_gran = 1;
593
594 pcb = lwp_getpcb(l);
595 kpreempt_disable();
596 if (which == 'f') {
597 memcpy(&pcb->pcb_fsd, &usd.sd,
598 sizeof(struct segment_descriptor));
599 if (l == curlwp) {
600 update_descriptor(&curcpu()->ci_gdt[GUFS_SEL], &usd);
601 }
602 tf->tf_fs = GSEL(GUFS_SEL, SEL_UPL);
603 } else /* which == 'g' */ {
604 memcpy(&pcb->pcb_gsd, &usd.sd,
605 sizeof(struct segment_descriptor));
606 if (l == curlwp) {
607 update_descriptor(&curcpu()->ci_gdt[GUGS_SEL], &usd);
608#if defined(__x86_64__) && defined(XENPV)
609 setusergs(GSEL(GUGS_SEL, SEL_UPL));
610#endif
611 }
612 tf->tf_gs = GSEL(GUGS_SEL, SEL_UPL);
613 }
614 kpreempt_enable();
615 return 0;
616}

618int
619x86_set_sdbase(void *arg, char which, lwp_t *l, bool direct)
620{
621#ifdef i386
622 return x86_set_sdbase32(arg, which, l, direct);
623#else
624 struct pcb *pcb;
625 vaddr_t base;
626
627 if (l->l_proc->p_flag & PK_32) {
628 return x86_set_sdbase32(arg, which, l, direct);
629 }
630
631 if (direct) {
632 base = (vaddr_t)arg;
633 } else {
634 int error = copyin(arg, &base, sizeof(base));
635 if (error != 0)
636 return error;
637 }
638
639 if (base >= VM_MAXUSER_ADDRESS)
640 return EINVAL;
641
642 pcb = lwp_getpcb(l);
643
644 kpreempt_disable();
645 switch(which) {
646 case 'f':
647 pcb->pcb_fs = base;
648 if (l == curlwp)
649 wrmsr(MSR_FSBASE, pcb->pcb_fs);
650 break;
651 case 'g':
652 pcb->pcb_gs = base;
653 if (l == curlwp)
654 wrmsr(MSR_KERNELGSBASE, pcb->pcb_gs);
655 break;
656 default:
657 panic("x86_set_sdbase");
658 }
659 kpreempt_enable();
660
661 return 0;
662#endif
663}

665int
666x86_get_sdbase32(void *arg, char which)
667{
668 struct segment_descriptor *sd;
669 uint32_t base;
670
671 switch (which) {
672 case 'f':
673 sd = (void *)&curpcb->pcb_fsd;
674 break;
675 case 'g':
676 sd = (void *)&curpcb->pcb_gsd;
677 break;
678 default:
679 panic("x86_get_sdbase32");
680 }
681
682 base = sd->sd_hibase << 24 | sd->sd_lobase;
683 return copyout(&base, arg, sizeof(base));
684}

687x86_get_sdbase(void *arg, char which)
688{
689#ifdef i386
690 return x86_get_sdbase32(arg, which);
691#else
692 vaddr_t base;

700

702 case 'f':
703 base = pcb->pcb_fs;
704 break;
705 case 'g':
706 base = pcb->pcb_gs;
707 break;
708 default:
709 panic("x86_get_sdbase");
710 }
711
712 return copyout(&base, arg, sizeof(base));
713#endif
714}

716int
717sys_sysarch(struct lwp *l, const struct sys_sysarch_args *uap, register_t *retval)
718{
719 /* {
720 syscallarg(int) op;
721 syscallarg(void *) parms;
722 } */
723 int error = 0;
724
725 switch(SCARG(uap, op)) {
726 case X86_IOPL:
727 error = x86_iopl(l, SCARG(uap, parms), retval);
728 break;
729
730#ifdef i386
731 /*
732 * On amd64, this is done via netbsd32_sysarch.
733 */
734 case X86_GET_LDT:
735 error = x86_get_ldt(l, SCARG(uap, parms), retval);
736 break;
737
738 case X86_SET_LDT:
739 error = x86_set_ldt(l, SCARG(uap, parms), retval);
740 break;
741#endif
742
743 case X86_GET_IOPERM:
744 error = x86_get_ioperm(l, SCARG(uap, parms), retval);
745 break;
746
747 case X86_SET_IOPERM:
748 error = x86_set_ioperm(l, SCARG(uap, parms), retval);
749 break;
750
751 case X86_GET_MTRR:
752 error = x86_get_mtrr(l, SCARG(uap, parms), retval);
753 break;
754 case X86_SET_MTRR:
755 error = x86_set_mtrr(l, SCARG(uap, parms), retval);
756 break;
757
758 case X86_SET_FSBASE:
759 error = x86_set_sdbase(SCARG(uap, parms), 'f', curlwp, false);
760 break;
761
762 case X86_SET_GSBASE:
763 error = x86_set_sdbase(SCARG(uap, parms), 'g', curlwp, false);
764 break;
765
766 case X86_GET_FSBASE:
767 error = x86_get_sdbase(SCARG(uap, parms), 'f');
768 break;
769
770 case X86_GET_GSBASE:
771 error = x86_get_sdbase(SCARG(uap, parms), 'g');
772 break;
773
774 default:
775 error = EINVAL;
776 break;
777 }
778 return error;
779}
