60md_child_return(struct lwp *l)
61{
62 struct trapframe *tf = l->l_md.md_regs;
63
64 X86_TF_RAX(tf) = 0;
65 X86_TF_RFLAGS(tf) &= ~PSL_C;
66
67 userret(l);
68}

80/*
81 * syscall(frame):
82 * System call request from POSIX system call gate interface to kernel.
83 * Like trap(), argument is call by reference.
84 */
85#ifdef KDTRACE_HOOKS
86void syscall(struct trapframe *);
87#else
88static
89#endif
90void
91syscall(struct trapframe *frame)
92{
93 const struct sysent *callp;
94 struct proc *p;
95 struct lwp *l;
96 int error;
97 register_t code, rval[2];
98#ifdef __x86_64__
99 /* Verify that the syscall args will fit in the trapframe space */
100 CTASSERT(offsetof(struct trapframe, tf_arg9) >=
101 sizeof(register_t) * (2 + SYS_MAXSYSARGS - 1));
102#define args (&frame->tf_rdi)
103#else
104 register_t args[2 + SYS_MAXSYSARGS];
105#endif
106
107 l = curlwp;
108 p = l->l_proc;
109 LWP_CACHE_CREDS(l, p);
110
111 code = X86_TF_RAX(frame) & (SYS_NSYSENT - 1);
112 callp = p->p_emul->e_sysent + code;
113
114 SYSCALL_COUNT(syscall_counts, code);
115 SYSCALL_TIME_SYS_ENTRY(l, syscall_times, code);
116
117#ifdef __x86_64__
118 /*
119 * The first 6 syscall args are passed in rdi, rsi, rdx, r10, r8 and r9
120 * (rcx gets copied to r10 in the libc stub because the syscall
121 * instruction overwrites %cx) and are together in the trap frame
122 * with space following for 4 more entries.
123 */
124 if (__predict_false(callp->sy_argsize > 6 * 8)) {
125 error = copyin((register_t *)frame->tf_rsp + 1,
126 &frame->tf_arg6, callp->sy_argsize - 6 * 8);
127 if (error != 0)
128 goto bad;
129 }
130#else
131 if (callp->sy_argsize) {
132 error = x86_copyargs((char *)frame->tf_esp + sizeof(int), args,
133 callp->sy_argsize);
134 if (__predict_false(error != 0))
135 goto bad;
136 }
137#endif
138 error = sy_invoke(callp, l, args, rval, code);
139
140 if (__predict_true(error == 0)) {
141 X86_TF_RAX(frame) = rval[0];
142 X86_TF_RDX(frame) = rval[1];
143 X86_TF_RFLAGS(frame) &= ~PSL_C; /* carry bit */
144 } else {
145 switch (error) {
146 case ERESTART:
147 /*
148 * The offset to adjust the PC by depends on whether we
149 * entered the kernel through the trap or call gate.
150 * We saved the instruction size in tf_err on entry.
151 */
152 X86_TF_RIP(frame) -= frame->tf_err;
153 break;
154 case EJUSTRETURN:
155 /* nothing to do */
156 break;
157 default:
158 bad:
159 X86_TF_RAX(frame) = error;
160 X86_TF_RFLAGS(frame) |= PSL_C; /* carry bit */
161 break;
162 }
163 }
164
165 SYSCALL_TIME_SYS_EXIT(l);
166 userret(l);
167}

169void
170syscall_intern(struct proc *p)
171{
172
173 p->p_md.md_syscall = syscall;
174}
