69bool
70tsc_is_invariant(void)
71{
72 struct cpu_info *ci;
73 uint32_t descs[4];
74 uint32_t family;
75 bool invariant;
76
77 if (!cpu_hascounter())
78 return false;
79
80 ci = curcpu();
81 invariant = false;
82
83 if (cpu_vendor == CPUVENDOR_INTEL) {
84 /*
85 * From Intel(tm) 64 and IA-32 Architectures Software
86 * Developer's Manual Volume 3A: System Programming Guide,
87 * Part 1, 17.13 TIME_STAMP COUNTER, these are the processors
88 * where the TSC is known invariant:
89 *
90 * Pentium 4, Intel Xeon (family 0f, models 03 and higher)
91 * Core Solo and Core Duo processors (family 06, model 0e)
92 * Xeon 5100 series and Core 2 Duo (family 06, model 0f)
93 * Core 2 and Xeon (family 06, model 17)
94 * Atom (family 06, model 1c)
95 *
96 * We'll also assume that it's safe on the Pentium, and
97 * that it's safe on P-II and P-III Xeons due to the
98 * typical configuration of those systems.
99 *
100 */
101 switch (CPUID_TO_BASEFAMILY(ci->ci_signature)) {
102 case 0x05:
103 invariant = true;
104 break;
105 case 0x06:
106 invariant = CPUID_TO_MODEL(ci->ci_signature) == 0x0e ||
107 CPUID_TO_MODEL(ci->ci_signature) == 0x0f ||
108 CPUID_TO_MODEL(ci->ci_signature) == 0x17 ||
109 CPUID_TO_MODEL(ci->ci_signature) == 0x1c;
110 break;
111 case 0x0f:
112 invariant = CPUID_TO_MODEL(ci->ci_signature) >= 0x03;
113 break;
114 }
115 } else if (cpu_vendor == CPUVENDOR_AMD) {
116 /*
117 * TSC and Power Management Events on AMD Processors
118 * Nov 2, 2005 Rich Brunner, AMD Fellow
119 * http://lkml.org/lkml/2005/11/4/173
120 *
121 * See Appendix E.4.7 CPUID Fn8000_0007_EDX Advanced Power
122 * Management Features, AMD64 Architecture Programmer's
123 * Manual Volume 3: General-Purpose and System Instructions.
124 * The check is done below.
125 */
126 }
127
128 /*
129 * The best way to check whether the TSC counter is invariant or not
130 * is to check CPUID 80000007.
131 */
132 family = CPUID_TO_BASEFAMILY(ci->ci_signature);
133 if (((cpu_vendor == CPUVENDOR_INTEL) || (cpu_vendor == CPUVENDOR_AMD))
134 && ((family == 0x06) || (family == 0x0f))) {
135 x86_cpuid(0x80000000, descs);
136 if (descs[0] >= 0x80000007) {
137 x86_cpuid(0x80000007, descs);
138 invariant = (descs[3] & CPUID_APM_TSC) != 0;
139 }
140 }
141
142 return invariant;
143}

145void
146tsc_tc_init(void)
147{
148 struct cpu_info *ci;
149 bool invariant;
150
151 if (!cpu_hascounter())
152 return;
153
154 ci = curcpu();
155 tsc_freq = ci->ci_data.cpu_cc_freq;
156 tsc_good = (cpu_feature[0] & CPUID_MSR) != 0 &&
157 (rdmsr(MSR_TSC) != 0 || rdmsr(MSR_TSC) != 0);
158
159 invariant = tsc_is_invariant();
160 if (!invariant) {
161 aprint_debug("TSC not known invariant on this CPU\n");
162 tsc_timecounter.tc_quality = -100;
163 } else if (tsc_drift_observed > tsc_drift_max) {
164 aprint_error("ERROR: %lld cycle TSC drift observed\n",
165 (long long)tsc_drift_observed);
166 tsc_timecounter.tc_quality = -100;
167 invariant = false;
168 }
169
170 if (tsc_freq != 0) {
171 tsc_timecounter.tc_frequency = tsc_freq;
172 tc_init(&tsc_timecounter);
173 }
174}

176/*
177 * Record drift (in clock cycles). Called during AP startup.
178 */
179void
180tsc_sync_drift(int64_t drift)
181{
182
183 if (drift < 0)
184 drift = -drift;
185 if (drift > tsc_drift_observed)
186 tsc_drift_observed = drift;
187}

189/*
190 * Called during startup of APs, by the boot processor. Interrupts
191 * are disabled on entry.
192 */
193static void
194tsc_read_bp(struct cpu_info *ci, uint64_t *bptscp, uint64_t *aptscp)
195{
196 uint64_t bptsc;
197
198 if (atomic_swap_ptr(&tsc_sync_cpu, ci) != NULL) {
199 panic("tsc_sync_bp: 1");
200 }
201
202 /* Flag it and read our TSC. */
203 atomic_or_uint(&ci->ci_flags, CPUF_SYNCTSC);
204 bptsc = cpu_counter_serializing() >> 1;
205
206 /* Wait for remote to complete, and read ours again. */
207 while ((ci->ci_flags & CPUF_SYNCTSC) != 0) {
208 __insn_barrier();
209 }
210 bptsc += (cpu_counter_serializing() >> 1);
211
212 /* Wait for the results to come in. */
213 while (tsc_sync_cpu == ci) {
214 x86_pause();
215 }
216 if (tsc_sync_cpu != NULL) {
217 panic("tsc_sync_bp: 2");
218 }
219
220 *bptscp = bptsc;
221 *aptscp = tsc_sync_val;
222}
223
224void
225tsc_sync_bp(struct cpu_info *ci)
226{
227 uint64_t bptsc, aptsc;
228
229 tsc_read_bp(ci, &bptsc, &aptsc); /* discarded - cache effects */
230 tsc_read_bp(ci, &bptsc, &aptsc);
231
232 /* Compute final value to adjust for skew. */
233 ci->ci_data.cpu_cc_skew = bptsc - aptsc;
234}

236/*
237 * Called during startup of AP, by the AP itself. Interrupts are
238 * disabled on entry.
239 */
240static void
241tsc_post_ap(struct cpu_info *ci)
242{
243 uint64_t tsc;
244
245 /* Wait for go-ahead from primary. */
246 while ((ci->ci_flags & CPUF_SYNCTSC) == 0) {
247 __insn_barrier();
248 }
249 tsc = (cpu_counter_serializing() >> 1);
250
251 /* Instruct primary to read its counter. */
252 atomic_and_uint(&ci->ci_flags, ~CPUF_SYNCTSC);
253 tsc += (cpu_counter_serializing() >> 1);
254
255 /* Post result. Ensure the whole value goes out atomically. */
256 (void)atomic_swap_64(&tsc_sync_val, tsc);
257
258 if (atomic_swap_ptr(&tsc_sync_cpu, NULL) != ci) {
259 panic("tsc_sync_ap");
260 }
261}
262
263void
264tsc_sync_ap(struct cpu_info *ci)
265{
266
267 tsc_post_ap(ci);
268 tsc_post_ap(ci);
269}

271static void
272tsc_apply_cpu(void *arg1, void *arg2)
273{
274 bool enable = (bool)arg1;
275 if (enable) {
276 lcr4(rcr4() & ~CR4_TSD);
277 } else {
278 lcr4(rcr4() | CR4_TSD);
279 }
280}

282void
283tsc_user_enable(void)
284{
285 uint64_t xc;
286
287 xc = xc_broadcast(0, tsc_apply_cpu, (void *)true, NULL);
288 xc_wait(xc);
289}
290
291void
292tsc_user_disable(void)
293{
294 uint64_t xc;
295
296 xc = xc_broadcast(0, tsc_apply_cpu, (void *)false, NULL);
297 xc_wait(xc);
298}

314uint64_t
315cpu_counter_serializing(void)
316{
317 if (tsc_good)
318 return rdmsr(MSR_TSC);
319 else
320 return cpu_counter();
321}
