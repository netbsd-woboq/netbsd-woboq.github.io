119/*
120 * cpu_lwp_fork: finish a new LWP (l2) operation.
121 *
122 * First LWP (l1) is the process being forked. If it is &lwp0, then we
123 * are creating a kthread, where return path and argument are specified
124 * with `func' and `arg'.
125 *
126 * If an alternate user-level stack is requested (with non-zero values
127 * in both the stack and stacksize arguments), then set up the user stack
128 * pointer accordingly.
129 */
130void
131cpu_lwp_fork(struct lwp *l1, struct lwp *l2, void *stack, size_t stacksize,
132 void (*func)(void *), void *arg)
133{
134 struct pcb *pcb1, *pcb2;
135 struct trapframe *tf;
136 struct switchframe *sf;
137 vaddr_t uv;
138
139 pcb1 = lwp_getpcb(l1);
140 pcb2 = lwp_getpcb(l2);
141
142 /*
143 * If parent LWP was using FPU, then we have to save the FPU h/w
144 * state to PCB so that we can copy it.
145 */
146 fpusave_lwp(l1, true);
147
148 /*
149 * Sync the PCB before we copy it.
150 */
151 if (l1 == curlwp) {
152 KASSERT(pcb1 == curpcb);
153 savectx(pcb1);
154 } else {
155 KASSERT(l1 == &lwp0);
156 }
157
158 /* Copy the PCB from parent, except the FPU state. */
159 memcpy(pcb2, pcb1, offsetof(struct pcb, pcb_savefpu));
160
161 /* FPU state not installed. */
162 pcb2->pcb_fpcpu = NULL;
163
164 /* Copy FPU state. */
165 fpu_save_area_fork(pcb2, pcb1);
166
167 /* Never inherit CPU Debug Registers */
168 pcb2->pcb_dbregs = NULL;
169 pcb2->pcb_flags &= ~PCB_DBREGS;
170
171#if defined(XENPV)
172 pcb2->pcb_iopl = IOPL_KPL;
173#endif
174
175 /*
176 * Set the kernel stack address (from the address to uarea) and
177 * trapframe address for child.
178 *
179 * Rig kernel stack so that it would start out in lwp_trampoline()
180 * and call child_return() with l2 as an argument. This causes the
181 * newly-created child process to go directly to user level with a
182 * parent return value of 0 from fork(), while the parent process
183 * returns normally.
184 */
185 uv = uvm_lwp_getuarea(l2);
186 KASSERT(uv % PAGE_SIZE == 0);
187
188#ifdef __x86_64__
189#ifdef SVS
190 pcb2->pcb_rsp0 = (uv + USPACE - PAGE_SIZE +
191 sizeof(struct trapframe));
192 KASSERT((pcb2->pcb_rsp0 & 0xF) == 0);
193#else
194 pcb2->pcb_rsp0 = (uv + USPACE - 16);
195#endif
196 tf = (struct trapframe *)pcb2->pcb_rsp0 - 1;
197#else
198 pcb2->pcb_esp0 = (uv + USPACE - 16);
199 tf = (struct trapframe *)pcb2->pcb_esp0 - 1;
200
201 pcb2->pcb_iomap = NULL;
202#endif
203 l2->l_md.md_regs = tf;
204
205 /*
206 * Copy the trapframe from parent, so that return to userspace
207 * will be to right address, with correct registers.
208 */
209 memcpy(tf, l1->l_md.md_regs, sizeof(struct trapframe));
210
211 /* Child LWP might get aston() before returning to userspace. */
212 tf->tf_trapno = T_ASTFLT;
213
214#if 0 /* DIAGNOSTIC */
215 /* Set a red zone in the kernel stack after the uarea. */
216 pmap_kremove(uv, PAGE_SIZE);
217 pmap_update(pmap_kernel());
218#endif
219
220 /* If specified, set a different user stack for a child. */
221 if (stack != NULL) {
222#ifdef __x86_64__
223 tf->tf_rsp = (uint64_t)stack + stacksize;
224#else
225 tf->tf_esp = (uint32_t)stack + stacksize;
226#endif
227 }
228
229 l2->l_md.md_flags = l1->l_md.md_flags;
230 l2->l_md.md_astpending = 0;
231
232 sf = (struct switchframe *)tf - 1;
233
234#ifdef __x86_64__
235 sf->sf_r12 = (uint64_t)func;
236 sf->sf_r13 = (uint64_t)arg;
237 sf->sf_rip = (uint64_t)lwp_trampoline;
238 pcb2->pcb_rsp = (uint64_t)sf;
239 pcb2->pcb_rbp = (uint64_t)l2;
240#else
241 /*
242 * XXX Is there a reason sf->sf_edi isn't initialized here?
243 * Could this leak potentially sensitive information to new
244 * userspace processes?
245 */
246 sf->sf_esi = (int)func;
247 sf->sf_ebx = (int)arg;
248 sf->sf_eip = (int)lwp_trampoline;
249 pcb2->pcb_esp = (int)sf;
250 pcb2->pcb_ebp = (int)l2;
251#endif
252}

254/*
255 * cpu_lwp_free is called from exit() to let machine-dependent
256 * code free machine-dependent resources. Note that this routine
257 * must not block.
258 */
259void
260cpu_lwp_free(struct lwp *l, int proc)
261{
262
263 /* If we were using the FPU, forget about it. */
264 fpusave_lwp(l, false);
265
266 /* Abandon the dbregs state. */
267 x86_dbregs_abandon(l);
268
269#ifdef MTRR
270 if (proc && l->l_proc->p_md.md_flags & MDP_USEDMTRR)
271 mtrr_clean(l->l_proc);
272#endif
273 /*
274 * Free deferred mappings if any.
275 */
276 struct vm_page *empty_ptps = l->l_md.md_gc_ptp;
277 l->l_md.md_gc_ptp = NULL;
278 pmap_free_ptps(empty_ptps);
279}

281/*
282 * cpu_lwp_free2 is called when an LWP is being reaped.
283 * This routine may block.
284 */
285void
286cpu_lwp_free2(struct lwp *l)
287{
288 struct pcb *pcb;
289
290 KASSERT(l->l_md.md_gc_ptp == NULL);
291 KASSERT(l->l_md.md_gc_pmap == NULL);
292
293 pcb = lwp_getpcb(l);
294 KASSERT((pcb->pcb_flags & PCB_DBREGS) == 0);
295 if (pcb->pcb_dbregs) {
296 pool_put(&x86_dbregspl, pcb->pcb_dbregs);
297 pcb->pcb_dbregs = NULL;
298 }
299}

301/*
302 * Convert kernel VA to physical address
303 */
304paddr_t
305kvtop(void *addr)
306{
307 paddr_t pa;
308 bool ret __diagused;
309
310 ret = pmap_extract(pmap_kernel(), (vaddr_t)addr, &pa);
311 KASSERT(ret == true);
312 return pa;
313}

315/*
316 * Map a user I/O request into kernel virtual address space.
317 * Note: the pages are already locked by uvm_vslock(), so we
318 * do not need to pass an access_type to pmap_enter().
319 */
320int
321vmapbuf(struct buf *bp, vsize_t len)
322{
323 vaddr_t faddr, taddr, off;
324 paddr_t fpa;
325
326 KASSERT((bp->b_flags & B_PHYS) != 0);
327
328 bp->b_saveaddr = bp->b_data;
329 faddr = trunc_page((vaddr_t)bp->b_data);
330 off = (vaddr_t)bp->b_data - faddr;
331 len = round_page(off + len);
332 taddr = uvm_km_alloc(phys_map, len, 0, UVM_KMF_VAONLY | UVM_KMF_WAITVA);
333 bp->b_data = (void *)(taddr + off);
334 /*
335 * The region is locked, so we expect that pmap_pte() will return
336 * non-NULL.
337 * XXX: unwise to expect this in a multithreaded environment.
338 * anything can happen to a pmap between the time we lock a
339 * region, release the pmap lock, and then relock it for
340 * the pmap_extract().
341 *
342 * no need to flush TLB since we expect nothing to be mapped
343 * where we we just allocated (TLB will be flushed when our
344 * mapping is removed).
345 */
346 while (len) {
347 (void) pmap_extract(vm_map_pmap(&bp->b_proc->p_vmspace->vm_map),
348 faddr, &fpa);
349 pmap_kenter_pa(taddr, fpa, VM_PROT_READ|VM_PROT_WRITE, 0);
350 faddr += PAGE_SIZE;
351 taddr += PAGE_SIZE;
352 len -= PAGE_SIZE;
353 }
354 pmap_update(pmap_kernel());
355
356 return 0;
357}

359/*
360 * Unmap a previously-mapped user I/O request.
361 */
362void
363vunmapbuf(struct buf *bp, vsize_t len)
364{
365 vaddr_t addr, off;
366
367 KASSERT((bp->b_flags & B_PHYS) != 0);
368
369 addr = trunc_page((vaddr_t)bp->b_data);
370 off = (vaddr_t)bp->b_data - addr;
371 len = round_page(off + len);
372 pmap_kremove(addr, len);
373 pmap_update(pmap_kernel());
374 uvm_km_free(phys_map, addr, len, UVM_KMF_VAONLY);
375 bp->b_data = bp->b_saveaddr;
376 bp->b_saveaddr = 0;
377}
