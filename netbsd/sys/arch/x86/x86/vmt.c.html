129static void
130vmt_probe_cmd(struct vm_backdoor *frame, uint16_t cmd)
131{
132 memset(frame, 0, sizeof(*frame));
133
134 (frame->eax).word = VM_MAGIC;
135 (frame->ebx).word = ~VM_MAGIC;
136 (frame->ecx).part.low = cmd;
137 (frame->ecx).part.high = 0xffff;
138 (frame->edx).part.low = VM_PORT_CMD;
139 (frame->edx).part.high = 0;
140
141 vm_cmd(frame);
142}
143
144static bool
145vmt_probe(void)
146{
147 struct vm_backdoor frame;
148
149 vmt_probe_cmd(&frame, VM_CMD_GET_VERSION);
150 if (frame.eax.word == 0xffffffff ||
151 frame.ebx.word != VM_MAGIC)
152 return false;
153
154 vmt_probe_cmd(&frame, VM_CMD_GET_SPEED);
155 if (frame.eax.word == VM_MAGIC)
156 return false;
157
158 return true;
159}

161static int
162vmt_match(device_t parent, cfdata_t match, void *aux)
163{
164 struct cpufeature_attach_args *cfaa = aux;
165 struct cpu_info *ci = cfaa->ci;
166
167 if (strcmp(cfaa->name, "vm") != 0)
168 return 0;
169 if ((ci->ci_flags & (CPUF_BSP|CPUF_SP|CPUF_PRIMARY)) == 0)
170 return 0;
171
172 return vmt_probe();
173}

175static void
176vmt_attach(device_t parent, device_t self, void *aux)
177{
178 int rv;
179 struct vmt_softc *sc = device_private(self);
180
181 aprint_naive("\n");
182 aprint_normal("\n");
183
184 sc->sc_dev = self;
185 sc->sc_log = NULL;
186
187 callout_init(&sc->sc_tick, 0);
188 callout_init(&sc->sc_tclo_tick, 0);
189 callout_init(&sc->sc_clock_sync_tick, 0);
190
191 sc->sc_clock_sync_period_seconds = VMT_CLOCK_SYNC_PERIOD_SECONDS;
192
193 rv = vmt_sysctl_setup_root(self);
194 if (rv != 0) {
195 aprint_error_dev(self, "failed to initialize sysctl "
196 "(err %d)\n", rv);
197 goto free;
198 }
199
200 sc->sc_rpc_buf = kmem_alloc(VMT_RPC_BUFLEN, KM_SLEEP);
201
202 if (vm_rpc_open(&sc->sc_tclo_rpc, VM_RPC_OPEN_TCLO) != 0) {
203 aprint_error_dev(self, "failed to open backdoor RPC channel (TCLO protocol)\n");
204 goto free;
205 }
206 sc->sc_tclo_rpc_open = true;
207
208 /* don't know if this is important at all yet */
209 if (vm_rpc_send_rpci_tx(sc, "tools.capability.hgfs_server toolbox 1") != 0) {
210 aprint_error_dev(self, "failed to set HGFS server capability\n");
211 goto free;
212 }
213
214 pmf_device_register1(self, NULL, NULL, vmt_shutdown);
215
216 sysmon_task_queue_init();
217
218 sc->sc_ev_power.ev_smpsw.smpsw_type = PSWITCH_TYPE_POWER;
219 sc->sc_ev_power.ev_smpsw.smpsw_name = device_xname(self);
220 sc->sc_ev_power.ev_code = PSWITCH_EVENT_PRESSED;
221 sysmon_pswitch_register(&sc->sc_ev_power.ev_smpsw);
222 sc->sc_ev_reset.ev_smpsw.smpsw_type = PSWITCH_TYPE_RESET;
223 sc->sc_ev_reset.ev_smpsw.smpsw_name = device_xname(self);
224 sc->sc_ev_reset.ev_code = PSWITCH_EVENT_PRESSED;
225 sysmon_pswitch_register(&sc->sc_ev_reset.ev_smpsw);
226 sc->sc_ev_sleep.ev_smpsw.smpsw_type = PSWITCH_TYPE_SLEEP;
227 sc->sc_ev_sleep.ev_smpsw.smpsw_name = device_xname(self);
228 sc->sc_ev_sleep.ev_code = PSWITCH_EVENT_RELEASED;
229 sysmon_pswitch_register(&sc->sc_ev_sleep.ev_smpsw);
230 sc->sc_smpsw_valid = true;
231
232 callout_setfunc(&sc->sc_tick, vmt_tick, sc);
233 callout_schedule(&sc->sc_tick, hz);
234
235 callout_setfunc(&sc->sc_tclo_tick, vmt_tclo_tick, sc);
236 callout_schedule(&sc->sc_tclo_tick, hz);
237 sc->sc_tclo_ping = 1;
238
239 callout_setfunc(&sc->sc_clock_sync_tick, vmt_clock_sync_tick, sc);
240 callout_schedule(&sc->sc_clock_sync_tick,
241 mstohz(sc->sc_clock_sync_period_seconds * 1000));
242
243 vmt_sync_guest_clock(sc);
244
245 return;
246
247free:
248 if (sc->sc_rpc_buf)
249 kmem_free(sc->sc_rpc_buf, VMT_RPC_BUFLEN);
250 pmf_device_register(self, NULL, NULL);
251 if (sc->sc_log)
252 sysctl_teardown(&sc->sc_log);
253}

255static int
256vmt_detach(device_t self, int flags)
257{
258 struct vmt_softc *sc = device_private(self);
259
260 if (sc->sc_tclo_rpc_open)
261 vm_rpc_close(&sc->sc_tclo_rpc);
262
263 if (sc->sc_smpsw_valid) {
264 sysmon_pswitch_unregister(&sc->sc_ev_sleep.ev_smpsw);
265 sysmon_pswitch_unregister(&sc->sc_ev_reset.ev_smpsw);
266 sysmon_pswitch_unregister(&sc->sc_ev_power.ev_smpsw);
267 }
268
269 callout_halt(&sc->sc_tick, NULL);
270 callout_destroy(&sc->sc_tick);
271
272 callout_halt(&sc->sc_tclo_tick, NULL);
273 callout_destroy(&sc->sc_tclo_tick);
274
275 callout_halt(&sc->sc_clock_sync_tick, NULL);
276 callout_destroy(&sc->sc_clock_sync_tick);
277
278 if (sc->sc_rpc_buf)
279 kmem_free(sc->sc_rpc_buf, VMT_RPC_BUFLEN);
280
281 if (sc->sc_log) {
282 sysctl_teardown(&sc->sc_log);
283 sc->sc_log = NULL;
284 }
285
286 return 0;
287}

289static int
290vmt_sysctl_setup_root(device_t self)
291{
292 const struct sysctlnode *machdep_node, *vmt_node;
293 struct vmt_softc *sc = device_private(self);
294 int rv;
295
296 rv = sysctl_createv(&sc->sc_log, 0, NULL, &machdep_node,
297 CTLFLAG_PERMANENT, CTLTYPE_NODE, "machdep", NULL,
298 NULL, 0, NULL, 0, CTL_MACHDEP, CTL_EOL);
299 if (rv != 0)
300 goto fail;
301
302 rv = sysctl_createv(&sc->sc_log, 0, &machdep_node, &vmt_node,
303 0, CTLTYPE_NODE, device_xname(self), NULL,
304 NULL, 0, NULL, 0, CTL_CREATE, CTL_EOL);
305 if (rv != 0)
306 goto fail;
307
308 rv = vmt_sysctl_setup_clock_sync(self, vmt_node);
309 if (rv != 0)
310 goto fail;
311
312 return 0;
313
314fail:
315 sysctl_teardown(&sc->sc_log);
316 sc->sc_log = NULL;
317
318 return rv;
319}

321static int
322vmt_sysctl_setup_clock_sync(device_t self, const struct sysctlnode *root_node)
323{
324 const struct sysctlnode *node, *period_node;
325 struct vmt_softc *sc = device_private(self);
326 int rv;
327
328 rv = sysctl_createv(&sc->sc_log, 0, &root_node, &node,
329 0, CTLTYPE_NODE, "clock_sync", NULL,
330 NULL, 0, NULL, 0, CTL_CREATE, CTL_EOL);
331 if (rv != 0)
332 return rv;
333
334 rv = sysctl_createv(&sc->sc_log, 0, &node, &period_node,
335 CTLFLAG_READWRITE, CTLTYPE_INT, "period",
336 SYSCTL_DESCR("Period, in seconds, at which to update the "
337 "guest's clock"),
338 vmt_sysctl_update_clock_sync_period, 0, (void *)sc, 0,
339 CTL_CREATE, CTL_EOL);
340 return rv;
341}

343static int
344vmt_sysctl_update_clock_sync_period(SYSCTLFN_ARGS)
345{
346 int error, period;
347 struct sysctlnode node;
348 struct vmt_softc *sc;
349
350 node = *rnode;
351 sc = (struct vmt_softc *)node.sysctl_data;
352
353 period = sc->sc_clock_sync_period_seconds;
354 node.sysctl_data = &period;
355 error = sysctl_lookup(SYSCTLFN_CALL(&node));
356 if (error || newp == NULL)
357 return error;
358
359 if (sc->sc_clock_sync_period_seconds != period) {
360 callout_halt(&sc->sc_clock_sync_tick, NULL);
361 sc->sc_clock_sync_period_seconds = period;
362 if (sc->sc_clock_sync_period_seconds > 0)
363 callout_schedule(&sc->sc_clock_sync_tick,
364 mstohz(sc->sc_clock_sync_period_seconds * 1000));
365 }
366 return 0;
367}

380static void
381vmt_update_guest_uptime(struct vmt_softc *sc)
382{
383 /* host wants uptime in hundredths of a second */
384 if (vm_rpc_send_rpci_tx(sc, "SetGuestInfo %d %" PRId64 "00",
385 VM_GUEST_INFO_UPTIME, time_uptime) != 0) {
386 device_printf(sc->sc_dev, "unable to set guest uptime\n");
387 sc->sc_rpc_error = 1;
388 }
389}

391static void
392vmt_update_guest_info(struct vmt_softc *sc)
393{
394 if (strncmp(sc->sc_hostname, hostname, sizeof(sc->sc_hostname)) != 0) {
395 strlcpy(sc->sc_hostname, hostname, sizeof(sc->sc_hostname));
396
397 if (vm_rpc_send_rpci_tx(sc, "SetGuestInfo %d %s",
398 VM_GUEST_INFO_DNS_NAME, sc->sc_hostname) != 0) {
399 device_printf(sc->sc_dev, "unable to set hostname\n");
400 sc->sc_rpc_error = 1;
401 }
402 }
403
404 /*
405 * we're supposed to pass the full network address information back here,
406 * but that involves xdr (sunrpc) data encoding, which seems a bit unreasonable.
407 */
408
409 if (sc->sc_set_guest_os == 0) {
410 if (vm_rpc_send_rpci_tx(sc, "SetGuestInfo %d %s %s %s",
411 VM_GUEST_INFO_OS_NAME_FULL, ostype, osrelease, machine_arch) != 0) {
412 device_printf(sc->sc_dev, "unable to set full guest OS\n");
413 sc->sc_rpc_error = 1;
414 }
415
416 /*
417 * host doesn't like it if we send an OS name it doesn't recognise,
418 * so use "other" for i386 and "other-64" for amd64
419 */
420 if (vm_rpc_send_rpci_tx(sc, "SetGuestInfo %d %s",
421 VM_GUEST_INFO_OS_NAME, VM_OS_NAME) != 0) {
422 device_printf(sc->sc_dev, "unable to set guest OS\n");
423 sc->sc_rpc_error = 1;
424 }
425
426 sc->sc_set_guest_os = 1;
427 }
428}

430static void
431vmt_sync_guest_clock(struct vmt_softc *sc)
432{
433 struct vm_backdoor frame;
434 struct timespec ts;
435
436 memset(&frame, 0, sizeof(frame));
437 frame.eax.word = VM_MAGIC;
438 frame.ecx.part.low = VM_CMD_GET_TIME_FULL;
439 frame.edx.part.low = VM_PORT_CMD;
440 vm_cmd(&frame);
441
442 if (frame.eax.word != 0xffffffff) {
443 ts.tv_sec = ((uint64_t)frame.esi.word << 32) | frame.edx.word;
444 ts.tv_nsec = frame.ebx.word * 1000;
445 tc_setclock(&ts);
446 }
447}

490static void
491vmt_do_resume(struct vmt_softc *sc)
492{
493 device_printf(sc->sc_dev, "guest resuming from suspended state\n");
494
495 vmt_sync_guest_clock(sc);
496
497 /* force guest info update */
498 sc->sc_hostname[0] = '\0';
499 sc->sc_set_guest_os = 0;
500 vmt_update_guest_info(sc);
501
502 vmt_tclo_state_change_success(sc, 1, VM_STATE_CHANGE_RESUME);
503 if (vm_rpc_send_str(&sc->sc_tclo_rpc, VM_RPC_REPLY_OK) != 0) {
504 device_printf(sc->sc_dev, "error sending resume response\n");
505 sc->sc_rpc_error = 1;
506 }
507
508 sysmon_task_queue_sched(0, vmt_pswitch_event, &sc->sc_ev_sleep);
509}

513{

525

527}

537static void
538vmt_tclo_tick(void *xarg)
539{
540 struct vmt_softc *sc = xarg;
541 u_int32_t rlen;
542 u_int16_t ack;
543
544 /* reopen tclo channel if it's currently closed */
545 if (sc->sc_tclo_rpc.channel == 0 &&
546 sc->sc_tclo_rpc.cookie1 == 0 &&
547 sc->sc_tclo_rpc.cookie2 == 0) {
548 if (vm_rpc_open(&sc->sc_tclo_rpc, VM_RPC_OPEN_TCLO) != 0) {
549 device_printf(sc->sc_dev, "unable to reopen TCLO channel\n");
550 callout_schedule(&sc->sc_tclo_tick, hz * 15);
551 return;
552 }
553
554 if (vm_rpc_send_str(&sc->sc_tclo_rpc, VM_RPC_RESET_REPLY) != 0) {
555 device_printf(sc->sc_dev, "failed to send reset reply\n");
556 sc->sc_rpc_error = 1;
557 goto out;
558 } else {
559 sc->sc_rpc_error = 0;
560 }
561 }
562
563 if (sc->sc_tclo_ping) {
564 if (vm_rpc_send(&sc->sc_tclo_rpc, NULL, 0) != 0) {
565 device_printf(sc->sc_dev, "failed to send TCLO outgoing ping\n");
566 sc->sc_rpc_error = 1;
567 goto out;
568 }
569 }
570
571 if (vm_rpc_get_length(&sc->sc_tclo_rpc, &rlen, &ack) != 0) {
572 device_printf(sc->sc_dev, "failed to get length of incoming TCLO data\n");
573 sc->sc_rpc_error = 1;
574 goto out;
575 }
576
577 if (rlen == 0) {
578 sc->sc_tclo_ping = 1;
579 goto out;
580 }
581
582 if (rlen >= VMT_RPC_BUFLEN) {
583 rlen = VMT_RPC_BUFLEN - 1;
584 }
585 if (vm_rpc_get_data(&sc->sc_tclo_rpc, sc->sc_rpc_buf, rlen, ack) != 0) {
586 device_printf(sc->sc_dev, "failed to get incoming TCLO data\n");
587 sc->sc_rpc_error = 1;
588 goto out;
589 }
590 sc->sc_tclo_ping = 0;
591
592#ifdef VMT_DEBUG
593 printf("vmware: received message '%s'\n", sc->sc_rpc_buf);
594#endif
595
596 if (strcmp(sc->sc_rpc_buf, "reset") == 0) {
597
598 if (sc->sc_rpc_error != 0) {
599 device_printf(sc->sc_dev, "resetting rpc\n");
600 vm_rpc_close(&sc->sc_tclo_rpc);
601 /* reopen and send the reset reply next time around */
602 goto out;
603 }
604
605 if (vm_rpc_send_str(&sc->sc_tclo_rpc, VM_RPC_RESET_REPLY) != 0) {
606 device_printf(sc->sc_dev, "failed to send reset reply\n");
607 sc->sc_rpc_error = 1;
608 }
609
610 } else if (strcmp(sc->sc_rpc_buf, "ping") == 0) {
611
612 vmt_update_guest_info(sc);
613 if (vm_rpc_send_str(&sc->sc_tclo_rpc, VM_RPC_REPLY_OK) != 0) {
614 device_printf(sc->sc_dev, "error sending ping response\n");
615 sc->sc_rpc_error = 1;
616 }
617
618 } else if (strcmp(sc->sc_rpc_buf, "OS_Halt") == 0) {
619 vmt_do_shutdown(sc);
620 } else if (strcmp(sc->sc_rpc_buf, "OS_Reboot") == 0) {
621 vmt_do_reboot(sc);
622 } else if (strcmp(sc->sc_rpc_buf, "OS_PowerOn") == 0) {
623 vmt_tclo_state_change_success(sc, 1, VM_STATE_CHANGE_POWERON);
624 if (vm_rpc_send_str(&sc->sc_tclo_rpc, VM_RPC_REPLY_OK) != 0) {
625 device_printf(sc->sc_dev, "error sending poweron response\n");
626 sc->sc_rpc_error = 1;
627 }
628 } else if (strcmp(sc->sc_rpc_buf, "OS_Suspend") == 0) {
629 log(LOG_KERN | LOG_NOTICE, "VMware guest entering suspended state\n");
630

665 if (vm_rpc_send_str(&sc->sc_tclo_rpc, VM_RPC_REPLY_OK) != 0) {
666 device_printf(sc->sc_dev, "error sending capabilities_register response\n");
667 sc->sc_rpc_error = 1;

714 device_printf(sc->sc_dev,
715 "error sending broadcastIP error response\n");

721 device_printf(sc->sc_dev, "error sending unknown command reply\n");
722 sc->sc_rpc_error = 1;
723 }
724 }

726out:
727 callout_schedule(&sc->sc_tclo_tick, sc->sc_tclo_ping ? hz : 1);

748static int
749vm_rpc_open(struct vm_rpc *rpc, uint32_t proto)
750{
751 struct vm_backdoor frame;
752
753 memset(&frame, 0, sizeof(frame));
754 frame.eax.word = VM_MAGIC;
755 frame.ebx.word = proto | VM_RPC_FLAG_COOKIE;
756 frame.ecx.part.low = VM_CMD_RPC;
757 frame.ecx.part.high = VM_RPC_OPEN;
758 frame.edx.part.low = VM_PORT_CMD;
759 frame.edx.part.high = 0;
760
761 vm_cmd(&frame);
762
763 if (frame.ecx.part.high != 1 || frame.edx.part.low != 0) {
764 /* open-vm-tools retries without VM_RPC_FLAG_COOKIE here.. */
765 printf("vmware: open failed, eax=%08x, ecx=%08x, edx=%08x\n",
766 frame.eax.word, frame.ecx.word, frame.edx.word);
767 return EIO;
768 }
769
770 rpc->channel = frame.edx.part.high;
771 rpc->cookie1 = frame.esi.word;
772 rpc->cookie2 = frame.edi.word;
773
774 return 0;
775}

777static int
778vm_rpc_close(struct vm_rpc *rpc)
779{
780 struct vm_backdoor frame;
781
782 memset(&frame, 0, sizeof(frame));
783 frame.eax.word = VM_MAGIC;
784 frame.ebx.word = 0;
785 frame.ecx.part.low = VM_CMD_RPC;
786 frame.ecx.part.high = VM_RPC_CLOSE;
787 frame.edx.part.low = VM_PORT_CMD;
788 frame.edx.part.high = rpc->channel;
789 frame.edi.word = rpc->cookie2;
790 frame.esi.word = rpc->cookie1;
791
792 vm_cmd(&frame);
793
794 if (frame.ecx.part.high == 0 || frame.ecx.part.low != 0) {
795 printf("vmware: close failed, eax=%08x, ecx=%08x\n",
796 frame.eax.word, frame.ecx.word);
797 return EIO;
798 }
799
800 rpc->channel = 0;
801 rpc->cookie1 = 0;
802 rpc->cookie2 = 0;
803
804 return 0;

807static int
808vm_rpc_send(const struct vm_rpc *rpc, const uint8_t *buf, uint32_t length)
809{
810 struct vm_backdoor frame;
811
812 /* Send the length of the command. */
813 memset(&frame, 0, sizeof(frame));
814 frame.eax.word = VM_MAGIC;
815 frame.ebx.word = length;
816 frame.ecx.part.low = VM_CMD_RPC;
817 frame.ecx.part.high = VM_RPC_SET_LENGTH;
818 frame.edx.part.low = VM_PORT_CMD;
819 frame.edx.part.high = rpc->channel;
820 frame.esi.word = rpc->cookie1;
821 frame.edi.word = rpc->cookie2;
822
823 vm_cmd(&frame);
824
825 if ((frame.ecx.part.high & VM_RPC_REPLY_SUCCESS) == 0) {
826 printf("vmware: sending length failed, eax=%08x, ecx=%08x\n",
827 frame.eax.word, frame.ecx.word);
828 return EIO;
829 }
830
831 if (length == 0)
832 return 0; /* Only need to poke once if command is null. */
833
834 /* Send the command using enhanced RPC. */
835 memset(&frame, 0, sizeof(frame));
836 frame.eax.word = VM_MAGIC;
837 frame.ebx.word = VM_RPC_ENH_DATA;
838 frame.ecx.word = length;
839 frame.edx.part.low = VM_PORT_RPC;
840 frame.edx.part.high = rpc->channel;
841 frame.ebp.word = rpc->cookie1;
842 frame.edi.word = rpc->cookie2;
843#ifdef __amd64__
844 frame.esi.quad = (uint64_t)buf;
845#else
846 frame.esi.word = (uint32_t)buf;
847#endif
848
849 vm_outs(&frame);
850
851 if (frame.ebx.word != VM_RPC_ENH_DATA) {
852 /* open-vm-tools retries on VM_RPC_REPLY_CHECKPOINT */
853 printf("vmware: send failed, ebx=%08x\n", frame.ebx.word);
854 return EIO;
855 }
856
857 return 0;
858}

866static int
867vm_rpc_get_data(const struct vm_rpc *rpc, char *data, uint32_t length,
868 uint16_t dataid)
869{
870 struct vm_backdoor frame;
871
872 /* Get data using enhanced RPC. */
873 memset(&frame, 0, sizeof(frame));
874 frame.eax.word = VM_MAGIC;
875 frame.ebx.word = VM_RPC_ENH_DATA;
876 frame.ecx.word = length;
877 frame.edx.part.low = VM_PORT_RPC;
878 frame.edx.part.high = rpc->channel;
879 frame.esi.word = rpc->cookie1;
880#ifdef __amd64__
881 frame.edi.quad = (uint64_t)data;
882#else
883 frame.edi.word = (uint32_t)data;
884#endif
885 frame.ebp.word = rpc->cookie2;
886
887 vm_ins(&frame);
888
889 /* NUL-terminate the data */
890 data[length] = '\0';
891
892 if (frame.ebx.word != VM_RPC_ENH_DATA) {
893 printf("vmware: get data failed, ebx=%08x\n",
894 frame.ebx.word);
895 return EIO;
896 }
897
898 /* Acknowledge data received. */
899 memset(&frame, 0, sizeof(frame));
900 frame.eax.word = VM_MAGIC;
901 frame.ebx.word = dataid;
902 frame.ecx.part.low = VM_CMD_RPC;
903 frame.ecx.part.high = VM_RPC_GET_END;
904 frame.edx.part.low = VM_PORT_CMD;
905 frame.edx.part.high = rpc->channel;
906 frame.esi.word = rpc->cookie1;
907 frame.edi.word = rpc->cookie2;
908
909 vm_cmd(&frame);
910
911 if (frame.ecx.part.high == 0) {
912 printf("vmware: ack data failed, eax=%08x, ecx=%08x\n",
913 frame.eax.word, frame.ecx.word);
914 return EIO;
915 }
916
917 return 0;
918}

922{
923 struct vm_backdoor frame;

925 memset(&frame, 0, sizeof(frame));
926 frame.eax.word = VM_MAGIC;
927 frame.ebx.word = 0;
928 frame.ecx.part.low = VM_CMD_RPC;
929 frame.ecx.part.high = VM_RPC_GET_LENGTH;
930 frame.edx.part.low = VM_PORT_CMD;
931 frame.edx.part.high = rpc->channel;
932 frame.esi.word = rpc->cookie1;
933 frame.edi.word = rpc->cookie2;
934

959static int
960vm_rpc_send_rpci_tx_buf(struct vmt_softc *sc, const uint8_t *buf, uint32_t length)
961{
962 struct vm_rpc rpci;
963 u_int32_t rlen;
964 u_int16_t ack;
965 int result = 0;
966
967 if (vm_rpc_open(&rpci, VM_RPC_OPEN_RPCI) != 0) {
968 device_printf(sc->sc_dev, "rpci channel open failed\n");
969 return EIO;
970 }
971
972 if (vm_rpc_send(&rpci, sc->sc_rpc_buf, length) != 0) {
973 device_printf(sc->sc_dev, "unable to send rpci command\n");
974 result = EIO;
975 goto out;
976 }
977
978 if (vm_rpc_get_length(&rpci, &rlen, &ack) != 0) {
979 device_printf(sc->sc_dev, "failed to get length of rpci response data\n");
980 result = EIO;
981 goto out;
982 }
983
984 if (rlen > 0) {
985 if (rlen >= VMT_RPC_BUFLEN) {
986 rlen = VMT_RPC_BUFLEN - 1;
987 }
988
989 if (vm_rpc_get_data(&rpci, sc->sc_rpc_buf, rlen, ack) != 0) {
990 device_printf(sc->sc_dev, "failed to get rpci response data\n");
991 result = EIO;
992 goto out;
993 }
994 }
995
996out:
997 if (vm_rpc_close(&rpci) != 0) {
998 device_printf(sc->sc_dev, "unable to close rpci channel\n");
999 }
1000
1001 return result;
1002}
