22/* OS name to report to host */
23#ifdef __i386__
24#define VM_OS_NAME "other"
25#else
26#define VM_OS_NAME "other-64"
27#endif
28
29/* "The" magic number, always occupies the EAX register. */
30#define VM_MAGIC 0x564D5868
31
32/* Port numbers, passed on EDX.LOW . */
33#define VM_PORT_CMD 0x5658
34#define VM_PORT_RPC 0x5659
35
36/* Commands, passed on ECX.LOW. */
37#define VM_CMD_GET_SPEED 0x01
38#define VM_CMD_APM 0x02
39#define VM_CMD_GET_MOUSEPOS 0x04
40#define VM_CMD_SET_MOUSEPOS 0x05
41#define VM_CMD_GET_CLIPBOARD_LEN 0x06
42#define VM_CMD_GET_CLIPBOARD 0x07
43#define VM_CMD_SET_CLIPBOARD_LEN 0x08
44#define VM_CMD_SET_CLIPBOARD 0x09
45#define VM_CMD_GET_VERSION 0x0a
46#define VM_VERSION_UNMANAGED 0x7fffffff
47#define VM_CMD_GET_DEVINFO 0x0b
48#define VM_CMD_DEV_ADDREMOVE 0x0c
49#define VM_CMD_GET_GUI_OPTIONS 0x0d
50#define VM_CMD_SET_GUI_OPTIONS 0x0e
51#define VM_CMD_GET_SCREEN_SIZE 0x0f
52#define VM_CMD_GET_HWVER 0x11
53#define VM_CMD_POPUP_OSNOTFOUND 0x12
54#define VM_CMD_GET_BIOS_UUID 0x13
55#define VM_CMD_GET_MEM_SIZE 0x14
56/*#define VM_CMD_GET_TIME 0x17 */ /* deprecated */
57#define VM_CMD_RPC 0x1e
58#define VM_CMD_GET_HZ 0x2d
59#define VM_CMD_GET_TIME_FULL 0x2e
60#define VM_CMD_GET_VCPU_INFO 0x44
61#define VCPUINFO_LEGACY_X2APIC __BIT(3)
62#define VCPUINFO_VCPU_RESERVED __BIT(31)
63
64/* RPC sub-commands, passed on ECX.HIGH. */
65#define VM_RPC_OPEN 0x00
66#define VM_RPC_SET_LENGTH 0x01
67#define VM_RPC_SET_DATA 0x02
68#define VM_RPC_GET_LENGTH 0x03
69#define VM_RPC_GET_DATA 0x04
70#define VM_RPC_GET_END 0x05
71#define VM_RPC_CLOSE 0x06
72
73/* RPC magic numbers, passed on EBX. */
74#define VM_RPC_OPEN_RPCI 0x49435052UL /* with VM_RPC_OPEN. */
75#define VM_RPC_OPEN_TCLO 0x4F4C4354UL /* with VP_RPC_OPEN. */
76#define VM_RPC_ENH_DATA 0x00010000UL /* with enhanced RPC data calls. */
77
78#define VM_RPC_FLAG_COOKIE 0x80000000UL
79
80/* RPC reply flags */
81#define VM_RPC_REPLY_SUCCESS 0x0001
82#define VM_RPC_REPLY_DORECV 0x0002 /* incoming message available */
83#define VM_RPC_REPLY_CLOSED 0x0004 /* RPC channel is closed */
84#define VM_RPC_REPLY_UNSENT 0x0008 /* incoming message was removed? */
85#define VM_RPC_REPLY_CHECKPOINT 0x0010 /* checkpoint occurred -> retry */
86#define VM_RPC_REPLY_POWEROFF 0x0020 /* underlying device is powering off */
87#define VM_RPC_REPLY_TIMEOUT 0x0040
88#define VM_RPC_REPLY_HB 0x0080 /* high-bandwidth tx/rx available */
89
90/* VM state change IDs */
91#define VM_STATE_CHANGE_HALT 1
92#define VM_STATE_CHANGE_REBOOT 2
93#define VM_STATE_CHANGE_POWERON 3
94#define VM_STATE_CHANGE_RESUME 4
95#define VM_STATE_CHANGE_SUSPEND 5
96
97/* VM guest info keys */
98#define VM_GUEST_INFO_DNS_NAME 1
99#define VM_GUEST_INFO_IP_ADDRESS 2
100#define VM_GUEST_INFO_DISK_FREE_SPACE 3
101#define VM_GUEST_INFO_BUILD_NUMBER 4
102#define VM_GUEST_INFO_OS_NAME_FULL 5
103#define VM_GUEST_INFO_OS_NAME 6
104#define VM_GUEST_INFO_UPTIME 7
105#define VM_GUEST_INFO_MEMORY 8
106#define VM_GUEST_INFO_IP_ADDRESS_V2 9
107
108/* RPC responses */

111#define VM_RPC_REPLY_ERROR "ERROR Unknown command"
