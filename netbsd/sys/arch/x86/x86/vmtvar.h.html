22/* A register. */
23union vm_reg {
24 struct {
25 uint16_t low;
26 uint16_t high;
27 } part;
28 uint32_t word;
29#ifdef __amd64__
30 struct {
31 uint32_t low;
32 uint32_t high;
33 } words;
34 uint64_t quad;
35#endif
36} __packed;
37
38/* A register frame. */
39/* XXX 'volatile' as a workaround because BACKDOOR_OP is likely broken */
40struct vm_backdoor {
41 volatile union vm_reg eax;
42 volatile union vm_reg ebx;
43 volatile union vm_reg ecx;
44 volatile union vm_reg edx;
45 volatile union vm_reg esi;
46 volatile union vm_reg edi;
47 volatile union vm_reg ebp;
48} __packed;
49
50/* RPC context. */
51struct vm_rpc {
52 uint16_t channel;
53 uint32_t cookie1;
54 uint32_t cookie2;
55};
56
57#define BACKDOOR_OP_I386(op, frame) \
58 __asm__ __volatile__ ( \
59 "pushal;" \
60 "pushl %%eax;" \
61 "movl 0x18(%%eax), %%ebp;" \
62 "movl 0x14(%%eax), %%edi;" \
63 "movl 0x10(%%eax), %%esi;" \
64 "movl 0x0c(%%eax), %%edx;" \
65 "movl 0x08(%%eax), %%ecx;" \
66 "movl 0x04(%%eax), %%ebx;" \
67 "movl 0x00(%%eax), %%eax;" \
68 op \
69 "xchgl %%eax, 0x00(%%esp);" \
70 "movl %%ebp, 0x18(%%eax);" \
71 "movl %%edi, 0x14(%%eax);" \
72 "movl %%esi, 0x10(%%eax);" \
73 "movl %%edx, 0x0c(%%eax);" \
74 "movl %%ecx, 0x08(%%eax);" \
75 "movl %%ebx, 0x04(%%eax);" \
76 "popl 0x00(%%eax);" \
77 "popal;" \
78 : \
79 :"a"(frame) \
80 )
81
82#define BACKDOOR_OP_AMD64(op, frame) \
83 __asm__ __volatile__ ( \
84 "pushq %%rbp; \n\t" \
85 "pushq %%rax; \n\t" \
86 "movq 0x30(%%rax), %%rbp; \n\t" \
87 "movq 0x28(%%rax), %%rdi; \n\t" \
88 "movq 0x20(%%rax), %%rsi; \n\t" \
89 "movq 0x18(%%rax), %%rdx; \n\t" \
90 "movq 0x10(%%rax), %%rcx; \n\t" \
91 "movq 0x08(%%rax), %%rbx; \n\t" \
92 "movq 0x00(%%rax), %%rax; \n\t" \
93 op "\n\t" \
94 "xchgq %%rax, 0x00(%%rsp); \n\t" \
95 "movq %%rbp, 0x30(%%rax); \n\t" \
96 "movq %%rdi, 0x28(%%rax); \n\t" \
97 "movq %%rsi, 0x20(%%rax); \n\t" \
98 "movq %%rdx, 0x18(%%rax); \n\t" \
99 "movq %%rcx, 0x10(%%rax); \n\t" \
100 "movq %%rbx, 0x08(%%rax); \n\t" \
101 "popq 0x00(%%rax); \n\t" \
102 "popq %%rbp; \n\t" \
103 : /* No outputs. */ \
104 : "a" (frame) \
105 /* No pushal on amd64 so warn gcc about the clobbered registers. */\
106 : "rbx", "rcx", "rdx", "rdi", "rsi", "cc", "memory" \
107 )
108
109#ifdef __i386__
110#define BACKDOOR_OP(op, frame) BACKDOOR_OP_I386(op, frame)
111#else
112#define BACKDOOR_OP(op, frame) BACKDOOR_OP_AMD64(op, frame)
113#endif

115static __inline void
116vmt_hvcall(uint8_t cmd, u_int regs[6])
117{
118 struct vm_backdoor frame;
119
120 memset(&frame, 0, sizeof(frame));
121 frame.eax.word = VM_MAGIC;
122 frame.ebx.word = UINT_MAX;
123 frame.ecx.part.low = cmd;
124 frame.edx.part.low = VM_PORT_CMD;
125
126 BACKDOOR_OP("inl %%dx, %%eax;", &frame);
127
128 regs[0] = frame.eax.word;
129 regs[1] = frame.ebx.word;
130 regs[2] = frame.ecx.word;
131 regs[3] = frame.edx.word;
132 regs[4] = frame.esi.word;
133 regs[5] = frame.edi.word;
134}
