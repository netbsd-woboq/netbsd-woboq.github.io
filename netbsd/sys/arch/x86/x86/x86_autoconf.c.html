82static void
83dmatch(const char *func, device_t dv, const char *method)
84{
85
86 printf("WARNING: %s: double match for boot device (%s:%s %s:%s)\n",
87 func, booted_method, device_xname(booted_device),
88 method, device_xname(dv));
89}

91static int
92is_valid_disk(device_t dv)
93{
94
95 if (device_class(dv) != DV_DISK)
96 return (0);
97
98 return (device_is_a(dv, "dk") ||
99 device_is_a(dv, "sd") ||
100 device_is_a(dv, "wd") ||
101 device_is_a(dv, "ld") ||
102 device_is_a(dv, "ed"));
103}

105/*
106 * XXX Ugly bit of code. But, this is the only safe time that the
107 * match between BIOS disks and native disks can be done.
108 */
109static void
110matchbiosdisks(void)
111{
112 struct btinfo_biosgeom *big;
113 struct bi_biosgeom_entry *be;
114 device_t dv;
115 deviter_t di;
116 int i, ck, error, m, n;
117 struct vnode *tv;
118 char mbr[DEV_BSIZE];
119 int dklist_size;
120 int numbig;
121
122 if (x86_ndisks)
123 return;
124 big = lookup_bootinfo(BTINFO_BIOSGEOM);
125
126 numbig = big ? big->num : 0;
127
128 /* First, count all native disks. */
129 for (dv = deviter_first(&di, DEVITER_F_ROOT_FIRST); dv != NULL;
130 dv = deviter_next(&di)) {
131 if (is_valid_disk(dv))
132 x86_ndisks++;
133 }
134 deviter_release(&di);
135
136 dklist_size = sizeof(struct disklist) + (x86_ndisks - 1) *
137 sizeof(struct nativedisk_info);
138
139 /* XXX M_TEMP is wrong */
140 x86_alldisks = malloc(dklist_size, M_TEMP, M_NOWAIT | M_ZERO);
141 if (x86_alldisks == NULL)
142 return;
143
144 x86_alldisks->dl_nnativedisks = x86_ndisks;
145 x86_alldisks->dl_nbiosdisks = numbig;
146 for (i = 0; i < numbig; i++) {
147 x86_alldisks->dl_biosdisks[i].bi_dev = big->disk[i].dev;
148 x86_alldisks->dl_biosdisks[i].bi_sec = big->disk[i].sec;
149 x86_alldisks->dl_biosdisks[i].bi_head = big->disk[i].head;
150 x86_alldisks->dl_biosdisks[i].bi_cyl = big->disk[i].cyl;
151 x86_alldisks->dl_biosdisks[i].bi_lbasecs = big->disk[i].totsec;
152 x86_alldisks->dl_biosdisks[i].bi_flags = big->disk[i].flags;
153 DPRINTF(("%s: disk %x: flags %x",
154 __func__, big->disk[i].dev, big->disk[i].flags));
155#ifdef BIOSDISK_EXTINFO_V3
156 DPRINTF((", interface %x, device %llx",
157 big->disk[i].interface_path, big->disk[i].device_path));
158#endif
159 DPRINTF(("\n"));
160 }
161
162 /* XXX Code duplication from findroot(). */
163 n = -1;
164 for (dv = deviter_first(&di, DEVITER_F_ROOT_FIRST); dv != NULL;
165 dv = deviter_next(&di)) {
166 if (!is_valid_disk(dv))
167 continue;
168 DPRINTF(("%s: trying to match (%s) %s: ", __func__,
169 device_xname(dv), device_cfdata(dv)->cf_name));
170 n++;
171 snprintf(x86_alldisks->dl_nativedisks[n].ni_devname,
172 sizeof(x86_alldisks->dl_nativedisks[n].ni_devname),
173 "%s", device_xname(dv));
174
175 if ((tv = opendisk(dv)) == NULL) {
176 DPRINTF(("cannot open\n"));
177 continue;
178 }
179
180 error = vn_rdwr(UIO_READ, tv, mbr, DEV_BSIZE, 0, UIO_SYSSPACE,
181 0, NOCRED, NULL, NULL);
182 VOP_CLOSE(tv, FREAD, NOCRED);
183 vput(tv);
184 if (error) {
185 DPRINTF(("MBR read failure %d\n", error));
186 continue;
187 }
188
189 for (ck = i = 0; i < DEV_BSIZE; i++)
190 ck += mbr[i];
191 for (m = i = 0; i < numbig; i++) {
192 be = &big->disk[i];
193 if (be->flags & BI_GEOM_INVALID)
194 continue;
195 DPRINTF(("matched with %d dev ck %x bios ck %x\n",
196 i, ck, be->cksum));
197 if (be->cksum == ck && memcmp(&mbr[MBR_PART_OFFSET],
198 be->mbrparts, MBR_PART_COUNT
199 * sizeof(struct mbr_partition)) == 0) {
200 DPRINTF(("%s: matched BIOS disk %x with %s\n",
201 __func__, be->dev, device_xname(dv)));
202 x86_alldisks->dl_nativedisks[n].
203 ni_biosmatches[m++] = i;
204 }
205 }
206 x86_alldisks->dl_nativedisks[n].ni_nmatches = m;
207 }
208 deviter_release(&di);
209}

211/*
212 * Helper function for findroot():
213 * Return non-zero if wedge device matches bootinfo.
214 */
215static int
216match_bootwedge(device_t dv, struct btinfo_bootwedge *biw)
217{
218 MD5_CTX ctx;
219 struct vnode *tmpvn;
220 int error;
221 uint8_t bf[DEV_BSIZE];
222 uint8_t hash[16];
223 int found = 0;
224 daddr_t blk;
225 uint64_t nblks;
226
227 /*
228 * If the boot loader didn't specify the sector, abort.
229 */
230 if (biw->matchblk == -1) {
231 DPRINTF(("%s: no sector specified for %s\n", __func__,
232 device_xname(dv)));
233 return 0;
234 }
235
236 if ((tmpvn = opendisk(dv)) == NULL) {
237 DPRINTF(("%s: can't open %s\n", __func__, device_xname(dv)));
238 return 0;
239 }
240
241 MD5Init(&ctx);
242 for (blk = biw->matchblk, nblks = biw->matchnblks;
243 nblks != 0; nblks--, blk++) {
244 error = vn_rdwr(UIO_READ, tmpvn, (void *) bf,
245 sizeof(bf), blk * DEV_BSIZE, UIO_SYSSPACE,
246 0, NOCRED, NULL, NULL);
247 if (error) {
248 if (error != EINVAL) {
249 aprint_error("%s: unable to read block %"
250 PRId64 " " "of dev %s (%d)\n", __func__,
251 blk, device_xname(dv), error);
252 }
253 goto closeout;
254 }
255 MD5Update(&ctx, bf, sizeof(bf));
256 }
257 MD5Final(hash, &ctx);
258
259 /* Compare with the provided hash. */
260 found = memcmp(biw->matchhash, hash, sizeof(hash)) == 0;
261 DPRINTF(("%s: %s found=%d\n", __func__, device_xname(dv), found));
262
263 closeout:
264 VOP_CLOSE(tmpvn, FREAD, NOCRED);
265 vput(tmpvn);
266 return found;
267}

269/*
270 * Helper function for findroot():
271 * Return non-zero if disk device matches bootinfo.
272 */
273static int
274match_bootdisk(device_t dv, struct btinfo_bootdisk *bid)
275{
276 struct vnode *tmpvn;
277 int error;
278 struct disklabel label;
279 int found = 0;
280
281 if (device_is_a(dv, "dk")) {
282 DPRINTF(("%s: dk %s\n", __func__, device_xname(dv)));
283 return 0;
284 }
285
286 /*
287 * A disklabel is required here. The boot loader doesn't refuse
288 * to boot from a disk without a label, but this is normally not
289 * wanted.
290 */
291 if (bid->labelsector == -1) {
292 DPRINTF(("%s: no label %s\n", __func__, device_xname(dv)));
293 return 0;
294 }
295
296 if ((tmpvn = opendisk(dv)) == NULL) {
297 DPRINTF(("%s: can't open %s\n", __func__, device_xname(dv)));
298 return 0;
299 }
300
301 error = VOP_IOCTL(tmpvn, DIOCGDINFO, &label, FREAD, NOCRED);
302 if (error) {
303 /*
304 * XXX Can't happen -- open() would have errored out
305 * or faked one up.
306 */
307 printf("%s: can't get label for dev %s (%d)\n", __func__,
308 device_xname(dv), error);
309 goto closeout;
310 }
311
312 /* Compare with our data. */
313 if (label.d_type == bid->label.type &&
314 label.d_checksum == bid->label.checksum &&
315 strncmp(label.d_packname, bid->label.packname, 16) == 0)
316 found = 1;
317
318 DPRINTF(("%s: %s found=%d\n", __func__, device_xname(dv), found));
319 closeout:
320 VOP_CLOSE(tmpvn, FREAD, NOCRED);
321 vput(tmpvn);
322 return (found);
323}

325/*
326 * Attempt to find the device from which we were booted. If we can do so,
327 * and not instructed not to do so, change rootdev to correspond to the
328 * load device.
329 */
330static void
331findroot(void)
332{
333 struct btinfo_rootdevice *biv;
334 struct btinfo_bootdisk *bid;
335 struct btinfo_bootwedge *biw;
336 struct btinfo_biosgeom *big;
337 device_t dv;
338 deviter_t di;
339 static char bootspecbuf[sizeof(biv->devname)+1];
340
341 if (booted_device)
342 return;
343
344 if (lookup_bootinfo(BTINFO_NETIF) != NULL) {
345 /*
346 * We got netboot interface information, but device_register()
347 * failed to match it to a configured device. Boot disk
348 * information cannot be present at the same time, so give
349 * up.
350 */
351 printf("%s: netboot interface not found.\n", __func__);
352 return;
353 }
354
355 if ((biv = lookup_bootinfo(BTINFO_ROOTDEVICE)) != NULL) {
356 for (dv = deviter_first(&di, DEVITER_F_ROOT_FIRST);
357 dv != NULL;
358 dv = deviter_next(&di)) {
359 cfdata_t cd;
360 size_t len;
361
362 if (device_class(dv) != DV_DISK)
363 continue;
364
365 cd = device_cfdata(dv);
366 len = strlen(cd->cf_name);
367
368 if (strncmp(cd->cf_name, biv->devname, len) == 0 &&
369 biv->devname[len] - '0' == device_unit(dv)) {
370 booted_device = dv;
371 booted_method = "bootinfo/rootdevice";
372 booted_partition = biv->devname[len + 1] - 'a';
373 booted_nblks = 0;
374 break;
375 }
376 }
377 DPRINTF(("%s: BTINFO_ROOTDEVICE %s\n", __func__,
378 booted_device ? device_xname(booted_device) : "not found"));
379 deviter_release(&di);
380 if (dv != NULL)
381 return;
382
383 if (biv->devname[0] != '\0') {
384 strlcpy(bootspecbuf, biv->devname, sizeof(bootspecbuf));
385 bootspec = bootspecbuf;
386 return;
387 }
388 }
389
390 bid = lookup_bootinfo(BTINFO_BOOTDISK);
391 biw = lookup_bootinfo(BTINFO_BOOTWEDGE);
392
393 if (biw != NULL) {
394 /*
395 * Scan all disk devices for ones that match the passed data.
396 * Don't break if one is found, to get possible multiple
397 * matches - for problem tracking. Use the first match anyway
398 * because lower devices numbers are more likely to be the
399 * boot device.
400 */
401 for (dv = deviter_first(&di, DEVITER_F_ROOT_FIRST);
402 dv != NULL;
403 dv = deviter_next(&di)) {
404 if (is_valid_disk(dv)) {
405 /*
406 * Don't trust BIOS device numbers, try
407 * to match the information passed by the
408 * boot loader instead.
409 */
410 if ((biw->biosdev & 0x80) == 0 ||
411 match_bootwedge(dv, biw) == 0)
412 continue;
413 goto bootwedge_found;
414 }
415
416 continue;
417 bootwedge_found:
418 if (booted_device) {
419 dmatch(__func__, dv, "bootinfo/bootwedge");
420 continue;
421 }
422 booted_device = dv;
423 booted_method = "bootinfo/bootwedge";
424 booted_partition = bid != NULL ? bid->partition : 0;
425 booted_nblks = biw->nblks;
426 booted_startblk = biw->startblk;
427 }
428 deviter_release(&di);
429
430 DPRINTF(("%s: BTINFO_BOOTWEDGE %s\n", __func__,
431 booted_device ? device_xname(booted_device) : "not found"));
432 if (booted_nblks)
433 return;
434 }
435
436 if (bid != NULL) {
437 /*
438 * Scan all disk devices for ones that match the passed data.
439 * Don't break if one is found, to get possible multiple
440 * matches - for problem tracking. Use the first match anyway
441 * because lower device numbers are more likely to be the
442 * boot device.
443 */
444 for (dv = deviter_first(&di, DEVITER_F_ROOT_FIRST);
445 dv != NULL;
446 dv = deviter_next(&di)) {
447
448 if (device_is_a(dv, "fd") &&
449 device_class(dv) == DV_DISK) {
450 /*
451 * Assume the configured unit number matches
452 * the BIOS device number. (This is the old
453 * behavior.) Needs some ideas how to handle
454 * the BIOS's "swap floppy drive" options.
455 */
456 /* XXX device_unit() abuse */
457 if ((bid->biosdev & 0x80) != 0 ||
458 device_unit(dv) != bid->biosdev)
459 continue;
460 goto bootdisk_found;
461 }
462
463 if (is_valid_disk(dv)) {
464 /*
465 * Don't trust BIOS device numbers, try
466 * to match the information passed by the
467 * boot loader instead.
468 */
469 if ((bid->biosdev & 0x80) == 0 ||
470 match_bootdisk(dv, bid) == 0)
471 continue;
472 goto bootdisk_found;
473 }
474
475 continue;
476 bootdisk_found:
477 if (booted_device) {
478 dmatch(__func__, dv, "bootinfo/bootdisk");
479 continue;
480 }
481 booted_device = dv;
482 booted_method = "bootinfo/bootdisk";
483 booted_partition = bid->partition;
484 booted_nblks = 0;
485 }
486 deviter_release(&di);
487
488 DPRINTF(("%s: BTINFO_BOOTDISK %s\n", __func__,
489 booted_device ? device_xname(booted_device) : "not found"));
490 if (booted_device)
491 return;
492
493 /*
494 * No booted device found; check CD-ROM boot at last.
495 *
496 * Our bootloader assumes CD-ROM boot if biosdev is larger
497 * or equal than the number of hard drives recognized by the
498 * BIOS. The number of drives can be found in BTINFO_BIOSGEOM
499 * here. For example, if we have wd0, wd1, and cd0:
500 *
501 * big->num = 2 (for wd0 and wd1)
502 * bid->biosdev = 0x80 (wd0)
503 * bid->biosdev = 0x81 (wd1)
504 * bid->biosdev = 0x82 (cd0)
505 *
506 * See src/sys/arch/i386/stand/boot/devopen.c and
507 * src/sys/arch/i386/stand/lib/bootinfo_biosgeom.c .
508 */
509 if ((big = lookup_bootinfo(BTINFO_BIOSGEOM)) != NULL &&
510 bid->biosdev >= 0x80 + big->num) {
511 /*
512 * XXX
513 * There is no proper way to detect which unit is
514 * recognized as a bootable CD-ROM drive by the BIOS.
515 * Assume the first unit is the one.
516 */
517 for (dv = deviter_first(&di, DEVITER_F_ROOT_FIRST);
518 dv != NULL;
519 dv = deviter_next(&di)) {
520 if (device_class(dv) == DV_DISK &&
521 device_is_a(dv, "cd")) {
522 booted_device = dv;
523 booted_method = "bootinfo/biosgeom";
524 booted_partition = 0;
525 booted_nblks = 0;
526 break;
527 }
528 }
529 deviter_release(&di);
530 DPRINTF(("%s: BTINFO_BIOSGEOM %s\n", __func__,
531 booted_device ? device_xname(booted_device) :
532 "not found"));
533 }
534 }

544void
545cpu_rootconf(void)
546{
547 cpu_bootconf();
548
549 aprint_normal("boot device: %s\n",
550 booted_device ? device_xname(booted_device) : "<unknown>");
551 rootconf();
552}

554void
555device_register(device_t dev, void *aux)
556{
557 device_t isaboot, pciboot;
558
559 /*
560 * The Intel Integrated Memory Controller has a built-in i2c
561 * controller that's rather limited in capability; it is intended
562 * only for reading memory module EERPOMs and sensors.
563 */
564 if (device_is_a(dev, "iic") &&
565 device_is_a(dev->dv_parent, "imcsmb")) {
566 static const char *imcsmb_device_whitelist[] = {
567 "spdmem",
568 "sdtemp",
569 NULL,
570 };
571 prop_array_t whitelist = prop_array_create();
572 prop_dictionary_t props = device_properties(dev);
573 int i;
574
575 for (i = 0; imcsmb_device_whitelist[i] != NULL; i++) {
576 prop_string_t pstr = prop_string_create_cstring_nocopy(
577 imcsmb_device_whitelist[i]);
578 (void) prop_array_add(whitelist, pstr);
579 prop_object_release(pstr);
580 }
581 (void) prop_dictionary_set(props,
582 I2C_PROP_INDIRECT_DEVICE_WHITELIST,
583 whitelist);
584 (void) prop_dictionary_set_cstring_nocopy(props,
585 I2C_PROP_INDIRECT_PROBE_STRATEGY,
586 I2C_PROBE_STRATEGY_NONE);
587 }
588
589 device_acpi_register(dev, aux);
590
591 isaboot = device_isa_register(dev, aux);
592 pciboot = device_pci_register(dev, aux);
593#if NHYPERV > 0
594 (void)device_hyperv_register(dev, aux);
595#endif
596
597 if (isaboot == NULL && pciboot == NULL)
598 return;
599
600 if (booted_device != NULL) {
601 /* XXX should be a panic() */
602 dmatch(__func__, dev, "device/register");
603 } else {
604 booted_device = (isaboot != NULL) ? isaboot : pciboot;
605 booted_method = "device/register";
606 }
