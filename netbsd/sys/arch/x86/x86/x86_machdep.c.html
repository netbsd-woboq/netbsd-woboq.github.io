127/*
128 * Given the type of a bootinfo entry, looks for a matching item inside
129 * the bootinfo structure. If found, returns a pointer to it (which must
130 * then be casted to the appropriate bootinfo_* type); otherwise, returns
131 * NULL.
132 */
133void *
134lookup_bootinfo(int type)
135{
136 bool found;
137 int i;
138 struct btinfo_common *bic;
139
140 bic = (struct btinfo_common *)(bootinfo.bi_data);
141 found = FALSE;
142 for (i = 0; i < bootinfo.bi_nentries && !found; i++) {
143 if (bic->type == type)
144 found = TRUE;
145 else
146 bic = (struct btinfo_common *)
147 ((uint8_t *)bic + bic->len);
148 }
149
150 return found ? bic : NULL;
151}

181/*
182 * mm_md_physacc: check if given pa is accessible.
183 */
184int
185mm_md_physacc(paddr_t pa, vm_prot_t prot)
186{
187 extern phys_ram_seg_t mem_clusters[VM_PHYSSEG_MAX];
188 extern int mem_cluster_cnt;
189 int i;
190
191 for (i = 0; i < mem_cluster_cnt; i++) {
192 const phys_ram_seg_t *seg = &mem_clusters[i];
193 paddr_t lstart = seg->start;
194
195 if (lstart <= pa && pa - lstart <= seg->size) {
196 return 0;
197 }
198 }
199 return kauth_authorize_machdep(kauth_cred_get(),
200 KAUTH_MACHDEP_UNMANAGEDMEM, NULL, NULL, NULL, NULL);
201}

204/*
205 * Push any modules loaded by the boot loader.
206 */
207void
208module_init_md(void)
209{
210 struct btinfo_modulelist *biml;
211 struct bi_modulelist_entry *bi, *bimax;
212
213 /* setup module path for XEN kernels */
214#ifdef XEN
215#ifdef __x86_64__
216 module_machine = module_machine_amd64_xen;
217#else
218 module_machine = module_machine_i386pae_xen;
219#endif
220#endif
221
222 biml = lookup_bootinfo(BTINFO_MODULELIST);
223 if (biml == NULL) {
224 aprint_debug("No module info at boot\n");
225 return;
226 }
227
228 bi = (struct bi_modulelist_entry *)((uint8_t *)biml + sizeof(*biml));
229 bimax = bi + biml->num;
230 for (; bi < bimax; bi++) {
231 switch (bi->type) {
232 case BI_MODULE_ELF:
233 aprint_debug("Prep module path=%s len=%d pa=%x\n",
234 bi->path, bi->len, bi->base);
235 KASSERT(trunc_page(bi->base) == bi->base);
236 module_prime(bi->path,
237#ifdef KASLR
238 (void *)PMAP_DIRECT_MAP((uintptr_t)bi->base),
239#else
240 (void *)((uintptr_t)bi->base + KERNBASE),
241#endif
242 bi->len);
243 break;
244 case BI_MODULE_IMAGE:
245#ifdef SPLASHSCREEN
246 aprint_debug("Splash image path=%s len=%d pa=%x\n",
247 bi->path, bi->len, bi->base);
248 KASSERT(trunc_page(bi->base) == bi->base);
249 splash_setimage(
250#ifdef KASLR
251 (void *)PMAP_DIRECT_MAP((uintptr_t)bi->base),
252#else
253 (void *)((uintptr_t)bi->base + KERNBASE),
254#endif
255 bi->len);
256#endif
257 break;
258 case BI_MODULE_RND:
259 aprint_debug("Random seed data path=%s len=%d pa=%x\n",
260 bi->path, bi->len, bi->base);
261 KASSERT(trunc_page(bi->base) == bi->base);
262 rnd_seed(
263#ifdef KASLR
264 (void *)PMAP_DIRECT_MAP((uintptr_t)bi->base),
265#else
266 (void *)((uintptr_t)bi->base + KERNBASE),
267#endif
268 bi->len);
269 break;
270 case BI_MODULE_FS:
271 aprint_debug("File-system image path=%s len=%d pa=%x\n",
272 bi->path, bi->len, bi->base);
273 KASSERT(trunc_page(bi->base) == bi->base);
274#if defined(MEMORY_DISK_HOOKS) && defined(MEMORY_DISK_DYNAMIC)
275 md_root_setconf(
276#ifdef KASLR
277 (void *)PMAP_DIRECT_MAP((uintptr_t)bi->base),
278#else
279 (void *)((uintptr_t)bi->base + KERNBASE),
280#endif
281 bi->len);
282#endif
283 break;
284 default:
285 aprint_debug("Skipping non-ELF module\n");
286 break;
287 }
288 }
289}

292void
293cpu_need_resched(struct cpu_info *ci, int flags)
294{
295 struct cpu_info *cur;
296 lwp_t *l;
297
298 KASSERT(kpreempt_disabled());
299 cur = curcpu();
300 l = ci->ci_data.cpu_onproc;
301 ci->ci_want_resched |= flags;
302
303 if (__predict_false((l->l_pflag & LP_INTR) != 0)) {
304 /*
305 * No point doing anything, it will switch soon.
306 * Also here to prevent an assertion failure in
307 * kpreempt() due to preemption being set on a
308 * soft interrupt LWP.
309 */
310 return;
311 }
312
313 if (l == ci->ci_data.cpu_idlelwp) {
314 if (ci == cur)
315 return;
316 if (x86_cpu_idle_ipi != false) {
317 cpu_kick(ci);
318 }
319 return;
320 }
321
322 if ((flags & RESCHED_KPREEMPT) != 0) {
323#ifdef __HAVE_PREEMPTION
324 atomic_or_uint(&l->l_dopreempt, DOPREEMPT_ACTIVE);
325 if (ci == cur) {
326 softint_trigger(1 << SIR_PREEMPT);
327 } else {
328 x86_send_ipi(ci, X86_IPI_KPREEMPT);
329 }
330 return;
331#endif
332 }
333
334 aston(l, X86_AST_PREEMPT);
335 if (ci == cur) {
336 return;
337 }
338 if ((flags & RESCHED_IMMED) != 0) {
339 cpu_kick(ci);
340 }
341}

366{
367 int idepth;
368
369 kpreempt_disable();

371 kpreempt_enable();

375#ifdef __HAVE_PREEMPTION
376/*
377 * Called to check MD conditions that would prevent preemption, and to
378 * arrange for those conditions to be rechecked later.
379 */
380bool
381cpu_kpreempt_enter(uintptr_t where, int s)
382{
383 struct pcb *pcb;
384 lwp_t *l;
385
386 KASSERT(kpreempt_disabled());
387 l = curlwp;
388
389 /*
390 * If SPL raised, can't go. Note this implies that spin
391 * mutexes at IPL_NONE are _not_ valid to use.
392 */
393 if (s > IPL_PREEMPT) {
394 softint_trigger(1 << SIR_PREEMPT);
395 aston(l, X86_AST_PREEMPT); /* paranoid */
396 return false;
397 }
398
399 /* Must save cr2 or it could be clobbered. */
400 pcb = lwp_getpcb(l);
401 pcb->pcb_cr2 = rcr2();
402
403 return true;
404}
405
406/*
407 * Called after returning from a kernel preemption, and called with
408 * preemption disabled.
409 */
410void
411cpu_kpreempt_exit(uintptr_t where)
412{
413 extern char x86_copyfunc_start, x86_copyfunc_end;
414 struct pcb *pcb;
415
416 KASSERT(kpreempt_disabled());
417
418 /*
419 * If we interrupted any of the copy functions we must reload
420 * the pmap when resuming, as they cannot tolerate it being
421 * swapped out.
422 */
423 if (where >= (uintptr_t)&x86_copyfunc_start &&
424 where < (uintptr_t)&x86_copyfunc_end) {
425 pmap_load();
426 }
427
428 /* Restore cr2 only after the pmap, as pmap_load can block. */
429 pcb = lwp_getpcb(curlwp);
430 lcr2(pcb->pcb_cr2);
431}

522int
523x86_select_freelist(uint64_t maxaddr)
524{
525 unsigned int i;
526
527 if (avail_end <= maxaddr)
528 return VM_NFREELIST;
529
530 for (i = 0; i < __arraycount(x86_freelists); i++) {
531 if ((x86_freelists[i].limit - 1) <= maxaddr)
532 return x86_freelists[i].freelist;
533 }
534
535 panic("no freelist for maximum address %"PRIx64, maxaddr);
536}

538static int
539x86_add_cluster(uint64_t seg_start, uint64_t seg_end, uint32_t type)
540{
541 extern struct extent *iomem_ex;
542 const uint64_t endext = MAXIOMEM + 1;
543 uint64_t new_physmem = 0;
544 phys_ram_seg_t *cluster;
545 int i;
546
547 if (seg_end > MAXPHYSMEM) {
548 aprint_verbose("WARNING: skipping large memory map entry: "
549 "0x%"PRIx64"/0x%"PRIx64"/0x%x\n",
550 seg_start, (seg_end - seg_start), type);
551 return 0;
552 }
553
554 /*
555 * XXX: Chop the last page off the size so that it can fit in avail_end.
556 */
557 if (seg_end == MAXPHYSMEM)
558 seg_end -= PAGE_SIZE;
559
560 if (seg_end <= seg_start)
561 return 0;
562
563 for (i = 0; i < mem_cluster_cnt; i++) {
564 cluster = &mem_clusters[i];
565 if ((cluster->start == round_page(seg_start)) &&
566 (cluster->size == trunc_page(seg_end) - cluster->start)) {
567#ifdef DEBUG_MEMLOAD
568 printf("WARNING: skipping duplicate segment entry\n");
569#endif
570 return 0;
571 }
572 }
573
574 /*
575 * This cluster is used by RAM. If it is included in the iomem extent,
576 * allocate it from there, so that we won't unintentionally reuse it
577 * later with extent_alloc_region. A way to avoid collision (with UVM
578 * for example).
579 *
580 * This is done before the addresses are page rounded just to make
581 * sure we get them all.
582 */
583 if (seg_start < endext) {
584 uint64_t io_end;
585
586 if (seg_end > endext)
587 io_end = endext;
588 else
589 io_end = seg_end;
590
591 if (iomem_ex != NULL && extent_alloc_region(iomem_ex, seg_start,
592 io_end - seg_start, EX_NOWAIT)) {
593 /* XXX What should we do? */
594 printf("WARNING: CAN't ALLOCATE MEMORY SEGMENT "
595 "(0x%"PRIx64"/0x%"PRIx64"/0x%x) FROM "
596 "IOMEM EXTENT MAP!\n",
597 seg_start, seg_end - seg_start, type);
598 return 0;
599 }
600 }
601
602 /* If it's not free memory, skip it. */
603 if (type != BIM_Memory)
604 return 0;
605
606 if (mem_cluster_cnt >= VM_PHYSSEG_MAX) {
607 printf("WARNING: too many memory segments"
608 "(increase VM_PHYSSEG_MAX)");
609 return -1;
610 }
611
612#ifdef PHYSMEM_MAX_ADDR
613 if (seg_start >= MBTOB(PHYSMEM_MAX_ADDR))
614 return 0;
615 if (seg_end > MBTOB(PHYSMEM_MAX_ADDR))
616 seg_end = MBTOB(PHYSMEM_MAX_ADDR);
617#endif
618
619 seg_start = round_page(seg_start);
620 seg_end = trunc_page(seg_end);
621
622 if (seg_start == seg_end)
623 return 0;
624
625 cluster = &mem_clusters[mem_cluster_cnt];
626 cluster->start = seg_start;
627 if (iomem_ex != NULL)
628 new_physmem = physmem + atop(seg_end - seg_start);
629
630#ifdef PHYSMEM_MAX_SIZE
631 if (iomem_ex != NULL) {
632 if (physmem >= atop(MBTOB(PHYSMEM_MAX_SIZE)))
633 return 0;
634 if (new_physmem > atop(MBTOB(PHYSMEM_MAX_SIZE))) {
635 seg_end = seg_start + MBTOB(PHYSMEM_MAX_SIZE) - ptoa(physmem);
636 new_physmem = atop(MBTOB(PHYSMEM_MAX_SIZE));
637 }
638 }
639#endif
640
641 cluster->size = seg_end - seg_start;
642
643 if (iomem_ex != NULL) {
644 if (avail_end < seg_end)
645 avail_end = seg_end;
646 physmem = new_physmem;
647 }
648 mem_cluster_cnt++;
649
650 return 0;
651}

653static int
654x86_parse_clusters(struct btinfo_memmap *bim)
655{
656 uint64_t seg_start, seg_end;
657 uint64_t addr, size;
658 uint32_t type;
659 int x;
660
661 KASSERT(bim != NULL);
662 KASSERT(bim->num > 0);
663
664#ifdef DEBUG_MEMLOAD
665 printf("MEMMAP: %s MEMORY MAP (%d ENTRIES):\n",
666 lookup_bootinfo(BTINFO_EFIMEMMAP) != NULL ? "UEFI" : "BIOS",
667 bim->num);
668#endif
669
670 for (x = 0; x < bim->num; x++) {
671 addr = bim->entry[x].addr;
672 size = bim->entry[x].size;
673 type = bim->entry[x].type;
674#ifdef DEBUG_MEMLOAD
675 printf("MEMMAP: 0x%016" PRIx64 "-0x%016" PRIx64
676 ", size=0x%016" PRIx64 ", type=%d(%s)\n",
677 addr, addr + size - 1, size, type,
678 (type == BIM_Memory) ? "Memory" :
679 (type == BIM_Reserved) ? "Reserved" :
680 (type == BIM_ACPI) ? "ACPI" :
681 (type == BIM_NVS) ? "NVS" :
682 (type == BIM_PMEM) ? "Persistent" :
683 (type == BIM_PRAM) ? "Persistent (Legacy)" :
684 "unknown");
685#endif
686
687 /* If the segment is not memory, skip it. */
688 switch (type) {
689 case BIM_Memory:
690 case BIM_ACPI:
691 case BIM_NVS:
692 break;
693 default:
694 continue;
695 }
696
697 /* If the segment is smaller than a page, skip it. */
698 if (size < PAGE_SIZE)
699 continue;
700
701 seg_start = addr;
702 seg_end = addr + size;
703
704 /*
705 * XXX XXX: Avoid the ISA I/O MEM.
706 *
707 * Some laptops (for example, Toshiba Satellite2550X) report
708 * this area as valid.
709 */
710 if (seg_start < IOM_END && seg_end > IOM_BEGIN) {
711 printf("WARNING: memory map entry overlaps "
712 "with ``Compatibility Holes'': "
713 "0x%"PRIx64"/0x%"PRIx64"/0x%x\n", seg_start,
714 seg_end - seg_start, type);
715
716 if (x86_add_cluster(seg_start, IOM_BEGIN, type) == -1)
717 break;
718 if (x86_add_cluster(IOM_END, seg_end, type) == -1)
719 break;
720 } else {
721 if (x86_add_cluster(seg_start, seg_end, type) == -1)
722 break;
723 }
724 }
725
726 return 0;
727}

729static int
730x86_fake_clusters(void)
731{
732 extern struct extent *iomem_ex;
733 phys_ram_seg_t *cluster;
734 KASSERT(mem_cluster_cnt == 0);
735
736 /*
737 * Allocate the physical addresses used by RAM from the iomem extent
738 * map. This is done before the addresses are page rounded just to make
739 * sure we get them all.
740 */
741 if (extent_alloc_region(iomem_ex, 0, KBTOB(biosbasemem), EX_NOWAIT)) {
742 /* XXX What should we do? */
743 printf("WARNING: CAN'T ALLOCATE BASE MEMORY FROM "
744 "IOMEM EXTENT MAP!\n");
745 }
746
747 cluster = &mem_clusters[0];
748 cluster->start = 0;
749 cluster->size = trunc_page(KBTOB(biosbasemem));
750 physmem += atop(cluster->size);
751
752 if (extent_alloc_region(iomem_ex, IOM_END, KBTOB(biosextmem),
753 EX_NOWAIT)) {
754 /* XXX What should we do? */
755 printf("WARNING: CAN'T ALLOCATE EXTENDED MEMORY FROM "
756 "IOMEM EXTENT MAP!\n");
757 }
758
759#if NISADMA > 0
760 /*
761 * Some motherboards/BIOSes remap the 384K of RAM that would
762 * normally be covered by the ISA hole to the end of memory
763 * so that it can be used. However, on a 16M system, this
764 * would cause bounce buffers to be allocated and used.
765 * This is not desirable behaviour, as more than 384K of
766 * bounce buffers might be allocated. As a work-around,
767 * we round memory down to the nearest 1M boundary if
768 * we're using any isadma devices and the remapped memory
769 * is what puts us over 16M.
770 */
771 if (biosextmem > (15*1024) && biosextmem < (16*1024)) {
772 char pbuf[9];
773
774 format_bytes(pbuf, sizeof(pbuf), biosextmem - (15*1024));
775 printf("Warning: ignoring %s of remapped memory\n", pbuf);
776 biosextmem = (15*1024);
777 }
778#endif
779
780 cluster = &mem_clusters[1];
781 cluster->start = IOM_END;
782 cluster->size = trunc_page(KBTOB(biosextmem));
783 physmem += atop(cluster->size);
784
785 mem_cluster_cnt = 2;
786
787 avail_end = IOM_END + trunc_page(KBTOB(biosextmem));
788
789 return 0;
790}

792/*
793 * x86_load_region: load the physical memory region from seg_start to seg_end
794 * into the VM system.
795 */
796static void
797x86_load_region(uint64_t seg_start, uint64_t seg_end)
798{
799 unsigned int i;
800 uint64_t tmp;
801
802 i = __arraycount(x86_freelists);
803 while (i--) {
804 if (x86_freelists[i].limit <= seg_start)
805 continue;
806 if (x86_freelists[i].freelist == VM_FREELIST_DEFAULT)
807 continue;
808 tmp = MIN(x86_freelists[i].limit, seg_end);
809 if (tmp == seg_start)
810 continue;
811
812#ifdef DEBUG_MEMLOAD
813 printf("loading freelist %d 0x%"PRIx64"-0x%"PRIx64
814 " (0x%"PRIx64"-0x%"PRIx64")\n", x86_freelists[i].freelist,
815 seg_start, tmp, (uint64_t)atop(seg_start),
816 (uint64_t)atop(tmp));
817#endif
818
819 uvm_page_physload(atop(seg_start), atop(tmp), atop(seg_start),
820 atop(tmp), x86_freelists[i].freelist);
821 seg_start = tmp;
822 }
823
824 if (seg_start != seg_end) {
825#ifdef DEBUG_MEMLOAD
826 printf("loading default 0x%"PRIx64"-0x%"PRIx64
827 " (0x%"PRIx64"-0x%"PRIx64")\n", seg_start, seg_end,
828 (uint64_t)atop(seg_start), (uint64_t)atop(seg_end));
829#endif
830 uvm_page_physload(atop(seg_start), atop(seg_end),
831 atop(seg_start), atop(seg_end), VM_FREELIST_DEFAULT);
832 }
833}

836 * init_x86_clusters: retrieve the memory clusters provided by the BIOS, and
837 * initialize mem_clusters.

839void
840init_x86_clusters(void)
841{
842 struct btinfo_memmap *bim;
843 struct btinfo_efimemmap *biem;
844
845 /*
846 * Check to see if we have a memory map from the BIOS (passed to us by
847 * the boot program).
848 */
849#ifdef i386
850 extern int biosmem_implicit;
851 biem = lookup_bootinfo(BTINFO_EFIMEMMAP);
852 if (biem != NULL)
853 bim = efi_get_e820memmap();
854 else
855 bim = lookup_bootinfo(BTINFO_MEMMAP);
856 if ((biosmem_implicit || (biosbasemem == 0 && biosextmem == 0)) &&
857 bim != NULL && bim->num > 0)
858 x86_parse_clusters(bim);
859#else
860#if !defined(REALBASEMEM) && !defined(REALEXTMEM)
861 biem = lookup_bootinfo(BTINFO_EFIMEMMAP);
862 if (biem != NULL)
863 bim = efi_get_e820memmap();
864 else
865 bim = lookup_bootinfo(BTINFO_MEMMAP);
866 if (bim != NULL && bim->num > 0)
867 x86_parse_clusters(bim);
868#else
869 (void)bim, (void)biem;
870#endif
871#endif
872
873 if (mem_cluster_cnt == 0) {
874 /*
875 * If x86_parse_clusters didn't find any valid segment, create
876 * fake clusters.
877 */
878 x86_fake_clusters();
879 }
880}

882/*
883 * init_x86_vm: initialize the VM system on x86. We basically internalize as
884 * many physical pages as we can, starting at lowmem_rsvd, but we don't
885 * internalize the kernel physical pages (from pa_kstart to pa_kend).
886 */
887int
888init_x86_vm(paddr_t pa_kend)
889{
890 extern struct bootspace bootspace;
891 paddr_t pa_kstart = bootspace.head.pa;
892 uint64_t seg_start, seg_end;
893 uint64_t seg_start1, seg_end1;
894 int x;
895 unsigned i;
896
897 for (i = 0; i < __arraycount(x86_freelists); i++) {
898 if (avail_end < x86_freelists[i].limit)
899 x86_freelists[i].freelist = VM_FREELIST_DEFAULT;
900 }
901
902 /*
903 * Now, load the memory clusters (which have already been rounded and
904 * truncated) into the VM system.
905 *
906 * NOTE: we assume that memory starts at 0.
907 */
908 for (x = 0; x < mem_cluster_cnt; x++) {
909 const phys_ram_seg_t *cluster = &mem_clusters[x];
910
911 seg_start = cluster->start;
912 seg_end = cluster->start + cluster->size;
913 seg_start1 = 0;
914 seg_end1 = 0;
915
916 /* Skip memory before our available starting point. */
917 if (seg_end <= lowmem_rsvd)
918 continue;
919
920 if (seg_start <= lowmem_rsvd && lowmem_rsvd < seg_end) {
921 seg_start = lowmem_rsvd;
922 if (seg_start == seg_end)
923 continue;
924 }
925
926 /*
927 * If this segment contains the kernel, split it in two, around
928 * the kernel.
929 */
930 if (seg_start <= pa_kstart && pa_kend <= seg_end) {
931 seg_start1 = pa_kend;
932 seg_end1 = seg_end;
933 seg_end = pa_kstart;
934 KASSERT(seg_end < seg_end1);
935 }
936
937 /* First hunk */
938 if (seg_start != seg_end) {
939 x86_load_region(seg_start, seg_end);
940 }
941
942 /* Second hunk */
943 if (seg_start1 != seg_end1) {
944 x86_load_region(seg_start1, seg_end1);
945 }
946 }
947
948 return 0;
949}

953void
954init_x86_msgbuf(void)
955{
956 /* Message buffer is located at end of core. */
957 psize_t sz = round_page(MSGBUFSIZE);
958 psize_t reqsz = sz;
959 uvm_physseg_t x;
960
961 search_again:
962 for (x = uvm_physseg_get_first();
963 uvm_physseg_valid_p(x);
964 x = uvm_physseg_get_next(x)) {
965
966 if (ctob(uvm_physseg_get_avail_end(x)) == avail_end)
967 break;
968 }
969
970 if (uvm_physseg_valid_p(x) == false)
971 panic("init_x86_msgbuf: can't find end of memory");
972
973 /* Shrink so it'll fit in the last segment. */
974 if (uvm_physseg_get_avail_end(x) - uvm_physseg_get_avail_start(x) < atop(sz))
975 sz = ctob(uvm_physseg_get_avail_end(x) - uvm_physseg_get_avail_start(x));
976
977 msgbuf_p_seg[msgbuf_p_cnt].sz = sz;
978 msgbuf_p_seg[msgbuf_p_cnt++].paddr = ctob(uvm_physseg_get_avail_end(x)) - sz;
979 uvm_physseg_unplug(uvm_physseg_get_end(x) - atop(sz), atop(sz));
980
981 /* Now find where the new avail_end is. */
982 avail_end = ctob(uvm_physseg_get_highest_frame());
983
984 if (sz == reqsz)
985 return;
986
987 reqsz -= sz;
988 if (msgbuf_p_cnt == VM_PHYSSEG_MAX) {
989 /* No more segments available, bail out. */
990 printf("WARNING: MSGBUFSIZE (%zu) too large, using %zu.\n",
991 (size_t)MSGBUFSIZE, (size_t)(MSGBUFSIZE - reqsz));
992 return;
993 }
994
995 sz = reqsz;
996 goto search_again;
997}

999void
1000x86_reset(void)
1001{
1002 uint8_t b;
1003
1004#if NACPICA > 0
1005 /*
1006 * If ACPI is active, try to reset using the reset register
1007 * defined in the FADT.
1008 */
1009 if (acpi_active) {
1010 if (acpi_reset() == 0) {
1011 delay(500000); /* wait 0.5 sec to see if that did it */
1012 }
1013 }
1014#endif
1015
1016 /*
1017 * The keyboard controller has 4 random output pins, one of which is
1018 * connected to the RESET pin on the CPU in many PCs. We tell the
1019 * keyboard controller to pulse this line a couple of times.
1020 */
1021 outb(IO_KBD + KBCMDP, KBC_PULSE0);
1022 delay(100000);
1023 outb(IO_KBD + KBCMDP, KBC_PULSE0);
1024 delay(100000);
1025
1026 /*
1027 * Attempt to force a reset via the Reset Control register at
1028 * I/O port 0xcf9. Bit 2 forces a system reset when it
1029 * transitions from 0 to 1. Bit 1 selects the type of reset
1030 * to attempt: 0 selects a "soft" reset, and 1 selects a
1031 * "hard" reset. We try a "hard" reset. The first write sets
1032 * bit 1 to select a "hard" reset and clears bit 2. The
1033 * second write forces a 0 -> 1 transition in bit 2 to trigger
1034 * a reset.
1035 */
1036 outb(0xcf9, 0x2);
1037 outb(0xcf9, 0x6);
1038 DELAY(500000); /* wait 0.5 sec to see if that did it */
1039
1040 /*
1041 * Attempt to force a reset via the Fast A20 and Init register
1042 * at I/O port 0x92. Bit 1 serves as an alternate A20 gate.
1043 * Bit 0 asserts INIT# when set to 1. We are careful to only
1044 * preserve bit 1 while setting bit 0. We also must clear bit
1045 * 0 before setting it if it isn't already clear.
1046 */
1047 b = inb(0x92);
1048 if (b != 0xff) {
1049 if ((b & 0x1) != 0)
1050 outb(0x92, b & 0xfe);
1051 outb(0x92, b | 0x1);
1052 DELAY(500000); /* wait 0.5 sec to see if that did it */
1053 }
1054}
