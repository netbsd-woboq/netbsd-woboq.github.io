32/*
33 * x86 pmap(9) module: TLB shootdowns.
34 *
35 * TLB shootdowns are hard interrupts that operate outside the SPL framework.
36 * They do not need to be blocked, provided that the pmap module gets the
37 * order of events correct. The calls are made by poking the LAPIC directly.
38 * The interrupt handler is short and does one of the following: invalidate
39 * a set of pages, all user TLB entries or the entire TLB.
40 */
41

61/*
62 * TLB shootdown structures.
63 */
64
65typedef struct {
66#ifdef _LP64
67 uintptr_t tp_va[14]; /* whole struct: 128 bytes */
68#else
69 uintptr_t tp_va[13]; /* whole struct: 64 bytes */
70#endif
71 uint16_t tp_count;
72 uint16_t tp_pte;
73 int tp_userpmap;
74 kcpuset_t * tp_cpumask;
75} pmap_tlb_packet_t;
76
77/*
78 * No more than N separate invlpg.
79 *
80 * Statistically, a value of six is big enough to cover the requested number
81 * of pages in ~ 95% of the TLB shootdowns we are getting. We therefore rarely
82 * reach the limit, and increasing it can actually reduce the performance due
83 * to the high cost of invlpg.
84 */
85#define TP_MAXVA 6
86
87/*
88 * TLB shootdown state.
89 */
90static pmap_tlb_packet_t pmap_tlb_packet __cacheline_aligned;
91static volatile u_int pmap_tlb_pendcount __cacheline_aligned;
92static volatile u_int pmap_tlb_gen __cacheline_aligned;
93static struct evcnt pmap_tlb_evcnt __cacheline_aligned;

122void
123pmap_tlb_init(void)
124{
125
126 memset(&pmap_tlb_packet, 0, sizeof(pmap_tlb_packet_t));
127 pmap_tlb_pendcount = 0;
128 pmap_tlb_gen = 0;
129
130 evcnt_attach_dynamic(&pmap_tlb_evcnt, EVCNT_TYPE_INTR,
131 NULL, "TLB", "shootdown");
132
133#ifdef TLBSTATS
134 int i;
135
136 for (i = 0; i < TLBSHOOT__MAX; i++) {
137 evcnt_attach_dynamic(&tlbstat_local[i], EVCNT_TYPE_MISC,
138 NULL, "tlbshoot local", tlbstat_name[i]);
139 }
140 for (i = 0; i < TLBSHOOT__MAX; i++) {
141 evcnt_attach_dynamic(&tlbstat_remote[i], EVCNT_TYPE_MISC,
142 NULL, "tlbshoot remote", tlbstat_name[i]);
143 }
144 for (i = 0; i < TLBSHOOT__MAX; i++) {
145 evcnt_attach_dynamic(&tlbstat_kernel[i], EVCNT_TYPE_MISC,
146 NULL, "tlbshoot kernel", tlbstat_name[i]);
147 }
148 evcnt_attach_dynamic(&tlbstat_single_req, EVCNT_TYPE_MISC,
149 NULL, "tlbshoot single page", "requests");
150 evcnt_attach_dynamic(&tlbstat_single_issue, EVCNT_TYPE_MISC,
151 NULL, "tlbshoot single page", "issues");
152#endif
153}

155void
156pmap_tlb_cpu_init(struct cpu_info *ci)
157{
158 pmap_tlb_packet_t *tp = (pmap_tlb_packet_t *)ci->ci_pmap_data;
159
160 memset(tp, 0, sizeof(pmap_tlb_packet_t));
161 kcpuset_create(&tp->tp_cpumask, true);
162}

164static inline void
165pmap_tlbstat_count(struct pmap *pm, vaddr_t va, tlbwhy_t why)
166{
167#ifdef TLBSTATS
168 const cpuid_t cid = cpu_index(curcpu());
169 bool local = false, remote = false;
170
171 if (va != (vaddr_t)-1LL) {
172 atomic_inc_64(&tlbstat_single_req.ev_count);
173 }
174 if (pm == pmap_kernel()) {
175 atomic_inc_64(&tlbstat_kernel[why].ev_count);
176 return;
177 }
178
179 if (va >= VM_MAXUSER_ADDRESS) {
180 remote = kcpuset_isotherset(pm->pm_kernel_cpus, cid);
181 local = kcpuset_isset(pm->pm_kernel_cpus, cid);
182 }
183 remote |= kcpuset_isotherset(pm->pm_cpus, cid);
184 local |= kcpuset_isset(pm->pm_cpus, cid);
185
186 if (local) {
187 atomic_inc_64(&tlbstat_local[why].ev_count);
188 }
189 if (remote) {
190 atomic_inc_64(&tlbstat_remote[why].ev_count);
191 }
192#endif
193}

195static inline void
196pmap_tlb_invalidate(const pmap_tlb_packet_t *tp)
197{
198 int i;
199
200 /* Find out what we need to invalidate. */
201 if (tp->tp_count == (uint16_t)-1) {
202 if (tp->tp_pte & PTE_G) {
203 /* Invalidating all TLB entries. */
204 tlbflushg();
205 } else {
206 /* Invalidating non-global TLB entries only. */
207 tlbflush();
208 }
209 } else {
210 /* Invalidating a single page or a range of pages. */
211 for (i = tp->tp_count - 1; i >= 0; i--) {
212 pmap_update_pg(tp->tp_va[i]);
213 }
214 }
215}

217/*
218 * pmap_tlb_shootdown: invalidate a page on all CPUs using pmap 'pm'.
219 */
220void
221pmap_tlb_shootdown(struct pmap *pm, vaddr_t va, pt_entry_t pte, tlbwhy_t why)
222{
223 pmap_tlb_packet_t *tp;
224 int s;
225
226#ifndef XENPV
227 KASSERT((pte & PTE_G) == 0 || pm == pmap_kernel());
228#endif
229
230 if (__predict_false(pm->pm_tlb_flush != NULL)) {
231 (*pm->pm_tlb_flush)(pm);
232 return;
233 }
234
235 /*
236 * If tearing down the pmap, do nothing. We will flush later
237 * when we are ready to recycle/destroy it.
238 */
239 if (__predict_false(curlwp->l_md.md_gc_pmap == pm)) {
240 return;
241 }
242
243 if ((pte & PTE_PS) != 0) {
244 va &= PTE_LGFRAME;
245 }
246
247 /*
248 * Add the shootdown operation to our pending set.
249 */
250 s = splvm();
251 tp = (pmap_tlb_packet_t *)curcpu()->ci_pmap_data;
252
253 /* Whole address flush will be needed if PTE_G is set. */
254 CTASSERT(PTE_G == (uint16_t)PTE_G);
255 tp->tp_pte |= (uint16_t)pte;
256
257 if (tp->tp_count == (uint16_t)-1) {
258 /*
259 * Already flushing everything.
260 */
261 } else if (tp->tp_count < TP_MAXVA && va != (vaddr_t)-1LL) {
262 /* Flush a single page. */
263 tp->tp_va[tp->tp_count++] = va;
264 KASSERT(tp->tp_count > 0);
265 } else {
266 /* Flush everything. */
267 tp->tp_count = (uint16_t)-1;
268 }
269
270 if (pm != pmap_kernel()) {
271 kcpuset_merge(tp->tp_cpumask, pm->pm_cpus);
272 if (va >= VM_MAXUSER_ADDRESS) {
273 kcpuset_merge(tp->tp_cpumask, pm->pm_kernel_cpus);
274 }
275 tp->tp_userpmap = 1;
276 } else {
277 kcpuset_copy(tp->tp_cpumask, kcpuset_running);
278 }
279 pmap_tlbstat_count(pm, va, why);
280 splx(s);
281}

286static inline void
287pmap_tlb_processpacket(pmap_tlb_packet_t *tp, kcpuset_t *target)
288{
289
290 if (tp->tp_count != (uint16_t)-1) {
291 /* Invalidating a single page or a range of pages. */
292 for (int i = tp->tp_count - 1; i >= 0; i--) {
293 xen_mcast_invlpg(tp->tp_va[i], target);
294 }
295 } else {
296 xen_mcast_tlbflush(target);
297 }
298
299 /* Remote CPUs have been synchronously flushed. */
300 pmap_tlb_pendcount = 0;
301}
302
303#else
304
305static inline void
306pmap_tlb_processpacket(pmap_tlb_packet_t *tp, kcpuset_t *target)
307{
308 int err = 0;
309
310 if (!kcpuset_match(target, kcpuset_attached)) {
311 const struct cpu_info * const self = curcpu();
312 CPU_INFO_ITERATOR cii;
313 struct cpu_info *lci;
314
315 for (CPU_INFO_FOREACH(cii, lci)) {
316 const cpuid_t lcid = cpu_index(lci);
317
318 if (__predict_false(lci == self) ||
319 !kcpuset_isset(target, lcid)) {
320 continue;
321 }
322 err |= x86_ipi(LAPIC_TLB_VECTOR,
323 lci->ci_cpuid, LAPIC_DLMODE_FIXED);
324 }
325 } else {
326 err = x86_ipi(LAPIC_TLB_VECTOR, LAPIC_DEST_ALLEXCL,
327 LAPIC_DLMODE_FIXED);
328 }
329 KASSERT(err == 0);
330}

335/*
336 * pmap_tlb_shootnow: process pending TLB shootdowns queued on current CPU.
337 *
338 * => Must be called with preemption disabled.
339 */
340void
341pmap_tlb_shootnow(void)
342{
343 pmap_tlb_packet_t *tp;
344 struct cpu_info *ci;
345 kcpuset_t *target;
346 u_int local, gen, rcpucount;
347 cpuid_t cid;
348 int s;
349
350 KASSERT(kpreempt_disabled());
351
352 ci = curcpu();
353 tp = (pmap_tlb_packet_t *)ci->ci_pmap_data;
354
355 /* Pre-check first. */
356 if (tp->tp_count == 0) {
357 return;
358 }
359
360 s = splvm();
361 if (tp->tp_count == 0) {
362 splx(s);
363 return;
364 }
365 cid = cpu_index(ci);
366
367 target = tp->tp_cpumask;
368 local = kcpuset_isset(target, cid) ? 1 : 0;
369 rcpucount = kcpuset_countset(target) - local;
370 gen = 0;
371
372#ifdef MULTIPROCESSOR
373 if (rcpucount) {
374 int count;
375
376 /*
377 * Gain ownership of the shootdown mailbox. We must stay
378 * at IPL_VM once we own it or could deadlock against an
379 * interrupt on this CPU trying to do the same.
380 */
381 KASSERT(rcpucount < ncpu);
382
383 while (atomic_cas_uint(&pmap_tlb_pendcount, 0, rcpucount)) {
384 splx(s);
385 count = SPINLOCK_BACKOFF_MIN;
386 while (pmap_tlb_pendcount) {
387 KASSERT(pmap_tlb_pendcount < ncpu);
388 SPINLOCK_BACKOFF(count);
389 }
390 s = splvm();
391 /* An interrupt might have done it for us. */
392 if (tp->tp_count == 0) {
393 splx(s);
394 return;
395 }
396 }
397
398 /*
399 * Start a new generation of updates. Copy our shootdown
400 * requests into the global buffer. Note that tp_cpumask
401 * will not be used by remote CPUs (it would be unsafe).
402 */
403 gen = ++pmap_tlb_gen;
404 memcpy(&pmap_tlb_packet, tp, sizeof(*tp));
405 pmap_tlb_evcnt.ev_count++;
406
407 /*
408 * Initiate shootdowns on remote CPUs.
409 */
410 pmap_tlb_processpacket(tp, target);
411 }
412#endif
413
414 /*
415 * Shootdowns on remote CPUs are now in flight. In the meantime,
416 * perform local shootdown if needed.
417 */
418 if (local) {
419 pmap_tlb_invalidate(tp);
420 }
421
422 /*
423 * Clear out our local buffer.
424 */
425#ifdef TLBSTATS
426 if (tp->tp_count != (uint16_t)-1) {
427 atomic_add_64(&tlbstat_single_issue.ev_count, tp->tp_count);
428 }
429#endif
430 kcpuset_zero(tp->tp_cpumask);
431 tp->tp_userpmap = 0;
432 tp->tp_count = 0;
433 tp->tp_pte = 0;
434 splx(s);
435
436 /*
437 * Now wait for the current generation of updates to be
438 * processed by remote CPUs.
439 */
440 if (rcpucount && pmap_tlb_pendcount) {
441 int count = SPINLOCK_BACKOFF_MIN;
442
443 while (pmap_tlb_pendcount && pmap_tlb_gen == gen) {
444 KASSERT(pmap_tlb_pendcount < ncpu);
445 SPINLOCK_BACKOFF(count);
446 }
447 }
448}

450/*
451 * pmap_tlb_intr: pmap shootdown interrupt handler to invalidate TLB entries.
452 *
453 * => Called from IPI only.
454 */
455void
456pmap_tlb_intr(void)
457{
458 const pmap_tlb_packet_t *tp = &pmap_tlb_packet;
459 struct cpu_info *ci = curcpu();
460
461 KASSERT(pmap_tlb_pendcount > 0);
462
463 /* First, TLB flush. */
464 pmap_tlb_invalidate(tp);
465
466 /*
467 * Check the current TLB state. If we do not want further
468 * invalidations for this pmap, then take the CPU out of
469 * the pmap's bitmask.
470 */
471 if (ci->ci_tlbstate == TLBSTATE_LAZY && tp->tp_userpmap) {
472 struct pmap *pm = ci->ci_pmap;
473 cpuid_t cid = cpu_index(ci);
474
475 kcpuset_atomic_clear(pm->pm_cpus, cid);
476 ci->ci_tlbstate = TLBSTATE_STALE;
477 }
478
479 /* Finally, ack the request. */
480 atomic_dec_uint(&pmap_tlb_pendcount);
481}
