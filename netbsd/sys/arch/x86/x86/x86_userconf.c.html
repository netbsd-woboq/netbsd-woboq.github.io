37#include <sys/systm.h>
38
39#include <sys/userconf.h>
40
41#include <machine/bootinfo.h>
42
43void
44userconf_bootinfo(void)
45{
46 struct btinfo_userconfcommands *biuc;
47 struct bi_userconfcommand *bi, *bimax;
48
49 biuc = lookup_bootinfo(BTINFO_USERCONFCOMMANDS);
50 if (biuc == NULL) {
51 aprint_debug("No bootinfo commands at boot\n");
52 return;
53 }
54
55 bi = (struct bi_userconfcommand *)((uint8_t *)biuc + sizeof(*biuc));
56 bimax = bi + biuc->num;
57 for (; bi < bimax; bi++) {
58 aprint_debug("Processing userconf command: %s\n", bi->text);
59 userconf_parse(bi->text);
60 }
61}
