27/* Interface to the Xen Grant tables */
28#include <xen/include/public/xen.h>
29#include <xen/include/public/grant_table.h>
30
31void xengnt_init(void);
32
33/* suspend/resume grant table, for save/restore operations */
34bool xengnt_suspend(void);
35bool xengnt_resume(void);
36
37/*
38 * grant access to a remote domain. Returns a handle on the allocated grant
39 * entry in table in grant_ref_t *.
40 */
41int xengnt_grant_access(domid_t, paddr_t, int, grant_ref_t *);
42
43/*
44 * Revoke access. Caller is responsible to ensure that the grant entry is
45 * not referenced any more.
46 */
47void xengnt_revoke_access(grant_ref_t);
48
49/* allow a page transfer from a remote domain */
50int xengnt_grant_transfer(domid_t, grant_ref_t *);
51
52/* end transfer, return the new page address or 0 */
53paddr_t xengnt_revoke_transfer(grant_ref_t);
54
55/*
56 * Query grant status (i.e. if remote has a valid mapping to this grant).
57 * Returns GTF_reading | GTF_writing (0 if remote does not use it anymore).
58 */
59int xengnt_status(grant_ref_t);
