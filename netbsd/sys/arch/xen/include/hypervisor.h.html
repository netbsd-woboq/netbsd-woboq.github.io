155extern int xen_version;
156#define XEN_MAJOR(x) (((x) & 0xffff0000) >> 16)
157#define XEN_MINOR(x) ((x) & 0x0000ffff)
158
159/*
160 * Does the hypervisor we're running on support an api
161 * call at the requested version number ?
162 */
163#define XEN_VERSION_SUPPORTED(major, minor) \
164 (XEN_MAJOR(xen_version) > (major) || \
165 (XEN_MAJOR(xen_version) == (major) && \
166 XEN_MINOR(xen_version) >= (minor)))

178/*
179 * Force a proper event-channel callback from Xen after clearing the
180 * callback mask. We do this in a very simple manner, by making a call
181 * down into Xen. The pending flag will be checked by Xen on return.
182 */
183static __inline void hypervisor_force_callback(void)
184{
185 (void)HYPERVISOR_xen_version(0, (void*)0);
186} __attribute__((no_instrument_function)) /* used by mcount */
187
188static __inline void
189hypervisor_notify_via_evtchn(unsigned int port)
190{
191 evtchn_op_t op;
192
193 op.cmd = EVTCHNOP_send;
194 op.u.send.port = port;
195 (void)HYPERVISOR_event_channel_op(&op);
196}
