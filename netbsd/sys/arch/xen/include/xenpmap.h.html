75#ifdef XENPV
76extern unsigned long *xpmap_phys_to_machine_mapping;
77
78static __inline paddr_t
79xpmap_mtop_masked(paddr_t mpa)
80{
81 return (
82 (paddr_t)machine_to_phys_mapping[mpa >> PAGE_SHIFT] << PAGE_SHIFT);
83}
84
85static __inline paddr_t
86xpmap_mtop(paddr_t mpa)
87{
88 return (xpmap_mtop_masked(mpa) | (mpa & ~PG_FRAME));
89}
90
91static __inline paddr_t
92xpmap_ptom_masked(paddr_t ppa)
93{
94 return (
95 (paddr_t)xpmap_phys_to_machine_mapping[ppa >> PAGE_SHIFT]
96 << PAGE_SHIFT);
97}
98
99static __inline paddr_t
100xpmap_ptom(paddr_t ppa)
101{
102 return (xpmap_ptom_masked(ppa) | (ppa & ~PG_FRAME));
103}
104
105static __inline void
106xpmap_ptom_map(paddr_t ppa, paddr_t mpa)
107{
108 xpmap_phys_to_machine_mapping[ppa >> PAGE_SHIFT] = mpa >> PAGE_SHIFT;
109}
110
111static __inline void
112xpmap_ptom_unmap(paddr_t ppa)
113{
114 xpmap_phys_to_machine_mapping[ppa >> PAGE_SHIFT] = INVALID_P2M_ENTRY;
115}
116
117static __inline bool
118xpmap_ptom_isvalid(paddr_t ppa)
119{
120 return (
121 xpmap_phys_to_machine_mapping[ppa >> PAGE_SHIFT]
122 != INVALID_P2M_ENTRY);
123}

127static inline void
128MULTI_update_va_mapping(
129 multicall_entry_t *mcl, vaddr_t va,
130 pt_entry_t new_val, unsigned long flags)
131{
132 mcl->op = __HYPERVISOR_update_va_mapping;
133 mcl->args[0] = va;
134#if defined(__x86_64__)
135 mcl->args[1] = new_val;
136 mcl->args[2] = flags;
137#else
138 mcl->args[1] = (new_val & 0xffffffff);
139 mcl->args[2] = (new_val >> 32);
140 mcl->args[3] = flags;
141#endif
142}
143
144static inline void
145MULTI_update_va_mapping_otherdomain(
146 multicall_entry_t *mcl, vaddr_t va,
147 pt_entry_t new_val, unsigned long flags, domid_t domid)
148{
149 mcl->op = __HYPERVISOR_update_va_mapping_otherdomain;
150 mcl->args[0] = va;
151#if defined(__x86_64__)
152 mcl->args[1] = new_val;
153 mcl->args[2] = flags;
154 mcl->args[3] = domid;
155#else
156 mcl->args[1] = (new_val & 0xffffffff);
157 mcl->args[2] = (new_val >> 32);
158 mcl->args[3] = flags;
159 mcl->args[4] = domid;
160#endif
161}
