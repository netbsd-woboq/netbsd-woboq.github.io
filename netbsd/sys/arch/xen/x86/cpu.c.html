149 * CPU, on uniprocessors). The CPU info list is initialized to

153 .ci_dev = 0,

193static void
194cpu_attach(device_t parent, device_t self, void *aux)
195{
196 struct cpu_softc *sc = device_private(self);
197 struct cpu_attach_args *caa = aux;
198 struct cpu_info *ci;
199 uintptr_t ptr;
200 static int nphycpu = 0;
201
202 sc->sc_dev = self;
203
204 /*
205 * If we're an Application Processor, allocate a cpu_info
206 * If we're the first attached CPU use the primary cpu_info,
207 * otherwise allocate a new one
208 */
209 aprint_naive("\n");
210 aprint_normal("\n");
211 if (nphycpu > 0) {
212 struct cpu_info *tmp;
213 ptr = (uintptr_t)kmem_zalloc(sizeof(*ci) + CACHE_LINE_SIZE - 1,
214 KM_SLEEP);
215 ci = (struct cpu_info *)roundup2(ptr, CACHE_LINE_SIZE);
216 ci->ci_curldt = -1;
217
218 tmp = phycpu_info_list;
219 while (tmp->ci_next)
220 tmp = tmp->ci_next;
221
222 tmp->ci_next = ci;
223 } else {
224 ci = &phycpu_info_primary;
225 }
226
227 ci->ci_self = ci;
228 sc->sc_info = ci;
229
230 ci->ci_dev = self;
231 ci->ci_acpiid = caa->cpu_id;
232 ci->ci_cpuid = caa->cpu_number;
233 ci->ci_vcpu = NULL;
234 ci->ci_index = nphycpu++;
235
236 if (!pmf_device_register(self, NULL, NULL))
237 aprint_error_dev(self, "couldn't establish power handler\n");
238
239 (void)config_defer(self, cpu_defer);
240}

248static int
249cpu_rescan(device_t self, const char *ifattr, const int *locators)
250{
251 struct cpu_softc *sc = device_private(self);
252 struct cpufeature_attach_args cfaa;
253 struct cpu_info *ci = sc->sc_info;
254
255 memset(&cfaa, 0, sizeof(cfaa));
256 cfaa.ci = ci;
257
258 if (ifattr_match(ifattr, "cpufeaturebus")) {
259
260 if (ci->ci_frequency == NULL) {
261 cfaa.name = "frequency";
262 ci->ci_frequency = config_found_ia(self,
263 "cpufeaturebus", &cfaa, NULL);
264 }
265 }
266
267 return 0;
268}

280static int
281vcpu_match(device_t parent, cfdata_t match, void *aux)
282{
283 struct vcpu_attach_args *vcaa = aux;
284 struct vcpu_runstate_info vcr;
285 int error;
286
287 if (strcmp(vcaa->vcaa_name, match->cf_name) == 0) {
288 error = HYPERVISOR_vcpu_op(VCPUOP_get_runstate_info,
289 vcaa->vcaa_caa.cpu_number, &vcr);
290 switch (error) {
291 case 0:
292 return 1;
293 case -ENOENT:
294 return 0;
295 default:
296 panic("Unknown hypervisor error %d returned on vcpu runstate probe\n", error);
297 }
298 }
299
300 return 0;
301}

303static void
304vcpu_attach(device_t parent, device_t self, void *aux)
305{
306 struct vcpu_attach_args *vcaa = aux;
307
308 KASSERT(vcaa->vcaa_caa.cpu_func == NULL);
309 vcaa->vcaa_caa.cpu_func = &mp_cpu_funcs;
310 cpu_attach_common(parent, self, &vcaa->vcaa_caa);
311
312 if (!pmf_device_register(self, NULL, NULL))
313 aprint_error_dev(self, "couldn't establish power handler\n");
314}

323static void
324cpu_vm_init(struct cpu_info *ci)
325{
326 int ncolors = 2, i;
327
328 for (i = CAI_ICACHE; i <= CAI_L2CACHE; i++) {
329 struct x86_cache_info *cai;
330 int tcolors;
331
332 cai = &ci->ci_cinfo[i];
333
334 tcolors = atop(cai->cai_totalsize);
335 switch (cai->cai_associativity) {
336 case 0xff:
337 tcolors = 1; /* fully associative */
338 break;
339 case 0:
340 case 1:
341 break;
342 default:
343 tcolors /= cai->cai_associativity;
344 }
345 ncolors = uimax(ncolors, tcolors);
346 }
347
348 /*
349 * Knowing the size of the largest cache on this CPU, potentially
350 * re-color our pages.
351 */
352 aprint_debug_dev(ci->ci_dev, "%d page colors\n", ncolors);
353 uvm_page_recolor(ncolors);
354 pmap_tlb_cpu_init(ci);
355#ifndef __HAVE_DIRECT_MAP
356 pmap_vpage_cpu_init(ci);
357#endif
358}

360static void
361cpu_attach_common(device_t parent, device_t self, void *aux)
362{
363 struct cpu_softc *sc = device_private(self);
364 struct cpu_attach_args *caa = aux;
365 struct cpu_info *ci;
366 uintptr_t ptr;
367 int cpunum = caa->cpu_number;
368 static bool again = false;
369
370 sc->sc_dev = self;
371
372 /*
373 * If we're an Application Processor, allocate a cpu_info
374 * structure, otherwise use the primary's.
375 */
376 if (caa->cpu_role == CPU_ROLE_AP) {
377 aprint_naive(": Application Processor\n");
378 ptr = (uintptr_t)kmem_alloc(sizeof(*ci) + CACHE_LINE_SIZE - 1,
379 KM_SLEEP);
380 ci = (struct cpu_info *)roundup2(ptr, CACHE_LINE_SIZE);
381 memset(ci, 0, sizeof(*ci));
382 cpu_init_tss(ci);
383 } else {
384 aprint_naive(": %s Processor\n",
385 caa->cpu_role == CPU_ROLE_SP ? "Single" : "Boot");
386 ci = &cpu_info_primary;
387 }
388
389 ci->ci_self = ci;
390 sc->sc_info = ci;
391 ci->ci_dev = self;
392 ci->ci_cpuid = cpunum;
393
394 KASSERT(HYPERVISOR_shared_info != NULL);
395 KASSERT(cpunum < XEN_LEGACY_MAX_VCPUS);
396 ci->ci_vcpu = &HYPERVISOR_shared_info->vcpu_info[cpunum];
397
398 KASSERT(ci->ci_func == 0);
399 ci->ci_func = caa->cpu_func;
400 aprint_normal("\n");
401
402 /* Must be called before mi_cpu_attach(). */
403 cpu_vm_init(ci);
404
405 if (caa->cpu_role == CPU_ROLE_AP) {
406 int error;
407
408 error = mi_cpu_attach(ci);
409
410 KASSERT(ci->ci_data.cpu_idlelwp != NULL);
411 if (error != 0) {
412 aprint_error_dev(self,
413 "mi_cpu_attach failed with %d\n", error);
414 return;
415 }
416
417 } else {
418 KASSERT(ci->ci_data.cpu_idlelwp != NULL);
419 }
420
421 KASSERT(ci->ci_cpuid == ci->ci_index);
422#ifdef __x86_64__
423 /* No user PGD mapped for this CPU yet */
424 ci->ci_xen_current_user_pgd = 0;
425#endif
426 mutex_init(&ci->ci_kpm_mtx, MUTEX_DEFAULT, IPL_VM);
427 pmap_reference(pmap_kernel());
428 ci->ci_pmap = pmap_kernel();
429 ci->ci_tlbstate = TLBSTATE_STALE;
430
431 /*
432 * Boot processor may not be attached first, but the below
433 * must be done to allow booting other processors.
434 */
435 if (!again) {
436 atomic_or_32(&ci->ci_flags, CPUF_PRESENT | CPUF_PRIMARY);
437 /* Basic init. */
438 cpu_intr_init(ci);
439 cpu_get_tsc_freq(ci);
440 cpu_init(ci);
441 pmap_cpu_init_late(ci);
442
443 /* Every processor needs to init its own ipi h/w (similar to lapic) */
444 xen_ipi_init();
445
446 /* Make sure DELAY() is initialized. */
447 DELAY(1);
448 again = true;
449 }
450
451 /* further PCB init done later. */
452
453 switch (caa->cpu_role) {
454 case CPU_ROLE_SP:
455 atomic_or_32(&ci->ci_flags, CPUF_SP);
456 cpu_identify(ci);
457 x86_cpu_idle_init();
458 break;
459
460 case CPU_ROLE_BP:
461 atomic_or_32(&ci->ci_flags, CPUF_BSP);
462 cpu_identify(ci);
463 x86_cpu_idle_init();
464 break;
465
466 case CPU_ROLE_AP:
467 atomic_or_32(&ci->ci_flags, CPUF_AP);
468
469 /*
470 * report on an AP
471 */
472
473#if defined(MULTIPROCESSOR)
474 /* interrupt handler stack */
475 cpu_intr_init(ci);
476
477 /* Setup per-cpu memory for gdt */
478 gdt_alloc_cpu(ci);
479
480 pmap_cpu_init_late(ci);
481 cpu_start_secondary(ci);
482
483 if (ci->ci_flags & CPUF_PRESENT) {
484 struct cpu_info *tmp;
485
486 cpu_identify(ci);
487 tmp = cpu_info_list;
488 while (tmp->ci_next)
489 tmp = tmp->ci_next;
490
491 tmp->ci_next = ci;
492 }
493#else
494 aprint_error_dev(ci->ci_dev, "not started\n");
495#endif
496 break;
497
498 default:
499 panic("unknown processor type??\n");
500 }
501
502#ifdef MPVERBOSE
503 if (mp_verbose) {
504 struct lwp *l = ci->ci_data.cpu_idlelwp;
505 struct pcb *pcb = lwp_getpcb(l);
506
507 aprint_verbose_dev(self,
508 "idle lwp at %p, idle sp at 0x%p\n",
509 l,
510#ifdef i386
511 (void *)pcb->pcb_esp
512#else
513 (void *)pcb->pcb_rsp
514#endif
515 );
516
517 }
518#endif /* MPVERBOSE */
519}

521/*
522 * Initialize the processor appropriately.
523 */
524
525void
526cpu_init(struct cpu_info *ci)
527{
528 uint32_t cr4 = 0;
529
530 /*
531 * If we have FXSAVE/FXRESTOR, use them.
532 */
533 if (cpu_feature[0] & CPUID_FXSR) {
534 cr4 |= CR4_OSFXSR;
535
536 /*
537 * If we have SSE/SSE2, enable XMM exceptions.
538 */
539 if (cpu_feature[0] & (CPUID_SSE|CPUID_SSE2))
540 cr4 |= CR4_OSXMMEXCPT;
541 }
542
543 /* If xsave is supported, enable it */
544 if (cpu_feature[1] & CPUID2_XSAVE && x86_fpu_save >= FPU_SAVE_XSAVE)
545 cr4 |= CR4_OSXSAVE;
546
547 if (cr4) {
548 cr4 |= rcr4();
549 lcr4(cr4);
550 }
551
552 if (x86_fpu_save >= FPU_SAVE_FXSAVE) {
553 fpuinit_mxcsr_mask();
554 }
555
556 /*
557 * Changing CR4 register may change cpuid values. For example, setting
558 * CR4_OSXSAVE sets CPUID2_OSXSAVE. The CPUID2_OSXSAVE is in
559 * ci_feat_val[1], so update it.
560 * XXX Other than ci_feat_val[1] might be changed.
561 */
562 if (cpuid_level >= 1) {
563 u_int descs[4];
564
565 x86_cpuid(1, descs);
566 ci->ci_feat_val[1] = descs[2];
567 }
568
569 /* If xsave is enabled, enable all fpu features */
570 if (cr4 & CR4_OSXSAVE) {
571 wrxcr(0, x86_xsave_features & XCR0_FPU);
572 }
573
574 atomic_or_32(&ci->ci_flags, CPUF_RUNNING);
575}

580void
581cpu_boot_secondary_processors(void)
582{
583 struct cpu_info *ci;
584 kcpuset_t *cpus;
585 u_long i;
586
587 kcpuset_create(&cpus, true);
588 kcpuset_set(cpus, cpu_index(curcpu()));
589 for (i = 0; i < maxcpus; i++) {
590 ci = cpu_lookup(i);
591 if (ci == NULL)
592 continue;
593 if (ci->ci_data.cpu_idlelwp == NULL)
594 continue;
595 if ((ci->ci_flags & CPUF_PRESENT) == 0)
596 continue;
597 if (ci->ci_flags & (CPUF_BSP|CPUF_SP|CPUF_PRIMARY))
598 continue;
599 cpu_boot_secondary(ci);
600 kcpuset_set(cpus, cpu_index(ci));
601 }
602 while (!kcpuset_match(cpus, kcpuset_running))
603 ;
604 kcpuset_destroy(cpus);
605
606 x86_mp_online = true;
607}

636static void
637cpu_start_secondary(struct cpu_info *ci)
638{
639 int i;
640
641 aprint_debug_dev(ci->ci_dev, "starting\n");
642
643 ci->ci_curlwp = ci->ci_data.cpu_idlelwp;
644
645 if (CPU_STARTUP(ci, (vaddr_t) cpu_hatch) != 0) {
646 return;
647 }
648
649 /*
650 * wait for it to become ready
651 */
652 for (i = 100000; (!(ci->ci_flags & CPUF_PRESENT)) && i > 0; i--) {
653 delay(10);
654 }
655 if ((ci->ci_flags & CPUF_PRESENT) == 0) {
656 aprint_error_dev(ci->ci_dev, "failed to become ready\n");
657#if defined(MPDEBUG) && defined(DDB)
658 printf("dropping into debugger; continue from here to resume boot\n");
659 Debugger();
660#endif
661 }
662
663 CPU_START_CLEANUP(ci);
664}
665
666void
667cpu_boot_secondary(struct cpu_info *ci)
668{
669 int i;
670 atomic_or_32(&ci->ci_flags, CPUF_GO);
671 for (i = 100000; (!(ci->ci_flags & CPUF_RUNNING)) && i > 0; i--) {
672 delay(10);
673 }
674 if ((ci->ci_flags & CPUF_RUNNING) == 0) {
675 aprint_error_dev(ci->ci_dev, "CPU failed to start\n");
676#if defined(MPDEBUG) && defined(DDB)
677 printf("dropping into debugger; continue from here to resume boot\n");
678 Debugger();
679#endif
680 }
681}

683/*
684 * APs end up here immediately after initialisation and VCPUOP_up in
685 * mp_cpu_start().
686 * At this point, we are running in the idle pcb/idle stack of the new
687 * CPU. This function jumps to the idle loop and starts looking for
688 * work.
689 */
690extern void x86_64_tls_switch(struct lwp *);
691void
692cpu_hatch(void *v)
693{
694 struct cpu_info *ci = (struct cpu_info *)v;
695 struct pcb *pcb;
696 int s, i;
697
698 /* Setup TLS and kernel GS/FS */
699 cpu_init_msrs(ci, true);
700 cpu_init_idt();
701 gdt_init_cpu(ci);
702
703 cpu_probe(ci);
704
705 atomic_or_32(&ci->ci_flags, CPUF_PRESENT);
706
707 while ((ci->ci_flags & CPUF_GO) == 0) {
708 /* Don't use delay, boot CPU may be patching the text. */
709 for (i = 10000; i != 0; i--)
710 x86_pause();
711 }
712
713 /* Because the text may have been patched in x86_patch(). */
714 x86_flush();
715 tlbflushg();
716
717 KASSERT((ci->ci_flags & CPUF_RUNNING) == 0);
718
719 pcb = lwp_getpcb(curlwp);
720 pcb->pcb_cr3 = pmap_pdirpa(pmap_kernel(), 0);
721 pcb = lwp_getpcb(ci->ci_data.cpu_idlelwp);
722
723 xen_ipi_init();
724
725 xen_initclocks();
726
727#ifdef __x86_64__
728 fpuinit(ci);
729#endif
730
731 lldt(GSEL(GLDT_SEL, SEL_KPL));
732
733 cpu_init(ci);
734 cpu_get_tsc_freq(ci);
735
736 s = splhigh();
737 x86_enable_intr();
738 splx(s);
739
740 aprint_debug_dev(ci->ci_dev, "running\n");
741
742 cpu_switchto(NULL, ci->ci_data.cpu_idlelwp, true);
743
744 idle_loop(NULL);
745 KASSERT(false);
746}

784/*
785 * Setup the "trampoline". On Xen, we setup nearly all cpu context
786 * outside a trampoline, so we prototype and call targetip like so:
787 * void targetip(struct cpu_info *);
788 */
789
790static void
791gdt_prepframes(paddr_t *frames, vaddr_t base, uint32_t entries)
792{
793 int i;
794 for (i = 0; i < entries; i++) {
795 frames[i] = ((paddr_t)xpmap_ptetomach(
796 (pt_entry_t *)(base + (i << PAGE_SHIFT)))) >> PAGE_SHIFT;
797
798 /* Mark Read-only */
799 pmap_pte_clearbits(kvtopte(base + (i << PAGE_SHIFT)),
800 PTE_W);
801 }
802}

807static void
808xen_init_amd64_vcpuctxt(struct cpu_info *ci, struct vcpu_guest_context *initctx,
809 void targetrip(struct cpu_info *))
810{
811 /* page frames to point at GDT */
812 extern int gdt_size;
813 paddr_t frames[16];
814 psize_t gdt_ents;
815
816 struct lwp *l;
817 struct pcb *pcb;
818
819 volatile struct vcpu_info *vci;
820
821 KASSERT(ci != NULL);
822 KASSERT(ci != &cpu_info_primary);
823 KASSERT(initctx != NULL);
824 KASSERT(targetrip != NULL);
825
826 memset(initctx, 0, sizeof(*initctx));
827
828 gdt_ents = roundup(gdt_size, PAGE_SIZE) >> PAGE_SHIFT;
829 KASSERT(gdt_ents <= 16);
830
831 gdt_prepframes(frames, (vaddr_t)ci->ci_gdt, gdt_ents);
832
833 /* Initialise the vcpu context: We use idle_loop()'s pcb context. */
834
835 l = ci->ci_data.cpu_idlelwp;
836
837 KASSERT(l != NULL);
838 pcb = lwp_getpcb(l);
839 KASSERT(pcb != NULL);
840
841 /* resume with interrupts off */
842 vci = ci->ci_vcpu;
843 vci->evtchn_upcall_mask = 1;
844 xen_mb();
845
846 /* resume in kernel-mode */
847 initctx->flags = VGCF_in_kernel | VGCF_online;
848
849 /* Stack and entry points:
850 * We arrange for the stack frame for cpu_hatch() to
851 * appear as a callee frame of lwp_trampoline(). Being a
852 * leaf frame prevents trampling on any of the MD stack setup
853 * that x86/vm_machdep.c:cpu_lwp_fork() does for idle_loop()
854 */
855
856 initctx->user_regs.rdi = (uint64_t) ci; /* targetrip(ci); */
857 initctx->user_regs.rip = (vaddr_t) targetrip;
858
859 initctx->user_regs.cs = GSEL(GCODE_SEL, SEL_KPL);
860
861 initctx->user_regs.rflags = pcb->pcb_flags;
862 initctx->user_regs.rsp = pcb->pcb_rsp;
863
864 /* Data segments */
865 initctx->user_regs.ss = GSEL(GDATA_SEL, SEL_KPL);
866 initctx->user_regs.es = GSEL(GDATA_SEL, SEL_KPL);
867 initctx->user_regs.ds = GSEL(GDATA_SEL, SEL_KPL);
868
869 /* GDT */
870 memcpy(initctx->gdt_frames, frames, sizeof(frames));
871 initctx->gdt_ents = gdt_ents;
872
873 /* LDT */
874 initctx->ldt_base = (unsigned long)ldtstore;
875 initctx->ldt_ents = LDT_SIZE >> 3;
876
877 /* Kernel context state */
878 initctx->kernel_ss = GSEL(GDATA_SEL, SEL_KPL);
879 initctx->kernel_sp = pcb->pcb_rsp0;
880 initctx->ctrlreg[0] = pcb->pcb_cr0;
881 initctx->ctrlreg[1] = 0; /* "resuming" from kernel - no User cr3. */
882 initctx->ctrlreg[2] = (vaddr_t)targetrip;
883 /*
884 * Use pmap_kernel() L4 PD directly, until we setup the
885 * per-cpu L4 PD in pmap_cpu_init_late()
886 */
887 initctx->ctrlreg[3] = xen_pfn_to_cr3(x86_btop(xpmap_ptom(ci->ci_kpm_pdirpa)));
888 initctx->ctrlreg[4] = CR4_PAE | CR4_OSFXSR | CR4_OSXMMEXCPT;
889
890 /* Xen callbacks */
891 initctx->event_callback_eip = (unsigned long)hypervisor_callback;
892 initctx->failsafe_callback_eip = (unsigned long)failsafe_callback;
893 initctx->syscall_callback_eip = (unsigned long)Xsyscall;
894
895 return;
896}

902xen_init_i386_vcpuctxt(struct cpu_info *ci, struct vcpu_guest_context *initctx,

906 extern int gdt_size;
907 paddr_t frames[16];
908 psize_t gdt_ents;
909
910 struct lwp *l;
911 struct pcb *pcb;
912
913 volatile struct vcpu_info *vci;
914
915 KASSERT(ci != NULL);
916 KASSERT(ci != &cpu_info_primary);
917 KASSERT(initctx != NULL);
918 KASSERT(targeteip != NULL);
919
920 memset(initctx, 0, sizeof(*initctx));
921
922 gdt_ents = roundup(gdt_size, PAGE_SIZE) >> PAGE_SHIFT;
923 KASSERT(gdt_ents <= 16);
924
925 gdt_prepframes(frames, (vaddr_t)ci->ci_gdt, gdt_ents);
926
927 /*
928 * Initialise the vcpu context:
929 * We use this cpu's idle_loop() pcb context.
930 */
931
932 l = ci->ci_data.cpu_idlelwp;
933
934 KASSERT(l != NULL);
935 pcb = lwp_getpcb(l);
936 KASSERT(pcb != NULL);
937
938 /* resume with interrupts off */
939 vci = ci->ci_vcpu;
940 vci->evtchn_upcall_mask = 1;
941 xen_mb();
942
943 /* resume in kernel-mode */
944 initctx->flags = VGCF_in_kernel | VGCF_online;
945
946 /* Stack frame setup for cpu_hatch():
947 * We arrange for the stack frame for cpu_hatch() to
948 * appear as a callee frame of lwp_trampoline(). Being a
949 * leaf frame prevents trampling on any of the MD stack setup
950 * that x86/vm_machdep.c:cpu_lwp_fork() does for idle_loop()
951 */
952
953 initctx->user_regs.esp = pcb->pcb_esp - 4; /* Leave word for
954 arg1 */
955 {
956 /* targeteip(ci); */
957 uint32_t *arg = (uint32_t *)initctx->user_regs.esp;
958 arg[1] = (uint32_t)ci; /* arg1 */
959 }
960
961 initctx->user_regs.eip = (vaddr_t)targeteip;
962 initctx->user_regs.cs = GSEL(GCODE_SEL, SEL_KPL);
963 initctx->user_regs.eflags |= pcb->pcb_iopl;
964
965 /* Data segments */
966 initctx->user_regs.ss = GSEL(GDATA_SEL, SEL_KPL);
967 initctx->user_regs.es = GSEL(GDATA_SEL, SEL_KPL);
968 initctx->user_regs.ds = GSEL(GDATA_SEL, SEL_KPL);
969 initctx->user_regs.fs = GSEL(GDATA_SEL, SEL_KPL);
970
971 /* GDT */
972 memcpy(initctx->gdt_frames, frames, sizeof(frames));
973 initctx->gdt_ents = gdt_ents;
974
975 /* LDT */
976 initctx->ldt_base = (unsigned long)ldtstore;
977 initctx->ldt_ents = NLDT;
978
979 /* Kernel context state */
980 initctx->kernel_ss = GSEL(GDATA_SEL, SEL_KPL);
981 initctx->kernel_sp = pcb->pcb_esp0;
982 initctx->ctrlreg[0] = pcb->pcb_cr0;
983 initctx->ctrlreg[1] = 0; /* "resuming" from kernel - no User cr3. */
984 initctx->ctrlreg[2] = (vaddr_t)targeteip;
985 initctx->ctrlreg[3] = xen_pfn_to_cr3(x86_btop(xpmap_ptom(ci->ci_pae_l3_pdirpa)));
986 initctx->ctrlreg[4] = /* CR4_PAE | */CR4_OSFXSR | CR4_OSXMMEXCPT;
987
988 /* Xen callbacks */
989 initctx->event_callback_eip = (unsigned long)hypervisor_callback;
990 initctx->event_callback_cs = GSEL(GCODE_SEL, SEL_KPL);
991 initctx->failsafe_callback_eip = (unsigned long)failsafe_callback;
992 initctx->failsafe_callback_cs = GSEL(GCODE_SEL, SEL_KPL);
993
994 return;
995}

998int
999mp_cpu_start(struct cpu_info *ci, vaddr_t target)
1000{
1001 int hyperror;
1002 struct vcpu_guest_context vcpuctx;
1003
1004 KASSERT(ci != NULL);
1005 KASSERT(ci != &cpu_info_primary);
1006 KASSERT(ci->ci_flags & CPUF_AP);
1007
1008#ifdef __x86_64__
1009 xen_init_amd64_vcpuctxt(ci, &vcpuctx, (void (*)(struct cpu_info *))target);
1010#else
1011 xen_init_i386_vcpuctxt(ci, &vcpuctx, (void (*)(struct cpu_info *))target);
1012#endif
1013
1014 /* Initialise the given vcpu to execute cpu_hatch(ci); */
1015 if ((hyperror = HYPERVISOR_vcpu_op(VCPUOP_initialise, ci->ci_cpuid, &vcpuctx))) {
1016 aprint_error(": context initialisation failed. errno = %d\n", hyperror);
1017 return hyperror;
1018 }
1019
1020 /* Start it up */
1021
1022 /* First bring it down */
1023 if ((hyperror = HYPERVISOR_vcpu_op(VCPUOP_down, ci->ci_cpuid, NULL))) {
1024 aprint_error(": VCPUOP_down hypervisor command failed. errno = %d\n", hyperror);
1025 return hyperror;
1026 }
1027
1028 if ((hyperror = HYPERVISOR_vcpu_op(VCPUOP_up, ci->ci_cpuid, NULL))) {
1029 aprint_error(": VCPUOP_up hypervisor command failed. errno = %d\n", hyperror);
1030 return hyperror;
1031 }
1032
1033 if (!vcpu_is_up(ci)) {
1034 aprint_error(": did not come up\n");
1035 return -1;
1036 }
1037
1038 return 0;
1039}

1076void
1077cpu_get_tsc_freq(struct cpu_info *ci)
1078{
1079 uint32_t vcpu_tversion;
1080 const volatile vcpu_time_info_t *tinfo = &ci->ci_vcpu->time;
1081
1082 vcpu_tversion = tinfo->version;
1083 while (tinfo->version == vcpu_tversion); /* Wait for a time update. XXX: timeout ? */
1084
1085 uint64_t freq = 1000000000ULL << 32;
1086 freq = freq / (uint64_t)tinfo->tsc_to_system_mul;
1087 if (tinfo->tsc_shift < 0)
1088 freq = freq << -tinfo->tsc_shift;
1089 else
1090 freq = freq >> tinfo->tsc_shift;
1091 ci->ci_data.cpu_cc_freq = freq;

1109/*
1110 * Loads pmap for the current CPU.
1111 */
1112void
1113cpu_load_pmap(struct pmap *pmap, struct pmap *oldpmap)
1114{
1115 struct cpu_info *ci = curcpu();
1116 cpuid_t cid = cpu_index(ci);
1117 int i;
1118
1119 KASSERT(pmap != pmap_kernel());
1120
1121 mutex_enter(&ci->ci_kpm_mtx);
1122 /* make new pmap visible to xen_kpm_sync() */
1123 kcpuset_atomic_set(pmap->pm_xen_ptp_cpus, cid);
1124
1125#ifdef __x86_64__
1126 pd_entry_t *new_pgd;
1127 paddr_t l4_pd_ma;
1128
1129 l4_pd_ma = xpmap_ptom_masked(ci->ci_kpm_pdirpa);
1130
1131 /*
1132 * Map user space address in kernel space and load
1133 * user cr3
1134 */
1135 new_pgd = pmap->pm_pdir;
1136 KASSERT(pmap == ci->ci_pmap);
1137
1138 /* Copy user pmap L4 PDEs (in user addr. range) to per-cpu L4 */
1139 for (i = 0; i < PDIR_SLOT_USERLIM; i++) {
1140 KASSERT(pmap != pmap_kernel() || new_pgd[i] == 0);
1141 if (ci->ci_kpm_pdir[i] != new_pgd[i]) {
1142 xpq_queue_pte_update(l4_pd_ma + i * sizeof(pd_entry_t),
1143 new_pgd[i]);
1144 }
1145 }
1146
1147 xen_set_user_pgd(pmap_pdirpa(pmap, 0));
1148 ci->ci_xen_current_user_pgd = pmap_pdirpa(pmap, 0);
1149#else
1150 paddr_t l3_pd = xpmap_ptom_masked(ci->ci_pae_l3_pdirpa);
1151 /* don't update the kernel L3 slot */
1152 for (i = 0; i < PDP_SIZE - 1; i++) {
1153 xpq_queue_pte_update(l3_pd + i * sizeof(pd_entry_t),
1154 xpmap_ptom(pmap->pm_pdirpa[i]) | PTE_P);
1155 }
1156#endif
1157
1158 tlbflush();
1159
1160 /* old pmap no longer visible to xen_kpm_sync() */
1161 if (oldpmap != pmap_kernel()) {
1162 kcpuset_atomic_clear(oldpmap->pm_xen_ptp_cpus, cid);
1163 }
1164 mutex_exit(&ci->ci_kpm_mtx);
1165}

1167/*
1168 * pmap_cpu_init_late: perform late per-CPU initialization.
1169 *
1170 * Short note about percpu PDIR pages. Both the PAE and __x86_64__ architectures
1171 * have per-cpu PDIR tables, for two different reasons:
1172 * - on PAE, this is to get around Xen's pagetable setup constraints (multiple
1173 * L3[3]s cannot point to the same L2 - Xen will refuse to pin a table set up
1174 * this way).
1175 * - on __x86_64__, this is for multiple CPUs to map in different user pmaps
1176 * (see cpu_load_pmap()).
1177 *
1178 * What this means for us is that the PDIR of the pmap_kernel() is considered
1179 * to be a canonical "SHADOW" PDIR with the following properties:
1180 * - its recursive mapping points to itself
1181 * - per-cpu recursive mappings point to themselves on __x86_64__
1182 * - per-cpu L4 pages' kernel entries are expected to be in sync with
1183 * the shadow
1184 */
1185
1186void
1187pmap_cpu_init_late(struct cpu_info *ci)
1188{
1189 int i;
1190
1191 /*
1192 * The BP has already its own PD page allocated during early
1193 * MD startup.
1194 */
1195
1196#ifdef __x86_64__
1197 /* Setup per-cpu normal_pdes */
1198 extern pd_entry_t * const normal_pdes[];
1199 for (i = 0;i < PTP_LEVELS - 1;i++) {
1200 ci->ci_normal_pdes[i] = normal_pdes[i];
1201 }
1202#endif
1203
1204 if (ci == &cpu_info_primary)
1205 return;
1206
1207 KASSERT(ci != NULL);
1208
1209#if defined(i386)
1210 cpu_alloc_l3_page(ci);
1211 KASSERT(ci->ci_pae_l3_pdirpa != 0);
1212
1213 /* Initialise L2 entries 0 - 2: Point them to pmap_kernel() */
1214 for (i = 0; i < PDP_SIZE - 1; i++) {
1215 ci->ci_pae_l3_pdir[i] =
1216 xpmap_ptom_masked(pmap_kernel()->pm_pdirpa[i]) | PTE_P;
1217 }
1218#endif
1219
1220 ci->ci_kpm_pdir = (pd_entry_t *)uvm_km_alloc(kernel_map, PAGE_SIZE, 0,
1221 UVM_KMF_WIRED | UVM_KMF_ZERO | UVM_KMF_NOWAIT);
1222
1223 if (ci->ci_kpm_pdir == NULL) {
1224 panic("%s: failed to allocate L4 per-cpu PD for CPU %d\n",
1225 __func__, cpu_index(ci));
1226 }
1227 ci->ci_kpm_pdirpa = vtophys((vaddr_t)ci->ci_kpm_pdir);
1228 KASSERT(ci->ci_kpm_pdirpa != 0);
1229
1230#ifdef __x86_64__
1231 extern pt_entry_t xpmap_pg_nx;
1232
1233 /* Copy over the pmap_kernel() shadow L4 entries */
1234 memcpy(ci->ci_kpm_pdir, pmap_kernel()->pm_pdir, PAGE_SIZE);
1235
1236 /* Recursive kernel mapping */
1237 ci->ci_kpm_pdir[PDIR_SLOT_PTE] = xpmap_ptom_masked(ci->ci_kpm_pdirpa)
1238 | PTE_P | xpmap_pg_nx;
1239#else
1240 /* Copy over the pmap_kernel() shadow L2 entries */
1241 memcpy(ci->ci_kpm_pdir, pmap_kernel()->pm_pdir + PDIR_SLOT_KERN,
1242 nkptp[PTP_LEVELS - 1] * sizeof(pd_entry_t));
1243#endif
1244
1245 /* Xen wants a RO pdir. */
1246 pmap_protect(pmap_kernel(), (vaddr_t)ci->ci_kpm_pdir,
1247 (vaddr_t)ci->ci_kpm_pdir + PAGE_SIZE, VM_PROT_READ);
1248 pmap_update(pmap_kernel());
1249
1250#ifdef __x86_64__
1251 xpq_queue_pin_l4_table(xpmap_ptom_masked(ci->ci_kpm_pdirpa));
1252#else
1253 /*
1254 * Initialize L3 entry 3. This mapping is shared across all pmaps and is
1255 * static, ie: loading a new pmap will not update this entry.
1256 */
1257 ci->ci_pae_l3_pdir[3] = xpmap_ptom_masked(ci->ci_kpm_pdirpa) | PTE_P;
1258
1259 /* Xen wants a RO L3. */
1260 pmap_protect(pmap_kernel(), (vaddr_t)ci->ci_pae_l3_pdir,
1261 (vaddr_t)ci->ci_pae_l3_pdir + PAGE_SIZE, VM_PROT_READ);
1262 pmap_update(pmap_kernel());
1263
1264 xpq_queue_pin_l3_table(xpmap_ptom_masked(ci->ci_pae_l3_pdirpa));
1265#endif
1266}
