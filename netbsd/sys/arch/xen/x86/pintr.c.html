163#if defined(DOM0OPS) || NPCI > 0
164int
165xen_vec_alloc(int gsi)
166{
167 physdev_op_t op;
168
169 KASSERT(gsi < 255);
170
171 if (irq2port[gsi] == 0) {
172 op.cmd = PHYSDEVOP_ASSIGN_VECTOR;
173 op.u.irq_op.irq = gsi;
174 if (HYPERVISOR_physdev_op(&op) < 0) {
175 panic("PHYSDEVOP_ASSIGN_VECTOR gsi %d", gsi);
176 }
177 KASSERT(irq2vect[gsi] == 0 ||
178 irq2vect[gsi] == op.u.irq_op.vector);
179 irq2vect[gsi] = op.u.irq_op.vector;
180 KASSERT(vect2irq[op.u.irq_op.vector] == 0 ||
181 vect2irq[op.u.irq_op.vector] == gsi);
182 vect2irq[op.u.irq_op.vector] = gsi;
183 }
184
185 return (irq2vect[gsi]);
186}

188/*
189 * This function doesn't "allocate" anything. It merely translates our
190 * understanding of PIC to the XEN 'gsi' namespace. In the case of
191 * MSIs, pirq == irq. In the case of everything else, the hypervisor
192 * doesn't really care, so we just use the native conventions that
193 * have been setup during boot by mpbios.c/mpacpi.c
194 */
195int
196xen_pic_to_gsi(struct pic *pic, int pin)
197{
198 int gsi;
199
200 KASSERT(pic != NULL);
201
202 /*
203 * We assume that mpbios/mpacpi have done the right thing.
204 * If so, legacy_irq should identity map correctly to gsi.
205 */
206 gsi = pic->pic_vecbase + pin;
207
208 switch (pic->pic_type) {
209 case PIC_I8259:
210 KASSERT(gsi < 16);
211 break;
212 case PIC_IOAPIC:
213 break;
214 default: /* XXX: MSI Support */
215 panic("XXX: MSI(X) Support");
216 break;
217 }
218
219 KASSERT(gsi < 255);
220 return gsi;
221}
