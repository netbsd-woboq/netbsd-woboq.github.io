143extern paddr_t pmap_pa_start; /* PA of first physical page for this domain */
144extern paddr_t pmap_pa_end; /* PA of last physical page for this domain */

146int
147pmap_enter(struct pmap *pmap, vaddr_t va, paddr_t pa, vm_prot_t prot, u_int flags)
148{
149 paddr_t ma;
150
151 if (__predict_false(pa < pmap_pa_start || pmap_pa_end <= pa)) {
152 ma = pa; /* XXX hack */
153 } else {
154 ma = xpmap_ptom(pa);
155 }
156
157 return pmap_enter_ma(pmap, va, ma, pa, prot, flags, DOMID_SELF);
158}

160/*
161 * pmap_kenter_ma: enter a kernel mapping without R/M (pv_entry) tracking
162 *
163 * => no need to lock anything, assume va is already allocated
164 * => should be faster than normal pmap enter function
165 * => we expect a MACHINE address
166 */
167
168void
169pmap_kenter_ma(vaddr_t va, paddr_t ma, vm_prot_t prot, u_int flags)
170{
171 pt_entry_t *pte, opte, npte;
172
173 if (va < VM_MIN_KERNEL_ADDRESS)
174 pte = vtopte(va);
175 else
176 pte = kvtopte(va);
177
178 npte = ma | ((prot & VM_PROT_WRITE) ? PTE_W : 0) | PTE_P;
179 if (flags & PMAP_NOCACHE)
180 npte |= PG_N;
181
182 if ((cpu_feature[2] & CPUID_NOX) && !(prot & VM_PROT_EXECUTE))
183 npte |= PG_NX;
184
185 opte = pmap_pte_testset(pte, npte); /* zap! */
186
187 if (pmap_valid_entry(opte)) {
188#if defined(MULTIPROCESSOR)
189 if (__predict_false(x86_mp_online == false)) {
190 pmap_update_pg(va);
191 } else {
192 kpreempt_disable();
193 pmap_tlb_shootdown(pmap_kernel(), va, opte,
194 TLBSHOOT_KENTER);
195 kpreempt_enable();
196 }
197#else
198 /* Don't bother deferring in the single CPU case. */
199 pmap_update_pg(va);
200#endif
201 }
202}

204/*
205 * pmap_extract_ma: extract a MA for the given VA
206 */
207
208bool
209pmap_extract_ma(struct pmap *pmap, vaddr_t va, paddr_t *pap)
210{
211 pt_entry_t *ptes, pte;
212 pd_entry_t pde;
213 pd_entry_t * const *pdes;
214 struct pmap *pmap2;
215 int lvl;
216
217 kpreempt_disable();
218 pmap_map_ptes(pmap, &pmap2, &ptes, &pdes);
219 if (!pmap_pdes_valid(va, pdes, &pde, &lvl)) {
220 pmap_unmap_ptes(pmap, pmap2);
221 kpreempt_enable();
222 return false;
223 }
224
225 KASSERT(lvl == 1);
226 pte = ptes[pl1_i(va)];
227 pmap_unmap_ptes(pmap, pmap2);
228 kpreempt_enable();
229
230 if (__predict_true((pte & PTE_P) != 0)) {
231 if (pap != NULL)
232 *pap = (pte & PG_FRAME) | (va & (NBPD_L1 - 1));
233 return true;
234 }
235
236 return false;
237}

258/*
259 * NetBSD uses L2 shadow pages to support PAE with Xen. However, Xen does not
260 * handle them correctly during save/restore, leading to incorrect page
261 * tracking and pinning during restore.
262 * For save/restore to succeed, two functions are introduced:
263 * - pmap_map_recursive_entries(), used by resume code to set the recursive
264 * mapping entries to their correct value
265 * - pmap_unmap_recursive_entries(), used by suspend code to clear all
266 * PDIR_SLOT_PTE entries
267 */
268void
269pmap_map_recursive_entries(void)
270{
271 int i;
272 struct pmap *pm;
273
274 mutex_enter(&pmaps_lock);
275 LIST_FOREACH(pm, &pmaps, pm_list) {
276 for (i = 0; i < PDP_SIZE; i++) {
277 xpq_queue_pte_update(
278 xpmap_ptom(pmap_pdirpa(pm, PDIR_SLOT_PTE + i)),
279 xpmap_ptom((pm)->pm_pdirpa[i]) | PTE_P);
280 }
281 }
282 mutex_exit(&pmaps_lock);
283
284 for (i = 0; i < PDP_SIZE; i++) {
285 xpq_queue_pte_update(
286 xpmap_ptom(pmap_pdirpa(pmap_kernel(), PDIR_SLOT_PTE + i)),
287 xpmap_ptom(pmap_kernel()->pm_pdirpa[i]) | PTE_P);
288 }
289}
290
291/*
292 * Unmap recursive entries found in pmaps. Required during Xen
293 * save/restore operations, as Xen does not handle recursive mappings
294 * properly.
295 */
296void
297pmap_unmap_recursive_entries(void)
298{
299 int i;
300 struct pmap *pm;
301
302 /*
303 * Invalidate pmap_pdp_cache as it contains L2-pinned objects with
304 * recursive entries.
305 * XXX jym@ : find a way to drain per-CPU caches to. pool_cache_inv
306 * does not do that.
307 */
308 pool_cache_invalidate(&pmap_pdp_cache);
309
310 mutex_enter(&pmaps_lock);
311 LIST_FOREACH(pm, &pmaps, pm_list) {
312 for (i = 0; i < PDP_SIZE; i++) {
313 xpq_queue_pte_update(
314 xpmap_ptom(pmap_pdirpa(pm, PDIR_SLOT_PTE + i)), 0);
315 }
316 }
317 mutex_exit(&pmaps_lock);
318
319 /* do it for pmap_kernel() too! */
320 for (i = 0; i < PDP_SIZE; i++) {
321 xpq_queue_pte_update(
322 xpmap_ptom(pmap_pdirpa(pmap_kernel(), PDIR_SLOT_PTE + i)),
323 0);
324 }
325}

327static __inline void
328pmap_kpm_setpte(struct cpu_info *ci, struct pmap *pmap, int index)
329{
330 KASSERT(mutex_owned(pmap->pm_lock));
331 KASSERT(mutex_owned(&ci->ci_kpm_mtx));
332 if (pmap == pmap_kernel()) {
333 KASSERT(index >= PDIR_SLOT_KERN);
334 }
335
336#ifdef __x86_64__
337 xpq_queue_pte_update(
338 xpmap_ptetomach(&ci->ci_kpm_pdir[index]),
339 pmap->pm_pdir[index]);
340#else
341 xpq_queue_pte_update(
342 xpmap_ptetomach(&ci->ci_kpm_pdir[l2tol2(index)]),
343 pmap->pm_pdir[index]);
344#endif
345
346 xpq_flush_queue();
347}

349/*
350 * Synchronise shadow pdir with the pmap on all cpus on which it is
351 * loaded.
352 */
353void
354xen_kpm_sync(struct pmap *pmap, int index)
355{
356 CPU_INFO_ITERATOR cii;
357 struct cpu_info *ci;
358
359 KASSERT(pmap != NULL);
360 KASSERT(kpreempt_disabled());
361
362 pmap_pte_flush();
363
364 for (CPU_INFO_FOREACH(cii, ci)) {
365 if (ci == NULL) {
366 continue;
367 }
368 cpuid_t cid = cpu_index(ci);
369 if (pmap != pmap_kernel() &&
370 !kcpuset_isset(pmap->pm_xen_ptp_cpus, cid))
371 continue;
372
373 /* take the lock and check again */
374 mutex_enter(&ci->ci_kpm_mtx);
375 if (pmap == pmap_kernel() ||
376 kcpuset_isset(pmap->pm_xen_ptp_cpus, cid)) {
377 pmap_kpm_setpte(ci, pmap, index);
378 }
379 mutex_exit(&ci->ci_kpm_mtx);
380 }
