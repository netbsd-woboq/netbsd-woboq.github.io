33/*
34 * The Xen balloon driver enables growing and shrinking PV domains
35 * memory on the fly, by allocating and freeing memory pages directly.
36 * This management needs domain cooperation to work properly, especially
37 * during balloon_inflate() operation where a domain gives back memory to
38 * the hypervisor.
39 *
40 * Shrinking memory on a live system is a difficult task, and may render
41 * it unstable or lead to crash. The driver takes a conservative approach
42 * there by doing memory operations in smal steps of a few MiB each time. It
43 * will also refuse to decrease reservation below a certain threshold
44 * (XEN_RESERVATION_MIN), so as to avoid a complete kernel memory exhaustion.
45 *
46 * The user can intervene at two different levels to manage the ballooning
47 * of a domain:
48 * - directly within the domain using a sysctl(9) interface.
49 * - through the Xentools, by modifying the memory/target entry associated
50 * to a domain. This is usually done in dom0.
51 *
52 * Modification of the reservation is signaled by writing inside the
53 * memory/target node in Xenstore. Writing new values will fire the xenbus
54 * watcher, and wakeup the balloon thread to inflate or deflate balloon.
55 *
56 * Both sysctl(9) nodes and memory/target entry assume that the values passed
57 * to them are in KiB. Internally, the driver will convert this value in
58 * pages (assuming a page is PAGE_SIZE bytes), and issue the correct hypercalls
59 * to decrease/increase domain's reservation accordingly.
60 *
61 * XXX Pages used by balloon are tracked through entries stored in a SLIST.
62 * This allows driver to conveniently add/remove wired pages from memory
63 * without the need to support these "memory gaps" inside uvm(9). Still, the
64 * driver does not currently "plug" new pages into uvm(9) when more memory
65 * is available than originally managed by balloon. For example, deflating
66 * balloon with a total number of pages above physmem is not supported for
67 * now. See balloon_deflate() for more details.
68 *
69 */

105/*
106 * Number of MFNs stored in the array passed back and forth between domain
107 * and balloon/hypervisor, during balloon_inflate() / balloon_deflate(). These
108 * should fit in a page, for performance reasons.
109 */
110#define BALLOON_DELTA (PAGE_SIZE / sizeof(xen_pfn_t))
111
112/*
113 * Safeguard value. Refuse to go below this threshold, so that domain
114 * can keep some free pages for its own use. Value is arbitrary, and may
115 * evolve with time.
116 */
117#define BALLOON_BALLAST 256 /* In pages - 1MiB */
118#define XEN_RESERVATION_MIN (uvmexp.freemin + BALLOON_BALLAST) /* In pages */
119
120/* KB <-> PAGEs */
121#define PAGE_SIZE_KB (PAGE_SIZE >> 10) /* page size in KB */
122#define BALLOON_PAGES_TO_KB(_pg) ((uint64_t)_pg * PAGE_SIZE_KB)
123#define BALLOON_KB_TO_PAGES(_kb) (roundup(_kb, PAGE_SIZE_KB) / PAGE_SIZE_KB)
124
125/*
126 * A balloon page entry. Needed to track pages put/reclaimed from balloon
127 */
128struct balloon_page_entry {
129 struct vm_page *pg;
130 SLIST_ENTRY(balloon_page_entry) entry;
131};

133struct balloon_xenbus_softc {
134 device_t sc_dev;
135 struct sysctllog *sc_log;
136
137 kmutex_t balloon_mtx; /* Protects condvar, target and res_min (below) */
138 kcondvar_t balloon_cv; /* Condvar variable for target (below) */
139 size_t balloon_target; /* Target domain reservation size in pages. */
140 /* Minimum amount of memory reserved by domain, in KiB */
141 uint64_t balloon_res_min;
142
143 xen_pfn_t *sc_mfn_list; /* List of MFNs passed from/to balloon */
144 pool_cache_t bpge_pool; /* pool cache for balloon page entries */
145 /* linked list for tracking pages used by balloon */
146 SLIST_HEAD(, balloon_page_entry) balloon_page_entries;
147 size_t balloon_num_page_entries;
148};

177static int
178balloon_xenbus_match(device_t parent, cfdata_t match, void *aux)
179{
180 struct xenbusdev_attach_args *xa = aux;
181
182 if (strcmp(xa->xa_type, "balloon") != 0)
183 return 0;
184
185 if (match->cf_loc[XENBUSCF_ID] != XENBUSCF_ID_DEFAULT &&
186 match->cf_loc[XENBUSCF_ID] != xa->xa_id)
187 return 0;
188
189 return 1;
190}

192static void
193balloon_xenbus_attach(device_t parent, device_t self, void *aux)
194{
195 xen_pfn_t *mfn_list;
196 size_t currentpages;
197 struct balloon_xenbus_softc *sc = balloon_sc = device_private(self);
198
199 aprint_normal(": Xen Balloon driver\n");
200 sc->sc_dev = self;
201
202 /* Initialize target mutex and condvar */
203 mutex_init(&sc->balloon_mtx, MUTEX_DEFAULT, IPL_NONE);
204 cv_init(&sc->balloon_cv, "xen_balloon");
205
206 SLIST_INIT(&sc->balloon_page_entries);
207 sc->balloon_num_page_entries = 0;
208
209 /* Get current number of pages */
210 currentpages = xenmem_get_currentreservation();
211
212 KASSERT(currentpages > 0);
213
214 /* Update initial target value - no need to lock for initialization */
215 sc->balloon_target = currentpages;
216
217 /* Set the values used by sysctl */
218 sc->balloon_res_min =
219 BALLOON_PAGES_TO_KB(XEN_RESERVATION_MIN);
220
221 aprint_normal_dev(self, "current reservation: %"PRIu64" KiB\n",
222 BALLOON_PAGES_TO_KB(currentpages));
223#if BALLOONDEBUG
224 aprint_normal_dev(self, "min reservation: %"PRIu64" KiB\n",
225 sc->balloon_res_min);
226 aprint_normal_dev(self, "max reservation: %"PRIu64" KiB\n",
227 BALLOON_PAGES_TO_KB(xenmem_get_maxreservation()));
228#endif
229
230 sc->bpge_pool = pool_cache_init(sizeof(struct balloon_page_entry),
231 0, 0, 0, "xen_bpge", NULL, IPL_NONE, bpge_ctor, bpge_dtor, NULL);
232
233 sysctl_kern_xen_balloon_setup(sc);
234
235 /* List of MFNs passed from/to balloon for inflating/deflating */
236 mfn_list = kmem_alloc(BALLOON_DELTA * sizeof(*mfn_list), KM_SLEEP);
237 sc->sc_mfn_list = mfn_list;
238
239 /* Setup xenbus node watch callback */
240 if (register_xenbus_watch(&balloon_xenbus_watch)) {
241 aprint_error_dev(self, "unable to watch memory/target\n");
242 goto error;
243 }
244
245 /* Setup kernel thread to asynchronously (in/de)-flate the balloon */
246 if (kthread_create(PRI_NONE, KTHREAD_MPSAFE, NULL, balloon_thread,
247 sc, NULL, "xen_balloon")) {
248 aprint_error_dev(self, "unable to create balloon thread\n");
249 unregister_xenbus_watch(&balloon_xenbus_watch);
250 goto error;
251 }
252
253 if (!pmf_device_register(self, NULL, NULL))
254 aprint_error_dev(self, "couldn't establish power handler\n");
255
256 return;
257
258error:
259 sysctl_teardown(&sc->sc_log);
260 cv_destroy(&sc->balloon_cv);
261 mutex_destroy(&sc->balloon_mtx);
262 return;
263
264}

266/*
267 * Returns maximum memory reservation available to current domain. In Xen
268 * with DOMID_SELF, this hypercall never fails: return value should be
269 * interpreted as unsigned.
270 *
271 */
272static size_t
273xenmem_get_maxreservation(void)
274{
275 int s;
276 unsigned int ret;
277
278 s = splvm(); /* XXXSMP */
279 ret = HYPERVISOR_memory_op(XENMEM_maximum_reservation,
280 & (domid_t) { DOMID_SELF });
281
282 splx(s);
283
284 if (ret == 0) {
285 /* well, a maximum reservation of 0 is really bogus */
286 panic("%s failed, maximum reservation returned 0", __func__);
287 }
288
289 return ret;
290}
291
292/* Returns current reservation, in pages */
293static size_t
294xenmem_get_currentreservation(void)
295{
296 int s, ret;
297
298 s = splvm(); /* XXXSMP */
299 ret = HYPERVISOR_memory_op(XENMEM_current_reservation,
300 & (domid_t) { DOMID_SELF });
301 splx(s);
302
303 if (ret < 0) {
304 panic("%s failed: %d", __func__, ret);
305 }
306
307 return ret;
308}

310/*
311 * Get value (in KiB) of memory/target in XenStore for current domain
312 * A return value of 0 can be considered as bogus or absent.
313 */
314static unsigned long long
315balloon_xenbus_read_target(void)
316{
317 unsigned long long new_target;
318 int err = xenbus_read_ull(NULL, "memory", "target", &new_target, 0);
319
320 switch(err) {
321 case 0:
322 return new_target;
323 case ENOENT:
324 break;
325 default:
326 device_printf(balloon_sc->sc_dev,
327 "error %d, couldn't read xenbus target node\n", err);
328 break;
329 }
330
331 return 0;
332}
333
334/* Set memory/target value (in KiB) in XenStore for current domain */
335static void
336balloon_xenbus_write_target(unsigned long long new_target)
337{
338 int err = xenbus_printf(NULL, "memory", "target", "%llu", new_target);
339
340 if (err != 0) {
341 device_printf(balloon_sc->sc_dev,
342 "error %d, couldn't write xenbus target node\n", err);
343 }
344
345 return;
346}

348static int
349bpge_ctor(void *arg, void *obj, int flags)
350{
351 struct balloon_page_entry *bpge = obj;
352
353 bpge->pg = uvm_pagealloc(NULL, 0, NULL, UVM_PGA_ZERO);
354 if (bpge->pg == NULL)
355 return ENOMEM;
356
357 return 0;
358
359}

361static void
362bpge_dtor(void *arg, void *obj)
363{
364 struct balloon_page_entry *bpge = obj;
365
366 uvm_pagefree(bpge->pg);
367}

369/*
370 * Inflate balloon. Pages are moved out of domain's memory towards balloon.
371 */
372static size_t
373balloon_inflate(struct balloon_xenbus_softc *sc, size_t tpages)
374{
375 int rpages, s, ret;
376 paddr_t pa;
377 struct balloon_page_entry *bpg_entry;
378 xen_pfn_t *mfn_list = sc->sc_mfn_list;
379
380 struct xen_memory_reservation reservation = {
381 .address_bits = 0,
382 .extent_order = 0,
383 .domid = DOMID_SELF
384 };
385
386 KASSERT(tpages > 0);
387 KASSERT(tpages <= BALLOON_DELTA);
388
389 memset(mfn_list, 0, BALLOON_DELTA * sizeof(*mfn_list));
390
391 /* allocate pages that will be given to Hypervisor */
392 for (rpages = 0; rpages < tpages; rpages++) {
393
394 bpg_entry = pool_cache_get(sc->bpge_pool, PR_WAITOK);
395 if (bpg_entry == NULL) {
396 /* failed reserving a page for balloon */
397 break;
398 }
399
400 pa = VM_PAGE_TO_PHYS(bpg_entry->pg);
401
402 mfn_list[rpages] = xpmap_ptom(pa) >> PAGE_SHIFT;
403
404 s = splvm(); /* XXXSMP */
405 /* Invalidate pg */
406 xpmap_ptom_unmap(pa);
407 splx(s);
408
409 SLIST_INSERT_HEAD(&balloon_sc->balloon_page_entries,
410 bpg_entry, entry);
411 balloon_sc->balloon_num_page_entries++;
412 }
413
414 /* Hand over pages to Hypervisor */
415 set_xen_guest_handle(reservation.extent_start, mfn_list);
416 reservation.nr_extents = rpages;
417
418 s = splvm(); /* XXXSMP */
419 ret = HYPERVISOR_memory_op(XENMEM_decrease_reservation,
420 &reservation);
421 splx(s);
422
423 if (ret != rpages) {
424 /*
425 * we are in bad shape: the operation failed for certain
426 * MFNs. As the API does not allow us to know which frame
427 * numbers were erroneous, we cannot really recover safely.
428 */
429 panic("%s: decrease reservation failed: was %d, "
430 "returned %d", device_xname(sc->sc_dev), rpages, ret);
431 }
432
433#if BALLOONDEBUG
434 device_printf(sc->sc_dev, "inflate %zu => inflated by %d\n",
435 tpages, rpages);
436#endif
437 return rpages;
438}

440/*
441 * Deflate balloon. Pages are given back to domain's memory.
442 */
443static size_t
444balloon_deflate(struct balloon_xenbus_softc *sc, size_t tpages)
445{
446 int rpages, s, ret;
447 paddr_t pa;
448 struct balloon_page_entry *bpg_entry;
449 xen_pfn_t *mfn_list = sc->sc_mfn_list;
450
451 struct xen_memory_reservation reservation = {
452 .address_bits = 0,
453 .extent_order = 0,
454 .domid = DOMID_SELF
455 };
456
457 KASSERT(tpages > 0);
458 KASSERT(tpages <= BALLOON_DELTA);
459
460 memset(mfn_list, 0, BALLOON_DELTA * sizeof(*mfn_list));
461
462#ifndef UVM_HOTPLUG
463 /*
464 * If the list is empty, we are deflating balloon beyond empty. This
465 * is currently unsupported as this would require to dynamically add
466 * new memory pages inside uvm(9) and instruct pmap(9) on how to
467 * handle them. For now, we clip reservation up to the point we
468 * can manage them, eg. the remaining bpg entries in the SLIST.
469 * XXX find a way to hotplug memory through uvm(9)/pmap(9).
470 */
471 if (tpages > sc->balloon_num_page_entries) {
472 device_printf(sc->sc_dev,
473 "memory 'hot-plug' unsupported - clipping "
474 "reservation %zu => %zu pages.\n",
475 tpages, sc->balloon_num_page_entries);
476 tpages = sc->balloon_num_page_entries;
477 }
478#endif
479
480 /* reclaim pages from balloon */
481 set_xen_guest_handle(reservation.extent_start, mfn_list);
482 reservation.nr_extents = tpages;
483
484 s = splvm(); /* XXXSMP */
485 ret = HYPERVISOR_memory_op(XENMEM_increase_reservation, &reservation);
486 splx(s);
487
488 if (ret < 0) {
489 panic("%s: increase reservation failed, ret %d",
490 device_xname(sc->sc_dev), ret);
491 }
492
493 if (ret != tpages) {
494 device_printf(sc->sc_dev,
495 "increase reservation incomplete: was %zu, "
496 "returned %d\n", tpages, ret);
497 }
498
499 /* plug pages back into memory through bpge entries */
500 for (rpages = 0; rpages < ret; rpages++) {
501#ifdef UVM_HOTPLUG
502 extern paddr_t pmap_pa_end;
503 if (sc->balloon_num_page_entries == 0) { /*XXX: consolidate */
504 /* "hot-plug": Stick it at the end of memory */
505 pa = pmap_pa_end;
506
507 /* P2M update */
508 s = splvm(); /* XXXSMP */
509 pmap_pa_end += PAGE_SIZE; /* XXX: TLB flush ?*/
510 xpmap_ptom_map(pa, ptoa(mfn_list[rpages]));
511 xpq_queue_machphys_update(ptoa(mfn_list[rpages]), pa);
512 splx(s);
513
514 if (uvm_physseg_plug(atop(pa), 1, NULL) == false) {
515 /* Undo P2M */
516 s = splvm(); /* XXXSMP */
517 xpmap_ptom_unmap(pa);
518 xpq_queue_machphys_update(ptoa(mfn_list[rpages]), 0);
519 pmap_pa_end -= PAGE_SIZE; /* XXX: TLB flush ?*/
520 splx(s);
521 break;
522 }
523 continue;
524 }
525#else
526 if (sc->balloon_num_page_entries == 0) {
527 /*
528 * XXX This is the case where extra "hot-plug"
529 * mem w.r.t boot comes in
530 */
531 device_printf(sc->sc_dev,
532 "List empty. Cannot be collapsed further!\n");
533 break;
534 }
535#endif
536 bpg_entry = SLIST_FIRST(&balloon_sc->balloon_page_entries);
537 SLIST_REMOVE_HEAD(&balloon_sc->balloon_page_entries, entry);
538 balloon_sc->balloon_num_page_entries--;
539
540 /* Update P->M */
541 pa = VM_PAGE_TO_PHYS(bpg_entry->pg);
542
543 s = splvm(); /* XXXSMP */
544
545 xpmap_ptom_map(pa, ptoa(mfn_list[rpages]));
546 xpq_queue_machphys_update(ptoa(mfn_list[rpages]), pa);
547
548 splx(s);
549
550 pool_cache_put(sc->bpge_pool, bpg_entry);
551 }
552
553 xpq_flush_queue();
554
555#if BALLOONDEBUG
556 device_printf(sc->sc_dev, "deflate %zu => deflated by %d\n",
557 tpages, rpages);
558#endif
559 return rpages;
560}

562/*
563 * The balloon thread is responsible for handling inflate/deflate balloon
564 * requests for the current domain given the new "target" value.
565 */
566static void
567balloon_thread(void *cookie)
568{
569 int ret;
570 size_t current, diff, target;
571 struct balloon_xenbus_softc *sc = cookie;
572
573 for/*ever*/ (;;) {
574 current = xenmem_get_currentreservation();
575
576 /*
577 * We assume that balloon_xenbus_watcher() and
578 * sysctl(9) handlers checked the sanity of the
579 * new target value.
580 */
581 mutex_enter(&sc->balloon_mtx);
582 target = sc->balloon_target;
583 if (current != target) {
584 /*
585 * There is work to do. Inflate/deflate in
586 * increments of BALLOON_DELTA pages at maximum. The
587 * risk of integer wrapping is mitigated by
588 * BALLOON_DELTA, which is the upper bound.
589 */
590 mutex_exit(&sc->balloon_mtx);
591 diff = MIN(target - current, BALLOON_DELTA);
592 if (current < target)
593 ret = balloon_deflate(sc, diff);
594 else
595 ret = balloon_inflate(sc, diff);
596
597 if (ret != diff) {
598 /*
599 * Something went wrong during operation.
600 * Log error then feedback current value in
601 * target so that thread gets back to waiting
602 * for the next iteration
603 */
604 device_printf(sc->sc_dev,
605 "WARNING: balloon could not reach target "
606 "%zu (current %zu)\n",
607 target, current);
608 current = xenmem_get_currentreservation();
609 mutex_enter(&sc->balloon_mtx);
610 sc->balloon_target = current;
611 mutex_exit(&sc->balloon_mtx);
612 }
613 } else {
614 /* no need for change -- wait for a signal */
615 cv_wait(&sc->balloon_cv, &sc->balloon_mtx);
616 mutex_exit(&sc->balloon_mtx);
617 }
618 }
619}

621/*
622 * Handler called when memory/target value changes inside Xenstore.
623 * All sanity checks must also happen in this handler, as it is the common
624 * entry point where controller domain schedules balloon operations.
625 */
626static void
627balloon_xenbus_watcher(struct xenbus_watch *watch, const char **vec,
628 unsigned int len)
629{
630 size_t new_target;
631 uint64_t target_kb, target_max, target_min;
632
633 target_kb = balloon_xenbus_read_target();
634 if (target_kb == 0) {
635 /* bogus -- just return */
636 return;
637 }
638
639 mutex_enter(&balloon_sc->balloon_mtx);
640 target_min = balloon_sc->balloon_res_min;
641 mutex_exit(&balloon_sc->balloon_mtx);
642 if (target_kb < target_min) {
643 device_printf(balloon_sc->sc_dev,
644 "new target %"PRIu64" is below min %"PRIu64"\n",
645 target_kb, target_min);
646 return;
647 }
648
649 target_max = BALLOON_PAGES_TO_KB(xenmem_get_maxreservation());
650 if (target_kb > target_max) {
651 /*
652 * Should not happen. Hypervisor should block balloon
653 * requests above mem-max.
654 */
655 device_printf(balloon_sc->sc_dev,
656 "new target %"PRIu64" is above max %"PRIu64"\n",
657 target_kb, target_max);
658 return;
659 }
660
661 new_target = BALLOON_KB_TO_PAGES(target_kb);
662
663 device_printf(balloon_sc->sc_dev,
664 "current reservation: %zu pages => target: %zu pages\n",
665 xenmem_get_currentreservation(), new_target);
666
667 /* Only update target if its value changes */
668 mutex_enter(&balloon_sc->balloon_mtx);
669 if (balloon_sc->balloon_target != new_target) {
670 balloon_sc->balloon_target = new_target;
671 cv_signal(&balloon_sc->balloon_cv);
672 }
673 mutex_exit(&balloon_sc->balloon_mtx);
674
675 return;
676}

681
682/* routine to control the minimum memory reserved for the domain */
683static int
684sysctl_kern_xen_balloon_min(SYSCTLFN_ARGS)
685{
686 struct sysctlnode node;
687 u_quad_t newval;
688 int error;
689
690 node = *rnode;
691 node.sysctl_data = &newval;
692
693 mutex_enter(&balloon_sc->balloon_mtx);
694 newval = balloon_sc->balloon_res_min;
695 mutex_exit(&balloon_sc->balloon_mtx);
696
697 error = sysctl_lookup(SYSCTLFN_CALL(&node));
698 if (error || newp == NULL)
699 return error;
700
701 /* Safeguard value: refuse to go below. */
702 if (newval < XEN_RESERVATION_MIN) {
703 device_printf(balloon_sc->sc_dev,
704 "cannot set min below minimum safe value (%d)\n",
705 XEN_RESERVATION_MIN);
706 return EPERM;
707 }
708
709 mutex_enter(&balloon_sc->balloon_mtx);
710 if (balloon_sc->balloon_res_min != newval)
711 balloon_sc->balloon_res_min = newval;
712 mutex_exit(&balloon_sc->balloon_mtx);
713
714 return 0;
715}

745/*
746 * Returns the target memory reservation of the domain
747 * When reading, this sysctl will return the value of the balloon_target
748 * variable, converted into KiB
749 * When used for writing, it will update the new memory/target value
750 * in XenStore, but will not update the balloon_target variable directly.
751 * This will be done by the Xenbus watch handler, balloon_xenbus_watcher().
752 */
753static int
754sysctl_kern_xen_balloon_target(SYSCTLFN_ARGS)
755{
756 struct sysctlnode node;
757 u_quad_t newval, res_min, res_max;
758 int error;
759
760 node = *rnode;
761 node.sysctl_data = &newval;
762
763 mutex_enter(&balloon_sc->balloon_mtx);
764 newval = BALLOON_PAGES_TO_KB(balloon_sc->balloon_target);
765 res_min = balloon_sc->balloon_res_min;
766 mutex_exit(&balloon_sc->balloon_mtx);
767
768 error = sysctl_lookup(SYSCTLFN_CALL(&node));
769 if (newp == NULL || error != 0) {
770 return error;
771 }
772
773 /*
774 * Sanity check new size
775 * We should not balloon below the minimum reservation
776 * set by the domain, nor above the maximum reservation set
777 * by domain controller.
778 * Note: domain is not supposed to receive balloon requests when
779 * they are above maximum reservation, but better be safe than
780 * sorry.
781 */
782 res_max = BALLOON_PAGES_TO_KB(xenmem_get_maxreservation());
783 if (newval < res_min || newval > res_max) {
784#if BALLOONDEBUG
785 device_printf(balloon_sc->sc_dev,
786 "new value out of bounds: %"PRIu64"\n", newval);
787 device_printf(balloon_sc->sc_dev,
788 "min %"PRIu64", max %"PRIu64"\n", res_min, res_max);
789#endif
790 return EPERM;
791 }
792
793 /*
794 * Write new value inside Xenstore. This will fire the memory/target
795 * watch handler, balloon_xenbus_watcher().
796 */
797 balloon_xenbus_write_target(newval);
798
799 return 0;
800}
