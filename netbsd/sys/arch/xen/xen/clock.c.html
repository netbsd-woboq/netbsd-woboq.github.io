169/*
170 * xen_rdtsc()
171 *
172 * Read the local pCPU's tsc.
173 */
174static inline uint64_t
175xen_rdtsc(void)
176{
177 uint32_t lo, hi;
178
179 asm volatile("rdtsc" : "=a"(lo), "=d"(hi));
180
181 return ((uint64_t)hi << 32) | lo;
182}

184/*
185 * struct xen_vcputime_ticket
186 *
187 * State for a vCPU read section, during which a caller may read
188 * from fields of a struct vcpu_time_info and call xen_rdtsc.
189 * Caller must enter with xen_vcputime_enter, exit with
190 * xen_vcputime_exit, and be prepared to retry if
191 * xen_vcputime_exit fails.
192 */
193struct xen_vcputime_ticket {
194 uint64_t version;
195};
196
197/*
198 * xen_vcputime_enter(tp)
199 *
200 * Enter a vCPU time read section and store a ticket in *tp, which
201 * the caller must use with xen_vcputime_exit. Return a pointer
202 * to the current CPU's vcpu_time_info structure. Caller must
203 * already be bound to the CPU.
204 */
205static inline volatile struct vcpu_time_info *
206xen_vcputime_enter(struct xen_vcputime_ticket *tp)
207{
208 volatile struct vcpu_time_info *vt = &curcpu()->ci_vcpu->time;
209
210 while (__predict_false(1 & (tp->version = vt->version)))
211 SPINLOCK_BACKOFF_HOOK;
212
213 /*
214 * Must read the version before reading the tsc on the local
215 * pCPU. We are racing only with interruption by the
216 * hypervisor, so no need for a stronger memory barrier.
217 */
218 __insn_barrier();
219
220 return vt;
221}
222
223/*
224 * xen_vcputime_exit(vt, tp)
225 *
226 * Exit a vCPU time read section with the ticket in *tp from
227 * xen_vcputime_enter. Return true on success, false if caller
228 * must retry.
229 */
230static inline bool
231xen_vcputime_exit(volatile struct vcpu_time_info *vt,
232 struct xen_vcputime_ticket *tp)
233{
234
235 KASSERT(vt == &curcpu()->ci_vcpu->time);
236
237 /*
238 * Must read the tsc before re-reading the version on the local
239 * pCPU. We are racing only with interruption by the
240 * hypervisor, so no need for a stronger memory barrier.
241 */
242 __insn_barrier();
243
244 return tp->version == vt->version;
245}

247/*
248 * xen_tsc_to_ns_delta(delta_tsc, mul_frac, shift)
249 *
250 * Convert a difference in tsc units to a difference in
251 * nanoseconds given a multiplier and shift for the unit
252 * conversion.
253 */
254static inline uint64_t
255xen_tsc_to_ns_delta(uint64_t delta_tsc, uint32_t tsc_to_system_mul,
256 int8_t tsc_shift)
257{
258 uint32_t delta_tsc_hi, delta_tsc_lo;
259
260 if (tsc_shift < 0)
261 delta_tsc >>= -tsc_shift;
262 else
263 delta_tsc <<= tsc_shift;
264
265 delta_tsc_hi = delta_tsc >> 32;
266 delta_tsc_lo = delta_tsc & 0xffffffffUL;
267
268 /* d*m/2^32 = (2^32 d_h + d_l)*m/2^32 = d_h*m + (d_l*m)/2^32 */
269 return ((uint64_t)delta_tsc_hi * tsc_to_system_mul) +
270 (((uint64_t)delta_tsc_lo * tsc_to_system_mul) >> 32);
271}

273/*
274 * xen_vcputime_systime_ns()
275 *
276 * Return a snapshot of the Xen system time plus an adjustment
277 * from the tsc, in units of nanoseconds. Caller must be bound to
278 * the current CPU.
279 */
280static uint64_t
281xen_vcputime_systime_ns(void)
282{
283 volatile struct vcpu_time_info *vt;
284 struct cpu_info *ci = curcpu();
285 struct xen_vcputime_ticket ticket;
286 uint64_t raw_systime_ns, tsc_timestamp, tsc, delta_tsc, delta_ns;
287 uint32_t tsc_to_system_mul;
288 int8_t tsc_shift;
289 uint64_t systime_ns;
290
291 /* We'd better be bound to the CPU in _some_ way. */
292 KASSERT(cpu_intr_p() || cpu_softintr_p() || kpreempt_disabled() ||
293 (curlwp->l_flag & LP_BOUND));
294
295 /*
296 * Repeatedly try to read the system time, corresponding tsc
297 * timestamp, and tsc frequency until we get a consistent view.
298 */
299 do {
300 vt = xen_vcputime_enter(&ticket);
301
302 /* Grab Xen's snapshot of raw system time and tsc. */
303 raw_systime_ns = vt->system_time;
304 tsc_timestamp = vt->tsc_timestamp;
305
306 /* Get Xen's current idea of how fast the tsc is counting. */
307 tsc_to_system_mul = vt->tsc_to_system_mul;
308 tsc_shift = vt->tsc_shift;
309
310 /* Read the CPU's tsc. */
311 tsc = xen_rdtsc();
312 } while (!xen_vcputime_exit(vt, &ticket));
313
314 /*
315 * Out of paranoia, check whether the tsc has gone backwards
316 * since Xen's timestamp.
317 *
318 * This shouldn't happen because the Xen hypervisor is supposed
319 * to have read the tsc _before_ writing to the vcpu_time_info
320 * page, _before_ we read the tsc.
321 *
322 * Further, if we switched pCPUs after reading the tsc
323 * timestamp but before reading the CPU's tsc, the hypervisor
324 * had better notify us by updating the version too and forcing
325 * us to retry the vCPU time read.
326 */
327 if (__predict_false(tsc < tsc_timestamp)) {
328 /*
329 * Notify the console that the CPU's tsc appeared to
330 * run behind Xen's idea of it, and pretend it hadn't.
331 */
332#if XEN_CLOCK_DEBUG /* XXX dtrace hook */
333 printf("xen cpu tsc %"PRIu64
334 " ran backwards from timestamp %"PRIu64
335 " by %"PRIu64"\n",
336 tsc, tsc_timestamp, tsc_timestamp - tsc);
337#endif
338 ci->ci_xen_cpu_tsc_backwards_evcnt.ev_count++;
339 delta_ns = delta_tsc = 0;
340 } else {
341 /* Find how far the CPU's tsc has advanced. */
342 delta_tsc = tsc - tsc_timestamp;
343
344 /* Convert the tsc delta to a nanosecond delta. */
345 delta_ns = xen_tsc_to_ns_delta(delta_tsc, tsc_to_system_mul,
346 tsc_shift);
347 }
348
349 /*
350 * Notify the console if the delta computation yielded a
351 * negative, and pretend it hadn't.
352 *
353 * This doesn't make sense but I include it out of paranoia.
354 */
355 if (__predict_false((int64_t)delta_ns < 0)) {
356#if XEN_CLOCK_DEBUG /* XXX dtrace hook */
357 printf("xen tsc delta in ns went negative: %"PRId64"\n",
358 delta_ns);
359#endif
360 ci->ci_xen_tsc_delta_negative_evcnt.ev_count++;
361 delta_ns = 0;
362 }
363
364 /*
365 * Compute the TSC-adjusted system time.
366 */
367 systime_ns = raw_systime_ns + delta_ns;
368
369 /*
370 * Notify the console if the addition wrapped around.
371 *
372 * This shouldn't happen because system time should be relative
373 * to a reasonable reference point, not centuries in the past.
374 * (2^64 ns is approximately half a millennium.)
375 */
376 if (__predict_false(systime_ns < raw_systime_ns)) {
377#if XEN_CLOCK_DEBUG /* XXX dtrace hook */
378 printf("xen raw systime + tsc delta wrapped around:"
379 " %"PRIu64" + %"PRIu64" = %"PRIu64"\n",
380 raw_systime_ns, delta_ns, systime_ns);
381#endif
382 ci->ci_xen_raw_systime_wraparound_evcnt.ev_count++;
383 }
384
385 /*
386 * Notify the console if the TSC-adjusted Xen system time
387 * appears to have gone backwards, and pretend we had gone
388 * forward. This seems to happen pretty regularly under load.
389 */
390 if (__predict_false(ci->ci_xen_last_systime_ns > systime_ns)) {
391#if XEN_CLOCK_DEBUG /* XXX dtrace hook */
392 printf("xen raw systime + tsc delta went backwards:"
393 " %"PRIu64" > %"PRIu64"\n",
394 ci->ci_xen_last_systime_ns, systime_ns);
395 printf(" raw_systime_ns=%"PRIu64"\n tsc_timestamp=%"PRIu64"\n"
396 " tsc=%"PRIu64"\n tsc_to_system_mul=%"PRIu32"\n"
397 " tsc_shift=%"PRId8"\n delta_tsc=%"PRIu64"\n"
398 " delta_ns=%"PRIu64"\n",
399 raw_systime_ns, tsc_timestamp, tsc, tsc_to_system_mul,
400 tsc_shift, delta_tsc, delta_ns);
401#endif
402 ci->ci_xen_raw_systime_backwards_evcnt.ev_count++;
403 systime_ns = ci->ci_xen_last_systime_ns + 1;
404 }
405
406 /* Remember the TSC-adjusted Xen system time. */
407 ci->ci_xen_last_systime_ns = systime_ns;
408
409 /* We had better not have migrated CPUs. */
410 KASSERT(ci == curcpu());
411
412 /* And we're done: return the TSC-adjusted systime in nanoseconds. */
413 return systime_ns;
414}

416/*
417 * xen_vcputime_raw_systime_ns()
418 *
419 * Return a snapshot of the current Xen system time to the
420 * resolution of the Xen hypervisor tick, in units of nanoseconds.
421 */
422static uint64_t
423xen_vcputime_raw_systime_ns(void)
424{
425 volatile struct vcpu_time_info *vt;
426 struct xen_vcputime_ticket ticket;
427 uint64_t raw_systime_ns;
428
429 do {
430 vt = xen_vcputime_enter(&ticket);
431 raw_systime_ns = vt->system_time;
432 } while (!xen_vcputime_exit(vt, &ticket));
433
434 return raw_systime_ns;
435}

437/*
438 * struct xen_wallclock_ticket
439 *
440 * State for a wall clock read section, during which a caller may
441 * read from the wall clock fields of HYPERVISOR_shared_info.
442 * Caller must enter with xen_wallclock_enter, exit with
443 * xen_wallclock_exit, and be prepared to retry if
444 * xen_wallclock_exit fails.
445 */
446struct xen_wallclock_ticket {
447 uint32_t version;
448};
449
450/*
451 * xen_wallclock_enter(tp)
452 *
453 * Enter a wall clock read section and store a ticket in *tp,
454 * which the caller must use with xen_wallclock_exit.
455 */
456static inline void
457xen_wallclock_enter(struct xen_wallclock_ticket *tp)
458{
459
460 while (__predict_false(1 & (tp->version =
461 HYPERVISOR_shared_info->wc_version)))
462 SPINLOCK_BACKOFF_HOOK;
463
464 /*
465 * Must read the version from memory before reading the
466 * timestamp from memory, as written potentially by another
467 * pCPU.
468 */
469 membar_consumer();
470}
471
472/*
473 * xen_wallclock_exit(tp)
474 *
475 * Exit a wall clock read section with the ticket in *tp from
476 * xen_wallclock_enter. Return true on success, false if caller
477 * must retry.
478 */
479static inline bool
480xen_wallclock_exit(struct xen_wallclock_ticket *tp)
481{
482
483 /*
484 * Must read the timestamp from memory before re-reading the
485 * version from memory, as written potentially by another pCPU.
486 */
487 membar_consumer();
488
489 return tp->version == HYPERVISOR_shared_info->wc_version;
490}

492/*
493 * xen_wallclock_time(tsp)
494 *
495 * Return a snapshot of the current low-resolution wall clock
496 * time, as reported by the hypervisor, in tsp.
497 */
498static void
499xen_wallclock_time(struct timespec *tsp)
500{
501 struct xen_wallclock_ticket ticket;
502 uint64_t systime_ns;
503
504 /* Read the last wall clock sample from the hypervisor. */
505 do {
506 xen_wallclock_enter(&ticket);
507 tsp->tv_sec = HYPERVISOR_shared_info->wc_sec;
508 tsp->tv_nsec = HYPERVISOR_shared_info->wc_nsec;
509 } while (!xen_wallclock_exit(&ticket));
510
511 /* Get the global system time. */
512 systime_ns = xen_global_systime_ns();
513
514 /* Add the system time to the wall clock time. */
515 systime_ns += tsp->tv_nsec;
516 tsp->tv_sec += systime_ns / 1000000000ull;
517 tsp->tv_nsec = systime_ns % 1000000000ull;
518}

520/*
521 * xen_global_systime_ns()
522 *
523 * Return a global monotonic view of the system time in
524 * nanoseconds, computed by the per-CPU Xen raw system time plus
525 * an rdtsc adjustment, and advance the view of the system time
526 * for all other CPUs.
527 */
528static uint64_t
529xen_global_systime_ns(void)
530{
531 struct cpu_info *ci;
532 uint64_t local, global, result;
533 int bound;
534
535 /*
536 * Find the local timecount on this CPU, and make sure it does
537 * not precede the latest global timecount witnessed so far by
538 * any CPU. If it does, add to the local CPU's skew from the
539 * fastest CPU.
540 *
541 * XXX Can we avoid retrying if the CAS fails?
542 */
543 bound = curlwp_bind();
544 ci = curcpu();
545 do {
546 local = xen_vcputime_systime_ns();
547 local += ci->ci_xen_systime_ns_skew;
548 global = xen_global_systime_ns_stamp;
549 if (__predict_false(local < global + 1)) {
550 result = global + 1;
551 ci->ci_xen_systime_ns_skew += global + 1 - local;
552 } else {
553 result = local;
554 }
555 } while (atomic_cas_64(&xen_global_systime_ns_stamp, global, result)
556 != global);
557 KASSERT(ci == curcpu());
558 curlwp_bindx(bound);
559
560 return result;
561}

595/*
596 * xen_rtc_set(todr, tv)
597 *
598 * Set the Xen wall clock time, if we can.
599 */
600static int
601xen_rtc_set(struct todr_chip_handle *todr, struct timeval *tvp)
602{
603#ifdef DOM0OPS
604 struct clock_ymdhms dt;
605 xen_platform_op_t op;
606 uint64_t systime_ns;
607
608 if (xendomain_is_privileged()) {
609 /* Convert to ymdhms and set the x86 ISA RTC. */
610 clock_secs_to_ymdhms(tvp->tv_sec, &dt);
611 rtc_set_ymdhms(NULL, &dt);
612
613 /* Get the global system time so we can preserve it. */
614 systime_ns = xen_global_systime_ns();
615
616 /* Set the hypervisor wall clock time. */
617 op.cmd = XENPF_settime;
618 op.u.settime.secs = tvp->tv_sec;
619 op.u.settime.nsecs = tvp->tv_usec * 1000;
620 op.u.settime.system_time = systime_ns;
621 return HYPERVISOR_platform_op(&op);
622 }
623#endif
624
625 /* XXX Should this fail if not on privileged dom0? */
626 return 0;
627}

629/*
630 * xen_delay(n)
631 *
632 * Wait approximately n microseconds.
633 */
634void
635xen_delay(unsigned n)
636{
637 int bound;
638
639 /* Bind to the CPU so we don't compare tsc on different CPUs. */
640 bound = curlwp_bind();
641
642 /* Short wait (<500us) or long wait? */
643 if (n < 500000) {
644 /*
645 * Xen system time is not precise enough for short
646 * delays, so use the tsc instead.
647 *
648 * We work with the current tsc frequency, and figure
649 * that if it changes while we're delaying, we've
650 * probably delayed long enough -- up to 500us.
651 *
652 * We do not use cpu_frequency(ci), which uses a
653 * quantity detected at boot time, and which may have
654 * changed by now if Xen has migrated this vCPU to
655 * another pCPU.
656 *
657 * XXX How long does it take to migrate pCPUs?
658 */
659 volatile struct vcpu_time_info *vt;
660 struct xen_vcputime_ticket ticket;
661 uint64_t tsc_start, last_tsc, tsc;
662 uint32_t tsc_to_system_mul;
663 int8_t tsc_shift;
664
665 /* Get the starting tsc and tsc frequency. */
666 do {
667 vt = xen_vcputime_enter(&ticket);
668 tsc_start = last_tsc = xen_rdtsc();
669 tsc_to_system_mul = vt->tsc_to_system_mul;
670 tsc_shift = vt->tsc_shift;
671 } while (!xen_vcputime_exit(vt, &ticket));
672
673 /*
674 * Wait until as many tsc ticks as there are in n
675 * microseconds have elapsed, or the tsc has gone
676 * backwards meaning we've probably migrated pCPUs.
677 */
678 for (;;) {
679 tsc = xen_rdtsc();
680 if (__predict_false(tsc < last_tsc))
681 break;
682 if (xen_tsc_to_ns_delta(tsc - tsc_start,
683 tsc_to_system_mul, tsc_shift)/1000 >= n)
684 break;
685 last_tsc = tsc;
686 }
687 } else {
688 /*
689 * Use the Xen system time for >=500us delays. From my
690 * testing, it seems to sometimes run backward by about
691 * 110us, which is not so bad.
692 */
693 uint64_t n_ns = 1000*(uint64_t)n;
694 uint64_t start_ns;
695
696 /* Get the start time. */
697 start_ns = xen_vcputime_raw_systime_ns();
698
699 /* Wait until the system time has passed the end. */
700 do {
701 HYPERVISOR_yield();
702 } while (xen_vcputime_raw_systime_ns() - start_ns < n_ns);
703 }
704
705 /* Unbind from the CPU if we weren't already bound. */
706 curlwp_bindx(bound);
707}

709/*
710 * xen_suspendclocks(ci)
711 *
712 * Stop handling the Xen timer event on the CPU of ci. Caller
713 * must be running on and bound to ci's CPU.
714 *
715 * Actually, caller must have kpreemption disabled, because that's
716 * easier to assert at the moment.
717 */
718void
719xen_suspendclocks(struct cpu_info *ci)
720{
721 int evtch;
722
723 KASSERT(ci == curcpu());
724 KASSERT(kpreempt_disabled());
725 KASSERT(ci->ci_xen_timer_intrhand != NULL);
726
727 evtch = unbind_virq_from_evtch(VIRQ_TIMER);
728 KASSERT(evtch != -1);
729
730 hypervisor_mask_event(evtch);
731 xen_intr_disestablish(ci->ci_xen_timer_intrhand);
732 ci->ci_xen_timer_intrhand = NULL;
733
734 aprint_verbose("Xen clock: removed event channel %d\n", evtch);
735
736 /* We'd better not have switched CPUs. */
737 KASSERT(ci == curcpu());
738}

740/*
741 * xen_resumeclocks(ci)
742 *
743 * Start handling the Xen timer event on the CPU of ci. Arm the
744 * Xen timer. Caller must be running on and bound to ci's CPU.
745 *
746 * Actually, caller must have kpreemption disabled, because that's
747 * easier to assert at the moment.
748 */
749void
750xen_resumeclocks(struct cpu_info *ci)
751{
752 char intr_xname[INTRDEVNAMEBUF];
753 int evtch;
754 int error __diagused;
755
756 KASSERT(ci == curcpu());
757 KASSERT(kpreempt_disabled());
758 KASSERT(ci->ci_xen_timer_intrhand == NULL);
759
760 evtch = bind_virq_to_evtch(VIRQ_TIMER);
761 KASSERT(evtch != -1);
762
763 snprintf(intr_xname, sizeof(intr_xname), "%s clock",
764 device_xname(ci->ci_dev));
765 /* XXX sketchy function pointer cast -- fix the API, please */
766 ci->ci_xen_timer_intrhand = xen_intr_establish_xname(-1, &xen_pic, evtch,
767 IST_LEVEL, IPL_CLOCK, (int (*)(void *))xen_timer_handler, ci, true,
768 intr_xname);
769 if (ci->ci_xen_timer_intrhand == NULL)
770 panic("failed to establish timer interrupt handler");
771
772 hypervisor_unmask_event(evtch);
773
774 aprint_verbose("Xen %s: using event channel %d\n", intr_xname, evtch);
775
776 /* Disarm the periodic timer on Xen>=3.1 which is allegedly buggy. */
777 if (XEN_MAJOR(xen_version) > 3 || XEN_MINOR(xen_version) > 0) {
778 error = HYPERVISOR_vcpu_op(VCPUOP_stop_periodic_timer,
779 ci->ci_cpuid, NULL);
780 KASSERT(error == 0);
781 }
782
783 /* Pretend the last hardclock happened right now. */
784 ci->ci_xen_hardclock_systime_ns = xen_vcputime_systime_ns();
785
786 /* Arm the one-shot timer. */
787 error = HYPERVISOR_set_timer_op(ci->ci_xen_hardclock_systime_ns +
788 NS_PER_TICK);
789 KASSERT(error == 0);
790
791 /* We'd better not have switched CPUs. */
792 KASSERT(ci == curcpu());
793}

795/*
796 * xen_timer_handler(cookie, frame)
797 *
798 * Periodic Xen timer event handler for NetBSD hardclock. Calls
799 * to this may get delayed, so we run hardclock as many times as
800 * we need to in order to cover the Xen system time that elapsed.
801 * After that, re-arm the timer to run again at the next tick.
802 * The cookie is the pointer to struct cpu_info.
803 */
804static int
805xen_timer_handler(void *cookie, struct clockframe *frame)
806{
807 struct cpu_info *ci = curcpu();
808 uint64_t last, now, delta, next;
809 int error;
810
811 KASSERT(cpu_intr_p());
812 KASSERT(cookie == ci);
813
814 frame = NULL; /* We use values cached in curcpu() */
815again:
816 /*
817 * Find how many nanoseconds of Xen system time has elapsed
818 * since the last hardclock tick.
819 */
820 last = ci->ci_xen_hardclock_systime_ns;
821 now = xen_vcputime_systime_ns();
822 if (now < last) {
823#if XEN_CLOCK_DEBUG /* XXX dtrace hook */
824 printf("xen systime ran backwards in hardclock %"PRIu64"ns\n",
825 last - now);
826#endif
827 ci->ci_xen_systime_backwards_hardclock_evcnt.ev_count++;
828 now = last;
829 }
830 delta = now - last;
831
832 /*
833 * Play hardclock catchup: run the hardclock timer as many
834 * times as appears necessary based on how much time has
835 * passed.
836 */
837 while (delta >= NS_PER_TICK) {
838 ci->ci_xen_hardclock_systime_ns += NS_PER_TICK;
839 delta -= NS_PER_TICK;
840 hardclock(frame);
841 if (__predict_false(delta >= NS_PER_TICK))
842 ci->ci_xen_missed_hardclock_evcnt.ev_count++;
843 }
844
845 /*
846 * Re-arm the timer. If it fails, it's probably because the
847 * time is in the past, so update our idea of what the Xen
848 * system time is and try again.
849 */
850 next = ci->ci_xen_hardclock_systime_ns + NS_PER_TICK;
851 error = HYPERVISOR_set_timer_op(next);
852 if (error)
853 goto again;
854
855 /* Success! */
856 return 0;
857}

859/*
860 * xen_initclocks()
861 *
862 * Initialize the Xen clocks on the current CPU.
863 */
864void
865xen_initclocks(void)
866{
867 struct cpu_info *ci = curcpu();
868
869 /* If this is the primary CPU, do global initialization first. */
870 if (ci == &cpu_info_primary) {
871 /* Initialize the systemwide Xen timecounter. */
872 tc_init(&xen_timecounter);
873
874#ifdef DOM0OPS
875 /*
876 * If this is a privileged dom0, start pushing the wall
877 * clock time back to the Xen hypervisor.
878 */
879 if (xendomain_is_privileged())
880 xen_timepush_init();
881#endif
882 }
883
884 /* Attach the event counters. */
885 evcnt_attach_dynamic(&ci->ci_xen_cpu_tsc_backwards_evcnt,
886 EVCNT_TYPE_INTR, NULL, device_xname(ci->ci_dev),
887 "cpu tsc ran backwards");
888 evcnt_attach_dynamic(&ci->ci_xen_tsc_delta_negative_evcnt,
889 EVCNT_TYPE_INTR, NULL, device_xname(ci->ci_dev),
890 "tsc delta went negative");
891 evcnt_attach_dynamic(&ci->ci_xen_raw_systime_wraparound_evcnt,
892 EVCNT_TYPE_INTR, NULL, device_xname(ci->ci_dev),
893 "raw systime wrapped around");
894 evcnt_attach_dynamic(&ci->ci_xen_raw_systime_backwards_evcnt,
895 EVCNT_TYPE_INTR, NULL, device_xname(ci->ci_dev),
896 "raw systime went backwards");
897 evcnt_attach_dynamic(&ci->ci_xen_systime_backwards_hardclock_evcnt,
898 EVCNT_TYPE_INTR, NULL, device_xname(ci->ci_dev),
899 "systime went backwards in hardclock");
900 evcnt_attach_dynamic(&ci->ci_xen_missed_hardclock_evcnt,
901 EVCNT_TYPE_INTR, NULL, device_xname(ci->ci_dev),
902 "missed hardclock");
903
904 /* Fire up the clocks. */
905 xen_resumeclocks(ci);
906}

910/*
911 * xen_timepush_init()
912 *
913 * Initialize callout to periodically set Xen hypervisor's wall
914 * clock time.
915 */
916static void
917xen_timepush_init(void)
918{
919 struct sysctllog *log = NULL;
920 const struct sysctlnode *node = NULL;
921 int error;
922
923 /* Start periodically updating the hypervisor's wall clock time. */
924 callout_init(&xen_timepush.ch, 0);
925 callout_setfunc(&xen_timepush.ch, xen_timepush_intr, NULL);
926
927 /* Pick a default frequency for timepush. */
928 xen_timepush.ticks = 53*hz + 3; /* avoid exact # of min/sec */
929
930 /* Create machdep.xen node. */
931 /* XXX Creation of the `machdep.xen' node should be elsewhere. */
932 error = sysctl_createv(&log, 0, NULL, &node, 0,
933 CTLTYPE_NODE, "xen",
934 SYSCTL_DESCR("Xen top level node"),
935 NULL, 0, NULL, 0,
936 CTL_MACHDEP, CTL_CREATE, CTL_EOL);
937 if (error)
938 goto fail;
939 KASSERT(node != NULL);
940
941 /* Create int machdep.xen.timepush_ticks knob. */
942 error = sysctl_createv(&log, 0, NULL, NULL, CTLFLAG_READWRITE,
943 CTLTYPE_INT, "timepush_ticks",
944 SYSCTL_DESCR("How often to update the hypervisor's time-of-day;"
945 " 0 to disable"),
946 sysctl_xen_timepush, 0, &xen_timepush.ticks, 0,
947 CTL_CREATE, CTL_EOL);
948 if (error)
949 goto fail;
950
951 /* Start the timepush callout. */
952 callout_schedule(&xen_timepush.ch, xen_timepush.ticks);
953
954 /* Success! */
955 return;
956
957fail: sysctl_teardown(&log);
958}

960/*
961 * xen_timepush_intr(cookie)
962 *
963 * Callout interrupt handler to push NetBSD's idea of the wall
964 * clock time, usually synchronized with NTP, back to the Xen
965 * hypervisor.
966 */
967static void
968xen_timepush_intr(void *cookie)
969{
970
971 resettodr();
972 if (xen_timepush.ticks)
973 callout_schedule(&xen_timepush.ch, xen_timepush.ticks);
974}
975
976/*
977 * sysctl_xen_timepush(...)
978 *
979 * Sysctl handler to set machdep.xen.timepush_ticks.
980 */
981static int
982sysctl_xen_timepush(SYSCTLFN_ARGS)
983{
984 struct sysctlnode node;
985 int ticks;
986 int error;
987
988 ticks = xen_timepush.ticks;
989 node = *rnode;
990 node.sysctl_data = &ticks;
991 error = sysctl_lookup(SYSCTLFN_CALL(&node));
992 if (error || newp == NULL)
993 return error;
994
995 if (ticks < 0)
996 return EINVAL;
997
998 if (ticks != xen_timepush.ticks) {
999 xen_timepush.ticks = ticks;
1000
1001 if (ticks == 0)
1002 callout_stop(&xen_timepush.ch);
1003 else
1004 callout_schedule(&xen_timepush.ch, ticks);
1005 }
1006
1007 return 0;
1008}
