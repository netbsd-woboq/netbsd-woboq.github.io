141/*
142 * We try to stick to the traditional x86 PIC semantics wrt Xen
143 * events.
144 *
145 * PIC pins exist in a global namespace which may be hierarchical, and
146 * are mapped to a cpu bus concept called 'IRQ' numbers, which are
147 * also global, but linear. Thus a PIC, pin tuple will always map to
148 * an IRQ number. These tuples can alias to the same IRQ number, thus
149 * causing IRQ "sharing". IRQ numbers can be bound to specific CPUs,
150 * and to specific callback vector indices on the CPU called idt_vec,
151 * which are aliases to handlers meant to run on destination
152 * CPUs. This binding can also happen at interrupt time and resolved
153 * 'round-robin' between all CPUs, depending on the lapic setup. In
154 * this case, all CPUs need to have identical idt_vec->handler
155 * mappings.
156 *
157 * The job of pic_addroute() is to setup the 'wiring' between the
158 * source pin, and the destination CPU handler, ideally on a specific
159 * CPU in MP systems (or 'round-robin').
160 *
161 * On Xen, a global namespace of 'events' exist, which are initially
162 * bound to nothing. This is similar to the relationship between
163 * realworld realworld IRQ numbers wrt PIC pins, since before routing,
164 * IRQ numbers by themselves have no causal connection setup with the
165 * real world. (Except for the hardwired cases on the PC Architecture,
166 * which we ignore for the purpose of this description). However the
167 * really important routing is from pin to idt_vec. On PIC_XEN, all
168 * three (pic, irq, idt_vec) belong to the same namespace and are
169 * identical. Further, the mapping between idt_vec and the actual
170 * callback handler is setup via calls to the evtchn.h api - this
171 * last bit is analogous to x86/idt.c:idt_vec_set() on real h/w
172 *
173 * For now we handle two cases:
174 * - IPC style events - eg: timer, PV devices, etc.
175 * - dom0 physical irq bound events.
176 *
177 * In the case of IPC style events, we currently externalise the
178 * event binding by using evtchn.h functions. From the POV of
179 * PIC_XEN , 'pin' , 'irq' and 'idt_vec' are all identical to the
180 * port number of the event.
181 *
182 * In the case of dom0 physical irq bound events, we currently
183 * event binding by exporting evtchn.h functions. From the POV of
184 * PIC_LAPIC/PIC_IOAPIC, the 'pin' is the hardware pin, the 'irq' is
185 * the x86 global irq number - the port number is extracted out of a
186 * global array (this is currently kludgy and breaks API abstraction)
187 * and the binding happens during pic_addroute() of the ioapic.
188 *
189 * Later when we integrate more tightly with x86/intr.c, we will be
190 * able to conform better to (PIC_LAPIC/PIC_IOAPIC)->PIC_XEN
191 * cascading model.
192 */

221void
222events_default_setup(void)
223{
224 int i;
225
226 /* No VCPU -> event mappings. */
227 for (i = 0; i < XEN_LEGACY_MAX_VCPUS; i++)
228 vcpu_ipi_to_evtch[i] = -1;
229
230 /* No VIRQ_TIMER -> event mappings. */
231 for (i = 0; i < XEN_LEGACY_MAX_VCPUS; i++)
232 virq_timer_to_evtch[i] = -1;
233
234 /* No VIRQ -> event mappings. */
235 for (i = 0; i < NR_VIRQS; i++)
236 virq_to_evtch[i] = -1;
237
238#if NPCI > 0 || NISA > 0
239 /* No PIRQ -> event mappings. */
240 for (i = 0; i < NR_PIRQS; i++)
241 pirq_to_evtch[i] = -1;
242 for (i = 0; i < NR_EVENT_CHANNELS / 32; i++)
243 pirq_needs_unmask_notify[i] = 0;
244#endif
245
246 /* No event-channel are 'live' right now. */
247 for (i = 0; i < NR_EVENT_CHANNELS; i++) {
248 evtsource[i] = NULL;
249 evtch_bindcount[i] = 0;
250 hypervisor_mask_event(i);
251 }
252
253}

255void
256events_init(void)
257{
258 mutex_init(&evtchn_lock, MUTEX_DEFAULT, IPL_NONE);
259#ifdef XENPV
260 debug_port = bind_virq_to_evtch(VIRQ_DEBUG);
261
262 KASSERT(debug_port != -1);
263
264 aprint_verbose("VIRQ_DEBUG interrupt using event channel %d\n",
265 debug_port);
266 /*
267 * Don't call event_set_handler(), we'll use a shortcut. Just set
268 * evtsource[] to a non-NULL value so that evtchn_do_event will
269 * be called.
270 */
271 evtsource[debug_port] = (void *)-1;
272 xen_atomic_set_bit(&curcpu()->ci_evtmask[0], debug_port);
273 hypervisor_unmask_event(debug_port);
274#if NPCI > 0 || NISA > 0
275 hypervisor_ack_pirq_event(debug_port);
276#endif /* NPCI > 0 || NISA > 0 */
277#endif /* XENPV */
278 x86_enable_intr(); /* at long last... */
279}

281bool
282events_suspend(void)
283{
284 int evtch;
285
286 x86_disable_intr();
287
288 /* VIRQ_DEBUG is the last interrupt to remove */
289 evtch = unbind_virq_from_evtch(VIRQ_DEBUG);
290
291 KASSERT(evtch != -1);
292
293 hypervisor_mask_event(evtch);
294 /* Remove the non-NULL value set in events_init() */
295 evtsource[evtch] = NULL;
296 aprint_verbose("VIRQ_DEBUG interrupt disabled, "
297 "event channel %d removed\n", evtch);
298
299 return true;
300}
301
302bool
303events_resume (void)
304{
305 events_init();
306
307 return true;
308}

311unsigned int
312evtchn_do_event(int evtch, struct intrframe *regs)
313{
314 struct cpu_info *ci;
315 int ilevel;
316 struct intrhand *ih;
317 int (*ih_fun)(void *, void *);
318 uint32_t iplmask;
319 int i;
320 uint32_t iplbit;
321
322 KASSERTMSG(evtch >= 0, "negative evtch: %d", evtch);
323 KASSERTMSG(evtch < NR_EVENT_CHANNELS,
324 "evtch number %d > NR_EVENT_CHANNELS", evtch);
325
326#ifdef IRQ_DEBUG
327 if (evtch == IRQ_DEBUG)
328 printf("evtchn_do_event: evtch %d\n", evtch);
329#endif
330 ci = curcpu();
331
332 /*
333 * Shortcut for the debug handler, we want it to always run,
334 * regardless of the IPL level.
335 */
336 if (__predict_false(evtch == debug_port)) {
337 xen_debug_handler(NULL);
338 hypervisor_unmask_event(debug_port);
339#if NPCI > 0 || NISA > 0
340 hypervisor_ack_pirq_event(debug_port);
341#endif /* NPCI > 0 || NISA > 0 */
342 return 0;
343 }
344
345 KASSERTMSG(evtsource[evtch] != NULL, "unknown event %d", evtch);
346 ci->ci_data.cpu_nintr++;
347 evtsource[evtch]->ev_evcnt.ev_count++;
348 ilevel = ci->ci_ilevel;
349
350 if (evtsource[evtch]->ev_cpu != ci /* XXX: get stats */) {
351 hypervisor_send_event(evtsource[evtch]->ev_cpu, evtch);
352 return 0;
353 }
354
355 if (evtsource[evtch]->ev_maxlevel <= ilevel) {
356#ifdef IRQ_DEBUG
357 if (evtch == IRQ_DEBUG)
358 printf("evtsource[%d]->ev_maxlevel %d <= ilevel %d\n",
359 evtch, evtsource[evtch]->ev_maxlevel, ilevel);
360#endif
361 hypervisor_set_ipending(evtsource[evtch]->ev_imask,
362 evtch >> LONG_SHIFT,
363 evtch & LONG_MASK);
364
365 /* leave masked */
366
367 return 0;
368 }
369 ci->ci_ilevel = evtsource[evtch]->ev_maxlevel;
370 iplmask = evtsource[evtch]->ev_imask;
371 x86_enable_intr();
372 mutex_spin_enter(&evtlock[evtch]);
373 ih = evtsource[evtch]->ev_handlers;
374 while (ih != NULL) {
375 if (ih->ih_cpu != ci) {
376 hypervisor_send_event(ih->ih_cpu, evtch);
377 iplmask &= ~XUNMASK(ci, ih->ih_level);
378 ih = ih->ih_evt_next;
379 continue;
380 }
381 if (ih->ih_level <= ilevel) {
382#ifdef IRQ_DEBUG
383 if (evtch == IRQ_DEBUG)
384 printf("ih->ih_level %d <= ilevel %d\n", ih->ih_level, ilevel);
385#endif
386 x86_disable_intr();
387 hypervisor_set_ipending(iplmask,
388 evtch >> LONG_SHIFT, evtch & LONG_MASK);
389 /* leave masked */
390 mutex_spin_exit(&evtlock[evtch]);
391 goto splx;
392 }
393 iplmask &= ~XUNMASK(ci, ih->ih_level);
394 ci->ci_ilevel = ih->ih_level;
395 ih_fun = (void *)ih->ih_fun;
396 ih_fun(ih->ih_arg, regs);
397 ih = ih->ih_evt_next;
398 }
399 mutex_spin_exit(&evtlock[evtch]);
400 x86_disable_intr();
401 hypervisor_unmask_event(evtch);
402#if NPCI > 0 || NISA > 0
403 hypervisor_ack_pirq_event(evtch);
404#endif /* NPCI > 0 || NISA > 0 */
405
406splx:
407 /*
408 * C version of spllower(). ASTs will be checked when
409 * hypevisor_callback() exits, so no need to check here.
410 */
411 iplmask = (XUNMASK(ci, ilevel) & ci->ci_xpending);
412 while (iplmask != 0) {
413 iplbit = 1 << (NIPL - 1);
414 i = (NIPL - 1);
415 while (iplmask != 0 && i > ilevel) {
416 while (iplmask & iplbit) {
417 ci->ci_xpending &= ~iplbit;
418 ci->ci_ilevel = i;
419 for (ih = ci->ci_xsources[i]->is_handlers;
420 ih != NULL; ih = ih->ih_next) {
421 KASSERT(ih->ih_cpu == ci);
422 x86_enable_intr();
423 ih_fun = (void *)ih->ih_fun;
424 ih_fun(ih->ih_arg, regs);
425 x86_disable_intr();
426 }
427 hypervisor_enable_ipl(i);
428 /* more pending IPLs may have been registered */
429 iplmask =
430 (XUNMASK(ci, ilevel) & ci->ci_xpending);
431 }
432 i--;
433 iplbit >>= 1;
434 }
435 }
436 ci->ci_ilevel = ilevel;
437 return 0;
438}

469static void
470xen_evtchn_addroute(struct pic *pic, struct cpu_info *ci, int pin, int idt_vec, int type)
471{
472
473 evtchn_port_t evtchn = pin;
474
475 /* Events are simulated as level triggered interrupts */
476 KASSERT(type == IST_LEVEL);
477
478 KASSERT(evtchn < NR_EVENT_CHANNELS);
479#if notyet
480 evtchn_port_t boundport = idt_vec;
481#endif
482
483 KASSERT(pic->pic_type == PIC_XEN);
484
485 xen_atomic_set_bit(&ci->ci_evtmask[0], evtchn);
486
487}

489static void
490xen_evtchn_delroute(struct pic *pic, struct cpu_info *ci, int pin, int idt_vec, int type)
491{
492 /*
493 * XXX: In the future, this is a great place to
494 * 'unbind' events to underlying events and cpus.
495 * For now, just disable interrupt servicing on this cpu for
496 * this pin aka cpu.
497 */
498 evtchn_port_t evtchn = pin;
499
500 /* Events are simulated as level triggered interrupts */
501 KASSERT(type == IST_LEVEL);
502
503 KASSERT(evtchn < NR_EVENT_CHANNELS);
504#if notyet
505 evtchn_port_t boundport = idt_vec;
506#endif
507
508 KASSERT(pic->pic_type == PIC_XEN);
509
510 xen_atomic_clear_bit(&ci->ci_evtmask[0], evtchn);
511}

513/*
514 * xen_evtchn_trymask(pic, pin)
515 *
516 * If there are interrupts pending on the bus-shared pic, return
517 * false. Otherwise, mask interrupts on the bus-shared pic and
518 * return true.
519 */
520static bool
521xen_evtchn_trymask(struct pic *pic, int pin)
522{
523 volatile struct shared_info *s = HYPERVISOR_shared_info;
524 unsigned long masked __diagused;
525
526 /* Mask it. */
527 masked = xen_atomic_test_and_set_bit(&s->evtchn_mask[0], pin);
528
529 /*
530 * Caller is responsible for calling trymask only when the
531 * interrupt pin is not masked, and for serializing calls to
532 * trymask.
533 */
534 KASSERT(!masked);
535
536 /*
537 * Check whether there were any interrupts pending when we
538 * masked it. If there were, unmask and abort.
539 */
540 if (xen_atomic_test_bit(&s->evtchn_pending[0], pin)) {
541 xen_atomic_clear_bit(&s->evtchn_mask[0], pin);
542 return false;
543 }
544
545 /* Success: masked, not pending. */
546 return true;
547}

549evtchn_port_t
550bind_vcpu_to_evtch(cpuid_t vcpu)
551{
552 evtchn_op_t op;
553 evtchn_port_t evtchn;
554
555 mutex_spin_enter(&evtchn_lock);
556
557 evtchn = vcpu_ipi_to_evtch[vcpu];
558 if (evtchn == -1) {
559 op.cmd = EVTCHNOP_bind_ipi;
560 op.u.bind_ipi.vcpu = (uint32_t) vcpu;
561 if (HYPERVISOR_event_channel_op(&op) != 0)
562 panic("Failed to bind ipi to VCPU %"PRIuCPUID"\n", vcpu);
563 evtchn = op.u.bind_ipi.port;
564
565 vcpu_ipi_to_evtch[vcpu] = evtchn;
566 }
567
568 evtch_bindcount[evtchn]++;
569
570 mutex_spin_exit(&evtchn_lock);
571
572 return evtchn;
573}
574
575int
576bind_virq_to_evtch(int virq)
577{
578 evtchn_op_t op;
579 int evtchn;
580
581 mutex_spin_enter(&evtchn_lock);
582
583 /*
584 * XXX: The only per-cpu VIRQ we currently use is VIRQ_TIMER.
585 * Please re-visit this implementation when others are used.
586 * Note: VIRQ_DEBUG is special-cased, and not used or bound on APs.
587 * XXX: event->virq/ipi can be unified in a linked-list
588 * implementation.
589 */
590 struct cpu_info *ci = curcpu();
591
592 if (virq == VIRQ_DEBUG && ci != &cpu_info_primary) {
593 mutex_spin_exit(&evtchn_lock);
594 return -1;
595 }
596
597 if (virq == VIRQ_TIMER) {
598 evtchn = virq_timer_to_evtch[ci->ci_cpuid];
599 } else {
600 evtchn = virq_to_evtch[virq];
601 }
602
603 /* Allocate a channel if there is none already allocated */
604 if (evtchn == -1) {
605 op.cmd = EVTCHNOP_bind_virq;
606 op.u.bind_virq.virq = virq;
607 op.u.bind_virq.vcpu = ci->ci_cpuid;
608 if (HYPERVISOR_event_channel_op(&op) != 0)
609 panic("Failed to bind virtual IRQ %d\n", virq);
610 evtchn = op.u.bind_virq.port;
611 }
612
613 /* Set event channel */
614 if (virq == VIRQ_TIMER) {
615 virq_timer_to_evtch[ci->ci_cpuid] = evtchn;
616 } else {
617 virq_to_evtch[virq] = evtchn;
618 }
619
620 /* Increase ref counter */
621 evtch_bindcount[evtchn]++;
622
623 mutex_spin_exit(&evtchn_lock);
624
625 return evtchn;
626}
627
628int
629unbind_virq_from_evtch(int virq)
630{
631 evtchn_op_t op;
632 int evtchn;
633
634 struct cpu_info *ci = curcpu();
635
636 if (virq == VIRQ_TIMER) {
637 evtchn = virq_timer_to_evtch[ci->ci_cpuid];
638 }
639 else {
640 evtchn = virq_to_evtch[virq];
641 }
642
643 if (evtchn == -1) {
644 return -1;
645 }
646
647 mutex_spin_enter(&evtchn_lock);
648
649 evtch_bindcount[evtchn]--;
650 if (evtch_bindcount[evtchn] == 0) {
651 op.cmd = EVTCHNOP_close;
652 op.u.close.port = evtchn;
653 if (HYPERVISOR_event_channel_op(&op) != 0)
654 panic("Failed to unbind virtual IRQ %d\n", virq);
655
656 if (virq == VIRQ_TIMER) {
657 virq_timer_to_evtch[ci->ci_cpuid] = -1;
658 } else {
659 virq_to_evtch[virq] = -1;
660 }
661 }
662
663 mutex_spin_exit(&evtchn_lock);
664
665 return evtchn;
666}

689int
690bind_pirq_to_evtch(int pirq)
691{
692 evtchn_op_t op;
693 int evtchn;
694
695 if (pirq >= NR_PIRQS) {
696 panic("pirq %d out of bound, increase NR_PIRQS", pirq);
697 }
698
699 mutex_spin_enter(&evtchn_lock);
700
701 evtchn = pirq_to_evtch[pirq];
702 if (evtchn == -1) {
703 op.cmd = EVTCHNOP_bind_pirq;
704 op.u.bind_pirq.pirq = pirq;
705 op.u.bind_pirq.flags = BIND_PIRQ__WILL_SHARE;
706 if (HYPERVISOR_event_channel_op(&op) != 0)
707 panic("Failed to bind physical IRQ %d\n", pirq);
708 evtchn = op.u.bind_pirq.port;
709
710#ifdef IRQ_DEBUG
711 printf("pirq %d evtchn %d\n", pirq, evtchn);
712#endif
713 pirq_to_evtch[pirq] = evtchn;
714 }
715
716 evtch_bindcount[evtchn]++;
717
718 mutex_spin_exit(&evtchn_lock);
719
720 return evtchn;
721}

723int
724unbind_pirq_from_evtch(int pirq)
725{
726 evtchn_op_t op;
727 int evtchn = pirq_to_evtch[pirq];
728
729 mutex_spin_enter(&evtchn_lock);
730
731 evtch_bindcount[evtchn]--;
732 if (evtch_bindcount[evtchn] == 0) {
733 op.cmd = EVTCHNOP_close;
734 op.u.close.port = evtchn;
735 if (HYPERVISOR_event_channel_op(&op) != 0)
736 panic("Failed to unbind physical IRQ %d\n", pirq);
737
738 pirq_to_evtch[pirq] = -1;
739 }
740
741 mutex_spin_exit(&evtchn_lock);
742
743 return evtchn;
744}

806/*
807 * Recalculate the interrupt from scratch for an event source.
808 */
809static void
810intr_calculatemasks(struct evtsource *evts, int evtch, struct cpu_info *ci)
811{
812 struct intrhand *ih;
813 int cpu_receive = 0;
814
815#ifdef MULTIPROCESSOR
816 KASSERT(!mutex_owned(&evtlock[evtch]));
817#endif
818 mutex_spin_enter(&evtlock[evtch]);
819 evts->ev_maxlevel = IPL_NONE;
820 evts->ev_imask = 0;
821 for (ih = evts->ev_handlers; ih != NULL; ih = ih->ih_evt_next) {
822 if (ih->ih_level > evts->ev_maxlevel)
823 evts->ev_maxlevel = ih->ih_level;
824 evts->ev_imask |= (1 << ih->ih_level);
825 if (ih->ih_cpu == ci)
826 cpu_receive = 1;
827 }
828 if (cpu_receive)
829 xen_atomic_set_bit(&curcpu()->ci_evtmask[0], evtch);
830 else
831 xen_atomic_clear_bit(&curcpu()->ci_evtmask[0], evtch);
832 mutex_spin_exit(&evtlock[evtch]);
833}

835int
836event_set_handler(int evtch, int (*func)(void *), void *arg, int level,
837 const char *intrname, const char *xname)
838{
839 struct cpu_info *ci = curcpu(); /* XXX: pass in ci ? */
840 struct evtsource *evts;
841 struct intrhand *ih, **ihp;
842 int s;
843#ifdef MULTIPROCESSOR
844 bool mpsafe = (level != IPL_VM);
845#endif /* MULTIPROCESSOR */
846
847#ifdef IRQ_DEBUG
848 printf("event_set_handler IRQ %d handler %p\n", evtch, func);
849#endif
850
851 KASSERTMSG(evtch >= 0, "negative evtch: %d", evtch);
852 KASSERTMSG(evtch < NR_EVENT_CHANNELS,
853 "evtch number %d > NR_EVENT_CHANNELS", evtch);
854 KASSERT(intrname != NULL && xname != NULL);
855
856#if 0
857 printf("event_set_handler evtch %d handler %p level %d\n", evtch,
858 handler, level);
859#endif
860 ih = kmem_zalloc(sizeof (struct intrhand), KM_NOSLEEP);
861 if (ih == NULL)
862 panic("can't allocate fixed interrupt source");
863
864
865 ih->ih_level = level;
866 ih->ih_fun = ih->ih_realfun = func;
867 ih->ih_arg = ih->ih_realarg = arg;
868 ih->ih_evt_next = NULL;
869 ih->ih_next = NULL;
870 ih->ih_cpu = ci;
871#ifdef MULTIPROCESSOR
872 if (!mpsafe) {
873 ih->ih_fun = xen_intr_biglock_wrapper;
874 ih->ih_arg = ih;
875 }
876#endif /* MULTIPROCESSOR */
877
878 s = splhigh();
879
880 /* register per-cpu handler for spllower() */
881 event_set_iplhandler(ci, ih, level);
882
883 /* register handler for event channel */
884 if (evtsource[evtch] == NULL) {
885 evts = kmem_zalloc(sizeof (struct evtsource),
886 KM_NOSLEEP);
887 if (evts == NULL)
888 panic("can't allocate fixed interrupt source");
889
890 evts->ev_handlers = ih;
891 /*
892 * XXX: We're assuming here that ci is the same cpu as
893 * the one on which this event/port is bound on. The
894 * api needs to be reshuffled so that this assumption
895 * is more explicitly implemented.
896 */
897 evts->ev_cpu = ci;
898 mutex_init(&evtlock[evtch], MUTEX_DEFAULT, IPL_HIGH);
899 evtsource[evtch] = evts;
900 strlcpy(evts->ev_intrname, intrname, sizeof(evts->ev_intrname));
901
902 evcnt_attach_dynamic(&evts->ev_evcnt, EVCNT_TYPE_INTR, NULL,
903 device_xname(ci->ci_dev), evts->ev_intrname);
904 } else {
905 evts = evtsource[evtch];
906 /* sort by IPL order, higher first */
907 mutex_spin_enter(&evtlock[evtch]);
908 for (ihp = &evts->ev_handlers; ; ihp = &((*ihp)->ih_evt_next)) {
909 if ((*ihp)->ih_level < ih->ih_level) {
910 /* insert before *ihp */
911 ih->ih_evt_next = *ihp;
912 *ihp = ih;
913 break;
914 }
915 if ((*ihp)->ih_evt_next == NULL) {
916 (*ihp)->ih_evt_next = ih;
917 break;
918 }
919 }
920 mutex_spin_exit(&evtlock[evtch]);
921 }
922
923
924 // append device name
925 if (evts->ev_xname[0] != '\0')
926 strlcat(evts->ev_xname, ", ", sizeof(evts->ev_xname));
927 strlcat(evts->ev_xname, xname, sizeof(evts->ev_xname));
928
929 intr_calculatemasks(evts, evtch, ci);
930 splx(s);
931
932 return 0;
933}

936event_set_iplhandler(struct cpu_info *ci,
937 struct intrhand *ih,
938 int level)
939{
940 struct intrsource *ipls;
941
942 KASSERT(ci == ih->ih_cpu);
943 if (ci->ci_xsources[level] == NULL) {
944 ipls = kmem_zalloc(sizeof (struct intrsource),
945 KM_NOSLEEP);
946 if (ipls == NULL)
947 panic("can't allocate fixed interrupt source");
948 ipls->is_recurse = xenev_stubs[level].ist_entry;
949 ipls->is_recurse = xenev_stubs[level].ist_recurse;
950 ipls->is_resume = xenev_stubs[level].ist_resume;
951 ipls->is_handlers = ih;
952 ci->ci_xsources[level] = ipls;
953 } else {
954 ipls = ci->ci_xsources[level];
955 ih->ih_next = ipls->is_handlers;
956 ipls->is_handlers = ih;
957 }

960int
961event_remove_handler(int evtch, int (*func)(void *), void *arg)
962{
963 struct intrsource *ipls;
964 struct evtsource *evts;
965 struct intrhand *ih;
966 struct intrhand **ihp;
967 struct cpu_info *ci;
968
969 evts = evtsource[evtch];
970 if (evts == NULL)
971 return ENOENT;
972
973 mutex_spin_enter(&evtlock[evtch]);
974 for (ihp = &evts->ev_handlers, ih = evts->ev_handlers;
975 ih != NULL;
976 ihp = &ih->ih_evt_next, ih = ih->ih_evt_next) {
977 if (ih->ih_realfun == func && ih->ih_realarg == arg)
978 break;
979 }
980 if (ih == NULL) {
981 mutex_spin_exit(&evtlock[evtch]);
982 return ENOENT;
983 }
984 ci = ih->ih_cpu;
985 *ihp = ih->ih_evt_next;
986
987 ipls = ci->ci_xsources[ih->ih_level];
988 for (ihp = &ipls->is_handlers, ih = ipls->is_handlers;
989 ih != NULL;
990 ihp = &ih->ih_next, ih = ih->ih_next) {
991 if (ih->ih_realfun == func && ih->ih_realarg == arg)
992 break;
993 }
994 if (ih == NULL)
995 panic("event_remove_handler");
996 *ihp = ih->ih_next;
997 mutex_spin_exit(&evtlock[evtch]);
998 kmem_free(ih, sizeof (struct intrhand));
999 if (evts->ev_handlers == NULL) {
1000 xen_atomic_clear_bit(&ci->ci_evtmask[0], evtch);
1001 evcnt_detach(&evts->ev_evcnt);
1002 kmem_free(evts, sizeof (struct evtsource));
1003 evtsource[evtch] = NULL;
1004 } else {
1005 intr_calculatemasks(evts, evtch, ci);
1006 }
1007 return 0;
1008}

1157/*
1158 * MI interface for subr_interrupt.
1159 */
1160struct intrids_handler *
1161interrupt_construct_intrids(const kcpuset_t *cpuset)
1162{
1163 struct intrids_handler *ii_handler;
1164 intrid_t *ids;
1165 int i, count, off;
1166 struct evtsource *evp;
1167
1168 if (kcpuset_iszero(cpuset))
1169 return 0;
1170
1171 /*
1172 * Count the number of interrupts which affinity to any cpu of "cpuset".
1173 */
1174 count = 0;
1175 for (i = 0; i < NR_EVENT_CHANNELS; i++) {
1176 evp = evtsource[i];
1177
1178 if (evp == NULL || i == debug_port)
1179 continue;
1180
1181 if (!kcpuset_isset(cpuset, cpu_index(evp->ev_cpu)))
1182 continue;
1183
1184 count++;
1185 }
1186
1187 ii_handler = kmem_zalloc(sizeof(int) + sizeof(intrid_t) * count,
1188 KM_SLEEP);
1189 if (ii_handler == NULL)
1190 return NULL;
1191 ii_handler->iih_nids = count;
1192 if (count == 0)
1193 return ii_handler;
1194
1195 ids = ii_handler->iih_intrids;
1196 mutex_spin_enter(&evtchn_lock);
1197 for (i = 0, off = 0; i < NR_EVENT_CHANNELS && off < count; i++) {
1198 evp = evtsource[i];
1199
1200 if (evp == NULL || i == debug_port)
1201 continue;
1202
1203 if (!kcpuset_isset(cpuset, cpu_index(evp->ev_cpu)))
1204 continue;
1205
1206 snprintf(ids[off], sizeof(intrid_t), "%s", evp->ev_intrname);
1207 off++;
1208 }
1209 mutex_spin_exit(&evtchn_lock);
1210
1211 return ii_handler;
1212}
