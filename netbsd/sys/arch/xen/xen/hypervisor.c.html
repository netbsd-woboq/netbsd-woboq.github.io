195/*
196 * Probe for the hypervisor; always succeeds.
197 */
198int
199hypervisor_match(device_t parent, cfdata_t match, void *aux)
200{
201 struct hypervisor_attach_args *haa = aux;
202
203 /* Attach path sanity check */
204 if (strncmp(haa->haa_busname, "hypervisor", sizeof("hypervisor")) != 0)
205 return 0;
206
207#if defined(XENPVHVM)
208 /*
209 * The strategy here is to setup hypercall and all PVHVM
210 * interfaces on match, or fail to match.
211 * Ideally this should happen under attach, but it's too late
212 * then and there's no way to bailout.
213 *
214 * If match fails, hypervisor does not attach, and the domain
215 * can boot with the minimal PC AT ISA configuration just
216 * enough to attach wd(4) and mount the rootfs.
217 */
218 int vec;
219 extern vaddr_t hypercall_page;
220
221 if (vm_guest == VM_GUEST_XEN) {
222 aprint_normal("%s: Identified Guest XEN in HVM mode.\n",
223 haa->haa_busname);
224
225 u_int descs[4];
226 x86_cpuid(XEN_CPUID_LEAF(2), descs);
227
228 /*
229 * Given 32 bytes per hypercall stub, and an optimistic number
230 * of 100 hypercalls ( the current max is 55), there shouldn't
231 * be any reason to spill over the arbitrary number of 1
232 * hypercall page. This is what we allocate in locore.S
233 * anyway. Make sure the allocation matches the registration.
234 */
235
236 KASSERT(descs[0] == 1);
237
238 /* XXX: vtophys(&hypercall_page) */
239 wrmsr(descs[1], (uintptr_t)&hypercall_page - KERNBASE);
240
241 vm_guest = VM_GUEST_XENPVHVM; /* Be more specific */
242
243 } else {
244 aprint_normal("%s: Xen HVM mode not identified. Exiting.\n",
245 haa->haa_busname);
246 return 0;
247 }
248
249 if (-1 != HYPERVISOR_xen_version(XENVER_version, NULL)) {
250 printf("%s: detected functional hypercall page.\n",
251 haa->haa_busname);
252
253 xen_init_features();
254 }
255
256 /* Init various preset boot time data structures */
257
258 /* XEN xenstore shared page address, event channel */
259 struct xen_hvm_param xen_hvm_param;
260
261 xen_hvm_param.domid = DOMID_SELF;
262 xen_hvm_param.index = HVM_PARAM_STORE_PFN;
263
264 if ( HYPERVISOR_hvm_op(HVMOP_get_param, &xen_hvm_param) < 0) {
265 aprint_error("%s: Unable to obtain xenstore page address\n",
266 haa->haa_busname);
267 return 0;
268 }
269
270 /* Re-use PV field */
271 xen_start_info.store_mfn = xen_hvm_param.value;
272
273 pmap_kenter_pa((vaddr_t) xenstore_interface, ptoa(xen_start_info.store_mfn),
274 VM_PROT_READ|VM_PROT_WRITE, 0);
275
276 xen_hvm_param.domid = DOMID_SELF;
277 xen_hvm_param.index = HVM_PARAM_STORE_EVTCHN;
278
279 if ( HYPERVISOR_hvm_op(HVMOP_get_param, &xen_hvm_param) < 0) {
280 aprint_error("%s: Unable to obtain xenstore event channel\n",
281 haa->haa_busname);
282 return 0;
283 }
284
285 xen_start_info.store_evtchn = xen_hvm_param.value;
286
287 xen_hvm_param.domid = DOMID_SELF;
288 xen_hvm_param.index = HVM_PARAM_CONSOLE_PFN;
289
290 if ( HYPERVISOR_hvm_op(HVMOP_get_param, &xen_hvm_param) < 0) {
291 aprint_error("%s: Unable to obtain xencons page address\n",
292 haa->haa_busname);
293 return 0;
294 }
295
296 /* Re-use PV field */
297 xen_start_info.console.domU.mfn = xen_hvm_param.value;
298
299 pmap_kenter_pa((vaddr_t) xencons_interface, ptoa(xen_start_info.console.domU.mfn),
300 VM_PROT_READ|VM_PROT_WRITE, 0);
301
302 xen_hvm_param.domid = DOMID_SELF;
303 xen_hvm_param.index = HVM_PARAM_CONSOLE_EVTCHN;
304
305 if ( HYPERVISOR_hvm_op(HVMOP_get_param, &xen_hvm_param) < 0) {
306 aprint_error("%s: Unable to obtain xencons event channel\n",
307 haa->haa_busname);
308 return 0;
309 }
310
311 xen_start_info.console.domU.evtchn = xen_hvm_param.value;
312
313 /* HYPERVISOR_shared_info */
314 struct xen_add_to_physmap xmap = {
315 .domid = DOMID_SELF,
316 .space = XENMAPSPACE_shared_info,
317 .idx = 0, /* Important - XEN checks for this */
318 .gpfn = atop(HYPERVISOR_shared_info_pa)
319 };
320
321 if (HYPERVISOR_memory_op(XENMEM_add_to_physmap, &xmap) < 0) {
322 aprint_error("%s: Unable to register HYPERVISOR_shared_info\n",
323 haa->haa_busname);
324 return 0;
325 }
326
327 /* HYPERVISOR_shared_info va,pa has been allocated in pmap_bootstrap() */
328 pmap_kenter_pa((vaddr_t) HYPERVISOR_shared_info,
329 HYPERVISOR_shared_info_pa, VM_PROT_READ|VM_PROT_WRITE, 0);
330
331 cpu_info_primary.ci_vcpu = &HYPERVISOR_shared_info->vcpu_info[0];
332
333 /*
334 * First register callback: here's why
335 * http://xenbits.xen.org/gitweb/?p=xen.git;a=commit;h=7b5b8ca7dffde866d851f0b87b994e0b13e5b867
336 */
337
338 /*
339 * Check for XENFEAT_hvm_callback_vector. Can't proceed
340 * without it.
341 */
342 if (!xen_feature(XENFEAT_hvm_callback_vector)) {
343 aprint_error("%s: XENFEAT_hvm_callback_vector"
344 "not available, cannot proceed", haa->haa_busname);
345
346 return 0;
347 }
348
349 /* Register event callback handler. */
350
351 /* We don't really care where it is, as long as it's free */
352 vec = idt_vec_alloc(129, 255);
353
354 idt_vec_set(vec, &IDTVEC(hypervisor_pvhvm_callback));
355
356 cpu_init_idt(); /* XXX remove and use only native one below ? */
357
358 xen_hvm_param.domid = DOMID_SELF;
359 xen_hvm_param.index = HVM_PARAM_CALLBACK_IRQ;
360
361 /* val[63:56] = 2, val[7:0] = vec */
362 xen_hvm_param.value = ((int64_t)0x2 << 56) | vec;
363
364 if (HYPERVISOR_hvm_op(HVMOP_set_param, &xen_hvm_param) < 0) {
365 aprint_error("%s: Unable to register event callback vector\n",
366 haa->haa_busname);
367 return 0;
368 }
369
370 /* Print out value. */
371 xen_hvm_param.domid = DOMID_SELF;
372 xen_hvm_param.index = HVM_PARAM_CALLBACK_IRQ;
373 xen_hvm_param.value = 0;
374
375 if (HYPERVISOR_hvm_op(HVMOP_get_param, &xen_hvm_param) < 0) {
376 printf("%s: Unable to get event callback vector\n",
377 haa->haa_busname);
378 return 0;
379 }
380
381 /*
382 * Afterwards vector callback is done, register VCPU info
383 * page. Here's why:
384 * http://xenbits.xen.org/gitweb/?p=xen.git;a=commit;h=7b5b8ca7dffde866d851f0b87b994e0b13e5b867
385 * XXX: Ideally this should happen at vcpu attach.
386 */
387 struct vcpu_register_vcpu_info vrvi;
388
389 paddr_t vcpu_info_pa = HYPERVISOR_shared_info_pa +
390 offsetof(struct shared_info, vcpu_info);
391
392 vrvi.mfn = atop(vcpu_info_pa);
393 vrvi.offset = vcpu_info_pa - trunc_page(vcpu_info_pa);
394
395 if (HYPERVISOR_vcpu_op(VCPUOP_register_vcpu_info, curcpu()->ci_cpuid /* VCPU0 */,
396 &vrvi) < 0) {
397 aprint_error("%s: Unable to register vcpu info page\n",
398 haa->haa_busname);
399 return 0;
400 }
401
402 /*
403 * Set the boot device to xbd0a.
404 * We claim this is a reasonable default which is picked up
405 * later as the rootfs device.
406 *
407 * We need to do this because the HVM domain loader uses the
408 * regular BIOS based native boot(8) procedure, which sets the
409 * boot device to the native driver/partition of whatever was
410 * detected by the native bootloader.
411 */
412
413 struct btinfo_rootdevice bi;
414 snprintf(bi.devname, 6, "xbd0a");
415 bi.common.type = BTINFO_ROOTDEVICE;
416 bi.common.len = sizeof(struct btinfo_rootdevice);
417
418 /* From i386/multiboot.c */
419 /* $NetBSD: hypervisor.c,v 1.72 2019/02/14 08:18:26 cherry Exp $ */
420 int i, len;
421 vaddr_t data;
422 extern struct bootinfo bootinfo;
423 struct bootinfo *bip = (struct bootinfo *)&bootinfo;
424 len = bi.common.len;
425
426 data = (vaddr_t)&bip->bi_data;
427 for (i = 0; i < bip->bi_nentries; i++) {
428 struct btinfo_common *tmp;
429
430 tmp = (struct btinfo_common *)data;
431 data += tmp->len;
432 }
433 if (data + len < (vaddr_t)&bip->bi_data + sizeof(bip->bi_data)) {
434 memcpy((void *)data, &bi, len);
435 bip->bi_nentries++;
436 }
437
438#endif /* XENPVHVM */
439
440 /* If we got here, it must mean we matched */
441 return 1;
442}

453/*
454 * Attach the hypervisor.
455 */
456void
457hypervisor_attach(device_t parent, device_t self, void *aux)
458{
459
460#if NPCI >0
461#ifdef PCI_BUS_FIXUP
462 int pci_maxbus = 0;
463#endif
464#endif /* NPCI */
465 union hypervisor_attach_cookie hac;
466 char xen_extra_version[XEN_EXTRAVERSION_LEN];
467 static char xen_version_string[20];
468 int rc;
469 const struct sysctlnode *node = NULL;
470
471 xenkernfs_init();
472
473 xen_version = HYPERVISOR_xen_version(XENVER_version, NULL);
474 memset(xen_extra_version, 0, sizeof(xen_extra_version));
475 HYPERVISOR_xen_version(XENVER_extraversion, xen_extra_version);
476 rc = snprintf(xen_version_string, 20, "%d.%d%s", XEN_MAJOR(xen_version),
477 XEN_MINOR(xen_version), xen_extra_version);
478 aprint_normal(": Xen version %s\n", xen_version_string);
479 if (rc >= 20)
480 aprint_debug(": xen_version_string truncated\n");
481
482 sysctl_createv(NULL, 0, NULL, &node, 0,
483 CTLTYPE_NODE, "xen",
484 SYSCTL_DESCR("Xen top level node"),
485 NULL, 0, NULL, 0, CTL_MACHDEP, CTL_CREATE, CTL_EOL);
486
487 if (node != NULL) {
488 sysctl_createv(NULL, 0, &node, NULL, CTLFLAG_READONLY,
489 CTLTYPE_STRING, "version",
490 SYSCTL_DESCR("Xen hypervisor version"),
491 NULL, 0, xen_version_string, 0, CTL_CREATE, CTL_EOL);
492 }
493
494 aprint_verbose_dev(self, "features: ");
495#define XEN_TST_F(n) \
496 if (xen_feature(XENFEAT_##n)) \
497 aprint_verbose(" %s", #n);
498
499 XEN_TST_F(writable_page_tables);
500 XEN_TST_F(writable_descriptor_tables);
501 XEN_TST_F(auto_translated_physmap);
502 XEN_TST_F(supervisor_mode_kernel);
503 XEN_TST_F(pae_pgdir_above_4gb);
504 XEN_TST_F(mmu_pt_update_preserve_ad);
505 XEN_TST_F(highmem_assist);
506 XEN_TST_F(gnttab_map_avail_bits);
507 XEN_TST_F(hvm_callback_vector);
508 XEN_TST_F(hvm_safe_pvclock);
509 XEN_TST_F(hvm_pirqs);
510#undef XEN_TST_F
511 aprint_verbose("\n");
512
513 xengnt_init();
514 events_init();
515
516 memset(&hac, 0, sizeof(hac));
517 hac.hac_vcaa.vcaa_name = "vcpu";
518 hac.hac_vcaa.vcaa_caa.cpu_number = 0;
519 hac.hac_vcaa.vcaa_caa.cpu_role = CPU_ROLE_BP;
520 hac.hac_vcaa.vcaa_caa.cpu_func = NULL; /* See xen/x86/cpu.c:vcpu_attach() */
521 config_found_ia(self, "xendevbus", &hac.hac_vcaa, hypervisor_print);
522
523#ifdef MULTIPROCESSOR
524
525 /*
526 * The xenstore contains the configured number of vcpus.
527 * The xenstore however, is not accessible until much later in
528 * the boot sequence. We therefore bruteforce check for
529 * allocated vcpus (See: cpu.c:vcpu_match()) by iterating
530 * through the maximum supported by NetBSD MP.
531 */
532 cpuid_t vcpuid;
533
534 for (vcpuid = 1; vcpuid < maxcpus; vcpuid++) {
535 memset(&hac, 0, sizeof(hac));
536 hac.hac_vcaa.vcaa_name = "vcpu";
537 hac.hac_vcaa.vcaa_caa.cpu_number = vcpuid;
538 hac.hac_vcaa.vcaa_caa.cpu_role = CPU_ROLE_AP;
539 hac.hac_vcaa.vcaa_caa.cpu_func = NULL; /* See xen/x86/cpu.c:vcpu_attach() */
540 if (NULL == config_found_ia(self, "xendevbus", &hac.hac_vcaa,
541 hypervisor_vcpu_print)) {
542 break;
543 }
544 }
545
546#endif /* MULTIPROCESSOR */
547
548#if NXENBUS > 0
549 memset(&hac, 0, sizeof(hac));
550 hac.hac_xenbus.xa_device = "xenbus";
551 config_found_ia(self, "xendevbus", &hac.hac_xenbus, hypervisor_print);
552#endif
553#if NXENCONS > 0
554 memset(&hac, 0, sizeof(hac));
555 hac.hac_xencons.xa_device = "xencons";
556 config_found_ia(self, "xendevbus", &hac.hac_xencons, hypervisor_print);
557#endif
558#ifdef DOM0OPS
559#if NPCI > 0
560#if NACPICA > 0
561 if (acpi_present) {
562 memset(&hac, 0, sizeof(hac));
563 hac.hac_acpi.aa_iot = x86_bus_space_io;
564 hac.hac_acpi.aa_memt = x86_bus_space_mem;
565 hac.hac_acpi.aa_pc = NULL;
566 hac.hac_acpi.aa_pciflags =
567 PCI_FLAGS_IO_OKAY | PCI_FLAGS_MEM_OKAY |
568 PCI_FLAGS_MRL_OKAY | PCI_FLAGS_MRM_OKAY |
569 PCI_FLAGS_MWI_OKAY;
570 hac.hac_acpi.aa_ic = &x86_isa_chipset;
571 hac.hac_acpi.aa_dmat = &pci_bus_dma_tag;
572#ifdef _LP64
573 hac.hac_acpi.aa_dmat64 = &pci_bus_dma64_tag;
574#else
575 hac.hac_acpi.aa_dmat64 = NULL;
576#endif /* _LP64 */
577 config_found_ia(self, "acpibus", &hac.hac_acpi, 0);
578 }
579#endif /* NACPICA */
580 memset(&hac, 0, sizeof(hac));
581 hac.hac_pba.pba_iot = x86_bus_space_io;
582 hac.hac_pba.pba_memt = x86_bus_space_mem;
583 hac.hac_pba.pba_dmat = &pci_bus_dma_tag;
584#ifdef _LP64
585 hac.hac_pba.pba_dmat64 = &pci_bus_dma64_tag;
586#else
587 hac.hac_pba.pba_dmat64 = NULL;
588#endif /* _LP64 */
589 hac.hac_pba.pba_flags = PCI_FLAGS_MEM_OKAY | PCI_FLAGS_IO_OKAY;
590 hac.hac_pba.pba_bridgetag = NULL;
591 hac.hac_pba.pba_bus = 0;
592#if NACPICA > 0 && defined(ACPI_SCANPCI)
593 if (mpacpi_active)
594 mp_pci_scan(self, &hac.hac_pba, pcibusprint);
595 else
596#endif
597#if defined(MPBIOS) && defined(MPBIOS_SCANPCI)
598 if (mpbios_scanned != 0)
599 mp_pci_scan(self, &hac.hac_pba, pcibusprint);
600 else
601#endif
602 config_found_ia(self, "pcibus", &hac.hac_pba, pcibusprint);
603#if NACPICA > 0
604 if (mp_verbose)
605 acpi_pci_link_state();
606#endif
607#if NISA > 0
608 if (isa_has_been_seen == 0) {
609 memset(&hac, 0, sizeof(hac));
610 hac.hac_iba._iba_busname = "isa";
611 hac.hac_iba.iba_iot = x86_bus_space_io;
612 hac.hac_iba.iba_memt = x86_bus_space_mem;
613 hac.hac_iba.iba_dmat = &isa_bus_dma_tag;
614 hac.hac_iba.iba_ic = NULL; /* No isa DMA yet */
615 config_found_ia(self, "isabus", &hac.hac_iba, isabusprint);
616 }
617#endif /* NISA */
618#endif /* NPCI */
619
620 if (xendomain_is_privileged()) {
621 xenprivcmd_init();
622 xen_shm_init();
623 }
624#endif /* DOM0OPS */
625
626 hypervisor_machdep_attach();
627
628 if (!pmf_device_register(self, hypervisor_suspend, hypervisor_resume))
629 aprint_error_dev(self, "couldn't establish power handler\n");
630
631}
