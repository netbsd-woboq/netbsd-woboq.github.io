52/*
53 * This file contains the xennet frontend code required for the network
54 * communication between two Xen domains.
55 * It ressembles xbd, but is a little more complex as it must deal with two
56 * rings:
57 * - the TX ring, to transmit packets to backend (inside => outside)
58 * - the RX ring, to receive packets from backend (outside => inside)
59 *
60 * Principles are following.
61 *
62 * For TX:
63 * Purpose is to transmit packets to the outside. The start of day is in
64 * xennet_start() (default output routine of xennet) that schedules a softint,
65 * xennet_softstart(). xennet_softstart() generates the requests associated
66 * to the TX mbufs queued (see altq(9)).
67 * The backend's responses are processed by xennet_tx_complete(), called either
68 * from:
69 * - xennet_start()
70 * - xennet_handler(), during an asynchronous event notification from backend
71 * (similar to an IRQ).
72 *
73 * for RX:
74 * Purpose is to process the packets received from the outside. RX buffers
75 * are pre-allocated through xennet_alloc_rx_buffer(), during xennet autoconf
76 * attach. During pre-allocation, frontend pushes requests in the I/O ring, in
77 * preparation for incoming packets from backend.
78 * When RX packets need to be processed, backend takes the requests previously
79 * offered by frontend and pushes the associated responses inside the I/O ring.
80 * When done, it notifies frontend through an event notification, which will
81 * asynchronously call xennet_handler() in frontend.
82 * xennet_handler() processes the responses, generates the associated mbuf, and
83 * passes it to the MI layer for further processing.
84 */

260static int
261xennet_xenbus_match(device_t parent, cfdata_t match, void *aux)
262{
263 struct xenbusdev_attach_args *xa = aux;
264
265 if (strcmp(xa->xa_type, "vif") != 0)
266 return 0;
267
268 if (match->cf_loc[XENBUSCF_ID] != XENBUSCF_ID_DEFAULT &&
269 match->cf_loc[XENBUSCF_ID] != xa->xa_id)
270 return 0;
271
272 return 1;

275static void
276xennet_xenbus_attach(device_t parent, device_t self, void *aux)
277{
278 struct xennet_xenbus_softc *sc = device_private(self);
279 struct xenbusdev_attach_args *xa = aux;
280 struct ifnet *ifp = &sc->sc_ethercom.ec_if;
281 int err;
282 netif_tx_sring_t *tx_ring;
283 netif_rx_sring_t *rx_ring;
284 RING_IDX i;
285 char *val, *e, *p;
286 int s;
287 extern int ifqmaxlen; /* XXX */
288#ifdef XENNET_DEBUG
289 char **dir;
290 int dir_n = 0;
291 char id_str[20];
292#endif
293
294 aprint_normal(": Xen Virtual Network Interface\n");
295 sc->sc_dev = self;
296
297#ifdef XENNET_DEBUG
298 printf("path: %s\n", xa->xa_xbusd->xbusd_path);
299 snprintf(id_str, sizeof(id_str), "%d", xa->xa_id);
300 err = xenbus_directory(NULL, "device/vif", id_str, &dir_n, &dir);
301 if (err) {
302 aprint_error_dev(self, "xenbus_directory err %d\n", err);
303 } else {
304 printf("%s/\n", xa->xa_xbusd->xbusd_path);
305 for (i = 0; i < dir_n; i++) {
306 printf("\t/%s", dir[i]);
307 err = xenbus_read(NULL, xa->xa_xbusd->xbusd_path,
308 dir[i], NULL, &val);
309 if (err) {
310 aprint_error_dev(self, "xenbus_read err %d\n",
311 err);
312 } else {
313 printf(" = %s\n", val);
314 free(val, M_DEVBUF);
315 }
316 }
317 }
318#endif /* XENNET_DEBUG */
319 sc->sc_xbusd = xa->xa_xbusd;
320 sc->sc_xbusd->xbusd_otherend_changed = xennet_backend_changed;
321
322 /* xenbus ensure 2 devices can't be probed at the same time */
323 if (if_xennetrxbuf_cache_inited == 0) {
324 if_xennetrxbuf_cache = pool_cache_init(PAGE_SIZE, 0, 0, 0,
325 "xnfrx", NULL, IPL_VM, NULL, NULL, NULL);
326 if_xennetrxbuf_cache_inited = 1;
327 }
328
329 /* initialize free RX and RX request lists */
330 mutex_init(&sc->sc_tx_lock, MUTEX_DEFAULT, IPL_NET);
331 SLIST_INIT(&sc->sc_txreq_head);
332 for (i = 0; i < NET_TX_RING_SIZE; i++) {
333 sc->sc_txreqs[i].txreq_id = i;
334 SLIST_INSERT_HEAD(&sc->sc_txreq_head, &sc->sc_txreqs[i],
335 txreq_next);
336 }
337 mutex_init(&sc->sc_rx_lock, MUTEX_DEFAULT, IPL_NET);
338 SLIST_INIT(&sc->sc_rxreq_head);
339 s = splvm(); /* XXXSMP */
340 for (i = 0; i < NET_RX_RING_SIZE; i++) {
341 struct xennet_rxreq *rxreq = &sc->sc_rxreqs[i];
342 rxreq->rxreq_id = i;
343 rxreq->rxreq_sc = sc;
344 rxreq->rxreq_va = (vaddr_t)pool_cache_get_paddr(
345 if_xennetrxbuf_cache, PR_WAITOK, &rxreq->rxreq_pa);
346 if (rxreq->rxreq_va == 0)
347 break;
348 rxreq->rxreq_gntref = GRANT_INVALID_REF;
349 SLIST_INSERT_HEAD(&sc->sc_rxreq_head, rxreq, rxreq_next);
350 }
351 splx(s);
352 sc->sc_free_rxreql = i;
353 if (sc->sc_free_rxreql == 0) {
354 aprint_error_dev(self, "failed to allocate rx memory\n");
355 return;
356 }
357
358 /* read mac address */
359 err = xenbus_read(NULL, xa->xa_xbusd->xbusd_path, "mac", NULL, &val);
360 if (err) {
361 aprint_error_dev(self, "can't read mac address, err %d\n", err);
362 return;
363 }
364 for (i = 0, p = val; i < 6; i++) {
365 sc->sc_enaddr[i] = strtoul(p, &e, 16);
366 if ((e[0] == '\0' && i != 5) && e[0] != ':') {
367 aprint_error_dev(self,
368 "%s is not a valid mac address\n", val);
369 free(val, M_DEVBUF);
370 return;
371 }
372 p = &e[1];
373 }
374 free(val, M_DEVBUF);
375 aprint_normal_dev(self, "MAC address %s\n",
376 ether_sprintf(sc->sc_enaddr));
377 /* Initialize ifnet structure and attach interface */
378 strlcpy(ifp->if_xname, device_xname(self), IFNAMSIZ);
379 sc->sc_ethercom.ec_capabilities |= ETHERCAP_VLAN_MTU;
380 ifp->if_softc = sc;
381 ifp->if_start = xennet_start;
382 ifp->if_ioctl = xennet_ioctl;
383 ifp->if_watchdog = xennet_watchdog;
384 ifp->if_init = xennet_init;
385 ifp->if_stop = xennet_stop;
386 ifp->if_flags = IFF_BROADCAST | IFF_SIMPLEX | IFF_MULTICAST;
387 ifp->if_timer = 0;
388 ifp->if_snd.ifq_maxlen = uimax(ifqmaxlen, NET_TX_RING_SIZE * 2);
389 ifp->if_capabilities = IFCAP_CSUM_TCPv4_Tx | IFCAP_CSUM_UDPv4_Tx;
390 IFQ_SET_READY(&ifp->if_snd);
391 if_attach(ifp);
392 ether_ifattach(ifp, sc->sc_enaddr);
393 sc->sc_softintr = softint_establish(SOFTINT_NET, xennet_softstart, sc);
394 if (sc->sc_softintr == NULL)
395 panic("%s: can't establish soft interrupt",
396 device_xname(self));
397
398 /* alloc shared rings */
399 tx_ring = (void *)uvm_km_alloc(kernel_map, PAGE_SIZE, 0,
400 UVM_KMF_WIRED);
401 rx_ring = (void *)uvm_km_alloc(kernel_map, PAGE_SIZE, 0,
402 UVM_KMF_WIRED);
403 if (tx_ring == NULL || rx_ring == NULL)
404 panic("%s: can't alloc rings", device_xname(self));
405
406 sc->sc_tx_ring.sring = tx_ring;
407 sc->sc_rx_ring.sring = rx_ring;
408
409 /* resume shared structures and tell backend that we are ready */
410 if (xennet_xenbus_resume(self, PMF_Q_NONE) == false) {
411 uvm_km_free(kernel_map, (vaddr_t)tx_ring, PAGE_SIZE,
412 UVM_KMF_WIRED);
413 uvm_km_free(kernel_map, (vaddr_t)rx_ring, PAGE_SIZE,
414 UVM_KMF_WIRED);
415 return;
416 }
417
418 rnd_attach_source(&sc->sc_rnd_source, device_xname(sc->sc_dev),
419 RND_TYPE_NET, RND_FLAG_DEFAULT);
420
421 if (!pmf_device_register(self, xennet_xenbus_suspend,
422 xennet_xenbus_resume))
423 aprint_error_dev(self, "couldn't establish power handler\n");
424 else
425 pmf_class_network_register(self, ifp);
426}

428static int
429xennet_xenbus_detach(device_t self, int flags)
430{
431 struct xennet_xenbus_softc *sc = device_private(self);
432 struct ifnet *ifp = &sc->sc_ethercom.ec_if;
433 int s0, s1;
434 RING_IDX i;
435
436 DPRINTF(("%s: xennet_xenbus_detach\n", device_xname(self)));
437 s0 = splnet();
438 xennet_stop(ifp, 1);
439 xen_intr_disestablish(sc->sc_ih);
440 /* wait for pending TX to complete, and collect pending RX packets */
441 xennet_handler(sc);
442 while (sc->sc_tx_ring.sring->rsp_prod != sc->sc_tx_ring.rsp_cons) {
443 /* XXXSMP */
444 tsleep(xennet_xenbus_detach, PRIBIO, "xnet_detach", hz/2);
445 xennet_handler(sc);
446 }
447 xennet_free_rx_buffer(sc);
448
449 s1 = splvm(); /* XXXSMP */
450 for (i = 0; i < NET_RX_RING_SIZE; i++) {
451 struct xennet_rxreq *rxreq = &sc->sc_rxreqs[i];
452 uvm_km_free(kernel_map, rxreq->rxreq_va, PAGE_SIZE,
453 UVM_KMF_WIRED);
454 }
455 splx(s1);
456
457 ether_ifdetach(ifp);
458 if_detach(ifp);
459
460 /* Unhook the entropy source. */
461 rnd_detach_source(&sc->sc_rnd_source);
462
463 while (xengnt_status(sc->sc_tx_ring_gntref)) {
464 /* XXXSMP */
465 tsleep(xennet_xenbus_detach, PRIBIO, "xnet_txref", hz/2);
466 }
467 xengnt_revoke_access(sc->sc_tx_ring_gntref);
468 uvm_km_free(kernel_map, (vaddr_t)sc->sc_tx_ring.sring, PAGE_SIZE,
469 UVM_KMF_WIRED);
470 while (xengnt_status(sc->sc_rx_ring_gntref)) {
471 /* XXXSMP */
472 tsleep(xennet_xenbus_detach, PRIBIO, "xnet_rxref", hz/2);
473 }
474 xengnt_revoke_access(sc->sc_rx_ring_gntref);
475 uvm_km_free(kernel_map, (vaddr_t)sc->sc_rx_ring.sring, PAGE_SIZE,
476 UVM_KMF_WIRED);
477 softint_disestablish(sc->sc_softintr);
478 splx(s0);
479
480 pmf_device_deregister(self);
481
482 DPRINTF(("%s: xennet_xenbus_detach done\n", device_xname(self)));
483 return 0;
484}

486static bool
487xennet_xenbus_resume(device_t dev, const pmf_qual_t *qual)
488{
489 struct xennet_xenbus_softc *sc = device_private(dev);
490 int error;
491 netif_tx_sring_t *tx_ring;
492 netif_rx_sring_t *rx_ring;
493 paddr_t ma;
494
495 /* invalidate the RX and TX rings */
496 if (sc->sc_backend_status == BEST_SUSPENDED) {
497 /*
498 * Device was suspended, so ensure that access associated to
499 * the previous RX and TX rings are revoked.
500 */
501 xengnt_revoke_access(sc->sc_tx_ring_gntref);
502 xengnt_revoke_access(sc->sc_rx_ring_gntref);
503 }
504
505 sc->sc_tx_ring_gntref = GRANT_INVALID_REF;
506 sc->sc_rx_ring_gntref = GRANT_INVALID_REF;
507
508 tx_ring = sc->sc_tx_ring.sring;
509 rx_ring = sc->sc_rx_ring.sring;
510
511 /* Initialize rings */
512 memset(tx_ring, 0, PAGE_SIZE);
513 SHARED_RING_INIT(tx_ring);
514 FRONT_RING_INIT(&sc->sc_tx_ring, tx_ring, PAGE_SIZE);
515
516 memset(rx_ring, 0, PAGE_SIZE);
517 SHARED_RING_INIT(rx_ring);
518 FRONT_RING_INIT(&sc->sc_rx_ring, rx_ring, PAGE_SIZE);
519
520 (void)pmap_extract_ma(pmap_kernel(), (vaddr_t)tx_ring, &ma);
521 error = xenbus_grant_ring(sc->sc_xbusd, ma, &sc->sc_tx_ring_gntref);
522 if (error)
523 goto abort_resume;
524 (void)pmap_extract_ma(pmap_kernel(), (vaddr_t)rx_ring, &ma);
525 error = xenbus_grant_ring(sc->sc_xbusd, ma, &sc->sc_rx_ring_gntref);
526 if (error)
527 goto abort_resume;
528 error = xenbus_alloc_evtchn(sc->sc_xbusd, &sc->sc_evtchn);
529 if (error)
530 goto abort_resume;
531 aprint_verbose_dev(dev, "using event channel %d\n",
532 sc->sc_evtchn);
533 sc->sc_ih = xen_intr_establish_xname(-1, &xen_pic, sc->sc_evtchn, IST_LEVEL,
534 IPL_NET, &xennet_handler, sc, false, device_xname(dev));
535 KASSERT(sc->sc_ih != NULL);
536 return true;
537
538abort_resume:
539 xenbus_dev_fatal(sc->sc_xbusd, error, "resuming device");
540 return false;
541}

543static bool
544xennet_talk_to_backend(struct xennet_xenbus_softc *sc)
545{
546 int error;
547 unsigned long rx_copy;
548 struct xenbus_transaction *xbt;
549 const char *errmsg;
550
551 error = xenbus_read_ul(NULL, sc->sc_xbusd->xbusd_otherend,
552 "feature-rx-copy", &rx_copy, 10);
553 if (error)
554 rx_copy = 0; /* default value if key is absent */
555
556 if (rx_copy == 1) {
557 aprint_normal_dev(sc->sc_dev, "using RX copy mode\n");
558 sc->sc_rx_feature = FEATURE_RX_COPY;
559 } else {
560 aprint_normal_dev(sc->sc_dev, "using RX flip mode\n");
561 sc->sc_rx_feature = FEATURE_RX_FLIP;
562 }
563
564again:
565 xbt = xenbus_transaction_start();
566 if (xbt == NULL)
567 return false;
568 error = xenbus_printf(xbt, sc->sc_xbusd->xbusd_path,
569 "vifname", "%s", device_xname(sc->sc_dev));
570 if (error) {
571 errmsg = "vifname";
572 goto abort_transaction;
573 }
574 error = xenbus_printf(xbt, sc->sc_xbusd->xbusd_path,
575 "tx-ring-ref","%u", sc->sc_tx_ring_gntref);
576 if (error) {
577 errmsg = "writing tx ring-ref";
578 goto abort_transaction;
579 }
580 error = xenbus_printf(xbt, sc->sc_xbusd->xbusd_path,
581 "rx-ring-ref","%u", sc->sc_rx_ring_gntref);
582 if (error) {
583 errmsg = "writing rx ring-ref";
584 goto abort_transaction;
585 }
586 error = xenbus_printf(xbt, sc->sc_xbusd->xbusd_path,
587 "request-rx-copy", "%lu", rx_copy);
588 if (error) {
589 errmsg = "writing request-rx-copy";
590 goto abort_transaction;
591 }
592 error = xenbus_printf(xbt, sc->sc_xbusd->xbusd_path,
593 "feature-rx-notify", "%u", 1);
594 if (error) {
595 errmsg = "writing feature-rx-notify";
596 goto abort_transaction;
597 }
598 error = xenbus_printf(xbt, sc->sc_xbusd->xbusd_path,
599 "event-channel", "%u", sc->sc_evtchn);
600 if (error) {
601 errmsg = "writing event channel";
602 goto abort_transaction;
603 }
604 error = xenbus_transaction_end(xbt, 0);
605 if (error == EAGAIN)
606 goto again;
607 if (error) {
608 xenbus_dev_fatal(sc->sc_xbusd, error, "completing transaction");
609 return false;
610 }
611 mutex_enter(&sc->sc_rx_lock);
612 xennet_alloc_rx_buffer(sc);
613 mutex_exit(&sc->sc_rx_lock);
614
615 if (sc->sc_backend_status == BEST_SUSPENDED) {
616 xenbus_device_resume(sc->sc_xbusd);
617 }
618
619 sc->sc_backend_status = BEST_CONNECTED;
620
621 return true;
622
623abort_transaction:
624 xenbus_transaction_end(xbt, 1);
625 xenbus_dev_fatal(sc->sc_xbusd, error, "%s", errmsg);
626 return false;
627}

629static bool
630xennet_xenbus_suspend(device_t dev, const pmf_qual_t *qual)
631{
632 int s;
633 struct xennet_xenbus_softc *sc = device_private(dev);
634
635 /*
636 * xennet_stop() is called by pmf(9) before xennet_xenbus_suspend(),
637 * so we do not mask event channel here
638 */
639
640 s = splnet();
641 /* process any outstanding TX responses, then collect RX packets */
642 xennet_handler(sc);
643 while (sc->sc_tx_ring.sring->rsp_prod != sc->sc_tx_ring.rsp_cons) {
644 /* XXXSMP */
645 tsleep(xennet_xenbus_suspend, PRIBIO, "xnet_suspend", hz/2);
646 xennet_handler(sc);
647 }
648
649 /*
650 * dom0 may still use references to the grants we gave away
651 * earlier during RX buffers allocation. So we do not free RX buffers
652 * here, as dom0 does not expect the guest domain to suddenly revoke
653 * access to these grants.
654 */
655
656 sc->sc_backend_status = BEST_SUSPENDED;
657 xen_intr_disestablish(sc->sc_ih);
658
659 splx(s);
660
661 xenbus_device_suspend(sc->sc_xbusd);
662 aprint_verbose_dev(dev, "removed event channel %d\n", sc->sc_evtchn);
663
664 return true;
665}

669 struct xennet_xenbus_softc *sc = device_private((device_t)arg);
670 DPRINTF(("%s: new backend state %d\n",

695/*
696 * Allocate RX buffers and put the associated request structures
697 * in the ring. This allows the backend to use them to communicate with
698 * frontend when some data is destined to frontend
699 */
700
701static void
702xennet_alloc_rx_buffer(struct xennet_xenbus_softc *sc)
703{
704 RING_IDX req_prod = sc->sc_rx_ring.req_prod_pvt;
705 RING_IDX i;
706 struct xennet_rxreq *req;
707 struct xen_memory_reservation reservation;
708 int s, otherend_id, notify;
709
710 otherend_id = sc->sc_xbusd->xbusd_otherend_id;
711
712 KASSERT(mutex_owned(&sc->sc_rx_lock));
713 for (i = 0; sc->sc_free_rxreql != 0; i++) {
714 req = SLIST_FIRST(&sc->sc_rxreq_head);
715 KASSERT(req != NULL);
716 KASSERT(req == &sc->sc_rxreqs[req->rxreq_id]);
717 RING_GET_REQUEST(&sc->sc_rx_ring, req_prod + i)->id =
718 req->rxreq_id;
719
720 switch (sc->sc_rx_feature) {
721 case FEATURE_RX_COPY:
722 if (xengnt_grant_access(otherend_id,
723 xpmap_ptom_masked(req->rxreq_pa),
724 0, &req->rxreq_gntref) != 0) {
725 goto out_loop;
726 }
727 break;
728 case FEATURE_RX_FLIP:
729 if (xengnt_grant_transfer(otherend_id,
730 &req->rxreq_gntref) != 0) {
731 goto out_loop;
732 }
733 break;
734 default:
735 panic("%s: unsupported RX feature mode: %ld\n",
736 __func__, sc->sc_rx_feature);
737 }
738
739 RING_GET_REQUEST(&sc->sc_rx_ring, req_prod + i)->gref =
740 req->rxreq_gntref;
741
742 SLIST_REMOVE_HEAD(&sc->sc_rxreq_head, rxreq_next);
743 sc->sc_free_rxreql--;
744
745 if (sc->sc_rx_feature == FEATURE_RX_FLIP) {
746 /* unmap the page */
747 MULTI_update_va_mapping(&rx_mcl[i],
748 req->rxreq_va, 0, 0);
749 /*
750 * Remove this page from pseudo phys map before
751 * passing back to Xen.
752 */
753 xennet_pages[i] =
754 xpmap_ptom(req->rxreq_pa) >> PAGE_SHIFT;
755 xpmap_ptom_unmap(req->rxreq_pa);
756 }
757 }
758
759out_loop:
760 if (i == 0) {
761 return;
762 }
763
764 if (sc->sc_rx_feature == FEATURE_RX_FLIP) {
765 /* also make sure to flush all TLB entries */
766 rx_mcl[i-1].args[MULTI_UVMFLAGS_INDEX] =
767 UVMF_TLB_FLUSH | UVMF_ALL;
768 /*
769 * We may have allocated buffers which have entries
770 * outstanding in the page update queue -- make sure we flush
771 * those first!
772 */
773 s = splvm(); /* XXXSMP */
774 xpq_flush_queue();
775 splx(s);
776 /* now decrease reservation */
777 set_xen_guest_handle(reservation.extent_start, xennet_pages);
778 reservation.nr_extents = i;
779 reservation.extent_order = 0;
780 reservation.address_bits = 0;
781 reservation.domid = DOMID_SELF;
782 rx_mcl[i].op = __HYPERVISOR_memory_op;
783 rx_mcl[i].args[0] = XENMEM_decrease_reservation;
784 rx_mcl[i].args[1] = (unsigned long)&reservation;
785 HYPERVISOR_multicall(rx_mcl, i+1);
786 if (__predict_false(rx_mcl[i].result != i)) {
787 panic("xennet_alloc_rx_buffer: "
788 "XENMEM_decrease_reservation");
789 }
790 }
791
792 sc->sc_rx_ring.req_prod_pvt = req_prod + i;
793 RING_PUSH_REQUESTS_AND_CHECK_NOTIFY(&sc->sc_rx_ring, notify);
794 if (notify)
795 hypervisor_notify_via_evtchn(sc->sc_evtchn);
796 return;
797}

799/*
800 * Reclaim all RX buffers used by the I/O ring between frontend and backend
801 */
802static void
803xennet_free_rx_buffer(struct xennet_xenbus_softc *sc)
804{
805 paddr_t ma, pa;
806 vaddr_t va;
807 RING_IDX i;
808 mmu_update_t mmu[1];
809 multicall_entry_t mcl[2];
810
811 mutex_enter(&sc->sc_rx_lock);
812
813 DPRINTF(("%s: xennet_free_rx_buffer\n", device_xname(sc->sc_dev)));
814 /* get back memory from RX ring */
815 for (i = 0; i < NET_RX_RING_SIZE; i++) {
816 struct xennet_rxreq *rxreq = &sc->sc_rxreqs[i];
817
818 if (rxreq->rxreq_gntref != GRANT_INVALID_REF) {
819 /*
820 * this req is still granted. Get back the page or
821 * allocate a new one, and remap it.
822 */
823 SLIST_INSERT_HEAD(&sc->sc_rxreq_head, rxreq,
824 rxreq_next);
825 sc->sc_free_rxreql++;
826
827 switch (sc->sc_rx_feature) {
828 case FEATURE_RX_COPY:
829 xengnt_revoke_access(rxreq->rxreq_gntref);
830 rxreq->rxreq_gntref = GRANT_INVALID_REF;
831 break;
832 case FEATURE_RX_FLIP:
833 ma = xengnt_revoke_transfer(
834 rxreq->rxreq_gntref);
835 rxreq->rxreq_gntref = GRANT_INVALID_REF;
836 if (ma == 0) {
837 u_long pfn;
838 struct xen_memory_reservation xenres;
839 /*
840 * transfer not complete, we lost the page.
841 * Get one from hypervisor
842 */
843 set_xen_guest_handle(
844 xenres.extent_start, &pfn);
845 xenres.nr_extents = 1;
846 xenres.extent_order = 0;
847 xenres.address_bits = 31;
848 xenres.domid = DOMID_SELF;
849 if (HYPERVISOR_memory_op(
850 XENMEM_increase_reservation, &xenres) < 0) {
851 panic("xennet_free_rx_buffer: "
852 "can't get memory back");
853 }
854 ma = pfn;
855 KASSERT(ma != 0);
856 }
857 pa = rxreq->rxreq_pa;
858 va = rxreq->rxreq_va;
859 /* remap the page */
860 mmu[0].ptr = (ma << PAGE_SHIFT) | MMU_MACHPHYS_UPDATE;
861 mmu[0].val = pa >> PAGE_SHIFT;
862 MULTI_update_va_mapping(&mcl[0], va,
863 (ma << PAGE_SHIFT) | PTE_P | PTE_W | xpmap_pg_nx,
864 UVMF_TLB_FLUSH|UVMF_ALL);
865 xpmap_ptom_map(pa, ptoa(ma));
866 mcl[1].op = __HYPERVISOR_mmu_update;
867 mcl[1].args[0] = (unsigned long)mmu;
868 mcl[1].args[1] = 1;
869 mcl[1].args[2] = 0;
870 mcl[1].args[3] = DOMID_SELF;
871 HYPERVISOR_multicall(mcl, 2);
872 break;
873 default:
874 panic("%s: unsupported RX feature mode: %ld\n",
875 __func__, sc->sc_rx_feature);
876 }
877 }
878
879 }
880 mutex_exit(&sc->sc_rx_lock);
881 DPRINTF(("%s: xennet_free_rx_buffer done\n", device_xname(sc->sc_dev)));
882}

901static void
902xennet_rx_free_req(struct xennet_rxreq *req)
903{
904 struct xennet_xenbus_softc *sc = req->rxreq_sc;
905
906 KASSERT(mutex_owned(&sc->sc_rx_lock));
907
908 /* puts back the RX request in the list of free RX requests */
909 SLIST_INSERT_HEAD(&sc->sc_rxreq_head, req, rxreq_next);
910 sc->sc_free_rxreql++;
911
912 /*
913 * ring needs more requests to be pushed in, allocate some
914 * RX buffers to catch-up with backend's consumption
915 */
916 req->rxreq_gntref = GRANT_INVALID_REF;
917
918 if (sc->sc_free_rxreql >= (NET_RX_RING_SIZE * 4 / 5) &&
919 __predict_true(sc->sc_backend_status == BEST_CONNECTED)) {
920 xennet_alloc_rx_buffer(sc);
921 }

924/*
925 * Process responses associated to the TX mbufs sent previously through
926 * xennet_softstart()
927 * Called at splnet.
928 */
929static void
930xennet_tx_complete(struct xennet_xenbus_softc *sc)
931{
932 struct xennet_txreq *req;
933 struct ifnet *ifp = &sc->sc_ethercom.ec_if;
934 RING_IDX resp_prod, i;
935
936 DPRINTFN(XEDB_EVENT, ("xennet_tx_complete prod %d cons %d\n",
937 sc->sc_tx_ring.sring->rsp_prod, sc->sc_tx_ring.rsp_cons));
938
939again:
940 resp_prod = sc->sc_tx_ring.sring->rsp_prod;
941 xen_rmb();
942 mutex_enter(&sc->sc_tx_lock);
943 for (i = sc->sc_tx_ring.rsp_cons; i != resp_prod; i++) {
944 req = &sc->sc_txreqs[RING_GET_RESPONSE(&sc->sc_tx_ring, i)->id];
945 KASSERT(req->txreq_id ==
946 RING_GET_RESPONSE(&sc->sc_tx_ring, i)->id);
947 if (__predict_false(xengnt_status(req->txreq_gntref))) {
948 aprint_verbose_dev(sc->sc_dev,
949 "grant still used by backend\n");
950 sc->sc_tx_ring.rsp_cons = i;
951 goto end;
952 }
953 if (__predict_false(
954 RING_GET_RESPONSE(&sc->sc_tx_ring, i)->status !=
955 NETIF_RSP_OKAY))
956 ifp->if_oerrors++;
957 else
958 ifp->if_opackets++;
959 xengnt_revoke_access(req->txreq_gntref);
960 m_freem(req->txreq_m);
961 SLIST_INSERT_HEAD(&sc->sc_txreq_head, req, txreq_next);
962 }
963 mutex_exit(&sc->sc_tx_lock);
964
965 sc->sc_tx_ring.rsp_cons = resp_prod;
966 /* set new event and check for race with rsp_cons update */
967 sc->sc_tx_ring.sring->rsp_event =
968 resp_prod + ((sc->sc_tx_ring.sring->req_prod - resp_prod) >> 1) + 1;
969 ifp->if_timer = 0;
970 xen_wmb();
971 if (resp_prod != sc->sc_tx_ring.sring->rsp_prod)
972 goto again;
973end:
974 if (ifp->if_flags & IFF_OACTIVE) {
975 ifp->if_flags &= ~IFF_OACTIVE;
976 softint_schedule(sc->sc_softintr);
977 }
978}

980/*
981 * Xennet event handler.
982 * Get outstanding responses of TX packets, then collect all responses of
983 * pending RX packets
984 * Called at splnet.
985 */
986static int
987xennet_handler(void *arg)
988{
989 struct xennet_xenbus_softc *sc = arg;
990 struct ifnet *ifp = &sc->sc_ethercom.ec_if;
991 RING_IDX resp_prod, i;
992 struct xennet_rxreq *req;
993 paddr_t ma, pa;
994 vaddr_t va;
995 mmu_update_t mmu[1];
996 multicall_entry_t mcl[2];
997 struct mbuf *m;
998 void *pktp;
999 int more_to_do;
1000
1001 if (sc->sc_backend_status != BEST_CONNECTED)
1002 return 1;
1003
1004 xennet_tx_complete(sc);
1005
1006 rnd_add_uint32(&sc->sc_rnd_source, sc->sc_tx_ring.req_prod_pvt);
1007
1008again:
1009 DPRINTFN(XEDB_EVENT, ("xennet_handler prod %d cons %d\n",
1010 sc->sc_rx_ring.sring->rsp_prod, sc->sc_rx_ring.rsp_cons));
1011
1012 mutex_enter(&sc->sc_rx_lock);
1013 resp_prod = sc->sc_rx_ring.sring->rsp_prod;
1014 xen_rmb(); /* ensure we see replies up to resp_prod */
1015
1016 for (i = sc->sc_rx_ring.rsp_cons; i != resp_prod; i++) {
1017 netif_rx_response_t *rx = RING_GET_RESPONSE(&sc->sc_rx_ring, i);
1018 req = &sc->sc_rxreqs[rx->id];
1019 KASSERT(req->rxreq_gntref != GRANT_INVALID_REF);
1020 KASSERT(req->rxreq_id == rx->id);
1021
1022 ma = 0;
1023 switch (sc->sc_rx_feature) {
1024 case FEATURE_RX_COPY:
1025 xengnt_revoke_access(req->rxreq_gntref);
1026 break;
1027 case FEATURE_RX_FLIP:
1028 ma = xengnt_revoke_transfer(req->rxreq_gntref);
1029 if (ma == 0) {
1030 DPRINTFN(XEDB_EVENT, ("xennet_handler ma == 0\n"));
1031 /*
1032 * the remote could't send us a packet.
1033 * we can't free this rxreq as no page will be mapped
1034 * here. Instead give it back immediatly to backend.
1035 */
1036 ifp->if_ierrors++;
1037 RING_GET_REQUEST(&sc->sc_rx_ring,
1038 sc->sc_rx_ring.req_prod_pvt)->id = req->rxreq_id;
1039 RING_GET_REQUEST(&sc->sc_rx_ring,
1040 sc->sc_rx_ring.req_prod_pvt)->gref =
1041 req->rxreq_gntref;
1042 sc->sc_rx_ring.req_prod_pvt++;
1043 RING_PUSH_REQUESTS(&sc->sc_rx_ring);
1044 continue;
1045 }
1046 break;
1047 default:
1048 panic("%s: unsupported RX feature mode: %ld\n",
1049 __func__, sc->sc_rx_feature);
1050 }
1051
1052 pa = req->rxreq_pa;
1053 va = req->rxreq_va;
1054
1055 if (sc->sc_rx_feature == FEATURE_RX_FLIP) {
1056 /* remap the page */
1057 mmu[0].ptr = (ma << PAGE_SHIFT) | MMU_MACHPHYS_UPDATE;
1058 mmu[0].val = pa >> PAGE_SHIFT;
1059 MULTI_update_va_mapping(&mcl[0], va,
1060 (ma << PAGE_SHIFT) | PTE_P | PTE_W | xpmap_pg_nx,
1061 UVMF_TLB_FLUSH|UVMF_ALL);
1062 xpmap_ptom_map(pa, ptoa(ma));
1063 mcl[1].op = __HYPERVISOR_mmu_update;
1064 mcl[1].args[0] = (unsigned long)mmu;
1065 mcl[1].args[1] = 1;
1066 mcl[1].args[2] = 0;
1067 mcl[1].args[3] = DOMID_SELF;
1068 HYPERVISOR_multicall(mcl, 2);
1069 }
1070
1071 pktp = (void *)(va + rx->offset);
1072#ifdef XENNET_DEBUG_DUMP
1073 xennet_hex_dump(pktp, rx->status, "r", rx->id);
1074#endif
1075 if ((ifp->if_flags & IFF_PROMISC) == 0) {
1076 struct ether_header *eh = pktp;
1077 if (ETHER_IS_MULTICAST(eh->ether_dhost) == 0 &&
1078 memcmp(CLLADDR(ifp->if_sadl), eh->ether_dhost,
1079 ETHER_ADDR_LEN) != 0) {
1080 DPRINTFN(XEDB_EVENT,
1081 ("xennet_handler bad dest\n"));
1082 /* packet not for us */
1083 xennet_rx_free_req(req);
1084 continue;
1085 }
1086 }
1087 MGETHDR(m, M_DONTWAIT, MT_DATA);
1088 if (__predict_false(m == NULL)) {
1089 printf("%s: rx no mbuf\n", ifp->if_xname);
1090 ifp->if_ierrors++;
1091 xennet_rx_free_req(req);
1092 continue;
1093 }
1094 MCLAIM(m, &sc->sc_ethercom.ec_rx_mowner);
1095
1096 m_set_rcvif(m, ifp);
1097 if (rx->status <= MHLEN) {
1098 /* small packet; copy to mbuf data area */
1099 m_copyback(m, 0, rx->status, pktp);
1100 KASSERT(m->m_pkthdr.len == rx->status);
1101 KASSERT(m->m_len == rx->status);
1102 } else {
1103 /* large packet; attach buffer to mbuf */
1104 req->rxreq_va = (vaddr_t)pool_cache_get_paddr(
1105 if_xennetrxbuf_cache, PR_NOWAIT, &req->rxreq_pa);
1106 if (__predict_false(req->rxreq_va == 0)) {
1107 printf("%s: rx no buf\n", ifp->if_xname);
1108 ifp->if_ierrors++;
1109 req->rxreq_va = va;
1110 req->rxreq_pa = pa;
1111 xennet_rx_free_req(req);
1112 m_freem(m);
1113 continue;
1114 }
1115 m->m_len = m->m_pkthdr.len = rx->status;
1116 MEXTADD(m, pktp, rx->status,
1117 M_DEVBUF, xennet_rx_mbuf_free, NULL);
1118 m->m_ext.ext_paddr = pa;
1119 m->m_flags |= M_EXT_RW; /* we own the buffer */
1120 }
1121 if ((rx->flags & NETRXF_csum_blank) != 0) {
1122 xennet_checksum_fill(&m);
1123 if (m == NULL) {
1124 ifp->if_ierrors++;
1125 xennet_rx_free_req(req);
1126 continue;
1127 }
1128 }
1129 /* free req may overwrite *rx, better doing it late */
1130 xennet_rx_free_req(req);
1131
1132 /* Pass the packet up. */
1133 if_percpuq_enqueue(ifp->if_percpuq, m);
1134 }
1135 xen_rmb();
1136 sc->sc_rx_ring.rsp_cons = i;
1137 RING_FINAL_CHECK_FOR_RESPONSES(&sc->sc_rx_ring, more_to_do);
1138 mutex_exit(&sc->sc_rx_lock);
1139
1140 if (more_to_do)
1141 goto again;
1142
1143 return 1;
1144}

1146/*
1147 * The output routine of a xennet interface
1148 * Called at splnet.
1149 */
1150void
1151xennet_start(struct ifnet *ifp)
1152{
1153 struct xennet_xenbus_softc *sc = ifp->if_softc;
1154
1155 DPRINTFN(XEDB_FOLLOW, ("%s: xennet_start()\n", device_xname(sc->sc_dev)));
1156
1157 rnd_add_uint32(&sc->sc_rnd_source, sc->sc_tx_ring.req_prod_pvt);
1158
1159 xennet_tx_complete(sc);
1160
1161 if (__predict_false(
1162 (ifp->if_flags & (IFF_RUNNING | IFF_OACTIVE)) != IFF_RUNNING))
1163 return;
1164
1165 /*
1166 * The Xen communication channel is much more efficient if we can
1167 * schedule batch of packets for domain0. To achieve this, we
1168 * schedule a soft interrupt, and just return. This way, the network
1169 * stack will enqueue all pending mbufs in the interface's send queue
1170 * before it is processed by xennet_softstart().
1171 */
1172 softint_schedule(sc->sc_softintr);
1173 return;
1174}

1176/*
1177 * Prepares mbufs for TX, and notify backend when finished
1178 * Called at splsoftnet
1179 */
1180void
1181xennet_softstart(void *arg)
1182{
1183 struct xennet_xenbus_softc *sc = arg;
1184 struct ifnet *ifp = &sc->sc_ethercom.ec_if;
1185 struct mbuf *m, *new_m;
1186 netif_tx_request_t *txreq;
1187 RING_IDX req_prod;
1188 paddr_t pa;
1189 struct xennet_txreq *req;
1190 int notify;
1191 int do_notify = 0;
1192
1193 mutex_enter(&sc->sc_tx_lock);
1194 if (__predict_false(
1195 (ifp->if_flags & (IFF_RUNNING | IFF_OACTIVE)) != IFF_RUNNING)) {
1196 mutex_exit(&sc->sc_tx_lock);
1197 return;
1198 }
1199
1200 req_prod = sc->sc_tx_ring.req_prod_pvt;
1201 while (/*CONSTCOND*/1) {
1202 uint16_t txflags;
1203
1204 req = SLIST_FIRST(&sc->sc_txreq_head);
1205 if (__predict_false(req == NULL)) {
1206 ifp->if_flags |= IFF_OACTIVE;
1207 break;
1208 }
1209 IFQ_POLL(&ifp->if_snd, m);
1210 if (m == NULL)
1211 break;
1212
1213 switch (m->m_flags & (M_EXT|M_EXT_CLUSTER)) {
1214 case M_EXT|M_EXT_CLUSTER:
1215 KASSERT(m->m_ext.ext_paddr != M_PADDR_INVALID);
1216 pa = m->m_ext.ext_paddr +
1217 (m->m_data - m->m_ext.ext_buf);
1218 break;
1219 case 0:
1220 KASSERT(m->m_paddr != M_PADDR_INVALID);
1221 pa = m->m_paddr + M_BUFOFFSET(m) +
1222 (m->m_data - M_BUFADDR(m));
1223 break;
1224 default:
1225 if (__predict_false(
1226 !pmap_extract(pmap_kernel(), (vaddr_t)m->m_data,
1227 &pa))) {
1228 panic("xennet_start: no pa");
1229 }
1230 break;
1231 }
1232
1233 if ((m->m_pkthdr.csum_flags &
1234 (M_CSUM_TCPv4 | M_CSUM_UDPv4)) != 0) {
1235 txflags = NETTXF_csum_blank;
1236 } else {
1237 txflags = 0;
1238 }
1239
1240 if (m->m_pkthdr.len != m->m_len ||
1241 (pa ^ (pa + m->m_pkthdr.len - 1)) & PG_FRAME) {
1242
1243 MGETHDR(new_m, M_DONTWAIT, MT_DATA);
1244 if (__predict_false(new_m == NULL)) {
1245 printf("%s: cannot allocate new mbuf\n",
1246 device_xname(sc->sc_dev));
1247 break;
1248 }
1249 if (m->m_pkthdr.len > MHLEN) {
1250 MCLGET(new_m, M_DONTWAIT);
1251 if (__predict_false(
1252 (new_m->m_flags & M_EXT) == 0)) {
1253 DPRINTF(("%s: no mbuf cluster\n",
1254 device_xname(sc->sc_dev)));
1255 m_freem(new_m);
1256 break;
1257 }
1258 }
1259
1260 m_copydata(m, 0, m->m_pkthdr.len, mtod(new_m, void *));
1261 new_m->m_len = new_m->m_pkthdr.len = m->m_pkthdr.len;
1262
1263 if ((new_m->m_flags & M_EXT) != 0) {
1264 pa = new_m->m_ext.ext_paddr;
1265 KASSERT(new_m->m_data == new_m->m_ext.ext_buf);
1266 KASSERT(pa != M_PADDR_INVALID);
1267 } else {
1268 pa = new_m->m_paddr;
1269 KASSERT(pa != M_PADDR_INVALID);
1270 KASSERT(new_m->m_data == M_BUFADDR(new_m));
1271 pa += M_BUFOFFSET(new_m);
1272 }
1273 if (__predict_false(xengnt_grant_access(
1274 sc->sc_xbusd->xbusd_otherend_id,
1275 xpmap_ptom_masked(pa),
1276 GNTMAP_readonly, &req->txreq_gntref) != 0)) {
1277 m_freem(new_m);
1278 ifp->if_flags |= IFF_OACTIVE;
1279 break;
1280 }
1281 /* we will be able to send new_m */
1282 IFQ_DEQUEUE(&ifp->if_snd, m);
1283 m_freem(m);
1284 m = new_m;
1285 } else {
1286 if (__predict_false(xengnt_grant_access(
1287 sc->sc_xbusd->xbusd_otherend_id,
1288 xpmap_ptom_masked(pa),
1289 GNTMAP_readonly, &req->txreq_gntref) != 0)) {
1290 ifp->if_flags |= IFF_OACTIVE;
1291 break;
1292 }
1293 /* we will be able to send m */
1294 IFQ_DEQUEUE(&ifp->if_snd, m);
1295 }
1296 MCLAIM(m, &sc->sc_ethercom.ec_tx_mowner);
1297
1298 KASSERT(((pa ^ (pa + m->m_pkthdr.len - 1)) & PG_FRAME) == 0);
1299
1300 SLIST_REMOVE_HEAD(&sc->sc_txreq_head, txreq_next);
1301 req->txreq_m = m;
1302
1303 DPRINTFN(XEDB_MBUF, ("xennet_start id %d, "
1304 "mbuf %p, buf %p/%p/%p, size %d\n",
1305 req->txreq_id, m, mtod(m, void *), (void *)pa,
1306 (void *)xpmap_ptom_masked(pa), m->m_pkthdr.len));
1307#ifdef XENNET_DEBUG
1308 paddr_t pa2;
1309 pmap_extract_ma(pmap_kernel(), mtod(m, vaddr_t), &pa2);
1310 DPRINTFN(XEDB_MBUF, ("xennet_start pa %p ma %p/%p\n",
1311 (void *)pa, (void *)xpmap_ptom_masked(pa), (void *)pa2));
1312#endif
1313
1314#ifdef XENNET_DEBUG_DUMP
1315 xennet_hex_dump(mtod(m, u_char *), m->m_pkthdr.len, "s",
1316 req->txreq_id);
1317#endif
1318
1319 txreq = RING_GET_REQUEST(&sc->sc_tx_ring, req_prod);
1320 txreq->id = req->txreq_id;
1321 txreq->gref = req->txreq_gntref;
1322 txreq->offset = pa & ~PG_FRAME;
1323 txreq->size = m->m_pkthdr.len;
1324 txreq->flags = txflags;
1325
1326 req_prod++;
1327 sc->sc_tx_ring.req_prod_pvt = req_prod;
1328 RING_PUSH_REQUESTS_AND_CHECK_NOTIFY(&sc->sc_tx_ring, notify);
1329 if (notify)
1330 do_notify = 1;
1331
1332#ifdef XENNET_DEBUG
1333 DPRINTFN(XEDB_MEM, ("packet addr %p/%p, physical %p/%p, "
1334 "m_paddr %p, len %d/%d\n", M_BUFADDR(m), mtod(m, void *),
1335 (void *)*kvtopte(mtod(m, vaddr_t)),
1336 (void *)xpmap_mtop(*kvtopte(mtod(m, vaddr_t))),
1337 (void *)m->m_paddr, m->m_pkthdr.len, m->m_len));
1338 DPRINTFN(XEDB_MEM, ("id %d gref %d offset %d size %d flags %d"
1339 " prod %d\n",
1340 txreq->id, txreq->gref, txreq->offset, txreq->size,
1341 txreq->flags, req_prod));
1342#endif
1343
1344 /*
1345 * Pass packet to bpf if there is a listener.
1346 */
1347 bpf_mtap(ifp, m, BPF_D_OUT);
1348 }
1349
1350 if (do_notify) {
1351 hypervisor_notify_via_evtchn(sc->sc_evtchn);
1352 ifp->if_timer = 5;
1353 }
1354
1355 mutex_exit(&sc->sc_tx_lock);
1356
1357 DPRINTFN(XEDB_FOLLOW, ("%s: xennet_start() done\n",
1358 device_xname(sc->sc_dev)));
1359}
