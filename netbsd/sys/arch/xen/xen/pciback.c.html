138/* emulate byte and word access to PCI config space */
139static inline u_int8_t
140pciback_read_byte(pci_chipset_tag_t pc, pcitag_t pa, int reg)
141{
142
143 return (pci_conf_read(pc, pa, (reg & ~0x03)) >>
144 ((reg & 0x03) * 8) & 0xff);
145}
146
147static inline u_int16_t
148pciback_read_word(pci_chipset_tag_t pc, pcitag_t pa, int reg)
149{
150 return (pci_conf_read(pc, pa, (reg & ~0x03)) >>
151 ((reg & 0x03) * 8) & 0xffff);
152}
153
154static inline void
155pciback_write_byte(pci_chipset_tag_t pc, pcitag_t pa, int reg, uint8_t val)
156{
157 pcireg_t pcival;
158
159 pcival = pci_conf_read(pc, pa, (reg & ~0x03));
160 pcival &= ~(0xff << ((reg & 0x03) * 8));
161 pcival |= (val << ((reg & 0x03) * 8));
162 pci_conf_write(pc, pa, (reg & ~0x03), pcival);
163}
164
165static inline void
166pciback_write_word(pci_chipset_tag_t pc, pcitag_t pa, int reg, uint16_t val)
167{
168 pcireg_t pcival;
169
170 pcival = pci_conf_read(pc, pa, (reg & ~0x03));
171 pcival &= ~(0xffff << ((reg & 0x03) * 8));
172 pcival |= (val << ((reg & 0x03) * 8));
173 pci_conf_write(pc, pa, (reg & ~0x03), pcival);
174}

205int
206pciback_pci_match(device_t parent, cfdata_t match, void *aux)
207{
208 struct pci_attach_args *pa = aux;
209 if (pciback_pci_inited == 0) {
210 pciback_pci_init();
211 pciback_pci_inited = 1;
212 }
213 if (pciback_pci_lookup(pa->pa_bus, pa->pa_device, pa->pa_function))
214 return 500; /* we really want to take over anything else */
215 return 0;
216}

218void
219pciback_pci_attach(device_t parent, device_t self, void *aux)
220{
221 struct pci_attach_args *pa = aux;
222 struct pciback_pci_softc *sc = device_private(self);
223 char devinfo[256];
224 int i;
225 const char *intrstr;
226 kernfs_entry_t *dkt;
227 kfstype kfst;
228 pcireg_t reg;
229 char buf[PCI_INTRSTR_LEN];
230
231 sc->sc_dev = self;
232 sc->sc_pb = pciback_pci_lookup(pa->pa_bus, pa->pa_device, pa->pa_function);
233 if (sc->sc_pb == NULL)
234 panic("pciback_pci_attach: pciback_lookup");
235 pci_devinfo(pa->pa_id, pa->pa_class, 0, devinfo, sizeof(devinfo));
236 aprint_normal(": %s (rev. 0x%02x)\n", devinfo,
237 PCI_REVISION(pa->pa_class));
238 sc->sc_pb->pb_pci_softc = sc;
239 sc->sc_pb->pb_pc = pa->pa_pc;
240 sc->sc_pb->pb_tag = pa->pa_tag;
241
242 sc->sc_id = pci_conf_read(pa->pa_pc, pa->pa_tag, PCI_ID_REG);
243 sc->sc_subid = pci_conf_read(pa->pa_pc, pa->pa_tag, PCI_SUBSYS_ID_REG);
244
245 for (i = 0; i < PCI_NBARS;) {
246 sc->sc_bars[i].b_type = pci_mapreg_type(pa->pa_pc, pa->pa_tag,
247 PCI_MAPREG_START + i * 4);
248 if (pci_mapreg_map(pa, PCI_MAPREG_START + i * 4,
249 sc->sc_bars[i].b_type, 0,
250 &sc->sc_bars[i].b_t, &sc->sc_bars[i].b_h,
251 &sc->sc_bars[i].b_addr, &sc->sc_bars[i].b_size) == 0)
252 sc->sc_bars[i].b_valid = 1;
253 if (sc->sc_bars[i].b_valid) {
254 aprint_verbose_dev(self, "%s: 0x%08jx - 0x%08jx\n",
255 (sc->sc_bars[i].b_type == PCI_MAPREG_TYPE_IO) ?
256 "I/O" : "mem",
257 (uintmax_t)sc->sc_bars[i].b_addr,
258 (uintmax_t)sc->sc_bars[i].b_size);
259 }
260
261 if (sc->sc_bars[i].b_type ==
262 (PCI_MAPREG_TYPE_MEM | PCI_MAPREG_MEM_TYPE_64BIT))
263 i += 2;
264 else
265 i += 1;
266 }
267 /* map the irq so interrupt routing is done */
268 if (pci_intr_map(pa, &sc->sc_intrhandle) != 0) {
269 aprint_error_dev(self, "couldn't map interrupt\n");
270 } else {
271 intrstr = pci_intr_string(pa->pa_pc, sc->sc_intrhandle,
272 buf, sizeof(buf));
273 aprint_normal_dev(self, "interrupting at %s\n", intrstr);
274 }
275 unbind_pirq_from_evtch(APIC_IRQ_LEGACY_IRQ(sc->sc_intrhandle));
276 sc->sc_irq = APIC_IRQ_LEGACY_IRQ(sc->sc_intrhandle);
277 /* XXX should be done elsewhere ? */
278 reg = pci_conf_read(pa->pa_pc, pa->pa_tag, PCI_INTERRUPT_REG);
279 reg &= ~ (PCI_INTERRUPT_LINE_MASK << PCI_INTERRUPT_LINE_SHIFT);
280 reg |= (sc->sc_irq << PCI_INTERRUPT_LINE_SHIFT);
281 pci_conf_write(pa->pa_pc, pa->pa_tag, PCI_INTERRUPT_REG, reg);
282 printf("irq line %d pin %d sc %p\n",
283 PCI_INTERRUPT_LINE(pci_conf_read(pa->pa_pc, pa->pa_tag, PCI_INTERRUPT_REG)),
284 PCI_INTERRUPT_PIN(pci_conf_read(pa->pa_pc, pa->pa_tag, PCI_INTERRUPT_REG)), sc);
285 /*
286 * don't establish the interrupt, we're not interested in
287 * getting it here.
288 */
289 /* publish the informations about this device to /kern/xen/pci */
290 snprintf(sc->sc_kernfsname, sizeof(sc->sc_kernfsname),
291 "0000:%02x:%02x.%x", pa->pa_bus, pa->pa_device, pa->pa_function);
292 kfst = KERNFS_ALLOCTYPE(pciback_dev_fileops);
293 KERNFS_ALLOCENTRY(dkt, M_TEMP, M_WAITOK);
294 KERNFS_INITENTRY(dkt, DT_REG, sc->sc_kernfsname, sc,
295 kfst, VREG, FILE_MODE);
296 kernfs_addentry(pciback_kern_pkt, dkt);
297}

299static int
300pciback_kernfs_read(void *v)
301{
302 struct vop_read_args /* {
303 struct vnode *a_vp;
304 struct uio *a_uio;
305 int a_ioflag;
306 struct ucred *a_cred;
307 } */ *ap = v;
308 struct kernfs_node *kfs = VTOKERN(ap->a_vp);
309 struct uio *uio = ap->a_uio;
310 struct pciback_pci_softc *sc = kfs->kfs_kt->kt_data;
311#define PCIBACK_KERNFS_SIZE 512
312 static char buf[PCIBACK_KERNFS_SIZE];
313 off_t off;
314 off_t len;
315 int error, i;
316
317 off = uio->uio_offset;
318 len = 0;
319 len += snprintf(&buf[len], sizeof(buf) - len,
320 "vendor: 0x%04x\nproduct: 0x%04x\n",
321 PCI_VENDOR(sc->sc_id), PCI_PRODUCT(sc->sc_id));
322 if (len > sizeof(buf))
323 return ENOSPC;
324 len += snprintf(&buf[len], sizeof(buf) - len,
325 "subsys_vendor: 0x%04x\nsubsys_product: 0x%04x\n",
326 PCI_VENDOR(sc->sc_subid), PCI_PRODUCT(sc->sc_subid));
327 if (len > sizeof(buf))
328 return ENOSPC;
329 for(i = 0; i < PCI_NBARS; i++) {
330 if (sc->sc_bars[i].b_valid) {
331 len += snprintf(&buf[len], sizeof(buf) - len,
332 "%s: 0x%08jx - 0x%08jx\n",
333 (sc->sc_bars[i].b_type == PCI_MAPREG_TYPE_IO) ?
334 "I/O" : "mem",
335 (uintmax_t)sc->sc_bars[i].b_addr,
336 (uintmax_t)(sc->sc_bars[i].b_addr + sc->sc_bars[i].b_size));
337 if (len > sizeof(buf))
338 return ENOSPC;
339 }
340 }
341 if (len > sizeof(buf))
342 return ENOSPC;
343 len += snprintf(&buf[len], sizeof(buf) - len,
344 "irq: %d\n", sc->sc_irq);
345 if (off >= len) {
346 error = uiomove(buf, 0, uio);
347 } else {
348 error = uiomove(&buf[off], len - off, uio);
349 }
350 return error;
351}

353static struct pciback_pci_dev*
354pciback_pci_lookup(u_int bus, u_int dev, u_int func)
355{
356 struct pciback_pci_dev *pbd;
357 /* Safe without lock, only written during init */
358 SLIST_FOREACH(pbd, &pciback_pci_devlist_head, pb_devlist_next) {
359 if (pbd->pb_bus == bus &&
360 pbd->pb_device == dev &&
361 pbd->pb_function == func)
362 return pbd;
363 }
364 return NULL;
365}

367static void
368pciback_pci_init(void)
369{
370 union xen_cmdline_parseinfo xi;
371 char *pcidevs, *c;
372 u_int bus, dev, func;
373 struct pciback_pci_dev *pb;
374 kernfs_entry_t *dkt;
375
376 memset(&xi, 0, sizeof(xi));
377 xen_parse_cmdline(XEN_PARSE_PCIBACK, &xi);
378 if (strlen(xi.xcp_pcidevs) == 0)
379 return;
380 pcidevs = xi.xcp_pcidevs;
381 for (pcidevs = xi.xcp_pcidevs; *pcidevs != '\0';) {
382 if (*pcidevs != '(')
383 goto error;
384 pcidevs++;
385 /* parse location */
386 c = strchr(pcidevs, ')');
387 if (c == NULL)
388 goto error;
389 *c = '\0';
390 if (pciback_parse_pci(pcidevs, &bus, &dev, &func) == 0) {
391 pb = kmem_zalloc(sizeof(*pb), KM_SLEEP);
392 pb->pb_bus = bus;
393 pb->pb_device = dev;
394 pb->pb_function = func;
395 aprint_verbose("pciback_pci_init: hide claim device "
396 "%x:%x:%x\n", bus, dev, func);
397 SLIST_INSERT_HEAD(&pciback_pci_devlist_head, pb,
398 pb_devlist_next);
399 }
400 pcidevs = c + 1;
401 }
402
403 SLIST_INIT(&pb_xenbus_instances);
404 mutex_init(&pb_xenbus_lock, MUTEX_DEFAULT, IPL_NONE);
405
406 xenbus_backend_register(&pci_backend_driver);
407
408 KERNFS_ALLOCENTRY(dkt, M_TEMP, M_WAITOK);
409 KERNFS_INITENTRY(dkt, DT_DIR, "pci", NULL, KFSsubdir, VDIR, DIR_MODE);
410 kernfs_addentry(kernxen_pkt, dkt);
411 pciback_kern_pkt = KERNFS_ENTOPARENTDIR(dkt);
412 return;
413error:
414 aprint_error("pciback_pci_init: syntax error at %s\n", pcidevs);
415 return;
416}

418static int
419pciback_parse_pci(const char *str, u_int *busp, u_int *devp, u_int *funcp)
420{
421 char *c;
422
423 /* parse location */
424 c = strchr(str, ':');
425 if (c == NULL)
426 goto error;
427 if (strncmp("0000", str, 4) == 0) {
428 /* must be domain number, get next */
429 str = c + 1;
430 }
431 *busp = strtoul(str, &c, 16);
432 if (*c != ':')
433 goto error;
434 str = c + 1;
435 *devp = strtoul(str, &c, 16);
436 if (*c != '.')
437 goto error;
438 str = c + 1;
439 *funcp = strtoul(str, &c, 16);
440 if (*c != '\0')
441 goto error;
442 return 0;
443error:
444 aprint_error("pciback_pci_init: syntax error at char %c\n", *c);
445 return EINVAL;
446}

448static int
449pciback_xenbus_create(struct xenbus_device *xbusd)
450{
451 struct pb_xenbus_instance *pbxi;
452 long domid;
453 char *val;
454 char path[10];
455 int i, err;
456 u_long num_devs;
457
458 if ((err = xenbus_read_ul(NULL, xbusd->xbusd_path,
459 "frontend-id", &domid, 10)) != 0) {
460 aprint_error("pciback: can' read %s/frontend-id: %d\n",
461 xbusd->xbusd_path, err);
462 return err;
463 }
464
465 if (pbxif_lookup(domid)) {
466 return EEXIST;
467 }
468 pbxi = kmem_zalloc(sizeof(*pbxi), KM_SLEEP);
469 pbxi->pbx_domid = domid;
470
471 xbusd->xbusd_u.b.b_cookie = pbxi;
472 xbusd->xbusd_u.b.b_detach = pciback_xenbus_destroy;
473 pbxi->pbx_xbusd = xbusd;
474
475 SLIST_INIT(&pbxi->pbx_pb_pci_dev);
476
477 mutex_enter(&pb_xenbus_lock);
478 SLIST_INSERT_HEAD(&pb_xenbus_instances, pbxi, pbx_next);
479 mutex_exit(&pb_xenbus_lock);
480
481 xbusd->xbusd_otherend_changed = pciback_xenbus_frontend_changed;
482
483 err = xenbus_switch_state(xbusd, NULL, XenbusStateInitWait);
484 if (err) {
485 printf("failed to switch state on %s: %d\n",
486 xbusd->xbusd_path, err);
487 goto fail;
488 }
489 if ((err = xenbus_read_ul(NULL, xbusd->xbusd_path, "num_devs",
490 &num_devs, 10)) != 0) {
491 aprint_error("pciback: can' read %s/num_devs %d\n",
492 xbusd->xbusd_path, err);
493 goto fail;
494 }
495 for (i = 0; i < num_devs; i++) {
496 snprintf(path, sizeof(path), "dev-%d", i);
497 if ((err = xenbus_read(NULL, xbusd->xbusd_path, path,
498 NULL, &val)) != 0) {
499 aprint_error("pciback: can' read %s/%s: %d\n",
500 xbusd->xbusd_path, path, err);
501 goto fail;
502 }
503 pciback_xenbus_export_device(pbxi, val);
504 free(val, M_DEVBUF);
505 }
506 pciback_xenbus_export_roots(pbxi);
507 if ((err = xenbus_switch_state(xbusd, NULL, XenbusStateInitialised))) {
508 printf("failed to switch state on %s: %d\n",
509 xbusd->xbusd_path, err);
510 goto fail;
511 }
512
513 return 0;
514fail:
515 kmem_free(pbxi, sizeof(*pbxi));
516 return err;
517}

519static int
520pciback_xenbus_destroy(void *arg)
521{
522 struct pb_xenbus_instance *pbxi = arg;
523 struct pciback_pci_dev *pbd;
524 struct gnttab_unmap_grant_ref op;
525 int err;
526
527 hypervisor_mask_event(pbxi->pbx_evtchn);
528 xen_intr_disestablish(pbxi->pbx_ih);
529 mutex_enter(&pb_xenbus_lock);
530 SLIST_REMOVE(&pb_xenbus_instances,
531 pbxi, pb_xenbus_instance, pbx_next);
532 mutex_exit(&pb_xenbus_lock);
533
534 if (pbxi->pbx_sh_info) {
535 op.host_addr = (vaddr_t)pbxi->pbx_sh_info;
536 op.handle = pbxi->pbx_shinfo_handle;
537 op.dev_bus_addr = 0;
538 err = HYPERVISOR_grant_table_op(GNTTABOP_unmap_grant_ref,
539 &op, 1);
540 if (err)
541 aprint_error("pciback: unmap_grant_ref failed: %d\n",
542 err);
543 }
544 SLIST_FOREACH(pbd, &pbxi->pbx_pb_pci_dev, pb_guest_next) {
545 pbd->pbx_instance = NULL;
546 }
547 uvm_km_free(kernel_map, (vaddr_t)pbxi->pbx_sh_info,
548 PAGE_SIZE, UVM_KMF_VAONLY);
549 kmem_free(pbxi, sizeof(*pbxi));
550 return 0;
551}

553static void
554pciback_xenbus_frontend_changed(void *arg, XenbusState new_state)
555{
556 struct pb_xenbus_instance *pbxi = arg;
557 struct xenbus_device *xbusd = pbxi->pbx_xbusd;
558 int err;
559 struct gnttab_map_grant_ref op;
560 evtchn_op_t evop;
561 u_long shared_ref;
562 u_long revtchn;
563
564 /* do it only once */
565 if (xenbus_read_driver_state(xbusd->xbusd_path) !=
566 XenbusStateInitialised)
567 return;
568
569 switch(new_state) {
570 case XenbusStateInitialising:
571 case XenbusStateConnected:
572 break;
573 case XenbusStateInitialised:
574 /* read comunication informations */
575 err = xenbus_read_ul(NULL, xbusd->xbusd_otherend,
576 "pci-op-ref", &shared_ref, 10);
577 if (err) {
578 xenbus_dev_fatal(xbusd, err, "reading %s/pci-op-ref",
579 xbusd->xbusd_otherend);
580 break;
581 }
582 err = xenbus_read_ul(NULL, xbusd->xbusd_otherend,
583 "event-channel", &revtchn, 10);
584 if (err) {
585 xenbus_dev_fatal(xbusd, err, "reading %s/event-channel",
586 xbusd->xbusd_otherend);
587 break;
588 }
589 /* allocate VA space and map rings */
590 pbxi->pbx_sh_info = (void *)uvm_km_alloc(kernel_map,
591 PAGE_SIZE, 0, UVM_KMF_VAONLY);
592 if (pbxi->pbx_sh_info == 0) {
593 xenbus_dev_fatal(xbusd, ENOMEM,
594 "can't get VA for shared infos",
595 xbusd->xbusd_otherend);
596 break;
597 }
598 op.host_addr = (vaddr_t)pbxi->pbx_sh_info;
599 op.flags = GNTMAP_host_map;
600 op.ref = shared_ref;
601 op.dom = pbxi->pbx_domid;
602 err = HYPERVISOR_grant_table_op(GNTTABOP_map_grant_ref, &op, 1);
603 if (err || op.status) {
604 aprint_error("pciback: can't map shared grant ref: "
605 "%d/%d\n", err, op.status);
606 goto err1;
607 }
608 pbxi->pbx_shinfo_handle = op.handle;
609
610 evop.cmd = EVTCHNOP_bind_interdomain;
611 evop.u.bind_interdomain.remote_dom = pbxi->pbx_domid;
612 evop.u.bind_interdomain.remote_port = revtchn;
613 err = HYPERVISOR_event_channel_op(&evop);
614 if (err) {
615 printf("pciback: can't get event channel: %d\n", err);
616 goto err1;
617 }
618 pbxi->pbx_evtchn = evop.u.bind_interdomain.local_port;
619 x86_sfence();
620 xenbus_switch_state(xbusd, NULL, XenbusStateConnected);
621 x86_sfence();
622 pbxi->pbx_ih = xen_intr_establish_xname(-1, &xen_pic, pbxi->pbx_evtchn,
623 IST_LEVEL, IPL_BIO, pciback_xenbus_evthandler, pbxi, true, "pciback");
624 KASSERT(pbxi->pbx_ih != NULL);
625 hypervisor_unmask_event(pbxi->pbx_evtchn);
626 hypervisor_notify_via_evtchn(pbxi->pbx_evtchn);
627 break;
628
629 case XenbusStateClosing:
630 xenbus_switch_state(xbusd, NULL, XenbusStateClosing);
631 break;
632
633 case XenbusStateClosed:
634 /* otherend_changed() should handle it for us */
635 panic("pciback_xenbus_frontend_changed: closed\n");
636 case XenbusStateUnknown:
637 case XenbusStateInitWait:
638 default:
639 aprint_error("pciback: invalid frontend state %d\n",
640 new_state);
641 break;
642 }
643 return;
644err1:
645 uvm_km_free(kernel_map, (vaddr_t)pbxi->pbx_sh_info,
646 PAGE_SIZE, UVM_KMF_VAONLY);
647}

649/* lookup a pbxi based on domain id and interface handle */
650static bool
651pbxif_lookup(domid_t dom)
652{
653 struct pb_xenbus_instance *pbxi;
654 bool found = false;
655
656 mutex_enter(&pb_xenbus_lock);
657 SLIST_FOREACH(pbxi, &pb_xenbus_instances, pbx_next) {
658 if (pbxi->pbx_domid == dom) {
659 found = true;
660 break;
661 }
662 }
663 mutex_exit(&pb_xenbus_lock);
664
665 return found;
666}

668static void
669pciback_xenbus_export_device(struct pb_xenbus_instance *pbxi, char *val)
670{
671 u_int bus, dev, func;
672 struct pciback_pci_dev *pbd;
673 if (pciback_parse_pci(val, &bus, &dev, &func)) {
674 aprint_error("pciback: can't parse %s\n", val);
675 return;
676 }
677 pbd = pciback_pci_lookup(bus, dev, func);
678 if (pbd == NULL) {
679 aprint_error("pciback: can't locate 0x%02x:0x%02x.0x%02x\n",
680 bus, dev, func);
681 return;
682 }
683 if (pbd->pb_pci_softc == NULL) {
684 aprint_error("pciback: 0x%02x:0x%02x.0x%02x not detected\n",
685 bus, dev, func);
686 return;
687 }
688 pbd->pbx_instance = pbxi;
689 SLIST_INSERT_HEAD(&pbxi->pbx_pb_pci_dev, pbd, pb_guest_next);
690 return;
691}

693static void
694pciback_xenbus_export_roots(struct pb_xenbus_instance *pbxi)
695{
696 char bus[256];
697 char root[10];
698 struct pciback_pci_dev *pbd;
699 int num_roots = 0;
700 int err;
701
702 memset(bus, 0, sizeof(bus));
703
704 SLIST_FOREACH(pbd, &pbxi->pbx_pb_pci_dev, pb_guest_next) {
705 if (bus[pbd->pb_bus] == 0) {
706 /* not published yet */
707 snprintf(root, sizeof(root), "root-%d", num_roots);
708 err = xenbus_printf(NULL,
709 pbxi->pbx_xbusd->xbusd_path,
710 root, "0000:%02x", pbd->pb_bus);
711 if (err) {
712 aprint_error("pciback: can't write to %s/%s: "
713 "%d\n", pbxi->pbx_xbusd->xbusd_path, root,
714 err);
715 }
716 num_roots++;
717 bus[pbd->pb_bus]++;
718 }
719 }
720 err = xenbus_printf(NULL, pbxi->pbx_xbusd->xbusd_path, "root_num",
721 "%d", num_roots);
722 if (err) {
723 aprint_error("pciback: can't write to %s/root_num: "
724 "%d\n", pbxi->pbx_xbusd->xbusd_path, err);
725 }
726}

728static int
729pciback_xenbus_evthandler(void * arg)
730{
731 struct pb_xenbus_instance *pbxi = arg;
732 struct pciback_pci_dev *pbd;
733 struct xen_pci_op *op = &pbxi->op;
734 u_int bus, dev, func;
735
736 hypervisor_clear_event(pbxi->pbx_evtchn);
737 if (xen_atomic_test_bit(&pbxi->pbx_sh_info->flags,
738 _XEN_PCIF_active) == 0)
739 return 0;
740
741 memcpy(op, &pbxi->pbx_sh_info->op, sizeof (struct xen_pci_op));
742 __insn_barrier();
743 if (op->domain != 0) {
744 aprint_error("pciback: domain %d != 0", op->domain);
745 op->err = XEN_PCI_ERR_dev_not_found;
746 goto end;
747 }
748 bus = op->bus;
749 dev = (op->devfn >> 3) & 0xff;
750 func = (op->devfn) & 0x7;
751 SLIST_FOREACH(pbd, &pbxi->pbx_pb_pci_dev, pb_guest_next) {
752 if (pbd->pb_bus == bus &&
753 pbd->pb_device == dev &&
754 pbd->pb_function == func)
755 break;
756 }
757 if (pbd == NULL) {
758 aprint_debug("pciback: %02x:%02x.%x not found\n",
759 bus, dev, func);
760 op->err = XEN_PCI_ERR_dev_not_found;
761 goto end;
762 }
763 switch(op->cmd) {
764 case XEN_PCI_OP_conf_read:
765 op->err = XEN_PCI_ERR_success;
766 switch (op->size) {
767 case 1:
768 op->value = pciback_read_byte(pbd->pb_pc, pbd->pb_tag,
769 op->offset);
770 break;
771 case 2:
772 op->value = pciback_read_word(pbd->pb_pc, pbd->pb_tag,
773 op->offset);
774 break;
775 case 4:
776 op->value = pci_conf_read(pbd->pb_pc, pbd->pb_tag,
777 op->offset);
778 break;
779 default:
780 aprint_error("pciback: bad size %d\n", op->size);
781 break;
782 }
783 break;
784 case XEN_PCI_OP_conf_write:
785 op->err = XEN_PCI_ERR_success;
786 switch(op->size) {
787 case 1:
788 pciback_write_byte(pbd->pb_pc, pbd->pb_tag,
789 op->offset, op->value);
790 break;
791 case 2:
792 pciback_write_word(pbd->pb_pc, pbd->pb_tag,
793 op->offset, op->value);
794 break;
795 case 4:
796 pci_conf_write(pbd->pb_pc, pbd->pb_tag,
797 op->offset, op->value);
798 break;
799 default:
800 aprint_error("pciback: bad size %d\n", op->size);
801 op->err = XEN_PCI_ERR_invalid_offset;
802 break;
803 }
804 break;
805 default:
806 aprint_error("pciback: unknown cmd %d\n", op->cmd);
807 op->err = XEN_PCI_ERR_not_implemented;
808 }
809 pbxi->pbx_sh_info->op.value = op->value;
810 pbxi->pbx_sh_info->op.err = op->err;
811end:
812 xen_atomic_clear_bit(&pbxi->pbx_sh_info->flags, _XEN_PCIF_active);
813 hypervisor_notify_via_evtchn(pbxi->pbx_evtchn);
814 return 1;
815}
