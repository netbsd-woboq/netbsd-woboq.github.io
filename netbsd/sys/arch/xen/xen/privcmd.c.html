254static int
255privcmd_ioctl(void *v)
256{
257 struct vop_ioctl_args /* {
258 const struct vnodeop_desc *a_desc;
259 struct vnode *a_vp;
260 u_long a_command;
261 void *a_data;
262 int a_fflag;
263 kauth_cred_t a_cred;
264 } */ *ap = v;
265 int error = 0;
266 paddr_t *maddr;
267
268 switch (ap->a_command) {
269 case IOCTL_PRIVCMD_HYPERCALL:
270 case IOCTL_PRIVCMD_HYPERCALL_OLD:
271 /*
272 * oprivcmd_hypercall_t is privcmd_hypercall_t without the last entry
273 */
274 {
275 privcmd_hypercall_t *hc = ap->a_data;
276 if (hc->op >= (PAGE_SIZE >> 5))
277 return EINVAL;
278 error = -EOPNOTSUPP;
279#if defined(__i386__)
280 __asm volatile (
281 "pushl %%ebx; pushl %%ecx; pushl %%edx;"
282 "pushl %%esi; pushl %%edi; "
283 "movl 4(%%eax),%%ebx ;"
284 "movl 8(%%eax),%%ecx ;"
285 "movl 12(%%eax),%%edx ;"
286 "movl 16(%%eax),%%esi ;"
287 "movl 20(%%eax),%%edi ;"
288 "movl (%%eax),%%eax ;"
289 "shll $5,%%eax ;"
290 "addl $hypercall_page,%%eax ;"
291 "call *%%eax ;"
292 "popl %%edi; popl %%esi; popl %%edx;"
293 "popl %%ecx; popl %%ebx"
294 : "=a" (error) : "0" (ap->a_data) : "memory" );
295#endif /* __i386__ */
296#if defined(__x86_64__)
297 {
298 long i1, i2, i3;
299 __asm volatile (
300 "movq %8,%%r10; movq %9,%%r8;"
301 "shll $5,%%eax ;"
302 "addq $hypercall_page,%%rax ;"
303 "call *%%rax"
304 : "=a" (error), "=D" (i1),
305 "=S" (i2), "=d" (i3)
306 : "0" ((unsigned int)hc->op),
307 "1" (hc->arg[0]),
308 "2" (hc->arg[1]),
309 "3" (hc->arg[2]),
310 "g" (hc->arg[3]),
311 "g" (hc->arg[4])
312 : "r8", "r10", "memory" );
313 }
314#endif /* __x86_64__ */
315 if (ap->a_command == IOCTL_PRIVCMD_HYPERCALL) {
316 if (error >= 0) {
317 hc->retval = error;
318 error = 0;
319 } else {
320 /* error occured, return the errno */
321 error = privcmd_xen2bsd_errno(error);
322 hc->retval = 0;
323 }
324 } else {
325 error = privcmd_xen2bsd_errno(error);
326 }
327 break;
328 }
329 case IOCTL_PRIVCMD_MMAP:
330 {
331 int i, j;
332 privcmd_mmap_t *mcmd = ap->a_data;
333 privcmd_mmap_entry_t mentry;
334 vaddr_t va;
335 paddr_t ma;
336 struct vm_map *vmm = &curlwp->l_proc->p_vmspace->vm_map;
337
338 for (i = 0; i < mcmd->num; i++) {
339 error = copyin(&mcmd->entry[i], &mentry, sizeof(mentry));
340 if (error)
341 return error;
342 if (mentry.npages == 0)
343 return EINVAL;
344 if (mentry.va > VM_MAXUSER_ADDRESS)
345 return EINVAL;
346#if 0
347 if (mentry.va + (mentry.npages << PGSHIFT) >
348 mrentry->vm_end)
349 return EINVAL;
350#endif
351 maddr = kmem_alloc(sizeof(paddr_t) * mentry.npages,
352 KM_SLEEP);
353 va = mentry.va & ~PAGE_MASK;
354 ma = ((paddr_t)mentry.mfn) << PGSHIFT; /* XXX ??? */
355 for (j = 0; j < mentry.npages; j++) {
356 maddr[j] = ma;
357 ma += PAGE_SIZE;
358 }
359 error = privcmd_map_obj(vmm, va, maddr,
360 mentry.npages, mcmd->dom);
361 if (error)
362 return error;
363 }
364 break;
365 }
366 case IOCTL_PRIVCMD_MMAPBATCH:
367 {
368 int i;
369 privcmd_mmapbatch_t* pmb = ap->a_data;
370 vaddr_t va0;
371 u_long mfn;
372 paddr_t ma;
373 struct vm_map *vmm;
374 struct vm_map_entry *entry;
375 vm_prot_t prot;
376 vaddr_t trymap;
377
378 vmm = &curlwp->l_proc->p_vmspace->vm_map;
379 va0 = pmb->addr & ~PAGE_MASK;
380
381 if (pmb->num == 0)
382 return EINVAL;
383 if (va0 > VM_MAXUSER_ADDRESS)
384 return EINVAL;
385 if (((VM_MAXUSER_ADDRESS - va0) >> PGSHIFT) < pmb->num)
386 return EINVAL;
387
388 vm_map_lock_read(vmm);
389 if (!uvm_map_lookup_entry(vmm, va0, &entry)) {
390 vm_map_unlock_read(vmm);
391 return EINVAL;
392 }
393 prot = entry->protection;
394 vm_map_unlock_read(vmm);
395
396 maddr = kmem_alloc(sizeof(paddr_t) * pmb->num, KM_SLEEP);
397 /* get a page of KVA to check mappins */
398 trymap = uvm_km_alloc(kernel_map, PAGE_SIZE, PAGE_SIZE,
399 UVM_KMF_VAONLY);
400 if (trymap == 0) {
401 kmem_free(maddr, sizeof(paddr_t) * pmb->num);
402 return ENOMEM;
403 }
404
405 for(i = 0; i < pmb->num; ++i) {
406 error = copyin(&pmb->arr[i], &mfn, sizeof(mfn));
407 if (error != 0) {
408 /* XXX: mappings */
409 pmap_update(pmap_kernel());
410 kmem_free(maddr, sizeof(paddr_t) * pmb->num);
411 uvm_km_free(kernel_map, trymap, PAGE_SIZE,
412 UVM_KMF_VAONLY);
413 return error;
414 }
415 ma = ((paddr_t)mfn) << PGSHIFT;
416 if (pmap_enter_ma(pmap_kernel(), trymap, ma, 0,
417 prot, PMAP_CANFAIL, pmb->dom)) {
418 mfn |= 0xF0000000;
419 copyout(&mfn, &pmb->arr[i], sizeof(mfn));
420 maddr[i] = INVALID_PAGE;
421 } else {
422 pmap_remove(pmap_kernel(), trymap,
423 trymap + PAGE_SIZE);
424 maddr[i] = ma;
425 }
426 }
427 pmap_update(pmap_kernel());
428
429 error = privcmd_map_obj(vmm, va0, maddr, pmb->num, pmb->dom);
430 uvm_km_free(kernel_map, trymap, PAGE_SIZE, UVM_KMF_VAONLY);
431
432 if (error != 0)
433 return error;
434
435 break;
436 }
437 default:
438 error = EINVAL;
439 }
440
441 return error;
442}

444static struct uvm_pagerops privpgops = {
445 .pgo_reference = privpgop_reference,
446 .pgo_detach = privpgop_detach,
447 .pgo_fault = privpgop_fault,
448};
449
450static void
451privpgop_reference(struct uvm_object *uobj)
452{
453 mutex_enter(uobj->vmobjlock);
454 uobj->uo_refs++;
455 mutex_exit(uobj->vmobjlock);
456}
457
458static void
459privpgop_detach(struct uvm_object *uobj)
460{
461 struct privcmd_object *pobj = (struct privcmd_object *)uobj;
462
463 mutex_enter(uobj->vmobjlock);
464 if (uobj->uo_refs > 1) {
465 uobj->uo_refs--;
466 mutex_exit(uobj->vmobjlock);
467 return;
468 }
469 mutex_exit(uobj->vmobjlock);
470 kmem_free(pobj->maddr, sizeof(paddr_t) * pobj->npages);
471 uvm_obj_destroy(uobj, true);
472 kmem_free(pobj, sizeof(struct privcmd_object));
473 privcmd_nobjects--;
474}

476static int
477privpgop_fault(struct uvm_faultinfo *ufi, vaddr_t vaddr, struct vm_page **pps,
478 int npages, int centeridx, vm_prot_t access_type, int flags)
479{
480 struct vm_map_entry *entry = ufi->entry;
481 struct uvm_object *uobj = entry->object.uvm_obj;
482 struct privcmd_object *pobj = (struct privcmd_object*)uobj;
483 int maddr_i, i, error = 0;
484
485 /* compute offset from start of map */
486 maddr_i = (entry->offset + (vaddr - entry->start)) >> PAGE_SHIFT;
487 if (maddr_i + npages > pobj->npages) {
488 return EINVAL;
489 }
490 for (i = 0; i < npages; i++, maddr_i++, vaddr+= PAGE_SIZE) {
491 if ((flags & PGO_ALLPAGES) == 0 && i != centeridx)
492 continue;
493 if (pps[i] == PGO_DONTCARE)
494 continue;
495 if (pobj->maddr[maddr_i] == INVALID_PAGE) {
496 /* This has already been flagged as error. */
497 error = EFAULT;
498 break;
499 }
500 error = pmap_enter_ma(ufi->orig_map->pmap, vaddr,
501 pobj->maddr[maddr_i], 0, ufi->entry->protection,
502 PMAP_CANFAIL | ufi->entry->protection,
503 pobj->domid);
504 if (error == ENOMEM) {
505 error = ERESTART;
506 break;
507 }
508 if (error) {
509 /* XXX for proper ptp accountings */
510 pmap_remove(ufi->orig_map->pmap, vaddr,
511 vaddr + PAGE_SIZE);
512 }
513 }
514 pmap_update(ufi->orig_map->pmap);
515 uvmfault_unlockall(ufi, ufi->entry->aref.ar_amap, uobj);
516
517 if (error == ERESTART) {
518 uvm_wait("privpgop_fault");
519 }
520 return error;
521}

523static int
524privcmd_map_obj(struct vm_map *map, vaddr_t start, paddr_t *maddr,
525 int npages, int domid)
526{
527 struct privcmd_object *obj;
528 int error;
529 uvm_flag_t uvmflag;
530 vaddr_t newstart = start;
531 vm_prot_t prot;
532 off_t size = ((off_t)npages << PGSHIFT);
533
534 vm_map_lock_read(map);
535 /* get protections. This also check for validity of mapping */
536 if (uvm_map_checkprot(map, start, start + size - 1, VM_PROT_WRITE))
537 prot = VM_PROT_READ | VM_PROT_WRITE;
538 else if (uvm_map_checkprot(map, start, start + size - 1, VM_PROT_READ))
539 prot = VM_PROT_READ;
540 else {
541 printf("uvm_map_checkprot 0x%lx -> 0x%lx "
542 "failed\n",
543 start, (unsigned long)(start + size - 1));
544 vm_map_unlock_read(map);
545 kmem_free(maddr, sizeof(paddr_t) * npages);
546 return EINVAL;
547 }
548 vm_map_unlock_read(map);
549
550 obj = kmem_alloc(sizeof(*obj), KM_SLEEP);
551 privcmd_nobjects++;
552 uvm_obj_init(&obj->uobj, &privpgops, true, 1);
553 obj->maddr = maddr;
554 obj->npages = npages;
555 obj->domid = domid;
556 uvmflag = UVM_MAPFLAG(prot, prot, UVM_INH_NONE, UVM_ADV_NORMAL,
557 UVM_FLAG_FIXED | UVM_FLAG_UNMAP | UVM_FLAG_NOMERGE);
558 error = uvm_map(map, &newstart, size, &obj->uobj, 0, 0, uvmflag);
559
560 if (error)
561 obj->uobj.pgops->pgo_detach(&obj->uobj);
562 return error;
563}

569void
570xenprivcmd_init(void)
571{
572 kernfs_entry_t *dkt;
573 kfstype kfst;
574
575 if (!xendomain_is_privileged())
576 return;
577
578 kfst = KERNFS_ALLOCTYPE(privcmd_fileops);
579
580 KERNFS_ALLOCENTRY(dkt, M_TEMP, M_WAITOK);
581 KERNFS_INITENTRY(dkt, DT_REG, "privcmd", NULL, kfst, VREG,
582 PRIVCMD_MODE);
583 kernfs_addentry(kernxen_pkt, dkt);
584}
