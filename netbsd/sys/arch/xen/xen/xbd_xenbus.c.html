28/*
29 * The file contains the xbd frontend code required for block-level
30 * communications (similar to hard disks) between two Xen domains.
31 *
32 * We are not supposed to receive solicitations spontaneously from backend. The
33 * protocol is therefore fairly simple and uses only one ring to communicate
34 * with backend: frontend posts requests to the ring then wait for their
35 * replies asynchronously.
36 *
37 * xbd follows NetBSD's disk(9) convention. At any time, a LWP can schedule
38 * an operation request for the device (be it open(), read(), write(), ...).
39 * Calls are typically processed that way:
40 * - initiate request: xbdread/write/open/ioctl/..
41 * - depending on operation, it is handled directly by disk(9) subsystem or
42 * goes through physio(9) first.
43 * - the request is ultimately processed by xbd_diskstart() that prepares the
44 * xbd requests, post them in the ring I/O queue, then signal the backend.
45 *
46 * When a response is available in the queue, the backend signals the frontend
47 * via its event channel. This triggers xbd_handler(), which will link back
48 * the response to its request through the request ID, and mark the I/O as
49 * completed.
50 */

100struct xbd_req {
101 SLIST_ENTRY(xbd_req) req_next;
102 uint16_t req_id; /* ID passed to backend */
103 union {
104 struct {
105 grant_ref_t req_gntref[BLKIF_MAX_SEGMENTS_PER_REQUEST];
106 int req_nr_segments; /* number of segments in this request */
107 struct buf *req_bp; /* buffer associated with this request */
108 void *req_data; /* pointer to the data buffer */
109 } req_rw;
110 struct {
111 int s_error;
112 volatile int s_done;
113 } req_sync;
114 } u;
115};
116#define req_gntref u.req_rw.req_gntref
117#define req_nr_segments u.req_rw.req_nr_segments
118#define req_bp u.req_rw.req_bp
119#define req_data u.req_rw.req_data
120#define req_sync u.req_sync

229static int
230xbd_xenbus_match(device_t parent, cfdata_t match, void *aux)
231{
232 struct xenbusdev_attach_args *xa = aux;
233
234 if (strcmp(xa->xa_type, "vbd") != 0)
235 return 0;
236
237 if (match->cf_loc[XENBUSCF_ID] != XENBUSCF_ID_DEFAULT &&
238 match->cf_loc[XENBUSCF_ID] != xa->xa_id)
239 return 0;
240
241 return 1;
242}

244static void
245xbd_xenbus_attach(device_t parent, device_t self, void *aux)
246{
247 struct xbd_xenbus_softc *sc = device_private(self);
248 struct xenbusdev_attach_args *xa = aux;
249 blkif_sring_t *ring;
250 RING_IDX i;
251#ifdef XBD_DEBUG
252 char **dir, *val;
253 int dir_n = 0;
254 char id_str[20];
255 int err;
256#endif
257
258 config_pending_incr(self);
259 aprint_normal(": Xen Virtual Block Device Interface\n");
260
261 dk_init(&sc->sc_dksc, self, DKTYPE_ESDI);
262 disk_init(&sc->sc_dksc.sc_dkdev, device_xname(self), &xbddkdriver);
263
264#ifdef XBD_DEBUG
265 printf("path: %s\n", xa->xa_xbusd->xbusd_path);
266 snprintf(id_str, sizeof(id_str), "%d", xa->xa_id);
267 err = xenbus_directory(NULL, "device/vbd", id_str, &dir_n, &dir);
268 if (err) {
269 aprint_error_dev(self, "xenbus_directory err %d\n", err);
270 } else {
271 printf("%s/\n", xa->xa_xbusd->xbusd_path);
272 for (i = 0; i < dir_n; i++) {
273 printf("\t/%s", dir[i]);
274 err = xenbus_read(NULL, xa->xa_xbusd->xbusd_path,
275 dir[i], NULL, &val);
276 if (err) {
277 aprint_error_dev(self, "xenbus_read err %d\n",
278 err);
279 } else {
280 printf(" = %s\n", val);
281 free(val, M_DEVBUF);
282 }
283 }
284 }
285#endif /* XBD_DEBUG */
286 sc->sc_xbusd = xa->xa_xbusd;
287 sc->sc_xbusd->xbusd_otherend_changed = xbd_backend_changed;
288
289 /* initialize free requests list */
290 SLIST_INIT(&sc->sc_xbdreq_head);
291 for (i = 0; i < XBD_RING_SIZE; i++) {
292 sc->sc_reqs[i].req_id = i;
293 SLIST_INSERT_HEAD(&sc->sc_xbdreq_head, &sc->sc_reqs[i],
294 req_next);
295 }
296
297 sc->sc_backend_status = BLKIF_STATE_DISCONNECTED;
298 sc->sc_shutdown = BLKIF_SHUTDOWN_REMOTE;
299
300 ring = (void *)uvm_km_alloc(kernel_map, PAGE_SIZE, 0, UVM_KMF_WIRED);
301 if (ring == NULL)
302 panic("%s: can't alloc ring", device_xname(self));
303 sc->sc_ring.sring = ring;
304
305 evcnt_attach_dynamic(&sc->sc_cnt_map_unalign, EVCNT_TYPE_MISC,
306 NULL, device_xname(self), "map unaligned");
307
308 /* resume shared structures and tell backend that we are ready */
309 if (xbd_xenbus_resume(self, PMF_Q_NONE) == false) {
310 uvm_km_free(kernel_map, (vaddr_t)ring, PAGE_SIZE,
311 UVM_KMF_WIRED);
312 return;
313 }
314
315 if (!pmf_device_register(self, xbd_xenbus_suspend, xbd_xenbus_resume))
316 aprint_error_dev(self, "couldn't establish power handler\n");
317
318}

320static int
321xbd_xenbus_detach(device_t dev, int flags)
322{
323 struct xbd_xenbus_softc *sc = device_private(dev);
324 int bmaj, cmaj, i, mn, rc, s;
325
326 rc = disk_begindetach(&sc->sc_dksc.sc_dkdev, NULL, dev, flags);
327 if (rc != 0)
328 return rc;
329
330 s = splbio(); /* XXXSMP */
331 DPRINTF(("%s: xbd_detach\n", device_xname(dev)));
332 if (sc->sc_shutdown == BLKIF_SHUTDOWN_RUN) {
333 sc->sc_shutdown = BLKIF_SHUTDOWN_LOCAL;
334 /* wait for requests to complete */
335 while (sc->sc_backend_status == BLKIF_STATE_CONNECTED &&
336 disk_isbusy(&sc->sc_dksc.sc_dkdev)) {
337 /* XXXSMP */
338 tsleep(xbd_xenbus_detach, PRIBIO, "xbddetach", hz/2);
339 }
340
341 xenbus_switch_state(sc->sc_xbusd, NULL, XenbusStateClosing);
342 }
343 if ((flags & DETACH_FORCE) == 0) {
344 /* xbd_xenbus_detach already in progress */
345 wakeup(xbd_xenbus_detach); /* XXXSMP */
346 splx(s);
347 return EALREADY;
348 }
349 while (xenbus_read_driver_state(sc->sc_xbusd->xbusd_otherend)
350 != XenbusStateClosed) {
351 /* XXXSMP */
352 tsleep(xbd_xenbus_detach, PRIBIO, "xbddetach2", hz/2);
353 }
354 splx(s);
355
356 /* locate the major number */
357 bmaj = bdevsw_lookup_major(&xbd_bdevsw);
358 cmaj = cdevsw_lookup_major(&xbd_cdevsw);
359
360 /* Nuke the vnodes for any open instances. */
361 for (i = 0; i < MAXPARTITIONS; i++) {
362 mn = DISKMINOR(device_unit(dev), i);
363 vdevgone(bmaj, mn, mn, VBLK);
364 vdevgone(cmaj, mn, mn, VCHR);
365 }
366 if (sc->sc_backend_status == BLKIF_STATE_CONNECTED) {
367 /* Delete all of our wedges. */
368 dkwedge_delall(&sc->sc_dksc.sc_dkdev);
369
370 /* Kill off any queued buffers. */
371 dk_drain(&sc->sc_dksc);
372 bufq_free(sc->sc_dksc.sc_bufq);
373
374 /* detach disk */
375 disk_detach(&sc->sc_dksc.sc_dkdev);
376 disk_destroy(&sc->sc_dksc.sc_dkdev);
377 dk_detach(&sc->sc_dksc);
378 }
379
380 hypervisor_mask_event(sc->sc_evtchn);
381 xen_intr_disestablish(sc->sc_ih);
382
383 while (xengnt_status(sc->sc_ring_gntref)) {
384 /* XXXSMP */
385 tsleep(xbd_xenbus_detach, PRIBIO, "xbd_ref", hz/2);
386 }
387 xengnt_revoke_access(sc->sc_ring_gntref);
388 uvm_km_free(kernel_map, (vaddr_t)sc->sc_ring.sring,
389 PAGE_SIZE, UVM_KMF_WIRED);
390
391 evcnt_detach(&sc->sc_cnt_map_unalign);
392
393 pmf_device_deregister(dev);
394
395 return 0;
396}

398static bool
399xbd_xenbus_suspend(device_t dev, const pmf_qual_t *qual) {
400
401 int s;
402 struct xbd_xenbus_softc *sc;
403
404 sc = device_private(dev);
405
406 s = splbio(); /* XXXSMP */
407 /* wait for requests to complete, then suspend device */
408 while (sc->sc_backend_status == BLKIF_STATE_CONNECTED &&
409 disk_isbusy(&sc->sc_dksc.sc_dkdev)) {
410 /* XXXSMP */
411 tsleep(xbd_xenbus_suspend, PRIBIO, "xbdsuspend", hz/2);
412 }
413
414 hypervisor_mask_event(sc->sc_evtchn);
415 sc->sc_backend_status = BLKIF_STATE_SUSPENDED;
416 xen_intr_disestablish(sc->sc_ih);
417
418 splx(s);
419
420 xenbus_device_suspend(sc->sc_xbusd);
421 aprint_verbose_dev(dev, "removed event channel %d\n", sc->sc_evtchn);
422
423 return true;
424}

426static bool
427xbd_xenbus_resume(device_t dev, const pmf_qual_t *qual)
428{
429 struct xbd_xenbus_softc *sc;
430 struct xenbus_transaction *xbt;
431 int error;
432 blkif_sring_t *ring;
433 paddr_t ma;
434 const char *errmsg;
435
436 sc = device_private(dev);
437
438 if (sc->sc_backend_status == BLKIF_STATE_SUSPENDED) {
439 /*
440 * Device was suspended, so ensure that access associated to
441 * the block I/O ring is revoked.
442 */
443 xengnt_revoke_access(sc->sc_ring_gntref);
444 }
445 sc->sc_ring_gntref = GRANT_INVALID_REF;
446
447 /* Initialize ring */
448 ring = sc->sc_ring.sring;
449 memset(ring, 0, PAGE_SIZE);
450 SHARED_RING_INIT(ring);
451 FRONT_RING_INIT(&sc->sc_ring, ring, PAGE_SIZE);
452
453 /*
454 * get MA address of the ring, and use it to set up the grant entry
455 * for the block device
456 */
457 (void)pmap_extract_ma(pmap_kernel(), (vaddr_t)ring, &ma);
458 error = xenbus_grant_ring(sc->sc_xbusd, ma, &sc->sc_ring_gntref);
459 if (error)
460 goto abort_resume;
461
462 error = xenbus_alloc_evtchn(sc->sc_xbusd, &sc->sc_evtchn);
463 if (error)
464 goto abort_resume;
465
466 aprint_verbose_dev(dev, "using event channel %d\n",
467 sc->sc_evtchn);
468 sc->sc_ih = xen_intr_establish_xname(-1, &xen_pic, sc->sc_evtchn, IST_LEVEL,
469 IPL_BIO, &xbd_handler, sc, false, device_xname(dev));
470 KASSERT(sc->sc_ih != NULL);
471
472again:
473 xbt = xenbus_transaction_start();
474 if (xbt == NULL)
475 return false;
476 error = xenbus_printf(xbt, sc->sc_xbusd->xbusd_path,
477 "ring-ref","%u", sc->sc_ring_gntref);
478 if (error) {
479 errmsg = "writing ring-ref";
480 goto abort_transaction;
481 }
482 error = xenbus_printf(xbt, sc->sc_xbusd->xbusd_path,
483 "event-channel", "%u", sc->sc_evtchn);
484 if (error) {
485 errmsg = "writing event channel";
486 goto abort_transaction;
487 }
488 error = xenbus_printf(xbt, sc->sc_xbusd->xbusd_path,
489 "protocol", "%s", XEN_IO_PROTO_ABI_NATIVE);
490 if (error) {
491 errmsg = "writing protocol";
492 goto abort_transaction;
493 }
494 error = xenbus_transaction_end(xbt, 0);
495 if (error == EAGAIN)
496 goto again;
497 if (error != 0) {
498 xenbus_dev_fatal(sc->sc_xbusd, error,
499 "completing transaction");
500 return false;
501 }
502
503 xenbus_switch_state(sc->sc_xbusd, NULL, XenbusStateInitialised);
504
505 if (sc->sc_backend_status == BLKIF_STATE_SUSPENDED) {
506 /*
507 * device was suspended, softc structures are
508 * already initialized - we use a shortcut
509 */
510 sc->sc_backend_status = BLKIF_STATE_CONNECTED;
511 xenbus_device_resume(sc->sc_xbusd);
512 hypervisor_unmask_event(sc->sc_evtchn);
513 xenbus_switch_state(sc->sc_xbusd, NULL, XenbusStateConnected);
514 }
515
516 return true;
517
518abort_resume:
519 xenbus_dev_fatal(sc->sc_xbusd, error, "resuming device");
520 return false;
521
522abort_transaction:
523 xenbus_transaction_end(xbt, 1);
524 xenbus_dev_fatal(sc->sc_xbusd, error, "%s", errmsg);
525 return false;
526}

528static void
529xbd_backend_changed(void *arg, XenbusState new_state)
530{
531 struct xbd_xenbus_softc *sc = device_private((device_t)arg);
532 struct disk_geom *dg;
533
534 char buf[9];
535 int s;
536 DPRINTF(("%s: new backend state %d\n",
537 device_xname(sc->sc_dksc.sc_dev), new_state));
538
539 switch (new_state) {
540 case XenbusStateUnknown:
541 case XenbusStateInitialising:
542 case XenbusStateInitWait:
543 case XenbusStateInitialised:
544 break;
545 case XenbusStateClosing:
546 s = splbio(); /* XXXSMP */
547 if (sc->sc_shutdown == BLKIF_SHUTDOWN_RUN)
548 sc->sc_shutdown = BLKIF_SHUTDOWN_REMOTE;
549 /* wait for requests to complete */
550 while (sc->sc_backend_status == BLKIF_STATE_CONNECTED &&
551 disk_isbusy(&sc->sc_dksc.sc_dkdev)) {
552 /* XXXSMP */
553 tsleep(xbd_xenbus_detach, PRIBIO, "xbddetach", hz/2);
554 }
555 splx(s);
556 xenbus_switch_state(sc->sc_xbusd, NULL, XenbusStateClosed);
557 break;
558 case XenbusStateConnected:
559 /*
560 * note that xbd_backend_changed() can only be called by
561 * the xenbus thread.
562 */
563
564 if (sc->sc_backend_status == BLKIF_STATE_CONNECTED ||
565 sc->sc_backend_status == BLKIF_STATE_SUSPENDED)
566 /* already connected */
567 return;
568
569 xbd_connect(sc);
570 sc->sc_shutdown = BLKIF_SHUTDOWN_RUN;
571 hypervisor_unmask_event(sc->sc_evtchn);
572 sc->sc_xbdsize =
573 sc->sc_sectors * (uint64_t)sc->sc_secsize / DEV_BSIZE;
574 dg = &sc->sc_dksc.sc_dkdev.dk_geom;
575 memset(dg, 0, sizeof(*dg));
576
577 dg->dg_secperunit = sc->sc_xbdsize;
578 dg->dg_secsize = DEV_BSIZE;
579 dg->dg_ntracks = 1;
580 // XXX: Ok to hard-code DEV_BSIZE?
581 dg->dg_nsectors = 1024 * (1024 / dg->dg_secsize);
582 dg->dg_ncylinders = dg->dg_secperunit / dg->dg_nsectors;
583
584 bufq_alloc(&sc->sc_dksc.sc_bufq, "fcfs", 0);
585 dk_attach(&sc->sc_dksc);
586 disk_attach(&sc->sc_dksc.sc_dkdev);
587
588 sc->sc_backend_status = BLKIF_STATE_CONNECTED;
589
590 /* try to read the disklabel */
591 dk_getdisklabel(&sc->sc_dksc, 0 /* XXX ? */);
592 format_bytes(buf, sizeof(buf), sc->sc_sectors * sc->sc_secsize);
593 aprint_verbose_dev(sc->sc_dksc.sc_dev,
594 "%s, %d bytes/sect x %" PRIu64 " sectors\n",
595 buf, (int)dg->dg_secsize, sc->sc_xbdsize);
596 /* Discover wedges on this disk. */
597 dkwedge_discover(&sc->sc_dksc.sc_dkdev);
598
599 disk_set_info(sc->sc_dksc.sc_dev, &sc->sc_dksc.sc_dkdev, NULL);
600
601 /* the disk should be working now */
602 config_pending_decr(sc->sc_dksc.sc_dev);
603 break;
604 default:
605 panic("bad backend state %d", new_state);
606 }
607}

609static void
610xbd_connect(struct xbd_xenbus_softc *sc)
611{
612 int err;
613 unsigned long long sectors;
614 u_long cache_flush;
615
616 err = xenbus_read_ul(NULL,
617 sc->sc_xbusd->xbusd_path, "virtual-device", &sc->sc_handle, 10);
618 if (err)
619 panic("%s: can't read number from %s/virtual-device\n",
620 device_xname(sc->sc_dksc.sc_dev),
621 sc->sc_xbusd->xbusd_otherend);
622 err = xenbus_read_ull(NULL,
623 sc->sc_xbusd->xbusd_otherend, "sectors", &sectors, 10);
624 if (err)
625 panic("%s: can't read number from %s/sectors\n",
626 device_xname(sc->sc_dksc.sc_dev),
627 sc->sc_xbusd->xbusd_otherend);
628 sc->sc_sectors = sectors;
629
630 err = xenbus_read_ul(NULL,
631 sc->sc_xbusd->xbusd_otherend, "info", &sc->sc_info, 10);
632 if (err)
633 panic("%s: can't read number from %s/info\n",
634 device_xname(sc->sc_dksc.sc_dev),
635 sc->sc_xbusd->xbusd_otherend);
636 err = xenbus_read_ul(NULL,
637 sc->sc_xbusd->xbusd_otherend, "sector-size", &sc->sc_secsize, 10);
638 if (err)
639 panic("%s: can't read number from %s/sector-size\n",
640 device_xname(sc->sc_dksc.sc_dev),
641 sc->sc_xbusd->xbusd_otherend);
642
643 err = xenbus_read_ul(NULL, sc->sc_xbusd->xbusd_otherend,
644 "feature-flush-cache", &cache_flush, 10);
645 if (err)
646 cache_flush = 0;
647 if (cache_flush > 0)
648 sc->sc_cache_flush = 1;
649 else
650 sc->sc_cache_flush = 0;
651
652 xenbus_switch_state(sc->sc_xbusd, NULL, XenbusStateConnected);
653}

655static int
656xbd_handler(void *arg)
657{
658 struct xbd_xenbus_softc *sc = arg;
659 struct buf *bp;
660 RING_IDX resp_prod, i;
661 int more_to_do;
662 int seg;
663
664 DPRINTF(("xbd_handler(%s)\n", device_xname(sc->sc_dksc.sc_dev)));
665
666 if (__predict_false(sc->sc_backend_status != BLKIF_STATE_CONNECTED))
667 return 0;
668again:
669 resp_prod = sc->sc_ring.sring->rsp_prod;
670 xen_rmb(); /* ensure we see replies up to resp_prod */
671 for (i = sc->sc_ring.rsp_cons; i != resp_prod; i++) {
672 blkif_response_t *rep = RING_GET_RESPONSE(&sc->sc_ring, i);
673 struct xbd_req *xbdreq = &sc->sc_reqs[rep->id];
674
675 if (rep->operation == BLKIF_OP_FLUSH_DISKCACHE) {
676 KASSERT(xbdreq->req_bp == NULL);
677 xbdreq->req_sync.s_error = rep->status;
678 xbdreq->req_sync.s_done = 1;
679 wakeup(xbdreq); /* XXXSMP */
680 /* caller will free the req */
681 continue;
682 }
683
684 if (rep->operation != BLKIF_OP_READ &&
685 rep->operation != BLKIF_OP_WRITE) {
686 aprint_error_dev(sc->sc_dksc.sc_dev,
687 "bad operation %d from backend\n", rep->operation);
688 continue;
689 }
690
691 for (seg = xbdreq->req_nr_segments - 1; seg >= 0; seg--) {
692 if (__predict_false(
693 xengnt_status(xbdreq->req_gntref[seg]))) {
694 aprint_verbose_dev(sc->sc_dksc.sc_dev,
695 "grant still used by backend\n");
696 sc->sc_ring.rsp_cons = i;
697 xbdreq->req_nr_segments = seg + 1;
698 goto done;
699 }
700 xengnt_revoke_access(xbdreq->req_gntref[seg]);
701 xbdreq->req_nr_segments--;
702 }
703 KASSERT(xbdreq->req_nr_segments == 0);
704
705 bp = xbdreq->req_bp;
706 KASSERT(bp != NULL);
707 DPRINTF(("%s(%p): b_bcount = %ld\n", __func__,
708 bp, (long)bp->b_bcount));
709
710 if (rep->status != BLKIF_RSP_OKAY) {
711 bp->b_error = EIO;
712 bp->b_resid = bp->b_bcount;
713 goto next;
714 }
715 /* b_resid was set in dk_start */
716 if (__predict_false(
717 xbdreq->req_data != NULL && bp->b_data != xbdreq->req_data))
718 xbd_unmap_align(xbdreq);
719next:
720 xbdreq->req_bp = NULL;
721 dk_done(&sc->sc_dksc, bp);
722
723 SLIST_INSERT_HEAD(&sc->sc_xbdreq_head, xbdreq, req_next);
724 }
725done:
726 xen_rmb();
727 sc->sc_ring.rsp_cons = i;
728
729 RING_FINAL_CHECK_FOR_RESPONSES(&sc->sc_ring, more_to_do);
730 if (more_to_do)
731 goto again;
732
733 if (sc->sc_xbdreq_wait)
734 wakeup(&sc->sc_xbdreq_wait); /* XXXSMP */
735 else
736 dk_start(&sc->sc_dksc, NULL);
737 return 1;
738}

840int
841xbdioctl(dev_t dev, u_long cmd, void *data, int flag, struct lwp *l)
842{
843 struct xbd_xenbus_softc *sc =
844 device_lookup_private(&xbd_cd, DISKUNIT(dev));
845 struct dk_softc *dksc;
846 int error;
847 int s;
848 struct xbd_req *xbdreq;
849 blkif_request_t *req;
850 int notify;
851
852 DPRINTF(("xbdioctl(%d, %08lx, %p, %d, %p)\n",
853 dev, cmd, data, flag, l));
854 dksc = &sc->sc_dksc;
855
856 switch (cmd) {
857 case DIOCSSTRATEGY:
858 error = EOPNOTSUPP;
859 break;
860 case DIOCCACHESYNC:
861 if (sc->sc_cache_flush <= 0) {
862 if (sc->sc_cache_flush == 0) {
863 aprint_error_dev(sc->sc_dksc.sc_dev,
864 "WARNING: cache flush not supported "
865 "by backend\n");
866 sc->sc_cache_flush = -1;
867 }
868 return EOPNOTSUPP;
869 }
870
871 s = splbio(); /* XXXSMP */
872
873 while (RING_FULL(&sc->sc_ring)) {
874 sc->sc_xbdreq_wait = 1;
875 /* XXXSMP */
876 tsleep(&sc->sc_xbdreq_wait, PRIBIO, "xbdreq", 0);
877 }
878 sc->sc_xbdreq_wait = 0;
879
880 xbdreq = SLIST_FIRST(&sc->sc_xbdreq_head);
881 if (__predict_false(xbdreq == NULL)) {
882 DPRINTF(("xbdioctl: no req\n"));
883 error = ENOMEM;
884 } else {
885 SLIST_REMOVE_HEAD(&sc->sc_xbdreq_head, req_next);
886 req = RING_GET_REQUEST(&sc->sc_ring,
887 sc->sc_ring.req_prod_pvt);
888 req->id = xbdreq->req_id;
889 req->operation = BLKIF_OP_FLUSH_DISKCACHE;
890 req->handle = sc->sc_handle;
891 xbdreq->req_sync.s_done = 0;
892 sc->sc_ring.req_prod_pvt++;
893 RING_PUSH_REQUESTS_AND_CHECK_NOTIFY(&sc->sc_ring,
894 notify);
895 if (notify)
896 hypervisor_notify_via_evtchn(sc->sc_evtchn);
897 /* request sent, now wait for completion */
898 while (xbdreq->req_sync.s_done == 0) {
899 /* XXXSMP */
900 tsleep(xbdreq, PRIBIO, "xbdsync", 0);
901 }
902 if (xbdreq->req_sync.s_error == BLKIF_RSP_EOPNOTSUPP)
903 error = EOPNOTSUPP;
904 else if (xbdreq->req_sync.s_error == BLKIF_RSP_OKAY)
905 error = 0;
906 else
907 error = EIO;
908 SLIST_INSERT_HEAD(&sc->sc_xbdreq_head, xbdreq,
909 req_next);
910 }
911 splx(s);
912 break;
913
914 default:
915 error = dk_ioctl(dksc, dev, cmd, data, flag, l);
916 break;
917 }
918
919 return error;
920}

936static int
937xbd_diskstart(device_t self, struct buf *bp)
938{
939 struct xbd_xenbus_softc *sc = device_private(self);
940 struct xbd_req *xbdreq;
941 blkif_request_t *req;
942 size_t bcount, off;
943 paddr_t ma;
944 vaddr_t va;
945 int nsects, nbytes, seg;
946 int notify, error = 0;
947 int s;
948
949 DPRINTF(("xbd_diskstart(%p): b_bcount = %ld\n",
950 bp, (long)bp->b_bcount));
951
952 if (sc->sc_shutdown != BLKIF_SHUTDOWN_RUN) {
953 error = EIO;
954 goto err;
955 }
956
957 if (bp->b_rawblkno < 0 || bp->b_rawblkno > sc->sc_xbdsize) {
958 /* invalid block number */
959 error = EINVAL;
960 goto err;
961 }
962
963 s = splbio(); /* XXX SMP */
964
965 if (__predict_false(
966 sc->sc_backend_status == BLKIF_STATE_SUSPENDED)) {
967 /* device is suspended, do not consume buffer */
968 DPRINTF(("%s: (xbd_diskstart) device suspended\n",
969 sc->sc_dksc.sc_xname));
970 error = EAGAIN;
971 goto out;
972 }
973
974 if (RING_FULL(&sc->sc_ring) || sc->sc_xbdreq_wait) {
975 DPRINTF(("xbd_diskstart: ring_full\n"));
976 error = EAGAIN;
977 goto out;
978 }
979
980 xbdreq = SLIST_FIRST(&sc->sc_xbdreq_head);
981 if (__predict_false(xbdreq == NULL)) {
982 DPRINTF(("xbd_diskstart: no req\n"));
983 error = EAGAIN;
984 goto out;
985 }
986
987 xbdreq->req_bp = bp;
988 xbdreq->req_data = bp->b_data;
989 if (__predict_false((vaddr_t)bp->b_data & (XEN_BSIZE - 1))) {
990 sc->sc_cnt_map_unalign.ev_count++;
991
992 if (__predict_false(xbd_map_align(xbdreq) != 0)) {
993 DPRINTF(("xbd_diskstart: no align\n"));
994 error = EAGAIN;
995 goto out;
996 }
997 }
998
999 SLIST_REMOVE_HEAD(&sc->sc_xbdreq_head, req_next);
1000 req = RING_GET_REQUEST(&sc->sc_ring, sc->sc_ring.req_prod_pvt);
1001 req->id = xbdreq->req_id;
1002 req->operation =
1003 bp->b_flags & B_READ ? BLKIF_OP_READ : BLKIF_OP_WRITE;
1004 req->sector_number = bp->b_rawblkno;
1005 req->handle = sc->sc_handle;
1006
1007 va = (vaddr_t)xbdreq->req_data & ~PAGE_MASK;
1008 off = (vaddr_t)xbdreq->req_data & PAGE_MASK;
1009 bcount = bp->b_bcount;
1010 bp->b_resid = 0;
1011 for (seg = 0; bcount > 0;) {
1012 pmap_extract_ma(pmap_kernel(), va, &ma);
1013 KASSERT((ma & (XEN_BSIZE - 1)) == 0);
1014 if (bcount > PAGE_SIZE - off)
1015 nbytes = PAGE_SIZE - off;
1016 else
1017 nbytes = bcount;
1018 nsects = nbytes >> XEN_BSHIFT;
1019 req->seg[seg].first_sect = off >> XEN_BSHIFT;
1020 req->seg[seg].last_sect =
1021 (off >> XEN_BSHIFT) + nsects - 1;
1022 KASSERT(req->seg[seg].first_sect <=
1023 req->seg[seg].last_sect);
1024 KASSERT(req->seg[seg].last_sect < 8);
1025 if (__predict_false(xengnt_grant_access(
1026 sc->sc_xbusd->xbusd_otherend_id, ma,
1027 (bp->b_flags & B_READ) == 0,
1028 &xbdreq->req_gntref[seg])))
1029 panic("xbd_diskstart: xengnt_grant_access"); /* XXX XXX !!! */
1030 req->seg[seg].gref = xbdreq->req_gntref[seg];
1031 seg++;
1032 KASSERT(seg <= BLKIF_MAX_SEGMENTS_PER_REQUEST);
1033 va += PAGE_SIZE;
1034 off = 0;
1035 bcount -= nbytes;
1036 }
1037 xbdreq->req_nr_segments = req->nr_segments = seg;
1038 sc->sc_ring.req_prod_pvt++;
1039
1040out:
1041 RING_PUSH_REQUESTS_AND_CHECK_NOTIFY(&sc->sc_ring, notify);
1042 if (notify)
1043 hypervisor_notify_via_evtchn(sc->sc_evtchn);
1044 splx(s); /* XXXSMP */
1045err:
1046 return error;
1047}

1049static int
1050xbd_map_align(struct xbd_req *req)
1051{
1052 int s = splvm(); /* XXXSMP - bogus? */
1053 int rc;
1054
1055 rc = uvm_km_kmem_alloc(kmem_va_arena,
1056 req->req_bp->b_bcount, (VM_NOSLEEP | VM_INSTANTFIT),
1057 (vmem_addr_t *)&req->req_data);
1058 splx(s);
1059 if (__predict_false(rc != 0))
1060 return ENOMEM;
1061 if ((req->req_bp->b_flags & B_READ) == 0)
1062 memcpy(req->req_data, req->req_bp->b_data,
1063 req->req_bp->b_bcount);
1064 return 0;
1065}
