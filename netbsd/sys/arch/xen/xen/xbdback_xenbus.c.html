83/*
84 * status of a xbdback instance:
85 * WAITING: xbdback instance is connected, waiting for requests
86 * RUN: xbdi thread must be woken up, I/Os have to be processed
87 * DISCONNECTING: the instance is closing, no more I/Os can be scheduled
88 * DISCONNECTED: no I/Os, no ring, the thread should terminate.
89 */
90typedef enum {WAITING, RUN, DISCONNECTING, DISCONNECTED} xbdback_state_t;
91
92/*
93 * Each xbdback instance is managed by a single thread that handles all
94 * the I/O processing. As there are a variety of conditions that can block,
95 * everything will be done in a sort of continuation-passing style.
96 *
97 * When the execution has to block to delay processing, for example to
98 * allow system to recover because of memory shortage (via shared memory
99 * callback), the return value of a continuation can be set to NULL. In that
100 * case, the thread will go back to sleeping and wait for the proper
101 * condition before it starts processing requests again from where it left.
102 * Continuation state is "stored" in the xbdback instance (xbdi_cont and
103 * xbdi_cont_aux), and should only be manipulated by the instance thread.
104 *
105 * As xbdback(4) has to handle different sort of asynchronous events (Xen
106 * event channels, biointr() soft interrupts, xenbus commands), the xbdi_lock
107 * mutex is used to protect specific elements of the xbdback instance from
108 * concurrent access: thread status and ring access (when pushing responses).
109 *
110 * Here's how the call graph is supposed to be for a single I/O:
111 *
112 * xbdback_co_main()
113 * |
114 * | --> xbdback_co_cache_doflush() or NULL
115 * | |
116 * | - xbdback_co_cache_flush2() <- xbdback_co_do_io() <-
117 * | | |
118 * | |-> xbdback_co_cache_flush() -> xbdback_co_map_io()-
119 * xbdback_co_main_loop()-|
120 * | |-> xbdback_co_main_done() ---> xbdback_co_map_io()-
121 * | | |
122 * | -- xbdback_co_main_done2() <-- xbdback_co_do_io() <-
123 * | |
124 * | --> xbdback_co_main() or NULL
125 * |
126 * xbdback_co_io() -> xbdback_co_main_incr() -> xbdback_co_main_loop()
127 * |
128 * xbdback_co_io_gotreq()--+--> xbdback_co_map_io() ---
129 * | | |
130 * -> xbdback_co_io_loop()----| <- xbdback_co_do_io() <--
131 * | | | |
132 * | | | |----------> xbdback_co_io_gotio()
133 * | | | |
134 * | | xbdback_co_main_incr() |
135 * | | | |
136 * | | xbdback_co_main_loop() |
137 * | | |
138 * | xbdback_co_io_gotio2() <-----------|
139 * | | |
140 * | | |----------> xbdback_co_io_gotfrag()
141 * | | |
142 * -- xbdback_co_io_gotfrag2() <---------|
143 * |
144 * xbdback_co_main_incr() -> xbdback_co_main_loop()
145 */
146typedef void *(* xbdback_cont_t)(struct xbdback_instance *, void *);

219/*
220 * For each request from a guest, a xbdback_request is allocated from
221 * a pool. This will describe the request until completion. The
222 * request may require multiple IO operations to perform, so the
223 * per-IO information is not stored here.
224 */
225struct xbdback_request {
226 struct xbdback_instance *rq_xbdi; /* our xbd instance */
227 uint64_t rq_id;
228 int rq_iocount; /* reference count; or, number of outstanding I/O's */
229 int rq_ioerrs;
230 uint8_t rq_operation;
231};
232
233/*
234 * For each I/O operation associated with one of those requests, an
235 * xbdback_io is allocated from a pool. It may correspond to multiple
236 * Xen disk requests, or parts of them, if several arrive at once that
237 * can be coalesced.
238 */
239struct xbdback_io {
240 /* The instance pointer is duplicated for convenience. */
241 struct xbdback_instance *xio_xbdi; /* our xbd instance */
242 uint8_t xio_operation;
243 union {
244 struct {
245 struct buf xio_buf; /* our I/O */
246 /* xbd requests involved */
247 SLIST_HEAD(, xbdback_fragment) xio_rq;
248 /* the virtual address to map the request at */
249 vaddr_t xio_vaddr;
250 /* grants to map */
251 grant_ref_t xio_gref[XENSHM_MAX_PAGES_PER_REQUEST];
252 /* grants release */
253 grant_handle_t xio_gh[XENSHM_MAX_PAGES_PER_REQUEST];
254 uint16_t xio_nrma; /* number of guest pages */
255 uint16_t xio_mapped; /* == 1: grants are mapped */
256 } xio_rw;
257 uint64_t xio_flush_id;
258 } u;
259};
260#define xio_buf u.xio_rw.xio_buf
261#define xio_rq u.xio_rw.xio_rq
262#define xio_vaddr u.xio_rw.xio_vaddr
263#define xio_gref u.xio_rw.xio_gref
264#define xio_gh u.xio_rw.xio_gh
265#define xio_nrma u.xio_rw.xio_nrma
266#define xio_mapped u.xio_rw.xio_mapped
267
268#define xio_flush_id u.xio_flush_id

359void
360xbdbackattach(int n)
361{
362 XENPRINTF(("xbdbackattach\n"));
363
364 /*
365 * initialize the backend driver, register the control message handler
366 * and send driver up message.
367 */
368 SLIST_INIT(&xbdback_instances);
369 mutex_init(&xbdback_lock, MUTEX_DEFAULT, IPL_NONE);
370 SIMPLEQ_INIT(&xbdback_shmq);
371 xbdback_shmcb = 0;
372
373 pool_cache_bootstrap(&xbdback_request_pool.pc,
374 sizeof(struct xbdback_request), 0, 0, 0, "xbbrp", NULL,
375 IPL_SOFTBIO, NULL, NULL, NULL);
376 pool_cache_bootstrap(&xbdback_io_pool.pc,
377 sizeof(struct xbdback_io), 0, 0, 0, "xbbip", NULL,
378 IPL_SOFTBIO, NULL, NULL, NULL);
379 pool_cache_bootstrap(&xbdback_fragment_pool.pc,
380 sizeof(struct xbdback_fragment), 0, 0, 0, "xbbfp", NULL,
381 IPL_SOFTBIO, NULL, NULL, NULL);
382
383 /* we allocate enough to handle a whole ring at once */
384 if (pool_prime(&xbdback_request_pool.pc.pc_pool, BLKIF_RING_SIZE) != 0)
385 printf("xbdback: failed to prime request pool\n");
386 if (pool_prime(&xbdback_io_pool.pc.pc_pool, BLKIF_RING_SIZE) != 0)
387 printf("xbdback: failed to prime io pool\n");
388 if (pool_prime(&xbdback_fragment_pool.pc.pc_pool,
389 BLKIF_MAX_SEGMENTS_PER_REQUEST * BLKIF_RING_SIZE) != 0)
390 printf("xbdback: failed to prime fragment pool\n");
391
392 xenbus_backend_register(&xbd_backend_driver);
393}

395static int
396xbdback_xenbus_create(struct xenbus_device *xbusd)
397{
398 struct xbdback_instance *xbdi;
399 long domid, handle;
400 int error, i;
401 char *ep;
402
403 if ((error = xenbus_read_ul(NULL, xbusd->xbusd_path,
404 "frontend-id", &domid, 10)) != 0) {
405 aprint_error("xbdback: can't read %s/frontend-id: %d\n",
406 xbusd->xbusd_path, error);
407 return error;
408 }
409
410 /*
411 * get handle: this is the last component of the path; which is
412 * a decimal number. $path/dev contains the device name, which is not
413 * appropriate.
414 */
415 for (i = strlen(xbusd->xbusd_path); i > 0; i--) {
416 if (xbusd->xbusd_path[i] == '/')
417 break;
418 }
419 if (i == 0) {
420 aprint_error("xbdback: can't parse %s\n",
421 xbusd->xbusd_path);
422 return EFTYPE;
423 }
424 handle = strtoul(&xbusd->xbusd_path[i+1], &ep, 10);
425 if (*ep != '\0') {
426 aprint_error("xbdback: can't parse %s\n",
427 xbusd->xbusd_path);
428 return EFTYPE;
429 }
430
431 if (xbdif_lookup(domid, handle)) {
432 return EEXIST;
433 }
434 xbdi = kmem_zalloc(sizeof(*xbdi), KM_SLEEP);
435
436 xbdi->xbdi_domid = domid;
437 xbdi->xbdi_handle = handle;
438 snprintf(xbdi->xbdi_name, sizeof(xbdi->xbdi_name), "xbdb%di%d",
439 xbdi->xbdi_domid, xbdi->xbdi_handle);
440
441 /* initialize status and reference counter */
442 xbdi->xbdi_status = DISCONNECTED;
443 xbdi_get(xbdi);
444
445 mutex_init(&xbdi->xbdi_lock, MUTEX_DEFAULT, IPL_BIO);
446 cv_init(&xbdi->xbdi_cv, xbdi->xbdi_name);
447 mutex_enter(&xbdback_lock);
448 SLIST_INSERT_HEAD(&xbdback_instances, xbdi, next);
449 mutex_exit(&xbdback_lock);
450
451 xbusd->xbusd_u.b.b_cookie = xbdi;
452 xbusd->xbusd_u.b.b_detach = xbdback_xenbus_destroy;
453 xbusd->xbusd_otherend_changed = xbdback_frontend_changed;
454 xbdi->xbdi_xbusd = xbusd;
455
456 error = xenbus_watch_path2(xbusd, xbusd->xbusd_path, "physical-device",
457 &xbdi->xbdi_watch, xbdback_backend_changed);
458 if (error) {
459 printf("failed to watch on %s/physical-device: %d\n",
460 xbusd->xbusd_path, error);
461 goto fail;
462 }
463 xbdi->xbdi_watch.xbw_dev = xbusd;
464 error = xenbus_switch_state(xbusd, NULL, XenbusStateInitWait);
465 if (error) {
466 printf("failed to switch state on %s: %d\n",
467 xbusd->xbusd_path, error);
468 goto fail2;
469 }
470 return 0;
471fail2:
472 unregister_xenbus_watch(&xbdi->xbdi_watch);
473fail:
474 kmem_free(xbdi, sizeof(*xbdi));
475 return error;
476}

478static int
479xbdback_xenbus_destroy(void *arg)
480{
481 struct xbdback_instance *xbdi = arg;
482 struct xenbus_device *xbusd = xbdi->xbdi_xbusd;
483 struct gnttab_unmap_grant_ref ungrop;
484 int err;
485
486 XENPRINTF(("xbdback_xenbus_destroy state %d\n", xbdi->xbdi_status));
487
488 xbdback_disconnect(xbdi);
489
490 /* unregister watch */
491 if (xbdi->xbdi_watch.node) {
492 unregister_xenbus_watch(&xbdi->xbdi_watch);
493 free(xbdi->xbdi_watch.node, M_DEVBUF);
494 xbdi->xbdi_watch.node = NULL;
495 }
496 /* unmap ring */
497 if (xbdi->xbdi_ring_va != 0) {
498 ungrop.host_addr = xbdi->xbdi_ring_va;
499 ungrop.handle = xbdi->xbdi_ring_handle;
500 ungrop.dev_bus_addr = 0;
501 err = HYPERVISOR_grant_table_op(GNTTABOP_unmap_grant_ref,
502 &ungrop, 1);
503 if (err)
504 printf("xbdback %s: unmap_grant_ref failed: %d\n",
505 xbusd->xbusd_otherend, err);
506 uvm_km_free(kernel_map, xbdi->xbdi_ring_va,
507 PAGE_SIZE, UVM_KMF_VAONLY);
508 }
509 /* close device */
510 if (xbdi->xbdi_size) {
511 const char *name;
512 struct dkwedge_info wi;
513 if (getdiskinfo(xbdi->xbdi_vp, &wi) == 0)
514 name = wi.dkw_devname;
515 else
516 name = "*unknown*";
517 printf("xbd backend: detach device %s for domain %d\n",
518 name, xbdi->xbdi_domid);
519 vn_close(xbdi->xbdi_vp, FREAD, NOCRED);
520 }
521 mutex_enter(&xbdback_lock);
522 SLIST_REMOVE(&xbdback_instances, xbdi, xbdback_instance, next);
523 mutex_exit(&xbdback_lock);
524 mutex_destroy(&xbdi->xbdi_lock);
525 cv_destroy(&xbdi->xbdi_cv);
526 kmem_free(xbdi, sizeof(*xbdi));
527 return 0;
528}

530static int
531xbdback_connect(struct xbdback_instance *xbdi)
532{
533 int len, err;
534 struct gnttab_map_grant_ref grop;
535 struct gnttab_unmap_grant_ref ungrop;
536 evtchn_op_t evop;
537 u_long ring_ref, revtchn;
538 char *xsproto;
539 const char *proto;
540 struct xenbus_device *xbusd = xbdi->xbdi_xbusd;
541
542 XENPRINTF(("xbdback %s: connect\n", xbusd->xbusd_path));
543 /* read comunication informations */
544 err = xenbus_read_ul(NULL, xbusd->xbusd_otherend,
545 "ring-ref", &ring_ref, 10);
546 if (err) {
547 xenbus_dev_fatal(xbusd, err, "reading %s/ring-ref",
548 xbusd->xbusd_otherend);
549 return -1;
550 }
551 XENPRINTF(("xbdback %s: connect ring-ref %lu\n", xbusd->xbusd_path, ring_ref));
552 err = xenbus_read_ul(NULL, xbusd->xbusd_otherend,
553 "event-channel", &revtchn, 10);
554 if (err) {
555 xenbus_dev_fatal(xbusd, err, "reading %s/event-channel",
556 xbusd->xbusd_otherend);
557 return -1;
558 }
559 XENPRINTF(("xbdback %s: connect revtchn %lu\n", xbusd->xbusd_path, revtchn));
560 err = xenbus_read(NULL, xbusd->xbusd_otherend, "protocol",
561 &len, &xsproto);
562 if (err) {
563 xbdi->xbdi_proto = XBDIP_NATIVE;
564 proto = "unspecified";
565 XENPRINTF(("xbdback %s: connect no xsproto\n", xbusd->xbusd_path));
566 } else {
567 XENPRINTF(("xbdback %s: connect xsproto %s\n", xbusd->xbusd_path, xsproto));
568 if (strcmp(xsproto, XEN_IO_PROTO_ABI_NATIVE) == 0) {
569 xbdi->xbdi_proto = XBDIP_NATIVE;
570 proto = XEN_IO_PROTO_ABI_NATIVE;
571 } else if (strcmp(xsproto, XEN_IO_PROTO_ABI_X86_32) == 0) {
572 xbdi->xbdi_proto = XBDIP_32;
573 proto = XEN_IO_PROTO_ABI_X86_32;
574 } else if (strcmp(xsproto, XEN_IO_PROTO_ABI_X86_64) == 0) {
575 xbdi->xbdi_proto = XBDIP_64;
576 proto = XEN_IO_PROTO_ABI_X86_64;
577 } else {
578 aprint_error("xbd domain %d: unknown proto %s\n",
579 xbdi->xbdi_domid, xsproto);
580 free(xsproto, M_DEVBUF);
581 return -1;
582 }
583 free(xsproto, M_DEVBUF);
584 }
585
586 /* allocate VA space and map rings */
587 xbdi->xbdi_ring_va = uvm_km_alloc(kernel_map, PAGE_SIZE, 0,
588 UVM_KMF_VAONLY);
589 if (xbdi->xbdi_ring_va == 0) {
590 xenbus_dev_fatal(xbusd, ENOMEM,
591 "can't get VA for ring", xbusd->xbusd_otherend);
592 return -1;
593 }
594 XENPRINTF(("xbdback %s: connect va 0x%" PRIxVADDR "\n", xbusd->xbusd_path, xbdi->xbdi_ring_va));
595
596 grop.host_addr = xbdi->xbdi_ring_va;
597 grop.flags = GNTMAP_host_map;
598 grop.ref = ring_ref;
599 grop.dom = xbdi->xbdi_domid;
600 err = HYPERVISOR_grant_table_op(GNTTABOP_map_grant_ref,
601 &grop, 1);
602 if (err || grop.status) {
603 aprint_error("xbdback %s: can't map grant ref: %d/%d\n",
604 xbusd->xbusd_path, err, grop.status);
605 xenbus_dev_fatal(xbusd, EINVAL,
606 "can't map ring", xbusd->xbusd_otherend);
607 goto err;
608 }
609 xbdi->xbdi_ring_handle = grop.handle;
610 XENPRINTF(("xbdback %s: connect grhandle %d\n", xbusd->xbusd_path, grop.handle));
611
612 switch(xbdi->xbdi_proto) {
613 case XBDIP_NATIVE:
614 {
615 blkif_sring_t *sring = (void *)xbdi->xbdi_ring_va;
616 BACK_RING_INIT(&xbdi->xbdi_ring.ring_n, sring, PAGE_SIZE);
617 break;
618 }
619 case XBDIP_32:
620 {
621 blkif_x86_32_sring_t *sring = (void *)xbdi->xbdi_ring_va;
622 BACK_RING_INIT(&xbdi->xbdi_ring.ring_32, sring, PAGE_SIZE);
623 break;
624 }
625 case XBDIP_64:
626 {
627 blkif_x86_64_sring_t *sring = (void *)xbdi->xbdi_ring_va;
628 BACK_RING_INIT(&xbdi->xbdi_ring.ring_64, sring, PAGE_SIZE);
629 break;
630 }
631 }
632
633 evop.cmd = EVTCHNOP_bind_interdomain;
634 evop.u.bind_interdomain.remote_dom = xbdi->xbdi_domid;
635 evop.u.bind_interdomain.remote_port = revtchn;
636 err = HYPERVISOR_event_channel_op(&evop);
637 if (err) {
638 aprint_error("blkback %s: "
639 "can't get event channel: %d\n",
640 xbusd->xbusd_otherend, err);
641 xenbus_dev_fatal(xbusd, err,
642 "can't bind event channel", xbusd->xbusd_otherend);
643 goto err2;
644 }
645 XENPRINTF(("xbdback %s: connect evchannel %d\n", xbusd->xbusd_path, xbdi->xbdi_evtchn));
646 xbdi->xbdi_evtchn = evop.u.bind_interdomain.local_port;
647
648 xbdi->xbdi_ih = xen_intr_establish_xname(-1, &xen_pic, xbdi->xbdi_evtchn,
649 IST_LEVEL, IPL_BIO, xbdback_evthandler, xbdi, false,
650 xbdi->xbdi_name);
651 KASSERT(xbdi->xbdi_ih != NULL);
652 aprint_verbose("xbd backend domain %d handle %#x (%d) "
653 "using event channel %d, protocol %s\n", xbdi->xbdi_domid,
654 xbdi->xbdi_handle, xbdi->xbdi_handle, xbdi->xbdi_evtchn, proto);
655
656 /* enable the xbdback event handler machinery */
657 xbdi->xbdi_status = WAITING;
658 hypervisor_unmask_event(xbdi->xbdi_evtchn);
659 hypervisor_notify_via_evtchn(xbdi->xbdi_evtchn);
660
661 if (kthread_create(PRI_NONE, KTHREAD_MPSAFE, NULL,
662 xbdback_thread, xbdi, NULL, "%s", xbdi->xbdi_name) == 0)
663 return 0;
664
665err2:
666 /* unmap ring */
667 ungrop.host_addr = xbdi->xbdi_ring_va;
668 ungrop.handle = xbdi->xbdi_ring_handle;
669 ungrop.dev_bus_addr = 0;
670 err = HYPERVISOR_grant_table_op(GNTTABOP_unmap_grant_ref,
671 &ungrop, 1);
672 if (err)
673 aprint_error("xbdback %s: unmap_grant_ref failed: %d\n",
674 xbusd->xbusd_path, err);
675
676err:
677 /* free ring VA space */
678 uvm_km_free(kernel_map, xbdi->xbdi_ring_va, PAGE_SIZE, UVM_KMF_VAONLY);
679 return -1;
680}

682/*
683 * Signal a xbdback thread to disconnect. Done in 'xenwatch' thread context.
684 */
685static void
686xbdback_disconnect(struct xbdback_instance *xbdi)
687{
688
689 mutex_enter(&xbdi->xbdi_lock);
690 if (xbdi->xbdi_status == DISCONNECTED) {
691 mutex_exit(&xbdi->xbdi_lock);
692 return;
693 }
694 hypervisor_mask_event(xbdi->xbdi_evtchn);
695 xen_intr_disestablish(xbdi->xbdi_ih);
696
697 /* signal thread that we want to disconnect, then wait for it */
698 xbdi->xbdi_status = DISCONNECTING;
699 cv_signal(&xbdi->xbdi_cv);
700
701 while (xbdi->xbdi_status != DISCONNECTED)
702 cv_wait(&xbdi->xbdi_cv, &xbdi->xbdi_lock);
703
704 mutex_exit(&xbdi->xbdi_lock);
705
706 xenbus_switch_state(xbdi->xbdi_xbusd, NULL, XenbusStateClosing);
707}

711{
712 struct xbdback_instance *xbdi = arg;

740static void
741xbdback_backend_changed(struct xenbus_watch *watch,
742 const char **vec, unsigned int len)
743{
744 struct xenbus_device *xbusd = watch->xbw_dev;
745 struct xbdback_instance *xbdi = xbusd->xbusd_u.b.b_cookie;
746 int err;
747 long dev;
748 char *mode;
749 struct xenbus_transaction *xbt;
750 const char *devname;
751 int major;
752
753 err = xenbus_read_ul(NULL, xbusd->xbusd_path, "physical-device",
754 &dev, 10);
755 /*
756 * An error can occur as the watch can fire up just after being
757 * registered. So we have to ignore error :(
758 */
759 if (err)
760 return;
761 /*
762 * we can also fire up after having opened the device, don't try
763 * to do it twice.
764 */
765 if (xbdi->xbdi_vp != NULL) {
766 if (xbdi->xbdi_status == WAITING || xbdi->xbdi_status == RUN) {
767 if (xbdi->xbdi_dev != dev) {
768 printf("xbdback %s: changing physical device "
769 "from %#"PRIx64" to %#lx not supported\n",
770 xbusd->xbusd_path, xbdi->xbdi_dev, dev);
771 }
772 }
773 return;
774 }
775 xbdi->xbdi_dev = dev;
776 err = xenbus_read(NULL, xbusd->xbusd_path, "mode", NULL, &mode);
777 if (err) {
778 printf("xbdback: failed to read %s/mode: %d\n",
779 xbusd->xbusd_path, err);
780 return;
781 }
782 if (mode[0] == 'w')
783 xbdi->xbdi_ro = false;
784 else
785 xbdi->xbdi_ro = true;
786 free(mode, M_DEVBUF);
787 major = major(xbdi->xbdi_dev);
788 devname = devsw_blk2name(major);
789 if (devname == NULL) {
790 printf("xbdback %s: unknown device 0x%"PRIx64"\n",
791 xbusd->xbusd_path, xbdi->xbdi_dev);
792 return;
793 }
794 xbdi->xbdi_bdevsw = bdevsw_lookup(xbdi->xbdi_dev);
795 if (xbdi->xbdi_bdevsw == NULL) {
796 printf("xbdback %s: no bdevsw for device 0x%"PRIx64"\n",
797 xbusd->xbusd_path, xbdi->xbdi_dev);
798 return;
799 }
800 err = bdevvp(xbdi->xbdi_dev, &xbdi->xbdi_vp);
801 if (err) {
802 printf("xbdback %s: can't open device 0x%"PRIx64": %d\n",
803 xbusd->xbusd_path, xbdi->xbdi_dev, err);
804 return;
805 }
806 err = vn_lock(xbdi->xbdi_vp, LK_EXCLUSIVE | LK_RETRY);
807 if (err) {
808 printf("xbdback %s: can't vn_lock device 0x%"PRIx64": %d\n",
809 xbusd->xbusd_path, xbdi->xbdi_dev, err);
810 vrele(xbdi->xbdi_vp);
811 return;
812 }
813 err = VOP_OPEN(xbdi->xbdi_vp, FREAD, NOCRED);
814 if (err) {
815 printf("xbdback %s: can't VOP_OPEN device 0x%"PRIx64": %d\n",
816 xbusd->xbusd_path, xbdi->xbdi_dev, err);
817 vput(xbdi->xbdi_vp);
818 return;
819 }
820 VOP_UNLOCK(xbdi->xbdi_vp);
821
822 /* dk device; get wedge data */
823 struct dkwedge_info wi;
824 if ((err = getdiskinfo(xbdi->xbdi_vp, &wi)) == 0) {
825 xbdi->xbdi_size = wi.dkw_size;
826 printf("xbd backend: attach device %s (size %" PRIu64 ") "
827 "for domain %d\n", wi.dkw_devname, xbdi->xbdi_size,
828 xbdi->xbdi_domid);
829 } else {
830 /* If both Ioctls failed set device size to 0 and return */
831 printf("xbdback %s: can't DIOCGWEDGEINFO device "
832 "0x%"PRIx64": %d\n", xbusd->xbusd_path,
833 xbdi->xbdi_dev, err);
834 xbdi->xbdi_size = xbdi->xbdi_dev = 0;
835 vn_close(xbdi->xbdi_vp, FREAD, NOCRED);
836 xbdi->xbdi_vp = NULL;
837 return;
838 }
839again:
840 xbt = xenbus_transaction_start();
841 if (xbt == NULL) {
842 printf("xbdback %s: can't start transaction\n",
843 xbusd->xbusd_path);
844 return;
845 }
846 err = xenbus_printf(xbt, xbusd->xbusd_path, "sectors", "%" PRIu64 ,
847 xbdi->xbdi_size);
848 if (err) {
849 printf("xbdback: failed to write %s/sectors: %d\n",
850 xbusd->xbusd_path, err);
851 goto abort;
852 }
853 err = xenbus_printf(xbt, xbusd->xbusd_path, "info", "%u",
854 xbdi->xbdi_ro ? VDISK_READONLY : 0);
855 if (err) {
856 printf("xbdback: failed to write %s/info: %d\n",
857 xbusd->xbusd_path, err);
858 goto abort;
859 }
860 err = xenbus_printf(xbt, xbusd->xbusd_path, "sector-size", "%lu",
861 (u_long)DEV_BSIZE);
862 if (err) {
863 printf("xbdback: failed to write %s/sector-size: %d\n",
864 xbusd->xbusd_path, err);
865 goto abort;
866 }
867 err = xenbus_printf(xbt, xbusd->xbusd_path, "feature-flush-cache",
868 "%u", 1);
869 if (err) {
870 printf("xbdback: failed to write %s/feature-flush-cache: %d\n",
871 xbusd->xbusd_path, err);
872 goto abort;
873 }
874 err = xenbus_transaction_end(xbt, 0);
875 if (err == EAGAIN)
876 goto again;
877 if (err) {
878 printf("xbdback %s: can't end transaction: %d\n",
879 xbusd->xbusd_path, err);
880 }
881 err = xenbus_switch_state(xbusd, NULL, XenbusStateConnected);
882 if (err) {
883 printf("xbdback %s: can't switch state: %d\n",
884 xbusd->xbusd_path, err);
885 }
886 return;
887abort:
888 xenbus_transaction_end(xbt, 1);
889}

936/*
937 * Main thread routine for one xbdback instance. Woken up by
938 * xbdback_evthandler when a domain has I/O work scheduled in a I/O ring.
939 */
940static void
941xbdback_thread(void *arg)
942{
943 struct xbdback_instance *xbdi = arg;
944
945 for (;;) {
946 mutex_enter(&xbdi->xbdi_lock);
947 switch (xbdi->xbdi_status) {
948 case WAITING:
949 cv_wait(&xbdi->xbdi_cv, &xbdi->xbdi_lock);
950 mutex_exit(&xbdi->xbdi_lock);
951 break;
952 case RUN:
953 xbdi->xbdi_status = WAITING; /* reset state */
954 mutex_exit(&xbdi->xbdi_lock);
955
956 if (xbdi->xbdi_cont == NULL) {
957 xbdi->xbdi_cont = xbdback_co_main;
958 }
959
960 xbdback_trampoline(xbdi, xbdi);
961 break;
962 case DISCONNECTING:
963 if (xbdi->xbdi_pendingreqs > 0) {
964 /* there are pending I/Os. Wait for them. */
965 cv_wait(&xbdi->xbdi_cv, &xbdi->xbdi_lock);
966 mutex_exit(&xbdi->xbdi_lock);
967 break;
968 }
969
970 /* All I/Os should have been processed by now,
971 * xbdi_refcnt should drop to 0 */
972 xbdi_put(xbdi);
973 KASSERT(xbdi->xbdi_refcnt == 0);
974 mutex_exit(&xbdi->xbdi_lock);
975 kthread_exit(0);
976 break;
977 default:
978 panic("%s: invalid state %d",
979 xbdi->xbdi_name, xbdi->xbdi_status);
980 }
981 }
982}

984static void *
985xbdback_co_main(struct xbdback_instance *xbdi, void *obj)
986{
987 (void)obj;
988
989 xbdi->xbdi_req_prod = xbdi->xbdi_ring.ring_n.sring->req_prod;
990 xen_rmb(); /* ensure we see all requests up to req_prod */
991 /*
992 * note that we'll eventually get a full ring of request.
993 * in this case, MASK_BLKIF_IDX(req_cons) == MASK_BLKIF_IDX(req_prod)
994 */
995 xbdi->xbdi_cont = xbdback_co_main_loop;
996 return xbdi;
997}

999/*
1000 * Fetch a blkif request from the ring, and pass control to the appropriate
1001 * continuation.
1002 * If someone asked for disconnection, do not fetch any more request from
1003 * the ring.
1004 */
1005static void *
1006xbdback_co_main_loop(struct xbdback_instance *xbdi, void *obj)
1007{
1008 blkif_request_t *req;
1009 blkif_x86_32_request_t *req32;
1010 blkif_x86_64_request_t *req64;
1011
1012 (void)obj;
1013 req = &xbdi->xbdi_xen_req;
1014 if (xbdi->xbdi_ring.ring_n.req_cons != xbdi->xbdi_req_prod) {
1015 switch(xbdi->xbdi_proto) {
1016 case XBDIP_NATIVE:
1017 memcpy(req, RING_GET_REQUEST(&xbdi->xbdi_ring.ring_n,
1018 xbdi->xbdi_ring.ring_n.req_cons),
1019 sizeof(blkif_request_t));
1020 break;
1021 case XBDIP_32:
1022 req32 = RING_GET_REQUEST(&xbdi->xbdi_ring.ring_32,
1023 xbdi->xbdi_ring.ring_n.req_cons);
1024 req->operation = req32->operation;
1025 req->nr_segments = req32->nr_segments;
1026 req->handle = req32->handle;
1027 req->id = req32->id;
1028 req->sector_number = req32->sector_number;
1029 break;
1030
1031 case XBDIP_64:
1032 req64 = RING_GET_REQUEST(&xbdi->xbdi_ring.ring_64,
1033 xbdi->xbdi_ring.ring_n.req_cons);
1034 req->operation = req64->operation;
1035 req->nr_segments = req64->nr_segments;
1036 req->handle = req64->handle;
1037 req->id = req64->id;
1038 req->sector_number = req64->sector_number;
1039 break;
1040 }
1041 __insn_barrier();
1042 XENPRINTF(("xbdback op %d req_cons 0x%x req_prod 0x%x "
1043 "resp_prod 0x%x id %" PRIu64 "\n", req->operation,
1044 xbdi->xbdi_ring.ring_n.req_cons,
1045 xbdi->xbdi_req_prod,
1046 xbdi->xbdi_ring.ring_n.rsp_prod_pvt,
1047 req->id));
1048 switch(req->operation) {
1049 case BLKIF_OP_READ:
1050 case BLKIF_OP_WRITE:
1051 xbdi->xbdi_cont = xbdback_co_io;
1052 break;
1053 case BLKIF_OP_FLUSH_DISKCACHE:
1054 xbdi_get(xbdi);
1055 xbdi->xbdi_cont = xbdback_co_cache_flush;
1056 break;
1057 default:
1058 if (ratecheck(&xbdi->xbdi_lasterr_time,
1059 &xbdback_err_intvl)) {
1060 printf("%s: unknown operation %d\n",
1061 xbdi->xbdi_name, req->operation);
1062 }
1063 xbdback_send_reply(xbdi, req->id, req->operation,
1064 BLKIF_RSP_ERROR);
1065 xbdi->xbdi_cont = xbdback_co_main_incr;
1066 break;
1067 }
1068 } else {
1069 xbdi->xbdi_cont = xbdback_co_main_done;
1070 }
1071 return xbdi;
1072}
1073
1074/*
1075 * Increment consumer index and move on to the next request. In case
1076 * we want to disconnect, leave continuation now.
1077 */
1078static void *
1079xbdback_co_main_incr(struct xbdback_instance *xbdi, void *obj)
1080{
1081 (void)obj;
1082 blkif_back_ring_t *ring = &xbdi->xbdi_ring.ring_n;
1083
1084 ring->req_cons++;
1085
1086 /*
1087 * Do not bother with locking here when checking for xbdi_status: if
1088 * we get a transient state, we will get the right value at
1089 * the next increment.
1090 */
1091 if (xbdi->xbdi_status == DISCONNECTING)
1092 xbdi->xbdi_cont = NULL;
1093 else
1094 xbdi->xbdi_cont = xbdback_co_main_loop;
1095
1096 /*
1097 * Each time the thread processes a full ring of requests, give
1098 * a chance to other threads to process I/Os too
1099 */
1100 if ((ring->req_cons % BLKIF_RING_SIZE) == 0)
1101 yield();
1102
1103 return xbdi;
1104}
1105
1106/*
1107 * Ring processing is over. If there are any I/O still present for this
1108 * instance, handle them first.
1109 */
1110static void *
1111xbdback_co_main_done(struct xbdback_instance *xbdi, void *obj)
1112{
1113 (void)obj;
1114 if (xbdi->xbdi_io != NULL) {
1115 KASSERT(xbdi->xbdi_io->xio_operation == BLKIF_OP_READ ||
1116 xbdi->xbdi_io->xio_operation == BLKIF_OP_WRITE);
1117 xbdi->xbdi_cont = xbdback_co_map_io;
1118 xbdi->xbdi_cont_aux = xbdback_co_main_done2;
1119 } else {
1120 xbdi->xbdi_cont = xbdback_co_main_done2;
1121 }
1122 return xbdi;
1123}
1124
1125/*
1126 * Check for requests in the instance's ring. In case there are, start again
1127 * from the beginning. If not, stall.
1128 */
1129static void *
1130xbdback_co_main_done2(struct xbdback_instance *xbdi, void *obj)
1131{
1132 int work_to_do;
1133
1134 RING_FINAL_CHECK_FOR_REQUESTS(&xbdi->xbdi_ring.ring_n, work_to_do);
1135 if (work_to_do)
1136 xbdi->xbdi_cont = xbdback_co_main;
1137 else
1138 xbdi->xbdi_cont = NULL;
1139
1140 return xbdi;
1141}

1143/*
1144 * Frontend requested a cache flush operation.
1145 */
1146static void *
1147xbdback_co_cache_flush(struct xbdback_instance *xbdi, void *obj)
1148{
1149 (void)obj;
1150
1151 XENPRINTF(("xbdback_co_cache_flush %p %p\n", xbdi, obj));
1152 if (xbdi->xbdi_io != NULL) {
1153 /* Some I/Os are required for this instance. Process them. */
1154 KASSERT(xbdi->xbdi_io->xio_operation == BLKIF_OP_READ ||
1155 xbdi->xbdi_io->xio_operation == BLKIF_OP_WRITE);
1156 KASSERT(xbdi->xbdi_pendingreqs > 0);
1157 xbdi->xbdi_cont = xbdback_co_map_io;
1158 xbdi->xbdi_cont_aux = xbdback_co_cache_flush2;
1159 } else {
1160 xbdi->xbdi_cont = xbdback_co_cache_flush2;
1161 }
1162 return xbdi;
1163}
1164
1165static void *
1166xbdback_co_cache_flush2(struct xbdback_instance *xbdi, void *obj)
1167{
1168 (void)obj;
1169 XENPRINTF(("xbdback_co_cache_flush2 %p %p\n", xbdi, obj));
1170 if (xbdi->xbdi_pendingreqs > 0) {
1171 /*

1177 return NULL;
1178 }
1179 xbdi->xbdi_cont = xbdback_co_cache_doflush;
1180 return xbdback_pool_get(&xbdback_io_pool, xbdi);
1181}

1198/*
1199 * A read or write I/O request must be processed. Do some checks first,
1200 * then get the segment information directly from the ring request.
1201 */
1202static void *
1203xbdback_co_io(struct xbdback_instance *xbdi, void *obj)
1204{
1205 int i, error;
1206 blkif_request_t *req;
1207 blkif_x86_32_request_t *req32;
1208 blkif_x86_64_request_t *req64;
1209
1210 (void)obj;
1211
1212 /* some sanity checks */
1213 req = &xbdi->xbdi_xen_req;
1214 if (req->nr_segments < 1 ||
1215 req->nr_segments > BLKIF_MAX_SEGMENTS_PER_REQUEST) {
1216 if (ratecheck(&xbdi->xbdi_lasterr_time,
1217 &xbdback_err_intvl)) {
1218 printf("%s: invalid number of segments: %d\n",
1219 xbdi->xbdi_name,
1220 xbdi->xbdi_xen_req.nr_segments);
1221 }
1222 error = EINVAL;
1223 goto end;
1224 }
1225
1226 KASSERT(req->operation == BLKIF_OP_READ ||
1227 req->operation == BLKIF_OP_WRITE);
1228 if (req->operation == BLKIF_OP_WRITE) {
1229 if (xbdi->xbdi_ro) {
1230 error = EROFS;
1231 goto end;
1232 }
1233 }
1234
1235 xbdi->xbdi_segno = 0;
1236
1237 /* copy request segments */
1238 switch(xbdi->xbdi_proto) {
1239 case XBDIP_NATIVE:
1240 /* already copied in xbdback_co_main_loop */
1241 break;
1242 case XBDIP_32:
1243 req32 = RING_GET_REQUEST(&xbdi->xbdi_ring.ring_32,
1244 xbdi->xbdi_ring.ring_n.req_cons);
1245 for (i = 0; i < req->nr_segments; i++)
1246 req->seg[i] = req32->seg[i];
1247 break;
1248 case XBDIP_64:
1249 req64 = RING_GET_REQUEST(&xbdi->xbdi_ring.ring_64,
1250 xbdi->xbdi_ring.ring_n.req_cons);
1251 for (i = 0; i < req->nr_segments; i++)
1252 req->seg[i] = req64->seg[i];
1253 break;
1254 }
1255
1256 xbdi->xbdi_cont = xbdback_co_io_gotreq;
1257 return xbdback_pool_get(&xbdback_request_pool, xbdi);
1258
1259 end:
1260 xbdback_send_reply(xbdi, xbdi->xbdi_xen_req.id,
1261 xbdi->xbdi_xen_req.operation, error);
1262 xbdi->xbdi_cont = xbdback_co_main_incr;
1263 return xbdi;
1264}

1266/*
1267 * We have fetched segment requests from the ring. In case there are already
1268 * I/Os prepared for this instance, we can try coalescing the requests
1269 * with these I/Os.
1270 */
1271static void *
1272xbdback_co_io_gotreq(struct xbdback_instance *xbdi, void *obj)
1273{
1274 struct xbdback_request *xrq;
1275
1276 xrq = xbdi->xbdi_req = obj;
1277
1278 xrq->rq_xbdi = xbdi;
1279 xrq->rq_iocount = 0;
1280 xrq->rq_ioerrs = 0;
1281 xrq->rq_id = xbdi->xbdi_xen_req.id;
1282 xrq->rq_operation = xbdi->xbdi_xen_req.operation;
1283 KASSERT(xbdi->xbdi_req->rq_operation == BLKIF_OP_READ ||
1284 xbdi->xbdi_req->rq_operation == BLKIF_OP_WRITE);
1285
1286 /*
1287 * Request-level reasons not to coalesce: different device,
1288 * different op, or noncontiguous disk sectors (vs. previous
1289 * request handed to us).
1290 */
1291 xbdi->xbdi_cont = xbdback_co_io_loop;
1292 if (xbdi->xbdi_io != NULL) {
1293 struct xbdback_request *last_req;
1294 last_req = SLIST_FIRST(&xbdi->xbdi_io->xio_rq)->car;
1295 XENPRINTF(("xbdback_io domain %d: hoping for sector %" PRIu64
1296 "; got %" PRIu64 "\n", xbdi->xbdi_domid,
1297 xbdi->xbdi_next_sector,
1298 xbdi->xbdi_xen_req.sector_number));
1299 if ((xrq->rq_operation != last_req->rq_operation)
1300 || (xbdi->xbdi_xen_req.sector_number !=
1301 xbdi->xbdi_next_sector)) {
1302 XENPRINTF(("xbdback_io domain %d: segment break\n",
1303 xbdi->xbdi_domid));
1304 xbdi->xbdi_next_sector =
1305 xbdi->xbdi_xen_req.sector_number;
1306 KASSERT(xbdi->xbdi_io->xio_operation == BLKIF_OP_READ ||
1307 xbdi->xbdi_io->xio_operation == BLKIF_OP_WRITE);
1308 xbdi->xbdi_cont_aux = xbdback_co_io_loop;
1309 xbdi->xbdi_cont = xbdback_co_map_io;
1310 }
1311 } else {
1312 xbdi->xbdi_next_sector = xbdi->xbdi_xen_req.sector_number;
1313 }
1314 return xbdi;
1315}

1317/* Handle coalescing of multiple segment requests into one I/O work */
1318static void *
1319xbdback_co_io_loop(struct xbdback_instance *xbdi, void *obj)
1320{
1321 (void)obj;
1322 KASSERT(xbdi->xbdi_req->rq_operation == BLKIF_OP_READ ||
1323 xbdi->xbdi_req->rq_operation == BLKIF_OP_WRITE);
1324 if (xbdi->xbdi_segno < xbdi->xbdi_xen_req.nr_segments) {
1325 uint8_t this_fs, this_ls, last_ls;
1326 grant_ref_t thisgrt;
1327 /*
1328 * Segment-level reason to coalesce: handling full
1329 * pages, or adjacent sector ranges from the same page
1330 * (and yes, this latter does happen). But not if the
1331 * array of client pseudo-physical pages is full.
1332 */
1333 this_fs = xbdi->xbdi_xen_req.seg[xbdi->xbdi_segno].first_sect;
1334 this_ls = xbdi->xbdi_xen_req.seg[xbdi->xbdi_segno].last_sect;
1335 thisgrt = xbdi->xbdi_xen_req.seg[xbdi->xbdi_segno].gref;
1336 XENPRINTF(("xbdback_io domain %d: "
1337 "first,last_sect[%d]=0%o,0%o\n",
1338 xbdi->xbdi_domid, xbdi->xbdi_segno,
1339 this_fs, this_ls));
1340 last_ls = xbdi->xbdi_last_ls = xbdi->xbdi_this_ls;
1341 xbdi->xbdi_this_fs = this_fs;
1342 xbdi->xbdi_this_ls = this_ls;
1343 xbdi->xbdi_thisgrt = thisgrt;
1344 if (xbdi->xbdi_io != NULL) {
1345 if (last_ls == VBD_MAXSECT
1346 && this_fs == 0
1347 && xbdi->xbdi_io->xio_nrma
1348 < XENSHM_MAX_PAGES_PER_REQUEST) {
1349 xbdi->xbdi_same_page = 0;
1350 } else if (last_ls + 1
1351 == this_fs
1352#ifdef notyet
1353 && (last_fas & ~PAGE_MASK)
1354 == (this_fas & ~PAGE_MASK)
1355#else
1356 && 0 /* can't know frame number yet */
1357#endif
1358 ) {
1359#ifdef DEBUG
1360 if (ratecheck(&xbdi->xbdi_lastfragio_time,
1361 &xbdback_fragio_intvl))
1362 printf("%s: domain is sending"
1363 " excessively fragmented I/O\n",
1364 xbdi->xbdi_name);
1365#endif
1366 printf("xbdback_io: would maybe glue "
1367 "same page sec %d (%d->%d)\n",
1368 xbdi->xbdi_segno, this_fs, this_ls);
1369 XENPRINTF(("xbdback_io domain %d: glue same "
1370 "page", xbdi->xbdi_domid));
1371 panic("notyet!");
1372 xbdi->xbdi_same_page = 1;
1373 } else {
1374 KASSERT(xbdi->xbdi_io->xio_operation ==
1375 BLKIF_OP_READ ||
1376 xbdi->xbdi_io->xio_operation ==
1377 BLKIF_OP_WRITE);
1378 xbdi->xbdi_cont_aux = xbdback_co_io_loop;
1379 xbdi->xbdi_cont = xbdback_co_map_io;
1380 return xbdi;
1381 }
1382 } else
1383 xbdi->xbdi_same_page = 0;
1384
1385 if (xbdi->xbdi_io == NULL) {
1386 xbdi->xbdi_cont = xbdback_co_io_gotio;
1387 return xbdback_pool_get(&xbdback_io_pool, xbdi);
1388 } else {
1389 xbdi->xbdi_cont = xbdback_co_io_gotio2;
1390 }
1391 } else {
1392 /* done with the loop over segments; get next request */
1393 xbdi->xbdi_cont = xbdback_co_main_incr;
1394 }
1395 return xbdi;
1396}

1472/* Last routine to manage segments fragments for one I/O */

1475{
1476 struct xbdback_io *xbd_io;

1478 uint8_t this_fs, this_ls;
1479
1480 this_fs = xbdi->xbdi_this_fs;
1481 this_ls = xbdi->xbdi_this_ls;
1482 xbd_io = xbdi->xbdi_io;
1483 seg_size = this_ls - this_fs + 1;
1484
1485 if (seg_size < 0) {

1500 xbd_io->xio_gref[xbd_io->xio_nrma++] = xbdi->xbdi_thisgrt;

1504 XENPRINTF(("xbdback_io domain %d: start sect %ld size %d\n",
1505 xbdi->xbdi_domid, (long)xbdi->xbdi_next_sector, seg_size));
1506
1507 /* Finally, the end of the segment loop! */
1508 xbdi->xbdi_next_sector += seg_size;
1509 ++xbdi->xbdi_segno;
1510 xbdi->xbdi_cont = xbdback_co_io_loop;
1511 return xbdi;

1535/*
1536 * Main xbdback I/O routine. It can either perform a flush operation or
1537 * schedule a read/write operation.
1538 */
1539static void *
1540xbdback_co_do_io(struct xbdback_instance *xbdi, void *obj)
1541{
1542 struct xbdback_io *xbd_io = xbdi->xbdi_io;
1543
1544 switch (xbd_io->xio_operation) {
1545 case BLKIF_OP_FLUSH_DISKCACHE:
1546 {
1547 int error;
1548 int force = 1;
1549
1550 error = VOP_IOCTL(xbdi->xbdi_vp, DIOCCACHESYNC, &force, FWRITE,
1551 kauth_cred_get());
1552 if (error) {
1553 aprint_error("xbdback %s: DIOCCACHESYNC returned %d\n",
1554 xbdi->xbdi_xbusd->xbusd_path, error);
1555 if (error == EOPNOTSUPP || error == ENOTTY)
1556 error = BLKIF_RSP_EOPNOTSUPP;
1557 else
1558 error = BLKIF_RSP_ERROR;
1559 } else
1560 error = BLKIF_RSP_OKAY;
1561 xbdback_send_reply(xbdi, xbd_io->xio_flush_id,
1562 xbd_io->xio_operation, error);
1563 xbdback_pool_put(&xbdback_io_pool, xbd_io);
1564 xbdi_put(xbdi);
1565 xbdi->xbdi_io = NULL;
1566 xbdi->xbdi_cont = xbdback_co_main_incr;
1567 return xbdi;
1568 }
1569 case BLKIF_OP_READ:
1570 case BLKIF_OP_WRITE:
1571 xbd_io->xio_buf.b_data = (void *)
1572 ((vaddr_t)xbd_io->xio_buf.b_data + xbd_io->xio_vaddr);
1573#ifdef DIAGNOSTIC
1574 {
1575 vaddr_t bdata = (vaddr_t)xbd_io->xio_buf.b_data;
1576 int nsegs =
1577 ((((bdata + xbd_io->xio_buf.b_bcount - 1) & ~PAGE_MASK) -
1578 (bdata & ~PAGE_MASK)) >> PAGE_SHIFT) + 1;
1579 if ((bdata & ~PAGE_MASK) != (xbd_io->xio_vaddr & ~PAGE_MASK)) {
1580 printf("xbdback_co_do_io: vaddr %#" PRIxVADDR
1581 " bdata %#" PRIxVADDR "\n",
1582 xbd_io->xio_vaddr, bdata);
1583 panic("xbdback_co_do_io: bdata page change");
1584 }
1585 if (nsegs > xbd_io->xio_nrma) {
1586 printf("xbdback_co_do_io: vaddr %#" PRIxVADDR
1587 " bcount %#x doesn't fit in %d pages\n",
1588 bdata, xbd_io->xio_buf.b_bcount, xbd_io->xio_nrma);
1589 panic("xbdback_co_do_io: not enough pages");
1590 }
1591 }
1592#endif
1593 if ((xbd_io->xio_buf.b_flags & B_READ) == 0) {
1594 mutex_enter(xbd_io->xio_buf.b_vp->v_interlock);
1595 xbd_io->xio_buf.b_vp->v_numoutput++;
1596 mutex_exit(xbd_io->xio_buf.b_vp->v_interlock);
1597 }
1598 bdev_strategy(&xbd_io->xio_buf);
1599 /* will call xbdback_iodone() asynchronously when done */
1600 xbdi->xbdi_io = NULL;
1601 xbdi->xbdi_cont = xbdi->xbdi_cont_aux;
1602 return xbdi;
1603 default:
1604 /* Should never happen */
1605 panic("xbdback_co_do_io: unsupported operation %d",
1606 xbd_io->xio_operation);
1607 }
1608}

1610/*
1611 * Called from softint(9) context when an I/O is done: for each request, send
1612 * back the associated reply to the domain.
1613 *
1614 * This gets reused by xbdback_io_error to report errors from other sources.
1615 */
1616static void
1617xbdback_iodone(struct buf *bp)
1618{
1619 struct xbdback_io *xbd_io;
1620 struct xbdback_instance *xbdi;
1621 int errp;
1622
1623 xbd_io = bp->b_private;
1624 xbdi = xbd_io->xio_xbdi;
1625
1626 XENPRINTF(("xbdback_io domain %d: iodone ptr 0x%lx\n",
1627 xbdi->xbdi_domid, (long)xbd_io));
1628
1629 if (xbd_io->xio_mapped == 1)
1630 xbdback_unmap_shm(xbd_io);
1631
1632 if (bp->b_error != 0) {
1633 printf("xbd IO domain %d: error %d\n",
1634 xbdi->xbdi_domid, bp->b_error);
1635 errp = 1;
1636 } else
1637 errp = 0;
1638
1639 /* for each constituent xbd request */
1640 while(!SLIST_EMPTY(&xbd_io->xio_rq)) {
1641 struct xbdback_fragment *xbd_fr;
1642 struct xbdback_request *xbd_req;
1643 struct xbdback_instance *rxbdi __diagused;
1644 int error;
1645
1646 xbd_fr = SLIST_FIRST(&xbd_io->xio_rq);
1647 xbd_req = xbd_fr->car;
1648 SLIST_REMOVE_HEAD(&xbd_io->xio_rq, cdr);
1649 xbdback_pool_put(&xbdback_fragment_pool, xbd_fr);
1650
1651 if (errp)
1652 ++xbd_req->rq_ioerrs;
1653
1654 /* finalize it only if this was its last I/O */
1655 if (--xbd_req->rq_iocount > 0)
1656 continue;
1657
1658 rxbdi = xbd_req->rq_xbdi;
1659 KASSERT(xbdi == rxbdi);
1660
1661 error = xbd_req->rq_ioerrs > 0
1662 ? BLKIF_RSP_ERROR
1663 : BLKIF_RSP_OKAY;
1664
1665 XENPRINTF(("xbdback_io domain %d: end request %"PRIu64
1666 " error=%d\n",
1667 xbdi->xbdi_domid, xbd_req->rq_id, error));
1668 xbdback_send_reply(xbdi, xbd_req->rq_id,
1669 xbd_req->rq_operation, error);
1670 xbdback_pool_put(&xbdback_request_pool, xbd_req);
1671 }
1672 xbdi_put(xbdi);
1673 atomic_dec_uint(&xbdi->xbdi_pendingreqs);
1674 buf_destroy(&xbd_io->xio_buf);
1675 xbdback_pool_put(&xbdback_io_pool, xbd_io);
1676
1677 xbdback_wakeup_thread(xbdi);
1678}

1695/*
1696 * called once a request has completed. Place the reply in the ring and
1697 * notify the guest OS.
1698 */
1699static void
1700xbdback_send_reply(struct xbdback_instance *xbdi, uint64_t id,
1701 int op, int status)
1702{
1703 blkif_response_t *resp_n;
1704 blkif_x86_32_response_t *resp32;
1705 blkif_x86_64_response_t *resp64;
1706 int notify;
1707
1708 /*
1709 * The ring can be accessed by the xbdback thread, xbdback_iodone()
1710 * handler, or any handler that triggered the shm callback. So
1711 * protect ring access via the xbdi_lock mutex.
1712 */
1713 mutex_enter(&xbdi->xbdi_lock);
1714 switch (xbdi->xbdi_proto) {
1715 case XBDIP_NATIVE:
1716 resp_n = RING_GET_RESPONSE(&xbdi->xbdi_ring.ring_n,
1717 xbdi->xbdi_ring.ring_n.rsp_prod_pvt);
1718 resp_n->id = id;
1719 resp_n->operation = op;
1720 resp_n->status = status;
1721 break;
1722 case XBDIP_32:
1723 resp32 = RING_GET_RESPONSE(&xbdi->xbdi_ring.ring_32,
1724 xbdi->xbdi_ring.ring_n.rsp_prod_pvt);
1725 resp32->id = id;
1726 resp32->operation = op;
1727 resp32->status = status;
1728 break;
1729 case XBDIP_64:
1730 resp64 = RING_GET_RESPONSE(&xbdi->xbdi_ring.ring_64,
1731 xbdi->xbdi_ring.ring_n.rsp_prod_pvt);
1732 resp64->id = id;
1733 resp64->operation = op;
1734 resp64->status = status;
1735 break;
1736 }
1737 xbdi->xbdi_ring.ring_n.rsp_prod_pvt++;
1738 RING_PUSH_RESPONSES_AND_CHECK_NOTIFY(&xbdi->xbdi_ring.ring_n, notify);
1739 mutex_exit(&xbdi->xbdi_lock);
1740
1741 if (notify) {
1742 XENPRINTF(("xbdback_send_reply notify %d\n", xbdi->xbdi_domid));
1743 hypervisor_notify_via_evtchn(xbdi->xbdi_evtchn);
1744 }
1745}

1747/*
1748 * Map multiple entries of an I/O request into backend's VA space.
1749 * The xbd_io->xio_gref array has to be filled out by the caller.
1750 */
1751static void *
1752xbdback_map_shm(struct xbdback_io *xbd_io)
1753{
1754 struct xbdback_instance *xbdi;
1755 struct xbdback_request *xbd_rq;
1756 int error, s;
1757
1758#ifdef XENDEBUG_VBD
1759 int i;
1760 printf("xbdback_map_shm map grant ");
1761 for (i = 0; i < xbd_io->xio_nrma; i++) {
1762 printf("%u ", (u_int)xbd_io->xio_gref[i]);
1763 }
1764#endif
1765
1766 KASSERT(xbd_io->xio_mapped == 0);
1767
1768 xbdi = xbd_io->xio_xbdi;
1769 xbd_rq = SLIST_FIRST(&xbd_io->xio_rq)->car;
1770
1771 error = xen_shm_map(xbd_io->xio_nrma, xbdi->xbdi_domid,
1772 xbd_io->xio_gref, &xbd_io->xio_vaddr, xbd_io->xio_gh,
1773 (xbd_rq->rq_operation == BLKIF_OP_WRITE) ? XSHM_RO : 0);
1774
1775 switch(error) {
1776 case 0:
1777#ifdef XENDEBUG_VBD
1778 printf("handle ");
1779 for (i = 0; i < xbd_io->xio_nrma; i++) {
1780 printf("%u ", (u_int)xbd_io->xio_gh[i]);
1781 }
1782 printf("\n");
1783#endif
1784 xbd_io->xio_mapped = 1;
1785 return xbdi;
1786 case ENOMEM:
1787 s = splvm(); /* XXXSMP */
1788 if (!xbdback_shmcb) {
1789 if (xen_shm_callback(xbdback_shm_callback, xbdi)
1790 != 0) {
1791 splx(s);
1792 panic("xbdback_map_shm: "
1793 "xen_shm_callback failed");
1794 }
1795 xbdback_shmcb = 1;
1796 }
1797 SIMPLEQ_INSERT_TAIL(&xbdback_shmq, xbdi, xbdi_on_hold);
1798 splx(s);
1799 /* Put the thread to sleep until the callback is called */
1800 xbdi->xbdi_cont = xbdback_co_wait_shm_callback;
1801 return NULL;
1802 default:
1803 if (ratecheck(&xbdi->xbdi_lasterr_time, &xbdback_err_intvl)) {
1804 printf("xbdback_map_shm: xen_shm error %d ", error);
1805 }
1806 xbdback_io_error(xbdi->xbdi_io, error);
1807 xbdi->xbdi_io = NULL;
1808 xbdi->xbdi_cont = xbdi->xbdi_cont_aux;
1809 return xbdi;
1810 }
1811}

1813static int
1814xbdback_shm_callback(void *arg)
1815{
1816 int error, s;
1817
1818 /*
1819 * The shm callback may be executed at any level, including
1820 * IPL_BIO and IPL_NET levels. Raise to the lowest priority level
1821 * that can mask both.
1822 */
1823 s = splvm(); /* XXXSMP */
1824 while(!SIMPLEQ_EMPTY(&xbdback_shmq)) {
1825 struct xbdback_instance *xbdi;
1826 struct xbdback_io *xbd_io;
1827 struct xbdback_request *xbd_rq;
1828
1829 xbdi = SIMPLEQ_FIRST(&xbdback_shmq);
1830 xbd_io = xbdi->xbdi_io;
1831 xbd_rq = SLIST_FIRST(&xbd_io->xio_rq)->car;
1832 KASSERT(xbd_io->xio_mapped == 0);
1833
1834 error = xen_shm_map(xbd_io->xio_nrma,
1835 xbdi->xbdi_domid, xbd_io->xio_gref,
1836 &xbd_io->xio_vaddr, xbd_io->xio_gh,
1837 XSHM_CALLBACK |
1838 ((xbd_rq->rq_operation == BLKIF_OP_WRITE) ? XSHM_RO: 0));
1839 switch(error) {
1840 case ENOMEM:
1841 splx(s);
1842 return -1; /* will try again later */
1843 case 0:
1844 SIMPLEQ_REMOVE_HEAD(&xbdback_shmq, xbdi_on_hold);
1845 xbd_io->xio_mapped = 1;
1846 xbdback_wakeup_thread(xbdi);
1847 break;
1848 default:
1849 SIMPLEQ_REMOVE_HEAD(&xbdback_shmq, xbdi_on_hold);
1850 printf("xbdback_shm_callback: xen_shm error %d\n",
1851 error);
1852 xbdback_io_error(xbd_io, error);
1853 xbdi->xbdi_io = NULL;
1854 xbdback_wakeup_thread(xbdi);
1855 break;
1856 }
1857 }
1858 xbdback_shmcb = 0;
1859 splx(s);
1860 return 0;
1861}

1863/*
1864 * Allows waiting for the shm callback to complete.
1865 */
1866static void *
1867xbdback_co_wait_shm_callback(struct xbdback_instance *xbdi, void *obj)
1868{
1869
1870 if (xbdi->xbdi_io == NULL || xbdi->xbdi_io->xio_mapped == 1) {
1871 /*
1872 * Only proceed to next step when the callback reported
1873 * success or failure.
1874 */
1875 xbdi->xbdi_cont = xbdi->xbdi_cont_aux;
1876 return xbdi;
1877 } else {
1878 /* go back to sleep */
1879 return NULL;
1880 }
1881}

1918/*
1919 * Trampoline routine. Calls continuations in a loop and only exits when
1920 * either the returned object or the next callback is NULL.
1921 */
1922static void
1923xbdback_trampoline(struct xbdback_instance *xbdi, void *obj)
1924{
1925 xbdback_cont_t cont;
1926
1927 while(obj != NULL && xbdi->xbdi_cont != NULL) {
1928 cont = xbdi->xbdi_cont;
1929#ifdef DIAGNOSTIC
1930 xbdi->xbdi_cont = (xbdback_cont_t)0xDEADBEEF;
1931#endif
1932 obj = (*cont)(xbdi, obj);
1933#ifdef DIAGNOSTIC
1934 if (xbdi->xbdi_cont == (xbdback_cont_t)0xDEADBEEF) {
1935 printf("xbdback_trampoline: 0x%lx didn't set "
1936 "xbdi->xbdi_cont!\n", (long)cont);
1937 panic("xbdback_trampoline: bad continuation");
1938 }
1939#endif
1940 }
