91void
92xen_parse_cmdline(int what, union xen_cmdline_parseinfo *xcp)
93{
94 char _cmd_line[256], *cmd_line, *opt, *s;
95 int b, i, ipidx = 0;
96 uint32_t xi_ip[5];
97 size_t len;
98
99 len = strlcpy(_cmd_line, xen_start_info.cmd_line, sizeof(_cmd_line));
100 if (len > sizeof(_cmd_line)) {
101 printf("command line exceeded limit of 255 chars. Truncated.\n");
102 }
103 cmd_line = _cmd_line;
104
105 switch (what) {
106 case XEN_PARSE_BOOTDEV:
107 xcp->xcp_bootdev[0] = 0;
108 break;
109 case XEN_PARSE_CONSOLE:
110 xcp->xcp_console[0] = 0;
111 break;
112 }
113
114 while (cmd_line && *cmd_line) {
115 opt = cmd_line;
116 cmd_line = strchr(opt, ' ');
117 if (cmd_line)
118 *cmd_line = 0;
119
120 switch (what) {
121 case XEN_PARSE_BOOTDEV:
122 if (strncasecmp(opt, "bootdev=", 8) == 0) {
123 strncpy(xcp->xcp_bootdev, opt + 8,
124 sizeof(xcp->xcp_bootdev));
125 break;
126 }
127 if (strncasecmp(opt, "root=", 5) == 0) {
128 strncpy(xcp->xcp_bootdev, opt + 5,
129 sizeof(xcp->xcp_bootdev));
130 break;
131 }
132 break;
133
134 case XEN_PARSE_NETINFO:
135 if (xcp->xcp_netinfo.xi_root &&
136 strncasecmp(opt, "nfsroot=", 8) == 0)
137 strncpy(xcp->xcp_netinfo.xi_root, opt + 8,
138 MNAMELEN);
139
140 if (strncasecmp(opt, "ip=", 3) == 0) {
141 memset(xi_ip, 0, sizeof(xi_ip));
142 opt += 3;
143 ipidx = 0;
144 while (opt && *opt) {
145 s = opt;
146 opt = strchr(opt, ':');
147 if (opt)
148 *opt = 0;
149
150 switch (ipidx) {
151 case 0: /* ip */
152 case 1: /* nfs server */
153 case 2: /* gw */
154 case 3: /* mask */
155 case 4: /* host */
156 if (*s == 0)
157 break;
158 for (i = 0; i < 4; i++) {
159 b = strtoul(s, &s, 10);
160 xi_ip[ipidx] = b + 256
161 * xi_ip[ipidx];
162 if (*s != '.')
163 break;
164 s++;
165 }
166 if (i < 3)
167 xi_ip[ipidx] = 0;
168 break;
169 case 5: /* interface */
170 if (!strncmp(s, "xennet", 6))
171 s += 6;
172 else if (!strncmp(s, "eth", 3))
173 s += 3;
174 else
175 break;
176 if (xcp->xcp_netinfo.xi_ifno
177 == strtoul(s, NULL, 10))
178 memcpy(xcp->
179 xcp_netinfo.xi_ip,
180 xi_ip,
181 sizeof(xi_ip));
182 break;
183 }
184 ipidx++;
185
186 if (opt)
187 *opt++ = ':';
188 }
189 }
190 break;
191
192 case XEN_PARSE_CONSOLE:
193 if (strncasecmp(opt, "console=", 8) == 0)
194 strncpy(xcp->xcp_console, opt + 8,
195 sizeof(xcp->xcp_console));
196 break;
197
198 case XEN_PARSE_BOOTFLAGS:
199 if (*opt == '-') {
200 opt++;
201 while(*opt != '\0') {
202 BOOT_FLAG(*opt, boothowto);
203 opt++;
204 }
205 }
206 break;
207 case XEN_PARSE_PCIBACK:
208 if (strncasecmp(opt, "pciback.hide=", 13) == 0)
209 strncpy(xcp->xcp_pcidevs, opt + 13,
210 sizeof(xcp->xcp_pcidevs));
211 break;
212 }
213
214 if (cmd_line)
215 *cmd_line++ = ' ';
216 }
217}

226/*
227 * this function sets up the machdep.xen.suspend sysctl(7) that
228 * controls domain suspend/save.
229 */
230void
231sysctl_xen_suspend_setup(void)
232{
233 const struct sysctlnode *node = NULL;
234
235 /*
236 * dom0 implements sleep support through ACPI. It should not call
237 * this function to register a suspend interface.
238 */
239 KASSERT(!(xendomain_is_dom0()));
240
241 sysctl_createv(NULL, 0, NULL, &node,
242 CTLFLAG_PERMANENT,
243 CTLTYPE_NODE, "machdep", NULL,
244 NULL, 0, NULL, 0,
245 CTL_MACHDEP, CTL_EOL);
246
247 sysctl_createv(NULL, 0, &node, &node,
248 CTLFLAG_PERMANENT,
249 CTLTYPE_NODE, "xen",
250 SYSCTL_DESCR("Xen top level node"),
251 NULL, 0, NULL, 0,
252 CTL_CREATE, CTL_EOL);
253
254 sysctl_createv(NULL, 0, &node, &node,
255 CTLFLAG_PERMANENT | CTLFLAG_READWRITE | CTLFLAG_IMMEDIATE,
256 CTLTYPE_INT, "suspend",
257 SYSCTL_DESCR("Suspend/save current Xen domain"),
258 sysctl_xen_suspend, 0, NULL, 0,
259 CTL_CREATE, CTL_EOL);
260}
261
262static int
263sysctl_xen_suspend(SYSCTLFN_ARGS)
264{
265 int error;
266 struct sysctlnode node;
267
268 node = *rnode;
269 error = sysctl_lookup(SYSCTLFN_CALL(&node));
270
271 if (error || newp == NULL)
272 return error;
273
274 /* only allow domain to suspend when dom0 instructed to do so */
275 if (xen_suspend_allow == false)
276 return EAGAIN;
277
278 xen_suspend_domain();
279
280 return 0;
281
282}

287/*
288 * Last operations before suspending domain
289 */
290static void
291xen_prepare_suspend(void)
292{
293
294 kpreempt_disable();
295
296 pmap_xen_suspend();
297 xc_wait(xc_broadcast(0, &xen_suspendclocks_xc, NULL, NULL));
298
299 /*
300 * save/restore code does not translate these MFNs to their
301 * associated PFNs, so we must do it
302 */
303 xen_start_info.store_mfn =
304 atop(xpmap_mtop(ptoa(xen_start_info.store_mfn)));
305 xen_start_info.console_mfn =
306 atop(xpmap_mtop(ptoa(xen_start_info.console_mfn)));
307
308 DPRINTK(("suspending domain\n"));
309 aprint_verbose("suspending domain\n");
310
311 /* invalidate the shared_info page */
312 if (HYPERVISOR_update_va_mapping((vaddr_t)HYPERVISOR_shared_info,
313 0, UVMF_INVLPG)) {
314 DPRINTK(("HYPERVISOR_shared_info page invalidation failed"));
315 HYPERVISOR_crash();
316 }
317
318}
319
320static void
321xen_suspendclocks_xc(void *a, void *b)
322{
323
324 kpreempt_disable();
325 xen_suspendclocks(curcpu());
326 kpreempt_enable();
327}
328
329/*
330 * First operations before restoring domain context
331 */
332static void
333xen_prepare_resume(void)
334{
335 /* map the new shared_info page */
336 if (HYPERVISOR_update_va_mapping((vaddr_t)HYPERVISOR_shared_info,
337 xen_start_info.shared_info | PTE_W | PTE_P,
338 UVMF_INVLPG)) {
339 DPRINTK(("could not map new shared info page"));
340 HYPERVISOR_crash();
341 }
342
343 pmap_xen_resume();
344
345 if (xen_start_info.nr_pages != physmem) {
346 /*
347 * XXX JYM for now, we crash - fix it with memory
348 * hotplug when supported
349 */
350 DPRINTK(("xen_start_info.nr_pages != physmem"));
351 HYPERVISOR_crash();
352 }
353
354 DPRINTK(("preparing domain resume\n"));
355 aprint_verbose("preparing domain resume\n");
356
357 xen_suspend_allow = false;
358
359 xc_wait(xc_broadcast(0, xen_resumeclocks_xc, NULL, NULL));
360
361 kpreempt_enable();
362
363}

365static void
366xen_resumeclocks_xc(void *a, void *b)
367{
368
369 kpreempt_disable();
370 xen_resumeclocks(curcpu());
371 kpreempt_enable();
372}

374static void
375xen_suspend_domain(void)
376{
377 paddr_t mfn;
378 int s = splvm(); /* XXXSMP */
379
380 /*
381 * console becomes unavailable when suspended, so
382 * direct communications to domain are hampered from there on.
383 * We can only rely on low level primitives like printk(), until
384 * console is fully restored
385 */
386 if (!pmf_system_suspend(PMF_Q_NONE)) {
387 DPRINTK(("devices suspend failed"));
388 HYPERVISOR_crash();
389 }
390
391 /*
392 * obtain the MFN of the start_info page now, as we will not be
393 * able to do it once pmap is locked
394 */
395 pmap_extract_ma(pmap_kernel(), (vaddr_t)&xen_start_info, &mfn);
396 mfn >>= PAGE_SHIFT;
397
398 xen_prepare_suspend();
399
400 DPRINTK(("calling HYPERVISOR_suspend()\n"));
401 if (HYPERVISOR_suspend(mfn) != 0) {
402 /* XXX JYM: implement checkpoint/snapshot (ret == 1) */
403 DPRINTK(("HYPERVISOR_suspend() failed"));
404 HYPERVISOR_crash();
405 }
406
407 DPRINTK(("left HYPERVISOR_suspend()\n"));
408
409 xen_prepare_resume();
410
411 DPRINTK(("resuming devices\n"));
412 if (!pmf_system_resume(PMF_Q_NONE)) {
413 DPRINTK(("devices resume failed\n"));
414 HYPERVISOR_crash();
415 }
416
417 splx(s);
418
419 /* xencons is back online, we can print to console */
420 aprint_verbose("domain resumed\n");
421
422}

425#define PRINTK_BUFSIZE 1024
426void
427printk(const char *fmt, ...)
428{
429 va_list ap;
430 int ret;
431 static char buf[PRINTK_BUFSIZE];
432
433 va_start(ap, fmt);
434 ret = vsnprintf(buf, PRINTK_BUFSIZE - 1, fmt, ap);
435 va_end(ap);
436 buf[ret] = 0;
437 (void)HYPERVISOR_console_io(CONSOLEIO_write, ret, buf);
438}

440bool xen_feature_tables[XENFEAT_NR_SUBMAPS * 32];
441
442void
443xen_init_features(void)
444{
445 xen_feature_info_t features;
446
447 for (int sm = 0; sm < XENFEAT_NR_SUBMAPS; sm++) {
448 features.submap_idx = sm;
449 if (HYPERVISOR_xen_version(XENVER_get_features, &features) < 0)
450 break;
451 for (int f = 0; f < 32; f++) {
452 xen_feature_tables[sm * 32 + f] =
453 (features.submap & (1 << f)) ? 1 : 0;
454 }
455 }
