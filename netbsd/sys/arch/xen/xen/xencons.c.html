159int
160xencons_match(device_t parent, cfdata_t match, void *aux)
161{

169void
170xencons_attach(device_t parent, device_t self, void *aux)
171{
172 struct xencons_softc *sc = device_private(self);
173
174 aprint_normal(": Xen Virtual Console Driver\n");
175
176 sc->sc_dev = self;
177 sc->sc_tty = tty_alloc();
178 tty_attach(sc->sc_tty);
179 sc->sc_tty->t_oproc = xencons_start;
180 sc->sc_tty->t_param = xencons_param;
181
182 if (xencons_isconsole) {
183 int maj;
184
185 /* Locate the major number. */
186 maj = cdevsw_lookup_major(&xencons_cdevsw);
187
188 /* There can be only one, but it can have any unit number. */
189 cn_tab->cn_dev = makedev(maj, device_unit(self));
190
191 aprint_verbose_dev(self, "console major %d, unit %d\n",
192 maj, device_unit(self));
193
194 sc->sc_tty->t_dev = cn_tab->cn_dev;
195
196#ifdef DDB
197 /* Set db_max_line to avoid paging. */
198 db_max_line = 0x7fffffff;
199#endif
200
201 xencons_console_device = sc;
202
203 xencons_resume(self, PMF_Q_NONE);
204 }
205 sc->polling = 0;
206
207 if (!pmf_device_register(self, xencons_suspend, xencons_resume))
208 aprint_error_dev(self, "couldn't establish power handler\n");
209}

211static bool
212xencons_suspend(device_t dev, const pmf_qual_t *qual) {
213
214 int evtch;
215
216 /* dom0 console should not be suspended */
217 if (!xendomain_is_dom0()) {
218 evtch = xen_start_info.console_evtchn;
219 hypervisor_mask_event(evtch);
220 xen_intr_disestablish(ih);
221 aprint_verbose_dev(dev, "removed event channel %d\n", ih->ih_pin);
222 }
223
224 return true;
225}
226
227static bool
228xencons_resume(device_t dev, const pmf_qual_t *qual) {
229
230 int evtch = -1;
231
232 if (xendomain_is_dom0()) {
233 /* dom0 console resume is required only during first start-up */
234 if (cold) {
235 evtch = bind_virq_to_evtch(VIRQ_CONSOLE);
236 ih = xen_intr_establish_xname(-1, &xen_pic, evtch,
237 IST_LEVEL, IPL_TTY, xencons_intr,
238 xencons_console_device, false,
239 device_xname(dev));
240 KASSERT(ih != NULL);
241 }
242 } else {
243 evtch = xen_start_info.console_evtchn;
244 ih = xen_intr_establish_xname(-1, &xen_pic, evtch,
245 IST_LEVEL, IPL_TTY, xencons_handler,
246 xencons_console_device, false, device_xname(dev));
247 KASSERT(ih != NULL);
248 }
249
250 if (evtch != -1) {
251 aprint_verbose_dev(dev, "using event channel %d\n", evtch);
252 hypervisor_unmask_event(evtch);
253 }
254
255 return true;
256}

258int
259xencons_open(dev_t dev, int flag, int mode, struct lwp *l)
260{
261 struct xencons_softc *sc;
262 struct tty *tp;
263
264 sc = device_lookup_private(&xencons_cd, XENCONS_UNIT(dev));
265 if (sc == NULL)
266 return (ENXIO);
267
268 tp = sc->sc_tty;
269
270 if (kauth_authorize_device_tty(l->l_cred, KAUTH_DEVICE_TTY_OPEN, tp))
271 return (EBUSY);
272
273 if ((tp->t_state & TS_ISOPEN) == 0 && tp->t_wopen == 0) {
274 tp->t_dev = dev;
275 ttychars(tp);
276 tp->t_iflag = TTYDEF_IFLAG;
277 tp->t_oflag = TTYDEF_OFLAG;
278 tp->t_cflag = TTYDEF_CFLAG;
279 tp->t_lflag = TTYDEF_LFLAG;
280 tp->t_ispeed = tp->t_ospeed = TTYDEF_SPEED;
281 xencons_param(tp, &tp->t_termios);
282 ttsetwater(tp);
283 }
284 tp->t_state |= TS_CARR_ON;
285
286 return ((*tp->t_linesw->l_open)(dev, tp));
287}

346int
347xencons_ioctl(dev_t dev, u_long cmd, void *data, int flag, struct lwp *l)
348{
349 struct xencons_softc *sc = device_lookup_private(&xencons_cd,
350 XENCONS_UNIT(dev));
351 struct tty *tp = sc->sc_tty;
352 int error;
353
354 error = (*tp->t_linesw->l_ioctl)(tp, cmd, data, flag, l);
355 if (error != EPASSTHROUGH)
356 return (error);
357
358 error = ttioctl(tp, cmd, data, flag, l);
359 if (error != EPASSTHROUGH)
360 return (error);
361
362 switch (cmd) {
363 default:
364 return (EPASSTHROUGH);
365 }
366
367#ifdef DIAGNOSTIC
368 panic("xencons_ioctl: impossible");
369#endif
370}

372void
373xencons_start(struct tty *tp)
374{
375 struct clist *cl;
376 int s;
377
378 s = spltty();
379 if (tp->t_state & (TS_TIMEOUT | TS_BUSY | TS_TTSTOP))
380 goto out;
381 tp->t_state |= TS_BUSY;
382 splx(s);
383
384 /*
385 * We need to do this outside spl since it could be fairly
386 * expensive and we don't want our serial ports to overflow.
387 */
388 cl = &tp->t_outq;
389 if (xendomain_is_dom0()) {
390 int len, r;
391 u_char buf[XENCONS_BURST+1];
392
393 len = q_to_b(cl, buf, XENCONS_BURST);
394 while (len > 0) {
395 r = HYPERVISOR_console_io(CONSOLEIO_write, len, buf);
396 if (r <= 0)
397 break;
398 len -= r;
399 }
400 } else {
401 XENCONS_RING_IDX cons, prod, len;
402
403#define XNC_OUT (xencons_interface->out)
404 cons = xencons_interface->out_cons;
405 prod = xencons_interface->out_prod;
406 xen_rmb();
407 while (prod != cons + sizeof(xencons_interface->out)) {
408 if (MASK_XENCONS_IDX(prod, XNC_OUT) <
409 MASK_XENCONS_IDX(cons, XNC_OUT)) {
410 len = MASK_XENCONS_IDX(cons, XNC_OUT) -
411 MASK_XENCONS_IDX(prod, XNC_OUT);
412 } else {
413 len = sizeof(XNC_OUT) -
414 MASK_XENCONS_IDX(prod, XNC_OUT);
415 }
416 len = q_to_b(cl, __UNVOLATILE(
417 &XNC_OUT[MASK_XENCONS_IDX(prod, XNC_OUT)]), len);
418 if (len == 0)
419 break;
420 prod = prod + len;
421 }
422 xen_wmb();
423 xencons_interface->out_prod = prod;
424 xen_wmb();
425 hypervisor_notify_via_evtchn(xen_start_info.console.domU.evtchn);
426#undef XNC_OUT
427 }
428
429 s = spltty();
430 tp->t_state &= ~TS_BUSY;
431 if (ttypull(tp)) {
432 tp->t_state |= TS_TIMEOUT;
433 callout_schedule(&tp->t_rstrt_ch, 1);
434 }
435out:
436 splx(s);
437}

446/* Non-privileged console interrupt routine */
447static int
448xencons_handler(void *arg)
449{
450 struct xencons_softc *sc = arg;
451 XENCONS_RING_IDX cons, prod, len;
452 int s = spltty();
453
454 if (sc->polling) {
455 splx(s);
456 return 1;
457 }
458
459
460#define XNC_IN (xencons_interface->in)
461
462 cons = xencons_interface->in_cons;
463 prod = xencons_interface->in_prod;
464 xen_rmb();
465 while (cons != prod) {
466 if (MASK_XENCONS_IDX(cons, XNC_IN) <
467 MASK_XENCONS_IDX(prod, XNC_IN))
468 len = MASK_XENCONS_IDX(prod, XNC_IN) -
469 MASK_XENCONS_IDX(cons, XNC_IN);
470 else
471 len = sizeof(XNC_IN) - MASK_XENCONS_IDX(cons, XNC_IN);
472
473 xencons_tty_input(sc, __UNVOLATILE(
474 &XNC_IN[MASK_XENCONS_IDX(cons, XNC_IN)]), len);
475 if (__predict_false(xencons_interface->in_cons != cons)) {
476 /* catch up with xenconscn_getc() */
477 cons = xencons_interface->in_cons;
478 prod = xencons_interface->in_prod;
479 xen_rmb();
480 } else {
481 cons += len;
482 xen_wmb();
483 xencons_interface->in_cons = cons;
484 xen_wmb();
485 }
486 }
487 hypervisor_notify_via_evtchn(xen_start_info.console.domU.evtchn);
488 splx(s);
489 return 1;
490#undef XNC_IN
491}

494void
495xencons_tty_input(struct xencons_softc *sc, char* buf, int len)
496{
497 struct tty *tp;
498 int i;
499
500 tp = sc->sc_tty;
501 if (tp == NULL)
502 return;
503
504 for (i = 0; i < len; i++) {
505 cn_check_magic(sc->sc_tty->t_dev, buf[i], xencons_cnm_state);
506 (*tp->t_linesw->l_rint)(buf[i], tp);
507 }
508}

510/* privileged receive callback */
511int
512xencons_intr(void *p)
513{
514 static char rbuf[16];
515 int len;
516 struct xencons_softc *sc = p;
517
518 if (sc == NULL)
519 /* Interrupt may happen during resume */
520 return 1;
521
522 if (sc->polling)
523 return 1;
524
525 while ((len =
526 HYPERVISOR_console_io(CONSOLEIO_read, sizeof(rbuf), rbuf)) > 0) {
527 xencons_tty_input(sc, rbuf, len);
528 }
529 return 1;
530}

546int
547xenconscn_getc(dev_t dev)
548{
549 char c;
550 int s = spltty();
551 XENCONS_RING_IDX cons, prod;
552
553 if (xencons_console_device && xencons_console_device->polling == 0) {
554 printf("xenconscn_getc() but not polling\n");
555 splx(s);
556 return 0;
557 }
558 if (xendomain_is_dom0()) {
559 while (HYPERVISOR_console_io(CONSOLEIO_read, 1, &c) == 0)
560 ;
561 cn_check_magic(dev, c, xencons_cnm_state);
562 splx(s);
563 return c;
564 }
565 if (xencons_console_device == NULL) {
566 printf("xenconscn_getc(): not console\n");
567 while (1)
568 ; /* loop here instead of in ddb */
569 splx(s);
570 return 0;
571 }
572
573 if (xencons_console_device->polling == 0) {
574 printf("xenconscn_getc() but not polling\n");
575 splx(s);
576 return 0;
577 }
578
579 cons = xencons_interface->in_cons;
580 prod = xencons_interface->in_prod;
581 xen_rmb();
582 while (cons == prod) {
583 HYPERVISOR_yield();
584 prod = xencons_interface->in_prod;
585 }
586 xen_rmb();
587 c = xencons_interface->in[MASK_XENCONS_IDX(xencons_interface->in_cons,
588 xencons_interface->in)];
589 xen_rmb();
590 xencons_interface->in_cons = cons + 1;
591 cn_check_magic(dev, c, xencons_cnm_state);
592 splx(s);
593 return c;
594}

596void
597xenconscn_putc(dev_t dev, int c)
598{
599 int s = spltty();
600 XENCONS_RING_IDX cons, prod;
601
602 if (xendomain_is_dom0()) {
603 u_char buf[1];
604
605 buf[0] = c;
606 (void)HYPERVISOR_console_io(CONSOLEIO_write, 1, buf);
607 } else {
608 XENPRINTK(("xenconscn_putc(%c)\n", c));
609
610 cons = xencons_interface->out_cons;
611 prod = xencons_interface->out_prod;
612 xen_rmb();
613 while (prod == cons + sizeof(xencons_interface->out)) {
614 cons = xencons_interface->out_cons;
615 prod = xencons_interface->out_prod;
616 xen_rmb();
617 }
618 xencons_interface->out[MASK_XENCONS_IDX(xencons_interface->out_prod,
619 xencons_interface->out)] = c;
620 xen_rmb();
621 xencons_interface->out_prod++;
622 xen_rmb();
623 hypervisor_notify_via_evtchn(xen_start_info.console.domU.evtchn);
624 splx(s);
625 }
626}
