57/*
58 * Interface between the event channel and userland.
59 * Each process with a xenevt device instance open can regiter events it
60 * wants to receive. It will get pending events by read(), eventually blocking
61 * until some event is available. Pending events are ack'd by a bitmask
62 * write()en to the device. Some special operations (such as events binding)
63 * are done though ioctl().
64 * Processes get a device instance by opening a cloning device.
65 */
66

107/* minor numbers */
108#define DEV_EVT 0
109#define DEV_XSD 1

111/* per-instance datas */
112#define XENEVT_RING_SIZE 2048
113#define XENEVT_RING_MASK 2047
114
115#define BYTES_PER_PORT (sizeof(evtchn_port_t) / sizeof(uint8_t))
116
117struct xenevt_d {
118 kmutex_t lock;
119 kcondvar_t cv;
120 STAILQ_ENTRY(xenevt_d) pendingq;
121 bool pending;
122 evtchn_port_t ring[2048];
123 u_int ring_read; /* pointer of the reader */
124 u_int ring_write; /* pointer of the writer */
125 u_int flags;
126#define XENEVT_F_OVERFLOW 0x01 /* ring overflow */
127#define XENEVT_F_FREE 0x02 /* free entry */
128 struct selinfo sel; /* used by poll */
129 struct cpu_info *ci; /* prefered CPU for events for this device */
130};

143/* pending events */
144long xenevt_ev1;

148static evtchn_port_t xenevt_alloc_event(void)
149{
150 evtchn_op_t op;
151 op.cmd = EVTCHNOP_alloc_unbound;
152 op.u.alloc_unbound.dom = DOMID_SELF;
153 op.u.alloc_unbound.remote_dom = DOMID_SELF;
154 if (HYPERVISOR_event_channel_op(&op) != 0)
155 panic("%s: Failed to allocate loopback event\n", __func__);
156
157 return op.u.alloc_unbound.port;
158}

160/* called at boot time */
161void
162xenevtattach(int n)
163{
164 struct intrhand *ih;
165 int level = IPL_HIGH;
166 bool mpsafe = (level != IPL_VM);
167
168 mutex_init(&devevent_lock, MUTEX_DEFAULT, IPL_HIGH);
169 STAILQ_INIT(&devevent_pending);
170
171 devevent_sih = softint_establish(SOFTINT_SERIAL,
172 (void (*)(void *))xenevt_notify, NULL);
173 memset(devevent, 0, sizeof(devevent));
174 xenevt_ev1 = 0;
175 memset(xenevt_ev2, 0, sizeof(xenevt_ev2));
176
177 /*
178 * Allocate a loopback event port.
179 * This helps us massage xenevt_processevt() into the
180 * callchain at the appropriate level using only
181 * xen_intr_establish_xname().
182 */
183 evtchn_port_t evtchn = xenevt_alloc_event();
184
185 /* The real objective here is to wiggle into the ih callchain for IPL level */
186 ih = xen_intr_establish_xname(-1, &xen_pic, evtchn, IST_LEVEL, level,
187 xenevt_processevt, NULL, mpsafe, "xenevt");
188
189 KASSERT(ih != NULL);
190}

192/* register pending event - always called with interrupt disabled */
193void
194xenevt_setipending(int l1, int l2)
195{
196 atomic_or_ulong(&xenevt_ev1, 1UL << l1);
197 atomic_or_ulong(&xenevt_ev2[l1], 1UL << l2);
198 atomic_or_32(&cpu_info_primary.ci_xpending, 1 << IPL_HIGH);
199}
200
201/* process pending events */
202static int
203xenevt_processevt(void *v)
204{
205 long l1, l2;
206 int l1i, l2i;
207 int port;
208
209 l1 = xen_atomic_xchg(&xenevt_ev1, 0);
210 while ((l1i = xen_ffs(l1)) != 0) {
211 l1i--;
212 l1 &= ~(1UL << l1i);
213 l2 = xen_atomic_xchg(&xenevt_ev2[l1i], 0);
214 while ((l2i = xen_ffs(l2)) != 0) {
215 l2i--;
216 l2 &= ~(1UL << l2i);
217 port = (l1i << LONG_SHIFT) + l2i;
218 xenevt_event(port);
219 }
220 }
221
222 return 0;
223}

226/* event callback, called at splhigh() */
227void
228xenevt_event(int port)
229{
230 struct xenevt_d *d;
231
232 mutex_enter(&devevent_lock);
233 d = devevent[port];
234 if (d != NULL) {
235 xenevt_record(d, port);
236
237 if (d->pending == false) {
238 STAILQ_INSERT_TAIL(&devevent_pending, d, pendingq);
239 d->pending = true;
240 mutex_exit(&devevent_lock);
241 softint_schedule(devevent_sih);
242 return;
243 }
244 }
245 mutex_exit(&devevent_lock);
246}
247
248void
249xenevt_notify(void)
250{
251 struct xenevt_d *d;
252
253 for (;;) {
254 mutex_enter(&devevent_lock);
255 d = STAILQ_FIRST(&devevent_pending);
256 if (d == NULL) {
257 mutex_exit(&devevent_lock);
258 break;
259 }
260 STAILQ_REMOVE_HEAD(&devevent_pending, pendingq);
261 d->pending = false;
262 mutex_enter(&d->lock);
263 if (d->flags & XENEVT_F_FREE) {
264 xenevt_free(d);
265 mutex_exit(&devevent_lock);
266 } else {
267 mutex_exit(&devevent_lock);
268 selnotify(&d->sel, 0, 1);
269 cv_broadcast(&d->cv);
270 mutex_exit(&d->lock);
271 }
272 }
273}

275static void
276xenevt_record(struct xenevt_d *d, evtchn_port_t port)
277{
278
279 /*
280 * This algorithm overflows for one less slot than available.
281 * Not really an issue, and the correct algorithm would be more
282 * complex
283 */
284
285 mutex_enter(&d->lock);
286 if (d->ring_read ==
287 ((d->ring_write + 1) & XENEVT_RING_MASK)) {
288 d->flags |= XENEVT_F_OVERFLOW;
289 printf("xenevt_event: ring overflow port %d\n", port);
290 } else {
291 d->ring[d->ring_write] = port;
292 d->ring_write = (d->ring_write + 1) & XENEVT_RING_MASK;
293 }
294 mutex_exit(&d->lock);
295}

297/* open the xenevt device; this is where we clone */
298int
299xenevtopen(dev_t dev, int flags, int mode, struct lwp *l)
300{
301 struct xenevt_d *d;
302 struct file *fp;
303 int fd, error;
304
305 switch(minor(dev)) {
306 case DEV_EVT:
307 /* falloc() will fill in the descriptor for us. */
308 if ((error = fd_allocfile(&fp, &fd)) != 0)
309 return error;
310
311 d = malloc(sizeof(*d), M_DEVBUF, M_WAITOK | M_ZERO);
312 d->ci = &cpu_info_primary;
313 mutex_init(&d->lock, MUTEX_DEFAULT, IPL_HIGH);
314 cv_init(&d->cv, "xenevt");
315 selinit(&d->sel);
316 return fd_clone(fp, fd, flags, &xenevt_fileops, d);
317 case DEV_XSD:
318 /* no clone for /dev/xsd_kva */
319 return (0);
320 default:
321 break;
322 }
323 return ENODEV;
324}

326/* read from device: only for /dev/xsd_kva, xenevt is done though fread */
327int
328xenevtread(dev_t dev, struct uio *uio, int flags)
329{
330#define LD_STRLEN 21 /* a 64bit integer needs 20 digits in base10 */
331 if (minor(dev) == DEV_XSD) {
332 char strbuf[LD_STRLEN], *bf;
333 int off, error;
334 size_t len;
335
336 off = (int)uio->uio_offset;
337 if (off < 0)
338 return EINVAL;
339 len = snprintf(strbuf, sizeof(strbuf), "%ld\n",
340 xen_start_info.store_mfn);
341 if (off >= len) {
342 bf = strbuf;
343 len = 0;
344 } else {
345 bf = &strbuf[off];
346 len -= off;
347 }
348 error = uiomove(bf, len, uio);
349 return error;
350 }
351 return ENODEV;
352}

354/* mmap: only for xsd_kva */
355paddr_t
356xenevtmmap(dev_t dev, off_t off, int prot)
357{
358 if (minor(dev) == DEV_XSD) {
359 /* only one page, so off is always 0 */
360 if (off != 0)
361 return -1;
362 return x86_btop(
363 xpmap_mtop((paddr_t)xen_start_info.store_mfn << PAGE_SHIFT));
364 }
365 return -1;
366}

368static void
369xenevt_free(struct xenevt_d *d)
370{
371 int i;
372 KASSERT(mutex_owned(&devevent_lock));
373 KASSERT(mutex_owned(&d->lock));
374
375 for (i = 0; i < NR_EVENT_CHANNELS; i++ ) {
376 if (devevent[i] == d) {
377 evtchn_op_t op = { .cmd = 0 };
378 int error;
379
380 hypervisor_mask_event(i);
381 xen_atomic_clear_bit(&d->ci->ci_evtmask[0], i);
382 devevent[i] = NULL;
383
384 op.cmd = EVTCHNOP_close;
385 op.u.close.port = i;
386 if ((error = HYPERVISOR_event_channel_op(&op))) {
387 printf("xenevt_fclose: error %d from "
388 "hypervisor\n", -error);
389 }
390 }
391 }
392 mutex_exit(&d->lock);
393 seldestroy(&d->sel);
394 cv_destroy(&d->cv);
395 mutex_destroy(&d->lock);
396 free(d, M_DEVBUF);
397}

399static int
400xenevt_fclose(struct file *fp)
401{
402 struct xenevt_d *d = fp->f_data;
403
404 mutex_enter(&devevent_lock);
405 mutex_enter(&d->lock);
406 if (d->pending) {
407 d->flags |= XENEVT_F_FREE;
408 mutex_exit(&d->lock);
409 } else {
410 xenevt_free(d);
411 }
412
413 mutex_exit(&devevent_lock);
414 fp->f_data = NULL;
415 return (0);
416}

418static int
419xenevt_fread(struct file *fp, off_t *offp, struct uio *uio,
420 kauth_cred_t cred, int flags)
421{
422 struct xenevt_d *d = fp->f_data;
423 int error, ring_read, ring_write;
424 size_t len, uio_len;
425
426 error = 0;
427 mutex_enter(&d->lock);
428 while (error == 0) {
429 ring_read = d->ring_read;
430 ring_write = d->ring_write;
431 if (ring_read != ring_write) {
432 break;
433 }
434 if (d->flags & XENEVT_F_OVERFLOW) {
435 break;
436 }
437
438 /* nothing to read */
439 if ((fp->f_flag & FNONBLOCK) == 0) {
440 error = cv_wait_sig(&d->cv, &d->lock);
441 } else {
442 error = EAGAIN;
443 }
444 }
445 if (error == 0 && (d->flags & XENEVT_F_OVERFLOW)) {
446 error = EFBIG;
447 }
448 mutex_exit(&d->lock);
449
450 if (error) {
451 return error;
452 }
453
454 uio_len = uio->uio_resid / BYTES_PER_PORT;
455 if (ring_read <= ring_write)
456 len = ring_write - ring_read;
457 else
458 len = XENEVT_RING_SIZE - ring_read;
459 if (len > uio_len)
460 len = uio_len;
461 error = uiomove(&d->ring[ring_read], len * BYTES_PER_PORT, uio);
462 if (error)
463 return error;
464 ring_read = (ring_read + len) & XENEVT_RING_MASK;
465 uio_len = uio->uio_resid / BYTES_PER_PORT;
466 if (uio_len == 0)
467 goto done;
468 /* ring wrapped, read the second part */
469 len = ring_write - ring_read;
470 if (len > uio_len)
471 len = uio_len;
472 error = uiomove(&d->ring[ring_read], len * BYTES_PER_PORT, uio);
473 if (error)
474 return error;
475 ring_read = (ring_read + len) & XENEVT_RING_MASK;
476
477done:
478 mutex_enter(&d->lock);
479 d->ring_read = ring_read;
480 mutex_exit(&d->lock);
481
482 return 0;
483}

485static int
486xenevt_fwrite(struct file *fp, off_t *offp, struct uio *uio,
487 kauth_cred_t cred, int flags)
488{
489 struct xenevt_d *d = fp->f_data;
490 uint16_t *chans;
491 int i, nentries, error;
492
493 if (uio->uio_resid == 0)
494 return (0);
495 nentries = uio->uio_resid / sizeof(uint16_t);
496 if (nentries >= NR_EVENT_CHANNELS)
497 return EMSGSIZE;
498 chans = kmem_alloc(nentries * sizeof(uint16_t), KM_SLEEP);
499 error = uiomove(chans, uio->uio_resid, uio);
500 if (error)
501 goto out;
502 mutex_enter(&devevent_lock);
503 for (i = 0; i < nentries; i++) {
504 if (chans[i] < NR_EVENT_CHANNELS &&
505 devevent[chans[i]] == d) {
506 hypervisor_unmask_event(chans[i]);
507 }
508 }
509 mutex_exit(&devevent_lock);
510out:
511 kmem_free(chans, nentries * sizeof(uint16_t));
512 return 0;
513}

515static int
516xenevt_fioctl(struct file *fp, u_long cmd, void *addr)
517{
518 struct xenevt_d *d = fp->f_data;
519 evtchn_op_t op = { .cmd = 0 };
520 int error;
521
522 switch(cmd) {
523 case EVTCHN_RESET:
524 case IOCTL_EVTCHN_RESET:
525 mutex_enter(&d->lock);
526 d->ring_read = d->ring_write = 0;
527 d->flags = 0;
528 mutex_exit(&d->lock);
529 break;
530 case IOCTL_EVTCHN_BIND_VIRQ:
531 {
532 struct ioctl_evtchn_bind_virq *bind_virq = addr;
533 op.cmd = EVTCHNOP_bind_virq;
534 op.u.bind_virq.virq = bind_virq->virq;
535 op.u.bind_virq.vcpu = 0;
536 if ((error = HYPERVISOR_event_channel_op(&op))) {
537 printf("IOCTL_EVTCHN_BIND_VIRQ failed: virq %d error %d\n", bind_virq->virq, error);
538 return -error;
539 }
540 bind_virq->port = op.u.bind_virq.port;
541 mutex_enter(&devevent_lock);
542 KASSERT(devevent[bind_virq->port] == NULL);
543 devevent[bind_virq->port] = d;
544 mutex_exit(&devevent_lock);
545 xen_atomic_set_bit(&d->ci->ci_evtmask[0], bind_virq->port);
546 hypervisor_unmask_event(bind_virq->port);
547 break;
548 }
549 case IOCTL_EVTCHN_BIND_INTERDOMAIN:
550 {
551 struct ioctl_evtchn_bind_interdomain *bind_intd = addr;
552 op.cmd = EVTCHNOP_bind_interdomain;
553 op.u.bind_interdomain.remote_dom = bind_intd->remote_domain;
554 op.u.bind_interdomain.remote_port = bind_intd->remote_port;
555 if ((error = HYPERVISOR_event_channel_op(&op)))
556 return -error;
557 bind_intd->port = op.u.bind_interdomain.local_port;
558 mutex_enter(&devevent_lock);
559 KASSERT(devevent[bind_intd->port] == NULL);
560 devevent[bind_intd->port] = d;
561 mutex_exit(&devevent_lock);
562 xen_atomic_set_bit(&d->ci->ci_evtmask[0], bind_intd->port);
563 hypervisor_unmask_event(bind_intd->port);
564 break;
565 }
566 case IOCTL_EVTCHN_BIND_UNBOUND_PORT:
567 {
568 struct ioctl_evtchn_bind_unbound_port *bind_unbound = addr;
569 op.cmd = EVTCHNOP_alloc_unbound;
570 op.u.alloc_unbound.dom = DOMID_SELF;
571 op.u.alloc_unbound.remote_dom = bind_unbound->remote_domain;
572 if ((error = HYPERVISOR_event_channel_op(&op)))
573 return -error;
574 bind_unbound->port = op.u.alloc_unbound.port;
575 mutex_enter(&devevent_lock);
576 KASSERT(devevent[bind_unbound->port] == NULL);
577 devevent[bind_unbound->port] = d;
578 mutex_exit(&devevent_lock);
579 xen_atomic_set_bit(&d->ci->ci_evtmask[0], bind_unbound->port);
580 hypervisor_unmask_event(bind_unbound->port);
581 break;
582 }
583 case IOCTL_EVTCHN_UNBIND:
584 {
585 struct ioctl_evtchn_unbind *unbind = addr;
586
587 if (unbind->port >= NR_EVENT_CHANNELS)
588 return EINVAL;
589 mutex_enter(&devevent_lock);
590 if (devevent[unbind->port] != d) {
591 mutex_exit(&devevent_lock);
592 return ENOTCONN;
593 }
594 devevent[unbind->port] = NULL;
595 mutex_exit(&devevent_lock);
596 hypervisor_mask_event(unbind->port);
597 xen_atomic_clear_bit(&d->ci->ci_evtmask[0], unbind->port);
598 op.cmd = EVTCHNOP_close;
599 op.u.close.port = unbind->port;
600 if ((error = HYPERVISOR_event_channel_op(&op)))
601 return -error;
602 break;
603 }
604 case IOCTL_EVTCHN_NOTIFY:
605 {
606 struct ioctl_evtchn_notify *notify = addr;
607
608 if (notify->port >= NR_EVENT_CHANNELS)
609 return EINVAL;
610 mutex_enter(&devevent_lock);
611 if (devevent[notify->port] != d) {
612 mutex_exit(&devevent_lock);
613 return ENOTCONN;
614 }
615 hypervisor_notify_via_evtchn(notify->port);
616 mutex_exit(&devevent_lock);
617 break;
618 }
619 case FIONBIO:
620 break;
621 default:
622 return EINVAL;
623 }
624 return 0;
625}

627/*
628 * Support for poll() system call
629 *
630 * Return true if the specific operation will not block indefinitely.
631 */
632
633static int
634xenevt_fpoll(struct file *fp, int events)
635{
636 struct xenevt_d *d = fp->f_data;
637 int revents = events & (POLLOUT | POLLWRNORM); /* we can always write */
638
639 mutex_enter(&d->lock);
640 if (events & (POLLIN | POLLRDNORM)) {
641 if (d->ring_read != d->ring_write) {
642 revents |= events & (POLLIN | POLLRDNORM);
643 } else {
644 /* Record that someone is waiting */
645 selrecord(curlwp, &d->sel);
646 }
647 }
648 mutex_exit(&d->lock);
649 return (revents);
650}
