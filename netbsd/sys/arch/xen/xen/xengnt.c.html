52#define NR_GRANT_ENTRIES_PER_PAGE (PAGE_SIZE / sizeof(grant_entry_t))
53
54/* External tools reserve first few grant table entries. */
55#define NR_RESERVED_ENTRIES 8
56
57/* Current number of frames making up the grant table */
58int gnt_nr_grant_frames;
59/* Maximum number of frames that can make up the grant table */
60int gnt_max_grant_frames;
61
62/* table of free grant entries */
63grant_ref_t *gnt_entries;
64/* last free entry */
65int last_gnt_entry;
66/* empty entry in the list */
67#define XENGNT_NO_ENTRY 0xffffffff
68
69/* VM address of the grant table */
70grant_entry_t *grant_table;
71kmutex_t grant_lock;
72
73static grant_ref_t xengnt_get_entry(void);
74static void xengnt_free_entry(grant_ref_t);
75static int xengnt_more_entries(void);

77void
78xengnt_init(void)
79{
80 struct gnttab_query_size query;
81 int rc;
82 int nr_grant_entries;
83 int i;
84
85 query.dom = DOMID_SELF;
86 rc = HYPERVISOR_grant_table_op(GNTTABOP_query_size, &query, 1);
87 if ((rc < 0) || (query.status != GNTST_okay))
88 gnt_max_grant_frames = 4; /* Legacy max number of frames */
89 else
90 gnt_max_grant_frames = query.max_nr_frames;
91 gnt_nr_grant_frames = 0;
92
93 nr_grant_entries =
94 gnt_max_grant_frames * NR_GRANT_ENTRIES_PER_PAGE;
95
96 grant_table = (void *)uvm_km_alloc(kernel_map,
97 gnt_max_grant_frames * PAGE_SIZE, 0, UVM_KMF_VAONLY);
98 if (grant_table == NULL)
99 panic("xengnt_init() no VM space");
100 gnt_entries = malloc((nr_grant_entries + 1) * sizeof(grant_ref_t),
101 M_DEVBUF, M_NOWAIT);
102 if (gnt_entries == NULL)
103 panic("xengnt_init() no space for bitmask");
104 for (i = 0; i <= nr_grant_entries; i++)
105 gnt_entries[i] = XENGNT_NO_ENTRY;
106
107 mutex_init(&grant_lock, MUTEX_DEFAULT, IPL_VM);
108
109 xengnt_resume();
110
111}
112
113/*
114 * Resume grant table state
115 */
116bool
117xengnt_resume(void)
118{
119 int previous_nr_grant_frames = gnt_nr_grant_frames;
120
121 last_gnt_entry = 0;
122 gnt_nr_grant_frames = 0;
123
124 mutex_enter(&grant_lock);
125 while (gnt_nr_grant_frames < previous_nr_grant_frames) {
126 if (xengnt_more_entries() != 0)
127 panic("xengnt_resume: can't restore grant frames");
128 }
129 mutex_exit(&grant_lock);
130 return true;
131}
132
133/*
134 * Suspend grant table state
135 */
136bool
137xengnt_suspend(void) {
138
139 int i;
140
141 mutex_enter(&grant_lock);
142 KASSERT(gnt_entries[last_gnt_entry] == XENGNT_NO_ENTRY);
143
144 for (i = 0; i < last_gnt_entry; i++) {
145 /* invalidate all grant entries (necessary for resume) */
146 gnt_entries[i] = XENGNT_NO_ENTRY;
147 }
148
149 /* Remove virtual => machine mapping */
150 pmap_kremove((vaddr_t)grant_table, gnt_nr_grant_frames * PAGE_SIZE);
151 pmap_update(pmap_kernel());
152 mutex_exit(&grant_lock);
153 return true;
154}

157/*
158 * Add another page to the grant table
159 * Returns 0 on success, ENOMEM on failure
160 */
161static int
162xengnt_more_entries(void)
163{
164 gnttab_setup_table_t setup;
165 u_long *pages;
166 int nframes_new = gnt_nr_grant_frames + 1;
167 int i, start_gnt;
168 KASSERT(mutex_owned(&grant_lock));
169
170 if (gnt_nr_grant_frames == gnt_max_grant_frames)
171 return ENOMEM;
172
173 pages = malloc(nframes_new * sizeof(u_long), M_DEVBUF, M_NOWAIT);
174 if (pages == NULL)
175 return ENOMEM;
176
177 if (xen_feature(XENFEAT_auto_translated_physmap)) {
178 /*
179 * Note: Although we allocate space for the entire
180 * table, in this mode we only update one entry at a
181 * time.
182 */
183 struct vm_page *pg;
184 struct xen_add_to_physmap xmap;
185
186 pg = uvm_pagealloc(NULL, 0, NULL, UVM_PGA_USERESERVE|UVM_PGA_ZERO);
187 pages[gnt_nr_grant_frames] = atop(uvm_vm_page_to_phys(pg));
188
189 xmap.domid = DOMID_SELF;
190 xmap.space = XENMAPSPACE_grant_table;
191 xmap.idx = gnt_nr_grant_frames;
192 xmap.gpfn = pages[gnt_nr_grant_frames];
193
194 if (HYPERVISOR_memory_op(XENMEM_add_to_physmap, &xmap) < 0)
195 panic("%s: Unable to register HYPERVISOR_shared_info\n", __func__);
196
197 } else {
198 setup.dom = DOMID_SELF;
199 setup.nr_frames = nframes_new;
200 set_xen_guest_handle(setup.frame_list, pages);
201
202 /*
203 * setup the grant table, made of nframes_new frames
204 * and return the list of their virtual addresses
205 * in 'pages'
206 */
207 if (HYPERVISOR_grant_table_op(GNTTABOP_setup_table, &setup, 1) != 0)
208 panic("%s: setup table failed", __func__);
209 if (setup.status != GNTST_okay) {
210 aprint_error("%s: setup table returned %d\n",
211 __func__, setup.status);
212 free(pages, M_DEVBUF);
213 return ENOMEM;
214 }
215 }
216
217 DPRINTF(("xengnt_more_entries: map 0x%lx -> %p\n",
218 pages[gnt_nr_grant_frames],
219 (char *)grant_table + gnt_nr_grant_frames * PAGE_SIZE));
220
221 /*
222 * map between grant_table addresses and the machine addresses of
223 * the grant table frames
224 */
225 pmap_kenter_ma(((vaddr_t)grant_table) + gnt_nr_grant_frames * PAGE_SIZE,
226 ((paddr_t)pages[gnt_nr_grant_frames]) << PAGE_SHIFT,
227 VM_PROT_WRITE, 0);
228 pmap_update(pmap_kernel());
229
230 /*
231 * add the grant entries associated to the last grant table frame
232 * and mark them as free. Prevent using the first grants (from 0 to 8)
233 * since they are used by the tools.
234 */
235 start_gnt = (gnt_nr_grant_frames * NR_GRANT_ENTRIES_PER_PAGE) <
236 (NR_RESERVED_ENTRIES + 1) ?
237 (NR_RESERVED_ENTRIES + 1) :
238 (gnt_nr_grant_frames * NR_GRANT_ENTRIES_PER_PAGE);
239 for (i = start_gnt;
240 i < nframes_new * NR_GRANT_ENTRIES_PER_PAGE;
241 i++) {
242 KASSERT(gnt_entries[last_gnt_entry] == XENGNT_NO_ENTRY);
243 gnt_entries[last_gnt_entry] = i;
244 last_gnt_entry++;
245 }
246 gnt_nr_grant_frames = nframes_new;
247 free(pages, M_DEVBUF);
248 return 0;
249}

251/*
252 * Returns a reference to the first free entry in grant table
253 */
254static grant_ref_t
255xengnt_get_entry(void)
256{
257 grant_ref_t entry;
258 static struct timeval xengnt_nonmemtime;
259 static const struct timeval xengnt_nonmemintvl = {5,0};
260
261 if (last_gnt_entry == 0) {
262 if (xengnt_more_entries()) {
263 if (ratecheck(&xengnt_nonmemtime, &xengnt_nonmemintvl))
264 printf("xengnt_get_entry: out of grant "
265 "table entries\n");
266 return XENGNT_NO_ENTRY;
267 }
268 }
269 KASSERT(gnt_entries[last_gnt_entry] == XENGNT_NO_ENTRY);
270 last_gnt_entry--;
271 entry = gnt_entries[last_gnt_entry];
272 gnt_entries[last_gnt_entry] = XENGNT_NO_ENTRY;
273 KASSERT(entry != XENGNT_NO_ENTRY && entry > NR_RESERVED_ENTRIES);
274 KASSERT(last_gnt_entry >= 0);
275 KASSERT(last_gnt_entry <= gnt_max_grant_frames * NR_GRANT_ENTRIES_PER_PAGE);
276 return entry;
277}

279/*
280 * Mark the grant table entry as free
281 */
282static void
283xengnt_free_entry(grant_ref_t entry)
284{
285 mutex_enter(&grant_lock);
286 KASSERT(entry > NR_RESERVED_ENTRIES);
287 KASSERT(gnt_entries[last_gnt_entry] == XENGNT_NO_ENTRY);
288 KASSERT(last_gnt_entry >= 0);
289 KASSERT(last_gnt_entry <= gnt_max_grant_frames * NR_GRANT_ENTRIES_PER_PAGE);
290 gnt_entries[last_gnt_entry] = entry;
291 last_gnt_entry++;
292 mutex_exit(&grant_lock);
293}

295int
296xengnt_grant_access(domid_t dom, paddr_t ma, int ro, grant_ref_t *entryp)
297{
298 mutex_enter(&grant_lock);
299
300 *entryp = xengnt_get_entry();
301 if (__predict_false(*entryp == XENGNT_NO_ENTRY)) {
302 mutex_exit(&grant_lock);
303 return ENOMEM;
304 }
305
306 grant_table[*entryp].frame = ma >> PAGE_SHIFT;
307 grant_table[*entryp].domid = dom;
308 /*
309 * ensure that the above values reach global visibility
310 * before permitting frame's access (done when we set flags)
311 */
312 xen_rmb();
313 grant_table[*entryp].flags =
314 GTF_permit_access | (ro ? GTF_readonly : 0);
315 mutex_exit(&grant_lock);
316 return 0;
317}

319void
320xengnt_revoke_access(grant_ref_t entry)
321{
322 uint16_t flags, nflags;
323
324 nflags = grant_table[entry].flags;
325
326 do {
327 if ((flags = nflags) & (GTF_reading|GTF_writing))
328 panic("xengnt_revoke_access: still in use");
329 nflags = xen_atomic_cmpxchg16(&grant_table[entry].flags,
330 flags, 0);
331 } while (nflags != flags);
332 xengnt_free_entry(entry);
333}
334
335int
336xengnt_grant_transfer(domid_t dom, grant_ref_t *entryp)
337{
338 mutex_enter(&grant_lock);
339
340 *entryp = xengnt_get_entry();
341 if (__predict_false(*entryp == XENGNT_NO_ENTRY)) {
342 mutex_exit(&grant_lock);
343 return ENOMEM;
344 }
345
346 grant_table[*entryp].frame = 0;
347 grant_table[*entryp].domid = dom;
348 /*
349 * ensure that the above values reach global visibility
350 * before permitting frame's transfer (done when we set flags)
351 */
352 xen_rmb();
353 grant_table[*entryp].flags = GTF_accept_transfer;
354 mutex_exit(&grant_lock);
355 return 0;
356}
357
358paddr_t
359xengnt_revoke_transfer(grant_ref_t entry)
360{
361 paddr_t page;
362 uint16_t flags;
363
364 /* if the transfer has not started, free the entry and return 0 */
365 while (!((flags = grant_table[entry].flags) & GTF_transfer_committed)) {
366 if (xen_atomic_cmpxchg16(&grant_table[entry].flags,
367 flags, 0) == flags ) {
368 xengnt_free_entry(entry);
369 return 0;
370 }
371 HYPERVISOR_yield();
372 }
373
374 /* If transfer in progress, wait for completion */
375 while (!((flags = grant_table[entry].flags) & GTF_transfer_completed))
376 HYPERVISOR_yield();
377
378 /* Read the frame number /after/ reading completion status. */
379 __insn_barrier();
380 page = grant_table[entry].frame;
381 if (page == 0)
382 printf("xengnt_revoke_transfer: guest sent pa 0\n");
383
384 xengnt_free_entry(entry);
385 return page;
386}

388int
389xengnt_status(grant_ref_t entry)
390{
391 return (grant_table[entry].flags & (GTF_reading|GTF_writing));
392}
