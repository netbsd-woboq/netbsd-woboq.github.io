200void
201xvifattach(int n)
202{
203 int i;
204 struct pglist mlist;
205 struct vm_page *pg;
206
207 XENPRINTF(("xennetback_init\n"));
208
209 /*
210 * steal some non-managed pages to the VM system, to replace
211 * mbuf cluster or xmit_pages_pool pages given to foreign domains.
212 */
213 if (uvm_pglistalloc(PAGE_SIZE * NB_XMIT_PAGES_BATCH, 0, 0xffffffff,
214 0, 0, &mlist, NB_XMIT_PAGES_BATCH, 0) != 0)
215 panic("xennetback_init: uvm_pglistalloc");
216 for (i = 0, pg = mlist.tqh_first; pg != NULL;
217 pg = pg->pageq.queue.tqe_next, i++)
218 mcl_pages[i] = xpmap_ptom(VM_PAGE_TO_PHYS(pg)) >> PAGE_SHIFT;
219 if (i != NB_XMIT_PAGES_BATCH)
220 panic("xennetback_init: %d mcl pages", i);
221 mcl_pages_alloc = NB_XMIT_PAGES_BATCH - 1;
222
223 /* initialise pools */
224 pool_init(&xni_pkt_pool, sizeof(struct xni_pkt), 0, 0, 0,
225 "xnbpkt", NULL, IPL_VM);
226 xmit_pages_cache = pool_cache_init(PAGE_SIZE, 0, 0, 0, "xnbxm", NULL,
227 IPL_VM, NULL, NULL, NULL);
228
229 SLIST_INIT(&xnetback_instances);
230 mutex_init(&xnetback_lock, MUTEX_DEFAULT, IPL_NONE);
231
232 xenbus_backend_register(&xvif_backend_driver);
233}

235static int
236xennetback_xenbus_create(struct xenbus_device *xbusd)
237{
238 struct xnetback_instance *xneti;
239 long domid, handle;
240 struct ifnet *ifp;
241 extern int ifqmaxlen; /* XXX */
242 char *val, *e, *p;
243 int i, err;
244 struct xenbus_transaction *xbt;
245
246 if ((err = xenbus_read_ul(NULL, xbusd->xbusd_path,
247 "frontend-id", &domid, 10)) != 0) {
248 aprint_error("xvif: can't read %s/frontend-id: %d\n",
249 xbusd->xbusd_path, err);
250 return err;
251 }
252 if ((err = xenbus_read_ul(NULL, xbusd->xbusd_path,
253 "handle", &handle, 10)) != 0) {
254 aprint_error("xvif: can't read %s/handle: %d\n",
255 xbusd->xbusd_path, err);
256 return err;
257 }
258
259 if (xnetif_lookup(domid, handle)) {
260 return EEXIST;
261 }
262 xneti = kmem_zalloc(sizeof(*xneti), KM_SLEEP);
263 xneti->xni_domid = domid;
264 xneti->xni_handle = handle;
265 xneti->xni_status = DISCONNECTED;
266
267 xbusd->xbusd_u.b.b_cookie = xneti;
268 xbusd->xbusd_u.b.b_detach = xennetback_xenbus_destroy;
269 xneti->xni_xbusd = xbusd;
270
271 ifp = &xneti->xni_if;
272 ifp->if_softc = xneti;
273 snprintf(ifp->if_xname, IFNAMSIZ, "xvif%di%d",
274 (int)domid, (int)handle);
275
276 /* read mac address */
277 if ((err = xenbus_read(NULL, xbusd->xbusd_path, "mac", NULL, &val))) {
278 aprint_error_ifnet(ifp, "can't read %s/mac: %d\n",
279 xbusd->xbusd_path, err);
280 goto fail;
281 }
282 for (i = 0, p = val; i < 6; i++) {
283 xneti->xni_enaddr[i] = strtoul(p, &e, 16);
284 if ((e[0] == '\0' && i != 5) && e[0] != ':') {
285 aprint_error_ifnet(ifp,
286 "%s is not a valid mac address\n", val);
287 free(val, M_DEVBUF);
288 err = EINVAL;
289 goto fail;
290 }
291 p = &e[1];
292 }
293 free(val, M_DEVBUF);
294
295 /* we can't use the same MAC addr as our guest */
296 xneti->xni_enaddr[3]++;
297 /* create pseudo-interface */
298 aprint_verbose_ifnet(ifp, "Ethernet address %s\n",
299 ether_sprintf(xneti->xni_enaddr));
300 xneti->xni_ec.ec_capabilities |= ETHERCAP_VLAN_MTU;
301 ifp->if_flags = IFF_BROADCAST | IFF_SIMPLEX | IFF_MULTICAST;
302 ifp->if_snd.ifq_maxlen =
303 uimax(ifqmaxlen, NET_TX_RING_SIZE * 2);
304 ifp->if_capabilities = IFCAP_CSUM_TCPv4_Tx | IFCAP_CSUM_UDPv4_Tx;
305 ifp->if_ioctl = xennetback_ifioctl;
306 ifp->if_start = xennetback_ifstart;
307 ifp->if_watchdog = xennetback_ifwatchdog;
308 ifp->if_init = xennetback_ifinit;
309 ifp->if_stop = xennetback_ifstop;
310 ifp->if_timer = 0;
311 IFQ_SET_READY(&ifp->if_snd);
312 if_attach(ifp);
313 ether_ifattach(&xneti->xni_if, xneti->xni_enaddr);
314
315 mutex_enter(&xnetback_lock);
316 SLIST_INSERT_HEAD(&xnetback_instances, xneti, next);
317 mutex_exit(&xnetback_lock);
318
319 xbusd->xbusd_otherend_changed = xennetback_frontend_changed;
320
321 do {
322 xbt = xenbus_transaction_start();
323 if (xbt == NULL) {
324 aprint_error_ifnet(ifp,
325 "%s: can't start transaction\n",
326 xbusd->xbusd_path);
327 goto fail;
328 }
329 err = xenbus_printf(xbt, xbusd->xbusd_path,
330 "vifname", "%s", ifp->if_xname);
331 if (err) {
332 aprint_error_ifnet(ifp,
333 "failed to write %s/vifname: %d\n",
334 xbusd->xbusd_path, err);
335 goto abort_xbt;
336 }
337 err = xenbus_printf(xbt, xbusd->xbusd_path,
338 "feature-rx-copy", "%d", 1);
339 if (err) {
340 aprint_error_ifnet(ifp,
341 "failed to write %s/feature-rx-copy: %d\n",
342 xbusd->xbusd_path, err);
343 goto abort_xbt;
344 }
345 err = xenbus_printf(xbt, xbusd->xbusd_path,
346 "feature-rx-flip", "%d", 1);
347 if (err) {
348 aprint_error_ifnet(ifp,
349 "failed to write %s/feature-rx-flip: %d\n",
350 xbusd->xbusd_path, err);
351 goto abort_xbt;
352 }
353 } while ((err = xenbus_transaction_end(xbt, 0)) == EAGAIN);
354 if (err) {
355 aprint_error_ifnet(ifp,
356 "%s: can't end transaction: %d\n",
357 xbusd->xbusd_path, err);
358 }
359
360 err = xenbus_switch_state(xbusd, NULL, XenbusStateInitWait);
361 if (err) {
362 aprint_error_ifnet(ifp,
363 "failed to switch state on %s: %d\n",
364 xbusd->xbusd_path, err);
365 goto fail;
366 }
367 return 0;
368
369abort_xbt:
370 xenbus_transaction_end(xbt, 1);
371fail:
372 kmem_free(xneti, sizeof(*xneti));
373 return err;
374}

376int
377xennetback_xenbus_destroy(void *arg)
378{
379 struct xnetback_instance *xneti = arg;
380 struct gnttab_unmap_grant_ref op;
381 int err;
382
383 aprint_verbose_ifnet(&xneti->xni_if, "disconnecting\n");
384
385 if (xneti->xni_ih != NULL) {
386 hypervisor_mask_event(xneti->xni_evtchn);
387 xen_intr_disestablish(xneti->xni_ih);
388 xneti->xni_ih = NULL;
389
390 if (xneti->xni_softintr) {
391 softint_disestablish(xneti->xni_softintr);
392 xneti->xni_softintr = NULL;
393 }
394 }
395
396 mutex_enter(&xnetback_lock);
397 SLIST_REMOVE(&xnetback_instances,
398 xneti, xnetback_instance, next);
399 mutex_exit(&xnetback_lock);
400
401 ether_ifdetach(&xneti->xni_if);
402 if_detach(&xneti->xni_if);
403
404 if (xneti->xni_txring.sring) {
405 op.host_addr = xneti->xni_tx_ring_va;
406 op.handle = xneti->xni_tx_ring_handle;
407 op.dev_bus_addr = 0;
408 err = HYPERVISOR_grant_table_op(GNTTABOP_unmap_grant_ref,
409 &op, 1);
410 if (err)
411 aprint_error_ifnet(&xneti->xni_if,
412 "unmap_grant_ref failed: %d\n", err);
413 }
414 if (xneti->xni_rxring.sring) {
415 op.host_addr = xneti->xni_rx_ring_va;
416 op.handle = xneti->xni_rx_ring_handle;
417 op.dev_bus_addr = 0;
418 err = HYPERVISOR_grant_table_op(GNTTABOP_unmap_grant_ref,
419 &op, 1);
420 if (err)
421 aprint_error_ifnet(&xneti->xni_if,
422 "unmap_grant_ref failed: %d\n", err);
423 }
424 if (xneti->xni_tx_ring_va != 0) {
425 uvm_km_free(kernel_map, xneti->xni_tx_ring_va,
426 PAGE_SIZE, UVM_KMF_VAONLY);
427 xneti->xni_tx_ring_va = 0;
428 }
429 if (xneti->xni_rx_ring_va != 0) {
430 uvm_km_free(kernel_map, xneti->xni_rx_ring_va,
431 PAGE_SIZE, UVM_KMF_VAONLY);
432 xneti->xni_rx_ring_va = 0;
433 }
434 kmem_free(xneti, sizeof(*xneti));
435 return 0;
436}

438static int
439xennetback_connect(struct xnetback_instance *xneti)
440{
441 int err;
442 netif_tx_sring_t *tx_ring;
443 netif_rx_sring_t *rx_ring;
444 struct gnttab_map_grant_ref op;
445 struct gnttab_unmap_grant_ref uop;
446 evtchn_op_t evop;
447 u_long tx_ring_ref, rx_ring_ref;
448 u_long revtchn, rx_copy;
449 struct xenbus_device *xbusd = xneti->xni_xbusd;
450
451 /* read communication information */
452 err = xenbus_read_ul(NULL, xbusd->xbusd_otherend,
453 "tx-ring-ref", &tx_ring_ref, 10);
454 if (err) {
455 xenbus_dev_fatal(xbusd, err, "reading %s/tx-ring-ref",
456 xbusd->xbusd_otherend);
457 return -1;
458 }
459 err = xenbus_read_ul(NULL, xbusd->xbusd_otherend,
460 "rx-ring-ref", &rx_ring_ref, 10);
461 if (err) {
462 xenbus_dev_fatal(xbusd, err, "reading %s/rx-ring-ref",
463 xbusd->xbusd_otherend);
464 return -1;
465 }
466 err = xenbus_read_ul(NULL, xbusd->xbusd_otherend,
467 "event-channel", &revtchn, 10);
468 if (err) {
469 xenbus_dev_fatal(xbusd, err, "reading %s/event-channel",
470 xbusd->xbusd_otherend);
471 return -1;
472 }
473 err = xenbus_read_ul(NULL, xbusd->xbusd_otherend,
474 "request-rx-copy", &rx_copy, 10);
475 if (err == ENOENT)
476 rx_copy = 0;
477 else if (err) {
478 xenbus_dev_fatal(xbusd, err, "reading %s/request-rx-copy",
479 xbusd->xbusd_otherend);
480 return -1;
481 }
482
483 if (rx_copy)
484 xneti->xni_softintr = softint_establish(SOFTINT_NET,
485 xennetback_ifsoftstart_copy, xneti);
486 else
487 xneti->xni_softintr = softint_establish(SOFTINT_NET,
488 xennetback_ifsoftstart_transfer, xneti);
489
490 if (xneti->xni_softintr == NULL) {
491 err = ENOMEM;
492 xenbus_dev_fatal(xbusd, ENOMEM,
493 "can't allocate softint", xbusd->xbusd_otherend);
494 return -1;
495 }
496
497 /* allocate VA space and map rings */
498 xneti->xni_tx_ring_va = uvm_km_alloc(kernel_map, PAGE_SIZE, 0,
499 UVM_KMF_VAONLY);
500 if (xneti->xni_tx_ring_va == 0) {
501 xenbus_dev_fatal(xbusd, ENOMEM,
502 "can't get VA for TX ring", xbusd->xbusd_otherend);
503 goto err1;
504 }
505 tx_ring = (void *)xneti->xni_tx_ring_va;
506
507 xneti->xni_rx_ring_va = uvm_km_alloc(kernel_map, PAGE_SIZE, 0,
508 UVM_KMF_VAONLY);
509 if (xneti->xni_rx_ring_va == 0) {
510 xenbus_dev_fatal(xbusd, ENOMEM,
511 "can't get VA for RX ring", xbusd->xbusd_otherend);
512 goto err1;
513 }
514 rx_ring = (void *)xneti->xni_rx_ring_va;
515
516 op.host_addr = xneti->xni_tx_ring_va;
517 op.flags = GNTMAP_host_map;
518 op.ref = tx_ring_ref;
519 op.dom = xneti->xni_domid;
520 err = HYPERVISOR_grant_table_op(GNTTABOP_map_grant_ref, &op, 1);
521 if (err || op.status) {
522 aprint_error_ifnet(&xneti->xni_if,
523 "can't map TX grant ref: err %d status %d\n",
524 err, op.status);
525 goto err2;
526 }
527 xneti->xni_tx_ring_handle = op.handle;
528 BACK_RING_INIT(&xneti->xni_txring, tx_ring, PAGE_SIZE);
529
530 op.host_addr = xneti->xni_rx_ring_va;
531 op.flags = GNTMAP_host_map;
532 op.ref = rx_ring_ref;
533 op.dom = xneti->xni_domid;
534 err = HYPERVISOR_grant_table_op(GNTTABOP_map_grant_ref, &op, 1);
535 if (err || op.status) {
536 aprint_error_ifnet(&xneti->xni_if,
537 "can't map RX grant ref: err %d status %d\n",
538 err, op.status);
539 goto err2;
540 }
541 xneti->xni_rx_ring_handle = op.handle;
542 BACK_RING_INIT(&xneti->xni_rxring, rx_ring, PAGE_SIZE);
543
544 evop.cmd = EVTCHNOP_bind_interdomain;
545 evop.u.bind_interdomain.remote_dom = xneti->xni_domid;
546 evop.u.bind_interdomain.remote_port = revtchn;
547 err = HYPERVISOR_event_channel_op(&evop);
548 if (err) {
549 aprint_error_ifnet(&xneti->xni_if,
550 "can't get event channel: %d\n", err);
551 goto err2;
552 }
553 xneti->xni_evtchn = evop.u.bind_interdomain.local_port;
554 xen_wmb();
555 xneti->xni_status = CONNECTED;
556 xen_wmb();
557
558 xneti->xni_ih = xen_intr_establish_xname(-1, &xen_pic, xneti->xni_evtchn,
559 IST_LEVEL, IPL_NET, xennetback_evthandler, xneti, false,
560 xneti->xni_if.if_xname);
561 KASSERT(xneti->xni_ih != NULL);
562 xennetback_ifinit(&xneti->xni_if);
563 hypervisor_unmask_event(xneti->xni_evtchn);
564 hypervisor_notify_via_evtchn(xneti->xni_evtchn);
565 return 0;
566
567err2:
568 /* unmap rings */
569 if (xneti->xni_tx_ring_handle != 0) {
570 uop.host_addr = xneti->xni_tx_ring_va;
571 uop.handle = xneti->xni_tx_ring_handle;
572 uop.dev_bus_addr = 0;
573 err = HYPERVISOR_grant_table_op(GNTTABOP_unmap_grant_ref,
574 &uop, 1);
575 if (err)
576 aprint_error_ifnet(&xneti->xni_if,
577 "unmap_grant_ref failed: %d\n", err);
578 }
579
580 if (xneti->xni_rx_ring_handle != 0) {
581 uop.host_addr = xneti->xni_rx_ring_va;
582 uop.handle = xneti->xni_rx_ring_handle;
583 uop.dev_bus_addr = 0;
584 err = HYPERVISOR_grant_table_op(GNTTABOP_unmap_grant_ref,
585 &uop, 1);
586 if (err)
587 aprint_error_ifnet(&xneti->xni_if,
588 "unmap_grant_ref failed: %d\n", err);
589 }
590
591err1:
592 /* free rings VA space */
593 if (xneti->xni_rx_ring_va != 0)
594 uvm_km_free(kernel_map, xneti->xni_rx_ring_va,
595 PAGE_SIZE, UVM_KMF_VAONLY);
596
597 if (xneti->xni_tx_ring_va != 0)
598 uvm_km_free(kernel_map, xneti->xni_tx_ring_va,
599 PAGE_SIZE, UVM_KMF_VAONLY);
600
601 softint_disestablish(xneti->xni_softintr);
602 return -1;
603
604}

606static void
607xennetback_frontend_changed(void *arg, XenbusState new_state)
608{
609 struct xnetback_instance *xneti = arg;
610 struct xenbus_device *xbusd = xneti->xni_xbusd;
611
612 XENPRINTF(("%s: new state %d\n", xneti->xni_if.if_xname, new_state));
613 switch(new_state) {
614 case XenbusStateInitialising:
615 case XenbusStateInitialised:
616 break;
617
618 case XenbusStateConnected:
619 if (xneti->xni_status == CONNECTED)
620 break;
621 if (xennetback_connect(xneti) == 0)
622 xenbus_switch_state(xbusd, NULL, XenbusStateConnected);
623 break;
624
625 case XenbusStateClosing:
626 xneti->xni_status = DISCONNECTING;
627 xneti->xni_if.if_flags &= ~(IFF_RUNNING | IFF_OACTIVE);
628 xneti->xni_if.if_timer = 0;
629 xenbus_switch_state(xbusd, NULL, XenbusStateClosing);
630 break;
631
632 case XenbusStateClosed:
633 /* otherend_changed() should handle it for us */
634 panic("xennetback_frontend_changed: closed\n");
635 case XenbusStateUnknown:
636 case XenbusStateInitWait:
637 default:
638 aprint_error("%s: invalid frontend state %d\n",
639 xneti->xni_if.if_xname, new_state);
640 break;
641 }
642 return;
643
644}

665/* get a page to replace a mbuf cluster page given to a domain */
666static int
667xennetback_get_mcl_page(paddr_t *map)
668{
669 if (mcl_pages_alloc < 0) {
670 /*
671 * we exhausted our allocation. We can't allocate new ones yet
672 * because the current pages may not have been loaned to
673 * the remote domain yet. We have to let the caller do this.
674 */
675 return -1;
676 }
677
678 *map = ((paddr_t)mcl_pages[mcl_pages_alloc]) << PAGE_SHIFT;
679 mcl_pages_alloc--;
680 return 0;
681}

727static inline const char *

729{
730 if (__predict_false(txreq->size < ETHER_HDR_LEN))

732

740 return "too big";

742 return NULL;

745static int
746xennetback_evthandler(void *arg)
747{
748 struct xnetback_instance *xneti = arg;
749 struct ifnet *ifp = &xneti->xni_if;
750 netif_tx_request_t txreq;
751 struct xni_pkt *pkt;
752 vaddr_t pkt_va;
753 struct mbuf *m;
754 int receive_pending, err;
755 RING_IDX req_cons;
756
757 XENPRINTF(("xennetback_evthandler "));
758 req_cons = xneti->xni_txring.req_cons;
759 xen_rmb();
760 while (1) {
761 xen_rmb(); /* be sure to read the request before updating */
762 xneti->xni_txring.req_cons = req_cons;
763 xen_wmb();
764 RING_FINAL_CHECK_FOR_REQUESTS(&xneti->xni_txring,
765 receive_pending);
766 if (receive_pending == 0)
767 break;
768 RING_COPY_REQUEST(&xneti->xni_txring, req_cons, &txreq);
769 xen_rmb();
770 XENPRINTF(("%s pkt size %d\n", xneti->xni_if.if_xname,
771 txreq.size));
772 req_cons++;
773 if (__predict_false((ifp->if_flags & (IFF_UP | IFF_RUNNING)) !=
774 (IFF_UP | IFF_RUNNING))) {
775 /* interface not up, drop */
776 xennetback_tx_response(xneti, txreq.id,
777 NETIF_RSP_DROPPED);
778 continue;
779 }
780
781 /*
782 * Do some sanity checks, and map the packet's page.
783 */
784 const char *msg = xennetback_tx_check_packet(&txreq,
785 xneti->xni_ec.ec_capenable & ETHERCAP_VLAN_MTU);
786 if (msg) {
787 printf("%s: packet with size %d is %s\n",
788 ifp->if_xname, txreq.size, msg);
789 xennetback_tx_response(xneti, txreq.id,
790 NETIF_RSP_ERROR);
791 ifp->if_ierrors++;
792 continue;
793 }
794
795 /* get a mbuf for this packet */
796 MGETHDR(m, M_DONTWAIT, MT_DATA);
797 if (__predict_false(m == NULL)) {
798 static struct timeval lasttime;
799 if (ratecheck(&lasttime, &xni_pool_errintvl))
800 printf("%s: mbuf alloc failed\n",
801 ifp->if_xname);
802 xennetback_tx_response(xneti, txreq.id,
803 NETIF_RSP_DROPPED);
804 ifp->if_ierrors++;
805 continue;
806 }
807
808 XENPRINTF(("%s pkt offset %d size %d id %d req_cons %d\n",
809 xneti->xni_if.if_xname, txreq.offset,
810 txreq.size, txreq.id, MASK_NETIF_TX_IDX(req_cons)));
811
812 pkt = pool_get(&xni_pkt_pool, PR_NOWAIT);
813 if (__predict_false(pkt == NULL)) {
814 static struct timeval lasttime;
815 if (ratecheck(&lasttime, &xni_pool_errintvl))
816 printf("%s: xnbpkt alloc failed\n",
817 ifp->if_xname);
818 xennetback_tx_response(xneti, txreq.id,
819 NETIF_RSP_DROPPED);
820 ifp->if_ierrors++;
821 m_freem(m);
822 continue;
823 }
824 err = xen_shm_map(1, xneti->xni_domid, &txreq.gref, &pkt_va,
825 &pkt->pkt_handle, XSHM_RO);
826 if (__predict_false(err == ENOMEM)) {
827 xennetback_tx_response(xneti, txreq.id,
828 NETIF_RSP_DROPPED);
829 ifp->if_ierrors++;
830 pool_put(&xni_pkt_pool, pkt);
831 m_freem(m);
832 continue;
833 }
834
835 if (__predict_false(err)) {
836 printf("%s: mapping foreing page failed: %d\n",
837 xneti->xni_if.if_xname, err);
838 xennetback_tx_response(xneti, txreq.id,
839 NETIF_RSP_ERROR);
840 ifp->if_ierrors++;
841 pool_put(&xni_pkt_pool, pkt);
842 m_freem(m);
843 continue;
844 }
845
846 if ((ifp->if_flags & IFF_PROMISC) == 0) {
847 struct ether_header *eh =
848 (void*)(pkt_va + txreq.offset);
849 if (ETHER_IS_MULTICAST(eh->ether_dhost) == 0 &&
850 memcmp(CLLADDR(ifp->if_sadl), eh->ether_dhost,
851 ETHER_ADDR_LEN) != 0) {
852 xni_pkt_unmap(pkt, pkt_va);
853 m_freem(m);
854 xennetback_tx_response(xneti, txreq.id,
855 NETIF_RSP_OKAY);
856 continue; /* packet is not for us */
857 }
858 }
859
860#ifdef notyet
861 /*
862 * A lot of work is needed in the tcp stack to handle read-only
863 * ext storage so always copy for now.
864 */
865 if (((req_cons + 1) & (NET_TX_RING_SIZE - 1)) ==
866 (xneti->xni_txring.rsp_prod_pvt & (NET_TX_RING_SIZE - 1)))
867#else
868 if (1)
869#endif /* notyet */
870 {
871 /*
872 * This is the last TX buffer. Copy the data and
873 * ack it. Delaying it until the mbuf is
874 * freed will stall transmit.
875 */
876 m->m_len = uimin(MHLEN, txreq.size);
877 m->m_pkthdr.len = 0;
878 m_copyback(m, 0, txreq.size,
879 (void *)(pkt_va + txreq.offset));
880 xni_pkt_unmap(pkt, pkt_va);
881 if (m->m_pkthdr.len < txreq.size) {
882 ifp->if_ierrors++;
883 m_freem(m);
884 xennetback_tx_response(xneti, txreq.id,
885 NETIF_RSP_DROPPED);
886 continue;
887 }
888 xennetback_tx_response(xneti, txreq.id,
889 NETIF_RSP_OKAY);
890 } else {
891
892 pkt->pkt_id = txreq.id;
893 pkt->pkt_xneti = xneti;
894
895 MEXTADD(m, pkt_va + txreq.offset,
896 txreq.size, M_DEVBUF, xennetback_tx_free, pkt);
897 m->m_pkthdr.len = m->m_len = txreq.size;
898 m->m_flags |= M_EXT_ROMAP;
899 }
900 if ((txreq.flags & NETTXF_csum_blank) != 0) {
901 xennet_checksum_fill(&m);
902 if (m == NULL) {
903 ifp->if_ierrors++;
904 continue;
905 }
906 }
907 m_set_rcvif(m, ifp);
908
909 if_percpuq_enqueue(ifp->if_percpuq, m);
910 }
911 xen_rmb(); /* be sure to read the request before updating pointer */
912 xneti->xni_txring.req_cons = req_cons;
913 xen_wmb();
914 /* check to see if we can transmit more packets */
915 softint_schedule(xneti->xni_softintr);
916
917 return 1;
918}

968static void
969xennetback_ifsoftstart_transfer(void *arg)
970{
971 struct xnetback_instance *xneti = arg;
972 struct ifnet *ifp = &xneti->xni_if;
973 struct mbuf *m;
974 vaddr_t xmit_va;
975 paddr_t xmit_pa;
976 paddr_t xmit_ma;
977 paddr_t newp_ma = 0; /* XXX gcc */
978 int i, j, nppitems;
979 mmu_update_t *mmup;
980 multicall_entry_t *mclp;
981 netif_rx_response_t *rxresp;
982 netif_rx_request_t rxreq;
983 RING_IDX req_prod, resp_prod;
984 int do_event = 0;
985 gnttab_transfer_t *gop;
986 int id, offset;
987
988 XENPRINTF(("xennetback_ifsoftstart_transfer "));
989 int s = splnet();
990 if (__predict_false(
991 (ifp->if_flags & (IFF_RUNNING|IFF_OACTIVE)) != IFF_RUNNING)) {
992 splx(s);
993 return;
994 }
995
996 while (!IFQ_IS_EMPTY(&ifp->if_snd)) {
997 XENPRINTF(("pkt\n"));
998 req_prod = xneti->xni_rxring.sring->req_prod;
999 resp_prod = xneti->xni_rxring.rsp_prod_pvt;
1000 xen_rmb();
1001
1002 mmup = xstart_mmu;
1003 mclp = xstart_mcl;
1004 gop = xstart_gop_transfer;
1005 for (nppitems = 0, i = 0; !IFQ_IS_EMPTY(&ifp->if_snd);) {
1006 XENPRINTF(("have a packet\n"));
1007 IFQ_POLL(&ifp->if_snd, m);
1008 if (__predict_false(m == NULL))
1009 panic("xennetback_ifstart: IFQ_POLL");
1010 if (__predict_false(
1011 req_prod == xneti->xni_rxring.req_cons ||
1012 xneti->xni_rxring.req_cons - resp_prod ==
1013 NET_RX_RING_SIZE)) {
1014 /* out of ring space */
1015 XENPRINTF(("xennetback_ifstart: ring full "
1016 "req_prod 0x%x req_cons 0x%x resp_prod "
1017 "0x%x\n",
1018 req_prod, xneti->xni_rxring.req_cons,
1019 resp_prod));
1020 ifp->if_timer = 1;
1021 break;
1022 }
1023 if (__predict_false(i == NB_XMIT_PAGES_BATCH))
1024 break; /* we filled the array */
1025 if (__predict_false(
1026 xennetback_get_mcl_page(&newp_ma) != 0))
1027 break; /* out of memory */
1028 if ((m->m_flags & M_EXT_CLUSTER) != 0 &&
1029 !M_READONLY(m) && MCLBYTES == PAGE_SIZE) {
1030 /* we can give this page away */
1031 xmit_pa = m->m_ext.ext_paddr;
1032 xmit_ma = xpmap_ptom(xmit_pa);
1033 xmit_va = (vaddr_t)m->m_ext.ext_buf;
1034 KASSERT(xmit_pa != M_PADDR_INVALID);
1035 KASSERT((xmit_va & PAGE_MASK) == 0);
1036 offset = m->m_data - m->m_ext.ext_buf;
1037 } else {
1038 /* we have to copy the packet */
1039 xmit_va = (vaddr_t)pool_cache_get_paddr(
1040 xmit_pages_cache, PR_NOWAIT, &xmit_pa);
1041 if (__predict_false(xmit_va == 0))
1042 break; /* out of memory */
1043
1044 KASSERT(xmit_pa != POOL_PADDR_INVALID);
1045 xmit_ma = xpmap_ptom(xmit_pa);
1046 XENPRINTF(("xennetback_get_xmit_page: got va "
1047 "0x%x ma 0x%x\n", (u_int)xmit_va,
1048 (u_int)xmit_ma));
1049 m_copydata(m, 0, m->m_pkthdr.len,
1050 (char *)xmit_va + LINUX_REQUESTED_OFFSET);
1051 offset = LINUX_REQUESTED_OFFSET;
1052 pages_pool_free[nppitems].va = xmit_va;
1053 pages_pool_free[nppitems].pa = xmit_pa;
1054 nppitems++;
1055 }
1056 /* start filling ring */
1057 RING_COPY_REQUEST(&xneti->xni_rxring,
1058 xneti->xni_rxring.req_cons, &rxreq);
1059 gop->ref = rxreq.gref;
1060 id = rxreq.id;
1061 xen_rmb();
1062 xneti->xni_rxring.req_cons++;
1063 rxresp = RING_GET_RESPONSE(&xneti->xni_rxring,
1064 resp_prod);
1065 rxresp->id = id;
1066 rxresp->offset = offset;
1067 rxresp->status = m->m_pkthdr.len;
1068 if ((m->m_pkthdr.csum_flags &
1069 (M_CSUM_TCPv4 | M_CSUM_UDPv4)) != 0) {
1070 rxresp->flags = NETRXF_csum_blank;
1071 } else {
1072 rxresp->flags = 0;
1073 }
1074 /*
1075 * transfers the page containing the packet to the
1076 * remote domain, and map newp in place.
1077 */
1078 xpmap_ptom_map(xmit_pa, newp_ma);
1079 MULTI_update_va_mapping(mclp, xmit_va,
1080 newp_ma | PTE_P | PTE_W | PTE_A | PTE_D | xpmap_pg_nx, 0);
1081 mclp++;
1082 gop->mfn = xmit_ma >> PAGE_SHIFT;
1083 gop->domid = xneti->xni_domid;
1084 gop++;
1085
1086 mmup->ptr = newp_ma | MMU_MACHPHYS_UPDATE;
1087 mmup->val = xmit_pa >> PAGE_SHIFT;
1088 mmup++;
1089
1090 /* done with this packet */
1091 IFQ_DEQUEUE(&ifp->if_snd, m);
1092 mbufs_sent[i] = m;
1093 resp_prod++;
1094 i++; /* this packet has been queued */
1095 ifp->if_opackets++;
1096 bpf_mtap(ifp, m, BPF_D_OUT);
1097 }
1098 if (i != 0) {
1099 /*
1100 * We may have allocated buffers which have entries
1101 * outstanding in the page update queue -- make sure
1102 * we flush those first!
1103 */
1104 int svm = splvm();
1105 xpq_flush_queue();
1106 splx(svm);
1107 mclp[-1].args[MULTI_UVMFLAGS_INDEX] =
1108 UVMF_TLB_FLUSH|UVMF_ALL;
1109 mclp->op = __HYPERVISOR_mmu_update;
1110 mclp->args[0] = (unsigned long)xstart_mmu;
1111 mclp->args[1] = i;
1112 mclp->args[2] = 0;
1113 mclp->args[3] = DOMID_SELF;
1114 mclp++;
1115 /* update the MMU */
1116 if (HYPERVISOR_multicall(xstart_mcl, i + 1) != 0) {
1117 panic("%s: HYPERVISOR_multicall failed",
1118 ifp->if_xname);
1119 }
1120 for (j = 0; j < i + 1; j++) {
1121 if (xstart_mcl[j].result != 0) {
1122 printf("%s: xstart_mcl[%d] "
1123 "failed (%lu)\n", ifp->if_xname,
1124 j, xstart_mcl[j].result);
1125 printf("%s: req_prod %u req_cons "
1126 "%u rsp_prod %u rsp_prod_pvt %u "
1127 "i %u\n",
1128 ifp->if_xname,
1129 xneti->xni_rxring.sring->req_prod,
1130 xneti->xni_rxring.req_cons,
1131 xneti->xni_rxring.sring->rsp_prod,
1132 xneti->xni_rxring.rsp_prod_pvt,
1133 i);
1134 }
1135 }
1136 if (HYPERVISOR_grant_table_op(GNTTABOP_transfer,
1137 xstart_gop_transfer, i) != 0) {
1138 panic("%s: GNTTABOP_transfer failed",
1139 ifp->if_xname);
1140 }
1141
1142 for (j = 0; j < i; j++) {
1143 if (xstart_gop_transfer[j].status != GNTST_okay) {
1144 printf("%s GNTTABOP_transfer[%d] %d\n",
1145 ifp->if_xname,
1146 j, xstart_gop_transfer[j].status);
1147 printf("%s: req_prod %u req_cons "
1148 "%u rsp_prod %u rsp_prod_pvt %u "
1149 "i %d\n",
1150 ifp->if_xname,
1151 xneti->xni_rxring.sring->req_prod,
1152 xneti->xni_rxring.req_cons,
1153 xneti->xni_rxring.sring->rsp_prod,
1154 xneti->xni_rxring.rsp_prod_pvt,
1155 i);
1156 rxresp = RING_GET_RESPONSE(
1157 &xneti->xni_rxring,
1158 xneti->xni_rxring.rsp_prod_pvt + j);
1159 rxresp->status = NETIF_RSP_ERROR;
1160 }
1161 }
1162
1163 /* update pointer */
1164 KASSERT(
1165 xneti->xni_rxring.rsp_prod_pvt + i == resp_prod);
1166 xneti->xni_rxring.rsp_prod_pvt = resp_prod;
1167 RING_PUSH_RESPONSES_AND_CHECK_NOTIFY(
1168 &xneti->xni_rxring, j);
1169 if (j)
1170 do_event = 1;
1171 /* now we can free the mbufs */
1172 for (j = 0; j < i; j++) {
1173 m_freem(mbufs_sent[j]);
1174 }
1175 for (j = 0; j < nppitems; j++) {
1176 pool_cache_put_paddr(xmit_pages_cache,
1177 (void *)pages_pool_free[j].va,
1178 pages_pool_free[j].pa);
1179 }
1180 }
1181 /* send event */
1182 if (do_event) {
1183 xen_rmb();
1184 XENPRINTF(("%s receive event\n",
1185 xneti->xni_if.if_xname));
1186 hypervisor_notify_via_evtchn(xneti->xni_evtchn);
1187 do_event = 0;
1188 }
1189 /* check if we need to get back some pages */
1190 if (mcl_pages_alloc < 0) {
1191 xennetback_get_new_mcl_pages();
1192 if (mcl_pages_alloc < 0) {
1193 /*
1194 * setup the watchdog to try again, because
1195 * xennetback_ifstart() will never be called
1196 * again if queue is full.
1197 */
1198 printf("xennetback_ifstart: no mcl_pages\n");
1199 ifp->if_timer = 1;
1200 break;
1201 }
1202 }
1203 /*
1204 * note that we don't use RING_FINAL_CHECK_FOR_REQUESTS()
1205 * here, as the frontend doesn't notify when adding
1206 * requests anyway
1207 */
1208 if (__predict_false(
1209 !RING_HAS_UNCONSUMED_REQUESTS(&xneti->xni_rxring))) {
1210 /* ring full */
1211 break;
1212 }
1213 }
1214 splx(s);
1215}

1217/*
1218 * sighly different from m_dup(); for some reason m_dup() can return
1219 * a chain where the data area can cross a page boundary.
1220 * This doesn't happens with the function below.
1221 */
1222static struct mbuf *
1223xennetback_copymbuf(struct mbuf *m)
1224{
1225 struct mbuf *new_m;
1226
1227 MGETHDR(new_m, M_DONTWAIT, MT_DATA);
1228 if (__predict_false(new_m == NULL)) {
1229 return NULL;
1230 }
1231 if (m->m_pkthdr.len > MHLEN) {
1232 MCLGET(new_m, M_DONTWAIT);
1233 if (__predict_false(
1234 (new_m->m_flags & M_EXT) == 0)) {
1235 m_freem(new_m);
1236 return NULL;
1237 }
1238 }
1239 m_copydata(m, 0, m->m_pkthdr.len,
1240 mtod(new_m, void *));
1241 new_m->m_len = new_m->m_pkthdr.len =
1242 m->m_pkthdr.len;
1243 return new_m;
1244}

1246/* return physical page address and offset of data area of an mbuf */
1247static void
1248xennetback_mbuf_addr(struct mbuf *m, paddr_t *xmit_pa, int *offset)
1249{
1250 switch (m->m_flags & (M_EXT|M_EXT_CLUSTER)) {
1251 case M_EXT|M_EXT_CLUSTER:
1252 KASSERT(m->m_ext.ext_paddr != M_PADDR_INVALID);
1253 *xmit_pa = m->m_ext.ext_paddr;
1254 *offset = m->m_data - m->m_ext.ext_buf;
1255 break;
1256 case 0:
1257 KASSERT(m->m_paddr != M_PADDR_INVALID);
1258 *xmit_pa = m->m_paddr;
1259 *offset = M_BUFOFFSET(m) +
1260 (m->m_data - M_BUFADDR(m));
1261 break;
1262 default:
1263 if (__predict_false(
1264 !pmap_extract(pmap_kernel(),
1265 (vaddr_t)m->m_data, xmit_pa))) {
1266 panic("xennet_start: no pa");
1267 }
1268 *offset = 0;
1269 break;
1270 }
1271 *offset += (*xmit_pa & ~PTE_FRAME);
1272 *xmit_pa = (*xmit_pa & PTE_FRAME);
1273}

1275static void
1276xennetback_ifsoftstart_copy(void *arg)
1277{
1278 struct xnetback_instance *xneti = arg;
1279 struct ifnet *ifp = &xneti->xni_if;
1280 struct mbuf *m, *new_m;
1281 paddr_t xmit_pa;
1282 paddr_t xmit_ma;
1283 int i, j;
1284 netif_rx_response_t *rxresp;
1285 netif_rx_request_t rxreq;
1286 RING_IDX req_prod, resp_prod;
1287 int do_event = 0;
1288 gnttab_copy_t *gop;
1289 int id, offset;
1290 bool abort;
1291
1292 XENPRINTF(("xennetback_ifsoftstart_copy "));
1293 int s = splnet();
1294 if (__predict_false(
1295 (ifp->if_flags & (IFF_RUNNING|IFF_OACTIVE)) != IFF_RUNNING)) {
1296 splx(s);
1297 return;
1298 }
1299
1300 while (!IFQ_IS_EMPTY(&ifp->if_snd)) {
1301 XENPRINTF(("pkt\n"));
1302 req_prod = xneti->xni_rxring.sring->req_prod;
1303 resp_prod = xneti->xni_rxring.rsp_prod_pvt;
1304 xen_rmb();
1305
1306 gop = xstart_gop_copy;
1307 abort = false;
1308 for (i = 0; !IFQ_IS_EMPTY(&ifp->if_snd);) {
1309 XENPRINTF(("have a packet\n"));
1310 IFQ_POLL(&ifp->if_snd, m);
1311 if (__predict_false(m == NULL))
1312 panic("xennetback_ifstart: IFQ_POLL");
1313 if (__predict_false(
1314 req_prod == xneti->xni_rxring.req_cons ||
1315 xneti->xni_rxring.req_cons - resp_prod ==
1316 NET_RX_RING_SIZE)) {
1317 /* out of ring space */
1318 XENPRINTF(("xennetback_ifstart: ring full "
1319 "req_prod 0x%x req_cons 0x%x resp_prod "
1320 "0x%x\n",
1321 req_prod, xneti->xni_rxring.req_cons,
1322 resp_prod));
1323 abort = true;
1324 break;
1325 }
1326 if (__predict_false(i == NB_XMIT_PAGES_BATCH))
1327 break; /* we filled the array */
1328
1329 xennetback_mbuf_addr(m, &xmit_pa, &offset);
1330 if (m->m_pkthdr.len != m->m_len ||
1331 (offset + m->m_pkthdr.len) > PAGE_SIZE) {
1332 new_m = xennetback_copymbuf(m);
1333 if (__predict_false(new_m == NULL)) {
1334 static struct timeval lasttime;
1335 if (ratecheck(&lasttime, &xni_pool_errintvl))
1336 printf("%s: cannot allocate new mbuf\n",
1337 ifp->if_xname);
1338 abort = 1;
1339 break;
1340 } else {
1341 IFQ_DEQUEUE(&ifp->if_snd, m);
1342 m_freem(m);
1343 m = new_m;
1344 xennetback_mbuf_addr(m,
1345 &xmit_pa, &offset);
1346 }
1347 } else {
1348 IFQ_DEQUEUE(&ifp->if_snd, m);
1349 }
1350
1351 KASSERT(xmit_pa != POOL_PADDR_INVALID);
1352 KASSERT((offset + m->m_pkthdr.len) <= PAGE_SIZE);
1353 xmit_ma = xpmap_ptom(xmit_pa);
1354 /* start filling ring */
1355 gop->flags = GNTCOPY_dest_gref;
1356 gop->source.offset = offset;
1357 gop->source.domid = DOMID_SELF;
1358 gop->source.u.gmfn = xmit_ma >> PAGE_SHIFT;
1359
1360 RING_COPY_REQUEST(&xneti->xni_rxring,
1361 xneti->xni_rxring.req_cons, &rxreq);
1362 gop->dest.u.ref = rxreq.gref;
1363 gop->dest.offset = 0;
1364 gop->dest.domid = xneti->xni_domid;
1365
1366 gop->len = m->m_pkthdr.len;
1367 gop++;
1368
1369 id = rxreq.id;
1370 xen_rmb();
1371 xneti->xni_rxring.req_cons++;
1372 rxresp = RING_GET_RESPONSE(&xneti->xni_rxring,
1373 resp_prod);
1374 rxresp->id = id;
1375 rxresp->offset = 0;
1376 rxresp->status = m->m_pkthdr.len;
1377 if ((m->m_pkthdr.csum_flags &
1378 (M_CSUM_TCPv4 | M_CSUM_UDPv4)) != 0) {
1379 rxresp->flags = NETRXF_csum_blank;
1380 } else {
1381 rxresp->flags = 0;
1382 }
1383
1384 mbufs_sent[i] = m;
1385 resp_prod++;
1386 i++; /* this packet has been queued */
1387 ifp->if_opackets++;
1388 bpf_mtap(ifp, m, BPF_D_OUT);
1389 }
1390 if (i != 0) {
1391 if (HYPERVISOR_grant_table_op(GNTTABOP_copy,
1392 xstart_gop_copy, i) != 0) {
1393 panic("%s: GNTTABOP_copy failed",
1394 ifp->if_xname);
1395 }
1396
1397 for (j = 0; j < i; j++) {
1398 if (xstart_gop_copy[j].status != GNTST_okay) {
1399 printf("%s GNTTABOP_copy[%d] %d\n",
1400 ifp->if_xname,
1401 j, xstart_gop_copy[j].status);
1402 printf("%s: req_prod %u req_cons "
1403 "%u rsp_prod %u rsp_prod_pvt %u "
1404 "i %d\n",
1405 ifp->if_xname,
1406 xneti->xni_rxring.sring->req_prod,
1407 xneti->xni_rxring.req_cons,
1408 xneti->xni_rxring.sring->rsp_prod,
1409 xneti->xni_rxring.rsp_prod_pvt,
1410 i);
1411 rxresp = RING_GET_RESPONSE(
1412 &xneti->xni_rxring,
1413 xneti->xni_rxring.rsp_prod_pvt + j);
1414 rxresp->status = NETIF_RSP_ERROR;
1415 }
1416 }
1417
1418 /* update pointer */
1419 KASSERT(
1420 xneti->xni_rxring.rsp_prod_pvt + i == resp_prod);
1421 xneti->xni_rxring.rsp_prod_pvt = resp_prod;
1422 RING_PUSH_RESPONSES_AND_CHECK_NOTIFY(
1423 &xneti->xni_rxring, j);
1424 if (j)
1425 do_event = 1;
1426 /* now we can free the mbufs */
1427 for (j = 0; j < i; j++) {
1428 m_freem(mbufs_sent[j]);
1429 }
1430 }
1431 /* send event */
1432 if (do_event) {
1433 xen_rmb();
1434 XENPRINTF(("%s receive event\n",
1435 xneti->xni_if.if_xname));
1436 hypervisor_notify_via_evtchn(xneti->xni_evtchn);
1437 do_event = 0;
1438 }
1439 /*
1440 * note that we don't use RING_FINAL_CHECK_FOR_REQUESTS()
1441 * here, as the frontend doesn't notify when adding
1442 * requests anyway
1443 */
1444 if (__predict_false(abort ||
1445 !RING_HAS_UNCONSUMED_REQUESTS(&xneti->xni_rxring))) {
1446 /* ring full */
1447 ifp->if_timer = 1;
1448 break;
1449 }
1450 }
1451 splx(s);
1452}

1454static void
1455xennetback_ifwatchdog(struct ifnet * ifp)
1456{
1457 /*
1458 * We can get to the following condition: transmit stalls because the
1459 * ring is full when the ifq is full too.
1460 *
1461 * In this case (as, unfortunately, we don't get an interrupt from xen
1462 * on transmit) nothing will ever call xennetback_ifstart() again.
1463 * Here we abuse the watchdog to get out of this condition.
1464 */
1465 XENPRINTF(("xennetback_ifwatchdog\n"));
1466 xennetback_ifstart(ifp);
