50#include <compat/common/compat_util.h>

52
53int
54compat_14_init(void)
55{
56
57 rtsock_14_init();
58 return 0;
59}
60
61int
62compat_14_fini(void)
63{
64
65 rtsock_14_fini();
66 return 0;
67}
68
69MODULE(MODULE_CLASS_EXEC, compat_14, "compat_16");
70
71static int
72compat_14_modcmd(modcmd_t cmd, void *arg)
73{
74
75 switch (cmd) {
76 case MODULE_CMD_INIT:
77 return compat_14_init();
78 case MODULE_CMD_FINI:
79 return compat_14_fini();
80 default:
81 return ENOTTY;
82 }
83}
