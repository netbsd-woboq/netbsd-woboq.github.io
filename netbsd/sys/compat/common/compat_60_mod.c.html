56int
57compat_60_init(void)
58{
59 int error = 0;
60
61 error = kern_time_60_init();
62 if (error != 0)
63 return error;
64
65 error = kern_sa_60_init();
66 if (error != 0) {
67 kern_time_60_fini();
68 return 0;
69 }
70
71 kern_tty_60_init();
72 ccd_60_init();
73#ifdef CPU_UCODE
74 kern_cpu_60_init();
75#endif
76
77 return error;
78}
79
80int
81compat_60_fini(void)
82{
83 int error = 0;
84
85#ifdef CPU_UCODE
86 kern_cpu_60_fini();
87#endif
88 ccd_60_fini();
89 kern_tty_60_fini();
90
91
92 error = kern_sa_60_fini();
93 if (error != 0) {
94 kern_tty_60_init();
95 ccd_60_init();
96#ifdef CPU_UCODE
97 kern_cpu_60_init();
98#endif
99 return error;
100 }
101
102 error = kern_time_60_fini();
103 if (error != 0) {
104 kern_sa_60_init();
105 kern_tty_60_init();
106 ccd_60_init();
107#ifdef CPU_UCODE
108 kern_cpu_60_init();
109#endif
110 return error;
111 }
112
113 return error;
114}
115
116MODULE(MODULE_CLASS_EXEC, compat_60, "compat_70");
117
118static int
119compat_60_modcmd(modcmd_t cmd, void *arg)
120{
121
122 switch (cmd) {
123 case MODULE_CMD_INIT:
124 return compat_60_init();
125 case MODULE_CMD_FINI:
126 return compat_60_fini();
127 default:
128 return ENOTTY;
129 }
130}
