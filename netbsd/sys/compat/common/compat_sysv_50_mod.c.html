50MODULE(MODULE_CLASS_EXEC, compat_sysv_50, "sysv_ipc");
51
52/* Build the syscall package based on options specified */
53
54static const struct syscall_package compat_sysv_50_syscalls[] = {
55#ifdef SYSVSHM
56 { SYS_compat_50___shmctl13, 0, (sy_call_t *)compat_50_sys___shmctl13 },
57#endif
58#ifdef SYSVSEM
59 { SYS_compat_50_____semctl13, 0, (sy_call_t *)compat_50_sys_____semctl13 },
60#endif
61#ifdef SYSVMSG
62 { SYS_compat_50___msgctl13, 0, (sy_call_t *)compat_50_sys___msgctl13 },
63#endif
64 { 0, 0, NULL }
65};
66
67static int
68compat_sysv_50_modcmd(modcmd_t cmd, void *arg)
69{
70 int error = 0;
71
72 switch (cmd) {
73 case MODULE_CMD_INIT:
74 error = syscall_establish(NULL, compat_sysv_50_syscalls);
75 if (error != 0) {
76 break;
77 }
78 MODULE_HOOK_SET(sysvipc_sysctl_50_hook, "sysv50",
79 sysctl_kern_sysvipc50);
80 break;
81
82 case MODULE_CMD_FINI:
83 error = syscall_disestablish(NULL, compat_sysv_50_syscalls);
84 if (error != 0) {
85 break;
86 }
87 MODULE_HOOK_UNSET(sysvipc_sysctl_50_hook);
88 break;
89
90 default:
91 error = ENOTTY;
92 break;
93 }
94 return error;
95}
