81static struct syscall_package kern_exit_43_syscalls[] = {
82 { SYS_compat_43_owait, 0, (sy_call_t *)compat_43_sys_wait },
83 { 0, 0, NULL}
84};
85
86int
87compat_43_sys_wait(struct lwp *l, const void *v, register_t *retval)
88{
89 int error, status, child_pid = WAIT_ANY;
90
91#ifdef PSL_ALLCC
92 if ((GETPS(l->l_md.md_regs) & PSL_ALLCC) != PSL_ALLCC) {
93 error = do_sys_wait(&child_pid, &status, 0, NULL);
94 } else {
95 error = do_sys_wait(&child_pid, &status,
96 l->l_md.md_regs[R0], (struct rusage *)l->l_md.md_regs[R1]);
97 }
98#else
99 error = do_sys_wait(&child_pid, &status, 0, NULL);
100#endif
101 retval[0] = child_pid;
102 retval[1] = status;
103 return error;
104}
105
106int
107kern_exit_43_init(void)
108{
109
110 return syscall_establish(NULL, kern_exit_43_syscalls);
111}
112
113int
114kern_exit_43_fini(void)
115{
116
117 return syscall_disestablish(NULL, kern_exit_43_syscalls);
118}
