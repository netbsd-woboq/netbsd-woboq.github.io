58static struct syscall_package kern_info_09_syscalls[] = {
59 { SYS_compat_09_ogetdomainname, 0,
60 (sy_call_t *)compat_09_sys_getdomainname },
61 { SYS_compat_09_osetdomainname, 0,
62 (sy_call_t *)compat_09_sys_setdomainname },
63 { SYS_compat_09_ouname, 0, (sy_call_t *)compat_09_sys_uname },
64 { 0, 0, NULL }
65};
66
67/* ARGSUSED */
68int
69compat_09_sys_getdomainname(struct lwp *l,
70 const struct compat_09_sys_getdomainname_args *uap, register_t *retval)
71{
72 /* {
73 syscallarg(char *) domainname;
74 syscallarg(int) len;
75 } */
76 int name[2];
77 size_t sz;
78
79 name[0] = CTL_KERN;
80 name[1] = KERN_DOMAINNAME;
81 sz = SCARG(uap,len);
82 return (old_sysctl(&name[0], 2, SCARG(uap, domainname), &sz, 0, 0, l));
83}
84
85
86/* ARGSUSED */
87int
88compat_09_sys_setdomainname(struct lwp *l,
89 const struct compat_09_sys_setdomainname_args *uap, register_t *retval)
90{
91 /* {
92 syscallarg(char *) domainname;
93 syscallarg(int) len;
94 } */
95 int name[2];
96
97 name[0] = CTL_KERN;
98 name[1] = KERN_DOMAINNAME;
99 return (old_sysctl(&name[0], 2, 0, 0, SCARG(uap, domainname),
100 SCARG(uap, len), l));
101}

103struct outsname {
104 char sysname[32];
105 char nodename[32];
106 char release[32];
107 char version[32];
108 char machine[32];
109};
110
111/* ARGSUSED */
112int
113compat_09_sys_uname(struct lwp *l,
114 const struct compat_09_sys_uname_args *uap, register_t *retval)
115{
116 /* {
117 syscallarg(struct outsname *) name;
118 } */
119 struct outsname outsname;
120 const char *cp;
121 char *dp, *ep;
122
123 strncpy(outsname.sysname, ostype, sizeof(outsname.sysname));
124 strncpy(outsname.nodename, hostname, sizeof(outsname.nodename));
125 strncpy(outsname.release, osrelease, sizeof(outsname.release));
126 dp = outsname.version;
127 ep = &outsname.version[sizeof(outsname.version) - 1];
128 for (cp = version; *cp && *cp != '('; cp++)
129 ;
130 for (cp++; *cp && *cp != ')' && dp < ep; cp++)
131 *dp++ = *cp;
132 for (; *cp && *cp != '#'; cp++)
133 ;
134 for (; *cp && *cp != ':' && dp < ep; cp++)
135 *dp++ = *cp;
136 *dp = '\0';
137 strncpy(outsname.machine, MACHINE, sizeof(outsname.machine));
138 return (copyout((void *)&outsname, (void *)SCARG(uap, name),
139 sizeof(struct outsname)));
140}
