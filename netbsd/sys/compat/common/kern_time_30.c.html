60static const struct syscall_package kern_time_30_syscalls[] = {
61 { SYS_compat_30_ntp_gettime, 0,
62 (sy_call_t *)compat_30_sys_ntp_gettime },
63 { 0, 0, NULL }
64};
65
66int
67compat_30_sys_ntp_gettime(struct lwp *l,
68 const struct compat_30_sys_ntp_gettime_args *uap, register_t *retval)
69{
70 /* {
71 syscallarg(struct ntptimeval30 *) ontvp;
72 } */
73 struct ntptimeval ntv;
74 struct ntptimeval30 ntv30;
75 struct timeval tv;
76 int error;
77
78 if (vec_ntp_gettime == NULL)
79 return ENOSYS;
80
81 if (SCARG(uap, ntvp)) {
82 (*vec_ntp_gettime)(&ntv);
83 memset(&ntv30, 0, sizeof(ntv30));
84 TIMESPEC_TO_TIMEVAL(&tv, &ntv.time);
85 timeval_to_timeval50(&tv, &ntv30.time);
86 ntv30.maxerror = ntv.maxerror;
87 ntv30.esterror = ntv.esterror;
88
89 error = copyout(&ntv30, SCARG(uap, ntvp), sizeof(ntv30));
90 if (error)
91 return error;
92 }
93 *retval = (*vec_ntp_timestatus)();
94 return 0;
95}

104int
105kern_time_30_fini(void)
