98void
99compat_14_rt_oifmsg(struct ifnet *ifp)
100{
101 struct if_msghdr14 oifm;
102 struct mbuf *m;
103 struct rt_addrinfo info;
104 struct timeval tv;
105
106 if (compat_50_route_info.ri_cb.any_count == 0)
107 return;
108 (void)memset(&info, 0, sizeof(info));
109 (void)memset(&oifm, 0, sizeof(oifm));
110 oifm.ifm_index = ifp->if_index;
111 oifm.ifm_flags = ifp->if_flags;
112 oifm.ifm_data.ifi_type = ifp->if_data.ifi_type;
113 oifm.ifm_data.ifi_addrlen = ifp->if_data.ifi_addrlen;
114 oifm.ifm_data.ifi_hdrlen = ifp->if_data.ifi_hdrlen;
115 oifm.ifm_data.ifi_mtu = ifp->if_data.ifi_mtu;
116 oifm.ifm_data.ifi_metric = ifp->if_data.ifi_metric;
117 oifm.ifm_data.ifi_baudrate = ifp->if_data.ifi_baudrate;
118 oifm.ifm_data.ifi_ipackets = ifp->if_data.ifi_ipackets;
119 oifm.ifm_data.ifi_ierrors = ifp->if_data.ifi_ierrors;
120 oifm.ifm_data.ifi_opackets = ifp->if_data.ifi_opackets;
121 oifm.ifm_data.ifi_oerrors = ifp->if_data.ifi_oerrors;
122 oifm.ifm_data.ifi_collisions = ifp->if_data.ifi_collisions;
123 oifm.ifm_data.ifi_ibytes = ifp->if_data.ifi_ibytes;
124 oifm.ifm_data.ifi_obytes = ifp->if_data.ifi_obytes;
125 oifm.ifm_data.ifi_imcasts = ifp->if_data.ifi_imcasts;
126 oifm.ifm_data.ifi_omcasts = ifp->if_data.ifi_omcasts;
127 oifm.ifm_data.ifi_iqdrops = ifp->if_data.ifi_iqdrops;
128 oifm.ifm_data.ifi_noproto = ifp->if_data.ifi_noproto;
129 TIMESPEC_TO_TIMEVAL(&tv, &ifp->if_data.ifi_lastchange);
130 timeval_to_timeval50(&tv, &oifm.ifm_data.ifi_lastchange);
131 oifm.ifm_addrs = 0;
132 m = compat_50_rt_msg1(RTM_OOIFINFO, &info, (void *)&oifm, sizeof(oifm));
133 if (m == NULL)
134 return;
135 compat_50_route_enqueue(m, 0);
136}

138int
139compat_14_iflist(struct ifnet *ifp, struct rt_walkarg *w,
140 struct rt_addrinfo *info, size_t len)
141{
142 struct if_msghdr14 *ifm;
143 struct timeval tv;
144 int error;
145
146 ifm = (struct if_msghdr14 *)w->w_tmem;
147 ifm->ifm_index = ifp->if_index;
148 ifm->ifm_flags = ifp->if_flags;
149 ifm->ifm_data.ifi_type = ifp->if_data.ifi_type;
150 ifm->ifm_data.ifi_addrlen = ifp->if_data.ifi_addrlen;
151 ifm->ifm_data.ifi_hdrlen = ifp->if_data.ifi_hdrlen;
152 ifm->ifm_data.ifi_mtu = ifp->if_data.ifi_mtu;
153 ifm->ifm_data.ifi_metric = ifp->if_data.ifi_metric;
154 ifm->ifm_data.ifi_baudrate = ifp->if_data.ifi_baudrate;
155 ifm->ifm_data.ifi_ipackets = ifp->if_data.ifi_ipackets;
156 ifm->ifm_data.ifi_ierrors = ifp->if_data.ifi_ierrors;
157 ifm->ifm_data.ifi_opackets = ifp->if_data.ifi_opackets;
158 ifm->ifm_data.ifi_oerrors = ifp->if_data.ifi_oerrors;
159 ifm->ifm_data.ifi_collisions = ifp->if_data.ifi_collisions;
160 ifm->ifm_data.ifi_ibytes = ifp->if_data.ifi_ibytes;
161 ifm->ifm_data.ifi_obytes = ifp->if_data.ifi_obytes;
162 ifm->ifm_data.ifi_imcasts = ifp->if_data.ifi_imcasts;
163 ifm->ifm_data.ifi_omcasts = ifp->if_data.ifi_omcasts;
164 ifm->ifm_data.ifi_iqdrops = ifp->if_data.ifi_iqdrops;
165 ifm->ifm_data.ifi_noproto = ifp->if_data.ifi_noproto;
166 TIMESPEC_TO_TIMEVAL(&tv, &ifp->if_data.ifi_lastchange);
167 timeval_to_timeval50(&tv, &ifm->ifm_data.ifi_lastchange);
168 ifm->ifm_addrs = info->rti_addrs;
169 error = copyout(ifm, w->w_where, len);
170 if (error)
171 return error;
172 w->w_where = (char *)w->w_where + len;
173 return 0;
174}

176void
177rtsock_14_init(void)
178{
179
180 MODULE_HOOK_SET(rtsock_oifmsg_14_hook, "rts_14", compat_14_rt_oifmsg);
181 MODULE_HOOK_SET(rtsock_iflist_14_hook, "rts_14", compat_14_iflist);
182}
183
184void
185rtsock_14_fini(void)
186{
187
188 MODULE_HOOK_UNSET(rtsock_oifmsg_14_hook);
189 MODULE_HOOK_UNSET(rtsock_iflist_14_hook);
190}
