51/* convert struct ptmget to struct compat_60_ptmget */
52static int
53ptmget_to_ptmget60(struct ptmget *pg, struct compat_60_ptmget *pg60)
54{
55 memset(pg60, 0, sizeof(*pg60));
56 pg60->cfd = pg->cfd;
57 pg60->sfd = pg->sfd;
58 strlcpy(pg60->cn, pg->cn, sizeof(pg60->cn));
59 strlcpy(pg60->sn, pg->sn, sizeof(pg60->sn));
60 if (strlen(pg->cn) >= sizeof(pg60->cn)
61 || strlen(pg->sn) >= sizeof(pg60->sn))
62 return E2BIG;
63 return 0;
64}

66/* Helper for compat ioctls that use struct compat_60_ptmget. */
67static int
68compat_60_ptmget_ioctl(dev_t dev, u_long cmd, void *data, int flag,
69 struct lwp *l)
70{
71 int ret;
72 u_long newcmd;
73 struct ptmget pg;
74 const struct cdevsw *cd = cdevsw_lookup(dev);
75
76 if (cd == NULL || cd->d_ioctl == NULL)
77 return ENXIO;
78
79 switch (cmd) {
80 case COMPAT_60_TIOCPTMGET: newcmd = TIOCPTMGET; break;
81 case COMPAT_60_TIOCPTSNAME: newcmd = TIOCPTSNAME; break;
82 default: return ENOTTY;
83 }
84
85 ret = (cd->d_ioctl)(dev, newcmd, &pg, flag, l);
86 if (ret != 0)
87 return ret;
88 ret = ptmget_to_ptmget60(&pg, data);
89 return ret;
90}

92/*
93 * COMPAT_60 versions of ttioctl and ptmioctl.
94 */
95int
96compat_60_ttioctl(struct tty *tp, u_long cmd, void *data, int flag,
97 struct lwp *l)
98{
99
100 switch (cmd) {
101 case COMPAT_60_TIOCPTMGET:
102 case COMPAT_60_TIOCPTSNAME:
103 return compat_60_ptmget_ioctl(tp->t_dev, cmd, data, flag, l);
104 default:
105 return EPASSTHROUGH;
106 }
107}
108
109int
110compat_60_ptmioctl(dev_t dev, u_long cmd, void *data, int flag,
111 struct lwp *l)
112{
113
114 switch (cmd) {
115 case COMPAT_60_TIOCPTMGET:
116 return compat_60_ptmget_ioctl(dev, cmd, data, flag, l);
117 default:
118 return EPASSTHROUGH;
119 }
120}

122void
123kern_tty_60_init(void)
124{
125
126 MODULE_HOOK_SET(tty_ttioctl_60_hook, "tty_60", compat_60_ttioctl);
127 MODULE_HOOK_SET(tty_ptmioctl_60_hook, "tty_60", compat_60_ptmioctl);
128}
129
130void
131kern_tty_60_fini(void)
132{
133 MODULE_HOOK_UNSET(tty_ttioctl_60_hook);
134 MODULE_HOOK_UNSET(tty_ptmioctl_60_hook);
135}
