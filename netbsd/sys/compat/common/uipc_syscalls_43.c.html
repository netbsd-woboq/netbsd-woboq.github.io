82/*
83 * Following 4.3 syscalls were not versioned, even through they should
84 * have been:
85 * connect(2), bind(2), sendto(2)

88static struct syscall_package uipc_syscalls_43_syscalls[] = {
89 { SYS_compat_43_oaccept, 0, (sy_call_t *)compat_43_sys_accept },
90 { SYS_compat_43_ogetpeername, 0,
91 (sy_call_t *)compat_43_sys_getpeername },
92 { SYS_compat_43_ogetsockname, 0,
93 (sy_call_t *)compat_43_sys_getsockname },
94 { SYS_compat_43_orecv, 0, (sy_call_t *)compat_43_sys_recv },
95 { SYS_compat_43_orecvfrom, 0, (sy_call_t *)compat_43_sys_recvfrom },
96 { SYS_compat_43_orecvmsg, 0, (sy_call_t *)compat_43_sys_recvmsg },
97 { SYS_compat_43_osend, 0, (sy_call_t *)compat_43_sys_send },
98 { SYS_compat_43_osendmsg, 0, (sy_call_t *)compat_43_sys_sendmsg },
99 { 0, 0, NULL }
100};

104int
105compat_43_sys_accept(struct lwp *l, const struct compat_43_sys_accept_args *uap, register_t *retval)
106{
107 /* {
108 syscallarg(int) s;
109 syscallarg(void *) name;
110 syscallarg(int *) anamelen;
111 } */
112 int error;
113
114 if ((error = sys_accept(l, (const struct sys_accept_args *)uap, retval)) != 0)
115 return error;
116
117 if (SCARG(uap, name)
118 && (error = compat_43_sa_put(SCARG(uap, name))))
119 return (error);
120
121 return 0;
122}

173

206/*
207 * Old recvmsg. Arrange necessary structures, calls generic code and
208 * adjusts results accordingly.
209 */
210int
211compat_43_sys_recvmsg(struct lwp *l, const struct compat_43_sys_recvmsg_args *uap, register_t *retval)
212{
213 /* {
214 syscallarg(int) s;
215 syscallarg(struct omsghdr *) msg;
216 syscallarg(int) flags;
217 } */
218 struct omsghdr omsg;
219 struct msghdr msg;
220 struct mbuf *from, *control;
221 int error;
222
223 error = copyin(SCARG(uap, msg), &omsg, sizeof (struct omsghdr));
224 if (error)
225 return (error);
226
227 if (omsg.msg_accrights == NULL)
228 omsg.msg_accrightslen = 0;
229 /* it was this way in 4.4BSD */
230 if (omsg.msg_accrightslen > MLEN)
231 return EINVAL;
232
233 msg.msg_name = omsg.msg_name;
234 msg.msg_namelen = omsg.msg_namelen;
235 msg.msg_iovlen = omsg.msg_iovlen;
236 msg.msg_iov = omsg.msg_iov;
237 msg.msg_flags = (SCARG(uap, flags) & MSG_USERFLAGS) | MSG_IOVUSRSPACE;
238
239 error = do_sys_recvmsg(l, SCARG(uap, s), &msg, &from,
240 omsg.msg_accrights != NULL ? &control : NULL, retval);
241 if (error != 0)
242 return error;
243
244 /*
245 * If there is any control information and it's SCM_RIGHTS,
246 * pass it back to the program.
247 * XXX: maybe there can be more than one chunk of control data?
248 */
249 if (omsg.msg_accrights && control != NULL) {
250 struct cmsghdr *cmsg = mtod(control, struct cmsghdr *);
251
252 if (cmsg->cmsg_level == SOL_SOCKET
253 && cmsg->cmsg_type == SCM_RIGHTS
254 && cmsg->cmsg_len < omsg.msg_accrightslen
255 && copyout(CMSG_DATA(cmsg), omsg.msg_accrights,
256 cmsg->cmsg_len) == 0) {
257 omsg.msg_accrightslen = cmsg->cmsg_len;
258 free_control_mbuf(l, control, control->m_next);
259 } else {
260 omsg.msg_accrightslen = 0;
261 free_control_mbuf(l, control, control);
262 }
263 } else
264 omsg.msg_accrightslen = 0;
265
266 if (from != NULL)
267 /* convert from sockaddr sa_family to osockaddr one here */
268 mtod(from, struct osockaddr *)->sa_family =
269 mtod(from, struct sockaddr *)->sa_family;
270
271 error = copyout_sockname((struct sockaddr *)omsg.msg_name, &omsg.msg_namelen, 0, from);
272 if (from != NULL)
273 m_free(from);
274
275 if (error != 0)
276 error = copyout(&omsg, SCARG(uap, msg), sizeof(omsg));
277
278 return error;
279}

302int
303compat43_set_accrights(struct msghdr *msg, void *accrights, int accrightslen)
304{
305 struct cmsghdr *cmsg;
306 int error;
307 struct mbuf *ctl;
308 u_int clen;
309
310 if (accrights == NULL || accrightslen == 0) {
311 msg->msg_control = NULL;
312 msg->msg_controllen = 0;
313 return 0;
314 }
315
316 clen = CMSG_SPACE(accrightslen);
317 /* it was (almost) this way in 4.4BSD */
318 if (accrightslen < 0 || clen > MLEN)
319 return EINVAL;
320
321 ctl = m_get(M_WAIT, MT_CONTROL);
322 ctl->m_len = clen;
323 cmsg = mtod(ctl, struct cmsghdr *);
324 cmsg->cmsg_len = CMSG_SPACE(accrightslen);
325 cmsg->cmsg_level = SOL_SOCKET;
326 cmsg->cmsg_type = SCM_RIGHTS;
327
328 error = copyin(accrights, CMSG_DATA(cmsg), accrightslen);
329 if (error) {
330 m_free(ctl);
331 return error;
332 }
333
334 msg->msg_control = ctl;
335 msg->msg_controllen = clen;
336 msg->msg_flags |= MSG_CONTROLMBUF;
337 return 0;
338}

340/*
341 * Old sendmsg. Arrange necessary structures, call generic code and
342 * adjust the results accordingly for old code.
343 */
344int
345compat_43_sys_sendmsg(struct lwp *l, const struct compat_43_sys_sendmsg_args *uap, register_t *retval)
346{
347 /* {
348 syscallarg(int) s;
349 syscallarg(void *) msg;
350 syscallarg(int) flags;
351 } */
352 struct omsghdr omsg;
353 struct msghdr msg;
354 int error;
355 struct mbuf *nam;
356 struct osockaddr *osa;
357 struct sockaddr *sa;
358
359 error = copyin(SCARG(uap, msg), &omsg, sizeof (struct omsghdr));
360 if (error != 0)
361 return (error);
362
363 msg.msg_iovlen = omsg.msg_iovlen;
364 msg.msg_iov = omsg.msg_iov;
365
366 error = sockargs(&nam, omsg.msg_name, omsg.msg_namelen,
367 UIO_USERSPACE, MT_SONAME);
368 if (error != 0)
369 return (error);
370
371 sa = mtod(nam, struct sockaddr *);
372 osa = mtod(nam, struct osockaddr *);
373 sa->sa_family = osa->sa_family;
374 sa->sa_len = omsg.msg_namelen;
375
376 msg.msg_flags = MSG_IOVUSRSPACE | MSG_NAMEMBUF;
377
378 msg.msg_name = nam;
379 msg.msg_namelen = omsg.msg_namelen;
380 error = compat43_set_accrights(&msg, omsg.msg_accrights,
381 omsg.msg_accrightslen);
382 if (error != 0)
383 goto bad;
384
385 return do_sys_sendmsg(l, SCARG(uap, s), &msg, SCARG(uap, flags),
386 retval);
387
388 bad:
389 if (nam != NULL)
390 m_free(nam);
391
392 return (error);
393}

395static int
396compat_43_sa_put(void *from)
397{
398 struct osockaddr *osa = (struct osockaddr *) from;
399 struct sockaddr sa;
400 struct osockaddr *kosa;
401 int error, len;
402
403 /*
404 * Only read/write the sockaddr family and length, the rest is
405 * not changed.
406 */
407 len = sizeof(sa.sa_len) + sizeof(sa.sa_family);
408
409 error = copyin((void *) osa, (void *) &sa, len);
410 if (error)
411 return (error);
412
413 /* Note: we convert from sockaddr sa_family to osockaddr one here */
414 kosa = (struct osockaddr *) &sa;
415 kosa->sa_family = sa.sa_family;
416 error = copyout(kosa, osa, len);
417 if (error)
418 return (error);
419
420 return (0);
421}
