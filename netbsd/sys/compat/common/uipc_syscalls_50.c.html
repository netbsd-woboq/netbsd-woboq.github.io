64/*ARGSUSED*/
65static int
66compat_ifdatareq(struct lwp *l, u_long cmd, void *data)
67{
68 struct oifdatareq *ifdr = data;
69 struct ifnet *ifp;
70 int error;
71
72 /* Validate arguments. */
73 switch (cmd) {
74 case OSIOCGIFDATA:
75 case OSIOCZIFDATA:
76 break;
77 default:
78 return ENOSYS;
79 }
80
81 ifp = ifunit(ifdr->ifdr_name);
82 if (ifp == NULL)
83 return ENXIO;
84
85 /* Do work. */
86 switch (cmd) {
87 case OSIOCGIFDATA:
88 ifdatan2o(&ifdr->ifdr_data, &ifp->if_data);
89 return 0;
90
91 case OSIOCZIFDATA:
92 if (l != NULL) {
93 error = kauth_authorize_network(l->l_cred,
94 KAUTH_NETWORK_INTERFACE,
95 KAUTH_REQ_NETWORK_INTERFACE_SETPRIV, ifp,
96 (void *)cmd, NULL);
97 if (error != 0)
98 return error;
99 }
100 ifdatan2o(&ifdr->ifdr_data, &ifp->if_data);
101 /*
102 * Assumes that the volatile counters that can be
103 * zero'ed are at the end of if_data.
104 */
105 memset(&ifp->if_data.ifi_ipackets, 0, sizeof(ifp->if_data) -
106 offsetof(struct if_data, ifi_ipackets));
107 return 0;
108
109 default:
110 /* Impossible due to above validation, but makes gcc happy. */
111 return ENOSYS;
112 }
113}
114
115void
116uipc_syscalls_50_init(void)
117{
118
119 MODULE_HOOK_SET(uipc_syscalls_50_hook, "uipc50", compat_ifdatareq);
120}
121
122void
123uipc_syscalls_50_fini(void)
124{
125
126 MODULE_HOOK_UNSET(uipc_syscalls_50_hook);
127}
