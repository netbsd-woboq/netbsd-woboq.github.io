57static void
58swapent13_cvt(void *p, const struct swapent *se)
59{
60 struct swapent13 *sep13 = p;
61
62 sep13->se13_dev = se->se_dev;
63 sep13->se13_flags = se->se_flags;
64 sep13->se13_nblks = se->se_nblks;
65 sep13->se13_inuse = se->se_inuse;
66 sep13->se13_priority = se->se_priority;
67}
68
69
70static int
71compat_uvm_swap_stats13(const struct sys_swapctl_args *uap, register_t *retval)
72{
73 return uvm_swap_stats(SCARG(uap, arg), SCARG(uap, misc),
74 swapent13_cvt, sizeof(struct swapent13), retval);
75
76}

82}

84void
