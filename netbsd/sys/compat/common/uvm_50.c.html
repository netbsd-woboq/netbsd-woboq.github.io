56static void
57swapent50_cvt(void *p, const struct swapent *se)
58{
59 struct swapent50 *sep50 = p;
60
61 sep50->se50_dev = se->se_dev;
62 sep50->se50_flags = se->se_flags;
63 sep50->se50_nblks = se->se_nblks;
64 sep50->se50_inuse = se->se_inuse;
65 sep50->se50_priority = se->se_priority;
66 KASSERT(sizeof(se->se_path) <= sizeof(sep50->se50_path));
67 strcpy(sep50->se50_path, se->se_path);
68}
69
70
71static int
72compat_uvm_swap_stats50(const struct sys_swapctl_args *uap, register_t *retval)
73{
74 return uvm_swap_stats(SCARG(uap, arg), SCARG(uap, misc),
75 swapent50_cvt, sizeof(struct swapent50), retval);
76
77}
