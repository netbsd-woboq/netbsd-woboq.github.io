47static int
48real_sys_openat_10(struct pathbuf **pb)
49{
50
51 *pb = pathbuf_create(".");
52 return (*pb == NULL ? ENOMEM : 0);
53}
54
55void vfs_syscalls_10_init(void)
56{
57
58 MODULE_HOOK_SET(vfs_openat_10_hook, "openat_10", real_sys_openat_10);
59}
60
61void vfs_syscalls_10_fini(void)
62{
63
64 MODULE_HOOK_UNSET(vfs_openat_10_hook);
65}
