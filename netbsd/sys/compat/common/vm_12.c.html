47static const struct syscall_package vm_12_syscalls[] = {
48 { SYS_compat_12_msync, 0, (sy_call_t *)compat_12_sys_msync },
49 { SYS_compat_12_oswapon, 0, (sy_call_t *)compat_12_sys_swapon },
50 { 0, 0, NULL }
51};
52
53int
54compat_12_sys_swapon(struct lwp *l,
55 const struct compat_12_sys_swapon_args *uap, register_t *retval)
56{
57 /* {
58 syscallarg(const char *) name;
59 } */
60 struct sys_swapctl_args ua;
61
62 SCARG(&ua, cmd) = SWAP_ON;
63 /*XXXUNCONST*/
64 SCARG(&ua, arg) = __UNCONST(SCARG(uap, name));
65 SCARG(&ua, misc) = 0; /* priority */
66 return (sys_swapctl(l, &ua, retval));
67}
68
69int
70compat_12_sys_msync(struct lwp *l,
71 const struct compat_12_sys_msync_args *uap, register_t *retval)
72{
73 /* {
74 syscallarg(void *) addr;
75 syscallarg(size_t) len;
76 } */
77 struct sys___msync13_args ua;
78
79 SCARG(&ua, addr) = SCARG(uap, addr);
80 SCARG(&ua, len) = SCARG(uap, len);
81 SCARG(&ua, flags) = MS_SYNC | MS_INVALIDATE;
82 return (sys___msync13(l, &ua, retval));
83}
84
85int
86vm_12_init(void)
87{
88
89 return syscall_establish(NULL, vm_12_syscalls);
90}
91
92int
93vm_12_fini(void)
94{
95
96 return syscall_disestablish(NULL, vm_12_syscalls);
97}
