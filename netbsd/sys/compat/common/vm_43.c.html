85int
86compat_43_sys_mmap(struct lwp *l, const struct compat_43_sys_mmap_args *uap, register_t *retval)
87{
88 /* {
89 syscallarg(void *) addr;
90 syscallarg(size_t) len;
91 syscallarg(int) prot;
92 syscallarg(int) flags;
93 syscallarg(int) fd;
94 syscallarg(long) pos;
95 } */
96 struct sys_mmap_args /* {
97 syscallarg(void *) addr;
98 syscallarg(size_t) len;
99 syscallarg(int) prot;
100 syscallarg(int) flags;
101 syscallarg(int) fd;
102 syscallarg(long) pad;
103 syscallarg(off_t) pos;
104 } */ nargs;
105 static const char cvtbsdprot[8] = {
106 0,
107 PROT_EXEC,
108 PROT_WRITE,
109 PROT_EXEC|PROT_WRITE,
110 PROT_READ,
111 PROT_EXEC|PROT_READ,
112 PROT_WRITE|PROT_READ,
113 PROT_EXEC|PROT_WRITE|PROT_READ,
114 };
115#define OMAP_ANON 0x0002
116#define OMAP_COPY 0x0020
117#define OMAP_SHARED 0x0010
118#define OMAP_FIXED 0x0100
119#define OMAP_INHERIT 0x0800
120
121 SCARG(&nargs, addr) = SCARG(uap, addr);
122 SCARG(&nargs, len) = SCARG(uap, len);
123 /* Note: index using prot is sign-safe due to mask */
124 SCARG(&nargs, prot) = cvtbsdprot[SCARG(uap, prot)&0x7];
125 SCARG(&nargs, flags) = 0;
126 if (SCARG(uap, flags) & OMAP_ANON)
127 SCARG(&nargs, flags) |= MAP_ANON;
128 if (SCARG(uap, flags) & OMAP_SHARED)
129 SCARG(&nargs, flags) |= MAP_SHARED;
130 else
131 SCARG(&nargs, flags) |= MAP_PRIVATE;
132 if (SCARG(uap, flags) & OMAP_COPY) {
133 SCARG(&nargs, flags) |= MAP_PRIVATE;
134#if defined(COMPAT_10) && defined(__i386__)
135 /*
136 * Ancient kernel on x86 did not obey PROT_EXEC on i386 at least
137 * and ld.so did not turn it on. We take care of this on amd64
138 * in compat32.
139 */
140 SCARG(&nargs, prot) |= PROT_EXEC;
141#endif
142 }
143 if (SCARG(uap, flags) & OMAP_FIXED)
144 SCARG(&nargs, flags) |= MAP_FIXED;
145 if (SCARG(uap, flags) & OMAP_INHERIT)
146 SCARG(&nargs, flags) |= MAP_INHERIT;
147 SCARG(&nargs, fd) = SCARG(uap, fd);
148 SCARG(&nargs, pos) = SCARG(uap, pos);
149 return (sys_mmap(l, &nargs, retval));
150}

152int
153vm_43_init(void)
154{
155
156 return syscall_establish(NULL, vm_43_syscalls);
157}

159int
160vm_43_fini(void)
161{
162
163 return syscall_disestablish(NULL, vm_43_syscalls);
