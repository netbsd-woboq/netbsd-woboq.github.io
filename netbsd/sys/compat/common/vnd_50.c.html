98static int compat_50_vndioctl(u_long, struct lwp *, void *, int, struct vattr *,
99 int (*)(struct lwp *, void *, int, struct vattr *));
100
101static int
102compat_50_vndioctl(u_long cmd, struct lwp *l, void *data, int unit,
103 struct vattr *vattr_p,
104 int (*get)(struct lwp *, void *, int, struct vattr *))
105{
106 struct vnd_user50 *vnu = data;
107 int error;
108
109 if (cmd != VNDIOCGET50)
110 return EPASSTHROUGH;
111
112 error = (*get)(l, data, unit, vattr_p);
113 if (error != 0)
114 return error;
115
116 vnu->vnu_dev = vattr_p->va_fsid;
117 vnu->vnu_ino = vattr_p->va_fileid;
118 return 0;
119}
120
121void
122vnd_50_init(void)
123{
124
125 MODULE_HOOK_SET(compat_vndioctl_50_hook, "vnd_50", compat_50_vndioctl);
126}
127
128void
129vnd_50_fini(void)
130{
131
132 MODULE_HOOK_UNSET(compat_vndioctl_50_hook);
133}
