62int
63linux_ioctl_blkio(struct lwp *l, const struct linux_sys_ioctl_args *uap,
64 register_t *retval)
65{
66 u_long com;
67 long size;
68 int error;
69 file_t *fp;
70 int (*ioctlf)(file_t *, u_long, void *);
71 struct partinfo pi;
72 struct disklabel label;
73
74 if ((fp = fd_getfile(SCARG(uap, fd))) == NULL)
75 return (EBADF);
76
77 error = 0;
78 ioctlf = fp->f_ops->fo_ioctl;
79 com = SCARG(uap, com);
80
81 switch (com) {
82 case LINUX_BLKGETSIZE:
83 /*
84 * Try to get the partition size of this device. If that
85 * fails, it may be a disk without label; try to get
86 * the default label and compute the size from it.
87 */
88 error = ioctlf(fp, DIOCGPARTINFO, &pi);
89 if (error != 0) {
90 error = ioctlf(fp, DIOCGDINFO, &label);
91 if (error != 0)
92 break;
93 size = label.d_nsectors * label.d_ntracks *
94 label.d_ncylinders;
95 } else
96 size = pi.pi_size;
97 error = copyout(&size, SCARG(uap, data), sizeof size);
98 break;
99 case LINUX_BLKSECTGET:
100 error = ioctlf(fp, DIOCGDINFO, &label);
101 if (error != 0)
102 break;
103 error = copyout(&label.d_secsize, SCARG(uap, data),
104 sizeof label.d_secsize);
105 break;
106 case LINUX_BLKROSET:
107 case LINUX_BLKROGET:
108 case LINUX_BLKRRPART:
109 case LINUX_BLKFLSBUF:
110 case LINUX_BLKRASET:
111 case LINUX_BLKRAGET:
112 case LINUX_BLKFRASET:
113 case LINUX_BLKFRAGET:
114 case LINUX_BLKSECTSET:
115 case LINUX_BLKSSZGET:
116 case LINUX_BLKPG:
117 default:
118 error = ENOTTY;
119 }
120
121 fd_putfile(SCARG(uap, fd));
122
123 return error;
124}
