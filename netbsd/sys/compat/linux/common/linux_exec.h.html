60/* Defines for a.out executables */
61#define LINUX_AOUT_HDR_SIZE (sizeof (struct exec))
62#define LINUX_AOUT_AUX_ARGSIZ 8
63
64#define LINUX_N_MAGIC(ep) ((ep)->a_midmag & 0xffff)
65#define LINUX_N_MACHTYPE(ep) (((ep)->a_midmag >> 16) & 0xff)
66
67#define LINUX_N_TXTOFF(x,m) \
68 ((m) == ZMAGIC ? 1024 : ((m) == QMAGIC ? 0 : sizeof (struct exec)))
69
70#define LINUX_N_DATOFF(x,m) (LINUX_N_TXTOFF(x,m) + (x).a_text)
71
72#define LINUX_N_TXTADDR(x,m) ((m) == QMAGIC ? PAGE_SIZE : 0)
73
74#define LINUX__N_SEGMENT_ROUND(x) (((x) + PAGE_SIZE - 1) & ~(PAGE_SIZE - 1))
75
76#define LINUX__N_TXTENDADDR(x,m) (LINUX_N_TXTADDR(x,m)+(x).a_text)
77
78#define LINUX_N_DATADDR(x,m) \
79 ((m)==OMAGIC? (LINUX__N_TXTENDADDR(x,m)) \
80 : (LINUX__N_SEGMENT_ROUND (LINUX__N_TXTENDADDR(x,m))))
81
82#define LINUX_N_BSSADDR(x,m) (LINUX_N_DATADDR(x,m) + (x).a_data)
