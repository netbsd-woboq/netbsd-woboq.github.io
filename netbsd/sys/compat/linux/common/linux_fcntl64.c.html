56conv_linux_flock(linux, flock64)
57
58int
59linux_sys_fcntl64(struct lwp *l, const struct linux_sys_fcntl64_args *uap, register_t *retval)
60{
61 /* {
62 syscallarg(int) fd;
63 syscallarg(int) cmd;
64 syscallarg(void *) arg;
65 } */
66 int cmd = SCARG(uap, cmd);
67
68 switch (cmd) {
69 case LINUX_F_GETLK64:
70 do_linux_getlk(SCARG(uap, fd), cmd, SCARG(uap, arg),
71 linux, flock64);
72 case LINUX_F_SETLK64:
73 case LINUX_F_SETLKW64:
74 do_linux_setlk(SCARG(uap, fd), cmd, SCARG(uap, arg),
75 linux, flock64, LINUX_F_SETLK64);
76 default:
77 return linux_sys_fcntl(l, (const void *)uap, retval);
78 }
79}
