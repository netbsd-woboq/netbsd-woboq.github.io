66int
67linux_ioctl_fdio(struct lwp *l, const struct linux_sys_ioctl_args *uap,
68 register_t *retval)
69{
70 file_t *fp;
71 int error;
72 int (*ioctlf)(file_t *, u_long, void *);
73 u_long com;
74 struct fdformat_parms fparams;
75 struct linux_floppy_struct lflop;
76 struct linux_floppy_drive_struct ldrive;
77
78 com = (u_long)SCARG(uap, data);
79
80 if ((fp = fd_getfile(SCARG(uap, fd))) == NULL)
81 return (EBADF);
82
83 com = SCARG(uap, com);
84 ioctlf = fp->f_ops->fo_ioctl;
85
86 retval[0] = error = 0;
87
88 switch (com) {
89 case LINUX_FDMSGON:
90 case LINUX_FDMSGOFF:
91 case LINUX_FDTWADDLE:
92 case LINUX_FDCLRPRM:
93 /* whatever you say */
94 break;
95 case LINUX_FDPOLLDRVSTAT:
96 /*
97 * Just fill in some innocent defaults.
98 */
99 memset(&ldrive, 0, sizeof ldrive);
100 ldrive.fd_ref = 1;
101 ldrive.maxblock = 2;
102 ldrive.maxtrack = ldrive.track = 1;
103 ldrive.flags = LINUX_FD_DISK_WRITABLE;
104 error = copyout(&ldrive, SCARG(uap, data), sizeof ldrive);
105 break;
106 case LINUX_FDGETPRM:
107 error = ioctlf(fp, FDIOCGETFORMAT, &fparams);
108 if (error != 0)
109 break;
110 lflop.size = fparams.ncyl * fparams.nspt * fparams.ntrk;
111 lflop.sect = fparams.nspt;
112 lflop.head = fparams.ntrk;
113 lflop.track = fparams.ncyl;
114 lflop.stretch = fparams.stepspercyl == 2 ? 1 : 0;
115 lflop.fmt_gap = fparams.gaplen;
116 /*
117 * XXXX Use knowledge of floppy for these fields.
118 * FDIOCGETFORMAT does not provide enough information.
119 * It would be better to have FDIOCGETFORMAT return more
120 * information.
121 */
122 switch (fparams.xfer_rate) {
123 case 500 * 1024:
124 lflop.spec1 = 0xcf;
125 lflop.gap = 0x1b;
126 lflop.rate = FDC_500KBPS;
127 break;
128 case 300 * 1024:
129 lflop.spec1 = 0xdf;
130 lflop.gap = 0x23;
131 lflop.rate = FDC_300KBPS;
132 break;
133 case 250 * 1024:
134 lflop.spec1 = 0xdf;
135 lflop.gap = 0x2a;
136 lflop.rate = FDC_250KBPS;
137 break;
138 }
139 error = copyout(&lflop, SCARG(uap, data), sizeof lflop);
140 break;
141 case LINUX_FDSETPRM:
142 /*
143 * Should use FDIOCSETFORMAT here, iff its interface
144 * is extended.
145 */
146 case LINUX_FDDEFPRM:
147 case LINUX_FDFMTBEG:
148 case LINUX_FDFMTTRK:
149 case LINUX_FDFMTEND:
150 case LINUX_FDSETEMSGTRESH:
151 case LINUX_FDFLUSH:
152 case LINUX_FDSETMAXERRS:
153 case LINUX_FDGETMAXERRS:
154 case LINUX_FDGETDRVTYP:
155 case LINUX_FDSETDRVPRM:
156 case LINUX_FDGETDRVPRM:
157 case LINUX_FDGETDRVSTAT:
158 case LINUX_FDRESET:
159 case LINUX_FDGETFDCSTAT:
160 case LINUX_FDWERRORCLR:
161 case LINUX_FDWERRORGET:
162 case LINUX_FDRAWCMD:
163 case LINUX_FDEJECT:
164 default:
165 error = EINVAL;
166 }
167
168 fd_putfile(SCARG(uap, fd));
169
170 return 0;
171}
