37#define LINUX_FUTEX_WAIT 0
38#define LINUX_FUTEX_WAKE 1
39#define LINUX_FUTEX_FD 2
40#define LINUX_FUTEX_REQUEUE 3
41#define LINUX_FUTEX_CMP_REQUEUE 4
42#define LINUX_FUTEX_WAKE_OP 5
43#define LINUX_FUTEX_LOCK_PI 6
44#define LINUX_FUTEX_UNLOCK_PI 7
45#define LINUX_FUTEX_TRYLOCK_PI 8
46#define LINUX_FUTEX_WAIT_BITSET 9
47#define LINUX_FUTEX_WAKE_BITSET 10
48#define LINUX_FUTEX_WAIT_REQUEUE_PI 11
49#define LINUX_FUTEX_CMP_REQUEUE_PI 12

51#define LINUX_FUTEX_PRIVATE_FLAG 128
52#define LINUX_FUTEX_CLOCK_REALTIME 256
53#define LINUX_FUTEX_CMD_MASK \
54 (~(LINUX_FUTEX_PRIVATE_FLAG|LINUX_FUTEX_CLOCK_REALTIME))

56#define FUTEX_OP_SET 0
57#define FUTEX_OP_ADD 1
58#define FUTEX_OP_OR 2
59#define FUTEX_OP_ANDN 3
60#define FUTEX_OP_XOR 4
61#define FUTEX_OP_OPARG_SHIFT 8
62
63#define FUTEX_OP_CMP_EQ 0
64#define FUTEX_OP_CMP_NE 1
65#define FUTEX_OP_CMP_LT 2
66#define FUTEX_OP_CMP_LE 3
67#define FUTEX_OP_CMP_GT 4
68#define FUTEX_OP_CMP_GE 5

70struct linux_robust_list {

72};

74struct linux_robust_list_head {

78};
