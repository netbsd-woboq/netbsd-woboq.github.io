66int
67linux_ioctl_hdio(struct lwp *l, const struct linux_sys_ioctl_args *uap,
68 register_t *retval)
69{
70 u_long com;
71 int error, error1;
72 struct file *fp;
73 int (*ioctlf)(struct file *, u_long, void *);
74 struct atareq req;
75 struct disklabel label;
76 struct partinfo pi;
77 struct linux_hd_geometry hdg;
78 struct linux_hd_big_geometry hdg_big;
79
80 if ((fp = fd_getfile(SCARG(uap, fd))) == NULL)
81 return (EBADF);
82
83 com = SCARG(uap, com);
84 ioctlf = fp->f_ops->fo_ioctl;
85 retval[0] = error = 0;
86
87 switch (com) {
88 case LINUX_HDIO_OBSOLETE_IDENTITY:
89 case LINUX_HDIO_GET_IDENTITY:
90 req.flags = ATACMD_READ;
91 req.command = WDCC_IDENTIFY;
92 req.databuf = SCARG(uap, data);
93 /*
94 * 142 is the size of the old structure used by Linux,
95 * which doesn't seem to be defined anywhere anymore.
96 * The new function should return the entire 512 byte area.
97 */
98 req.datalen = com == LINUX_HDIO_GET_IDENTITY ? 512 : 142;
99 req.timeout = 1000;
100 error = ioctlf(fp, ATAIOCCOMMAND, &req);
101 if (error != 0)
102 break;
103 if (req.retsts != ATACMD_OK)
104 error = EIO;
105 break;
106 case LINUX_HDIO_GETGEO:
107 error = linux_machdepioctl(l, uap, retval);
108 if (error == 0)
109 break;
110 error = ioctlf(fp, DIOCGDINFO, &label);
111 error1 = ioctlf(fp, DIOCGPARTINFO, &pi);
112 if (error != 0 && error1 != 0) {
113 error = error1;
114 break;
115 }
116 hdg.start = error1 != 0 ? pi.pi_offset : 0;
117 hdg.heads = label.d_ntracks;
118 hdg.cylinders = label.d_ncylinders;
119 hdg.sectors = label.d_nsectors;
120 error = copyout(&hdg, SCARG(uap, data), sizeof hdg);
121 break;
122 case LINUX_HDIO_GETGEO_BIG:
123 error = linux_machdepioctl(l, uap, retval);
124 if (error == 0)
125 break;
126 /* FALLTHROUGH */
127 case LINUX_HDIO_GETGEO_BIG_RAW:
128 error = ioctlf(fp, DIOCGDINFO, &label);
129 error1 = ioctlf(fp, DIOCGPARTINFO, &pi);
130 if (error != 0 && error1 != 0) {
131 error = error1;
132 break;
133 }
134 hdg_big.start = error1 != 0 ? pi.pi_offset : 0;
135 hdg_big.heads = label.d_ntracks;
136 hdg_big.cylinders = label.d_ncylinders;
137 hdg_big.sectors = label.d_nsectors;
138 error = copyout(&hdg_big, SCARG(uap, data), sizeof hdg_big);
139 break;
140 case LINUX_HDIO_GET_UNMASKINTR:
141 case LINUX_HDIO_GET_MULTCOUNT:
142 case LINUX_HDIO_GET_KEEPSETTINGS:
143 case LINUX_HDIO_GET_32BIT:
144 case LINUX_HDIO_GET_NOWERR:
145 case LINUX_HDIO_GET_DMA:
146 case LINUX_HDIO_GET_NICE:
147 case LINUX_HDIO_DRIVE_RESET:
148 case LINUX_HDIO_TRISTATE_HWIF:
149 case LINUX_HDIO_DRIVE_TASK:
150 case LINUX_HDIO_DRIVE_CMD:
151 case LINUX_HDIO_SET_MULTCOUNT:
152 case LINUX_HDIO_SET_UNMASKINTR:
153 case LINUX_HDIO_SET_KEEPSETTINGS:
154 case LINUX_HDIO_SET_32BIT:
155 case LINUX_HDIO_SET_NOWERR:
156 case LINUX_HDIO_SET_DMA:
157 case LINUX_HDIO_SET_PIO_MODE:
158 case LINUX_HDIO_SCAN_HWIF:
159 case LINUX_HDIO_SET_NICE:
160 case LINUX_HDIO_UNREGISTER_HWIF:
161 error = EINVAL;
162 }
163
164 fd_putfile(SCARG(uap, fd));
165
166 return error;
167}
