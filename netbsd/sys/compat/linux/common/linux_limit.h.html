33#ifndef _LINUX_LIMIT_H_
34#define _LINUX_LIMIT_H_
35
36static int linux_to_bsd_limit(int);
37
38#ifdef LINUX_LARGEFILE64
39#define bsd_to_linux_rlimit1(l, b, f) \
40 (l)->f = ((b)->f == RLIM_INFINITY || \
41 ((b)->f & 0x8000000000000000UL) != 0) ? \
42 LINUX_RLIM_INFINITY : (b)->f
43#else
44#define bsd_to_linux_rlimit1(l, b, f) \
45 (l)->f = ((b)->f == RLIM_INFINITY || \
46 ((b)->f & 0xffffffff00000000ULL) != 0) ? \
47 LINUX_RLIM_INFINITY : (int32_t)(b)->f
48#endif
49#define bsd_to_linux_rlimit(l, b) \
50 bsd_to_linux_rlimit1(l, b, rlim_cur); \
51 bsd_to_linux_rlimit1(l, b, rlim_max)
52
53#define linux_to_bsd_rlimit1(b, l, f) \
54 (b)->f = (l)->f == LINUX_RLIM_INFINITY ? RLIM_INFINITY : (l)->f
55#define linux_to_bsd_rlimit(b, l) \
56 linux_to_bsd_rlimit1(b, l, rlim_cur); \
57 linux_to_bsd_rlimit1(b, l, rlim_max)
58
59#define linux32_to_bsd_rlimit1(b, l, f) \
60 (b)->f = (l)->f == LINUX32_RLIM_INFINITY ? RLIM_INFINITY : (uint32_t)(l)->f
61#define linux32_to_bsd_rlimit(b, l) \
62 linux32_to_bsd_rlimit1(b, l, rlim_cur); \
63 linux32_to_bsd_rlimit1(b, l, rlim_max)
64
65static int
66linux_to_bsd_limit(int lim)
67{
68 switch (lim) {
69 case LINUX_RLIMIT_CPU:
70 return RLIMIT_CPU;
71 case LINUX_RLIMIT_FSIZE:
72 return RLIMIT_FSIZE;
73 case LINUX_RLIMIT_DATA:
74 return RLIMIT_DATA;
75 case LINUX_RLIMIT_STACK:
76 return RLIMIT_STACK;
77 case LINUX_RLIMIT_CORE:
78 return RLIMIT_CORE;
79 case LINUX_RLIMIT_RSS:
80 return RLIMIT_RSS;
81 case LINUX_RLIMIT_NPROC:
82 return RLIMIT_NPROC;
83 case LINUX_RLIMIT_NOFILE:
84 return RLIMIT_NOFILE;
85 case LINUX_RLIMIT_MEMLOCK:
86 return RLIMIT_MEMLOCK;
87 case LINUX_RLIMIT_AS:
88 return RLIMIT_AS;
89 case LINUX_RLIMIT_LOCKS:
90 return -EOPNOTSUPP;
91 default:
92 return -EINVAL;
93 }
94}
