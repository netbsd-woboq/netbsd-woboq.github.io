54/*
55 * This appears to be part of a Linux attempt to switch to 64 bits file sizes.
56 */
57int
58linux_sys_llseek(struct lwp *l, const struct linux_sys_llseek_args *uap, register_t *retval)
59{
60 /* {
61 syscallarg(int) fd;
62 syscallarg(uint32_t) ohigh;
63 syscallarg(uint32_t) olow;
64 syscallarg(void *) res;
65 syscallarg(int) whence;
66 } */
67 struct sys_lseek_args bla;
68 int error;
69 off_t off;
70
71 off = SCARG(uap, olow) | (((off_t) SCARG(uap, ohigh)) << 32);
72
73 SCARG(&bla, fd) = SCARG(uap, fd);
74 SCARG(&bla, offset) = off;
75 SCARG(&bla, whence) = SCARG(uap, whence);
76
77 if ((error = sys_lseek(l, &bla, retval)))
78 return error;
79
80 if ((error = copyout(retval, SCARG(uap, res), sizeof (off_t))))
81 return error;
82
83 retval[0] = 0;
84 return 0;
85}
