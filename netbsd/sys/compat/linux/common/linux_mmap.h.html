75/*
76 * Ensure the linux_sys_mmap syscall appears to exist:
77 */
78struct linux_sys_mmap_args;
79struct linux_sys_mmap_args {
80 syscallarg(unsigned long) addr;
81 syscallarg(size_t) len;
82 syscallarg(int) prot;
83 syscallarg(int) flags;
84 syscallarg(int) fd;
85 syscallarg(linux_off_t) offset;
86};
87
88/*
89 * Same arguments, but different meaning for "offset".
90 * See linux_misc.c:linux_sys_mmap2().
91 */
92#define linux_sys_mmap2_args linux_sys_mmap_args
93
94#ifdef _KERNEL
95__BEGIN_DECLS
96int linux_sys_mmap(struct lwp *p, const struct linux_sys_mmap_args *v, register_t *retval);
97int linux_sys_mmap2(struct lwp *p, const struct linux_sys_mmap2_args *v, register_t *retval);
98
99__END_DECLS
100#endif /* !_KERNEL */
101
102#endif /* !_LINUX_MMAP_H */
