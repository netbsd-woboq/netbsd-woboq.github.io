56/* Used on: arm, i386, m68k, ppc, amd64 */
57/* Not used on: alpha, mips, sparc, sparc64 */
58/* Alpha, mips, sparc and sparc64 pass one of the fds in a register */
59
60int
61linux_sys_pipe(struct lwp *l, const struct linux_sys_pipe_args *uap,
62 register_t *retval)
63{
64 /* {
65 syscallarg(int *) pfds;
66 } */
67 int fd[2], error;
68
69 if ((error = pipe1(l, fd, 0)))
70 return error;
71
72 if ((error = copyout(fd, SCARG(uap, pfds), sizeof(fd))) != 0)
73 return error;
74 retval[0] = 0;
75 return 0;
76}
77
78int
79linux_sys_pipe2(struct lwp *l, const struct linux_sys_pipe2_args *uap,
80 register_t *retval)
81{
82 /* {
83 syscallarg(int *) pfds;
84 syscallarg(int) flags;
85 } */
86 int fd[2], error, flags;
87
88 flags = linux_to_bsd_ioflags(SCARG(uap, flags));
89 if ((flags & ~(O_CLOEXEC|O_NONBLOCK)) != 0)
90 return EINVAL;
91
92 if ((error = pipe1(l, fd, flags)))
93 return error;
94
95 if ((error = copyout(fd, SCARG(uap, pfds), sizeof(fd))) != 0)
96 return error;
97 retval[0] = 0;
98 return 0;
99}
