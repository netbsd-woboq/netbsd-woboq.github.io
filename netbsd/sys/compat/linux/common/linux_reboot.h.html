32#ifndef _LINUX_REBOOT_H
33#define _LINUX_REBOOT_H
34
35/*
36 * Magic values that Linux requires for the reboot syscall.
37 */
38#define LINUX_REBOOT_MAGIC1 0xfee1dead
39#define LINUX_REBOOT_MAGIC2 672274793
40#define LINUX_REBOOT_MAGIC2A 85072278
41#define LINUX_REBOOT_MAGIC2B 369367448
42
43/*
44 * Reboot commands.
45 * RESTART Normal reboot
46 * HALT
47 * CAD_ON Controls action of pressing ctrl-alt-del.
48 * CAD_OFF Not supported by NetBSD.
49 * POWER_OFF
50 * RESTART2 Reboot and pass a command string.
51 */
52#define LINUX_REBOOT_CMD_RESTART 0x01234567
53#define LINUX_REBOOT_CMD_HALT 0xCDEF0123
54#define LINUX_REBOOT_CMD_CAD_ON 0x89ABCDEF
55#define LINUX_REBOOT_CMD_CAD_OFF 0x00000000
56#define LINUX_REBOOT_CMD_POWER_OFF 0x4321FEDC
57#define LINUX_REBOOT_CMD_RESTART2 0xA1B2C3D4
58
59#endif /* !_LINUX_REBOOT_H */
