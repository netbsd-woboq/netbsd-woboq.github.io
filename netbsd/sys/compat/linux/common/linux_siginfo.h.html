53/* si_code values for non signal */
54#define LINUX_SI_USER 0
55#define LINUX_SI_KERNEL 0x80
56#define LINUX_SI_QUEUE -1
57#ifndef LINUX_SI_TIMER /* all except mips */
58#define LINUX_SI_TIMER -2
59#define LINUX_SI_MESGQ -3
60#define LINUX_SI_ASYNCIO -4
61#endif /* LINUX_SI_TIMER */
62#define LINUX_SI_SIGIO -5
63#define LINUX_SI_TKILL -6
64#define LINUX_SI_DETHREAD -7
65
66/* si_code values for SIGILL */
67#define LINUX_ILL_ILLOPC 1
68#define LINUX_ILL_ILLOPN 2
69#define LINUX_ILL_ILLADR 3
70#define LINUX_ILL_ILLTRP 4
71#define LINUX_ILL_PRVOPC 5
72#define LINUX_ILL_PRVREG 6
73#define LINUX_ILL_COPROC 7
74#define LINUX_ILL_BADSTK 8
75
76/* si_code values for SIGFPE */
77#define LINUX_FPE_INTDIV 1
78#define LINUX_FPE_INTOVF 2
79#define LINUX_FPE_FLTDIV 3
80#define LINUX_FPE_FLTOVF 4
81#define LINUX_FPE_FLTUND 5
82#define LINUX_FPE_FLTRES 6
83#define LINUX_FPE_FLTINV 7
84#define LINUX_FPE_FLTSUB 8
85
86/* si_code values for SIGSEGV */
87#define LINUX_SEGV_MAPERR 1
88#define LINUX_SEGV_ACCERR 2
89
90/* si_code values for SIGBUS */
91#define LINUX_BUS_ADRALN 1
92#define LINUX_BUS_ADRERR 2
93#define LINUX_BUS_OBJERR 3
94
95/* si_code values for SIGTRAP */
96#define LINUX_TRAP_BRKPT 1
97#define LINUX_TRAP_TRACE 2
98
99/* si_code values for SIGCHLD */
100#define LINUX_CLD_EXITED 1
101#define LINUX_CLD_KILLED 2
102#define LINUX_CLD_DUMPED 3
103#define LINUX_CLD_TRAPPED 4
104#define LINUX_CLD_STOPPED 5
105#define LINUX_CLD_CONTINUED 6
106
107/* si_code values for SIGPOLL */
108#define LINUX_POLL_IN 1
109#define LINUX_POLL_OUT 2
110#define LINUX_POLL_MSG 3
111#define LINUX_POLL_ERR 4
112#define LINUX_POLL_PRI 5
113#define LINUX_POLL_HUP 6
