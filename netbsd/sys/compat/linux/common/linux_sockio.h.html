52#define LINUX_IFNAMSIZ 16
53
54struct linux_ifmap {
55 unsigned long mem_start;
56 unsigned long mem_end;
57 unsigned short base_addr;
58 unsigned char irq;
59 unsigned char dma;
60 unsigned char port;
61};
62
63struct linux_ifreq {
64 union {
65 char ifrn_name[LINUX_IFNAMSIZ]; /* if name, e.g. "en0" */
66 } ifr_ifrn;
67 union {
68 struct osockaddr ifru_addr;
69 struct osockaddr ifru_hwaddr;
70 struct linux_ifmap ifru_map;
71 int ifru_ifindex;
72 } ifr_ifru;
73#define ifr_name ifr_ifrn.ifrn_name /* interface name */
74#define ifr_addr ifr_ifru.ifru_addr /* address */
75#define ifr_hwaddr ifr_ifru.ifru_hwaddr /* MAC address */
76#define ifr_map ifr_ifru.ifru_map /* device map */
77};
78
79struct linux_ifconf {
80 int ifc_len;
81 union {
82 char *ifcu_buf;
83 struct linux_ifreq *ifcu_req;
84 } ifc_ifcu;
85};
86
87#define ifc_buf ifc_ifcu.ifcu_buf
88#define ifc_req ifc_ifcu.ifcu_req
