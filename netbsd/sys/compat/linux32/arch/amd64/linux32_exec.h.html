57static char linux32_kernel_vsyscall[] = {
58 0x55, /* push %ebp */ \
59 0x89, 0xcd, /* mov %ecx,%ebp */ \
60 0x0f, 0x05, /* syscall */ \
61 0xb9, 0x7b, 0x00, 0x00, 0x00, /* mov $0x7b,%ecx */ \
62 0x8e, 0xd1, /* movl %ecx,%ss */ \
63 0x89, 0xe9, /* mov %ebp,%ecx */ \
64 0x5d, /* pop %ebp */ \
65 0xc3, /* ret */ \
66};
