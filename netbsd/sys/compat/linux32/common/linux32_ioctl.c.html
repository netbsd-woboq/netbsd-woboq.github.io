62int
63linux32_sys_ioctl(struct lwp *l, const struct linux32_sys_ioctl_args *uap, register_t *retval)
64{
65 /* {
66 syscallarg(int) fd;
67 syscallarg(netbsd32_u_long) com;
68 syscallarg(netbsd32_charp) data;
69 } */
70 struct oss_sys_ioctl_args ossuap;
71 int group;
72 int error;
73
74 group = LINUX32_IOCGROUP((int)SCARG(uap, com));
75
76#ifdef DEBUG_LINUX
77 printf("linux32_sys_ioctl(%d, 0x%x/\'%c\', %p)\n", SCARG(uap, fd),
78 SCARG(uap, com), (char)group, SCARG_P32(uap, data));
79#endif
80
81 switch(group) {
82 case 'T':
83 error = linux32_ioctl_termios(l, uap, retval);
84 break;
85 case 'M':
86 case 'Q':
87 case 'P':
88 SCARG(&ossuap, fd) = SCARG(uap, fd);
89 SCARG(&ossuap, com) = (u_long)SCARG(uap, com);
90 SCARG(&ossuap, data) = SCARG_P32(uap, data);
91 switch (group) {
92 case 'M':
93 error = oss_ioctl_mixer(l, &ossuap, retval);
94 break;
95 case 'Q':
96 error = oss_ioctl_sequencer(l, &ossuap, retval);
97 break;
98 case 'P':
99 error = oss_ioctl_audio(l, &ossuap, retval);
100 break;
101 default:
102 error = EINVAL; /* shutup gcc */
103 break;
104 }
105 break;
106 case 'V': /* video4linux2 */
107 case 'd': /* drm */
108 {
109 struct sys_ioctl_args ua;
110 u_long com = 0;
111 if (SCARG(uap, com) & IOC_IN)
112 com |= IOC_OUT;
113 if (SCARG(uap, com) & IOC_OUT)
114 com |= IOC_IN;
115 SCARG(&ua, fd) = SCARG(uap, fd);
116 SCARG(&ua, com) = SCARG(uap, com);
117 SCARG(&ua, com) &= ~IOC_DIRMASK;
118 SCARG(&ua, com) |= com;
119 SCARG(&ua, data) = SCARG_P32(uap, data);
120 error = sys_ioctl(l, (const void *)&ua, retval);
121 break;
122 }
123 case 0x89:
124 error = linux32_ioctl_socket(l, uap, retval);
125 break;
126 default:
127 printf("Not yet implemented ioctl group \'%c\'\n", group);
128 error = EINVAL;
129 break;
130 }
131
132 if (error == EPASSTHROUGH)
133 error = EINVAL;
134
135 return error;
136}
