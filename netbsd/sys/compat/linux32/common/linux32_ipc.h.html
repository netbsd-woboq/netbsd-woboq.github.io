29#ifndef _LINUX32_IPC_H
30#define _LINUX32_IPC_H
31
32typedef int32_t linux32_key_t;
33
34struct linux32_ipc_perm {
35 linux32_key_t l_key;
36 ushort l_uid;
37 ushort l_gid;
38 ushort l_cuid;
39 ushort l_cgid;
40 ushort l_mode;
41 ushort l_seq;
42};
43
44struct linux32_ipc64_perm {
45 linux32_key_t l_key;
46 uint l_uid;
47 uint l_gid;
48 uint l_cuid;
49 uint l_cgid;
50 ushort l_mode;
51 ushort l___pad1;
52 ushort l_seq;
53 ushort l___pad2;
54 netbsd32_u_long l___unused1;
55 netbsd32_u_long l___unused2;
56};
57
58struct linux32_msqid_ds {
59 struct linux32_ipc_perm l_msg_perm;
60 netbsd32_pointer_t l_msg_first;
61 netbsd32_pointer_t l_msg_last;
62 linux32_time_t l_msg_stime;
63 linux32_time_t l_msg_rtime;
64 linux32_time_t l_msg_ctime;
65 netbsd32_pointer_t l_wwait; /* Linux internal */
66 netbsd32_pointer_t l_rwait; /* Linux internal */
67 ushort l_msg_cbytes;
68 ushort l_msg_qnum;
69 ushort l_msg_qbytes;
70 ushort l_msg_lspid;
71 ushort l_msg_lrpid;
72};
73
74struct linux32_msqid64_ds {
75 struct linux32_ipc64_perm l_msg_perm;
76 linux32_time_t l_msg_stime;
77 netbsd32_u_long l___unused1;
78 linux32_time_t l_msg_rtime;
79 netbsd32_u_long l___unused2;
80 linux32_time_t l_msg_ctime;
81 netbsd32_u_long l___unused3;
82 netbsd32_u_long l_msg_cbytes;
83 netbsd32_u_long l_msg_qnum;
84 netbsd32_u_long l_msg_qbytes;
85 int l_msg_lspid;
86 int l_msg_lrpid;
87 netbsd32_u_long l___unused4;
88 netbsd32_u_long l___unused5;
89
90};

95#define LINUX32_IPC_INFO 3
96
97#define LINUX32_IPC_64 0x0100

99#endif /* _LINUX32_IPC_H */
