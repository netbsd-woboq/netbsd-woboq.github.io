75int
76linux32_sys_old_mmap(struct lwp *l, const struct linux32_sys_old_mmap_args *uap, register_t *retval)
77{
78 /* {
79 syscallarg(linux32_oldmmapp) lmp;
80 } */
81 struct linux_sys_old_mmap_args ua;
82
83 NETBSD32TOP_UAP(lmp, struct linux_oldmmap);
84 return linux_sys_old_mmap(l, &ua, retval);
85}

87int
88linux32_sys_mprotect(struct lwp *l, const struct linux32_sys_mprotect_args *uap, register_t *retval)
89{
90 /* {
91 syscallarg(netbsd32_voidp) start;
92 syscallarg(netbsd32_long) len;
93 syscallarg(int) prot;
94 } */
95 struct linux_sys_mprotect_args ua;
96
97 NETBSD32TOP_UAP(start, void);
98 NETBSD32TOX_UAP(len, long);
99 NETBSD32TO64_UAP(prot);
100 return (linux_sys_mprotect(l, &ua, retval));
101}

103int
104linux32_sys_mremap(struct lwp *l, const struct linux32_sys_mremap_args *uap, register_t *retval)
105{
106 /* {
107 syscallarg(netbsd32_voidp) old_address;
108 syscallarg(netbsd32_size_t) old_size;
109 syscallarg(netbsd32_size_t) new_size;
110 syscallarg(netbsd32_u_long) flags;
111 } */
112 struct linux_sys_mremap_args ua;
113
114 NETBSD32TOP_UAP(old_address, void);
115 NETBSD32TOX_UAP(old_size, size_t);
116 NETBSD32TOX_UAP(new_size, size_t);
117 NETBSD32TOX_UAP(flags, u_long);
118
119 return linux_sys_mremap(l, &ua, retval);
120}

123linux32_sys_mmap2(struct lwp *l, const struct linux32_sys_mmap2_args *uap, register_t *retval)
124{
125 /* {
126 syscallarg(netbsd32_u_long) addr;

128 syscallarg(int) prot;
129 syscallarg(int) flags;
130 syscallarg(int) fd;
131 syscallarg(linux32_off_t) offset;
132 } */
133 struct linux_sys_mmap_args ua;
134
135 NETBSD32TOX64_UAP(addr, u_long);
136 NETBSD32TOX64_UAP(len, size_t);
137 NETBSD32TO64_UAP(prot);
138 NETBSD32TO64_UAP(flags);
139 NETBSD32TO64_UAP(fd);
140 NETBSD32TOX64_UAP(offset, linux_off_t);
141
142 return linux_sys_mmap2(l, &ua, retval);
143}
