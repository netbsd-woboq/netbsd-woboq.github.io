86int
87linux32_sysctl_enable(SYSCTLFN_ARGS)
88{
89 struct sysctlnode node;
90 int error, val;
91
92 val = *(int *)rnode->sysctl_data;
93
94 node = *rnode;
95 node.sysctl_data = &val;
96
97 error = sysctl_lookup(SYSCTLFN_CALL(&node));
98 if (error != 0 || newp == NULL)
99 return error;
100
101 if (val == *(int *)rnode->sysctl_data)
102 return 0;
103
104 if (val == 1)
105 error = exec_add(linux32_execsw, __arraycount(linux32_execsw));
106 else if (val == 0)
107 error =
108 exec_remove(linux32_execsw, __arraycount(linux32_execsw));
109 else
110 error = EINVAL;
111
112 if (error)
113 return error;
114
115 *(int *)rnode->sysctl_data = val;
116
117 return 0;
118}

120static int
121compat_linux32_modcmd(modcmd_t cmd, void *arg)
122{
123 int error;
124
125 switch (cmd) {
126 case MODULE_CMD_INIT:
127 linux32_sysctl_init();
128 error = exec_add(linux32_execsw, __arraycount(linux32_execsw));
129 if (error != 0)
130 linux32_sysctl_fini();
131 return error;
132
133 case MODULE_CMD_FINI:
134 error =
135 exec_remove(linux32_execsw, __arraycount(linux32_execsw));
136 if (error)
137 return error;
138 linux32_sysctl_fini();
139 return 0;
140
141 default:
142 return ENOTTY;
143 }
144}
