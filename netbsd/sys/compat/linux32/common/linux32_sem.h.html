29#ifndef _LINUX32_SEM_H
30#define _LINUX32_SEM_H
31
32struct linux32_semid_ds {
33 struct linux32_ipc_perm l_sem_perm;
34 linux32_time_t l_sem_otime;
35 linux32_time_t l_sem_ctime;
36 netbsd32_voidp l_sem_base;
37 netbsd32_voidp l_eventn;
38 netbsd32_voidp l_eventz;
39 netbsd32_voidp l_undo;
40 ushort l_sem_nsems;
41};
42
43struct linux32_semid64_ds {
44 struct linux32_ipc64_perm l_sem_perm;
45 linux32_time_t l_sem_otime;
46 netbsd32_u_long l___unused1;
47 linux32_time_t l_sem_ctime;
48 netbsd32_u_long l___unused2;
49 netbsd32_u_long l_sem_nsems;
50 netbsd32_u_long l___unused3;
51 netbsd32_u_long l___unused4;
52};
53
54typedef netbsd32_pointer_t linux32_semid_dsp;
55
56union linux32_semun {
57 int l_val;
58 linux32_semid_dsp l_buf;
59 netbsd32_u_shortp l_array;
60 netbsd32_voidp l___buf;
61 netbsd32_voidp l___pad;
62};
