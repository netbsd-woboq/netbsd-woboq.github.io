390int
391linux32_getifname(struct lwp *l, register_t *retval, void *data)
392{
393 struct ifnet *ifp;
394 struct linux32_ifreq ifr;
395 int error;
396 int s;
397
398 error = copyin(data, &ifr, sizeof(ifr));
399 if (error)
400 return error;
401
402 s = pserialize_read_enter();
403 ifp = if_byindex(ifr.ifr_ifru.ifru_ifindex);
404 if (ifp == NULL) {
405 pserialize_read_exit(s);
406 return ENODEV;
407 }
408
409 strncpy(ifr.ifr_name, ifp->if_xname, sizeof(ifr.ifr_name));
410 pserialize_read_exit(s);
411
412 return copyout(&ifr, data, sizeof(ifr));
413}

415int
416linux32_getifconf(struct lwp *l, register_t *retval, void *data)
417{
418 struct linux32_ifreq ifr, *ifrp = NULL;
419 struct linux32_ifconf ifc;
420 struct ifnet *ifp;
421 struct sockaddr *sa;
422 struct osockaddr *osa;
423 int space = 0, error;
424 const int sz = (int)sizeof(ifr);
425 bool docopy;
426 int s;
427 int bound;
428 struct psref psref;
429
430 error = copyin(data, &ifc, sizeof(ifc));
431 if (error)
432 return error;
433
434 memset(&ifr, 0, sizeof(ifr));
435 docopy = NETBSD32PTR64(ifc.ifc_req) != NULL;
436 if (docopy) {
437 space = ifc.ifc_len;
438 ifrp = NETBSD32PTR64(ifc.ifc_req);
439 }
440
441 bound = curlwp_bind();
442 s = pserialize_read_enter();
443 IFNET_READER_FOREACH(ifp) {
444 struct ifaddr *ifa;
445 if_acquire(ifp, &psref);
446 pserialize_read_exit(s);
447
448 (void)strncpy(ifr.ifr_name, ifp->if_xname,
449 sizeof(ifr.ifr_name));
450 if (ifr.ifr_name[sizeof(ifr.ifr_name) - 1] != '\0') {
451 error = ENAMETOOLONG;
452 goto release_exit;
453 }
454
455 s = pserialize_read_enter();
456 IFADDR_READER_FOREACH(ifa, ifp) {
457 struct psref psref_ifa;
458 ifa_acquire(ifa, &psref_ifa);
459 pserialize_read_exit(s);
460
461 sa = ifa->ifa_addr;
462 if (sa->sa_family != AF_INET ||
463 sa->sa_len > sizeof(*osa))
464 goto next;
465 memcpy(&ifr.ifr_addr, sa, sa->sa_len);
466 osa = (struct osockaddr *)&ifr.ifr_addr;
467 osa->sa_family = sa->sa_family;
468 if (space >= sz) {
469 error = copyout(&ifr, ifrp, sz);
470 if (error != 0) {
471 ifa_release(ifa, &psref_ifa);
472 goto release_exit;
473 }
474 ifrp++;
475 }
476 space -= sz;
477 next:
478 s = pserialize_read_enter();
479 ifa_release(ifa, &psref_ifa);
480 }
481
482 KASSERT(pserialize_in_read_section());
483 if_release(ifp, &psref);
484 }
485 pserialize_read_exit(s);
486 curlwp_bindx(bound);
487
488 if (docopy)
489 ifc.ifc_len -= space;
490 else
491 ifc.ifc_len = -space;
492
493 return copyout(&ifc, data, sizeof(ifc));
494
495release_exit:
496 if_release(ifp, &psref);
497 curlwp_bindx(bound);
498 return error;
499}

501int
502linux32_getifhwaddr(struct lwp *l, register_t *retval, u_int fd,
503 void *data)
504{
505 struct linux32_ifreq lreq;
506 file_t *fp;
507 struct ifaddr *ifa;
508 struct ifnet *ifp;
509 struct sockaddr_dl *sadl;
510 int error, found;
511 int index, ifnum;
512 int s;
513
514 /*
515 * We can't emulate this ioctl by calling sys_ioctl() to run
516 * SIOCGIFCONF, because the user buffer is not of the right
517 * type to take those results. We can't use kernel buffers to
518 * receive the results, as the implementation of sys_ioctl()
519 * and ifconf() [which implements SIOCGIFCONF] use
520 * copyin()/copyout() which will fail on kernel addresses.
521 *
522 * So, we must duplicate code from sys_ioctl() and ifconf(). Ugh.
523 */
524
525 if ((fp = fd_getfile(fd)) == NULL)
526 return (EBADF);
527
528 KERNEL_LOCK(1, NULL);
529
530 if ((fp->f_flag & (FREAD | FWRITE)) == 0) {
531 error = EBADF;
532 goto out;
533 }
534
535 error = copyin(data, &lreq, sizeof(lreq));
536 if (error)
537 goto out;
538 lreq.ifr_name[LINUX32_IFNAMSIZ-1] = '\0'; /* just in case */
539
540 /*
541 * Try real interface name first, then fake "ethX"
542 */
543 found = 0;
544 s = pserialize_read_enter();
545 IFNET_READER_FOREACH(ifp) {
546 if (found)
547 break;
548 if (strcmp(lreq.ifr_name, ifp->if_xname))
549 /* not this interface */
550 continue;
551 found=1;
552 if (IFADDR_READER_EMPTY(ifp)) {
553 error = ENODEV;
554 goto out;
555 }
556 IFADDR_READER_FOREACH(ifa, ifp) {
557 sadl = satosdl(ifa->ifa_addr);
558 /* only return ethernet addresses */
559 /* XXX what about FDDI, etc. ? */
560 if (sadl->sdl_family != AF_LINK ||
561 sadl->sdl_type != IFT_ETHER)
562 continue;
563 memcpy(&lreq.ifr_hwaddr.sa_data, CLLADDR(sadl),
564 MIN(sadl->sdl_alen,
565 sizeof(lreq.ifr_hwaddr.sa_data)));
566 lreq.ifr_hwaddr.sa_family =
567 sadl->sdl_family;
568 pserialize_read_exit(s);
569
570 error = copyout(&lreq, data, sizeof(lreq));
571 goto out;
572 }
573 }
574 pserialize_read_exit(s);
575
576 if (strncmp(lreq.ifr_name, "eth", 3) == 0) {
577 for (ifnum = 0, index = 3;
578 index < LINUX32_IFNAMSIZ && lreq.ifr_name[index] != '\0';
579 index++) {
580 ifnum *= 10;
581 ifnum += lreq.ifr_name[index] - '0';
582 }
583
584 error = EINVAL; /* in case we don't find one */
585 s = pserialize_read_enter();
586 IFNET_READER_FOREACH(ifp) {
587 memcpy(lreq.ifr_name, ifp->if_xname,
588 MIN(LINUX32_IFNAMSIZ, IFNAMSIZ));
589 IFADDR_READER_FOREACH(ifa, ifp) {
590 sadl = satosdl(ifa->ifa_addr);
591 /* only return ethernet addresses */
592 /* XXX what about FDDI, etc. ? */
593 if (sadl->sdl_family != AF_LINK ||
594 sadl->sdl_type != IFT_ETHER)
595 continue;
596 if (ifnum--)
597 /* not the reqested iface */
598 continue;
599 memcpy(&lreq.ifr_hwaddr.sa_data,
600 CLLADDR(sadl),
601 MIN(sadl->sdl_alen,
602 sizeof(lreq.ifr_hwaddr.sa_data)));
603 lreq.ifr_hwaddr.sa_family =
604 sadl->sdl_family;
605 pserialize_read_exit(s);
606
607 error = copyout(&lreq, data, sizeof(lreq));
608 goto out;
609 }
610 }
611 pserialize_read_exit(s);
612 } else {
613 /* unknown interface, not even an "eth*" name */
614 error = ENODEV;
615 }
616
617out:
618 KERNEL_UNLOCK_ONE(NULL);
619 fd_putfile(fd);
620 return error;
621}

623int
624linux32_ioctl_socket(struct lwp *l, const struct linux32_sys_ioctl_args *uap, register_t *retval)
625{
626 /* {
627 syscallarg(int) fd;
628 syscallarg(u_long) com;
629 syscallarg(void *) data;
630 } */
631 u_long com;
632 int error = 0, isdev = 0, dosys = 1;
633 struct netbsd32_ioctl_args ia;
634 file_t *fp;
635 struct vnode *vp;
636 int (*ioctlf)(file_t *, u_long, void *);
637 struct ioctl_pt pt;
638
639 if ((fp = fd_getfile(SCARG(uap, fd))) == NULL)
640 return (EBADF);
641
642 if (fp->f_type == DTYPE_VNODE) {
643 vp = (struct vnode *)fp->f_data;
644 isdev = vp->v_type == VCHR;
645 }
646
647 /*
648 * Don't try to interpret socket ioctl calls that are done
649 * on a device filedescriptor, just pass them through, to
650 * emulate Linux behaviour. Use PTIOCLINUX so that the
651 * device will only handle these if it's prepared to do
652 * so, to avoid unexpected things from happening.
653 */
654 if (isdev) {
655 dosys = 0;
656 ioctlf = fp->f_ops->fo_ioctl;
657 pt.com = SCARG(uap, com);
658 pt.data = (void *)NETBSD32PTR64(SCARG(uap, data));
659 error = ioctlf(fp, PTIOCLINUX, &pt);
660 /*
661 * XXX hack: if the function returns EJUSTRETURN,
662 * it has stuffed a sysctl return value in pt.data.
663 */
664 if (error == EJUSTRETURN) {
665 retval[0] = (register_t)pt.data;
666 error = 0;
667 }
668 goto out;
669 }
670
671 com = SCARG(uap, com);
672 retval[0] = 0;
673
674 switch (com) {
675 case LINUX_SIOCGIFNAME:
676 error = linux32_getifname(l, retval, SCARG_P32(uap, data));
677 dosys = 0;
678 break;
679 case LINUX_SIOCGIFCONF:
680 error = linux32_getifconf(l, retval, SCARG_P32(uap, data));
681 dosys = 0;
682 break;
683 case LINUX_SIOCGIFFLAGS:
684 SCARG(&ia, com) = OSIOCGIFFLAGS32;
685 break;
686 case LINUX_SIOCSIFFLAGS:
687 SCARG(&ia, com) = OSIOCSIFFLAGS32;
688 break;
689 case LINUX_SIOCGIFADDR:
690 SCARG(&ia, com) = OOSIOCGIFADDR32;
691 break;
692 case LINUX_SIOCGIFDSTADDR:
693 SCARG(&ia, com) = OOSIOCGIFDSTADDR32;
694 break;
695 case LINUX_SIOCGIFBRDADDR:
696 SCARG(&ia, com) = OOSIOCGIFBRDADDR32;
697 break;
698 case LINUX_SIOCGIFNETMASK:
699 SCARG(&ia, com) = OOSIOCGIFNETMASK32;
700 break;
701 case LINUX_SIOCGIFMTU:
702 SCARG(&ia, com) = OSIOCGIFMTU32;
703 break;
704 case LINUX_SIOCADDMULTI:
705 SCARG(&ia, com) = OSIOCADDMULTI32;
706 break;
707 case LINUX_SIOCDELMULTI:
708 SCARG(&ia, com) = OSIOCDELMULTI32;
709 break;
710 case LINUX_SIOCGIFHWADDR:
711 error = linux32_getifhwaddr(l, retval, SCARG(uap, fd),
712 SCARG_P32(uap, data));
713 dosys = 0;
714 break;
715 default:
716 error = EINVAL;
717 }
718
719 out:
720 fd_putfile(SCARG(uap, fd));
721
722 if (error == 0 && dosys) {
723 SCARG(&ia, fd) = SCARG(uap, fd);
724 SCARG(&ia, data) = SCARG(uap, data);
725 error = netbsd32_ioctl(curlwp, &ia, retval);
726 }
727
728 return error;
729}
