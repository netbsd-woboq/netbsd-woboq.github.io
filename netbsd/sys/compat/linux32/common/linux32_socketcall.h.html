42#define syscallarg(x) \
43 union { \
44 register32_t pad; \
45 \
46 struct { x datum; } le; \
47 struct { /* LINTED zero array dimension */ \
48 int8_t pad[ /* CONSTCOND */ \
49 (sizeof (register32_t) < sizeof (x)) \
50 ? 0 \
51 : sizeof (register32_t) - sizeof (x)]; \
52 x datum; \
53 } be; \
54 }
