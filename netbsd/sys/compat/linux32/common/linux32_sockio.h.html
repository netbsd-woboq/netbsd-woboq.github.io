29#ifndef _LINUX32_SOCKIO_H
30#define _LINUX32_SOCKIO_H
31
32#define LINUX32_IFNAMSIZ 16
33
34struct linux32_ifmap {
35 netbsd32_u_long mem_start;
36 netbsd32_u_long mem_end;
37 unsigned short base_addr;
38 unsigned char irq;
39 unsigned char dma;
40 unsigned char port;
41};
42
43struct linux32_ifreq {
44 union {
45 char ifrn_name[LINUX32_IFNAMSIZ];
46 } ifr_ifrn;
47 union {
48 struct osockaddr ifru_addr;
49 struct osockaddr ifru_hwaddr;
50 struct linux32_ifmap ifru_map;
51 int ifru_ifindex;
52 } ifr_ifru;
53#define ifr_name ifr_ifrn.ifrn_name /* interface name */
54#define ifr_addr ifr_ifru.ifru_addr /* address */
55#define ifr_hwaddr ifr_ifru.ifru_hwaddr /* MAC address */
56#define ifr_map ifr_ifru.ifru_map /* device map */
57};
58
59struct linux32_ifconf {
60 int ifc_len;
61 union {
62 netbsd32_caddr_t ifcu_buf;
63 netbsd32_ifreq_tp_t ifcu_req;
64 } ifc_ifcu;
65};
66
67#define ifc_buf ifc_ifcu.ifcu_buf
68#define ifc_req ifc_ifcu.ifcu_req
69
70#endif /* !_LINUX32_SOCKIO_H */
