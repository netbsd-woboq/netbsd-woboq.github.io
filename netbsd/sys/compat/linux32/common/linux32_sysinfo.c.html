63/* ARGSUSED */
64int
65linux32_sys_sysinfo(struct lwp *l, const struct linux32_sys_sysinfo_args *uap, register_t *retval)
66{
67 /* {
68 syscallarg(struct linux32_sysinfo *) arg;
69 } */
70 struct linux32_sysinfo si;
71 struct loadavg *la;
72
73 si.uptime = time_uptime;
74 la = &averunnable;
75 si.loads[0] = la->ldavg[0] * LINUX_SYSINFO_LOADS_SCALE / la->fscale;
76 si.loads[1] = la->ldavg[1] * LINUX_SYSINFO_LOADS_SCALE / la->fscale;
77 si.loads[2] = la->ldavg[2] * LINUX_SYSINFO_LOADS_SCALE / la->fscale;
78 si.totalram = ctob((u_long)physmem);
79 si.freeram = (u_long)uvmexp.free * uvmexp.pagesize;
80 si.sharedram = 0; /* XXX */
81 si.bufferram = (u_long)uvmexp.filepages * uvmexp.pagesize;
82 si.totalswap = (u_long)uvmexp.swpages * uvmexp.pagesize;
83 si.freeswap =
84 (u_long)(uvmexp.swpages - uvmexp.swpginuse) * uvmexp.pagesize;
85 si.procs = nprocs;
86
87 /* The following are only present in newer Linux kernels. */
88 si.totalbig = 0;
89 si.freebig = 0;
90 si.mem_unit = 1;
91
92 return (copyout(&si, SCARG_P32(uap, arg), sizeof si));
93}
