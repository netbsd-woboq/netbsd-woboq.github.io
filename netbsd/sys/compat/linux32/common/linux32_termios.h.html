34#ifndef _LINUX32_TERMIOS_H
35#define _LINUX32_TERMIOS_H
36
37#if defined(__amd64__)
38#include <compat/linux32/arch/amd64/linux32_termios.h>
39#endif
40
41/*
42 * LINUX32_NCC is architecture dependent
43 */
44struct linux32_termio {
45 unsigned short c_iflag;
46 unsigned short c_oflag;
47 unsigned short c_cflag;
48 unsigned short c_lflag;
49 unsigned char c_line;
50 unsigned char c_cc[LINUX32_NCC];
51};
52
53struct linux32_termios {
54 linux32_tcflag_t c_iflag;
55 linux32_tcflag_t c_oflag;
56 linux32_tcflag_t c_cflag;
57 linux32_tcflag_t c_lflag;
58 linux32_cc_t c_line;
59 linux32_cc_t c_cc[LINUX32_NCCS];
60#ifdef LINUX32_LARGE_STRUCT_TERMIOS
61 linux32_speed_t c_ispeed;
62 linux32_speed_t c_ospeed;
63#endif
64};
65
66#endif /* !_LINUX32_TERMIOS_H */
