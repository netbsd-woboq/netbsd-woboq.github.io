75int
76linux32_sys_uname(struct lwp *l, const struct linux32_sys_uname_args *uap, register_t *retval)
77{
78 /* {
79 syscallarg(linux32_utsnamep) up;
80 } */
81 struct linux_utsname luts;
82 struct linux_utsname *lp;
83
84 strlcpy(luts.l_sysname, linux32_sysname, sizeof(luts.l_sysname));
85 strlcpy(luts.l_nodename, hostname, sizeof(luts.l_nodename));
86 strlcpy(luts.l_release, linux32_release, sizeof(luts.l_release));
87 strlcpy(luts.l_version, linux32_version, sizeof(luts.l_version));
88 strlcpy(luts.l_machine, LINUX_UNAME_ARCH, sizeof(luts.l_machine));
89 strlcpy(luts.l_domainname, domainname, sizeof(luts.l_domainname));
90
91 lp = SCARG_P32(uap, up);
92
93 return copyout(&luts, lp, sizeof(luts));
94}
