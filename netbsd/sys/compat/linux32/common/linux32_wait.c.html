75int
76linux32_sys_waitpid(struct lwp *l, const struct linux32_sys_waitpid_args *uap, register_t *retval)
77{
78 /* {
79 syscallarg(int) pid;
80 syscallarg(netbsd32_intp) status;
81 syscallarg(int) options;
82 } */
83 struct linux32_sys_wait4_args ua;
84
85 SCARG(&ua, pid) = SCARG(uap, pid);
86 SCARG(&ua, status) = SCARG(uap, status);
87 SCARG(&ua, options) = SCARG(uap, options);
88 NETBSD32PTR32(SCARG(&ua, rusage), NULL);
89
90 return linux32_sys_wait4(l, &ua, retval);
91}

93int
94linux32_sys_wait4(struct lwp *l, const struct linux32_sys_wait4_args *uap, register_t *retval)
95{
96 /* {
97 syscallarg(int) pid;
98 syscallarg(netbsd32_intp) status;
99 syscallarg(int) options;
100 syscallarg(netbsd32_rusage50p_t) rusage;
101 } */
102 int error, status, linux_options, options, pid;
103 struct netbsd32_rusage50 ru32;
104 struct rusage ru;
105 proc_t *p;
106
107 linux_options = SCARG(uap, options);
108 if (linux_options & ~(LINUX_WAIT4_KNOWNFLAGS))
109 return EINVAL;
110
111 options = 0;
112 if (linux_options & LINUX_WAIT4_WNOHANG)
113 options |= WNOHANG;
114 if (linux_options & LINUX_WAIT4_WUNTRACED)
115 options |= WUNTRACED;
116 if (linux_options & LINUX_WAIT4_WALL)
117 options |= WALLSIG;
118 if (linux_options & LINUX_WAIT4_WCLONE)
119 options |= WALTSIG;
120
121 pid = SCARG(uap, pid);
122 error = do_sys_wait(&pid, &status, options,
123 SCARG_P32(uap, rusage) != NULL ? &ru : NULL);
124 retval[0] = pid;
125 if (pid == 0)
126 return error;
127
128 p = curproc;
129 mutex_enter(p->p_lock);
130 sigdelset(&p->p_sigpend.sp_set, SIGCHLD); /* XXXAD ksiginfo leak */
131 mutex_exit(p->p_lock);
132
133 if (SCARG_P32(uap, rusage) != NULL) {
134 netbsd32_from_rusage50(&ru, &ru32);
135 error = copyout(&ru32, SCARG_P32(uap, rusage), sizeof(ru32));
136 }
137
138 if (error == 0 && SCARG_P32(uap, status) != NULL) {
139 status = bsd_to_linux_wstat(status);
140 error = copyout(&status, SCARG_P32(uap, status), sizeof(status));
141 }
142
143 return error;
144}
