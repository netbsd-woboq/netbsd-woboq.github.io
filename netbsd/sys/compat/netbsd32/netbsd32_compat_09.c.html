49int
50compat_09_netbsd32_ogetdomainname(struct lwp *l, const struct compat_09_netbsd32_ogetdomainname_args *uap, register_t *retval)
51{
52 /* {
53 syscallarg(netbsd32_charp) domainname;
54 syscallarg(int) len;
55 } */
56 int name[2];
57 size_t sz;
58
59 name[0] = CTL_KERN;
60 name[1] = KERN_DOMAINNAME;
61 sz = SCARG(uap, len);
62 return (old_sysctl(&name[0], 2,
63 (char *)SCARG_P32(uap, domainname), &sz, 0, 0, l));
64}
65
66int
67compat_09_netbsd32_osetdomainname(struct lwp *l, const struct compat_09_netbsd32_osetdomainname_args *uap, register_t *retval)
68{
69 /* {
70 syscallarg(netbsd32_charp) domainname;
71 syscallarg(int) len;
72 } */
73 int name[2];
74
75 name[0] = CTL_KERN;
76 name[1] = KERN_DOMAINNAME;
77 return (old_sysctl(&name[0], 2, 0, 0,
78 (char *)SCARG_P32(uap, domainname), SCARG(uap, len), l));
79}

81int
82compat_09_netbsd32_uname(struct lwp *l, const struct compat_09_netbsd32_uname_args *uap, register_t *retval)
83{
84 /* {
85 syscallarg(netbsd32_outsnamep_t) name;
86 } */
87 struct compat_09_sys_uname_args ua;
88
89 NETBSD32TOP_UAP(name, struct outsname);
90 return (compat_09_sys_uname(l, &ua, retval));
91}

93static struct syscall_package compat_netbsd32_09_syscalls[] = {
94 { NETBSD32_SYS_compat_09_netbsd32_ogetdomainname, 0,
95 (sy_call_t *)compat_09_netbsd32_ogetdomainname },
96 { NETBSD32_SYS_compat_09_netbsd32_osetdomainname, 0,
97 (sy_call_t *)compat_09_netbsd32_osetdomainname },
98 { NETBSD32_SYS_compat_09_netbsd32_uname, 0,
99 (sy_call_t *)compat_09_netbsd32_uname },
100 { 0, 0, NULL }
101};
102
103MODULE(MODULE_CLASS_EXEC, compat_netbsd32_09, "compat_netbsd32_12,compat_09");
104
105static int
106compat_netbsd32_09_modcmd(modcmd_t cmd, void *arg)
107{
108
109 switch (cmd) {
110 case MODULE_CMD_INIT:
111 return syscall_establish(&emul_netbsd32,
112 compat_netbsd32_09_syscalls);
113
114 case MODULE_CMD_FINI:
115 return syscall_disestablish(&emul_netbsd32,
116 compat_netbsd32_09_syscalls);
117
118 default:
119 return ENOTTY;
120 }
121}
