55int
56compat_60_netbsd32__lwp_park(struct lwp *l,
57 const struct compat_60_netbsd32__lwp_park_args *uap, register_t *retval)
58{
59 /* {
60 syscallarg(const netbsd32_timespecp) ts;
61 syscallarg(lwpid_t) unpark;
62 syscallarg(netbsd32_voidp) hint;
63 syscallarg(netbsd32_voidp) unparkhint;
64 } */
65 struct timespec ts, *tsp;
66 struct netbsd32_timespec ts32;
67 int error;
68
69 if (SCARG_P32(uap, ts) == NULL)
70 tsp = NULL;
71 else {
72 error = copyin(SCARG_P32(uap, ts), &ts32, sizeof ts32);
73 if (error != 0)
74 return error;
75 netbsd32_to_timespec(&ts32, &ts);
76 tsp = &ts;
77 }
78
79 if (SCARG(uap, unpark) != 0) {
80 error = lwp_unpark(SCARG(uap, unpark),
81 SCARG_P32(uap, unparkhint));
82 if (error != 0)
83 return error;
84 }
85
86 return lwp_park(CLOCK_REALTIME, TIMER_ABSTIME, tsp,
87 SCARG_P32(uap, hint));
88}

90static struct syscall_package compat_netbsd32_60_syscalls[] = {
91 { NETBSD32_SYS_compat_60_netbsd32__lwp_park, 0,
92 (sy_call_t *)compat_60_netbsd32__lwp_park },
93 { 0, 0, NULL }
94};
95
96
97MODULE(MODULE_CLASS_EXEC, compat_netbsd32_60, "compat_netbsd32_80,compat_60");
98
99static int
100compat_netbsd32_60_modcmd(modcmd_t cmd, void *arg)
101{
102
103 switch (cmd) {
104 case MODULE_CMD_INIT:
105 return syscall_establish(&emul_netbsd32,
106 compat_netbsd32_60_syscalls);
107
108 case MODULE_CMD_FINI:
109 return syscall_disestablish(&emul_netbsd32,
110 compat_netbsd32_60_syscalls);
111
112 default:
113 return ENOTTY;
114 }
115}
