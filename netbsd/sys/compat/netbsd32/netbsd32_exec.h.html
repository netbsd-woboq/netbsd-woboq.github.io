39/*
40 * Header prepended to each a.out file.
41 * only manipulate the a_midmag field via the
42 * N_SETMAGIC/N_GET{MAGIC,MID,FLAG} macros below.
43 */
44struct netbsd32_exec {
45 netbsd32_u_long a_midmag; /* htonl(flags<<26 | mid<<16 | magic) */
46 netbsd32_u_long a_text; /* text segment size */
47 netbsd32_u_long a_data; /* initialized data size */
48 netbsd32_u_long a_bss; /* uninitialized data size */
49 netbsd32_u_long a_syms; /* symbol table size */
50 netbsd32_u_long a_entry; /* entry point */
51 netbsd32_u_long a_trsize; /* text relocation size */
52 netbsd32_u_long a_drsize; /* data relocation size */
53};

79/*
80 * We need to copy out all pointers as 32-bit values.
81 */
82static __inline int
83netbsd32_copyargs(struct lwp *l, struct exec_package *pack,
84 struct ps_strings *arginfo, char **stackp, void *argp)
85{
86 u_int32_t *cpp = (u_int32_t *)*stackp;
87 netbsd32_pointer_t dp;
88 u_int32_t nullp = 0;
89 char *sp;
90 size_t len;
91 int argc = arginfo->ps_nargvstr;
92 int envc = arginfo->ps_nenvstr;
93 int error;
94
95 NETBSD32PTR32(dp, (char *)(cpp +
96 1 + /* int argc */
97 argc + /* char *argv[] */
98 1 + /* \0 */
99 envc + /* char *env[] */
100 1) + /* \0 */
101 pack->ep_esch->es_arglen); /* auxinfo */
102 sp = argp;
103
104 if ((error = copyout(&argc, cpp++, sizeof(argc))) != 0)
105 return error;
106
107 /* XXX don't copy them out, remap them! */
108 /* remember location of argv for later */
109 arginfo->ps_argvstr = (char **)(u_long)cpp;
110
111 for (; --argc >= 0; sp += len, NETBSD32PTR32PLUS(dp, len)) {
112 if ((error = copyout(&dp, cpp++, sizeof(dp))) != 0)
113 return error;
114 if ((error = copyoutstr(sp, NETBSD32PTR64(dp),
115 ARG_MAX, &len)) != 0)
116 return error;
117 }
118 if ((error = copyout(&nullp, cpp++, sizeof(nullp))) != 0)
119 return error;
120
121 /* remember location of envp for later */
122 arginfo->ps_envstr = (char **)(u_long)cpp;
123
124 for (; --envc >= 0; sp += len, NETBSD32PTR32PLUS(dp, len)) {
125 if ((error = copyout(&dp, cpp++, sizeof(dp))) != 0)
126 return error;
127 if ((error = copyoutstr(sp, NETBSD32PTR64(dp),
128 ARG_MAX, &len)) != 0)
129 return error;
130 }
131 if ((error = copyout(&nullp, cpp++, sizeof(nullp))) != 0)
132 return error;
133
134 *stackp = (char *)cpp;
135 return 0;
136}
