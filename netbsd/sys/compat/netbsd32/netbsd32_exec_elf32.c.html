84int ELFNAME2(netbsd32,probe_noteless)(struct lwp *, struct exec_package *epp,
85 void *eh, char *itp, vaddr_t *pos);
86extern int ELFNAME2(netbsd,signature)(struct lwp *, struct exec_package *,
87 Elf_Ehdr *);
88
89int
90ELFNAME2(netbsd32,probe)(struct lwp *l, struct exec_package *epp,
91 void *eh, char *itp, vaddr_t *pos)
92{
93 int error;
94
95 if ((error = ELFNAME2(netbsd,signature)(l, epp, eh)) != 0)
96 return error;
97
98#ifdef ELF_MD_PROBE_FUNC
99 if ((error = ELF_MD_PROBE_FUNC(l, epp, eh, itp, pos)) != 0)
100 return error;
101#elif defined(ELF_INTERP_NON_RELOCATABLE)
102 *pos = ELF_LINK_ADDR;
103#endif
104
105 return ELFNAME2(netbsd32,probe_noteless)(l, epp, eh, itp, pos);
106}
107
108int
109ELFNAME2(netbsd32,probe_noteless)(struct lwp *l, struct exec_package *epp,
110 void *eh, char *itp, vaddr_t *pos)
111{
112 const char *m;
113
114 if (itp && epp->ep_interp == NULL) {
115 MODULE_HOOK_CALL(netbsd32_machine32_hook, (), machine, m);
116 (void)compat_elf_check_interp(epp, itp, m);
117 }
118#ifdef _LP64
119 epp->ep_flags |= EXEC_32 | EXEC_FORCEAUX;
120#endif
121 epp->ep_vm_minaddr = exec_vm_minaddr(VM_MIN_ADDRESS);
122 epp->ep_vm_maxaddr = USRSTACK32;
123#ifdef ELF_INTERP_NON_RELOCATABLE
124 *pos = ELF_LINK_ADDR;
125#endif
126 return 0;
127}

129/*
130 * Copy arguments onto the stack in the normal way, but add some
131 * extra information in case of dynamic binding.
132 */
133int
134netbsd32_elf32_copyargs(struct lwp *l, struct exec_package *pack,
135 struct ps_strings *arginfo, char **stackp, void *argp)
136{
137 int error;
138
139 if ((error = netbsd32_copyargs(l, pack, arginfo, stackp, argp)) != 0)
140 return error;
141
142 return elf32_populate_auxv(l, pack, stackp);
143}
