111static int
112copyin_psstrings_32(struct proc *p, struct ps_strings *arginfo)
113{
114 struct ps_strings32 arginfo32;
115
116 int error = copyin_proc(p, (void *)p->p_psstrp, &arginfo32,
117 sizeof(arginfo32));
118 if (error)
119 return error;
120 arginfo->ps_argvstr = (void *)(uintptr_t)arginfo32.ps_argvstr;
121 arginfo->ps_nargvstr = arginfo32.ps_nargvstr;
122 arginfo->ps_envstr = (void *)(uintptr_t)arginfo32.ps_envstr;
123 arginfo->ps_nenvstr = arginfo32.ps_nenvstr;
124 return 0;
125}
126
127static vaddr_t
128get_base32(char **argv, size_t i)
129{
130
131 netbsd32_charp *argv32;
132
133 argv32 = (netbsd32_charp *)argv;
134 return (vaddr_t)NETBSD32PTR64(argv32[i]);
135}
136
137void
138netbsd32_kern_proc_32_init(void)
139{
140
141#if !defined(_RUMPSERVER)
142 MODULE_HOOK_SET(kern_proc32_copyin_hook, "kern_proc_32",
143 copyin_psstrings_32);
144 MODULE_HOOK_SET(kern_proc32_base_hook, "kern_proc_32", get_base32);
145#endif
146}
147
148void
149netbsd32_kern_proc_32_fini(void)
150{
151
152#if !defined(_RUMPSERVER)
153 MODULE_HOOK_UNSET(kern_proc32_copyin_hook);
154 MODULE_HOOK_UNSET(kern_proc32_base_hook);
155#endif
156}
