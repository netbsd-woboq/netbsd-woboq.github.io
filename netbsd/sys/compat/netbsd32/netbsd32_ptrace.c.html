58static int
59netbsd32_copyin_piod(struct ptrace_io_desc *piod, const void *addr, size_t len)
60{
61 struct netbsd32_ptrace_io_desc piod32;
62
63 if (len != 0 && sizeof(piod32) != len)
64 return EINVAL;
65
66 int error = copyin(addr, &piod32, sizeof(piod32));
67 if (error)
68 return error;
69 piod->piod_op = piod32.piod_op;
70 piod->piod_offs = NETBSD32PTR64(piod32.piod_offs);
71 piod->piod_addr = NETBSD32PTR64(piod32.piod_addr);
72 piod->piod_len = (size_t)piod32.piod_len;
73
74 return 0;
75}
76
77static int
78netbsd32_copyout_piod(const struct ptrace_io_desc *piod, void *addr, size_t len)
79{
80 struct netbsd32_ptrace_io_desc piod32;
81
82 if (len != 0 && sizeof(piod32) != len)
83 return EINVAL;
84
85 piod32.piod_op = piod->piod_op;
86 NETBSD32PTR32(piod32.piod_offs, piod->piod_offs);
87 NETBSD32PTR32(piod32.piod_addr, piod->piod_addr);
88 piod32.piod_len = (netbsd32_size_t)piod->piod_len;
89 return copyout(&piod32, addr, sizeof(piod32));
90}

92static int
93netbsd32_copyin_siginfo(struct ptrace_siginfo *psi, const void *addr, size_t len)
94{
95 struct netbsd32_ptrace_siginfo psi32;
96
97 if (sizeof(psi32) != len)
98 return EINVAL;
99
100 int error = copyin(addr, &psi32, sizeof(psi32));
101 if (error)
102 return error;
103 psi->psi_lwpid = psi32.psi_lwpid;
104 netbsd32_si32_to_si(&psi->psi_siginfo, &psi32.psi_siginfo);
105 return 0;
106}
107
108static int
109netbsd32_copyout_siginfo(const struct ptrace_siginfo *psi, void *addr, size_t len)
110{
111 struct netbsd32_ptrace_siginfo psi32;
112
113 if (sizeof(psi32) != len)
114 return EINVAL;
115
116 psi32.psi_lwpid = psi->psi_lwpid;
117 netbsd32_si_to_si32(&psi32.psi_siginfo, &psi->psi_siginfo);
118 return copyout(&psi32, addr, sizeof(psi32));
119}

121static int
122netbsd32_doregs(struct lwp *curl /*tracer*/,
123 struct lwp *l /*traced*/,
124 struct uio *uio)
125{
126#if defined(PT_GETREGS) || defined(PT_SETREGS)
127 process_reg32 r32;
128 int error;
129 char *kv;
130 int kl;
131
132 if (uio->uio_offset < 0 || uio->uio_offset > (off_t)sizeof(r32))
133 return EINVAL;
134
135 kl = sizeof(r32);
136 kv = (char *)&r32;
137
138 kv += uio->uio_offset;
139 kl -= uio->uio_offset;
140 if ((size_t)kl > uio->uio_resid)
141 kl = uio->uio_resid;
142 error = process_read_regs32(l, &r32);
143 if (error == 0)
144 error = uiomove(kv, kl, uio);
145 if (error == 0 && uio->uio_rw == UIO_WRITE) {
146 if (l->l_stat != LSSTOP)
147 error = EBUSY;
148 else
149 error = process_write_regs32(l, &r32);
150 }
151
152 uio->uio_offset = 0;
153 return error;
154#else
155 return EINVAL;
156#endif
157}

246int
247netbsd32_ptrace(struct lwp *l, const struct netbsd32_ptrace_args *uap,
248 register_t *retval)
249{
250 int req;
251
252 /* {
253 syscallarg(int) req;
254 syscallarg(pid_t) pid;
255 syscallarg(netbsd32_voidp *) addr;
256 syscallarg(int) data;
257 } */
258
259 req = PTRACE_TRANSLATE_REQUEST32(SCARG(uap, req));
260 if (req == -1)
261 return EOPNOTSUPP;
262
263 return do_ptrace(&netbsd32_ptm, l, req, SCARG(uap, pid),
264 SCARG_P32(uap, addr), SCARG(uap, data), retval);
265}

267static const struct syscall_package compat_ptrace_syscalls[] = {
268 { NETBSD32_SYS_netbsd32_ptrace, 0, (sy_call_t *)netbsd32_ptrace },
269 { 0, 0, NULL },
270};

272#define DEPS "compat_netbsd32,ptrace_common"

274MODULE(MODULE_CLASS_EXEC, compat_netbsd32_ptrace, DEPS);

276static int
277compat_netbsd32_ptrace_modcmd(modcmd_t cmd, void *arg)
278{
279 int error;
280
281 switch (cmd) {
282 case MODULE_CMD_INIT:
283 error = syscall_establish(&emul_netbsd32,
284 compat_ptrace_syscalls);
285 break;
286 case MODULE_CMD_FINI:
287 error = syscall_disestablish(&emul_netbsd32,
288 compat_ptrace_syscalls);
289 break;
290 default:
291 error = ENOTTY;
292 break;
293 }
294 return error;
295}
