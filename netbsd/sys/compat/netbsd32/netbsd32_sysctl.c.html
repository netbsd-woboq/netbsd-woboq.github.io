66struct sysctlnode netbsd32_sysctl_root = {
67 .sysctl_flags = SYSCTL_VERSION|CTLFLAG_ROOT|CTLTYPE_NODE,
68 .sysctl_num = 0,
69 .sysctl_name = "(netbsd32_root)",
70 .sysctl_size = sizeof(struct sysctlnode),
71};
72
73static struct sysctllog *netbsd32_clog;
74
75/*
76 * sysctl helper routine for netbsd32's kern.boottime node
77 */
78static int
79netbsd32_sysctl_kern_boottime(SYSCTLFN_ARGS)
80{
81 struct sysctlnode node;
82 struct netbsd32_timespec bt32;
83
84 netbsd32_from_timespec(&boottime, &bt32);
85
86 node = *rnode;
87 node.sysctl_data = &bt32;
88 return (sysctl_lookup(SYSCTLFN_CALL(&node)));
89}
90
91/*
92 * sysctl helper routine for netbsd32's vm.loadavg node
93 */
94static int
95netbsd32_sysctl_vm_loadavg(SYSCTLFN_ARGS)
96{
97 struct sysctlnode node;
98 struct netbsd32_loadavg av32;
99
100 netbsd32_from_loadavg(&av32, &averunnable);
101
102 node = *rnode;
103 node.sysctl_data = &av32;
104 return (sysctl_lookup(SYSCTLFN_CALL(&node)));
105}
106
107static int
108sysctl_hw_machine_arch32(SYSCTLFN_ARGS)
109{
110 struct sysctlnode node = *rnode;
111#ifndef PROC_MACHINE_ARCH32
112 extern const char machine_arch32[];
113#define PROC_MACHINE_ARCH32(P) __UNCONST(machine_arch32)
114#endif
115
116 node.sysctl_data = PROC_MACHINE_ARCH32(l->l_proc);
117 node.sysctl_size = strlen(node.sysctl_data) + 1;
118 return sysctl_lookup(SYSCTLFN_CALL(&node));
119}

121void
122netbsd32_sysctl_init(void)
123{
124 const struct sysctlnode *_root = &netbsd32_sysctl_root;
125 extern const char machine32[];
126
127 sysctl_createv(&netbsd32_clog, 0, &_root, NULL,
128 CTLFLAG_PERMANENT,
129 CTLTYPE_NODE, "kern", NULL,
130 NULL, 0, NULL, 0,
131 CTL_KERN, CTL_EOL);
132 sysctl_createv(&netbsd32_clog, 0, &_root, NULL,
133 CTLFLAG_PERMANENT,
134 CTLTYPE_STRUCT, "boottime", NULL,
135 netbsd32_sysctl_kern_boottime, 0, NULL,
136 sizeof(struct netbsd32_timeval),
137 CTL_KERN, KERN_BOOTTIME, CTL_EOL);
138
139 sysctl_createv(&netbsd32_clog, 0, &_root, NULL,
140 CTLFLAG_PERMANENT,
141 CTLTYPE_NODE, "vm", NULL,
142 NULL, 0, NULL, 0,
143 CTL_VM, CTL_EOL);
144 sysctl_createv(&netbsd32_clog, 0, &_root, NULL,
145 CTLFLAG_PERMANENT,
146 CTLTYPE_STRUCT, "loadavg", NULL,
147 netbsd32_sysctl_vm_loadavg, 0, NULL,
148 sizeof(struct netbsd32_loadavg),
149 CTL_VM, VM_LOADAVG, CTL_EOL);
150 sysctl_createv(&netbsd32_clog, 0, &_root, NULL,
151 CTLFLAG_PERMANENT|CTLFLAG_IMMEDIATE,
152 CTLTYPE_INT, "maxaddress",
153 SYSCTL_DESCR("Maximum user address"),
154 NULL, VM_MAXUSER_ADDRESS32, NULL, 0,
155 CTL_VM, VM_MAXADDRESS, CTL_EOL);
156 sysctl_createv(&netbsd32_clog, 0, &_root, NULL,
157 CTLFLAG_PERMANENT|CTLFLAG_IMMEDIATE,
158 CTLTYPE_INT, "minaddress",
159 SYSCTL_DESCR("Minimum user address"),
160 NULL, VM_MIN_ADDRESS, NULL, 0,
161 CTL_VM, VM_MINADDRESS, CTL_EOL);
162
163 sysctl_createv(&netbsd32_clog, 0, &_root, NULL,
164 CTLFLAG_PERMANENT,
165 CTLTYPE_NODE, "hw", NULL,
166 NULL, 0, NULL, 0,
167 CTL_HW, CTL_EOL);
168 sysctl_createv(&netbsd32_clog, 0, &_root, NULL,
169 CTLFLAG_PERMANENT|CTLFLAG_IMMEDIATE,
170 CTLTYPE_INT, "alignbytes", NULL,
171 NULL, ALIGNBYTES32, NULL, 0,
172 CTL_HW, HW_ALIGNBYTES, CTL_EOL);
173 sysctl_createv(&netbsd32_clog, 0, &_root, NULL,
174 CTLFLAG_PERMANENT,
175 CTLTYPE_STRING, "machine", NULL,
176 NULL, 0, __UNCONST(&machine32), 0,
177 CTL_HW, HW_MACHINE, CTL_EOL);
178 sysctl_createv(&netbsd32_clog, 0, &_root, NULL,
179 CTLFLAG_PERMANENT|CTLFLAG_READONLY,
180 CTLTYPE_STRING, "machine_arch", NULL,
181 sysctl_hw_machine_arch32, 0, NULL, 0,
182 CTL_HW, HW_MACHINE_ARCH, CTL_EOL);
183}

185void
186netbsd32_sysctl_fini(void)
187{
188
189 sysctl_teardown(&netbsd32_clog);
190 sysctl_free(&netbsd32_sysctl_root);

193int
194netbsd32___sysctl(struct lwp *l, const struct netbsd32___sysctl_args *uap, register_t *retval)
195{
196 /* {
197 syscallarg(netbsd32_intp) name;
198 syscallarg(u_int) namelen;
199 syscallarg(netbsd32_voidp) old;
200 syscallarg(netbsd32_size_tp) oldlenp;
201 syscallarg(netbsd32_voidp) new;
202 syscallarg(netbsd32_size_t) newlen;
203 } */
204 const struct sysctlnode *pnode;
205 netbsd32_size_t netbsd32_oldlen;
206 size_t oldlen, *oldlenp, savelen;
207 int name[CTL_MAXNAME], error, nerror, *namep;
208 void *newp, *oldp;
209
210 /*
211 * get and convert 32 bit size_t to native size_t
212 */
213 namep = SCARG_P32(uap, name);
214 oldp = SCARG_P32(uap, oldv);
215 newp = SCARG_P32(uap, newv);
216 oldlenp = SCARG_P32(uap, oldlenp);
217 oldlen = 0;
218 if (oldlenp != NULL) {
219 error = copyin(oldlenp, &netbsd32_oldlen,
220 sizeof(netbsd32_oldlen));
221 if (error)
222 return (error);
223 oldlen = netbsd32_oldlen;
224 }
225 savelen = oldlen;
226
227 /*
228 * retrieve name and see if we need to dispatch this query to
229 * the shadow tree. if we find it in the shadow tree,
230 * dispatch to there, otherwise NULL means use the built-in
231 * default main tree.
232 */
233 if (SCARG(uap, namelen) > CTL_MAXNAME || SCARG(uap, namelen) < 1)
234 return (EINVAL);
235 error = copyin(namep, &name[0], SCARG(uap, namelen) * sizeof(int));
236 if (error)
237 return (error);
238
239 ktrmib(name, SCARG(uap, namelen));
240
241 sysctl_lock(newp != NULL);
242 pnode = &netbsd32_sysctl_root;
243 error = sysctl_locate(l, &name[0], SCARG(uap, namelen), &pnode, NULL);
244 pnode = (error == 0) ? &netbsd32_sysctl_root : NULL;
245 error = sysctl_dispatch(&name[0], SCARG(uap, namelen),
246 oldp, &oldlen,
247 newp, SCARG(uap, newlen),
248 &name[0], l, pnode);
249 sysctl_unlock();
250
251 /*
252 * reset caller's oldlen, even if we got an error
253 */
254 if (oldlenp) {
255 netbsd32_oldlen = oldlen;
256 nerror = copyout(&netbsd32_oldlen, oldlenp,
257 sizeof(netbsd32_oldlen));
258 if (error == 0)
259 error = nerror;
260 }
261
262 /*
263 * if the only problem is that we weren't given enough space,
264 * that's an ENOMEM error
265 */
266 if (error == 0 && oldp != NULL && savelen < oldlen)
267 error = ENOMEM;
268
269 return (error);
270}
