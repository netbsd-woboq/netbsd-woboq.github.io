48int
49netbsd32___wait450(struct lwp *l, const struct netbsd32___wait450_args *uap,
50 register_t *retval)
51{
52 /* {
53 syscallarg(int) pid;
54 syscallarg(netbsd32_intp) status;
55 syscallarg(int) options;
56 syscallarg(netbsd32_rusagep_t) rusage;
57 } */
58 int error, status, pid = SCARG(uap, pid);
59 struct netbsd32_rusage ru32;
60 struct rusage ru;
61
62 error = do_sys_wait(&pid, &status, SCARG(uap, options),
63 SCARG_P32(uap, rusage) != NULL ? &ru : NULL);
64
65 retval[0] = pid;
66 if (pid == 0)
67 return error;
68
69 if (SCARG_P32(uap, rusage)) {
70 netbsd32_from_rusage(&ru, &ru32);
71 error = copyout(&ru32, SCARG_P32(uap, rusage), sizeof(ru32));
72 }
73
74 if (error == 0 && SCARG_P32(uap, status))
75 error = copyout(&status, SCARG_P32(uap, status), sizeof(status));
76
77 return error;
78}

80int
81netbsd32_wait6(struct lwp *l, const struct netbsd32_wait6_args *uap,
82 register_t *retval)
83{
84 /* {
85 syscallarg(idtype_t) idtype;
86 syscallarg(id_t) id;
87 syscallarg(netbsd32_intp) status;
88 syscallarg(int) options;
89 syscallarg(netbsd32_wrusagep_t) wru;
90 syscallarg(netbsd32_siginfop_t) info;
91 } */
92 idtype_t idtype = SCARG(uap, idtype);
93 id_t id = SCARG(uap, id);
94 struct wrusage wru, *wrup;
95 siginfo_t si, *sip;
96 int status;
97 int pid;
98
99 if (SCARG_P32(uap, wru) != NULL)
100 wrup = &wru;
101 else
102 wrup = NULL;
103
104 if (SCARG_P32(uap, info) != NULL)
105 sip = &si;
106 else
107 sip = NULL;
108
109 /*
110 * We expect all callers of wait6() to know about WEXITED and
111 * WTRAPPED.
112 */
113 int error = do_sys_waitid(idtype, id, &pid, &status,
114 SCARG(uap, options), wrup, sip);
115
116 retval[0] = pid; /* tell userland who it was */
117
118#if 0
119 /*
120 * should we copyout if there was no process, hence no useful data?
121 * We don't for an old sytle wait4() (etc) but I believe
122 * FreeBSD does for wait6(), so a tossup... Go with FreeBSD for now.
123 */
124 if (pid == 0)
125 return error;
126#endif
127
128
129 if (error == 0 && SCARG_P32(uap, status))
130 error = copyout(&status, SCARG_P32(uap, status),
131 sizeof(status));
132 if (wrup != NULL && error == 0) {
133 struct netbsd32_wrusage wru32;
134
135 netbsd32_from_rusage(&wrup->wru_self, &wru32.wru_self);
136 netbsd32_from_rusage(&wrup->wru_children, &wru32.wru_children);
137 error = copyout(&wru32, SCARG_P32(uap, wru), sizeof(wru32));
138 }
139 if (sip != NULL && error == 0) {
140 siginfo32_t si32;
141
142 netbsd32_si_to_si32(&si32, sip);
143 error = copyout(&si32, SCARG_P32(uap, info), sizeof(si32));
144 }
145
146 return error;
147}

149int
150netbsd32___getrusage50(struct lwp *l,
151 const struct netbsd32___getrusage50_args *uap, register_t *retval)
152{
153 /* {
154 syscallarg(int) who;
155 syscallarg(netbsd32_rusagep_t) rusage;
156 } */
157 int error;
158 struct proc *p = l->l_proc;
159 struct rusage ru;
160 struct netbsd32_rusage ru32;
161
162 error = getrusage1(p, SCARG(uap, who), &ru);
163 if (error != 0)
164 return error;
165
166 netbsd32_from_rusage(&ru, &ru32);
167 return copyout(&ru32, SCARG_P32(uap, rusage), sizeof(ru32));
168}
