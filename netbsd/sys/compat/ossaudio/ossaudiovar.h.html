35#define OSS_IOCPARM_MASK 0xfff /* parameters must be < 4096 bytes */
36#define OSS_IOC_VOID 0x00000000 /* no parameters */
37#define OSS_IOC_IN 0x40000000 /* copy in parameters */
38#define OSS_IOC_OUT 0x80000000 /* copy out parameters */
39#define OSS_IOC_INOUT (OSS_IOC_IN | OSS_IOC_OUT)
40#define _OSS_IOCTL(w,x,y,z) ((uint)((w)|(((z)&OSS_IOCPARM_MASK)<<16)|((x)<<8)|(y)))
41#define _OSS_IO(x,y) _OSS_IOCTL(OSS_IOC_VOID, x, y, 0)
42#define _OSS_IOR(x,y,t) _OSS_IOCTL(OSS_IOC_OUT, x, y, sizeof(t))
43#define _OSS_IOW(x,y,t) _OSS_IOCTL(OSS_IOC_IN, x, y, sizeof(t))
44#define _OSS_IOWR(x,y,t) _OSS_IOCTL(OSS_IOC_INOUT, x, y, sizeof(t))
45
46#define OSS_IOCTL_SIZE(x) (((x) >> 16) & OSS_IOCPARM_MASK)
