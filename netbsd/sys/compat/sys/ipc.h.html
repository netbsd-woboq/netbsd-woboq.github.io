57/*
58 * Old IPC permission structure used before NetBSD 1.5.
59 */
60struct ipc_perm14 {
61 unsigned short cuid; /* creator user id */
62 unsigned short cgid; /* creator group id */
63 unsigned short uid; /* user id */
64 unsigned short gid; /* group id */
65 unsigned short mode; /* r/w permission */
66 unsigned short seq; /* sequence # (to generate unique
67 msg/sem/shm id) */
68 key_t key; /* user specified msg/sem/shm key */
69};
70
71static __inline void __ipc_perm14_to_native(const struct ipc_perm14 *, struct ipc_perm *);
72static __inline void __native_to_ipc_perm14(const struct ipc_perm *, struct ipc_perm14 *);
73static __inline void
74__ipc_perm14_to_native(const struct ipc_perm14 *operm, struct ipc_perm *perm)
75{
76
77 memset(perm, 0, sizeof *perm);
78#define CVT(x) perm->x = operm->x
79 CVT(uid);
80 CVT(gid);
81 CVT(cuid);
82 CVT(cgid);
83 CVT(mode);
84#undef CVT
85}
86
87static __inline void
88__native_to_ipc_perm14(const struct ipc_perm *perm, struct ipc_perm14 *operm)
89{
90
91 memset(operm, 0, sizeof *operm);
92#define CVT(x) operm->x = perm->x
93 CVT(uid);
94 CVT(gid);
95 CVT(cuid);
96 CVT(cgid);
97 CVT(mode);
98#undef CVT
99
100 /*
101 * Not part of the API, but some programs might look at it.
102 */
103 operm->seq = perm->_seq;
104 operm->key = perm->_key;
105}
