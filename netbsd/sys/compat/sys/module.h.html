30#define _COMPAT_SYS_MODULE_H_
31
32/*
33 * This structure intentionally has the same layout for 32 and 64
34 * bit builds.
35 */
36typedef struct omodstat {
37 char oms_name[MAXMODNAME];
38 char oms_required[MAXMODNAME * MAXMODDEPS];
39 uint64_t oms_addr;
40 modsrc_t oms_source;
41 modclass_t oms_class;
42 u_int oms_size;
43 u_int oms_refcnt;
44 u_int oms_flags;
45 u_int oms_reserved[3];
46} omodstat_t;
47
48#endif /* !_COMPAT_SYS_MODULE_H_ */
