32#ifndef _COMPAT_SYS_SIGINFO_H_
33#define _COMPAT_SYS_SIGINFO_H_
34
35#ifdef _KERNEL
36
37typedef union sigval32 {
38 int sival_int;
39 uint32_t sival_ptr;
40} sigval32_t;
41
42struct __ksiginfo32 {
43 int _signo;
44 int _code;
45 int _errno;
46
47 union {
48 struct {
49 pid_t _pid;
50 uid_t _uid;
51 sigval32_t _value;
52 } _rt;
53
54 struct {
55 pid_t _pid;
56 uid_t _uid;
57 int _status;
58 clock_t _utime;
59 clock_t _stime;
60 } _child;
61
62 struct {
63 uint32_t _addr;
64 int _trap;
65 } _fault;
66
67 struct {
68 int32_t _band;
69 int _fd;
70 } _poll;
71
72 struct {
73 int _sysnum;
74 int _retval[2];
75 int _error;
76 uint64_t _args[8]; /* SYS_MAXSYSARGS */
77 } _syscall;
78 } _reason;
79};
80
81typedef union siginfo32 {
82 char si_pad[128];
83 struct __ksiginfo32 _info;
84} siginfo32_t;
85
86#endif /* _KERNEL */
87
88#endif /* !_COMPAT_SYS_SIGINFO_H_ */
