39#ifndef _COMPAT_SYS_SIGTYPES_H_
40#define _COMPAT_SYS_SIGTYPES_H_
41
42#if defined(_POSIX_C_SOURCE) || defined(_XOPEN_SOURCE) || \
43 defined(_NETBSD_SOURCE)
44
45typedef unsigned int sigset13_t;
46
47/*
48 * Macro for manipulating signal masks.
49 */
50#define __sigmask13(n) (1 << ((n) - 1))
51#define __sigaddset13(s, n) (*(s) |= __sigmask13(n))
52#define __sigdelset13(s, n) (*(s) &= ~__sigmask13(n))
53#define __sigismember13(s, n) (*(s) & __sigmask13(n))
54#define __sigemptyset13(s) (*(s) = 0)
55#define __sigfillset13(s) (*(s) = ~(sigset13_t)0)
56
57/* Not strictly a defined type, but is logically associated with stack_t. */
58struct sigaltstack13 {
59 char *ss_sp; /* signal stack base */
60 int ss_size; /* signal stack length */
61 int ss_flags; /* SS_DISABLE and/or SS_ONSTACK */
62};
63
64#endif /* _POSIX_C_SOURCE || _XOPEN_SOURCE || ... */

66#ifdef _KERNEL
67
68struct __sigaltstack32 {
69 uint32_t ss_sp;
70 uint32_t ss_size;
71 int32_t ss_flags;
72};
73
74typedef struct __sigaltstack32 stack32_t;
75
76#endif /* _KERNEL */
