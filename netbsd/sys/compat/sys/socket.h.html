53/*
54 * 4.3 compat sockaddr
55 */
56struct osockaddr {
57 uint16_t sa_family; /* address family */
58 char sa_data[14]; /* up to 14 bytes of direct address */
59};
60
61/*
62 * 4.3-compat message header
63 */
64struct omsghdr {
65 void * msg_name; /* optional address */
66 int msg_namelen; /* size of address */
67 struct iovec *msg_iov; /* scatter/gather array */
68 int msg_iovlen; /* # elements in msg_iov */
69 void * msg_accrights; /* access rights sent/received */
70 int msg_accrightslen;
71};
72
73/*
74 * 7.0 compat sockcred
75 */
76struct sockcred70 {
77 uid_t sc_uid; /* real user id */
78 uid_t sc_euid; /* effective user id */
79 gid_t sc_gid; /* real group id */
80 gid_t sc_egid; /* effective group id */
81 int sc_ngroups; /* number of supplemental groups */
82 gid_t sc_groups[1]; /* variable length */
83};
84#define SOCKCRED70SIZE(ngrps) \
85 (/*CONSTCOND*/sizeof(struct sockcred70) + (sizeof(gid_t) * \
86 ((ngrps) ? ((ngrps) - 1) : 0)))
