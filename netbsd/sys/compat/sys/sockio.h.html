37#define OIFNAMSIZ 16
38
39struct oifreq {
40 char ifr_name[OIFNAMSIZ]; /* if name, e.g. "en0" */
41 union {
42 struct sockaddr ifru_addr;
43 struct sockaddr ifru_dstaddr;
44 struct sockaddr ifru_broadaddr;
45 short ifru_flags;
46 int ifru_metric;
47 int ifru_mtu;
48 int ifru_dlt;
49 u_int ifru_value;
50 void * ifru_data;
51 struct {
52 uint32_t b_buflen;
53 void *b_buf;
54 } ifru_b;
55 } ifr_ifru;
56};
57struct oifconf {
58 int ifc_len; /* size of associated buffer */
59 union {
60 void * ifcu_buf;
61 struct oifreq *ifcu_req;
62 } ifc_ifcu;
63#define ifc_buf ifc_ifcu.ifcu_buf /* buffer address */
64#define ifc_req ifc_ifcu.ifcu_req /* array of structures returned */
65};
66
67#include <compat/sys/time.h>
68/*
69 * Structure defining statistics and other data kept regarding a network
70 * interface.
71 */
72struct oif_data {
73 /* generic interface information */
74 u_char ifi_type; /* ethernet, tokenring, etc. */
75 u_char ifi_addrlen; /* media address length */
76 u_char ifi_hdrlen; /* media header length */
77 int ifi_link_state; /* current link state */
78 u_quad_t ifi_mtu; /* maximum transmission unit */
79 u_quad_t ifi_metric; /* routing metric (external only) */
80 u_quad_t ifi_baudrate; /* linespeed */
81 /* volatile statistics */
82 u_quad_t ifi_ipackets; /* packets received on interface */
83 u_quad_t ifi_ierrors; /* input errors on interface */
84 u_quad_t ifi_opackets; /* packets sent on interface */
85 u_quad_t ifi_oerrors; /* output errors on interface */
86 u_quad_t ifi_collisions; /* collisions on csma interfaces */
87 u_quad_t ifi_ibytes; /* total number of octets received */
88 u_quad_t ifi_obytes; /* total number of octets sent */
89 u_quad_t ifi_imcasts; /* packets received via multicast */
90 u_quad_t ifi_omcasts; /* packets sent via multicast */
91 u_quad_t ifi_iqdrops; /* dropped on input, this interface */
92 u_quad_t ifi_noproto; /* destined for unsupported protocol */
93 struct timeval50 ifi_lastchange;/* last operational state change */
94};
95
96struct oifdatareq {
97 char ifdr_name[OIFNAMSIZ]; /* if name, e.g. "en0" */
98 struct oif_data ifdr_data;
99};
