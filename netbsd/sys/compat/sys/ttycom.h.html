44#ifdef COMPAT_60
45
46/*
47 * The cn and sn fields in struct ptmget changed size from
48 * char[16] to char[PATH_MAX] in NetBSD-6.99.14.
49 */
50struct compat_60_ptmget {
51 int cfd;
52 int sfd;
53 char cn[16];
54 char sn[16];
55};
56
57#define COMPAT_60_TIOCPTMGET _IOR('t', 70, struct compat_60_ptmget)
58#define COMPAT_60_TIOCPTSNAME _IOR('t', 72, struct compat_60_ptmget)

65#endif /* COMPAT_60 */
