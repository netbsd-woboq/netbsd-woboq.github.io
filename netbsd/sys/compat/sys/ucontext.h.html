32#ifndef _COMPAT_SYS_UCONTEXT_H_
33#define _COMPAT_SYS_UCONTEXT_H_
34
35#include <compat/sys/sigtypes.h>
36
37#ifdef _KERNEL
38
39typedef struct __ucontext32 ucontext32_t;
40
41struct __ucontext32 {
42 unsigned int uc_flags; /* properties */
43 uint32_t uc_link; /* context to resume */
44 sigset_t uc_sigmask; /* signals blocked in this context */
45 stack32_t uc_stack; /* the stack used by this context */
46 mcontext32_t uc_mcontext; /* machine state */
47#if defined(_UC_MACHINE32_PAD)
48 int __uc_pad[_UC_MACHINE32_PAD];
49#endif
50};
51
52#ifdef __UCONTEXT32_SIZE
53__CTASSERT(sizeof(ucontext32_t) == __UCONTEXT32_SIZE);
54#endif
55
56
57struct lwp;
58void getucontext32(struct lwp *, ucontext32_t *);
59int setucontext32(struct lwp *, const ucontext32_t *);
60int cpu_mcontext32_validate(struct lwp *, const mcontext32_t *);
61void cpu_getmcontext32(struct lwp *, mcontext32_t *, unsigned int *);
62int cpu_setmcontext32(struct lwp *, const mcontext32_t *, unsigned int);
63
64#endif /* _KERNEL */
65
66#endif /* !_COMPAT_SYS_UCONTEXT_H_ */
