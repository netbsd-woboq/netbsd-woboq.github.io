69void BF_cbc_encrypt(const unsigned char *in, unsigned char *out, long length,
70 const BF_KEY *schedule, unsigned char *ivec, int encrypt)
71 {
72 register BF_LONG tin0,tin1;
73 register BF_LONG tout0,tout1,xor0,xor1;
74 register long l=length;
75 BF_LONG tin[2];
76
77 if (encrypt)
78 {
79 n2l(ivec,tout0);
80 n2l(ivec,tout1);
81 ivec-=8;
82 for (l-=8; l>=0; l-=8)
83 {
84 n2l(in,tin0);
85 n2l(in,tin1);
86 tin0^=tout0;
87 tin1^=tout1;
88 tin[0]=tin0;
89 tin[1]=tin1;
90 BF_encrypt(tin,(const BF_KEY *)schedule);
91 tout0=tin[0];
92 tout1=tin[1];
93 l2n(tout0,out);
94 l2n(tout1,out);
95 }
96 if (l != -8)
97 {
98 n2ln(in,tin0,tin1,l+8);
99 tin0^=tout0;
100 tin1^=tout1;
101 tin[0]=tin0;
102 tin[1]=tin1;
103 BF_encrypt(tin,(const BF_KEY *)schedule);
104 tout0=tin[0];
105 tout1=tin[1];
106 l2n(tout0,out);
107 l2n(tout1,out);
108 }
109 l2n(tout0,ivec);
110 l2n(tout1,ivec);
111 }
112 else
113 {
114 n2l(ivec,xor0);
115 n2l(ivec,xor1);
116 ivec-=8;
117 for (l-=8; l>=0; l-=8)
118 {
119 n2l(in,tin0);
120 n2l(in,tin1);
121 tin[0]=tin0;
122 tin[1]=tin1;
123 BF_decrypt(tin,(const BF_KEY *)schedule);
124 tout0=tin[0]^xor0;
125 tout1=tin[1]^xor1;
126 l2n(tout0,out);
127 l2n(tout1,out);
128 xor0=tin0;
129 xor1=tin1;
130 }
131 if (l != -8)
132 {
133 n2l(in,tin0);
134 n2l(in,tin1);
135 tin[0]=tin0;
136 tin[1]=tin1;
137 BF_decrypt(tin,(const BF_KEY *)schedule);
138 tout0=tin[0]^xor0;
139 tout1=tin[1]^xor1;
140 l2nn(tout0,tout1,out,l+8);
141 xor0=tin0;
142 xor1=tin1;
143 }
144 l2n(xor0,ivec);
145 l2n(xor1,ivec);
146 }
147 tin0=tin1=tout0=tout1=xor0=xor1=0;
148 tin[0]=tin[1]=0;
149 }
