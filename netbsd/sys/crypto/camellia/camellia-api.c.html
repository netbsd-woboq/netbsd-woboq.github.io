30#include <sys/cdefs.h>
31
32#include <sys/types.h>
33#include <crypto/camellia/camellia.h>
34
35void
36camellia_set_key(camellia_ctx *ctx, const u_char *key, int bits)
37{
38
39 Camellia_Ekeygen(bits, key, ctx->subkey);
40 ctx->bits = bits;
41}
42
43void
44camellia_decrypt(const camellia_ctx *ctx, const u_char *src, u_char *dst)
45{
46
47 Camellia_DecryptBlock(ctx->bits, src, ctx->subkey, dst);
48}
49
50void
51camellia_encrypt(const camellia_ctx *ctx, const u_char *src, u_char *dst)
52{
53
54 Camellia_EncryptBlock(ctx->bits, src, ctx->subkey, dst);
55}
