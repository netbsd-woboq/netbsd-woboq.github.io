30#ifndef _CAMELLIA_H
31#define _CAMELLIA_H
32
33#define CAMELLIA_BLOCK_SIZE 16
34#define CAMELLIA_SUBKEYWORD 68 /* (34*8/4) */
35
36typedef struct {
37 int bits; /* key-length */
38 uint32_t subkey[CAMELLIA_SUBKEYWORD]; /* encrypt/decrypt key schedule */
39} camellia_ctx;
40
41void camellia_set_key(camellia_ctx *, const u_char *, int);
42void camellia_decrypt(const camellia_ctx *, const u_char *, u_char *);
43void camellia_encrypt(const camellia_ctx *, const u_char *, u_char *);
44
45
46void Camellia_Ekeygen(const int keyBitLength,
47 const unsigned char *rawKey,
48 uint32_t *subkey);
49
50void Camellia_EncryptBlock(const int keyBitLength,
51 const unsigned char *plaintext,
52 const uint32_t *subkey,
53 unsigned char *cipherText);
54
55void Camellia_DecryptBlock(const int keyBitLength,
56 const unsigned char *cipherText,
57 const uint32_t *subkey,
58 unsigned char *plaintext);
59
60void camellia_setup128(const unsigned char *key, uint32_t *subkey);
61void camellia_setup192(const unsigned char *key, uint32_t *subkey);
62void camellia_setup256(const unsigned char *key, uint32_t *subkey);
63void camellia_encrypt128(const uint32_t *subkey, uint32_t *io);
64void camellia_encrypt256(const uint32_t *subkey, uint32_t *io);
65void camellia_decrypt128(const uint32_t *subkey, uint32_t *io);
66void camellia_decrypt256(const uint32_t *subkey, uint32_t *io);
67
68
69#endif /* _CAMELLIA_H */
