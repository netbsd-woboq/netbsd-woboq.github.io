51#ifndef HEADER_DES_H
52#define HEADER_DES_H

54#ifdef __cplusplus
55extern "C" {
56#endif
57
58/* must be 32bit quantity */
59#define DES_LONG u_int32_t
60
61typedef unsigned char des_cblock[8];
62typedef struct des_ks_struct
63 {
64 union {
65 des_cblock cblock;
66 /* make sure things are correct size on machines with
67 * 8 byte longs */
68 DES_LONG deslong[2];
69 } ks;
70 int weak_key;
71} des_key_schedule[16];
72
73#define DES_KEY_SZ (sizeof(des_cblock))
74#define DES_SCHEDULE_SZ (sizeof(des_key_schedule))
75
76#define DES_ENCRYPT 1
77#define DES_DECRYPT 0
78
79#define DES_CBC_MODE 0
80#define DES_PCBC_MODE 1
81
82extern int des_check_key; /* defaults to false */
83
84char *des_options(void);
85void des_ecb_encrypt(des_cblock *, des_cblock *,
86 des_key_schedule, int);
87
88void des_encrypt1(DES_LONG *, des_key_schedule, int);
89void des_encrypt2(DES_LONG *, des_key_schedule, int);
90void des_encrypt3(DES_LONG *, des_key_schedule, des_key_schedule,
91 des_key_schedule);
92void des_decrypt3(DES_LONG *, des_key_schedule, des_key_schedule,
93 des_key_schedule);
94
95void des_ecb3_encrypt(des_cblock *, des_cblock *, des_key_schedule,
96 des_key_schedule, des_key_schedule, int);
97
98void des_ncbc_encrypt(const unsigned char *, unsigned char *, long,
99 des_key_schedule, des_cblock *, int);
100
101void des_ede3_cbc_encrypt(const unsigned char *, unsigned char *, long,
102 des_key_schedule, des_key_schedule,
103 des_key_schedule, des_cblock *, int);
104
105void des_set_odd_parity(des_cblock *);
106void des_fixup_key_parity(des_cblock *);
107int des_is_weak_key(des_cblock *);
108int des_set_key(des_cblock *, des_key_schedule);
109int des_key_sched(des_cblock *, des_key_schedule);
110int des_set_key_checked(des_cblock *, des_key_schedule);
111void des_set_key_unchecked(des_cblock *, des_key_schedule);
112int des_check_key_parity(des_cblock *);
113

115}
