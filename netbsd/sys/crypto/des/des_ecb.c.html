67char *des_options(void)
68 {
69 static int init=1;
70 static char buf[32];
71
72 if (init)
73 {
74 const char *ptr,*unroll,*risc,*size;
75
76#ifdef DES_PTR
77 ptr="ptr";
78#else
79 ptr="idx";
80#endif
81#if defined(DES_RISC1) || defined(DES_RISC2)
82#ifdef DES_RISC1
83 risc="risc1";
84#endif
85#ifdef DES_RISC2
86 risc="risc2";
87#endif
88#else
89 risc="cisc";
90#endif
91#ifdef DES_UNROLL
92 unroll="16";
93#else
94 unroll="4";
95#endif
96 /*CONSTCOND*/
97 if (sizeof(DES_LONG) != sizeof(long))
98 size="int";
99 else
100 size="long";
101 snprintf(buf, sizeof(buf), "des(%s,%s,%s,%s)",
102 ptr, risc, unroll, size);
103 init=0;
104 }
105 return(buf);
106}
107void des_ecb_encrypt(des_cblock *input, des_cblock *output,
108 des_key_schedule ks, int enc)
109{
110 register DES_LONG l;
111 DES_LONG ll[2];
112 const unsigned char *in=&(*input)[0];
113 unsigned char *out = &(*output)[0];
114
115 c2l(in,l); ll[0]=l;
116 c2l(in,l); ll[1]=l;
117 des_encrypt1(ll,ks,enc);
118 l=ll[0]; l2c(l,out);
119 l=ll[1]; l2c(l,out);
120 l=ll[0]=ll[1]=0;
121}
122
123void des_ecb3_encrypt(des_cblock *input, des_cblock *output,
124 des_key_schedule ks1, des_key_schedule ks2, des_key_schedule ks3,
125 int enc)
126{
127 register DES_LONG l0,l1;
128 DES_LONG ll[2];
129 const unsigned char *in = &(*input)[0];
130 unsigned char *out = &(*output)[0];
131
132 c2l(in,l0);
133 c2l(in,l1);
134 ll[0]=l0;
135 ll[1]=l1;
136
137 if (enc)
138 des_encrypt3(ll,ks1,ks2,ks3);
139 else
140 des_decrypt3(ll,ks1,ks2,ks3);
141
142 l0=ll[0];
143 l1=ll[1];
144 l2c(l0,out);
145 l2c(l1,out);
146}
