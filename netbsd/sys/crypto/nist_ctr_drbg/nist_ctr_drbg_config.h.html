48/*
49 * NIST SP 800-90 Configuration (Random Number Generator)
50 */
51
52#ifndef NIST_CTR_DRBG_CONFIG_H
53#define NIST_CTR_DRBG_CONFIG_H
54
55/*
56 * The test vectors will indicate failure if the
57 * byte ordering is set incorrectly.
58 * Pretending to be little endian will improve performance
59 * slightly but should not have an impact on
60 * security (a byte-swapped nonce is still a nonce).
61 */
62#define NIST_IS_LITTLE_ENDIAN 1
63
64/* Without NIST_ZEROIZE, we leave some intermediaries on the stack. */
65#define NIST_ZEROIZE 1
66
67#include <crypto/nist_ctr_drbg/nist_ctr_aes_rijndael.h>
68
69/* Use AES-128 as the block cipher */
70#include <crypto/nist_ctr_drbg/nist_ctr_drbg_aes128.h>
71
72#endif /* NIST_CTR_DRBG_CONFIG_H */
