820/**

822 *
823 * @return the number of rounds for the given cipher key size.
824 */
825int rijndaelKeySetupDec(u32 rk[/*4*(Nr + 1)*/], const u8 cipherKey[], int keyBits) {
826 int Nr, i, j;
827 u32 temp;
828
829 /* expand the cipher key: */
830 Nr = rijndaelKeySetupEnc(rk, cipherKey, keyBits);
831 /* invert the order of the round keys: */
832 for (i = 0, j = 4*Nr; i < j; i += 4, j -= 4) {
833 temp = rk[i ]; rk[i ] = rk[j ]; rk[j ] = temp;
834 temp = rk[i + 1]; rk[i + 1] = rk[j + 1]; rk[j + 1] = temp;
835 temp = rk[i + 2]; rk[i + 2] = rk[j + 2]; rk[j + 2] = temp;
836 temp = rk[i + 3]; rk[i + 3] = rk[j + 3]; rk[j + 3] = temp;
837 }
838 /* apply the inverse MixColumn transform to all round keys but the first and the last: */
839 for (i = 1; i < Nr; i++) {
840 rk += 4;
841 rk[0] =
842 Td0[Te4[(rk[0] >> 24) ] & 0xff] ^
843 Td1[Te4[(rk[0] >> 16) & 0xff] & 0xff] ^
844 Td2[Te4[(rk[0] >> 8) & 0xff] & 0xff] ^

849 Td2[Te4[(rk[1] >> 8) & 0xff] & 0xff] ^
