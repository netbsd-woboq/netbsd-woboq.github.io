55#define XTS_ALPHA 0x87

57static void xor16(uint8_t *d, const uint8_t *a, const uint8_t* b)
58{
59 for (size_t i = 0; i < 4; i++) {
60 *d++ = *a++ ^ *b++;
61 *d++ = *a++ ^ *b++;
62 *d++ = *a++ ^ *b++;
63 *d++ = *a++ ^ *b++;
64 }
65}

67static void
68xts_exponentiate(uint8_t *iv)
69{
70 unsigned int carry = 0;
71
72 for (size_t i = 0; i < 16; i++) {
73 unsigned int msb = iv[i] >> 7;
74
75 iv[i] = (iv[i] << 1) | carry;
76 carry = msb;
77 }
78
79 if (carry != 0)
80 iv[0] ^= XTS_ALPHA;
81}

83int
84rijndael_makeKey(keyInstance *key, BYTE direction, int keyLen,
85 const char *keyMaterial)
86{
87 u_int8_t cipherKey[RIJNDAEL_MAXKB];
88
89 if (key == NULL) {
90 return BAD_KEY_INSTANCE;
91 }
92
93 if ((direction == DIR_ENCRYPT) || (direction == DIR_DECRYPT)) {
94 key->direction = direction;
95 } else {
96 return BAD_KEY_DIR;
97 }
98
99 if ((keyLen == 128) || (keyLen == 192) || (keyLen == 256)) {
100 key->keyLen = keyLen;
101 } else {
102 return BAD_KEY_MAT;
103 }
104
105 if (keyMaterial != NULL) {
106 memcpy(key->keyMaterial, keyMaterial, keyLen/8);
107 }
108
109 /* initialize key schedule: */
110 memcpy(cipherKey, key->keyMaterial, keyLen/8);
111 if (direction == DIR_ENCRYPT) {
112 key->Nr = rijndaelKeySetupEnc(key->rk, cipherKey, keyLen);
113 } else {
114 key->Nr = rijndaelKeySetupDec(key->rk, cipherKey, keyLen);
115 }
116 rijndaelKeySetupEnc(key->ek, cipherKey, keyLen);
117 return TRUE;
118}

120int
121rijndael_cipherInit(cipherInstance *cipher, BYTE mode, const char *IV)
122{
123 if ((mode == MODE_ECB) || (mode == MODE_CBC) ||
124 (mode == MODE_XTS) || (mode == MODE_CFB1)) {
125 cipher->mode = mode;
126 } else {
127 return BAD_CIPHER_MODE;
128 }
129 if (IV != NULL) {
130 memcpy(cipher->IV, IV, RIJNDAEL_MAX_IV_SIZE);
131 } else {
132 memset(cipher->IV, 0, RIJNDAEL_MAX_IV_SIZE);
133 }
134 return TRUE;
135}

137int
138rijndael_blockEncrypt(cipherInstance *cipher, keyInstance *key,
139 const BYTE *input, int inputLen, BYTE *outBuffer)
140{
141 int i, k, t, numBlocks;
142 u_int8_t block[16], *iv;
143
144 if (cipher == NULL ||
145 key == NULL ||
146 key->direction == DIR_DECRYPT) {
147 return BAD_CIPHER_STATE;
148 }
149 if (input == NULL || inputLen <= 0) {
150 return 0; /* nothing to do */
151 }
152
153 numBlocks = inputLen/128;
154
155 switch (cipher->mode) {
156 case MODE_ECB:
157 for (i = numBlocks; i > 0; i--) {
158 rijndaelEncrypt(key->rk, key->Nr, input, outBuffer);
159 input += 16;
160 outBuffer += 16;
161 }
162 break;
163
164 case MODE_CBC:
165 iv = (u_int8_t *)cipher->IV;
166 for (i = numBlocks; i > 0; i--) {
167 xor16(block, input, iv);
168 rijndaelEncrypt(key->rk, key->Nr, block, outBuffer);
169 iv = outBuffer;
170 input += 16;
171 outBuffer += 16;
172 }
173 break;
174
175 case MODE_XTS:
176 iv = (u_int8_t *)cipher->IV;
177 for (i = numBlocks; i > 0; i--) {
178 xor16(block, input, iv);
179 rijndaelEncrypt(key->rk, key->Nr, block, block);
180 xor16(outBuffer, block, iv);
181 xts_exponentiate(iv);
182 input += 16;
183 outBuffer += 16;
184 }
185 break;
186
187 case MODE_CFB1:
188 iv = (u_int8_t *)cipher->IV;
189 for (i = numBlocks; i > 0; i--) {
190 memcpy(outBuffer, input, 16);
191 for (k = 0; k < 128; k++) {
192 rijndaelEncrypt(key->ek, key->Nr, iv, block);
193 outBuffer[k >> 3] ^=
194 (block[0] & 0x80U) >> (k & 7);
195 for (t = 0; t < 15; t++) {
196 iv[t] = (iv[t] << 1) | (iv[t + 1] >> 7);
197 }
198 iv[15] = (iv[15] << 1) |
199 ((outBuffer[k >> 3] >> (7 - (k & 7))) & 1);
200 }
201 outBuffer += 16;
202 input += 16;
203 }
204 break;
205
206 default:
207 return BAD_CIPHER_STATE;
208 }
209
210 return 128 * numBlocks;
211}

213/**
214 * Encrypt data partitioned in octets, using RFC 2040-like padding.
215 *
216 * @param input data to be encrypted (octet sequence)
217 * @param inputOctets input length in octets (not bits)
218 * @param outBuffer encrypted output data
219 *
220 * @return length in octets (not bits) of the encrypted output buffer.
221 */
222int
223rijndael_padEncrypt(cipherInstance *cipher, keyInstance *key,
224 const BYTE *input, int inputOctets, BYTE *outBuffer)
225{
226 int i, numBlocks, padLen;
227 u_int8_t block[16], *iv;
228
229 if (cipher == NULL ||
230 key == NULL ||
231 key->direction == DIR_DECRYPT) {
232 return BAD_CIPHER_STATE;
233 }
234 if (input == NULL || inputOctets <= 0) {
235 return 0; /* nothing to do */
236 }
237
238 numBlocks = inputOctets / 16;
239
240 switch (cipher->mode) {
241 case MODE_ECB:
242 for (i = numBlocks; i > 0; i--) {
243 rijndaelEncrypt(key->rk, key->Nr, input, outBuffer);
244 input += 16;
245 outBuffer += 16;
246 }
247 padLen = 16 - (inputOctets - 16*numBlocks);
248 memcpy(block, input, 16 - padLen);
249 memset(block + 16 - padLen, padLen, padLen);
250 rijndaelEncrypt(key->rk, key->Nr, block, outBuffer);
251 break;
252
253 case MODE_CBC:
254 iv = (u_int8_t *)cipher->IV;
255 for (i = numBlocks; i > 0; i--) {
256 xor16(block, input, iv);
257 rijndaelEncrypt(key->rk, key->Nr, block, outBuffer);
258 iv = outBuffer;
259 input += 16;
260 outBuffer += 16;
261 }
262 padLen = 16 - (inputOctets - 16*numBlocks);
263 for (i = 0; i < 16 - padLen; i++) {
264 block[i] = input[i] ^ iv[i];
265 }
266 for (i = 16 - padLen; i < 16; i++) {
267 block[i] = (BYTE)padLen ^ iv[i];
268 }
269 rijndaelEncrypt(key->rk, key->Nr, block, outBuffer);
270 break;
271
272 default:
273 return BAD_CIPHER_STATE;
274 }
275
276 return 16 * (numBlocks + 1);
277}

279int
280rijndael_blockDecrypt(cipherInstance *cipher, keyInstance *key,
281 const BYTE *input, int inputLen, BYTE *outBuffer)
282{
283 int i, k, t, numBlocks;
284 u_int8_t block[16], *iv;
285
286 if (cipher == NULL ||
287 key == NULL ||
288 (cipher->mode != MODE_CFB1 && key->direction == DIR_ENCRYPT)) {
289 return BAD_CIPHER_STATE;
290 }
291 if (input == NULL || inputLen <= 0) {
292 return 0; /* nothing to do */
293 }
294
295 numBlocks = inputLen/128;
296
297 switch (cipher->mode) {
298 case MODE_ECB:
299 for (i = numBlocks; i > 0; i--) {
300 rijndaelDecrypt(key->rk, key->Nr, input, outBuffer);
301 input += 16;
302 outBuffer += 16;
303 }
304 break;
305
306 case MODE_CBC:
307 iv = (u_int8_t *)cipher->IV;
308 for (i = numBlocks; i > 0; i--) {
309 rijndaelDecrypt(key->rk, key->Nr, input, block);
310 xor16(block, block, iv);
311 memcpy(cipher->IV, input, 16);
312 memcpy(outBuffer, block, 16);
313 input += 16;
314 outBuffer += 16;
315 }
316 break;
317
318 case MODE_XTS:
319 iv = (u_int8_t *)cipher->IV;
320 for (i = numBlocks; i > 0; i--) {
321 xor16(block, input, iv);
322 rijndaelDecrypt(key->rk, key->Nr, block, block);
323 xor16(outBuffer, block, iv);
324 xts_exponentiate(iv);
325 input += 16;
326 outBuffer += 16;
327 }
328 break;
329
330 case MODE_CFB1:
331 iv = (u_int8_t *)cipher->IV;
332 for (i = numBlocks; i > 0; i--) {
333 memcpy(outBuffer, input, 16);
334 for (k = 0; k < 128; k++) {
335 rijndaelEncrypt(key->ek, key->Nr, iv, block);
336 for (t = 0; t < 15; t++) {
337 iv[t] = (iv[t] << 1) | (iv[t + 1] >> 7);
338 }
339 iv[15] = (iv[15] << 1) |
340 ((input[k >> 3] >> (7 - (k & 7))) & 1);
341 outBuffer[k >> 3] ^= (block[0] & 0x80U) >>
342 (k & 7);
343 }
344 outBuffer += 16;
345 input += 16;
346 }
347 break;
348
349 default:
350 return BAD_CIPHER_STATE;
351 }
352
353 return 128 * numBlocks;
354}

356int
357rijndael_padDecrypt(cipherInstance *cipher, keyInstance *key,
358 const BYTE *input, int inputOctets, BYTE *outBuffer)
359{
360 int i, numBlocks, padLen;
361 u_int8_t block[16], *iv;
362
363 if (cipher == NULL ||
364 key == NULL ||
365 key->direction == DIR_ENCRYPT) {
366 return BAD_CIPHER_STATE;
367 }
368 if (input == NULL || inputOctets <= 0) {
369 return 0; /* nothing to do */
370 }
371 if (inputOctets % 16 != 0) {
372 return BAD_DATA;
373 }
374
375 numBlocks = inputOctets/16;
376
377 switch (cipher->mode) {
378 case MODE_ECB:
379 /* all blocks but last */
380 for (i = numBlocks - 1; i > 0; i--) {
381 rijndaelDecrypt(key->rk, key->Nr, input, outBuffer);
382 input += 16;
383 outBuffer += 16;
384 }
385 /* last block */
386 rijndaelDecrypt(key->rk, key->Nr, input, block);
387 padLen = block[15];
388 if (padLen >= 16) {
389 return BAD_DATA;
390 }
391 for (i = 16 - padLen; i < 16; i++) {
392 if (block[i] != padLen) {
393 return BAD_DATA;
394 }
395 }
396 memcpy(outBuffer, block, 16 - padLen);
397 break;
398
399 case MODE_CBC:
400 iv = (u_int8_t *)cipher->IV;
401 /* all blocks but last */
402 for (i = numBlocks - 1; i > 0; i--) {
403 rijndaelDecrypt(key->rk, key->Nr, input, block);
404 xor16(block, block, iv);
405 memcpy(cipher->IV, input, 16);
406 memcpy(outBuffer, block, 16);
407 input += 16;
408 outBuffer += 16;
409 }
410 /* last block */
411 rijndaelDecrypt(key->rk, key->Nr, input, block);
412 xor16(block, block, iv);
413 padLen = block[15];
414 if (padLen <= 0 || padLen > 16) {
415 return BAD_DATA;
416 }
417 for (i = 16 - padLen; i < 16; i++) {
418 if (block[i] != padLen) {
419 return BAD_DATA;
420 }
421 }
422 memcpy(outBuffer, block, 16 - padLen);
423 break;
424
425 default:
426 return BAD_CIPHER_STATE;
427 }
428
429 return 16 * numBlocks - padLen;
430}
