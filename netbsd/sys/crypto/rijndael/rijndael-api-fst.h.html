40#ifndef __RIJNDAEL_API_FST_H
41#define __RIJNDAEL_API_FST_H
42
43#include "rijndael-alg-fst.h"
44
45/* Generic Defines */
46#define DIR_ENCRYPT 0 /* Are we encrpyting? */
47#define DIR_DECRYPT 1 /* Are we decrpyting? */
48#define MODE_ECB 1 /* Are we ciphering in ECB mode? */
49#define MODE_CBC 2 /* Are we ciphering in CBC mode? */
50#define MODE_CFB1 3 /* Are we ciphering in 1-bit CFB mode? */
51#define MODE_XTS 4 /* Are we ciphering in XTS mode? */
52#define TRUE 1
53#define FALSE 0
54#define BITSPERBLOCK 128 /* Default number of bits in a cipher block */
55
56/* Error Codes */
57#define BAD_KEY_DIR -1 /* Key direction is invalid, e.g., unknown value */
58#define BAD_KEY_MAT -2 /* Key material not of correct length */
59#define BAD_KEY_INSTANCE -3 /* Key passed is not valid */
60#define BAD_CIPHER_MODE -4 /* Params struct passed to cipherInit invalid */
61#define BAD_CIPHER_STATE -5 /* Cipher in wrong state (e.g., not initialized) */
62#define BAD_BLOCK_LENGTH -6
63#define BAD_CIPHER_INSTANCE -7
64#define BAD_DATA -8 /* Data contents are invalid, e.g., invalid padding */
65#define BAD_OTHER -9 /* Unknown error */
66
67/* Algorithm-specific Defines */
68#define RIJNDAEL_MAX_KEY_SIZE 64 /* # of ASCII char's needed to represent a key */
69#define RIJNDAEL_MAX_IV_SIZE 16 /* # bytes needed to represent an IV */
70
71/* Typedefs */
72
73typedef unsigned char BYTE;
74
75/* The structure for key information */
76typedef struct {
77 BYTE direction; /* Key used for encrypting or decrypting? */
78 int keyLen; /* Length of the key */
79 char keyMaterial[RIJNDAEL_MAX_KEY_SIZE+1]; /* Raw key data in ASCII, e.g., user input or KAT values */
80 int Nr; /* key-length-dependent number of rounds */
81 u_int32_t rk[4*(RIJNDAEL_MAXNR + 1)]; /* key schedule */
82 u_int32_t ek[4*(RIJNDAEL_MAXNR + 1)]; /* CFB1 key schedule (encryption only) */
83} keyInstance;
84
85/* The structure for cipher information */
86typedef struct { /* changed order of the components */
87 u_int32_t IV[RIJNDAEL_MAX_IV_SIZE / sizeof(u_int32_t)];
88 /* A possible Initialization Vector for ciphering */
89 BYTE mode; /* MODE_ECB, MODE_CBC, or MODE_CFB1 */
90} cipherInstance;
91
92/* Function prototypes */
93
94int rijndael_makeKey(keyInstance *, BYTE, int, const char *);
95
96int rijndael_cipherInit(cipherInstance *, BYTE, const char *);
97
98int rijndael_blockEncrypt(cipherInstance *, keyInstance *, const BYTE *, int, BYTE *);
99
100int rijndael_padEncrypt(cipherInstance *, keyInstance *, const BYTE *, int, BYTE *);
101
102int rijndael_blockDecrypt(cipherInstance *, keyInstance *, const BYTE *, int, BYTE *);
103
104int rijndael_padDecrypt(cipherInstance *, keyInstance *, const BYTE *, int, BYTE *);
105
106#endif /* __RIJNDAEL_API_FST_H */
