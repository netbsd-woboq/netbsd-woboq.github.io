41static const char *classnames[] = {
42 "dull", "cpu", "disk", "ifnet", "tape", "tty",
43 "audiodev", "displaydev", "bus", "virtual"
44};
45
46struct device *
47db_device_first(void)
48{
49
50 return db_read_ptr("alldevs");
51}
52
53struct device *
54db_device_next(struct device *dv)
55{
56
57 db_read_bytes((db_addr_t)&dv->dv_list.tqe_next, sizeof(dv),
58 (char *)&dv);
59 return dv;
60}
61
62void
63db_show_all_devices(db_expr_t addr, bool haddr, db_expr_t count,
64 const char *modif)
65{
66 struct device buf;
67 struct device *dv;
68 const char *cl;
69
70 db_printf("%-16s %10s %4s %18s %18s\n",
71 "NAME","CLASS","UNIT","DEVICE_T","PRIVATE");
72
73 for (dv = db_device_first(); dv != NULL; dv = db_device_next(dv)) {
74 db_read_bytes((db_addr_t)dv, sizeof(buf), (char *)&buf);
75
76 unsigned i = buf.dv_class;
77
78 if (i >= __arraycount(classnames))
79 cl = "????";
80 else
81 cl = classnames[i];
82
83 db_printf("%-16.16s", buf.dv_xname);
84 db_printf(" %10s", cl);
85 db_printf(" %4u", buf.dv_unit);
86 db_printf(" %18" PRIxPTR, (uintptr_t)dv);
87 db_printf(" %18" PRIxPTR, (uintptr_t)buf.dv_private);
88 db_printf("\n");
89 }
90}
