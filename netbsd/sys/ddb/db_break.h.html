32#ifndef _DDB_DB_BREAK_H_
33#define _DDB_DB_BREAK_H_
34
35#include <uvm/uvm_extern.h>
36
37/*
38 * Breakpoints.
39 */
40typedef struct db_breakpoint {
41 struct vm_map *map; /* in this map */
42 db_addr_t address; /* set here */
43 int init_count; /* number of times to skip bkpt */
44 int count; /* current count */
45 int flags; /* flags: */
46#define BKPT_SINGLE_STEP 0x2 /* to simulate single step */
47#define BKPT_TEMP 0x4 /* temporary */
48 db_expr_t bkpt_inst; /* saved instruction at bkpt */
49 struct db_breakpoint *link; /* link in in-use or free chain */
50} *db_breakpoint_t;
51
52db_breakpoint_t db_find_breakpoint_here(db_addr_t);
53void db_set_breakpoints(void);
54void db_clear_breakpoints(void);
55void db_delete_cmd(db_expr_t, bool, db_expr_t, const char *);
56void db_breakpoint_cmd(db_expr_t, bool, db_expr_t, const char *);
57void db_listbreak_cmd(db_expr_t, bool, db_expr_t, const char *);
58bool db_map_equal(struct vm_map *, struct vm_map *);
59bool db_map_current(struct vm_map *);
60struct vm_map *db_map_addr(vaddr_t);
61
62#endif /* _DDB_DB_BREAK_H_ */
