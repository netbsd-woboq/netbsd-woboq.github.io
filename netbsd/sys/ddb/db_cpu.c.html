35#ifndef _KERNEL
36#include <stdbool.h>
37#endif

39#include <sys/param.h>
40#include <sys/cpu.h>
41#include <sys/proc.h>
42
43#include <ddb/ddb.h>
44
45static int db_ncpu;
46static struct cpu_info **db_cpu_infos;
47
48static void
49db_cpu_init(void)
50{
51 db_expr_t addr;
52
53 db_value_of_name("ncpu", &addr);
54 db_read_bytes((db_addr_t)addr, sizeof(db_ncpu), (char *)&db_ncpu);
55
56 db_value_of_name("cpu_infos", &addr);
57 db_read_bytes((db_addr_t)addr, sizeof(db_cpu_infos),
58 (char *)&db_cpu_infos);
59}
60
61static struct cpu_info *
62db_cpu_by_index(u_int idx)
63{
64 struct cpu_info *ci;
65
66 db_read_bytes((db_addr_t)&db_cpu_infos[idx], sizeof(ci), (char *)&ci);
67 return ci;
68}
69
70struct cpu_info *
71db_cpu_first(void)
72{
73
74 db_cpu_init();
75 return db_cpu_by_index(0);
76}
77
78struct cpu_info *
79db_cpu_next(struct cpu_info *ci)
80{
81 u_int idx;
82
83 db_read_bytes((db_addr_t)&ci->ci_index, sizeof(idx), (char *)&idx);
84 if ((idx + 1) >= (u_int)db_ncpu)
85 return NULL;
86 return db_cpu_by_index(idx + 1);
87}
