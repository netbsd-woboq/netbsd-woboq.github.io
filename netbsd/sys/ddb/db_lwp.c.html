48lwp_t *
49db_lwp_first(void)
50{
51
52 return db_read_ptr("alllwp");
53}
54
55lwp_t *
56db_lwp_next(lwp_t *l)
57{
58
59 db_read_bytes((db_addr_t)&l->l_list.le_next, sizeof(l), (char *)&l);
60 return l;
61}
62
63void
64db_lwp_whatis(uintptr_t addr, void (*pr)(const char *, ...))
65{
66 uintptr_t stack;
67 lwp_t l, *lp;
68
69 for (lp = db_lwp_first(); lp != NULL; lp = db_lwp_next(lp)) {
70 db_read_bytes((db_addr_t)lp, sizeof(l), (char *)&l);
71 stack = (uintptr_t)KSTACK_LOWEST_ADDR((&l));
72 if (addr < stack || stack + KSTACK_SIZE <= addr) {
73 continue;
74 }
75 (*pr)("%p is %p+%zu, LWP %p's stack\n",
76 (void *)addr, (void *)stack,
77 (size_t)(addr - stack), lp);
78 }
79}
