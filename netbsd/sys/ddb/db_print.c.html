45/*ARGSUSED*/
46void
47db_show_regs(db_expr_t addr, bool have_addr,
48 db_expr_t count, const char *modif)
49{
50 const struct db_variable *regp;
51 struct db_variable vs;
52 db_expr_t value, offset;
53 const char *name;
54
55 for (regp = db_regs; regp < db_eregs; regp++) {
56 vs = *regp;
57 vs.modif = modif;
58 db_read_variable(&vs, &value);
59 db_printf("%-12s%s", vs.name, db_num_to_str(value));
60 db_find_xtrn_sym_and_offset((db_addr_t)value, &name, &offset);
61 if (name != NULL &&
62 (unsigned int) offset <= db_maxoff && offset != value) {
63 db_printf("\t%s", name);
64 if (offset != 0) {
65 char tbuf[24];
66
67 db_format_radix(tbuf, 24, offset, true);
68 db_printf("+%s", tbuf);
69 }
70 }
71 db_printf("\n");
72 }
73 db_print_loc_and_inst(PC_REGS(DDB_REGS));
74}
