32#ifndef _DB_VARIABLES_H_
33#define _DB_VARIABLES_H_
34
35/*
36 * Debugger variables.
37 */
38struct db_variable {
39 const char *name; /* Name of variable */
40 void *valuep; /* value of variable */
41 /* function to call when reading/writing */
42 int (*fcn)(const struct db_variable *, db_expr_t *, int);
43 const char *modif;
44#define DB_VAR_GET 0
45#define DB_VAR_SET 1
46};
47#define FCN_NULL ((int (*)(const struct db_variable *, db_expr_t *, int))0)
48
49extern const struct db_variable db_vars[]; /* debugger variables */
50extern const struct db_variable * const db_evars;
51extern const struct db_variable db_regs[]; /* machine registers */
52extern const struct db_variable * const db_eregs;
53
54int db_get_variable(db_expr_t *);
55int db_set_variable(db_expr_t);
56void db_read_variable(const struct db_variable *, db_expr_t *);
57void db_write_variable(const struct db_variable *, db_expr_t *);
58void db_set_cmd(db_expr_t, bool, db_expr_t, const char *);
59
60#endif /* _DB_VARIABLES_H_ */
