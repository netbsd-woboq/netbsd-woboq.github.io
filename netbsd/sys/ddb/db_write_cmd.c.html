40/*
41 * Write to file.
42 */
43/*ARGSUSED*/
44void
45db_write_cmd(db_expr_t address, bool have_addr,
46 db_expr_t count, const char *modif)
47{
48 db_addr_t addr;
49 db_expr_t old_value;
50 db_expr_t new_value;
51 int size;
52 bool wrote_one;
53 bool show_old_val;
54
55 addr = (db_addr_t) address;
56 wrote_one = false;
57 show_old_val = islower((unsigned char)modif[0]);
58
59 switch (tolower((unsigned char)modif[0])) {
60 case 'b':
61 size = 1;
62 break;
63 case 'h':
64 size = 2;
65 break;
66 case 'l':
67 case '\0':
68 size = 4;
69 break;
70 case 'q':
71 if (sizeof(db_expr_t) != sizeof(uint64_t)) {
72 size = -1;
73 db_error("q not supported\n");
74 /*NOTREACHED*/
75 }
76 /* FALLTHROUGH */
77 case 'L':
78 size = sizeof(db_expr_t);
79 break;
80 default:
81 size = -1;
82 db_error("Unknown size\n");
83 /*NOTREACHED*/
84 }
85
86 while (db_expression(&new_value)) {
87 db_printsym(addr, DB_STGY_ANY, db_printf);
88 if (show_old_val) {
89 old_value = db_get_value(addr, size, false);
90 db_printf("\t\t%s = ", db_num_to_str(old_value));
91 db_printf("%s\n", db_num_to_str(new_value));
92 }
93 else
94 db_printf("\t\t= %s\n", db_num_to_str(new_value));
95 db_put_value(addr, size, new_value);
96 addr += size;
97
98 wrote_one = true;
99 }
100
101 if (!wrote_one) {
102 db_error("Nothing written.\n");
103 /*NOTREACHED*/
104 }
105
106 db_next = addr;
107 db_prev = addr - size;
108
109 db_skip_to_eol();
110}
