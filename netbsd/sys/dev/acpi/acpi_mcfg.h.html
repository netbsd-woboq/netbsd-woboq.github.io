28#ifndef _SYS_DEV_ACPI_ACPI_MCFG_H
29#define _SYS_DEV_ACPI_ACPI_MCFG_H
30
31struct acpimcfg_ops;
32void acpimcfg_probe(struct acpi_softc *);
33int acpimcfg_init(bus_space_tag_t, const struct acpimcfg_ops *);
34int acpimcfg_map_bus(device_t, pci_chipset_tag_t, int);
35int acpimcfg_configure_bus(device_t, pci_chipset_tag_t, ACPI_HANDLE, int, int);
36
37int acpimcfg_conf_read(pci_chipset_tag_t, pcitag_t, int, pcireg_t *);
38int acpimcfg_conf_write(pci_chipset_tag_t, pcitag_t, int, pcireg_t);
39
40struct acpimcfg_ops {
41 /* validate MCFG memory region */
42 bool (*ao_validate)(uint64_t, int, int *);
43
44 /* override default bus_space(9) function */
45 uint32_t (*ao_read)(bus_space_tag_t, bus_space_handle_t,
46 bus_addr_t);
47 void (*ao_write)(bus_space_tag_t, bus_space_handle_t,
48 bus_addr_t, uint32_t);
49};
50
51#define ACPIMCFG_SIZE_PER_BUS (PCI_EXTCONF_SIZE * 32/*dev*/ * 8/*func*/)
52
53bool acpimcfg_default_validate(uint64_t, int, int *);
54uint32_t acpimcfg_default_read(bus_space_tag_t, bus_space_handle_t,
55 bus_addr_t);
56void acpimcfg_default_write(bus_space_tag_t, bus_space_handle_t,
57 bus_addr_t, uint32_t);
58
59#endif /* _SYS_DEV_ACPI_ACPI_PCI_H */
