41static ACPI_TABLE_SLIT *slit;
42
43bool
44acpislit_exist(void)
45{
46 ACPI_TABLE_HEADER *table;
47 ACPI_STATUS rv;
48
49 rv = AcpiGetTable(ACPI_SIG_SLIT, 1, (ACPI_TABLE_HEADER **)&table);
50 if (ACPI_FAILURE(rv))
51 return false;
52
53 /* Check if header is valid */
54 if (table == NULL)
55 return false;
56
57 if (table->Length == 0xffffffff)
58 return false;
59
60 slit = (ACPI_TABLE_SLIT *)table;
61
62 return true;
63}
64
65int
66acpislit_init(void)
67{
68 if (!acpislit_exist())
69 return EEXIST;
70 return 0;
71}
72
73void
74acpislit_dump(void)
75{
76 uint64_t idx, count;
77
78 count = acpislit_cpus();
79
80 aprint_debug("SLIT: matrix with %"PRIu64" nodes\n", count);
81 aprint_debug("SLIT: ");
82 for (idx = 0; idx < (count * count); idx++) {
83 aprint_debug("%u ", slit->Entry[idx]);
84 if ((idx % count) == (count - 1)) {
85 aprint_debug("\n");
86 if (idx < (count * count) - 1)
87 aprint_debug("SLIT: ");
88 }
89 }
90}
91
92uint32_t
93acpislit_cpus(void)
94{
95 return slit->LocalityCount;
96}
97
98uint32_t
99acpislit_distance(uint32_t cpu1, uint32_t cpu2)
100{
101 uint64_t idx = cpu1 * slit->LocalityCount + cpu2;
102 return slit->Entry[idx];
103}
