32#ifndef _SYS_DEV_ACPI_ACPI_SLIT_H
33#define _SYS_DEV_ACPI_ACPI_SLIT_H
34
35/* Returns true if ACPI SLIT table is available
36 * If table does not exist, all functions below
37 * have undefined behaviour.
38 */
39bool acpislit_exist(void);
40
41/* Initializes parser. Must be the first function
42 * being called when table is available.
43 */
44int acpislit_init(void);
45
46void acpislit_dump(void);
47
48/* Retrieve number of cpus */
49uint32_t acpislit_cpus(void);
50
51/* Get distance between cpu1 and cpu2 */
52uint32_t acpislit_distance(uint32_t, uint32_t);
53
54#endif /* !_SYS_DEV_ACPI_ACPI_SLIT_H */
