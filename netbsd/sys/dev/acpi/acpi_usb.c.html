32/*
33 * USB Device-Specific Method (_DSM) support:
34 *
35 * https://docs.microsoft.com/en-us/windows-hardware/drivers/bringup/usb-device-specific-method---dsm-
36 */
37
38#include <sys/cdefs.h>
39__KERNEL_RCSID(0, "$NetBSD: acpi_usb.c,v 1.1 2018/10/26 23:33:38 jmcneill Exp $");
40
41#include <sys/param.h>
42
43#include <dev/acpi/acpireg.h>
44#include <dev/acpi/acpivar.h>
45#include <dev/acpi/acpi_usb.h>
46
47/* Windows-defined USB subsystem device-specific method (_DSM) */
48static UINT8 ehci_acpi_dsm_uuid[ACPI_UUID_LENGTH] = {
49 0x85, 0xe3, 0x2e, 0xce, 0xe6, 0x00, 0xcb, 0x48,
50 0x9f, 0x05, 0x2e, 0xdb, 0x92, 0x7c, 0x48, 0x99
51};
52
53void
54acpi_usb_post_reset(ACPI_HANDLE handle)
55{
56 ACPI_OBJECT_LIST objs;
57 ACPI_OBJECT obj[4];
58 ACPI_HANDLE method;
59
60 /*
61 * Invoke the _DSM control method for post-reset processing function
62 * for dual-role USB controllers. If supported, this will perform any
63 * controller-specific initialization required to put the dual-role
64 * device into host mode.
65 */
66
67 objs.Count = 4;
68 objs.Pointer = obj;
69 obj[0].Type = ACPI_TYPE_BUFFER;
70 obj[0].Buffer.Length = ACPI_UUID_LENGTH;
71 obj[0].Buffer.Pointer = ehci_acpi_dsm_uuid;
72 obj[1].Type = ACPI_TYPE_INTEGER;
73 obj[1].Integer.Value = 0; /* Revision ID = 0 */
74 obj[2].Type = ACPI_TYPE_INTEGER;
75 obj[2].Integer.Value = 1; /* Function index = 1 */
76 obj[3].Type = ACPI_TYPE_PACKAGE;
77 obj[3].Package.Count = 0; /* Empty package (not used) */
78 obj[3].Package.Elements = NULL;
79
80 if (ACPI_FAILURE(AcpiGetHandle(handle, "_DSM", &method)))
81 return;
82
83 (void)AcpiEvaluateObject(method, NULL, &objs, NULL);
84}
