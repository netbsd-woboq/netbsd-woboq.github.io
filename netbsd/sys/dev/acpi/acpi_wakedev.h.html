29#ifndef _SYS_DEV_ACPI_ACPI_WAKEDEV_H
30#define _SYS_DEV_ACPI_ACPI_WAKEDEV_H
31
32struct acpi_wakedev {
33 ACPI_HANDLE aw_power[8]; /* Power resources */
34 ACPI_HANDLE aw_handle; /* Wake GPE handle */
35 ACPI_INTEGER aw_number; /* Wake GPE number */
36 ACPI_INTEGER aw_state; /* Highest sleep state for wake */
37 bool aw_enable; /* Wake enabled (sysctl)? */
38};
39
40void acpi_wakedev_init(struct acpi_devnode *);
41void acpi_wakedev_add(struct acpi_devnode *);
42void acpi_wakedev_commit(struct acpi_softc *, int);
43
44#endif /* !_SYS_DEV_ACPI_ACPI_WAKEDEV_H */
