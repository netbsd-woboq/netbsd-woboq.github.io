58extern int acpi_indebugger;
59
60/*
61 * AcpiOsPrintf:
62 *
63 * Formatted stream output.
64 */
65void ACPI_INTERNAL_VAR_XFACE
66AcpiOsPrintf(const char *Format, ...)
67{
68
69 va_list ap;
70
71 va_start(ap, Format);
72 AcpiOsVprintf(Format, ap);
73 va_end(ap);
74}
75
76void
77AcpiOsVprintf(const char *Format, va_list Args)
78{
79
80#ifdef DDB
81 if (acpi_indebugger)
82 db_vprintf(Format, Args);
83 else
84#endif
85 vprintf(Format, Args);
86}
