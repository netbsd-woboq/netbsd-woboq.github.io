38#ifndef _SYS_DEV_ACPI_ACPIREG_H
39#define _SYS_DEV_ACPI_ACPIREG_H
40
41/*
42 * A common device status mask.
43 */
44#define ACPI_STA_OK (ACPI_STA_DEVICE_PRESENT | \
45 ACPI_STA_DEVICE_ENABLED | \
46 ACPI_STA_DEVICE_FUNCTIONING)
47
48/*
49 * Operating System Capabilities, _OSC.
50 */
51#define ACPI_OSC_QUERY __BIT(0)
52#define ACPI_OSC_ERROR __BIT(1)
53#define ACPI_OSC_ERROR_UUID __BIT(2)
54#define ACPI_OSC_ERROR_REV __BIT(3)
55#define ACPI_OSC_ERROR_MASKED __BIT(4)
56
57/*
58 * PCI functions.
59 */
60#define ACPI_ADR_PCI_DEV(x) (((x) >> 16) & 0xffff)
61#define ACPI_ADR_PCI_FUNC(x) ((x) & 0xffff)
62#define ACPI_ADR_PCI_ALLFUNCS 0xffff
63
64#define ACPI_GAS_PCI_DEV(x) (((x) >> 32) & 0xffff)
65#define ACPI_GAS_PCI_FUNC(x) (((x) >> 16) & 0xffff)
66#define ACPI_GAS_PCI_REGOFF(x) ((x) & 0xffff)
67
68/*
69 * ACPI driver components.
70 */
71#define ACPI_BUS_COMPONENT 0x00010000
72#define ACPI_ACAD_COMPONENT 0x00020000
73#define ACPI_BAT_COMPONENT 0x00040000
74#define ACPI_BUTTON_COMPONENT 0x00080000
75#define ACPI_EC_COMPONENT 0x00100000
76#define ACPI_LID_COMPONENT 0x00200000
77#define ACPI_RESOURCE_COMPONENT 0x00400000
78#define ACPI_TZ_COMPONENT 0x00800000
79#define ACPI_DISPLAY_COMPONENT 0x01000000
80
81#endif /* !_SYS_DEV_ACPI_ACPIREG_H */
