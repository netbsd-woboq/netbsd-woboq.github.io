45static int mpu_acpi_match(device_t, cfdata_t, void *);
46static void mpu_acpi_attach(device_t, device_t, void *);
47
48struct mpu_acpi_softc {
49 struct mpu_softc sc_mpu;
50 kmutex_t sc_lock;
51};
52
53CFATTACH_DECL_NEW(mpu_acpi, sizeof(struct mpu_acpi_softc), mpu_acpi_match,
54 mpu_acpi_attach, NULL, NULL);
55
56/*
57 * Supported device IDs
58 */
59
60static const char * const mpu_acpi_ids[] = {
61 "PNPB006", /* Roland MPU-401 (compatible) MIDI UART */
62 NULL
63};
64
65/*
66 * mpu_acpi_match: autoconf(9) match routine
67 */
68static int
69mpu_acpi_match(device_t parent, cfdata_t match, void *aux)
70{
71 struct acpi_attach_args *aa = aux;
72
73 if (aa->aa_node->ad_type != ACPI_TYPE_DEVICE)
74 return 0;
75
76 return acpi_match_hid(aa->aa_node->ad_devinfo, mpu_acpi_ids);
77}
78
79/*
80 * mpu_acpi_attach: autoconf(9) attach routine
81 */
82static void
83mpu_acpi_attach(device_t parent, device_t self, void *aux)
84{
85 struct mpu_acpi_softc *asc = device_private(self);
86 struct mpu_softc *sc = &asc->sc_mpu;
87 struct acpi_attach_args *aa = aux;
88 struct acpi_resources res;
89 struct acpi_io *io;
90 struct acpi_irq *irq;
91 ACPI_STATUS rv;
92
93 /* parse resources */
94 rv = acpi_resource_parse(self, aa->aa_node->ad_handle, "_CRS",
95 &res, &acpi_resource_parse_ops_default);
96 if (ACPI_FAILURE(rv))
97 return;
98
99 /* find our i/o registers */
100 io = acpi_res_io(&res, 0);
101 if (io == NULL) {
102 aprint_error_dev(self,
103 "unable to find i/o register resource\n");
104 goto out;
105 }
106
107 /* find our IRQ */
108 irq = acpi_res_irq(&res, 0);
109 if (irq == NULL) {
110 aprint_error_dev(self, "unable to find irq resource\n");
111 goto out;
112 }
113
114 sc->iot = aa->aa_iot;
115 if (bus_space_map(sc->iot, io->ar_base, io->ar_length, 0, &sc->ioh)) {
116 aprint_error_dev(self, "can't map i/o space\n");
117 goto out;
118 }
119
120 sc->model = "Roland MPU-401 MIDI UART";
121 sc->sc_dev = self;
122 sc->lock = &asc->sc_lock;
123 mutex_init(&asc->sc_lock, MUTEX_DEFAULT, IPL_AUDIO);
124 mpu_attach(sc);
125
126 sc->arg = isa_intr_establish(aa->aa_ic, irq->ar_irq,
127 (irq->ar_type == ACPI_EDGE_SENSITIVE) ? IST_EDGE : IST_LEVEL,
128 IPL_AUDIO, mpu_intr, sc);
129
130 out:
131 acpi_resource_cleanup(&res);
132}
