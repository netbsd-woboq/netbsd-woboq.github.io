40/* autoconf(9) functions */
41static int ug_acpi_match(device_t, cfdata_t, void *);
42static void ug_acpi_attach(device_t, device_t, void *);
43
44CFATTACH_DECL_NEW(ug_acpi, sizeof(struct ug_softc), ug_acpi_match,
45 ug_acpi_attach, NULL, NULL);
46
47/*
48 * Supported devices
49 * XXX: only uGuru 2005 for now
50 */
51
52static const char* const ug_acpi_ids[] = {
53 "ABT2005", /* uGuru 2005 */
54 NULL
55};
56
57static int
58ug_acpi_match(device_t parent, cfdata_t match, void *aux)
59{
60 struct acpi_attach_args *aa = aux;
61
62 if (aa->aa_node->ad_type != ACPI_TYPE_DEVICE)
63 return 0;
64
65 return acpi_match_hid(aa->aa_node->ad_devinfo, ug_acpi_ids);
66}
67
68static void
69ug_acpi_attach(device_t parent, device_t self, void *aux)
70{
71 struct ug_softc *sc = device_private(self);
72 struct acpi_attach_args *aa = aux;
73 struct acpi_resources res;
74 struct acpi_io *io;
75 bus_space_handle_t ioh;
76 ACPI_STATUS rv;
77
78 /* parse resources */
79 rv = acpi_resource_parse(self, aa->aa_node->ad_handle, "_CRS",
80 &res, &acpi_resource_parse_ops_default);
81 if (ACPI_FAILURE(rv))
82 return;
83
84 /* find our i/o registers */
85 io = acpi_res_io(&res, 0);
86 if (io == NULL) {
87 aprint_error_dev(self,
88 "unable to find i/o register resource\n");
89 acpi_resource_cleanup(&res);
90 return;
91 }
92
93 if (bus_space_map(aa->aa_iot, io->ar_base, io->ar_length,
94 0, &ioh)) {
95 aprint_error_dev(self, "can't map i/o space\n");
96 acpi_resource_cleanup(&res);
97 return;
98 }
99
100 aprint_normal("%s", device_xname(self));
101
102 sc->version = 2; /* uGuru 2005 */
103 sc->sc_ioh = ioh;
104 sc->sc_iot = aa->aa_iot;
105 ug2_attach(self);
106
107 acpi_resource_cleanup(&res);
108}
