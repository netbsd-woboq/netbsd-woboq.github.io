30#ifndef _DEV_APM_APMIO_H_
31#define _DEV_APM_APMIO_H_
32
33#include <sys/types.h>
34#include <sys/ioccom.h>
35
36struct apm_event_info {
37 u_int type;
38 u_int index;
39 u_int spare[8];
40};
41
42struct apm_power_info {
43 u_char battery_state;
44 u_char ac_state;
45 u_char battery_life;
46 u_char minutes_valid;
47 u_int minutes_left; /* estimate */
48 u_int nbattery;
49 u_int batteryid;
50 u_int battery_flags;
51 u_int spare2[3];
52};
53
54struct apm_ctl {
55 u_int dev;
56 u_int mode;
57};
58
59#define APM_IOC_REJECT _IOW('A', 0, struct apm_event_info) /* reject request # */
60#define APM_IOC_STANDBY _IO('A', 1) /* put system into standby */
61#define APM_IOC_SUSPEND _IO('A', 2) /* put system into suspend */
62#define OAPM_IOC_GETPOWER _IOR('A', 3, struct apm_power_info) /* fetch battery state */
63#define APM_IOC_GETPOWER _IOWR('A', 3, struct apm_power_info) /* fetch battery state */
64#define APM_IOC_NEXTEVENT _IOR('A', 4, struct apm_event_info) /* fetch event */
65#define APM_IOC_DEV_CTL _IOW('A', 5, struct apm_ctl) /* put device into mode */
66
67#endif /* _DEV_APM_APMIO_H_ */
