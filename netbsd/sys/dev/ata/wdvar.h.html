27#ifndef _DEV_ATA_WDVAR_H_
28#define _DEV_ATA_WDVAR_H_
29
30#ifdef _KERNEL_OPT
31#include "opt_wd.h"
32#endif
33
34#include <dev/dkvar.h>
35#include <sys/sysctl.h>
36
37struct wd_softc {
38 /* General disk infos */
39 struct dk_softc sc_dksc;
40 kmutex_t sc_lock;
41 int sc_quirks; /* any quirks drive might have */
42
43 /* IDE disk soft states */
44 struct ata_drive_datas *drvp; /* Our controller's infos */
45 const struct ata_bustype *atabus;
46 struct ataparams sc_params;/* drive characteristics found */
47 int sc_flags;
48/*
49 * XXX Nothing resets this yet, but disk change sensing will when ATA-4 is
50 * more fully implemented.
51 */
52#define WDF_LOADED 0x010 /* parameters loaded */
53#define WDF_WAIT 0x020 /* waiting for resources */
54#define WDF_LBA 0x040 /* using LBA mode */
55#define WDF_LBA48 0x100 /* using 48-bit LBA mode */
56#define WDF_DIRTY 0x200 /* disk cache dirty */
57#define WDF_OPEN 0x400 /* device is open */
58 uint64_t sc_capacity; /* full capacity of the device */
59 uint64_t sc_capacity512; /* ... in DEV_BSIZE blocks */
60 uint32_t sc_capacity28; /* capacity accessible with LBA28 commands */
61 uint32_t sc_blksize; /* logical block size, in bytes */
62
63#ifdef WD_SOFTBADSECT
64 SLIST_HEAD(, disk_badsectors) sc_bslist;
65 u_int sc_bscount;
66#endif
67
68 /* Retry/requeue failed transfers */
69 SLIST_HEAD(, ata_xfer) sc_retry_list;
70 struct callout sc_retry_callout; /* retry callout handle */
71 struct callout sc_restart_diskqueue; /* restart queue processing */
72
73 SLIST_HEAD(, ata_xfer) sc_requeue_list;
74 struct callout sc_requeue_callout; /* requeue callout handle */
75
76 struct ata_xfer dump_xfer;
77
78 /* Sysctl nodes specific for the disk */
79 struct sysctllog *nodelog;
80 bool drv_ncq;
81#define WD_USE_NCQ(wd) \
82 ((wd)->drv_ncq && ((wd)->drvp->drive_flags & ATA_DRIVE_NCQ))
83 bool drv_ncq_prio;
84#define WD_USE_NCQ_PRIO(wd) \
85 ((wd)->drv_ncq_prio && ((wd)->drvp->drive_flags & ATA_DRIVE_NCQ_PRIO))
86#ifdef WD_CHAOS_MONKEY
87 int drv_chaos_freq; /* frequency of simulated bio errors */
88 int drv_chaos_cnt; /* count of processed bio read xfers */
89#endif
90 unsigned inflight;
91 char *sc_typename;
92};
93
94#endif /* _DEV_ATA_WDVAR_H_ */
