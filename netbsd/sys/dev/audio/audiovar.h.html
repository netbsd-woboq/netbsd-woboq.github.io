91 * Frequency range.
92 * For lower limit, there are some antique machines who supports under

115struct au_mixer_ports {
116 int index; /* index of port-selector mixerctl */
117 int master; /* index of master mixerctl */
118 int nports; /* number of selectable ports */
119 bool isenum; /* selector is enum type */
120 u_int allports; /* all aumasks or'd */
121 u_int aumask[AUDIO_N_PORTS]; /* exposed value of "ports" */

259/*
260 * Return true if 'fmt' is the internal format.
261 * It does not check for frequency and number of channels.
262 */
263static __inline bool
264audio_format2_is_internal(const audio_format2_t *fmt)
265{
266
267 if (fmt->encoding != AUDIO_ENCODING_SLINEAR_NE)
268 return false;
269 if (fmt->precision != AUDIO_INTERNAL_BITS)
270 return false;
271 if (fmt->stride != AUDIO_INTERNAL_BITS)
272 return false;
273 return true;
274}
275
276/*
277 * Return true if fmt's encoding is one of LINEAR.
278 */
279static __inline bool
280audio_format2_is_linear(const audio_format2_t *fmt)
281{
282 return (fmt->encoding == AUDIO_ENCODING_SLINEAR_LE)
283 || (fmt->encoding == AUDIO_ENCODING_SLINEAR_BE)
284 || (fmt->encoding == AUDIO_ENCODING_ULINEAR_LE)
285 || (fmt->encoding == AUDIO_ENCODING_ULINEAR_BE);
286}
287
288/*
289 * Return true if fmt's encoding is one of SLINEAR.
290 */
291static __inline bool
292audio_format2_is_signed(const audio_format2_t *fmt)
293{
294 return (fmt->encoding == AUDIO_ENCODING_SLINEAR_LE)
295 || (fmt->encoding == AUDIO_ENCODING_SLINEAR_BE);
296}
297
298/*
299 * Return fmt's endian as LITTLE_ENDIAN or BIG_ENDIAN.
300 */
301static __inline int
302audio_format2_endian(const audio_format2_t *fmt)
303{
304 if (fmt->stride == 8) {
305 /* HOST ENDIAN */
306 return BYTE_ORDER;
307 }
308
309 if (fmt->encoding == AUDIO_ENCODING_SLINEAR_LE ||
310 fmt->encoding == AUDIO_ENCODING_ULINEAR_LE) {
311 return LITTLE_ENDIAN;
312 }
313 if (fmt->encoding == AUDIO_ENCODING_SLINEAR_BE ||
314 fmt->encoding == AUDIO_ENCODING_ULINEAR_BE) {
315 return BIG_ENDIAN;
316 }
317 return BYTE_ORDER;
318}
