53struct bio_mapping {
54 LIST_ENTRY(bio_mapping) bm_link;
55 device_t bm_dev;
56 int (*bm_ioctl)(device_t, u_long, void *);

88static void
89bio_initialize(void)
90{
91 if (bio_lock_initialized)
92 return;
93
94 mutex_init(&bio_lock, MUTEX_DEFAULT, IPL_VM);
95 bio_lock_initialized = true;
96}

98void
99bioattach(int nunits)
100{
101 if (!bio_lock_initialized)
102 bio_initialize();
103}

117static int
118bioioctl(dev_t dev, u_long cmd, void *addr, int flag, struct lwp *l)
119{
120 struct bio_locate *locate;
121 struct bio_common *common;
122 char name[16];
123 int error;
124
125 switch(cmd) {
126 case BIOCLOCATE:
127 case BIOCINQ:
128 case BIOCDISK:
129 case BIOCDISK_NOVOL:
130 case BIOCVOL:
131 case OBIOCDISK:
132 case OBIOCVOL:
133 error = kauth_authorize_device_passthru(l->l_cred, dev,
134 KAUTH_REQ_DEVICE_RAWIO_PASSTHRU_READCONF, addr);
135 if (error)
136 return error;
137 break;
138 case BIOCBLINK:
139 case BIOCSETSTATE:
140 case BIOCVOLOPS:
141 error = kauth_authorize_device_passthru(l->l_cred, dev,
142 KAUTH_REQ_DEVICE_RAWIO_PASSTHRU_WRITECONF, addr);
143 if (error)
144 return error;
145 break;
146 case BIOCALARM: {
147 struct bioc_alarm *alarm = (struct bioc_alarm *)addr;
148 switch (alarm->ba_opcode) {
149 case BIOC_SADISABLE:
150 case BIOC_SAENABLE:
151 case BIOC_SASILENCE:
152 case BIOC_SATEST:
153 error = kauth_authorize_device_passthru(l->l_cred, dev,
154 KAUTH_REQ_DEVICE_RAWIO_PASSTHRU_WRITECONF, addr);
155 if (error)
156 return error;
157 break;
158 case BIOC_GASTATUS:
159 error = kauth_authorize_device_passthru(l->l_cred, dev,
160 KAUTH_REQ_DEVICE_RAWIO_PASSTHRU_READCONF, addr);
161 if (error)
162 return error;
163 break;
164 default:
165 return EINVAL;
166 }
167 break;
168 }
169 default:
170 return ENOTTY;
171 }
172
173 switch (cmd) {
174 case BIOCLOCATE:
175 locate = addr;
176 error = copyinstr(locate->bl_name, name, sizeof(name), NULL);
177 if (error != 0)
178 return error;
179 locate->bl_cookie = bio_lookup(name);
180 if (locate->bl_cookie == NULL)
181 return ENOENT;
182 break;
183
184 default:
185 common = addr;
186 mutex_enter(&bio_lock);
187 if (!bio_validate(common->bc_cookie)) {
188 mutex_exit(&bio_lock);
189 return ENOENT;
190 }
191 mutex_exit(&bio_lock);
192 MODULE_HOOK_CALL(compat_bio_30_hook,
193 (common->bc_cookie, cmd, addr, bio_delegate_ioctl),
194 enosys(), error);
195 if (error == ENOSYS)
196 error = bio_delegate_ioctl(common->bc_cookie, cmd,
197 addr);
198 return error;
199 }
200 return 0;
201}

203int
204bio_register(device_t dev, int (*ioctl)(device_t, u_long, void *))
205{
206 struct bio_mapping *bm;
207
208 if (!bio_lock_initialized)
209 bio_initialize();
210
211 bm = malloc(sizeof(*bm), M_DEVBUF, M_NOWAIT|M_ZERO);
212 if (bm == NULL)
213 return ENOMEM;
214 bm->bm_dev = dev;
215 bm->bm_ioctl = ioctl;
216 mutex_enter(&bio_lock);
217 LIST_INSERT_HEAD(&bios, bm, bm_link);
218 mutex_exit(&bio_lock);
219 return 0;
220}
221
222void
223bio_unregister(device_t dev)
224{
225 struct bio_mapping *bm, *next;
226
227 mutex_enter(&bio_lock);
228 for (bm = LIST_FIRST(&bios); bm != NULL; bm = next) {
229 next = LIST_NEXT(bm, bm_link);
230
231 if (dev == bm->bm_dev) {
232 LIST_REMOVE(bm, bm_link);
233 free(bm, M_DEVBUF);
234 }
235 }
236 mutex_exit(&bio_lock);
237}
238
239static struct bio_mapping *
240bio_lookup(char *name)
241{
242 struct bio_mapping *bm;
243
244 mutex_enter(&bio_lock);
245 LIST_FOREACH(bm, &bios, bm_link) {
246 if (strcmp(name, device_xname(bm->bm_dev)) == 0) {
247 mutex_exit(&bio_lock);
248 return bm;
249 }
250 }
251 mutex_exit(&bio_lock);
252 return NULL;
253}
254
255static int
256bio_validate(void *cookie)
257{
258 struct bio_mapping *bm;
259
260 LIST_FOREACH(bm, &bios, bm_link)
261 if (bm == cookie)
262 return 1;
263
264 return 0;
265}
266
267static int
268bio_delegate_ioctl(void *cookie, u_long cmd, void *addr)
269{
270 struct bio_mapping *bm = cookie;
271
272 return bm->bm_ioctl(bm->bm_dev, cmd, addr);
273}

275void
276bio_disk_to_envsys(envsys_data_t *edata, const struct bioc_disk *bd)
277{
278 switch (bd->bd_status) {
279 case BIOC_SDONLINE:
280 edata->value_cur = ENVSYS_DRIVE_ONLINE;
281 edata->state = ENVSYS_SVALID;
282 break;
283 case BIOC_SDOFFLINE:
284 edata->value_cur = ENVSYS_DRIVE_OFFLINE;
285 edata->state = ENVSYS_SCRITICAL;
286 break;
287 default:
288 edata->value_cur = ENVSYS_DRIVE_FAIL;
289 edata->state = ENVSYS_SCRITICAL;
290 break;
291 }
292}

294void
295bio_vol_to_envsys(envsys_data_t *edata, const struct bioc_vol *bv)
296{
297 switch (bv->bv_status) {
298 case BIOC_SVOFFLINE:
299 edata->value_cur = ENVSYS_DRIVE_OFFLINE;
300 edata->state = ENVSYS_SCRITICAL;
301 break;
302 case BIOC_SVDEGRADED:
303 edata->value_cur = ENVSYS_DRIVE_PFAIL;
304 edata->state = ENVSYS_SCRITICAL;
305 break;
306 case BIOC_SVBUILDING:
307 edata->value_cur = ENVSYS_DRIVE_BUILD;
308 edata->state = ENVSYS_SVALID;
309 break;
310 case BIOC_SVMIGRATING:
311 edata->value_cur = ENVSYS_DRIVE_MIGRATING;
312 edata->state = ENVSYS_SVALID;
313 break;
314 case BIOC_SVCHECKING:
315 edata->value_cur = ENVSYS_DRIVE_CHECK;
316 edata->state = ENVSYS_SVALID;
317 break;
318 case BIOC_SVREBUILD:
319 edata->value_cur = ENVSYS_DRIVE_REBUILD;
320 edata->state = ENVSYS_SCRITICAL;
321 break;
322 case BIOC_SVSCRUB:
323 case BIOC_SVONLINE:
324 edata->value_cur = ENVSYS_DRIVE_ONLINE;
325 edata->state = ENVSYS_SVALID;
326 break;
327 case BIOC_SVINVALID:
328 /* FALLTHROUGH */
329 default:
330 edata->value_cur = ENVSYS_DRIVE_EMPTY; /* unknown state */
331 edata->state = ENVSYS_SINVALID;
332 break;
333 }
