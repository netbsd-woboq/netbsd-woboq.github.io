105static void
106bthub_attach(device_t parent, device_t self, void *aux)
107{
108 bdaddr_t *addr = aux;
109 prop_dictionary_t dict;
110 prop_object_t obj;
111
112 dict = device_properties(self);
113 obj = prop_data_create_data(addr, sizeof(*addr));
114 prop_dictionary_set(dict, BTDEVladdr, obj);
115 prop_object_release(obj);
116
117 aprint_verbose(" %s %2.2x:%2.2x:%2.2x:%2.2x:%2.2x:%2.2x",
118 BTDEVladdr,
119 addr->b[5], addr->b[4], addr->b[3],
120 addr->b[2], addr->b[1], addr->b[0]);
121
122 aprint_normal("\n");
123
124 if (!pmf_device_register(self, NULL, NULL)) {
125 /*
126 * XXX this should not be allowed to happen, but
127 * avoiding it needs a pretty big rearrangement of
128 * device attachments.
129 */
130 aprint_error_dev(self, "couldn't establish power handler\n");
131 }
132}

134static int
135bthub_detach(device_t self, int flags)
136{
137
138 pmf_device_deregister(self);
139
140 return config_detach_children(self, flags);
141}

143/*****************************************************************************
144 *
145 * bthub access functions to control device
146 */
147
148int
149bthubioctl(dev_t devno, unsigned long cmd, void *data, int flag, struct lwp *l)
150{
151 prop_dictionary_t dict;
152 int err;
153
154 switch(cmd) {
155 case BTDEV_ATTACH:
156 case BTDEV_DETACH:
157 /* load dictionary */
158 err = prop_dictionary_copyin_ioctl(data, cmd, &dict);
159 if (err == 0) {
160 err = bthub_pioctl(devno, cmd, dict, flag, l);
161 prop_object_release(dict);
162 }
163 break;
164
165 default:
166 err = EPASSTHROUGH;
167 break;
168 }
169
170 return err;
171}

173static int
174bthub_pioctl(dev_t devno, unsigned long cmd, prop_dictionary_t dict,
175 int flag, struct lwp *l)
176{
177 prop_data_t laddr, raddr;
178 prop_string_t service;
179 prop_dictionary_t prop;
180 prop_object_t obj;
181 device_t dev, self;
182 deviter_t di;
183 int unit;
184
185 /* validate local address */
186 laddr = prop_dictionary_get(dict, BTDEVladdr);
187 if (prop_data_size(laddr) != sizeof(bdaddr_t))
188 return EINVAL;
189
190 /* locate the relevant bthub */
191 for (unit = 0 ; ; unit++) {
192 if (unit == bthub_cd.cd_ndevs)
193 return ENXIO;
194
195 self = device_lookup(&bthub_cd, unit);
196 if (self == NULL)
197 continue;
198
199 prop = device_properties(self);
200 obj = prop_dictionary_get(prop, BTDEVladdr);
201 if (prop_data_equals(laddr, obj))
202 break;
203 }
204
205 /* validate remote address */
206 raddr = prop_dictionary_get(dict, BTDEVraddr);
207 if (prop_data_size(raddr) != sizeof(bdaddr_t)
208 || bdaddr_any(prop_data_data_nocopy(raddr)))
209 return EINVAL;
210
211 /* validate service name */
212 service = prop_dictionary_get(dict, BTDEVservice);
213 if (prop_object_type(service) != PROP_TYPE_STRING)
214 return EINVAL;
215
216 /* locate matching child device, if any */
217 deviter_init(&di, 0);
218 while ((dev = deviter_next(&di)) != NULL) {
219 if (device_parent(dev) != self)
220 continue;
221
222 prop = device_properties(dev);
223
224 obj = prop_dictionary_get(prop, BTDEVraddr);
225 if (!prop_object_equals(raddr, obj))
226 continue;
227
228 obj = prop_dictionary_get(prop, BTDEVservice);
229 if (!prop_object_equals(service, obj))
230 continue;
231
232 break;
233 }
234 deviter_release(&di);
235
236 switch (cmd) {
237 case BTDEV_ATTACH: /* attach BTDEV */
238 if (dev != NULL)
239 return EADDRINUSE;
240
241 dev = config_found(self, dict, bthub_print);
242 if (dev == NULL)
243 return ENXIO;
244
245 prop = device_properties(dev);
246 prop_dictionary_set(prop, BTDEVladdr, laddr);
247 prop_dictionary_set(prop, BTDEVraddr, raddr);
248 prop_dictionary_set(prop, BTDEVservice, service);
249 break;
250
251 case BTDEV_DETACH: /* detach BTDEV */
252 if (dev == NULL)
253 return ENXIO;
254
255 config_detach(dev, DETACH_FORCE);
256 break;
257 }
258
259 return 0;
260}

262static int
263bthub_print(void *aux, const char *pnp)
264{
265 prop_dictionary_t dict = aux;
266 prop_object_t obj;
267 const bdaddr_t *raddr;
268
269 if (pnp != NULL) {
270 obj = prop_dictionary_get(dict, BTDEVtype);
271 aprint_normal("%s: %s '%s',", pnp, BTDEVtype,
272 prop_string_cstring_nocopy(obj));
273 }
274
275 obj = prop_dictionary_get(dict, BTDEVraddr);
276 raddr = prop_data_data_nocopy(obj);
277
278 aprint_verbose(" %s %2.2x:%2.2x:%2.2x:%2.2x:%2.2x:%2.2x",
279 BTDEVraddr,
280 raddr->b[5], raddr->b[4], raddr->b[3],
281 raddr->b[2], raddr->b[1], raddr->b[0]);
282
283 return UNCONF;
284}
