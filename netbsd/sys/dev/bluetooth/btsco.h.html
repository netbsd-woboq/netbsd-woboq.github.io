34#ifndef _DEV_BLUETOOTH_BTSCO_H_
35#define _DEV_BLUETOOTH_BTSCO_H_
36
37#include <sys/ioccom.h>
38#include <netbt/bluetooth.h>
39
40/* btsco(4) properties */
41#define BTSCOlisten "listen"
42#define BTSCOchannel "rfcomm-channel"
43
44/*
45 * We need to provide a way to get the SCO Audio contact information
46 * from the audio device so that the control channel can be set up
47 */
48
49struct btsco_info {
50 bdaddr_t laddr; /* controller addr */
51 bdaddr_t raddr; /* remote addr */
52 uint8_t channel; /* RFCOMM channel */
53 int vgs; /* mixer index speaker */
54 int vgm; /* mixer index mic */
55};
56
57#define BTSCO_GETINFO _IOR('b', 16, struct btsco_info)
58
59#define BTSCO_DELTA 16 /* volume delta */
60
61#endif /* _DEV_BLUETOOTH_BTSCO_H_ */
