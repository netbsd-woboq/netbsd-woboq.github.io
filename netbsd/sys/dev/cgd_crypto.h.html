32#ifndef _DEV_CGD_CRYPTO_H_
33#define _DEV_CGD_CRYPTO_H_
34
35#ifdef _KERNEL
36#define CGD_CIPHER_DECRYPT 1
37#define CGD_CIPHER_ENCRYPT 2
38#define CGD_AES_BLOCK_SIZE 16
39#define CGD_3DES_BLOCK_SIZE 8
40#define CGD_BF_BLOCK_SIZE 8
41
42typedef void *(cfunc_init)(size_t, const void *, size_t *);
43typedef void (cfunc_destroy)(void *);
44typedef void (cfunc_cipher)(void *, struct uio *, struct uio *, const void *,
45 int);
46typedef void (cfunc_cipher_prep)(void *, char *, const char *, size_t, int);
47
48struct cryptfuncs {
49 const char *cf_name; /* cipher name */
50 cfunc_init *cf_init; /* Initialisation function */
51 cfunc_destroy *cf_destroy; /* destruction function */
52 cfunc_cipher *cf_cipher; /* the cipher itself */
53 cfunc_cipher_prep *cf_cipher_prep; /* prepare to cipher */
54};
55
56const struct cryptfuncs *cryptfuncs_find(const char *);
57#endif /* _KERNEL */
58
59#endif /* _DEV_CGD_CRYPTO_H_ */
