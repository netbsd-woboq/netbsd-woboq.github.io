37/*
38 * "POSIX time" to/from "YY/MM/DD/hh/mm/ss"
39 */
40struct clock_ymdhms {
41 uint64_t dt_year;
42 uint8_t dt_mon;
43 uint8_t dt_day;
44 uint8_t dt_wday; /* Day of week */
45 uint8_t dt_hour;
46 uint8_t dt_min;
47 uint8_t dt_sec;
48};
49
50time_t clock_ymdhms_to_secs(struct clock_ymdhms *);
51int clock_secs_to_ymdhms(time_t, struct clock_ymdhms *);
52
53/*
54 * BCD to binary and binary to BCD.
55 */
56static __inline unsigned int
57bcdtobin(unsigned int bcd)
58{
59 return ((bcd >> 4) & 0x0f) * 10 + (bcd & 0x0f);
60}
61
62static __inline unsigned int
63bintobcd(unsigned int bin)
64{
65 return (((bin / 10) << 4) & 0xf0) | (bin % 10);
66}

68/*
69 * Interface to time-of-day clock devices.
70 *
71 * todr_gettime: convert time-of-day clock into a `struct timeval'
72 * todr_settime: set time-of-day clock from a `struct timeval'
73 *
74 * (this is probably not so useful:)
75 * todr_setwen: provide a machine-dependent TOD clock write-enable callback
76 * function which takes one boolean argument:
77 * 1 to enable writes; 0 to disable writes.
78 */
79struct timeval;
80struct todr_chip_handle {
81 void *cookie; /* Device specific data */
82 void *bus_cookie; /* Bus specific data */
83 time_t base_time; /* Base time (e.g. rootfs time) */
84
85 int (*todr_gettime)(struct todr_chip_handle *, struct timeval *);
86 int (*todr_settime)(struct todr_chip_handle *, struct timeval *);
87 int (*todr_gettime_ymdhms)(struct todr_chip_handle *,
88 struct clock_ymdhms *);
89 int (*todr_settime_ymdhms)(struct todr_chip_handle *,
90 struct clock_ymdhms *);
91 int (*todr_setwen)(struct todr_chip_handle *, int);
92
93};
94typedef struct todr_chip_handle *todr_chip_handle_t;
95
96#define todr_wenable(ct, v) if ((ct)->todr_setwen) \
97 ((*(ct)->todr_setwen)(ct, v))
98
99/*
100 * Probably these should evolve into internal routines in kern_todr.c.
101 */
102extern int todr_gettime(todr_chip_handle_t, struct timeval *);
103extern int todr_settime(todr_chip_handle_t, struct timeval *);
104
105/*
106 * Machine-dependent function that machine-independent RTC drivers can
107 * use to register their todr_chip_handle_t with inittodr()/resettodr().
108 */
109void todr_attach(todr_chip_handle_t);
