52#define SWAP16(x) ap->x = be16toh(ap->x)
53#define SWAP32(x) ap->x = be32toh(ap->x)

61static void
62swap_apple_drvr_descriptor(struct apple_drvr_descriptor *ap)
63{
64 SWAP32(descBlock);
65 SWAP16(descSize);
66 SWAP16(descType);
67}
68
69static void
70swap_apple_drvr_map(struct apple_drvr_map *ap)
71{
72 uint16_t i;
73
74 SWAP16(sbSig);
75 SWAP16(sbBlockSize);
76 SWAP32(sbBlkCount);
77 SWAP16(sbDevType);
78 SWAP16(sbDevID);
79 SWAP32(sbData);
80 SWAP16(sbDrvrCount);
81
82 if (ap->sbDrvrCount >= APPLE_DRVR_MAP_MAX_DESCRIPTORS)
83 ap->sbDrvrCount = APPLE_DRVR_MAP_MAX_DESCRIPTORS;
84
85 for (i = 0; i < ap->sbDrvrCount; i++)
86 swap_apple_drvr_descriptor(&ap->sb_dd[i]);
87}
88
89static void
90swap_apple_part_map_entry(struct apple_part_map_entry *ap)
91{
92 SWAP16(pmSig);
93 SWAP16(pmSigPad);
94 SWAP32(pmMapBlkCnt);
95 SWAP32(pmPyPartStart);
96 SWAP32(pmPartBlkCnt);
97 SWAP32(pmLgDataStart);
98 SWAP32(pmDataCnt);
99 SWAP32(pmPartStatus);
100 SWAP32(pmLgBootStart);
101 SWAP32(pmBootSize);
102 SWAP32(pmBootLoad);
103 SWAP32(pmBootLoad2);
104 SWAP32(pmBootEntry);
105 SWAP32(pmBootEntry2);
106 SWAP32(pmBootCksum);
107}

109static void
110swap_apple_blockzeroblock(struct apple_blockzeroblock *ap)
111{
112 SWAP32(bzbMagic);
113 SWAP16(bzbBadBlockInode);
114 SWAP16(bzbFlags);
115 SWAP16(bzbReserved);
116 SWAP32(bzbCreationTime);
117 SWAP32(bzbMountTime);
118 SWAP32(bzbUMountTime);
119}

134static struct {
135 const char *name;
136 const char *type;
137} map[] = {
138 { APPLE_PART_TYPE_UNIX, DKW_PTYPE_SYSV },
139 { APPLE_PART_TYPE_MAC, DKW_PTYPE_APPLEHFS },
140};

143static int
144dkwedge_discover_apple(struct disk *pdk, struct vnode *vp)
145{
146 size_t i, n;
147 int error;
148 void *buf;
149 uint32_t blocksize, blockcount, offset, rsize;
150 struct apple_drvr_map *am;
151 struct apple_part_map_entry *ae;
152 struct apple_blockzeroblock ab;
153 const char *ptype;
154
155 buf = DKW_MALLOC(ASIZE);
156 if ((error = dkwedge_read(pdk, vp, 0, buf, ASIZE)) != 0) {
157 DPRINTF("%s: read @%u %d\n", __func__, 0, error);
158 goto out;
159 }
160
161 am = buf;
162 swap_apple_drvr_map(am);
163
164 error = ESRCH;
165
166 if (am->sbSig != APPLE_DRVR_MAP_MAGIC) {
167 DPRINTF("%s: drvr magic %x != %x\n", __func__, am->sbSig,
168 APPLE_DRVR_MAP_MAGIC);
169 goto out;
170 }
171
172 blocksize = am->sbBlockSize;
173
174 rsize = 1 << (ilog2(MAX(sizeof(*ae), blocksize) - 1) + 1);
175 if (ASIZE < rsize) {
176 DPRINTF("%s: buffer too small %u < %u\n", __func__, ASIZE,
177 rsize);
178 goto out;
179 }
180
181 /* XXX Clamp entries at 512 for now. */
182 blockcount = am->sbBlkCount;
183 if (blockcount > 512) {
184 aprint_error("%s: WARNING: clamping number of blocks to "
185 "512 (was %u)\n", pdk->dk_name, blockcount);
186 blockcount = 512;
187 }
188
189 ae = buf;
190 offset = blocksize;
191 for (n = 0; n < blockcount; n++, offset += rsize) {
192 DPRINTF("%s: offset %x rsize %x\n", __func__, offset, rsize);
193 if ((error = dkwedge_read(pdk, vp, offset / DEV_BSIZE, buf,
194 rsize)) != 0) {
195 DPRINTF("%s: read @%u %d\n", __func__, offset,
196 error);
197 goto out;
198 }
199
200 swap_apple_part_map_entry(ae);
201 if (ae->pmSig != APPLE_PART_MAP_ENTRY_MAGIC) {
202 DPRINTF("%s: part magic %x != %x\n", __func__,
203 ae->pmSig, APPLE_PART_MAP_ENTRY_MAGIC);
204 break;
205 }
206
207 for (i = 0; i < __arraycount(map); i++)
208 if (strcasecmp(map[i].name, ae->pmPartType) == 0)
209 break;
210
211 DPRINTF("%s: %s/%s PH=%u/%u LG=%u/%u\n", __func__,
212 ae->pmPartName, ae->pmPartType,
213 ae->pmPyPartStart, ae->pmPartBlkCnt,
214 ae->pmLgDataStart, ae->pmDataCnt);
215
216 if (i == __arraycount(map))
217 continue;
218
219 ptype = map[i].type;
220 memcpy(&ab, ae->pmBootArgs, sizeof(ab));
221 swap_apple_blockzeroblock(&ab);
222 if (ab.bzbMagic == APPLE_BZB_MAGIC) {
223 if (ab.bzbType == APPLE_BZB_TYPESWAP)
224 ptype = DKW_PTYPE_SWAP;
225 }
226
227 struct dkwedge_info dkw;
228 memset(&dkw, 0, sizeof(dkw));
229
230 strlcpy(dkw.dkw_ptype, ptype, sizeof(dkw.dkw_ptype));
231 strlcpy(dkw.dkw_parent, pdk->dk_name, sizeof(dkw.dkw_parent));
232 dkw.dkw_offset = ae->pmPyPartStart;
233 dkw.dkw_size = ae->pmPartBlkCnt;
234 strlcpy(dkw.dkw_wname, ae->pmPartName, sizeof(dkw.dkw_wname));
235 error = dkwedge_add(&dkw);
236 if (error == EEXIST)
237 aprint_error("%s: wedge named '%s' already "
238 "exists, manual intervention required\n",
239 pdk->dk_name, dkw.dkw_wname);
240 else if (error)
241 aprint_error("%s: error %d adding partition "
242 "%s type %s\n", pdk->dk_name, error,
243 ae->pmPartType, dkw.dkw_ptype);
244 }
245
246out:
247 DKW_FREE(buf);
248 DPRINTF("%s: return %d\n", __func__, error);
249 return error;
250}

253DKWEDGE_DISCOVERY_METHOD_DECL(APPLE, -5, dkwedge_discover_apple);
254#endif
