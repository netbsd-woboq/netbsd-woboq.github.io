38#ifndef _DMOVER_DMOVER_IO_H_
39#define _DMOVER_DMOVER_IO_H_
40
41#include <sys/types.h>
42#include <sys/ioccom.h>
43#include <sys/uio.h>
44
45typedef struct {
46 struct iovec *dmbuf_iov;
47 u_int dmbuf_iovcnt;
48} dmio_buffer;
49
50/*
51 * dmio_usrreq:
52 *
53 * Request structure passed from user-space.
54 */
55struct dmio_usrreq {
56 /* Output buffer. */
57 dmio_buffer req_outbuf;
58
59 /*
60 * General purpose immediate value. Can be used as an
61 * input, output, or both, depending on the function.
62 * The output is transmitted via the usrresp structure.
63 */
64 uint8_t req_immediate[8];
65
66 /* Input buffer. */
67 dmio_buffer *req_inbuf;
68
69 uint32_t req_id; /* request ID; passed in response */
70};
71
72/*
73 * dmio_usrresp:
74 *
75 * Response structure passed to user-space.
76 */
77struct dmio_usrresp {
78 uint32_t resp_id; /* request ID */
79 int resp_error; /* error, 0 if success */
80 uint8_t resp_immediate[8];
81};
82
83/*
84 * DMIO_SETFUNC:
85 *
86 * Ioctl to set the function type for the session.
87 */
88#define DMIO_SETFUNC _IOW('D', 0, struct dmio_setfunc)
89
90#define DMIO_MAX_FUNCNAME 64
91struct dmio_setfunc {
92 char dsf_name[DMIO_MAX_FUNCNAME];
93};
94
95#endif /* _DMOVER_DMOVER_IO_H_ */
