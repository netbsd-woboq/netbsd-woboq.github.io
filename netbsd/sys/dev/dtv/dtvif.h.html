35#ifndef _DEV_DTV_DTVIF_H
36#define _DEV_DTV_DTVIF_H
37
38#include <sys/device.h>
39
40#include <dev/dtv/dtvio.h>
41
42#define DTV_DEVICE_FRONTEND 0
43#define DTV_DEVICE_DEMUX 1
44#define DTV_DEVICE_DVR 2
45
46#define DTV_NUM_DEVICES 3
47
48#define DTVUNIT(x) (minor(x) & 0x0f)
49#define DTVDEV(x) ((minor(x) & 0xf0) >> 4)
50
51#define ISDTVFRONTEND(x) (DTVDEV((x)) == DTV_DEVICE_FRONTEND)
52#define ISDTVDEMUX(x) (DTVDEV((x)) == DTV_DEVICE_DEMUX)
53#define ISDTVDVR(x) (DTVDEV((x)) == DTV_DEVICE_DVR)
54
55struct dtv_payload;
56
57struct dtv_hw_if {
58 void (*get_devinfo)(void *, struct dvb_frontend_info *);
59
60 int (*open)(void *, int);
61 void (*close)(void *);
62 int (*set_tuner)(void *, const struct dvb_frontend_parameters *);
63 fe_status_t (*get_status)(void *);
64 uint16_t (*get_signal_strength)(void *);
65 uint16_t (*get_snr)(void *);
66 int (*start_transfer)(void *,
67 void (*)(void *, const struct dtv_payload *),
68 void *);
69 int (*stop_transfer)(void *);
70};
71
72struct dtv_attach_args {
73 const struct dtv_hw_if *hw;
74 void *priv;
75};
76
77struct dtv_payload {
78 const uint8_t *data;
79 size_t size;
80};
81
82static __inline int
83dtv_print(void *priv, const char *pnp)
84{
85 if (pnp)
86 aprint_normal("dtv at %s", pnp);
87 return UNCONF;
88}

90#endif /* !_DEV_DTV_DTVIF_H */
