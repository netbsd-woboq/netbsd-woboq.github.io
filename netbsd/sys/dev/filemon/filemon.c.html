93void
94filemon_output(struct filemon * filemon, char *msg, size_t len)
95{
96 struct uio auio;
97 struct iovec aiov;
98
99 if (filemon->fm_fp == NULL)
100 return;
101
102 aiov.iov_base = msg;
103 aiov.iov_len = len;
104 auio.uio_iov = &aiov;
105 auio.uio_iovcnt = 1;
106 auio.uio_resid = len;
107 auio.uio_rw = UIO_WRITE;
108 auio.uio_offset = (off_t) - 1;
109 uio_setup_sysspace(&auio);
110
111#ifdef DEBUG
112 {
113 char *cp;
114 int x = 16;
115
116 cp = strchr(msg, '\n');
117 if (cp && cp - msg <= 16)
118 x = (cp - msg) - 2;
119 log(logLevel, "filemon_output:('%.*s%s'", x,
120 (x < 16) ? "..." : "", msg);
121 }
122#endif
123 (*filemon->fm_fp->f_ops->fo_write) (filemon->fm_fp,
124 &(filemon->fm_fp->f_offset),
125 &auio, curlwp->l_cred, FOF_UPDATE_OFFSET);
126}

128void
129filemon_printf(struct filemon *filemon, const char *fmt, ...)
130{
131 size_t len;
132 va_list ap;
133
134 va_start(ap, fmt);
135 len = vsnprintf(filemon->fm_msgbufr, sizeof(filemon->fm_msgbufr),
136 fmt, ap);
137 va_end(ap);
138 if (len > sizeof(filemon->fm_msgbufr))
139 len = sizeof(filemon->fm_msgbufr);
140 filemon_output(filemon, filemon->fm_msgbufr, len);
141}

152static struct filemon *
153filemon_pid_check(struct proc * p)
154{
155 struct filemon *filemon;
156 struct proc * lp;
157
158 KASSERT(p != NULL);
159 if (!TAILQ_EMPTY(&filemons_inuse)) {
160 /*
161 * make sure p cannot exit
162 * until we have moved on to p_pptr
163 */
164 rw_enter(&p->p_reflock, RW_READER);
165 while (p) {
166 TAILQ_FOREACH(filemon, &filemons_inuse, fm_link) {
167 if (p->p_pid == filemon->fm_pid) {
168 rw_exit(&p->p_reflock);
169 return (filemon);
170 }
171 }
172 lp = p;
173 p = p->p_pptr;
174
175 /* lock parent before releasing child */
176 if (p != NULL)
177 rw_enter(&p->p_reflock, RW_READER);
178 rw_exit(&lp->p_reflock);
179 }
180 }
181 return (NULL);
182}

184/*
185 * return exclusive access to a filemon struct
186 */
187struct filemon *
188filemon_lookup(struct proc * p)
189{
190 struct filemon *filemon;
191
192 rw_enter(&filemon_mtx, RW_READER);
193 filemon = filemon_pid_check(p);
194 if (filemon) {
195 rw_enter(&filemon->fm_mtx, RW_WRITER);
196 }
197 rw_exit(&filemon_mtx);
198 return filemon;
199}

201static struct filemon *
202filemon_fp_data(struct file * fp, int lck)
203{
204 struct filemon *filemon;
205
206 rw_enter(&filemon_mtx, RW_READER);
207 filemon = fp->f_data;
208 if (filemon && lck) {
209 rw_enter(&filemon->fm_mtx, lck);
210 }
211 rw_exit(&filemon_mtx);
212 return filemon;
213}

215static int n_open = 0;
216
217static int
218filemon_open(dev_t dev, int oflags __unused, int mode __unused,
219 struct lwp * l __unused)
220{
221 struct filemon *filemon;
222 struct file *fp;
223 int error, fd;
224
225 /* falloc() will fill in the descriptor for us. */
226 if ((error = fd_allocfile(&fp, &fd)) != 0)
227 return error;
228
229 filemon = kmem_alloc(sizeof(struct filemon), KM_SLEEP);
230 rw_init(&filemon->fm_mtx);
231 filemon->fm_fp = NULL;
232 filemon->fm_pid = curproc->p_pid;
233
234 rw_enter(&filemon_mtx, RW_WRITER);
235 TAILQ_INSERT_TAIL(&filemons_inuse, filemon, fm_link);
236 n_open++;
237 rw_exit(&filemon_mtx);
238
239 return fd_clone(fp, fd, oflags, &filemon_fileops, filemon);
240}

243static int
244filemon_close(struct file * fp)
245{
246 struct filemon *filemon;
247
248#ifdef DEBUG
249 log(logLevel, "filemon_close()");
250#endif
251 /*
252 * Follow the same lock order as filemon_lookup()
253 * and filemon_fp_data() but hold exclusive access to
254 * filemon_mtx until we are done.
255 */
256 rw_enter(&filemon_mtx, RW_WRITER);
257 filemon = fp->f_data;
258 if (!filemon) {
259 rw_exit(&filemon_mtx);
260 return EBADF;
261 }
262 /* ensure that filemon_lookup() will now fail */
263 TAILQ_REMOVE(&filemons_inuse, filemon, fm_link);
264 n_open--;
265 /* ensure that filemon_fp_data() will now fail */
266 fp->f_data = NULL;
267
268 /*
269 * once we have exclusive access, it should never be used again
270 */
271 rw_enter(&filemon->fm_mtx, RW_WRITER);
272 if (filemon->fm_fp) {
273 closef(filemon->fm_fp); /* release our reference */
274 filemon->fm_fp = NULL;
275 }
276 rw_exit(&filemon->fm_mtx);
277 rw_destroy(&filemon->fm_mtx);
278 kmem_free(filemon, sizeof(struct filemon));
279 rw_exit(&filemon_mtx);
280 return (0);
281}

283static int
284filemon_ioctl(struct file * fp, u_long cmd, void *data)
285{
286 int error = 0;
287 int fd;
288 struct filemon *filemon;
289 struct proc *tp;
290
291#ifdef DEBUG
292 log(logLevel, "filemon_ioctl(%lu)", cmd);
293#endif
294
295 /*
296 * this ensures we cannot get filemon if it is closing.
297 */
298 filemon = filemon_fp_data(fp, RW_WRITER);
299 if (!filemon)
300 return EBADF;
301
302 /* filemon_fp_data() has locked the entry - make sure to unlock! */
303
304 switch (cmd) {
305 case FILEMON_SET_FD:
306 /* Set the output file descriptor. */
307
308 /* First, release any current output file descriptor */
309 if (filemon->fm_fp)
310 closef(filemon->fm_fp);
311
312 /* Now set up the new one */
313 fd = *((int *) data);
314 if ((filemon->fm_fp = fd_getfile2(curproc, fd)) == NULL) {
315 error = EBADF;
316 break;
317 }
318 /* Write the file header. */
319 filemon_comment(filemon);
320 break;
321
322 case FILEMON_SET_PID:
323 /* Set the monitored process ID - if allowed. */
324 mutex_enter(proc_lock);
325 tp = proc_find(*((pid_t *) data));
326 if (tp == NULL ||
327 tp->p_emul != &emul_netbsd) {
328 error = ESRCH;
329 mutex_exit(proc_lock);
330 break;
331 }
332
333 error = kauth_authorize_process(curproc->p_cred,
334 KAUTH_PROCESS_CANSEE, tp,
335 KAUTH_ARG(KAUTH_REQ_PROCESS_CANSEE_ENTRY), NULL, NULL);
336 if (!error) {
337 filemon->fm_pid = tp->p_pid;
338 }
339 mutex_exit(proc_lock);
340 break;
341
342 default:
343 error = EINVAL;
344 break;
345 }
346
347 rw_exit(&filemon->fm_mtx);
348 return (error);
349}

351static int
352filemon_load(void *dummy __unused)
353{
354 rw_init(&filemon_mtx);
355
356 /* Install the syscall wrappers. */
357 return filemon_wrapper_install();
358}

377static int
378filemon_unload(void)
379{
380 int error = 0;
381
382 rw_enter(&filemon_mtx, RW_WRITER);
383
384 if (TAILQ_FIRST(&filemons_inuse) != NULL)
385 error = EBUSY;
386 else {
387 /* Deinstall the syscall wrappers. */
388 error = filemon_wrapper_deinstall();
389 }
390 rw_exit(&filemon_mtx);
391
392 if (error == 0) {
393 rw_destroy(&filemon_mtx);
394 }
395 return (error);
396}

398static int
399filemon_modcmd(modcmd_t cmd, void *data)
400{
401 int error = 0;
402#ifdef _MODULE
403 int bmajor = -1;
404 int cmajor = -1;
405#endif
406
407 switch (cmd) {
408 case MODULE_CMD_INIT:
409#ifdef DEBUG
410 logLevel = LOG_INFO;
411#endif
412
413 error = filemon_load(data);
414#ifdef _MODULE
415 if (!error)
416 error = devsw_attach("filemon", NULL, &bmajor,
417 &filemon_cdevsw, &cmajor);
418#endif
419 break;
420
421 case MODULE_CMD_FINI:
422 error = filemon_unload();
423#ifdef _MODULE
424 if (!error)
425 error = devsw_detach(NULL, &filemon_cdevsw);
426#endif
427 break;
428
429 case MODULE_CMD_STAT:
430 log(LOG_INFO, "filemon: open=%d", n_open);
431 break;
432
433 default:
434 error = ENOTTY;
435 break;
436
437 }
438
439 return (error);
440}
