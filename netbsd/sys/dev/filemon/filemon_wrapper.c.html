89int
90syscall_hijack(struct sysent *sv_table, const struct hijack *hj_pkg,
91 bool restore)
92{
93 int from, to;
94 const struct hijack *entry;
95
96 if (restore) /* Which entry should currently match? */
97 from = 1;
98 else
99 from = 0;
100 to = 1 - from; /* Which entry will we replace with? */
101
102 KASSERT(kernconfig_is_held());
103
104 /* First, make sure that all of the old values match */
105 for (entry = hj_pkg; entry->hj_index >= 0; entry++)
106 if (sv_table[entry->hj_index].sy_call != entry->hj_funcs[from])
107 break;
108
109 if (entry->hj_index >= 0)
110 return EBUSY;
111
112 /* Now replace the old values with the new ones */
113 for (entry = hj_pkg; entry->hj_index >= 0; entry++)
114 sv_table[entry->hj_index].sy_call = entry->hj_funcs[to];
115
116 return 0;
117}

119static int
120filemon_wrapper_chdir(struct lwp * l, const struct sys_chdir_args * uap,
121 register_t * retval)
122{
123 int error;
124 size_t done;
125 struct filemon *filemon;
126
127 if ((error = sys_chdir(l, uap, retval)) != 0)
128 return error;
129
130 filemon = filemon_lookup(curproc);
131 if (filemon == NULL)
132 return 0;
133
134 error = copyinstr(SCARG(uap, path), filemon->fm_fname1,
135 sizeof(filemon->fm_fname1), &done);
136 if (error)
137 goto out;
138
139 filemon_printf(filemon, "C %d %s\n",
140 curproc->p_pid, filemon->fm_fname1);
141out:
142 rw_exit(&filemon->fm_mtx);
143 return 0;
144}

146static int
147filemon_wrapper_execve(struct lwp * l, struct sys_execve_args * uap,
148 register_t * retval)
149{
150 char fname[MAXPATHLEN];
151 int error, cerror;
152 size_t done;
153 struct filemon *filemon;
154
155 cerror = copyinstr(SCARG(uap, path), fname, sizeof(fname), &done);
156
157 if ((error = sys_execve(l, uap, retval)) != EJUSTRETURN)
158 return error;
159
160 filemon = filemon_lookup(curproc);
161 if (filemon == NULL)
162 return EJUSTRETURN;
163
164 if (cerror)
165 goto out;
166
167 filemon_printf(filemon, "E %d %s\n", curproc->p_pid, fname);
168out:
169 rw_exit(&filemon->fm_mtx);
170 return EJUSTRETURN;
171}

212static void
213filemon_flags(struct filemon * filemon, int f)
214{
215 if (f & O_RDWR) {
216 /* we want a separate R record */
217 filemon_printf(filemon, "R %d %s\n", curproc->p_pid,
218 filemon->fm_fname1);
219 }
220
221 filemon_printf(filemon, "%c %d %s\n", (f & O_ACCMODE) ? 'W' : 'R',
222 curproc->p_pid, filemon->fm_fname1);
223}

251static int
252filemon_wrapper_openat(struct lwp * l, struct sys_openat_args * uap,
253 register_t * retval)
254{
255 int error;
256 size_t done;
257 struct filemon *filemon;
258
259 if ((error = sys_openat(l, uap, retval)) != 0)
260 return error;
261
262 filemon = filemon_lookup(curproc);
263 if (filemon == NULL)
264 return 0;
265
266 error = copyinstr(SCARG(uap, path), filemon->fm_fname1,
267 sizeof(filemon->fm_fname1), &done);
268 if (error)
269 goto out;
270
271 if (filemon->fm_fname1[0] != '/' && SCARG(uap, fd) != AT_FDCWD) {
272 /*
273 * Rats we cannot just treat like open.
274 * Output an 'A' record as a clue.
275 */
276 filemon_printf(filemon, "A %d %s\n", curproc->p_pid,
277 filemon->fm_fname1);
278 }
279
280 filemon_flags(filemon, SCARG(uap, oflags));
281out:
282 rw_exit(&filemon->fm_mtx);
283 return 0;
284}

383static void
384filemon_wrapper_sys_exit(struct lwp * l, struct sys_exit_args * uap,
385 register_t * retval)
386{
387 struct filemon *filemon;
388
389 filemon = filemon_lookup(curproc);
390
391 if (filemon) {
392 filemon_printf(filemon, "X %d %d\n",
393 curproc->p_pid, SCARG(uap, rval));
394
395 /* Check if the monitored process is about to exit. */
396 if (filemon->fm_pid == curproc->p_pid)
397 filemon_printf(filemon, "# Bye bye\n");
398
399 rw_exit(&filemon->fm_mtx);
400 }
401 sys_exit(l, uap, retval);
402 /* NOT REACHED */
403}

433int
434filemon_wrapper_install(void)
435{
436 int error;
437 struct sysent *sv_table = emul_netbsd.e_sysent;
438
439 kernconfig_lock();
440 error = syscall_hijack(sv_table, filemon_hijack, false);
441 kernconfig_unlock();
442
443 return error;
444}
445
446int
447filemon_wrapper_deinstall(void)
448{
449 int error;
450 struct sysent *sv_table = emul_netbsd.e_sysent;
451
452 kernconfig_lock();
453 error = syscall_hijack(sv_table, filemon_hijack, true);
454 kernconfig_unlock();
455
456 return error;
457}
