20#ifndef _DEV_GPIO_GPIOVAR_H_
21#define _DEV_GPIO_GPIOVAR_H_
22
23#include <sys/device.h>
24
25/* GPIO controller description */
26typedef struct gpio_chipset_tag {
27 void *gp_cookie;
28
29 int (*gp_gc_open)(void *, device_t);
30 void (*gp_gc_close)(void *, device_t);
31 int (*gp_pin_read)(void *, int);
32 void (*gp_pin_write)(void *, int, int);
33 void (*gp_pin_ctl)(void *, int, int);
34
35 void * (*gp_intr_establish)(void *, int, int, int,
36 int (*)(void *), void *);
37 void (*gp_intr_disestablish)(void *, void *);
38 bool (*gp_intr_str)(void *, int, int, char *, size_t);
39} *gpio_chipset_tag_t;
40
41/* GPIO pin description */
42typedef struct gpio_pin {
43 int pin_num; /* number */
44 int pin_caps; /* capabilities */
45 int pin_flags; /* current configuration */
46 int pin_state; /* current state */
47 int pin_mapped; /* is mapped */
48 gpio_chipset_tag_t pin_gc; /* reference the controller */
49 char pin_defname[GPIOMAXNAME]; /* default name */
50 int pin_intrcaps; /* interrupt capabilities */
51} gpio_pin_t;
52
53/* Attach GPIO framework to the controller */
54struct gpiobus_attach_args {
55 gpio_chipset_tag_t gba_gc; /* underlying controller */
56 gpio_pin_t *gba_pins; /* pins array */
57 int gba_npins; /* total number of pins */
58};
59
60int gpiobus_print(void *, const char *);
61
62/* GPIO framework private methods */
63#define gpiobus_open(gc, dev) \
64 ((gc)->gp_gc_open ? ((gc)->gp_gc_open((gc)->gp_cookie, dev)) : 0)
65#define gpiobus_close(gc, dev) \
66 ((gc)->gp_gc_close ? ((gc)->gp_gc_close((gc)->gp_cookie, dev)), 1 : 0)
67#define gpiobus_pin_read(gc, pin) \
68 ((gc)->gp_pin_read((gc)->gp_cookie, (pin)))
69#define gpiobus_pin_write(gc, pin, value) \
70 ((gc)->gp_pin_write((gc)->gp_cookie, (pin), (value)))
71#define gpiobus_pin_ctl(gc, pin, flags) \
72 ((gc)->gp_pin_ctl((gc)->gp_cookie, (pin), (flags)))
73
74/* Attach devices connected to the GPIO pins */
75struct gpio_attach_args {
76 void *ga_gpio;
77 int ga_offset;
78 uint32_t ga_mask;
79 char *ga_dvname;
80 uint32_t ga_flags; /* driver-specific flags */
81};

83/* GPIO pin map */
84struct gpio_pinmap {
85 int *pm_map; /* pin map */
86 int pm_size; /* map size */
87};

89struct gpio_dev {
90 device_t sc_dev; /* the gpio device */

92};

94struct gpio_name {
95 char gp_name[GPIOMAXNAME];
96 int gp_pin;
97 LIST_ENTRY(gpio_name) gp_next;
98};
