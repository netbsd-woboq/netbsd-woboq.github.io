47int
48hdafg_dd_parse_info(uint8_t *data, size_t datalen, struct hdafg_dd_info *hdi)
49{
50 struct eld_baseline_block *block = &hdi->eld;
51 unsigned int i;
52
53#ifdef HDAFG_HDMI_DEBUG
54 printf("hdafg_dd_parse_info: datalen=%u\n", (unsigned int)datalen);
55#endif
56
57 memset(hdi, 0, sizeof(*hdi));
58
59 if (datalen < sizeof(block->header)) {
60#ifdef HDAFG_HDMI_DEBUG
61 printf(" no room for header\n");
62#endif
63 return EINVAL;
64 }
65
66 memcpy(&block->header, data, sizeof(block->header));
67 data += sizeof(block->header);
68 datalen -= sizeof(block->header);
69
70 if (datalen < block->header.baseline_eld_len * 4 ||
71 datalen < sizeof(*block) - sizeof(block->header)) {
72#ifdef HDAFG_HDMI_DEBUG
73 printf(" ack!\n");
74#endif
75 return EINVAL;
76 }
77
78 datalen = block->header.baseline_eld_len * 4;
79
80 memcpy(&block->flags[0], data, sizeof(*block) - sizeof(block->header));
81 data += sizeof(*block) - sizeof(block->header);
82 datalen -= sizeof(*block) - sizeof(block->header);
83
84 if (datalen < ELD_MNL(block)) {
85#ifdef HDAFG_HDMI_DEBUG
86 printf(" MNL=%u\n", ELD_MNL(block));
87#endif
88 return EINVAL;
89 }
90
91 memcpy(hdi->monitor, data, ELD_MNL(block));
92 data += ELD_MNL(block);
93 datalen -= ELD_MNL(block);
94
95 if (datalen < ELD_SAD_COUNT(block) * sizeof(hdi->sad[0])) {
96#ifdef HDAFG_HDMI_DEBUG
97 printf(" datalen %u sadcount %u sizeof sad %u\n",
98 (unsigned int)datalen,
99 ELD_SAD_COUNT(block),
100 (unsigned int)sizeof(hdi->sad[0]));
101#endif
102 return EINVAL;
103 }
104 hdi->nsad = ELD_SAD_COUNT(block);
105 for (i = 0; i < hdi->nsad; i++) {
106 memcpy(&hdi->sad[i], data, sizeof(hdi->sad[i]));
107 data += sizeof(hdi->sad[i]);
108 datalen -= sizeof(hdi->sad[i]);
109 }
110
111#ifdef HDAFG_HDMI_DEBUG
112 printf("hdafg eld padding ignored = %u\n", (unsigned int)datalen);
113#endif
114 return 0;
115}

117void
118hdafg_dd_hdmi_ai_cksum(struct hdmi_audio_infoframe *hdmi)
119{
120 uint8_t *dip = (uint8_t *)hdmi, c = 0;
121 int i;
122
123 hdmi->checksum = 0;
124 for (i = 0; i < sizeof(*hdmi); i++)
125 c += dip[i];
126 hdmi->checksum = -c;
127}
