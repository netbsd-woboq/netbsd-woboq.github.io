33#include <dev/wscons/wsconsio.h>
34#include <dev/wscons/wsmousevar.h>
35#include <dev/wscons/tpcalibvar.h>
36
37#define MAX_BUTTONS 31 /* must not exceed size of sc_buttons */
38
39struct hidms {
40 struct hid_location hidms_loc_x, hidms_loc_y, hidms_loc_z, hidms_loc_w;
41 struct hid_location hidms_loc_btn[MAX_BUTTONS];
42
43 u_int flags; /* device configuration */
44#define HIDMS_Z 0x001 /* z direction available */
45#define HIDMS_SPUR_BUT_UP 0x002 /* spurious button up events */
46#define HIDMS_REVZ 0x004 /* Z-axis is reversed */
47#define HIDMS_W 0x008 /* w direction/tilt available */
48#define HIDMS_ABS 0x010 /* absolute position, touchpanel */
49#define HIDMS_TIP_SWITCH 0x020 /* digitizer tip switch */
50#define HIDMS_SEC_TIP_SWITCH 0x040 /* digitizer secondary tip switch */
51#define HIDMS_BARREL_SWITCH 0x080 /* digitizer barrel switch */
52#define HIDMS_ERASER 0x100 /* digitizer eraser */
53#define HIDMS_DIGITIZER 0x200 /* digitizer */
54
55 int nbuttons;
56
57 uint32_t hidms_buttons; /* mouse button status */
58 device_t hidms_wsmousedev;
59
60 struct tpcalib_softc sc_tpcalib; /* calibration */
61 struct wsmouse_calibcoords sc_calibcoords;
62};
63
64bool hidms_setup(device_t, struct hidms *, int, void *, int);
65void hidms_attach(device_t, struct hidms *, const struct wsmouse_accessops *);
66void hidms_intr(struct hidms *, void *, u_int);
