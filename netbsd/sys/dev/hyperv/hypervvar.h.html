94/*
95 * Hyper-V bus_dma utilities.
96 */
97struct hyperv_dma {
98 bus_dmamap_t map;
99 bus_dma_segment_t *segs;
100 void *addr;
101 int nsegs;
102};
103
104static __inline bus_addr_t
105hyperv_dma_get_paddr(struct hyperv_dma *dma)
106{
107 return dma->map->dm_segs[0].ds_addr;
108}
