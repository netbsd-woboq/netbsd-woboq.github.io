209static __inline void
210clear_bit(u_int b, volatile void *p)
211{
212 atomic_and_32(((volatile u_int *)p) + (b >> 5), ~(1 << (b & 0x1f)));
213}
214
215static __inline void
216set_bit(u_int b, volatile void *p)
217{
218 atomic_or_32(((volatile u_int *)p) + (b >> 5), 1 << (b & 0x1f));
219}
220
221static __inline int
222test_bit(u_int b, volatile void *p)
223{
224 return !!(((volatile u_int *)p)[b >> 5] & (1 << (b & 0x1f)));
225}
