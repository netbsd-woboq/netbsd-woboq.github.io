38#ifndef _DEV_I2C_I2C_BITBANG_H_
39#define _DEV_I2C_I2C_BITBANG_H_
40
41#define I2C_BIT_SDA 0 /* SDA signal */
42#define I2C_BIT_SCL 1 /* SCL signal */
43#define I2C_BIT_OUTPUT 2 /* controller: SDA is output */
44#define I2C_BIT_INPUT 3 /* controller: SDA is input */
45#define I2C_NBITS 4
46
47struct i2c_bitbang_ops {
48 void (*ibo_set_bits)(void *, uint32_t);
49 void (*ibo_set_dir)(void *, uint32_t);
50 uint32_t (*ibo_read_bits)(void *);
51 uint32_t ibo_bits[I2C_NBITS];
52};
53
54typedef const struct i2c_bitbang_ops *i2c_bitbang_ops_t;
55
56int i2c_bitbang_send_start(void *, int, i2c_bitbang_ops_t);
57int i2c_bitbang_send_stop(void *, int, i2c_bitbang_ops_t);
58int i2c_bitbang_initiate_xfer(void *, i2c_addr_t, int, i2c_bitbang_ops_t);
59int i2c_bitbang_read_byte(void *, uint8_t *, int, i2c_bitbang_ops_t);
60int i2c_bitbang_write_byte(void *, uint8_t, int, i2c_bitbang_ops_t);
61
62#endif /* _DEV_I2C_I2C_BITBANG_H_ */
