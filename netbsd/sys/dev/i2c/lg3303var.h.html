29#ifndef _LG3303VAR_H
30#define _LG3303VAR_H
31
32#include <dev/i2c/i2cvar.h>
33#include <dev/dtv/dtvio.h>
34
35#define LG3303_CFG_SERIAL_INPUT 0x01
36
37struct lg3303 {
38 device_t parent;
39 i2c_tag_t i2c;
40 i2c_addr_t i2c_addr;
41
42 fe_modulation_t current_modulation;
43
44 int flags;
45};
46
47struct lg3303 * lg3303_open(device_t, i2c_tag_t, i2c_addr_t, int);
48void lg3303_close(struct lg3303 *);

51uint16_t lg3303_get_snr(struct lg3303 *);
52uint16_t lg3303_get_signal_strength(struct lg3303 *);
53uint32_t lg3303_get_ucblocks(struct lg3303 *);
54
55#endif /* !_LG3303VAR_H */
