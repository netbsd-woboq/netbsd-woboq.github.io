56/* Constants for matching i2c bus address */
57#define SPDMEM_I2C_ADDRMASK 0xfff8
58#define SPDMEM_I2C_ADDR 0x50
59#define SPDCTL_I2C_ADDR 0x30
60
61/* set write protection */
62#define SPDCTL_SWP0 (SPDCTL_I2C_ADDR + 1)
63#define SPDCTL_SWP1 (SPDCTL_I2C_ADDR + 4)
64#define SPDCTL_SWP2 (SPDCTL_I2C_ADDR + 5)
65#define SPDCTL_SWP3 (SPDCTL_I2C_ADDR + 0)
66
67/* clear write protections */
68#define SPDCTL_CWP (SPDCTL_I2C_ADDR + 3)
69
70/* read protection status */
71#define SPDCTL_RPS0 (SPDCTL_I2C_ADDR + 1)
72#define SPDCTL_RPS1 (SPDCTL_I2C_ADDR + 4)
73#define SPDCTL_RPS2 (SPDCTL_I2C_ADDR + 5)
74#define SPDCTL_RPS3 (SPDCTL_I2C_ADDR + 0)
75
76/* select page address */
77#define SPDCTL_SPA0 (SPDCTL_I2C_ADDR + 6)
78#define SPDCTL_SPA1 (SPDCTL_I2C_ADDR + 7)
79
80/* read page address */
81#define SPDCTL_RPA (SPDCTL_I2C_ADDR + 6)
82
83struct spdmem_i2c_softc {
84 struct spdmem_softc sc_base;
85 i2c_tag_t sc_tag;
86 i2c_addr_t sc_addr; /* EEPROM */
87 i2c_addr_t sc_page0;
88 i2c_addr_t sc_page1;
89};

101static int
102spdmem_reset_page(struct spdmem_i2c_softc *sc)
103{
104 uint8_t reg, byte0, byte2;
105 static uint8_t dummy = 0;
106 int rv;
107
108 reg = 0;
109
110 iic_acquire_bus(sc->sc_tag, 0);
111
112 /*
113 * Try to read byte 0 and 2. If it failed, it's not spdmem or a device
114 * doesn't exist at the address.
115 */
116 rv = iic_exec(sc->sc_tag, I2C_OP_READ_WITH_STOP, sc->sc_addr, &reg, 1,
117 &byte0, 1, I2C_F_POLL);
118 rv |= iic_exec(sc->sc_tag, I2C_OP_READ_WITH_STOP, sc->sc_addr, &reg, 1,
119 &byte2, 1, I2C_F_POLL);
120 if (rv != 0)
121 goto error;
122
123 /*
124 * Quirk for BIOSes that leave page 1 of a 4kbit EEPROM selected.
125 *
126 * byte0 is the length, byte2 is the memory type. Both of them should
127 * not be zero. If zero, the current page might be 1 (DDR4 and newer).
128 * If page 1 is selected, offset 0 can be 0 (Module Characteristics
129 * (Energy backup is not available)) and also offset 2 can be 0
130 * (Megabytes, and a part of Capacity digits).
131 *
132 * Note: The encoding of byte0 is vary in memory type, so we check
133 * just with zero to be simple.
134 *
135 * Try to see if we are not at page 0. If it's not, select page 0.
136 */
137 if ((byte0 == 0) || (byte2 == 0)) {
138 /*
139 * Note that SDCTL_RPA is the same as sc->sc_page0(SPDCTL_SPA0)
140 * Write is SPA0, read is RPA.
141 *
142 * This call returns 0 on page 0 and returns -1 on page 1.
143 * I don't know whether our icc_exec()'s API is good or not.
144 */
145 rv = iic_exec(sc->sc_tag, I2C_OP_READ_WITH_STOP, sc->sc_page0,
146 &reg, 1, &dummy, 1, I2C_F_POLL);
147 if (rv != 0) {
148 /*
149 * The possibilities are:
150 * a) page 1 is selected.
151 * b) The device doesn't support page select and
152 * it's not a SPD ROM.
153 * Is there no way to distinguish them now?
154 */
155 rv = iic_exec(sc->sc_tag, I2C_OP_WRITE_WITH_STOP,
156 sc->sc_page0, &reg, 1, &dummy, 1, I2C_F_POLL);
157 if (rv == 0) {
158 aprint_debug("Page 1 was selected. Page 0 is "
159 "selected now.\n");
160 } else {
161 aprint_debug("Failed to select page 0. This "
162 "device isn't SPD ROM\n");
163 }
164 } else {
165 /* This device isn't SPD ROM */
166 rv = -1;
167 }
168 }
169error:
170 iic_release_bus(sc->sc_tag, 0);
171
172 return rv;
173}

177 { "i2c-at34c02", 0 },

181static int
182spdmem_i2c_match(device_t parent, cfdata_t match, void *aux)
183{
184 struct i2c_attach_args *ia = aux;
185 struct spdmem_i2c_softc sc;
186 int match_result;
187
188 if (iic_use_direct_match(ia, match, compat_data, &match_result))
189 return match_result;
190
191 /*
192 * XXXJRT
193 * Should do this with "compatible" strings. There are also
194 * other problems with this "match" routine. Specifically, if
195 * we are doing direct-config, we know the device is already
196 * there aren't do need to probe. I'll leave the logic for
197 * now and let someone who knows better clean it later.
198 */
199
200 if (ia->ia_name) {
201 /* add other names as we find more firmware variations */
202 if (strcmp(ia->ia_name, "dimm-spd") &&
203 strcmp(ia->ia_name, "dimm"))
204 return 0;
205 }
206
207 /* only do this lame test when not using direct config */
208 if (ia->ia_name == NULL) {
209 if ((ia->ia_addr & SPDMEM_I2C_ADDRMASK) != SPDMEM_I2C_ADDR)
210 return 0;
211 }
212
213 sc.sc_tag = ia->ia_tag;
214 sc.sc_addr = ia->ia_addr;
215 sc.sc_page0 = SPDCTL_SPA0;
216 sc.sc_page1 = SPDCTL_SPA1;
217 sc.sc_base.sc_read = spdmem_i2c_read;
218
219 /* Check the bank and reset to the page 0 */
220 if (spdmem_reset_page(&sc) != 0)
221 return 0;
222
223 if (spdmem_common_probe(&sc.sc_base)) {
224 return ia->ia_name ? I2C_MATCH_DIRECT_SPECIFIC
225 : I2C_MATCH_ADDRESS_AND_PROBE;
226 }
227 return 0;
228}

230static void
231spdmem_i2c_attach(device_t parent, device_t self, void *aux)
232{
233 struct spdmem_i2c_softc *sc = device_private(self);
234 struct i2c_attach_args *ia = aux;
235
236 sc->sc_tag = ia->ia_tag;
237 sc->sc_addr = ia->ia_addr;
238 sc->sc_page0 = SPDCTL_SPA0;
239 sc->sc_page1 = SPDCTL_SPA1;
240 sc->sc_base.sc_read = spdmem_i2c_read;
241
242 if (!pmf_device_register(self, NULL, NULL))
243 aprint_error_dev(self, "couldn't establish power handler\n");
244
245 spdmem_common_attach(&sc->sc_base, self);
246}
247
248static int
249spdmem_i2c_detach(device_t self, int flags)
250{
251 struct spdmem_i2c_softc *sc = device_private(self);
252
253 pmf_device_deregister(self);
254
255 return spdmem_common_detach(&sc->sc_base, self);
256}

258static int
259spdmem_i2c_read(struct spdmem_softc *softc, uint16_t addr, uint8_t *val)
260{
261 uint8_t reg;
262 struct spdmem_i2c_softc *sc = (struct spdmem_i2c_softc *)softc;
263 static uint8_t dummy = 0;
264 int rv;
265
266 reg = addr & 0xff;
267
268 iic_acquire_bus(sc->sc_tag, 0);
269
270 if (addr & 0x100) {
271 rv = iic_exec(sc->sc_tag, I2C_OP_WRITE_WITH_STOP, sc->sc_page1,
272 &dummy, 1, &dummy, 1, I2C_F_POLL);
273 rv |= iic_exec(sc->sc_tag, I2C_OP_READ_WITH_STOP, sc->sc_addr,
274 &reg, 1, val, 1, I2C_F_POLL);
275 rv |= iic_exec(sc->sc_tag, I2C_OP_WRITE_WITH_STOP,
276 sc->sc_page0, &dummy, 1, &dummy, 1, I2C_F_POLL);
277 } else {
278 rv = iic_exec(sc->sc_tag, I2C_OP_READ_WITH_STOP, sc->sc_addr,
279 &reg, 1, val, 1, I2C_F_POLL);
280 }
281
282 iic_release_bus(sc->sc_tag, 0);
283
284 return rv;
285}

293static int
294spdmem_modcmd(modcmd_t cmd, void *opaque)
295{
296 int error = 0;
297#ifdef _MODULE
298 static struct sysctllog *spdmem_sysctl_clog;
299#endif
300
301 switch (cmd) {
302 case MODULE_CMD_INIT:
303#ifdef _MODULE
304 error = config_init_component(cfdriver_ioconf_spdmem,
305 cfattach_ioconf_spdmem, cfdata_ioconf_spdmem);
306#endif
307 return error;
308 case MODULE_CMD_FINI:
309#ifdef _MODULE
310 error = config_fini_component(cfdriver_ioconf_spdmem,
311 cfattach_ioconf_spdmem, cfdata_ioconf_spdmem);
312 sysctl_teardown(&spdmem_sysctl_clog);
313#endif
314 return error;
315 default:
316 return ENOTTY;
317 }
