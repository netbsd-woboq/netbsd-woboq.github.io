31
32#include <sys/param.h>
33#include <sys/types.h>
34#include <dev/i2c/i2cvar.h>
35#include <dev/dtv/dtvio.h>
36
37#define TVPLL_IGNORE_AUX 0xff
38
39struct tvpll_data {
40 const char * name;
41 uint32_t min;
42 uint32_t max;
43 uint32_t iffreq;
44 uint8_t *initdata;
45 uint8_t *sleepdata;
46 int count;
47 struct tvpll_entry{
48 uint32_t limit;
49 uint32_t stepsize;
50 uint8_t config;
51 uint8_t cb;
52 uint8_t aux;
53 } *entries;
54};

56struct tvpll * tvpll_open(device_t, i2c_tag_t, i2c_addr_t, struct tvpll_data *);
57void tvpll_close(struct tvpll *);
58int tvpll_tune_dtv(struct tvpll *, const struct dvb_frontend_parameters *);
59int tvpll_get_status(struct tvpll *);

61#endif /* !_DEV_I2C_TVPLLVAR_H_ */
