31
32#include <dev/i2c/i2cvar.h>
33#include <dev/dtv/dtvio.h>
34
35typedef int (*xc3028_reset_cb)(void *);
36
37enum xc3028_type {
38 XC3028,
39 XC3028L,
40};
41
42struct xc3028_fw {
43 uint32_t type;
44 uint64_t id;
45 uint16_t int_freq;
46 uint8_t *data;
47 uint32_t data_size;
48};
49
50struct xc3028 {
51 device_t parent;
52 i2c_tag_t i2c;
53 i2c_addr_t i2c_addr;
54
55 xc3028_reset_cb reset;
56 void *reset_priv;
57
58 enum xc3028_type type;
59
60 struct xc3028_fw *fw;
61 unsigned int nfw;
62};

64struct xc3028 * xc3028_open(device_t, i2c_tag_t, i2c_addr_t,
65 xc3028_reset_cb, void *, enum xc3028_type);
66void xc3028_close(struct xc3028 *);
67int xc3028_tune_dtv(struct xc3028 *,
68 const struct dvb_frontend_parameters *);

70#endif /* !_XC3028VAR_H */
