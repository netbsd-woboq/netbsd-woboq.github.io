29#ifndef _XC5KVAR_H
30#define _XC5KVAR_H
31
32#include <dev/i2c/i2cvar.h>
33#include <dev/dtv/dtvio.h>
34#include <dev/video_if.h>
35
36typedef int (*xc5k_reset_cb)(void *);
37
38struct xc5k {
39 device_t parent;
40 i2c_tag_t i2c;
41 i2c_addr_t i2c_addr;
42
43 xc5k_reset_cb reset;
44 void *reset_priv;
45
46 unsigned int if_freq;
47 fe_type_t fe_type;
48};
49
50struct xc5k_params {
51 uint16_t signal_source;
52 uint32_t frequency;
53 enum video_standard standard;
54};
55
56struct xc5k * xc5k_open(device_t, i2c_tag_t, i2c_addr_t,
57 xc5k_reset_cb, void *, unsigned int, fe_type_t);
58void xc5k_close(struct xc5k *);
59int xc5k_tune_video(struct xc5k *, struct xc5k_params *);
60int xc5k_tune_dtv(struct xc5k *, const struct dvb_frontend_parameters *);
61fe_status_t xc5k_get_status(struct xc5k *);
62
63#endif /* !_XC5KVAR_H */
