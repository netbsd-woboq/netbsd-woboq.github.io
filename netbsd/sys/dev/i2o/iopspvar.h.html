32#ifndef _I2O_IOPSPVAR_H_
33#define _I2O_IOPSPVAR_H_
34
35#define IOPSP_MAX_LUN 8
36#define IOPSP_MAX_SCSI_TARGET 15
37#define IOPSP_MAX_FC_TARGET 127
38
39#define IOPSP_TIDMAP(map, t, l) (map[(t) * IOPSP_MAX_LUN + (l)])
40#define IOPSP_TID_ABSENT 0x0000 /* Device is absent */
41#define IOPSP_TID_INUSE 0xffff /* Device in use by another module */
42
43struct iopsp_target {
44 u_int8_t it_width;
45 u_int8_t it_syncrate;
46 u_int8_t it_offset;
47 u_int8_t it_flags;
48};
49#define IT_PRESENT 0x01 /* Target is present */
50
51struct iopsp_softc {
52 device_t sc_dev; /* Generic device data */
53 struct scsipi_adapter sc_adapter; /* scsipi adapter */
54 struct scsipi_channel sc_channel; /* Prototype link */
55 struct iop_initiator sc_ii; /* I2O initiator state */
56 u_short *sc_tidmap; /* Target/LUN -> TID map */
57 u_int sc_chgind; /* Last LCT change # */
58 int sc_openings; /* # command openings */
59 struct iopsp_target *sc_targetmap; /* Target information */
60};
61
62#endif /* !_I2O_IOPSPVAR_H_ */
