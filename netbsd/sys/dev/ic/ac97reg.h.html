210#define AC97_CODEC_ID(a0, a1, a2, x) \
211 (((a0) << 24) | ((a1) << 16) | ((a2) << 8) | (x))
212
213#define AC97_GET_CODEC_ID(id, cp) \
214do { \
215 (cp)[0] = ((id) >> 24) & 0xff; \
216 (cp)[1] = ((id) >> 16) & 0xff; \
217 (cp)[2] = ((id) >> 8) & 0xff; \
218 (cp)[3] = (id) & 0xff; \
219} while (0)
