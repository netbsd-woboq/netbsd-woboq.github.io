31#ifndef _DEV_IC_AC97VAR_H_
32#define _DEV_IC_AC97VAR_H_
33#include <sys/types.h>
34
35struct ac97_codec_if;
36
37/*
38 * This is the interface used to attach the AC97 compliant CODEC.
39 */
40enum ac97_host_flags {
41 AC97_HOST_DONT_READ = 0x1,
42 AC97_HOST_SWAPPED_CHANNELS = 0x2, /* l/r is reversed */
43 AC97_HOST_SKIP_AUDIO = 0x4,
44 AC97_HOST_SKIP_MODEM = 0x8,
45 AC97_HOST_INVERTED_EAMP = 0x10
46};
47
48struct ac97_host_if {
49 void *arg;
50
51 int (*attach)(void *, struct ac97_codec_if *);
52 int (*read)(void *, uint8_t, uint16_t *);
53 int (*write)(void *, uint8_t, uint16_t);
54 int (*reset)(void *);
55
56 enum ac97_host_flags (*flags)(void *);
57 void (*spdif_event)(void *, bool);
58};
59
60/*
61 * This is the interface exported by the AC97 compliant CODEC
62 */
63struct ac97_codec_if_vtbl {
64 int (*mixer_get_port)(struct ac97_codec_if *, mixer_ctrl_t *);
65 int (*mixer_set_port)(struct ac97_codec_if *, mixer_ctrl_t *);
66 int (*query_devinfo)(struct ac97_codec_if *, mixer_devinfo_t *);
67 int (*get_portnum_by_name)(struct ac97_codec_if *, const char *,
68 const char *, const char *);
69 /*
70 * The AC97 codec driver records the various port settings. This
71 * function can be used to restore the port settings, e.g. after
72 * resume from a laptop suspend to disk.
73 * Also restores AC97_POWER_REG.
74 */
75 void (*restore_ports)(struct ac97_codec_if *);
76
77 u_int16_t (*get_extcaps)(struct ac97_codec_if *);
78 int (*set_rate)(struct ac97_codec_if *, int, u_int *);
79 void (*set_clock)(struct ac97_codec_if *, unsigned int);
80 void (*detach)(struct ac97_codec_if *);
81 /**
82 * To enable SPDIF,
83 * - call unlock() just after ac97_attach()
84 * - call lock() in audio_hw_if::open()
85 * - call unlock() in audio_hw_if::close()
86 */
87 void (*lock)(struct ac97_codec_if *);
88 void (*unlock)(struct ac97_codec_if *);
89};
90
91struct ac97_codec_if {
92 struct ac97_codec_if_vtbl *vtbl;
93};
94
95int ac97_attach_type(struct ac97_host_if *, device_t, int, kmutex_t *);
96int ac97_attach(struct ac97_host_if *, device_t, kmutex_t *);
97
98#define AC97_IS_FIXED_RATE(codec) \
99 !((codec)->vtbl->get_extcaps(codec) & AC97_EXT_AUDIO_VRA)
100#define AC97_IS_4CH(codec) \
101 ((codec)->vtbl->get_extcaps(codec) & AC97_EXT_AUDIO_SDAC)
102#define AC97_IS_6CH(codec) \
103 (((codec)->vtbl->get_extcaps(codec) \
104 & (AC97_EXT_AUDIO_SDAC | AC97_EXT_AUDIO_CDAC | AC97_EXT_AUDIO_LDAC)) \
105 == (AC97_EXT_AUDIO_SDAC | AC97_EXT_AUDIO_CDAC | AC97_EXT_AUDIO_LDAC))
106#define AC97_HAS_SPDIF(codec) \
107 ((codec)->vtbl->get_extcaps(codec) & AC97_EXT_AUDIO_SPDIF)
108
109#define AC97_CODEC_TYPE_AUDIO 1
110#define AC97_CODEC_TYPE_MODEM 2
