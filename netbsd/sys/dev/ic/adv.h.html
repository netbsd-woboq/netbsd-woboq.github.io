41#ifndef _ADVANSYS_NARROW_H_
42#define _ADVANSYS_NARROW_H_
43
44/******************************************************************************/
45
46/* second level interrupt callback type definition */
47typedef int (* ASC_ISR_CALLBACK) (ASC_SOFTC *, ASC_QDONE_INFO *);
48
49struct adv_ccb
50{
51 ASC_SG_HEAD sghead;
52 ASC_SCSI_Q scsiq;
53
54 struct scsi_sense_data scsi_sense;
55
56 struct callout ccb_watchdog;
57
58 TAILQ_ENTRY(adv_ccb) chain;
59 struct adv_ccb *nexthash;
60 u_long hashkey;
61 struct scsipi_xfer *xs; /* the scsipi_xfer for this cmd */
62 int flags; /* see below */
63
64 int timeout;
65 /*
66 * This DMA map maps the buffer involved in the transfer.
67 */
68 bus_dmamap_t dmamap_xfer;
69};
70
71typedef struct adv_ccb ADV_CCB;
72
73/* flags for ADV_CCB */
74#define CCB_ALLOC 0x01
75#define CCB_ABORT 0x02
76#define CCB_WATCHDOG 0x10
77
78
79#define ADV_MAX_CCB 32
80
81struct adv_control
82{
83 ADV_CCB ccbs[ADV_MAX_CCB]; /* all our control blocks */
84 u_int8_t overrun_buf[ASC_OVERRUN_BSIZE];
85};
86
87/*
88 * Offset of a CCB from the beginning of the control DMA mapping.
89 */
90#define ADV_CCB_OFF(c) (offsetof(struct adv_control, ccbs[0]) + \
91 (((u_long)(c)) - ((u_long)&sc->sc_control->ccbs[0])))
92

97int adv_detach(ASC_SOFTC *, int);
98int adv_intr(void *);
99ADV_CCB *adv_ccb_phys_kv(ASC_SOFTC *, u_long);
