41#ifndef _ADVANSYS_WIDE_H_
42#define _ADVANSYS_WIDE_H_
43
44/******************************************************************************/
45
46typedef int (* ADW_ISR_CALLBACK) (ADW_SOFTC *, ADW_SCSI_REQ_Q *);
47typedef void (* ADW_ASYNC_CALLBACK) (ADW_SOFTC *, u_int8_t);
48
49
50/*
51 * per request scatter-gather element limit
52 * We could have up to 256 SG lists.
53 */
54#define ADW_MAX_SG_LIST 255
55
56/*
57 * Scatter-Gather Definitions per request.
58 */
59
60#define NO_OF_SG_PER_BLOCK 15
61
62/* Number of SG blocks needed. */
63#define ADW_NUM_SG_BLOCK \
64 ((ADW_MAX_SG_LIST + (NO_OF_SG_PER_BLOCK - 1))/NO_OF_SG_PER_BLOCK)
65
66
67struct adw_ccb {
68 ADW_SCSI_REQ_Q scsiq;
69 ADW_SG_BLOCK sg_block[ADW_NUM_SG_BLOCK];
70
71 struct scsi_sense_data scsi_sense;
72
73 TAILQ_ENTRY(adw_ccb) chain;
74 struct adw_ccb *nexthash;
75 u_int32_t hashkey;
76
77 struct scsipi_xfer *xs; /* the scsipi_xfer for this cmd */
78 int flags; /* see below */
79
80 int timeout;
81 /*
82 * This DMA map maps the buffer involved in the transfer.
83 */
84 bus_dmamap_t dmamap_xfer;
85};
86
87typedef struct adw_ccb ADW_CCB;
88
89/* flags for ADW_CCB */
90#define CCB_ALLOC 0x01
91#define CCB_ABORTING 0x02
92#define CCB_ABORTED 0x04
93
94
95#define ADW_MAX_CCB 63 /* Max. number commands per device (63) */
96
97struct adw_control {
98 ADW_CCB ccbs[ADW_MAX_CCB]; /* all our control blocks */
99 ADW_CARRIER *carriers; /* all our carriers */
100};
101
102/*
103 * Offset of a CCB from the beginning of the control DMA mapping.
104 */
105#define ADW_CCB_OFF(c) (offsetof(struct adw_control, ccbs[0]) + \
106 (((u_long)(c)) - ((u_long)&sc->sc_control->ccbs[0])))
107

109

111void adw_attach(ADW_SOFTC *);
112int adw_intr(void *);
113ADW_CCB *adw_ccb_phys_kv(ADW_SOFTC *, u_int32_t);
