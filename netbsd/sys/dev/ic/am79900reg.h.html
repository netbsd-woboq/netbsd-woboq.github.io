66/*
67 * Receive message descriptor
68 */
69struct lermd {
70 u_int32_t rmd0;
71 u_int32_t rmd1;
72 u_int32_t rmd2;
73 int32_t rmd3;
74};
75
76/*
77 * Transmit message descriptor
78 */
79struct letmd {
80 u_int32_t tmd0;
81 u_int32_t tmd1;
82 u_int32_t tmd2;
83 int32_t tmd3;
84};
85
86/*
87 * Initialization block
88 */
89struct leinit {
90 u_int32_t init_mode; /* +0x0000 */
91 u_int32_t init_padr[2]; /* +0x0002 */
92 u_int16_t init_ladrf[4]; /* +0x0008 */
93 u_int32_t init_rdra; /* +0x0010 */
94 u_int32_t init_tdra; /* +0x0014 */
95 int32_t pad; /* Pad to 16 shorts */
96};
97
98/* Receive message descriptor 1 (rmd1_bits) */
99#define LE_R1_OWN (1<<31) /* LANCE owns the packet */
100#define LE_R1_ERR (1<<30) /* error summary */
101#define LE_R1_FRAM (1<<29) /* framing error */
102#define LE_R1_OFLO (1<<28) /* overflow error */
103#define LE_R1_CRC (1<<27) /* CRC error */
104#define LE_R1_BUFF (1<<26) /* buffer error */
105#define LE_R1_STP (1<<25) /* start of packet */
106#define LE_R1_ENP (1<<24) /* end of packet */
107#define LE_R1_ONES (0xf<<12) /* must be ones */
108#define LE_R1_BCNT_MASK (0xfff) /* byte count mask */
109
110#define LE_R1_BITS \
111 "\20\40OWN\37ERR\36FRAM\35OFLO\34CRC\33BUFF\32STP\31ENP"
112
113/* Transmit message descriptor 1 (tmd1_bits) */
114#define LE_T1_OWN (1<<31) /* LANCE owns the packet */
115#define LE_T1_ERR (1<<30) /* error summary */
116#define LE_T1_ADD_FCS (1<<29) /* add FCS (PCnet-PCI) */
117#define LE_T1_NO_FCS (1<<29) /* no FCS (ILACC) */
118#define LE_T1_MORE (1<<28) /* multiple collisions */
119#define LE_T1_LTINT (1<<28) /* transmit interrupt (if LTINTEN) */
120#define LE_T1_ONE (1<<27) /* single collision */
121#define LE_T1_DEF (1<<26) /* deferred transmit */
122#define LE_T1_STP (1<<25) /* start of packet */
123#define LE_T1_ENP (1<<24) /* end of packet */
124#define LE_T1_ONES (0xf<<12) /* must be ones */
125#define LE_T1_BCNT_MASK (0xfff) /* byte count mask */
126
127#define LE_T1_BITS \
128 "\20\40OWN\37ERR\36RES\35MORE\34ONE\33DEF\32STP\31ENP"
129
130/* Transmit message descriptor 3 (tmd3) */
131#define LE_T2_BUFF (1<<31) /* buffer error */
132#define LE_T2_UFLO (1<<30) /* underflow error */
133#define LE_T2_EXDEF (1<<29) /* excessive defferral */
134#define LE_T2_LCOL (1<<28) /* late collision */
135#define LE_T2_LCAR (1<<27) /* loss of carrier */
136#define LE_T2_RTRY (1<<26) /* retry error */
137#if 0
138#define LE_T3_TDR_MASK 0x03ff /* time domain reflectometry counter */
139#endif
140
141#define LE_T3_BITS \
142 "\12\40BUFF\37UFLO\35LCOL\34LCAR\33RTRY"
