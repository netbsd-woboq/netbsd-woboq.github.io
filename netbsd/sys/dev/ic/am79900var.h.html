36/*
37 * Ethernet software status per device.
38 *
39 * NOTE: this structure MUST be the first element in machine-dependent
40 * le_softc structures! This is designed SPECIFICALLY to make it possible
41 * to simply cast a "void *" to "struct le_softc *" or to
42 * "struct am79900_softc *". Among other things, this saves a lot of hair
43 * in the interrupt handlers.
44 */
45struct am79900_softc {
46 struct lance_softc lsc;
47};
48
49void am79900_config(struct am79900_softc *);
50int am79900_intr(void *);
51
52#endif /* _DEV_IC_AM79900VAR_H_ */
