38/*
39 * Transmit descriptor list size. This is arbitrary, but allocate
40 * enough descriptors for 64 pending transmissions and 16 segments
41 * per packet. Since a descriptor holds 2 buffer addresses, that's
42 * 8 descriptors per packet. This MUST work out to a power of 2.
43 */
44#define ATW_NTXSEGS 16
45
46#define ATW_TXQUEUELEN 64
47#define ATW_NTXDESC (ATW_TXQUEUELEN * ATW_NTXSEGS)
48#define ATW_NTXDESC_MASK (ATW_NTXDESC - 1)
49#define ATW_NEXTTX(x) ((x + 1) & ATW_NTXDESC_MASK)
50
51/*
52 * Receive descriptor list size. We have one Rx buffer per incoming
53 * packet, so this logic is a little simpler.
54 */
55#define ATW_NRXDESC 64
56#define ATW_NRXDESC_MASK (ATW_NRXDESC - 1)
57#define ATW_NEXTRX(x) ((x + 1) & ATW_NRXDESC_MASK)

59/*
60 * Control structures are DMA'd to the ADM8211 chip. We allocate them in
61 * a single clump that maps to a single DMA segment to make several things
62 * easier.
63 */

66 * The transmit descriptors.

367#define ATW_CDTXADDR(sc, x) ((sc)->sc_cddma + ATW_CDTXOFF((x)))
368#define ATW_CDRXADDR(sc, x) ((sc)->sc_cddma + ATW_CDRXOFF((x)))

370#define ATW_CDTXSYNC(sc, x, n, ops) \
371do { \
372 int __x, __n; \
373 \
374 __x = (x); \
375 __n = (n); \
376 \
377 /* If it will wrap around, sync to the end of the ring. */ \
378 if ((__x + __n) > ATW_NTXDESC) { \
379 bus_dmamap_sync((sc)->sc_dmat, (sc)->sc_cddmamap, \
380 ATW_CDTXOFF(__x), sizeof(struct atw_txdesc) * \
381 (ATW_NTXDESC - __x), (ops)); \
382 __n -= (ATW_NTXDESC - __x); \
383 __x = 0; \
384 } \
385 \
386 /* Now sync whatever is left. */ \
387 bus_dmamap_sync((sc)->sc_dmat, (sc)->sc_cddmamap, \
388 ATW_CDTXOFF(__x), sizeof(struct atw_txdesc) * __n, (ops)); \
389} while (0)
390
391#define ATW_CDRXSYNC(sc, x, ops) \
392 bus_dmamap_sync((sc)->sc_dmat, (sc)->sc_cddmamap, \
393 ATW_CDRXOFF((x)), sizeof(struct atw_rxdesc), (ops))

395/*
396 * Note we rely on MCLBYTES being a power of two. Because the `length'
397 * field is only 11 bits, we must subtract 1 from the length to avoid
398 * having it truncated to 0!
399 */
400static __inline void
401atw_init_rxdesc(struct atw_softc *sc, int x)
402{
403 struct atw_rxsoft *rxs = &sc->sc_rxsoft[x];
404 struct atw_rxdesc *rxd = &sc->sc_rxdescs[x];
405 struct mbuf *m = rxs->rxs_mbuf;
406
407 rxd->ar_buf1 =
408 htole32(rxs->rxs_dmamap->dm_segs[0].ds_addr);
409 rxd->ar_buf2 = /* for descriptor chaining */
410 htole32(ATW_CDRXADDR((sc), ATW_NEXTRX(x)));
411 rxd->ar_ctlrssi =
412 htole32(__SHIFTIN(((m->m_ext.ext_size - 1) & ~0x3U),
413 ATW_RXCTL_RBS1_MASK) |
414 0 /* ATW_RXCTL_RCH */ |
415 (x == (ATW_NRXDESC - 1) ? ATW_RXCTL_RER : 0));
416 rxd->ar_stat = htole32(ATW_RXSTAT_OWN);
417
418 ATW_CDRXSYNC((sc), x, BUS_DMASYNC_PREREAD|BUS_DMASYNC_PREWRITE);
419}
