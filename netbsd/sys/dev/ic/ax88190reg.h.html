32#ifndef _DEV_IC_AX88190REG_H_
33#define _DEV_IC_AX88190REG_H_
34
35#define AX88190_MEMR 0x04 /* MII/EEPROM/ Management Register */
36#define AX88190_MEMR_MDC 0x01 /* MII Clock */
37#define AX88190_MEMR_MDIR 0x02 /* MII STA MDIO signal direction
38 assert -> input */
39#define AX88190_MEMR_MDI 0x04 /* MII Data In */
40#define AX88190_MEMR_MDO 0x08 /* MII Data Out */
41#define AX88190_MEMR_EECS 0x10 /* EEPROM Chip Select */
42#define AX88190_MEMR_EEI 0x20 /* EEPROM Data In */
43#define AX88190_MEMR_EEO 0x40 /* EEPROM Data Out */
44#define AX88190_MEMR_EECLK 0x80 /* EEPROM Clock */
45
46/*
47 * Offset of LAN IOBASE0 and IOBASE1, and its size.
48 */
49#define AX88190_LAN_IOBASE 0x3ca
50#define AX88190_LAN_IOSIZE 4
51#define AX88790_CSR 0x3c2
52#define AX88790_CSR_SIZE 2
53
54/*
55 * Offset of NODE ID in SRAM memory of ASIX AX88190.
56 */
57#define AX88190_NODEID_OFFSET 0x400
58
59/*
60 * Start of SRAM buffer.
61 */
62#define AX88190_BUFFER_START 0x800
63
64#endif /* _DEV_IC_AX88190REG_H_ */
