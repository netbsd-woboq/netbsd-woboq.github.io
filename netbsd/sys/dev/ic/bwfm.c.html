133void
134bwfm_attach(struct bwfm_softc *sc)
135{
136 struct ieee80211com *ic = &sc->sc_ic;
137 struct ifnet *ifp = &sc->sc_if;
138 struct bwfm_task *t;
139 char fw_version[BWFM_DCMD_SMLEN];
140 uint32_t bandlist[3];
141 uint32_t tmp;
142 int i, j, error;
143
144 error = workqueue_create(&sc->sc_taskq, DEVNAME(sc),
145 bwfm_task, sc, PRI_NONE, IPL_NET, 0);
146 if (error != 0) {
147 printf("%s: could not create workqueue\n", DEVNAME(sc));
148 return;
149 }
150 sc->sc_freetask = pcq_create(BWFM_TASK_COUNT, KM_SLEEP);
151 for (i = 0; i < BWFM_TASK_COUNT; i++) {
152 t = &sc->sc_task[i];
153 t->t_sc = sc;
154 pcq_put(sc->sc_freetask, t);
155 }
156
157 /* Stop the device in case it was previously initialized */
158 bwfm_fwvar_cmd_set_int(sc, BWFM_C_DOWN, 1);
159
160 if (bwfm_fwvar_cmd_get_int(sc, BWFM_C_GET_VERSION, &tmp)) {
161 printf("%s: could not read io type\n", DEVNAME(sc));
162 return;
163 } else
164 sc->sc_io_type = tmp;
165 if (bwfm_fwvar_var_get_data(sc, "cur_etheraddr", ic->ic_myaddr,
166 sizeof(ic->ic_myaddr))) {
167 printf("%s: could not read mac address\n", DEVNAME(sc));
168 return;
169 }
170
171 memset(fw_version, 0, sizeof(fw_version));
172 if (bwfm_fwvar_var_get_data(sc, "ver", fw_version, sizeof(fw_version)) == 0)
173 printf("%s: %s", DEVNAME(sc), fw_version);
174 printf("%s: address %s\n", DEVNAME(sc), ether_sprintf(ic->ic_myaddr));
175
176 ic->ic_ifp = ifp;
177 ic->ic_phytype = IEEE80211_T_OFDM;
178 ic->ic_opmode = IEEE80211_M_STA;
179 ic->ic_state = IEEE80211_S_INIT;
180
181 ic->ic_caps =
182 IEEE80211_C_WEP |
183 IEEE80211_C_TKIP |
184 IEEE80211_C_AES |
185 IEEE80211_C_AES_CCM |
186#if notyet
187 IEEE80211_C_MONITOR | /* monitor mode suported */
188 IEEE80211_C_IBSS |
189 IEEE80211_C_TXPMGT |
190 IEEE80211_C_WME |
191#endif
192 IEEE80211_C_SHSLOT | /* short slot time supported */
193 IEEE80211_C_SHPREAMBLE | /* short preamble supported */
194 IEEE80211_C_WPA | /* 802.11i */
195 /* IEEE80211_C_WPA_4WAY */0; /* WPA 4-way handshake in hw */
196
197 /* IBSS channel undefined for now. */
198 ic->ic_ibss_chan = &ic->ic_channels[0];
199
200 if (bwfm_fwvar_cmd_get_data(sc, BWFM_C_GET_BANDLIST, bandlist,
201 sizeof(bandlist))) {
202 printf("%s: couldn't get supported band list\n", DEVNAME(sc));
203 return;
204 }
205 const u_int nbands = le32toh(bandlist[0]);
206 for (i = 1; i <= MIN(nbands, __arraycount(bandlist) - 1); i++) {
207 switch (le32toh(bandlist[i])) {
208 case BWFM_BAND_2G:
209 ic->ic_sup_rates[IEEE80211_MODE_11B] = ieee80211_std_rateset_11b;
210 ic->ic_sup_rates[IEEE80211_MODE_11G] = ieee80211_std_rateset_11g;
211
212 for (j = 0; j < __arraycount(bwfm_2ghz_channels); j++) {
213 uint8_t chan = bwfm_2ghz_channels[j];
214 ic->ic_channels[chan].ic_freq =
215 ieee80211_ieee2mhz(chan, IEEE80211_CHAN_2GHZ);
216 ic->ic_channels[chan].ic_flags =
217 IEEE80211_CHAN_CCK | IEEE80211_CHAN_OFDM |
218 IEEE80211_CHAN_DYN | IEEE80211_CHAN_2GHZ;
219 }
220 break;
221 case BWFM_BAND_5G:
222 ic->ic_sup_rates[IEEE80211_MODE_11A] = ieee80211_std_rateset_11a;
223
224 for (j = 0; j < __arraycount(bwfm_5ghz_channels); j++) {
225 uint8_t chan = bwfm_5ghz_channels[j];
226 ic->ic_channels[chan].ic_freq =
227 ieee80211_ieee2mhz(chan, IEEE80211_CHAN_5GHZ);
228 ic->ic_channels[chan].ic_flags =
229 IEEE80211_CHAN_A;
230 }
231 break;
232 }
233 }
234
235 ifp->if_softc = sc;
236 ifp->if_flags = IFF_BROADCAST | IFF_SIMPLEX | IFF_MULTICAST;
237 ifp->if_init = bwfm_init;
238 ifp->if_ioctl = bwfm_ioctl;
239 ifp->if_start = bwfm_start;
240 ifp->if_stop = bwfm_stop;
241 ifp->if_watchdog = bwfm_watchdog;
242 IFQ_SET_READY(&ifp->if_snd);
243 memcpy(ifp->if_xname, DEVNAME(sc), IFNAMSIZ);
244
245 error = if_initialize(ifp);
246 if (error != 0) {
247 printf("%s: if_initialize failed(%d)\n", DEVNAME(sc), error);
248 pcq_destroy(sc->sc_freetask);
249 workqueue_destroy(sc->sc_taskq);
250
251 return; /* Error */
252 }
253
254 ieee80211_ifattach(ic);
255 ifp->if_percpuq = if_percpuq_create(ifp);
256 if_deferred_start_init(ifp, NULL);
257 if_register(ifp);
258
259 sc->sc_newstate = ic->ic_newstate;
260 ic->ic_newstate = bwfm_newstate;
261 ic->ic_newassoc = bwfm_newassoc;
262 ic->ic_send_mgmt = bwfm_send_mgmt;
263 ic->ic_recv_mgmt = bwfm_recv_mgmt;
264 ic->ic_crypto.cs_key_set = bwfm_key_set;
265 ic->ic_crypto.cs_key_delete = bwfm_key_delete;
266 ieee80211_media_init(ic, bwfm_media_change, ieee80211_media_status);
267
268 ieee80211_announce(ic);
269
270 sc->sc_if_attached = true;
271}

274bwfm_detach(struct bwfm_softc *sc, int flags)
275{
276 struct ieee80211com *ic = &sc->sc_ic;
277 struct ifnet *ifp = ic->ic_ifp;
278

280 bpf_detach(ifp);

293void
294bwfm_start(struct ifnet *ifp)
295{
296 struct bwfm_softc *sc = ifp->if_softc;
297 struct ieee80211com *ic = &sc->sc_ic;
298 struct mbuf *m;
299 int error;
300
301 if ((ifp->if_flags & (IFF_RUNNING | IFF_OACTIVE)) != IFF_RUNNING)
302 return;
303
304 /* TODO: return if no link? */
305
306 for (;;) {
307 struct ieee80211_node *ni;
308 struct ether_header *eh;
309
310 /* Discard management packets (fw handles this for us) */
311 IF_DEQUEUE(&ic->ic_mgtq, m);
312 if (m != NULL) {
313 m_freem(m);
314 continue;
315 }
316
317 if (sc->sc_bus_ops->bs_txcheck(sc)) {
318 ifp->if_flags |= IFF_OACTIVE;
319 break;
320 }
321
322 IFQ_DEQUEUE(&ifp->if_snd, m);
323 if (m == NULL)
324 break;
325
326 eh = mtod(m, struct ether_header *);
327 ni = ieee80211_find_txnode(ic, eh->ether_dhost);
328 if (ni == NULL) {
329 ifp->if_oerrors++;
330 m_freem(m);
331 continue;
332 }
333
334 if (ieee80211_classify(ic, m, ni) != 0) {
335 ifp->if_oerrors++;
336 m_freem(m);
337 ieee80211_free_node(ni);
338 continue;
339 }
340
341 error = sc->sc_bus_ops->bs_txdata(sc, &m);
342 if (error == ENOBUFS) {
343 IF_PREPEND(&ifp->if_snd, m);
344 ifp->if_flags |= IFF_OACTIVE;
345 break;
346 }
347
348 if (error != 0) {
349 ifp->if_oerrors++;
350 m_freem(m);
351 if (ni != NULL)
352 ieee80211_free_node(ni);
353 } else {
354 bpf_mtap3(ic->ic_rawbpf, m, BPF_D_OUT);
355 }
356 }
357}

359int
360bwfm_init(struct ifnet *ifp)
361{
362 struct bwfm_softc *sc = ifp->if_softc;
363 struct ieee80211com *ic = &sc->sc_ic;
364 uint8_t evmask[BWFM_EVENT_MASK_LEN];
365 struct bwfm_join_pref_params join_pref[2];
366
367 if (bwfm_fwvar_var_set_int(sc, "mpc", 1)) {
368 printf("%s: could not set mpc\n", DEVNAME(sc));
369 return EIO;
370 }
371
372 /* Select target by RSSI (boost on 5GHz) */
373 join_pref[0].type = BWFM_JOIN_PREF_RSSI_DELTA;
374 join_pref[0].len = 2;
375 join_pref[0].rssi_gain = BWFM_JOIN_PREF_RSSI_BOOST;
376 join_pref[0].band = BWFM_JOIN_PREF_BAND_5G;
377 join_pref[1].type = BWFM_JOIN_PREF_RSSI;
378 join_pref[1].len = 2;
379 join_pref[1].rssi_gain = 0;
380 join_pref[1].band = 0;
381 if (bwfm_fwvar_var_set_data(sc, "join_pref", join_pref,
382 sizeof(join_pref))) {
383 printf("%s: could not set join pref\n", DEVNAME(sc));
384 return EIO;
385 }
386
387 memset(evmask, 0, sizeof(evmask));
388
389#define ENABLE_EVENT(e) evmask[(e) / 8] |= 1 << ((e) % 8)
390 /* Events used to drive the state machine */
391 ENABLE_EVENT(BWFM_E_ASSOC);
392 ENABLE_EVENT(BWFM_E_ESCAN_RESULT);
393 ENABLE_EVENT(BWFM_E_SET_SSID);
394 ENABLE_EVENT(BWFM_E_LINK);
395#undef ENABLE_EVENT
396
397#ifdef BWFM_DEBUG
398 memset(evmask, 0xff, sizeof(evmask));
399#endif
400
401 if (bwfm_fwvar_var_set_data(sc, "event_msgs", evmask, sizeof(evmask))) {
402 printf("%s: could not set event mask\n", DEVNAME(sc));
403 return EIO;
404 }
405
406 if (bwfm_fwvar_cmd_set_int(sc, BWFM_C_SET_SCAN_CHANNEL_TIME,
407 BWFM_DEFAULT_SCAN_CHANNEL_TIME)) {
408 printf("%s: could not set scan channel time\n", DEVNAME(sc));
409 return EIO;
410 }
411 if (bwfm_fwvar_cmd_set_int(sc, BWFM_C_SET_SCAN_UNASSOC_TIME,
412 BWFM_DEFAULT_SCAN_UNASSOC_TIME)) {
413 printf("%s: could not set scan unassoc time\n", DEVNAME(sc));
414 return EIO;
415 }
416 if (bwfm_fwvar_cmd_set_int(sc, BWFM_C_SET_SCAN_PASSIVE_TIME,
417 BWFM_DEFAULT_SCAN_PASSIVE_TIME)) {
418 printf("%s: could not set scan passive time\n", DEVNAME(sc));
419 return EIO;
420 }
421
422 if (bwfm_fwvar_cmd_set_int(sc, BWFM_C_SET_PM, 2)) {
423 printf("%s: could not set power\n", DEVNAME(sc));
424 return EIO;
425 }
426
427 bwfm_fwvar_var_set_int(sc, "txbf", 1);
428 bwfm_fwvar_cmd_set_int(sc, BWFM_C_UP, 0);
429 bwfm_fwvar_cmd_set_int(sc, BWFM_C_SET_INFRA, 1);
430 bwfm_fwvar_cmd_set_int(sc, BWFM_C_SET_AP, 0);
431
432 /* Disable all offloading (ARP, NDP, TCP/UDP cksum). */
433 bwfm_fwvar_var_set_int(sc, "arp_ol", 0);
434 bwfm_fwvar_var_set_int(sc, "arpoe", 0);
435 bwfm_fwvar_var_set_int(sc, "ndoe", 0);
436 bwfm_fwvar_var_set_int(sc, "toe", 0);
437
438 /* Accept all multicast frames. */
439 bwfm_fwvar_var_set_int(sc, "allmulti", 1);
440
441 /* Setup promiscuous mode */
442 bwfm_fwvar_cmd_set_int(sc, BWFM_C_SET_PROMISC,
443 (ifp->if_flags & IFF_PROMISC) ? 1 : 0);
444
445 /*
446 * Tell the firmware supplicant that we are going to handle the
447 * WPA handshake ourselves.
448 */
449 bwfm_fwvar_var_set_int(sc, "sup_wpa", 0);
450
451 ifp->if_flags |= IFF_RUNNING;
452 ifp->if_flags &= ~IFF_OACTIVE;
453
454 if (ic->ic_opmode != IEEE80211_M_MONITOR) {
455 if (ic->ic_roaming != IEEE80211_ROAMING_MANUAL)
456 ieee80211_new_state(ic, IEEE80211_S_SCAN, -1);
457 } else {
458 ieee80211_new_state(ic, IEEE80211_S_RUN, -1);
459 }
460
461 return 0;
462}

464void
465bwfm_stop(struct ifnet *ifp, int disable)
466{
467 struct bwfm_softc *sc = ifp->if_softc;
468 struct ieee80211com *ic = &sc->sc_ic;
469
470 sc->sc_tx_timer = 0;
471 ifp->if_timer = 0;
472 ifp->if_flags &= ~(IFF_RUNNING | IFF_OACTIVE);
473
474 bwfm_fwvar_cmd_set_int(sc, BWFM_C_DOWN, 1);
475 bwfm_fwvar_cmd_set_int(sc, BWFM_C_SET_PM, 0);
476
477 ieee80211_new_state(ic, IEEE80211_S_INIT, -1);
478}

499int
500bwfm_ioctl(struct ifnet *ifp, u_long cmd, void *data)
501{
502 struct bwfm_softc *sc = ifp->if_softc;
503 struct ieee80211com *ic = &sc->sc_ic;
504 int s, error = 0;
505
506 s = splnet();
507
508 switch (cmd) {
509 case SIOCSIFFLAGS:
510 if ((error = ifioctl_common(ifp, cmd, data)) != 0)
511 break;
512 switch (ifp->if_flags & (IFF_UP | IFF_RUNNING)) {
513 case IFF_UP | IFF_RUNNING:
514 break;
515 case IFF_UP:
516 bwfm_init(ifp);
517 break;
518 case IFF_RUNNING:
519 bwfm_stop(ifp, 1);
520 break;
521 case 0:
522 break;
523 }
524 break;
525
526 case SIOCADDMULTI:
527 case SIOCDELMULTI:
528 if ((error = ether_ioctl(ifp, cmd, data)) == ENETRESET) {
529 /* setup multicast filter, etc */
530 error = 0;
531 }
532 break;
533
534 default:
535 error = ieee80211_ioctl(ic, cmd, data);
536 }
537
538 if (error == ENETRESET) {
539 if ((ifp->if_flags & IFF_UP) != 0 &&
540 (ifp->if_flags & IFF_RUNNING) != 0 &&
541 ic->ic_roaming != IEEE80211_ROAMING_MANUAL) {
542 bwfm_init(ifp);
543 }
544 error = 0;
545 }
546
547 splx(s);
548
549 return error;
550}

565int
566bwfm_key_set(struct ieee80211com *ic, const struct ieee80211_key *wk,
567 const uint8_t mac[IEEE80211_ADDR_LEN])
568{
569 struct bwfm_softc *sc = ic->ic_ifp->if_softc;
570 struct bwfm_task *t;
571
572 t = pcq_get(sc->sc_freetask);
573 if (t == NULL) {
574 printf("%s: no free tasks\n", DEVNAME(sc));
575 return 0;
576 }
577
578 t->t_cmd = BWFM_TASK_KEY_SET;
579 t->t_key.key = wk;
580 memcpy(t->t_key.mac, mac, sizeof(t->t_key.mac));
581 workqueue_enqueue(sc->sc_taskq, (struct work *)t, NULL);
582 return 1;
583}

585static void
586bwfm_key_set_cb(struct bwfm_softc *sc, struct bwfm_cmd_key *ck)
587{
588 const struct ieee80211_key *wk = ck->key;
589 const uint8_t *mac = ck->mac;
590 struct bwfm_wsec_key wsec_key;
591 uint32_t wsec_enable, wsec;
592 bool ext_key;
593
594#ifdef BWFM_DEBUG
595 printf("key_set: key cipher %s len %d: ", wk->wk_cipher->ic_name, wk->wk_keylen);
596 for (int j = 0; j < sizeof(wk->wk_key); j++)
597 printf("%02x", wk->wk_key[j]);
598#endif
599
600 if ((wk->wk_flags & IEEE80211_KEY_GROUP) == 0 &&
601 wk->wk_cipher->ic_cipher != IEEE80211_CIPHER_WEP) {
602 ext_key = true;
603 } else {
604 ext_key = false;
605 }
606
607#ifdef BWFM_DEBUG
608 printf(", ext_key = %d", ext_key);
609 printf(", mac = %02x:%02x:%02x:%02x:%02x:%02x",
610 mac[0], mac[1], mac[2], mac[3], mac[4], mac[5]);
611 printf("\n");
612#endif
613
614 memset(&wsec_key, 0, sizeof(wsec_key));
615 if (ext_key && !IEEE80211_IS_MULTICAST(mac))
616 memcpy(wsec_key.ea, mac, sizeof(wsec_key.ea));
617 wsec_key.index = htole32(wk->wk_keyix);
618 wsec_key.len = htole32(wk->wk_keylen);
619 memcpy(wsec_key.data, wk->wk_key, sizeof(wsec_key.data));
620 if (!ext_key)
621 wsec_key.flags = htole32(BWFM_PRIMARY_KEY);
622
623 switch (wk->wk_cipher->ic_cipher) {
624 case IEEE80211_CIPHER_WEP:
625 if (wk->wk_keylen == 5)
626 wsec_key.algo = htole32(BWFM_CRYPTO_ALGO_WEP1);
627 else if (wk->wk_keylen == 13)
628 wsec_key.algo = htole32(BWFM_CRYPTO_ALGO_WEP128);
629 else
630 return;
631 wsec_enable = BWFM_WSEC_WEP;
632 break;
633 case IEEE80211_CIPHER_TKIP:
634 wsec_key.algo = htole32(BWFM_CRYPTO_ALGO_TKIP);
635 wsec_enable = BWFM_WSEC_TKIP;
636 break;
637 case IEEE80211_CIPHER_AES_CCM:
638 wsec_key.algo = htole32(BWFM_CRYPTO_ALGO_AES_CCM);
639 wsec_enable = BWFM_WSEC_AES;
640 break;
641 default:
642 printf("%s: %s: cipher %s not supported\n", DEVNAME(sc),
643 __func__, wk->wk_cipher->ic_name);
644 return;
645 }
646
647 if (bwfm_fwvar_var_set_data(sc, "wsec_key", &wsec_key, sizeof(wsec_key)))
648 return;
649
650 bwfm_fwvar_var_set_int(sc, "wpa_auth", BWFM_WPA_AUTH_WPA2_PSK);
651
652 bwfm_fwvar_var_get_int(sc, "wsec", &wsec);
653 wsec |= wsec_enable;
654 bwfm_fwvar_var_set_int(sc, "wsec", wsec);
655}

786/* Chip initialization (SDIO, PCIe) */
787int
788bwfm_chip_attach(struct bwfm_softc *sc)
789{
790 struct bwfm_core *core;
791 int need_socram = 0;
792 int has_socram = 0;
793 int cpu_found = 0;
794 uint32_t val;
795
796 LIST_INIT(&sc->sc_chip.ch_list);
797
798 if (sc->sc_buscore_ops->bc_prepare(sc) != 0) {
799 printf("%s: failed buscore prepare\n", DEVNAME(sc));
800 return 1;
801 }
802
803 val = sc->sc_buscore_ops->bc_read(sc,
804 BWFM_CHIP_BASE + BWFM_CHIP_REG_CHIPID);
805 sc->sc_chip.ch_chip = BWFM_CHIP_CHIPID_ID(val);
806 sc->sc_chip.ch_chiprev = BWFM_CHIP_CHIPID_REV(val);
807
808 if ((sc->sc_chip.ch_chip > 0xa000) || (sc->sc_chip.ch_chip < 0x4000))
809 snprintf(sc->sc_chip.ch_name, sizeof(sc->sc_chip.ch_name),
810 "%d", sc->sc_chip.ch_chip);
811 else
812 snprintf(sc->sc_chip.ch_name, sizeof(sc->sc_chip.ch_name),
813 "%x", sc->sc_chip.ch_chip);
814
815 switch (BWFM_CHIP_CHIPID_TYPE(val))
816 {
817 case BWFM_CHIP_CHIPID_TYPE_SOCI_SB:
818 printf("%s: SoC interconnect SB not implemented\n",
819 DEVNAME(sc));
820 return 1;
821 case BWFM_CHIP_CHIPID_TYPE_SOCI_AI:
822 sc->sc_chip.ch_core_isup = bwfm_chip_ai_isup;
823 sc->sc_chip.ch_core_disable = bwfm_chip_ai_disable;
824 sc->sc_chip.ch_core_reset = bwfm_chip_ai_reset;
825 bwfm_chip_dmp_erom_scan(sc);
826 break;
827 default:
828 printf("%s: SoC interconnect %d unknown\n",
829 DEVNAME(sc), BWFM_CHIP_CHIPID_TYPE(val));
830 return 1;
831 }
832
833 LIST_FOREACH(core, &sc->sc_chip.ch_list, co_link) {
834 DPRINTF(("%s: 0x%x:%-2d base 0x%08x wrap 0x%08x\n",
835 DEVNAME(sc), core->co_id, core->co_rev,
836 core->co_base, core->co_wrapbase));
837
838 switch (core->co_id) {
839 case BWFM_AGENT_CORE_ARM_CM3:
840 need_socram = true;
841 /* FALLTHROUGH */
842 case BWFM_AGENT_CORE_ARM_CR4:
843 case BWFM_AGENT_CORE_ARM_CA7:
844 cpu_found = true;
845 break;
846 case BWFM_AGENT_INTERNAL_MEM:
847 has_socram = true;
848 break;
849 default:
850 break;
851 }
852 }
853
854 if (!cpu_found) {
855 printf("%s: CPU core not detected\n", DEVNAME(sc));
856 return 1;
857 }
858 if (need_socram && !has_socram) {
859 printf("%s: RAM core not provided\n", DEVNAME(sc));
860 return 1;
861 }
862
863 bwfm_chip_set_passive(sc);
864
865 if (sc->sc_buscore_ops->bc_reset) {
866 sc->sc_buscore_ops->bc_reset(sc);
867 bwfm_chip_set_passive(sc);
868 }
869
870 if ((core = bwfm_chip_get_core(sc, BWFM_AGENT_CORE_ARM_CR4)) != NULL) {
871 bwfm_chip_tcm_ramsize(sc, core);
872 bwfm_chip_tcm_rambase(sc);
873 } else if ((core = bwfm_chip_get_core(sc, BWFM_AGENT_SYS_MEM)) != NULL) {
874 bwfm_chip_sysmem_ramsize(sc, core);
875 bwfm_chip_tcm_rambase(sc);
876 } else if ((core = bwfm_chip_get_core(sc, BWFM_AGENT_INTERNAL_MEM)) != NULL) {
877 bwfm_chip_socram_ramsize(sc, core);
878 }
879
880 core = bwfm_chip_get_core(sc, BWFM_AGENT_CORE_CHIPCOMMON);
881 sc->sc_chip.ch_cc_caps = sc->sc_buscore_ops->bc_read(sc,
882 core->co_base + BWFM_CHIP_REG_CAPABILITIES);
883 sc->sc_chip.ch_cc_caps_ext = sc->sc_buscore_ops->bc_read(sc,
884 core->co_base + BWFM_CHIP_REG_CAPABILITIES_EXT);
885
886 core = bwfm_chip_get_pmu(sc);
887 if (sc->sc_chip.ch_cc_caps & BWFM_CHIP_REG_CAPABILITIES_PMU) {
888 sc->sc_chip.ch_pmucaps = sc->sc_buscore_ops->bc_read(sc,
889 core->co_base + BWFM_CHIP_REG_PMUCAPABILITIES);
890 sc->sc_chip.ch_pmurev = sc->sc_chip.ch_pmucaps &
891 BWFM_CHIP_REG_PMUCAPABILITIES_REV_MASK;
892 }
893
894 if (sc->sc_buscore_ops->bc_setup)
895 sc->sc_buscore_ops->bc_setup(sc);
896
897 return 0;
898}

949bwfm_chip_ai_disable(struct bwfm_softc *sc, struct bwfm_core *core,
950 uint32_t prereset, uint32_t reset)
951{
952 uint32_t val;
953 int i;
954
955 val = sc->sc_buscore_ops->bc_read(sc,
956 core->co_wrapbase + BWFM_AGENT_RESET_CTL);
957 if ((val & BWFM_AGENT_RESET_CTL_RESET) == 0) {
958
959 sc->sc_buscore_ops->bc_write(sc,
960 core->co_wrapbase + BWFM_AGENT_IOCTL,
961 prereset | BWFM_AGENT_IOCTL_FGC | BWFM_AGENT_IOCTL_CLK);
962 sc->sc_buscore_ops->bc_read(sc,
963 core->co_wrapbase + BWFM_AGENT_IOCTL);
964
965 sc->sc_buscore_ops->bc_write(sc,
966 core->co_wrapbase + BWFM_AGENT_RESET_CTL,
967 BWFM_AGENT_RESET_CTL_RESET);
968 delay(20);
969
970 for (i = 300; i > 0; i--) {
971 if (sc->sc_buscore_ops->bc_read(sc,
972 core->co_wrapbase + BWFM_AGENT_RESET_CTL) ==
973 BWFM_AGENT_RESET_CTL_RESET)
974 break;
975 }
976 if (i == 0)
977 printf("%s: timeout on core reset\n", DEVNAME(sc));
978 }
979
980 sc->sc_buscore_ops->bc_write(sc,
981 core->co_wrapbase + BWFM_AGENT_IOCTL,
982 reset | BWFM_AGENT_IOCTL_FGC | BWFM_AGENT_IOCTL_CLK);
983 sc->sc_buscore_ops->bc_read(sc,
984 core->co_wrapbase + BWFM_AGENT_IOCTL);
985}

988bwfm_chip_ai_reset(struct bwfm_softc *sc, struct bwfm_core *core,
989 uint32_t prereset, uint32_t reset, uint32_t postreset)
990{
991 int i;
992
993 bwfm_chip_ai_disable(sc, core, prereset, reset);
994
995 for (i = 50; i > 0; i--) {
996 if ((sc->sc_buscore_ops->bc_read(sc,
997 core->co_wrapbase + BWFM_AGENT_RESET_CTL) &
998 BWFM_AGENT_RESET_CTL_RESET) == 0)
999 break;
1000 sc->sc_buscore_ops->bc_write(sc,

1002 delay(60);
1003 }
1004 if (i == 0)
1005 printf("%s: timeout on core reset\n", DEVNAME(sc));
1006
1007 sc->sc_buscore_ops->bc_write(sc,
1008 core->co_wrapbase + BWFM_AGENT_IOCTL,
1009 postreset | BWFM_AGENT_IOCTL_CLK);
1010 sc->sc_buscore_ops->bc_read(sc,
1011 core->co_wrapbase + BWFM_AGENT_IOCTL);
1012}

1014void
1015bwfm_chip_dmp_erom_scan(struct bwfm_softc *sc)
1016{
1017 uint32_t erom, val, base, wrap;
1018 uint8_t type = 0;
1019 uint16_t id;
1020 uint8_t nmw, nsw, rev;
1021 struct bwfm_core *core;
1022
1023 erom = sc->sc_buscore_ops->bc_read(sc,
1024 BWFM_CHIP_BASE + BWFM_CHIP_REG_EROMPTR);
1025 while (type != BWFM_DMP_DESC_EOT) {
1026 val = sc->sc_buscore_ops->bc_read(sc, erom);
1027 type = val & BWFM_DMP_DESC_MASK;
1028 erom += 4;
1029
1030 if (type != BWFM_DMP_DESC_COMPONENT)
1031 continue;
1032
1033 id = (val & BWFM_DMP_COMP_PARTNUM)
1034 >> BWFM_DMP_COMP_PARTNUM_S;
1035
1036 val = sc->sc_buscore_ops->bc_read(sc, erom);
1037 type = val & BWFM_DMP_DESC_MASK;
1038 erom += 4;
1039
1040 if (type != BWFM_DMP_DESC_COMPONENT) {
1041 printf("%s: not component descriptor\n", DEVNAME(sc));
1042 return;
1043 }
1044
1045 nmw = (val & BWFM_DMP_COMP_NUM_MWRAP)
1046 >> BWFM_DMP_COMP_NUM_MWRAP_S;
1047 nsw = (val & BWFM_DMP_COMP_NUM_SWRAP)
1048 >> BWFM_DMP_COMP_NUM_SWRAP_S;
1049 rev = (val & BWFM_DMP_COMP_REVISION)
1050 >> BWFM_DMP_COMP_REVISION_S;
1051
1052 if (nmw + nsw == 0 && id != BWFM_AGENT_CORE_PMU)
1053 continue;
1054
1055 if (bwfm_chip_dmp_get_regaddr(sc, &erom, &base, &wrap))
1056 continue;
1057
1058 core = kmem_alloc(sizeof(*core), KM_SLEEP);
1059 core->co_id = id;
1060 core->co_base = base;
1061 core->co_wrapbase = wrap;
1062 core->co_rev = rev;
1063 LIST_INSERT_HEAD(&sc->sc_chip.ch_list, core, co_link);
1064 }
1065}

1067int
1068bwfm_chip_dmp_get_regaddr(struct bwfm_softc *sc, uint32_t *erom,
1069 uint32_t *base, uint32_t *wrap)
1070{
1071 uint8_t type = 0, mpnum __unused = 0;
1072 uint8_t stype, sztype, wraptype;
1073 uint32_t val;
1074
1075 *base = 0;
1076 *wrap = 0;
1077
1078 val = sc->sc_buscore_ops->bc_read(sc, *erom);
1079 type = val & BWFM_DMP_DESC_MASK;
1080 if (type == BWFM_DMP_DESC_MASTER_PORT) {
1081 mpnum = (val & BWFM_DMP_MASTER_PORT_NUM)
1082 >> BWFM_DMP_MASTER_PORT_NUM_S;
1083 wraptype = BWFM_DMP_SLAVE_TYPE_MWRAP;
1084 *erom += 4;
1085 } else if ((type & ~BWFM_DMP_DESC_ADDRSIZE_GT32) ==
1086 BWFM_DMP_DESC_ADDRESS)
1087 wraptype = BWFM_DMP_SLAVE_TYPE_SWRAP;
1088 else
1089 return 1;
1090
1091 do {
1092 do {
1093 val = sc->sc_buscore_ops->bc_read(sc, *erom);
1094 type = val & BWFM_DMP_DESC_MASK;
1095 if (type == BWFM_DMP_DESC_COMPONENT)
1096 return 0;
1097 if (type == BWFM_DMP_DESC_EOT)
1098 return 1;
1099 *erom += 4;
1100 } while ((type & ~BWFM_DMP_DESC_ADDRSIZE_GT32) !=
1101 BWFM_DMP_DESC_ADDRESS);
1102
1103 if (type & BWFM_DMP_DESC_ADDRSIZE_GT32)
1104 *erom += 4;
1105
1106 sztype = (val & BWFM_DMP_SLAVE_SIZE_TYPE)
1107 >> BWFM_DMP_SLAVE_SIZE_TYPE_S;
1108 if (sztype == BWFM_DMP_SLAVE_SIZE_DESC) {
1109 val = sc->sc_buscore_ops->bc_read(sc, *erom);
1110 type = val & BWFM_DMP_DESC_MASK;
1111 if (type & BWFM_DMP_DESC_ADDRSIZE_GT32)
1112 *erom += 8;
1113 else
1114 *erom += 4;
1115 }
1116 if (sztype != BWFM_DMP_SLAVE_SIZE_4K)
1117 continue;
1118
1119 stype = (val & BWFM_DMP_SLAVE_TYPE) >> BWFM_DMP_SLAVE_TYPE_S;
1120 if (*base == 0 && stype == BWFM_DMP_SLAVE_TYPE_SLAVE)
1121 *base = val & BWFM_DMP_SLAVE_ADDR_BASE;
1122 if (*wrap == 0 && stype == wraptype)
1123 *wrap = val & BWFM_DMP_SLAVE_ADDR_BASE;
1124 } while (*base == 0 || *wrap == 0);
1125
1126 return 0;
1127}

1264/* RAM size helpers */
1265void
1266bwfm_chip_socram_ramsize(struct bwfm_softc *sc, struct bwfm_core *core)
1267{
1268 uint32_t coreinfo, nb, lss, banksize, bankinfo;
1269 uint32_t ramsize = 0, srsize = 0;
1270 int i;
1271
1272 if (!sc->sc_chip.ch_core_isup(sc, core))
1273 sc->sc_chip.ch_core_reset(sc, core, 0, 0, 0);
1274
1275 coreinfo = sc->sc_buscore_ops->bc_read(sc,
1276 core->co_base + BWFM_SOCRAM_COREINFO);
1277 nb = (coreinfo & BWFM_SOCRAM_COREINFO_SRNB_MASK)
1278 >> BWFM_SOCRAM_COREINFO_SRNB_SHIFT;
1279
1280 if (core->co_rev <= 7 || core->co_rev == 12) {
1281 banksize = coreinfo & BWFM_SOCRAM_COREINFO_SRBSZ_MASK;
1282 lss = (coreinfo & BWFM_SOCRAM_COREINFO_LSS_MASK)
1283 >> BWFM_SOCRAM_COREINFO_LSS_SHIFT;
1284 if (lss != 0)
1285 nb--;
1286 ramsize = nb * (1 << (banksize + BWFM_SOCRAM_COREINFO_SRBSZ_BASE));
1287 if (lss != 0)
1288 ramsize += (1 << ((lss - 1) + BWFM_SOCRAM_COREINFO_SRBSZ_BASE));
1289 } else {
1290 for (i = 0; i < nb; i++) {
1291 sc->sc_buscore_ops->bc_write(sc,
1292 core->co_base + BWFM_SOCRAM_BANKIDX,
1293 (BWFM_SOCRAM_BANKIDX_MEMTYPE_RAM <<
1294 BWFM_SOCRAM_BANKIDX_MEMTYPE_SHIFT) | i);
1295 bankinfo = sc->sc_buscore_ops->bc_read(sc,
1296 core->co_base + BWFM_SOCRAM_BANKINFO);
1297 banksize = ((bankinfo & BWFM_SOCRAM_BANKINFO_SZMASK) + 1)
1298 * BWFM_SOCRAM_BANKINFO_SZBASE;
1299 ramsize += banksize;
1300 if (bankinfo & BWFM_SOCRAM_BANKINFO_RETNTRAM_MASK)
1301 srsize += banksize;
1302 }
1303 }
1304
1305 switch (sc->sc_chip.ch_chip) {
1306 case BRCM_CC_4334_CHIP_ID:
1307 if (sc->sc_chip.ch_chiprev < 2)
1308 srsize = 32 * 1024;
1309 break;
1310 case BRCM_CC_43430_CHIP_ID:
1311 srsize = 64 * 1024;
1312 break;
1313 default:
1314 break;
1315 }
1316
1317 sc->sc_chip.ch_ramsize = ramsize;
1318 sc->sc_chip.ch_srsize = srsize;
1319}

1411/* BCDC protocol implementation */
1412int
1413bwfm_proto_bcdc_query_dcmd(struct bwfm_softc *sc, int ifidx,
1414 int cmd, char *buf, size_t *len)
1415{
1416 struct bwfm_proto_bcdc_dcmd *dcmd;
1417 size_t size = sizeof(dcmd->hdr) + *len;
1418 static int reqid = 0;
1419 int ret = 1;
1420
1421 reqid++;
1422
1423 dcmd = kmem_zalloc(sizeof(*dcmd), KM_SLEEP);
1424 if (*len > sizeof(dcmd->buf))
1425 goto err;
1426
1427 dcmd->hdr.cmd = htole32(cmd);
1428 dcmd->hdr.len = htole32(*len);
1429 dcmd->hdr.flags |= BWFM_BCDC_DCMD_GET;
1430 dcmd->hdr.flags |= BWFM_BCDC_DCMD_ID_SET(reqid);
1431 dcmd->hdr.flags |= BWFM_BCDC_DCMD_IF_SET(ifidx);
1432 dcmd->hdr.flags = htole32(dcmd->hdr.flags);
1433 memcpy(&dcmd->buf, buf, *len);
1434
1435 if (sc->sc_bus_ops->bs_txctl(sc, (void *)dcmd,
1436 sizeof(dcmd->hdr) + *len)) {
1437 DPRINTF(("%s: tx failed\n", DEVNAME(sc)));
1438 goto err;
1439 }
1440
1441 do {
1442 if (sc->sc_bus_ops->bs_rxctl(sc, (void *)dcmd, &size)) {
1443 DPRINTF(("%s: rx failed\n", DEVNAME(sc)));
1444 goto err;
1445 }
1446 dcmd->hdr.cmd = le32toh(dcmd->hdr.cmd);
1447 dcmd->hdr.len = le32toh(dcmd->hdr.len);
1448 dcmd->hdr.flags = le32toh(dcmd->hdr.flags);
1449 dcmd->hdr.status = le32toh(dcmd->hdr.status);
1450 } while (BWFM_BCDC_DCMD_ID_GET(dcmd->hdr.flags) != reqid);
1451
1452 if (BWFM_BCDC_DCMD_ID_GET(dcmd->hdr.flags) != reqid) {
1453 printf("%s: unexpected request id\n", DEVNAME(sc));
1454 goto err;
1455 }
1456
1457 if (buf) {
1458 if (size > *len)
1459 size = *len;
1460 if (size < *len)
1461 *len = size;
1462 memcpy(buf, dcmd->buf, *len);
1463 }
1464
1465 if (dcmd->hdr.flags & BWFM_BCDC_DCMD_ERROR)
1466 ret = dcmd->hdr.status;
1467 else
1468 ret = 0;
1469err:
1470 kmem_free(dcmd, sizeof(*dcmd));
1471 return ret;
1472}

1656/*
1657 * Derive wireless security settings from WPA/RSN IE.
1658 */
1659static uint32_t
1660bwfm_get_wsec(struct bwfm_softc *sc)
1661{
1662 struct ieee80211com *ic = &sc->sc_ic;
1663 uint8_t *wpa = ic->ic_opt_ie;
1664
1665 KASSERT(ic->ic_opt_ie_len > 0);
1666
1667 if (wpa[0] != IEEE80211_ELEMID_RSN) {
1668 if (ic->ic_opt_ie_len < 12)
1669 return BWFM_WSEC_NONE;
1670
1671 /* non-RSN IE, expect that we are doing WPA1 */
1672 if ((ic->ic_flags & IEEE80211_F_WPA1) == 0)
1673 return BWFM_WSEC_NONE;
1674
1675 /* Must contain WPA OUI */
1676 if (!bwfm_iswpaoui(wpa))
1677 return BWFM_WSEC_NONE;
1678
1679 switch (le32dec(wpa + 8)) {
1680 case ((WPA_CSE_TKIP<<24)|WPA_OUI):
1681 return BWFM_WSEC_TKIP;
1682 case ((WPA_CSE_CCMP<<24)|WPA_OUI):
1683 return BWFM_WSEC_AES;
1684 default:
1685 return BWFM_WSEC_NONE;
1686 }
1687 } else {
1688 if (ic->ic_opt_ie_len < 14)
1689 return BWFM_WSEC_NONE;
1690
1691 /* RSN IE, expect that we are doing WPA2 */
1692 if ((ic->ic_flags & IEEE80211_F_WPA2) == 0)
1693 return BWFM_WSEC_NONE;
1694
1695 switch (le32dec(wpa + 10)) {
1696 case ((RSN_CSE_TKIP<<24)|RSN_OUI):
1697 return BWFM_WSEC_TKIP;
1698 case ((RSN_CSE_CCMP<<24)|RSN_OUI):
1699 return BWFM_WSEC_AES;
1700 default:
1701 return BWFM_WSEC_NONE;
1702 }
1703 }
1704}

1706void
1707bwfm_connect(struct bwfm_softc *sc)
1708{
1709 struct ieee80211com *ic = &sc->sc_ic;
1710 struct ieee80211_node *ni = ic->ic_bss;
1711 struct bwfm_ext_join_params *params;
1712
1713 if (ic->ic_flags & IEEE80211_F_WPA) {
1714 uint32_t wsec = 0;
1715 uint32_t wpa = 0;
1716
1717 if (ic->ic_opt_ie_len)
1718 bwfm_fwvar_var_set_data(sc, "wpaie", ic->ic_opt_ie, ic->ic_opt_ie_len);
1719
1720 if (ic->ic_flags & IEEE80211_F_WPA1)
1721 wpa |= BWFM_WPA_AUTH_WPA_PSK;
1722 if (ic->ic_flags & IEEE80211_F_WPA2)
1723 wpa |= BWFM_WPA_AUTH_WPA2_PSK;
1724
1725 wsec |= bwfm_get_wsec(sc);
1726
1727 DPRINTF(("%s: WPA enabled, ic_flags = 0x%x, wpa 0x%x, wsec 0x%x\n",
1728 DEVNAME(sc), ic->ic_flags, wpa, wsec));
1729
1730 bwfm_fwvar_var_set_int(sc, "wpa_auth", wpa);
1731 bwfm_fwvar_var_set_int(sc, "wsec", wsec);
1732 } else {
1733 bwfm_fwvar_var_set_int(sc, "wpa_auth", BWFM_WPA_AUTH_DISABLED);
1734 bwfm_fwvar_var_set_int(sc, "wsec", BWFM_WSEC_NONE);
1735 }
1736
1737 bwfm_fwvar_var_set_int(sc, "auth", BWFM_AUTH_OPEN);
1738 bwfm_fwvar_var_set_int(sc, "mfp", BWFM_MFP_NONE);
1739
1740 if (ni->ni_esslen && ni->ni_esslen < BWFM_MAX_SSID_LEN) {
1741 params = kmem_zalloc(sizeof(*params), KM_SLEEP);
1742 memcpy(params->ssid.ssid, ni->ni_essid, ni->ni_esslen);
1743 params->ssid.len = htole32(ni->ni_esslen);
1744 memcpy(params->assoc.bssid, ni->ni_bssid, sizeof(params->assoc.bssid));
1745 params->scan.scan_type = -1;
1746 params->scan.nprobes = htole32(-1);
1747 params->scan.active_time = htole32(-1);
1748 params->scan.passive_time = htole32(-1);
1749 params->scan.home_time = htole32(-1);
1750 if (bwfm_fwvar_var_set_data(sc, "join", params, sizeof(*params))) {

1772 ntohs(e->ehdr.ether_type) == BWFM_ETHERTYPE_LINK_CTL &&
1773 memcmp(BWFM_BRCM_OUI, e->hdr.oui, sizeof(e->hdr.oui)) == 0 &&
1774 ntohs(e->hdr.usr_subtype) == BWFM_BRCM_SUBTYPE_EVENT) {
1775 bwfm_rx_event(sc, mtod(m, char *), m->m_len);

1790void
1791bwfm_rx_event(struct bwfm_softc *sc, char *buf, size_t len)
1792{
1793 struct ieee80211com *ic = &sc->sc_ic;
1794 struct bwfm_event *e = (void *)buf;
1795 int s;
1796
1797 DPRINTF(("%s: buf %p len %lu datalen %u code %u status %u"
1798 " reason %u\n", __func__, buf, len, ntohl(e->msg.datalen),
1799 ntohl(e->msg.event_type), ntohl(e->msg.status),
1800 ntohl(e->msg.reason)));
1801
1802 if (ntohl(e->msg.event_type) >= BWFM_E_LAST)
1803 return;
1804
1805 switch (ntohl(e->msg.event_type)) {
1806 case BWFM_E_ESCAN_RESULT: {
1807 struct bwfm_escan_results *res = (void *)(buf + sizeof(*e));
1808 struct bwfm_bss_info *bss;
1809 int i;
1810 if (ntohl(e->msg.status) != BWFM_E_STATUS_PARTIAL) {
1811 /* Scan complete */
1812 s = splnet();
1813 if (ic->ic_opmode != IEEE80211_M_MONITOR)
1814 ieee80211_end_scan(ic);
1815 splx(s);
1816 break;
1817 }
1818 len -= sizeof(*e);
1819 if (len < sizeof(*res) || len < le32toh(res->buflen)) {
1820 printf("%s: results too small\n", DEVNAME(sc));
1821 return;
1822 }
1823 len -= sizeof(*res);
1824 if (len < le16toh(res->bss_count) * sizeof(struct bwfm_bss_info)) {
1825 printf("%s: results too small\n", DEVNAME(sc));
1826 return;
1827 }
1828 bss = &res->bss_info[0];
1829 for (i = 0; i < le16toh(res->bss_count); i++) {
1830 /* Fix alignment of bss_info */
1831 union {
1832 struct bwfm_bss_info bss_info;
1833 uint8_t padding[BWFM_BSS_INFO_BUFLEN];
1834 } bss_buf;
1835 if (len > sizeof(bss_buf)) {
1836 printf("%s: bss_info buffer too big\n", DEVNAME(sc));
1837 } else {
1838 memcpy(&bss_buf, &res->bss_info[i], len);
1839 bwfm_scan_node(sc, &bss_buf.bss_info, len);
1840 }
1841 len -= sizeof(*bss) + le32toh(bss->length);
1842 bss = (void *)(((uintptr_t)bss) + le32toh(bss->length));
1843 if (len <= 0)
1844 break;
1845 }
1846 break;
1847 }
1848
1849 case BWFM_E_SET_SSID:
1850 if (ntohl(e->msg.status) == BWFM_E_STATUS_SUCCESS) {
1851 ieee80211_new_state(ic, IEEE80211_S_RUN, -1);
1852 } else {
1853 ieee80211_new_state(ic, IEEE80211_S_SCAN, -1);
1854 }
1855 break;
1856
1857 case BWFM_E_ASSOC:
1858 if (ntohl(e->msg.status) == BWFM_E_STATUS_SUCCESS) {
1859 ieee80211_new_state(ic, IEEE80211_S_ASSOC, -1);
1860 } else {
1861 ieee80211_new_state(ic, IEEE80211_S_SCAN, -1);
1862 }
1863 break;
1864
1865 case BWFM_E_LINK:
1866 if (ntohl(e->msg.status) == BWFM_E_STATUS_SUCCESS &&
1867 ntohl(e->msg.reason) == 0)
1868 break;
1869
1870 /* Link status has changed */
1871 ieee80211_new_state(ic, IEEE80211_S_SCAN, -1);
1872 break;
1873
1874 default:
1875 break;
1876 }
1877}

1879void
1880bwfm_scan_node(struct bwfm_softc *sc, struct bwfm_bss_info *bss, size_t len)
1881{
1882 struct ieee80211com *ic = &sc->sc_ic;
1883 struct ieee80211_frame wh;
1884 struct ieee80211_scanparams scan;
1885 uint8_t rates[sizeof(bss->rates) + 2];
1886 uint8_t ssid[sizeof(bss->ssid) + 2];
1887 uint8_t *frm, *sfrm, *efrm;
1888 uint64_t tsf;
1889
1890 tsf = 0;
1891 sfrm = ((uint8_t *)bss) + le16toh(bss->ie_offset);
1892 efrm = sfrm + le32toh(bss->ie_length);
1893
1894 /* Fake a wireless header with the scan result's BSSID */
1895 memset(&wh, 0, sizeof(wh));
1896 IEEE80211_ADDR_COPY(wh.i_addr2, bss->bssid);
1897 IEEE80211_ADDR_COPY(wh.i_addr3, bss->bssid);
1898
1899 if (efrm - sfrm < 12) {
1900 ic->ic_stats.is_rx_elem_toosmall++;
1901 return;
1902 }
1903
1904 rates[0] = 0;
1905 rates[1] = le32toh(bss->nrates);
1906 memcpy(&rates[2], bss->rates, sizeof(bss->rates));
1907
1908 ssid[0] = 0;
1909 ssid[1] = bss->ssid_len;
1910 memcpy(&ssid[2], bss->ssid, sizeof(bss->ssid));
1911
1912 /* Build scan result */
1913 memset(&scan, 0, sizeof(scan));
1914 scan.sp_tstamp = (uint8_t *)&tsf;
1915 scan.sp_bintval = le16toh(bss->beacon_period);
1916 scan.sp_capinfo = le16toh(bss->capability);
1917 scan.sp_bchan = ieee80211_chan2ieee(ic, ic->ic_curchan);
1918 scan.sp_chan = scan.sp_bchan;
1919 scan.sp_rates = rates;
1920 scan.sp_ssid = ssid;
1921
1922 for (frm = sfrm; frm < efrm; frm += frm[1] + 2) {
1923 switch (frm[0]) {
1924 case IEEE80211_ELEMID_COUNTRY:
1925 scan.sp_country = frm;
1926 break;
1927 case IEEE80211_ELEMID_FHPARMS:
1928 if (ic->ic_phytype == IEEE80211_T_FH) {
1929 if (frm + 6 >= efrm)
1930 break;
1931 scan.sp_fhdwell = le16dec(&frm[2]);
1932 scan.sp_chan = IEEE80211_FH_CHAN(frm[4], frm[5]);
1933 scan.sp_fhindex = frm[6];
1934 }
1935 break;
1936 case IEEE80211_ELEMID_DSPARMS:
1937 if (ic->ic_phytype != IEEE80211_T_FH) {
1938 if (frm + 2 >= efrm)
1939 break;
1940 scan.sp_chan = frm[2];
1941 }
1942 break;
1943 case IEEE80211_ELEMID_TIM:
1944 scan.sp_tim = frm;
1945 scan.sp_timoff = frm - sfrm;
1946 break;
1947 case IEEE80211_ELEMID_XRATES:
1948 scan.sp_xrates = frm;
1949 break;
1950 case IEEE80211_ELEMID_ERP:
1951 if (frm + 1 >= efrm)
1952 break;
1953 if (frm[1] != 1) {
1954 ic->ic_stats.is_rx_elem_toobig++;
1955 break;
1956 }
1957 scan.sp_erp = frm[2];
1958 break;
1959 case IEEE80211_ELEMID_RSN:
1960 scan.sp_wpa = frm;
1961 break;
1962 case IEEE80211_ELEMID_VENDOR:
1963 if (frm + 1 >= efrm)
1964 break;
1965 if (frm + frm[1] + 2 >= efrm)
1966 break;
1967 if (bwfm_iswpaoui(frm))
1968 scan.sp_wpa = frm;
1969 break;
1970 }
1971 if (frm + 1 >= efrm)
1972 break;
1973 }
1974
1975 if (ic->ic_flags & IEEE80211_F_SCAN)
1976 ieee80211_add_scan(ic, &scan, &wh, IEEE80211_FC0_SUBTYPE_BEACON,
1977 le32toh(bss->rssi), 0);
1978}
