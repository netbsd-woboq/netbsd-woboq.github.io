118enum bwfm_task_cmd {
119 BWFM_TASK_NEWSTATE,
120 BWFM_TASK_KEY_SET,
121 BWFM_TASK_KEY_DELETE,
122};
123
124struct bwfm_cmd_newstate {
125 enum ieee80211_state state;
126 int arg;
127};
128
129struct bwfm_cmd_key {
130 const struct ieee80211_key *key;
131 uint8_t mac[IEEE80211_ADDR_LEN];
132};
133
134struct bwfm_task {
135 struct work t_work;
136 struct bwfm_softc *t_sc;
137 enum bwfm_task_cmd t_cmd;
138 union {
139 struct bwfm_cmd_newstate newstate;
140 struct bwfm_cmd_key key;
141 } t_u;
142#define t_newstate t_u.newstate
143#define t_key t_u.key
144};
