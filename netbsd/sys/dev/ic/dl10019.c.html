72const struct mii_bitbang_ops dl10019_mii_bitbang_ops = {
73 dl10019_mii_bitbang_read,
74 dl10019_mii_bitbang_write,
75 {
76 DL0_GPIO_MII_DATAOUT, /* MII_BIT_MDO */
77 DL0_GPIO_MII_DATAIN, /* MII_BIT_MDI */
78 DL0_GPIO_MII_CLK, /* MII_BIT_MDC */
79 DL0_19_GPIO_MII_DIROUT, /* MII_BIT_DIR_HOST_PHY */
80 0, /* MII_BIT_DIR_PHY_HOST */
81 }
82};
83
84const struct mii_bitbang_ops dl10022_mii_bitbang_ops = {
85 dl10019_mii_bitbang_read,
86 dl10019_mii_bitbang_write,
87 {
88 DL0_GPIO_MII_DATAOUT, /* MII_BIT_MDO */
89 DL0_GPIO_MII_DATAIN, /* MII_BIT_MDI */
90 DL0_GPIO_MII_CLK, /* MII_BIT_MDC */
91 DL0_22_GPIO_MII_DIROUT, /* MII_BIT_DIR_HOST_PHY */
92 0, /* MII_BIT_DIR_PHY_HOST */
93 }
94};

96static void
97dl10019_mii_reset(struct dp8390_softc *sc)
98{
99 struct ne2000_softc *nsc = (void *) sc;
100 int i;
101
102 if (nsc->sc_type != NE2000_TYPE_DL10022)
103 return;
104
105 for (i = 0; i < 2; i++) {
106 bus_space_write_1(sc->sc_regt, sc->sc_regh, NEDL_DL0_GPIO,
107 0x08);
108 delay(1);
109 bus_space_write_1(sc->sc_regt, sc->sc_regh, NEDL_DL0_GPIO,
110 0x0c);
111 delay(1);
112 }
113 bus_space_write_1(sc->sc_regt, sc->sc_regh, NEDL_DL0_GPIO, 0x00);
114}

116void
117dl10019_media_init(struct dp8390_softc *sc)
118{
119 struct ifnet *ifp = &sc->sc_ec.ec_if;
120 struct mii_data *mii = &sc->sc_mii;
121
122 mii->mii_ifp = ifp;
123 mii->mii_readreg = dl10019_mii_readreg;
124 mii->mii_writereg = dl10019_mii_writereg;
125 mii->mii_statchg = dl10019_mii_statchg;
126 ifmedia_init(&mii->mii_media, IFM_IMASK, dp8390_mediachange,
127 dp8390_mediastatus);
128
129 dl10019_mii_reset(sc);
130
131 mii_attach(sc->sc_dev, mii, 0xffffffff, MII_PHY_ANY,
132 MII_OFFSET_ANY, 0);
133
134 if (LIST_FIRST(&mii->mii_phys) == NULL) {
135 ifmedia_add(&mii->mii_media, IFM_ETHER | IFM_NONE, 0, NULL);
136 ifmedia_set(&mii->mii_media, IFM_ETHER | IFM_NONE);
137 } else
138 ifmedia_set(&mii->mii_media, IFM_ETHER | IFM_AUTO);
139}
140
141void
142dl10019_media_fini(struct dp8390_softc *sc)
143{
144
145 mii_detach(&sc->sc_mii, MII_PHY_ANY, MII_OFFSET_ANY);
146}
147
148int
149dl10019_mediachange(struct dp8390_softc *sc)
150{
151 int rc;
152
153 if ((rc = mii_mediachg(&sc->sc_mii)) == ENXIO)
154 return 0;
155 return rc;
156}
157
158void
159dl10019_mediastatus(struct dp8390_softc *sc, struct ifmediareq *ifmr)
160{
161
162 mii_pollstat(&sc->sc_mii);
163 ifmr->ifm_status = sc->sc_mii.mii_media_status;
164 ifmr->ifm_active = sc->sc_mii.mii_media_active;
165}

167void
168dl10019_init_card(struct dp8390_softc *sc)
169{
170
171 dl10019_mii_reset(sc);
172 mii_mediachg(&sc->sc_mii);
173}

175void
176dl10019_stop_card(struct dp8390_softc *sc)
177{
178
179 mii_down(&sc->sc_mii);
180}

182u_int32_t
183dl10019_mii_bitbang_read(device_t self)
184{
185 struct dp8390_softc *sc = device_private(self);
186
187 /* We're already in Page 0. */
188 return (bus_space_read_1(sc->sc_regt, sc->sc_regh, NEDL_DL0_GPIO) &
189 ~DL0_GPIO_PRESERVE);
190}
191
192void
193dl10019_mii_bitbang_write(device_t self, u_int32_t val)
194{
195 struct dp8390_softc *sc = device_private(self);
196 u_int8_t gpio;
197
198 /* We're already in Page 0. */
199 gpio = bus_space_read_1(sc->sc_regt, sc->sc_regh, NEDL_DL0_GPIO);
200 bus_space_write_1(sc->sc_regt, sc->sc_regh, NEDL_DL0_GPIO,
201 (val & ~DL0_GPIO_PRESERVE) | (gpio & DL0_GPIO_PRESERVE));
202}

204int
205dl10019_mii_readreg(device_t self, int phy, int reg, uint16_t *val)
206{
207 struct ne2000_softc *nsc = device_private(self);
208 const struct mii_bitbang_ops *ops;
209
210 ops = (nsc->sc_type == NE2000_TYPE_DL10022) ?
211 &dl10022_mii_bitbang_ops : &dl10019_mii_bitbang_ops;
212
213 return mii_bitbang_readreg(self, ops, phy, reg, val);
214}
215
216int
217dl10019_mii_writereg(device_t self, int phy, int reg, uint16_t val)
218{
219 struct ne2000_softc *nsc = device_private(self);
220 const struct mii_bitbang_ops *ops;
221
222 ops = (nsc->sc_type == NE2000_TYPE_DL10022) ?
223 &dl10022_mii_bitbang_ops : &dl10019_mii_bitbang_ops;
224
225 return mii_bitbang_writereg(self, ops, phy, reg, val);
226}

228void
229dl10019_mii_statchg(struct ifnet *ifp)
230{
231 struct ne2000_softc *nsc = ifp->if_softc;
232 struct dp8390_softc *sc = &nsc->sc_dp8390;
233
234 /*
235 * Disable collision detection on the DL10022 if
236 * we are on a full-duplex link.
237 */
238 if (nsc->sc_type == NE2000_TYPE_DL10022) {
239 u_int8_t diag;
240
241 if (sc->sc_mii.mii_media_active & IFM_FDX)
242 diag = DL0_DIAG_NOCOLLDETECT;
243 else
244 diag = 0;
245 bus_space_write_1(sc->sc_regt, sc->sc_regh,
246 NEDL_DL0_DIAG, diag);
247 }
