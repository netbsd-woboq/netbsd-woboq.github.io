155bool
156dwiic_attach(struct dwiic_softc *sc)
157{
158 if (sc->sc_power != NULL) {
159 if (!sc->sc_power(sc, 1)) {
160 aprint_error_dev(sc->sc_dev, "failed to power up\n");
161 return 0;
162 }
163 }
164
165 /* fetch timing parameters */
166 if (sc->ss_hcnt == 0)
167 sc->ss_hcnt = dwiic_read(sc, DW_IC_SS_SCL_HCNT);
168 if (sc->ss_lcnt == 0)
169 sc->ss_lcnt = dwiic_read(sc, DW_IC_SS_SCL_LCNT);
170 if (sc->fs_hcnt == 0)
171 sc->fs_hcnt = dwiic_read(sc, DW_IC_FS_SCL_HCNT);
172 if (sc->fs_lcnt == 0)
173 sc->fs_lcnt = dwiic_read(sc, DW_IC_FS_SCL_LCNT);
174 if (sc->sda_hold_time == 0)
175 sc->sda_hold_time = dwiic_read(sc, DW_IC_SDA_HOLD);
176
177 if (dwiic_init(sc)) {
178 aprint_error_dev(sc->sc_dev, "failed initializing\n");
179 return 0;
180 }
181
182 /* leave the controller disabled */
183 dwiic_write(sc, DW_IC_INTR_MASK, 0);
184 dwiic_enable(sc, 0);
185 dwiic_read(sc, DW_IC_CLR_INTR);
186
187 mutex_init(&sc->sc_i2c_lock, MUTEX_DEFAULT, IPL_NONE);
188 mutex_init(&sc->sc_int_lock, MUTEX_DEFAULT, IPL_VM);
189 cv_init(&sc->sc_int_readwait, "dwiicr");
190 cv_init(&sc->sc_int_writewait, "dwiicw");
191 cv_init(&sc->sc_int_stopwait, "dwiics");
192
193 /* setup and attach iic bus */
194 sc->sc_i2c_tag.ic_cookie = sc;
195 sc->sc_i2c_tag.ic_acquire_bus = dwiic_i2c_acquire_bus;
196 sc->sc_i2c_tag.ic_release_bus = dwiic_i2c_release_bus;
197 sc->sc_i2c_tag.ic_exec = dwiic_i2c_exec;
198
199 sc->sc_iba.iba_tag = &sc->sc_i2c_tag;
200
201 /* config_found_ia for "i2cbus" is done in the bus-attachment glue */
202
203 return 1;
204}

206int
207dwiic_detach(device_t self, int flags)
208{
209 struct dwiic_softc *sc = device_private(self);
210
211 dwiic_enable(sc, 0);
212 if (sc->sc_ih != NULL) {
213 intr_disestablish(sc->sc_ih);
214 sc->sc_ih = NULL;
215 }
216
217 return 0;
218}
219
220bool
221dwiic_suspend(device_t self, const pmf_qual_t *qual)
222{
223 struct dwiic_softc *sc = device_private(self);
224
225 /* disable controller */
226 dwiic_enable(sc, 0);
227
228 /* disable interrupts */
229 dwiic_write(sc, DW_IC_INTR_MASK, 0);
230 dwiic_read(sc, DW_IC_CLR_INTR);
231 if (sc->sc_power != NULL) {
232 if (!sc->sc_power(sc, 0)) {
233 device_printf(sc->sc_dev, "failed to power off\n");
234 }
235 }
236 return true;
237}
238
239bool
240dwiic_resume(device_t self, const pmf_qual_t *qual)
241{
242 struct dwiic_softc *sc = device_private(self);
243 if (sc->sc_power != NULL) {
244 if (!sc->sc_power(sc, 1)) {
245 device_printf(sc->sc_dev, "failed to power up\n");
246 return false;
247 }
248 }
249 dwiic_init(sc);
250 return true;
251}

272static int
273dwiic_i2c_acquire_bus(void *cookie, int flags)
274{
275 struct dwiic_softc *sc = cookie;
276
277 if (cold || sc->sc_poll || (flags & I2C_F_POLL))
278 return (0);
279
280 mutex_enter(&sc->sc_i2c_lock);
281 return 0;
282}
283
284void
285dwiic_i2c_release_bus(void *cookie, int flags)
286{
287 struct dwiic_softc *sc = cookie;
288
289 if (cold || sc->sc_poll || (flags & I2C_F_POLL))
290 return;
291
292 mutex_exit(&sc->sc_i2c_lock);
293}

295static int
296dwiic_init(struct dwiic_softc *sc)
297{
298 uint32_t reg;
299
300 /* make sure we're talking to a device we know */
301 reg = dwiic_read(sc, DW_IC_COMP_TYPE);
302 if (reg != DW_IC_COMP_TYPE_VALUE) {
303 DPRINTF(("%s: invalid component type 0x%x\n",
304 device_xname(sc->sc_dev), reg));
305 return 1;
306 }
307
308 /* disable the adapter */
309 dwiic_enable(sc, 0);
310
311 /* write standard-mode SCL timing parameters */
312 dwiic_write(sc, DW_IC_SS_SCL_HCNT, sc->ss_hcnt);
313 dwiic_write(sc, DW_IC_SS_SCL_LCNT, sc->ss_lcnt);
314
315 /* and fast-mode SCL timing parameters */
316 dwiic_write(sc, DW_IC_FS_SCL_HCNT, sc->fs_hcnt);
317 dwiic_write(sc, DW_IC_FS_SCL_LCNT, sc->fs_lcnt);
318
319 /* SDA hold time */
320 reg = dwiic_read(sc, DW_IC_COMP_VERSION);
321 if (reg >= DW_IC_SDA_HOLD_MIN_VERS)
322 dwiic_write(sc, DW_IC_SDA_HOLD, sc->sda_hold_time);
323
324 /* FIFO threshold levels */
325 sc->tx_fifo_depth = 32;
326 sc->rx_fifo_depth = 32;
327 dwiic_write(sc, DW_IC_TX_TL, sc->tx_fifo_depth / 2);
328 dwiic_write(sc, DW_IC_RX_TL, 0);
329
330 /* configure as i2c master with fast speed */
331 sc->master_cfg = DW_IC_CON_MASTER | DW_IC_CON_SLAVE_DISABLE |
332 DW_IC_CON_RESTART_EN | DW_IC_CON_SPEED_FAST;
333 dwiic_write(sc, DW_IC_CON, sc->master_cfg);
334
335 return 0;
336}

338static void
339dwiic_enable(struct dwiic_softc *sc, bool enable)
340{
341 int retries;
342
343 for (retries = 100; retries > 0; retries--) {
344 dwiic_write(sc, DW_IC_ENABLE, enable);
345 if ((dwiic_read(sc, DW_IC_ENABLE_STATUS) & 1) == enable)
346 return;
347
348 DELAY(25);
349 }
350
351 device_printf(sc->sc_dev, "failed to %sable\n",
352 (enable ? "en" : "dis"));
353}

355static int
356dwiic_i2c_exec(void *cookie, i2c_op_t op, i2c_addr_t addr, const void *cmdbuf,
357 size_t cmdlen, void *buf, size_t len, int flags)
358{
359 struct dwiic_softc *sc = cookie;
360 u_int32_t ic_con, st, cmd, resp;
361 int retries, tx_limit, rx_avail, x, readpos;
362 const uint8_t *bcmd;
363 uint8_t *bdata;
364
365 if (cold || sc->sc_poll)
366 flags |= I2C_F_POLL;
367
368 DPRINTF(("%s: %s: op %d, addr 0x%02x, cmdlen %zu, len %zu, "
369 "flags 0x%02x\n", device_xname(sc->sc_dev), __func__, op, addr, cmdlen,
370 len, flags));
371
372 /* setup transfer */
373 sc->sc_i2c_xfer.op = op;
374 sc->sc_i2c_xfer.buf = buf;
375 sc->sc_i2c_xfer.len = len;
376 sc->sc_i2c_xfer.flags = flags;
377 sc->sc_i2c_xfer.error = 0;
378
379 /* wait for bus to be idle */
380 for (retries = 100; retries > 0; retries--) {
381 st = dwiic_read(sc, DW_IC_STATUS);
382 if (!(st & DW_IC_STATUS_ACTIVITY))
383 break;
384 DELAY(1000);
385 }
386 DPRINTF(("%s: %s: status 0x%x\n", device_xname(sc->sc_dev), __func__, st));
387 if (st & DW_IC_STATUS_ACTIVITY) {
388 return (1);
389 }
390
391 /* disable controller */
392 dwiic_enable(sc, 0);
393
394 /* set slave address */
395 ic_con = dwiic_read(sc, DW_IC_CON);
396 ic_con &= ~DW_IC_CON_10BITADDR_MASTER;
397 dwiic_write(sc, DW_IC_CON, ic_con);
398 dwiic_write(sc, DW_IC_TAR, addr);
399
400 /* disable interrupts */
401 dwiic_write(sc, DW_IC_INTR_MASK, 0);
402 dwiic_read(sc, DW_IC_CLR_INTR);
403
404 /* enable controller */
405 dwiic_enable(sc, 1);
406
407 /* wait until the controller is ready for commands */
408 if (flags & I2C_F_POLL)
409 DELAY(200);
410 else {
411 mutex_enter(&sc->sc_int_lock);
412 dwiic_read(sc, DW_IC_CLR_INTR);
413 dwiic_write(sc, DW_IC_INTR_MASK, DW_IC_INTR_TX_EMPTY);
414
415 if (cv_timedwait(&sc->sc_int_writewait,
416 &sc->sc_int_lock, hz / 2) != 0)
417 device_printf(sc->sc_dev,
418 "timed out waiting for tx_empty intr\n");
419 dwiic_write(sc, DW_IC_INTR_MASK, 0);
420 dwiic_read(sc, DW_IC_CLR_INTR);
421 mutex_exit(&sc->sc_int_lock);
422 }
423
424 /* send our command, one byte at a time */
425 if (cmdlen > 0) {
426 bcmd = (const void *)cmdbuf;
427
428 DPRINTF(("%s: %s: sending cmd (len %zu):", device_xname(sc->sc_dev),
429 __func__, cmdlen));
430 for (x = 0; x < cmdlen; x++)
431 DPRINTF((" %02x", bcmd[x]));
432 DPRINTF(("\n"));
433
434 tx_limit = sc->tx_fifo_depth - dwiic_read(sc, DW_IC_TXFLR);
435 if (cmdlen > tx_limit) {
436 /* TODO */
437 device_printf(sc->sc_dev, "can't write %zu (> %d)\n",
438 cmdlen, tx_limit);
439 sc->sc_i2c_xfer.error = 1;
440 return (1);
441 }
442
443 for (x = 0; x < cmdlen; x++) {
444 cmd = bcmd[x];
445 /*
446 * Generate STOP condition if this is the last
447 * byte of the transfer.
448 */
449 if (x == (cmdlen - 1) && len == 0 && I2C_OP_STOP_P(op))
450 cmd |= DW_IC_DATA_CMD_STOP;
451 dwiic_write(sc, DW_IC_DATA_CMD, cmd);
452 }
453 }
454
455 bdata = (void *)buf;
456 x = readpos = 0;
457 tx_limit = sc->tx_fifo_depth - dwiic_read(sc, DW_IC_TXFLR);
458
459 DPRINTF(("%s: %s: need to read %zu bytes, can send %d read reqs\n",
460 device_xname(sc->sc_dev), __func__, len, tx_limit));
461
462 while (x < len) {
463 if (I2C_OP_WRITE_P(op))
464 cmd = bdata[x];
465 else
466 cmd = DW_IC_DATA_CMD_READ;
467
468 /*
469 * Generate RESTART condition if we're reversing
470 * direction.
471 */
472 if (x == 0 && cmdlen > 0 && I2C_OP_READ_P(op))
473 cmd |= DW_IC_DATA_CMD_RESTART;
474 /*
475 * Generate STOP conditon on the last byte of the
476 * transfer.
477 */
478 if (x == (len - 1) && I2C_OP_STOP_P(op))
479 cmd |= DW_IC_DATA_CMD_STOP;
480
481 dwiic_write(sc, DW_IC_DATA_CMD, cmd);
482
483 tx_limit--;
484 x++;
485
486 /*
487 * As TXFLR fills up, we need to clear it out by reading all
488 * available data.
489 */
490 while (I2C_OP_READ_P(op) && (tx_limit == 0 || x == len)) {
491 DPRINTF(("%s: %s: tx_limit %d, sent %d read reqs\n",
492 device_xname(sc->sc_dev), __func__, tx_limit, x));
493
494 if (flags & I2C_F_POLL) {
495 for (retries = 100; retries > 0; retries--) {
496 rx_avail = dwiic_read(sc, DW_IC_RXFLR);
497 if (rx_avail > 0)
498 break;
499 DELAY(50);
500 }
501 } else {
502 mutex_enter(&sc->sc_int_lock);
503 dwiic_read(sc, DW_IC_CLR_INTR);
504 dwiic_write(sc, DW_IC_INTR_MASK,
505 DW_IC_INTR_RX_FULL);
506
507 if (cv_timedwait(&sc->sc_int_readwait,
508 &sc->sc_int_lock, hz / 2) != 0)
509 device_printf(sc->sc_dev,
510 "timed out waiting for "
511 "rx_full intr\n");
512
513 dwiic_write(sc, DW_IC_INTR_MASK, 0);
514 dwiic_read(sc, DW_IC_CLR_INTR);
515 mutex_exit(&sc->sc_int_lock);
516 rx_avail = dwiic_read(sc, DW_IC_RXFLR);
517 }
518
519 if (rx_avail == 0) {
520 device_printf(sc->sc_dev,
521 "timed out reading remaining %d\n",
522 (int)(len - 1 - readpos));
523 sc->sc_i2c_xfer.error = 1;
524 return (1);
525 }
526
527 DPRINTF(("%s: %s: %d avail to read (%zu remaining)\n",
528 device_xname(sc->sc_dev), __func__, rx_avail,
529 len - readpos));
530
531 while (rx_avail > 0) {
532 resp = dwiic_read(sc, DW_IC_DATA_CMD);
533 if (readpos < len) {
534 bdata[readpos] = resp;
535 readpos++;
536 }
537 rx_avail--;
538 }
539
540 if (readpos >= len)
541 break;
542
543 DPRINTF(("%s: still need to read %d bytes\n",
544 device_xname(sc->sc_dev), (int)(len - readpos)));
545 tx_limit = sc->tx_fifo_depth -
546 dwiic_read(sc, DW_IC_TXFLR);
547 }
548 }
549
550 if (I2C_OP_STOP_P(op) && I2C_OP_WRITE_P(op)) {
551 if (flags & I2C_F_POLL) {
552 /* wait for bus to be idle */
553 for (retries = 100; retries > 0; retries--) {
554 st = dwiic_read(sc, DW_IC_STATUS);
555 if (!(st & DW_IC_STATUS_ACTIVITY))
556 break;
557 DELAY(1000);
558 }
559 if (st & DW_IC_STATUS_ACTIVITY)
560 device_printf(sc->sc_dev, "timed out waiting "
561 "for bus idle\n");
562 } else {
563 mutex_enter(&sc->sc_int_lock);
564 dwiic_read(sc, DW_IC_CLR_INTR);
565 dwiic_write(sc, DW_IC_INTR_MASK,
566 DW_IC_INTR_STOP_DET);
567 if (cv_timedwait(&sc->sc_int_stopwait,
568 &sc->sc_int_lock, hz / 2) != 0)
569 device_printf(sc->sc_dev, "timed out waiting "
570 "for stop intr\n");
571 dwiic_write(sc, DW_IC_INTR_MASK, 0);
572 dwiic_read(sc, DW_IC_CLR_INTR);
573 mutex_exit(&sc->sc_int_lock);
574 }
575 }
576
577 return 0;
578}

580static uint32_t
581dwiic_read_clear_intrbits(struct dwiic_softc *sc)
582{
583 uint32_t stat;
584
585 stat = dwiic_read(sc, DW_IC_INTR_STAT);
586
587 if (stat & DW_IC_INTR_RX_UNDER)
588 dwiic_read(sc, DW_IC_CLR_RX_UNDER);
589 if (stat & DW_IC_INTR_RX_OVER)
590 dwiic_read(sc, DW_IC_CLR_RX_OVER);
591 if (stat & DW_IC_INTR_TX_OVER)
592 dwiic_read(sc, DW_IC_CLR_TX_OVER);
593 if (stat & DW_IC_INTR_RD_REQ)
594 dwiic_read(sc, DW_IC_CLR_RD_REQ);
595 if (stat & DW_IC_INTR_TX_ABRT)
596 dwiic_read(sc, DW_IC_CLR_TX_ABRT);
597 if (stat & DW_IC_INTR_RX_DONE)
598 dwiic_read(sc, DW_IC_CLR_RX_DONE);
599 if (stat & DW_IC_INTR_ACTIVITY)
600 dwiic_read(sc, DW_IC_CLR_ACTIVITY);
601 if (stat & DW_IC_INTR_STOP_DET)
602 dwiic_read(sc, DW_IC_CLR_STOP_DET);
603 if (stat & DW_IC_INTR_START_DET)
604 dwiic_read(sc, DW_IC_CLR_START_DET);
605 if (stat & DW_IC_INTR_GEN_CALL)
606 dwiic_read(sc, DW_IC_CLR_GEN_CALL);
607
608 return stat;
609}

611int
612dwiic_intr(void *arg)
613{
614 struct dwiic_softc *sc = arg;
615 uint32_t en, stat;
616
617 en = dwiic_read(sc, DW_IC_ENABLE);
618 /* probably for the other controller */
619 if (!en)
620 return 0;
621
622 stat = dwiic_read_clear_intrbits(sc);
623 DPRINTF(("%s: %s: enabled=0x%x stat=0x%x\n", device_xname(sc->sc_dev),
624 __func__, en, stat));
625 if (!(stat & ~DW_IC_INTR_ACTIVITY))
626 return 1;
627
628 if (stat & DW_IC_INTR_TX_ABRT)
629 sc->sc_i2c_xfer.error = 1;
630
631 if (sc->sc_i2c_xfer.flags & I2C_F_POLL)
632 DPRINTF(("%s: %s: intr in poll mode?\n", device_xname(sc->sc_dev),
633 __func__));
634 else {
635 mutex_enter(&sc->sc_int_lock);
636 if (stat & DW_IC_INTR_RX_FULL) {
637 dwiic_write(sc, DW_IC_INTR_MASK, 0);
638 DPRINTF(("%s: %s: waking up reader\n",
639 device_xname(sc->sc_dev), __func__));
640 cv_signal(&sc->sc_int_readwait);
641 }
642 if (stat & DW_IC_INTR_TX_EMPTY) {
643 dwiic_write(sc, DW_IC_INTR_MASK, 0);
644 DPRINTF(("%s: %s: waking up writer\n",
645 device_xname(sc->sc_dev), __func__));
646 cv_signal(&sc->sc_int_writewait);
647 }
648 if (stat & DW_IC_INTR_STOP_DET) {
649 dwiic_write(sc, DW_IC_INTR_MASK, 0);
650 DPRINTF(("%s: %s: waking up stopper\n",
651 device_xname(sc->sc_dev), __func__));
652 cv_signal(&sc->sc_int_stopwait);
653 }
654 mutex_exit(&sc->sc_int_lock);
655 }
656
657 return 1;
658}
