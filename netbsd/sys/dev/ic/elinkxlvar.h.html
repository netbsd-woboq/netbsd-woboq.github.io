32#include <sys/rndsource.h>
33
34/*
35 * Ethernet software status per interface.
36 */
37struct ex_softc {
38 device_t sc_dev;
39 device_suspensor_t sc_suspensor;
40 pmf_qual_t sc_qual;
41 void *sc_ih;
42
43 struct ethercom sc_ethercom; /* Ethernet common part */
44 bus_space_tag_t sc_iot; /* bus cookie */
45 bus_space_handle_t sc_ioh; /* bus i/o handle */
46 bus_dma_tag_t sc_dmat; /* bus dma tag */
47 bus_dmamap_t sc_dpd_dmamap;
48 bus_dmamap_t sc_upd_dmamap;
49#define sc_upddma sc_upd_dmamap->dm_segs[0].ds_addr
50#define sc_dpddma sc_dpd_dmamap->dm_segs[0].ds_addr
51 struct ex_upd *sc_upd;
52 struct ex_dpd *sc_dpd;
53 bus_dmamap_t sc_tx_dmamaps[EX_NDPD]; /* DMA maps for DPDs */
54 bus_dmamap_t sc_rx_dmamaps[EX_NUPD]; /* DMA maps for UPDs */
55 struct ex_rxdesc sc_rxdescs[EX_NUPD];
56 struct ex_txdesc sc_txdescs[EX_NDPD];
57
58 struct ex_rxdesc *rx_head;
59 struct ex_rxdesc *rx_tail;
60
61 struct ex_txdesc *tx_head;
62 struct ex_txdesc *tx_tail;
63 struct ex_txdesc *tx_free;
64 struct ex_txdesc *tx_ftail;
65
66 int tx_start_thresh; /* Current TX_start_thresh. */
67 int tx_succ_ok; /* # packets sent in sequence */
68
69 u_int ex_connectors; /* Connectors on this card. */
70 mii_data_t ex_mii; /* mii bus data */
71 struct callout ex_mii_callout; /* mii callout */
72 u_int ex_conf; /* config flags */
73
74#define EX_CONF_MII 0x0001 /* has MII bus */
75#define EX_CONF_INTPHY 0x0002 /* has internal PHY */
76#define EX_CONF_90XB 0x0004 /* is 90xB */
77#define EX_CONF_INV_LED_POLARITY 0x0010 /* CardBus & MiniPCI: LED polarity */
78#define EX_CONF_PHY_POWER 0x0020 /* CardBus & MiniPCI: PHY power */
79#define EX_CONF_EEPROM_OFF 0x0040 /* EEPROM is offset by 0x30 */
80#define EX_CONF_EEPROM_8BIT 0x0080 /* 8 bit EEPROM */
81#define EX_CONF_PCI_FUNCREG 0x0100 /* Has PCI function registers */
82#define EX_CONF_RESETHACK 0x0200 /* Hack to make reset work on 556B */
83#define EX_CONF_NO_XCVR_PWR 0x0400 /* Hack to enable later 556B xcvr */
84
85#define EX_XCVR_PWR_MAGICBITS 0x0900 /* NO_XCVR_PWR magic value */
86
87
88 /*
89 * XXX code duplication from elink3var.h
90 */
91 u_int ex_flags; /* capabilities flag (from EEPROM) */
92#define EX_FLAGS_PNP 0x0001
93#define EX_FLAGS_FULLDUPLEX 0x0002
94#define EX_FLAGS_LARGEPKT 0x0004 /* 4k packet support */
95#define EX_FLAGS_SLAVEDMA 0x0008
96#define EX_FLAGS_SECONDDMA 0x0010
97#define EX_FLAGS_FULLDMA 0x0020
98#define EX_FLAGS_FRAGMENTDMA 0x0040
99#define EX_FLAGS_CRC_PASSTHRU 0x0080
100#define EX_FLAGS_TXDONE 0x0100
101#define EX_FLAGS_NO_TXLENGTH 0x0200
102#define EX_FLAGS_RXREPEAT 0x0400
103#define EX_FLAGS_SNOOPING 0x0800
104#define EX_FLAGS_100MBIT 0x1000
105#define EX_FLAGS_POWERMGMT 0x2000
106#define EX_FLAGS_ATTACHED 0x4000 /* attach has succeeded */
107
108 krndsource_t rnd_source;
109
110 /* power management hooks */
111 int (*enable)(struct ex_softc *);
112 void (*disable)(struct ex_softc *);
113 int enabled;
114
115 /* interrupt acknowledge hook */
116 void (*intr_ack)(struct ex_softc *);
117
118 bus_dma_segment_t sc_useg, sc_dseg;
119 int sc_urseg, sc_drseg;
120
121 short sc_if_flags;
122};

124#define ex_waitcmd(sc) \
125 do { \
126 int stat; \
127 do { \
128 stat = bus_space_read_2((sc)->sc_iot, (sc)->sc_ioh, \
129 ELINK_STATUS); \
130 } while ((stat & COMMAND_IN_PROGRESS) && (stat != 0xffff)); \
131 } while (0)\

133u_int16_t exreadeeprom(bus_space_tag_t, bus_space_handle_t, int);
134void ex_config(struct ex_softc *);

137void ex_stop(struct ifnet *, int);
138void ex_watchdog(struct ifnet *);
139int ex_ioctl(struct ifnet *ifp, u_long, void *);
140int ex_activate(device_t, enum devact);

142
