34/*
35 * Device driver for Fujitsu MB86960A/MB86965A based Ethernet cards.
36 * Contributed by M.S. <seki@sysrap.cs.fujitsu.co.jp>
37 *
38 * This version is intended to be a generic template for various
39 * MB86960A/MB86965A based Ethernet cards. It currently supports
40 * Fujitsu FMV-180 series (i.e., FMV-181 and FMV-182) and Allied-
41 * Telesis AT1700 series and RE2000 series. There are some
42 * unnecessary hooks embedded, which are primarily intended to support
43 * other types of Ethernet cards, but the author is not sure whether
44 * they are useful.
45 */
46
47#include <sys/rndsource.h>
48
49/*
50 * Default settings for fe driver specific options.
51 * They can be set in config file by "options" statements.
52 */
53
54/*
55 * Debug control.
56 * 0: No debug at all. All debug specific codes are stripped off.
57 * 1: Silent. No debug messages are logged except emergent ones.
58 * 2: Brief. Lair events and/or important information are logged.
59 * 3: Detailed. Logs all information which *may* be useful for debugging.
60 * 4: Trace. All actions in the driver is logged. Super verbose.
61 */
62#ifndef FE_DEBUG
63#define FE_DEBUG 1
64#endif
65
66/*
67 * Delay padding of short transmission packets to minimum Ethernet size.
68 * This may or may not gain performance. An EXPERIMENTAL option.
69 */
70#ifndef FE_DELAYED_PADDING
71#define FE_DELAYED_PADDING 0
72#endif
73
74/*
75 * Transmit just one packet per a "send" command to 86960.
76 * This option is intended for performance test. An EXPERIMENTAL option.
77 */
78#ifndef FE_SINGLE_TRANSMISSION
79#define FE_SINGLE_TRANSMISSION 0
80#endif
81

109 FE_TYPE_AT1700T,

120/*
121 * fe_softc: per line info and status
122 */
123struct mb86960_softc {
124 device_t sc_dev;
125 struct ethercom sc_ec; /* ethernet common */
126 struct ifmedia sc_media; /* supported media information */
127
128 bus_space_tag_t sc_bst; /* bus space */
129 bus_space_handle_t sc_bsh;
130
131 /* Set by probe() and not modified in later phases. */
132 uint32_t sc_flags; /* controller quirks */
133#define FE_FLAGS_MB86960 0x0001 /* DLCR7 is differnt on MB86960 */
134#define FE_FLAGS_SBW_BYTE 0x0002 /* byte access mode for system bus */
135#define FE_FLAGS_SRAM_150ns 0x0004 /* The board has slow SRAM */
136
137 uint8_t proto_dlcr4; /* DLCR4 prototype. */
138 uint8_t proto_dlcr5; /* DLCR5 prototype. */
139 uint8_t proto_dlcr6; /* DLCR6 prototype. */
140 uint8_t proto_dlcr7; /* DLCR7 prototype. */
141 uint8_t proto_bmpr13; /* BMPR13 prototype. */
142
143 /* Vendor specific hooks. */
144 void (*init_card)(struct mb86960_softc *);
145 void (*stop_card)(struct mb86960_softc *);
146
147 /* Transmission buffer management. */
148 int txb_size; /* total bytes in TX buffer */
149 int txb_free; /* free bytes in TX buffer */
150 int txb_count; /* number of packets in TX buffer */
151 int txb_sched; /* number of scheduled packets */
152 int txb_padding; /* number of delayed padding bytes */
153
154 /* Multicast address filter management. */
155 int filter_change; /* MARs must be changed ASAP. */
156 uint8_t filter[FE_FILTER_LEN]; /* new filter value. */
157
158 uint8_t sc_enaddr[ETHER_ADDR_LEN];
159
160 krndsource_t rnd_source;
161
162 uint32_t sc_stat; /* driver status */
163#define FE_STAT_ENABLED 0x0001 /* power enabled on interface */
164#define FE_STAT_ATTACHED 0x0002 /* attach has succeeded */
165
166 int (*sc_enable)(struct mb86960_softc *);
167 void (*sc_disable)(struct mb86960_softc *);
168
169 int (*sc_mediachange)(struct mb86960_softc *);
170 void (*sc_mediastatus)(struct mb86960_softc *, struct ifmediareq *);
171};

173/*
174 * Fe driver specific constants which relate to 86960/86965.
175 */
176
177/* Interrupt masks. */
178#define FE_TMASK (FE_D2_COLL16 | FE_D2_TXDONE)
179#define FE_RMASK (FE_D3_OVRFLO | FE_D3_CRCERR | \
180 FE_D3_ALGERR | FE_D3_SRTPKT | FE_D3_PKTRDY)
181
182/* Maximum number of iterrations for a receive interrupt. */
183#define FE_MAX_RECV_COUNT ((65536 - 2048 * 2) / 64)
184 /* Maximum size of SRAM is 65536,
185 * minimum size of transmission buffer in fe is 2x2KB,
186 * and minimum amount of received packet including headers
187 * added by the chip is 64 bytes.
188 * Hence FE_MAX_RECV_COUNT is the upper limit for number
189 * of packets in the receive buffer. */
