36#define NCR_TCL 0x00 /* RW - Transfer Count Low */
37#define NCR_TCM 0x01 /* RW - Transfer Count Mid */
38#define NCR_TCH 0x0e /* RW - Transfer Count High */
39 /* NOT on 53C90 */
40
41#define NCR_FIFO 0x02 /* RW - FIFO data */
42
43#define NCR_CMD 0x03 /* RW - Command (2 deep) */
44#define NCRCMD_DMA 0x80 /* DMA Bit */
45#define NCRCMD_NOP 0x00 /* No Operation */
46#define NCRCMD_FLUSH 0x01 /* Flush FIFO */
47#define NCRCMD_RSTCHIP 0x02 /* Reset Chip */
48#define NCRCMD_RSTSCSI 0x03 /* Reset SCSI Bus */
49#define NCRCMD_RESEL 0x40 /* Reselect Sequence */
50#define NCRCMD_SELNATN 0x41 /* Select without ATN */
51#define NCRCMD_SELATN 0x42 /* Select with ATN */
52#define NCRCMD_SELATNS 0x43 /* Select with ATN & Stop */
53#define NCRCMD_ENSEL 0x44 /* Enable (Re)Selection */
54#define NCRCMD_DISSEL 0x45 /* Disable (Re)Selection */
55#define NCRCMD_SELATN3 0x46 /* Select with ATN3 */
56#define NCRCMD_RESEL3 0x47 /* Reselect3 Sequence */
57#define NCRCMD_SNDMSG 0x20 /* Send Message */

59#define NCRCMD_SNDDATA 0x22 /* Send Data */
60#define NCRCMD_DISCSEQ 0x23 /* Disconnect Sequence */
61#define NCRCMD_TERMSEQ 0x24 /* Terminate Sequence */

63#define NCRCMD_DISC 0x27 /* Disconnect */
64#define NCRCMD_RECMSG 0x28 /* Receive Message */

74#define NCRCMD_RSTATN 0x1b /* Reset ATN */

76#define NCR_STAT 0x04 /* RO - Status */

79#define NCRSTAT_PE 0x20 /* Parity Error */
80#define NCRSTAT_TC 0x10 /* Terminal Count */

84#define NCR_SELID 0x04 /* WO - Select/Reselect Bus ID */

88#define NCR_INTR 0x05 /* RO - Interrupt */
89#define NCRINTR_SBR 0x80 /* SCSI Bus Reset */
90#define NCRINTR_ILL 0x40 /* Illegal Command */
91#define NCRINTR_DIS 0x20 /* Disconnect */
92#define NCRINTR_BS 0x10 /* Bus Service */
93#define NCRINTR_FC 0x08 /* Function Complete */
94#define NCRINTR_RESEL 0x04 /* Reselected */
95#define NCRINTR_SELATN 0x02 /* Select with ATN */
96#define NCRINTR_SEL 0x01 /* Selected */

98#define NCR_TIMEOUT 0x05 /* WO - Select/Reselect Timeout */
99
100#define NCR_STEP 0x06 /* RO - Sequence Step */
101#define NCRSTEP_MASK 0x07 /* the last 3 bits */
102#define NCRSTEP_DONE 0x04 /* command went out */
103
104#define NCR_SYNCTP 0x06 /* WO - Synch Transfer Period */
105 /* Default 5 (53C9X) */
106
107#define NCR_FFLAG 0x07 /* RO - FIFO Flags */
108#define NCRFIFO_SS 0xe0 /* Sequence Step (Dup) */
109#define NCRFIFO_FF 0x1f /* Bytes in FIFO */
110
111#define NCR_SYNCOFF 0x07 /* WO - Synch Offset */
112 /* 0 = ASYNC */
113 /* 1 - 15 = SYNC bytes */
114
115#define NCR_CFG1 0x08 /* RW - Configuration #1 */
116#define NCRCFG1_SLOW 0x80 /* Slow Cable Mode */
117#define NCRCFG1_SRR 0x40 /* SCSI Reset Rep Int Dis */
118#define NCRCFG1_PTEST 0x20 /* Parity Test Mod */
119#define NCRCFG1_PARENB 0x10 /* Enable Parity Check */
120#define NCRCFG1_CTEST 0x08 /* Enable Chip Test */
121#define NCRCFG1_BUSID 0x07 /* Bus ID */
122
123#define NCR_CCF 0x09 /* WO - Clock Conversion Factor */
124 /* 0 = 35.01 - 40MHz */
125 /* NEVER SET TO 1 */
126 /* 2 = 10MHz */
127 /* 3 = 10.01 - 15MHz */
128 /* 4 = 15.01 - 20MHz */
129 /* 5 = 20.01 - 25MHz */
130 /* 6 = 25.01 - 30MHz */
131 /* 7 = 30.01 - 35MHz */
132
133#define NCR_TEST 0x0a /* WO - Test (Chip Test Only) */
134
135#define NCR_CFG2 0x0b /* RW - Configuration #2 */
136#define NCRCFG2_RSVD 0xa0 /* reserved */
137#define NCRCFG2_FE 0x40 /* Features Enable */
138#define NCRCFG2_DREQ 0x10 /* DREQ High Impedance */
139#define NCRCFG2_SCSI2 0x08 /* SCSI-2 Enable */
140#define NCRCFG2_BPA 0x04 /* Target Bad Parity Abort */
141#define NCRCFG2_RPE 0x02 /* Register Parity Error */
142#define NCRCFG2_DPE 0x01 /* DMA Parity Error */
143
144#define NCRCFG2_HMEFE 0x10 /* HME feature enable */
145#define NCRCFG2_HME32 0x80 /* HME 32 extended */
146
147/* Config #3 only on 53C9X */
148#define NCR_CFG3 0x0c /* RW - Configuration #3 */
149#define NCRCFG3_RSVD 0xe0 /* reserved */
150#define NCRCFG3_IDM 0x10 /* ID Message Res Check */
151#define NCRCFG3_QTE 0x08 /* Queue Tag Enable */
152#define NCRCFG3_CDB 0x04 /* CDB 10-bytes OK */
153#define NCRCFG3_FSCSI 0x02 /* Fast SCSI */
154#define NCRCFG3_FCLK 0x01 /* Fast Clock (>25MHz) */

156/*
157 * For some unknown reason, the ESP406/FAS408 looks like every
158 * other ncr53c9x, except for configuration #3 register. At any
159 * rate, if you're dealing with these chips, you need to use these
160 * defines instead.
161 */
162
163/* Config #3 different on ESP406/FAS408 */
164#define NCR_ESPCFG3 0x0c /* RW - Configuration #3 */
165#define NCRESPCFG3_IDM 0x80 /* ID Message Res Check */
166#define NCRESPCFG3_QTE 0x40 /* Queue Tag Enable */
167#define NCRESPCFG3_CDB 0x20 /* CDB 10-bytes OK */
168#define NCRESPCFG3_FSCSI 0x10 /* Fast SCSI */
169#define NCRESPCFG3_SRESB 0x08 /* Save Residual Byte */
170#define NCRESPCFG3_FCLK 0x04 /* Fast Clock (>25MHz) */
171#define NCRESPCFG3_ADMA 0x02 /* Alternate DMA Mode */
172#define NCRESPCFG3_T8M 0x01 /* Threshold 8 Mode */
173
174/* Config #3 also different on NCR53CF9x/FAS216 */
175#define NCR_F9XCFG3 0x0c /* RW - Configuration #3 */
176#define NCRF9XCFG3_IDM 0x80 /* ID Message Res Check */
177#define NCRF9XCFG3_QTE 0x40 /* Queue Tag Enable */
178#define NCRF9XCFG3_CDB 0x20 /* CDB 10-bytes OK */
179#define NCRF9XCFG3_FSCSI 0x10 /* Fast SCSI */
180#define NCRF9XCFG3_FCLK 0x08 /* Fast Clock (>25MHz) */
181#define NCRF9XCFG3_SRESB 0x04 /* Save Residual Byte */
182#define NCRF9XCFG3_ADMA 0x02 /* Alternate DMA Mode */
183#define NCRF9XCFG3_T8M 0x01 /* Threshold 8 Mode */
184
185/* Config #3 on FAS366 */
186#define NCRFASCFG3_OBAUTO 0x80 /* auto push odd-byte to DMA */
187#define NCRFASCFG3_EWIDE 0x40 /* Enable Wide-SCSI */
188#define NCRFASCFG3_IDBIT3 0x20 /* Bit 3 of HME SCSI-ID */
189#define NCRFASCFG3_IDRESCHK 0x10 /* ID message checking */
190#define NCRFASCFG3_QUENB 0x08 /* 3-byte msg support */
191#define NCRFASCFG3_CDB10 0x04 /* group 2 scsi-2 support */
192#define NCRFASCFG3_FASTSCSI 0x02 /* 10 MB/S fast scsi mode */
193#define NCRFASCFG3_FASTCLK 0x01 /* fast clock mode */
194
195/* Config #4 only on ESP406/FAS408 */
196#define NCR_CFG4 0x0d /* RW - Configuration #4 */
197#define NCRCFG4_CRS1 0x80 /* Select register set #1 */
198#define NCRCFG4_RSVD 0x7b /* reserved */
199#define NCRCFG4_ACTNEG 0x04 /* Active negation */

201/*
202 The following registers are only on the ESP406/FAS408. The
203 documentation refers to them as "Control Register Set #1".
204 These are the registers that are visible when bit 7 of
205 register 0x0d is set. This bit is common to both register sets.
206*/

211#define NCRJMP_J4 0x10 /* Jumper #4 */
212#define NCRJMP_J3 0x08 /* Jumper #3 */
213#define NCRJMP_J2 0x04 /* Jumper #2 */

215#define NCRJMP_J0 0x01 /* Jumper #0 */

217#define NCR_PIOFIFO 0x04 /* WO - PIO FIFO, 4 bytes deep */

229#define NCR_PIOI 0x0b /* RW - PIO Interrupt Enable */

258/* Am53c974 Config #4 */
