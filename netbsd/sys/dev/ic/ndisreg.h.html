58#define NDIS_HASH_FUNCTION_MASK 0x000000FF /* see hash function */
59#define NDIS_HASH_TYPE_MASK 0x00FFFF00 /* see hash type */

61/* hash function */
62#define NDIS_HASH_FUNCTION_TOEPLITZ 0x00000001
63
64/* hash type */
65#define NDIS_HASH_IPV4 0x00000100
66#define NDIS_HASH_TCP_IPV4 0x00000200
67#define NDIS_HASH_IPV6 0x00000400
68#define NDIS_HASH_IPV6_EX 0x00000800
69#define NDIS_HASH_TCP_IPV6 0x00001000
70#define NDIS_HASH_TCP_IPV6_EX 0x00002000
71#define NDIS_HASH_UDP_IPV4_X 0x00004000 /* XXX non-standard */
72
73#define NDIS_HASH_ALL (NDIS_HASH_IPV4 | \
74 NDIS_HASH_TCP_IPV4 | \
75 NDIS_HASH_IPV6 | \
76 NDIS_HASH_IPV6_EX | \
77 NDIS_HASH_TCP_IPV6 | \
78 NDIS_HASH_TCP_IPV6_EX |\
79 NDIS_HASH_UDP_IPV4_X)
80
81#define NDIS_HASH_STD (NDIS_HASH_IPV4 | \
82 NDIS_HASH_TCP_IPV4 | \
83 NDIS_HASH_IPV6 | \
84 NDIS_HASH_IPV6_EX | \
85 NDIS_HASH_TCP_IPV6 | \
86 NDIS_HASH_TCP_IPV6_EX)
87
88/* Hash description for use with printf(9) %b identifier. */
89#define NDIS_HASH_BITS \
90 "\20\1TOEPLITZ\11IP4\12TCP4\13IP6\14IP6EX\15TCP6\16TCP6EX\17UDP4_X"

92#define NDIS_HASH_KEYSIZE_TOEPLITZ 40
93#define NDIS_HASH_INDCNT 128

95#define NDIS_OBJTYPE_DEFAULT 0x80
96#define NDIS_OBJTYPE_RSS_CAPS 0x88
97#define NDIS_OBJTYPE_RSS_PARAMS 0x89
98#define NDIS_OBJTYPE_OFFLOAD 0xa7

100struct ndis_object_hdr {
101 uint8_t ndis_type; /* NDIS_OBJTYPE_ */
102 uint8_t ndis_rev; /* type specific */
103 uint16_t ndis_size; /* incl. this hdr */
104};

107 * OID_TCP_OFFLOAD_PARAMETERS

110struct ndis_offload_params {
111 struct ndis_object_hdr ndis_hdr;
112 uint8_t ndis_ip4csum; /* NDIS_OFFLOAD_PARAM_ */

114 uint8_t ndis_udp4csum; /* NDIS_OFFLOAD_PARAM_ */

132};

175struct ndis_rss_caps {
176 struct ndis_object_hdr ndis_hdr;
177 uint32_t ndis_caps; /* NDIS_RSS_CAP_ */
178 uint32_t ndis_nmsi; /* # of MSIs */
179 uint32_t ndis_nrxr; /* # of RX rings */
180 /* NDIS >= 6.30 */
181 uint16_t ndis_nind; /* # of indtbl ent. */
182 uint16_t ndis_pad;
183};
184
185#define NDIS_RSS_CAPS_SIZE \
186 offsetof(struct ndis_rss_caps, ndis_pad)
187#define NDIS_RSS_CAPS_SIZE_6_0 \
188 offsetof(struct ndis_rss_caps, ndis_nind)
189
190#define NDIS_RSS_CAPS_REV_1 1 /* NDIS 6.{0,1,20} */
191#define NDIS_RSS_CAPS_REV_2 2 /* NDIS 6.30 */
192

203/*
204 * OID_GEN_RECEIVE_SCALE_PARAMETERS
205 * ndis_type: NDIS_OBJTYPE_RSS_PARAMS
206 */
207struct ndis_rss_params {
208 struct ndis_object_hdr ndis_hdr;
209 uint16_t ndis_flags; /* NDIS_RSS_FLAG_ */
210 uint16_t ndis_bcpu; /* base cpu 0 */
211 uint32_t ndis_hash; /* NDIS_HASH_ */
212 uint16_t ndis_indsize; /* indirect table */
213 uint32_t ndis_indoffset;
214 uint16_t ndis_keysize; /* hash key */
215 uint32_t ndis_keyoffset;
216 /* NDIS >= 6.20 */
217 uint32_t ndis_cpumaskoffset;
218 uint32_t ndis_cpumaskcnt;
219 uint32_t ndis_cpumaskentsz;
220};
221
222#define NDIS_RSS_PARAMS_SIZE sizeof(struct ndis_rss_params)
223#define NDIS_RSS_PARAMS_SIZE_6_0 \
224 offsetof(struct ndis_rss_params, ndis_cpumaskoffset)
225
226#define NDIS_RSS_PARAMS_REV_1 1 /* NDIS 6.0 */
227#define NDIS_RSS_PARAMS_REV_2 2 /* NDIS 6.20 */
228
229#define NDIS_RSS_FLAG_NONE 0x0000
230#define NDIS_RSS_FLAG_BCPU_UNCHG 0x0001
231#define NDIS_RSS_FLAG_HASH_UNCHG 0x0002
232#define NDIS_RSS_FLAG_IND_UNCHG 0x0004
233#define NDIS_RSS_FLAG_KEY_UNCHG 0x0008
234#define NDIS_RSS_FLAG_DISABLE 0x0010

371/*
372 * Per-packet-info

374
375/* VLAN */
376#define NDIS_VLAN_INFO_SIZE sizeof(uint32_t)
377#define NDIS_VLAN_INFO_PRI_MASK 0x0007
378#define NDIS_VLAN_INFO_CFI_MASK 0x0008
379#define NDIS_VLAN_INFO_ID_MASK 0xfff0
380#define NDIS_VLAN_INFO_MAKE(id, pri, cfi) \
381 (((pri) & NDIS_VLAN_INFO_PRI_MASK) | \
382 (((cfi) & 0x1) << 3) | (((id) & 0xfff) << 4))
383#define NDIS_VLAN_INFO_ID(inf) (((inf) & NDIS_VLAN_INFO_ID_MASK) >> 4)
384#define NDIS_VLAN_INFO_CFI(inf) (((inf) & NDIS_VLAN_INFO_CFI_MASK) >> 3)
385#define NDIS_VLAN_INFO_PRI(inf) ((inf) & NDIS_VLAN_INFO_PRI_MASK)
386
387/* Reception checksum */
388#define NDIS_RXCSUM_INFO_SIZE sizeof(uint32_t)
389#define NDIS_RXCSUM_INFO_TCPCS_FAILED 0x0001
390#define NDIS_RXCSUM_INFO_UDPCS_FAILED 0x0002
391#define NDIS_RXCSUM_INFO_IPCS_FAILED 0x0004
392#define NDIS_RXCSUM_INFO_TCPCS_OK 0x0008
393#define NDIS_RXCSUM_INFO_UDPCS_OK 0x0010
394#define NDIS_RXCSUM_INFO_IPCS_OK 0x0020
395#define NDIS_RXCSUM_INFO_LOOPBACK 0x0040
396#define NDIS_RXCSUM_INFO_TCPCS_INVAL 0x0080
397#define NDIS_RXCSUM_INFO_IPCS_INVAL 0x0100
398
399/* LSOv2 */
400#define NDIS_LSO2_INFO_SIZE sizeof(uint32_t)
401#define NDIS_LSO2_INFO_MSS_MASK 0x000fffff
402#define NDIS_LSO2_INFO_THOFF_MASK 0x3ff00000
403#define NDIS_LSO2_INFO_ISLSO2 0x40000000
404#define NDIS_LSO2_INFO_ISIPV6 0x80000000
405
406#define NDIS_LSO2_INFO_MAKE(thoff, mss) \
407 ((((uint32_t)(mss)) & NDIS_LSO2_INFO_MSS_MASK) | \
408 ((((uint32_t)(thoff)) & 0x3ff) << 20) | \
409 NDIS_LSO2_INFO_ISLSO2)
410
411#define NDIS_LSO2_INFO_MAKEIPV4(thoff, mss) \
412 NDIS_LSO2_INFO_MAKE((thoff), (mss))
413
414#define NDIS_LSO2_INFO_MAKEIPV6(thoff, mss) \
415 (NDIS_LSO2_INFO_MAKE((thoff), (mss)) | NDIS_LSO2_INFO_ISIPV6)
416
417/* Transmission checksum */
418#define NDIS_TXCSUM_INFO_SIZE sizeof(uint32_t)
419#define NDIS_TXCSUM_INFO_IPV4 0x00000001
420#define NDIS_TXCSUM_INFO_IPV6 0x00000002
421#define NDIS_TXCSUM_INFO_TCPCS 0x00000004
422#define NDIS_TXCSUM_INFO_UDPCS 0x00000008
423#define NDIS_TXCSUM_INFO_IPCS 0x00000010
424#define NDIS_TXCSUM_INFO_THOFF 0x03ff0000
425
426#define NDIS_TXCSUM_INFO_MKL4CS(thoff, flag) \
427 ((((uint32_t)(thoff)) << 16) | (flag))
428
429#define NDIS_TXCSUM_INFO_MKTCPCS(thoff) \
430 NDIS_TXCSUM_INFO_MKL4CS((thoff), NDIS_TXCSUM_INFO_TCPCS)
431
432#define NDIS_TXCSUM_INFO_MKUDPCS(thoff) \
433 NDIS_TXCSUM_INFO_MKL4CS((thoff), NDIS_TXCSUM_INFO_UDPCS)
