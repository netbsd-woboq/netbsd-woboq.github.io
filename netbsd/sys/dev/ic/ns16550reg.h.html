34/*
35 * NS16550 UART registers
36 */
37
38#define com_data 0 /* data register (R/W) */
39#define com_dlbl 0 /* divisor latch low (W) */
40#define com_dlbh 1 /* divisor latch high (W) */
41#define com_ier 1 /* interrupt enable (W) */
42#define com_iir 2 /* interrupt identification (R) */
43#define com_fifo 2 /* FIFO control (W) */
44#define com_lctl 3 /* line control register (R/W) */
45#define com_cfcr 3 /* line control register (R/W) */
46#define com_lcr com_cfcr
47#define com_mcr 4 /* modem control register (R/W) */
48#define com_lsr 5 /* line status register (R/W) */
49#define com_msr 6 /* modem status register (R/W) */
50#define com_scratch 7 /* scratch register (R/W) */
51
52/*
53 * Additional registers present on TI OMAP hardware
54 */
55#define com_mdr1 8 /* mode definition register 1 (OMAP) */
56
57/*
58 * Additional register present in NS16750
59 */
60#define com_usr 31 /* status register (R) (16750/SUNXI) */
61
62/*
63 * Additional registers present on Allwinner hardware
64 */
65#define com_tfl 32 /* transmit fifo level (R) (SUNXI) */
66#define com_rfl 33 /* receive fifo level (R) (SUNXI) */
67#define com_halt 41 /* halt tx (R/W) (SUNXI) */
