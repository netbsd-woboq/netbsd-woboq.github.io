32#include <dev/midivar.h>
33#include <dev/midisynvar.h>
34
35struct opl_voice {
36 int voiceno;
37 int iooffs;
38 u_int8_t op[4];
39 const struct opl_operators *patch;
40 u_int8_t rB0;
41};
42
43struct opl_softc {
44 device_t dev;
45 bus_space_tag_t iot;
46 bus_space_handle_t ioh;
47 kmutex_t *lock;
48 int offs;
49 int model;
50#define OPL_2 2
51#define OPL_3 3
52 struct midisyn syn;
53 device_t sc_mididev;
54
55 struct opl_voice voices[OPL3_NVOICE];
56 u_int8_t pan[MIDI_MAX_CHANS];
57 u_int8_t panl, panr;
58
59 int (*spkrctl)(void *, int);
60 void *spkrarg;
61
62#ifndef AUDIO_NO_POWER_CTL
63 int (*powerctl)(void *, int);
64 void *powerarg;
65#endif
66};
67
68/* for panpot */
69#define OPL_MIDI_CENTER_MIN (64 - 20)
70#define OPL_MIDI_CENTER_MAX (64 + 20)
71
72/* config flags */
73#define OPL_FLAGS_SWAP_LR 0x0001 /* swap L and R channels */
74
75struct opl_attach_arg {
76 bus_space_tag_t iot;
77 bus_space_handle_t ioh;
78 int offs;
79 int done;
80};
81
82struct opl_operators {
83 u_int8_t opl3;
84 u_int8_t ops[22];
85#define OO_CHARS 0
86#define OO_KSL_LEV 2
87#define OO_ATT_DEC 4
88#define OO_SUS_REL 6
89#define OO_WAV_SEL 8
90#define OO_FB_CONN 10
91#define OO_4OP_OFFS 11
92};
93
94#define OPL_NINSTR 256

103int opl_detach(struct opl_softc *, int);
104#endif
