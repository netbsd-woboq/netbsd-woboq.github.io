103void
104pdq_ifinit(pdq_softc_t *sc)
105{
106 if (sc->sc_if.if_flags & IFF_UP) {
107 sc->sc_if.if_flags |= IFF_RUNNING;
108 if (sc->sc_if.if_flags & IFF_PROMISC)
109 sc->sc_pdq->pdq_flags |= PDQ_PROMISC;
110 else
111 sc->sc_pdq->pdq_flags &= ~PDQ_PROMISC;
112
113 if (sc->sc_if.if_flags & IFF_LINK1)
114 sc->sc_pdq->pdq_flags |= PDQ_PASS_SMT;
115 else
116 sc->sc_pdq->pdq_flags &= ~PDQ_PASS_SMT;
117
118 sc->sc_pdq->pdq_flags |= PDQ_RUNNING;
119 pdq_run(sc->sc_pdq);
120 } else {
121 sc->sc_if.if_flags &= ~IFF_RUNNING;
122 sc->sc_pdq->pdq_flags &= ~PDQ_RUNNING;
123 pdq_stop(sc->sc_pdq);
124 }
125}

127void
128pdq_ifwatchdog(struct ifnet *ifp)
129{
130 /*
131 * No progress was made on the transmit queue for PDQ_OS_TX_TRANSMIT
132 * seconds. Remove all queued packets.
133 */
134
135 ifp->if_flags &= ~IFF_OACTIVE;
136 ifp->if_timer = 0;
137 for (;;) {
138 struct mbuf *m;
139 IFQ_DEQUEUE(&ifp->if_snd, m);
140 if (m == NULL)
141 return;
142 PDQ_OS_DATABUF_FREE(PDQ_OS_IFP_TO_SOFTC(ifp)->sc_pdq, m);
143 }
144}

146ifnet_ret_t
147pdq_ifstart(struct ifnet *ifp)
148{
149 pdq_softc_t * const sc = PDQ_OS_IFP_TO_SOFTC(ifp);
150 struct mbuf *m;
151 int tx = 0;
152
153 if ((ifp->if_flags & IFF_RUNNING) == 0)
154 return;
155
156 if (sc->sc_if.if_timer == 0)
157 sc->sc_if.if_timer = PDQ_OS_TX_TIMEOUT;
158
159 if ((sc->sc_pdq->pdq_flags & PDQ_TXOK) == 0) {
160 sc->sc_if.if_flags |= IFF_OACTIVE;
161 return;
162 }
163 sc->sc_flags |= PDQIF_DOWNCALL;
164 for (;; tx = 1) {
165 IFQ_POLL(&ifp->if_snd, m);
166 if (m == NULL)
167 break;
168#if defined(PDQ_BUS_DMA) && !defined(PDQ_BUS_DMA_NOTX)
169 if ((m->m_flags & M_HASTXDMAMAP) == 0) {
170 bus_dmamap_t map;
171 if (PDQ_OS_HDR_OFFSET != PDQ_RX_FC_OFFSET) {
172 m->m_data[0] = PDQ_FDDI_PH0;
173 m->m_data[1] = PDQ_FDDI_PH1;
174 m->m_data[2] = PDQ_FDDI_PH2;
175 }
176 if (!bus_dmamap_create(sc->sc_dmatag, m->m_pkthdr.len,
177 255, m->m_pkthdr.len, 0, BUS_DMA_NOWAIT, &map)) {
178 if (!bus_dmamap_load_mbuf(sc->sc_dmatag, map,
179 m, BUS_DMA_WRITE | BUS_DMA_NOWAIT)) {
180 bus_dmamap_sync(sc->sc_dmatag, map, 0,
181 m->m_pkthdr.len,
182 BUS_DMASYNC_PREWRITE);
183 M_SETCTX(m, map);
184 m->m_flags |= M_HASTXDMAMAP;
185 }
186 }
187 if ((m->m_flags & M_HASTXDMAMAP) == 0)
188 break;
189 }
190#else
191 if (PDQ_OS_HDR_OFFSET != PDQ_RX_FC_OFFSET) {
192 m->m_data[0] = PDQ_FDDI_PH0;
193 m->m_data[1] = PDQ_FDDI_PH1;
194 m->m_data[2] = PDQ_FDDI_PH2;
195 }
196#endif
197
198 if (pdq_queue_transmit_data(sc->sc_pdq, m) == PDQ_FALSE)
199 break;
200 IFQ_DEQUEUE(&ifp->if_snd, m);
201 }
202 if (m != NULL)
203 ifp->if_flags |= IFF_OACTIVE;
204 if (tx)
205 PDQ_DO_TYPE2_PRODUCER(sc->sc_pdq);
206 sc->sc_flags &= ~PDQIF_DOWNCALL;
207}

209void
210pdq_os_receive_pdu(pdq_t *pdq, struct mbuf *m, size_t pktlen, int drop)
211{
212 pdq_softc_t *sc = pdq->pdq_os_ctx;
213 struct fddi_header *fh;
214
215#if defined(PDQ_BUS_DMA)
216 {
217 /*
218 * Even though the first mbuf start at the first fddi header octet,
219 * the dmamap starts PDQ_OS_HDR_OFFSET octets earlier. Any additional
220 * mbufs will start normally.
221 */
222 int offset = PDQ_OS_HDR_OFFSET;
223 struct mbuf *m0;
224 for (m0 = m; m0 != NULL; m0 = m0->m_next, offset = 0) {
225 pdq_os_databuf_sync(sc, m0, offset, m0->m_len,
226 BUS_DMASYNC_POSTREAD);
227 bus_dmamap_unload(sc->sc_dmatag, M_GETCTX(m0, bus_dmamap_t));
228 bus_dmamap_destroy(sc->sc_dmatag, M_GETCTX(m0, bus_dmamap_t));
229 m0->m_flags &= ~M_HASRXDMAMAP;
230 M_SETCTX(m0, NULL);
231 }
232 }
233#endif
234 m->m_pkthdr.len = pktlen;
235 fh = mtod(m, struct fddi_header *);
236 if (drop || (fh->fddi_fc & (FDDIFC_L | FDDIFC_F))
237 != FDDIFC_LLC_ASYNC) {
238 PDQ_OS_DATABUF_FREE(pdq, m);
239 return;
240 }
241
242 m_set_rcvif(m, &sc->sc_if);
243 if_percpuq_enqueue((&sc->sc_if)->if_percpuq, m);
244}

246void
247pdq_os_restart_transmitter(pdq_t *pdq)
248{
249 pdq_softc_t *sc = pdq->pdq_os_ctx;
250 sc->sc_if.if_flags &= ~IFF_OACTIVE;
251 if (IFQ_IS_EMPTY(&sc->sc_if.if_snd) == 0) {
252 sc->sc_if.if_timer = PDQ_OS_TX_TIMEOUT;
253 if ((sc->sc_flags & PDQIF_DOWNCALL) == 0)
254 pdq_ifstart(&sc->sc_if);
255 } else
256 sc->sc_if.if_timer = 0;
257}

259void
260pdq_os_transmit_done(pdq_t *pdq, struct mbuf *m)
261{
262 pdq_softc_t *sc = pdq->pdq_os_ctx;
263 if (sc->sc_bpf != NULL)
264 PDQ_BPF_MTAP(sc, m, BPF_D_OUT);
265 PDQ_OS_DATABUF_FREE(pdq, m);
266 sc->sc_if.if_opackets++;
267}

269void
270pdq_os_addr_fill(pdq_t *pdq, pdq_lanaddr_t *addr, size_t num_addrs)
271{
272 pdq_softc_t *sc = pdq->pdq_os_ctx;
273 struct ethercom *ec = PDQ_FDDICOM(sc);
274 struct ether_multistep step;
275 struct ether_multi *enm;
276
277 /*
278 * ADDR_FILTER_SET is always issued before FILTER_SET so
279 * we can play with PDQ_ALLMULTI and not worry about
280 * queueing a FILTER_SET ourselves.
281 */
282
283 pdq->pdq_flags &= ~PDQ_ALLMULTI;
284#if defined(IFF_ALLMULTI)
285 sc->sc_if.if_flags &= ~IFF_ALLMULTI;
286#endif
287
288 ETHER_LOCK(ec);
289 ETHER_FIRST_MULTI(step, ec, enm);
290 while (enm != NULL && num_addrs > 0) {
291 if (memcmp(enm->enm_addrlo, enm->enm_addrhi, 6) == 0) {
292 ((u_short *)addr->lanaddr_bytes)[0] = ((u_short *)enm->enm_addrlo)[0];
293 ((u_short *)addr->lanaddr_bytes)[1] = ((u_short *)enm->enm_addrlo)[1];
294 ((u_short *)addr->lanaddr_bytes)[2] = ((u_short *)enm->enm_addrlo)[2];
295 addr++;
296 num_addrs--;
297 } else {
298 pdq->pdq_flags |= PDQ_ALLMULTI;
299#if defined(IFF_ALLMULTI)
300 sc->sc_if.if_flags |= IFF_ALLMULTI;
301#endif
302 }
303 ETHER_NEXT_MULTI(step, enm);
304 }
305 ETHER_UNLOCK(ec);
306 /*
307 * If not all the address fit into the CAM, turn on all-multicast mode.
308 */
309 if (enm != NULL) {
310 pdq->pdq_flags |= PDQ_ALLMULTI;
311#if defined(IFF_ALLMULTI)
312 sc->sc_if.if_flags |= IFF_ALLMULTI;
313#endif
314 }
315}

317#if defined(IFM_FDDI)
318static int
319pdq_ifmedia_change(
320 struct ifnet *ifp)
321{
322 pdq_softc_t * const sc = PDQ_OS_IFP_TO_SOFTC(ifp);
323
324 if (sc->sc_ifmedia.ifm_media & IFM_FDX) {
325 if ((sc->sc_pdq->pdq_flags & PDQ_WANT_FDX) == 0) {
326 sc->sc_pdq->pdq_flags |= PDQ_WANT_FDX;
327 if (sc->sc_pdq->pdq_flags & PDQ_RUNNING)
328 pdq_run(sc->sc_pdq);
329 }
330 } else if (sc->sc_pdq->pdq_flags & PDQ_WANT_FDX) {
331 sc->sc_pdq->pdq_flags &= ~PDQ_WANT_FDX;
332 if (sc->sc_pdq->pdq_flags & PDQ_RUNNING)
333 pdq_run(sc->sc_pdq);
334 }
335
336 return 0;
337}

339static void
340pdq_ifmedia_status(struct ifnet *ifp, struct ifmediareq *ifmr)
341{
342 pdq_softc_t * const sc = PDQ_OS_IFP_TO_SOFTC(ifp);
343
344 ifmr->ifm_status = IFM_AVALID;
345 if (sc->sc_pdq->pdq_flags & PDQ_IS_ONRING)
346 ifmr->ifm_status |= IFM_ACTIVE;
347
348 ifmr->ifm_active = (ifmr->ifm_current & ~IFM_FDX);
349 if (sc->sc_pdq->pdq_flags & PDQ_IS_FDX)
350 ifmr->ifm_active |= IFM_FDX;
351}

353void
354pdq_os_update_status(pdq_t *pdq, const void *arg)
355{
356 pdq_softc_t * const sc = pdq->pdq_os_ctx;
357 const pdq_response_status_chars_get_t *rsp = arg;
358 int media = 0;
359
360 switch (rsp->status_chars_get.pmd_type[0]) {
361 case PDQ_PMD_TYPE_ANSI_MUTLI_MODE: media = IFM_FDDI_MMF; break;
362 case PDQ_PMD_TYPE_ANSI_SINGLE_MODE_TYPE_1: media = IFM_FDDI_SMF; break;
363 case PDQ_PMD_TYPE_ANSI_SIGNLE_MODE_TYPE_2: media = IFM_FDDI_SMF; break;
364 case PDQ_PMD_TYPE_UNSHIELDED_TWISTED_PAIR: media = IFM_FDDI_UTP; break;
365 default: media |= IFM_MANUAL;
366 }
367
368 if (rsp->status_chars_get.station_type == PDQ_STATION_TYPE_DAS)
369 media |= IFM_FDDI_DA;
370
371 sc->sc_ifmedia.ifm_media = media | IFM_FDDI;
372}

375int
376pdq_ifioctl(struct ifnet *ifp, ioctl_cmd_t cmd, void *data)
377{
378 pdq_softc_t *sc = PDQ_OS_IFP_TO_SOFTC(ifp);
379 int s, error = 0;
380
381 s = PDQ_OS_SPL_RAISE();
382
383 switch (cmd) {
384 case SIOCINITIFADDR: {
385 struct ifaddr *ifa = (struct ifaddr *)data;
386
387 ifp->if_flags |= IFF_UP;
388 pdq_ifinit(sc);
389 switch (ifa->ifa_addr->sa_family) {
390#if defined(INET)
391 case AF_INET:
392 PDQ_ARP_IFINIT(sc, ifa);
393 break;
394#endif /* INET */
395 default:
396 break;
397 }
398 break;
399 }
400 case SIOCSIFFLAGS: {
401 if ((error = ifioctl_common(ifp, cmd, data)) != 0)
402 break;
403 pdq_ifinit(sc);
404 break;
405 }
406
407 case SIOCADDMULTI:
408 case SIOCDELMULTI: {
409 /*
410 * Update multicast listeners
411 */
412 if ((error = ether_ioctl(ifp, cmd, data)) == ENETRESET) {
413 if (sc->sc_if.if_flags & IFF_RUNNING)
414 pdq_run(sc->sc_pdq);
415 error = 0;
416 }
417 break;
418 }
419
420#if defined(SIOCSIFMTU)
421#if !defined(ifr_mtu)
422#define ifr_mtu ifr_metric
423#endif
424 case SIOCSIFMTU: {
425 struct ifreq *ifr = (struct ifreq *)data;
426 /*
427 * Set the interface MTU.
428 */
429 if (ifr->ifr_mtu > FDDIMTU) {
430 error = EINVAL;
431 break;
432 }
433 if ((error = ifioctl_common(ifp, cmd, data)) == ENETRESET)
434 error = 0;
435 break;
436 }
437#endif /* SIOCSIFMTU */
438
439 default: {
440 error = ether_ioctl(ifp, cmd, data);
441 break;
442 }
443 }
444
445 PDQ_OS_SPL_LOWER(s);
446 return error;
447}

450void
451pdq_ifattach(pdq_softc_t *sc, ifnet_ret_t (*ifwatchdog)(int unit))
452{
453 struct ifnet *ifp = &sc->sc_if;
454
455 ifp->if_flags = IFF_BROADCAST | IFF_SIMPLEX | IFF_MULTICAST;
456
457#if (defined(__FreeBSD__) && BSD >= 199506) || defined(__NetBSD__)
458 ifp->if_watchdog = pdq_ifwatchdog;
459#else
460 ifp->if_watchdog = ifwatchdog;
461#endif
462
463 ifp->if_ioctl = pdq_ifioctl;
464#if !defined(__NetBSD__)
465 ifp->if_output = fddi_output;
466#endif
467 ifp->if_start = pdq_ifstart;
468 IFQ_SET_READY(&ifp->if_snd);
469
470#if defined(IFM_FDDI)
471 {
472 const int media = sc->sc_ifmedia.ifm_media;
473
474 /* Initialize ifmedia structures. */
475 PDQ_FDDICOM(sc)->ec_ifmedia = &sc->sc_ifmedia;
476 ifmedia_init(&sc->sc_ifmedia, IFM_FDX,
477 pdq_ifmedia_change, pdq_ifmedia_status);
478 ifmedia_add(&sc->sc_ifmedia, media, 0, 0);
479 ifmedia_set(&sc->sc_ifmedia, media);
480 }
481#endif
482
483 if_attach(ifp);
484#if defined(__NetBSD__)
485 fddi_ifattach(ifp, (void *)&sc->sc_pdq->pdq_hwaddr);
486#else
487 fddi_ifattach(ifp);
488#endif
489}

491#if defined(PDQ_BUS_DMA)
492int
493pdq_os_memalloc_contig(
494 pdq_t *pdq)
495{
496 pdq_softc_t * const sc = pdq->pdq_os_ctx;
497 bus_dma_segment_t db_segs[1], ui_segs[1], cb_segs[1];
498 int db_nsegs = 0, ui_nsegs = 0;
499 int steps = 0;
500 int not_ok;
501
502 not_ok = bus_dmamem_alloc(sc->sc_dmatag,
503 sizeof(*pdq->pdq_dbp), sizeof(*pdq->pdq_dbp),
504 sizeof(*pdq->pdq_dbp), db_segs, 1, &db_nsegs,
505#if defined(__sparc__) || defined(__sparc64__)
506 BUS_DMA_NOWAIT | BUS_DMA_COHERENT);
507#else
508 BUS_DMA_NOWAIT);
509#endif
510 if (!not_ok) {
511 steps = 1;
512 not_ok = bus_dmamem_map(sc->sc_dmatag, db_segs, db_nsegs,
513 sizeof(*pdq->pdq_dbp), (void **) &pdq->pdq_dbp,
514 BUS_DMA_NOWAIT);
515 }
516 if (!not_ok) {
517 steps = 2;
518 not_ok = bus_dmamap_create(sc->sc_dmatag, db_segs[0].ds_len, 1,
519 0x2000, 0, BUS_DMA_NOWAIT, &sc->sc_dbmap);
520 }
521 if (!not_ok) {
522 steps = 3;
523 not_ok = bus_dmamap_load(sc->sc_dmatag, sc->sc_dbmap,
524 pdq->pdq_dbp, sizeof(*pdq->pdq_dbp),
525 NULL, BUS_DMA_NOWAIT);
526 }
527 if (!not_ok) {
528 steps = 4;
529 pdq->pdq_pa_descriptor_block = sc->sc_dbmap->dm_segs[0].ds_addr;
530 not_ok = bus_dmamem_alloc(sc->sc_dmatag,
531 PDQ_OS_PAGESIZE, PDQ_OS_PAGESIZE, PDQ_OS_PAGESIZE,
532 ui_segs, 1, &ui_nsegs, BUS_DMA_NOWAIT);
533 }
534 if (!not_ok) {
535 steps = 5;
536 not_ok = bus_dmamem_map(sc->sc_dmatag, ui_segs, ui_nsegs,
537 PDQ_OS_PAGESIZE,
538 (void **) &pdq->pdq_unsolicited_info.ui_events,
539 BUS_DMA_NOWAIT);
540 }
541 if (!not_ok) {
542 steps = 6;
543 not_ok = bus_dmamap_create(sc->sc_dmatag, ui_segs[0].ds_len, 1,
544 PDQ_OS_PAGESIZE, 0, BUS_DMA_NOWAIT, &sc->sc_uimap);
545 }
546 if (!not_ok) {
547 steps = 7;
548 not_ok = bus_dmamap_load(sc->sc_dmatag, sc->sc_uimap,
549 pdq->pdq_unsolicited_info.ui_events,
550 PDQ_OS_PAGESIZE, NULL, BUS_DMA_NOWAIT);
551 }
552 if (!not_ok) {
553 steps = 8;
554 pdq->pdq_unsolicited_info.ui_pa_bufstart
555 = sc->sc_uimap->dm_segs[0].ds_addr;
556 cb_segs[0] = db_segs[0];
557 cb_segs[0].ds_addr
558 += offsetof(pdq_descriptor_block_t, pdqdb_consumer);
559 cb_segs[0].ds_len = sizeof(pdq_consumer_block_t);
560#if defined(__sparc__) || defined(__sparc64__)
561 pdq->pdq_cbp
562 = (pdq_consumer_block_t*)((unsigned long int)pdq->pdq_dbp +
563 (unsigned long int)offsetof(pdq_descriptor_block_t,
564 pdqdb_consumer));
565#else
566 not_ok = bus_dmamem_map(sc->sc_dmatag, cb_segs, 1,
567 sizeof(*pdq->pdq_cbp), (void **)&pdq->pdq_cbp,
568 BUS_DMA_NOWAIT | BUS_DMA_COHERENT);
569#endif
570 }
571 if (!not_ok) {
572 steps = 9;
573 not_ok = bus_dmamap_create(sc->sc_dmatag, cb_segs[0].ds_len, 1,
574 0x2000, 0, BUS_DMA_NOWAIT, &sc->sc_cbmap);
575 }
576 if (!not_ok) {
577 steps = 10;
578 not_ok = bus_dmamap_load(sc->sc_dmatag, sc->sc_cbmap,
579 pdq->pdq_cbp, sizeof(*pdq->pdq_cbp),
580 NULL, BUS_DMA_NOWAIT);
581 }
582 if (!not_ok) {
583 pdq->pdq_pa_consumer_block = sc->sc_cbmap->dm_segs[0].ds_addr;
584 return not_ok;
585 }
586
587 switch (steps) {
588 case 11:
589 bus_dmamap_unload(sc->sc_dmatag, sc->sc_cbmap);
590 /* FALLTHROUGH */
591 case 10:
592 bus_dmamap_destroy(sc->sc_dmatag, sc->sc_cbmap);
593 /* FALLTHROUGH */
594 case 9:
595 bus_dmamem_unmap(sc->sc_dmatag,
596 (void *)pdq->pdq_cbp, sizeof(*pdq->pdq_cbp));
597 /* FALLTHROUGH */
598 case 8:
599 bus_dmamap_unload(sc->sc_dmatag, sc->sc_uimap);
600 /* FALLTHROUGH */
601 case 7:
602 bus_dmamap_destroy(sc->sc_dmatag, sc->sc_uimap);
603 /* FALLTHROUGH */
604 case 6:
605 bus_dmamem_unmap(sc->sc_dmatag,
606 (void *) pdq->pdq_unsolicited_info.ui_events,
607 PDQ_OS_PAGESIZE);
608 /* FALLTHROUGH */
609 case 5:
610 bus_dmamem_free(sc->sc_dmatag, ui_segs, ui_nsegs);
611 /* FALLTHROUGH */
612 case 4:
613 bus_dmamap_unload(sc->sc_dmatag, sc->sc_dbmap);
614 /* FALLTHROUGH */
615 case 3:
616 bus_dmamap_destroy(sc->sc_dmatag, sc->sc_dbmap);
617 /* FALLTHROUGH */
618 case 2:
619 bus_dmamem_unmap(sc->sc_dmatag,
620 (void *) pdq->pdq_dbp,
621 sizeof(*pdq->pdq_dbp));
622 /* FALLTHROUGH */
623 case 1:
624 bus_dmamem_free(sc->sc_dmatag, db_segs, db_nsegs);
625 }
626
627 return not_ok;
628}

675extern struct mbuf *
676pdq_os_databuf_alloc(pdq_os_ctx_t *sc)
677{
678 struct mbuf *m;
679 bus_dmamap_t map;
680
681 MGETHDR(m, M_DONTWAIT, MT_DATA);
682 if (m == NULL) {
683 aprint_error_dev(sc->sc_dev, "can't alloc small buf\n");
684 return NULL;
685 }
686 MCLGET(m, M_DONTWAIT);
687 if ((m->m_flags & M_EXT) == 0) {
688 aprint_error_dev(sc->sc_dev, "can't alloc cluster\n");
689 m_free(m);
690 return NULL;
691 }
692 MCLAIM(m, &PDQ_FDDICOM(sc)->ec_rx_mowner);
693 m->m_pkthdr.len = m->m_len = PDQ_OS_DATABUF_SIZE;
694
695 if (bus_dmamap_create(sc->sc_dmatag, PDQ_OS_DATABUF_SIZE,
696 1, PDQ_OS_DATABUF_SIZE, 0, BUS_DMA_NOWAIT, &map)) {
697 aprint_error_dev(sc->sc_dev, "can't create dmamap\n");
698 m_free(m);
699 return NULL;
700 }
701 if (bus_dmamap_load_mbuf(sc->sc_dmatag, map, m,
702 BUS_DMA_READ | BUS_DMA_NOWAIT)) {
703 aprint_error_dev(sc->sc_dev, "can't load dmamap\n");
704 bus_dmamap_destroy(sc->sc_dmatag, map);
705 m_free(m);
706 return NULL;
707 }
708 m->m_flags |= M_HASRXDMAMAP;
709 M_SETCTX(m, map);
710 return m;
711}
