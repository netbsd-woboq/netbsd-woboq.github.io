174/*
175 * Back-end attach and configure. Allocate DMA space and initialize
176 * all structures.
177 */
178
179void
180eshconfig(struct esh_softc *sc)
181{
182 struct ifnet *ifp = &sc->sc_if;
183 bus_space_tag_t iot = sc->sc_iot;
184 bus_space_handle_t ioh = sc->sc_ioh;
185 u_int32_t misc_host_ctl;
186 u_int32_t misc_local_ctl;
187 u_int32_t header_format;
188 u_int32_t ula_tmp;
189 bus_size_t size;
190 int rseg;
191 int error;
192 int i;
193
194 esh_softc_debug[device_unit(sc->sc_dev)] = sc;
195 sc->sc_flags = 0;
196
197 TAILQ_INIT(&sc->sc_dmainfo_freelist);
198 sc->sc_dmainfo_freelist_count = 0;
199
200 /*
201 * Allocate and divvy up some host side memory that can hold
202 * data structures that will be DMA'ed over to the NIC
203 */
204
205 sc->sc_dma_size = sizeof(struct rr_gen_info) +
206 sizeof(struct rr_ring_ctl) * RR_ULP_COUNT +
207 sizeof(struct rr_descr) * RR_SEND_RING_SIZE +
208 sizeof(struct rr_descr) * RR_SNAP_RECV_RING_SIZE +
209 sizeof(struct rr_event) * RR_EVENT_RING_SIZE;
210
211 error = bus_dmamem_alloc(sc->sc_dmat, sc->sc_dma_size,
212 0, RR_DMA_BOUNDARY, &sc->sc_dmaseg, 1,
213 &rseg, BUS_DMA_NOWAIT);
214 if (error) {
215 aprint_error_dev(sc->sc_dev, "couldn't allocate space for host-side"
216 "data structures\n");
217 return;
218 }
219 if (rseg > 1) {
220 aprint_error_dev(sc->sc_dev, "contiguous memory not available\n");
221 goto bad_dmamem_map;
222 }
223
224 error = bus_dmamem_map(sc->sc_dmat, &sc->sc_dmaseg, rseg,
225 sc->sc_dma_size, (void **)&sc->sc_dma_addr,
226 BUS_DMA_NOWAIT | BUS_DMA_COHERENT);
227 if (error) {
228 aprint_error_dev(sc->sc_dev,
229 "couldn't map memory for host-side structures\n");
230 goto bad_dmamem_map;
231 }
232
233 if (bus_dmamap_create(sc->sc_dmat, sc->sc_dma_size,
234 1, sc->sc_dma_size, RR_DMA_BOUNDARY,
235 BUS_DMA_ALLOCNOW | BUS_DMA_NOWAIT,
236 &sc->sc_dma)) {
237 aprint_error_dev(sc->sc_dev, "couldn't create DMA map\n");
238 goto bad_dmamap_create;
239 }
240
241 if (bus_dmamap_load(sc->sc_dmat, sc->sc_dma, sc->sc_dma_addr,
242 sc->sc_dma_size, NULL, BUS_DMA_NOWAIT)) {
243 aprint_error_dev(sc->sc_dev, "couldn't load DMA map\n");
244 goto bad_dmamap_load;
245 }
246
247 memset(sc->sc_dma_addr, 0, sc->sc_dma_size);
248
249 sc->sc_gen_info_dma = sc->sc_dma->dm_segs->ds_addr;
250 sc->sc_gen_info = (struct rr_gen_info *) sc->sc_dma_addr;
251 size = sizeof(struct rr_gen_info);
252
253 sc->sc_recv_ring_table_dma = sc->sc_dma->dm_segs->ds_addr + size;
254 sc->sc_recv_ring_table =
255 (struct rr_ring_ctl *) (sc->sc_dma_addr + size);
256 size += sizeof(struct rr_ring_ctl) * RR_ULP_COUNT;
257
258 sc->sc_send_ring_dma = sc->sc_dma->dm_segs->ds_addr + size;
259 sc->sc_send_ring = (struct rr_descr *) (sc->sc_dma_addr + size);
260 sc->sc2_send_ring = (struct rr2_descr *) (sc->sc_dma_addr + size);
261 size += sizeof(struct rr_descr) * RR_SEND_RING_SIZE;
262
263 sc->sc_snap_recv_ring_dma = sc->sc_dma->dm_segs->ds_addr + size;
264 sc->sc_snap_recv_ring = (struct rr_descr *) (sc->sc_dma_addr + size);
265 sc->sc2_snap_recv_ring = (struct rr2_descr *) (sc->sc_dma_addr + size);
266 size += sizeof(struct rr_descr) * RR_SNAP_RECV_RING_SIZE;
267
268 sc->sc_event_ring_dma = sc->sc_dma->dm_segs->ds_addr + size;
269 sc->sc_event_ring = (struct rr_event *) (sc->sc_dma_addr + size);
270 size += sizeof(struct rr_event) * RR_EVENT_RING_SIZE;
271
272#ifdef DIAGNOSTIC
273 if (size > sc->sc_dmaseg.ds_len) {
274 aprint_error_dev(sc->sc_dev, "bogus size calculation\n");
275 goto bad_other;
276 }
277#endif
278
279 /*
280 * Allocate DMA maps for transfers. We do this here and now
281 * so we won't have to wait for them in the middle of sending
282 * or receiving something.
283 */
284
285 if (bus_dmamap_create(sc->sc_dmat, ESH_MAX_NSEGS * RR_DMA_MAX,
286 ESH_MAX_NSEGS, RR_DMA_MAX, RR_DMA_BOUNDARY,
287 BUS_DMA_ALLOCNOW | BUS_DMA_NOWAIT,
288 &sc->sc_send.ec_dma)) {
289 aprint_error_dev(sc->sc_dev, "failed bus_dmamap_create\n");
290 goto bad_other;
291 }
292 sc->sc_send.ec_offset = 0;
293 sc->sc_send.ec_descr = sc->sc_send_ring;
294 TAILQ_INIT(&sc->sc_send.ec_di_queue);
295 bufq_alloc(&sc->sc_send.ec_buf_queue, "fcfs", 0);
296
297 for (i = 0; i < RR_MAX_SNAP_RECV_RING_SIZE; i++)
298 if (bus_dmamap_create(sc->sc_dmat, RR_DMA_MAX, 1, RR_DMA_MAX,
299 RR_DMA_BOUNDARY,
300 BUS_DMA_ALLOCNOW | BUS_DMA_NOWAIT,
301 &sc->sc_snap_recv.ec_dma[i])) {
302 aprint_error_dev(sc->sc_dev, "failed bus_dmamap_create\n");
303 for (i--; i >= 0; i--)
304 bus_dmamap_destroy(sc->sc_dmat,
305 sc->sc_snap_recv.ec_dma[i]);
306 goto bad_ring_dmamap_create;
307 }
308
309 /*
310 * If this is a coldboot, the NIC RunCode should be operational.
311 * If it is a warmboot, it may or may not be operational.
312 * Just to be sure, we'll stop the RunCode and reset everything.
313 */
314
315 /* Halt the processor (preserve NO_SWAP, if set) */
316
317 misc_host_ctl = bus_space_read_4(iot, ioh, RR_MISC_HOST_CTL);
318 bus_space_write_4(iot, ioh, RR_MISC_HOST_CTL,
319 (misc_host_ctl & RR_MH_NO_SWAP) | RR_MH_HALT_PROC);
320
321 /* Make the EEPROM readable */
322
323 misc_local_ctl = bus_space_read_4(iot, ioh, RR_MISC_LOCAL_CTL);
324 bus_space_write_4(iot, ioh, RR_MISC_LOCAL_CTL,
325 misc_local_ctl & ~(RR_LC_FAST_PROM | RR_LC_ADD_SRAM |
326 RR_LC_PARITY_ON));
327
328 /* Extract interesting information from the EEPROM: */
329
330 header_format = esh_read_eeprom(sc, RR_EE_HEADER_FORMAT);
331 if (header_format != RR_EE_HEADER_FORMAT_MAGIC) {
332 aprint_error_dev(sc->sc_dev, "bogus EEPROM header format value %x\n",
333 header_format);
334 goto bad_other;
335 }
336
337 /*
338 * As it is now, the runcode version in the EEPROM doesn't
339 * reflect the actual runcode version number. That is only
340 * available once the runcode starts up. We should probably
341 * change the firmware update code to modify this value,
342 * but Essential itself doesn't do it right now.
343 */
344
345 sc->sc_sram_size = 4 * esh_read_eeprom(sc, RR_EE_SRAM_SIZE);
346 sc->sc_runcode_start = esh_read_eeprom(sc, RR_EE_RUNCODE_START);
347 sc->sc_runcode_version = esh_read_eeprom(sc, RR_EE_RUNCODE_VERSION);
348
349 sc->sc_pci_latency = esh_read_eeprom(sc, RR_EE_PCI_LATENCY);
350 sc->sc_pci_lat_gnt = esh_read_eeprom(sc, RR_EE_PCI_LAT_GNT);
351
352 /* General tuning values */
353
354 sc->sc_tune.rt_mode_and_status =
355 esh_read_eeprom(sc, RR_EE_MODE_AND_STATUS);
356 sc->sc_tune.rt_conn_retry_count =
357 esh_read_eeprom(sc, RR_EE_CONN_RETRY_COUNT);
358 sc->sc_tune.rt_conn_retry_timer =
359 esh_read_eeprom(sc, RR_EE_CONN_RETRY_TIMER);
360 sc->sc_tune.rt_conn_timeout =
361 esh_read_eeprom(sc, RR_EE_CONN_TIMEOUT);
362 sc->sc_tune.rt_interrupt_timer =
363 esh_read_eeprom(sc, RR_EE_INTERRUPT_TIMER);
364 sc->sc_tune.rt_tx_timeout =
365 esh_read_eeprom(sc, RR_EE_TX_TIMEOUT);
366 sc->sc_tune.rt_rx_timeout =
367 esh_read_eeprom(sc, RR_EE_RX_TIMEOUT);
368 sc->sc_tune.rt_stats_timer =
369 esh_read_eeprom(sc, RR_EE_STATS_TIMER);
370 sc->sc_tune.rt_stats_timer = ESH_STATS_TIMER_DEFAULT;
371
372 /* DMA tuning values */
373
374 sc->sc_tune.rt_pci_state =
375 esh_read_eeprom(sc, RR_EE_PCI_STATE);
376 sc->sc_tune.rt_dma_write_state =
377 esh_read_eeprom(sc, RR_EE_DMA_WRITE_STATE);
378 sc->sc_tune.rt_dma_read_state =
379 esh_read_eeprom(sc, RR_EE_DMA_READ_STATE);
380 sc->sc_tune.rt_driver_param =
381 esh_read_eeprom(sc, RR_EE_DRIVER_PARAM);
382
383 /*
384 * Snag the ULA. The first two bytes are reserved.
385 * We don't really use it immediately, but it would be good to
386 * have for building IPv6 addresses, etc.
387 */
388
389 ula_tmp = esh_read_eeprom(sc, RR_EE_ULA_HI);
390 sc->sc_ula[0] = (ula_tmp >> 8) & 0xff;
391 sc->sc_ula[1] = ula_tmp & 0xff;
392
393 ula_tmp = esh_read_eeprom(sc, RR_EE_ULA_LO);
394 sc->sc_ula[2] = (ula_tmp >> 24) & 0xff;
395 sc->sc_ula[3] = (ula_tmp >> 16) & 0xff;
396 sc->sc_ula[4] = (ula_tmp >> 8) & 0xff;
397 sc->sc_ula[5] = ula_tmp & 0xff;
398
399 /* Reset EEPROM readability */
400
401 bus_space_write_4(iot, ioh, RR_MISC_LOCAL_CTL, misc_local_ctl);
402
403 strlcpy(ifp->if_xname, device_xname(sc->sc_dev), IFNAMSIZ);
404 ifp->if_softc = sc;
405 ifp->if_start = eshstart;
406 ifp->if_ioctl = eshioctl;
407 ifp->if_watchdog = eshwatchdog;
408 ifp->if_flags = IFF_SIMPLEX | IFF_NOARP;
409 IFQ_SET_READY(&ifp->if_snd);
410
411 if_attach(ifp);
412 hippi_ifattach(ifp, sc->sc_ula);
413
414 sc->sc_misaligned_bufs = sc->sc_bad_lens = 0;
415 sc->sc_fp_rings = 0;
416
417 return;
418
419bad_ring_dmamap_create:
420 bus_dmamap_destroy(sc->sc_dmat, sc->sc_send.ec_dma);
421bad_other:
422 bus_dmamap_unload(sc->sc_dmat, sc->sc_dma);
423bad_dmamap_load:
424 bus_dmamap_destroy(sc->sc_dmat, sc->sc_dma);
425bad_dmamap_create:
426 bus_dmamem_unmap(sc->sc_dmat, sc->sc_dma_addr, sc->sc_dma_size);
427bad_dmamem_map:
428 bus_dmamem_free(sc->sc_dmat, &sc->sc_dmaseg, rseg);
429 return;
430}

433/*
434 * Bring device up.
435 *
436 * Assume that the on-board processor has already been stopped,
437 * the rings have been cleared of valid buffers, and everything
438 * is pretty much as it was when the system started.
439 *
440 * Stop the processor (just for good measure), clear the SRAM,
441 * reload the boot code, and start it all up again, with the PC
442 * pointing at the boot code. Once the boot code has had a chance
443 * to come up, adjust all of the appropriate parameters, and send
444 * the 'start firmware' command.
445 *
446 * The NIC won't actually be up until it gets an interrupt with an
447 * event indicating the RunCode is up.
448 */
449
450void
451eshinit(struct esh_softc *sc)
452{
453 struct ifnet *ifp = &sc->sc_if;
454 bus_space_tag_t iot = sc->sc_iot;
455 bus_space_handle_t ioh = sc->sc_ioh;
456 struct rr_ring_ctl *ring;
457 u_int32_t misc_host_ctl;
458 u_int32_t misc_local_ctl;
459 u_int32_t value;
460 u_int32_t mode;
461
462 /* If we're already doing an init, don't try again simultaniously */
463
464 if ((sc->sc_flags & ESH_FL_INITIALIZING) != 0)
465 return;
466 sc->sc_flags = ESH_FL_INITIALIZING;
467
468 bus_dmamap_sync(sc->sc_dmat, sc->sc_dma, 0, sc->sc_dma_size,
469 BUS_DMASYNC_POSTREAD | BUS_DMASYNC_POSTWRITE);
470
471 /* Halt the processor (preserve NO_SWAP, if set) */
472
473 misc_host_ctl = bus_space_read_4(iot, ioh, RR_MISC_HOST_CTL);
474 bus_space_write_4(iot, ioh, RR_MISC_HOST_CTL,
475 (misc_host_ctl & RR_MH_NO_SWAP)
476 | RR_MH_HALT_PROC | RR_MH_CLEAR_INT);
477
478 /* Make the EEPROM readable */
479
480 misc_local_ctl = bus_space_read_4(iot, ioh, RR_MISC_LOCAL_CTL);
481 bus_space_write_4(iot, ioh, RR_MISC_LOCAL_CTL,
482 misc_local_ctl & ~(RR_LC_FAST_PROM |
483 RR_LC_ADD_SRAM |
484 RR_LC_PARITY_ON));
485
486 /* Reset DMA */
487
488 bus_space_write_4(iot, ioh, RR_RX_STATE, RR_RS_RESET);
489 bus_space_write_4(iot, ioh, RR_TX_STATE, 0);
490 bus_space_write_4(iot, ioh, RR_DMA_READ_STATE, RR_DR_RESET);
491 bus_space_write_4(iot, ioh, RR_DMA_WRITE_STATE, RR_DW_RESET);
492 bus_space_write_4(iot, ioh, RR_PCI_STATE, 0);
493 bus_space_write_4(iot, ioh, RR_TIMER, 0);
494 bus_space_write_4(iot, ioh, RR_TIMER_REF, 0);
495
496 /*
497 * Reset the assist register that the documentation suggests
498 * resetting. Too bad that the docs don't mention anything
499 * else about the register!
500 */
501
502 bus_space_write_4(iot, ioh, 0x15C, 1);
503
504 /* Clear BIST, set the PC to the start of the code and let 'er rip */
505
506 value = bus_space_read_4(iot, ioh, RR_PCI_BIST);
507 bus_space_write_4(iot, ioh, RR_PCI_BIST, (value & ~0xff) | 8);
508
509 sc->sc_bist_write(sc, 0);
510 esh_reset_runcode(sc);
511
512 bus_space_write_4(iot, ioh, RR_PROC_PC, sc->sc_runcode_start);
513 bus_space_write_4(iot, ioh, RR_PROC_BREAKPT, 0x00000001);
514
515 misc_host_ctl &= ~RR_MH_HALT_PROC;
516 bus_space_write_4(iot, ioh, RR_MISC_HOST_CTL, misc_host_ctl);
517
518 /* XXX: should we sleep rather than delaying for 1ms!? */
519
520 delay(1000); /* Need 500 us, but we'll give it more */
521
522 value = sc->sc_bist_read(sc);
523 if (value != 0) {
524 aprint_error_dev(sc->sc_dev, "BIST is %d, not 0!\n",
525 value);
526 goto bad_init;
527 }
528
529#ifdef ESH_PRINTF
530 printf("%s: BIST is %x\n", device_xname(sc->sc_dev), value);
531 eshstatus(sc);
532#endif
533
534 /* RunCode is up. Initialize NIC */
535
536 esh_write_addr(iot, ioh, RR_GEN_INFO_PTR, sc->sc_gen_info_dma);
537 esh_write_addr(iot, ioh, RR_RECV_RING_PTR, sc->sc_recv_ring_table_dma);
538
539 sc->sc_event_consumer = 0;
540 bus_space_write_4(iot, ioh, RR_EVENT_CONSUMER, sc->sc_event_consumer);
541 sc->sc_event_producer = bus_space_read_4(iot, ioh, RR_EVENT_PRODUCER);
542 sc->sc_cmd_producer = RR_INIT_CMD;
543 sc->sc_cmd_consumer = 0;
544
545 mode = bus_space_read_4(iot, ioh, RR_MODE_AND_STATUS);
546 mode |= (RR_MS_WARNINGS |
547 RR_MS_ERR_TERM |
548 RR_MS_NO_RESTART |
549 RR_MS_SWAP_DATA);
550 mode &= ~RR_MS_PH_MODE;
551 bus_space_write_4(iot, ioh, RR_MODE_AND_STATUS, mode);
552
553#if 0
554#ifdef ESH_PRINTF
555 printf("eshinit: misc_local_ctl %x, SRAM size %d\n", misc_local_ctl,
556 sc->sc_sram_size);
557#endif
558/*
559 misc_local_ctl |= (RR_LC_FAST_PROM | RR_LC_PARITY_ON);
560*/
561 if (sc->sc_sram_size > 256 * 1024) {
562 misc_local_ctl |= RR_LC_ADD_SRAM;
563 }
564#endif
565
566#ifdef ESH_PRINTF
567 printf("eshinit: misc_local_ctl %x\n", misc_local_ctl);
568#endif
569 bus_space_write_4(iot, ioh, RR_MISC_LOCAL_CTL, misc_local_ctl);
570
571 /* Set tuning parameters */
572
573 bus_space_write_4(iot, ioh, RR_CONN_RETRY_COUNT,
574 sc->sc_tune.rt_conn_retry_count);
575 bus_space_write_4(iot, ioh, RR_CONN_RETRY_TIMER,
576 sc->sc_tune.rt_conn_retry_timer);
577 bus_space_write_4(iot, ioh, RR_CONN_TIMEOUT,
578 sc->sc_tune.rt_conn_timeout);
579 bus_space_write_4(iot, ioh, RR_INTERRUPT_TIMER,
580 sc->sc_tune.rt_interrupt_timer);
581 bus_space_write_4(iot, ioh, RR_TX_TIMEOUT,
582 sc->sc_tune.rt_tx_timeout);
583 bus_space_write_4(iot, ioh, RR_RX_TIMEOUT,
584 sc->sc_tune.rt_rx_timeout);
585 bus_space_write_4(iot, ioh, RR_STATS_TIMER,
586 sc->sc_tune.rt_stats_timer);
587 bus_space_write_4(iot, ioh, RR_PCI_STATE,
588 sc->sc_tune.rt_pci_state);
589 bus_space_write_4(iot, ioh, RR_DMA_WRITE_STATE,
590 sc->sc_tune.rt_dma_write_state);
591 bus_space_write_4(iot, ioh, RR_DMA_READ_STATE,
592 sc->sc_tune.rt_dma_read_state);
593
594 sc->sc_max_rings = bus_space_read_4(iot, ioh, RR_MAX_RECV_RINGS);
595
596 sc->sc_runcode_version =
597 bus_space_read_4(iot, ioh, RR_RUNCODE_VERSION);
598 sc->sc_version = sc->sc_runcode_version >> 16;
599 if (sc->sc_version != 1 && sc->sc_version != 2) {
600 aprint_error_dev(sc->sc_dev, "bad version number %d in runcode\n",
601 sc->sc_version);
602 goto bad_init;
603 }
604
605 if (sc->sc_version == 1) {
606 sc->sc_options = 0;
607 } else {
608 value = bus_space_read_4(iot, ioh, RR_ULA);
609 sc->sc_options = value >> 16;
610 }
611
612 if (sc->sc_options & (RR_OP_LONG_TX | RR_OP_LONG_RX)) {
613 aprint_error_dev(sc->sc_dev, "unsupported firmware -- long descriptors\n");
614 goto bad_init;
615 }
616
617 printf("%s: startup runcode version %d.%d.%d, options %x\n",
618 device_xname(sc->sc_dev),
619 sc->sc_version,
620 (sc->sc_runcode_version >> 8) & 0xff,
621 sc->sc_runcode_version & 0xff,
622 sc->sc_options);
623
624 /* Initialize the general ring information */
625
626 memset(sc->sc_recv_ring_table, 0,
627 sizeof(struct rr_ring_ctl) * RR_ULP_COUNT);
628
629 ring = &sc->sc_gen_info->ri_event_ring_ctl;
630 ring->rr_ring_addr = sc->sc_event_ring_dma;
631 ring->rr_entry_size = sizeof(struct rr_event);
632 ring->rr_free_bufs = RR_EVENT_RING_SIZE / 4;
633 ring->rr_entries = RR_EVENT_RING_SIZE;
634 ring->rr_prod_index = 0;
635
636 ring = &sc->sc_gen_info->ri_cmd_ring_ctl;
637 ring->rr_free_bufs = 8;
638 ring->rr_entry_size = sizeof(union rr_cmd);
639 ring->rr_prod_index = RR_INIT_CMD;
640
641 ring = &sc->sc_gen_info->ri_send_ring_ctl;
642 ring->rr_ring_addr = sc->sc_send_ring_dma;
643 if (sc->sc_version == 1) {
644 ring->rr_free_bufs = RR_RR_DONT_COMPLAIN;
645 } else {
646 ring->rr_free_bufs = 0;
647 }
648
649 ring->rr_entries = RR_SEND_RING_SIZE;
650 ring->rr_entry_size = sizeof(struct rr_descr);
651
652 ring->rr_prod_index = sc->sc_send.ec_producer =
653 sc->sc_send.ec_consumer = 0;
654 sc->sc_send.ec_cur_mbuf = NULL;
655 sc->sc_send.ec_cur_buf = NULL;
656
657 sc->sc_snap_recv.ec_descr = sc->sc_snap_recv_ring;
658 sc->sc_snap_recv.ec_consumer = sc->sc_snap_recv.ec_producer = 0;
659
660 bus_dmamap_sync(sc->sc_dmat, sc->sc_dma, 0, sc->sc_dma_size,
661 BUS_DMASYNC_PREREAD | BUS_DMASYNC_PREWRITE);
662
663 /* Set up the watchdog to make sure something happens! */
664
665 sc->sc_watchdog = 0;
666 ifp->if_timer = 5;
667
668 /*
669 * Can't actually turn on interface until we see some events,
670 * so set initialized flag, but don't start sending.
671 */
672
673 sc->sc_flags = ESH_FL_INITIALIZED;
674 esh_send_cmd(sc, RR_CC_START_RUNCODE, 0, 0);
675 return;
676
677bad_init:
678 sc->sc_flags = 0;
679 wakeup((void *) sc);
680 return;
681}

684/*
685 * Code to handle the Framing Protocol (FP) interface to the esh.
686 * This will allow us to write directly to the wire, with no
687 * intervening memcpy's to slow us down.
688 */
689
690int
691esh_fpopen(dev_t dev, int oflags, int devtype,
692 struct lwp *l)
693{
694 struct esh_softc *sc;
695 struct rr_ring_ctl *ring_ctl;
696 struct esh_fp_ring_ctl *recv;
697 int ulp = ESHULP(dev);
698 int error = 0;
699 bus_size_t size;
700 int rseg;
701 int s;
702
703 sc = device_lookup_private(&esh_cd, ESHUNIT(dev));
704 if (sc == NULL || ulp == HIPPI_ULP_802)
705 return (ENXIO);
706
707#ifdef ESH_PRINTF
708 printf("esh_fpopen: opening board %d, ulp %d\n",
709 device_unit(sc->sc_dev), ulp);
710#endif
711
712 /* If the card is not up, initialize it. */
713
714 s = splnet();
715
716 if (sc->sc_fp_rings >= sc->sc_max_rings - 1) {
717 splx(s);
718 return (ENOSPC);
719 }
720
721 if ((sc->sc_flags & ESH_FL_INITIALIZED) == 0) {
722 eshinit(sc);
723 if ((sc->sc_flags & ESH_FL_INITIALIZED) == 0) {
724 splx(s);
725 return EIO;
726 }
727 }
728
729 if ((sc->sc_flags & ESH_FL_RUNCODE_UP) == 0) {
730 /*
731 * Wait for the runcode to indicate that it is up,
732 * while watching to make sure we haven't crashed.
733 */
734
735 error = 0;
736 while (error == 0 &&
737 (sc->sc_flags & ESH_FL_INITIALIZED) != 0 &&
738 (sc->sc_flags & ESH_FL_RUNCODE_UP) == 0) {
739 error = tsleep((void *) sc, PCATCH | PRIBIO,
740 "eshinit", 0);
741#ifdef ESH_PRINTF
742 printf("esh_fpopen: tslept\n");
743#endif
744 }
745
746 if (error != 0) {
747 splx(s);
748 return error;
749 }
750
751 if ((sc->sc_flags & ESH_FL_RUNCODE_UP) == 0) {
752 splx(s);
753 return EIO;
754 }
755 }
756
757
758#ifdef ESH_PRINTF
759 printf("esh_fpopen: card up\n");
760#endif
761
762 /* Look at the ring descriptor to see if the ULP is in use */
763
764 ring_ctl = &sc->sc_recv_ring_table[ulp];
765 bus_dmamap_sync(sc->sc_dmat, sc->sc_dma,
766 (char *) ring_ctl - (char *) sc->sc_dma_addr,
767 sizeof(*ring_ctl),
768 BUS_DMASYNC_POSTREAD | BUS_DMASYNC_POSTWRITE);
769 if (ring_ctl->rr_entry_size != 0) {
770 splx(s);
771 return (EBUSY);
772 }
773
774#ifdef ESH_PRINTF
775 printf("esh_fpopen: ring %d okay\n", ulp);
776#endif
777
778 /*
779 * Allocate the DMA space for the ring; space for the
780 * ring control blocks has already been staticly allocated.
781 */
782
783 recv = (struct esh_fp_ring_ctl *)
784 malloc(sizeof(*recv), M_DEVBUF, M_WAITOK|M_ZERO);
785 TAILQ_INIT(&recv->ec_queue);
786
787 size = RR_FP_RECV_RING_SIZE * sizeof(struct rr_descr);
788 error = bus_dmamem_alloc(sc->sc_dmat, size, 0, RR_DMA_BOUNDARY,
789 &recv->ec_dmaseg, 1,
790 &rseg, BUS_DMA_WAITOK);
791
792 if (error) {
793 aprint_error_dev(sc->sc_dev, "couldn't allocate space for FP receive ring"
794 "data structures\n");
795 goto bad_fp_dmamem_alloc;
796 }
797
798 if (rseg > 1) {
799 aprint_error_dev(sc->sc_dev, "contiguous memory not available for "
800 "FP receive ring\n");
801 goto bad_fp_dmamem_map;
802 }
803
804 error = bus_dmamem_map(sc->sc_dmat, &recv->ec_dmaseg, rseg,
805 size, (void **) &recv->ec_descr,
806 BUS_DMA_WAITOK | BUS_DMA_COHERENT);
807 if (error) {
808 aprint_error_dev(sc->sc_dev, "couldn't map memory for FP receive ring\n");
809 goto bad_fp_dmamem_map;
810 }
811
812 if (bus_dmamap_create(sc->sc_dmat, size, 1, size, RR_DMA_BOUNDARY,
813 BUS_DMA_ALLOCNOW | BUS_DMA_WAITOK,
814 &recv->ec_dma)) {
815 aprint_error_dev(sc->sc_dev, "couldn't create DMA map for FP receive ring\n");
816 goto bad_fp_dmamap_create;
817 }
818
819 if (bus_dmamap_load(sc->sc_dmat, recv->ec_dma, recv->ec_descr,
820 size, NULL, BUS_DMA_WAITOK)) {
821 aprint_error_dev(sc->sc_dev, "couldn't load DMA map for FP receive ring\n");
822 goto bad_fp_dmamap_load;
823 }
824
825 memset(recv->ec_descr, 0, size);
826
827 /*
828 * Create the ring:
829 *
830 * XXX: HTF are we gonna deal with the fact that we don't know
831 * if the open succeeded until we get a response from
832 * the event handler? I guess we could go to sleep waiting
833 * for the interrupt, and get woken up by the eshintr
834 * case handling it.
835 */
836
837 ring_ctl->rr_ring_addr = recv->ec_dma->dm_segs->ds_addr;
838 ring_ctl->rr_free_bufs = RR_FP_RECV_RING_SIZE / 4;
839 ring_ctl->rr_entries = RR_FP_RECV_RING_SIZE;
840 ring_ctl->rr_entry_size = sizeof(struct rr_descr);
841 ring_ctl->rr_prod_index = recv->ec_producer = recv->ec_consumer = 0;
842 ring_ctl->rr_mode = RR_RR_CHARACTER;
843 recv->ec_ulp = ulp;
844 recv->ec_index = -1;
845
846 sc->sc_fp_recv[ulp] = recv;
847
848 bus_dmamap_sync(sc->sc_dmat, sc->sc_dma,
849 (char *) ring_ctl - (char *) sc->sc_dma_addr,
850 sizeof(*ring_ctl),
851 BUS_DMASYNC_PREREAD | BUS_DMASYNC_PREWRITE);
852
853 bus_dmamap_sync(sc->sc_dmat, recv->ec_dma, 0, size,
854 BUS_DMASYNC_PREREAD | BUS_DMASYNC_PREWRITE);
855
856 esh_send_cmd(sc, RR_CC_ENABLE_RING, ulp, recv->ec_producer);
857
858#ifdef ESH_PRINTF
859 printf("esh_fpopen: sent create ring cmd\n");
860#endif
861
862 while (recv->ec_index == -1) {
863 error = tsleep((void *) &recv->ec_ulp, PCATCH | PRIBIO,
864 "eshfpopen", 0);
865 if (error != 0 || recv->ec_index == -1) {
866 goto bad_fp_ring_create;
867 }
868 }
869#ifdef ESH_PRINTF
870 printf("esh_fpopen: created ring\n");
871#endif
872
873 /*
874 * Ring is created. Set up various pointers to the ring
875 * information, fill the ring, and get going...
876 */
877
878 sc->sc_fp_rings++;
879 splx(s);
880 return 0;
881
882bad_fp_ring_create:
883#ifdef ESH_PRINTF
884 printf("esh_fpopen: bad ring create\n");
885#endif
886 sc->sc_fp_recv[ulp] = NULL;
887 memset(ring_ctl, 0, sizeof(*ring_ctl));
888 bus_dmamap_unload(sc->sc_dmat, recv->ec_dma);
889bad_fp_dmamap_load:
890 bus_dmamap_destroy(sc->sc_dmat, recv->ec_dma);
891bad_fp_dmamap_create:
892 bus_dmamem_unmap(sc->sc_dmat, (void *) recv->ec_descr, size);
893bad_fp_dmamem_map:
894 bus_dmamem_free(sc->sc_dmat, &recv->ec_dmaseg, rseg);
895bad_fp_dmamem_alloc:
896 free(recv, M_DEVBUF);
897 if (error == 0)
898 error = ENOMEM;
899 splx(s);
900 return (error);
901}

904int
905esh_fpclose(dev_t dev, int fflag, int devtype,
906 struct lwp *l)
907{
908 struct esh_softc *sc;
909 struct rr_ring_ctl *ring_ctl;
910 struct esh_fp_ring_ctl *ring;
911 int ulp = ESHULP(dev);
912 int index;
913 int error = 0;
914 int s;
915
916 sc = device_lookup_private(&esh_cd, ESHUNIT(dev));
917 if (sc == NULL || ulp == HIPPI_ULP_802)
918 return (ENXIO);
919
920 s = splnet();
921
922 ring = sc->sc_fp_recv[ulp];
923 ring_ctl = &sc->sc_recv_ring_table[ulp];
924 index = ring->ec_index;
925
926#ifdef ESH_PRINTF
927 printf("esh_fpclose: closing unit %d, ulp %d\n",
928 device_unit(sc->sc_dev), ulp);
929#endif
930 assert(ring);
931 assert(ring_ctl);
932
933 /*
934 * Disable the ring, wait for notification, and get rid of DMA
935 * stuff and dynamically allocated memory. Loop, waiting to
936 * learn that the ring has been disabled, or the card
937 * has been shut down.
938 */
939
940 do {
941 esh_send_cmd(sc, RR_CC_DISABLE_RING, ulp, ring->ec_producer);
942
943 error = tsleep((void *) &ring->ec_index, PCATCH | PRIBIO,
944 "esh_fpclose", 0);
945 if (error != 0 && error != EAGAIN) {
946 aprint_error_dev(sc->sc_dev, "esh_fpclose: wait on ring disable bad\n");
947 ring->ec_index = -1;
948 break;
949 }
950 } while (ring->ec_index != -1 && sc->sc_flags != 0);
951
952 /*
953 * XXX: Gotta unload the ring, removing old descriptors!
954 * *Can* there be outstanding reads with a close issued!?
955 */
956
957 bus_dmamap_unload(sc->sc_dmat, ring->ec_dma);
958 bus_dmamap_destroy(sc->sc_dmat, ring->ec_dma);
959 bus_dmamem_unmap(sc->sc_dmat, (void *) ring->ec_descr,
960 RR_FP_RECV_RING_SIZE * sizeof(struct rr_descr));
961 bus_dmamem_free(sc->sc_dmat, &ring->ec_dmaseg, ring->ec_dma->dm_nsegs);
962 free(ring, M_DEVBUF);
963 memset(ring_ctl, 0, sizeof(*ring_ctl));
964 sc->sc_fp_recv[ulp] = NULL;
965 sc->sc_fp_recv_index[index] = NULL;
966
967 sc->sc_fp_rings--;
968 if (sc->sc_fp_rings == 0)
969 sc->sc_flags &= ~ESH_FL_FP_RING_UP;
970
971 splx(s);
972 return 0;
973}

975int
976esh_fpread(dev_t dev, struct uio *uio, int ioflag)
977{
978 struct lwp *l = curlwp;
979 struct proc *p = l->l_proc;
980 struct iovec *iovp;
981 struct esh_softc *sc;
982 struct esh_fp_ring_ctl *ring;
983 struct esh_dmainfo *di;
984 int ulp = ESHULP(dev);
985 int error;
986 int i;
987 int s;
988
989#ifdef ESH_PRINTF
990 printf("esh_fpread: dev %x\n", dev);
991#endif
992
993 sc = device_lookup_private(&esh_cd, ESHUNIT(dev));
994 if (sc == NULL || ulp == HIPPI_ULP_802)
995 return (ENXIO);
996
997 s = splnet();
998
999 ring = sc->sc_fp_recv[ulp];
1000
1001 if ((sc->sc_flags & ESH_FL_INITIALIZED) == 0) {
1002 error = ENXIO;
1003 goto fpread_done;
1004 }
1005
1006 /* Check for validity */
1007 for (i = 0; i < uio->uio_iovcnt; i++) {
1008 /* Check for valid offsets and sizes */
1009 if (((u_long) uio->uio_iov[i].iov_base & 3) != 0 ||
1010 (i < uio->uio_iovcnt - 1 &&
1011 (uio->uio_iov[i].iov_len & 3) != 0)) {
1012 error = EFAULT;
1013 goto fpread_done;
1014 }
1015 }
1016
1017 /* Lock down the pages */
1018 for (i = 0; i < uio->uio_iovcnt; i++) {
1019 iovp = &uio->uio_iov[i];
1020 error = uvm_vslock(p->p_vmspace, iovp->iov_base, iovp->iov_len,
1021 VM_PROT_WRITE);
1022 if (error) {
1023 /* Unlock what we've locked so far. */
1024 for (--i; i >= 0; i--) {
1025 iovp = &uio->uio_iov[i];
1026 uvm_vsunlock(p->p_vmspace, iovp->iov_base,
1027 iovp->iov_len);
1028 }
1029 goto fpread_done;
1030 }
1031 }
1032
1033 /*
1034 * Perform preliminary DMA mapping and throw the buffers
1035 * onto the queue to be sent.
1036 */
1037
1038 di = esh_new_dmainfo(sc);
1039 if (di == NULL) {
1040 error = ENOMEM;
1041 goto fpread_done;
1042 }
1043 di->ed_buf = NULL;
1044 di->ed_error = 0;
1045 di->ed_read_len = 0;
1046
1047#ifdef ESH_PRINTF
1048 printf("esh_fpread: ulp %d, uio offset %qd, resid %d, iovcnt %d\n",
1049 ulp, uio->uio_offset, uio->uio_resid, uio->uio_iovcnt);
1050#endif
1051
1052 error = bus_dmamap_load_uio(sc->sc_dmat, di->ed_dma,
1053 uio, BUS_DMA_READ|BUS_DMA_WAITOK);
1054 if (error) {
1055 aprint_error_dev(sc->sc_dev, "esh_fpread: bus_dmamap_load_uio "
1056 "failed\terror code %d\n",
1057 error);
1058 error = ENOBUFS;
1059 esh_free_dmainfo(sc, di);
1060 goto fpread_done;
1061 }
1062
1063 bus_dmamap_sync(sc->sc_dmat, di->ed_dma,
1064 0, di->ed_dma->dm_mapsize,
1065 BUS_DMASYNC_PREREAD);
1066
1067#ifdef ESH_PRINTF
1068 printf("esh_fpread: ulp %d, di %p, nsegs %d, uio len %d\n",
1069 ulp, di, di->ed_dma->dm_nsegs, uio->uio_resid);
1070#endif
1071
1072 di->ed_flags |= ESH_DI_BUSY;
1073
1074 TAILQ_INSERT_TAIL(&ring->ec_queue, di, ed_list);
1075 esh_fill_fp_ring(sc, ring);
1076
1077 while ((di->ed_flags & ESH_DI_BUSY) != 0 && error == 0) {
1078 error = tsleep((void *) di, PCATCH | PRIBIO, "esh_fpread", 0);
1079#ifdef ESH_PRINTF
1080 printf("esh_fpread: ulp %d, tslept %d\n", ulp, error);
1081#endif
1082 if (error) {
1083 /*
1084 * Remove the buffer entries from the ring; this
1085 * is gonna require a DISCARD_PKT command, and
1086 * will certainly disrupt things. This is why we
1087 * can have only one outstanding read on a ring
1088 * at a time. :-(
1089 */
1090
1091 printf("esh_fpread: was that a ^C!? error %d, ulp %d\n",
1092 error, ulp);
1093 if (error == EINTR || error == ERESTART)
1094 error = 0;
1095 if ((di->ed_flags & ESH_DI_BUSY) != 0) {
1096 esh_flush_fp_ring(sc, ring, di);
1097 error = EINTR;
1098 break;
1099 }
1100 }
1101 }
1102
1103 if (error == 0 && di->ed_error != 0)
1104 error = EIO;
1105
1106 /*
1107 * How do we let the caller know how much has been read?
1108 * Adjust the uio_resid stuff!?
1109 */
1110
1111 assert(uio->uio_resid >= di->ed_read_len);
1112
1113 uio->uio_resid -= di->ed_read_len;
1114 for (i = 0; i < uio->uio_iovcnt; i++) {
1115 iovp = &uio->uio_iov[i];
1116 uvm_vsunlock(p->p_vmspace, iovp->iov_base, iovp->iov_len);
1117 }
1118 esh_free_dmainfo(sc, di);
1119
1120fpread_done:
1121#ifdef ESH_PRINTF
1122 printf("esh_fpread: ulp %d, error %d\n", ulp, error);
1123#endif
1124 splx(s);
1125 return error;
1126}

1203#endif
1204
1205 error = bus_dmamap_load_uio(sc->sc_dmat, di->ed_dma,
1206 uio, BUS_DMA_WRITE|BUS_DMA_WAITOK);
1207 if (error) {
1208 aprint_error_dev(sc->sc_dev, "esh_fpwrite: bus_dmamap_load_uio "
1209 "failed\terror code %d\n",
1210 error);
1211 error = ENOBUFS;
1212 esh_free_dmainfo(sc, di);
1213 goto fpwrite_done;
1214 }
1215
1216 bus_dmamap_sync(sc->sc_dmat, di->ed_dma,
1217 0, di->ed_dma->dm_mapsize,
1218 BUS_DMASYNC_PREWRITE);
1219
1220#ifdef ESH_PRINTF
1221 printf("esh_fpwrite: di %p, nsegs %d, uio len %d\n",
1222 di, di->ed_dma->dm_nsegs, uio->uio_resid);
1223#endif
1224
1225 len = di->ed_dma->dm_mapsize;
1226 di->ed_flags |= ESH_DI_BUSY;
1227
1228 TAILQ_INSERT_TAIL(&ring->ec_di_queue, di, ed_list);
1229 eshstart(&sc->sc_if);
1230
1231 while ((di->ed_flags & ESH_DI_BUSY) != 0 && error == 0) {
1232 error = tsleep((void *) di, PRIBIO, "esh_fpwrite", 0);
1233#ifdef ESH_PRINTF
1234 printf("esh_fpwrite: tslept %d\n", error);
1235#endif
1236 if (error) {
1237 printf("esh_fpwrite: was that a ^C!? Shouldn't be! Error %d\n",
1238 error);
1239 if (error == EINTR || error == ERESTART)
1240 error = 0;
1241 if ((di->ed_flags & ESH_DI_BUSY) != 0) {
1242 panic("interrupted eshwrite!");
1243#if 0
1244 /* Better do *something* here! */
1245 esh_flush_send_ring(sc, di);
1246#endif
1247 error = EINTR;
1248 break;
1249 }
1250 }
1251 }
1252
1253 if (error == 0 && di->ed_error != 0)
1254 error = EIO;
1255
1256 /*
1257 * How do we let the caller know how much has been written?
1258 * Adjust the uio_resid stuff!?
1259 */
1260
1261 uio->uio_resid -= len;
1262 uio->uio_offset += len;
1263
1264 for (i = 0; i < uio->uio_iovcnt; i++) {
1265 iovp = &uio->uio_iov[i];
1266 uvm_vsunlock(p->p_vmspace, iovp->iov_base, iovp->iov_len);
1267 }
1268
1269 esh_free_dmainfo(sc, di);
1270
1271fpwrite_done:
1272#ifdef ESH_PRINTF
1273 printf("esh_fpwrite: error %d\n", error);
1274#endif
1275 splx(s);
1276 return error;
1277}
1278
1279void
1280esh_fpstrategy(struct buf *bp)
1281{
1282 struct esh_softc *sc;
1283 int ulp = ESHULP(bp->b_dev);
1284 int error = 0;
1285 int s;
1286
1287#ifdef ESH_PRINTF
1288 printf("esh_fpstrategy: starting, bcount %ld, flags %lx, dev %x\n"
1289 "\tunit %x, ulp %d\n",
1290 bp->b_bcount, bp->b_flags, bp->b_dev, unit, ulp);
1291#endif
1292
1293 sc = device_lookup_private(&esh_cd, ESHUNIT(bp->b_dev));
1294
1295 s = splnet();
1296 if (sc == NULL || ulp == HIPPI_ULP_802) {
1297 bp->b_error = ENXIO;
1298 goto done;
1299 }
1300
1301 if (bp->b_bcount == 0)
1302 goto done;
1303
1304#define UP_FLAGS (ESH_FL_INITIALIZED | ESH_FL_RUNCODE_UP)
1305
1306 if ((sc->sc_flags & UP_FLAGS) != UP_FLAGS) {
1307 bp->b_error = EBUSY;
1308 goto done;
1309 }
1310#undef UP_FLAGS
1311
1312 if (bp->b_flags & B_READ) {
1313 /*
1314 * Perform preliminary DMA mapping and throw the buffers
1315 * onto the queue to be sent.
1316 */
1317
1318 struct esh_fp_ring_ctl *ring = sc->sc_fp_recv[ulp];
1319 struct esh_dmainfo *di = esh_new_dmainfo(sc);
1320
1321 if (di == NULL) {
1322 bp->b_error = ENOMEM;
1323 goto done;
1324 }
1325 di->ed_buf = bp;
1326 error = bus_dmamap_load(sc->sc_dmat, di->ed_dma,
1327 bp->b_data, bp->b_bcount,
1328 bp->b_proc,
1329 BUS_DMA_READ|BUS_DMA_WAITOK);
1330 if (error) {
1331 aprint_error_dev(sc->sc_dev, "esh_fpstrategy: "
1332 "bus_dmamap_load "
1333 "failed\terror code %d\n",
1334 error);
1335 bp->b_error = ENOBUFS;
1336 esh_free_dmainfo(sc, di);
1337 goto done;
1338 }
1339
1340 bus_dmamap_sync(sc->sc_dmat, di->ed_dma,
1341 0, di->ed_dma->dm_mapsize,
1342 BUS_DMASYNC_PREREAD);
1343
1344#ifdef ESH_PRINTF
1345 printf("fpstrategy: di %p\n", di);
1346#endif
1347
1348 TAILQ_INSERT_TAIL(&ring->ec_queue, di, ed_list);
1349 esh_fill_fp_ring(sc, ring);
1350 } else {
1351 /*
1352 * Queue up the buffer for future sending. If the card
1353 * isn't already transmitting, give it a kick.
1354 */
1355
1356 struct esh_send_ring_ctl *ring = &sc->sc_send;
1357 bufq_put(ring->ec_buf_queue, bp);
1358#ifdef ESH_PRINTF
1359 printf("esh_fpstrategy: ready to call eshstart to write!\n");
1360#endif
1361 eshstart(&sc->sc_if);
1362 }
1363 splx(s);
1364 return;
1365
1366done:
1367 splx(s);
1368#ifdef ESH_PRINTF
1369 printf("esh_fpstrategy: failing, bp->b_error %d!\n",
1370 bp->b_error);
1371#endif
1372 biodone(bp);
1373}
1374
1375/*
1376 * Handle interrupts. This is basicly event handling code; version two
1377 * firmware tries to speed things up by just telling us the location
1378 * of the producer and consumer indices, rather than sending us an event.
1379 */
1380
1381int
1382eshintr(void *arg)
1383{
1384 struct esh_softc *sc = arg;
1385 bus_space_tag_t iot = sc->sc_iot;
1386 bus_space_handle_t ioh = sc->sc_ioh;
1387 struct ifnet *ifp = &sc->sc_if;
1388 u_int32_t rc_offsets;
1389 u_int32_t misc_host_ctl;
1390 int rc_send_consumer = 0; /* shut up compiler */
1391 int rc_snap_ring_consumer = 0; /* ditto */
1392 u_int8_t fp_ring_consumer[RR_MAX_RECV_RING];
1393 int start_consumer;
1394 int ret = 0;
1395
1396 int okay = 0;
1397 int blah = 0;
1398 char sbuf[100];
1399 char t[100];
1400
1401
1402 /* Check to see if this is our interrupt. */
1403
1404 misc_host_ctl = bus_space_read_4(iot, ioh, RR_MISC_HOST_CTL);
1405 if ((misc_host_ctl & RR_MH_INTERRUPT) == 0)
1406 return 0;
1407
1408 /* If we can't do anything with the interrupt, just drop it */
1409
1410 if (sc->sc_flags == 0)
1411 return 1;
1412
1413 rc_offsets = bus_space_read_4(iot, ioh, RR_EVENT_PRODUCER);
1414 sc->sc_event_producer = rc_offsets & 0xff;
1415 if (sc->sc_version == 2) {
1416 int i;
1417
1418 sbuf[0] = '\0';
1419 strlcat(sbuf, "rc: ", sizeof(sbuf));
1420 rc_send_consumer = (rc_offsets >> 8) & 0xff;
1421 rc_snap_ring_consumer = (rc_offsets >> 16) & 0xff;
1422 for (i = 0; i < RR_MAX_RECV_RING; i += 4) {
1423 rc_offsets =
1424 bus_space_read_4(iot, ioh,
1425 RR_RUNCODE_RECV_CONS + i);
1426 /* XXX: should do this right! */
1427 NTOHL(rc_offsets);
1428 *((u_int32_t *) &fp_ring_consumer[i]) = rc_offsets;
1429 snprintf(t, sizeof(t), "%.8x|", rc_offsets);
1430 strlcat(sbuf, t, sizeof(sbuf));
1431 }
1432 }
1433 start_consumer = sc->sc_event_consumer;
1434
1435 /* Take care of synchronizing DMA with entries we read... */
1436
1437 esh_dma_sync(sc, sc->sc_event_ring,
1438 start_consumer, sc->sc_event_producer,
1439 RR_EVENT_RING_SIZE, sizeof(struct rr_event), 0,
1440 BUS_DMASYNC_POSTREAD | BUS_DMASYNC_POSTWRITE);
1441
1442 while (sc->sc_event_consumer != sc->sc_event_producer) {
1443 struct rr_event *event =
1444 &sc->sc_event_ring[sc->sc_event_consumer];
1445
1446#ifdef ESH_PRINTF
1447 if (event->re_code != RR_EC_WATCHDOG &&
1448 event->re_code != RR_EC_STATS_UPDATE &&
1449 event->re_code != RR_EC_SET_CMD_CONSUMER) {
1450 printf("%s: event code %x, ring %d, index %d\n",
1451 device_xname(sc->sc_dev), event->re_code,
1452 event->re_ring, event->re_index);
1453 if (okay == 0)
1454 printf("%s\n", sbuf);
1455 okay = 1;
1456 }
1457#endif
1458 ret = 1; /* some action was taken by card */
1459
1460 switch(event->re_code) {
1461 case RR_EC_RUNCODE_UP:
1462 printf("%s: firmware up\n", device_xname(sc->sc_dev));
1463 sc->sc_flags |= ESH_FL_RUNCODE_UP;
1464 esh_send_cmd(sc, RR_CC_WATCHDOG, 0, 0);
1465 esh_send_cmd(sc, RR_CC_UPDATE_STATS, 0, 0);
1466#ifdef ESH_PRINTF
1467 eshstatus(sc);
1468#endif
1469 if ((ifp->if_flags & IFF_UP) != 0)
1470 esh_init_snap_ring(sc);
1471 if (sc->sc_fp_rings > 0)
1472 esh_init_fp_rings(sc);
1473
1474 /*
1475 * XXX: crank up FP rings that might be
1476 * in use after a reset!
1477 */
1478 wakeup((void *) sc);
1479 break;
1480
1481 case RR_EC_WATCHDOG:
1482 /*
1483 * Record the watchdog event.
1484 * This is checked by eshwatchdog
1485 */
1486
1487 sc->sc_watchdog = 1;
1488 break;
1489
1490 case RR_EC_SET_CMD_CONSUMER:
1491 sc->sc_cmd_consumer = event->re_index;
1492 break;
1493
1494 case RR_EC_LINK_ON:
1495 printf("%s: link up\n", device_xname(sc->sc_dev));
1496 sc->sc_flags |= ESH_FL_LINK_UP;
1497
1498 esh_send_cmd(sc, RR_CC_WATCHDOG, 0, 0);
1499 esh_send_cmd(sc, RR_CC_UPDATE_STATS, 0, 0);
1500 if ((sc->sc_flags & ESH_FL_SNAP_RING_UP) != 0) {
1501 /*
1502 * Interface is now `running', with no
1503 * output active.
1504 */
1505 ifp->if_flags |= IFF_RUNNING;
1506 ifp->if_flags &= ~IFF_OACTIVE;
1507
1508 /* Attempt to start output, if any. */
1509 }
1510 eshstart(ifp);
1511 break;
1512
1513 case RR_EC_LINK_OFF:
1514 sc->sc_flags &= ~ESH_FL_LINK_UP;
1515 printf("%s: link down\n", device_xname(sc->sc_dev));
1516 break;
1517
1518 /*
1519 * These are all unexpected. We need to handle all
1520 * of them, though.
1521 */
1522
1523 case RR_EC_INVALID_CMD:
1524 case RR_EC_INTERNAL_ERROR:
1525 case RR2_EC_INTERNAL_ERROR:
1526 case RR_EC_BAD_SEND_RING:
1527 case RR_EC_BAD_SEND_BUF:
1528 case RR_EC_BAD_SEND_DESC:
1529 case RR_EC_RECV_RING_FLUSH:
1530 case RR_EC_RECV_ERROR_INFO:
1531 case RR_EC_BAD_RECV_BUF:
1532 case RR_EC_BAD_RECV_DESC:
1533 case RR_EC_BAD_RECV_RING:
1534 case RR_EC_UNIMPLEMENTED:
1535 aprint_error_dev(sc->sc_dev, "unexpected event %x;"
1536 "shutting down interface\n",
1537 event->re_code);
1538 ifp->if_flags &= ~(IFF_RUNNING | IFF_OACTIVE);
1539 sc->sc_flags = ESH_FL_CRASHED;
1540#ifdef ESH_PRINTF
1541 eshstatus(sc);
1542#endif
1543 break;
1544
1545#define CALLOUT(a) case a: \
1546 printf("%s: Event " #a " received -- " \
1547 "ring %d index %d timestamp %x\n", \
1548 device_xname(sc->sc_dev), event->re_ring, event->re_index, \
1549 event->re_timestamp); \
1550 break;
1551
1552 CALLOUT(RR_EC_NO_RING_FOR_ULP);
1553 CALLOUT(RR_EC_REJECTING); /* dropping packets */
1554#undef CALLOUT
1555
1556 /* Send events */
1557
1558 case RR_EC_PACKET_SENT: /* not used in firmware 2.x */
1559 ifp->if_opackets++;
1560 /* FALLTHROUGH */
1561
1562 case RR_EC_SET_SND_CONSUMER:
1563 assert(sc->sc_version == 1);
1564 /* FALLTHROUGH */
1565
1566 case RR_EC_SEND_RING_LOW:
1567 eshstart_cleanup(sc, event->re_index, 0);
1568 break;
1569
1570
1571 case RR_EC_CONN_REJECT:
1572 case RR_EC_CAMPON_TIMEOUT:
1573 case RR_EC_CONN_TIMEOUT:
1574 case RR_EC_DISCONN_ERR:
1575 case RR_EC_INTERNAL_PARITY:
1576 case RR_EC_TX_IDLE:
1577 case RR_EC_SEND_LINK_OFF:
1578 eshstart_cleanup(sc, event->re_index, event->re_code);
1579 break;
1580
1581 /* Receive events */
1582
1583 case RR_EC_RING_ENABLED:
1584 if (event->re_ring == HIPPI_ULP_802) {
1585 rc_snap_ring_consumer = 0; /* prevent read */
1586 sc->sc_flags |= ESH_FL_SNAP_RING_UP;
1587 esh_fill_snap_ring(sc);
1588
1589 if (sc->sc_flags & ESH_FL_LINK_UP) {
1590 /*
1591 * Interface is now `running', with no
1592 * output active.
1593 */
1594 ifp->if_flags |= IFF_RUNNING;
1595 ifp->if_flags &= ~IFF_OACTIVE;
1596
1597 /* Attempt to start output, if any. */
1598
1599 eshstart(ifp);
1600 }
1601#ifdef ESH_PRINTF
1602 if (event->re_index != 0)
1603 printf("ENABLE snap ring -- index %d instead of 0!\n",
1604 event->re_index);
1605#endif
1606 } else {
1607 struct esh_fp_ring_ctl *ring =
1608 sc->sc_fp_recv[event->re_ring];
1609
1610 sc->sc_flags |= ESH_FL_FP_RING_UP;
1611#ifdef ESH_PRINTF
1612 printf("eshintr: FP ring %d up\n",
1613 event->re_ring);
1614#endif
1615
1616 sc->sc_fp_recv_index[event->re_index] = ring;
1617 ring->ec_index = event->re_index;
1618 wakeup((void *) &ring->ec_ulp);
1619 }
1620 break;
1621
1622 case RR_EC_RING_DISABLED:
1623#ifdef ESH_PRINTF
1624 printf("eshintr: disabling ring %d\n",
1625 event->re_ring);
1626#endif
1627 if (event->re_ring == HIPPI_ULP_802) {
1628 struct rr_ring_ctl *ring =
1629 sc->sc_recv_ring_table + HIPPI_ULP_802;
1630 memset(ring, 0, sizeof(*ring));
1631 sc->sc_flags &= ~ESH_FL_CLOSING_SNAP;
1632 sc->sc_flags &= ~ESH_FL_SNAP_RING_UP;
1633 while (sc->sc_snap_recv.ec_consumer
1634 != sc->sc_snap_recv.ec_producer) {
1635 u_int16_t offset = sc->sc_snap_recv.ec_consumer;
1636
1637 bus_dmamap_unload(sc->sc_dmat,
1638 sc->sc_snap_recv.ec_dma[offset]);
1639 m_free(sc->sc_snap_recv.ec_m[offset]);
1640 sc->sc_snap_recv.ec_m[offset] = NULL;
1641 sc->sc_snap_recv.ec_consumer =
1642 NEXT_RECV(sc->sc_snap_recv.ec_consumer);
1643 }
1644 sc->sc_snap_recv.ec_consumer =
1645 rc_snap_ring_consumer;
1646 sc->sc_snap_recv.ec_producer =
1647 rc_snap_ring_consumer;
1648 wakeup((void *) &sc->sc_snap_recv);
1649 } else {
1650 struct esh_fp_ring_ctl *recv =
1651 sc->sc_fp_recv[event->re_ring];
1652 assert(recv != NULL);
1653 recv->ec_consumer = recv->ec_producer =
1654 fp_ring_consumer[recv->ec_index];
1655 recv->ec_index = -1;
1656 wakeup((void *) &recv->ec_index);
1657 }
1658 break;
1659
1660 case RR_EC_RING_ENABLE_ERR:
1661 if (event->re_ring == HIPPI_ULP_802) {
1662 aprint_error_dev(sc->sc_dev, "unable to enable SNAP ring!?\n\t"
1663 "shutting down interface\n");
1664 ifp->if_flags &= ~(IFF_RUNNING | IFF_OACTIVE);
1665#ifdef ESH_PRINTF
1666 eshstatus(sc);
1667#endif
1668 } else {
1669 /*
1670 * If we just leave the ring index as-is,
1671 * the driver will figure out that
1672 * we failed to open the ring.
1673 */
1674 wakeup((void *) &(sc->sc_fp_recv[event->re_ring]->ec_ulp));
1675 }
1676 break;
1677
1678 case RR_EC_PACKET_DISCARDED:
1679 /*
1680 * Determine the dmainfo for the current packet
1681 * we just discarded and wake up the waiting
1682 * process.
1683 *
1684 * This should never happen on the network ring!
1685 */
1686
1687 if (event->re_ring == HIPPI_ULP_802) {
1688 aprint_error_dev(sc->sc_dev, "discard on SNAP ring!?\n\t"
1689 "shutting down interface\n");
1690 ifp->if_flags &= ~(IFF_RUNNING | IFF_OACTIVE);
1691 sc->sc_flags = ESH_FL_CRASHED;
1692 } else {
1693 struct esh_fp_ring_ctl *ring =
1694 sc->sc_fp_recv[event->re_ring];
1695 struct esh_dmainfo *di =
1696 ring->ec_cur_dmainfo;
1697
1698 if (di == NULL)
1699 di = ring->ec_dmainfo[ring->ec_producer];
1700 printf("eshintr: DISCARD: index %d,"
1701 "ring prod %d, di %p, ring[index] %p\n",
1702 event->re_index, ring->ec_producer, di,
1703 ring->ec_dmainfo[event->re_index]);
1704
1705 if (di == NULL)
1706 di = ring->ec_dmainfo[event->re_index];
1707
1708 if (di == NULL) {
1709 printf("eshintr: DISCARD: NULL di, skipping...\n");
1710 break;
1711 }
1712
1713 di->ed_flags &=
1714 ~(ESH_DI_READING | ESH_DI_BUSY);
1715 wakeup((void *) &di->ed_flags);
1716 }
1717 break;
1718
1719 case RR_EC_OUT_OF_BUF:
1720 case RR_EC_RECV_RING_OUT:
1721 case RR_EC_RECV_RING_LOW:
1722 break;
1723
1724 case RR_EC_SET_RECV_CONSUMER:
1725 case RR_EC_PACKET_RECVED:
1726 if (event->re_ring == HIPPI_ULP_802)
1727 esh_read_snap_ring(sc, event->re_index, 0);
1728 else if (sc->sc_fp_recv[event->re_ring] != NULL)
1729 esh_read_fp_ring(sc, event->re_index, 0,
1730 event->re_ring);
1731 break;
1732
1733 case RR_EC_RECV_IDLE:
1734 case RR_EC_PARITY_ERR:
1735 case RR_EC_LLRC_ERR:
1736 case RR_EC_PKT_LENGTH_ERR:
1737 case RR_EC_IP_HDR_CKSUM_ERR:
1738 case RR_EC_DATA_CKSUM_ERR:
1739 case RR_EC_SHORT_BURST_ERR:
1740 case RR_EC_RECV_LINK_OFF:
1741 case RR_EC_FLAG_SYNC_ERR:
1742 case RR_EC_FRAME_ERR:
1743 case RR_EC_STATE_TRANS_ERR:
1744 case RR_EC_NO_READY_PULSE:
1745 if (event->re_ring == HIPPI_ULP_802) {
1746 esh_read_snap_ring(sc, event->re_index,
1747 event->re_code);
1748 } else {
1749 struct esh_fp_ring_ctl *r;
1750
1751 r = sc->sc_fp_recv[event->re_ring];
1752 if (r)
1753 r->ec_error = event->re_code;
1754 }
1755 break;
1756
1757 /*
1758 * Statistics events can be ignored for now. They might become
1759 * necessary if we have to deliver stats on demand, rather than
1760 * just returning the statistics block of memory.
1761 */
1762
1763 case RR_EC_STATS_UPDATE:
1764 case RR_EC_STATS_RETRIEVED:
1765 case RR_EC_TRACE:
1766 break;
1767
1768 default:
1769 aprint_error_dev(sc->sc_dev, "Bogus event code %x, "
1770 "ring %d, index %d, timestamp %x\n",
1771 event->re_code,
1772 event->re_ring, event->re_index,
1773 event->re_timestamp);
1774 break;
1775 }
1776
1777 sc->sc_event_consumer = NEXT_EVENT(sc->sc_event_consumer);
1778 }
1779
1780 /* Do the receive and send ring processing for version 2 RunCode */
1781
1782 if (sc->sc_version == 2) {
1783 int i;
1784 if (sc->sc_send.ec_consumer != rc_send_consumer) {
1785 eshstart_cleanup(sc, rc_send_consumer, 0);
1786 ret = 1;
1787 blah++;
1788 }
1789 if (sc->sc_snap_recv.ec_consumer != rc_snap_ring_consumer &&
1790 (sc->sc_flags & ESH_FL_SNAP_RING_UP) != 0) {
1791 esh_read_snap_ring(sc, rc_snap_ring_consumer, 0);
1792 ret = 1;
1793 blah++;
1794 }
1795 for (i = 0; i < RR_MAX_RECV_RING; i++) {
1796 struct esh_fp_ring_ctl *r = sc->sc_fp_recv_index[i];
1797
1798 if (r != NULL &&
1799 r->ec_consumer != fp_ring_consumer[i]) {
1800#ifdef ESH_PRINTF
1801 printf("eshintr: performed read on ring %d, index %d\n",
1802 r->ec_ulp, i);
1803#endif
1804 blah++;
1805 esh_read_fp_ring(sc, fp_ring_consumer[i],
1806 0, r->ec_ulp);
1807 fp_ring_consumer[i] = r->ec_consumer;
1808 }
1809 }
1810 if (blah != 0 && okay == 0) {
1811 okay = 1;
1812#ifdef ESH_PRINTF
1813 printf("%s\n", sbuf);
1814#endif
1815 }
1816 rc_offsets = (sc->sc_snap_recv.ec_consumer << 16) |
1817 (sc->sc_send.ec_consumer << 8) | sc->sc_event_consumer;
1818 } else {
1819 rc_offsets = sc->sc_event_consumer;
1820 }
1821
1822 esh_dma_sync(sc, sc->sc_event_ring,
1823 start_consumer, sc->sc_event_producer,
1824 RR_EVENT_RING_SIZE, sizeof(struct rr_event), 0,
1825 BUS_DMASYNC_POSTREAD | BUS_DMASYNC_POSTWRITE);
1826
1827 /* Write out new values for the FP segments... */
1828
1829 if (sc->sc_version == 2) {
1830 int i;
1831 u_int32_t u;
1832
1833 sbuf[0] = '\0';
1834 strlcat(sbuf, "drv: ", sizeof(sbuf));
1835 for (i = 0; i < RR_MAX_RECV_RING; i += 4) {
1836 /* XXX: should do this right! */
1837 u = *((u_int32_t *) &fp_ring_consumer[i]);

1872/*
1873 * Start output on the interface. Always called at splnet().
1874 * Check to see if there are any mbufs that didn't get sent the
1875 * last time this was called. If there are none, get more mbufs
1876 * and send 'em.
1877 *
1878 * For now, we only send one packet at a time.
1879 */
1880
1881void
1882eshstart(struct ifnet *ifp)
1883{
1884 struct esh_softc *sc = ifp->if_softc;
1885 struct esh_send_ring_ctl *send = &sc->sc_send;
1886 struct mbuf *m = NULL;
1887 int error;
1888
1889 /* Don't transmit if interface is busy or not running */
1890
1891#ifdef ESH_PRINTF
1892 printf("eshstart: ready to look; flags %x\n", sc->sc_flags);
1893#endif
1894
1895#define LINK_UP_FLAGS (ESH_FL_LINK_UP | ESH_FL_INITIALIZED | ESH_FL_RUNCODE_UP)
1896 if ((sc->sc_flags & LINK_UP_FLAGS) != LINK_UP_FLAGS)
1897 return;
1898#undef LINK_UP_FLAGS
1899
1900#ifdef ESH_PRINTF
1901 if (esh_check(sc))
1902 return;
1903#endif
1904
1905 /* If we have sent the current packet, get another */
1906
1907 while ((sc->sc_flags & ESH_FL_SNAP_RING_UP) != 0 &&
1908 (m = send->ec_cur_mbuf) == NULL && send->ec_cur_buf == NULL &&
1909 send->ec_cur_dmainfo == NULL) {
1910 IFQ_DEQUEUE(&ifp->if_snd, m);
1911 if (m == 0) /* not really needed */
1912 break;
1913
1914 if (ifp->if_bpf) {
1915 /*
1916 * On output, the raw packet has a eight-byte CCI
1917 * field prepended. On input, there is no such field.
1918 * The bpf expects the packet to look the same in both
1919 * places, so we temporarily lop off the prepended CCI
1920 * field here, then replace it. Ugh.
1921 *
1922 * XXX: Need to use standard mbuf manipulation
1923 * functions, first mbuf may be less than
1924 * 8 bytes long.
1925 */
1926
1927 m->m_len -= 8;
1928 m->m_data += 8;
1929 m->m_pkthdr.len -= 8;
1930 bpf_mtap(ifp, m, BPF_D_OUT);
1931 m->m_len += 8;
1932 m->m_data -= 8;
1933 m->m_pkthdr.len += 8;
1934 }
1935
1936 send->ec_len = m->m_pkthdr.len;
1937 m = send->ec_cur_mbuf = esh_adjust_mbufs(sc, m);
1938 if (m == NULL)
1939 continue;
1940
1941 error = bus_dmamap_load_mbuf(sc->sc_dmat, send->ec_dma,
1942 m, BUS_DMA_WRITE|BUS_DMA_NOWAIT);
1943 if (error)
1944 panic("%s: eshstart: "
1945 "bus_dmamap_load_mbuf failed err %d\n",
1946 device_xname(sc->sc_dev), error);
1947 send->ec_offset = 0;
1948 }
1949
1950 /*
1951 * If there are no network packets to send, see if there
1952 * are any FP packets to send.
1953 *
1954 * XXX: Some users may disagree with these priorities;
1955 * this reduces network latency by increasing FP latency...
1956 * Note that it also means that FP packets can get
1957 * locked out so that they *never* get sent, if the
1958 * network constantly fills up the pipe. Not good!
1959 */
1960
1961 if ((sc->sc_flags & ESH_FL_FP_RING_UP) != 0 &&
1962 send->ec_cur_mbuf == NULL && send->ec_cur_buf == NULL &&
1963 send->ec_cur_dmainfo == NULL &&
1964 bufq_peek(send->ec_buf_queue) != NULL) {
1965 struct buf *bp;
1966
1967#ifdef ESH_PRINTF
1968 printf("eshstart: getting a buf from send->ec_queue %p\n",
1969 send->ec_queue);
1970#endif
1971
1972 bp = send->ec_cur_buf = bufq_get(send->ec_buf_queue);
1973 send->ec_offset = 0;
1974 send->ec_len = bp->b_bcount;
1975
1976 /*
1977 * Determine the DMA mapping for the buffer.
1978 * If this is too large, what do we do!?
1979 */
1980
1981 error = bus_dmamap_load(sc->sc_dmat, send->ec_dma,
1982 bp->b_data, bp->b_bcount,
1983 bp->b_proc,
1984 BUS_DMA_WRITE|BUS_DMA_NOWAIT);
1985
1986 if (error)
1987 panic("%s: eshstart: "
1988 "bus_dmamap_load failed err %d\n",
1989 device_xname(sc->sc_dev), error);
1990 }
1991
1992 /*
1993 * If there are no packets from strategy to send, see if there
1994 * are any FP packets to send from fpwrite.
1995 */
1996
1997 if ((sc->sc_flags & ESH_FL_FP_RING_UP) != 0 &&
1998 send->ec_cur_mbuf == NULL && send->ec_cur_buf == NULL &&
1999 send->ec_cur_dmainfo == NULL) {
2000 struct esh_dmainfo *di;
2001
2002 di = TAILQ_FIRST(&send->ec_di_queue);
2003 if (di == NULL)
2004 return;
2005 TAILQ_REMOVE(&send->ec_di_queue, di, ed_list);
2006
2007#ifdef ESH_PRINTF
2008 printf("eshstart: getting a di from send->ec_di_queue %p\n",
2009 &send->ec_di_queue);
2010#endif
2011
2012 send->ec_cur_dmainfo = di;
2013 send->ec_offset = 0;
2014 send->ec_len = di->ed_dma->dm_mapsize;
2015 }
2016
2017 if (send->ec_cur_mbuf == NULL && send->ec_cur_buf == NULL &&
2018 send->ec_cur_dmainfo == NULL)
2019 return;
2020
2021 assert(send->ec_len);
2022 assert(send->ec_dma->dm_nsegs ||
2023 send->ec_cur_dmainfo->ed_dma->dm_nsegs);
2024 assert(send->ec_cur_mbuf || send->ec_cur_buf || send->ec_cur_dmainfo);
2025
2026 esh_send(sc);
2027 return;
2028}

2031/*
2032 * Put the buffers from the send dmamap into the descriptors and
2033 * send 'em off...
2034 */
2035
2036static void
2037esh_send(struct esh_softc *sc)
2038{
2039 struct esh_send_ring_ctl *send = &sc->sc_send;
2040 u_int start_producer = send->ec_producer;
2041 bus_dmamap_t dma;
2042
2043 if (send->ec_cur_dmainfo != NULL)
2044 dma = send->ec_cur_dmainfo->ed_dma;
2045 else
2046 dma = send->ec_dma;
2047
2048#ifdef ESH_PRINTF
2049 printf("esh_send: producer %x consumer %x nsegs %d\n",
2050 send->ec_producer, send->ec_consumer, dma->dm_nsegs);
2051#endif
2052
2053 esh_dma_sync(sc, send->ec_descr, send->ec_producer, send->ec_consumer,
2054 RR_SEND_RING_SIZE, sizeof(struct rr_descr), 1,
2055 BUS_DMASYNC_POSTREAD | BUS_DMASYNC_POSTWRITE);
2056
2057 while (NEXT_SEND(send->ec_producer) != send->ec_consumer &&
2058 send->ec_offset < dma->dm_nsegs) {
2059 int offset = send->ec_producer;
2060
2061 send->ec_descr[offset].rd_buffer_addr =
2062 dma->dm_segs[send->ec_offset].ds_addr;
2063 send->ec_descr[offset].rd_length =
2064 dma->dm_segs[send->ec_offset].ds_len;
2065 send->ec_descr[offset].rd_control = 0;
2066
2067 if (send->ec_offset == 0) {
2068 /* Start of the dmamap... */
2069 send->ec_descr[offset].rd_control |=
2070 RR_CT_PACKET_START;
2071 }
2072
2073 if (send->ec_offset + 1 == dma->dm_nsegs) {
2074 send->ec_descr[offset].rd_control |= RR_CT_PACKET_END;
2075 }
2076
2077 send->ec_offset++;
2078 send->ec_producer = NEXT_SEND(send->ec_producer);
2079 }
2080
2081 /*
2082 * XXX: we could optimize the dmamap_sync to just get what we've
2083 * just set up, rather than the whole buffer...
2084 */
2085
2086 bus_dmamap_sync(sc->sc_dmat, dma, 0, dma->dm_mapsize,
2087 BUS_DMASYNC_PREWRITE);
2088 esh_dma_sync(sc, send->ec_descr,
2089 start_producer, send->ec_consumer,
2090 RR_SEND_RING_SIZE, sizeof(struct rr_descr), 1,
2091 BUS_DMASYNC_PREREAD | BUS_DMASYNC_PREWRITE);
2092
2093#ifdef ESH_PRINTF
2094 if (send->ec_offset != dma->dm_nsegs)
2095 printf("eshstart: couldn't fit packet in send ring!\n");
2096#endif
2097
2098 if (sc->sc_version == 1) {
2099 esh_send_cmd(sc, RR_CC_SET_SEND_PRODUCER,
2100 0, send->ec_producer);
2101 } else {
2102 bus_space_write_4(sc->sc_iot, sc->sc_ioh,
2103 RR_SEND_PRODUCER, send->ec_producer);
2104 }
2105 return;
2106}

2195/*
2196 * XXX: Ouch: The NIC can only send word-aligned buffers, and only
2197 * the last buffer in the packet can have a length that is not
2198 * a multiple of four!
2199 *
2200 * Here we traverse the packet, pick out the bogus mbufs, and fix 'em
2201 * if possible. The fix is amazingly expensive, so we sure hope that
2202 * this is a rare occurance (it seems to be).
2203 */
2204
2205static struct mbuf *
2206esh_adjust_mbufs(struct esh_softc *sc, struct mbuf *m)
2207{
2208 struct mbuf *m0, *n, *n0;
2209 u_int32_t write_len;
2210
2211 write_len = m->m_pkthdr.len;
2212#ifdef DIAGNOSTIC
2213 if (write_len > max_write_len)
2214 max_write_len = write_len;
2215#endif
2216
2217 for (n0 = n = m; n; n = n->m_next) {
2218 while (n && n->m_len == 0) {
2219 m0 = m_free(n);
2220 if (n == m)
2221 n = n0 = m = m0;
2222 else
2223 n = n0->m_next = m0;
2224 }
2225 if (n == NULL)
2226 break;
2227
2228 if (mtod(n, long) & 3 || (n->m_next && n->m_len & 3)) {
2229 /* Gotta clean it up */
2230 struct mbuf *o;
2231 u_int32_t len;
2232
2233 sc->sc_misaligned_bufs++;
2234 MGETHDR(o, M_DONTWAIT, MT_DATA);
2235 if (!o)
2236 goto bogosity;
2237
2238 MCLGET(o, M_DONTWAIT);
2239 if (!(o->m_flags & M_EXT)) {
2240 m0 = m_free(o);
2241 goto bogosity;
2242 }
2243
2244 /*
2245 * XXX: Copy as much as we can into the
2246 * cluster. For now we can't have more
2247 * than a cluster in there. May change.
2248 * I'd prefer not to get this
2249 * down-n-dirty, but we have to be able
2250 * to do this kind of funky copy.
2251 */
2252
2253 len = uimin(MCLBYTES, write_len);
2254#ifdef DIAGNOSTIC
2255 assert(n->m_len <= len);
2256 assert(len <= MCLBYTES);
2257#endif
2258
2259 m_copydata(n, 0, len, mtod(o, void *));
2260 o->m_pkthdr.len = len;
2261 m_adj(n, len);
2262 o->m_len = len;
2263 o->m_next = n;
2264
2265 if (n == m)
2266 m = o;
2267 else
2268 n0->m_next = o;
2269 n = o;
2270 }
2271 n0 = n;
2272 write_len -= n->m_len;
2273 }
2274 return m;
2275
2276bogosity:
2277 aprint_error_dev(sc->sc_dev, "esh_adjust_mbuf: unable to allocate cluster for "
2278 "mbuf %p, len %x\n",
2279 mtod(m, void *), m->m_len);
2280 m_freem(m);
2281 return NULL;
2282}

2285/*
2286 * Read in the current valid entries from the ring and forward
2287 * them to the upper layer protocols. It is possible that we
2288 * haven't received the whole packet yet, in which case we just
2289 * add each of the buffers into the packet until we have the whole
2290 * thing.
2291 */
2292
2293static void
2294esh_read_snap_ring(struct esh_softc *sc, u_int16_t consumer, int error)
2295{
2296 struct ifnet *ifp = &sc->sc_if;
2297 struct esh_snap_ring_ctl *recv = &sc->sc_snap_recv;
2298 int start_consumer = recv->ec_consumer;
2299 u_int16_t control;
2300
2301 if ((sc->sc_flags & ESH_FL_SNAP_RING_UP) == 0)
2302 return;
2303
2304 if (error)
2305 recv->ec_error = error;
2306
2307 esh_dma_sync(sc, recv->ec_descr,
2308 start_consumer, consumer,
2309 RR_SNAP_RECV_RING_SIZE,
2310 sizeof(struct rr_descr), 0,
2311 BUS_DMASYNC_POSTREAD | BUS_DMASYNC_POSTWRITE);
2312
2313 while (recv->ec_consumer != consumer) {
2314 u_int16_t offset = recv->ec_consumer;
2315 struct mbuf *m;
2316
2317 m = recv->ec_m[offset];
2318 m->m_len = recv->ec_descr[offset].rd_length;
2319 control = recv->ec_descr[offset].rd_control;
2320 bus_dmamap_sync(sc->sc_dmat, recv->ec_dma[offset], 0, m->m_len,
2321 BUS_DMASYNC_POSTREAD);
2322 bus_dmamap_unload(sc->sc_dmat, recv->ec_dma[offset]);
2323
2324#ifdef ESH_PRINTF
2325 printf("esh_read_snap_ring: offset %x addr %p len %x flags %x\n",
2326 offset, mtod(m, void *), m->m_len, control);
2327#endif
2328 if (control & RR_CT_PACKET_START || !recv->ec_cur_mbuf) {
2329 if (recv->ec_cur_pkt) {
2330 m_freem(recv->ec_cur_pkt);
2331 recv->ec_cur_pkt = NULL;
2332 printf("%s: possible skipped packet!\n",
2333 device_xname(sc->sc_dev));
2334 }
2335 recv->ec_cur_pkt = recv->ec_cur_mbuf = m;
2336 /* allocated buffers all have pkthdrs... */
2337 m_set_rcvif(m, ifp);
2338 m->m_pkthdr.len = m->m_len;
2339 } else {
2340 if (!recv->ec_cur_pkt)
2341 panic("esh_read_snap_ring: no cur_pkt");
2342
2343 recv->ec_cur_mbuf->m_next = m;
2344 recv->ec_cur_mbuf = m;
2345 recv->ec_cur_pkt->m_pkthdr.len += m->m_len;
2346 }
2347
2348 recv->ec_m[offset] = NULL;
2349 recv->ec_descr[offset].rd_length = 0;
2350 recv->ec_descr[offset].rd_buffer_addr = 0;
2351
2352 /* Note that we can START and END on the same buffer */
2353
2354 if (control & RR_CT_PACKET_END) { /* XXX: RR2_ matches */
2355 m = recv->ec_cur_pkt;
2356 if (!error && !recv->ec_error) {
2357 if ((ifp->if_flags & IFF_RUNNING) == 0) {
2358 m_freem(m);
2359 } else {
2360 m = m_pullup(m,
2361 sizeof(struct hippi_header));
2362 if_percpuq_enqueue(ifp->if_percpuq, m);
2363 }
2364 } else {
2365 ifp->if_ierrors++;
2366 recv->ec_error = 0;
2367 m_freem(m);
2368 }
2369 recv->ec_cur_pkt = recv->ec_cur_mbuf = NULL;
2370 }
2371
2372 recv->ec_descr[offset].rd_control = 0;
2373 recv->ec_consumer = NEXT_RECV(recv->ec_consumer);
2374 }
2375
2376 esh_dma_sync(sc, recv->ec_descr,
2377 start_consumer, consumer,
2378 RR_SNAP_RECV_RING_SIZE,
2379 sizeof(struct rr_descr), 0,
2380 BUS_DMASYNC_PREREAD | BUS_DMASYNC_PREWRITE);
2381
2382 esh_fill_snap_ring(sc);
2383}

2466 struct mbuf *m;
2467
2468 MGETHDR(m, M_DONTWAIT, MT_DATA);
2469 if (!m)
2470 break;
2471 MCLGET(m, M_DONTWAIT);
2472 if ((m->m_flags & M_EXT) == 0) {
2473 m_free(m);
2474 break;
2475 }
2476
2477 error = bus_dmamap_load(sc->sc_dmat, recv->ec_dma[offset],
2478 mtod(m, void *), MCLBYTES,
2479 NULL, BUS_DMA_READ|BUS_DMA_NOWAIT);
2480 if (error) {
2481 printf("%s: esh_fill_recv_ring: bus_dmamap_load "
2482 "failed\toffset %x, error code %d\n",
2483 device_xname(sc->sc_dev), offset, error);
2484 m_free(m);
2485 break;
2486 }
2487
2488 /*
2489 * In this implementation, we should only see one segment
2490 * per DMA.
2491 */
2492
2493 assert(recv->ec_dma[offset]->dm_nsegs == 1);
2494
2495 /*
2496 * Load into the descriptors.
2497 */
2498
2499 recv->ec_descr[offset].rd_ring =
2500 (sc->sc_version == 1) ? HIPPI_ULP_802 : 0;
2501 recv->ec_descr[offset].rd_buffer_addr =
2502 recv->ec_dma[offset]->dm_segs->ds_addr;
2503 recv->ec_descr[offset].rd_length =
2504 recv->ec_dma[offset]->dm_segs->ds_len;
2505 recv->ec_descr[offset].rd_control = 0;
2506
2507 bus_dmamap_sync(sc->sc_dmat, recv->ec_dma[offset], 0, MCLBYTES,
2508 BUS_DMASYNC_PREREAD);
2509
2510 recv->ec_m[offset] = m;
2511
2512 recv->ec_producer = NEXT_RECV(recv->ec_producer);
2513 }
2514
2515 esh_dma_sync(sc, recv->ec_descr,
2516 start_producer, recv->ec_consumer,
2517 RR_SNAP_RECV_RING_SIZE,
2518 sizeof(struct rr_descr), 1,
2519 BUS_DMASYNC_PREREAD | BUS_DMASYNC_PREWRITE);
2520
2521 if (sc->sc_version == 1)
2522 esh_send_cmd(sc, RR_CC_SET_RECV_PRODUCER, HIPPI_ULP_802,
2523 recv->ec_producer);
2524 else
2525 bus_space_write_4(sc->sc_iot, sc->sc_ioh,
2526 RR_SNAP_RECV_PRODUCER, recv->ec_producer);
2527}

2557static void
2558esh_read_fp_ring(struct esh_softc *sc, u_int16_t consumer, int error, int ulp)
2559{
2560 struct esh_fp_ring_ctl *recv = sc->sc_fp_recv[ulp];
2561 int start_consumer = recv->ec_consumer;
2562 u_int16_t control;
2563
2564#ifdef ESH_PRINTF
2565 printf("esh_read_fp_ring: ulp %d, consumer %d, producer %d, old consumer %d\n",
2566 recv->ec_ulp, consumer, recv->ec_producer, recv->ec_consumer);
2567#endif
2568 if ((sc->sc_flags & ESH_FL_FP_RING_UP) == 0)
2569 return;
2570
2571 if (error != 0)
2572 recv->ec_error = error;
2573
2574 esh_dma_sync(sc, recv->ec_descr,
2575 start_consumer, consumer,
2576 RR_FP_RECV_RING_SIZE,
2577 sizeof(struct rr_descr), 0,
2578 BUS_DMASYNC_POSTREAD | BUS_DMASYNC_POSTWRITE);
2579
2580 while (recv->ec_consumer != consumer) {
2581 u_int16_t offset = recv->ec_consumer;
2582
2583 control = recv->ec_descr[offset].rd_control;
2584
2585 if (control & RR_CT_PACKET_START) {
2586 if (recv->ec_read_len) {
2587 recv->ec_error = 0;
2588 printf("%s: ulp %d: possible skipped FP packet!\n",
2589 device_xname(sc->sc_dev), recv->ec_ulp);
2590 }
2591 recv->ec_seen_end = 0;
2592 recv->ec_read_len = 0;
2593 }
2594 if (recv->ec_seen_end == 0)
2595 recv->ec_read_len += recv->ec_descr[offset].rd_length;
2596
2597#if NOT_LAME
2598 recv->ec_descr[offset].rd_length = 0;
2599 recv->ec_descr[offset].rd_buffer_addr = 0;
2600#endif
2601
2602#ifdef ESH_PRINTF
2603 printf("esh_read_fp_ring: offset %d addr %d len %d flags %x, total %d\n",
2604 offset, recv->ec_descr[offset].rd_buffer_addr,
2605 recv->ec_descr[offset].rd_length, control, recv->ec_read_len);
2606#endif
2607 /* Note that we can START and END on the same buffer */
2608
2609 if ((control & RR_CT_PACKET_END) == RR_CT_PACKET_END) {
2610 if (recv->ec_dmainfo[offset] != NULL) {
2611 struct esh_dmainfo *di =
2612 recv->ec_dmainfo[offset];
2613
2614 recv->ec_dmainfo[offset] = NULL;
2615 bus_dmamap_sync(sc->sc_dmat, di->ed_dma,
2616 0, recv->ec_read_len,
2617 BUS_DMASYNC_POSTREAD);
2618 bus_dmamap_unload(sc->sc_dmat, di->ed_dma);
2619
2620 if (!error && !recv->ec_error) {
2621 /*
2622 * XXX: we oughta do this right, with full
2623 * BPF support and the rest...
2624 */
2625 if (di->ed_buf != NULL) {
2626 di->ed_buf->b_resid =
2627 di->ed_buf->b_bcount -
2628 recv->ec_read_len;
2629 } else {
2630 di->ed_read_len =
2631 recv->ec_read_len;
2632 }
2633 } else {
2634 if (di->ed_buf != NULL) {
2635 di->ed_buf->b_resid =
2636 di->ed_buf->b_bcount;
2637 di->ed_buf->b_error = EIO;
2638 } else {
2639 di->ed_error = EIO;
2640 recv->ec_error = 0;
2641 }
2642 }
2643
2644#ifdef ESH_PRINTF
2645 printf("esh_read_fp_ring: ulp %d, read %d, resid %ld\n",
2646 recv->ec_ulp, recv->ec_read_len, (di->ed_buf ? di->ed_buf->b_resid : di->ed_read_len));
2647#endif
2648 di->ed_flags &=
2649 ~(ESH_DI_BUSY | ESH_DI_READING);
2650 if (di->ed_buf != NULL)
2651 biodone(di->ed_buf);
2652 else
2653 wakeup((void *) di);
2654 recv->ec_read_len = 0;
2655 } else {
2656#ifdef ESH_PRINTF
2657 printf("esh_read_fp_ring: ulp %d, seen end at %d\n",
2658 recv->ec_ulp, offset);
2659#endif
2660 recv->ec_seen_end = 1;
2661 }
2662 }
2663
2664#if NOT_LAME
2665 recv->ec_descr[offset].rd_control = 0;
2666#endif
2667 recv->ec_consumer = NEXT_RECV(recv->ec_consumer);
2668 }
2669
2670 esh_dma_sync(sc, recv->ec_descr,
2671 start_consumer, consumer,
2672 RR_SNAP_RECV_RING_SIZE,
2673 sizeof(struct rr_descr), 0,
2674 BUS_DMASYNC_PREREAD | BUS_DMASYNC_PREWRITE);
2675
2676 esh_fill_fp_ring(sc, recv);
2677}

2680static void
2681esh_fill_fp_ring(struct esh_softc *sc, struct esh_fp_ring_ctl *recv)
2682{
2683 struct esh_dmainfo *di = recv->ec_cur_dmainfo;
2684 int start_producer = recv->ec_producer;
2685
2686#ifdef ESH_PRINTF
2687 printf("esh_fill_fp_ring: ulp %d, di %p, producer %d\n",
2688 recv->ec_ulp, di, start_producer);
2689#endif
2690
2691 esh_dma_sync(sc, recv->ec_descr,
2692 recv->ec_producer, recv->ec_consumer,
2693 RR_SNAP_RECV_RING_SIZE,
2694 sizeof(struct rr_descr), 1,
2695 BUS_DMASYNC_POSTREAD | BUS_DMASYNC_POSTWRITE);
2696
2697 while (NEXT_RECV(recv->ec_producer) != recv->ec_consumer) {
2698 int offset = recv->ec_producer;
2699
2700 if (di == NULL) {
2701 /*
2702 * Must allow only one reader at a time; see
2703 * esh_flush_fp_ring().
2704 */
2705
2706 if (offset != start_producer)
2707 goto fp_fill_done;
2708
2709 di = TAILQ_FIRST(&recv->ec_queue);
2710 if (di == NULL)
2711 goto fp_fill_done;
2712 TAILQ_REMOVE(&recv->ec_queue, di, ed_list);
2713 recv->ec_offset = 0;
2714 recv->ec_cur_dmainfo = di;
2715 di->ed_flags |= ESH_DI_READING;
2716#ifdef ESH_PRINTF
2717 printf("\toffset %d nsegs %d\n",
2718 recv->ec_offset, di->ed_dma->dm_nsegs);
2719#endif
2720 }
2721
2722 /*
2723 * Load into the descriptors.
2724 */
2725
2726 recv->ec_descr[offset].rd_ring = 0;
2727 recv->ec_descr[offset].rd_buffer_addr =
2728 di->ed_dma->dm_segs[recv->ec_offset].ds_addr;
2729 recv->ec_descr[offset].rd_length =
2730 di->ed_dma->dm_segs[recv->ec_offset].ds_len;
2731 recv->ec_descr[offset].rd_control = 0;
2732 recv->ec_dmainfo[offset] = NULL;
2733
2734 if (recv->ec_offset == 0) {
2735 /* Start of the dmamap... */
2736 recv->ec_descr[offset].rd_control |=
2737 RR_CT_PACKET_START;
2738 }
2739
2740 assert(recv->ec_offset < di->ed_dma->dm_nsegs);
2741
2742 recv->ec_offset++;
2743 if (recv->ec_offset == di->ed_dma->dm_nsegs) {
2744 recv->ec_descr[offset].rd_control |= RR_CT_PACKET_END;
2745 recv->ec_dmainfo[offset] = di;
2746 di = NULL;
2747 recv->ec_offset = 0;
2748 recv->ec_cur_dmainfo = NULL;
2749 }
2750
2751 recv->ec_producer = NEXT_RECV(recv->ec_producer);
2752 }
2753
2754fp_fill_done:
2755 esh_dma_sync(sc, recv->ec_descr,
2756 start_producer, recv->ec_consumer,
2757 RR_SNAP_RECV_RING_SIZE,
2758 sizeof(struct rr_descr), 1,
2759 BUS_DMASYNC_PREREAD | BUS_DMASYNC_PREWRITE);
2760
2761
2762 if (sc->sc_version == 1) {
2763 esh_send_cmd(sc, RR_CC_SET_RECV_PRODUCER, recv->ec_ulp,
2764 recv->ec_producer);
2765 } else {
2766 union {
2767 u_int32_t producer;
2768 u_int8_t indices[4];
2769 } v;
2770 int which;
2771 int i;
2772 struct esh_fp_ring_ctl *r;
2773
2774 which = (recv->ec_index / 4) * 4;
2775#if BAD_PRODUCER
2776 v.producer = bus_space_read_4(sc->sc_iot, sc->sc_ioh,
2777 RR_RECVS_PRODUCER + which);
2778 NTOHL(v.producer);
2779#endif
2780 for (i = 0; i < 4; i++) {
2781 r = sc->sc_fp_recv_index[i + which];
2782 if (r != NULL)
2783 v.indices[i] = r->ec_producer;
2784 else
2785 v.indices[i] = 0;
2786 }
2787#ifdef ESH_PRINTF
2788 printf("esh_fill_fp_ring: ulp %d, updating producer %d: %.8x\n",
2789 recv->ec_ulp, which, v.producer);
2790#endif
2791 HTONL(v.producer);
2792 bus_space_write_4(sc->sc_iot, sc->sc_ioh,
2793 RR_RECVS_PRODUCER + which, v.producer);
2794 }
2795#ifdef ESH_PRINTF
2796 printf("esh_fill_fp_ring: ulp %d, final producer %d\n",
2797 recv->ec_ulp, recv->ec_producer);
2798#endif
2799}

2801/*
2802 * When a read is interrupted, we need to flush the buffers out of
2803 * the ring; otherwise, a driver error could lock a process up,
2804 * with no way to exit.
2805 */
2806
2807static void
2808esh_flush_fp_ring(struct esh_softc *sc, struct esh_fp_ring_ctl *recv, struct esh_dmainfo *di)
2809{
2810 int error = 0;
2811
2812 /*
2813 * If the read request hasn't yet made it to the top of the queue,
2814 * just remove it from the queue, and return.
2815 */
2816
2817 if ((di->ed_flags & ESH_DI_READING) != ESH_DI_READING) {
2818 TAILQ_REMOVE(&recv->ec_queue, di, ed_list);
2819 return;
2820 }
2821
2822#ifdef ESH_PRINTF
2823 printf("esh_flush_fp_ring: di->ed_flags %x, ulp %d, producer %x\n",
2824 di->ed_flags, recv->ec_ulp, recv->ec_producer);
2825#endif
2826
2827 /* Now we gotta get tough. Issue a discard packet command */
2828
2829 esh_send_cmd(sc, RR_CC_DISCARD_PKT, recv->ec_ulp,
2830 recv->ec_producer - 1);
2831
2832 /* Wait for it to finish */
2833
2834 while ((di->ed_flags & ESH_DI_READING) != ESH_DI_READING &&
2835 error == 0) {
2836 error = tsleep((void *) &di->ed_flags, PRIBIO,
2837 "esh_flush_fp_ring", hz);
2838 printf("esh_flush_fp_ring: di->ed_flags %x, error %d\n",
2839 di->ed_flags, error);
2840 /*
2841 * What do I do if this times out or gets interrupted?
2842 * Reset the card? I could get an interrupt before
2843 * giving it a chance to check. Perhaps I oughta wait
2844 * awhile? What about not giving the user a chance
2845 * to interrupt, and just expecting a quick answer?
2846 * That way I could reset the card if it doesn't
2847 * come back right away!
2848 */
2849 if (error != 0) {
2850 eshreset(sc);
2851 break;
2852 }
2853 }
2854
2855 /* XXX: Do we need to clear out the dmainfo pointers */
2856}

2859int
2860eshioctl(struct ifnet *ifp, u_long cmd, void *data)
2861{
2862 int error = 0;
2863 struct esh_softc *sc = ifp->if_softc;
2864 struct ifaddr *ifa = (struct ifaddr *)data;
2865 struct ifdrv *ifd = (struct ifdrv *) data;
2866 u_long len;
2867 int s;
2868
2869 s = splnet();
2870
2871 while (sc->sc_flags & ESH_FL_EEPROM_BUSY) {
2872 error = tsleep(&sc->sc_flags, PCATCH | PRIBIO,
2873 "esheeprom", 0);
2874 if (error != 0)
2875 goto ioctl_done;
2876 }
2877
2878 switch (cmd) {
2879
2880 case SIOCINITIFADDR:
2881 ifp->if_flags |= IFF_UP;

2883 eshinit(sc);

2885 error = EIO;

2887 }
2888 }

2893 error = tsleep((void *) &sc->sc_snap_recv,

2895 hz);

2903#ifdef INET

2905 /* The driver doesn't really care about IP addresses */
2906 break;
2907#endif
2908 default:
2909 break;
2910 }
2911 break;
2912
2913 case SIOCSIFFLAGS:

2930 goto ioctl_done;

2932

2937 eshinit(sc);

2946 error = tsleep((void *) &sc->sc_snap_recv, PRIBIO, "esh_closing_fp_ring", hz);
2947 if (error != 0)
2948 goto ioctl_done;

2951 }
2952 }
2953 break;

2956 cmd = ifd->ifd_cmd;
2957 len = ifd->ifd_len;
2958 data = ifd->ifd_data;
2959
2960 esh_generic_ioctl(sc, cmd, data, len, NULL);
2961 break;
2962
2963 default:
2964 error = ether_ioctl(ifp, cmd, data);
2965 break;
2966 }
2967
2968ioctl_done:
2969 splx(s);
2970 return (error);
2971}

2974static int
2975esh_generic_ioctl(struct esh_softc *sc, u_long cmd, void *data,
2976 u_long len, struct lwp *l)
2977{
2978 struct ifnet *ifp = &sc->sc_if;
2979 struct rr_eeprom rr_eeprom;
2980 bus_space_tag_t iot = sc->sc_iot;
2981 bus_space_handle_t ioh = sc->sc_ioh;
2982 u_int32_t misc_host_ctl;
2983 u_int32_t misc_local_ctl;
2984 u_int32_t address;
2985 u_int32_t value;
2986 u_int32_t offset;
2987 u_int32_t length;
2988 int error = 0;
2989 int i;
2990
2991 /*
2992 * If we have a LWP pointer, check to make sure that the
2993 * user is privileged before performing any destruction operations.
2994 */
2995
2996 if (l != NULL) {
2997 switch (cmd) {
2998 case EIOCGTUNE:
2999 case EIOCGEEPROM:
3000 case EIOCGSTATS:
3001 break;
3002
3003 default:
3004 error = kauth_authorize_network(l->l_cred,
3005 KAUTH_NETWORK_INTERFACE,
3006 KAUTH_REQ_NETWORK_INTERFACE_SETPRIV,
3007 ifp, KAUTH_ARG(cmd), NULL);
3008 if (error)
3009 return (error);
3010 }
3011 }
3012
3013 switch (cmd) {
3014 case EIOCGTUNE:
3015 if (len != sizeof(struct rr_tuning))
3016 error = EMSGSIZE;
3017 else {
3018 error = copyout((void *) &sc->sc_tune, data,
3019 sizeof(struct rr_tuning));
3020 }
3021 break;
3022
3023 case EIOCSTUNE:
3024 if ((ifp->if_flags & IFF_UP) == 0) {
3025 if (len != sizeof(struct rr_tuning)) {
3026 error = EMSGSIZE;
3027 } else {
3028 error = copyin(data, (void *) &sc->sc_tune,
3029 sizeof(struct rr_tuning));
3030 }
3031 } else {
3032 error = EBUSY;
3033 }
3034 break;
3035
3036 case EIOCGSTATS:
3037 if (len != sizeof(struct rr_stats))
3038 error = EMSGSIZE;
3039 else
3040 error = copyout((void *) &sc->sc_gen_info->ri_stats,
3041 data, sizeof(struct rr_stats));
3042 break;
3043
3044 case EIOCGEEPROM:
3045 case EIOCSEEPROM:
3046 if ((ifp->if_flags & IFF_UP) != 0) {
3047 error = EBUSY;
3048 break;
3049 }
3050
3051 if (len != sizeof(struct rr_eeprom)) {
3052 error = EMSGSIZE;
3053 break;
3054 }
3055
3056 error = copyin(data, (void *) &rr_eeprom, sizeof(rr_eeprom));
3057 if (error != 0)
3058 break;
3059
3060 offset = rr_eeprom.ifr_offset;
3061 length = rr_eeprom.ifr_length;
3062
3063 if (length > RR_EE_MAX_LEN * sizeof(u_int32_t)) {
3064 error = EFBIG;
3065 break;
3066 }
3067
3068 if (offset + length > RR_EE_MAX_LEN * sizeof(u_int32_t)) {
3069 error = EFAULT;
3070 break;
3071 }
3072
3073 if (offset % 4 || length % 4) {
3074 error = EIO;
3075 break;
3076 }
3077
3078 /* Halt the processor (preserve NO_SWAP, if set) */
3079
3080 misc_host_ctl = bus_space_read_4(iot, ioh, RR_MISC_HOST_CTL);
3081 bus_space_write_4(iot, ioh, RR_MISC_HOST_CTL,
3082 (misc_host_ctl & RR_MH_NO_SWAP) |
3083 RR_MH_HALT_PROC);
3084
3085 /* Make the EEPROM accessible */
3086
3087 misc_local_ctl = bus_space_read_4(iot, ioh, RR_MISC_LOCAL_CTL);
3088 value = misc_local_ctl &
3089 ~(RR_LC_FAST_PROM | RR_LC_ADD_SRAM | RR_LC_PARITY_ON);
3090 if (cmd == EIOCSEEPROM) /* make writable! */
3091 value |= RR_LC_WRITE_PROM;
3092 bus_space_write_4(iot, ioh, RR_MISC_LOCAL_CTL, value);
3093
3094 if (cmd == EIOCSEEPROM) {
3095 printf("%s: writing EEPROM\n", device_xname(sc->sc_dev));
3096 sc->sc_flags |= ESH_FL_EEPROM_BUSY;
3097 }
3098
3099 /* Do that EEPROM voodoo that you do so well... */
3100
3101 address = offset * RR_EE_BYTE_LEN;
3102 for (i = 0; i < length; i += 4) {

3326/*
3327 * Read a value from the eeprom. This expects that the NIC has already
3328 * been tweaked to put it into the right state for reading from the
3329 * EEPROM -- the HALT bit is set in the MISC_HOST_CTL register,
3330 * and the FAST_PROM, ADD_SRAM, and PARITY flags have been cleared
3331 * in the MISC_LOCAL_CTL register.
3332 *
3333 * The EEPROM layout is a little weird. There is a valid byte every
3334 * eight bytes. Words are then smeared out over 32 bytes.
3335 * All addresses listed here are the actual starting addresses.
3336 */
3337
3338static u_int32_t
3339esh_read_eeprom(struct esh_softc *sc, u_int32_t addr)
3340{
3341 int i;
3342 u_int32_t tmp;
3343 u_int32_t value = 0;
3344
3345 /* If the offset hasn't been added, add it. Otherwise pass through */
3346
3347 if (!(addr & RR_EE_OFFSET))
3348 addr += RR_EE_OFFSET;
3349
3350 for (i = 0; i < 4; i++, addr += RR_EE_BYTE_LEN) {
3351 bus_space_write_4(sc->sc_iot, sc->sc_ioh,
3352 RR_WINDOW_BASE, addr);
3353 tmp = bus_space_read_4(sc->sc_iot, sc->sc_ioh,
3354 RR_WINDOW_DATA);
3355 value = (value << 8) | ((tmp >> 24) & 0xff);
3356 }
3357 return value;
3358}

3361/*
3362 * Write a value to the eeprom. Just like esh_read_eeprom, this routine
3363 * expects that the NIC has already been tweaked to put it into the right
3364 * state for reading from the EEPROM. Things are further complicated
3365 * in that we need to read each byte after we write it to ensure that
3366 * the new value has been successfully written. It can take as long
3367 * as 1ms (!) to write a byte.
3368 */
3369
3370static int
3371esh_write_eeprom(struct esh_softc *sc, u_int32_t addr, u_int32_t value)
3372{
3373 int i, j;
3374 u_int32_t shifted_value, tmp = 0;
3375
3376 /* If the offset hasn't been added, add it. Otherwise pass through */
3377
3378 if (!(addr & RR_EE_OFFSET))
3379 addr += RR_EE_OFFSET;
3380
3381 for (i = 0; i < 4; i++, addr += RR_EE_BYTE_LEN) {
3382 bus_space_write_4(sc->sc_iot, sc->sc_ioh,
3383 RR_WINDOW_BASE, addr);
3384
3385 /*
3386 * Get the byte out of value, starting with the top, and
3387 * put it into the top byte of the word to write.
3388 */
3389
3390 shifted_value = ((value >> ((3 - i) * 8)) & 0xff) << 24;
3391 bus_space_write_4(sc->sc_iot, sc->sc_ioh, RR_WINDOW_DATA,
3392 shifted_value);
3393 for (j = 0; j < 50; j++) {
3394 tmp = bus_space_read_4(sc->sc_iot, sc->sc_ioh,
3395 RR_WINDOW_DATA);
3396 if (tmp == shifted_value)
3397 break;
3398 delay(500); /* 50us break * 20 = 1ms */
3399 }
3400 if (tmp != shifted_value)
3401 return -1;
3402 }
3403
3404 return 0;
3405}
