487/*
488 * RX/TX descriptor definition. When large send mode is enabled, the
489 * lower 11 bits of the TX rtk_cmd word are used to hold the MSS, and
490 * the checksum offload bits are disabled. The structure layout is
491 * the same for RX and TX descriptors
492 */

496 volatile uint32_t re_vlanctl;
497 volatile uint32_t re_bufaddr_lo;
498 volatile uint32_t re_bufaddr_hi;
