34/*
35 * ST16650A UART registers
36 */
37
38#define com_data 0 /* data register (R/W) */
39#define com_dlbl 0 /* divisor latch low (W) */
40#define com_dlbh 1 /* divisor latch high (W) */
41#define com_ier 1 /* interrupt enable (W) */
42#define com_iir 2 /* interrupt identification (R) */
43#define com_fifo 2 /* FIFO control (W) */
44#define com_efr 2 /* Enhanced feature register (R/W) */
45#define com_lctl 3 /* line control register (R/W) */
46#define com_cfcr 3 /* line control register (R/W) */
47#define com_mcr 4 /* modem control register (R/W) */
48#define com_xon1 4 /* XON 1 character (R/W) */
49#define com_lsr 5 /* line status register (R/W) */
50#define com_xon2 5 /* XON 2 character (R/W) */
51#define com_msr 6 /* modem status register (R/W) */
52#define com_xoff1 6 /* XOFF 1 character (R/W) */
53#define com_scratch 7 /* scratch register (R/W) */
54#define com_xoff2 7 /* XOFF 2 character (R/W) */
