28#ifndef _SYS_DEV_IC_W83L518DVAR_H
29#define _SYS_DEV_IC_W83L518DVAR_H

31struct wb_softc {
32 device_t wb_dev;
33
34 /* to be filled in by bus driver */
35 bus_space_tag_t wb_iot;
36 bus_space_handle_t wb_ioh;
37 uint8_t wb_type;
38 uint16_t wb_base;
39 uint8_t wb_irq;
40
41 /* private */
42 device_t wb_sdmmc_dev;
43 int wb_sdmmc_width;
44 uint8_t wb_sdmmc_clk;
45 uint8_t wb_sdmmc_intsts;
46 callout_t wb_sdmmc_callout;
47};
48
49void wb_attach(struct wb_softc *);
50int wb_detach(struct wb_softc *, int);
51int wb_intr(void *);
52
53uint8_t wb_read(struct wb_softc *, uint8_t);
54void wb_write(struct wb_softc *, uint8_t, uint8_t);
55uint8_t wb_idx_read(struct wb_softc *, uint8_t);
56void wb_idx_write(struct wb_softc *, uint8_t, uint8_t);
57
58void wb_led(struct wb_softc *, bool);
59
60bool wb_suspend(struct wb_softc *);
61bool wb_resume(struct wb_softc *);
62
63#endif
