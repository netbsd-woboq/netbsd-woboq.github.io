49/*
50 * This header defines a simple command interface to the FreeBSD
51 * WaveLAN/IEEE driver (wi) driver, which is used to set certain
52 * device-specific parameters which can't be easily managed through
53 * ifconfig(8). No, sysctl(2) is not the answer. I said a _simple_
54 * interface, didn't I.
55 */

62#define SIOCGWAVELAN SIOCGIFGENERIC
63#endif

65/*
66 * Technically I don't think there's a limit to a record
67 * length. The largest record is the one that contains the CIS
68 * data, which is 240 words long, so 256 should be a safe
69 * value.
70 */
71#define WI_MAX_DATALEN 512

75 u_int16_t wi_type;

148/*
149 * Lucent/wavelan IEEE signal strength cache
150 *
151 * driver keeps cache of last
152 * MAXWICACHE packets to arrive including signal strength info.
153 * daemons may read this via ioctl
154 *
155 * Each entry in the wi_sigcache has a unique macsrc.
156 */
157#ifdef WICACHE
158#define MAXWICACHE 10
159
160struct wi_sigcache {
161 char macsrc[6]; /* unique MAC address for entry */
162 int ipsrc; /* ip address associated with packet */
163 int signal; /* signal strength of the packet */
164 int noise; /* noise value */
165 int quality; /* quality of the packet */
166};
167#endif
