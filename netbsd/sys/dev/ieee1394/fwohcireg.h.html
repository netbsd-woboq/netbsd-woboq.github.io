103#define FWOHCI_DMA_SET(x, y) ((x) |= htole32(y))

367#if BYTE_ORDER == BIG_ENDIAN
368 uint32_t :8,
369 srcbus:1,
370 :4,
371 spd:3,
372 tlrt:8,
373 tcode:4,

377 tcode:4,
378 tlrt:8,
379 spd:3,

395 tcode:4,
