40#ifndef _IF_FWIPVAR_H_
41#define _IF_FWIPVAR_H_
42
43#define MTAG_FIREWIRE_HWADDR 0
44#define MTAG_FIREWIRE_SENDER_EUID 1
45
46
47struct fwip_softc {
48 /* XXX this must be first for fd.post_explore() */
49 struct firewire_dev_comm sc_fd;
50 short sc_dma_ch;
51 struct fw_bind sc_fwb;
52 struct fw_eui64 sc_last_dest;
53 struct fw_pkt sc_last_hdr;
54 STAILQ_HEAD(, fw_xfer) sc_xferlist;
55 struct crom_chunk sc_unit4; /* unit directory for IPv4 */
56 struct crom_chunk sc_spec4; /* specifier description IPv4 */
57 struct crom_chunk sc_ver4; /* version description IPv4 */
58 struct crom_chunk sc_unit6; /* unit directory for IPv6 */
59 struct crom_chunk sc_spec6; /* specifier description IPv6 */
60 struct crom_chunk sc_ver6; /* version description IPv6 */
61 struct fwip_eth_softc {
62 struct ieee1394com fwcom;
63 struct ifnet *fwip_ifp;
64 struct fwip_softc *fwip;
65 } sc_eth;
66 kmutex_t sc_mtx;
67};
68
69#endif /* !_IF_FWIPVAR_H_ */
