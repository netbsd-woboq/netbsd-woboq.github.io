37/* Protocol constants */
38#define SIR_EXTRA_BOF 0xff
39#define SIR_BOF 0xc0
40#define SIR_CE 0x7d
41#define SIR_EOF 0xc1

43#define SIR_ESC_BIT 0x20
44
45enum framefsmstate {
46 FSTATE_END_OF_FRAME,
47 FSTATE_START_OF_FRAME,
48 FSTATE_IN_DATA,
49 FSTATE_IN_END
50};
51
52enum frameresult {
53 FR_IDLE,
54 FR_INPROGRESS,
55 FR_FRAMEOK,
56 FR_FRAMEBADFCS,
57 FR_FRAMEMALFORMED,
58 FR_BUFFEROVERRUN
59};
60
61struct framestate {
62 u_int8_t *buffer;
63 size_t buflen;
64 size_t bufindex;
65
66 enum framefsmstate fsmstate;
67 u_int escaped;
68 u_int state_index;
69};
70
71#define deframe_isclear(fs) ((fs)->fsmstate == FSTATE_END_OF_FRAME)

80 * CRC computation

82#define INITFCS 0xffff
83#define GOODFCS 0xf0b8

85extern const u_int16_t irda_fcstab[];
86
87static __inline u_int16_t updateFCS(u_int16_t fcs, int c) {
88 return (fcs >> 8) ^ irda_fcstab[(fcs^c) & 0xff];
89}
