33#ifndef _DEV_ISA_ISADMAREG_H_
34#define _DEV_ISA_ISADMAREG_H_
35
36#include <dev/ic/i8237reg.h>
37
38/*
39 * By default, ISA DMA controllers can do 64k or 128k transfers, depending
40 * on the width of the channel being used. However, this may be modified
41 * by our parent based on bus constraints, etc. Clients of ISA DMA should
42 * query the ISA DMA to determine the maximum transfer size allowed.
43 */
44#define ISA_DMA_MAXSIZE_8BIT (64 * 1024)
45#define ISA_DMA_MAXSIZE_16BIT (ISA_DMA_MAXSIZE_8BIT * 2)
46
47#define ISA_DMA_MAXSIZE_DEFAULT(chan) \
48 (((chan) & 4) ? ISA_DMA_MAXSIZE_16BIT : ISA_DMA_MAXSIZE_8BIT)
49
50/* DMA Page Address Registers size */
51#define DMAPG_IOSIZE (1*15)
52
53/*
54 * Register definitions for DMA controller 1 (channels 0..3):
55 */
56#define DMA1_CHN(c) (1*(2*(c))) /* addr reg for channel c */
57#define DMA1_SR (1*8) /* status register */
58#define DMA1_SMSK (1*10) /* single mask register */
59#define DMA1_MODE (1*11) /* mode register */
60#define DMA1_FFC (1*12) /* clear first/last FF */
61#define DMA1_MASK (1*15) /* mask register */
62
63#define DMA1_IOSIZE (1*16)
64
65/*
66 * Register definitions for DMA controller 2 (channels 4..7):
67 */
68#define DMA2_CHN(c) (2*(2*(c))) /* addr reg for channel c */
69#define DMA2_SR (2*8) /* status register */
70#define DMA2_SMSK (2*10) /* single mask register */
71#define DMA2_MODE (2*11) /* mode register */
72#define DMA2_FFC (2*12) /* clear first/last FF */
73#define DMA2_MASK (2*15) /* mask register */
74
75#define DMA2_IOSIZE (2*16)
76
77#endif /* _DEV_ISA_ISADMAREG_H_ */
