52MODULE(MODULE_CLASS_DRIVER, lm_wbsio, "lm_isa_common,wbsio");
53
54#ifdef _MODULE
55#include "ioconf.c"
56#endif
57
58static int
59lm_wbsio_modcmd(modcmd_t cmd, void *priv)
60{
61 int error = 0;
62
63 switch (cmd) {
64 case MODULE_CMD_INIT:
65#ifdef _MODULE
66 error = config_init_component(cfdriver_ioconf_lm_wbsio,
67 cfattach_ioconf_lm_wbsio, cfdata_ioconf_lm_wbsio);
68#endif
69 return error;
70 case MODULE_CMD_FINI:
71#ifdef _MODULE
72 error = config_fini_component(cfdriver_ioconf_lm_wbsio,
73 cfattach_ioconf_lm_wbsio, cfdata_ioconf_lm_wbsio);
74#endif
75 return error;
76 default:
77 return ENOTTY;
78 }
79}
