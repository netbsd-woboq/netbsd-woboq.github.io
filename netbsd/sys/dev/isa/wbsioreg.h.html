76/* Make the above WBSIO_ID_* vaue from WBSIO_ID, WBSIO_REV and IDbits */
77#define WBSIO_MAKEID(id, rev, bits) \
78 (((bits) == 13) ? (((uint16_t)(id) << 8) | ((rev) & 0xf8)) : \
79 ((bits) == 12) ? (((uint16_t)(id) << 4) | ((rev) >> 4)) : (id))
80
81#define WBSIO_MAKEREV(rev, bits) \
82 (((bits) == 13) ? ((rev) & 0x07) : ((bits) == 12) ? ((rev) & 0x0f) : \
83 (rev))
