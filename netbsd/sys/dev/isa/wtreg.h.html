53#include <sys/ioccom.h>
54
55/* ioctl for direct QIC commands */
56#define WTQICMD _IO('W', 0)
57
58/* QIC-02 commands allowed for WTQICMD */
59#define QIC_ERASE 0x22 /* erase the tape */
60#define QIC_RETENS 0x24 /* retension the tape */
61
62/* internal QIC-02 commands */
63#define QIC_RDDATA 0x80 /* read data */
64#define QIC_READFM 0xa0 /* read file mark */
65#define QIC_WRTDATA 0x40 /* write data */
66#define QIC_WRITEFM 0x60 /* write file mark */
67#define QIC_RDSTAT 0xc0 /* read status command */
68#define QIC_REWIND 0x21 /* rewind command (position+bot) */
69#define QIC_FMT11 0x26 /* set format QIC-11 */
70#define QIC_FMT24 0x27 /* set format QIC-24 */
71#define QIC_FMT120 0x28 /* set format QIC-120 */
72#define QIC_FMT150 0x29 /* set format QIC-150 */
73#define QIC_FMT300 0x2a /* set format QIC-300/QIC-2100 */
74#define QIC_FMT600 0x2b /* set format QIC-600/QIC-2200 */
75
76/* tape driver flags */
77#define TPINUSE 0x0001 /* tape is already open */
78#define TPREAD 0x0002 /* tape is only open for reading */
79#define TPWRITE 0x0004 /* tape is only open for writing */
80#define TPSTART 0x0008 /* tape must be rewound and reset */
81#define TPRMARK 0x0010 /* read file mark command outstanding */
82#define TPWMARK 0x0020 /* write file mark command outstanding */
83#define TPREW 0x0040 /* rewind command outstanding */
84#define TPEXCEP 0x0080 /* i/o exception flag */
85#define TPVOL 0x0100 /* read file mark or hit end of tape */
86#define TPWO 0x0200 /* write command outstanding */
87#define TPRO 0x0400 /* read command outstanding */
88#define TPWANY 0x0800 /* write command requested */
89#define TPRANY 0x1000 /* read command requested */
90#define TPWP 0x2000 /* write protect error seen */
91#define TPTIMER 0x4000 /* timer() is active */
92#define TPACTIVE 0x8000 /* DMA i/o active */
93
94/* controller error register bits */
95#define TP_FIL 0x0001 /* File mark detected */
96#define TP_BNL 0x0002 /* Block not located */
97#define TP_UDA 0x0004 /* Unrecoverable data error */
98#define TP_EOM 0x0008 /* End of media */
99#define TP_WRP 0x0010 /* Write protected cartridge */
100#define TP_USL 0x0020 /* Unselected drive */
101#define TP_CNI 0x0040 /* Cartridge not in place */
102#define TP_ST0 0x0080 /* Status byte 0 bits */
103#define TP_ST0MASK 0x00ff /* Status byte 0 mask */
104#define TP_POR 0x0100 /* Power on/reset occurred */
105#define TP_ERM 0x0200 /* Reserved for end of recorded media */
106#define TP_BPE 0x0400 /* Reserved for bus parity error */
107#define TP_BOM 0x0800 /* Beginning of media */
108#define TP_MBD 0x1000 /* Marginal block detected */
109#define TP_NDT 0x2000 /* No data detected */
110#define TP_ILL 0x4000 /* Illegal command - should not happen! */
111#define TP_ST1 0x8000 /* Status byte 1 bits */
112#define TP_ST1MASK 0xff00 /* Status byte 1 mask */
113
114/* formats for printing flags and error values */
115#define WTDS_BITS "\20\1inuse\2read\3write\4start\5rmark\6wmark\7rew\10excep\11vol\12wo\13ro\14wany\15rany\16wp\17timer\20active"
116#define WTER_BITS "\20\1eof\2bnl\3uda\4eom\5wrp\6usl\7cni\11por\12erm\13bpe\14bom\15mbd\16ndt\17ill"
117
118/* device minor number */
119#define WT_BSIZE 0100 /* long block flag */
120#define WT_DENSEL 0070 /* density select mask */
121#define WT_DENSDFLT 0000 /* default density */
122#define WT_QIC11 0010 /* 11 megabytes? */
123#define WT_QIC24 0020 /* 60 megabytes */
124#define WT_QIC120 0030 /* 120 megabytes */
125#define WT_QIC150 0040 /* 150 megabytes */
126#define WT_QIC300 0050 /* 300 megabytes? */
127#define WT_QIC600 0060 /* 600 megabytes? */
