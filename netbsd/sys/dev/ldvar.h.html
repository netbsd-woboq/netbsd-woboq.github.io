32#ifndef _DEV_LDVAR_H_
33#define _DEV_LDVAR_H_
34
35#include <sys/mutex.h>
36#include <sys/rndsource.h>
37
38#include <dev/dkvar.h> /* for dk_softc */
39
40struct ld_softc {
41 struct dk_softc sc_dksc;
42 kmutex_t sc_mutex;
43 kcondvar_t sc_drain;
44
45 int sc_queuecnt; /* current h/w queue depth */
46 int sc_ncylinders; /* # cylinders */
47 int sc_nheads; /* # heads */
48 int sc_nsectors; /* # sectors per track */
49 uint64_t sc_disksize512;
50
51 /*
52 * The following are filled by hardware specific attachment code.
53 */
54 device_t sc_dv;
55 int sc_flags; /* control flags */
56 uint64_t sc_secperunit; /* # sectors in total */
57 int sc_secsize; /* sector size in bytes */
58 int sc_maxxfer; /* max xfer size in bytes */
59 int sc_maxqueuecnt; /* maximum h/w queue depth */
60 char *sc_typename; /* inquiry data */
61
62 int (*sc_dump)(struct ld_softc *, void *, int, int);
63 int (*sc_ioctl)(struct ld_softc *, u_long, void *, int32_t, bool);
64 int (*sc_start)(struct ld_softc *, struct buf *);
65 int (*sc_discard)(struct ld_softc *, struct buf *);
66};
67
68/* sc_flags */
69#define LDF_ENABLED 0x001 /* device enabled */
70#define LDF_DRAIN 0x020 /* maxqueuecnt has changed; drain */
71#define LDF_NO_RND 0x040 /* do not attach rnd source */
72#define LDF_MPSAFE 0x080 /* backend is MPSAFE */
73
74int ldadjqparam(struct ld_softc *, int);
75void ldattach(struct ld_softc *, const char *);
76int ldbegindetach(struct ld_softc *, int);
77void ldenddetach(struct ld_softc *);
78void lddone(struct ld_softc *, struct buf *);
79void lddiscardend(struct ld_softc *, struct buf *);
80
81#endif /* !_DEV_LDVAR_H_ */
