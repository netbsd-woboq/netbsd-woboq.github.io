90static int
91ciphymatch(device_t parent, cfdata_t match,
92 void *aux)
93{
94 struct mii_attach_args *ma = aux;
95
96 if (mii_phy_match(ma, ciphys) != NULL)
97 return 10;
98
99 return 0;
100}

102static void
103ciphyattach(device_t parent, device_t self, void *aux)
104{
105 struct mii_softc *sc = device_private(self);
106 struct mii_attach_args *ma = aux;
107 struct mii_data *mii = ma->mii_data;
108 const struct mii_phydesc *mpd;
109
110 mpd = mii_phy_match(ma, ciphys);
111 aprint_naive(": Media interface\n");
112 aprint_normal(": %s, rev. %d\n", mpd->mpd_name, MII_REV(ma->mii_id2));
113
114 sc->mii_dev = self;
115 sc->mii_inst = mii->mii_instance;
116 sc->mii_phy = ma->mii_phyno;
117 sc->mii_funcs = &ciphy_funcs;
118 sc->mii_pdata = mii;
119 sc->mii_flags = ma->mii_flags;
120 sc->mii_anegticks = MII_ANEGTICKS;
121
122 sc->mii_flags |= MIIF_NOISOLATE;
123
124 ciphy_reset(sc);
125
126 PHY_READ(sc, MII_BMSR, &sc->mii_capabilities);
127 sc->mii_capabilities &= ma->mii_capmask;
128 if (sc->mii_capabilities & BMSR_EXTSTAT)
129 PHY_READ(sc, MII_EXTSR, &sc->mii_extcapabilities);
130 aprint_normal_dev(self, "");
131 if ((sc->mii_capabilities & BMSR_MEDIAMASK) == 0)
132 aprint_error("no media present");
133 else
134 mii_phy_add_media(sc);
135 aprint_normal("\n");
136}

138static int
139ciphy_service(struct mii_softc *sc, struct mii_data *mii, int cmd)
140{
141 struct ifmedia_entry *ife = mii->mii_media.ifm_cur;
142 uint16_t reg, speed, gig;
143
144 switch (cmd) {
145 case MII_POLLSTAT:
146 /* If we're not polling our PHY instance, just return. */
147 if (IFM_INST(ife->ifm_media) != sc->mii_inst)
148 return 0;
149 break;
150
151 case MII_MEDIACHG:
152 /*
153 * If the media indicates a different PHY instance,
154 * isolate ourselves.
155 */
156 if (IFM_INST(ife->ifm_media) != sc->mii_inst) {
157 PHY_READ(sc, MII_BMCR, &reg);
158 PHY_WRITE(sc, MII_BMCR, reg | BMCR_ISO);
159 return 0;
160 }
161
162 /* If the interface is not up, don't do anything. */
163 if ((mii->mii_ifp->if_flags & IFF_UP) == 0)
164 break;
165
166 ciphy_fixup(sc); /* XXX hardware bug work-around */
167
168 switch (IFM_SUBTYPE(ife->ifm_media)) {
169 case IFM_AUTO:
170#ifdef foo
171 /* If we're already in auto mode, just return. */
172 PHY_READ(sc, MII_BMCR, &reg);
173 if (reg & BMCR_AUTOEN)
174 return 0;
175#endif
176 (void) mii_phy_auto(sc, 0);
177 break;
178 case IFM_1000_T:
179 speed = BMCR_S1000;
180 goto setit;
181 case IFM_100_TX:
182 speed = BMCR_S100;
183 goto setit;
184 case IFM_10_T:
185 speed = BMCR_S10;
186setit:
187 if ((ife->ifm_media & IFM_FDX) != 0) {
188 speed |= BMCR_FDX;
189 gig = GTCR_ADV_1000TFDX;
190 } else
191 gig = GTCR_ADV_1000THDX;
192
193 PHY_WRITE(sc, MII_GTCR, 0);
194 PHY_WRITE(sc, MII_BMCR, speed);
195 PHY_WRITE(sc, MII_ANAR, ANAR_CSMA);
196
197 if (IFM_SUBTYPE(ife->ifm_media) != IFM_1000_T)
198 break;
199
200 PHY_WRITE(sc, MII_GTCR, gig);
201 PHY_WRITE(sc, MII_BMCR,
202 speed | BMCR_AUTOEN | BMCR_STARTNEG);
203
204 /*
205 * When setting the link manually, one side must
206 * be the master and the other the slave. However
207 * ifmedia doesn't give us a good way to specify
208 * this, so we fake it by using one of the LINK
209 * flags. If LINK0 is set, we program the PHY to
210 * be a master, otherwise it's a slave.
211 */
212 if ((mii->mii_ifp->if_flags & IFF_LINK0)) {
213 PHY_WRITE(sc, MII_GTCR,
214 gig | GTCR_MAN_MS | GTCR_ADV_MS);
215 } else
216 PHY_WRITE(sc, MII_GTCR, gig | GTCR_MAN_MS);
217 break;
218 case IFM_NONE:
219 PHY_WRITE(sc, MII_BMCR, BMCR_ISO | BMCR_PDOWN);
220 break;
221 case IFM_100_T4:
222 default:
223 return EINVAL;
224 }
225 break;
226
227 case MII_TICK:
228 /* If we're not currently selected, just return. */
229 if (IFM_INST(ife->ifm_media) != sc->mii_inst)
230 return 0;
231
232 /* Is the interface even up? */
233 if ((mii->mii_ifp->if_flags & IFF_UP) == 0)
234 return 0;
235
236 /* Only used for autonegotiation. */
237 if ((IFM_SUBTYPE(ife->ifm_media) != IFM_AUTO) &&
238 (IFM_SUBTYPE(ife->ifm_media) != IFM_1000_T)) {
239 /*
240 * Reset autonegotiation timer to 0 just to make sure
241 * the future autonegotiation start with 0.
242 */
243 sc->mii_ticks = 0;
244 break;
245 }
246
247 /*
248 * Check to see if we have link. If we do, we don't
249 * need to restart the autonegotiation process. Read
250 * the BMSR twice in case it's latched.
251 */
252 PHY_READ(sc, MII_BMSR, &reg);
253 PHY_READ(sc, MII_BMSR, &reg);
254 if (reg & BMSR_LINK) {
255 /*
256 * Reset autonegotiation timer to 0 in case the link
257 * goes down in the next tick.
258 */
259 sc->mii_ticks = 0;
260 /* See above. */
261 break;
262 }
263
264 /*
265 * mii_ticks == 0 means it's the first tick after changing the
266 * media or the link became down since the last tick
267 * (see above), so return with 0 to update the status.
268 */
269 if (sc->mii_ticks++ == 0)
270 break;
271
272 /* Only retry autonegotiation every N seconds. */
273 if (sc->mii_ticks <= MII_ANEGTICKS_GIGE)
274 break;
275
276 mii_phy_auto(sc, 0);
277 return 0;
278 }
279
280 /* Update the media status. */
281 ciphy_status(sc);
282
283 /*
284 * Callback if something changed. Note that we need to poke
285 * apply fixups for certain PHY revs.
286 */
287 if (sc->mii_media_active != mii->mii_media_active ||
288 sc->mii_media_status != mii->mii_media_status ||
289 cmd == MII_MEDIACHG) {
290 ciphy_fixup(sc);
291 }
292 mii_phy_update(sc, cmd);
293 return 0;
294}

296static void
297ciphy_status(struct mii_softc *sc)
298{
299 struct mii_data *mii = sc->mii_pdata;
300 uint16_t bmsr, bmcr;
301
302 mii->mii_media_status = IFM_AVALID;
303 mii->mii_media_active = IFM_ETHER;
304
305 PHY_READ(sc, MII_BMSR, &bmsr);
306 PHY_READ(sc, MII_BMSR, &bmsr);
307
308 if (bmsr & BMSR_LINK)
309 mii->mii_media_status |= IFM_ACTIVE;
310
311 PHY_READ(sc, MII_BMCR, &bmcr);
312
313 if (bmcr & BMCR_LOOP)
314 mii->mii_media_active |= IFM_LOOP;
315
316 if (bmcr & BMCR_AUTOEN) {
317 if ((bmsr & BMSR_ACOMP) == 0) {
318 /* Erg, still trying, I guess... */
319 mii->mii_media_active |= IFM_NONE;
320 return;
321 }
322 }
323
324 PHY_READ(sc, CIPHY_MII_AUXCSR, &bmsr);
325 switch (bmsr & CIPHY_AUXCSR_SPEED) {
326 case CIPHY_SPEED10:
327 mii->mii_media_active |= IFM_10_T;
328 break;
329 case CIPHY_SPEED100:
330 mii->mii_media_active |= IFM_100_TX;
331 break;
332 case CIPHY_SPEED1000:
333 mii->mii_media_active |= IFM_1000_T;
334 break;
335 default:
336 aprint_error_dev(sc->mii_dev, "unknown PHY speed %x\n",
337 bmsr & CIPHY_AUXCSR_SPEED);
338 break;
339 }
340
341 if (bmsr & CIPHY_AUXCSR_FDX)
342 mii->mii_media_active |= IFM_FDX;
343 else
344 mii->mii_media_active |= IFM_HDX;
345
346 return;
347}

349static void
350ciphy_reset(struct mii_softc *sc)
351{
352 mii_phy_reset(sc);
353 DELAY(1000);
354
355 return;
356}

358static inline int
359PHY_SETBIT(struct mii_softc *sc, int y, uint16_t z)
360{
361 uint16_t _tmp;
362 int rv;
363
364 if ((rv = PHY_READ(sc, y, &_tmp)) != 0)
365 return rv;
366 return PHY_WRITE(sc, y, _tmp | z);
367}
368
369static inline int
370PHY_CLRBIT(struct mii_softc *sc, int y, uint16_t z)
371{
372 uint16_t _tmp;
373 int rv;
374
375 if ((rv = PHY_READ(sc, y, &_tmp)) != 0)
376 return rv;
377 return PHY_WRITE(sc, y, _tmp & ~z);
378}

380static void
381ciphy_fixup(struct mii_softc *sc)
382{
383 uint16_t model, status, speed;
384 uint16_t reg;
385
386 PHY_READ(sc, MII_PHYIDR2, &reg);
387 model = MII_MODEL(reg);
388 PHY_READ(sc, CIPHY_MII_AUXCSR, &status);
389 speed = status & CIPHY_AUXCSR_SPEED;
390
391 if (device_is_a(device_parent(sc->mii_dev), "nfe")) {
392 /* Need to set for 2.5V RGMII for NVIDIA adapters */
393 PHY_SETBIT(sc, CIPHY_MII_ECTL1, CIPHY_INTSEL_RGMII);
394 PHY_SETBIT(sc, CIPHY_MII_ECTL1, CIPHY_IOVOL_2500MV);
395 }
396
397 switch (model) {
398 case MII_MODEL_CICADA_CS8201:
399 case MII_MODEL_CICADA_CS8204:
400 /* Turn off "aux mode" (whatever that means) */
401 PHY_SETBIT(sc, CIPHY_MII_AUXCSR, CIPHY_AUXCSR_MDPPS);
402
403 /*
404 * Work around speed polling bug in VT3119/VT3216
405 * when using MII in full duplex mode.
406 */
407 if ((speed == CIPHY_SPEED10 || speed == CIPHY_SPEED100) &&
408 (status & CIPHY_AUXCSR_FDX))
409 PHY_SETBIT(sc, CIPHY_MII_10BTCSR, CIPHY_10BTCSR_ECHO);
410 else
411 PHY_CLRBIT(sc, CIPHY_MII_10BTCSR, CIPHY_10BTCSR_ECHO);
412
413 /* Enable link/activity LED blink. */
414 PHY_SETBIT(sc, CIPHY_MII_LED, CIPHY_LED_LINKACTBLINK);
415
416 break;
417
418 case MII_MODEL_CICADA_CS8201A:
419 case MII_MODEL_CICADA_CS8201B:
420 /*
421 * Work around speed polling bug in VT3119/VT3216
422 * when using MII in full duplex mode.
423 */
424 if ((speed == CIPHY_SPEED10 || speed == CIPHY_SPEED100) &&
425 (status & CIPHY_AUXCSR_FDX))
426 PHY_SETBIT(sc, CIPHY_MII_10BTCSR, CIPHY_10BTCSR_ECHO);
427 else
428 PHY_CLRBIT(sc, CIPHY_MII_10BTCSR, CIPHY_10BTCSR_ECHO);
429
430 break;
431 case MII_MODEL_CICADA_VSC8211:
432 case MII_MODEL_CICADA_CS8244:
433 case MII_MODEL_VITESSE_VSC8601:
434 break;
435 default:
436 aprint_error_dev(sc->mii_dev, "unknown CICADA PHY model %x\n",
437 model);
438 break;
439 }
440
441 return;
442}
