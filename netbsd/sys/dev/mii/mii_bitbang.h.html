33#define MII_BIT_MDO 0 /* data out (host->PHY) */
34#define MII_BIT_MDI 1 /* data in (PHY->host) */
35#define MII_BIT_MDC 2 /* clock */
36#define MII_BIT_DIR_HOST_PHY 3 /* set direction: host->PHY */
37#define MII_BIT_DIR_PHY_HOST 4 /* set direction: PHY->host */
38#define MII_NBITS 5
39
40struct mii_bitbang_ops {
41 uint32_t (*mbo_read)(device_t);
42 void (*mbo_write)(device_t, uint32_t);
43 uint32_t mbo_bits[MII_NBITS];
44};
45
46typedef const struct mii_bitbang_ops *mii_bitbang_ops_t;
47
48int mii_bitbang_readreg(device_t, mii_bitbang_ops_t, int, int, uint16_t *);
49int mii_bitbang_writereg(device_t, mii_bitbang_ops_t, int, int, uint16_t);
