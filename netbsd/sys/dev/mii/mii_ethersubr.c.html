80int
81ether_mediachange(struct ifnet *ifp)
82{
83 struct ethercom *ec = (struct ethercom *)ifp;
84 int rc;
85
86 KASSERT(ec->ec_mii != NULL);
87
88 if ((ifp->if_flags & IFF_UP) == 0)
89 return 0;
90 if ((rc = mii_mediachg(ec->ec_mii)) == ENXIO)
91 return 0;
92 return rc;
93}
94
95void
96ether_mediastatus(struct ifnet *ifp, struct ifmediareq *ifmr)
97{
98 struct ethercom *ec = (struct ethercom *)ifp;
99 struct mii_data *mii;
100
101 KASSERT(ec->ec_mii != NULL);
102
103#ifdef notyet
104 if ((ifp->if_flags & IFF_RUNNING) == 0) {
105 ifmr->ifm_active = IFM_ETHER | IFM_NONE;
106 ifmr->ifm_status = 0;
107 return;
108 }
109#endif
110
111 mii = ec->ec_mii;
112
113 mii_pollstat(mii);
114 ifmr->ifm_active = mii->mii_media_active;
115 ifmr->ifm_status = mii->mii_media_status;
116}
