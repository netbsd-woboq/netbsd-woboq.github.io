64
65const char * mii_get_descr_real(int, int);
66
67MODULE(MODULE_CLASS_MISC, miiverbose, NULL);
68
69static int
70miiverbose_modcmd(modcmd_t cmd, void *arg)
71{
72 static const char *(*saved_mii_get_descr)(int, int);
73
74 switch (cmd) {
75 case MODULE_CMD_INIT:
76 saved_mii_get_descr = mii_get_descr;
77 mii_get_descr = mii_get_descr_real;
78 mii_verbose_loaded = 1;
79 return 0;
80 case MODULE_CMD_FINI:
81 mii_get_descr = saved_mii_get_descr;
82 mii_verbose_loaded = 0;
83 return 0;
84 default:
85 return ENOTTY;
86 }
87}
88
89const char *
90mii_get_descr_real(int oui, int model)
91{
92 int i;
93
94 for (i = 0; mii_knowndevs[i].descr != NULL; i++)
95 if (mii_knowndevs[i].oui == oui &&
96 mii_knowndevs[i].model == model)
97 break;
98 return mii_knowndevs[i].descr;
99}
