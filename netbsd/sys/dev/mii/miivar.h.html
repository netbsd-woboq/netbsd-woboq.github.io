56/*
57 * A network interface driver has one of these structures in its softc.
58 * It is the interface from the network interface driver to the MII
59 * layer.
60 */
61struct mii_data {
62 struct ifmedia mii_media; /* media information */
63 struct ifnet *mii_ifp; /* pointer back to network interface */
64
65 int mii_flags; /* misc. flags; see below */
66
67 /*
68 * For network interfaces with multiple PHYs, a list of all
69 * PHYs is required so they can all be notified when a media
70 * request is made.
71 */
72 LIST_HEAD(mii_listhead, mii_softc) mii_phys;
73 u_int mii_instance;
74
75 /* PHY driver fills this in with active media status. */
76 int mii_media_status;
77 u_int mii_media_active;
78
79 /* Calls from MII layer into network interface driver. */
80 mii_readreg_t mii_readreg;
81 mii_writereg_t mii_writereg;
82 mii_statchg_t mii_statchg;
83};
84typedef struct mii_data mii_data_t;
85
86/*
87 * Functions provided by the PHY to perform various functions.
88 */
89struct mii_phy_funcs {
90 int (*pf_service)(struct mii_softc *, struct mii_data *, int);
91 void (*pf_status)(struct mii_softc *);
92 void (*pf_reset)(struct mii_softc *);
93};

95/*
96 * Requests that can be made to the downcall.
97 */
98#define MII_TICK 1 /* once-per-second tick */
99#define MII_MEDIACHG 2 /* user changed media; perform the switch */
100#define MII_POLLSTAT 3 /* user requested media status; fill it in */
101#define MII_DOWN 4 /* interface is down */
102
103/*
104 * Each PHY driver's softc has one of these as the first member.
105 * XXX This would be better named "phy_softc", but this is the name
106 * XXX BSDI used, and we would like to have the same interface.
107 */
108struct mii_softc {
109 device_t mii_dev; /* generic device glue */
110
111 LIST_ENTRY(mii_softc) mii_list; /* entry on parent's PHY list */
112
113 uint32_t mii_mpd_oui; /* the PHY's OUI (MII_OUI())*/
114 uint32_t mii_mpd_model; /* the PHY's model (MII_MODEL())*/
115 uint32_t mii_mpd_rev; /* the PHY's revision (MII_REV())*/
116 int mii_phy; /* our MII address */
117 int mii_offset; /* first PHY, second PHY, etc. */
118 u_int mii_inst; /* instance for ifmedia */
119
120 /* Our PHY functions. */
121 const struct mii_phy_funcs *mii_funcs;
122
123 struct mii_data *mii_pdata; /* pointer to parent's mii_data */
124
125 int mii_flags; /* misc. flags; see below */
126 uint16_t mii_capabilities; /* capabilities from BMSR */
127 uint16_t mii_extcapabilities; /* extended capabilities from EXTSR */
128 int mii_ticks; /* MII_TICK counter */
129 int mii_anegticks; /* ticks before retrying aneg */
130
131 struct callout mii_nway_ch; /* NWAY callout */
132
133 u_int mii_media_active; /* last active media */
134 int mii_media_status; /* last active status */
135};
136typedef struct mii_softc mii_softc_t;

138/* Default mii_anegticks values. */
139#define MII_ANEGTICKS 5

142/* mii_flags */
143#define MIIF_INITDONE 0x0001 /* has been initialized (mii_data) */
144#define MIIF_NOISOLATE 0x0002 /* do not isolate the PHY */
145#define MIIF_NOLOOP 0x0004 /* no loopback capability */
146#define MIIF_DOINGAUTO 0x0008 /* doing autonegotiation (mii_softc) */
147#define MIIF_AUTOTSLEEP 0x0010 /* use tsleep(), not callout() */
148#define MIIF_HAVEFIBER 0x0020 /* from parent: has fiber interface */
149#define MIIF_HAVE_GTCR 0x0040 /* has 100base-T2/1000base-T CR */
150#define MIIF_IS_1000X 0x0080 /* is a 1000BASE-X device */
151#define MIIF_DOPAUSE 0x0100 /* advertise PAUSE capability */
152#define MIIF_IS_HPNA 0x0200 /* is a HomePNA device */
153#define MIIF_FORCEANEG 0x0400 /* force auto-negotiation */
154
155#define MIIF_INHERIT_MASK (MIIF_NOISOLATE | MIIF_NOLOOP | MIIF_AUTOTSLEEP)

222/*
223 * Setup MDD indirect access. Set device address and register number.
224 * "addr" variable takes an address ORed with the function (MMDACR_FN_*).
225 *
226 */
227static inline int
228MMD_INDIRECT(struct mii_softc *sc, uint16_t daddr, uint16_t regnum)
229{
230 int rv;
231
232 /*
233 * Set the MMD device address and set the access mode (function)
234 * to address.
235 */
236 if ((rv = PHY_WRITE(sc, MII_MMDACR, (daddr & ~MMDACR_FUNCMASK))) != 0)
237 return rv;
238
239 /* Set the register number */
240 if ((rv = PHY_WRITE(sc, MII_MMDAADR, regnum)) != 0)
241 return rv;
242
243 /* Set the access mode (function) */
244 rv = PHY_WRITE(sc, MII_MMDACR, daddr);
245
246 return rv;
247}
248
249static inline int
250MMD_INDIRECT_READ(struct mii_softc *sc, uint16_t daddr, uint16_t regnum,
251 uint16_t *valp)
252{
253 int rv;
254
255 if ((rv = MMD_INDIRECT(sc, daddr, regnum)) != 0)
256 return rv;
257
258 return PHY_READ(sc, MII_MMDAADR, valp);
259}
260
261static inline int
262MMD_INDIRECT_WRITE(struct mii_softc *sc, uint16_t daddr, uint16_t regnum,
263 uint16_t val)
264{
265 int rv;
266
267 if ((rv = MMD_INDIRECT(sc, daddr, regnum)) != 0)
268 return rv;
269
270 return PHY_WRITE(sc, MII_MMDAADR, val);
271}
