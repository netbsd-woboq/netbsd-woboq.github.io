92static int
93rgephy_match(device_t parent, cfdata_t match, void *aux)
94{
95 struct mii_attach_args *ma = aux;
96
97 if (mii_phy_match(ma, rgephys) != NULL)
98 return 10;
99
100 return 0;
101}

103static void
104rgephy_attach(device_t parent, device_t self, void *aux)
105{
106 struct rgephy_softc *rsc = device_private(self);
107 prop_dictionary_t prop = device_properties(self);
108 struct mii_softc *sc = &rsc->mii_sc;
109 struct mii_attach_args *ma = aux;
110 struct mii_data *mii = ma->mii_data;
111 const struct mii_phydesc *mpd;
112 int rev;
113 const char *sep = "";
114
115 rev = MII_REV(ma->mii_id2);
116 mpd = mii_phy_match(ma, rgephys);
117 aprint_naive(": Media interface\n");
118
119 sc->mii_dev = self;
120 sc->mii_inst = mii->mii_instance;
121 sc->mii_phy = ma->mii_phyno;
122 sc->mii_mpd_oui = MII_OUI(ma->mii_id1, ma->mii_id2);
123 sc->mii_mpd_model = MII_MODEL(ma->mii_id2);
124 sc->mii_mpd_rev = MII_REV(ma->mii_id2);
125
126 if (sc->mii_mpd_model == MII_MODEL_REALTEK_RTL8169S) {
127 aprint_normal(": RTL8211");
128 if (sc->mii_mpd_rev != 0)
129 aprint_normal("%c",'@' + sc->mii_mpd_rev);
130 aprint_normal(" 1000BASE-T media interface\n");
131 } else
132 aprint_normal(": %s, rev. %d\n", mpd->mpd_name, rev);
133
134 sc->mii_pdata = mii;
135 sc->mii_flags = ma->mii_flags;
136 sc->mii_anegticks = MII_ANEGTICKS_GIGE;
137
138 sc->mii_funcs = &rgephy_funcs;
139
140 prop_dictionary_get_bool(prop, "no-rx-delay", &rsc->mii_no_rx_delay);
141
142#define ADD(m, c) ifmedia_add(&mii->mii_media, (m), (c), NULL)
143#define PRINT(n) aprint_normal("%s%s", sep, (n)); sep = ", "
144
145#ifdef __FreeBSD__
146 ADD(IFM_MAKEWORD(IFM_ETHER, IFM_100_TX, IFM_LOOP, sc->mii_inst),
147 BMCR_LOOP | BMCR_S100);
148#endif
149
150 PHY_READ(sc, MII_BMSR, &sc->mii_capabilities);
151 sc->mii_capabilities &= ma->mii_capmask;
152 sc->mii_capabilities &= ~BMSR_ANEG;
153
154 /*
155 * FreeBSD does not check EXSTAT, but instead adds gigabit
156 * media explicitly. Why?
157 */
158 aprint_normal_dev(self, "");
159 if (sc->mii_capabilities & BMSR_EXTSTAT)
160 PHY_READ(sc, MII_EXTSR, &sc->mii_extcapabilities);
161
162 mii_phy_add_media(sc);
163
164 /* rtl8169S does not report auto-sense; add manually. */
165 ADD(IFM_MAKEWORD(IFM_ETHER, IFM_AUTO, 0, sc->mii_inst), MII_NMEDIA);
166 sep =", ";
167 PRINT("auto");
168
169#undef ADD
170#undef PRINT
171
172 rgephy_reset(sc);
173 aprint_normal("\n");
174}

176static int
177rgephy_service(struct mii_softc *sc, struct mii_data *mii, int cmd)
178{
179 struct ifmedia_entry *ife = mii->mii_media.ifm_cur;
180 uint16_t reg, speed, gig, anar;
181
182 switch (cmd) {
183 case MII_POLLSTAT:
184 /* If we're not polling our PHY instance, just return. */
185 if (IFM_INST(ife->ifm_media) != sc->mii_inst)
186 return 0;
187 break;
188
189 case MII_MEDIACHG:
190 /*
191 * If the media indicates a different PHY instance,
192 * isolate ourselves.
193 */
194 if (IFM_INST(ife->ifm_media) != sc->mii_inst) {
195 PHY_READ(sc, MII_BMCR, &reg);
196 PHY_WRITE(sc, MII_BMCR, reg | BMCR_ISO);
197 return 0;
198 }
199
200 /* If the interface is not up, don't do anything. */
201 if ((mii->mii_ifp->if_flags & IFF_UP) == 0)
202 break;
203
204 rgephy_reset(sc); /* XXX hardware bug work-around */
205
206 PHY_READ(sc, MII_ANAR, &anar);
207 anar &= ~(ANAR_TX_FD | ANAR_TX | ANAR_10_FD | ANAR_10);
208
209 switch (IFM_SUBTYPE(ife->ifm_media)) {
210 case IFM_AUTO:
211#ifdef foo
212 /* If we're already in auto mode, just return. */
213 PHY_READ(sc, MII_BMCR, &reg);
214 if (reg & BMCR_AUTOEN)
215 return 0;
216#endif
217 (void)rgephy_mii_phy_auto(sc);
218 break;
219 case IFM_1000_T:
220 speed = BMCR_S1000;
221 goto setit;
222 case IFM_100_TX:
223 speed = BMCR_S100;
224 anar |= ANAR_TX_FD | ANAR_TX;
225 goto setit;
226 case IFM_10_T:
227 speed = BMCR_S10;
228 anar |= ANAR_10_FD | ANAR_10;
229 setit:
230 rgephy_loop(sc);
231 if ((ife->ifm_media & IFM_FDX) != 0) {
232 speed |= BMCR_FDX;
233 gig = GTCR_ADV_1000TFDX;
234 anar &= ~(ANAR_TX | ANAR_10);
235 } else {
236 gig = GTCR_ADV_1000THDX;
237 anar &= ~(ANAR_TX_FD | ANAR_10_FD);
238 }
239
240 if (IFM_SUBTYPE(ife->ifm_media) != IFM_1000_T) {
241 PHY_WRITE(sc, MII_100T2CR, 0);
242 PHY_WRITE(sc, MII_ANAR, anar);
243 PHY_WRITE(sc, MII_BMCR,
244 speed | BMCR_AUTOEN | BMCR_STARTNEG);
245 break;
246 }
247
248 /*
249 * When setting the link manually, one side must be the
250 * master and the other the slave. However ifmedia
251 * doesn't give us a good way to specify this, so we
252 * fake it by using one of the LINK flags. If LINK0 is
253 * set, we program the PHY to be a master, otherwise
254 * it's a slave.
255 */
256 if ((mii->mii_ifp->if_flags & IFF_LINK0)) {
257 PHY_WRITE(sc, MII_100T2CR,
258 gig | GTCR_MAN_MS | GTCR_ADV_MS);
259 } else
260 PHY_WRITE(sc, MII_100T2CR, gig | GTCR_MAN_MS);
261 PHY_WRITE(sc, MII_BMCR,
262 speed | BMCR_AUTOEN | BMCR_STARTNEG);
263 break;
264 case IFM_NONE:
265 PHY_WRITE(sc, MII_BMCR, BMCR_ISO | BMCR_PDOWN);
266 break;
267 case IFM_100_T4:
268 default:
269 return EINVAL;
270 }
271 break;
272
273 case MII_TICK:
274 /* If we're not currently selected, just return. */
275 if (IFM_INST(ife->ifm_media) != sc->mii_inst)
276 return 0;
277
278 /* Is the interface even up? */
279 if ((mii->mii_ifp->if_flags & IFF_UP) == 0)
280 return 0;
281
282 /* Only used for autonegotiation. */
283 if ((IFM_SUBTYPE(ife->ifm_media) != IFM_AUTO) &&
284 (IFM_SUBTYPE(ife->ifm_media) != IFM_1000_T)) {
285 /*
286 * Reset autonegotiation timer to 0 to make sure
287 * the future autonegotiation start with 0.
288 */
289 sc->mii_ticks = 0;
290 break;
291 }
292
293 /*
294 * Check to see if we have link. If we do, we don't
295 * need to restart the autonegotiation process. Read
296 * the BMSR twice in case it's latched.
297 */
298 if (sc->mii_mpd_rev >= RGEPHY_8211F) {
299 /* RTL8211F */
300 PHY_READ(sc, RGEPHY_MII_PHYSR, &reg);
301 if (reg & RGEPHY_PHYSR_LINK) {
302 sc->mii_ticks = 0;
303 break;
304 }
305 } else if (sc->mii_mpd_rev >= RGEPHY_8211B) {
306 /* RTL8211B(L) */
307 PHY_READ(sc, RGEPHY_MII_SSR, &reg);
308 if (reg & RGEPHY_SSR_LINK) {
309 sc->mii_ticks = 0;
310 break;
311 }
312 } else {
313 PHY_READ(sc, RTK_GMEDIASTAT, &reg);
314 if ((reg & RTK_GMEDIASTAT_LINK) != 0) {
315 sc->mii_ticks = 0;
316 break;
317 }
318 }
319
320 /* Announce link loss right after it happens. */
321 if (sc->mii_ticks++ == 0)
322 break;
323
324 /* Only retry autonegotiation every mii_anegticks seconds. */
325 if (sc->mii_ticks <= sc->mii_anegticks)
326 return 0;
327
328 rgephy_mii_phy_auto(sc);
329 break;
330 }
331
332 /* Update the media status. */
333 rgephy_status(sc);
334
335 /*
336 * Callback if something changed. Note that we need to poke
337 * the DSP on the RealTek PHYs if the media changes.
338 */
339 if (sc->mii_media_active != mii->mii_media_active ||
340 sc->mii_media_status != mii->mii_media_status ||
341 cmd == MII_MEDIACHG) {
342 rgephy_load_dspcode(sc);
343 }
344 mii_phy_update(sc, cmd);
345 return 0;
346}

348static void
349rgephy_status(struct mii_softc *sc)
350{
351 struct mii_data *mii = sc->mii_pdata;
352 uint16_t gstat, bmsr, bmcr, physr, ssr;
353
354 mii->mii_media_status = IFM_AVALID;
355 mii->mii_media_active = IFM_ETHER;
356
357 if (sc->mii_mpd_rev >= RGEPHY_8211F) {
358 PHY_READ(sc, RGEPHY_MII_PHYSR, &physr);
359 if (physr & RGEPHY_PHYSR_LINK)
360 mii->mii_media_status |= IFM_ACTIVE;
361 } else if (sc->mii_mpd_rev >= RGEPHY_8211B) {
362 PHY_READ(sc, RGEPHY_MII_SSR, &ssr);
363 if (ssr & RGEPHY_SSR_LINK)
364 mii->mii_media_status |= IFM_ACTIVE;
365 } else {
366 PHY_READ(sc, RTK_GMEDIASTAT, &gstat);
367 if ((gstat & RTK_GMEDIASTAT_LINK) != 0)
368 mii->mii_media_status |= IFM_ACTIVE;
369 }
370
371 PHY_READ(sc, MII_BMSR, &bmsr);
372 PHY_READ(sc, MII_BMCR, &bmcr);
373
374 if ((bmcr & BMCR_ISO) != 0) {
375 mii->mii_media_active |= IFM_NONE;
376 mii->mii_media_status = 0;
377 return;
378 }
379
380 if ((bmcr & BMCR_LOOP) != 0)
381 mii->mii_media_active |= IFM_LOOP;
382
383 if ((bmcr & BMCR_AUTOEN) != 0) {
384 if ((bmsr & BMSR_ACOMP) == 0) {
385 /* Erg, still trying, I guess... */
386 mii->mii_media_active |= IFM_NONE;
387 return;
388 }
389 }
390
391 if (sc->mii_mpd_rev >= RGEPHY_8211F) {
392 PHY_READ(sc, RGEPHY_MII_PHYSR, &physr);
393 switch (__SHIFTOUT(physr, RGEPHY_PHYSR_SPEED)) {
394 case RGEPHY_PHYSR_SPEED_1000:
395 mii->mii_media_active |= IFM_1000_T;
396 break;
397 case RGEPHY_PHYSR_SPEED_100:
398 mii->mii_media_active |= IFM_100_TX;
399 break;
400 case RGEPHY_PHYSR_SPEED_10:
401 mii->mii_media_active |= IFM_10_T;
402 break;
403 default:
404 mii->mii_media_active |= IFM_NONE;
405 break;
406 }
407 if (physr & RGEPHY_PHYSR_DUPLEX)
408 mii->mii_media_active |= mii_phy_flowstatus(sc) |
409 IFM_FDX;
410 else
411 mii->mii_media_active |= IFM_HDX;
412 } else if (sc->mii_mpd_rev >= RGEPHY_8211B) {
413 PHY_READ(sc, RGEPHY_MII_SSR, &ssr);
414 switch (ssr & RGEPHY_SSR_SPD_MASK) {
415 case RGEPHY_SSR_S1000:
416 mii->mii_media_active |= IFM_1000_T;
417 break;
418 case RGEPHY_SSR_S100:
419 mii->mii_media_active |= IFM_100_TX;
420 break;
421 case RGEPHY_SSR_S10:
422 mii->mii_media_active |= IFM_10_T;
423 break;
424 default:
425 mii->mii_media_active |= IFM_NONE;
426 break;
427 }
428 if (ssr & RGEPHY_SSR_FDX)
429 mii->mii_media_active |= mii_phy_flowstatus(sc) |
430 IFM_FDX;
431 else
432 mii->mii_media_active |= IFM_HDX;
433 } else {
434 PHY_READ(sc, RTK_GMEDIASTAT, &gstat);
435 if ((gstat & RTK_GMEDIASTAT_1000MBPS) != 0)
436 mii->mii_media_active |= IFM_1000_T;
437 else if ((gstat & RTK_GMEDIASTAT_100MBPS) != 0)
438 mii->mii_media_active |= IFM_100_TX;
439 else if ((gstat & RTK_GMEDIASTAT_10MBPS) != 0)
440 mii->mii_media_active |= IFM_10_T;
441 else
442 mii->mii_media_active |= IFM_NONE;
443 if ((gstat & RTK_GMEDIASTAT_FDX) != 0)
444 mii->mii_media_active |= mii_phy_flowstatus(sc) |
445 IFM_FDX;
446 else
447 mii->mii_media_active |= IFM_HDX;
448 }
449}

451static int
452rgephy_mii_phy_auto(struct mii_softc *mii)
453{
454 int anar;
455
456 mii->mii_ticks = 0;
457 rgephy_loop(mii);
458 rgephy_reset(mii);
459
460 anar = BMSR_MEDIA_TO_ANAR(mii->mii_capabilities) | ANAR_CSMA;
461 if (mii->mii_flags & MIIF_DOPAUSE)
462 anar |= ANAR_FC | ANAR_PAUSE_ASYM;
463
464 PHY_WRITE(mii, MII_ANAR, anar);
465 DELAY(1000);
466 PHY_WRITE(mii, MII_100T2CR, GTCR_ADV_1000THDX | GTCR_ADV_1000TFDX);
467 DELAY(1000);
468 PHY_WRITE(mii, MII_BMCR, BMCR_AUTOEN | BMCR_STARTNEG);
469 DELAY(100);
470
471 return EJUSTRETURN;
472}

474static void
475rgephy_loop(struct mii_softc *sc)
476{
477 uint16_t bmsr;
478 int i;
479
480 if (sc->mii_mpd_model != MII_MODEL_REALTEK_RTL8251 &&
481 sc->mii_mpd_rev < RGEPHY_8211B) {
482 PHY_WRITE(sc, MII_BMCR, BMCR_PDOWN);
483 DELAY(1000);
484 }
485
486 for (i = 0; i < 15000; i++) {
487 PHY_READ(sc, MII_BMSR, &bmsr);
488 if ((bmsr & BMSR_LINK) == 0) {
489#if 0
490 device_printf(sc->mii_dev, "looped %d\n", i);
491#endif
492 break;
493 }
494 DELAY(10);
495 }
496}

498static inline int
499PHY_SETBIT(struct mii_softc *sc, int y, uint16_t z)
500{
501 uint16_t _tmp;
502 int rv;
503
504 if ((rv = PHY_READ(sc, y, &_tmp)) != 0)
505 return rv;
506 return PHY_WRITE(sc, y, _tmp | z);
507}
508
509static inline int
510PHY_CLRBIT(struct mii_softc *sc, int y, uint16_t z)
511{
512 uint16_t _tmp;
513 int rv;
514
515 if ((rv = PHY_READ(sc, y, &_tmp)) != 0)
516 return rv;
517 return PHY_WRITE(sc, y, _tmp & ~z);
518}

524 * PHY. Later versions work without any fixups.

629static void
630rgephy_reset(struct mii_softc *sc)
631{
632 struct rgephy_softc *rsc = (struct rgephy_softc *)sc;
633 uint16_t ssr, phycr1;
634
635 mii_phy_reset(sc);
636 DELAY(1000);
637
638 if (sc->mii_mpd_model != MII_MODEL_REALTEK_RTL8251 &&
639 sc->mii_mpd_rev < RGEPHY_8211B) {
640 rgephy_load_dspcode(sc);
641 } else if (sc->mii_mpd_rev == RGEPHY_8211C) {
642 /* RTL8211C(L) */
643 PHY_READ(sc, RGEPHY_MII_SSR, &ssr);
644 if ((ssr & RGEPHY_SSR_ALDPS) != 0) {
645 ssr &= ~RGEPHY_SSR_ALDPS;
646 PHY_WRITE(sc, RGEPHY_MII_SSR, ssr);
647 }
648 } else if (sc->mii_mpd_rev == RGEPHY_8211E) {
649 /* RTL8211E */
650 if (rsc->mii_no_rx_delay) {
651 /* Disable RX internal delay (undocumented) */
652 PHY_WRITE(sc, 0x1f, 0x0007);
653 PHY_WRITE(sc, 0x1e, 0x00a4);
654 PHY_WRITE(sc, 0x1c, 0xb591);
655 PHY_WRITE(sc, 0x1f, 0x0000);
656 }
657 } else if (sc->mii_mpd_rev == RGEPHY_8211F) {
658 /* RTL8211F */
659 PHY_READ(sc, RGEPHY_MII_PHYCR1, &phycr1);
660 phycr1 &= ~RGEPHY_PHYCR1_MDI_MMCE;
661 phycr1 &= ~RGEPHY_PHYCR1_ALDPS_EN;
662 PHY_WRITE(sc, RGEPHY_MII_PHYCR1, phycr1);
663 } else {
664 PHY_WRITE(sc, 0x1F, 0x0000);
665 PHY_WRITE(sc, 0x0e, 0x0000);
666 }
667
668 /* Reset capabilities */
669 /* Step1: write our capability */
670 /* 10/100 capability */
671 PHY_WRITE(sc, MII_ANAR,
672 ANAR_TX_FD | ANAR_TX | ANAR_10_FD | ANAR_10 | ANAR_CSMA);
673 /* 1000 capability */
674 PHY_WRITE(sc, MII_100T2CR, GTCR_ADV_1000TFDX | GTCR_ADV_1000THDX);
675
676 /* Step2: Restart NWay */
677 /* NWay enable and Restart NWay */
678 PHY_WRITE(sc, MII_BMCR, BMCR_RESET | BMCR_AUTOEN | BMCR_STARTNEG);
679
680 if (sc->mii_mpd_rev >= RGEPHY_8211D) {
681 /* RTL8211F */
682 delay(10000);
683 /* disable EEE */
684 MMD_INDIRECT_WRITE(sc, MDIO_MMD_AN | MMDACR_FN_DATA,
685 MDIO_AN_EEEADVERT, 0x0000);
686 }
687}
