37#ifndef _DEV_MII_RGEPHYREG_H_
38#define _DEV_MII_RGEPHYREG_H_
39
40#define RGEPHY_8211B 2
41#define RGEPHY_8211C 3
42#define RGEPHY_8211D 4
43#define RGEPHY_8211E 5
44#define RGEPHY_8211F 6
45
46/*
47 * RealTek 8169S/8110S gigE PHY registers
48 */
49
50/* RTL8211B(L)/RTL8211C(L) */
51#define RGEPHY_MII_SSR 0x11 /* PHY Specific status register */
52#define RGEPHY_SSR_S1000 0x8000 /* 1000Mbps */
53#define RGEPHY_SSR_S100 0x4000 /* 100Mbps */
54#define RGEPHY_SSR_S10 0x0000 /* 10Mbps */
55#define RGEPHY_SSR_SPD_MASK 0xc000
56#define RGEPHY_SSR_FDX 0x2000 /* full duplex */
57#define RGEPHY_SSR_PAGE_RECEIVED 0x1000 /* new page received */
58#define RGEPHY_SSR_SPD_DPLX_RESOLVED 0x0800 /* speed/duplex resolved */
59#define RGEPHY_SSR_LINK 0x0400 /* link up */
60#define RGEPHY_SSR_MDI_XOVER 0x0040 /* MDI crossover */
61#define RGEPHY_SSR_ALDPS 0x0008 /* RTL8211C(L) only */
62#define RGEPHY_SSR_JABBER 0x0001 /* Jabber */
63
64/* RTL8211F */
65#define RGEPHY_MII_PHYCR1 0x18 /* PHY Specific control register 1 */
66#define RGEPHY_PHYCR1_MDI_MMCE __BIT(9)
67#define RGEPHY_PHYCR1_ALDPS_EN __BIT(2)
68
69#define RGEPHY_MII_PHYSR 0x1a /* PHY Specific status register */
70#define RGEPHY_PHYSR_ALDPS __BIT(14)
71#define RGEPHY_PHYSR_MDI_PLUG __BIT(13)
72#define RGEPHY_PHYSR_NWAY_EN __BIT(12)
73#define RGEPHY_PHYSR_MASTER __BIT(11)
74#define RGEPHY_PHYSR_EEE __BIT(8)
75#define RGEPHY_PHYSR_RXFLOW_EN __BIT(7)
76#define RGEPHY_PHYSR_TXFLOW_EN __BIT(6)
77#define RGEPHY_PHYSR_SPEED __BITS(5,4)
78#define RGEPHY_PHYSR_SPEED_10 0
79#define RGEPHY_PHYSR_SPEED_100 1
80#define RGEPHY_PHYSR_SPEED_1000 2
81#define RGEPHY_PHYSR_DUPLEX __BIT(3)
82#define RGEPHY_PHYSR_LINK __BIT(2)
83#define RGEPHY_PHYSR_MDI_XOVER __BIT(1)
84#define RGEPHY_PHYSR_JABBER __BIT(0)
85
86#endif /* _DEV_MII_RGEPHYREG_H_ */
