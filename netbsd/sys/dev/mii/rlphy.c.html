30/*
31 * Driver for the internal PHY found on RTL8139 based nics, based
32 * on drivers for the 'exphy' (Internal 3Com phys) and 'nsphy'
33 * (National Semiconductor DP83840).
34 */
35
36/*
37 * Ported to NetBSD by Juan Romero Pardines <xtraeme@NetBSD.org>
38 */

59struct rlphy_softc {
60 struct mii_softc sc_mii;

64static int rlphymatch(device_t, cfdata_t, void *);

67CFATTACH_DECL_NEW(rlphy, sizeof(struct rlphy_softc),
68 rlphymatch, rlphyattach, mii_phy_detach, mii_phy_activate);

70static int rlphy_service(struct mii_softc *, struct mii_data *, int);

72static void rlphy_reset(struct mii_softc *);

74const struct mii_phy_funcs rlphy_funcs = {
75 rlphy_service, rlphy_status, rlphy_reset,
76};
77
78static const struct mii_phydesc rlphys[] = {
79 MII_PHY_DESC(yyREALTEK, RTL8201L),
80 MII_PHY_DESC(REALTEK, RTL8201E),
81 MII_PHY_DESC(ICPLUS, IP101),
82 MII_PHY_END,
83};
84
85static int
86rlphymatch(device_t parent, cfdata_t match, void *aux)
87{
88 struct mii_attach_args *ma = aux;
89 struct mii_data *mii = ma->mii_data;
90
91 if (mii->mii_instance != 0)
92 return 0;
93
94 if (mii_phy_match(ma, rlphys) != NULL)
95 return 10;
96
97 if (MII_OUI(ma->mii_id1, ma->mii_id2) != 0 ||
98 MII_MODEL(ma->mii_id2) != 0)
99 return 0;
100
101 if (!device_is_a(parent, "rtk") && !device_is_a(parent, "re"))
102 return 0;
103
104 /*
105 * A "real" phy should get preference, but on the 8139 there
106 * is no phyid register.
107 */
108 return 5;
109}
110
111static void
112rlphyattach(device_t parent, device_t self, void *aux)
113{
114 struct rlphy_softc *rsc = device_private(self);
115 struct mii_softc *sc = &rsc->sc_mii;
116 struct mii_attach_args *ma = aux;
117 struct mii_data *mii = ma->mii_data;
118
119 aprint_naive("\n");
120 if (MII_MODEL(ma->mii_id2) == MII_MODEL_yyREALTEK_RTL8201L) {
121 rsc->sc_rtl8201 = 1;
122 aprint_normal(": %s, rev. %d\n", MII_STR_yyREALTEK_RTL8201L,
123 MII_REV(ma->mii_id2));
124 } else if (MII_MODEL(ma->mii_id2) == MII_MODEL_REALTEK_RTL8201E) {
125 rsc->sc_rtl8201 = 1;
126 aprint_normal(": %s, rev. %d\n", MII_STR_REALTEK_RTL8201E,
127 MII_REV(ma->mii_id2));
128 } else
129 aprint_normal(": Realtek internal PHY\n");
130
131 sc->mii_dev = self;
132 sc->mii_inst = mii->mii_instance;
133 sc->mii_phy = ma->mii_phyno;
134 sc->mii_funcs = &rlphy_funcs;
135 sc->mii_pdata = mii;
136 sc->mii_flags = ma->mii_flags;
137
138 sc->mii_flags |= MIIF_NOISOLATE;
139
140 PHY_RESET(sc);
141
142 aprint_normal_dev(self, "");
143 PHY_READ(sc, MII_BMSR, &sc->mii_capabilities);
144 sc->mii_capabilities &= ma->mii_capmask;
145 if (sc->mii_capabilities & BMSR_MEDIAMASK)
146 mii_phy_add_media(sc);
147 aprint_normal("\n");
148}

150static int
151rlphy_service(struct mii_softc *sc, struct mii_data *mii, int cmd)
152{
153 struct ifmedia_entry *ife = mii->mii_media.ifm_cur;
154
155 /* Can't isolate the RTL8139 phy, so it has to be the only one. */
156 if (IFM_INST(ife->ifm_media) != sc->mii_inst)
157 panic("rlphy_service: attempt to isolate phy");
158
159 switch (cmd) {
160 case MII_POLLSTAT:
161 break;
162
163 case MII_MEDIACHG:
164 /* If the interface is not up, don't do anything. */
165 if ((mii->mii_ifp->if_flags & IFF_UP) == 0)
166 break;
167
168 mii_phy_setmedia(sc);
169 break;
170
171 case MII_TICK:
172 /* Is the interface even up? */
173 if ((mii->mii_ifp->if_flags & IFF_UP) == 0)
174 return 0;
175
176 /* Only used for autonegotiation. */
177 if (IFM_SUBTYPE(ife->ifm_media) != IFM_AUTO)
178 break;
179
180 /*
181 * The RealTek PHY's autonegotiation doesn't need to be
182 * kicked; it continues in the background.
183 */
184 break;
185
186 case MII_DOWN:
187 mii_phy_down(sc);
188 return 0;
189 }
190
191 /* Update the media status. */
192 mii_phy_status(sc);
193
194 /* Callback if something changed. */
195 mii_phy_update(sc, cmd);
196 return 0;
197}

199static void
200rlphy_status(struct mii_softc *sc)
201{
202 struct rlphy_softc *rsc = (void *)sc;
203 struct mii_data *mii = sc->mii_pdata;
204 struct ifmedia_entry *ife = mii->mii_media.ifm_cur;
205 uint16_t bmsr, bmcr, anar, anlpar, result, reg;
206
207 mii->mii_media_status = IFM_AVALID;
208 mii->mii_media_active = IFM_ETHER;
209
210 PHY_READ(sc, MII_BMSR, &bmsr);
211 PHY_READ(sc, MII_BMSR, &bmsr);
212 if (bmsr & BMSR_LINK)
213 mii->mii_media_status |= IFM_ACTIVE;
214
215 PHY_READ(sc, MII_BMCR, &bmcr);
216 if (bmcr & BMCR_ISO) {
217 mii->mii_media_active |= IFM_NONE;
218 mii->mii_media_status = 0;
219 return;
220 }
221
222 if (bmcr & BMCR_LOOP)
223 mii->mii_media_active |= IFM_LOOP;
224
225 if (bmcr & BMCR_AUTOEN) {
226 /*
227 * NWay autonegotiation takes the highest-order common
228 * bit of the ANAR and ANLPAR (i.e. best media advertised
229 * both by us and our link partner).
230 */
231 if ((bmsr & BMSR_ACOMP) == 0) {
232 /* Erg, still trying, I guess... */
233 mii->mii_media_active |= IFM_NONE;
234 return;
235 }
236
237 PHY_READ(sc, MII_ANAR, &anar);
238 PHY_READ(sc, MII_ANLPAR, &anlpar);
239 result = anar & anlpar;
240 if (result != 0) {
241 if (result & ANLPAR_TX_FD)
242 mii->mii_media_active |= IFM_100_TX | IFM_FDX;
243 else if (result & ANLPAR_T4)
244 mii->mii_media_active |= IFM_100_T4 | IFM_HDX;
245 else if (result & ANLPAR_TX)
246 mii->mii_media_active |= IFM_100_TX | IFM_HDX;
247 else if (result & ANLPAR_10_FD)
248 mii->mii_media_active |= IFM_10_T | IFM_FDX;
249 else if (result & ANLPAR_10)
250 mii->mii_media_active |= IFM_10_T | IFM_HDX;
251 else
252 mii->mii_media_active |= IFM_NONE;
253 return;
254 }
255
256 /*
257 * If the other side doesn't support NWAY, then the
258 * best we can do is determine if we have a 10Mbps or
259 * 100Mbps link. There's no way to know if the link
260 * is full or half duplex, so we default to half duplex
261 * and hope that the user is clever enough to manually
262 * change the media settings if we're wrong.
263 */
264
265 /*
266 * The RealTek PHY supports non-NWAY link speed
267 * detection, however it does not report the link
268 * detection results via the ANLPAR or BMSR registers.
269 * (What? RealTek doesn't do things the way everyone
270 * else does? I'm just shocked, shocked I tell you.)
271 * To determine the link speed, we have to do one
272 * of two things:
273 *
274 * - If this is a standalone RealTek RTL8201 PHY,
275 * we can determine the link speed by testing bit 0
276 * in the magic, vendor-specific register at offset
277 * 0x19.
278 *
279 * - If this is a RealTek MAC with integrated PHY, we
280 * can test the 'SPEED10' bit of the MAC's media status
281 * register.
282 */
283 if (rsc->sc_rtl8201) {
284 PHY_READ(sc, 0x0019, &reg);
285 if (reg & 0x01)
286 mii->mii_media_active |= IFM_100_TX;
287 else
288 mii->mii_media_active |= IFM_10_T;
289 } else {
290 PHY_READ(sc, RTK_MEDIASTAT, &reg);
291 if (reg & RTK_MEDIASTAT_SPEED10)
292 mii->mii_media_active |= IFM_10_T;
293 else
294 mii->mii_media_active |= IFM_100_TX;
295 }
296 mii->mii_media_active |= IFM_HDX;
297 } else
298 mii->mii_media_active = ife->ifm_media;
299}
300
301static void
302rlphy_reset(struct mii_softc *sc)
303{
304
305 mii_phy_reset(sc);
306
307 /*
308 * XXX RealTek PHY doesn't set the BMCR properly after
309 * XXX reset, which breaks autonegotiation.
310 */
311 PHY_WRITE(sc, MII_BMCR, BMCR_AUTOEN);
312}
