109static const struct mii_phydesc tlphys[] = {
110 MII_PHY_DESC(TI, TLAN10T),
111 MII_PHY_END,
112};

114static int
115tlphymatch(device_t parent, cfdata_t match, void *aux)
116{
117 struct mii_attach_args *ma = aux;
118
119 if (mii_phy_match(ma, tlphys) != NULL)
120 return 10;
121
122 return 0;
123}

125static void
126tlphyattach(device_t parent, device_t self, void *aux)
127{
128 struct tlphy_softc *tsc = device_private(self);
129 struct mii_softc *sc = &tsc->sc_mii;
130 struct tl_softc *tlsc = device_private(device_parent(self));
131 struct mii_attach_args *ma = aux;
132 struct mii_data *mii = ma->mii_data;
133 const struct mii_phydesc *mpd;
134 const char *sep = "";
135
136 mpd = mii_phy_match(ma, tlphys);
137 aprint_naive(": Media interface\n");
138 aprint_normal(": %s, rev. %d\n", mpd->mpd_name, MII_REV(ma->mii_id2));
139
140 sc->mii_dev = self;
141 sc->mii_inst = mii->mii_instance;
142 sc->mii_phy = ma->mii_phyno;
143 sc->mii_funcs = &tlphy_funcs;
144 sc->mii_pdata = mii;
145 sc->mii_flags = ma->mii_flags;
146 sc->mii_anegticks = MII_ANEGTICKS;
147
148 PHY_RESET(sc);
149
150 /*
151 * Note that if we're on a device that also supports 100baseTX, we are
152 * not going to want to use the built-in 10baseT port, since there will
153 * be another PHY on the MII wired up to the UTP connector. The parent
154 * indicates this to us by specifying the TLPHY_MEDIA_NO_10_T bit.
155 */
156 tsc->sc_tlphycap = tlsc->tl_product->tp_tlphymedia;
157 if ((tsc->sc_tlphycap & TLPHY_MEDIA_NO_10_T) == 0) {
158 PHY_READ(sc, MII_BMSR, &sc->mii_capabilities);
159 sc->mii_capabilities &= ma->mii_capmask;
160 } else
161 sc->mii_capabilities = 0;
162
163
164#define ADD(m, c) ifmedia_add(&mii->mii_media, (m), (c), NULL)
165#define PRINT(str) aprint_normal("%s%s", sep, str); sep = ", "
166
167 aprint_normal_dev(self, "");
168 if (tsc->sc_tlphycap) {
169 if (tsc->sc_tlphycap & TLPHY_MEDIA_10_2) {
170 ADD(IFM_MAKEWORD(IFM_ETHER, IFM_10_2, 0, sc->mii_inst),
171 0);
172 PRINT("10base2");
173 } else if (tsc->sc_tlphycap & TLPHY_MEDIA_10_5) {
174 ADD(IFM_MAKEWORD(IFM_ETHER, IFM_10_5, 0, sc->mii_inst),
175 0);
176 PRINT("10base5");
177 }
178 }
179 if (sc->mii_capabilities & BMSR_MEDIAMASK) {
180 aprint_normal("%s", sep);
181 mii_phy_add_media(sc);
182 } else {
183 if ((tsc->sc_tlphycap &
184 (TLPHY_MEDIA_10_2 | TLPHY_MEDIA_10_5)) == 0)
185 aprint_error("no media present");
186 /*
187 * mii_phy_add_media() automatically install power handler,
188 * but if_media_add() doesn't. Do it now.
189 */
190 if (!pmf_device_register(self, NULL, mii_phy_resume)) {
191 aprint_normal("\n");
192 aprint_error_dev(self,
193 "couldn't establish power handler");
194 }
195 }
196 aprint_normal("\n");
197#undef ADD
198#undef PRINT
199}

201static int
202tlphy_service(struct mii_softc *sc, struct mii_data *mii, int cmd)
203{
204 struct tlphy_softc *tsc = (struct tlphy_softc *)sc;
205 struct ifmedia_entry *ife = mii->mii_media.ifm_cur;
206 uint16_t reg;
207
208 if ((sc->mii_flags & MIIF_DOINGAUTO) == 0 && tsc->sc_need_acomp)
209 tlphy_acomp(tsc);
210
211 switch (cmd) {
212 case MII_POLLSTAT:
213 /* If we're not polling our PHY instance, just return. */
214 if (IFM_INST(ife->ifm_media) != sc->mii_inst)
215 return 0;
216 break;
217
218 case MII_MEDIACHG:
219 /*
220 * If the media indicates a different PHY instance,
221 * isolate ourselves.
222 */
223 if (IFM_INST(ife->ifm_media) != sc->mii_inst) {
224 PHY_READ(sc, MII_BMCR, &reg);
225 PHY_WRITE(sc, MII_BMCR, reg | BMCR_ISO);
226 return 0;
227 }
228
229 /* If the interface is not up, don't do anything. */
230 if ((mii->mii_ifp->if_flags & IFF_UP) == 0)
231 break;
232
233 switch (IFM_SUBTYPE(ife->ifm_media)) {
234 case IFM_AUTO:
235 /*
236 * The ThunderLAN PHY doesn't self-configure after
237 * an autonegotiation cycle, so there's no such
238 * thing as "already in auto mode".
239 */
240 (void) tlphy_auto(tsc, 1);
241 break;
242 case IFM_10_2:
243 case IFM_10_5:
244 PHY_WRITE(sc, MII_BMCR, 0);
245 PHY_WRITE(sc, MII_TLPHY_CTRL, CTRL_AUISEL);
246 delay(100000);
247 break;
248 default:
249 PHY_WRITE(sc, MII_TLPHY_CTRL, 0);
250 delay(100000);
251 mii_phy_setmedia(sc);
252 }
253 break;
254
255 case MII_TICK:
256 /* If we're not currently selected, just return. */
257 if (IFM_INST(ife->ifm_media) != sc->mii_inst)
258 return 0;
259
260 /*
261 * XXX WHAT ABOUT CHECKING LINK ON THE BNC/AUI?!
262 */
263
264 if (mii_phy_tick(sc) == EJUSTRETURN)
265 return 0;
266 break;
267
268 case MII_DOWN:
269 mii_phy_down(sc);
270 return 0;
271 }
272
273 /* Update the media status. */
274 mii_phy_status(sc);
275
276 /* Callback if something changed. */
277 mii_phy_update(sc, cmd);
278 return 0;
279}

281static void
282tlphy_status(struct mii_softc *sc)
283{
284 struct tlphy_softc *tsc = (struct tlphy_softc *)sc;
285 struct mii_data *mii = sc->mii_pdata;
286 uint16_t bmsr, bmcr, tlctrl;
287
288 mii->mii_media_status = IFM_AVALID;
289 mii->mii_media_active = IFM_ETHER;
290
291 PHY_READ(sc, MII_BMCR, &bmcr);
292 if (bmcr & BMCR_ISO) {
293 mii->mii_media_active |= IFM_NONE;
294 mii->mii_media_status = 0;
295 return;
296 }
297
298 PHY_READ(sc, MII_TLPHY_CTRL, &tlctrl);
299 if (tlctrl & CTRL_AUISEL) {
300 if (tsc->sc_tlphycap & TLPHY_MEDIA_10_2)
301 mii->mii_media_active |= IFM_10_2;
302 else if (tsc->sc_tlphycap & TLPHY_MEDIA_10_5)
303 mii->mii_media_active |= IFM_10_5;
304 else
305 printf("%s: AUI selected with no matching media !\n",
306 device_xname(sc->mii_dev));
307 mii->mii_media_status |= IFM_ACTIVE;
308 return;
309 }
310
311 PHY_READ(sc, MII_BMSR, &bmsr);
312 PHY_READ(sc, MII_BMSR, &bmsr);
313 if (bmsr & BMSR_LINK)
314 mii->mii_media_status |= IFM_ACTIVE;
315
316 if (bmcr & BMCR_LOOP)
317 mii->mii_media_active |= IFM_LOOP;
318
319 /*
320 * Grr, braindead ThunderLAN PHY doesn't have any way to tell which
321 * media is actually active. (Note it also doesn't self-configure
322 * after autonegotiation.) We just have to report what's in the BMCR.
323 */
324 if (bmcr & BMCR_FDX)
325 mii->mii_media_active |= IFM_FDX;
326 else
327 mii->mii_media_active |= IFM_HDX;
328 mii->mii_media_active |= IFM_10_T;
329}

331static int
332tlphy_auto(struct tlphy_softc *tsc, int waitfor)
333{
334 struct mii_softc *sc = &tsc->sc_mii;
335 int error;
336
337 switch ((error = mii_phy_auto(sc, waitfor))) {
338 case EIO:
339 /*
340 * Just assume we're not in full-duplex mode.
341 * XXX Check link and try AUI/BNC?
342 */
343 PHY_WRITE(sc, MII_BMCR, 0);
344 break;
345
346 case EJUSTRETURN:
347 /* Flag that we need to program when it completes. */
348 tsc->sc_need_acomp = 1;
349 break;
350
351 default:
352 tlphy_acomp(tsc);
353 }
354
355 return error;
356}
357
358static void
359tlphy_acomp(struct tlphy_softc *tsc)
360{
361 struct mii_softc *sc = &tsc->sc_mii;
362 uint16_t aner, anar, anlpar, result;
363
364 tsc->sc_need_acomp = 0;
365
366 /*
367 * Grr, braindead ThunderLAN PHY doesn't self-configure
368 * after autonegotiation. We have to do it ourselves
369 * based on the link partner status.
370 */
371
372 PHY_READ(sc, MII_ANER, &aner);
373 if (aner & ANER_LPAN) {
374 PHY_READ(sc, MII_ANAR, &anar);
375 PHY_READ(sc, MII_ANLPAR, &anlpar);
376 result = anar & anlpar;
377 if (result & ANAR_10_FD) {
378 PHY_WRITE(sc, MII_BMCR, BMCR_FDX);
379 return;
380 }
381 }
382 PHY_WRITE(sc, MII_BMCR, 0);
383}
