81static int tqphymatch(device_t, cfdata_t, void *);
82static void tqphyattach(device_t, device_t, void *);

84CFATTACH_DECL_NEW(tqphy, sizeof(struct mii_softc),
85 tqphymatch, tqphyattach, mii_phy_detach, mii_phy_activate);

87static int tqphy_service(struct mii_softc *, struct mii_data *, int);
88static void tqphy_status(struct mii_softc *);
89
90static const struct mii_phy_funcs tqphy_funcs = {
91 tqphy_service, tqphy_status, mii_phy_reset,
92};
93
94static const struct mii_phydesc tqphys[] = {
95 MII_PHY_DESC(xxTSC, 78Q2120),
96#if 0
97 MII_PHY_DESC(xxTSC, 78Q2121),
98#endif
99 MII_PHY_END,
100};
101
102static int
103tqphymatch(device_t parent, cfdata_t match, void *aux)
104{
105 struct mii_attach_args *ma = aux;
106
107 if (mii_phy_match(ma, tqphys) != NULL) {
108 /* The DIAG register is unreliable on early revisions. */
109 if (MII_MODEL(ma->mii_id2) == MII_MODEL_xxTSC_78Q2120 &&
110 MII_REV(ma->mii_id2) <= 3)
111 return 0;
112 return 10;
113 }
114
115 return 0;
116}
117
118static void
119tqphyattach(device_t parent, device_t self, void *aux)
120{
121 struct mii_softc *sc = device_private(self);
122 struct mii_attach_args *ma = aux;
123 struct mii_data *mii = ma->mii_data;
124 const struct mii_phydesc *mpd;
125
126 mpd = mii_phy_match(ma, tqphys);
127 aprint_naive(": Media interface\n");
128 aprint_normal(": %s, rev. %d\n", mpd->mpd_name, MII_REV(ma->mii_id2));
129
130 sc->mii_dev = self;
131 sc->mii_inst = mii->mii_instance;
132 sc->mii_phy = ma->mii_phyno;
133 sc->mii_funcs = &tqphy_funcs;
134 sc->mii_pdata = mii;
135 sc->mii_flags = ma->mii_flags;
136 sc->mii_anegticks = MII_ANEGTICKS;
137
138 /* Apparently, we can't do loopback on this PHY. */
139 sc->mii_flags |= MIIF_NOLOOP;
140
141 PHY_RESET(sc);
142
143 PHY_READ(sc, MII_BMSR, &sc->mii_capabilities);
144 sc->mii_capabilities &= ma->mii_capmask;
145 aprint_normal_dev(self, "");
146 if ((sc->mii_capabilities & BMSR_MEDIAMASK) == 0)
147 aprint_error("no media present");
148 else
149 mii_phy_add_media(sc);
150 aprint_normal("\n");
151}

153static int
154tqphy_service(struct mii_softc *sc, struct mii_data *mii, int cmd)
155{
156 struct ifmedia_entry *ife = mii->mii_media.ifm_cur;
157 uint16_t reg;
158
159 switch (cmd) {
160 case MII_POLLSTAT:
161 /* If we're not polling our PHY instance, just return. */
162 if (IFM_INST(ife->ifm_media) != sc->mii_inst)
163 return 0;
164 break;
165
166 case MII_MEDIACHG:
167 /*
168 * If the media indicates a different PHY instance,
169 * isolate ourselves.
170 */
171 if (IFM_INST(ife->ifm_media) != sc->mii_inst) {
172 PHY_READ(sc, MII_BMCR, &reg);
173 PHY_WRITE(sc, MII_BMCR, reg | BMCR_ISO);
174 return 0;
175 }
176
177 /* If the interface is not up, don't do anything. */
178 if ((mii->mii_ifp->if_flags & IFF_UP) == 0)
179 break;
180
181 mii_phy_setmedia(sc);
182 break;
183
184 case MII_TICK:
185 /* If we're not currently selected, just return. */
186 if (IFM_INST(ife->ifm_media) != sc->mii_inst)
187 return 0;
188
189 if (mii_phy_tick(sc) == EJUSTRETURN)
190 return 0;
191 break;
192
193 case MII_DOWN:
194 mii_phy_down(sc);
195 return 0;
196 }
197
198 /* Update the media status. */
199 mii_phy_status(sc);
200
201 /* Callback if something changed. */
202 mii_phy_update(sc, cmd);
203 return 0;
204}

206static void
207tqphy_status(struct mii_softc *sc)
208{
209 struct mii_data *mii = sc->mii_pdata;
210 struct ifmedia_entry *ife = mii->mii_media.ifm_cur;
211 uint16_t bmsr, bmcr, diag;
212
213 mii->mii_media_status = IFM_AVALID;
214 mii->mii_media_active = IFM_ETHER;
215
216 PHY_READ(sc, MII_BMSR, &bmsr);
217 PHY_READ(sc, MII_BMSR, &bmsr);
218 if (bmsr & BMSR_LINK)
219 mii->mii_media_status |= IFM_ACTIVE;
220
221 PHY_READ(sc, MII_BMCR, &bmcr);
222 if (bmcr & BMCR_ISO) {
223 mii->mii_media_active |= IFM_NONE;
224 mii->mii_media_status = 0;
225 return;
226 }
227
228 if (bmcr & BMCR_LOOP)
229 mii->mii_media_active |= IFM_LOOP;
230
231 if (bmcr & BMCR_AUTOEN) {
232 if ((bmsr & BMSR_ACOMP) == 0) {
233 /* Erg, still trying, I guess... */
234 mii->mii_media_active |= IFM_NONE;
235 return;
236 }
237 PHY_READ(sc, MII_TQPHY_DIAG, &diag);
238 if (diag & DIAG_RATE)
239 mii->mii_media_active |= IFM_100_TX;
240 else
241 mii->mii_media_active |= IFM_10_T;
242
243 if (diag & DIAG_DPLX)
244 mii->mii_media_active |= IFM_FDX;
245 else
246 mii->mii_media_active |= IFM_HDX;
247 } else
248 mii->mii_media_active = ife->ifm_media;
249}
