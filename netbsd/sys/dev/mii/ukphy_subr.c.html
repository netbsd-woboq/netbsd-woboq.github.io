52/*
53 * Media status subroutine. If a PHY driver does media detection simply
54 * by decoding the NWay autonegotiation, use this routine.
55 */
56void
57ukphy_status(struct mii_softc *phy)
58{
59 struct mii_data *mii = phy->mii_pdata;
60 struct ifmedia_entry *ife = mii->mii_media.ifm_cur;
61 uint16_t bmsr, bmcr, anar, anlpar, gtcr, gtsr, result;
62
63 mii->mii_media_status = IFM_AVALID;
64 mii->mii_media_active = IFM_ETHER;
65
66 PHY_READ(phy, MII_BMSR, &bmsr);
67 PHY_READ(phy, MII_BMSR, &bmsr);
68 if (bmsr & BMSR_LINK)
69 mii->mii_media_status |= IFM_ACTIVE;
70
71 PHY_READ(phy, MII_BMCR, &bmcr);
72 if (bmcr & BMCR_ISO) {
73 mii->mii_media_active |= IFM_NONE;
74 mii->mii_media_status = 0;
75 return;
76 }
77
78 if (bmcr & BMCR_LOOP)
79 mii->mii_media_active |= IFM_LOOP;
80
81 if (bmcr & BMCR_AUTOEN) {
82 /*
83 * NWay autonegotiation takes the highest-order common
84 * bit of the ANAR and ANLPAR (i.e. best media advertised
85 * both by us and our link partner).
86 */
87 if ((bmsr & BMSR_ACOMP) == 0) {
88 /* Erg, still trying, I guess... */
89 mii->mii_media_active |= IFM_NONE;
90 return;
91 }
92
93 PHY_READ(phy, MII_ANAR, &anar);
94 PHY_READ(phy, MII_ANLPAR, &anlpar);
95 result = anar & anlpar;
96 if ((phy->mii_flags & MIIF_HAVE_GTCR) != 0 &&
97 (phy->mii_extcapabilities &
98 (EXTSR_1000THDX | EXTSR_1000TFDX)) != 0) {
99 PHY_READ(phy, MII_100T2CR, &gtcr);
100 PHY_READ(phy, MII_100T2SR, &gtsr);
101 } else
102 gtcr = gtsr = 0;
103
104 if ((gtcr & GTCR_ADV_1000TFDX) && (gtsr & GTSR_LP_1000TFDX))
105 mii->mii_media_active |= IFM_1000_T | IFM_FDX;
106 else if ((gtcr & GTCR_ADV_1000THDX) &&
107 (gtsr & GTSR_LP_1000THDX))
108 mii->mii_media_active |= IFM_1000_T | IFM_HDX;
109 else if (result & ANLPAR_TX_FD)
110 mii->mii_media_active |= IFM_100_TX | IFM_FDX;
111 else if (result & ANLPAR_T4)
112 mii->mii_media_active |= IFM_100_T4 | IFM_HDX;
113 else if (result & ANLPAR_TX)
114 mii->mii_media_active |= IFM_100_TX | IFM_HDX;
115 else if (result & ANLPAR_10_FD)
116 mii->mii_media_active |= IFM_10_T | IFM_FDX;
117 else if (result & ANLPAR_10)
118 mii->mii_media_active |= IFM_10_T | IFM_HDX;
119 else
120 mii->mii_media_active |= IFM_NONE;
121
122 if ((mii->mii_media_active & IFM_1000_T) &&
123 (gtsr & GTSR_MS_RES))
124 mii->mii_media_active |= IFM_ETH_MASTER;
125
126 if (mii->mii_media_active & IFM_FDX)
127 mii->mii_media_active |= mii_phy_flowstatus(phy);
128 } else
129 mii->mii_media_active = ife->ifm_media;
130}
