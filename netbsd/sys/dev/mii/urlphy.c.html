73};

75static int
76urlphy_match(device_t parent, cfdata_t match, void *aux)
77{
78 struct mii_attach_args *ma = aux;
79
80 /* RTL8150 reports OUI == 0, MODEL == 0 */
81 if (MII_OUI(ma->mii_id1, ma->mii_id2) != 0 &&
82 MII_MODEL(ma->mii_id2) != 0)
83 return 0;
84
85 /* Make sure the parent is an 'url' device. */
86 if (!device_is_a(parent, "url"))
87 return 0;
88
89 return 10;
90}

92static void
93urlphy_attach(device_t parent, device_t self, void *aux)
94{
95 struct mii_softc *sc = device_private(self);
96 struct mii_attach_args *ma = aux;
97 struct mii_data *mii = ma->mii_data;
98
99 aprint_naive(": Media interface\n");
100 aprint_normal(": Realtek RTL8150L internal media interface\n");
101
102 DPRINTF(("%s: %s: enter\n", device_xname(self), __func__));
103
104 sc->mii_dev = self;
105 sc->mii_inst = mii->mii_instance;
106 sc->mii_phy = ma->mii_phyno;
107 sc->mii_funcs = &urlphy_funcs;
108 sc->mii_pdata = mii;
109 sc->mii_flags = ma->mii_flags;
110 sc->mii_anegticks = MII_ANEGTICKS_GIGE;
111
112 /* Don't do loopback on this PHY. */
113 sc->mii_flags |= MIIF_NOLOOP;
114 /* Don't do isolate on this PHY. */
115 sc->mii_flags |= MIIF_NOISOLATE;
116
117 if (mii->mii_instance != 0) {
118 aprint_error_dev(self,
119 "ignoring this PHY, non-zero instance\n");
120 return;
121 }
122 PHY_RESET(sc);
123
124 PHY_READ(sc, MII_BMSR, &sc->mii_capabilities);
125 sc->mii_capabilities &= ma->mii_capmask;
126 aprint_normal_dev(self, "");
127 if ((sc->mii_capabilities & BMSR_MEDIAMASK) == 0)
128 aprint_error("no media present");
129 else
130 mii_phy_add_media(sc);
131 aprint_normal("\n");
132}

134static int
135urlphy_service(struct mii_softc *sc, struct mii_data *mii, int cmd)
136{
137 struct ifmedia_entry *ife = mii->mii_media.ifm_cur;
138 uint16_t reg;
139
140 DPRINTF(("%s: %s: enter\n", device_xname(sc->mii_dev), __func__));
141
142 switch (cmd) {
143 case MII_POLLSTAT:
144 /* If we're not polling our PHY instance, just return. */
145 if (IFM_INST(ife->ifm_media) != sc->mii_inst)
146 return 0;
147 break;
148
149 case MII_MEDIACHG:
150 /* If we're not currently selected, just return. */
151 if (IFM_INST(ife->ifm_media) != sc->mii_inst)
152 return 0;
153
154 /* If the interface is not up, don't do anything. */
155 if ((mii->mii_ifp->if_flags & IFF_UP) == 0)
156 break;
157
158 mii_phy_setmedia(sc);
159 break;
160
161 case MII_TICK:
162 /* If we're not currently selected, just return. */
163 if (IFM_INST(ife->ifm_media) != sc->mii_inst)
164 return 0;
165
166 /* Just bail now if the interface is down. */
167 if ((mii->mii_ifp->if_flags & IFF_UP) == 0)
168 return 0;
169
170 /*
171 * If we're not doing autonegotiation, we don't need to do
172 * any extra work here. However, we need to check the link
173 * status so we can generate an announcement if the status
174 * changes.
175 */
176 if (IFM_SUBTYPE(ife->ifm_media) != IFM_AUTO)
177 break;
178
179 /* Read the status register twice; MSR_LINK is latch-low. */
180 PHY_READ(sc, URLPHY_MSR, &reg);
181 PHY_READ(sc, URLPHY_MSR, &reg);
182 if (reg & URLPHY_MSR_LINK) {
183 /*
184 * Reset autonegotiation timer to 0 in case the link
185 * goes down in the next tick.
186 */
187 sc->mii_ticks = 0;
188 /* See above. */
189 break;
190 }
191
192 /*
193 * mii_ticks == 0 means it's the first tick after changing the
194 * media or the link became down since the last tick (see
195 * above), so break to update the status.
196 */
197 if (sc->mii_ticks++ == 0)
198 break;
199
200 /* Only retry autonegotiation every N seconds. */
201 KASSERT(sc->mii_anegticks != 0);
202 if (sc->mii_ticks <= sc->mii_anegticks)
203 return 0;
204
205 PHY_RESET(sc);
206
207 if (mii_phy_auto(sc, 0) == EJUSTRETURN)
208 return 0;
209
210 break;
211
212 case MII_DOWN:
213 mii_phy_down(sc);
214 return 0;
215 }
216
217 /* Update the media status. */
218 mii_phy_status(sc);
219
220 /* Callback if something changed. */
221 mii_phy_update(sc, cmd);
222
223 return 0;
224}

226static void
227urlphy_status(struct mii_softc *sc)
228{
229 struct mii_data *mii = sc->mii_pdata;
230 struct ifmedia_entry *ife = mii->mii_media.ifm_cur;
231 uint16_t msr, bmsr, bmcr;
232
233 DPRINTF(("%s: %s: enter\n", device_xname(sc->mii_dev), __func__));
234
235 mii->mii_media_status = IFM_AVALID;
236 mii->mii_media_active = IFM_ETHER;
237
238 /*
239 * The link status bit is not exist in the BMSR register,
240 * so we need to read the MSR register to get link status.
241 */
242 PHY_READ(sc, URLPHY_MSR, &msr);
243 PHY_READ(sc, URLPHY_MSR, &msr);
244 if (msr & URLPHY_MSR_LINK)
245 mii->mii_media_status |= IFM_ACTIVE;
246
247 DPRINTF(("%s: %s: link %s\n", device_xname(sc->mii_dev), __func__,
248 mii->mii_media_status & IFM_ACTIVE ? "up" : "down"));
249
250 PHY_READ(sc, MII_BMCR, &bmcr);
251 if (bmcr & BMCR_AUTOEN) {
252 PHY_READ(sc, MII_BMSR, &bmsr);
253 PHY_READ(sc, MII_BMSR, &bmsr);
254 if ((bmsr & BMSR_ACOMP) == 0) {
255 /* Erg, still trying, I guess... */
256 mii->mii_media_active |= IFM_NONE;
257 return;
258 }
259
260 if (msr & URLPHY_MSR_SPEED_100)
261 mii->mii_media_active |= IFM_100_TX;
262 else
263 mii->mii_media_active |= IFM_10_T;
264
265 if (msr & URLPHY_MSR_DUPLEX)
266 mii->mii_media_active |= IFM_FDX;
267 else
268 mii->mii_media_active |= IFM_HDX;
269 } else
270 mii->mii_media_active = ife->ifm_media;
271}
