318bool
319nvmm_x86_pat_validate(uint64_t val)
320{
321 uint8_t *pat = (uint8_t *)&val;
322 size_t i;
323
324 for (i = 0; i < 8; i++) {
325 if (__predict_false(pat[i] & ~__BITS(2,0)))
326 return false;
327 if (__predict_false(pat[i] == 2 || pat[i] == 3))
328 return false;
329 }
330
331 return true;
332}
