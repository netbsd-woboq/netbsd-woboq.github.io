42#ifndef _DEV_OFW_OPENFIRMIO_H_
43#define _DEV_OFW_OPENFIRMIO_H_
44
45#include <sys/ioccom.h>
46
47struct ofiocdesc {
48 int of_nodeid; /* passed or returned node id */
49 int of_namelen; /* length of op_name */
50 char *of_name; /* pointer to field name */
51 int of_buflen; /* length of op_buf (value-result) */
52 char *of_buf; /* pointer to field value */
53};
54
55#define OFIOCGET _IOWR('O', 1, struct ofiocdesc) /* get openprom field */
56#define OFIOCSET _IOW('O', 2, struct ofiocdesc) /* set openprom field */
57#define OFIOCNEXTPROP _IOWR('O', 3, struct ofiocdesc) /* get next property */
58#define OFIOCGETOPTNODE _IOR('O', 4, int) /* get options node */
59#define OFIOCGETNEXT _IOWR('O', 5, int) /* get next node of node */
60#define OFIOCGETCHILD _IOWR('O', 6, int) /* get first child of node */
61#define OFIOCFINDDEVICE _IOWR('O', 7, struct ofiocdesc) /* find a specific device */
62
63#endif /* _DEV_OFW_OPENFIRMIO_H_ */
