78int
79onewire_crc(const void *buf, int len)
80{
81 const u_int8_t *p = buf;
82 u_int8_t crc = 0;
83
84 while (len--)
85 crc = crctable[crc ^ *p++];
86
87 return (crc);
88}

90const char *
91onewire_famname(int type)
92{
93#ifdef ONEWIREVERBOSE
94 const struct onewire_family *of;
95
96 for (of = onewire_famtab; of->of_name != NULL; of++)
97 if (of->of_type == type)
98 return (of->of_name);
99#endif
100
101 return (NULL);
102}

104int
105onewire_matchbyfam(struct onewire_attach_args *oa,
106 const struct onewire_matchfam *fams, int nent)
107{
108 const struct onewire_matchfam *om;
109 int i;
110
111 for (i = 0, om = fams; i < nent; i++, om++)
112 if (ONEWIRE_ROM_FAMILY_TYPE(oa->oa_rom) == om->om_type)
113 return (1);
114
115 return (0);
116}
