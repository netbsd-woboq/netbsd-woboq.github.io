30#include <sys/types.h>
31#include <sys/bus.h>
32
33#include <dev/pci/pcivar.h>

35#include <dev/pci/agpvar.h>
36
37struct agp_i810_softc {
38 struct pci_attach_args vga_pa; /* integrated graphics device args */
39 int chiptype; /* chipset family: i810, i830, &c. */
40 uint32_t stolen; /* pages of stolen graphics memory */
41
42 /* Memory-mapped I/O for integrated graphics device registers. */
43 bus_space_tag_t bst;
44 bus_space_handle_t bsh;
45 bus_size_t size;
46
47 /* Graphics translation table. */
48 bus_space_tag_t gtt_bst;
49 bus_space_handle_t gtt_bsh;
50 bus_size_t gtt_size;
51
52 /* Chipset flush page. */
53 bus_space_tag_t flush_bst;
54 bus_space_handle_t flush_bsh;
55 bus_addr_t flush_addr;
56
57 /* i810-only fields. */
58 struct agp_gatt *gatt; /* i810-only OS-allocated GTT */
59 uint32_t dcache_size; /* i810-only on-chip memory size */
60
61 /* Cached pgtblctl register for resume. */
62 pcireg_t pgtblctl;
63};
64
65extern struct agp_softc *agp_i810_sc;
66
67#define AGP_I810_GTT_VALID 0x01
68#define AGP_I810_GTT_I810_DCACHE 0x02 /* i810-only */
69#define AGP_I810_GTT_CACHED 0x06 /* >=i830 */

75void agp_i810_reset(struct agp_i810_softc *);
