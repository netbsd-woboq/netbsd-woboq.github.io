84static inline uint64_t
85ahc_compose_id(u_int device, u_int vendor, u_int subdevice, u_int subvendor)
86{
87 uint64_t id;
88
89 id = subvendor
90 | (subdevice << 16)
91 | ((uint64_t)vendor << 32)
92 | ((uint64_t)device << 48);
93
94 return (id);
95}

712static const struct ahc_pci_identity *
713ahc_find_pci_device(pcireg_t id, pcireg_t subid, u_int func)
714{
715 u_int64_t full_id;
716 const struct ahc_pci_identity *entry;
717 u_int i;
718
719 full_id = ahc_compose_id(PCI_PRODUCT(id), PCI_VENDOR(id),
720 PCI_PRODUCT(subid), PCI_VENDOR(subid));
721
722 /*
723 * If the second function is not hooked up, ignore it.
724 * Unfortunately, not all MB vendors implement the
725 * subdevice ID as per the Adaptec spec, so do our best
726 * to sanity check it prior to accepting the subdevice
727 * ID as valid.
728 */
729 if (func > 0
730 && ahc_9005_subdevinfo_valid(PCI_VENDOR(id), PCI_PRODUCT(id),
731 PCI_VENDOR(subid), PCI_PRODUCT(subid))
732 && SUBID_9005_MFUNCENB(PCI_PRODUCT(subid)) == 0)
733 return (NULL);
734
735 for (i = 0; i < ahc_num_pci_devs; i++) {
736 entry = &ahc_pci_ident_table[i];
737 if (entry->full_id == (full_id & entry->id_mask))
738 return (entry);
739 }
740 return (NULL);
741}

755static void
756ahc_pci_attach(device_t parent, device_t self, void *aux)
757{
758 struct pci_attach_args *pa = aux;
759 const struct ahc_pci_identity *entry;
760 struct ahc_softc *ahc = device_private(self);
761 pcireg_t command;
762 u_int our_id = 0;
763 u_int sxfrctl1;
764 u_int scsiseq;
765 u_int sblkctl;
766 uint8_t dscommand0;
767 uint32_t devconfig;
768 int error;
769 pcireg_t subid;
770 int ioh_valid;
771 bus_space_tag_t st, iot;
772 bus_space_handle_t sh, ioh;
773#ifdef AHC_ALLOW_MEMIO
774 int memh_valid;
775 bus_space_tag_t memt;
776 bus_space_handle_t memh;
777 pcireg_t memtype;
778#endif
779 pci_intr_handle_t ih;
780 const char *intrstr;
781 struct ahc_pci_busdata *bd;
782 bool override_ultra;
783 char intrbuf[PCI_INTRSTR_LEN];
784
785 ahc->sc_dev = self;
786 ahc_set_name(ahc, device_xname(ahc->sc_dev));
787 ahc->parent_dmat = pa->pa_dmat;
788
789 command = pci_conf_read(pa->pa_pc, pa->pa_tag, PCI_COMMAND_STATUS_REG);
790 subid = pci_conf_read(pa->pa_pc, pa->pa_tag, PCI_SUBSYS_ID_REG);
791 entry = ahc_find_pci_device(pa->pa_id, subid, pa->pa_function);
792 if (entry == NULL)
793 return;
794 aprint_naive("\n");
795 aprint_normal(": %s\n", entry->name);
796
797 /* Keep information about the PCI bus */
798 bd = malloc(sizeof (struct ahc_pci_busdata), M_DEVBUF, M_NOWAIT);
799 if (bd == NULL) {
800 aprint_error("%s: unable to allocate bus-specific data\n",
801 ahc_name(ahc));
802 return;
803 }
804 memset(bd, 0, sizeof(struct ahc_pci_busdata));
805
806 bd->pc = pa->pa_pc;
807 bd->tag = pa->pa_tag;
808 bd->func = pa->pa_function;
809 bd->dev = pa->pa_device;
810 bd->class = pa->pa_class;
811
812 ahc->bd = bd;
813
814 ahc->description = entry->name;
815
816 error = entry->setup(ahc);
817 if (error != 0)
818 return;
819
820 ioh_valid = 0;
821
822#ifdef AHC_ALLOW_MEMIO
823 memh_valid = 0;
824 memtype = pci_mapreg_type(pa->pa_pc, pa->pa_tag, AHC_PCI_MEMADDR);
825 switch (memtype) {
826 case PCI_MAPREG_TYPE_MEM | PCI_MAPREG_MEM_TYPE_32BIT:
827 case PCI_MAPREG_TYPE_MEM | PCI_MAPREG_MEM_TYPE_64BIT:
828 memh_valid = (pci_mapreg_map(pa, AHC_PCI_MEMADDR,
829 memtype, 0, &memt, &memh, NULL, NULL) == 0);
830 break;
831 default:
832 memh_valid = 0;
833 }
834#endif
835 ioh_valid = (pci_mapreg_map(pa, AHC_PCI_IOADDR,
836 PCI_MAPREG_TYPE_IO, 0, &iot,
837 &ioh, NULL, NULL) == 0);
838
839#if 0
840 printf("%s: bus info: memt 0x%lx, memh 0x%lx, iot 0x%lx, ioh 0x%lx\n",
841 ahc_name(ahc), (u_long)memt, (u_long)memh, (u_long)iot,
842 (u_long)ioh);
843#endif
844
845#ifdef AHC_ALLOW_MEMIO
846 if (memh_valid) {
847 st = memt;
848 sh = memh;
849 } else
850#endif
851 if (ioh_valid) {
852 st = iot;
853 sh = ioh;
854 } else {
855 aprint_error(": unable to map registers\n");
856 return;
857 }
858 ahc->tag = st;
859 ahc->bsh = sh;
860
861 ahc->chip |= AHC_PCI;
862 /*
863 * Before we continue probing the card, ensure that
864 * its interrupts are *disabled*. We don't want
865 * a misstep to hang the machine in an interrupt
866 * storm.
867 */
868 ahc_intr_enable(ahc, FALSE);
869
870 /*
871 * XXX somehow reading this once fails on some sparc64 systems.
872 * This may be a problem in the sparc64 PCI code. Doing it
873 * twice works around it.
874 */
875 devconfig = pci_conf_read(pa->pa_pc, pa->pa_tag, DEVCONFIG);
876 devconfig = pci_conf_read(pa->pa_pc, pa->pa_tag, DEVCONFIG);
877
878 /*
879 * If we need to support high memory, enable dual
880 * address cycles. This bit must be set to enable
881 * high address bit generation even if we are on a
882 * 64bit bus (PCI64BIT set in devconfig).
883 */
884 if ((ahc->flags & AHC_39BIT_ADDRESSING) != 0) {
885
886 if (1/*bootverbose*/)
887 aprint_normal("%s: Enabling 39Bit Addressing\n",
888 ahc_name(ahc));
889 devconfig |= DACEN;
890 }
891
892 /* Ensure that pci error generation, a test feature, is disabled. */
893 devconfig |= PCIERRGENDIS;
894
895 pci_conf_write(pa->pa_pc, pa->pa_tag, DEVCONFIG, devconfig);
896
897 /* Ensure busmastering is enabled */
898 command |= PCI_COMMAND_MASTER_ENABLE;
899 pci_conf_write(pa->pa_pc, pa->pa_tag, PCI_COMMAND_STATUS_REG, command);
900
901 /*
902 * Disable PCI parity error reporting. Users typically
903 * do this to work around broken PCI chipsets that get
904 * the parity timing wrong and thus generate lots of spurious
905 * errors.
906 */
907 if ((ahc->flags & AHC_DISABLE_PCI_PERR) != 0)
908 command &= ~PCI_COMMAND_PARITY_ENABLE;
909 pci_conf_write(pa->pa_pc, pa->pa_tag, PCI_COMMAND_STATUS_REG, command);
910
911 /* On all PCI adapters, we allow SCB paging */
912 ahc->flags |= AHC_PAGESCBS;
913 error = ahc_softc_init(ahc);
914 if (error != 0)
915 goto error_out;
916
917 ahc->bus_intr = ahc_pci_intr;
918
919 /* Remember how the card was setup in case there is no SEEPROM */
920 if ((ahc_inb(ahc, HCNTRL) & POWRDN) == 0) {
921 ahc_pause(ahc);
922 if ((ahc->features & AHC_ULTRA2) != 0)
923 our_id = ahc_inb(ahc, SCSIID_ULTRA2) & OID;
924 else
925 our_id = ahc_inb(ahc, SCSIID) & OID;
926 sxfrctl1 = ahc_inb(ahc, SXFRCTL1) & STPWEN;
927 scsiseq = ahc_inb(ahc, SCSISEQ);
928 } else {
929 sxfrctl1 = STPWEN;
930 our_id = 7;
931 scsiseq = 0;
932 }
933
934 error = ahc_reset(ahc);
935 if (error != 0)
936 goto error_out;
937
938 if ((ahc->features & AHC_DT) != 0) {
939 u_int sfunct;
940
941 /* Perform ALT-Mode Setup */
942 sfunct = ahc_inb(ahc, SFUNCT) & ~ALT_MODE;
943 ahc_outb(ahc, SFUNCT, sfunct | ALT_MODE);
944 ahc_outb(ahc, OPTIONMODE,
945 OPTIONMODE_DEFAULTS|AUTOACKEN|BUSFREEREV|EXPPHASEDIS);
946 ahc_outb(ahc, SFUNCT, sfunct);
947
948 /* Normal mode setup */
949 ahc_outb(ahc, CRCCONTROL1, CRCVALCHKEN|CRCENDCHKEN|CRCREQCHKEN
950 |TARGCRCENDEN);
951 }
952
953 if (pci_intr_map(pa, &ih)) {
954 aprint_error("%s: couldn't map interrupt\n", ahc_name(ahc));
955 ahc_free(ahc);
956 return;
957 }
958 intrstr = pci_intr_string(pa->pa_pc, ih, intrbuf, sizeof(intrbuf));
959 ahc->ih = pci_intr_establish_xname(pa->pa_pc, ih, IPL_BIO, ahc_intr,
960 ahc, device_xname(self));
961 if (ahc->ih == NULL) {
962 aprint_error_dev(ahc->sc_dev,
963 "couldn't establish interrupt\n");
964 if (intrstr != NULL)
965 aprint_error(" at %s", intrstr);
966 aprint_error("\n");
967 ahc_free(ahc);
968 return;
969 }
970 if (intrstr != NULL)
971 aprint_normal("%s: interrupting at %s\n", ahc_name(ahc),
972 intrstr);
973
974 dscommand0 = ahc_inb(ahc, DSCOMMAND0);
975 dscommand0 |= MPARCKEN|CACHETHEN;
976 if ((ahc->features & AHC_ULTRA2) != 0) {
977
978 /*
979 * DPARCKEN doesn't work correctly on
980 * some MBs so don't use it.
981 */
982 dscommand0 &= ~DPARCKEN;
983 }
984
985 /*
986 * Handle chips that must have cache line
987 * streaming (dis/en)abled.
988 */
989 if ((ahc->bugs & AHC_CACHETHEN_DIS_BUG) != 0)
990 dscommand0 |= CACHETHEN;
991
992 if ((ahc->bugs & AHC_CACHETHEN_BUG) != 0)
993 dscommand0 &= ~CACHETHEN;
994
995 ahc_outb(ahc, DSCOMMAND0, dscommand0);
996
997 ahc->pci_cachesize =
998 pci_conf_read(pa->pa_pc, pa->pa_tag, CSIZE_LATTIME) & CACHESIZE;
999 ahc->pci_cachesize *= 4;
1000
1001 if ((ahc->bugs & AHC_PCI_2_1_RETRY_BUG) != 0
1002 && ahc->pci_cachesize == 4) {
1003 pci_conf_write(pa->pa_pc, pa->pa_tag, CSIZE_LATTIME, 0);
1004 ahc->pci_cachesize = 0;
1005 }
1006
1007 /*
1008 * We cannot perform ULTRA speeds without the presence
1009 * of the external precision resistor.
1010 * Allow override for the SGI O2 though, which has two onboard ahc
1011 * that fail here but are perfectly capable of ultra speeds.
1012 */
1013 override_ultra = FALSE;
1014 prop_dictionary_get_bool(device_properties(self),
1015 "aic7xxx-override-ultra", &override_ultra);
1016
1017 if (((ahc->features & AHC_ULTRA) != 0) && (!override_ultra)) {
1018 uint32_t dvconfig;
1019
1020 dvconfig = pci_conf_read(pa->pa_pc, pa->pa_tag, DEVCONFIG);
1021 if ((dvconfig & REXTVALID) == 0)
1022 ahc->features &= ~AHC_ULTRA;
1023 }
1024
1025 ahc->seep_config = malloc(sizeof(*ahc->seep_config),
1026 M_DEVBUF, M_NOWAIT);
1027 if (ahc->seep_config == NULL)
1028 goto error_out;
1029
1030 memset(ahc->seep_config, 0, sizeof(*ahc->seep_config));
1031
1032 /* See if we have a SEEPROM and perform auto-term */
1033 ahc_check_extport(ahc, &sxfrctl1);
1034
1035 /*
1036 * Take the LED out of diagnostic mode
1037 */
1038 sblkctl = ahc_inb(ahc, SBLKCTL);
1039 ahc_outb(ahc, SBLKCTL, (sblkctl & ~(DIAGLEDEN|DIAGLEDON)));
1040
1041 if ((ahc->features & AHC_ULTRA2) != 0) {
1042 ahc_outb(ahc, DFF_THRSH, RD_DFTHRSH_MAX|WR_DFTHRSH_MAX);
1043 } else {
1044 ahc_outb(ahc, DSPCISTATUS, DFTHRSH_100);
1045 }
1046
1047 if (ahc->flags & AHC_USEDEFAULTS) {
1048 /*
1049 * PCI Adapter default setup
1050 * Should only be used if the adapter does not have
1051 * a SEEPROM.
1052 */
1053 /* See if someone else set us up already */
1054 if ((ahc->flags & AHC_NO_BIOS_INIT) == 0
1055 && scsiseq != 0) {
1056 prop_bool_t usetd;
1057
1058 printf("%s: Using left over BIOS settings\n",
1059 ahc_name(ahc));
1060 ahc->flags &= ~AHC_USEDEFAULTS;
1061 /*
1062 * Ignore target device settings and use default
1063 * if BIOS initializes chip's SRAM with some
1064 * conservative settings (async, no tagged
1065 * queuing etc.) and machine dependent device
1066 * property is set.
1067 */
1068 usetd = prop_dictionary_get(
1069 device_properties(ahc->sc_dev),
1070 "aic7xxx-use-target-defaults");
1071 if (usetd != NULL) {
1072 KASSERT(prop_object_type(usetd) ==
1073 PROP_TYPE_BOOL);
1074 if (prop_bool_true(usetd))
1075 ahc->flags |= AHC_USETARGETDEFAULTS;
1076 }
1077 ahc->flags |= AHC_BIOS_ENABLED;
1078 } else {
1079 /*
1080 * Assume only one connector and always turn
1081 * on termination.
1082 */
1083 our_id = 0x07;
1084 sxfrctl1 = STPWEN;
1085 }
1086 ahc_outb(ahc, SCSICONF, our_id|ENSPCHK|RESET_SCSI);
1087
1088 ahc->our_id = our_id;
1089 }
1090
1091 /*
1092 * Take a look to see if we have external SRAM.
1093 * We currently do not attempt to use SRAM that is
1094 * shared among multiple controllers.
1095 */
1096 ahc_probe_ext_scbram(ahc);
1097
1098 /*
1099 * Record our termination setting for the
1100 * generic initialization routine.
1101 */
1102 if ((sxfrctl1 & STPWEN) != 0)
1103 ahc->flags |= AHC_TERM_ENB_A;
1104
1105 if (ahc_init(ahc))
1106 goto error_out;
1107
1108 ahc_attach(ahc);
1109
1110 return;
1111
1112 error_out:
1113 ahc_free(ahc);
1114 return;
1115}

1155/*
1156 * Test for the presense of external sram in an
1157 * "unshared" configuration.
1158 */
1159static int
1160ahc_ext_scbram_present(struct ahc_softc *ahc)
1161{
1162 u_int chip;
1163 int ramps;
1164 int single_user;
1165 uint32_t devconfig;
1166
1167 chip = ahc->chip & AHC_CHIPID_MASK;
1168 devconfig = pci_conf_read(ahc->bd->pc, ahc->bd->tag, DEVCONFIG);
1169 single_user = (devconfig & MPORTMODE) != 0;
1170
1171 if ((ahc->features & AHC_ULTRA2) != 0)
1172 ramps = (ahc_inb(ahc, DSCOMMAND0) & RAMPS) != 0;
1173 else if (chip == AHC_AIC7895 || chip == AHC_AIC7895C)
1174 /*
1175 * External SCBRAM arbitration is flakey
1176 * on these chips. Unfortunately this means
1177 * we don't use the extra SCB ram space on the
1178 * 3940AUW.
1179 */
1180 ramps = 0;
1181 else if (chip >= AHC_AIC7870)
1182 ramps = (devconfig & RAMPSM) != 0;
1183 else
1184 ramps = 0;
1185
1186 if (ramps && single_user)
1187 return (1);
1188 return (0);
1189}

1192 * Enable external scbram.
1193 */
1194static void
1195ahc_scbram_config(struct ahc_softc *ahc, int enable, int pcheck,
1196 int fast, int large)
1197{
1198 uint32_t devconfig;
1199
1200 if (ahc->features & AHC_MULTI_FUNC) {
1201 /*
1202 * Set the SCB Base addr (highest address bit)
1203 * depending on which channel we are.
1204 */
1205 ahc_outb(ahc, SCBBADDR, ahc->bd->func);
1206 }
1207
1208 ahc->flags &= ~AHC_LSCBS_ENABLED;
1209 if (large)
1210 ahc->flags |= AHC_LSCBS_ENABLED;
1211 devconfig = pci_conf_read(ahc->bd->pc, ahc->bd->tag, DEVCONFIG);
1212 if ((ahc->features & AHC_ULTRA2) != 0) {
1213 u_int dscommand0;
1214
1215 dscommand0 = ahc_inb(ahc, DSCOMMAND0);
1216 if (enable)
1217 dscommand0 &= ~INTSCBRAMSEL;
1218 else
1219 dscommand0 |= INTSCBRAMSEL;
1220 if (large)
1221 dscommand0 &= ~USCBSIZE32;
1222 else
1223 dscommand0 |= USCBSIZE32;
1224 ahc_outb(ahc, DSCOMMAND0, dscommand0);
1225 } else {
1226 if (fast)
1227 devconfig &= ~EXTSCBTIME;
1228 else
1229 devconfig |= EXTSCBTIME;
1230 if (enable)
1231 devconfig &= ~SCBRAMSEL;
1232 else
1233 devconfig |= SCBRAMSEL;
1234 if (large)
1235 devconfig &= ~SCBSIZE32;
1236 else
1237 devconfig |= SCBSIZE32;
1238 }
1239 if (pcheck)
1240 devconfig |= EXTSCBPEN;
1241 else
1242 devconfig &= ~EXTSCBPEN;
1243
1244 pci_conf_write(ahc->bd->pc, ahc->bd->tag, DEVCONFIG, devconfig);
1245}

1247/*
1248 * Take a look to see if we have external SRAM.
1249 * We currently do not attempt to use SRAM that is
1250 * shared among multiple controllers.
1251 */
1252static void
1253ahc_probe_ext_scbram(struct ahc_softc *ahc)
1254{
1255 int num_scbs;
1256 int test_num_scbs;
1257 int enable;
1258 int pcheck;
1259 int fast;
1260 int large;
1261
1262 enable = FALSE;
1263 pcheck = FALSE;
1264 fast = FALSE;
1265 large = FALSE;
1266 num_scbs = 0;
1267
1268 if (ahc_ext_scbram_present(ahc) == 0)
1269 goto done;
1270
1271 /*
1272 * Probe for the best parameters to use.
1273 */
1274 ahc_scbram_config(ahc, /*enable*/TRUE, pcheck, fast, large);
1275 num_scbs = ahc_probe_scbs(ahc);
1276 if (num_scbs == 0) {
1277 /* The SRAM wasn't really present. */
1278 goto done;
1279 }
1280 enable = TRUE;
1281
1282 /*
1283 * Clear any outstanding parity error
1284 * and ensure that parity error reporting
1285 * is enabled.
1286 */
1287 ahc_outb(ahc, SEQCTL, 0);
1288 ahc_outb(ahc, CLRINT, CLRPARERR);
1289 ahc_outb(ahc, CLRINT, CLRBRKADRINT);
1290
1291 /* Now see if we can do parity */
1292 ahc_scbram_config(ahc, enable, /*pcheck*/TRUE, fast, large);
1293 num_scbs = ahc_probe_scbs(ahc);
1294 if ((ahc_inb(ahc, INTSTAT) & BRKADRINT) == 0
1295 || (ahc_inb(ahc, ERROR) & MPARERR) == 0)
1296 pcheck = TRUE;
1297
1298 /* Clear any resulting parity error */
1299 ahc_outb(ahc, CLRINT, CLRPARERR);
1300 ahc_outb(ahc, CLRINT, CLRBRKADRINT);
1301
1302 /* Now see if we can do fast timing */
1303 ahc_scbram_config(ahc, enable, pcheck, /*fast*/TRUE, large);
1304 test_num_scbs = ahc_probe_scbs(ahc);
1305 if (test_num_scbs == num_scbs
1306 && ((ahc_inb(ahc, INTSTAT) & BRKADRINT) == 0
1307 || (ahc_inb(ahc, ERROR) & MPARERR) == 0))
1308 fast = TRUE;
1309
1310 /*
1311 * See if we can use large SCBs and still maintain
1312 * the same overall count of SCBs.
1313 */
1314 if ((ahc->features & AHC_LARGE_SCBS) != 0) {
1315 ahc_scbram_config(ahc, enable, pcheck, fast, /*large*/TRUE);
1316 test_num_scbs = ahc_probe_scbs(ahc);
1317 if (test_num_scbs >= num_scbs) {
1318 large = TRUE;
1319 num_scbs = test_num_scbs;
1320 if (num_scbs >= 64) {
1321 /*
1322 * We have enough space to move the
1323 * "busy targets table" into SCB space
1324 * and make it qualify all the way to the
1325 * lun level.
1326 */
1327 ahc->flags |= AHC_SCB_BTT;
1328 }
1329 }
1330 }
1331done:
1332 /*
1333 * Disable parity error reporting until we
1334 * can load instruction ram.
1335 */
1336 ahc_outb(ahc, SEQCTL, PERRORDIS|FAILDIS);
1337 /* Clear any latched parity error */
1338 ahc_outb(ahc, CLRINT, CLRPARERR);
1339 ahc_outb(ahc, CLRINT, CLRBRKADRINT);
1340 if (1/*bootverbose*/ && enable) {
1341 printf("%s: External SRAM, %s access%s, %dbytes/SCB\n",
1342 ahc_name(ahc), fast ? "fast" : "slow",
1343 pcheck ? ", parity checking enabled" : "",
1344 large ? 64 : 32);
1345 }
1346 ahc_scbram_config(ahc, enable, pcheck, fast, large);
1347}

1420static void
1421ahc_pci_intr(struct ahc_softc *ahc)
1422{
1423 u_int error;
1424 u_int status1;
1425
1426 error = ahc_inb(ahc, ERROR);
1427 if ((error & PCIERRSTAT) == 0)
1428 return;
1429
1430 status1 = pci_conf_read(ahc->bd->pc, ahc->bd->tag,
1431 PCI_COMMAND_STATUS_REG);
1432
1433 printf("%s: PCI error Interrupt at seqaddr = 0x%x\n",
1434 ahc_name(ahc),
1435 ahc_inb(ahc, SEQADDR0) | (ahc_inb(ahc, SEQADDR1) << 8));
1436
1437 if (status1 & DPE) {
1438 printf("%s: Data Parity Error Detected during address "
1439 "or write data phase\n", ahc_name(ahc));
1440 }
1441 if (status1 & SSE) {
1442 printf("%s: Signal System Error Detected\n", ahc_name(ahc));
1443 }
1444 if (status1 & RMA) {
1445 printf("%s: Received a Master Abort\n", ahc_name(ahc));
1446 }
1447 if (status1 & RTA) {
1448 printf("%s: Received a Target Abort\n", ahc_name(ahc));
1449 }
1450 if (status1 & STA) {
1451 printf("%s: Signaled a Target Abort\n", ahc_name(ahc));
1452 }
1453 if (status1 & DPR) {
1454 printf("%s: Data Parity Error has been reported via PERR#\n",
1455 ahc_name(ahc));
1456 }
1457
1458 /* Clear latched errors. */
1459 pci_conf_write(ahc->bd->pc, ahc->bd->tag, PCI_COMMAND_STATUS_REG,
1460 status1);
1461
1462 if ((status1 & (DPE|SSE|RMA|RTA|STA|DPR)) == 0) {
1463 printf("%s: Latched PCIERR interrupt with "
1464 "no status bits set\n", ahc_name(ahc));
1465 } else {
1466 ahc_outb(ahc, CLRINT, CLRPARERR);
1467 }
1468
1469 ahc_unpause(ahc);
1470}
