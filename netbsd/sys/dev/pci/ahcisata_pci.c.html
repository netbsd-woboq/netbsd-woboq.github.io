234#define AHCI_PCI_ABAR_CAVIUM 0x10

236static int
237ahci_pci_has_quirk(pci_vendor_id_t vendor, pci_product_id_t product)
238{
239 int i;
240
241 for (i = 0; i < __arraycount(ahci_pci_quirks); i++)
242 if (vendor == ahci_pci_quirks[i].vendor &&
243 product == ahci_pci_quirks[i].product)
244 return ahci_pci_quirks[i].quirks;
245 return 0;
246}
247
248static int
249ahci_pci_abar(struct pci_attach_args *pa)
250{
251 if (PCI_VENDOR(pa->pa_id) == PCI_VENDOR_CAVIUM) {
252 if (PCI_PRODUCT(pa->pa_id) == PCI_PRODUCT_CAVIUM_THUNDERX_AHCI ||
253 PCI_PRODUCT(pa->pa_id) == PCI_PRODUCT_CAVIUM_THUNDERX_RAID) {
254 return AHCI_PCI_ABAR_CAVIUM;
255 }
256 }
257
258 return AHCI_PCI_ABAR;
259}

262static int
263ahci_pci_match(device_t parent, cfdata_t match, void *aux)
264{
265 struct pci_attach_args *pa = aux;
266 bus_space_tag_t regt;
267 bus_space_handle_t regh;
268 bus_size_t size;
269 int ret = 0;
270 bool force;
271
272 force = ((ahci_pci_has_quirk( PCI_VENDOR(pa->pa_id),
273 PCI_PRODUCT(pa->pa_id)) & AHCI_PCI_QUIRK_FORCE) != 0);
274
275 /* if wrong class and not forced by quirks, don't match */
276 if ((PCI_CLASS(pa->pa_class) != PCI_CLASS_MASS_STORAGE ||
277 ((PCI_SUBCLASS(pa->pa_class) != PCI_SUBCLASS_MASS_STORAGE_SATA ||
278 PCI_INTERFACE(pa->pa_class) != PCI_INTERFACE_SATA_AHCI) &&
279 PCI_SUBCLASS(pa->pa_class) != PCI_SUBCLASS_MASS_STORAGE_RAID)) &&
280 (force == false))
281 return 0;
282
283 int bar = ahci_pci_abar(pa);
284 pcireg_t memtype = pci_mapreg_type(pa->pa_pc, pa->pa_tag, bar);
285 if (pci_mapreg_map(pa, bar, memtype, 0, &regt, &regh, NULL, &size) != 0)
286 return 0;
287
288 if ((PCI_SUBCLASS(pa->pa_class) == PCI_SUBCLASS_MASS_STORAGE_SATA &&
289 PCI_INTERFACE(pa->pa_class) == PCI_INTERFACE_SATA_AHCI) ||
290 (bus_space_read_4(regt, regh, AHCI_GHC) & AHCI_GHC_AE) ||
291 (force == true))
292 ret = 3;
293
294 bus_space_unmap(regt, regh, size);
295 return ret;
296}

298static int
299ahci_pci_intr_establish(struct ahci_softc *sc, int port)
300{
301 struct ahci_pci_softc *psc = (struct ahci_pci_softc *)sc;
302 device_t self = sc->sc_atac.atac_dev;
303 char intrbuf[PCI_INTRSTR_LEN];
304 char intr_xname[INTRDEVNAMEBUF];
305 const char *intrstr;
306 int vec;
307 int (*intr_handler)(void *);
308 void *intr_arg;
309
310 KASSERT(psc->sc_pihp != NULL);
311 KASSERT(psc->sc_nintr > 0);
312
313 snprintf(intr_xname, sizeof(intr_xname), "%s", device_xname(self));
314
315 if (psc->sc_nintr == 1 || sc->sc_ghc_mrsm) {
316 /* Only one interrupt, established on vector 0 */
317 intr_handler = ahci_intr;
318 intr_arg = sc;
319 vec = 0;
320
321 if (psc->sc_ih[vec] != NULL) {
322 /* Already established, nothing more to do */
323 goto out;
324 }
325
326 } else {
327 /*
328 * Theoretically AHCI device can have less MSI/MSI-X vectors
329 * than supported ports. Hardware is allowed to revert
330 * to single message MSI, but not required to do so.
331 * So handle the case when it did not revert to single MSI.
332 * In this case last available interrupt vector is used
333 * for port == max vector, and all further ports.
334 * This last vector must use the general interrupt handler,
335 * since it needs to be able to handle several ports.
336 * NOTE: such case was never actually observed yet
337 */
338 if (sc->sc_atac.atac_nchannels > psc->sc_nintr
339 && port >= (psc->sc_nintr - 1)) {
340 intr_handler = ahci_intr;
341 intr_arg = sc;
342 vec = psc->sc_nintr - 1;
343
344 if (psc->sc_ih[vec] != NULL) {
345 /* Already established, nothing more to do */
346 goto out;
347 }
348
349 if (port == vec) {
350 /* Print error once */
351 aprint_error_dev(self,
352 "port %d independant interrupt vector not "
353 "available, sharing with further ports",
354 port);
355 }
356 } else {
357 /* Vector according to port */
358 KASSERT(port < psc->sc_nintr);
359 KASSERT(psc->sc_ih[port] == NULL);
360 intr_handler = ahci_intr_port;
361 intr_arg = &sc->sc_channels[port];
362 vec = port;
363
364 snprintf(intr_xname, sizeof(intr_xname), "%s port%d",
365 device_xname(self), port);
366 }
367 }
368
369 intrstr = pci_intr_string(psc->sc_pc, psc->sc_pihp[vec], intrbuf,
370 sizeof(intrbuf));
371 psc->sc_ih[vec] = pci_intr_establish_xname(psc->sc_pc,
372 psc->sc_pihp[vec], IPL_BIO, intr_handler, intr_arg, intr_xname);
373 if (psc->sc_ih == NULL) {
374 aprint_error_dev(self, "couldn't establish interrupt");
375 if (intrstr != NULL)
376 aprint_error(" at %s", intrstr);
377 aprint_error("\n");
378 goto fail;
379 }
380 aprint_normal_dev(self, "interrupting at %s\n", intrstr);
381
382out:
383 return 0;
384
385fail:
386 return EAGAIN;
387}

389static void
390ahci_pci_attach(device_t parent, device_t self, void *aux)
391{
392 struct pci_attach_args *pa = aux;
393 struct ahci_pci_softc *psc = device_private(self);
394 struct ahci_softc *sc = &psc->ah_sc;
395 bool ahci_cap_64bit;
396 bool ahci_bad_64bit;
397
398 sc->sc_atac.atac_dev = self;
399
400 int bar = ahci_pci_abar(pa);
401 pcireg_t memtype = pci_mapreg_type(pa->pa_pc, pa->pa_tag, bar);
402 if (pci_mapreg_map(pa, bar, memtype, 0, &sc->sc_ahcit, &sc->sc_ahcih,
403 NULL, &sc->sc_ahcis) != 0) {
404 aprint_error_dev(self, "can't map ahci registers\n");
405 return;
406 }
407 psc->sc_pc = pa->pa_pc;
408 psc->sc_pcitag = pa->pa_tag;
409
410 pci_aprint_devinfo(pa, "AHCI disk controller");
411
412 int counts[PCI_INTR_TYPE_SIZE] = {
413 [PCI_INTR_TYPE_INTX] = 1,
414 [PCI_INTR_TYPE_MSI] = 1,
415 [PCI_INTR_TYPE_MSIX] = -1,
416 };
417
418 /* Allocate and establish the interrupt. */
419 if (pci_intr_alloc(pa, &psc->sc_pihp, counts, PCI_INTR_TYPE_MSIX)) {
420 aprint_error_dev(self, "can't allocate handler\n");
421 goto fail;
422 }
423
424 psc->sc_nintr = counts[pci_intr_type(pa->pa_pc, psc->sc_pihp[0])];
425 psc->sc_ih = kmem_zalloc(sizeof(void *) * psc->sc_nintr, KM_SLEEP);
426 sc->sc_intr_establish = ahci_pci_intr_establish;
427
428 sc->sc_dmat = pa->pa_dmat;
429
430 sc->sc_ahci_quirks = ahci_pci_has_quirk(PCI_VENDOR(pa->pa_id),
431 PCI_PRODUCT(pa->pa_id));
432
433 ahci_cap_64bit = (AHCI_READ(sc, AHCI_CAP) & AHCI_CAP_64BIT) != 0;
434 ahci_bad_64bit = ((sc->sc_ahci_quirks & AHCI_PCI_QUIRK_BAD64) != 0);
435
436 if (pci_dma64_available(pa) && ahci_cap_64bit) {
437 if (!ahci_bad_64bit)
438 sc->sc_dmat = pa->pa_dmat64;
439 aprint_verbose_dev(self, "64-bit DMA%s\n",
440 (sc->sc_dmat == pa->pa_dmat) ? " unavailable" : "");
441 }
442
443 if (PCI_SUBCLASS(pa->pa_class) == PCI_SUBCLASS_MASS_STORAGE_RAID) {
444 AHCIDEBUG_PRINT(("%s: RAID mode\n", AHCINAME(sc)), DEBUG_PROBE);
445 sc->sc_atac_capflags = ATAC_CAP_RAID;
446 } else {
447 AHCIDEBUG_PRINT(("%s: SATA mode\n", AHCINAME(sc)), DEBUG_PROBE);
448 }
449
450 ahci_attach(sc);
451
452 if (!pmf_device_register(self, NULL, ahci_pci_resume))
453 aprint_error_dev(self, "couldn't establish power handler\n");
454
455 return;
456fail:
457 if (psc->sc_pihp != NULL) {
458 pci_intr_release(psc->sc_pc, psc->sc_pihp, psc->sc_nintr);
459 psc->sc_pihp = NULL;
460 }
461 if (sc->sc_ahcis) {
462 bus_space_unmap(sc->sc_ahcit, sc->sc_ahcih, sc->sc_ahcis);
463 sc->sc_ahcis = 0;
464 }
465
466 return;
467
468}

479static int
480ahci_pci_detach(device_t dv, int flags)
481{
482 struct ahci_pci_softc *psc;
483 struct ahci_softc *sc;
484 int rv;
485
486 psc = device_private(dv);
487 sc = &psc->ah_sc;
488
489 if ((rv = ahci_detach(sc, flags)))
490 return rv;
491
492 pmf_device_deregister(dv);
493
494 if (psc->sc_ih != NULL) {
495 for (int intr = 0; intr < psc->sc_nintr; intr++) {
496 if (psc->sc_ih[intr] != NULL) {
497 pci_intr_disestablish(psc->sc_pc,
498 psc->sc_ih[intr]);
499 psc->sc_ih[intr] = NULL;
500 }
501 }
502
503 kmem_free(psc->sc_ih, sizeof(void *) * psc->sc_nintr);
504 psc->sc_ih = NULL;
505 }
506
507 if (psc->sc_pihp != NULL) {
508 pci_intr_release(psc->sc_pc, psc->sc_pihp, psc->sc_nintr);
509 psc->sc_pihp = NULL;
510 }
511
512 bus_space_unmap(sc->sc_ahcit, sc->sc_ahcih, sc->sc_ahcis);
513
514 return 0;
515}

517static bool
518ahci_pci_resume(device_t dv, const pmf_qual_t *qual)
519{
520 struct ahci_pci_softc *psc = device_private(dv);
521 struct ahci_softc *sc = &psc->ah_sc;
522 int s;
523
524 s = splbio();
525 ahci_resume(sc);
526 splx(s);
527
528 return true;
529}
