32#ifndef _DEV_PCI_AMDPMVAR_H_
33#define _DEV_PCI_AMDPMVAR_H_
34
35#include <sys/mutex.h>
36#include <sys/rndsource.h>
37
38struct amdpm_softc {
39 device_t sc_dev;
40
41 pci_chipset_tag_t sc_pc;
42 pcitag_t sc_tag;
43 struct pci_attach_args *sc_pa;
44
45 bus_space_tag_t sc_iot;
46 bus_space_handle_t sc_ioh; /* PMxx space */
47 bus_space_handle_t sc_sm_ioh; /* SM space */
48
49 i2c_addr_t sc_smbus_slaveaddr; /* address of smbus slave */
50 struct i2c_controller sc_i2c; /* i2c controller info */
51 kmutex_t sc_mutex;
52
53 void *sc_ih;
54
55 struct callout sc_rnd_ch;
56 krndsource_t sc_rnd_source;
57 int sc_rnd_need;
58#ifdef AMDPM_RND_COUNTERS
59 struct evcnt sc_rnd_hits;
60 struct evcnt sc_rnd_miss;
61 struct evcnt sc_rnd_data[256];
62#endif
63
64 char sc_nforce;
65};
66
67#endif /* _DEV_PCI_AMDPMVAR_H_ */
