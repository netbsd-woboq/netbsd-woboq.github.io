33#ifndef _DEV_PCI_AUVIAVAR_H_
34#define _DEV_PCI_AUVIAVAR_H_
35
36struct auvia_softc_chan {
37 void (*sc_intr)(void *);
38 void *sc_arg;
39
40 struct auvia_dma_op *sc_dma_ops;
41 struct auvia_dma *sc_dma_ops_dma;
42 u_int16_t sc_dma_op_count;
43 int sc_base;
44 u_int16_t sc_reg;
45};
46
47struct auvia_softc {
48 device_t sc_dev;
49 kmutex_t sc_lock;
50 kmutex_t sc_intr_lock;
51
52 char sc_revision[8];
53 u_int sc_flags;
54#define AUVIA_FLAGS_VT8233 0x0001
55
56 void *sc_ih; /* interrupt handle */
57
58 pci_chipset_tag_t sc_pc;
59 pcitag_t sc_pt;
60
61 bus_space_tag_t sc_iot;
62 bus_space_handle_t sc_ioh;
63 bus_size_t sc_iosize;
64 bus_dma_tag_t sc_dmat;
65
66 struct ac97_host_if host_if;
67 struct ac97_codec_if *codec_if;
68 bool sc_spdif;
69
70 struct auvia_dma *sc_dmas;
71
72 struct auvia_softc_chan sc_play, sc_record;
73
74#define AUVIA_NFORMATS 4
75 struct audio_format sc_formats[AUVIA_NFORMATS];
76};
77
78#endif /* !_DEV_PCI_AUVIAVAR_H_ */
