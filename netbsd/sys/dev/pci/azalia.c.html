276static int
277azalia_pci_match(device_t parent, cfdata_t match, void *aux)
278{
279 struct pci_attach_args *pa;
280
281 pa = aux;
282 if (PCI_CLASS(pa->pa_class) == PCI_CLASS_MULTIMEDIA
283 && PCI_SUBCLASS(pa->pa_class) == PCI_SUBCLASS_MULTIMEDIA_HDAUDIO)
284 return 1;
285 return 0;
286}

288static void
289azalia_pci_attach(device_t parent, device_t self, void *aux)
290{
291 azalia_t *sc = device_private(self);
292 struct pci_attach_args *pa = aux;
293 pcireg_t v;
294 pci_intr_handle_t ih;
295 const char *intrrupt_str;
296 char vendor[PCI_VENDORSTR_LEN];
297 char product[PCI_PRODUCTSTR_LEN];
298 char intrbuf[PCI_INTRSTR_LEN];
299
300 sc->dev = self;
301 sc->dmat = pa->pa_dmat;
302
303 aprint_normal(": Generic High Definition Audio Controller\n");
304
305 v = pci_conf_read(pa->pa_pc, pa->pa_tag, ICH_PCI_HDBARL);
306 v &= PCI_MAPREG_TYPE_MASK | PCI_MAPREG_MEM_TYPE_MASK;
307 if (pci_mapreg_map(pa, ICH_PCI_HDBARL, v, 0,
308 &sc->iot, &sc->ioh, NULL, &sc->map_size)) {
309 aprint_error_dev(self, "can't map device i/o space\n");
310 return;
311 }
312
313 /* enable bus mastering */
314 v = pci_conf_read(pa->pa_pc, pa->pa_tag, PCI_COMMAND_STATUS_REG);
315 pci_conf_write(pa->pa_pc, pa->pa_tag, PCI_COMMAND_STATUS_REG,
316 v | PCI_COMMAND_MASTER_ENABLE | PCI_COMMAND_BACKTOBACK_ENABLE);
317
318 /* interrupt */
319 if (pci_intr_map(pa, &ih)) {
320 aprint_error_dev(self, "can't map interrupt\n");
321 return;
322 }
323
324 mutex_init(&sc->lock, MUTEX_DEFAULT, IPL_NONE);
325 mutex_init(&sc->intr_lock, MUTEX_DEFAULT, IPL_AUDIO);
326
327 sc->pc = pa->pa_pc;
328 sc->tag = pa->pa_tag;
329 intrrupt_str = pci_intr_string(pa->pa_pc, ih, intrbuf, sizeof(intrbuf));
330 sc->ih = pci_intr_establish_xname(pa->pa_pc, ih, IPL_AUDIO, azalia_intr,
331 sc, device_xname(self));
332 if (sc->ih == NULL) {
333 aprint_error_dev(self, "can't establish interrupt");
334 if (intrrupt_str != NULL)
335 aprint_error(" at %s", intrrupt_str);
336 aprint_error("\n");
337 mutex_destroy(&sc->lock);
338 mutex_destroy(&sc->intr_lock);
339 return;
340 }
341 aprint_normal_dev(self, "interrupting at %s\n", intrrupt_str);
342
343 if (!pmf_device_register(self, NULL, azalia_pci_resume))
344 aprint_error_dev(self, "couldn't establish power handler\n");
345
346 sc->pciid = pa->pa_id;
347 pci_findvendor(vendor, sizeof(vendor), PCI_VENDOR(pa->pa_id));
348 pci_findproduct(product, sizeof(product), PCI_VENDOR(pa->pa_id),
349 PCI_PRODUCT(pa->pa_id));
350 aprint_normal_dev(self, "host: %s %s (rev. %d)",
351 vendor, product, PCI_REVISION(pa->pa_class));
352
353 if (azalia_attach(sc)) {
354 aprint_error_dev(self, "initialization failure\n");
355 azalia_pci_detach(self, 0);
356 mutex_destroy(&sc->lock);
357 mutex_destroy(&sc->intr_lock);
358 return;
359 }
360 sc->subid = pci_conf_read(pa->pa_pc, pa->pa_tag, PCI_SUBSYS_ID_REG);
361
362 config_interrupts(self, azalia_attach_intr);
363}

374static int
375azalia_pci_detach(device_t self, int flags)
376{
377 azalia_t *az;
378 int i;
379
380 DPRINTF(("%s\n", __func__));
381 az = device_private(self);
382 if (az->audiodev != NULL)
383 config_detach(az->audiodev, flags);
384
385 mutex_enter(&az->lock);
386
387#if notyet
388 DPRINTF(("%s: halt streams\n", __func__));
389 azalia_stream_halt(&az->rstream);
390 azalia_stream_halt(&az->pstream);
391#endif
392
393
394 DPRINTF(("%s: delete streams\n", __func__));
395 azalia_stream_delete(&az->rstream, az);
396 azalia_stream_delete(&az->pstream, az);
397
398 DPRINTF(("%s: delete codecs\n", __func__));
399 for (i = 0; i < az->ncodecs; i++) {
400 azalia_codec_delete(&az->codecs[i]);
401 }
402 az->ncodecs = 0;
403
404 DPRINTF(("%s: delete CORB and RIRB\n", __func__));
405 azalia_delete_corb(az);
406 azalia_delete_rirb(az);
407
408 mutex_exit(&az->lock);
409 mutex_destroy(&az->lock);
410 mutex_destroy(&az->intr_lock);
411
412 DPRINTF(("%s: delete PCI resources\n", __func__));
413 if (az->ih != NULL) {
414 pci_intr_disestablish(az->pc, az->ih);
415 az->ih = NULL;
416 }
417 if (az->map_size != 0) {
418 bus_space_unmap(az->iot, az->ioh, az->map_size);
419 az->map_size = 0;
420 }
421 return 0;
422}

424static bool
425azalia_pci_resume(device_t dv, const pmf_qual_t *qual)
426{
427 azalia_t *az = device_private(dv);
428
429 mutex_enter(&az->lock);

434 mutex_exit(&az->lock);
435
436 return true;
437}

439static int
440azalia_intr(void *v)
441{
442 azalia_t *az;
443 int ret;
444 uint32_t intsts;
445 uint8_t rirbsts;
446
447 az = v;
448
449 if (!device_has_power(az->dev))
450 return 0;
451
452 mutex_spin_enter(&az->intr_lock);
453
454 intsts = AZ_READ_4(az, INTSTS);
455 if (intsts == 0) {
456 mutex_spin_exit(&az->intr_lock);
457 return 0;
458 }
459
460 ret = azalia_stream_intr(&az->pstream, intsts) +
461 azalia_stream_intr(&az->rstream, intsts);
462
463 rirbsts = AZ_READ_1(az, RIRBSTS);
464 if (rirbsts & (HDA_RIRBSTS_RIRBOIS | HDA_RIRBSTS_RINTFL)) {
465 if (rirbsts & HDA_RIRBSTS_RINTFL) {
466 azalia_rirb_intr(az);
467 } else {
468 /*printf("[Overflow!]");*/
469 }
470 AZ_WRITE_1(az, RIRBSTS,
471 rirbsts | HDA_RIRBSTS_RIRBOIS | HDA_RIRBSTS_RINTFL);
472 ret++;
473 }
474
475 mutex_spin_exit(&az->intr_lock);
476 return ret;
477}

483static int
484azalia_attach(azalia_t *az)
485{
486 int i, n;
487 uint32_t gctl;
488 uint16_t gcap;
489 uint16_t statests;
490
491 if (az->audiodev == NULL)
492 aprint_normal(", HDA rev. %d.%d\n",
493 AZ_READ_1(az, VMAJ), AZ_READ_1(az, VMIN));
494
495 gcap = AZ_READ_2(az, GCAP);
496 az->nistreams = HDA_GCAP_ISS(gcap);
497 az->nostreams = HDA_GCAP_OSS(gcap);
498 az->nbstreams = HDA_GCAP_BSS(gcap);
499 az->ok64 = (gcap & HDA_GCAP_64OK) != 0;
500 DPRINTF(("%s: host: %d output, %d input, and %d bidi streams\n",
501 device_xname(az->dev), az->nostreams, az->nistreams, az->nbstreams));
502 if (az->nistreams > 0)
503 az->mode_cap |= AUMODE_RECORD;
504 if (az->nostreams > 0)
505 az->mode_cap |= AUMODE_PLAY;
506
507 /* 4.2.2 Starting the High Definition Audio Controller */
508 DPRINTF(("%s: resetting\n", __func__));
509 gctl = AZ_READ_4(az, GCTL);
510 AZ_WRITE_4(az, GCTL, gctl & ~HDA_GCTL_CRST);
511 for (i = 5000; i >= 0; i--) {
512 DELAY(10);
513 if ((AZ_READ_4(az, GCTL) & HDA_GCTL_CRST) == 0)
514 break;
515 }
516 DPRINTF(("%s: reset counter = %d\n", __func__, i));
517 if (i <= 0) {
518 aprint_error_dev(az->dev, "reset failure\n");
519 return ETIMEDOUT;
520 }
521 DELAY(1000);
522 gctl = AZ_READ_4(az, GCTL);
523 AZ_WRITE_4(az, GCTL, gctl | HDA_GCTL_CRST);
524 for (i = 5000; i >= 0; i--) {
525 DELAY(10);
526 if (AZ_READ_4(az, GCTL) & HDA_GCTL_CRST)
527 break;
528 }
529 DPRINTF(("%s: reset counter = %d\n", __func__, i));
530 if (i <= 0) {
531 aprint_error_dev(az->dev, "reset-exit failure\n");
532 return ETIMEDOUT;
533 }
534
535 /* enable unsolicited response */
536 gctl = AZ_READ_4(az, GCTL);
537 AZ_WRITE_4(az, GCTL, gctl | HDA_GCTL_UNSOL);
538
539 /* 4.3 Codec discovery */
540 DELAY(1000);
541 statests = AZ_READ_2(az, STATESTS);
542 for (i = 0, n = 0; i < 15; i++) {
543 if ((statests >> i) & 1) {
544 DPRINTF(("%s: found a codec at #%d\n",
545 device_xname(az->dev), i));
546 az->codecs[n].address = i;
547 az->codecs[n++].dev = az->dev;
548 }
549 }
550 az->ncodecs = n;
551 if (az->ncodecs < 1) {
552 aprint_error_dev(az->dev, "No HD-Audio codecs\n");
553 return -1;
554 }
555 return 0;
556}

558static void
559azalia_attach_intr(device_t self)
560{
561 azalia_t *az;
562 int err, i, c, reinit;
563
564 az = device_private(self);
565 reinit = az->audiodev == NULL ? 0 : 1;
566
567 AZ_WRITE_2(az, STATESTS, HDA_STATESTS_SDIWAKE);
568 AZ_WRITE_1(az, RIRBSTS, HDA_RIRBSTS_RINTFL | HDA_RIRBSTS_RIRBOIS);
569 AZ_WRITE_4(az, INTSTS, HDA_INTSTS_CIS | HDA_INTSTS_GIS);
570 AZ_WRITE_4(az, DPLBASE, 0);
571 AZ_WRITE_4(az, DPUBASE, 0);
572
573 /* 4.4.1 Command Outbound Ring Buffer */
574 if (azalia_init_corb(az, reinit))
575 goto err_exit;
576 /* 4.4.2 Response Inbound Ring Buffer */
577 if (azalia_init_rirb(az, reinit))
578 goto err_exit;
579
580 AZ_WRITE_4(az, INTCTL,
581 AZ_READ_4(az, INTCTL) | HDA_INTCTL_CIE | HDA_INTCTL_GIE);
582
583 c = -1;
584 for (i = 0; i < az->ncodecs; i++) {
585 err = azalia_codec_init(&az->codecs[i], reinit, az->subid);
586 if (!err && c < 0)
587 c = i;
588 }
589 if (c < 0)
590 goto err_exit;
591 /* Use the first audio codec */
592 az->codecno = c;
593 DPRINTF(("%s: using the #%d codec\n",
594 device_xname(az->dev), az->codecno));
595 if (az->codecs[c].dacs.ngroups <= 0)
596 az->mode_cap &= ~AUMODE_PLAY;
597 if (az->codecs[c].adcs.ngroups <= 0)
598 az->mode_cap &= ~AUMODE_RECORD;
599
600 /* Use stream#1 and #2. Don't use stream#0. */
601 if (reinit == 0) {
602 if (azalia_stream_init(&az->pstream, az, az->nistreams + 0,
603 1, AUMODE_PLAY))
604 goto err_exit;
605 if (azalia_stream_init(&az->rstream, az, 0, 2, AUMODE_RECORD))
606 goto err_exit;
607
608 az->audiodev = audio_attach_mi(&azalia_hw_if, az, az->dev);
609 }
610 return;
611err_exit:
612 azalia_pci_detach(self, 0);
613 return;

616static int
617azalia_init_corb(azalia_t *az, int reinit)
618{
619 int entries, err, i;
620 uint16_t corbrp, corbwp;
621 uint8_t corbsize, cap, corbctl;
622
623 /* stop the CORB */
624 corbctl = AZ_READ_1(az, CORBCTL);
625 if (corbctl & HDA_CORBCTL_CORBRUN) { /* running? */
626 AZ_WRITE_1(az, CORBCTL, corbctl & ~HDA_CORBCTL_CORBRUN);
627 for (i = 5000; i >= 0; i--) {
628 DELAY(10);
629 corbctl = AZ_READ_1(az, CORBCTL);
630 if ((corbctl & HDA_CORBCTL_CORBRUN) == 0)
631 break;
632 }
633 if (i <= 0) {
634 aprint_error_dev(az->dev, "CORB is running\n");
635 return EBUSY;
636 }
637 }
638
639 /* determine CORB size */
640 corbsize = AZ_READ_1(az, CORBSIZE);
641 cap = corbsize & HDA_CORBSIZE_CORBSZCAP_MASK;
642 corbsize &= ~HDA_CORBSIZE_CORBSIZE_MASK;
643 if (cap & HDA_CORBSIZE_CORBSZCAP_256) {
644 entries = 256;
645 corbsize |= HDA_CORBSIZE_CORBSIZE_256;
646 } else if (cap & HDA_CORBSIZE_CORBSZCAP_16) {
647 entries = 16;
648 corbsize |= HDA_CORBSIZE_CORBSIZE_16;
649 } else if (cap & HDA_CORBSIZE_CORBSZCAP_2) {
650 entries = 2;
651 corbsize |= HDA_CORBSIZE_CORBSIZE_2;
652 } else {
653 aprint_error_dev(az->dev, "Invalid CORBSZCAP: 0x%2x\n", cap);
654 return -1;
655 }
656
657 if (reinit == 0) {
658 err = azalia_alloc_dmamem(az, entries * sizeof(corb_entry_t),
659 128, &az->corb_dma);
660 if (err) {
661 aprint_error_dev(az->dev, "can't allocate CORB buffer\n");
662 return err;
663 }
664 }
665 AZ_WRITE_4(az, CORBLBASE, (uint32_t)AZALIA_DMA_DMAADDR(&az->corb_dma));
666 AZ_WRITE_4(az, CORBUBASE, PTR_UPPER32(AZALIA_DMA_DMAADDR(&az->corb_dma)));
667 AZ_WRITE_1(az, CORBSIZE, corbsize);
668 az->corb_size = entries;
669
670 DPRINTF(("%s: CORB allocation succeeded.\n", __func__));
671
672 /* reset CORBRP */
673 corbrp = AZ_READ_2(az, CORBRP);
674 AZ_WRITE_2(az, CORBRP, corbrp | HDA_CORBRP_CORBRPRST);
675 AZ_WRITE_2(az, CORBRP, corbrp & ~HDA_CORBRP_CORBRPRST);
676 for (i = 5000; i >= 0; i--) {
677 DELAY(10);
678 corbrp = AZ_READ_2(az, CORBRP);
679 if ((corbrp & HDA_CORBRP_CORBRPRST) == 0)
680 break;
681 }
682 if (i <= 0) {
683 aprint_error_dev(az->dev, "CORBRP reset failure\n");
684 return -1;
685 }
686 DPRINTF(("%s: CORBWP=%d; size=%d\n", __func__,
687 AZ_READ_2(az, CORBRP) & HDA_CORBRP_CORBRP, az->corb_size));
688
689 /* clear CORBWP */
690 corbwp = AZ_READ_2(az, CORBWP);
691 AZ_WRITE_2(az, CORBWP, corbwp & ~HDA_CORBWP_CORBWP);
692
693 /* Run! */
694 corbctl = AZ_READ_1(az, CORBCTL);
695 AZ_WRITE_1(az, CORBCTL, corbctl | HDA_CORBCTL_CORBRUN);
696 return 0;
697}

814 break;
815 }
816 if (i <= 0) {
817 aprint_error_dev(az->dev, "RIRB is not running\n");
818 return EBUSY;
819 }
820
821 return 0;
822}

879static int
880azalia_get_response(azalia_t *az, uint32_t *result)
881{
882 const rirb_entry_t *rirb;
883 int i;
884 uint16_t wp;
885
886#ifdef DIAGNOSTIC
887 if ((AZ_READ_1(az, RIRBCTL) & HDA_RIRBCTL_RIRBDMAEN) == 0) {
888 aprint_error_dev(az->dev, "RIRB is not running.\n");
889 return -1;
890 }
891#endif
892 for (i = 5000; i >= 0; i--) {
893 wp = AZ_READ_2(az, RIRBWP) & HDA_RIRBWP_RIRBWP;
894 if (az->rirb_rp != wp)
895 break;
896 DELAY(10);
897 }
898 if (i <= 0) {
899 aprint_error_dev(az->dev, "RIRB time out\n");
900 return ETIMEDOUT;
901 }
902 rirb = (rirb_entry_t*)az->rirb_dma.addr;
903 for (;;) {
904 if (++az->rirb_rp >= az->rirb_size)
905 az->rirb_rp = 0;
906 if (rirb[az->rirb_rp].resp_ex & RIRB_RESP_UNSOL) {
907 az->unsolq[az->unsolq_wp].resp = rirb[az->rirb_rp].resp;
908 az->unsolq[az->unsolq_wp++].resp_ex = rirb[az->rirb_rp].resp_ex;
909 az->unsolq_wp %= UNSOLQ_SIZE;
910 } else
911 break;
912 }
913 if (result != NULL)
914 *result = rirb[az->rirb_rp].resp;
915 azalia_rirb_kick_unsol_events(az);
916#if 0
917 DPRINTF(("%s: rirbwp=%d rp=%d resp1=0x%8.8x resp2=0x%8.8x\n",
918 __func__, wp, az->rirb_rp, rirb[az->rirb_rp].resp,
919 rirb[az->rirb_rp].resp_ex));
920 for (i = 0; i < 16 /*az->rirb_size*/; i++) {
921 DPRINTF(("rirb[%d] 0x%8.8x:0x%8.8x ", i, rirb[i].resp, rirb[i].resp_ex));
922 if ((i % 2) == 1)
923 DPRINTF(("\n"));
924 }
925#endif
926 return 0;
927}

929static void
930azalia_rirb_kick_unsol_events(azalia_t *az)
931{
932 if (az->unsolq_kick)
933 return;
934 az->unsolq_kick = TRUE;
935 while (az->unsolq_rp != az->unsolq_wp) {
936 int i;
937 int tag;
938 codec_t *codec;
939 i = RIRB_RESP_CODEC(az->unsolq[az->unsolq_rp].resp_ex);
940 tag = RIRB_UNSOL_TAG(az->unsolq[az->unsolq_rp].resp);
941 codec = &az->codecs[i];
942 DPRINTF(("%s: codec#=%d tag=%d\n", __func__, i, tag));
943 az->unsolq_rp++;
944 az->unsolq_rp %= UNSOLQ_SIZE;
945 if (codec->unsol_event != NULL)
946 codec->unsol_event(codec, tag);
947 }
948 az->unsolq_kick = FALSE;
949}

951static void
952azalia_rirb_intr(azalia_t *az)
953{
954 const rirb_entry_t *rirb;
955 uint16_t wp, newrp;
956
957 wp = AZ_READ_2(az, RIRBWP) & HDA_RIRBWP_RIRBWP;
958 if (az->rirb_rp == wp)
959 return; /* interrupted but no data in RIRB */
960 /* Copy the first sequence of unsolicited reponses in the RIRB to
961 * unsolq. Don't consume non-unsolicited responses. */
962 rirb = (rirb_entry_t*)az->rirb_dma.addr;
963 while (az->rirb_rp != wp) {
964 newrp = az->rirb_rp + 1;
965 if (newrp >= az->rirb_size)
966 newrp = 0;
967 if (rirb[newrp].resp_ex & RIRB_RESP_UNSOL) {
968 az->unsolq[az->unsolq_wp].resp = rirb[newrp].resp;
969 az->unsolq[az->unsolq_wp++].resp_ex = rirb[newrp].resp_ex;
970 az->unsolq_wp %= UNSOLQ_SIZE;
971 az->rirb_rp = newrp;
972 } else {
973 break;
974 }
975 }
976 azalia_rirb_kick_unsol_events(az);
977}

979static int
980azalia_alloc_dmamem(azalia_t *az, size_t size, size_t align, azalia_dma_t *d)
981{
982 int err;
983 int nsegs;
984
985 d->size = size;
986 err = bus_dmamem_alloc(az->dmat, size, align, 0, d->segments, 1,
987 &nsegs, BUS_DMA_WAITOK);
988 if (err)
989 return err;
990 if (nsegs != 1)
991 goto free;
992 err = bus_dmamem_map(az->dmat, d->segments, 1, size,
993 &d->addr, BUS_DMA_WAITOK | BUS_DMA_COHERENT | BUS_DMA_NOCACHE);
994 if (err)
995 goto free;
996 err = bus_dmamap_create(az->dmat, size, 1, size, 0,
997 BUS_DMA_WAITOK, &d->map);
998 if (err)
999 goto unmap;
1000 err = bus_dmamap_load(az->dmat, d->map, d->addr, size,
1001 NULL, BUS_DMA_WAITOK);
1002 if (err)
1003 goto destroy;
1004
1005 if (!az->ok64 && PTR_UPPER32(AZALIA_DMA_DMAADDR(d)) != 0) {
1006 azalia_free_dmamem(az, d);
1007 return -1;
1008 }
1009 return 0;
1010
1011destroy:
1012 bus_dmamap_destroy(az->dmat, d->map);
1013unmap:
1014 bus_dmamem_unmap(az->dmat, d->addr, size);
1015free:
1016 bus_dmamem_free(az->dmat, d->segments, 1);
1017 d->addr = NULL;
1018 return err;
1019}

1038static int
1039azalia_codec_init(codec_t *this, int reinit, uint32_t subid)
1040{
1041#define LEAD_LEN 100
1042 char lead[LEAD_LEN];
1043 uint32_t rev, id, result;
1044 int err, addr, n, i;
1045
1046 this->comresp = azalia_codec_comresp;
1047 addr = this->address;
1048 DPRINTF(("%s: information of codec[%d] follows:\n",
1049 device_xname(this->dev), addr));
1050 /* codec vendor/device/revision */
1051 err = this->comresp(this, CORB_NID_ROOT, CORB_GET_PARAMETER,
1052 COP_REVISION_ID, &rev);
1053 if (err)
1054 return err;
1055 err = this->comresp(this, CORB_NID_ROOT, CORB_GET_PARAMETER,
1056 COP_VENDOR_ID, &id);
1057 if (err)
1058 return err;
1059 this->vid = id;
1060 this->subid = subid;
1061
1062 if (!reinit) {
1063 err = azalia_codec_init_vtbl(this);
1064 if (err)
1065 return err;
1066 }
1067
1068 if (!reinit) {
1069 aprint_normal("%s: codec[%d]: ", device_xname(this->dev), addr);
1070 if (this->name == NULL) {
1071 aprint_normal("0x%4.4x/0x%4.4x (rev. %u.%u)",
1072 id >> 16, id & 0xffff,
1073 COP_RID_REVISION(rev), COP_RID_STEPPING(rev));
1074 } else {
1075 aprint_normal("%s (rev. %u.%u)", this->name,
1076 COP_RID_REVISION(rev), COP_RID_STEPPING(rev));
1077 }
1078 aprint_normal(", HDA rev. %u.%u\n",
1079 COP_RID_MAJ(rev), COP_RID_MIN(rev));
1080 }
1081
1082 /* identify function nodes */
1083 err = this->comresp(this, CORB_NID_ROOT, CORB_GET_PARAMETER,
1084 COP_SUBORDINATE_NODE_COUNT, &result);
1085 if (err)
1086 return err;
1087 this->nfunctions = COP_NSUBNODES(result);
1088 if (COP_NSUBNODES(result) <= 0) {
1089 aprint_error("%s: No function groups\n",
1090 device_xname(this->dev));
1091 return -1;
1092 }
1093 /* iterate function nodes and find an audio function */
1094 n = COP_START_NID(result);
1095 DPRINTF(("%s: nidstart=%d #functions=%d\n",
1096 __func__, n, this->nfunctions));
1097 this->audiofunc = -1;
1098 for (i = 0; i < this->nfunctions; i++) {
1099 err = this->comresp(this, n + i, CORB_GET_PARAMETER,
1100 COP_FUNCTION_GROUP_TYPE, &result);
1101 if (err)
1102 continue;
1103 DPRINTF(("%s: FTYPE result = 0x%8.8x\n", __func__, result));
1104 if (COP_FTYPE(result) == COP_FTYPE_AUDIO) {
1105 this->audiofunc = n + i;
1106 break; /* XXX multiple audio functions? */
1107 } else if (COP_FTYPE(result) == COP_FTYPE_MODEM && !reinit) {
1108 aprint_normal("%s: codec[%d]: No support for modem "
1109 "function groups\n",
1110 device_xname(this->dev), addr);
1111 }
1112 }
1113 if (this->audiofunc < 0 && !reinit) {
1114 aprint_verbose("%s: codec[%d] has no audio function groups\n",
1115 device_xname(this->dev), addr);
1116 return -1;
1117 }
1118
1119 /* power the audio function */
1120 this->comresp(this, this->audiofunc, CORB_SET_POWER_STATE, CORB_PS_D0, &result);
1121 DELAY(100);
1122
1123 /* check widgets in the audio function */
1124 err = this->comresp(this, this->audiofunc,
1125 CORB_GET_PARAMETER, COP_SUBORDINATE_NODE_COUNT, &result);
1126 if (err)
1127 return err;
1128 DPRINTF(("%s: There are %d widgets in the audio function.\n",
1129 __func__, COP_NSUBNODES(result)));
1130 this->wstart = COP_START_NID(result);
1131 if (this->wstart < 2) {
1132 if (!reinit)
1133 aprint_error("%s: invalid node structure\n",
1134 device_xname(this->dev));
1135 return -1;
1136 }
1137 this->wend = this->wstart + COP_NSUBNODES(result);
1138 if (!reinit) {
1139 this->w = kmem_zalloc(sizeof(widget_t) * this->wend,
1140 KM_SLEEP);
1141 if (this->w == NULL) {
1142 aprint_error("%s: out of memory\n",
1143 device_xname(this->dev));
1144 return ENOMEM;
1145 }
1146 } else
1147 memset(this->w, 0, sizeof(widget_t) * this->wend);
1148
1149 /* query the base parameters */
1150 this->comresp(this, this->audiofunc, CORB_GET_PARAMETER,
1151 COP_STREAM_FORMATS, &result);
1152 this->w[this->audiofunc].d.audio.encodings = result;
1153 this->comresp(this, this->audiofunc, CORB_GET_PARAMETER,
1154 COP_PCM, &result);
1155 this->w[this->audiofunc].d.audio.bits_rates = result;
1156 this->comresp(this, this->audiofunc, CORB_GET_PARAMETER,
1157 COP_INPUT_AMPCAP, &result);
1158 this->w[this->audiofunc].inamp_cap = result;
1159 this->comresp(this, this->audiofunc, CORB_GET_PARAMETER,
1160 COP_OUTPUT_AMPCAP, &result);
1161 this->w[this->audiofunc].outamp_cap = result;
1162 lead[0] = 0;
1163#ifdef AZALIA_DEBUG
1164 snprintf(lead, LEAD_LEN, "%s: ", device_xname(this->dev));
1165 azalia_widget_print_audio(&this->w[this->audiofunc], lead, -1);
1166 result = this->w[this->audiofunc].inamp_cap;
1167 DPRINTF(("%sinamp: mute=%u size=%u steps=%u offset=%u\n", lead,
1168 (result & COP_AMPCAP_MUTE) != 0, COP_AMPCAP_STEPSIZE(result),

1323 chan = 0;

1325 if (newdac >= 0 && newdac < this->dacs.ngroups) {

1348 if (newadc >= 0 && newadc < this->adcs.ngroups) {

1352 nbits = 0;

1369 }

1384 nid = group->conv[i];

1386 bits_rates &= this->w[nid].d.audio.bits_rates;

1389 bits_rates, AUMODE_PLAY);
1390 }

1398 }
1399#endif
1400 }
1401
1402 /* register formats for recording */
1403 if (this->adcs.cur >= 0 && this->adcs.cur < this->adcs.ngroups) {
1404 group = &this->adcs.groups[this->adcs.cur];
1405 for (c = 0; c < group->nconv; c++) {
1406 chan = 0;
1407 bits_rates = ~0;
1408 for (i = 0; i <= c; i++) {
1409 nid = group->conv[i];
1410 chan += WIDGET_CHANNELS(&this->w[nid]);
1411 bits_rates &= this->w[nid].d.audio.bits_rates;
1412 }
1413 azalia_codec_add_bits(this, chan,
1414 bits_rates, AUMODE_RECORD);
1415 }
1416#ifdef AZALIA_DEBUG
1417 /* print recording capability */
1418 if (prev_adc != this->adcs.cur) {
1419 snprintf(flagbuf, FLAGBUFLEN, "%s: recording: ",
1420 device_xname(this->dev));
1421 azalia_widget_print_audio(&this->w[group->conv[0]],
1422 flagbuf, chan);
1423 }
1424#endif
1425 }
1426
1427#ifdef DIAGNOSTIC
1428 if (this->nformats > variation) {
1429 aprint_error("%s: Internal error: the format buffer is too small: "
1430 "nformats=%d variation=%d\n", device_xname(this->dev),
1431 this->nformats, variation);
1432 return ENOMEM;
1433 }
1434#endif
1435
1436 return 0;
1437}

1528static int
1529azalia_codec_connect_stream(codec_t *this, int dir, uint16_t fmt, int number)
1530{
1531 const convgroup_t *group;
1532 uint32_t v;
1533 int i, err, startchan, nchan;
1534 nid_t nid;
1535 bool flag222;
1536
1537 DPRINTF(("%s: fmt=0x%4.4x number=%d\n", __func__, fmt, number));
1538 err = 0;
1539 if (dir == AUMODE_RECORD)
1540 group = &this->adcs.groups[this->adcs.cur];
1541 else
1542 group = &this->dacs.groups[this->dacs.cur];
1543 flag222 = group->nconv >= 3 &&
1544 (WIDGET_CHANNELS(&this->w[group->conv[0]]) == 2) &&
1545 (WIDGET_CHANNELS(&this->w[group->conv[1]]) == 2) &&
1546 (WIDGET_CHANNELS(&this->w[group->conv[2]]) == 2);
1547 nchan = (fmt & HDA_SD_FMT_CHAN) + 1;
1548 startchan = 0;
1549 for (i = 0; i < group->nconv; i++) {
1550 uint32_t stream_chan;
1551 nid = group->conv[i];
1552
1553 /* surround and c/lfe handling */
1554 if (nchan >= 6 && flag222 && i == 1) {
1555 nid = group->conv[2];
1556 } else if (nchan >= 6 && flag222 && i == 2) {
1557 nid = group->conv[1];
1558 }
1559
1560 err = this->comresp(this, nid, CORB_SET_CONVERTER_FORMAT, fmt, NULL);
1561 if (err)
1562 goto exit;
1563 stream_chan = (number << 4) | startchan;
1564 if (startchan >= nchan)
1565 stream_chan = 0; /* stream#0 */
1566 err = this->comresp(this, nid, CORB_SET_CONVERTER_STREAM_CHANNEL,
1567 stream_chan, NULL);
1568 if (err)
1569 goto exit;
1570 if (this->w[nid].widgetcap & COP_AWCAP_DIGITAL) {
1571 /* enable S/PDIF */
1572 this->comresp(this, nid, CORB_GET_DIGITAL_CONTROL,
1573 0, &v);
1574 v = (v & 0xff) | CORB_DCC_DIGEN;
1575 this->comresp(this, nid, CORB_SET_DIGITAL_CONTROL_L,
1576 v, NULL);
1577 }
1578 startchan += WIDGET_CHANNELS(&this->w[nid]);
1579 }
1580
1581exit:
1582 DPRINTF(("%s: leave with %d\n", __func__, err));
1583 return err;
1584}

1988static int
1989azalia_stream_start(stream_t *this, void *start, void *end, int blk,
1990 void (*intr)(void *), void *arg, uint16_t fmt)
1991{
1992 bdlist_entry_t *bdlist;
1993 bus_addr_t dmaaddr;
1994 int err, index;
1995 uint32_t intctl;
1996 uint8_t ctl, ctl2;
1997
1998 DPRINTF(("%s: start=%p end=%p\n", __func__, start, end));
1999 if (this->bdlist.addr == NULL)
2000 return EINVAL;
2001 this->intr = intr;
2002 this->intr_arg = arg;
2003
2004 err = azalia_stream_reset(this);
2005 if (err)
2006 return err;
2007
2008 /* setup BDL */
2009 dmaaddr = AZALIA_DMA_DMAADDR(&this->buffer);
2010 this->dmaend = dmaaddr + ((char *)end - (char *)start);
2011 bdlist = (bdlist_entry_t*)this->bdlist.addr;
2012 for (index = 0; index < HDA_BDL_MAX; index++) {
2013 bdlist[index].low = dmaaddr;
2014 bdlist[index].high = PTR_UPPER32(dmaaddr);
2015 bdlist[index].length = blk;
2016 bdlist[index].flags = BDLIST_ENTRY_IOC;
2017 dmaaddr += blk;
2018 if (dmaaddr >= this->dmaend) {
2019 index++;
2020 break;
2021 }
2022 }
2023 /* The BDL covers the whole of the buffer. */
2024 this->dmanext = AZALIA_DMA_DMAADDR(&this->buffer);
2025
2026 dmaaddr = AZALIA_DMA_DMAADDR(&this->bdlist);
2027 STR_WRITE_4(this, BDPL, dmaaddr);
2028 STR_WRITE_4(this, BDPU, PTR_UPPER32(dmaaddr));
2029 STR_WRITE_2(this, LVI, (index - 1) & HDA_SD_LVI_LVI);
2030 ctl2 = STR_READ_1(this, CTL2);
2031 STR_WRITE_1(this, CTL2,
2032 (ctl2 & ~HDA_SD_CTL2_STRM) | (this->number << HDA_SD_CTL2_STRM_SHIFT));
2033 STR_WRITE_4(this, CBL, ((char *)end - (char *)start));
2034
2035 STR_WRITE_2(this, FMT, fmt);
2036
2037 err = azalia_codec_connect_stream(&this->az->codecs[this->az->codecno],
2038 this->dir, fmt, this->number);
2039 if (err)
2040 return EINVAL;
2041
2042 intctl = AZ_READ_4(this->az, INTCTL);
2043 intctl |= this->intr_bit;
2044 AZ_WRITE_4(this->az, INTCTL, intctl);
2045
2046 ctl = STR_READ_1(this, CTL);
2047 ctl |= ctl | HDA_SD_CTL_DEIE | HDA_SD_CTL_FEIE | HDA_SD_CTL_IOCE | HDA_SD_CTL_RUN;
2048 STR_WRITE_1(this, CTL, ctl);
2049 return 0;
2050}
