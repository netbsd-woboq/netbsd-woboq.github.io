33/*
34 * Machine-independent ns16x50 serial port ('com') driver attachment to
35 * "PCI Universal Communications" controller driver.
36 *
37 * Author: Christopher G. Demetriou, May 17, 1998.
38 */
39
40#include <sys/cdefs.h>
41__KERNEL_RCSID(0, "$NetBSD: com_puc.c,v 1.26 2018/12/08 17:46:14 thorpej Exp $");
42
43#include <sys/param.h>
44#include <sys/systm.h>
45#include <sys/device.h>
46#include <sys/tty.h>
47
48#include <sys/bus.h>
49
50#include <dev/pci/pcivar.h>
51#include <dev/pci/pucvar.h>
52#include <dev/ic/comreg.h>
53#include <dev/ic/comvar.h>
54#include <dev/pci/cybervar.h>
55
56struct com_puc_softc {
57 struct com_softc sc_com; /* real "com" softc */
58
59 /* puc-specific goo. */
60 void *sc_ih; /* interrupt handler */
61};
62
63/* Interface field in PCI Class register */
64static const char *serialtype[] = {
65 "Generic XT",
66 "16450",
67 "16550",
68 "16650",
69 "16750",
70 "16850",
71 "16950",
72};
73
74static int
75com_puc_probe(device_t parent, cfdata_t match, void *aux)
76{
77 struct puc_attach_args *aa = aux;
78
79 /*
80 * Locators already matched, just check the type.
81 */
82 if (aa->type != PUC_PORT_TYPE_COM)
83 return (0);
84
85 return (1);
86}
87
88static void
89com_puc_attach(device_t parent, device_t self, void *aux)
90{
91 struct com_puc_softc *psc = device_private(self);
92 struct com_softc *sc = &psc->sc_com;
93 struct puc_attach_args *aa = aux;
94 const char *intrstr = NULL;
95 char intrbuf[PCI_INTRSTR_LEN];
96 unsigned int iface;
97
98 sc->sc_dev = self;
99
100 iface = PCI_INTERFACE(pci_conf_read(aa->pc, aa->tag, PCI_CLASS_REG));
101 aprint_naive(": Serial port");
102 if (iface < __arraycount(serialtype))
103 aprint_normal(" (%s-compatible)", serialtype[iface]);
104 aprint_normal(": ");
105
106 com_init_regs(&sc->sc_regs, aa->t, aa->h, aa->a);
107 sc->sc_frequency = aa->flags & PUC_COM_CLOCKMASK;
108
109 if (!aa->poll) {
110 intrstr = pci_intr_string(aa->pc, aa->intrhandle, intrbuf,
111 sizeof(intrbuf));
112 psc->sc_ih = pci_intr_establish_xname(aa->pc, aa->intrhandle,
113 IPL_SERIAL, comintr, sc, device_xname(self));
114 if (psc->sc_ih == NULL) {
115 aprint_error("couldn't establish interrupt");
116 if (intrstr != NULL)
117 aprint_error(" at %s", intrstr);
118 aprint_error("\n");
119 return;
120 }
121 } else {
122 sc->sc_hwflags |= COM_HW_POLL;
123 }
124
125#if defined(amd64) || defined(i386)
126 /*
127 * Since puc(4) serial ports are typically not identified in the
128 * BIOS COM[1234] table, the I/O address must be manually set using
129 * installboot(8) in order to enable a serial console.
130 * Print the address here so the user doesn't have to dig through
131 * PCI configuration space to find it.
132 */
133 if (aa->h < 0x10000)
134 aprint_normal("ioaddr 0x%04lx, ", aa->h);
135#endif
136 if (!aa->poll)
137 aprint_normal("interrupting at %s\n", intrstr);
138 else
139 aprint_normal("polling\n");
140
141 /* Enable Cyberserial 8X clock. */
142 if (aa->flags & (PUC_COM_SIIG10x|PUC_COM_SIIG20x)) {
143 int usrregno;
144
145 if (aa->flags & PUC_PORT_USR3) usrregno = 3;
146 else if (aa->flags & PUC_PORT_USR2) usrregno = 2;
147 else if (aa->flags & PUC_PORT_USR1) usrregno = 1;
148 else /* (aa->flags & PUC_PORT_USR0) */ usrregno = 0;
149
150 if (aa->flags & PUC_COM_SIIG10x)
151 write_siig10x_usrreg(aa->pc, aa->tag, usrregno, 1);
152 else
153 write_siig20x_usrreg(aa->pc, aa->tag, usrregno, 1);
154 } else {
155 if (!pmf_device_register(self, NULL, com_resume))
156 aprint_error_dev(self,
157 "couldn't establish power handler\n");
158 }
159
160 aprint_normal("%s", device_xname(self));
161 com_attach_subr(sc);
162}
163
164CFATTACH_DECL_NEW(com_puc, sizeof(struct com_puc_softc),
165 com_puc_probe, com_puc_attach, NULL, NULL);
