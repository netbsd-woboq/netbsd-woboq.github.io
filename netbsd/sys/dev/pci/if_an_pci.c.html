72#define AN_PCI_PLX_IOBA 0x14 /* i/o base for PLX chip */
73#define AN_PCI_IOBA PCI_BAR(2) /* i/o base */

75struct an_pci_softc {
76 struct an_softc sc_an; /* real "an" softc */
77 pci_chipset_tag_t sc_pct;
78 pcitag_t sc_pcitag;
79
80 /* PCI-specific goo. */
81 void *sc_ih; /* interrupt handle */
82};
83
84static int an_pci_match(device_t, cfdata_t, void *);
85static void an_pci_attach(device_t, device_t, void *);
86
87CFATTACH_DECL_NEW(an_pci, sizeof(struct an_pci_softc),
88 an_pci_match, an_pci_attach, NULL, NULL);
89
90static const struct an_pci_product {
91 u_int32_t app_vendor; /* PCI vendor ID */
92 u_int32_t app_product; /* PCI product ID */
93} an_pci_products[] = {
94 { PCI_VENDOR_AIRONET, PCI_PRODUCT_AIRONET_PC4xxx },
95 { PCI_VENDOR_AIRONET, PCI_PRODUCT_AIRONET_PC4500 },
96 { PCI_VENDOR_AIRONET, PCI_PRODUCT_AIRONET_PC4800 },
97 { PCI_VENDOR_AIRONET, PCI_PRODUCT_AIRONET_PCI350 },
98 { PCI_VENDOR_AIRONET, PCI_PRODUCT_AIRONET_MPI350 },
99 { 0, 0 }
100};
101
102static int
103an_pci_match(device_t parent, cfdata_t match, void *aux)
104{
105 struct pci_attach_args *pa = aux;
106 const struct an_pci_product *app;
107
108 for (app = an_pci_products; app->app_vendor != 0; app++) {
109 if (PCI_VENDOR(pa->pa_id) == app->app_vendor &&
110 PCI_PRODUCT(pa->pa_id) == app->app_product)
111 return 1;
112 }
113 return 0;
114}
115
116static void
117an_pci_attach(device_t parent, device_t self, void *aux)
118{
119 struct pci_attach_args *pa = (struct pci_attach_args *)aux;
120 struct an_pci_softc *psc = device_private(self);
121 struct an_softc *sc = &psc->sc_an;
122 char const *intrstr;
123 pci_intr_handle_t ih;
124 bus_size_t iosize;
125 u_int32_t csr;
126 char intrbuf[PCI_INTRSTR_LEN];
127
128 sc->sc_dev = self;
129 psc->sc_pct = pa->pa_pc;
130 psc->sc_pcitag = pa->pa_tag;
131
132 pci_aprint_devinfo(pa, "802.11 controller");
133
134 /* Map I/O registers */
135 if (pci_mapreg_map(pa, AN_PCI_IOBA, PCI_MAPREG_TYPE_IO, 0,
136 &sc->sc_iot, &sc->sc_ioh, NULL, &iosize) != 0) {
137 aprint_error_dev(self, "unable to map registers\n");
138 return;
139 }
140
141 /* Enable the device. */
142 csr = pci_conf_read(pa->pa_pc, pa->pa_tag, PCI_COMMAND_STATUS_REG);
143 pci_conf_write(pa->pa_pc, pa->pa_tag, PCI_COMMAND_STATUS_REG,
144 csr | PCI_COMMAND_MASTER_ENABLE);
145
146 /* Map and establish the interrupt. */
147 if (pci_intr_map(pa, &ih)) {
148 aprint_error_dev(self, "unable to map interrupt\n");
149 return;
150 }
151 intrstr = pci_intr_string(pa->pa_pc, ih, intrbuf, sizeof(intrbuf));
152 psc->sc_ih = pci_intr_establish_xname(pa->pa_pc, ih, IPL_NET, an_intr,
153 sc, device_xname(self));
154 if (psc->sc_ih == NULL) {
155 aprint_error_dev(self, "unable to establish interrupt");
156 if (intrstr != NULL)
157 aprint_error(" at %s", intrstr);
158 aprint_error("\n");
159 return;
160 }
161 aprint_normal_dev(self, "interrupting at %s\n", intrstr);
162 sc->sc_enabled = 1;
163
164 if (an_attach(sc) != 0) {
165 aprint_error_dev(self, "failed to attach controller\n");
166 pci_intr_disestablish(pa->pa_pc, psc->sc_ih);
167 bus_space_unmap(sc->sc_iot, sc->sc_ioh, iosize);
168 }
169
170 if (pmf_device_register(self, NULL, NULL))
171 pmf_class_network_register(self, &sc->sc_if);
172 else
173 aprint_error_dev(self, "couldn't establish power handler\n");
174}
