98#define DEC_VENDORID 0x1011
99#define DEFPA_CHIPID 0x000F
100#define PCI_VENDORID(x) ((x) & 0xFFFF)
101#define PCI_CHIPID(x) (((x) >> 16) & 0xFFFF)
102
103#define DEFPA_LATENCY 0x88
104
105#define PCI_CFLT 0x0C /* Configuration Latency */
106#define PCI_CBMA 0x10 /* Configuration Base Memory Address */
107#define PCI_CBIO 0x14 /* Configuration Base I/O Address */

163/*
164 * This is the PCI configuration support. Since the PDQ is available
165 * on both EISA and PCI boards, one must be careful in how defines the
166 * PDQ in the config file.
167 */
168static char *
169pdq_pci_probe(
170 pcici_t config_id,
171 pcidi_t device_id)
172{
173 if (PCI_VENDORID(device_id) == DEC_VENDORID &&
174 PCI_CHIPID(device_id) == DEFPA_CHIPID)
175 return "Digital DEFPA PCI FDDI Controller";
176 return NULL;
177}

179static void
180pdq_pci_attach(
181 pcici_t config_id,
182 int unit)
183{
184 pdq_softc_t *sc;
185 vaddr_t va_csrs;
186 paddr_t pa_csrs;
187 pdq_uint32_t data;
188
189 if (unit == NFPA) {
190 printf("fpa%d: not configured; kernel is built for only %d device%s.\n",
191 unit, NFPA, NFPA == 1 ? "" : "s");
192 return;
193 }
194
195 data = pci_conf_read(config_id, PCI_CFLT);
196 if ((data & 0xFF00) < (DEFPA_LATENCY << 8)) {
197 data &= ~0xFF00;
198 data |= DEFPA_LATENCY << 8;
199 pci_conf_write(config_id, PCI_CFLT, data);
200 }
201
202 sc = (pdq_softc_t *) malloc(sizeof(*sc), M_DEVBUF, M_NOWAIT|M_ZERO);
203 if (sc == NULL)
204 return;
205
206 if (!pci_map_mem(config_id, PCI_CBMA, &va_csrs, &pa_csrs)) {
207 free((void *) sc, M_DEVBUF);
208 return;
209 }
210
211 sc->sc_if.if_name = "fpa";
212 sc->sc_if.if_unit = unit;
213 sc->sc_membase = (pdq_bus_memaddr_t) va_csrs;
214 sc->sc_pdq = pdq_initialize(PDQ_BUS_PCI, sc->sc_membase,
215 sc->sc_if.if_name, sc->sc_if.if_unit,
216 (void *) sc, PDQ_DEFPA);
217 if (sc->sc_pdq == NULL) {
218 free((void *) sc, M_DEVBUF);
219 return;
220 }
221 memcpy(sc->sc_ac.ac_enaddr, (void *) sc->sc_pdq->pdq_hwaddr.lanaddr_bytes, 6);
222 pdqs_pci[unit] = sc;
223 pdq_ifattach(sc, pdq_pci_ifwatchdog);
224 pci_map_int(config_id, pdq_pci_ifintr, (void*) sc, &net_imask);
225#if BSD >= 199506
226 at_shutdown(pdq_pci_shutdown, (void *) sc, SHUTDOWN_POST_SYNC);
227#endif
228}

268static int
269pdq_pci_match(
270 pci_devaddr_t *pa)
271{
272 int irq;
273 int id;
274
275 id = pci_inl(pa, PCI_VENDOR_ID);
276 if (PCI_VENDORID(id) != DEC_VENDORID || PCI_CHIPID(id) != DEFPA_CHIPID)
277 return 0;
278
279 irq = pci_inl(pa, PCI_I_LINE) & 0xFF;
280 if (irq == 0 || irq >= 16)
281 return 0;
282
283 return 1;
284}

286int
287pdq_pci_probe(device_t parent, cfdata_t cf, void *aux)
288{
289 struct isa_attach_args *ia = (struct isa_attach_args *) aux;
290 pdq_uint32_t irq, data;
291 pci_devaddr_t *pa;
292
293 pa = pci_scan(pdq_pci_match);
294 if (pa == NULL)
295 return 0;
296
297 irq = (1 << (pci_inl(pa, PCI_I_LINE) & 0xFF));
298
299 if (ia->ia_irq != IRQUNK && irq != ia->ia_irq) {
300 printf("fpa%d: error: desired IRQ of %d does not match device's actual IRQ of %d\n",
301 cf->cf_unit,
302 ffs(ia->ia_irq) - 1, ffs(irq) - 1);
303 return 0;
304 }
305 if (ia->ia_irq == IRQUNK) {
306 (void) isa_irqalloc(irq);
307 ia->ia_irq = irq;
308 }
309
310 /* PCI bus masters don't use host DMA channels */
311 ia->ia_drq = DRQNONE;
312
313 /* Get the memory base address; assume the BIOS set it up correctly */
314 ia->ia_maddr = (void *) (pci_inl(pa, PCI_CBMA) & ~7);
315 pci_outl(pa, PCI_CBMA, 0xFFFFFFFF);
316 ia->ia_msize = ((~pci_inl(pa, PCI_CBMA)) | 7) + 1;
317 pci_outl(pa, PCI_CBMA, (int) ia->ia_maddr);
318
319 /* Disable I/O space access */
320 pci_outl(pa, PCI_COMMAND, pci_inl(pa, PCI_COMMAND) & ~1);
321 ia->ia_iobase = 0;
322 ia->ia_iosize = 0;
323
324 /* Make sure the latency timer is what the DEFPA likes */
325 data = pci_inl(pa, PCI_CFLT);
326 if ((data & 0xFF00) < (DEFPA_LATENCY << 8)) {
327 data &= ~0xFF00;
328 data |= DEFPA_LATENCY << 8;
329 pci_outl(pa, PCI_CFLT, data);
330 }
331 ia->ia_irq |= IRQSHARE;
332
333 return 1;
334}

336void
337pdq_pci_attach(device_t parent, device_t self, void *aux)
338{
339 pdq_softc_t *sc = device_private(self);
340 struct isa_attach_args *ia = (struct isa_attach_args *) aux;
341 struct ifnet *ifp = &sc->sc_if;
342 int i;
343
344 sc->sc_dev = self;
345 sc->sc_if.if_unit = sc->sc_dev.dv_unit;
346 sc->sc_if.if_name = "fpa";
347 sc->sc_if.if_flags = 0;
348 sc->sc_membase = (pdq_bus_memaddr_t) mapphys((vaddr_t)ia->ia_maddr, ia->ia_msize);
349
350 sc->sc_pdq = pdq_initialize(PDQ_BUS_PCI, sc->sc_membase,
351 sc->sc_if.if_name, sc->sc_if.if_unit,
352 (void *) sc, PDQ_DEFPA);
353 if (sc->sc_pdq == NULL) {
354 printf("fpa%d: initialization failed\n", sc->sc_if.if_unit);
355 return;
356 }
357
358 memcpy(sc->sc_ac.ac_enaddr, (void *) sc->sc_pdq->pdq_hwaddr.lanaddr_bytes, 6);
359
360 pdq_ifattach(sc, pdq_pci_ifwatchdog);
361
362 isa_establish(&sc->sc_id, &sc->sc_dev);
363
364 sc->sc_ih.ih_fun = pdq_pci_ifintr;
365 sc->sc_ih.ih_arg = (void *)sc;
366 intr_establish(ia->ia_irq, &sc->sc_ih, DV_NET);
367
368 sc->sc_ats.func = (void (*)(void *)) pdq_hwreset;
369 sc->sc_ats.arg = (void *) sc->sc_pdq;
370 atshutdown(&sc->sc_ats, ATSH_ADD);
371}

383static int
384pdq_pci_match(device_t parent, cfdata_t match, void *aux)
385{
386 struct pci_attach_args *pa = (struct pci_attach_args *) aux;
387
388 if (PCI_VENDOR(pa->pa_id) != PCI_VENDOR_DEC)
389 return 0;
390 if (PCI_PRODUCT(pa->pa_id) != PCI_PRODUCT_DEC_DEFPA)
391 return 0;
392
393 return 1;
394}

396static void
397pdq_pci_attach(device_t const parent, device_t const self, void *const aux)
398{
399 pdq_softc_t * const sc = device_private(self);
400 struct pci_attach_args * const pa = (struct pci_attach_args *) aux;
401 pdq_uint32_t data;
402 pci_intr_handle_t intrhandle;
403 const char *intrstr;
404 bus_space_tag_t iot, memt;
405 bus_space_handle_t ioh, memh;
406 int ioh_valid, memh_valid;
407 char intrbuf[PCI_INTRSTR_LEN];
408
409 aprint_naive(": FDDI controller\n");
410
411 sc->sc_dev = self;
412
413 data = pci_conf_read(pa->pa_pc, pa->pa_tag, PCI_CFLT);
414 if ((data & 0xFF00) < (DEFPA_LATENCY << 8)) {
415 data &= ~0xFF00;
416 data |= DEFPA_LATENCY << 8;
417 pci_conf_write(pa->pa_pc, pa->pa_tag, PCI_CFLT, data);
418 }
419
420 strlcpy(sc->sc_if.if_xname, device_xname(sc->sc_dev), IFNAMSIZ);
421 sc->sc_if.if_flags = 0;
422 sc->sc_if.if_softc = sc;
423
424 ioh_valid = (pci_mapreg_map(pa, PCI_CBIO, PCI_MAPREG_TYPE_IO, 0,
425 &iot, &ioh, NULL, NULL) == 0);
426 memh_valid = (pci_mapreg_map(pa, PCI_CBMA,
427 PCI_MAPREG_TYPE_MEM | PCI_MAPREG_MEM_TYPE_32BIT, 0,
428 &memt, &memh, NULL, NULL) == 0);
429
430#if defined(DEFPA_IOMAPED)
431 if (ioh_valid) {
432 sc->sc_csrtag = iot;
433 sc->sc_membase = ioh;
434 } else if (memh_valid) {
435 sc->sc_csrtag = memt;
436 sc->sc_membase = memh;
437 }
438#else /* defined(DEFPA_IOMAPPED) */
439 if (memh_valid) {
440 sc->sc_csrtag = memt;
441 sc->sc_membase = memh;
442 } else if (ioh_valid) {
443 sc->sc_csrtag = iot;
444 sc->sc_membase = ioh;
445 }
446#endif /* DEFPA_IOMAPPED */
447 else {
448 aprint_error(": unable to map device registers\n");
449 return;
450 }
451
452 sc->sc_dmatag = pa->pa_dmat;
453
454 /* Make sure bus mastering is enabled. */
455 pci_conf_write(pa->pa_pc, pa->pa_tag, PCI_COMMAND_STATUS_REG,
456 pci_conf_read(pa->pa_pc, pa->pa_tag,
457 PCI_COMMAND_STATUS_REG) |
458 PCI_COMMAND_MASTER_ENABLE);
459
460 sc->sc_pdq = pdq_initialize(sc->sc_csrtag, sc->sc_membase,
461 sc->sc_if.if_xname, 0,
462 (void *) sc, PDQ_DEFPA);
463 if (sc->sc_pdq == NULL) {
464 aprint_error_dev(sc->sc_dev, "initialization failed\n");
465 return;
466 }
467
468 pdq_ifattach(sc, pdq_pci_ifwatchdog);
469
470 if (pci_intr_map(pa, &intrhandle)) {
471 aprint_error_dev(self, "couldn't map interrupt\n");
472 return;
473 }
474 intrstr = pci_intr_string(pa->pa_pc, intrhandle, intrbuf, sizeof(intrbuf));
475 sc->sc_ih = pci_intr_establish_xname(pa->pa_pc, intrhandle, IPL_NET,
476 pdq_pci_ifintr, sc, device_xname(self));
477 if (sc->sc_ih == NULL) {
478 aprint_error_dev(self, "couldn't establish interrupt");
479 if (intrstr != NULL)
480 aprint_error(" at %s", intrstr);
481 aprint_error("\n");
482 return;
483 }
484
485 sc->sc_ats = shutdownhook_establish((void (*)(void *)) pdq_hwreset, sc->sc_pdq);
486 if (sc->sc_ats == NULL)
487 aprint_error_dev(self, "warning: couldn't establish shutdown hook\n");
488 if (intrstr != NULL)
489 aprint_normal_dev(self, "interrupting at %s\n", intrstr);
490}
