41struct ipw_soft_bd {
42 struct ipw_bd *bd;
43 int type;
44#define IPW_SBD_TYPE_NOASSOC 0
45#define IPW_SBD_TYPE_COMMAND 1
46#define IPW_SBD_TYPE_HEADER 2
47#define IPW_SBD_TYPE_DATA 3

49};

58struct ipw_soft_buf {

63};

65struct ipw_rx_radiotap_header {
66 struct ieee80211_radiotap_header wr_ihdr;
67 uint16_t wr_chan_freq;
68 uint16_t wr_chan_flags;
69 uint8_t wr_antsignal;
70} __packed;
71
72#define IPW_RX_RADIOTAP_PRESENT \
73 ((1 << IEEE80211_RADIOTAP_CHANNEL) | \
74 (1 << IEEE80211_RADIOTAP_DB_ANTSIGNAL))
75
76struct ipw_tx_radiotap_header {
77 struct ieee80211_radiotap_header wt_ihdr;
78 uint16_t wt_chan_freq;
79 uint16_t wt_chan_flags;
80} __packed;
81
82#define IPW_TX_RADIOTAP_PRESENT \
83 ((1 << IEEE80211_RADIOTAP_CHANNEL))
