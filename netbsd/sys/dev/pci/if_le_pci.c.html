111static int le_pci_supmedia[] = {
112 IFM_ETHER | IFM_AUTO,
113 IFM_ETHER | IFM_AUTO | IFM_FDX,
114 IFM_ETHER | IFM_10_T,
115 IFM_ETHER | IFM_10_T | IFM_FDX,
116 IFM_ETHER | IFM_10_5,
117 IFM_ETHER | IFM_10_5 | IFM_FDX,

120static void
121le_pci_wrcsr(struct lance_softc *sc, uint16_t port, uint16_t val)
122{
123 struct le_softc *lesc = (struct le_softc *)sc;
124 bus_space_tag_t iot = lesc->sc_iot;
125 bus_space_handle_t ioh = lesc->sc_ioh;
126
127 bus_space_write_2(iot, ioh, lesc->sc_rap, port);
128 bus_space_write_2(iot, ioh, lesc->sc_rdp, val);
129}
130
131static uint16_t
132le_pci_rdcsr(struct lance_softc *sc, uint16_t port)
133{
134 struct le_softc *lesc = (struct le_softc *)sc;
135 bus_space_tag_t iot = lesc->sc_iot;
136 bus_space_handle_t ioh = lesc->sc_ioh;
137 uint16_t val;
138
139 bus_space_write_2(iot, ioh, lesc->sc_rap, port);
140 val = bus_space_read_2(iot, ioh, lesc->sc_rdp);
141 return val;
142}

144static int
145le_pci_mediachange(struct lance_softc *sc)
146{
147 struct le_softc *lesc = (struct le_softc *)sc;
148 bus_space_tag_t iot = lesc->sc_iot;
149 bus_space_handle_t ioh = lesc->sc_ioh;
150 int newmedia = sc->sc_media.ifm_media;
151 uint16_t reg;
152
153 if (IFM_SUBTYPE(newmedia) !=
154 IFM_SUBTYPE(lesc->sc_currentmedia)) {
155 if (IFM_SUBTYPE(newmedia) == IFM_AUTO) {
156 /* Switch to autoselect - BCR2 bit 1 */
157 bus_space_write_2(iot, ioh, PCNET_PCI_RAP, 2);
158 reg = bus_space_read_2(iot, ioh, PCNET_PCI_BDP);
159 reg |= 2;
160 bus_space_write_2(iot, ioh, PCNET_PCI_RAP, 2);
161 bus_space_write_2(iot, ioh, PCNET_PCI_BDP, reg);
162 } else {
163 /* Force media type (in init block) */
164 lance_reset(sc);
165 if (IFM_SUBTYPE(newmedia) == IFM_10_T)
166 sc->sc_initmodemedia = 1; /* UTP */
167 else
168 sc->sc_initmodemedia = 0; /* AUI */
169 lance_init(&sc->sc_ethercom.ec_if);
170
171 if (IFM_SUBTYPE(lesc->sc_currentmedia) == IFM_AUTO) {
172 /* Take away autoselect - BCR2 bit 1 */
173 bus_space_write_2(iot, ioh, PCNET_PCI_RAP, 2);
174 reg = bus_space_read_2(iot, ioh, PCNET_PCI_BDP);
175 reg &= ~2;
176 bus_space_write_2(iot, ioh, PCNET_PCI_RAP, 2);
177 bus_space_write_2(iot, ioh, PCNET_PCI_BDP, reg);
178 }
179 }
180
181 }
182
183 if ((IFM_OPTIONS(newmedia) ^ IFM_OPTIONS(lesc->sc_currentmedia))
184 & IFM_FDX) {
185 /* Toggle full duplex - BCR9 */
186 bus_space_write_2(iot, ioh, PCNET_PCI_RAP, 9);
187 reg = bus_space_read_2(iot, ioh, PCNET_PCI_BDP);
188 if (IFM_OPTIONS(newmedia) & IFM_FDX) {
189 reg |= 1; /* FDEN */
190 /*
191 * Allow FDX on AUI only if explicitly chosen, not in
192 * autoselect mode
193 */
194 if (IFM_SUBTYPE(newmedia) == IFM_10_5)
195 reg |= 2; /* AUIFD */
196 else
197 reg &= ~2;
198 } else
199 reg &= ~1;
200 bus_space_write_2(iot, ioh, PCNET_PCI_RAP, 9);
201 bus_space_write_2(iot, ioh, PCNET_PCI_BDP, reg);
202 }
203
204 lesc->sc_currentmedia = newmedia;
205 return 0;
206}

208static int
209le_pci_match(device_t parent, cfdata_t cf, void *aux)
210{
211 struct pci_attach_args *pa = aux;
212
213 if (PCI_VENDOR(pa->pa_id) != PCI_VENDOR_AMD)
214 return 0;
215
216 switch (PCI_PRODUCT(pa->pa_id)) {
217 case PCI_PRODUCT_AMD_PCNET_PCI:
218 return 1;
219 }
220
221 return 0;
222}
223
224static void
225le_pci_attach(device_t parent, device_t self, void *aux)
226{
227 struct le_softc *lesc = device_private(self);
228 struct lance_softc *sc = &lesc->sc_am79900.lsc;
229 struct pci_attach_args *pa = aux;
230 pci_intr_handle_t ih;
231 bus_space_tag_t iot;
232 bus_space_handle_t ioh;
233 bus_dma_tag_t dmat = pa->pa_dmat;
234 bus_dma_segment_t seg;
235 pci_chipset_tag_t pc = pa->pa_pc;
236 pcireg_t csr;
237 int i, rseg;
238 const char *model, *intrstr;
239 char intrbuf[PCI_INTRSTR_LEN];
240
241 sc->sc_dev = self;
242
243 switch (PCI_PRODUCT(pa->pa_id)) {
244 case PCI_PRODUCT_AMD_PCNET_PCI:
245 model = "PCnet-PCI Ethernet";
246 lesc->sc_rap = PCNET_PCI_RAP;
247 lesc->sc_rdp = PCNET_PCI_RDP;
248 break;
249
250 default:
251 model = "unknown model!";
252 }
253
254 aprint_normal(": %s\n", model);
255
256 if (pci_mapreg_map(pa, PCI_CBIO, PCI_MAPREG_TYPE_IO, 0,
257 &iot, &ioh, NULL, NULL)) {
258 aprint_error_dev(self, "can't map I/O space\n");
259 return;
260 }
261
262 /* Extract the physical MAC address from the ROM. */
263 for (i = 0; i < sizeof(sc->sc_enaddr); i++)
264 sc->sc_enaddr[i] = bus_space_read_1(iot, ioh, i);
265
266 lesc->sc_iot = iot;
267 lesc->sc_ioh = ioh;
268 lesc->sc_dmat = dmat;
269
270 /* Allocate a DMA area for the card. */
271 if (bus_dmamem_alloc(dmat, LE_PCI_MEMSIZE, PAGE_SIZE, 0, &seg, 1,
272 &rseg, BUS_DMA_NOWAIT)) {
273 aprint_error_dev(self, "couldn't allocate memory for card\n");
274 return;
275 }
276 if (bus_dmamem_map(dmat, &seg, rseg, LE_PCI_MEMSIZE,
277 (void **)&sc->sc_mem,
278 BUS_DMA_NOWAIT|BUS_DMA_COHERENT)) {
279 aprint_error_dev(self, "couldn't map memory for card\n");
280 return;
281 }
282
283 /* Create and load the DMA map for the DMA area. */
284 if (bus_dmamap_create(dmat, LE_PCI_MEMSIZE, 1,
285 LE_PCI_MEMSIZE, 0, BUS_DMA_NOWAIT, &lesc->sc_dmam)) {
286 aprint_error_dev(self, "couldn't create DMA map\n");
287 bus_dmamem_free(dmat, &seg, rseg);
288 return;
289 }
290 if (bus_dmamap_load(dmat, lesc->sc_dmam,
291 sc->sc_mem, LE_PCI_MEMSIZE, NULL, BUS_DMA_NOWAIT)) {
292 aprint_error_dev(self, "coundn't load DMA map\n");
293 bus_dmamem_free(dmat, &seg, rseg);
294 return;
295 }
296
297 sc->sc_conf3 = 0;
298 sc->sc_addr = lesc->sc_dmam->dm_segs[0].ds_addr;
299 sc->sc_memsize = LE_PCI_MEMSIZE;
300
301 sc->sc_copytodesc = lance_copytobuf_contig;
302 sc->sc_copyfromdesc = lance_copyfrombuf_contig;
303 sc->sc_copytobuf = lance_copytobuf_contig;
304 sc->sc_copyfrombuf = lance_copyfrombuf_contig;
305 sc->sc_zerobuf = lance_zerobuf_contig;
306
307 sc->sc_rdcsr = le_pci_rdcsr;
308 sc->sc_wrcsr = le_pci_wrcsr;
309 sc->sc_hwinit = NULL;
310
311 sc->sc_supmedia = le_pci_supmedia;
312 sc->sc_nsupmedia = __arraycount(le_pci_supmedia);
313 sc->sc_defaultmedia = le_pci_supmedia[0];
314 sc->sc_mediachange = le_pci_mediachange;
315 lesc->sc_currentmedia = le_pci_supmedia[0];
316
317 aprint_normal("%s", device_xname(self));
318 am79900_config(&lesc->sc_am79900);
319
320 /* Chip is stopped. Set "software style" to 32-bit. */
321 bus_space_write_2(iot, ioh, PCNET_PCI_RAP, 20);
322 bus_space_write_2(iot, ioh, PCNET_PCI_BDP, 2);
323
324 /* Enable the card. */
325 csr = pci_conf_read(pc, pa->pa_tag,
326 PCI_COMMAND_STATUS_REG);
327 pci_conf_write(pc, pa->pa_tag, PCI_COMMAND_STATUS_REG,
328 csr | PCI_COMMAND_MASTER_ENABLE);
329
330 /* Map and establish the interrupt. */
331 if (pci_intr_map(pa, &ih)) {
332 aprint_error_dev(self, "couldn't map interrupt\n");
333 return;
334 }
335 intrstr = pci_intr_string(pc, ih, intrbuf, sizeof(intrbuf));
336 lesc->sc_ih = pci_intr_establish_xname(pc, ih, IPL_NET, am79900_intr,
337 sc, device_xname(self));
338 if (lesc->sc_ih == NULL) {
339 aprint_error_dev(self, "couldn't establish interrupt");
340 if (intrstr != NULL)
341 aprint_error(" at %s", intrstr);
342 aprint_error("\n");
343 return;
344 }
345 aprint_normal_dev(self, "interrupting at %s\n", intrstr);
346}
