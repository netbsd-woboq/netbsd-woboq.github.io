32#define PCNET_PCI_RDP 0x10
33#define PCNET_PCI_RAP 0x12
34#define PCNET_PCI_BDP 0x16
35
36/*
37 * Ethernet software status per interface.
38 *
39 * Each interface is referenced by a network interface structure,
40 * ethercom.ec_if, which the routing code uses to locate the interface.
41 * This structure contains the output queue for the interface, its address, ...
42 */
43struct le_softc {
44 struct am79900_softc sc_am79900; /* glue to MI code */
45
46 void *sc_ih;
47 bus_space_tag_t sc_iot; /* space cookie */
48 bus_space_handle_t sc_ioh; /* bus space handle */
49 bus_dma_tag_t sc_dmat; /* bus dma tag */
50 bus_dmamap_t sc_dmam; /* bus dma map */
51 int sc_rap, sc_rdp; /* offsets to LANCE registers */
52 int sc_currentmedia; /* currently used media type */
53};
