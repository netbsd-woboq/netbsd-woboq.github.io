97/*
98 * Number of DMA segments in a TxCB. Note that this is carefully
99 * chosen to make the total struct size an even power of two. It's
100 * critical that no TxCB be split across a page boundary since
101 * no attempt is made to allocate physically contiguous memory.
102 *
103 */
104#define SK_NTXSEG 30

132struct msk_ring_data {
133 struct msk_tx_desc sk_tx_ring[MSK_TX_RING_CNT];
134 struct msk_rx_desc sk_rx_ring[MSK_RX_RING_CNT];
135};
136
137#define MSK_TX_RING_ADDR(sc, i) \
138 ((sc)->sk_ring_map->dm_segs[0].ds_addr + \
139 offsetof(struct msk_ring_data, sk_tx_ring[(i)]))
140
141#define MSK_RX_RING_ADDR(sc, i) \
142 ((sc)->sk_ring_map->dm_segs[0].ds_addr + \
143 offsetof(struct msk_ring_data, sk_rx_ring[(i)]))
144
145#define MSK_CDOFF(x) offsetof(struct msk_ring_data, x)
146#define MSK_CDTXOFF(x) MSK_CDOFF(sk_tx_ring[(x)])
147#define MSK_CDRXOFF(x) MSK_CDOFF(sk_rx_ring[(x)])
148#define MSK_CDSTOFF(x) ((x) * sizeof(struct msk_status_desc))
149
150#define MSK_CDTXSYNC(sc, x, n, ops) \
151do { \
152 int __x, __n; \
153 \
154 __x = (x); \
155 __n = (n); \
156 \
157 /* If it will wrap around, sync to the end of the ring. */ \
158 if ((__x + __n) > MSK_TX_RING_CNT) { \
159 bus_dmamap_sync((sc)->sk_softc->sc_dmatag, \
160 (sc)->sk_ring_map, MSK_CDTXOFF(__x), \
161 sizeof(struct msk_tx_desc) * (MSK_TX_RING_CNT - __x), \
162 (ops)); \
163 __n -= (MSK_TX_RING_CNT - __x); \
164 __x = 0; \
165 } \
166 \
167 /* Now sync whatever is left. */ \
168 bus_dmamap_sync((sc)->sk_softc->sc_dmatag, (sc)->sk_ring_map, \
169 MSK_CDTXOFF((__x)), sizeof(struct msk_tx_desc) * __n, (ops)); \
170} while (/*CONSTCOND*/0)
171
172#define MSK_CDRXSYNC(sc, x, ops) \
173do { \
174 bus_dmamap_sync((sc)->sk_softc->sc_dmatag, (sc)->sk_ring_map, \
175 MSK_CDRXOFF((x)), sizeof(struct msk_rx_desc), (ops)); \
176} while (/*CONSTCOND*/0)
177
178#define MSK_CDSTSYNC(sc, x, ops) \
179do { \
180 bus_dmamap_sync((sc)->sc_dmatag, (sc)->sk_status_map, \
181 MSK_CDSTOFF((x)), sizeof(struct msk_status_desc), (ops)); \
182} while (/*CONSTCOND*/0)
183
184#define SK_INC(x, y) (x) = (x + 1) % y

189/* Softc for the Yukon-II controller. */
190struct sk_softc {
191 device_t sk_dev;
192 bus_space_handle_t sk_bhandle; /* bus space handle */
193 bus_space_tag_t sk_btag; /* bus space tag */
194 bus_size_t sk_bsize; /* bus space size */
195 void *sk_intrhand; /* irq handler handle */
196 pci_intr_handle_t *sk_pihp;
197 pci_chipset_tag_t sk_pc;
198 u_int8_t sk_fibertype;
199 u_int8_t sk_type;
200 u_int8_t sk_rev;
201 u_int8_t sk_macs; /* # of MACs */
202 const char *sk_name;
203 u_int32_t sk_ramsize; /* amount of RAM on NIC */
204 u_int32_t sk_intrmask;
205 struct sysctllog *sk_clog;
206 int sk_int_mod;
207 int sk_int_mod_pending;
208 bus_dma_tag_t sc_dmatag;
209 struct sk_if_softc *sk_if[2];
210 struct msk_status_desc *sk_status_ring;
211 bus_dmamap_t sk_status_map;
212 bus_dma_segment_t sk_status_seg;
213 int sk_status_nseg;
214 int sk_status_idx;
215 int sk_status_own_idx;
216 u_int8_t rnd_attached;
217 krndsource_t rnd_source;
218};

233 struct callout sk_tick_ch;
234 struct callout sk_tick_rx;
235 struct msk_chain_data sk_cdata;
236 struct msk_ring_data *sk_rdata;
237 bus_dmamap_t sk_ring_map;
238 bus_dma_segment_t sk_ring_seg;
239 int sk_ring_nseg;
240 int sk_status_idx;
241 struct sk_softc *sk_softc; /* parent controller */
242 int sk_if_flags;

244 LIST_HEAD(__sk_jfreehead, sk_jpool_entry) sk_jfree_listhead;
