57#define PCI_IO_MAP_REG PCI_BAR(0)
58#define PCI_MEM_MAP_REG PCI_BAR(1)
59
60struct mtd_pci_device_id {
61 pci_vendor_id_t vendor; /* PCI vendor ID */
62 pci_product_id_t product; /* PCI product ID */
63};
64
65static struct mtd_pci_device_id mtd_ids[] = {
66 { PCI_VENDOR_MYSON, PCI_PRODUCT_MYSON_MTD803 },
67 { 0, 0 }
68};
69
70static int mtd_pci_match(device_t, cfdata_t, void *);
71static void mtd_pci_attach(device_t, device_t, void *);
72
73CFATTACH_DECL_NEW(mtd_pci, sizeof(struct mtd_softc), mtd_pci_match,
74 mtd_pci_attach, NULL, NULL);
75
76static int
77mtd_pci_match(device_t parent, cfdata_t match, void *aux)
78{
79 struct pci_attach_args *pa = aux;
80 struct mtd_pci_device_id *id;
81
82 for (id = mtd_ids; id->vendor != 0; ++id) {
83 if (PCI_VENDOR(pa->pa_id) == id->vendor &&
84 PCI_PRODUCT(pa->pa_id) == id->product)
85 return (1);
86 }
87 return (0);
88}
89
90static void
91mtd_pci_attach(device_t parent, device_t self, void *aux)
92{
93 struct pci_attach_args * const pa = aux;
94 struct mtd_softc * const sc = device_private(self);
95 pci_intr_handle_t ih;
96 const char *intrstring = NULL;
97 bus_space_tag_t iot, memt;
98 bus_space_handle_t ioh, memh;
99 int io_valid, mem_valid;
100 char intrbuf[PCI_INTRSTR_LEN];
101
102 sc->dev = self;
103 pci_aprint_devinfo(pa, NULL);
104
105 io_valid = (pci_mapreg_map(pa, PCI_IO_MAP_REG, PCI_MAPREG_TYPE_IO,
106 0, &iot, &ioh, NULL, NULL) == 0);
107 mem_valid = (pci_mapreg_map(pa, PCI_MEM_MAP_REG, PCI_MAPREG_TYPE_MEM
108 | PCI_MAPREG_MEM_TYPE_32BIT, 0, &memt, &memh,
109 NULL, NULL) == 0);
110
111 if (mem_valid) {
112 sc->bus_tag = memt;
113 sc->bus_handle = memh;
114 } else if (io_valid) {
115 sc->bus_tag = iot;
116 sc->bus_handle = ioh;
117 } else {
118 aprint_error_dev(sc->dev,
119 "could not map memory or i/o space\n");
120 return;
121 }
122 sc->dma_tag = pa->pa_dmat;
123
124 /* Do generic attach. Seems this must be done before setting IRQ */
125 mtd_config(sc);
126
127 if (pci_intr_map(pa, &ih)) {
128 aprint_error_dev(sc->dev, "could not map interrupt\n");
129 return;
130 }
131 intrstring = pci_intr_string(pa->pa_pc, ih, intrbuf, sizeof(intrbuf));
132
133 if (pci_intr_establish_xname(pa->pa_pc, ih, IPL_NET, mtd_irq_h, sc,
134 device_xname(self)) == NULL) {
135 aprint_error_dev(sc->dev, "could not establish interrupt");
136 if (intrstring != NULL)
137 aprint_error(" at %s", intrstring);
138 aprint_error("\n");
139 return;
140 } else {
141 aprint_normal_dev(sc->dev, "using %s for interrupt\n",
142 intrstring ? intrstring : "unknown interrupt");
143 }
144}
