39 * Device driver for the AMD PCnet-PCI series of Ethernet
40 * chips:
41 *
42 * * Am79c970 PCnet-PCI Single-Chip Ethernet Controller for PCI
43 * Local Bus
44 *
45 * * Am79c970A PCnet-PCI II Single-Chip Full-Duplex Ethernet Controller
46 * for PCI Local Bus
47 *
48 * * Am79c971 PCnet-FAST Single-Chip Full-Duplex 10/100Mbps
49 * Ethernet Controller for PCI Local Bus
50 *
51 * * Am79c972 PCnet-FAST+ Enhanced 10/100Mbps PCI Ethernet Controller
52 * with OnNow Support
53 *
54 * * Am79c973/Am79c975 PCnet-FAST III Single-Chip 10/100Mbps PCI
55 * Ethernet Controller with Integrated PHY
56 *
57 * This also supports the virtual PCnet-PCI Ethernet interface found
58 * in VMware.
59 *
60 * TODO:

107/*
108 * Transmit descriptor list size. This is arbitrary, but allocate
109 * enough descriptors for 128 pending transmissions, and 4 segments
110 * per packet. This MUST work out to a power of 2.
111 *
112 * NOTE: We can't have any more than 512 Tx descriptors, SO BE CAREFUL!
113 *
114 * So we play a little trick here. We give each packet up to 16
115 * DMA segments, but only allocate the max of 512 descriptors. The
116 * transmit logic can deal with this, we just are hoping to sneak by.
117 */
118#define PCN_NTXSEGS 16
119#define PCN_NTXSEGS_VMWARE 8 /* bug in VMware's emulation */
120
121#define PCN_TXQUEUELEN 128
122#define PCN_TXQUEUELEN_MASK (PCN_TXQUEUELEN - 1)
123#define PCN_NTXDESC 512
124#define PCN_NTXDESC_MASK (PCN_NTXDESC - 1)
125#define PCN_NEXTTX(x) (((x) + 1) & PCN_NTXDESC_MASK)
126#define PCN_NEXTTXS(x) (((x) + 1) & PCN_TXQUEUELEN_MASK)
127
128/* Tx interrupt every N + 1 packets. */
129#define PCN_TXINTR_MASK 7
130
131/*
132 * Receive descriptor list size. We have one Rx buffer per incoming
133 * packet, so this logic is a little simpler.
134 */
135#define PCN_NRXDESC 128
136#define PCN_NRXDESC_MASK (PCN_NRXDESC - 1)
137#define PCN_NEXTRX(x) (((x) + 1) & PCN_NRXDESC_MASK)

140 * Control structures are DMA'd to the PCnet chip. We allocate them in

323#define PCN_CDTXADDR(sc, x) ((sc)->sc_cddma + PCN_CDTXOFF((x)))
324#define PCN_CDRXADDR(sc, x) ((sc)->sc_cddma + PCN_CDRXOFF((x)))

327#define PCN_CDTXSYNC(sc, x, n, ops) \
328do { \
329 int __x, __n; \
330 \
331 __x = (x); \
332 __n = (n); \
333 \
334 /* If it will wrap around, sync to the end of the ring. */ \
335 if ((__x + __n) > PCN_NTXDESC) { \
336 bus_dmamap_sync((sc)->sc_dmat, (sc)->sc_cddmamap, \
337 PCN_CDTXOFF(__x), sizeof(struct letmd) * \
338 (PCN_NTXDESC - __x), (ops)); \
339 __n -= (PCN_NTXDESC - __x); \
340 __x = 0; \
341 } \
342 \
343 /* Now sync whatever is left. */ \
344 bus_dmamap_sync((sc)->sc_dmat, (sc)->sc_cddmamap, \
345 PCN_CDTXOFF(__x), sizeof(struct letmd) * __n, (ops)); \
346} while (/*CONSTCOND*/0)
347
348#define PCN_CDRXSYNC(sc, x, ops) \
349 bus_dmamap_sync((sc)->sc_dmat, (sc)->sc_cddmamap, \
350 PCN_CDRXOFF((x)), sizeof(struct lermd), (ops))
351
352#define PCN_CDINITSYNC(sc, ops) \
353 bus_dmamap_sync((sc)->sc_dmat, (sc)->sc_cddmamap, \
354 PCN_CDINITOFF, sizeof(struct leinit), (ops))
355
356#define PCN_INIT_RXDESC(sc, x) \
357do { \
358 struct pcn_rxsoft *__rxs = &(sc)->sc_rxsoft[(x)]; \
359 struct lermd *__rmd = &(sc)->sc_rxdescs[(x)]; \
360 struct mbuf *__m = __rxs->rxs_mbuf; \
361 \
362 /* \
363 * Note: We scoot the packet forward 2 bytes in the buffer \
364 * so that the payload after the Ethernet header is aligned \
365 * to a 4-byte boundary. \
366 */ \
367 __m->m_data = __m->m_ext.ext_buf + 2; \
368 \
369 if ((sc)->sc_swstyle == LE_B20_SSTYLE_PCNETPCI3) { \
370 __rmd->rmd2 = \
371 htole32(__rxs->rxs_dmamap->dm_segs[0].ds_addr + 2); \
372 __rmd->rmd0 = 0; \
373 } else { \
374 __rmd->rmd2 = 0; \
375 __rmd->rmd0 = \
376 htole32(__rxs->rxs_dmamap->dm_segs[0].ds_addr + 2); \
377 } \
378 __rmd->rmd1 = htole32(LE_R1_OWN | LE_R1_ONES | \
379 (LE_BCNT(MCLBYTES - 2) & LE_R1_BCNT_MASK)); \
380 PCN_CDRXSYNC((sc), (x), BUS_DMASYNC_PREREAD | BUS_DMASYNC_PREWRITE);\
381} while(/*CONSTCOND*/0)

496static bool
497pcn_is_vmware(const char *enaddr)
498{
499
500 /*
501 * VMware uses the OUI 00:0c:29 for auto-generated MAC
502 * addresses.
503 */
504 if (enaddr[0] == 0x00 && enaddr[1] == 0x0c && enaddr[2] == 0x29)
505 return TRUE;
506
507 /*
508 * VMware uses the OUI 00:50:56 for manually-set MAC
509 * addresses (and some auto-generated ones).
510 */
511 if (enaddr[0] == 0x00 && enaddr[1] == 0x50 && enaddr[2] == 0x56)
512 return TRUE;
513
514 return FALSE;
515}

517static const struct pcn_variant *
518pcn_lookup_variant(uint16_t chipid)
519{
520 const struct pcn_variant *pcv;
521
522 for (pcv = pcn_variants; pcv->pcv_chipid != 0; pcv++) {
523 if (chipid == pcv->pcv_chipid)
524 return pcv;
525 }
526
527 /*
528 * This covers unknown chips, which we simply treat like
529 * a generic PCnet-FAST.
530 */
531 return pcv;
532}

534static int
535pcn_match(device_t parent, cfdata_t cf, void *aux)
536{
537 struct pci_attach_args *pa = aux;
538
539 /*
540 * IBM Makes a PCI variant of this card which shows up as a
541 * Trident Microsystems 4DWAVE DX (ethernet network, revision 0x25)
542 * this card is truly a pcn card, so we have a special case match for
543 * it
544 */
545
546 if (PCI_VENDOR(pa->pa_id) == PCI_VENDOR_TRIDENT &&
547 PCI_PRODUCT(pa->pa_id) == PCI_PRODUCT_TRIDENT_4DWAVE_DX &&
548 PCI_CLASS(pa->pa_class) == PCI_CLASS_NETWORK)
549 return 1;
550
551 if (PCI_VENDOR(pa->pa_id) != PCI_VENDOR_AMD)
552 return 0;
553
554 switch (PCI_PRODUCT(pa->pa_id)) {
555 case PCI_PRODUCT_AMD_PCNET_PCI:
556 /* Beat if_le_pci.c */
557 return 10;
558 }
559
560 return 0;
561}

563static void
564pcn_attach(device_t parent, device_t self, void *aux)
565{
566 struct pcn_softc *sc = device_private(self);
567 struct pci_attach_args *pa = aux;
568 struct ifnet *ifp = &sc->sc_ethercom.ec_if;
569 pci_chipset_tag_t pc = pa->pa_pc;
570 pci_intr_handle_t ih;
571 const char *intrstr = NULL;
572 bus_space_tag_t iot, memt;
573 bus_space_handle_t ioh, memh;
574 bus_dma_segment_t seg;
575 int ioh_valid, memh_valid;
576 int ntxsegs, i, rseg, error;
577 uint32_t chipid, reg;
578 uint8_t enaddr[ETHER_ADDR_LEN];
579 prop_object_t obj;
580 bool is_vmware;
581 char intrbuf[PCI_INTRSTR_LEN];
582
583 sc->sc_dev = self;
584 callout_init(&sc->sc_tick_ch, 0);
585
586 aprint_normal(": AMD PCnet-PCI Ethernet\n");
587
588 /*
589 * Map the device.
590 */
591 ioh_valid = (pci_mapreg_map(pa, PCN_PCI_CBIO, PCI_MAPREG_TYPE_IO, 0,
592 &iot, &ioh, NULL, NULL) == 0);
593 memh_valid = (pci_mapreg_map(pa, PCN_PCI_CBMEM,
594 PCI_MAPREG_TYPE_MEM | PCI_MAPREG_MEM_TYPE_32BIT, 0,
595 &memt, &memh, NULL, NULL) == 0);
596
597 if (memh_valid) {
598 sc->sc_st = memt;
599 sc->sc_sh = memh;
600 } else if (ioh_valid) {
601 sc->sc_st = iot;
602 sc->sc_sh = ioh;
603 } else {
604 aprint_error_dev(self, "unable to map device registers\n");
605 return;
606 }
607
608 sc->sc_dmat = pa->pa_dmat;
609
610 /* Make sure bus mastering is enabled. */
611 pci_conf_write(pc, pa->pa_tag, PCI_COMMAND_STATUS_REG,
612 pci_conf_read(pc, pa->pa_tag, PCI_COMMAND_STATUS_REG) |
613 PCI_COMMAND_MASTER_ENABLE);
614
615 /* power up chip */
616 if ((error = pci_activate(pa->pa_pc, pa->pa_tag, self,
617 NULL)) && error != EOPNOTSUPP) {
618 aprint_error_dev(self, "cannot activate %d\n", error);
619 return;
620 }
621
622 /*
623 * Reset the chip to a known state. This also puts the
624 * chip into 32-bit mode.
625 */
626 pcn_reset(sc);
627
628 /*
629 * On some systems with the chip is an on-board device, the
630 * EEPROM is not used. Handle this by reading the MAC address
631 * from the CSRs (assuming that boot firmware has written
632 * it there).
633 */
634 obj = prop_dictionary_get(device_properties(sc->sc_dev),
635 "am79c970-no-eeprom");
636 if (prop_bool_true(obj)) {
637 for (i = 0; i < 3; i++) {
638 uint32_t val;
639 val = pcn_csr_read(sc, LE_CSR12 + i);
640 enaddr[2 * i] = val & 0xff;
641 enaddr[2 * i + 1] = (val >> 8) & 0xff;
642 }
643 } else {
644 for (i = 0; i < ETHER_ADDR_LEN; i++) {
645 enaddr[i] = bus_space_read_1(sc->sc_st, sc->sc_sh,
646 PCN32_APROM + i);
647 }
648 }
649
650 /* Check to see if this is a VMware emulated network interface. */
651 is_vmware = pcn_is_vmware(enaddr);
652
653 /*
654 * Now that the device is mapped, attempt to figure out what
655 * kind of chip we have. Note that IDL has all 32 bits of
656 * the chip ID when we're in 32-bit mode.
657 */
658 chipid = pcn_csr_read(sc, LE_CSR88);
659 sc->sc_variant = pcn_lookup_variant(CHIPID_PARTID(chipid));
660
661 aprint_normal_dev(self, "%s rev %d, Ethernet address %s\n",
662 sc->sc_variant->pcv_desc, CHIPID_VER(chipid),
663 ether_sprintf(enaddr));
664
665 /*
666 * VMware has a bug in its network interface emulation; we must
667 * limit the number of Tx segments.
668 */
669 if (is_vmware) {
670 ntxsegs = PCN_NTXSEGS_VMWARE;
671 prop_dictionary_set_bool(device_properties(sc->sc_dev),
672 "am79c970-vmware-tx-bug", TRUE);
673 aprint_verbose_dev(self,
674 "VMware Tx segment count bug detected\n");
675 } else {
676 ntxsegs = PCN_NTXSEGS;
677 }
678
679 /*
680 * Map and establish our interrupt.
681 */
682 if (pci_intr_map(pa, &ih)) {
683 aprint_error_dev(self, "unable to map interrupt\n");
684 return;
685 }
686 intrstr = pci_intr_string(pc, ih, intrbuf, sizeof(intrbuf));
687 sc->sc_ih = pci_intr_establish_xname(pc, ih, IPL_NET, pcn_intr, sc,
688 device_xname(self));
689 if (sc->sc_ih == NULL) {
690 aprint_error_dev(self, "unable to establish interrupt");
691 if (intrstr != NULL)
692 aprint_error(" at %s", intrstr);
693 aprint_error("\n");
694 return;
695 }
696 aprint_normal_dev(self, "interrupting at %s\n", intrstr);
697
698 /*
699 * Allocate the control data structures, and create and load the
700 * DMA map for it.
701 */
702 if ((error = bus_dmamem_alloc(sc->sc_dmat,
703 sizeof(struct pcn_control_data), PAGE_SIZE, 0, &seg, 1, &rseg,
704 0)) != 0) {
705 aprint_error_dev(self, "unable to allocate control data, "
706 "error = %d\n", error);
707 goto fail_0;
708 }
709
710 if ((error = bus_dmamem_map(sc->sc_dmat, &seg, rseg,
711 sizeof(struct pcn_control_data), (void **)&sc->sc_control_data,
712 BUS_DMA_COHERENT)) != 0) {
713 aprint_error_dev(self, "unable to map control data, "
714 "error = %d\n", error);
715 goto fail_1;
716 }
717
718 if ((error = bus_dmamap_create(sc->sc_dmat,
719 sizeof(struct pcn_control_data), 1,
720 sizeof(struct pcn_control_data), 0, 0, &sc->sc_cddmamap)) != 0) {
721 aprint_error_dev(self, "unable to create control data DMA map, "
722 "error = %d\n", error);
723 goto fail_2;
724 }
725
726 if ((error = bus_dmamap_load(sc->sc_dmat, sc->sc_cddmamap,
727 sc->sc_control_data, sizeof(struct pcn_control_data), NULL,
728 0)) != 0) {
729 aprint_error_dev(self,
730 "unable to load control data DMA map, error = %d\n", error);
731 goto fail_3;
732 }
733
734 /* Create the transmit buffer DMA maps. */
735 for (i = 0; i < PCN_TXQUEUELEN; i++) {
736 if ((error = bus_dmamap_create(sc->sc_dmat, MCLBYTES,
737 ntxsegs, MCLBYTES, 0, 0,
738 &sc->sc_txsoft[i].txs_dmamap)) != 0) {
739 aprint_error_dev(self,
740 "unable to create tx DMA map %d, error = %d\n",
741 i, error);
742 goto fail_4;
743 }
744 }
745
746 /* Create the receive buffer DMA maps. */
747 for (i = 0; i < PCN_NRXDESC; i++) {
748 if ((error = bus_dmamap_create(sc->sc_dmat, MCLBYTES, 1,
749 MCLBYTES, 0, 0, &sc->sc_rxsoft[i].rxs_dmamap)) != 0) {
750 aprint_error_dev(self,
751 "unable to create rx DMA map %d, error = %d\n",
752 i, error);
753 goto fail_5;
754 }
755 sc->sc_rxsoft[i].rxs_mbuf = NULL;
756 }
757
758 /* Initialize our media structures. */
759 (*sc->sc_variant->pcv_mediainit)(sc);
760
761 /*
762 * Initialize FIFO watermark info.
763 */
764 switch (sc->sc_variant->pcv_chipid) {
765 case PARTID_Am79c970:
766 case PARTID_Am79c970A:
767 sc->sc_rcvfw_desc = pcn_79c970_rcvfw;
768 sc->sc_xmtsp_desc = pcn_79c970_xmtsp;
769 sc->sc_xmtfw_desc = pcn_79c970_xmtfw;
770 break;
771
772 default:
773 sc->sc_rcvfw_desc = pcn_79c971_rcvfw;
774 /*
775 * Read BCR25 to determine how much SRAM is
776 * on the board. If > 0, then we the chip
777 * uses different Start Point thresholds.
778 *
779 * Note BCR25 and BCR26 are loaded from the
780 * EEPROM on RST, and unaffected by S_RESET,
781 * so we don't really have to worry about
782 * them except for this.
783 */
784 reg = pcn_bcr_read(sc, LE_BCR25) & 0x00ff;
785 if (reg != 0)
786 sc->sc_xmtsp_desc = pcn_79c971_xmtsp_sram;
787 else
788 sc->sc_xmtsp_desc = pcn_79c971_xmtsp;
789 sc->sc_xmtfw_desc = pcn_79c971_xmtfw;
790 break;
791 }
792
793 /*
794 * Set up defaults -- see the tables above for what these
795 * values mean.
796 *
797 * XXX How should we tune RCVFW and XMTFW?
798 */
799 sc->sc_rcvfw = 1; /* minimum for full-duplex */
800 sc->sc_xmtsp = 1;
801 sc->sc_xmtfw = 0;
802
803 ifp = &sc->sc_ethercom.ec_if;
804 strcpy(ifp->if_xname, device_xname(self));
805 ifp->if_softc = sc;
806 ifp->if_flags = IFF_BROADCAST | IFF_SIMPLEX | IFF_MULTICAST;
807 ifp->if_ioctl = pcn_ioctl;
808 ifp->if_start = pcn_start;
809 ifp->if_watchdog = pcn_watchdog;
810 ifp->if_init = pcn_init;
811 ifp->if_stop = pcn_stop;
812 IFQ_SET_READY(&ifp->if_snd);
813
814 /* Attach the interface. */
815 if_attach(ifp);
816 if_deferred_start_init(ifp, NULL);
817 ether_ifattach(ifp, enaddr);
818 rnd_attach_source(&sc->rnd_source, device_xname(self),
819 RND_TYPE_NET, RND_FLAG_DEFAULT);
820
821#ifdef PCN_EVENT_COUNTERS
822 /* Attach event counters. */
823 evcnt_attach_dynamic(&sc->sc_ev_txsstall, EVCNT_TYPE_MISC,
824 NULL, device_xname(self), "txsstall");
825 evcnt_attach_dynamic(&sc->sc_ev_txdstall, EVCNT_TYPE_MISC,
826 NULL, device_xname(self), "txdstall");
827 evcnt_attach_dynamic(&sc->sc_ev_txintr, EVCNT_TYPE_INTR,
828 NULL, device_xname(self), "txintr");
829 evcnt_attach_dynamic(&sc->sc_ev_rxintr, EVCNT_TYPE_INTR,
830 NULL, device_xname(self), "rxintr");
831 evcnt_attach_dynamic(&sc->sc_ev_babl, EVCNT_TYPE_MISC,
832 NULL, device_xname(self), "babl");
833 evcnt_attach_dynamic(&sc->sc_ev_miss, EVCNT_TYPE_MISC,
834 NULL, device_xname(self), "miss");
835 evcnt_attach_dynamic(&sc->sc_ev_merr, EVCNT_TYPE_MISC,
836 NULL, device_xname(self), "merr");
837
838 evcnt_attach_dynamic(&sc->sc_ev_txseg1, EVCNT_TYPE_MISC,
839 NULL, device_xname(self), "txseg1");
840 evcnt_attach_dynamic(&sc->sc_ev_txseg2, EVCNT_TYPE_MISC,
841 NULL, device_xname(self), "txseg2");
842 evcnt_attach_dynamic(&sc->sc_ev_txseg3, EVCNT_TYPE_MISC,
843 NULL, device_xname(self), "txseg3");
844 evcnt_attach_dynamic(&sc->sc_ev_txseg4, EVCNT_TYPE_MISC,
845 NULL, device_xname(self), "txseg4");
846 evcnt_attach_dynamic(&sc->sc_ev_txseg5, EVCNT_TYPE_MISC,
847 NULL, device_xname(self), "txseg5");
848 evcnt_attach_dynamic(&sc->sc_ev_txsegmore, EVCNT_TYPE_MISC,
849 NULL, device_xname(self), "txsegmore");
850 evcnt_attach_dynamic(&sc->sc_ev_txcopy, EVCNT_TYPE_MISC,
851 NULL, device_xname(self), "txcopy");
852#endif /* PCN_EVENT_COUNTERS */
853
854 /*
855 * Establish power handler with shutdown hook, to make sure
856 * the interface is shutdown during reboot.
857 */
858 if (pmf_device_register1(self, NULL, NULL, pcn_shutdown))
859 pmf_class_network_register(self, ifp);
860 else
861 aprint_error_dev(self, "couldn't establish power handler\n");
862
863 return;
864
865 /*
866 * Free any resources we've allocated during the failed attach
867 * attempt. Do this in reverse order and fall through.
868 */
869 fail_5:
870 for (i = 0; i < PCN_NRXDESC; i++) {
871 if (sc->sc_rxsoft[i].rxs_dmamap != NULL)
872 bus_dmamap_destroy(sc->sc_dmat,
873 sc->sc_rxsoft[i].rxs_dmamap);
874 }
875 fail_4:
876 for (i = 0; i < PCN_TXQUEUELEN; i++) {
877 if (sc->sc_txsoft[i].txs_dmamap != NULL)
878 bus_dmamap_destroy(sc->sc_dmat,
879 sc->sc_txsoft[i].txs_dmamap);
880 }
881 bus_dmamap_unload(sc->sc_dmat, sc->sc_cddmamap);
882 fail_3:
883 bus_dmamap_destroy(sc->sc_dmat, sc->sc_cddmamap);
884 fail_2:
885 bus_dmamem_unmap(sc->sc_dmat, (void *)sc->sc_control_data,
886 sizeof(struct pcn_control_data));
887 fail_1:
888 bus_dmamem_free(sc->sc_dmat, &seg, rseg);
889 fail_0:
890 return;
891}

910/*
911 * pcn_start: [ifnet interface function]
912 *
913 * Start packet transmission on the interface.
914 */
915static void
916pcn_start(struct ifnet *ifp)
917{
918 struct pcn_softc *sc = ifp->if_softc;
919 struct mbuf *m0, *m;
920 struct pcn_txsoft *txs;
921 bus_dmamap_t dmamap;
922 int error, nexttx, lasttx = -1, ofree, seg;
923
924 if ((ifp->if_flags & (IFF_RUNNING | IFF_OACTIVE)) != IFF_RUNNING)
925 return;
926
927 /*
928 * Remember the previous number of free descriptors and
929 * the first descriptor we'll use.
930 */
931 ofree = sc->sc_txfree;
932
933 /*
934 * Loop through the send queue, setting up transmit descriptors
935 * until we drain the queue, or use up all available transmit
936 * descriptors.
937 */
938 for (;;) {
939 /* Grab a packet off the queue. */
940 IFQ_POLL(&ifp->if_snd, m0);
941 if (m0 == NULL)
942 break;
943 m = NULL;
944
945 /* Get a work queue entry. */
946 if (sc->sc_txsfree == 0) {
947 PCN_EVCNT_INCR(&sc->sc_ev_txsstall);
948 break;
949 }
950
951 txs = &sc->sc_txsoft[sc->sc_txsnext];
952 dmamap = txs->txs_dmamap;
953
954 /*
955 * Load the DMA map. If this fails, the packet either
956 * didn't fit in the alloted number of segments, or we
957 * were short on resources. In this case, we'll copy
958 * and try again.
959 */
960 if (bus_dmamap_load_mbuf(sc->sc_dmat, dmamap, m0,
961 BUS_DMA_WRITE | BUS_DMA_NOWAIT) != 0) {
962 PCN_EVCNT_INCR(&sc->sc_ev_txcopy);
963 MGETHDR(m, M_DONTWAIT, MT_DATA);
964 if (m == NULL) {
965 printf("%s: unable to allocate Tx mbuf\n",
966 device_xname(sc->sc_dev));
967 break;
968 }
969 if (m0->m_pkthdr.len > MHLEN) {
970 MCLGET(m, M_DONTWAIT);
971 if ((m->m_flags & M_EXT) == 0) {
972 printf("%s: unable to allocate Tx "
973 "cluster\n",
974 device_xname(sc->sc_dev));
975 m_freem(m);
976 break;
977 }
978 }
979 m_copydata(m0, 0, m0->m_pkthdr.len, mtod(m, void *));
980 m->m_pkthdr.len = m->m_len = m0->m_pkthdr.len;
981 error = bus_dmamap_load_mbuf(sc->sc_dmat, dmamap,
982 m, BUS_DMA_WRITE | BUS_DMA_NOWAIT);
983 if (error) {
984 printf("%s: unable to load Tx buffer, "
985 "error = %d\n", device_xname(sc->sc_dev),
986 error);
987 m_freem(m);
988 break;
989 }
990 }
991
992 /*
993 * Ensure we have enough descriptors free to describe
994 * the packet. Note, we always reserve one descriptor
995 * at the end of the ring as a termination point, to
996 * prevent wrap-around.
997 */
998 if (dmamap->dm_nsegs > (sc->sc_txfree - 1)) {
999 /*
1000 * Not enough free descriptors to transmit this
1001 * packet. We haven't committed anything yet,
1002 * so just unload the DMA map, put the packet
1003 * back on the queue, and punt. Notify the upper
1004 * layer that there are not more slots left.
1005 *
1006 * XXX We could allocate an mbuf and copy, but
1007 * XXX is it worth it?
1008 */
1009 ifp->if_flags |= IFF_OACTIVE;
1010 bus_dmamap_unload(sc->sc_dmat, dmamap);
1011 if (m != NULL)
1012 m_freem(m);
1013 PCN_EVCNT_INCR(&sc->sc_ev_txdstall);
1014 break;
1015 }
1016
1017 IFQ_DEQUEUE(&ifp->if_snd, m0);
1018 if (m != NULL) {
1019 m_freem(m0);
1020 m0 = m;
1021 }
1022
1023 /*
1024 * WE ARE NOW COMMITTED TO TRANSMITTING THE PACKET.
1025 */
1026
1027 /* Sync the DMA map. */
1028 bus_dmamap_sync(sc->sc_dmat, dmamap, 0, dmamap->dm_mapsize,
1029 BUS_DMASYNC_PREWRITE);
1030
1031#ifdef PCN_EVENT_COUNTERS
1032 switch (dmamap->dm_nsegs) {
1033 case 1:
1034 PCN_EVCNT_INCR(&sc->sc_ev_txseg1);
1035 break;
1036 case 2:
1037 PCN_EVCNT_INCR(&sc->sc_ev_txseg2);
1038 break;
1039 case 3:
1040 PCN_EVCNT_INCR(&sc->sc_ev_txseg3);
1041 break;
1042 case 4:
1043 PCN_EVCNT_INCR(&sc->sc_ev_txseg4);
1044 break;
1045 case 5:
1046 PCN_EVCNT_INCR(&sc->sc_ev_txseg5);
1047 break;
1048 default:
1049 PCN_EVCNT_INCR(&sc->sc_ev_txsegmore);
1050 break;
1051 }
1052#endif /* PCN_EVENT_COUNTERS */
1053
1054 /*
1055 * Initialize the transmit descriptors.
1056 */
1057 if (sc->sc_swstyle == LE_B20_SSTYLE_PCNETPCI3) {
1058 for (nexttx = sc->sc_txnext, seg = 0;
1059 seg < dmamap->dm_nsegs;
1060 seg++, nexttx = PCN_NEXTTX(nexttx)) {
1061 /*
1062 * If this is the first descriptor we're
1063 * enqueueing, don't set the OWN bit just
1064 * yet. That could cause a race condition.
1065 * We'll do it below.
1066 */
1067 sc->sc_txdescs[nexttx].tmd0 = 0;
1068 sc->sc_txdescs[nexttx].tmd2 =
1069 htole32(dmamap->dm_segs[seg].ds_addr);
1070 sc->sc_txdescs[nexttx].tmd1 =
1071 htole32(LE_T1_ONES |
1072 (nexttx == sc->sc_txnext ? 0 : LE_T1_OWN) |
1073 (LE_BCNT(dmamap->dm_segs[seg].ds_len) &
1074 LE_T1_BCNT_MASK));
1075 lasttx = nexttx;
1076 }
1077 } else {
1078 for (nexttx = sc->sc_txnext, seg = 0;
1079 seg < dmamap->dm_nsegs;
1080 seg++, nexttx = PCN_NEXTTX(nexttx)) {
1081 /*
1082 * If this is the first descriptor we're
1083 * enqueueing, don't set the OWN bit just
1084 * yet. That could cause a race condition.
1085 * We'll do it below.
1086 */
1087 sc->sc_txdescs[nexttx].tmd0 =
1088 htole32(dmamap->dm_segs[seg].ds_addr);
1089 sc->sc_txdescs[nexttx].tmd2 = 0;
1090 sc->sc_txdescs[nexttx].tmd1 =
1091 htole32(LE_T1_ONES |
1092 (nexttx == sc->sc_txnext ? 0 : LE_T1_OWN) |
1093 (LE_BCNT(dmamap->dm_segs[seg].ds_len) &
1094 LE_T1_BCNT_MASK));
1095 lasttx = nexttx;
1096 }
1097 }
1098
1099 KASSERT(lasttx != -1);
1100 /* Interrupt on the packet, if appropriate. */
1101 if ((sc->sc_txsnext & PCN_TXINTR_MASK) == 0)
1102 sc->sc_txdescs[lasttx].tmd1 |= htole32(LE_T1_LTINT);
1103
1104 /* Set `start of packet' and `end of packet' appropriately. */
1105 sc->sc_txdescs[lasttx].tmd1 |= htole32(LE_T1_ENP);
1106 sc->sc_txdescs[sc->sc_txnext].tmd1 |=
1107 htole32(LE_T1_OWN | LE_T1_STP);
1108
1109 /* Sync the descriptors we're using. */
1110 PCN_CDTXSYNC(sc, sc->sc_txnext, dmamap->dm_nsegs,
1111 BUS_DMASYNC_PREREAD | BUS_DMASYNC_PREWRITE);
1112
1113 /* Kick the transmitter. */
1114 pcn_csr_write(sc, LE_CSR0, LE_C0_INEA | LE_C0_TDMD);
1115
1116 /*
1117 * Store a pointer to the packet so we can free it later,
1118 * and remember what txdirty will be once the packet is
1119 * done.
1120 */
1121 txs->txs_mbuf = m0;
1122 txs->txs_firstdesc = sc->sc_txnext;
1123 txs->txs_lastdesc = lasttx;
1124
1125 /* Advance the tx pointer. */
1126 sc->sc_txfree -= dmamap->dm_nsegs;
1127 sc->sc_txnext = nexttx;
1128
1129 sc->sc_txsfree--;
1130 sc->sc_txsnext = PCN_NEXTTXS(sc->sc_txsnext);
1131
1132 /* Pass the packet to any BPF listeners. */
1133 bpf_mtap(ifp, m0, BPF_D_OUT);
1134 }
1135
1136 if (sc->sc_txsfree == 0 || sc->sc_txfree == 0) {
1137 /* No more slots left; notify upper layer. */
1138 ifp->if_flags |= IFF_OACTIVE;
1139 }
1140
1141 if (sc->sc_txfree != ofree) {
1142 /* Set a watchdog timer in case the chip flakes out. */
1143 ifp->if_timer = 5;
1144 }
1145}

1147/*
1148 * pcn_watchdog: [ifnet interface function]
1149 *
1150 * Watchdog timer handler.
1151 */
1152static void
1153pcn_watchdog(struct ifnet *ifp)
1154{
1155 struct pcn_softc *sc = ifp->if_softc;
1156
1157 /*
1158 * Since we're not interrupting every packet, sweep
1159 * up before we report an error.
1160 */
1161 pcn_txintr(sc);
1162
1163 if (sc->sc_txfree != PCN_NTXDESC) {
1164 printf("%s: device timeout (txfree %d txsfree %d)\n",
1165 device_xname(sc->sc_dev), sc->sc_txfree, sc->sc_txsfree);
1166 ifp->if_oerrors++;
1167
1168 /* Reset the interface. */
1169 (void) pcn_init(ifp);
1170 }
1171
1172 /* Try to get more packets going. */
1173 pcn_start(ifp);
1174}

1176/*
1177 * pcn_ioctl: [ifnet interface function]
1178 *
1179 * Handle control requests from the operator.
1180 */
1181static int
1182pcn_ioctl(struct ifnet *ifp, u_long cmd, void *data)
1183{
1184 int s, error;
1185
1186 s = splnet();
1187
1188 switch (cmd) {
1189 default:
1190 error = ether_ioctl(ifp, cmd, data);
1191 if (error == ENETRESET) {
1192 /*
1193 * Multicast list has changed; set the hardware filter
1194 * accordingly.
1195 */
1196 if (ifp->if_flags & IFF_RUNNING)
1197 error = pcn_init(ifp);
1198 else
1199 error = 0;
1200 }
1201 break;
1202 }
1203
1204 /* Try to get more packets going. */
1205 pcn_start(ifp);
1206
1207 splx(s);
1208 return error;
1209}

1211/*
1212 * pcn_intr:
1213 *
1214 * Interrupt service routine.
1215 */
1216static int
1217pcn_intr(void *arg)
1218{
1219 struct pcn_softc *sc = arg;
1220 struct ifnet *ifp = &sc->sc_ethercom.ec_if;
1221 uint32_t csr0;
1222 int wantinit, handled = 0;
1223
1224 for (wantinit = 0; wantinit == 0;) {
1225 csr0 = pcn_csr_read(sc, LE_CSR0);
1226 if ((csr0 & LE_C0_INTR) == 0)
1227 break;
1228
1229 rnd_add_uint32(&sc->rnd_source, csr0);
1230
1231 /* ACK the bits and re-enable interrupts. */
1232 pcn_csr_write(sc, LE_CSR0, csr0 &
1233 (LE_C0_INEA | LE_C0_BABL | LE_C0_MISS | LE_C0_MERR |
1234 LE_C0_RINT | LE_C0_TINT | LE_C0_IDON));
1235
1236 handled = 1;
1237
1238 if (csr0 & LE_C0_RINT) {
1239 PCN_EVCNT_INCR(&sc->sc_ev_rxintr);
1240 wantinit = pcn_rxintr(sc);
1241 }
1242
1243 if (csr0 & LE_C0_TINT) {
1244 PCN_EVCNT_INCR(&sc->sc_ev_txintr);
1245 pcn_txintr(sc);
1246 }
1247
1248 if (csr0 & LE_C0_ERR) {
1249 if (csr0 & LE_C0_BABL) {
1250 PCN_EVCNT_INCR(&sc->sc_ev_babl);
1251 ifp->if_oerrors++;
1252 }
1253 if (csr0 & LE_C0_MISS) {
1254 PCN_EVCNT_INCR(&sc->sc_ev_miss);
1255 ifp->if_ierrors++;
1256 }
1257 if (csr0 & LE_C0_MERR) {
1258 PCN_EVCNT_INCR(&sc->sc_ev_merr);
1259 printf("%s: memory error\n",
1260 device_xname(sc->sc_dev));
1261 wantinit = 1;
1262 break;
1263 }
1264 }
1265
1266 if ((csr0 & LE_C0_RXON) == 0) {
1267 printf("%s: receiver disabled\n",
1268 device_xname(sc->sc_dev));
1269 ifp->if_ierrors++;
1270 wantinit = 1;
1271 }
1272
1273 if ((csr0 & LE_C0_TXON) == 0) {
1274 printf("%s: transmitter disabled\n",
1275 device_xname(sc->sc_dev));
1276 ifp->if_oerrors++;
1277 wantinit = 1;
1278 }
1279 }
1280
1281 if (handled) {
1282 if (wantinit)
1283 pcn_init(ifp);
1284
1285 /* Try to get more packets going. */
1286 if_schedule_deferred_start(ifp);
1287 }
1288
1289 return handled;
1290}

1293 * pcn_spnd:
1294 *
1295 * Suspend the chip.
1296 */
1297static void
1298pcn_spnd(struct pcn_softc *sc)
1299{

1301
1302 pcn_csr_write(sc, LE_CSR5, sc->sc_csr5 | LE_C5_SPND);
1303

1311 device_xname(sc->sc_dev));

1314/*
1315 * pcn_txintr:
1316 *
1317 * Helper; handle transmit interrupts.
1318 */
1319static void
1320pcn_txintr(struct pcn_softc *sc)
1321{
1322 struct ifnet *ifp = &sc->sc_ethercom.ec_if;
1323 struct pcn_txsoft *txs;
1324 uint32_t tmd1, tmd2, tmd;
1325 int i, j;
1326
1327 ifp->if_flags &= ~IFF_OACTIVE;
1328
1329 /*
1330 * Go through our Tx list and free mbufs for those
1331 * frames which have been transmitted.
1332 */
1333 for (i = sc->sc_txsdirty; sc->sc_txsfree != PCN_TXQUEUELEN;
1334 i = PCN_NEXTTXS(i), sc->sc_txsfree++) {
1335 txs = &sc->sc_txsoft[i];
1336
1337 PCN_CDTXSYNC(sc, txs->txs_firstdesc, txs->txs_dmamap->dm_nsegs,
1338 BUS_DMASYNC_POSTREAD | BUS_DMASYNC_POSTWRITE);
1339
1340 tmd1 = le32toh(sc->sc_txdescs[txs->txs_lastdesc].tmd1);
1341 if (tmd1 & LE_T1_OWN)
1342 break;
1343
1344 /*
1345 * Slightly annoying -- we have to loop through the
1346 * descriptors we've used looking for ERR, since it
1347 * can appear on any descriptor in the chain.
1348 */
1349 for (j = txs->txs_firstdesc;; j = PCN_NEXTTX(j)) {
1350 tmd = le32toh(sc->sc_txdescs[j].tmd1);
1351 if (tmd & LE_T1_ERR) {
1352 ifp->if_oerrors++;
1353 if (sc->sc_swstyle == LE_B20_SSTYLE_PCNETPCI3)
1354 tmd2 = le32toh(sc->sc_txdescs[j].tmd0);
1355 else
1356 tmd2 = le32toh(sc->sc_txdescs[j].tmd2);
1357 if (tmd2 & LE_T2_UFLO) {
1358 if (sc->sc_xmtsp < LE_C80_XMTSP_MAX) {
1359 sc->sc_xmtsp++;
1360 printf("%s: transmit "
1361 "underrun; new threshold: "
1362 "%s\n",
1363 device_xname(sc->sc_dev),
1364 sc->sc_xmtsp_desc[
1365 sc->sc_xmtsp]);
1366 pcn_spnd(sc);
1367 pcn_csr_write(sc, LE_CSR80,
1368 LE_C80_RCVFW(sc->sc_rcvfw) |
1369 LE_C80_XMTSP(sc->sc_xmtsp) |
1370 LE_C80_XMTFW(sc->sc_xmtfw));
1371 pcn_csr_write(sc, LE_CSR5,
1372 sc->sc_csr5);
1373 } else {
1374 printf("%s: transmit "
1375 "underrun\n",
1376 device_xname(sc->sc_dev));
1377 }
1378 } else if (tmd2 & LE_T2_BUFF) {
1379 printf("%s: transmit buffer error\n",
1380 device_xname(sc->sc_dev));
1381 }
1382 if (tmd2 & LE_T2_LCOL)
1383 ifp->if_collisions++;
1384 if (tmd2 & LE_T2_RTRY)
1385 ifp->if_collisions += 16;
1386 goto next_packet;
1387 }
1388 if (j == txs->txs_lastdesc)
1389 break;
1390 }
1391 if (tmd1 & LE_T1_ONE)
1392 ifp->if_collisions++;
1393 else if (tmd & LE_T1_MORE) {
1394 /* Real number is unknown. */
1395 ifp->if_collisions += 2;
1396 }
1397 ifp->if_opackets++;
1398 next_packet:
1399 sc->sc_txfree += txs->txs_dmamap->dm_nsegs;
1400 bus_dmamap_sync(sc->sc_dmat, txs->txs_dmamap,
1401 0, txs->txs_dmamap->dm_mapsize, BUS_DMASYNC_POSTWRITE);
1402 bus_dmamap_unload(sc->sc_dmat, txs->txs_dmamap);
1403 m_freem(txs->txs_mbuf);
1404 txs->txs_mbuf = NULL;
1405 }
1406
1407 /* Update the dirty transmit buffer pointer. */
1408 sc->sc_txsdirty = i;
1409
1410 /*
1411 * If there are no more pending transmissions, cancel the watchdog
1412 * timer.
1413 */
1414 if (sc->sc_txsfree == PCN_TXQUEUELEN)
1415 ifp->if_timer = 0;
1416}

1418/*
1419 * pcn_rxintr:
1420 *
1421 * Helper; handle receive interrupts.
1422 */
1423static int
1424pcn_rxintr(struct pcn_softc *sc)
1425{
1426 struct ifnet *ifp = &sc->sc_ethercom.ec_if;
1427 struct pcn_rxsoft *rxs;
1428 struct mbuf *m;
1429 uint32_t rmd1;
1430 int i, len;
1431
1432 for (i = sc->sc_rxptr;; i = PCN_NEXTRX(i)) {
1433 rxs = &sc->sc_rxsoft[i];
1434
1435 PCN_CDRXSYNC(sc, i,
1436 BUS_DMASYNC_POSTREAD | BUS_DMASYNC_POSTWRITE);
1437
1438 rmd1 = le32toh(sc->sc_rxdescs[i].rmd1);
1439
1440 if (rmd1 & LE_R1_OWN)
1441 break;
1442
1443 /*
1444 * Check for errors and make sure the packet fit into
1445 * a single buffer. We have structured this block of
1446 * code the way it is in order to compress it into
1447 * one test in the common case (no error).
1448 */
1449 if (__predict_false((rmd1 & (LE_R1_STP | LE_R1_ENP |LE_R1_ERR))
1450 != (LE_R1_STP | LE_R1_ENP))) {
1451 /* Make sure the packet is in a single buffer. */
1452 if ((rmd1 & (LE_R1_STP | LE_R1_ENP)) !=
1453 (LE_R1_STP | LE_R1_ENP)) {
1454 printf("%s: packet spilled into next buffer\n",
1455 device_xname(sc->sc_dev));
1456 return 1; /* pcn_intr() will re-init */
1457 }
1458
1459 /*
1460 * If the packet had an error, simple recycle the
1461 * buffer.
1462 */
1463 if (rmd1 & LE_R1_ERR) {
1464 ifp->if_ierrors++;
1465 /*
1466 * If we got an overflow error, chances
1467 * are there will be a CRC error. In
1468 * this case, just print the overflow
1469 * error, and skip the others.
1470 */
1471 if (rmd1 & LE_R1_OFLO)
1472 printf("%s: overflow error\n",
1473 device_xname(sc->sc_dev));
1474 else {
1475#define PRINTIT(x, str) \
1476 if (rmd1 & (x)) \
1477 printf("%s: %s\n", \
1478 device_xname(sc->sc_dev), \
1479 str);
1480 PRINTIT(LE_R1_FRAM, "framing error");
1481 PRINTIT(LE_R1_CRC, "CRC error");
1482 PRINTIT(LE_R1_BUFF, "buffer error");
1483 }
1484#undef PRINTIT
1485 PCN_INIT_RXDESC(sc, i);
1486 continue;
1487 }
1488 }
1489
1490 bus_dmamap_sync(sc->sc_dmat, rxs->rxs_dmamap, 0,
1491 rxs->rxs_dmamap->dm_mapsize, BUS_DMASYNC_POSTREAD);
1492
1493 /*
1494 * No errors; receive the packet.
1495 */
1496 if (sc->sc_swstyle == LE_B20_SSTYLE_PCNETPCI3)
1497 len = le32toh(sc->sc_rxdescs[i].rmd0) & LE_R1_BCNT_MASK;
1498 else
1499 len = le32toh(sc->sc_rxdescs[i].rmd2) & LE_R1_BCNT_MASK;
1500
1501 /*
1502 * The LANCE family includes the CRC with every packet;
1503 * trim it off here.
1504 */
1505 len -= ETHER_CRC_LEN;
1506
1507 /*
1508 * If the packet is small enough to fit in a
1509 * single header mbuf, allocate one and copy
1510 * the data into it. This greatly reduces
1511 * memory consumption when we receive lots
1512 * of small packets.
1513 *
1514 * Otherwise, we add a new buffer to the receive
1515 * chain. If this fails, we drop the packet and
1516 * recycle the old buffer.
1517 */
1518 if (pcn_copy_small != 0 && len <= (MHLEN - 2)) {
1519 MGETHDR(m, M_DONTWAIT, MT_DATA);
1520 if (m == NULL)
1521 goto dropit;
1522 m->m_data += 2;
1523 memcpy(mtod(m, void *),
1524 mtod(rxs->rxs_mbuf, void *), len);
1525 PCN_INIT_RXDESC(sc, i);
1526 bus_dmamap_sync(sc->sc_dmat, rxs->rxs_dmamap, 0,
1527 rxs->rxs_dmamap->dm_mapsize,
1528 BUS_DMASYNC_PREREAD);
1529 } else {
1530 m = rxs->rxs_mbuf;
1531 if (pcn_add_rxbuf(sc, i) != 0) {
1532 dropit:
1533 ifp->if_ierrors++;
1534 PCN_INIT_RXDESC(sc, i);
1535 bus_dmamap_sync(sc->sc_dmat,
1536 rxs->rxs_dmamap, 0,
1537 rxs->rxs_dmamap->dm_mapsize,
1538 BUS_DMASYNC_PREREAD);
1539 continue;
1540 }
1541 }
1542
1543 m_set_rcvif(m, ifp);
1544 m->m_pkthdr.len = m->m_len = len;
1545
1546 /* Pass it on. */
1547 if_percpuq_enqueue(ifp->if_percpuq, m);
1548 }
1549
1550 /* Update the receive pointer. */
1551 sc->sc_rxptr = i;
1552 return 0;
1553}

1573/*
1574 * pcn_reset:
1575 *
1576 * Perform a soft reset on the PCnet-PCI.
1577 */
1578static void
1579pcn_reset(struct pcn_softc *sc)
1580{
1581
1582 /*
1583 * The PCnet-PCI chip is reset by reading from the
1584 * RESET register. Note that while the NE2100 LANCE
1585 * boards require a write after the read, the PCnet-PCI
1586 * chips do not require this.
1587 *
1588 * Since we don't know if we're in 16-bit or 32-bit
1589 * mode right now, issue both (it's safe) in the
1590 * hopes that one will succeed.
1591 */
1592 (void) bus_space_read_2(sc->sc_st, sc->sc_sh, PCN16_RESET);
1593 (void) bus_space_read_4(sc->sc_st, sc->sc_sh, PCN32_RESET);
1594
1595 /* Wait 1ms for it to finish. */
1596 delay(1000);
1597
1598 /*
1599 * Select 32-bit I/O mode by issuing a 32-bit write to the
1600 * RDP. Since the RAP is 0 after a reset, writing a 0
1601 * to RDP is safe (since it simply clears CSR0).
1602 */
1603 bus_space_write_4(sc->sc_st, sc->sc_sh, PCN32_RDP, 0);
1604}

1606/*
1607 * pcn_init: [ifnet interface function]
1608 *
1609 * Initialize the interface. Must be called at splnet().
1610 */
1611static int
1612pcn_init(struct ifnet *ifp)
1613{
1614 struct pcn_softc *sc = ifp->if_softc;
1615 struct pcn_rxsoft *rxs;
1616 const uint8_t *enaddr = CLLADDR(ifp->if_sadl);
1617 int i, error = 0;
1618 uint32_t reg;
1619
1620 /* Cancel any pending I/O. */
1621 pcn_stop(ifp, 0);
1622
1623 /* Reset the chip to a known state. */
1624 pcn_reset(sc);
1625
1626 /*
1627 * On the Am79c970, select SSTYLE 2, and SSTYLE 3 on everything
1628 * else.
1629 *
1630 * XXX It'd be really nice to use SSTYLE 2 on all the chips,
1631 * because the structure layout is compatible with ILACC,
1632 * but the burst mode is only available in SSTYLE 3, and
1633 * burst mode should provide some performance enhancement.
1634 */
1635 if (sc->sc_variant->pcv_chipid == PARTID_Am79c970)
1636 sc->sc_swstyle = LE_B20_SSTYLE_PCNETPCI2;
1637 else
1638 sc->sc_swstyle = LE_B20_SSTYLE_PCNETPCI3;
1639 pcn_bcr_write(sc, LE_BCR20, sc->sc_swstyle);
1640
1641 /* Initialize the transmit descriptor ring. */
1642 memset(sc->sc_txdescs, 0, sizeof(sc->sc_txdescs));
1643 PCN_CDTXSYNC(sc, 0, PCN_NTXDESC,
1644 BUS_DMASYNC_PREREAD | BUS_DMASYNC_PREWRITE);
1645 sc->sc_txfree = PCN_NTXDESC;
1646 sc->sc_txnext = 0;
1647
1648 /* Initialize the transmit job descriptors. */
1649 for (i = 0; i < PCN_TXQUEUELEN; i++)
1650 sc->sc_txsoft[i].txs_mbuf = NULL;
1651 sc->sc_txsfree = PCN_TXQUEUELEN;
1652 sc->sc_txsnext = 0;
1653 sc->sc_txsdirty = 0;
1654
1655 /*
1656 * Initialize the receive descriptor and receive job
1657 * descriptor rings.
1658 */
1659 for (i = 0; i < PCN_NRXDESC; i++) {
1660 rxs = &sc->sc_rxsoft[i];
1661 if (rxs->rxs_mbuf == NULL) {
1662 if ((error = pcn_add_rxbuf(sc, i)) != 0) {
1663 printf("%s: unable to allocate or map rx "
1664 "buffer %d, error = %d\n",
1665 device_xname(sc->sc_dev), i, error);
1666 /*
1667 * XXX Should attempt to run with fewer receive
1668 * XXX buffers instead of just failing.
1669 */
1670 pcn_rxdrain(sc);
1671 goto out;
1672 }
1673 } else
1674 PCN_INIT_RXDESC(sc, i);
1675 }
1676 sc->sc_rxptr = 0;
1677
1678 /* Initialize MODE for the initialization block. */
1679 sc->sc_mode = 0;
1680 if (ifp->if_flags & IFF_PROMISC)
1681 sc->sc_mode |= LE_C15_PROM;
1682 if ((ifp->if_flags & IFF_BROADCAST) == 0)
1683 sc->sc_mode |= LE_C15_DRCVBC;
1684
1685 /*
1686 * If we have MII, simply select MII in the MODE register,
1687 * and clear ASEL. Otherwise, let ASEL stand (for now),
1688 * and leave PORTSEL alone (it is ignored with ASEL is set).
1689 */
1690 if (sc->sc_flags & PCN_F_HAS_MII) {
1691 pcn_bcr_write(sc, LE_BCR2,
1692 pcn_bcr_read(sc, LE_BCR2) & ~LE_B2_ASEL);
1693 sc->sc_mode |= LE_C15_PORTSEL(PORTSEL_MII);
1694
1695 /*
1696 * Disable MII auto-negotiation. We handle that in
1697 * our own MII layer.
1698 */
1699 pcn_bcr_write(sc, LE_BCR32,
1700 pcn_bcr_read(sc, LE_BCR32) | LE_B32_DANAS);
1701 }
1702
1703 /*
1704 * Set the Tx and Rx descriptor ring addresses in the init
1705 * block, the TLEN and RLEN other fields of the init block
1706 * MODE register.
1707 */
1708 sc->sc_initblock.init_rdra = htole32(PCN_CDRXADDR(sc, 0));
1709 sc->sc_initblock.init_tdra = htole32(PCN_CDTXADDR(sc, 0));
1710 sc->sc_initblock.init_mode = htole32(sc->sc_mode |
1711 ((ffs(PCN_NTXDESC) - 1) << 28) |
1712 ((ffs(PCN_NRXDESC) - 1) << 20));
1713
1714 /* Set the station address in the init block. */
1715 sc->sc_initblock.init_padr[0] = htole32(enaddr[0] |
1716 (enaddr[1] << 8) | (enaddr[2] << 16) | (enaddr[3] << 24));
1717 sc->sc_initblock.init_padr[1] = htole32(enaddr[4] |
1718 (enaddr[5] << 8));
1719
1720 /* Set the multicast filter in the init block. */
1721 pcn_set_filter(sc);
1722
1723 /* Initialize CSR3. */
1724 pcn_csr_write(sc, LE_CSR3, LE_C3_MISSM | LE_C3_IDONM | LE_C3_DXSUFLO);
1725
1726 /* Initialize CSR4. */
1727 pcn_csr_write(sc, LE_CSR4, LE_C4_DMAPLUS | LE_C4_APAD_XMT |
1728 LE_C4_MFCOM | LE_C4_RCVCCOM | LE_C4_TXSTRTM);
1729
1730 /* Initialize CSR5. */
1731 sc->sc_csr5 = LE_C5_LTINTEN | LE_C5_SINTE;
1732 pcn_csr_write(sc, LE_CSR5, sc->sc_csr5);
1733
1734 /*
1735 * If we have an Am79c971 or greater, initialize CSR7.
1736 *
1737 * XXX Might be nice to use the MII auto-poll interrupt someday.
1738 */
1739 switch (sc->sc_variant->pcv_chipid) {
1740 case PARTID_Am79c970:
1741 case PARTID_Am79c970A:
1742 /* Not available on these chips. */
1743 break;
1744
1745 default:
1746 pcn_csr_write(sc, LE_CSR7, LE_C7_FASTSPNDE);
1747 break;
1748 }
1749
1750 /*
1751 * On the Am79c970A and greater, initialize BCR18 to
1752 * enable burst mode.
1753 *
1754 * Also enable the "no underflow" option on the Am79c971 and
1755 * higher, which prevents the chip from generating transmit
1756 * underflows, yet sill provides decent performance. Note if
1757 * chip is not connected to external SRAM, then we still have
1758 * to handle underflow errors (the NOUFLO bit is ignored in
1759 * that case).
1760 */
1761 reg = pcn_bcr_read(sc, LE_BCR18);
1762 switch (sc->sc_variant->pcv_chipid) {
1763 case PARTID_Am79c970:
1764 break;
1765
1766 case PARTID_Am79c970A:
1767 reg |= LE_B18_BREADE | LE_B18_BWRITE;
1768 break;
1769
1770 default:
1771 reg |= LE_B18_BREADE | LE_B18_BWRITE | LE_B18_NOUFLO;
1772 break;
1773 }
1774 pcn_bcr_write(sc, LE_BCR18, reg);
1775
1776 /*
1777 * Initialize CSR80 (FIFO thresholds for Tx and Rx).
1778 */
1779 pcn_csr_write(sc, LE_CSR80, LE_C80_RCVFW(sc->sc_rcvfw) |
1780 LE_C80_XMTSP(sc->sc_xmtsp) | LE_C80_XMTFW(sc->sc_xmtfw));
1781
1782 /*
1783 * Send the init block to the chip, and wait for it
1784 * to be processed.
1785 */
1786 PCN_CDINITSYNC(sc, BUS_DMASYNC_PREWRITE);
1787 pcn_csr_write(sc, LE_CSR1, PCN_CDINITADDR(sc) & 0xffff);
1788 pcn_csr_write(sc, LE_CSR2, (PCN_CDINITADDR(sc) >> 16) & 0xffff);
1789 pcn_csr_write(sc, LE_CSR0, LE_C0_INIT);
1790 delay(100);
1791 for (i = 0; i < 10000; i++) {
1792 if (pcn_csr_read(sc, LE_CSR0) & LE_C0_IDON)
1793 break;
1794 delay(10);
1795 }
1796 PCN_CDINITSYNC(sc, BUS_DMASYNC_POSTWRITE);
1797 if (i == 10000) {
1798 printf("%s: timeout processing init block\n",
1799 device_xname(sc->sc_dev));
1800 error = EIO;
1801 goto out;
1802 }
1803
1804 /* Set the media. */
1805 if ((error = mii_ifmedia_change(&sc->sc_mii)) != 0)
1806 goto out;
1807
1808 /* Enable interrupts and external activity (and ACK IDON). */
1809 pcn_csr_write(sc, LE_CSR0, LE_C0_INEA | LE_C0_STRT | LE_C0_IDON);
1810
1811 if (sc->sc_flags & PCN_F_HAS_MII) {
1812 /* Start the one second MII clock. */
1813 callout_reset(&sc->sc_tick_ch, hz, pcn_tick, sc);
1814 }
1815
1816 /* ...all done! */
1817 ifp->if_flags |= IFF_RUNNING;
1818 ifp->if_flags &= ~IFF_OACTIVE;
1819
1820 out:
1821 if (error)
1822 printf("%s: interface not running\n", device_xname(sc->sc_dev));
1823 return error;
1824}

1847/*
1848 * pcn_stop: [ifnet interface function]
1849 *
1850 * Stop transmission on the interface.
1851 */
1852static void
1853pcn_stop(struct ifnet *ifp, int disable)
1854{
1855 struct pcn_softc *sc = ifp->if_softc;
1856 struct pcn_txsoft *txs;
1857 int i;
1858
1859 if (sc->sc_flags & PCN_F_HAS_MII) {
1860 /* Stop the one second clock. */
1861 callout_stop(&sc->sc_tick_ch);
1862
1863 /* Down the MII. */
1864 mii_down(&sc->sc_mii);
1865 }
1866
1867 /* Stop the chip. */
1868 pcn_csr_write(sc, LE_CSR0, LE_C0_STOP);
1869
1870 /* Release any queued transmit buffers. */
1871 for (i = 0; i < PCN_TXQUEUELEN; i++) {
1872 txs = &sc->sc_txsoft[i];
1873 if (txs->txs_mbuf != NULL) {
1874 bus_dmamap_unload(sc->sc_dmat, txs->txs_dmamap);
1875 m_freem(txs->txs_mbuf);
1876 txs->txs_mbuf = NULL;
1877 }
1878 }
1879
1880 /* Mark the interface as down and cancel the watchdog timer. */
1881 ifp->if_flags &= ~(IFF_RUNNING | IFF_OACTIVE);
1882 ifp->if_timer = 0;
1883
1884 if (disable)
1885 pcn_rxdrain(sc);
1886}

1888/*
1889 * pcn_add_rxbuf:
1890 *
1891 * Add a receive buffer to the indicated descriptor.
1892 */
1893static int
1894pcn_add_rxbuf(struct pcn_softc *sc, int idx)
1895{
1896 struct pcn_rxsoft *rxs = &sc->sc_rxsoft[idx];
1897 struct mbuf *m;
1898 int error;
1899
1900 MGETHDR(m, M_DONTWAIT, MT_DATA);
1901 if (m == NULL)
1902 return ENOBUFS;
1903
1904 MCLGET(m, M_DONTWAIT);
1905 if ((m->m_flags & M_EXT) == 0) {
1906 m_freem(m);
1907 return ENOBUFS;
1908 }
1909
1910 if (rxs->rxs_mbuf != NULL)
1911 bus_dmamap_unload(sc->sc_dmat, rxs->rxs_dmamap);
1912
1913 rxs->rxs_mbuf = m;
1914
1915 error = bus_dmamap_load(sc->sc_dmat, rxs->rxs_dmamap,
1916 m->m_ext.ext_buf, m->m_ext.ext_size, NULL,
1917 BUS_DMA_READ | BUS_DMA_NOWAIT);
1918 if (error) {
1919 printf("%s: can't load rx DMA map %d, error = %d\n",
1920 device_xname(sc->sc_dev), idx, error);
1921 panic("pcn_add_rxbuf");
1922 }
1923
1924 bus_dmamap_sync(sc->sc_dmat, rxs->rxs_dmamap, 0,
1925 rxs->rxs_dmamap->dm_mapsize, BUS_DMASYNC_PREREAD);
1926
1927 PCN_INIT_RXDESC(sc, idx);
1928
1929 return 0;
1930}

1932/*
1933 * pcn_set_filter:
1934 *
1935 * Set up the receive filter.
1936 */
1937static void
1938pcn_set_filter(struct pcn_softc *sc)
1939{
1940 struct ethercom *ec = &sc->sc_ethercom;
1941 struct ifnet *ifp = &sc->sc_ethercom.ec_if;
1942 struct ether_multi *enm;
1943 struct ether_multistep step;
1944 uint32_t crc;
1945
1946 /*
1947 * Set up the multicast address filter by passing all multicast
1948 * addresses through a CRC generator, and then using the high
1949 * order 6 bits as an index into the 64-bit logical address
1950 * filter. The high order bits select the word, while the rest
1951 * of the bits select the bit within the word.
1952 */
1953
1954 if (ifp->if_flags & IFF_PROMISC)
1955 goto allmulti;
1956
1957 sc->sc_initblock.init_ladrf[0] =
1958 sc->sc_initblock.init_ladrf[1] =
1959 sc->sc_initblock.init_ladrf[2] =
1960 sc->sc_initblock.init_ladrf[3] = 0;
1961
1962 ETHER_LOCK(ec);
1963 ETHER_FIRST_MULTI(step, ec, enm);
1964 while (enm != NULL) {
1965 if (memcmp(enm->enm_addrlo, enm->enm_addrhi, ETHER_ADDR_LEN)) {
1966 /*
1967 * We must listen to a range of multicast addresses.
1968 * For now, just accept all multicasts, rather than
1969 * trying to set only those filter bits needed to match
1970 * the range. (At this time, the only use of address
1971 * ranges is for IP multicast routing, for which the
1972 * range is big enough to require all bits set.)
1973 */
1974 ETHER_UNLOCK(ec);
1975 goto allmulti;
1976 }
1977
1978 crc = ether_crc32_le(enm->enm_addrlo, ETHER_ADDR_LEN);
1979
1980 /* Just want the 6 most significant bits. */
1981 crc >>= 26;
1982
1983 /* Set the corresponding bit in the filter. */
1984 sc->sc_initblock.init_ladrf[crc >> 4] |=
1985 htole16(1 << (crc & 0xf));
1986
1987 ETHER_NEXT_MULTI(step, enm);
1988 }
1989 ETHER_UNLOCK(ec);
1990
1991 ifp->if_flags &= ~IFF_ALLMULTI;
1992 return;
1993
1994 allmulti:
1995 ifp->if_flags |= IFF_ALLMULTI;
1996 sc->sc_initblock.init_ladrf[0] =
1997 sc->sc_initblock.init_ladrf[1] =
1998 sc->sc_initblock.init_ladrf[2] =
1999 sc->sc_initblock.init_ladrf[3] = 0xffff;
2000}
