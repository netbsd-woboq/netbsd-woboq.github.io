38#ifndef _DEV_PCI_IF_PCNREG_H_
39#define _DEV_PCI_IF_PCNREG_H_
40
41/*
42 * Register definitions for the AMD PCnet-PCI series of Ethernet
43 * chips.
44 *
45 * These are only the registers that we access directly from PCI
46 * space. Everything else (accessed via the RAP + RDP/BDP) is
47 * defined in <dev/ic/lancereg.h>.
48 */
49
50/*
51 * PCI configuration space.
52 */
53
54#define PCN_PCI_CBIO (PCI_MAPREG_START + 0x00)
55#define PCN_PCI_CBMEM (PCI_MAPREG_START + 0x04)
56
57/*
58 * I/O map in Word I/O mode.
59 */
60
61#define PCN16_APROM 0x00
62#define PCN16_RDP 0x10
63#define PCN16_RAP 0x12
64#define PCN16_RESET 0x14
65#define PCN16_BDP 0x16
66
67/*
68 * I/O map in DWord I/O mode.
69 */
70
71#define PCN32_APROM 0x00
72#define PCN32_RDP 0x10
73#define PCN32_RAP 0x14
74#define PCN32_RESET 0x18
75#define PCN32_BDP 0x1c
76
77#endif /* _DEV_PCI_IF_PCNREG_H_ */
