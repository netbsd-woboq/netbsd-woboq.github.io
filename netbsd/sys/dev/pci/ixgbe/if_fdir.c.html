39void
40ixgbe_init_fdir(struct adapter *adapter)
41{
42 u32 hdrm = 32 << fdir_pballoc;
43
44 if (!(adapter->feat_en & IXGBE_FEATURE_FDIR))
45 return;
46
47 adapter->hw.mac.ops.setup_rxpba(&adapter->hw, 0, hdrm,
48 PBA_STRATEGY_EQUAL);
49 ixgbe_init_fdir_signature_82599(&adapter->hw, fdir_pballoc);
50} /* ixgbe_init_fdir */
51
52void
53ixgbe_reinit_fdir(void *context)
54{
55 struct adapter *adapter = context;
56 struct ifnet *ifp = adapter->ifp;
57
58 if (!(adapter->feat_en & IXGBE_FEATURE_FDIR))
59 return;
60 if (adapter->fdir_reinit != 1) /* Shouldn't happen */
61 return;
62 ixgbe_reinit_fdir_tables_82599(&adapter->hw);
63 adapter->fdir_reinit = 0;
64 /* re-enable flow director interrupts */
65 IXGBE_WRITE_REG(&adapter->hw, IXGBE_EIMS, IXGBE_EIMS_FLOW_DIR);
66 /* Restart the interface */
67 ifp->if_drv_flags |= IFF_DRV_RUNNING;
68} /* ixgbe_reinit_fdir */

70/************************************************************************
71 * ixgbe_atr
72 *
73 * Parse packet headers so that Flow Director can make
74 * a hashed filter table entry allowing traffic flows
75 * to be identified and kept on the same cpu. This
76 * would be a performance hit, but we only do it at
77 * IXGBE_FDIR_RATE of packets.
78 ************************************************************************/
79void
80ixgbe_atr(struct tx_ring *txr, struct mbuf *mp)
81{
82 struct adapter *adapter = txr->adapter;
83 struct ix_queue *que;
84 struct ip *ip;
85 struct tcphdr *th;
86 struct udphdr *uh;
87 struct ether_vlan_header *eh;
88 union ixgbe_atr_hash_dword input = {.dword = 0};
89 union ixgbe_atr_hash_dword common = {.dword = 0};
90 int ehdrlen, ip_hlen;
91 u16 etype;
92
93 eh = mtod(mp, struct ether_vlan_header *);
94 if (eh->evl_encap_proto == htons(ETHERTYPE_VLAN)) {
95 ehdrlen = ETHER_HDR_LEN + ETHER_VLAN_ENCAP_LEN;
96 etype = eh->evl_proto;
97 } else {
98 ehdrlen = ETHER_HDR_LEN;
99 etype = eh->evl_encap_proto;
100 }
101
102 /* Only handling IPv4 */
103 if (etype != htons(ETHERTYPE_IP))
104 return;
105
106 ip = (struct ip *)(mp->m_data + ehdrlen);
107 ip_hlen = ip->ip_hl << 2;
108
109 /* check if we're UDP or TCP */
110 switch (ip->ip_p) {
111 case IPPROTO_TCP:
112 th = (struct tcphdr *)((caddr_t)ip + ip_hlen);
113 /* src and dst are inverted */
114 common.port.dst ^= th->th_sport;
115 common.port.src ^= th->th_dport;
116 input.formatted.flow_type ^= IXGBE_ATR_FLOW_TYPE_TCPV4;
117 break;
118 case IPPROTO_UDP:
119 uh = (struct udphdr *)((caddr_t)ip + ip_hlen);
120 /* src and dst are inverted */
121 common.port.dst ^= uh->uh_sport;
122 common.port.src ^= uh->uh_dport;
123 input.formatted.flow_type ^= IXGBE_ATR_FLOW_TYPE_UDPV4;
124 break;
125 default:
126 return;
127 }
128
129 input.formatted.vlan_id = htobe16(mp->m_pkthdr.ether_vtag);
130 if (mp->m_pkthdr.ether_vtag)
131 common.flex_bytes ^= htons(ETHERTYPE_VLAN);
132 else
133 common.flex_bytes ^= etype;
134 common.ip ^= ip->ip_src.s_addr ^ ip->ip_dst.s_addr;
135
136 que = &adapter->queues[txr->me];
137 /*
138 * This assumes the Rx queue and Tx
139 * queue are bound to the same CPU
140 */
141 ixgbe_fdir_add_signature_filter_82599(&adapter->hw,
142 input, common, que->msix);
143} /* ixgbe_atr */
