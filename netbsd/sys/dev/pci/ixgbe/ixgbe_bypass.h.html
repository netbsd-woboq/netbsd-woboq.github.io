39/*
40 * The bypass driver needs to set FW to a epoc of the number of
41 * seconds we are into this year. This macro's help support that.
42 */
43#define SEC_PER_DAY (60 * 60 * 24)
44#define SEC_PER_YEAR (SEC_PER_DAY * 365)
45#define SEC_PER_LYEAR (SEC_PER_DAY * 366)
46#define LEAP_YR(y) ((y % 400 == 0) || ((y % 4 == 0) && (y % 100 != 0)))
47#define SEC_THIS_YEAR(y) (LEAP_YR(y) ? SEC_PER_LYEAR : SEC_PER_YEAR)
