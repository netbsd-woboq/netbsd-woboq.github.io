36#ifndef _IXGBE_FEATURES_H_
37#define _IXGBE_FEATURES_H_
38
39/*
40 * Feature defines. Eventually, we'd like to get to a point where we
41 * can remove MAC/Phy type checks scattered throughout the code in
42 * favor of checking these feature flags. If the feature expects OS
43 * support, make sure to add an #undef below if expected to run on
44 * OSs that don't support said feature.
45 */
46#define IXGBE_FEATURE_VF (u32)(1 << 0)
47#define IXGBE_FEATURE_SRIOV (u32)(1 << 1)
48#define IXGBE_FEATURE_RSS (u32)(1 << 2)
49#define IXGBE_FEATURE_NETMAP (u32)(1 << 3)
50#define IXGBE_FEATURE_FAN_FAIL (u32)(1 << 4)
51#define IXGBE_FEATURE_TEMP_SENSOR (u32)(1 << 5)
52#define IXGBE_FEATURE_BYPASS (u32)(1 << 6)
53#define IXGBE_FEATURE_LEGACY_TX (u32)(1 << 7)
54#define IXGBE_FEATURE_FDIR (u32)(1 << 8)
55#define IXGBE_FEATURE_MSI (u32)(1 << 9)
56#define IXGBE_FEATURE_MSIX (u32)(1 << 10)
57#define IXGBE_FEATURE_EEE (u32)(1 << 11)
58#define IXGBE_FEATURE_LEGACY_IRQ (u32)(1 << 12)
59#define IXGBE_FEATURE_NEEDS_CTXD (u32)(1 << 13)
60#define IXGBE_FEATURE_RECOVERY_MODE (u32)(1 << 15)
61
62#define IXGBE_FEATURE_FLAGS "\20" \
63 "\1" "VF" "\2" "SRIOV" "\3" "RSS" "\4" "NETMAP" \
64 "\5" "FAN_FAIL" "\6" "TEMP_SENSOR" "\7" "BYPASS" "\10" "LEGACY_TX" \
65 "\11" "FDIR" "\12" "MSI" "\13" "MSIX" "\14" "EEE" \
66 "\15" "LEGACY_IRQ" "\16" "NEEDS_CTXD" "\20" "RECOVERY_MODE"
67
68/* Check for OS support. Undefine features if not included in the OS */
69#ifndef PCI_IOV
70#undef IXGBE_FEATURE_SRIOV
71#define IXGBE_FEATURE_SRIOV 0
72#endif
73
74#ifndef RSS
75#undef IXGBE_FEATURE_RSS
76#define IXGBE_FEATURE_RSS 0
77#endif
78
79#ifndef DEV_NETMAP
80#undef IXGBE_FEATURE_NETMAP
81#define IXGBE_FEATURE_NETMAP 0
82#endif
83
84#endif /* _IXGBE_FEATURES_H_ */
