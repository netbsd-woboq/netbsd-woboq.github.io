45void
46ixgbe_dma_tag_destroy(ixgbe_dma_tag_t *dt)
47{
48 kmem_free(dt, sizeof(*dt));
49}

51int
52ixgbe_dma_tag_create(bus_dma_tag_t dmat, bus_size_t alignment,
53 bus_size_t boundary, bus_size_t maxsize, int nsegments,
54 bus_size_t maxsegsize, int flags, ixgbe_dma_tag_t **dtp)
55{
56 ixgbe_dma_tag_t *dt;
57
58 *dtp = NULL;
59
60 dt = kmem_zalloc(sizeof(*dt), KM_SLEEP);
61 dt->dt_dmat = dmat;
62 dt->dt_alignment = alignment;
63 dt->dt_boundary = boundary;
64 dt->dt_maxsize = maxsize;
65 dt->dt_nsegments = nsegments;
66 dt->dt_maxsegsize = maxsegsize;
67 dt->dt_flags = flags;
68 *dtp = dt;
69
70 return 0;
71}

98static void
99ixgbe_putext(ixgbe_extmem_t *em)
100{
101 ixgbe_extmem_head_t *eh = em->em_head;
102
103 mutex_enter(&eh->eh_mtx);
104
105 TAILQ_INSERT_HEAD(&eh->eh_freelist, em, em_link);
106
107 mutex_exit(&eh->eh_mtx);
108
109 return;
110}
111
112static ixgbe_extmem_t *
113ixgbe_getext(ixgbe_extmem_head_t *eh, size_t size)
114{
115 ixgbe_extmem_t *em;
116
117 mutex_enter(&eh->eh_mtx);
118
119 TAILQ_FOREACH(em, &eh->eh_freelist, em_link) {
120 if (em->em_size >= size)
121 break;
122 }
123
124 if (em != NULL)
125 TAILQ_REMOVE(&eh->eh_freelist, em, em_link);
126
127 mutex_exit(&eh->eh_mtx);
128
129 return em;
130}

132static ixgbe_extmem_t *
133ixgbe_newext(ixgbe_extmem_head_t *eh, bus_dma_tag_t dmat, size_t size)
134{
135 ixgbe_extmem_t *em;
136 int nseg, rc;
137
138 em = kmem_zalloc(sizeof(*em), KM_SLEEP);
139
140 rc = bus_dmamem_alloc(dmat, size, PAGE_SIZE, 0, &em->em_seg, 1, &nseg,
141 BUS_DMA_WAITOK);
142
143 if (rc != 0)
144 goto post_zalloc_err;
145
146 rc = bus_dmamem_map(dmat, &em->em_seg, 1, size, &em->em_vaddr,
147 BUS_DMA_WAITOK);
148
149 if (rc != 0)
150 goto post_dmamem_err;
151
152 em->em_dmat = dmat;
153 em->em_size = size;
154 em->em_head = eh;
155
156 return em;
157post_dmamem_err:
158 bus_dmamem_free(dmat, &em->em_seg, 1);
159post_zalloc_err:
160 kmem_free(em, sizeof(*em));
161 return NULL;
162}

164void
165ixgbe_jcl_reinit(struct adapter *adapter, bus_dma_tag_t dmat,
166 struct rx_ring *rxr, int nbuf, size_t size)
167{
168 ixgbe_extmem_head_t *eh = &rxr->jcl_head;
169 ixgbe_extmem_t *em;
170 int i;
171
172 if (!eh->eh_initialized) {
173 TAILQ_INIT(&eh->eh_freelist);
174 mutex_init(&eh->eh_mtx, MUTEX_DEFAULT, IPL_NET);
175 eh->eh_initialized = true;
176 }
177
178 /*
179 * Check previous parameters. If it's not required to reinit, just
180 * return.
181 *
182 * Note that the num_rx_desc is currently fixed value. It's never
183 * changed after device is attached.
184 */
185 if ((rxr->last_rx_mbuf_sz == rxr->mbuf_sz)
186 && (rxr->last_num_rx_desc == adapter->num_rx_desc))
187 return;
188
189 /* Free all dmamem */
190 while ((em = ixgbe_getext(eh, 0)) != NULL) {
191 KASSERT(em->em_vaddr != NULL);
192 bus_dmamem_unmap(dmat, em->em_vaddr, em->em_size);
193 bus_dmamem_free(dmat, &em->em_seg, 1);
194 memset(em, 0, sizeof(*em));
195 kmem_free(em, sizeof(*em));
196 }
197
198 for (i = 0; i < nbuf; i++) {
199 if ((em = ixgbe_newext(eh, dmat, size)) == NULL) {
200 printf("%s: only %d of %d jumbo buffers allocated\n",
201 __func__, i, nbuf);
202 break;
203 }
204 ixgbe_putext(em);
205 }
206
207 /* Keep current parameters */
208 rxr->last_rx_mbuf_sz = adapter->rx_mbuf_sz;
209 rxr->last_num_rx_desc = adapter->num_rx_desc;
210}

212static void
213ixgbe_jcl_free(struct mbuf *m, void *buf, size_t size, void *arg)
214{
215 ixgbe_extmem_t *em = arg;
216
217 KASSERT(em->em_size == size);
218
219 ixgbe_putext(em);
220 /* this is an abstraction violation, but it does not lead to a
221 * double-free
222 */
223 if (__predict_true(m != NULL)) {
224 KASSERT(m->m_type != MT_FREE);
225 m->m_type = MT_FREE;
226 pool_cache_put(mb_cache, m);
227 }
228}

230/* XXX need to wait for the system to finish with each jumbo mbuf and
231 * free it before detaching the driver from the device.
232 */
233struct mbuf *
234ixgbe_getjcl(ixgbe_extmem_head_t *eh, int nowait /* M_DONTWAIT */,
235 int type /* MT_DATA */, int flags /* M_PKTHDR */, size_t size)
236{
237 ixgbe_extmem_t *em;
238 struct mbuf *m;
239
240 if ((flags & M_PKTHDR) != 0)
241 m = m_gethdr(nowait, type);
242 else
243 m = m_get(nowait, type);
244
245 if (m == NULL)
246 return NULL;
247
248 em = ixgbe_getext(eh, size);
249 if (em == NULL) {
250 m_freem(m);
251 return NULL;
252 }
253
254 MEXTADD(m, em->em_vaddr, em->em_size, M_DEVBUF, &ixgbe_jcl_free, em);
255
256 if ((m->m_flags & M_EXT) == 0) {
257 ixgbe_putext(em);
258 m_freem(m);
259 return NULL;
260 }
261
262 return m;
263}

265void
266ixgbe_pci_enable_busmaster(pci_chipset_tag_t pc, pcitag_t tag)
267{
268 pcireg_t pci_cmd_word;
269
270 pci_cmd_word = pci_conf_read(pc, tag, PCI_COMMAND_STATUS_REG);
271 if (!(pci_cmd_word & PCI_COMMAND_MASTER_ENABLE)) {
272 pci_cmd_word |= PCI_COMMAND_MASTER_ENABLE;
273 pci_conf_write(pc, tag, PCI_COMMAND_STATUS_REG, pci_cmd_word);
274 }
275}

277u_int
278atomic_load_acq_uint(volatile u_int *p)
279{
280 u_int rv;
281
282 rv = *p;
283 /*
284 * XXX
285 * membar_sync() is far more than we need on most CPUs;
286 * we just don't have an MI load-acqure operation.
287 */
288 membar_sync();
289
290 return rv;
291}
