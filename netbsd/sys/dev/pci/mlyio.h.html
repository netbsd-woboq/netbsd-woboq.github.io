60/*
61 * Control structures exchanged through the GAM interface with userland
62 * management tools.
63 *
64 * The member naming here is consistent with the Linux driver, with which this
65 * interface is basically compatible.
66 */
67
68#ifndef _PCI_MLYIO_H_
69#define _PCI_MLYIO_H_
70
71#include <sys/ioccom.h>
72
73
74struct mly_user_command {
75 unsigned char ControllerNumber;
76 union mly_cmd_packet CommandMailbox;
77 int DataTransferLength;
78 int RequestSenseLength;
79 void *DataTransferBuffer;
80 void *RequestSenseBuffer;
81 int CommandStatus; /* not in the Linux structure */
82};
83
84#define MLYIO_COMMAND _IOWR('M', 200, struct mly_user_command)
85
86struct mly_user_health {
87 unsigned char ControllerNumber;
88 void *HealthStatusBuffer;
89};
90
91#define MLYIO_HEALTH _IOW('M', 201, struct mly_user_health)
92
93/*
94 * Command queue statistics
95 */
96
97#define MLYQ_FREE 0
98#define MLYQ_BUSY 1
99#define MLYQ_COMPLETE 2
100#define MLYQ_COUNT 3
101
102struct mly_qstat {
103 u_int32_t q_length;
104 u_int32_t q_max;
105};
106
107#endif /* !defined _PCI_MLYIO_ */
