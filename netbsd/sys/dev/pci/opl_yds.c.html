61static int
62opl_yds_match(device_t parent, cfdata_t match, void *aux)
63{
64 struct audio_attach_args *aa = (struct audio_attach_args *)aux;
65 struct yds_softc *ssc = device_private(parent);
66
67 if (aa->type != AUDIODEV_TYPE_OPL)
68 return (0);
69 return opl_match(ssc->sc_opl_iot, ssc->sc_opl_ioh, 0);
70}
71
72static void
73opl_yds_attach(device_t parent, device_t self, void *aux)
74{
75 struct yds_softc *ssc = device_private(parent);
76 struct opl_softc *sc = device_private(self);
77
78 sc->dev = self;
79 sc->ioh = ssc->sc_opl_ioh;
80 sc->iot = ssc->sc_opl_iot;
81 sc->offs = 0;
82 sc->lock = &ssc->sc_intr_lock;
83 strcpy(sc->syn.name, "DS-1 integrated ");
84
85 opl_attach(sc);
86}
87
88CFATTACH_DECL_NEW(opl_yds, sizeof (struct opl_softc),
89 opl_yds_match, opl_yds_attach, NULL, NULL);
