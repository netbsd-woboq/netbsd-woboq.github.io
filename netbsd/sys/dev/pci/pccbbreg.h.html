198/* socket event register (CB_SOCKET_EVENT) elements */
199#define CB_SOCKET_EVENT_CSTS 0x01 /* CARDSTS event occurs */
200#define CB_SOCKET_EVENT_CD 0x06 /* CD event occurs */
201#define CB_SOCKET_EVENT_CD1 0x02 /* CD1 event occurs */
202#define CB_SOCKET_EVENT_CD2 0x04 /* CD2 event occurs */
203#define CB_SOCKET_EVENT_POWER 0x08 /* Power cycle event occurs */
204
205
206/* socket mask register (CB_SOCKET_MASK) elements */
207#define CB_SOCKET_MASK_CSTS 0x01 /* CARDSTS event mask */
208#define CB_SOCKET_MASK_CD 0x06 /* CD event mask */
209#define CB_SOCKET_MASK_POWER 0x08 /* Power cycle event mask */
210
211/* socket present-state register (CB_SOCKET_STAT) elements */
212#define CB_SOCKET_STAT_CARDSTS 0x01 /* card status change bit */
213#define CB_SOCKET_STAT_CD1 0x02 /* card detect 1 */
214#define CB_SOCKET_STAT_CD2 0x04 /* card detect 2 */
215#define CB_SOCKET_STAT_CD 0x06 /* card detect 1 and 2 */
216#define CB_SOCKET_STAT_PWRCYCLE 0x08 /* power cycle */
217#define CB_SOCKET_STAT_16BIT 0x010 /* 16-bit card */
218#define CB_SOCKET_STAT_CB 0x020 /* cardbus card */
219#define CB_SOCKET_STAT_IREQ 0x040 /* READY(~IREQ)//(~CINT) bit */
220#define CB_SOCKET_STAT_NOTCARD 0x080 /* Inserted card is unrecognisable */
221#define CB_SOCKET_STAT_DATALOST 0x0100 /* data lost */
222#define CB_SOCKET_STAT_BADVCC 0x0200 /* Bad Vcc Request */
223#define CB_SOCKET_STAT_5VCARD 0x0400 /* 5 V Card */
224#define CB_SOCKET_STAT_3VCARD 0x0800 /* 3.3 V Card */
225#define CB_SOCKET_STAT_XVCARD 0x01000 /* X.X V Card */
226#define CB_SOCKET_STAT_YVCARD 0x02000 /* Y.Y V Card */
227#define CB_SOCKET_STAT_5VSOCK 0x10000000 /* 5 V Socket */
228#define CB_SOCKET_STAT_3VSOCK 0x20000000 /* 3.3 V Socket */
229#define CB_SOCKET_STAT_XVSOCK 0x40000000 /* X.X V Socket */
230#define CB_SOCKET_STAT_YVSOCK 0x80000000 /* Y.Y V Socket */
