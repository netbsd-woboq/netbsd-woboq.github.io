33/*
34 * PCI Quirk data table and lookup function.
35 */
36
37#include <sys/cdefs.h>
38__KERNEL_RCSID(0, "$NetBSD: pci_quirks.c,v 1.12 2018/10/17 01:16:50 jmcneill Exp $");
39
40#include <sys/param.h>
41#include <sys/device.h>
42
43#include <dev/pci/pcireg.h>
44#include <dev/pci/pcivar.h>
45#include <dev/pci/pcidevs.h>
46
47static const struct pci_quirkdata pci_quirks[] = {
48 { PCI_VENDOR_INTEL, PCI_PRODUCT_INTEL_82371FB_ISA,
49 PCI_QUIRK_MULTIFUNCTION },
50 { PCI_VENDOR_NVIDIA, PCI_PRODUCT_NVIDIA_XBOX_PCHB,
51 PCI_QUIRK_SKIP_FUNC1 | PCI_QUIRK_SKIP_FUNC2 },
52 { PCI_VENDOR_INTEL, PCI_PRODUCT_INTEL_XEOND_MEM_0_TTR_1,
53 PCI_QUIRK_HASEXTCNF },
54 { PCI_VENDOR_INTEL, PCI_PRODUCT_INTEL_COREI76K_IMC_0,
55 PCI_QUIRK_HASEXTCNF },
56};
57
58const struct pci_quirkdata *
59pci_lookup_quirkdata(pci_vendor_id_t vendor, pci_product_id_t product)
60{
61 int i;
62
63 for (i = 0; i < __arraycount(pci_quirks); i++)
64 if (vendor == pci_quirks[i].vendor &&
65 product == pci_quirks[i].product)
66 return (&pci_quirks[i]);
67 return (NULL);
68}
