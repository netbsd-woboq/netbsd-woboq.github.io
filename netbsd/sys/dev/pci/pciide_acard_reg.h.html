27#define ATP850_IDETIME(channel) (0x40 + (channel) * 4)
28#define ATP860_IDETIME 0x40
29
30#define ATP850_SETTIME(drive, act, rec) \
31 (((((act) & 0xf) << 8) | ((rec) & 0xf)) << ((drive) * 16))
32#define ATP860_SETTIME(channel, drive, act, rec) \
33 (((((act) & 0xf) << 4) | ((rec) & 0xf)) << \
34 ((channel) * 16 + (drive) * 8))
35#define ATP860_SETTIME_MASK(channel) (0xffff << ((channel) * 16))
36
37static const u_int8_t acard_act_udma[] = {0x3, 0x3, 0x3, 0x3, 0x3, 0x3, 0x3};
38static const u_int8_t acard_rec_udma[] = {0x1, 0x1, 0x1, 0x1, 0x1, 0x1, 0x1};
39static const u_int8_t acard_act_dma[] = {0x0, 0x3, 0x3};
40static const u_int8_t acard_rec_dma[] = {0xa, 0x3, 0x1};
41static const u_int8_t acard_act_pio[] = {0x0, 0x0, 0x0, 0x3, 0x3};
42static const u_int8_t acard_rec_pio[] = {0x0, 0xa, 0x8, 0x3, 0x1};
43
44#define ATP850_UDMA 0x54
45#define ATP860_UDMA 0x44
46
47#define ATP850_UDMA_MODE(channel, drive, x) \
48 (((x) & 0x3) << ((channel) * 4 + (drive) * 2))
49#define ATP860_UDMA_MODE(channel, drive, x) \
50 (((x) & 0xf) << ((channel) * 8 + (drive) * 4))
51#define ATP850_UDMA_MASK(channel) (0xf << ((channel) * 4))
52#define ATP860_UDMA_MASK(channel) (0xff << ((channel) * 8))
53
54static const u_int8_t acard_udma_conf[] __unused =
55 {0x1, 0x2, 0x3, 0x4, 0x5, 0x6, 0x7};
