28#define CS5530_PIO_REG(chan, drv) (0x20 + (chan) * 0x10 + (drv) * 0x8)
29#define CS5530_DMA_REG(chan, drv) (0x24 + (chan) * 0x10 + (drv) * 0x8)
30#define CS5530_DMA_REG_PIO_FORMAT 0x80000000 /* select PIO format 1 */
31#define CS5530_DMA_REG_UDMA 0x00100000 /* enable Ultra-DMA */
32
33/*
34 * Recommended values from the cs5530 data sheet.
35 * Note that the udma values include CS5530_DMA_REG_UDMA.
36 * However, from the datasheet bits 30-21 should be reserved, yet
37 * geode_udma sets bit 23 to 1. I don't know if it's the definition of
38 * DMA_REG_UDMA which is wrong (bit 23 instead of 20) or these values.
39 */
40static const int32_t geode_cs5530_pio[] __unused =
41 {0x9172d132, 0x21717121, 0x00803020, 0x20102010, 0x00100010};
42static const int32_t geode_cs5530_dma[] __unused =
43 {0x00077771, 0x00012121, 0x00002020};
44static const int32_t geode_cs5530_udma[] __unused =
45 {0x00921250, 0x00911140, 0x00911030};
46
47#define SC1100_PIO_REG(chan, drv) (0x40 + (chan) * 0x10 + (drv) * 0x8)
48#define SC1100_DMA_REG(chan, drv) (0x44 + (chan) * 0x10 + (drv) * 0x8)
49
50/* Timings from FreeBSD */
51static const int32_t geode_sc1100_pio[] __unused =
52 {0x9172d132, 0x21717121, 0x00803020, 0x20102010, 0x00100010, 0x00803020,
53 0x20102010, 0x00100010, 0x00100010, 0x00100010, 0x00100010};
54static const int32_t geode_sc1100_dma[] __unused =
55 {0x80077771, 0x80012121, 0x80002020};
56static const int32_t geode_sc1100_udma[] __unused =
57 {0x80921250, 0x80911140, 0x80911030};
