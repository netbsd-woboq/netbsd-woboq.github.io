417 (((multi)?0x80:0) << PCI_HDRTYPE_SHIFT) | \
418 (((latency) & PCI_LATTIMER_MASK) << PCI_LATTIMER_SHIFT) | \
419 (((cacheline) & PCI_CACHELINE_MASK) << PCI_CACHELINE_SHIFT))
