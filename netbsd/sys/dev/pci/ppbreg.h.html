35/*
36 * Routine to translate between secondary bus interrupt pin/device number and
37 * primary bus interrupt pin number.
38 */
39#define PPB_INTERRUPT_SWIZZLE(pin, device) \
40 ((((pin) + (device) - 1) % 4) + 1)
