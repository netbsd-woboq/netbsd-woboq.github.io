32#ifndef _DEV_PCI_TWEIO_H_
33#define _DEV_PCI_TWEIO_H_
34
35#include <sys/ioccom.h>
36#include <dev/pci/twereg.h>
37
38/*
39 * User-space command
40 *
41 * Note that the command's scatter/gather list will be computed by the
42 * driver, and cannot be filled in by the consumer.
43 */
44struct twe_usercommand {
45 struct twe_cmd tu_cmd; /* command ready for the controller */
46 void *tu_data; /* pointer to data in userspace */
47 size_t tu_size; /* userspace data length */
48};
49
50#define TWEIO_COMMAND _IOWR('T', 100, struct twe_usercommand)
51
52/*
53 * Command queue statistics
54 */
55#define TWEQ_FREE 0
56#define TWEQ_BIO 1
57#define TWEQ_READY 2
58#define TWEQ_BUSY 3
59#define TWEQ_COMPLETE 4
60#define TWEQ_COUNT 5 /* total number of queues */
61
62struct twe_qstat {
63 u_int32_t q_length;
64 u_int32_t q_max;
65};
66
67/*
68 * Statistics request
69 */
70union twe_statrequest {
71 u_int32_t ts_item;
72 struct twe_qstat ts_qstat;
73};
74
75#define TWEIO_STATS _IOWR('T', 101, union twe_statrequest)
76
77/*
78 * AEN listen
79 */
80#define TWEIO_AEN_POLL _IOR('T', 102, int)
81#define TWEIO_AEN_WAIT _IOR('T', 103, int)
82
83/*
84 * Controller parameter access
85 */
86struct twe_paramcommand {
87 u_int16_t tp_table_id;
88 u_int8_t tp_param_id;
89 void *tp_data;
90 u_int8_t tp_size;
91};
92
93#define TWEIO_SET_PARAM _IOW('T', 104, struct twe_paramcommand)
94#define TWEIO_GET_PARAM _IOW('T', 105, struct twe_paramcommand)
95
96/*
97 * Request a controller soft-reset
98 */
99#define TWEIO_RESET _IO('T', 106)
100
101/*
102 * Request a drive addition or deletion
103 */
104struct twe_drivecommand {
105 int td_unit;
106};
107#define TWEIO_ADD_UNIT _IOW('U', 107, struct twe_drivecommand)
108#define TWEIO_DEL_UNIT _IOW('U', 108, struct twe_drivecommand)
109
110#endif /* _DEV_PCI_TWEIO_H_ */
