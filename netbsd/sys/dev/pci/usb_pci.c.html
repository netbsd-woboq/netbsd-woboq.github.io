52struct usb_pci_alldevs ehci_pci_alldevs =
53 TAILQ_HEAD_INITIALIZER(ehci_pci_alldevs);
54
55void
56usb_pci_add(struct usb_pci *up, struct pci_attach_args *pa, device_t bu)
57{
58 TAILQ_INSERT_TAIL(&ehci_pci_alldevs, up, next);
59 up->bus = pa->pa_bus;
60 up->device = pa->pa_device;
61 up->function = pa->pa_function;
62 up->usb = bu;
63}
64
65void
66usb_pci_rem(struct usb_pci *up)
67{
68 TAILQ_REMOVE(&ehci_pci_alldevs, up, next);
69}
