33/*
34 * EHCI controllers need a way to find their compainion controllers
35 * so we keep track of them as we attach.
36 */
37
38struct usb_pci {
39 TAILQ_ENTRY(usb_pci) next;
40 u_int bus;
41 u_int device;
42 u_int function;
43 device_t usb;
44 bool claimed;
45};
46
47TAILQ_HEAD(usb_pci_alldevs, usb_pci);
48extern struct usb_pci_alldevs ehci_pci_alldevs;
49

51void usb_pci_rem(struct usb_pci *);
