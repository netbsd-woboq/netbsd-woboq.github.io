31#include <sys/cdefs.h>
32__KERNEL_RCSID(0, "$NetBSD: wsdisplay_pci.c,v 1.1 2011/01/22 15:14:28 cegger Exp $");
33
34#include <sys/param.h>
35#include <sys/bus.h>
36#include <dev/pci/pcivar.h>
37#include <dev/wscons/wsconsio.h>
38#include <dev/pci/wsdisplay_pci.h>
39
40int
41wsdisplayio_busid_pci(device_t self, pci_chipset_tag_t pc,
42 pcitag_t tag, void *data)
43{
44 struct wsdisplayio_bus_id *busid = data;
45
46 KASSERT(device_is_a(device_parent(self), "pci"));
47 busid->bus_type = WSDISPLAYIO_BUS_PCI;
48 busid->ubus.pci.domain = device_unit(device_parent(self));
49 pci_decompose_tag(pc, tag,
50 &busid->ubus.pci.bus, &busid->ubus.pci.device,
51 &busid->ubus.pci.function);
52 return 0;
53}
