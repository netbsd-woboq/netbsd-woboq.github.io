61struct xhci_pci_softc {
62 struct xhci_softc sc_xhci;
63 pci_chipset_tag_t sc_pc;
64 pcitag_t sc_tag;
65 void *sc_ih;
66 pci_intr_handle_t *sc_pihp;
67};

69static int
70xhci_pci_match(device_t parent, cfdata_t match, void *aux)
71{
72 struct pci_attach_args *pa = (struct pci_attach_args *) aux;
73
74 if (PCI_CLASS(pa->pa_class) == PCI_CLASS_SERIALBUS &&
75 PCI_SUBCLASS(pa->pa_class) == PCI_SUBCLASS_SERIALBUS_USB &&
76 PCI_INTERFACE(pa->pa_class) == PCI_INTERFACE_XHCI)
77 return 1;
78
79 return 0;
80}

82static int
83xhci_pci_port_route(struct xhci_pci_softc *psc)
84{
85 struct xhci_softc * const sc = &psc->sc_xhci;
86
87 pcireg_t val;
88
89 /*
90 * Check USB3 Port Routing Mask register that indicates the ports
91 * can be changed from OS, and turn on by USB3 Port SS Enable register.
92 */
93 val = pci_conf_read(psc->sc_pc, psc->sc_tag, PCI_XHCI_INTEL_USB3PRM);
94 aprint_debug_dev(sc->sc_dev,
95 "USB3PRM / USB3.0 configurable ports: 0x%08x\n", val);
96
97 pci_conf_write(psc->sc_pc, psc->sc_tag, PCI_XHCI_INTEL_USB3_PSSEN, val);
98 val = pci_conf_read(psc->sc_pc, psc->sc_tag,PCI_XHCI_INTEL_USB3_PSSEN);
99 aprint_debug_dev(sc->sc_dev,
100 "USB3_PSSEN / Enabled USB3.0 ports under xHCI: 0x%08x\n", val);
101
102 /*
103 * Check USB2 Port Routing Mask register that indicates the USB2.0
104 * ports to be controlled by xHCI HC, and switch them to xHCI HC.
105 */
106 val = pci_conf_read(psc->sc_pc, psc->sc_tag, PCI_XHCI_INTEL_USB2PRM);
107 aprint_debug_dev(sc->sc_dev,
108 "XUSB2PRM / USB2.0 ports can switch from EHCI to xHCI:"
109 "0x%08x\n", val);
110 pci_conf_write(psc->sc_pc, psc->sc_tag, PCI_XHCI_INTEL_XUSB2PR, val);
111 val = pci_conf_read(psc->sc_pc, psc->sc_tag, PCI_XHCI_INTEL_XUSB2PR);
112 aprint_debug_dev(sc->sc_dev,
113 "XUSB2PR / USB2.0 ports under xHCI: 0x%08x\n", val);
114
115 return 0;
116}

118static void
119xhci_pci_attach(device_t parent, device_t self, void *aux)
120{
121 struct xhci_pci_softc * const psc = device_private(self);
122 struct xhci_softc * const sc = &psc->sc_xhci;
123 struct pci_attach_args *const pa = (struct pci_attach_args *)aux;
124 const pci_chipset_tag_t pc = pa->pa_pc;
125 const pcitag_t tag = pa->pa_tag;
126 char const *intrstr;
127 pcireg_t csr, memtype, usbrev;
128 uint32_t hccparams;
129 char intrbuf[PCI_INTRSTR_LEN];
130 bus_addr_t memaddr;
131 int flags, msixoff;
132 int err;
133
134 sc->sc_dev = self;
135
136 pci_aprint_devinfo(pa, "USB Controller");
137
138 /* Check for quirks */
139 sc->sc_quirks = 0;
140
141 csr = pci_conf_read(pc, tag, PCI_COMMAND_STATUS_REG);
142 if ((csr & PCI_COMMAND_MEM_ENABLE) == 0) {
143 /*
144 * Enable address decoding for memory range in case BIOS or
145 * UEFI didn't set it.
146 */
147 csr = pci_conf_read(pa->pa_pc, pa->pa_tag,
148 PCI_COMMAND_STATUS_REG);
149 csr |= PCI_COMMAND_MEM_ENABLE;
150 pci_conf_write(pa->pa_pc, pa->pa_tag, PCI_COMMAND_STATUS_REG,
151 csr);
152 }
153
154 /* map MMIO registers */
155 memtype = pci_mapreg_type(pa->pa_pc, pa->pa_tag, PCI_CBMEM);
156 if (PCI_MAPREG_TYPE(memtype) != PCI_MAPREG_TYPE_MEM) {
157 sc->sc_ios = 0;
158 aprint_error_dev(self, "BAR not 64 or 32-bit MMIO\n");
159 return;
160 }
161
162 sc->sc_iot = pa->pa_memt;
163 if (pci_mapreg_info(pa->pa_pc, pa->pa_tag, PCI_CBMEM, memtype,
164 &memaddr, &sc->sc_ios, &flags) != 0) {
165 sc->sc_ios = 0;
166 aprint_error_dev(self, "can't get map info\n");
167 return;
168 }
169
170 if (pci_get_capability(pa->pa_pc, pa->pa_tag, PCI_CAP_MSIX, &msixoff,
171 NULL)) {
172 pcireg_t msixtbl;
173 uint32_t table_offset;
174 int bir;
175
176 msixtbl = pci_conf_read(pa->pa_pc, pa->pa_tag,
177 msixoff + PCI_MSIX_TBLOFFSET);
178 table_offset = msixtbl & PCI_MSIX_TBLOFFSET_MASK;
179 bir = msixtbl & PCI_MSIX_PBABIR_MASK;
180 /* Shrink map area for MSI-X table */
181 if (bir == PCI_MAPREG_NUM(PCI_CBMEM))
182 sc->sc_ios = table_offset;
183 }
184 if (bus_space_map(sc->sc_iot, memaddr, sc->sc_ios, flags,
185 &sc->sc_ioh)) {
186 sc->sc_ios = 0;
187 aprint_error_dev(self, "can't map mem space\n");
188 return;
189 }
190
191 psc->sc_pc = pc;
192 psc->sc_tag = tag;
193
194 hccparams = bus_space_read_4(sc->sc_iot, sc->sc_ioh, XHCI_HCCPARAMS);
195
196 if (pci_dma64_available(pa) && (XHCI_HCC_AC64(hccparams) != 0))
197 sc->sc_bus.ub_dmatag = pa->pa_dmat64;
198 else
199 sc->sc_bus.ub_dmatag = pa->pa_dmat;
200
201 /* Enable the device. */
202 pci_conf_write(pc, tag, PCI_COMMAND_STATUS_REG,
203 csr | PCI_COMMAND_MASTER_ENABLE);
204
205 /* Allocate and establish the interrupt. */
206 if (pci_intr_alloc(pa, &psc->sc_pihp, NULL, 0)) {
207 aprint_error_dev(self, "can't allocate handler\n");
208 goto fail;
209 }
210 intrstr = pci_intr_string(pc, psc->sc_pihp[0], intrbuf,
211 sizeof(intrbuf));
212 psc->sc_ih = pci_intr_establish_xname(pc, psc->sc_pihp[0], IPL_USB,
213 xhci_intr, sc, device_xname(sc->sc_dev));
214 if (psc->sc_ih == NULL) {
215 pci_intr_release(pc, psc->sc_pihp, 1);
216 psc->sc_ih = NULL;
217 aprint_error_dev(self, "couldn't establish interrupt");
218 if (intrstr != NULL)
219 aprint_error(" at %s", intrstr);
220 aprint_error("\n");
221 goto fail;
222 }
223 aprint_normal_dev(self, "interrupting at %s\n", intrstr);
224
225 usbrev = pci_conf_read(pc, tag, PCI_USBREV) & PCI_USBREV_MASK;
226 switch (usbrev) {
227 case PCI_USBREV_3_0:
228 sc->sc_bus.ub_revision = USBREV_3_0;
229 break;
230 case PCI_USBREV_3_1:
231 sc->sc_bus.ub_revision = USBREV_3_1;
232 break;
233 default:
234 if (usbrev < PCI_USBREV_3_0) {
235 aprint_error_dev(self, "Unknown revision (%02x). Set to 3.0.\n",
236 usbrev);
237 sc->sc_bus.ub_revision = USBREV_3_0;
238 } else {
239 /* Default to the latest revision */
240 aprint_normal_dev(self,
241 "Unknown revision (%02x). Set to 3.1.\n", usbrev);
242 sc->sc_bus.ub_revision = USBREV_3_1;
243 }
244 break;
245 }
246
247 /* Intel chipset requires SuperSpeed enable and USB2 port routing */
248 switch (PCI_VENDOR(pa->pa_id)) {
249 case PCI_VENDOR_INTEL:
250 sc->sc_quirks |= XHCI_QUIRK_INTEL;
251 break;
252 default:
253 break;
254 }
255
256 err = xhci_init(sc);
257 if (err) {
258 aprint_error_dev(self, "init failed, error=%d\n", err);
259 goto fail;
260 }
261
262 if ((sc->sc_quirks & XHCI_QUIRK_INTEL) != 0)
263 xhci_pci_port_route(psc);
264
265 if (!pmf_device_register1(self, xhci_suspend, xhci_resume,
266 xhci_shutdown))
267 aprint_error_dev(self, "couldn't establish power handler\n");
268
269 /* Attach usb buses. */
270 sc->sc_child = config_found(self, &sc->sc_bus, usbctlprint);
271
272 sc->sc_child2 = config_found(self, &sc->sc_bus2, usbctlprint);
273
274 return;
275
276fail:
277 if (psc->sc_ih != NULL) {
278 pci_intr_disestablish(psc->sc_pc, psc->sc_ih);
279 psc->sc_ih = NULL;
280 }
281 if (psc->sc_pihp != NULL) {
282 pci_intr_release(psc->sc_pc, psc->sc_pihp, 1);
283 psc->sc_pihp = NULL;
284 }
285 if (sc->sc_ios) {
286 bus_space_unmap(sc->sc_iot, sc->sc_ioh, sc->sc_ios);
287 sc->sc_ios = 0;
288 }
289 return;
290}

292static int
293xhci_pci_detach(device_t self, int flags)
294{
295 struct xhci_pci_softc * const psc = device_private(self);
296 struct xhci_softc * const sc = &psc->sc_xhci;
297 int rv;
298
299 if (sc->sc_ios != 0) {
300 rv = xhci_detach(sc, flags);
301 if (rv)
302 return rv;
303
304 pmf_device_deregister(self);
305
306 xhci_shutdown(self, flags);
307
308#if 0
309 /* Disable interrupts, so we don't get any spurious ones. */
310 bus_space_write_4(sc->sc_iot, sc->sc_ioh,
311 OHCI_INTERRUPT_DISABLE, OHCI_ALL_INTRS);
312#endif
313 }
314
315 if (psc->sc_ih != NULL) {
316 pci_intr_disestablish(psc->sc_pc, psc->sc_ih);
317 psc->sc_ih = NULL;
318 }
319 if (psc->sc_pihp != NULL) {
320 pci_intr_release(psc->sc_pc, psc->sc_pihp, 1);
321 psc->sc_pihp = NULL;
322 }
323 if (sc->sc_ios) {
324 bus_space_unmap(sc->sc_iot, sc->sc_ioh, sc->sc_ios);
325 sc->sc_ios = 0;
326 }
327
328 return 0;
329}
