29#ifndef _DEV_PCKBCPORT_ELANTECHVAR_H
30#define _DEV_PCKBCPORT_ELANTECHVAR_H
31
32struct elantech_softc {
33 uint32_t version;
34
35 bool initializing;
36 int16_t last_x, last_y, last_z;
37 int last_nfingers;
38};
39
40/* minimum versions supported by this driver */
41#define ELANTECH_MIN_VERSION 0x0230
42
43#define ELANTECH_MAGIC(magic) \
44 ((magic)[0] == 0x3c && (magic)[1] == 0x03 && (magic)[2] == 0xc8)
45
46int pms_elantech_probe_init(void *);
47void pms_elantech_enable(void *);
48void pms_elantech_resume(void *);
49
50#endif /* !_DEV_PCKBCPORT_ELANTECHVAR_H */
