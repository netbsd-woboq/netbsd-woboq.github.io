38#ifndef _DEV_PCKBCPORT_SYNAPTICSVAR_H_
39#define _DEV_PCKBCPORT_SYNAPTICSVAR_H_
40
41struct synaptics_softc {
42 int caps;
43
44 int flags;
45#define SYN_FLAG_HAS_MIDDLE_BUTTON (1 << 0)
46#define SYN_FLAG_HAS_BUTTONS_4_5 (1 << 1)
47#define SYN_FLAG_HAS_UP_DOWN_BUTTONS (1 << 2)
48#define SYN_FLAG_HAS_PASSTHROUGH (1 << 3)
49#define SYN_FLAG_HAS_PALM_DETECT (1 << 4)
50#define SYN_FLAG_HAS_MULTI_FINGER (1 << 5)
51#define SYN_FLAG_HAS_MULTI_FINGER_REPORT (1 << 6)
52#define SYN_FLAG_HAS_VERTICAL_SCROLL (1 << 7)
53#define SYN_FLAG_HAS_HORIZONTAL_SCROLL (1 << 8)
54#define SYN_FLAG_HAS_ONE_BUTTON_CLICKPAD (1 << 9)
55#define SYN_FLAG_HAS_TWO_BUTTON_CLICKPAD (1 << 10)
56#define SYN_FLAG_HAS_EXTENDED_WMODE (1 << 11)
57#define SYN_FLAG_HAS_ADV_GESTURE_MODE (1 << 12)
58
59 u_int total_packets[2]; /* Total number of packets received */
60#define SYN_TIME(sc,c,n) (((sc)->total_packets[(n)] >= (c)) ? \
61 ((sc)->total_packets[(n)] - (c)) : \
62 ((c) - (sc)->total_packets[(n)]))
63
64 int up_down;
65 int prev_fingers;
66
67 int gesture_start_x, gesture_start_y;
68 int gesture_move_x, gesture_move_y;
69 u_int gesture_start_packet;
70 u_int gesture_tap_packet;
71
72 int gesture_buttons;
73 int gesture_type;
74#define SYN_GESTURE_SINGLE 0x01
75#define SYN_GESTURE_DOUBLE 0x02
76#define SYN_GESTURE_DRAG 0x04
77#define SYN_IS_SINGLE_TAP(t) ((t) & SYN_GESTURE_SINGLE)
78#define SYN_IS_DOUBLE_TAP(t) ((t) & SYN_GESTURE_DOUBLE)
79#define SYN_IS_DRAG(t) ((t) & SYN_GESTURE_DRAG)
80
81#define SYN_HIST_SIZE 4
82#define SYN_MAX_FINGERS 2
83 char button_history;
84 int dz_hold;
85 int rem_x[SYN_MAX_FINGERS];
86 int rem_y[SYN_MAX_FINGERS];
87 int rem_z[SYN_MAX_FINGERS];
88 u_int movement_history[SYN_MAX_FINGERS];
89 int history_x[SYN_MAX_FINGERS][SYN_HIST_SIZE];
90 int history_y[SYN_MAX_FINGERS][SYN_HIST_SIZE];
91 int history_z[SYN_MAX_FINGERS][SYN_HIST_SIZE];
92};
93
94int pms_synaptics_probe_init(void *vsc);
95void pms_synaptics_enable(void *vsc);
96void pms_synaptics_resume(void *vsc);
97
98#endif
