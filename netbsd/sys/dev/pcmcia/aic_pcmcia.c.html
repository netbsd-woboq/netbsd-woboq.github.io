54struct aic_pcmcia_softc {
55 struct aic_softc sc_aic; /* real "aic" softc */
56
57 /* PCMCIA-specific goo. */
58 struct pcmcia_function *sc_pf; /* our PCMCIA function */
59 void *sc_ih; /* interrupt handler */
60
61 int sc_state;
62#define AIC_PCMCIA_ATTACHED 3
63};

68static int aic_pcmcia_detach(device_t, int);

72 aic_pcmcia_match, aic_pcmcia_attach, aic_pcmcia_detach, NULL);

74static const struct pcmcia_product aic_pcmcia_products[] = {
75 { PCMCIA_VENDOR_ADAPTEC, PCMCIA_PRODUCT_ADAPTEC_APA1460,
76 PCMCIA_CIS_INVALID },
77
78 { PCMCIA_VENDOR_ADAPTEC, PCMCIA_PRODUCT_ADAPTEC_APA1460A,
79 PCMCIA_CIS_INVALID },
80
81 { PCMCIA_VENDOR_NEWMEDIA, PCMCIA_PRODUCT_NEWMEDIA_BUSTOASTER,
82 PCMCIA_CIS_INVALID },
83};
84static const size_t aic_pcmcia_nproducts = __arraycount(aic_pcmcia_products);
85
86int
87aic_pcmcia_match(device_t parent, cfdata_t match, void *aux)
88{
89 struct pcmcia_attach_args *pa = aux;
90
91 if (pcmcia_product_lookup(pa, aic_pcmcia_products, aic_pcmcia_nproducts,
92 sizeof(aic_pcmcia_products[0]), NULL))
93 return (1);
94 return (0);
95}

97int
98aic_pcmcia_validate_config(struct pcmcia_config_entry *cfe)
99{
100 if (cfe->iftype != PCMCIA_IFTYPE_IO ||
101 cfe->num_memspace != 0 ||
102 cfe->num_iospace != 1)
103 return (EINVAL);
104
105 cfe->iomask = 0; /* XXX: wrong from cfe?, quirk? */
106 return (0);
107}
108
109void
110aic_pcmcia_attach(device_t parent, device_t self, void *aux)
111{
112 struct aic_pcmcia_softc *psc = device_private(self);
113 struct aic_softc *sc = &psc->sc_aic;
114 struct pcmcia_attach_args *pa = aux;
115 struct pcmcia_config_entry *cfe;
116 struct pcmcia_function *pf = pa->pf;
117 int error;
118
119 sc->sc_dev = self;
120 psc->sc_pf = pf;
121
122 error = pcmcia_function_configure(pf, aic_pcmcia_validate_config);
123 if (error) {
124 aprint_error_dev(self, "configure failed, error=%d\n", error);
125 return;
126 }
127
128 cfe = pf->cfe;
129 sc->sc_iot = cfe->iospace[0].handle.iot;
130 sc->sc_ioh = cfe->iospace[0].handle.ioh;
131
132 error = aic_pcmcia_enable(self, 1);
133 if (error)
134 goto fail;
135
136 if (!aic_find(sc->sc_iot, sc->sc_ioh)) {
137 aprint_error_dev(self, "unable to detect chip!\n");
138 goto fail2;
139 }
140
141 /* We can enable and disable the controller. */
142 sc->sc_adapter.adapt_enable = aic_pcmcia_enable;
143 sc->sc_adapter.adapt_refcnt = 1;
144
145 aicattach(sc);
146 scsipi_adapter_delref(&sc->sc_adapter);
147 psc->sc_state = AIC_PCMCIA_ATTACHED;
148 return;
149
150fail2:
151 aic_pcmcia_enable(self, 0);
152fail:
153 pcmcia_function_unconfigure(pf);
154}
155
156int
157aic_pcmcia_detach(device_t self, int flags)
158{
159 struct aic_pcmcia_softc *psc = device_private(self);
160 int error;
161
162 if (psc->sc_state != AIC_PCMCIA_ATTACHED)
163 return (0);
164
165 error = aic_detach(self, flags);
166 if (error)
167 return (error);
168
169 pcmcia_function_unconfigure(psc->sc_pf);
170
171 return (0);
172}
173
174int
175aic_pcmcia_enable(device_t self, int onoff)
176{
177 struct aic_pcmcia_softc *psc = device_private(self);
178 struct aic_softc *sc = &psc->sc_aic;
179 int error;
180
181 if (onoff) {
182 /* Establish the interrupt handler. */
183 psc->sc_ih = pcmcia_intr_establish(psc->sc_pf, IPL_BIO,
184 aicintr, sc);
185 if (!psc->sc_ih)
186 return (EIO);
187
188 error = pcmcia_function_enable(psc->sc_pf);
189 if (error) {
190 pcmcia_intr_disestablish(psc->sc_pf, psc->sc_ih);
191 psc->sc_ih = 0;
192 return (error);
193 }
194
195 /* Initialize only chip. */
196 aic_init(sc, 0);
197 } else {
198 pcmcia_function_disable(psc->sc_pf);
199 pcmcia_intr_disestablish(psc->sc_pf, psc->sc_ih);
200 psc->sc_ih = 0;
201 }
202
203 return (0);
204}
