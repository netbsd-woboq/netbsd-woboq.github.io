38struct pud_touser {

43};

45struct pud_dev {
46 dev_t pd_dev;

48
49 kmutex_t pd_mtx;
50
51 TAILQ_HEAD(,pud_touser) pd_waitq_req;

53 kcondvar_t pd_waitq_req_cv;
54 size_t pd_waitcount;
55
56 kcondvar_t pd_draincv;
57
58 LIST_ENTRY(pud_dev) pd_entries;
59
60 uint64_t pd_nextreq;
61};
