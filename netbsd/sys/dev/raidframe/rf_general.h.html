35#ifndef _RF__RF_GENERAL_H_
36#define _RF__RF_GENERAL_H_
37
38#ifdef _KERNEL_OPT
39#include "opt_raid_diagnostic.h"
40#endif /* _KERNEL_OPT */
41
42/* error reporting and handling */
43
44#include <sys/systm.h> /* printf, snprintf, and friends */
45
46#define RF_ERRORMSG(s) printf((s))
47#define RF_ERRORMSG1(s,a) printf((s),(a))
48#define RF_ERRORMSG2(s,a,b) printf((s),(a),(b))
49#define RF_ERRORMSG3(s,a,b,c) printf((s),(a),(b),(c))
50
51void rf_print_panic_message(int, const char *);
52void rf_print_assert_panic_message(int, const char *, const char *);
53void rf_print_unable_to_init_mutex(const char *, int, int);
54void rf_print_unable_to_add_shutdown(const char *, int, int);
55
56
57#define RF_PANIC() {rf_print_panic_message(__LINE__,__FILE__);}
58
59#if defined(RAID_DIAGNOSTIC) || defined(__COVERITY__)
60#define RF_ASSERT(_x_) { \
61 if (!(_x_)) { \
62 rf_print_assert_panic_message(__LINE__, __FILE__, #_x_); \
63 } \
64}
65#else /* RAID_DIAGNOSTIC */
66#define RF_ASSERT(x) { /*noop*/ (void)(x); }
67#endif /* RAID_DIAGNOSTIC */
68
69/* random stuff */
70#define RF_MAX(a,b) (((a) > (b)) ? (a) : (b))
71#define RF_MIN(a,b) (((a) < (b)) ? (a) : (b))
72
73/* divide-by-zero check */
74#define RF_DB0_CHECK(a,b) ( ((b)==0) ? 0 : (a)/(b) )
75
76/* get time of day */
77#define RF_GETTIME(_t) microtime(&(_t))
78
79#define RF_UL(x) ((unsigned long) (x))
80#define RF_PGMASK PAGE_MASK
81#define RF_BLIP(x) (PAGE_SIZE - (RF_UL(x) & RF_PGMASK)) /* bytes left in page */
82#define RF_PAGE_ALIGNED(x) ((RF_UL(x) & RF_PGMASK) == 0)
83
84#ifdef __STDC__
85#define RF_STRING(_str_) #_str_
86#else /* __STDC__ */
87#define RF_STRING(_str_) "_str_"
88#endif /* __STDC__ */
89
90#endif /* !_RF__RF_GENERAL_H_ */
