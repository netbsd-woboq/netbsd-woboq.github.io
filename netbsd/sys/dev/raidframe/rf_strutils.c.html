45/* finds a non-white character in the line */
46char *
47rf_find_non_white(char *p)
48{
49 for (; *p != '\0' && (*p == ' ' || *p == '\t'); p++);
50 return (p);
51}
52/* finds a white character in the line */
53char *
54rf_find_white(char *p)
55{
56 for (; *p != '\0' && (*p != ' ' && *p != '\t'); p++);
57 return (p);
58}
