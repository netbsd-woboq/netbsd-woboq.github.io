53#define RF_USEC_PER_SEC 1000000
54#define RF_TIMEVAL_TO_US(_t_) (((_t_).tv_sec) \
55 * RF_USEC_PER_SEC + (_t_).tv_usec)
56
57#define RF_TIMEVAL_DIFF(_start_,_end_,_diff_) { \
58 if ((_end_)->tv_usec < (_start_)->tv_usec) { \
59 (_diff_)->tv_usec = ((_end_)->tv_usec + RF_USEC_PER_SEC) \
60 - (_start_)->tv_usec; \
61 (_diff_)->tv_sec = ((_end_)->tv_sec-1) - (_start_)->tv_sec; \
62 } \
63 else { \
64 (_diff_)->tv_usec = (_end_)->tv_usec - (_start_)->tv_usec; \
65 (_diff_)->tv_sec = (_end_)->tv_sec - (_start_)->tv_sec; \
66 } \
67}
