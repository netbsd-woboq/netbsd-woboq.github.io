155/*
156 * Generate a 32-bit counter.
157 */
158static inline uint32_t
159rndpseudo_counter(void)
160{
161 struct bintime bt;
162 uint32_t ret;
163
164#if defined(__HAVE_CPU_COUNTER)
165 if (cpu_hascounter())
166 return (cpu_counter32());
167#endif
168
169 binuptime(&bt);
170 ret = bt.sec;
171 ret ^= bt.sec >> 32;
172 ret ^= bt.frac;
173 ret ^= bt.frac >> 32;
174
175 return ret;
176}

178/*
179 * Used by ioconf.c to attach the rnd pseudo-device.
180 * `Attach' the random device. We use the timing of this event as
181 * another potential source of initial entropy.
182 */
183void
184rndattach(int num)
185{
186 uint32_t c;
187
188 /* Trap unwary players who don't call rnd_init() early. */
189 KASSERT(rnd_ready);
190
191 rnd_temp_buffer_cache = pool_cache_init(RND_TEMP_BUFFER_SIZE, 0, 0, 0,
192 "rndtemp", NULL, IPL_NONE, NULL, NULL, NULL);
193 rnd_ctx_cache = pool_cache_init(sizeof(struct rnd_ctx), 0, 0, 0,
194 "rndctx", NULL, IPL_NONE, NULL, NULL, NULL);
195 percpu_urandom_cprng = percpu_alloc(sizeof(struct cprng_strong *));
196
197 /* Mix in another counter. */
198 c = rndpseudo_counter();
199 rnd_add_data(NULL, &c, sizeof(c), 1);
200}

202int
203rndopen(dev_t dev, int flags, int fmt, struct lwp *l)
204{
205 bool hard;
206 struct file *fp;
207 int fd;
208 int error;
209
210 switch (minor(dev)) {
211 case RND_DEV_URANDOM:
212 hard = false;
213 rndpseudo_soft.ev_count++;
214 break;
215
216 case RND_DEV_RANDOM:
217 hard = true;
218 rndpseudo_hard.ev_count++;
219 break;
220
221 default:
222 return ENXIO;
223 }
224
225 error = fd_allocfile(&fp, &fd);
226 if (error)
227 return error;
228
229 /*
230 * Allocate a context, but don't create a CPRNG yet -- do that
231 * lazily because it consumes entropy from the system entropy
232 * pool, which (currently) has the effect of depleting it and
233 * causing readers from /dev/random to block. If this is
234 * /dev/urandom and the process is about to send only short
235 * reads to it, then we will be using a per-CPU CPRNG anyway.
236 */
237 struct rnd_ctx *const ctx = pool_cache_get(rnd_ctx_cache, PR_WAITOK);
238 ctx->rc_cprng = NULL;
239 ctx->rc_hard = hard;
240
241 error = fd_clone(fp, fd, flags, &rnd_fileops, ctx);
242 KASSERT(error == EMOVEFD);
243
244 return error;
245}

247/*
248 * Fetch a /dev/u?random context's CPRNG, or create and save one if
249 * necessary.
250 */
251static struct cprng_strong *
252rnd_ctx_cprng(struct rnd_ctx *ctx)
253{
254 struct cprng_strong *cprng, *tmp = NULL;
255
256 /* Fast path: if someone has already allocated a CPRNG, use it. */
257 cprng = ctx->rc_cprng;
258 if (__predict_true(cprng != NULL)) {
259 /* Make sure the CPU hasn't prefetched cprng's guts. */
260 membar_consumer();
261 goto out;
262 }
263
264 /* Slow path: create a CPRNG. Allocate before taking locks. */
265 char name[64];
266 struct lwp *const l = curlwp;
267 (void)snprintf(name, sizeof(name), "%d %"PRIu64" %u",
268 (int)l->l_proc->p_pid, l->l_ncsw, l->l_cpticks);
269 const int flags = (ctx->rc_hard? (CPRNG_USE_CV | CPRNG_HARD) :
270 (CPRNG_INIT_ANY | CPRNG_REKEY_ANY));
271 tmp = cprng_strong_create(name, IPL_NONE, flags);
272
273 /* Publish cprng's guts before the pointer to them. */
274 membar_producer();
275
276 /* Attempt to publish tmp, unless someone beat us. */
277 cprng = atomic_cas_ptr(&ctx->rc_cprng, NULL, tmp);
278 if (__predict_false(cprng != NULL)) {
279 /* Make sure the CPU hasn't prefetched cprng's guts. */
280 membar_consumer();
281 goto out;
282 }
283
284 /* Published. Commit tmp. */
285 cprng = tmp;
286 tmp = NULL;
287
288out: if (tmp != NULL)
289 cprng_strong_destroy(tmp);
290 KASSERT(cprng != NULL);
291 return cprng;
292}

294/*
295 * Fetch a per-CPU CPRNG, or create and save one if necessary.
296 */
297static struct cprng_strong *
298rnd_percpu_cprng(void)
299{
300 struct cprng_strong **cprngp, *cprng, *tmp = NULL;
301
302 /* Fast path: if there already is a CPRNG for this CPU, use it. */
303 cprngp = percpu_getref(percpu_urandom_cprng);
304 cprng = *cprngp;
305 if (__predict_true(cprng != NULL))
306 goto out;
307 percpu_putref(percpu_urandom_cprng);
308
309 /*
310 * Slow path: create a CPRNG named by this CPU.
311 *
312 * XXX The CPU of the name may be different from the CPU to
313 * which it is assigned, because we need to choose a name and
314 * allocate a cprng while preemption is enabled. This could be
315 * fixed by changing the cprng_strong API (e.g., by adding a
316 * cprng_strong_setname or by separating allocation from
317 * initialization), but it's not clear that's worth the
318 * trouble.
319 */
320 char name[32];
321 (void)snprintf(name, sizeof(name), "urandom%u", cpu_index(curcpu()));
322 tmp = cprng_strong_create(name, IPL_NONE,
323 (CPRNG_INIT_ANY | CPRNG_REKEY_ANY));
324
325 /* Try again, but we may have been preempted and lost a race. */
326 cprngp = percpu_getref(percpu_urandom_cprng);
327 cprng = *cprngp;
328 if (__predict_false(cprng != NULL))
329 goto out;
330
331 /* Commit the CPRNG we just created. */
332 cprng = tmp;
333 tmp = NULL;
334 *cprngp = cprng;
335
336out: percpu_putref(percpu_urandom_cprng);
337 if (tmp != NULL)
338 cprng_strong_destroy(tmp);
339 KASSERT(cprng != NULL);
340 return cprng;
341}

343static int
344rnd_read(struct file *fp, off_t *offp, struct uio *uio, kauth_cred_t cred,
345 int flags)
346{
347 int error = 0;
348
349 DPRINTF(RND_DEBUG_READ,
350 ("Random: Read of %zu requested, flags 0x%08x\n",
351 uio->uio_resid, flags));
352
353 if (uio->uio_resid == 0)
354 return 0;
355
356 struct rnd_ctx *const ctx = fp->f_rndctx;
357 uint8_t *const buf = pool_cache_get(rnd_temp_buffer_cache, PR_WAITOK);
358
359 /*
360 * Choose a CPRNG to use -- either the per-open CPRNG, if this
361 * is /dev/random or a long read, or the per-CPU one otherwise.
362 *
363 * XXX NIST_BLOCK_KEYLEN_BYTES is a detail of the cprng(9)
364 * implementation and as such should not be mentioned here.
365 */
366 struct cprng_strong *const cprng =
367 ((ctx->rc_hard || (uio->uio_resid > NIST_BLOCK_KEYLEN_BYTES))?
368 rnd_ctx_cprng(ctx) : rnd_percpu_cprng());
369
370 /*
371 * Generate the data in RND_TEMP_BUFFER_SIZE chunks.
372 */
373 while (uio->uio_resid > 0) {
374 const size_t n_req = MIN(uio->uio_resid, RND_TEMP_BUFFER_SIZE);
375
376 CTASSERT(RND_TEMP_BUFFER_SIZE <= CPRNG_MAX_LEN);
377 const size_t n_read = cprng_strong(cprng, buf, n_req,
378 ((ctx->rc_hard && ISSET(fp->f_flag, FNONBLOCK))?
379 FNONBLOCK : 0));
380
381 /*
382 * Equality will hold unless this is /dev/random, in
383 * which case we get only as many bytes as are left
384 * from the CPRNG's `information-theoretic strength'
385 * since the last rekey.
386 */
387 KASSERT(n_read <= n_req);
388 KASSERT(ctx->rc_hard || (n_read == n_req));
389
390 error = uiomove(buf, n_read, uio);
391 if (error)
392 goto out;
393
394 /*
395 * For /dev/urandom: Reads always succeed in full, no
396 * matter how many iterations that takes. (XXX But
397 * this means the computation can't be interrupted,
398 * wihch seems suboptimal.)
399 *
400 * For /dev/random, nonblocking: Reads succeed with as
401 * many bytes as a single request can return without
402 * blocking, or fail with EAGAIN if a request would
403 * block. (There is no sense in trying multiple
404 * requests because if the first one didn't fill the
405 * buffer, the second one would almost certainly
406 * block.)
407 *
408 * For /dev/random, blocking: Reads succeed with as
409 * many bytes as a single request -- which may block --
410 * can return if uninterrupted, or fail with EINTR if
411 * the request is interrupted.
412 */
413 KASSERT((0 < n_read) || ctx->rc_hard);
414 if (ctx->rc_hard) {
415 if (0 < n_read)
416 error = 0;
417 else if (ISSET(fp->f_flag, FNONBLOCK))
418 error = EAGAIN;
419 else
420 error = EINTR;
421 goto out;
422 }
423 }
424
425out: pool_cache_put(rnd_temp_buffer_cache, buf);
426 return error;
427}

429static int
430rnd_write(struct file *fp, off_t *offp, struct uio *uio,
431 kauth_cred_t cred, int flags)
432{
433 uint8_t *bf;
434 int n, ret = 0, estimate_ok = 0, estimate = 0, added = 0;
435
436 ret = kauth_authorize_device(cred,
437 KAUTH_DEVICE_RND_ADDDATA, NULL, NULL, NULL, NULL);
438 if (ret) {
439 return (ret);
440 }
441 estimate_ok = !kauth_authorize_device(cred,
442 KAUTH_DEVICE_RND_ADDDATA_ESTIMATE, NULL, NULL, NULL, NULL);
443
444 DPRINTF(RND_DEBUG_WRITE,
445 ("Random: Write of %zu requested\n", uio->uio_resid));
446
447 if (uio->uio_resid == 0)
448 return (0);
449 ret = 0;
450 bf = pool_cache_get(rnd_temp_buffer_cache, PR_WAITOK);
451 while (uio->uio_resid > 0) {
452 /*
453 * Don't flood the pool.
454 */
455 if (added > RND_POOLWORDS * sizeof(int)) {
456#ifdef RND_VERBOSE
457 printf("rnd: added %d already, adding no more.\n",
458 added);
459#endif
460 break;
461 }
462 n = uimin(RND_TEMP_BUFFER_SIZE, uio->uio_resid);
463
464 ret = uiomove((void *)bf, n, uio);
465 if (ret != 0)
466 break;
467
468 if (estimate_ok) {
469 /*
470 * Don't cause samples to be discarded by taking
471 * the pool's entropy estimate to the max.
472 */
473 if (added > RND_POOLWORDS / 2)
474 estimate = 0;
475 else
476 estimate = n * NBBY / 2;
477#ifdef RND_VERBOSE
478 printf("rnd: adding on write, %d bytes, estimate %d\n",
479 n, estimate);
480#endif
481 } else {
482#ifdef RND_VERBOSE
483 printf("rnd: kauth says no entropy.\n");
484#endif
485 }
486
487 /*
488 * Mix in the bytes.
489 */
490 rnd_add_data(NULL, bf, n, estimate);
491
492 added += n;
493 DPRINTF(RND_DEBUG_WRITE, ("Random: Copied in %d bytes\n", n));
494 }
495 pool_cache_put(rnd_temp_buffer_cache, bf);
496 return (ret);
497}

512static int
513rnd_poll(struct file *fp, int events)
514{
515 struct rnd_ctx *const ctx = fp->f_rndctx;
516 int revents;
517
518 /*
519 * We are always writable.
520 */
521 revents = events & (POLLOUT | POLLWRNORM);
522
523 /*
524 * Save some work if not checking for reads.
525 */
526 if ((events & (POLLIN | POLLRDNORM)) == 0)
527 return revents;
528
529 /*
530 * For /dev/random, ask the CPRNG, which may require creating
531 * one. For /dev/urandom, we're always readable.
532 */
533 if (ctx->rc_hard)
534 revents |= cprng_strong_poll(rnd_ctx_cprng(ctx), events);
535 else
536 revents |= (events & (POLLIN | POLLRDNORM));
537
538 return revents;
539}

541static int
542rnd_stat(struct file *fp, struct stat *st)
543{
544 struct rnd_ctx *const ctx = fp->f_rndctx;
545
546 /* XXX lock, if cprng allocated? why? */
547 memset(st, 0, sizeof(*st));
548 st->st_dev = makedev(cdevsw_lookup_major(&rnd_cdevsw),
549 (ctx->rc_hard? RND_DEV_RANDOM : RND_DEV_URANDOM));
550 /* XXX leave atimespect, mtimespec, ctimespec = 0? */
551
552 st->st_uid = kauth_cred_geteuid(fp->f_cred);
553 st->st_gid = kauth_cred_getegid(fp->f_cred);
554 st->st_mode = S_IFCHR;
555 return 0;
556}

558static int
559rnd_close(struct file *fp)
560{
561 struct rnd_ctx *const ctx = fp->f_rndctx;
562
563 if (ctx->rc_cprng != NULL)
564 cprng_strong_destroy(ctx->rc_cprng);
565 fp->f_rndctx = NULL;
566 pool_cache_put(rnd_ctx_cache, ctx);
567
568 return 0;
569}
