28#include <sys/ioctl.h>
29
30struct mbpp_param {
31 int bp_burst; /* chars to send/recv in one call */
32 int bp_timeout; /* timeout: -1 blocking, 0 non blocking >0 ms */
33 int bp_delay; /* delay between polls (ms) */
34};
35
36#define MBPP_BLOCK -1
37#define MBPP_NOBLOCK 0
38
39/* defaults */
40#define MBPP_BURST 1024
41#define MBPP_TIMEOUT MBPP_BLOCK
42#define MBPP_DELAY 10
43
44/* limits */
45#define MBPP_BURST_MIN 1
46#define MBPP_BURST_MAX 1024
47#define MBPP_DELAY_MIN 0
48#define MBPP_DELAY_MAX 30000
49
50/* status bits */
51#define MBPP_BUSY (1<<0)
52#define MBPP_PAPER (1<<1)
53
54/* ioctl commands */
55#define MBPPIOCSPARAM _IOW('P', 0x1, struct mbpp_param)
56#define MBPPIOCGPARAM _IOR('P', 0x2, struct mbpp_param)
57#define MBPPIOCGSTAT _IOR('P', 0x4, int)
