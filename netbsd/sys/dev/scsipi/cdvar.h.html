27#include <dev/dkvar.h>
28
29#define CDRETRIES 4
30
31struct cd_softc {
32 struct dk_softc sc_dksc;
33
34 int flags;
35#define CDF_ANCIENT 0x10 /* disk is ancient; for minphys */
36#define CDF_EJECTED 0x20 /* be silent when flushing cache */
37
38 struct scsipi_periph *sc_periph;
39
40 struct cd_parms {
41 u_int blksize;
42 u_long disksize; /* total number sectors */
43 u_long disksize512; /* total number sectors */
44 } params;
45
46 struct callout sc_callout;
47};
48
49#define CDGP_RESULT_OK 0 /* parameters obtained */
50#define CDGP_RESULT_OFFLINE 1 /* no media, or otherwise losing */
51#define CDGP_RESULT_UNFORMATTED 2 /* unformatted media (max params) */
