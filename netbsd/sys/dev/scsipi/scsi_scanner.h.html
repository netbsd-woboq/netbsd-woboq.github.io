33/*
34 * SCSI scanner interface description
35 */
36
37/* SCSI scanner commands */
38#define GET_IMAGE_STATUS 0x0f
39#define READ_10 0x28
40#define WRITE_10 0x2a
41#define OBJECT_POSITION 0x31
42#define GET_BUFFER_STATUS 0x34
43
44/* generic scanner command formats */
45
46struct scsi_rw_scanner {
47#define READ 0x08
48#define WRITE 0x0a
49 u_int8_t opcode;
50 u_int8_t byte2;
51#define SRW_FIXED 0x01
52 u_int8_t len[3];
53 u_int8_t control;
54};
55
56struct scsi_set_window {
57#define SET_WINDOW 0x24 /* set params of image area and windows */
58#define GET_WINDOW 0x25
59 u_int8_t opcode;
60 u_int8_t byte2;
61 u_int8_t reserved[4];
62 u_int8_t len[3];
63 u_int8_t control;
64};
65
66struct scsi_window_header {
67 u_int8_t reserved[6];
68 u_int8_t len[2];
69};
70
71struct scsi_window_data {
72 u_int8_t window_id; /* must be zero */
73 u_int8_t res1:7;
74 u_int8_t auto_bit:1;
75 u_int8_t x_res[2];
76 u_int8_t y_res[2];
77 u_int8_t x_org[4];
78 u_int8_t y_org[4];
79 u_int8_t width[4];
80 u_int8_t length[4];
81 u_int8_t brightness;
82 u_int8_t threshold;
83 u_int8_t contrast;
84 u_int8_t image_comp; /* image composition (data type) */
85 u_int8_t bits_per_pixel;
86 u_int8_t halftone_pattern[2];
87 u_int8_t rif:1; /* reverse image format (mono negative) */
88 u_int8_t res2:4;
89 u_int8_t pad_type:3;
90 u_int8_t bit_ordering[2];
91 u_int8_t compression_type;
92 u_int8_t compression_arg;
93 u_int8_t res3[6];
94};
95

99#define MUSTEK_ADF 0x10 /* ADF and backtracking selection */
100#define MUSTEK_LUT 0x55 /* look up table download */
