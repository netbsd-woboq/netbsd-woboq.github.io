867static void
868asc2ascii(u_char asc, u_char ascq, char *result, size_t l)
869{
870 int i = 0;
871
872 while (adesc[i].description != NULL) {
873 if (adesc[i].asc == asc && adesc[i].ascq == ascq)
874 break;
875 i++;
876 }
877 if (adesc[i].description == NULL) {
878 if (asc == 0x40 && ascq != 0)
879 (void)snprintf(result, l,
880 "Diagnostic Failure on Component 0x%02x",
881 ascq & 0xff);
882 else
883 (void)snprintf(result, l, "ASC 0x%02x ASCQ 0x%02x",
884 asc & 0xff, ascq & 0xff);
885 } else
886 (void)strlcpy(result, adesc[i].description, l);
887}

889void
890scsipi_print_sense_data_real(struct scsi_sense_data *sense, int verbosity)
891{
892 int32_t info;
893 int i, j, k;
894 char *sbs, *s = (char *) sense;
895
896 /*
897 * Basics- print out SENSE KEY
898 */
899 printf(" SENSE KEY: %s", scsipi_decode_sense(s, 0));
900
901 /*
902 * Print out, unqualified but aligned, FMK, EOM and ILI status.
903 */
904 if (s[2] & 0xe0) {
905 char pad;
906 printf("\n ");
907 pad = ' ';
908 if (s[2] & SSD_FILEMARK) {
909 printf("%c Filemark Detected", pad);
910 pad = ',';
911 }
912 if (s[2] & SSD_EOM) {
913 printf("%c EOM Detected", pad);
914 pad = ',';
915 }
916 if (s[2] & SSD_ILI)
917 printf("%c Incorrect Length Indicator Set", pad);
918 }
919
920 /*
921 * Now we should figure out, based upon device type, how
922 * to format the information field. Unfortunately, that's
923 * not convenient here, so we'll print it as a signed
924 * 32 bit integer.
925 */
926 info = _4btol(&s[3]);
927 if (info)
928 printf("\n INFO FIELD: %d", info);
929
930 /*
931 * Now we check additional length to see whether there is
932 * more information to extract.
933 */
934
935 /* enough for command specific information? */
936 if (((unsigned int) s[7]) < 4) {
937 printf("\n");
938 return;
939 }
940 info = _4btol(&s[8]);
941 if (info)
942 printf("\n COMMAND INFO: %d (0x%x)", info, info);
943
944 /*
945 * Decode ASC && ASCQ info, plus FRU, plus the rest...
946 */
947
948 sbs = scsipi_decode_sense(s, 1);
949 if (sbs)
950 printf("\n ASC/ASCQ: %s", sbs);
951 if (s[14] != 0)
952 printf("\n FRU CODE: 0x%x", s[14] & 0xff);
953 sbs = scsipi_decode_sense(s, 3);
954 if (sbs)
955 printf("\n SKSV: %s", sbs);
956 printf("\n");
957 if (verbosity == 0)
958 return;
959
960 /*
961 * Now figure whether we should print any additional informtion.
962 *
963 * Where should we start from? If we had SKSV data,
964 * start from offset 18, else from offset 15.
965 *
966 * From that point until the end of the buffer, check for any
967 * nonzero data. If we have some, go back and print the lot,
968 * otherwise we're done.
969 */
970 if (sbs)
971 i = 18;
972 else
973 i = 15;
974 for (j = i; j < sizeof (*sense); j++)
975 if (s[j])
976 break;
977 if (j == sizeof (*sense))
978 return;
979
980 printf("\n Additional Sense Information (byte %d out...):\n", i);
981 if (i == 15) {
982 printf("\n\t%2d:", i);
983 k = 7;
984 } else {
985 printf("\n\t%2d:", i);
986 k = 2;
987 j -= 2;
988 }
989 while (j > 0) {
990 if (i >= sizeof (*sense))
991 break;
992 if (k == 8) {
993 k = 0;
994 printf("\n\t%2d:", i);
995 }
996 printf(" 0x%02x", s[i] & 0xff);
997 k++;
998 j--;
999 i++;
1000 }
1001 printf("\n\n");
1002}

1004static char *
1005scsipi_decode_sense(void *sinfo, int flag)
1006{
1007 unsigned char *snsbuf;
1008 unsigned char skey;
1009 static char rqsbuf[132];
1010
1011 skey = 0;
1012
1013 snsbuf = (unsigned char *) sinfo;
1014 if (flag == 0 || flag == 2 || flag == 3)
1015 skey = snsbuf[2] & 0xf;
1016 if (flag == 0) { /* Sense Key Only */
1017 (void) strlcpy(rqsbuf, sense_keys[skey], sizeof(rqsbuf));
1018 return (rqsbuf);
1019 } else if (flag == 1) { /* ASC/ASCQ Only */
1020 asc2ascii(snsbuf[12], snsbuf[13], rqsbuf, sizeof(rqsbuf));
1021 return (rqsbuf);
1022 } else if (flag == 2) { /* Sense Key && ASC/ASCQ */
1023 auto char localbuf[64];
1024 asc2ascii(snsbuf[12], snsbuf[13], localbuf, sizeof(localbuf));
1025 (void) snprintf(rqsbuf, sizeof(rqsbuf), "%s, %s",
1026 sense_keys[skey], localbuf);
1027 return (rqsbuf);
1028 } else if (flag == 3 && snsbuf[7] >= 9 && (snsbuf[15] & 0x80)) {
1029 /*
1030 * SKSV Data
1031 */
1032 switch (skey) {
1033 case SKEY_ILLEGAL_REQUEST:
1034 if (snsbuf[15] & 0x8)
1035 (void)snprintf(rqsbuf, sizeof(rqsbuf),
1036 "Error in %s, Offset %d, bit %d",
1037 (snsbuf[15] & 0x40)? "CDB" : "Parameters",
1038 (snsbuf[16] & 0xff) << 8 |
1039 (snsbuf[17] & 0xff), snsbuf[15] & 0x7);
1040 else
1041 (void)snprintf(rqsbuf, sizeof(rqsbuf),
1042 "Error in %s, Offset %d",
1043 (snsbuf[15] & 0x40)? "CDB" : "Parameters",
1044 (snsbuf[16] & 0xff) << 8 |
1045 (snsbuf[17] & 0xff));
1046 return (rqsbuf);
1047 case SKEY_RECOVERED_ERROR:
1048 case SKEY_MEDIUM_ERROR:
1049 case SKEY_HARDWARE_ERROR:
1050 (void)snprintf(rqsbuf, sizeof(rqsbuf),
1051 "Actual Retry Count: %d",
1052 (snsbuf[16] & 0xff) << 8 | (snsbuf[17] & 0xff));
1053 return (rqsbuf);
1054 case SKEY_NOT_READY:
1055 (void)snprintf(rqsbuf, sizeof(rqsbuf),
1056 "Progress Indicator: %d",
1057 (snsbuf[16] & 0xff) << 8 | (snsbuf[17] & 0xff));
1058 return (rqsbuf);
1059 default:
1060 break;
1061 }
1062 }
1063 return (NULL);
1064}
