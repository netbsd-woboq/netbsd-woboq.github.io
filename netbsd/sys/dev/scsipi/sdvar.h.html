49#ifndef _DEV_SCSIPI_SDVAR_H_
50#define _DEV_SCSIPI_SDVAR_H_
51
52#ifdef _KERNEL_OPT
53#include "opt_scsi.h"
54#endif
55
56#include <dev/dkvar.h>
57
58#ifndef SDRETRIES
59#define SDRETRIES 4
60#endif
61
62#ifndef SD_IO_TIMEOUT
63#define SD_IO_TIMEOUT (60 * 1000)
64#endif
65
66struct disk_parms {
67 u_long heads; /* number of heads */
68 u_long cyls; /* number of cylinders */
69 u_long sectors; /* number of sectors/track */
70 u_long blksize; /* number of bytes/sector */
71 u_long rot_rate; /* rotational rate, in RPM */
72 u_int64_t disksize; /* total number sectors */
73 u_int64_t disksize512; /* total number sectors */
74};
75
76struct sd_softc {
77 struct dk_softc sc_dksc;
78
79 int flags;
80#define SDF_ANCIENT 0x10 /* disk is ancient; for minphys */
81#define SDF_DIRTY 0x20 /* disk is dirty; needs cache flush */
82#define SDF_FLUSHING 0x40 /* flushing, for sddone() */
83
84 struct scsipi_periph *sc_periph;/* contains our targ, lun, etc. */
85
86 struct disk_parms params;
87
88 callout_t sc_callout;
89 u_int8_t type;
90 char name[16]; /* product name, for default disklabel */
91 char typename[128+4+1]; /* stored in disk info */
92};
93
94#define SDGP_RESULT_OK 0 /* parameters obtained */
95#define SDGP_RESULT_OFFLINE 1 /* no media, or otherwise losing */
96#define SDGP_RESULT_UNFORMATTED 2 /* unformatted media (max params) */
97
98#endif /* _DEV_SCSIPI_SDVAR_H_ */
