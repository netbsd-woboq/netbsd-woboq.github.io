251/* SDHC_CAPABILITIES decoding */
252#define SDHC_BASE_V3_FREQ_KHZ(cap) \
253 ((((cap) >> SDHC_BASE_FREQ_SHIFT) & SDHC_BASE_V3_FREQ_MASK) * 1000)
254#define SDHC_BASE_FREQ_KHZ(cap) \
255 ((((cap) >> SDHC_BASE_FREQ_SHIFT) & SDHC_BASE_FREQ_MASK) * 1000)
256#define SDHC_TIMEOUT_FREQ(cap) \
257 (((cap) >> SDHC_TIMEOUT_FREQ_SHIFT) & SDHC_TIMEOUT_FREQ_MASK)
258#define SDHC_TIMEOUT_FREQ_KHZ(cap) \
259 (((cap) & SDHC_TIMEOUT_FREQ_UNIT) ? \
260 SDHC_TIMEOUT_FREQ(cap) * 1000: \
261 SDHC_TIMEOUT_FREQ(cap))
