29#ifndef _DEV_SPI_SPI_IO_H_
30#define _DEV_SPI_SPI_IO_H_
31
32#include <sys/types.h>
33#include <sys/ioccom.h>
34#include <sys/uio.h>
35
36typedef struct spi_ioctl_configure {
37 int sic_addr;
38 int sic_mode;
39 int sic_speed;
40} spi_ioctl_configure_t;
41
42typedef struct spi_ioctl_transfer {
43 int sit_addr;
44 const void *sit_send;
45 size_t sit_sendlen;
46 void *sit_recv;
47 size_t sit_recvlen;
48} spi_ioctl_transfer_t;
49
50#define SPI_IOCTL_CONFIGURE _IOW('S', 0, spi_ioctl_configure_t)
51#define SPI_IOCTL_TRANSFER _IOW('S', 1, spi_ioctl_transfer_t)
52
53#endif /* _DEV_SPI_SPI_IO_H_ */
