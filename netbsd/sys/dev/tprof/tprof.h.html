29#ifndef _DEV_TPROF_TPROF_H_
30#define _DEV_TPROF_TPROF_H_
31
32/*
33 * definitions used by backend drivers
34 */
35
36#include <sys/types.h>
37
38#include <dev/tprof/tprof_types.h>
39
40typedef struct tprof_backend_ops {
41 uint64_t (*tbo_estimate_freq)(void); /* samples per second */
42 uint32_t (*tbo_ident)(void);
43 int (*tbo_start)(const tprof_param_t *);
44 void (*tbo_stop)(const tprof_param_t *);
45} tprof_backend_ops_t;
46
47#define TPROF_BACKEND_VERSION 3
48int tprof_backend_register(const char *, const tprof_backend_ops_t *, int);
49int tprof_backend_unregister(const char *);
50
51typedef struct {
52 uintptr_t tfi_pc; /* program counter */
53 bool tfi_inkernel; /* if tfi_pc is in the kernel address space */
54} tprof_frame_info_t;
55
56void tprof_sample(void *, const tprof_frame_info_t *);
57
58#endif /* _DEV_TPROF_TPROF_H_ */
