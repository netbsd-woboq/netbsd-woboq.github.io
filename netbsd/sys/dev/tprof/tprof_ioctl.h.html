29#ifndef _DEV_TPROF_TPROF_IOCTL_H_
30#define _DEV_TPROF_TPROF_IOCTL_H_
31
32/*
33 * definitions for userland consumer
34 */
35
36#include <sys/ioccom.h>
37
38#include <dev/tprof/tprof_types.h>
39
40#define TPROF_VERSION 4 /* kernel-userland ABI version */
41
42struct tprof_info {
43 uint32_t ti_version;
44 uint32_t ti_ident;
45};
46#define TPROF_IOC_GETINFO _IOR('T', 1, struct tprof_info)
47
48#define TPROF_IOC_START _IOW('T', 2, tprof_param_t)
49
50#define TPROF_IOC_STOP _IO('T', 3)
51
52struct tprof_stat {
53 uint64_t ts_sample; /* samples successfully recorded */
54 uint64_t ts_overflow; /* samples dropped due to overflow */
55 uint64_t ts_buf; /* buffers successfully queued for read(2) */
56 uint64_t ts_emptybuf; /* empty buffers dropped */
57 uint64_t ts_dropbuf; /* buffers dropped due to the global limit */
58 uint64_t ts_dropbuf_sample; /* samples dropped with ts_dropbuf */
59};
60#define TPROF_IOC_GETSTAT _IOR('T', 4, struct tprof_stat)
61
62#endif /* _DEV_TPROF_TPROF_IOCTL_H_ */
