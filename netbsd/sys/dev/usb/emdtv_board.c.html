78const struct emdtv_board *
79emdtv_board_lookup(uint16_t vendor, uint16_t product)
80{
81 const struct emdtv_board *eb;
82 unsigned int i;
83
84 for (i = 0; i < __arraycount(emdtv_boards); i++) {
85 eb = &emdtv_boards[i];
86 if (vendor == eb->eb_vendor && product == eb->eb_product)
87 return eb;
88 }
89
90 return NULL;
91}
