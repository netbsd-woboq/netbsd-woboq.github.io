29#ifndef _DEV_USB_EMDTV_BOARD_H
30#define _DEV_USB_EMDTV_BOARD_H
31
32typedef enum emdtv_tuner_type {
33 EMDTV_TUNER_XC3028L,
34 EMDTV_TUNER_XC3028,
35 EMDTV_TUNER_XC5000,
36} emdtv_tuner_type_t;
37
38typedef enum emdtv_demod_type {
39 EMDTV_DEMOD_LG3303,
40 EMDTV_DEMOD_LG3304,
41} emdtv_demod_type_t;
42
43typedef enum emdtv_gpio_reg {
44 EMDTV_GPIO_DEMOD1_RESET = 16,
45 EMDTV_GPIO_TUNER1_RESET = 17,
46 EMDTV_GPIO_TUNER1_ON = 23,
47 EMDTV_GPIO_ANALOG_ON = 24,
48 EMDTV_GPIO_TS1_ON = 25,
49} emdtv_gpio_reg_t;
50
51#define EMDTV_GPIO_BIT_VAL(reg, val, reset) \
52 ((reg) | 0x80 | (reset ? (1 << 6) : 0) | (val << 5))
53
54struct emdtv_gpio_regs {
55 uint16_t ts1_on;
56 uint16_t a_on;
57 uint16_t t1_on;
58 uint16_t t1_reset;
59 uint16_t d1_reset;
60};
61
62struct emdtv_board {
63 const char *eb_name;
64 uint16_t eb_vendor;
65 uint16_t eb_product;
66 emdtv_tuner_type_t eb_tuner;
67 emdtv_demod_type_t eb_demod;
68 bool eb_manual_gpio;
69 struct emdtv_gpio_regs eb_gpio_regs;
70};
71
72const struct emdtv_board *emdtv_board_lookup(uint16_t, uint16_t);
73
74#endif /* !_DEV_USB_EMDTV_BOARD_H */
