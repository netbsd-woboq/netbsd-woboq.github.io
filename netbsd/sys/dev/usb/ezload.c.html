52/*
53 * Vendor specific request code for Anchor Upload/Download
54 */
55
56/* This one is implemented in the core */
57#define ANCHOR_LOAD_INTERNAL 0xA0
58
59/* This is the highest internal RAM address for the AN2131Q */
60#define ANCHOR_MAX_INTERNAL_ADDRESS 0x1B3F
61
62/*
63 * EZ-USB Control and Status Register. Bit 0 controls 8051 reset
64 */
65#define ANCHOR_CPUCS_REG 0x7F92
66#define ANCHOR_RESET 0x01
67
68/*
69 * Although USB does not limit you here, the Anchor docs
70 * quote 64 as a limit, and Mato@activewireinc.com suggested
71 * to use 16.
72 */
73#define ANCHOR_CHUNK 16
74
75/*
76 * This is a firmware loader for ezusb (Anchor) devices. When the firmware
77 * has been downloaded the device will simulate a disconnect and when it
78 * is next recognized by the USB software it will appear as another
79 * device.
80 */

91usbd_status
92ezload_reset(struct usbd_device *dev, int reset)
93{
94 usb_device_request_t req;
95 uByte rst;
96
97 DPRINTF(("ezload_reset: reset=%d\n", reset));
98
99 rst = reset ? ANCHOR_RESET : 0;
100 req.bmRequestType = UT_WRITE_VENDOR_DEVICE;
101 req.bRequest = ANCHOR_LOAD_INTERNAL;
102 USETW(req.wValue, ANCHOR_CPUCS_REG);
103 USETW(req.wIndex, 0);
104 USETW(req.wLength, 1);
105 return usbd_do_request(dev, &req, &rst);
106}

108usbd_status
109ezload_download(struct usbd_device *dev, const struct ezdata *rec)
110{
111 usb_device_request_t req;
112 const struct ezdata *ptr;
113 u_int len, offs;
114 int err;
115
116 DPRINTF(("ezload_down record=%p\n", rec));
117
118 for (ptr = rec; ptr->length != 0; ptr++) {
119
120#if 0
121 if (ptr->address + ptr->length > ANCHOR_MAX_INTERNAL_ADDRESS)
122 return USBD_INVAL;
123#endif
124
125 req.bmRequestType = UT_WRITE_VENDOR_DEVICE;
126 req.bRequest = ANCHOR_LOAD_INTERNAL;
127 USETW(req.wIndex, 0);
128 for (offs = 0; offs < ptr->length; offs += ANCHOR_CHUNK) {
129 len = ptr->length - offs;
130 if (len > ANCHOR_CHUNK)
131 len = ANCHOR_CHUNK;
132 USETW(req.wValue, ptr->address + offs);
133 USETW(req.wLength, len);
134 DPRINTFN(2,("ezload_download: addr=0x%x len=%d\n",
135 ptr->address + offs, len));
136 /*XXXUNCONST*/
137 err = usbd_do_request(dev, &req,
138 __UNCONST(ptr->data + offs));
139 if (err)
140 return err;
141 }
142 }
143
144 return 0;
145}

147usbd_status
148ezload_downloads_and_reset(struct usbd_device *dev, const struct ezdata **recs)
149{
150 usbd_status err;
151
152 /*(void)ezload_reset(dev, 1);*/
153 err = ezload_reset(dev, 1);
154 if (err)
155 return err;
156 usbd_delay_ms(dev, 250);
157 while (*recs != NULL) {
158 err = ezload_download(dev, *recs++);
159 if (err)
160 return err;
161 }
162 usbd_delay_ms(dev, 250);
163 err = ezload_reset(dev, 0);
164 usbd_delay_ms(dev, 250);
165 return err;
166}
