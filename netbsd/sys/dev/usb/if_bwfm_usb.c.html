88/* Control messages: bRequest values */
89#define DL_GETSTATE 0 /* returns the rdl_state_t struct */
90#define DL_CHECK_CRC 1 /* currently unused */
91#define DL_GO 2 /* execute downloaded image */
92#define DL_START 3 /* initialize dl state */
93#define DL_REBOOT 4 /* reboot the device in 2 seconds */
94#define DL_GETVER 5 /* returns the bootrom_id_t struct */
95#define DL_GO_PROTECTED 6 /* execute the downloaded code and set reset
96 * event to occur in 2 seconds. It is the
97 * responsibility of the downloaded code to
98 * clear this event
99 */
100#define DL_EXEC 7 /* jump to a supplied address */
101#define DL_RESETCFG 8 /* To support single enum on dongle
102 * - Not used by bootloader
103 */
104#define DL_DEFER_RESP_OK 9 /* Potentially defer the response to setup
105 * if resp unavailable
106 */
107
108/* states */
109#define DL_WAITING 0 /* waiting to rx first pkt */
110#define DL_READY 1 /* hdr was good, waiting for more of the
111 * compressed image
112 */
113#define DL_BAD_HDR 2 /* hdr was corrupted */
114#define DL_BAD_CRC 3 /* compressed image was corrupted */
115#define DL_RUNNABLE 4 /* download was successful,waiting for go cmd */
116#define DL_START_FAIL 5 /* failed to initialize correctly */
117#define DL_NVRAM_TOOBIG 6 /* host specified nvram data exceeds DL_NVRAM
118 * value
119 */
120#define DL_IMAGE_TOOBIG 7 /* firmware image too big */

235void
236bwfm_usb_attach(device_t parent, device_t self, void *aux)
237{
238 struct bwfm_usb_softc *sc = device_private(self);
239 struct usb_attach_arg *uaa = aux;
240 usb_device_descriptor_t *dd;
241 usb_interface_descriptor_t *id;
242 usb_endpoint_descriptor_t *ed;
243 char *devinfop;
244 int i;
245
246 sc->sc_sc.sc_dev = self;
247 sc->sc_udev = uaa->uaa_device;
248 mutex_init(&sc->sc_rx_lock, MUTEX_DEFAULT, IPL_NET);
249 mutex_init(&sc->sc_tx_lock, MUTEX_DEFAULT, IPL_NET);
250
251 aprint_naive("\n");
252
253 devinfop = usbd_devinfo_alloc(sc->sc_udev, 0);
254 aprint_normal(": %s\n", devinfop);
255 usbd_devinfo_free(devinfop);
256
257 if (usbd_set_config_no(sc->sc_udev, 1, 1) != 0) {
258 aprint_error_dev(self, "failed to set configuration\n");
259 return;
260 }
261 if (usbd_device2interface_handle(sc->sc_udev, 0, &sc->sc_iface) != 0) {
262 aprint_error_dev(self, "failed to get interface handle\n");
263 return;
264 }
265
266 sc->sc_ifaceno = 0;
267 sc->sc_vendor = uaa->uaa_vendor;
268 sc->sc_product = uaa->uaa_product;
269 sc->sc_sc.sc_bus_ops = &bwfm_usb_bus_ops;
270 sc->sc_sc.sc_proto_ops = &bwfm_proto_bcdc_ops;
271
272 /* Check number of configurations. */
273 dd = usbd_get_device_descriptor(sc->sc_udev);
274 if (dd->bNumConfigurations != 1) {
275 printf("%s: number of configurations not supported\n",
276 DEVNAME(sc));
277 return;
278 }
279
280 /* Get endpoints. */
281 id = usbd_get_interface_descriptor(sc->sc_iface);
282
283 sc->sc_rx_no = sc->sc_tx_no = -1;
284 for (i = 0; i < id->bNumEndpoints; i++) {
285 ed = usbd_interface2endpoint_descriptor(sc->sc_iface, i);
286 if (ed == NULL) {
287 printf("%s: no endpoint descriptor for iface %d\n",
288 DEVNAME(sc), i);
289 return;
290 }
291
292 if (UE_GET_DIR(ed->bEndpointAddress) == UE_DIR_IN &&
293 UE_GET_XFERTYPE(ed->bmAttributes) == UE_BULK &&
294 sc->sc_rx_no == -1)
295 sc->sc_rx_no = ed->bEndpointAddress;
296 else if (UE_GET_DIR(ed->bEndpointAddress) == UE_DIR_OUT &&
297 UE_GET_XFERTYPE(ed->bmAttributes) == UE_BULK &&
298 sc->sc_tx_no == -1)
299 sc->sc_tx_no = ed->bEndpointAddress;
300 }
301 if (sc->sc_rx_no == -1 || sc->sc_tx_no == -1) {
302 printf("%s: missing endpoint\n", DEVNAME(sc));
303 return;
304 }
305
306 config_mountroot(self, bwfm_usb_attachhook);
307}

309void
310bwfm_usb_attachhook(device_t self)
311{
312 struct bwfm_usb_softc *sc = device_private(self);
313 struct bwfm_usb_rx_data *data;
314 const char *name = NULL;
315 struct bootrom_id brom;
316 firmware_handle_t fwh;
317 usbd_status error;
318 u_char *ucode;
319 size_t size;
320 int i;
321
322 /* Read chip id and chip rev to check the firmware. */
323 memset(&brom, 0, sizeof(brom));
324 bwfm_usb_dl_cmd(sc, DL_GETVER, &brom, sizeof(brom));
325 sc->sc_chip = le32toh(brom.chip);
326 sc->sc_chiprev = le32toh(brom.chiprev);
327
328 /* Setup data pipes */
329 error = usbd_open_pipe(sc->sc_iface, sc->sc_rx_no, USBD_EXCLUSIVE_USE,
330 &sc->sc_rx_pipeh);
331 if (error != 0) {
332 printf("%s: could not open rx pipe: %s\n",
333 DEVNAME(sc), usbd_errstr(error));
334 return;
335 }
336 error = usbd_open_pipe(sc->sc_iface, sc->sc_tx_no, USBD_EXCLUSIVE_USE,
337 &sc->sc_tx_pipeh);
338 if (error != 0) {
339 printf("%s: could not open tx pipe: %s\n",
340 DEVNAME(sc), usbd_errstr(error));
341 return;
342 }
343
344 /* Firmware not yet loaded? */
345 if (sc->sc_chip != BRCMF_POSTBOOT_ID) {
346 switch (sc->sc_chip)
347 {
348 case BRCM_CC_43143_CHIP_ID:
349 name = "brcmfmac43143.bin";
350 break;
351 case BRCM_CC_43235_CHIP_ID:
352 case BRCM_CC_43236_CHIP_ID:
353 case BRCM_CC_43238_CHIP_ID:
354 if (sc->sc_chiprev == 3)
355 name = "brcmfmac43236b.bin";
356 break;
357 case BRCM_CC_43242_CHIP_ID:
358 name = "brcmfmac43242a.bin";
359 break;
360 case BRCM_CC_43566_CHIP_ID:
361 case BRCM_CC_43569_CHIP_ID:
362 name = "brcmfmac43569.bin";
363 break;
364 default:
365 break;
366 }
367
368 if (name == NULL) {
369 printf("%s: unknown firmware\n", DEVNAME(sc));
370 return;
371 }
372
373 if (firmware_open("if_bwfm", name, &fwh) != 0) {
374 printf("%s: failed firmware_open of file %s\n",
375 DEVNAME(sc), name);
376 return;
377 }
378 size = firmware_get_size(fwh);
379 ucode = firmware_malloc(size);
380 if (ucode == NULL) {
381 printf("%s: failed to allocate firmware memory\n",
382 DEVNAME(sc));
383 firmware_close(fwh);
384 return;
385 }
386 error = firmware_read(fwh, 0, ucode, size);
387 firmware_close(fwh);
388 if (error != 0) {
389 printf("%s: failed to read firmware (error %d)\n",
390 DEVNAME(sc), error);
391 firmware_free(ucode, size);
392 return;
393 }
394
395 if (bwfm_usb_load_microcode(sc, ucode, size) != 0) {
396 printf("%s: could not load microcode\n",
397 DEVNAME(sc));
398 return;
399 }
400
401 firmware_free(ucode, size);
402
403 for (i = 0; i < 10; i++) {
404 delay(100 * 1000);
405 memset(&brom, 0, sizeof(brom));
406 bwfm_usb_dl_cmd(sc, DL_GETVER, &brom, sizeof(brom));
407 if (le32toh(brom.chip) == BRCMF_POSTBOOT_ID)
408 break;
409 }
410
411 if (le32toh(brom.chip) != BRCMF_POSTBOOT_ID) {
412 printf("%s: firmware did not start up\n",
413 DEVNAME(sc));
414 return;
415 }
416
417 sc->sc_chip = le32toh(brom.chip);
418 sc->sc_chiprev = le32toh(brom.chiprev);
419 printf("%s: firmware loaded\n", DEVNAME(sc));
420 }
421
422 bwfm_usb_dl_cmd(sc, DL_RESETCFG, &brom, sizeof(brom));
423
424 if (bwfm_usb_alloc_rx_list(sc) || bwfm_usb_alloc_tx_list(sc)) {
425 printf("%s: cannot allocate rx/tx lists\n", DEVNAME(sc));
426 return;
427 }
428
429 bwfm_attach(&sc->sc_sc);
430
431 for (i = 0; i < BWFM_RX_LIST_COUNT; i++) {
432 data = &sc->sc_rx_data[i];
433
434 usbd_setup_xfer(data->xfer, data, data->buf,
435 BWFM_RXBUFSZ, USBD_SHORT_XFER_OK, USBD_NO_TIMEOUT,
436 bwfm_usb_rxeof);
437 error = usbd_transfer(data->xfer);
438 if (error != 0 && error != USBD_IN_PROGRESS)
439 printf("%s: could not set up new transfer: %s\n",
440 DEVNAME(sc), usbd_errstr(error));
441 }
442}

444struct mbuf *
445bwfm_usb_newbuf(void)
446{
447 struct mbuf *m;
448
449 MGETHDR(m, M_DONTWAIT, MT_DATA);
450 if (m == NULL)
451 return (NULL);
452
453 MCLGET(m, M_DONTWAIT);
454 if (!(m->m_flags & M_EXT)) {
455 m_freem(m);
456 return (NULL);
457 }
458
459 m->m_len = m->m_pkthdr.len = MCLBYTES;
460
461 return (m);
462}

464void
465bwfm_usb_rxeof(struct usbd_xfer *xfer, void *priv, usbd_status status)
466{
467 struct bwfm_usb_rx_data *data = priv;
468 struct bwfm_usb_softc *sc = data->sc;
469 struct bwfm_proto_bcdc_hdr *hdr;
470 usbd_status error;
471 uint32_t len, off;
472 struct mbuf *m;
473
474 DPRINTFN(2, ("%s: %s status %s\n", DEVNAME(sc), __func__,
475 usbd_errstr(status)));
476
477 if (__predict_false(status != USBD_NORMAL_COMPLETION)) {
478 usbd_clear_endpoint_stall_async(sc->sc_rx_pipeh);
479 if (status != USBD_CANCELLED)
480 goto resubmit;
481 return;
482 }
483 usbd_get_xfer_status(xfer, NULL, NULL, &len, NULL);
484
485 off = 0;
486 hdr = (void *)data->buf;
487 if (len < sizeof(*hdr))
488 goto resubmit;
489 len -= sizeof(*hdr);
490 off += sizeof(*hdr);
491 if (len <= hdr->data_offset << 2)
492 goto resubmit;
493 len -= hdr->data_offset << 2;
494 off += hdr->data_offset << 2;
495
496 m = bwfm_usb_newbuf();
497 if (m == NULL)
498 goto resubmit;
499
500 memcpy(mtod(m, char *), data->buf + off, len);
501 m->m_len = m->m_pkthdr.len = len;
502 mutex_enter(&sc->sc_rx_lock); /* XXX */
503 bwfm_rx(&sc->sc_sc, m);
504 mutex_exit(&sc->sc_rx_lock);
505
506resubmit:
507 usbd_setup_xfer(data->xfer, data, data->buf,
508 BWFM_RXBUFSZ, USBD_SHORT_XFER_OK, USBD_NO_TIMEOUT,
509 bwfm_usb_rxeof);
510 error = usbd_transfer(data->xfer);
511 if (error != 0 && error != USBD_IN_PROGRESS)
512 printf("%s: could not set up new transfer: %s\n",
513 DEVNAME(sc), usbd_errstr(error));
514}

516int
517bwfm_usb_alloc_rx_list(struct bwfm_usb_softc *sc)
518{
519 struct bwfm_usb_rx_data *data;
520 int i, error = 0;
521
522 for (i = 0; i < BWFM_RX_LIST_COUNT; i++) {
523 data = &sc->sc_rx_data[i];
524
525 data->sc = sc; /* Backpointer for callbacks. */
526
527 if (usbd_create_xfer(sc->sc_rx_pipeh, BWFM_RXBUFSZ,
528 0, 0, &data->xfer) != 0) {
529 printf("%s: could not create xfer\n",
530 DEVNAME(sc));
531 error = ENOMEM;
532 break;
533 }
534 data->buf = usbd_get_buffer(data->xfer);
535 }
536 if (error != 0)
537 bwfm_usb_free_rx_list(sc);
538 return (error);
539}

541void
542bwfm_usb_free_rx_list(struct bwfm_usb_softc *sc)
543{
544 int i;
545
546 /* NB: Caller must abort pipe first. */
547 for (i = 0; i < BWFM_RX_LIST_COUNT; i++) {
548 if (sc->sc_rx_data[i].xfer != NULL)
549 usbd_destroy_xfer(sc->sc_rx_data[i].xfer);
550 sc->sc_rx_data[i].xfer = NULL;
551 }
552}

554int
555bwfm_usb_alloc_tx_list(struct bwfm_usb_softc *sc)
556{
557 struct bwfm_usb_tx_data *data;
558 int i, error = 0;
559
560 TAILQ_INIT(&sc->sc_tx_free_list);
561 for (i = 0; i < BWFM_TX_LIST_COUNT; i++) {
562 data = &sc->sc_tx_data[i];
563
564 data->sc = sc; /* Backpointer for callbacks. */
565
566 if (usbd_create_xfer(sc->sc_tx_pipeh, BWFM_TXBUFSZ,
567 USBD_FORCE_SHORT_XFER, 0, &data->xfer) != 0) {
568 printf("%s: could not create xfer\n",
569 DEVNAME(sc));
570 error = ENOMEM;
571 break;
572 }
573 data->buf = usbd_get_buffer(data->xfer);
574
575 /* Append this Tx buffer to our free list. */
576 TAILQ_INSERT_TAIL(&sc->sc_tx_free_list, data, next);
577 }
578 if (error != 0)
579 bwfm_usb_free_tx_list(sc);
580 return (error);
581}

596void
597bwfm_usb_txeof(struct usbd_xfer *xfer, void *priv, usbd_status status)
598{
599 struct bwfm_usb_tx_data *data = priv;
600 struct bwfm_usb_softc *sc = data->sc;
601 struct ifnet *ifp = sc->sc_sc.sc_ic.ic_ifp;
602 int s;
603
604 DPRINTFN(2, ("%s: %s status %s\n", DEVNAME(sc), __func__,
605 usbd_errstr(status)));
606
607 m_freem(data->mbuf);
608 data->mbuf = NULL;
609
610 mutex_enter(&sc->sc_tx_lock);
611 /* Put this Tx buffer back to our free list. */
612 TAILQ_INSERT_TAIL(&sc->sc_tx_free_list, data, next);
613 mutex_exit(&sc->sc_tx_lock);
614
615 s = splnet();
616
617 if (__predict_false(status != USBD_NORMAL_COMPLETION)) {
618 if (status == USBD_CANCELLED)
619 usbd_clear_endpoint_stall_async(sc->sc_tx_pipeh);
620 ifp->if_oerrors++;
621 splx(s);
622 return;
623 }
624
625 ifp->if_opackets++;
626
627 /* We just released a Tx buffer, notify Tx. */
628 if ((ifp->if_flags & IFF_OACTIVE) != 0) {
629 ifp->if_flags &= ~IFF_OACTIVE;
630 if_schedule_deferred_start(ifp);
631 }
632 splx(s);
633}

635int
636bwfm_usb_detach(device_t self, int flags)
637{
638 struct bwfm_usb_softc *sc = device_private(self);
639
640 bwfm_detach(&sc->sc_sc, flags);
641
642 if (sc->sc_rx_pipeh != NULL) {
643 usbd_abort_pipe(sc->sc_rx_pipeh);
644 usbd_close_pipe(sc->sc_rx_pipeh);
645 }
646 if (sc->sc_tx_pipeh != NULL) {
647 usbd_abort_pipe(sc->sc_tx_pipeh);
648 usbd_close_pipe(sc->sc_tx_pipeh);
649 }
650
651 bwfm_usb_free_rx_list(sc);
652 bwfm_usb_free_tx_list(sc);
653
654 mutex_destroy(&sc->sc_rx_lock);
655 mutex_destroy(&sc->sc_tx_lock);
656
657 return 0;
658}

681int
682bwfm_usb_load_microcode(struct bwfm_usb_softc *sc, const u_char *ucode, size_t size)
683{
684 const struct trx_header *trx = (const struct trx_header *)ucode;
685 struct rdl_state state;
686 uint32_t rdlstate, rdlbytes, sent = 0, sendlen = 0;
687 struct usbd_xfer *xfer;
688 usbd_status error;
689 char *buf;
690
691 if (le32toh(trx->magic) != TRX_MAGIC ||
692 (le32toh(trx->flag_version) & TRX_UNCOMP_IMAGE) == 0) {
693 printf("%s: invalid firmware\n", DEVNAME(sc));
694 return 1;
695 }
696
697 bwfm_usb_dl_cmd(sc, DL_START, &state, sizeof(state));
698 rdlstate = le32toh(state.state);
699 rdlbytes = le32toh(state.bytes);
700
701 if (rdlstate != DL_WAITING) {
702 printf("%s: cannot start fw download\n", DEVNAME(sc));
703 return 1;
704 }
705
706 error = usbd_create_xfer(sc->sc_tx_pipeh, TRX_RDL_CHUNK,
707 0, 0, &xfer);
708 if (error != 0) {
709 printf("%s: cannot create xfer\n", DEVNAME(sc));
710 goto err;
711 }
712
713 buf = usbd_get_buffer(xfer);
714
715 while (rdlbytes != size) {
716 sendlen = MIN(size - sent, TRX_RDL_CHUNK);
717 memcpy(buf, ucode + sent, sendlen);
718
719 usbd_setup_xfer(xfer, NULL, buf, sendlen,
720 USBD_SYNCHRONOUS, USBD_NO_TIMEOUT, NULL);
721 error = usbd_transfer(xfer);
722 if (error != 0 && error != USBD_IN_PROGRESS) {
723 printf("%s: transfer error\n", DEVNAME(sc));
724 goto err;
725 }
726 sent += sendlen;
727
728 bwfm_usb_dl_cmd(sc, DL_GETSTATE, &state, sizeof(state));
729 rdlstate = le32toh(state.state);
730 rdlbytes = le32toh(state.bytes);
731
732 if (rdlbytes != sent) {
733 printf("%s: device reported different size\n",
734 DEVNAME(sc));
735 goto err;
736 }
737
738 if (rdlstate == DL_BAD_HDR || rdlstate == DL_BAD_CRC) {
739 printf("%s: device reported bad hdr/crc\n",
740 DEVNAME(sc));
741 goto err;
742 }
743 }
744
745 bwfm_usb_dl_cmd(sc, DL_GETSTATE, &state, sizeof(state));
746 rdlstate = le32toh(state.state);
747 rdlbytes = le32toh(state.bytes);
748
749 if (rdlstate != DL_RUNNABLE) {
750 printf("%s: dongle not runnable\n", DEVNAME(sc));
751 goto err;
752 }
753
754 bwfm_usb_dl_cmd(sc, DL_GO, &state, sizeof(state));
755
756 usbd_destroy_xfer(xfer);
757
758 return 0;
759err:
760 if (sc->sc_tx_pipeh != NULL) {
761 usbd_abort_pipe(sc->sc_tx_pipeh);
762 usbd_close_pipe(sc->sc_tx_pipeh);
763 sc->sc_tx_pipeh = NULL;
764 }
765 if (xfer != NULL)
766 usbd_destroy_xfer(xfer);
767 return 1;
768}

787int
788bwfm_usb_txdata(struct bwfm_softc *bwfm, struct mbuf **mp)
789{
790 struct bwfm_usb_softc *sc = (void *)bwfm;
791 struct mbuf *m = *mp;
792 struct bwfm_proto_bcdc_hdr *hdr;
793 struct bwfm_usb_tx_data *data;
794 struct ether_header *eh;
795 uint32_t len = 0;
796 int error, ac;
797
798 DPRINTFN(2, ("%s: %s\n", DEVNAME(sc), __func__));
799
800 mutex_enter(&sc->sc_tx_lock);
801
802 if (TAILQ_EMPTY(&sc->sc_tx_free_list)) {
803 mutex_exit(&sc->sc_tx_lock);
804 return ENOBUFS;
805 }
806
807 /* No QoS for EAPOL frames. */
808 eh = mtod(m, struct ether_header *);
809 ac = (eh->ether_type != htons(ETHERTYPE_PAE)) ?
810 M_WME_GETAC(m) : WME_AC_BE;
811
812 /* Grab a Tx buffer from our free list. */
813 data = TAILQ_FIRST(&sc->sc_tx_free_list);
814 TAILQ_REMOVE(&sc->sc_tx_free_list, data, next);
815
816 mutex_exit(&sc->sc_tx_lock);
817
818 hdr = (void *)&data->buf[len];
819 hdr->data_offset = 0;
820 hdr->priority = ac;
821 hdr->flags = BWFM_BCDC_FLAG_VER(BWFM_BCDC_FLAG_PROTO_VER);
822 hdr->flags2 = 0;
823 len += sizeof(*hdr);
824
825 m_copydata(m, 0, m->m_pkthdr.len, &data->buf[len]);
826 len += m->m_pkthdr.len;
827
828 data->mbuf = m;
829
830 usbd_setup_xfer(data->xfer, data, data->buf,
831 len, USBD_FORCE_SHORT_XFER, USBD_NO_TIMEOUT,
832 bwfm_usb_txeof);
833 error = usbd_transfer(data->xfer);
834 if (error != 0 && error != USBD_IN_PROGRESS)
835 printf("%s: could not set up new transfer: %s\n",
836 DEVNAME(sc), usbd_errstr(error));
837 return 0;
838}
