177int
178cdce_match(device_t parent, cfdata_t match, void *aux)
179{
180 struct usbif_attach_arg *uiaa = aux;
181
182 if (cdce_lookup(uiaa->uiaa_vendor, uiaa->uiaa_product) != NULL)
183 return UMATCH_VENDOR_PRODUCT;
184
185 if (uiaa->uiaa_class == UICLASS_CDC && uiaa->uiaa_subclass ==
186 UISUBCLASS_ETHERNET_NETWORKING_CONTROL_MODEL)
187 return UMATCH_IFACECLASS_GENERIC;
188
189 return UMATCH_NONE;
190}

192void
193cdce_attach(device_t parent, device_t self, void *aux)
194{
195 struct cdce_softc *sc = device_private(self);
196 struct usbif_attach_arg *uiaa = aux;
197 char *devinfop;
198 struct ifnet *ifp;
199 struct usbd_device *dev = uiaa->uiaa_device;
200 const struct cdce_type *t;
201 usb_interface_descriptor_t *id;
202 usb_endpoint_descriptor_t *ed;
203 const usb_cdc_union_descriptor_t *ud;
204 usb_config_descriptor_t *cd;
205 int data_ifcno;
206 int i, j, numalts;
207 u_char eaddr[ETHER_ADDR_LEN];
208 const usb_cdc_ethernet_descriptor_t *ue;
209 char eaddr_str[USB_MAX_ENCODED_STRING_LEN];
210
211 sc->cdce_dev = self;
212
213 aprint_naive("\n");
214 aprint_normal("\n");
215
216 devinfop = usbd_devinfo_alloc(dev, 0);
217 aprint_normal_dev(self, "%s\n", devinfop);
218 usbd_devinfo_free(devinfop);
219
220 sc->cdce_udev = uiaa->uiaa_device;
221 sc->cdce_ctl_iface = uiaa->uiaa_iface;
222
223 t = cdce_lookup(uiaa->uiaa_vendor, uiaa->uiaa_product);
224 if (t)
225 sc->cdce_flags = t->cdce_flags;
226
227 if (sc->cdce_flags & CDCE_NO_UNION)
228 sc->cdce_data_iface = sc->cdce_ctl_iface;
229 else {
230 ud = (const usb_cdc_union_descriptor_t *)usb_find_desc(sc->cdce_udev,
231 UDESC_CS_INTERFACE, UDESCSUB_CDC_UNION);
232 if (ud == NULL) {
233 aprint_error_dev(self, "no union descriptor\n");
234 return;
235 }
236 data_ifcno = ud->bSlaveInterface[0];
237
238 for (i = 0; i < uiaa->uiaa_nifaces; i++) {
239 if (uiaa->uiaa_ifaces[i] != NULL) {
240 id = usbd_get_interface_descriptor(
241 uiaa->uiaa_ifaces[i]);
242 if (id != NULL && id->bInterfaceNumber ==
243 data_ifcno) {
244 sc->cdce_data_iface = uiaa->uiaa_ifaces[i];
245 uiaa->uiaa_ifaces[i] = NULL;
246 }
247 }
248 }
249 }
250
251 if (sc->cdce_data_iface == NULL) {
252 aprint_error_dev(self, "no data interface\n");
253 return;
254 }
255
256 /*
257 * <quote>
258 * The Data Class interface of a networking device shall have a minimum
259 * of two interface settings. The first setting (the default interface
260 * setting) includes no endpoints and therefore no networking traffic is
261 * exchanged whenever the default interface setting is selected. One or
262 * more additional interface settings are used for normal operation, and
263 * therefore each includes a pair of endpoints (one IN, and one OUT) to
264 * exchange network traffic. Select an alternate interface setting to
265 * initialize the network aspects of the device and to enable the
266 * exchange of network traffic.
267 * </quote>
268 *
269 * Some devices, most notably cable modems, include interface settings
270 * that have no IN or OUT endpoint, therefore loop through the list of all
271 * available interface settings looking for one with both IN and OUT
272 * endpoints.
273 */
274 id = usbd_get_interface_descriptor(sc->cdce_data_iface);
275 cd = usbd_get_config_descriptor(sc->cdce_udev);
276 numalts = usbd_get_no_alts(cd, id->bInterfaceNumber);
277
278 for (j = 0; j < numalts; j++) {
279 if (usbd_set_interface(sc->cdce_data_iface, j)) {
280 aprint_error_dev(sc->cdce_dev,
281 "setting alternate interface failed\n");
282 return;
283 }
284 /* Find endpoints. */
285 id = usbd_get_interface_descriptor(sc->cdce_data_iface);
286 sc->cdce_bulkin_no = sc->cdce_bulkout_no = -1;
287 for (i = 0; i < id->bNumEndpoints; i++) {
288 ed = usbd_interface2endpoint_descriptor(sc->cdce_data_iface, i);
289 if (!ed) {
290 aprint_error_dev(self,
291 "could not read endpoint descriptor\n");
292 return;
293 }
294 if (UE_GET_DIR(ed->bEndpointAddress) == UE_DIR_IN &&
295 UE_GET_XFERTYPE(ed->bmAttributes) == UE_BULK) {
296 sc->cdce_bulkin_no = ed->bEndpointAddress;
297 } else if (UE_GET_DIR(ed->bEndpointAddress) == UE_DIR_OUT &&
298 UE_GET_XFERTYPE(ed->bmAttributes) == UE_BULK) {
299 sc->cdce_bulkout_no = ed->bEndpointAddress;
300 } else if (UE_GET_DIR(ed->bEndpointAddress) == UE_DIR_IN &&
301 UE_GET_XFERTYPE(ed->bmAttributes) == UE_INTERRUPT) {
302 /* XXX: CDC spec defines an interrupt pipe, but it is not
303 * needed for simple host-to-host applications. */
304 } else {
305 aprint_error_dev(self, "unexpected endpoint\n");
306 }
307 }
308 /* If we found something, try and use it... */
309 if ((sc->cdce_bulkin_no != -1) && (sc->cdce_bulkout_no != -1))
310 break;
311 }
312
313 if (sc->cdce_bulkin_no == -1) {
314 aprint_error_dev(self, "could not find data bulk in\n");
315 return;
316 }
317 if (sc->cdce_bulkout_no == -1 ) {
318 aprint_error_dev(self, "could not find data bulk out\n");
319 return;
320 }
321
322 ue = (const usb_cdc_ethernet_descriptor_t *)usb_find_desc(dev,
323 UDESC_CS_INTERFACE, UDESCSUB_CDC_ENF);
324 if (!ue || usbd_get_string(dev, ue->iMacAddress, eaddr_str) ||
325 ether_aton_r(eaddr, sizeof(eaddr), eaddr_str)) {
326 aprint_normal_dev(self, "faking address\n");
327 eaddr[0]= 0x2a;
328 memcpy(&eaddr[1], &hardclock_ticks, sizeof(uint32_t));
329 eaddr[5] = (uint8_t)(device_unit(sc->cdce_dev));
330 }
331
332 mutex_init(&sc->cdce_txlock, MUTEX_DEFAULT, IPL_SOFTUSB);
333 mutex_init(&sc->cdce_rxlock, MUTEX_DEFAULT, IPL_SOFTUSB);
334 mutex_init(&sc->cdce_lock, MUTEX_DEFAULT, IPL_NONE);
335
336 usb_init_task(&sc->cdce_tick_task, cdce_tick_task, sc, USB_TASKQ_MPSAFE);
337
338 aprint_normal_dev(self, "address %s\n", ether_sprintf(eaddr));
339
340 ifp = GET_IFP(sc);
341 ifp->if_softc = sc;
342 ifp->if_flags = IFF_BROADCAST | IFF_SIMPLEX | IFF_MULTICAST;
343 ifp->if_extflags = IFEF_MPSAFE;
344 ifp->if_ioctl = cdce_ioctl;
345 ifp->if_start = cdce_start;
346 strlcpy(ifp->if_xname, device_xname(sc->cdce_dev), IFNAMSIZ);
347
348 IFQ_SET_READY(&ifp->if_snd);
349
350 if_attach(ifp);
351 ether_ifattach(ifp, eaddr);
352
353 callout_init(&sc->cdce_stat_ch, CALLOUT_MPSAFE);
354 callout_setfunc(&sc->cdce_stat_ch, cdce_tick, sc);
355
356 sc->cdce_attached = true;
357
358 usbd_add_drv_event(USB_EVENT_DRIVER_ATTACH, sc->cdce_udev,
359 sc->cdce_dev);
360
361 if (!pmf_device_register(self, NULL, NULL))
362 aprint_error_dev(self, "couldn't establish power handler\n");
363
364 return;
365}

367int
368cdce_detach(device_t self, int flags)
369{
370 struct cdce_softc *sc = device_private(self);
371 struct ifnet *ifp = GET_IFP(sc);
372
373 mutex_enter(&sc->cdce_lock);
374 sc->cdce_dying = true;
375 mutex_exit(&sc->cdce_lock);
376
377 if (!sc->cdce_attached)
378 return 0;
379
380 pmf_device_deregister(self);
381
382 callout_halt(&sc->cdce_stat_ch, NULL);
383 usb_rem_task_wait(sc->cdce_udev, &sc->cdce_tick_task,
384 USB_TASKQ_DRIVER, NULL);
385
386 if (ifp->if_flags & IFF_RUNNING) {
387 IFNET_LOCK(ifp);
388 cdce_stop(sc);
389 IFNET_UNLOCK(ifp);
390 }
391
392 callout_destroy(&sc->cdce_stat_ch);
393 ether_ifdetach(ifp);
394 if_detach(ifp);
395 sc->cdce_attached = false;
396
397 usbd_add_drv_event(USB_EVENT_DRIVER_DETACH, sc->cdce_udev,sc->cdce_dev);
398
399 mutex_destroy(&sc->cdce_lock);
400 mutex_destroy(&sc->cdce_rxlock);
401 mutex_destroy(&sc->cdce_txlock);
402
403 return 0;
404}

406static void
407cdce_start_locked(struct ifnet *ifp)
408{
409 struct cdce_softc *sc = ifp->if_softc;
410 struct mbuf *m_head = NULL;
411
412 KASSERT(mutex_owned(&sc->cdce_txlock));
413
414 if (sc->cdce_dying || sc->cdce_cdata.cdce_tx_cnt == CDCE_TX_LIST_CNT)
415 return;
416
417 IFQ_POLL(&ifp->if_snd, m_head);
418 if (m_head == NULL)
419 return;
420
421 if (cdce_encap(sc, m_head, 0)) {
422 ifp->if_flags |= IFF_OACTIVE;
423 return;
424 }
425
426 IFQ_DEQUEUE(&ifp->if_snd, m_head);
427
428 bpf_mtap(ifp, m_head, BPF_D_OUT);
429
430 /*
431 * Set a timeout in case the chip goes out to lunch.
432 */
433 sc->cdce_timer = 6;
434}

436static void
437cdce_start(struct ifnet *ifp)
438{
439 struct cdce_softc * const sc = ifp->if_softc;
440
441 mutex_enter(&sc->cdce_txlock);
442 cdce_start_locked(ifp);
443 mutex_exit(&sc->cdce_txlock);
444}

446static int
447cdce_encap(struct cdce_softc *sc, struct mbuf *m, int idx)
448{
449 struct cdce_chain *c;
450 usbd_status err;
451 int extra = 0;
452
453 KASSERT(mutex_owned(&sc->cdce_txlock));
454
455 c = &sc->cdce_cdata.cdce_tx_chain[idx];
456
457 m_copydata(m, 0, m->m_pkthdr.len, c->cdce_buf);
458 if (sc->cdce_flags & CDCE_ZAURUS) {
459 /* Zaurus wants a 32-bit CRC appended to every frame */
460 uint32_t crc;
461
462 crc = htole32(~ether_crc32_le(c->cdce_buf, m->m_pkthdr.len));
463 memcpy(c->cdce_buf + m->m_pkthdr.len, &crc, sizeof(crc));
464 extra = sizeof(crc);
465 }
466 c->cdce_mbuf = m;
467
468 usbd_setup_xfer(c->cdce_xfer, c, c->cdce_buf, m->m_pkthdr.len + extra,
469 USBD_FORCE_SHORT_XFER, 10000, cdce_txeof);
470 err = usbd_transfer(c->cdce_xfer);
471 if (err != USBD_IN_PROGRESS) {
472 /* XXXSMP IFNET_LOCK */
473 cdce_stop(sc);
474 return EIO;
475 }
476
477 sc->cdce_cdata.cdce_tx_cnt++;
478 KASSERT(sc->cdce_cdata.cdce_tx_cnt <= CDCE_TX_LIST_CNT);
479
480 return 0;
481}

483static void
484cdce_stop_locked(struct cdce_softc *sc)
485{
486 usbd_status err;
487 struct ifnet *ifp = GET_IFP(sc);
488 int i;
489
490 /* XXXSMP can't KASSERT(IFNET_LOCKED(ifp)); */
491 KASSERT(mutex_owned(&sc->cdce_lock));
492
493 mutex_enter(&sc->cdce_rxlock);
494 mutex_enter(&sc->cdce_txlock);
495 sc->cdce_stopping = true;
496 mutex_exit(&sc->cdce_txlock);
497 mutex_exit(&sc->cdce_rxlock);
498
499 /*
500 * XXXSMP Would like to
501 * KASSERT(IFNET_LOCKED(ifp))
502 * here but the locking order is:
503 * ifnet -> sc lock -> rxlock -> txlock
504 * and sc lock is already held.
505 */
506 ifp->if_flags &= ~IFF_RUNNING;
507 sc->cdce_timer = 0;
508
509 callout_stop(&sc->cdce_stat_ch);
510
511 if (sc->cdce_bulkin_pipe != NULL) {
512 err = usbd_abort_pipe(sc->cdce_bulkin_pipe);
513 if (err)
514 printf("%s: abort rx pipe failed: %s\n",
515 device_xname(sc->cdce_dev), usbd_errstr(err));
516 }
517
518 if (sc->cdce_bulkout_pipe != NULL) {
519 err = usbd_abort_pipe(sc->cdce_bulkout_pipe);
520 if (err)
521 printf("%s: abort tx pipe failed: %s\n",
522 device_xname(sc->cdce_dev), usbd_errstr(err));
523 }
524
525 for (i = 0; i < CDCE_RX_LIST_CNT; i++) {
526 if (sc->cdce_cdata.cdce_rx_chain[i].cdce_mbuf != NULL) {
527 m_freem(sc->cdce_cdata.cdce_rx_chain[i].cdce_mbuf);
528 sc->cdce_cdata.cdce_rx_chain[i].cdce_mbuf = NULL;
529 }
530 if (sc->cdce_cdata.cdce_rx_chain[i].cdce_xfer != NULL) {
531 usbd_destroy_xfer
532 (sc->cdce_cdata.cdce_rx_chain[i].cdce_xfer);
533 sc->cdce_cdata.cdce_rx_chain[i].cdce_xfer = NULL;
534 }
535 }
536
537 for (i = 0; i < CDCE_TX_LIST_CNT; i++) {
538 if (sc->cdce_cdata.cdce_tx_chain[i].cdce_mbuf != NULL) {
539 m_freem(sc->cdce_cdata.cdce_tx_chain[i].cdce_mbuf);
540 sc->cdce_cdata.cdce_tx_chain[i].cdce_mbuf = NULL;
541 }
542 if (sc->cdce_cdata.cdce_tx_chain[i].cdce_xfer != NULL) {
543 usbd_destroy_xfer(
544 sc->cdce_cdata.cdce_tx_chain[i].cdce_xfer);
545 sc->cdce_cdata.cdce_tx_chain[i].cdce_xfer = NULL;
546 }
547 }
548
549 if (sc->cdce_bulkin_pipe != NULL) {
550 err = usbd_close_pipe(sc->cdce_bulkin_pipe);
551 if (err)
552 printf("%s: close rx pipe failed: %s\n",
553 device_xname(sc->cdce_dev), usbd_errstr(err));
554 sc->cdce_bulkin_pipe = NULL;
555 }
556
557 if (sc->cdce_bulkout_pipe != NULL) {
558 err = usbd_close_pipe(sc->cdce_bulkout_pipe);
559 if (err)
560 printf("%s: close tx pipe failed: %s\n",
561 device_xname(sc->cdce_dev), usbd_errstr(err));
562 sc->cdce_bulkout_pipe = NULL;
563 }
564}

575static int
576cdce_ioctl(struct ifnet *ifp, u_long command, void *data)
577{
578 struct cdce_softc *sc = ifp->if_softc;
579 struct ifaddr *ifa = (struct ifaddr *)data;
580 struct ifreq *ifr = (struct ifreq *)data;
581 int error = 0;
582
583 if (sc->cdce_dying)
584 return EIO;
585
586 switch(command) {
587 case SIOCINITIFADDR:
588 ifp->if_flags |= IFF_UP;
589 cdce_init(sc);
590 switch (ifa->ifa_addr->sa_family) {
591#ifdef INET
592 case AF_INET:
593 arp_ifinit(ifp, ifa);
594 break;
595#endif /* INET */
596 }
597 break;
598
599 case SIOCSIFMTU:
600 if (ifr->ifr_mtu < ETHERMIN || ifr->ifr_mtu > ETHERMTU)
601 error = EINVAL;
602 else if ((error = ifioctl_common(ifp, command, data)) == ENETRESET)
603 error = 0;
604 break;
605
606 case SIOCSIFFLAGS:
607 if ((error = ifioctl_common(ifp, command, data)) != 0)
608 break;
609 /* XXX re-use ether_ioctl() */
610 switch (ifp->if_flags & (IFF_UP|IFF_RUNNING)) {
611 case IFF_UP:
612 cdce_init(sc);
613 break;
614 case IFF_RUNNING:
615 cdce_stop(sc);
616 break;
617 default:
618 break;
619 }
620 break;
621
622 default:
623 error = ether_ioctl(ifp, command, data);
624 break;
625 }
626
627 if (error == ENETRESET)
628 error = 0;
629
630 return error;
631}

633static void
634cdce_watchdog(struct ifnet *ifp)
635{
636 struct cdce_softc *sc = ifp->if_softc;
637 struct cdce_chain *c;
638 usbd_status stat;
639
640 KASSERT(mutex_owned(&sc->cdce_lock));
641
642 if (sc->cdce_dying)
643 return;
644
645 ifp->if_oerrors++;
646 aprint_error_dev(sc->cdce_dev, "watchdog timeout\n");
647
648 mutex_enter(&sc->cdce_txlock);
649
650 c = &sc->cdce_cdata.cdce_rx_chain[0];
651 usbd_get_xfer_status(c->cdce_xfer, NULL, NULL, NULL, &stat);
652 cdce_txeof(c->cdce_xfer, c, stat);
653
654 if (!IFQ_IS_EMPTY(&ifp->if_snd))
655 cdce_start_locked(ifp);
656
657 mutex_exit(&sc->cdce_txlock);
658}

660static void
661cdce_init(void *xsc)
662{
663 struct cdce_softc *sc = xsc;
664 struct ifnet *ifp = GET_IFP(sc);
665 struct cdce_chain *c;
666 usbd_status err;
667 int i;
668
669 mutex_enter(&sc->cdce_lock);
670 if (ifp->if_flags & IFF_RUNNING)
671 goto out;
672
673 /* Maybe set multicast / broadcast here??? */
674
675 err = usbd_open_pipe(sc->cdce_data_iface, sc->cdce_bulkin_no,
676 USBD_EXCLUSIVE_USE | USBD_MPSAFE, &sc->cdce_bulkin_pipe);
677 if (err) {
678 printf("%s: open rx pipe failed: %s\n", device_xname(sc->cdce_dev),
679 usbd_errstr(err));
680 goto out;
681 }
682
683 err = usbd_open_pipe(sc->cdce_data_iface, sc->cdce_bulkout_no,
684 USBD_EXCLUSIVE_USE | USBD_MPSAFE, &sc->cdce_bulkout_pipe);
685 if (err) {
686 printf("%s: open tx pipe failed: %s\n",
687 device_xname(sc->cdce_dev), usbd_errstr(err));
688 goto out;
689 }
690
691 if (cdce_tx_list_init(sc)) {
692 printf("%s: tx list init failed\n", device_xname(sc->cdce_dev));
693 goto out;
694 }
695
696 if (cdce_rx_list_init(sc)) {
697 printf("%s: rx list init failed\n", device_xname(sc->cdce_dev));
698 goto out;
699 }
700
701 mutex_enter(&sc->cdce_rxlock);
702 mutex_enter(&sc->cdce_txlock);
703 sc->cdce_stopping = false;
704
705 for (i = 0; i < CDCE_RX_LIST_CNT; i++) {
706 c = &sc->cdce_cdata.cdce_rx_chain[i];
707
708 usbd_setup_xfer(c->cdce_xfer, c, c->cdce_buf, CDCE_BUFSZ,
709 USBD_SHORT_XFER_OK, USBD_NO_TIMEOUT, cdce_rxeof);
710 usbd_transfer(c->cdce_xfer);
711 }
712
713 mutex_exit(&sc->cdce_txlock);
714 mutex_exit(&sc->cdce_rxlock);
715
716 ifp->if_flags |= IFF_RUNNING;
717
718 callout_schedule(&sc->cdce_stat_ch, hz);
719
720out:
721 mutex_exit(&sc->cdce_lock);
722}

724static int
725cdce_newbuf(struct cdce_softc *sc, struct cdce_chain *c, struct mbuf *m)
726{
727 struct mbuf *m_new = NULL;
728
729 if (m == NULL) {
730 MGETHDR(m_new, M_DONTWAIT, MT_DATA);
731 if (m_new == NULL) {
732 printf("%s: no memory for rx list "
733 "-- packet dropped!\n", device_xname(sc->cdce_dev));
734 return ENOBUFS;
735 }
736 MCLGET(m_new, M_DONTWAIT);
737 if (!(m_new->m_flags & M_EXT)) {
738 printf("%s: no memory for rx list "
739 "-- packet dropped!\n", device_xname(sc->cdce_dev));
740 m_freem(m_new);
741 return ENOBUFS;
742 }
743 m_new->m_len = m_new->m_pkthdr.len = MCLBYTES;
744 } else {
745 m_new = m;
746 m_new->m_len = m_new->m_pkthdr.len = MCLBYTES;
747 m_new->m_data = m_new->m_ext.ext_buf;
748 }
749 c->cdce_mbuf = m_new;
750 return 0;
751}

753static int
754cdce_rx_list_init(struct cdce_softc *sc)
755{
756 struct cdce_cdata *cd;
757 struct cdce_chain *c;
758 int i;
759
760 cd = &sc->cdce_cdata;
761 for (i = 0; i < CDCE_RX_LIST_CNT; i++) {
762 c = &cd->cdce_rx_chain[i];
763 c->cdce_sc = sc;
764 c->cdce_idx = i;
765 if (cdce_newbuf(sc, c, NULL) == ENOBUFS)
766 return ENOBUFS;
767 if (c->cdce_xfer == NULL) {
768 int err = usbd_create_xfer(sc->cdce_bulkin_pipe,
769 CDCE_BUFSZ, 0, 0, &c->cdce_xfer);
770 if (err)
771 return err;
772 c->cdce_buf = usbd_get_buffer(c->cdce_xfer);
773 }
774 }
775
776 return 0;
777}

783 struct cdce_chain *c;

805static void
806cdce_rxeof(struct usbd_xfer *xfer, void *priv, usbd_status status)
807{
808 struct cdce_chain *c = priv;
809 struct cdce_softc *sc = c->cdce_sc;
810 struct ifnet *ifp = GET_IFP(sc);
811 struct mbuf *m;
812 int total_len = 0;
813
814 mutex_enter(&sc->cdce_rxlock);
815
816 if (sc->cdce_dying || sc->cdce_stopping ||
817 status == USBD_INVAL || status == USBD_NOT_STARTED ||
818 status == USBD_CANCELLED || !(ifp->if_flags & IFF_RUNNING)) {
819 mutex_exit(&sc->cdce_rxlock);
820 return;
821 }
822
823 if (status != USBD_NORMAL_COMPLETION) {
824 if (sc->cdce_rxeof_errors == 0)
825 printf("%s: usb error on rx: %s\n",
826 device_xname(sc->cdce_dev), usbd_errstr(status));
827 if (status == USBD_STALLED)
828 usbd_clear_endpoint_stall_async(sc->cdce_bulkin_pipe);
829 usbd_delay_ms(sc->cdce_udev, 10);
830 sc->cdce_rxeof_errors++;
831 goto done;
832 }
833
834 sc->cdce_rxeof_errors = 0;
835
836 usbd_get_xfer_status(xfer, NULL, NULL, &total_len, NULL);
837 if (sc->cdce_flags & CDCE_ZAURUS)
838 total_len -= 4; /* Strip off CRC added by Zaurus */
839 if (total_len <= 1) {
840 ifp->if_ierrors++;
841 goto done;
842 }
843
844 m = c->cdce_mbuf;
845 memcpy(mtod(m, char *), c->cdce_buf, total_len);
846
847 if (total_len < sizeof(struct ether_header)) {
848 ifp->if_ierrors++;
849 goto done;
850 }
851
852 m->m_pkthdr.len = m->m_len = total_len;
853 m_set_rcvif(m, ifp);
854
855 if (cdce_newbuf(sc, c, NULL) == ENOBUFS) {
856 ifp->if_ierrors++;
857 goto done;
858 }
859
860 mutex_exit(&sc->cdce_rxlock);
861 if_percpuq_enqueue(ifp->if_percpuq, m);
862 mutex_enter(&sc->cdce_rxlock);
863
864done:
865 if (sc->cdce_stopping || sc->cdce_dying) {
866 mutex_exit(&sc->cdce_rxlock);
867 return;
868 }
869
870 mutex_exit(&sc->cdce_rxlock);
871
872 /* Setup new transfer. */
873 usbd_setup_xfer(c->cdce_xfer, c, c->cdce_buf, CDCE_BUFSZ,
874 USBD_SHORT_XFER_OK, USBD_NO_TIMEOUT, cdce_rxeof);
875 usbd_transfer(c->cdce_xfer);
876}

878static void
879cdce_txeof(struct usbd_xfer *xfer, void *priv, usbd_status status)
880{
881 struct cdce_chain *c = priv;
882 struct cdce_softc *sc = c->cdce_sc;
883 struct ifnet *ifp = GET_IFP(sc);
884 usbd_status err;
885
886 mutex_enter(&sc->cdce_txlock);
887
888 if (sc->cdce_dying)
889 goto out;
890
891 KASSERT(sc->cdce_cdata.cdce_tx_cnt > 0);
892 sc->cdce_cdata.cdce_tx_cnt--;
893
894 sc->cdce_timer = 0;
895
896 switch (status) {
897 case USBD_NOT_STARTED:
898 case USBD_CANCELLED:
899 goto out;
900
901 case USBD_NORMAL_COMPLETION:
902 break;
903
904 default:
905 ifp->if_oerrors++;
906 printf("%s: usb error on tx: %s\n", device_xname(sc->cdce_dev),
907 usbd_errstr(status));
908 if (status == USBD_STALLED)
909 usbd_clear_endpoint_stall_async(sc->cdce_bulkout_pipe);
910 goto out;
911 }
912
913 usbd_get_xfer_status(c->cdce_xfer, NULL, NULL, NULL, &err);
914
915 if (c->cdce_mbuf != NULL) {
916 m_freem(c->cdce_mbuf);
917 c->cdce_mbuf = NULL;
918 }
919
920 if (err)
921 ifp->if_oerrors++;
922 else
923 ifp->if_opackets++;
924
925 if (IFQ_IS_EMPTY(&ifp->if_snd) == 0)
926 cdce_start_locked(ifp);
927
928out:
929 mutex_exit(&sc->cdce_txlock);
930}

932static void
933cdce_tick(void *xsc)
934{
935 struct cdce_softc * const sc = xsc;
936
937 if (sc == NULL)
938 return;
939
940 mutex_enter(&sc->cdce_lock);
941
942 if (sc->cdce_stopping || sc->cdce_dying) {
943 mutex_exit(&sc->cdce_lock);
944 return;
945 }
946
947 /* Perform periodic stuff in process context */
948 usb_add_task(sc->cdce_udev, &sc->cdce_tick_task, USB_TASKQ_DRIVER);
949
950 mutex_exit(&sc->cdce_lock);
951}
952
953static void
954cdce_tick_task(void *xsc)
955{
956 struct cdce_softc * const sc = xsc;
957 struct ifnet *ifp;
958
959 if (sc == NULL)
960 return;
961
962 mutex_enter(&sc->cdce_lock);
963
964 if (sc->cdce_stopping || sc->cdce_dying) {
965 mutex_exit(&sc->cdce_lock);
966 return;
967 }
968
969 ifp = GET_IFP(sc);
970
971 if (sc->cdce_timer != 0 && --sc->cdce_timer == 0)
972 cdce_watchdog(ifp);
973 if (!sc->cdce_stopping && !sc->cdce_dying)
974 callout_schedule(&sc->cdce_stat_ch, hz);
975
976 mutex_exit(&sc->cdce_lock);
977}

979int
980cdce_activate(device_t self, enum devact act)
981{
982 struct cdce_softc *sc = device_private(self);
983
984 switch (act) {
985 case DVACT_DEACTIVATE:
986 if_deactivate(GET_IFP(sc));
987
988 mutex_enter(&sc->cdce_lock);
989 sc->cdce_dying = true;
990 mutex_exit(&sc->cdce_lock);
991
992 mutex_enter(&sc->cdce_rxlock);
993 mutex_enter(&sc->cdce_txlock);
994 sc->cdce_stopping = true;
995 mutex_exit(&sc->cdce_txlock);
996 mutex_exit(&sc->cdce_rxlock);
997
998 return 0;
999 default:
1000 return EOPNOTSUPP;
1001 }
