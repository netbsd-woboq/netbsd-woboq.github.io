32/* USB vendor requests */
33#define MUE_UR_WRITEREG 0xa0
34#define MUE_UR_READREG 0xa1

41#define MUE_LED_CFG 0x018
42#define MUE_DP_SEL 0x024
43#define MUE_DP_CMD 0x028
44#define MUE_DP_ADDR 0x02c
45#define MUE_DP_DATA 0x030
46#define MUE_7500_BURST_CAP 0x034
47#define MUE_7500_INT_EP_CTL 0x038
48#define MUE_7500_BULKIN_DELAY 0x03c
49#define MUE_E2P_CMD 0x040
50#define MUE_E2P_DATA 0x044
51#define MUE_E2P_IND 0x0a5
52#define MUE_7500_RFE_CTL 0x060
53#define MUE_USB_CFG0 0x080
54#define MUE_USB_CFG1 0x084
55#define MUE_7500_FCT_RX_CTL 0x090
56#define MUE_7800_BURST_CAP 0x090
57#define MUE_7500_FCT_TX_CTL 0x094
58#define MUE_7800_BULKIN_DELAY 0x094
59#define MUE_7500_FCT_RX_FIFO_END 0x098
60#define MUE_7800_INT_EP_CTL 0x098
61#define MUE_7500_FCT_TX_FIFO_END 0x09c
62#define MUE_7500_FCT_FLOW 0x0a0
63#define MUE_7800_RFE_CTL 0x0b0
64#define MUE_7800_FCT_RX_CTL 0x0c0
65#define MUE_7800_FCT_TX_CTL 0x0c4
66#define MUE_7800_FCT_RX_FIFO_END 0x0c8
67#define MUE_7800_FCT_TX_FIFO_END 0x0cc
68#define MUE_7800_FCT_FLOW 0x0d0
69#define MUE_LTM_INDEX(idx) (0x0e0 + (idx) * 4)
70#define MUE_NUM_LTM_INDEX 6
71#define MUE_MAC_CR 0x100
72#define MUE_MAC_RX 0x104
73#define MUE_MAC_TX 0x108
74#define MUE_FLOW 0x10c
75#define MUE_RX_ADDRH 0x118
76#define MUE_RX_ADDRL 0x11c
77#define MUE_MII_ACCESS 0x120
78#define MUE_MII_DATA 0x124
79#define MUE_7500_ADDR_FILTX_BASE 0x300
80#define MUE_7500_ADDR_FILTX(i) (MUE_7500_ADDR_FILTX_BASE + 8 * (i))
81#define MUE_7800_ADDR_FILTX_BASE 0x400
82#define MUE_7800_ADDR_FILTX(i) (MUE_7800_ADDR_FILTX_BASE + 8 * (i))
83#define MUE_NUM_ADDR_FILTX 33

85/* device ID and revision register */
86#define MUE_ID_REV_ID __BITS(16,31)
87#define MUE_ID_REV_REV __BITS(0,15)
88
89/* hardware configuration register */
90#define MUE_HW_CFG_SRST 0x00000001
91#define MUE_HW_CFG_LRST 0x00000002
92#define MUE_HW_CFG_BCE 0x00000004
93#define MUE_HW_CFG_MEF 0x00000010
94#define MUE_HW_CFG_BIR 0x00000080
95#define MUE_HW_CFG_LED0_EN 0x00100000
96#define MUE_HW_CFG_LED1_EN 0x00200000
97
98/* power management control register */
99#define MUE_PMT_CTL_PHY_RST 0x00000010
100#define MUE_PMT_CTL_READY 0x00000080
101
102/* LED configuration register */
103#define MUE_LED_CFG_LEDGPIO_EN 0x0000f000
104#define MUE_LED_CFG_LED10_FUN_SEL 0x40000000
105#define MUE_LED_CFG_LED2_FUN_SEL 0x80000000
106
107/* data port select register */
108#define MUE_DP_SEL_RSEL_MASK 0x0000000f
109#define MUE_DP_SEL_VHF 0x00000001
110#define MUE_DP_SEL_DPRDY 0x80000000
111#define MUE_DP_SEL_VHF_HASH_LEN 16
112#define MUE_DP_SEL_VHF_VLAN_LEN 128
113
114/* data port command register */
115#define MUE_DP_CMD_WRITE 0x00000001
116
117/* burst cap register and etc */
118#define MUE_SS_USB_PKT_SIZE 1024
119#define MUE_HS_USB_PKT_SIZE 512
120#define MUE_FS_USB_PKT_SIZE 64
121#define MUE_7500_HS_RX_BUFSIZE \
122 (16 * 1024 + 5 * MUE_HS_USB_PKT_SIZE)
123#define MUE_7500_FS_RX_BUFSIZE \
124 (6 * 1024 + 33 * MUE_FS_USB_PKT_SIZE)
125#define MUE_7500_MAX_RX_FIFO_SIZE (20 * 1024)
126#define MUE_7500_MAX_TX_FIFO_SIZE (12 * 1024)
127#define MUE_7800_RX_BUFSIZE (12 * 1024)
128#define MUE_7800_MAX_RX_FIFO_SIZE MUE_7800_RX_BUFSIZE
129#define MUE_7800_MAX_TX_FIFO_SIZE MUE_7800_RX_BUFSIZE
130#define MUE_FRAME_LEN(mtu) \
131 (mtu + ETHER_HDR_LEN + ETHER_CRC_LEN + ETHER_VLAN_ENCAP_LEN)
132#define MUE_TSO_FRAME_LEN MUE_FRAME_LEN(IP_MAXPACKET)
133#define MUE_TX_BUFSIZE \
134 (sizeof(struct mue_txbuf_hdr) + MUE_TSO_FRAME_LEN)
135
136/* interrupt endpoint control register */
137#define MUE_INT_EP_CTL_PHY_INT 0x20000
138
139/* bulk-in delay register */
140#define MUE_7500_DEFAULT_BULKIN_DELAY 0x00002000
141#define MUE_7800_DEFAULT_BULKIN_DELAY 0x00000800
142
143/* EEPROM command register */
144#define MUE_E2P_CMD_ADDR_MASK 0x000001ff
145#define MUE_E2P_CMD_READ 0x00000000
146#define MUE_E2P_CMD_LOADED 0x00000200
147#define MUE_E2P_CMD_TIMEOUT 0x00000400
148#define MUE_E2P_CMD_BUSY 0x80000000
149#define MUE_E2P_IND_OFFSET 0x000
150#define MUE_E2P_MAC_OFFSET 0x001
151#define MUE_E2P_LTM_OFFSET 0x03f
152
153/* Receive Filtering Engine control register */
154#define MUE_RFE_CTL_PERFECT 0x00000002
155#define MUE_RFE_CTL_MULTICAST_HASH 0x00000008
156#define MUE_RFE_CTL_VLAN_FILTER 0x00000020
157#define MUE_RFE_CTL_UNICAST 0x00000100
158#define MUE_RFE_CTL_MULTICAST 0x00000200
159#define MUE_RFE_CTL_BROADCAST 0x00000400
160#define MUE_RFE_CTL_IP_COE 0x00000800
161#define MUE_RFE_CTL_TCPUDP_COE 0x00001000
162#define MUE_RFE_CTL_ICMP_COE 0x00002000
163#define MUE_RFE_CTL_IGMP_COE 0x00004000

165/* USB configuration register 0 */
166#define MUE_USB_CFG0_BCE 0x00000020
167#define MUE_USB_CFG0_BIR 0x00000040
168
169/* USB configuration register 1 */
170#define MUE_USB_CFG1_LTM_ENABLE 0x00000100

172#define MUE_USB_CFG1_DEV_U2_INIT_EN 0x00001000
173
174/* RX FIFO control register */
175#define MUE_FCT_RX_CTL_EN 0x80000000
176
177/* TX FIFO control register */
178#define MUE_FCT_TX_CTL_EN 0x80000000
179
180/* MAC control register */
181#define MUE_MAC_CR_RST 0x00000001
182#define MUE_MAC_CR_FULL_DUPLEX 0x00000008
183#define MUE_MAC_CR_AUTO_SPEED 0x00000800
184#define MUE_MAC_CR_AUTO_DUPLEX 0x00001000
185#define MUE_MAC_CR_GMII_EN 0x00080000
186
187/* MAC receive register */
188#define MUE_MAC_RX_RXEN 0x00000001
189#define MUE_MAC_RX_MAX_SIZE_MASK 0x3fff0000
190#define MUE_MAC_RX_MAX_SIZE_SHIFT 16
191#define MUE_MAC_RX_MAX_LEN(x) \
192 (((x) << MUE_MAC_RX_MAX_SIZE_SHIFT) & MUE_MAC_RX_MAX_SIZE_MASK)
193
194/* MAC transmit register */
195#define MUE_MAC_TX_TXEN 0x00000001
196
197/* flow control register */
198#define MUE_FLOW_PAUSE_TIME 0x0000ffff
199#define MUE_FLOW_RX_FCEN 0x20000000
200#define MUE_FLOW_TX_FCEN 0x40000000
201
202/* MII access register */
203#define MUE_MII_ACCESS_READ 0x00000000
204#define MUE_MII_ACCESS_BUSY 0x00000001
205#define MUE_MII_ACCESS_WRITE 0x00000002
206#define MUE_MII_ACCESS_REGADDR_MASK 0x000007c0
207#define MUE_MII_ACCESS_REGADDR_SHIFT 6
208#define MUE_MII_ACCESS_PHYADDR_MASK 0x0000f800
209#define MUE_MII_ACCESS_PHYADDR_SHIFT 11
210#define MUE_MII_ACCESS_REGADDR(x) \
211 (((x) << MUE_MII_ACCESS_REGADDR_SHIFT) & MUE_MII_ACCESS_REGADDR_MASK)
212#define MUE_MII_ACCESS_PHYADDR(x) \
213 (((x) << MUE_MII_ACCESS_PHYADDR_SHIFT) & MUE_MII_ACCESS_PHYADDR_MASK)
214
215/* MAC address perfect filter register */
216#define MUE_ADDR_FILTX_VALID 0x80000000

228#define MUE_OTP_ADDR3_MASK 0x03

242#define MUE_OTP_STATUS_BUSY 0x01
