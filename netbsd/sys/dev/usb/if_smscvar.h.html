32#ifndef _IF_SMSCVAR_H_
33#define _IF_SMSCVAR_H_
34
35#include <sys/rndsource.h>
36
37struct smsc_chain {
38 struct smsc_softc *sc_sc;
39 struct usbd_xfer *sc_xfer;
40 char *sc_buf;
41 struct mbuf *sc_mbuf;
42 int sc_accum;
43 int sc_idx;
44};
45
46struct smsc_cdata {
47 struct smsc_chain tx_chain[SMSC_TX_LIST_CNT];
48 struct smsc_chain rx_chain[SMSC_RX_LIST_CNT];
49 int tx_free;
50 int tx_next;
51 int rx_next;
52};
53
54struct smsc_softc {
55 device_t sc_dev;
56 struct usbd_device * sc_udev;
57 bool sc_dying;
58 bool sc_stopping;
59 bool sc_ttpending;
60
61 uint8_t sc_enaddr[ETHER_ADDR_LEN];
62 struct ethercom sc_ec;
63 struct mii_data sc_mii;
64 krndsource_t sc_rnd_source;
65 int sc_phyno;
66 struct usbd_interface * sc_iface;
67
68 /*
69 * The following stores the settings in the mac control (MAC_CSR)
70 * register
71 */
72 uint32_t sc_mac_csr;
73 uint32_t sc_rev_id;
74
75 uint32_t sc_coe_ctrl;
76
77 int sc_if_flags;
78 int sc_refcnt;
79
80 struct usb_task sc_tick_task;
81
82 int sc_ed[SMSC_ENDPT_MAX];
83 struct usbd_pipe * sc_ep[SMSC_ENDPT_MAX];
84
85 kmutex_t sc_lock;
86 kmutex_t sc_txlock;
87 kmutex_t sc_rxlock;
88 kmutex_t sc_mii_lock;
89 kcondvar_t sc_detachcv;
90
91 struct smsc_cdata sc_cdata;
92 callout_t sc_stat_ch;
93
94 struct timeval sc_rx_notice;
95 u_int sc_bufsz;
96
97 uint32_t sc_flags;
98#define SMSC_FLAG_LINK 0x0001
99
100 struct if_percpuq *sc_ipq; /* softint-based input queues */
101
102};
103
104#define SMSC_MIN_BUFSZ 2048
105#define SMSC_MAX_BUFSZ 18944
106
107#endif /* _IF_SMSCVAR_H_ */
