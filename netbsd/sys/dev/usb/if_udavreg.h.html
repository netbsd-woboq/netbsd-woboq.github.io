35#define UDAV_IFACE_INDEX 0
36#define UDAV_CONFIG_NO 1
37
38#define UDAV_TX_LIST_CNT 1
39#define UDAV_RX_LIST_CNT 1
40
41#define UDAV_TX_TIMEOUT 1000
42#define UDAV_TIMEOUT 10000
43
44
45/* Packet length */
46#define UDAV_MAX_MTU 1536 /* XXX: max frame size is unknown */
47#define UDAV_MIN_FRAME_LEN 60
48#define UDAV_BUFSZ UDAV_MAX_MTU
49
50/* Request */
51#define UDAV_REQ_REG_READ 0x00 /* Read from register(s) */
52#define UDAV_REQ_REG_WRITE 0x01 /* Write to register(s) */
53#define UDAV_REQ_REG_WRITE1 0x03 /* Write to a register */
54
55#define UDAV_REQ_MEM_READ 0x02 /* Read from memory */
56#define UDAV_REQ_MEM_WRITE 0x05 /* Write to memory */
57#define UDAV_REQ_MEM_WRITE1 0x07 /* Write a byte to memory */
58
59/* Registers */
60#define UDAV_NCR 0x00 /* Network Control Register */
61#define UDAV_NCR_EXT_PHY (1<<7) /* Select External PHY */
62#define UDAV_NCR_WAKEEN (1<<6) /* Wakeup Event Enable */
63#define UDAV_NCR_FCOL (1<<4) /* Force Collision Mode */
64#define UDAV_NCR_FDX (1<<3) /* Full-Duplex Mode (RO on Int. PHY) */
65#define UDAV_NCR_LBK1 (1<<2) /* Lookback Mode */
66#define UDAV_NCR_LBK0 (1<<1) /* Lookback Mode */
67#define UDAV_NCR_RST (1<<0) /* Software reset */
68
69#define UDAV_RCR 0x05 /* RX Control Register */
70#define UDAV_RCR_WTDIS (1<<6) /* Watchdog Timer Disable */
71#define UDAV_RCR_DIS_LONG (1<<5) /* Discard Long Packet(over 1522Byte) */
72#define UDAV_RCR_DIS_CRC (1<<4) /* Discard CRC Error Packet */
73#define UDAV_RCR_ALL (1<<3) /* Pass All Multicast */
74#define UDAV_RCR_RUNT (1<<2) /* Pass Runt Packet */
75#define UDAV_RCR_PRMSC (1<<1) /* Promiscuous Mode */
76#define UDAV_RCR_RXEN (1<<0) /* RX Enable */
77
78#define UDAV_RSR 0x06 /* RX Status Register */
79#define UDAV_RSR_RF (1<<7) /* Runt Frame */
80#define UDAV_RSR_MF (1<<6) /* Multicast Frame */
81#define UDAV_RSR_LCS (1<<5) /* Late Collision Seen */
82#define UDAV_RSR_RWTO (1<<4) /* Receive Watchdog Time-Out */
83#define UDAV_RSR_PLE (1<<3) /* Physical Layer Error */
84#define UDAV_RSR_AE (1<<2) /* Alignment Error */
85#define UDAV_RSR_CE (1<<1) /* CRC Error */
86#define UDAV_RSR_FOE (1<<0) /* FIFO Overflow Error */
87#define UDAV_RSR_ERR (UDAV_RSR_RF | UDAV_RSR_LCS | UDAV_RSR_RWTO |\
88 UDAV_RSR_PLE | UDAV_RSR_AE | UDAV_RSR_CE |\
89 UDAV_RSR_FOE)
90
91#define UDAV_EPCR 0x0b /* EEPROM & PHY Control Register */
92#define UDAV_EPCR_REEP (1<<5) /* Reload EEPROM */
93#define UDAV_EPCR_WEP (1<<4) /* Write EEPROM enable */
94#define UDAV_EPCR_EPOS (1<<3) /* EEPROM or PHY Operation Select */
95#define UDAV_EPCR_ERPRR (1<<2) /* EEPROM/PHY Register Read Command */
96#define UDAV_EPCR_ERPRW (1<<1) /* EEPROM/PHY Register Write Command */
97#define UDAV_EPCR_ERRE (1<<0) /* EEPROM/PHY Access Status */
98
99#define UDAV_EPAR 0x0c /* EEPROM & PHY Control Register */
100#define UDAV_EPAR_PHY_ADR1 (1<<7) /* PHY Address bit 1 */
101#define UDAV_EPAR_PHY_ADR0 (1<<6) /* PHY Address bit 0 */
102#define UDAV_EPAR_EROA (1<<0) /* EEPROM Word/PHY Register Address */
103#define UDAV_EPAR_EROA_MASK (0x1f) /* [5:0] */
104
105#define UDAV_EPDRL 0x0d /* EEPROM & PHY Data Register */
106#define UDAV_EPDRH 0x0e /* EEPROM & PHY Data Register */
107
108#define UDAV_PAR0 0x10 /* Ethernet Address, load from EEPROM */
109#define UDAV_PAR1 0x11 /* Ethernet Address, load from EEPROM */
110#define UDAV_PAR2 0x12 /* Ethernet Address, load from EEPROM */
111#define UDAV_PAR3 0x13 /* Ethernet Address, load from EEPROM */
112#define UDAV_PAR4 0x14 /* Ethernet Address, load from EEPROM */
113#define UDAV_PAR5 0x15 /* Ethernet Address, load from EEPROM */
114#define UDAV_PAR UDAV_PAR0
115
116#define UDAV_MAR0 0x16 /* Multicast Register */
117#define UDAV_MAR1 0x17 /* Multicast Register */
118#define UDAV_MAR2 0x18 /* Multicast Register */
119#define UDAV_MAR3 0x19 /* Multicast Register */
120#define UDAV_MAR4 0x1a /* Multicast Register */
121#define UDAV_MAR5 0x1b /* Multicast Register */
122#define UDAV_MAR6 0x1c /* Multicast Register */
123#define UDAV_MAR7 0x1d /* Multicast Register */
124#define UDAV_MAR UDAV_MAR0
125
126#define UDAV_GPCR 0x1e /* General purpose control register */
127#define UDAV_GPCR_GEP_CNTL6 (1<<6) /* General purpose control 6 */
128#define UDAV_GPCR_GEP_CNTL5 (1<<5) /* General purpose control 5 */
129#define UDAV_GPCR_GEP_CNTL4 (1<<4) /* General purpose control 4 */
130#define UDAV_GPCR_GEP_CNTL3 (1<<3) /* General purpose control 3 */
131#define UDAV_GPCR_GEP_CNTL2 (1<<2) /* General purpose control 2 */
132#define UDAV_GPCR_GEP_CNTL1 (1<<1) /* General purpose control 1 */
133#define UDAV_GPCR_GEP_CNTL0 (1<<0) /* General purpose control 0 */
134
135#define UDAV_GPR 0x1f /* General purpose register */
136#define UDAV_GPR_GEPIO6 (1<<6) /* General purpose 6 */
137#define UDAV_GPR_GEPIO5 (1<<5) /* General purpose 5 */
138#define UDAV_GPR_GEPIO4 (1<<4) /* General purpose 4 */
139#define UDAV_GPR_GEPIO3 (1<<3) /* General purpose 3 */
140#define UDAV_GPR_GEPIO2 (1<<2) /* General purpose 2 */
141#define UDAV_GPR_GEPIO1 (1<<1) /* General purpose 1 */
142#define UDAV_GPR_GEPIO0 (1<<0) /* General purpose 0 */

148 struct udav_softc *udav_sc;

155struct udav_cdata {
156 struct udav_chain udav_tx_chain[UDAV_TX_LIST_CNT];
157 struct udav_chain udav_rx_chain[UDAV_TX_LIST_CNT];
158#if 0

168struct udav_softc {
169 device_t sc_dev; /* base device */
170 struct usbd_device * sc_udev;
171
172 /* USB */
173 struct usbd_interface * sc_ctl_iface;
174 /* int sc_ctl_iface_no; */
175 int sc_bulkin_no; /* bulk in endpoint */
176 int sc_bulkout_no; /* bulk out endpoint */
177 int sc_intrin_no; /* intr in endpoint */
178 struct usbd_pipe * sc_pipe_rx;
179 struct usbd_pipe * sc_pipe_tx;
180 struct usbd_pipe * sc_pipe_intr;
181 struct callout sc_stat_ch;
182 u_int sc_rx_errs;
183 /* u_int sc_intr_errs; */
184 struct timeval sc_rx_notice;
185
186 /* Ethernet */
187 struct ethercom sc_ec; /* ethernet common */
188 struct mii_data sc_mii;
189 kmutex_t sc_mii_lock;
190 int sc_link;
191#define sc_media udav_mii.mii_media
192 krndsource_t rnd_source;
193 struct udav_cdata sc_cdata;
194
195 int sc_attached;
196 int sc_dying;
197 int sc_refcnt;
198
199 struct usb_task sc_tick_task;
200 struct usb_task sc_stop_task;
201
202 uint16_t sc_flags;
203};
