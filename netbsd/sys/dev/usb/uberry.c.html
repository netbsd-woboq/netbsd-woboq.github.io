75struct uberry_softc {
76 device_t sc_dev;
77 struct usbd_device * sc_udev;
78};

80/*
81 * Note that we do not attach to USB_PRODUCT_RIM_BLACKBERRY_PEARL_DUAL
82 * as we let umass claim the device instead.
83 */
84static const struct usb_devno uberry_devs[] = {
85 { USB_VENDOR_RIM, USB_PRODUCT_RIM_BLACKBERRY },
86 { USB_VENDOR_RIM, USB_PRODUCT_RIM_BLACKBERRY_PEARL },
87};
88
89#define uberry_lookup(v, p) usb_lookup(uberry_devs, v, p)
90#define UBERRY_CONFIG_NO 1

100static void
101uberry_cmd(struct uberry_softc *sc, uint8_t requestType, uint8_t reqno,
102 uint8_t value, uint8_t index, void *data, uint8_t length)
103{
104 usb_device_request_t req;
105 usbd_status err;
106
107 DPRINTF(("berry cmd type=%x, number=%x, value=%d, index=%d, len=%d\n",
108 requestType, reqno, value, index, length));
109 req.bmRequestType = requestType;
110 req.bRequest = reqno;
111 USETW(req.wValue, value);
112 USETW(req.wIndex, index);
113 USETW(req.wLength, length);
114
115 if ((err = usbd_do_request(sc->sc_udev, &req, data)) != 0)
116 aprint_error_dev(sc->sc_dev, "sending command failed %d\n",
117 err);
118}
119
120static void
121uberry_charge(struct uberry_softc *sc)
122{
123 char dummy[2];
124 usbd_status err;
125
126 if (sc->sc_udev->ud_power != USB_MAX_POWER) {
127 uberry_cmd(sc, UT_READ | UT_VENDOR, 0xa5, 0, 1, dummy, 2);
128 uberry_cmd(sc, UT_WRITE | UT_VENDOR, 0xa2, 0, 1, dummy, 0);
129 }
130
131 err = usbd_set_config_no(sc->sc_udev, UBERRY_CONFIG_NO, 1);
132 if (err) {
133 aprint_error_dev(sc->sc_dev, "failed to set configuration"
134 ", err=%s\n", usbd_errstr(err));
135 return;
136 }
137}
138
139/*
140 * Expose both the USB mass storage interface and the database access one
141 */
142static void
143uberry_dual_mode(struct uberry_softc *sc)
144{
145 char dummy[2];
146 usbd_status err;
147
148 uberry_cmd(sc, UT_READ | UT_VENDOR, 0xa9, 1, 1, dummy, 2);
149
150 err = usbd_set_config_no(sc->sc_udev, UBERRY_CONFIG_NO, 1);
151 if (err) {
152 aprint_error_dev(sc->sc_dev, "failed to set configuration"
153 ", err=%s\n", usbd_errstr(err));
154 return;
155 }
156}

159int
160uberry_match(device_t parent, cfdata_t match, void *aux)
161{
162 struct usb_attach_arg *uaa = aux;
163
164 DPRINTFN(50, ("uberry_match\n"));
165 return (uberry_lookup(uaa->uaa_vendor, uaa->uaa_product) != NULL ?
166 UMATCH_VENDOR_PRODUCT : UMATCH_NONE);
167}
168
169void
170uberry_attach(device_t parent, device_t self, void *aux)
171{
172 struct uberry_softc *sc = device_private(self);
173 struct usb_attach_arg *uaa = aux;
174 struct usbd_device * dev = uaa->uaa_device;
175 char *devinfop;
176
177 DPRINTFN(10,("uberry_attach: sc=%p\n", sc));
178
179 sc->sc_dev = self;
180 sc->sc_udev = dev;
181
182 aprint_naive("\n");
183 aprint_normal("\n");
184
185 devinfop = usbd_devinfo_alloc(dev, 0);
186 aprint_normal_dev(self, "%s\n", devinfop);
187 usbd_devinfo_free(devinfop);
188
189 uberry_charge(sc);
190 if (uaa->uaa_product == USB_PRODUCT_RIM_BLACKBERRY_PEARL)
191 uberry_dual_mode(sc);
192
193 DPRINTFN(10, ("uberry_attach: %p\n", sc->sc_udev));
194
195 if (!pmf_device_register(self, NULL, NULL))
196 aprint_error_dev(self, "couldn't establish power handler\n");
197
198 usbd_add_drv_event(USB_EVENT_DRIVER_ATTACH, sc->sc_udev, sc->sc_dev);
199 return;
200}
201
202int
203uberry_detach(device_t self, int flags)
204{
205 struct uberry_softc *sc = device_private(self);
206 DPRINTF(("uberry_detach: sc=%p flags=%d\n", sc, flags));
207
208 pmf_device_deregister(self);
209
210 usbd_add_drv_event(USB_EVENT_DRIVER_DETACH, sc->sc_udev, sc->sc_dev);
211
212 return 0;
213}
