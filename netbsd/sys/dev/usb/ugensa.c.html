63struct ugensa_softc {
64 device_t sc_dev; /* base device */
65 struct usbd_device * sc_udev; /* device */
66 struct usbd_interface * sc_iface; /* interface */
67
68 device_t sc_subdev;
69 int sc_numcon;
70
71 bool sc_dying;
72};

116static int
117ugensa_match(device_t parent, cfdata_t match, void *aux)
118{
119 struct usb_attach_arg *uaa = aux;
120
121 DPRINTFN(20,("ugensa: vendor=0x%x, product=0x%x\n",
122 uaa->uaa_vendor, uaa->uaa_product));
123
124 return ugensa_lookup(uaa->uaa_vendor, uaa->uaa_product) != NULL ?
125 UMATCH_VENDOR_PRODUCT : UMATCH_NONE;
126}

128static void
129ugensa_attach(device_t parent, device_t self, void *aux)
130{
131 struct ugensa_softc *sc = device_private(self);
132 struct usb_attach_arg *uaa = aux;
133 struct usbd_device *dev = uaa->uaa_device;
134 struct usbd_interface *iface;
135 usb_interface_descriptor_t *id;
136 usb_endpoint_descriptor_t *ed;
137 char *devinfop;
138 const char *devname = device_xname(self);
139 usbd_status err;
140 struct ucom_attach_args ucaa;
141 int i;
142
143 DPRINTFN(10,("\nugensa_attach: sc=%p\n", sc));
144
145 sc->sc_dev = self;
146 sc->sc_dying = false;
147
148 aprint_naive("\n");
149 aprint_normal("\n");
150
151 devinfop = usbd_devinfo_alloc(dev, 0);
152 aprint_normal_dev(self, "%s\n", devinfop);
153 usbd_devinfo_free(devinfop);
154
155 /* Move the device into the configured state. */
156 err = usbd_set_config_index(dev, UGENSA_CONFIG_INDEX, 1);
157 if (err) {
158 aprint_error("\n%s: failed to set configuration, err=%s\n",
159 devname, usbd_errstr(err));
160 goto bad;
161 }
162
163 err = usbd_device2interface_handle(dev, UGENSA_IFACE_INDEX, &iface);
164 if (err) {
165 aprint_error("\n%s: failed to get interface, err=%s\n",
166 devname, usbd_errstr(err));
167 goto bad;
168 }
169
170 if (ugensa_lookup(uaa->uaa_vendor, uaa->uaa_product)->ugensa_flags & UNTESTED)
171 aprint_normal_dev(self, "WARNING: This device is marked as "
172 "untested. Please submit a report via send-pr(1).\n");
173
174 id = usbd_get_interface_descriptor(iface);
175
176 sc->sc_udev = dev;
177 sc->sc_iface = iface;
178
179 ucaa.ucaa_info = "Generic Serial Device";
180 ucaa.ucaa_ibufsize = UGENSA_BUFSIZE;
181 ucaa.ucaa_obufsize = UGENSA_BUFSIZE;
182 ucaa.ucaa_ibufsizepad = UGENSA_BUFSIZE;
183 ucaa.ucaa_portno = UCOM_UNK_PORTNO;
184 ucaa.ucaa_opkthdrlen = 0;
185 ucaa.ucaa_device = dev;
186 ucaa.ucaa_iface = iface;
187 ucaa.ucaa_methods = &ugensa_methods;
188 ucaa.ucaa_arg = sc;
189
190 usbd_add_drv_event(USB_EVENT_DRIVER_ATTACH, sc->sc_udev, sc->sc_dev);
191
192 ucaa.ucaa_bulkin = ucaa.ucaa_bulkout = -1;
193 for (i = 0; i < id->bNumEndpoints; i++) {
194 int addr, dir, attr;
195
196 ed = usbd_interface2endpoint_descriptor(iface, i);
197 if (ed == NULL) {
198 aprint_error_dev(self,
199 "could not read endpoint descriptor: %s\n",
200 usbd_errstr(err));
201 goto bad;
202 }
203
204 addr = ed->bEndpointAddress;
205 dir = UE_GET_DIR(ed->bEndpointAddress);
206 attr = ed->bmAttributes & UE_XFERTYPE;
207 if (attr == UE_BULK) {
208 if (ucaa.ucaa_bulkin == -1 && dir == UE_DIR_IN) {
209 DPRINTF(("%s: Bulk in %d\n", devname, i));
210 ucaa.ucaa_bulkin = addr;
211 continue;
212 }
213 if (ucaa.ucaa_bulkout == -1 && dir == UE_DIR_OUT) {
214 DPRINTF(("%s: Bulk out %d\n", devname, i));
215 ucaa.ucaa_bulkout = addr;
216 continue;
217 }
218 }
219 aprint_error_dev(self, "unexpected endpoint\n");
220 }
221 if (ucaa.ucaa_bulkin == -1) {
222 aprint_error_dev(self, "Could not find data bulk in\n");
223 goto bad;
224 }
225 if (ucaa.ucaa_bulkout == -1) {
226 aprint_error_dev(self, "Could not find data bulk out\n");
227 goto bad;
228 }
229
230 DPRINTF(("ugensa: in=0x%x out=0x%x\n", ucaa.ucaa_bulkin,
231 ucaa.ucaa_bulkout));
232 sc->sc_subdev = config_found_sm_loc(self, "ucombus", NULL, &ucaa,
233 ucomprint, ucomsubmatch);
234
235 if (!pmf_device_register(self, NULL, NULL))
236 aprint_error_dev(self, "couldn't establish power handler\n");
237 return;
238
239bad:
240 DPRINTF(("ugensa_attach: ATTACH ERROR\n"));
241 sc->sc_dying = true;
242 return;
243}

245static void
246ugensa_childdet(device_t self, device_t child)
247{
248 struct ugensa_softc *sc = device_private(self);
249
250 KASSERT(sc->sc_subdev == child);
251 sc->sc_subdev = NULL;
252}
253
254static int
255ugensa_detach(device_t self, int flags)
256{
257 struct ugensa_softc *sc = device_private(self);
258 int rv = 0;
259
260 DPRINTF(("ugensa_detach: sc=%p flags=%d\n", sc, flags));
261
262 sc->sc_dying = true;
263
264 if (sc->sc_subdev != NULL) {
265 rv = config_detach(sc->sc_subdev, flags);
266 sc->sc_subdev = NULL;
267 }
268
269 usbd_add_drv_event(USB_EVENT_DRIVER_DETACH, sc->sc_udev, sc->sc_dev);
270
271 pmf_device_deregister(self);
272
273 return rv;
274}
