34#include <sys/rndsource.h>
35
36struct uhidev_softc {
37 device_t sc_dev; /* base device */
38 struct usbd_device *sc_udev;
39 struct usbd_interface *sc_iface; /* interface */
40 struct usbd_pipe *sc_ipipe; /* input interrupt pipe */
41 int sc_iep_addr;
42
43 u_char *sc_ibuf;
44 u_int sc_isize;
45
46 struct usbd_pipe *sc_opipe; /* output interrupt pipe */
47 struct usbd_xfer *sc_oxfer; /* write request */
48 int sc_oep_addr;
49
50 void *sc_repdesc;
51 int sc_repdesc_size;
52
53 u_int sc_nrepid;
54 device_t *sc_subdevs;
55
56 int sc_refcnt;
57 u_char sc_dying;
58
59 kmutex_t sc_lock; /* protects writes to sc_state */
60
61 u_int sc_flags;
62#define UHIDEV_F_XB1 0x0001 /* Xbox 1 controller */
63};
64
65struct uhidev {
66 device_t sc_dev; /* base device */
67 struct uhidev_softc *sc_parent;
68 uByte sc_report_id;
69 uint8_t sc_state;
70#define UHIDEV_OPEN 0x01 /* device is open */
71 int sc_in_rep_size;
72 void (*sc_intr)(struct uhidev *, void *, u_int);
73 krndsource_t rnd_source;
74};
75
76struct uhidev_attach_arg {
77 struct usbif_attach_arg *uiaa;
78 struct uhidev_softc *parent;
79 int reportid;
80 int reportsize;
81};

87usbd_status uhidev_set_report(struct uhidev *, int, void *, int);
