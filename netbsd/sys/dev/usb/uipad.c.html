107static void
108uipad_cmd(struct uipad_softc *sc, uint8_t requestType, uint8_t reqno,
109 uint16_t value, uint16_t index)
110{
111 usb_device_request_t req;
112 usbd_status err;
113
114 DPRINTF(("ipad cmd type=%x, number=%x, value=%d, index=%d\n",
115 requestType, reqno, value, index));
116 req.bmRequestType = requestType;
117 req.bRequest = reqno;
118 USETW(req.wValue, value);
119 USETW(req.wIndex, index);
120 USETW(req.wLength, 0);
121
122 if ((err = usbd_do_request(sc->sc_udev, &req, NULL)) != 0)
123 aprint_error_dev(sc->sc_dev, "sending command failed %d\n",
124 err);
125}
126
127static void
128uipad_charge(struct uipad_softc *sc)
129{
130 if (sc->sc_udev->ud_power != USB_MAX_POWER)
131 uipad_cmd(sc, UT_VENDOR | UT_WRITE, 0x40, 0x6400, 0x6400);
132}

134int
135uipad_match(device_t parent, cfdata_t match, void *aux)
136{
137 struct usb_attach_arg *uaa = aux;
138
139 DPRINTFN(50, ("uipad_match\n"));
140 return uipad_lookup(uaa->uaa_vendor, uaa->uaa_product) != NULL ?
141 UMATCH_VENDOR_PRODUCT : UMATCH_NONE;
142}
143
144void
145uipad_attach(device_t parent, device_t self, void *aux)
146{
147 struct uipad_softc *sc = device_private(self);
148 struct usb_attach_arg *uaa = aux;
149 struct usbd_device * dev = uaa->uaa_device;
150 char *devinfop;
151
152 DPRINTFN(10,("uipad_attach: sc=%p\n", sc));
153
154 sc->sc_dev = self;
155 sc->sc_udev = dev;
156
157 aprint_naive("\n");
158 aprint_normal("\n");
159
160 devinfop = usbd_devinfo_alloc(dev, 0);
161 aprint_normal_dev(self, "%s\n", devinfop);
162 usbd_devinfo_free(devinfop);
163
164 uipad_charge(sc);
165
166 DPRINTFN(10, ("uipad_attach: %p\n", sc->sc_udev));
167
168 if (!pmf_device_register(self, NULL, NULL))
169 aprint_error_dev(self, "couldn't establish power handler\n");
170
171 usbd_add_drv_event(USB_EVENT_DRIVER_ATTACH, sc->sc_udev, sc->sc_dev);
172 return;
173}
174
175int
176uipad_detach(device_t self, int flags)
177{
178 struct uipad_softc *sc = device_private(self);
179 DPRINTF(("uipad_detach: sc=%p flags=%d\n", sc, flags));
180
181 pmf_device_deregister(self);
182
183 usbd_add_drv_event(USB_EVENT_DRIVER_DETACH, sc->sc_udev, sc->sc_dev);
184
185 return 0;
186}
