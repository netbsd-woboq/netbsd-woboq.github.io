112static int
113ukyopon_match(device_t parent, cfdata_t match, void *aux)
114{
115 struct usbif_attach_arg *uiaa = aux;
116
117 if (uiaa->uiaa_vendor == USB_VENDOR_KYOCERA &&
118 uiaa->uiaa_product == USB_PRODUCT_KYOCERA_AHK3001V &&
119 (uiaa->uiaa_ifaceno == UKYOPON_MODEM_IFACE_INDEX ||
120 uiaa->uiaa_ifaceno == UKYOPON_DATA_IFACE_INDEX))
121 return UMATCH_VENDOR_PRODUCT;
122
123 return UMATCH_NONE;
124}
125
126static void
127ukyopon_attach(device_t parent, device_t self, void *aux)
128{
129 struct ukyopon_softc *sc = device_private(self);
130 struct usbif_attach_arg *uiaa = aux;
131 struct ucom_attach_args ucaa;
132
133 ucaa.ucaa_portno = (uiaa->uiaa_ifaceno == UKYOPON_MODEM_IFACE_INDEX) ?
134 UKYOPON_PORT_MODEM : UKYOPON_PORT_DATA;
135 ucaa.ucaa_methods = &ukyopon_methods;
136 ucaa.ucaa_info = (uiaa->uiaa_ifaceno == UKYOPON_MODEM_IFACE_INDEX) ?
137 "modem port" : "data transfer port";
138
139 if (umodem_common_attach(self, &sc->sc_umodem, uiaa, &ucaa))
140 return;
141 return;
142}

144static void
145ukyopon_get_status(void *addr, int portno, u_char *lsr, u_char *msr)
146{
147 struct ukyopon_softc *sc = addr;
148
149 /*
150 * The device doesn't set DCD (Data Carrier Detect) bit properly.
151 * Assume DCD is always present.
152 */
153 if ((sc->sc_umodem.sc_msr & UMSR_DCD) == 0)
154 sc->sc_umodem.sc_msr |= UMSR_DCD;
155
156 umodem_get_status(&sc->sc_umodem, portno, lsr, msr);
157}

192static int
193ukyopon_ioctl(void *addr, int portno, u_long cmd, void *data, int flag,
194 proc_t *p)
195{
196 struct ukyopon_softc *sc = addr;
197 struct ukyopon_identify *arg_id = (void*)data;
198 int error = 0;
199
200 switch (cmd) {
201 case UKYOPON_IDENTIFY:
202 strncpy(arg_id->ui_name, UKYOPON_NAME, sizeof(arg_id->ui_name));
203 arg_id->ui_busno =
204 device_unit(sc->sc_umodem.sc_udev->ud_bus->ub_usbctl);
205 arg_id->ui_address = sc->sc_umodem.sc_udev->ud_addr;
206 arg_id->ui_model = UKYOPON_MODEL_UNKNOWN;
207 arg_id->ui_porttype = portno;
208 break;
209
210 default:
211 error = umodem_ioctl(&sc->sc_umodem, portno, cmd, data, flag, p);
212 break;
213 }
214
215 return error;
216}
