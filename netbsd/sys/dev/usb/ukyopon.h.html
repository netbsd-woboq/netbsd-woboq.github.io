32#include <sys/ioccom.h>
33
34#ifdef _KERNEL
35#include <machine/limits.h>
36#else
37#include <limits.h>
38#endif
39
40struct ukyopon_identify {
41 char ui_name[16]; /* driver name */
42
43 int ui_busno; /* usb bus number */
44 uint8_t ui_address; /* device address */
45
46 enum ukyopon_model {
47 UKYOPON_MODEL_UNKNOWN,
48 /* UKYOPON_MODEL_AHK3001V, ... */
49 _UKYOPON_MODEL_KEEPSZ = INT_MAX /* fix size of this field */
50 } ui_model; /* possibly future use */
51 enum ukyopon_port {
52 UKYOPON_PORT_UNKNOWN,
53 UKYOPON_PORT_MODEM, /* modem port */
54 UKYOPON_PORT_DATA, /* data transfer port */
55 _UKYOPON_PORT_KEEPSZ = INT_MAX /* fix size of this field */
56 } ui_porttype; /* port type */
57 int ui_rsvd1, ui_rsvd2;
58};
59
60#define UKYOPON_NAME "ukyopon"
61#define UKYOPON_IDENTIFY _IOR ('U', 210, struct ukyopon_identify)
