145struct ata_cmd {
146 uint8_t ac_signature0;
147 uint8_t ac_signature1;
148
149 uint8_t ac_action_select;
150#define AC_ReadRegisterAccess 0x01
151#define AC_NoDeviceSelectionBit 0x02
152#define AC_NoBSYPollBit 0x04
153#define AC_IgnorePhaseErrorBit 0x08
154#define AC_IgnoreDeviceErrorBit 0x10
155
156 uint8_t ac_register_select;
157#define AC_SelectAlternateStatus 0x01 /* R */
158#define AC_SelectDeviceControl 0x01 /* W */
159#define AC_SelectError 0x02 /* R */
160#define AC_SelectFeatures 0x02 /* W */
161#define AC_SelectSectorCount 0x04 /* RW */
162#define AC_SelectSectorNumber 0x08 /* RW */
163#define AC_SelectCylinderLow 0x10 /* RW */
164#define AC_SelectCylinderHigh 0x20 /* RW */
165#define AC_SelectDeviceHead 0x40 /* RW */
166#define AC_SelectStatus 0x80 /* R */
167#define AC_SelectCommand 0x80 /* W */
168
169 uint8_t ac_transfer_blocksize;
170
171 uint8_t ac_alternate_status;
172#define ac_device_control ac_alternate_status
173 uint8_t ac_error;
174#define ac_features ac_error
175
176 uint8_t ac_sector_count;
177 uint8_t ac_sector_number;
178 uint8_t ac_cylinder_low;
179 uint8_t ac_cylinder_high;
180 uint8_t ac_device_head;
181
182 uint8_t ac_status;
183#define ac_command ac_status
184
185 uint8_t ac_reserved[3];
186};

190int
191umass_isdata_attach(struct umass_softc *sc)
192{
193 usb_device_request_t req;
194 usbd_status err;
195 struct ata_device adev;
196 struct uisdata_softc *scbus;
197 struct isd200_config *cf;
198
199 scbus = kmem_zalloc(sizeof(*scbus), KM_SLEEP);
200 sc->bus = &scbus->base;
201 cf = &scbus->sc_isd_config;
202
203 req.bmRequestType = UT_READ_VENDOR_DEVICE;
204 req.bRequest = 0x02;
205 USETW(req.wValue, 0);
206 USETW(req.wIndex, 2);
207 USETW(req.wLength, sizeof(*cf));
208
209 err = usbd_do_request(sc->sc_udev, &req, cf);
210 if (err) {
211 sc->bus = NULL;
212 kmem_free(scbus, sizeof(*scbus));
213 return EIO;
214 }
215 DPRINTF(("umass_wd_attach info:\n EventNotification=0x%02x "
216 "ExternalClock=0x%02x ATAInitTimeout=0x%02x\n"
217 " ATAMisc1=0x%02x ATAMajorCommand=0x%02x "
218 "ATAMinorCommand=0x%02x\n"
219 " ATAMisc2=0x%02x ATAMisc3=0x%02x\n",
220 cf->EventNotification, cf->ExternalClock, cf->ATAInitTimeout,
221 cf->ATAMisc1, cf->ATAMajorCommand, cf->ATAMinorCommand,
222 cf->ATAMisc2, cf->ATAMisc3));
223
224 memset(&adev, 0, sizeof(struct ata_device));
225 adev.adev_bustype = &uisdata_bustype;
226 adev.adev_channel = 1; /* XXX */
227 adev.adev_drv_data = &scbus->sc_drv_data;
228
229 /* Fake ATA channel so wd(4) ata_{get,free}_xfer() work */
230 ata_channel_init(&scbus->sc_channel);
231 scbus->sc_channel.atabus = (device_t)scbus;
232
233 scbus->sc_drv_data.drive_type = ATA_DRIVET_ATA;
234 scbus->sc_drv_data.chnl_softc = &scbus->sc_channel;
235
236 scbus->base.sc_child = config_found(sc->sc_dev, &adev, uwdprint);
237
238 return 0;
239}

252void
253uisdata_bio_cb(struct umass_softc *sc, void *priv, int residue, int status)
254{
255 struct uisdata_softc *scbus = (struct uisdata_softc *)sc->bus;
256 struct ata_xfer *xfer = priv;
257 struct ata_bio *ata_bio = &xfer->c_bio;
258 int s;
259
260 DPRINTF(("%s: residue=%d status=%d\n", __func__, residue, status));
261
262 s = splbio();
263 if (status != STATUS_CMD_OK)
264 ata_bio->error = ERR_DF; /* ??? */
265 else
266 ata_bio->error = NOERROR;
267 ata_bio->flags |= ATA_ITSDONE;
268
269 ata_bio->blkdone += ata_bio->nblks;
270 ata_bio->blkno += ata_bio->nblks;
271 ata_bio->bcount -= ata_bio->nbytes;
272 scbus->sc_skip += ata_bio->nbytes;
273 if (residue != 0) {
274 ata_bio->bcount += residue;
275 } else if (ata_bio->bcount > 0) {
276 DPRINTF(("%s: continue\n", __func__));
277 (void)uisdata_bio1(&scbus->sc_drv_data, xfer); /*XXX save drv*/
278 splx(s);
279 return;
280 }
281
282 if (ata_bio->flags & ATA_POLL) {
283 DPRINTF(("%s: wakeup %p\n", __func__, ata_bio));
284 wakeup(ata_bio);
285 } else {
286 (*scbus->sc_drv_data.drv_done)(scbus->sc_drv_data.drv_softc,
287 xfer);
288 }
289 splx(s);
290}

302int
303uisdata_bio1(struct ata_drive_datas *drv, struct ata_xfer *xfer)
304{
305 struct umass_softc *sc = CH2SELF(drv->chnl_softc);
306 struct uisdata_softc *scbus = (struct uisdata_softc *)sc->bus;
307 struct isd200_config *cf = &scbus->sc_isd_config;
308 struct ata_bio *ata_bio = &xfer->c_bio;
309 struct ata_cmd ata;
310 uint16_t cyl;
311 uint8_t head, sect;
312 int dir;
313 long nbytes;
314 u_int nblks;
315
316 DPRINTF(("%s\n", __func__));
317 /* XXX */
318
319 if (ata_bio->flags & ATA_POLL) {
320 printf("%s: ATA_POLL not supported\n", __func__);
321 ata_bio->error = TIMEOUT;
322 ata_bio->flags |= ATA_ITSDONE;
323 return ATACMD_COMPLETE;
324 }
325
326 if (ata_bio->flags & ATA_LBA) {
327 sect = (ata_bio->blkno >> 0) & 0xff;
328 cyl = (ata_bio->blkno >> 8) & 0xffff;
329 head = (ata_bio->blkno >> 24) & 0x0f;
330 head |= WDSD_LBA;
331 } else {
332 int blkno = ata_bio->blkno;
333 sect = blkno % drv->lp->d_nsectors;
334 sect++; /* Sectors begin with 1, not 0. */
335 blkno /= drv->lp->d_nsectors;
336 head = blkno % drv->lp->d_ntracks;
337 blkno /= drv->lp->d_ntracks;
338 cyl = blkno;
339 head |= WDSD_CHS;
340 }
341
342 nbytes = ata_bio->bcount;
343 if (ata_bio->flags & ATA_SINGLE)
344 nblks = 1;
345 else
346 nblks = uimin(drv->multi, nbytes / drv->lp->d_secsize);
347 nbytes = nblks * drv->lp->d_secsize;
348 ata_bio->nblks = nblks;
349 ata_bio->nbytes = nbytes;
350
351 memset(&ata, 0, sizeof(ata));
352 ata.ac_signature0 = cf->ATAMajorCommand;
353 ata.ac_signature1 = cf->ATAMinorCommand;
354 ata.ac_transfer_blocksize = 1;
355 ata.ac_sector_count = nblks;
356 ata.ac_sector_number = sect;
357 ata.ac_cylinder_high = cyl >> 8;
358 ata.ac_cylinder_low = cyl;
359 ata.ac_device_head = head;
360 ata.ac_register_select = AC_SelectSectorCount | AC_SelectSectorNumber |
361 AC_SelectCylinderLow | AC_SelectCylinderHigh | AC_SelectDeviceHead |
362 AC_SelectCommand;
363
364 dir = DIR_NONE;
365 if (ata_bio->bcount != 0) {
366 if (ata_bio->flags & ATA_READ)
367 dir = DIR_IN;
368 else
369 dir = DIR_OUT;
370 }
371
372 if (ata_bio->flags & ATA_READ) {
373 ata.ac_command = WDCC_READ;
374 } else {
375 ata.ac_command = WDCC_WRITE;
376 }
377 DPRINTF(("%s: bno=%" PRId64 " LBA=%d cyl=%d head=%d sect=%d "
378 "count=%d multi=%d\n",
379 __func__, ata_bio->blkno,
380 (ata_bio->flags & ATA_LBA) != 0, cyl, head, sect,
381 ata.ac_sector_count, drv->multi));
382 DPRINTF((" data=%p bcount=%ld, drive=%d\n", ata_bio->databuf,
383 ata_bio->bcount, drv->drive));
384 sc->sc_methods->wire_xfer(sc, drv->drive, &ata, sizeof(ata),
385 ata_bio->databuf + scbus->sc_skip, nbytes,
386 dir, ATA_DELAY, 0, uisdata_bio_cb, xfer);
387
388 while (ata_bio->flags & ATA_POLL) {
389 DPRINTF(("%s: tsleep %p\n", __func__, ata_bio));
390 if (tsleep(ata_bio, PZERO, "uisdatabl", 0)) {
391 ata_bio->error = TIMEOUT;
392 ata_bio->flags |= ATA_ITSDONE;
393 return ATACMD_COMPLETE;
394 }
395 }
396
397 return ata_bio->flags & ATA_ITSDONE ? ATACMD_COMPLETE : ATACMD_QUEUED;
398}

415void
416uisdata_exec_cb(struct umass_softc *sc, void *priv,
417 int residue, int status)
418{
419 struct ata_command *cmd = priv;
420
421 DPRINTF(("%s: status=%d\n", __func__, status));
422 if (status != STATUS_CMD_OK)
423 cmd->flags |= AT_DF; /* XXX */
424 cmd->flags |= AT_DONE;
425 if (cmd->flags & (AT_READ | AT_WRITE))
426 cmd->flags |= AT_XFDONE;
427 if (cmd->flags & (AT_POLL | AT_WAIT)) {
428 DPRINTF(("%s: wakeup %p\n", __func__, cmd));
429 wakeup(cmd);
430 }
431}

433int
434uisdata_exec_command(struct ata_drive_datas *drv, struct ata_xfer *xfer)
435{
436 struct umass_softc *sc = CH2SELF(drv->chnl_softc);
437 struct uisdata_softc *scbus = (struct uisdata_softc *)sc->bus;
438 struct isd200_config *cf = &scbus->sc_isd_config;
439 int dir;
440 struct ata_command *cmd = &xfer->c_ata_c;
441 struct ata_cmd ata;
442
443 DPRINTF(("%s\n", __func__));
444 DPRINTF((" r_command=0x%02x timeout=%d flags=0x%x bcount=%d\n",
445 cmd->r_command, cmd->timeout, cmd->flags, cmd->bcount));
446
447 dir = DIR_NONE;
448 if (cmd->bcount != 0) {
449 if (cmd->flags & AT_READ)
450 dir = DIR_IN;
451 else
452 dir = DIR_OUT;
453 }
454
455 if (cmd->bcount > UMASS_MAX_TRANSFER_SIZE) {
456 printf("uisdata_exec_command: large datalen %d\n", cmd->bcount);
457 cmd->flags |= AT_ERROR;
458 goto done;
459 }
460
461 memset(&ata, 0, sizeof(ata));
462 ata.ac_signature0 = cf->ATAMajorCommand;
463 ata.ac_signature1 = cf->ATAMinorCommand;
464 ata.ac_transfer_blocksize = 1;
465
466 switch (cmd->r_command) {
467 case WDCC_IDENTIFY:
468 ata.ac_register_select |= AC_SelectCommand;
469 ata.ac_command = WDCC_IDENTIFY;
470 break;
471 default:
472 printf("uisdata_exec_command: bad command 0x%02x\n",
473 cmd->r_command);
474 cmd->flags |= AT_ERROR;
475 goto done;
476 }
477
478 DPRINTF(("%s: execute ATA command 0x%02x, drive=%d\n", __func__,
479 ata.ac_command, drv->drive));
480 sc->sc_methods->wire_xfer(sc, drv->drive, &ata,
481 sizeof(ata), cmd->data, cmd->bcount, dir,
482 cmd->timeout, 0, uisdata_exec_cb, cmd);
483 if (cmd->flags & (AT_POLL | AT_WAIT)) {
484#if 0
485 if (cmd->flags & AT_POLL)
486 printf("%s: AT_POLL not supported\n", __func__);
487#endif
488 DPRINTF(("%s: tsleep %p\n", __func__, cmd));
489 if (tsleep(cmd, PZERO, "uisdataex", 0)) {
490 cmd->flags |= AT_ERROR;
491 goto done;
492 }
493 }
494
495done:
496 return ATACMD_COMPLETE;
497}

534int
535uisdata_get_params(struct ata_drive_datas *drvp, uint8_t flags,
536 struct ataparams *prms)
537{
538 char tb[DEV_BSIZE];
539 struct ata_xfer *xfer;
540 int rv;
541
542#if BYTE_ORDER == LITTLE_ENDIAN
543 int i;
544 uint16_t *p;
545#endif
546
547 DPRINTF(("%s\n", __func__));
548
549 memset(tb, 0, DEV_BSIZE);
550 memset(prms, 0, sizeof(struct ataparams));
551
552 xfer = ata_get_xfer(drvp->chnl_softc, false);
553 if (!xfer) {
554 rv = CMD_AGAIN;
555 goto out;
556 }
557
558 xfer->c_ata_c.r_command = WDCC_IDENTIFY;
559 xfer->c_ata_c.timeout = 1000; /* 1s */
560 xfer->c_ata_c.flags = AT_READ | flags;
561 xfer->c_ata_c.data = tb;
562 xfer->c_ata_c.bcount = DEV_BSIZE;
563 if (uisdata_exec_command(drvp, xfer) != ATACMD_COMPLETE) {
564 DPRINTF(("uisdata_get_parms: uisdata_exec_command failed\n"));
565 rv = CMD_AGAIN;
566 goto out;
567 }
568 if (xfer->c_ata_c.flags & (AT_ERROR | AT_TIMEOU | AT_DF)) {
569 DPRINTF(("uisdata_get_parms: ata_c.flags=0x%x\n",
570 xfer->c_ata_c.flags));
571 rv = CMD_ERR;
572 goto out;
573 }
574
575 /* Read in parameter block. */
576 memcpy(prms, tb, sizeof(struct ataparams));
577#if BYTE_ORDER == LITTLE_ENDIAN
578 /* XXX copied from ata.c */
579 /*
580 * Shuffle string byte order.
581 * ATAPI Mitsumi and NEC drives don't need this.
582 */
583 if (prms->atap_config != WDC_CFG_CFA_MAGIC &&
584 (prms->atap_config & WDC_CFG_ATAPI) &&
585 ((prms->atap_model[0] == 'N' &&
586 prms->atap_model[1] == 'E') ||
587 (prms->atap_model[0] == 'F' &&
588 prms->atap_model[1] == 'X'))) {
589 rv = 0;
590 goto out;
591 }
592 for (i = 0; i < sizeof(prms->atap_model); i += 2) {
593 p = (u_short *)(prms->atap_model + i);
594 *p = ntohs(*p);
595 }
596 for (i = 0; i < sizeof(prms->atap_serial); i += 2) {
597 p = (u_short *)(prms->atap_serial + i);
598 *p = ntohs(*p);
599 }
600 for (i = 0; i < sizeof(prms->atap_revision); i += 2) {
601 p = (u_short *)(prms->atap_revision + i);
602 *p = ntohs(*p);
603 }
604#endif
605 rv = CMD_OK;
606
607out:
608 ata_free_xfer(drvp->chnl_softc, xfer);
609 return rv;
610}
