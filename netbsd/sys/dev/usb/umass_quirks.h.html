33#ifndef _DEV_USB_UMASS_QUIRKS_H_
34#define _DEV_USB_UMASS_QUIRKS_H_
35
36typedef usbd_status (*umass_init_quirk)(struct umass_softc *);
37typedef void (*umass_fixup_quirk)(struct umass_softc *);
38
39struct umass_quirk {
40 struct usb_devno uq_dev;
41
42 uint8_t uq_wire;
43 uint8_t uq_cmd;
44 uint32_t uq_flags;
45 uint32_t uq_busquirks;
46 int uq_match;
47
48 umass_init_quirk uq_init;
49 umass_fixup_quirk uq_fixup;
50};
51
52const struct umass_quirk *umass_lookup(uint16_t, uint16_t);
53
54#endif /* _DEV_USB_UMASS_QUIRKS_H_ */
