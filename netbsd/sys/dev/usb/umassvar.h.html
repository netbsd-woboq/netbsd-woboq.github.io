77/* Command Block Wrapper */
78typedef struct {
79 uDWord dCBWSignature;
80#define CBWSIGNATURE 0x43425355
81 uDWord dCBWTag;
82 uDWord dCBWDataTransferLength;
83 uByte bCBWFlags;
84#define CBWFLAGS_OUT 0x00
85#define CBWFLAGS_IN 0x80
86 uByte bCBWLUN;
87 uByte bCDBLength;
88#define CBWCDBLENGTH 16
89 uByte CBWCDB[CBWCDBLENGTH];
90} umass_bbb_cbw_t;
91#define UMASS_BBB_CBW_SIZE 31
92
93/* Command Status Wrapper */
94typedef struct {
95 uDWord dCSWSignature;
96#define CSWSIGNATURE 0x53425355
97#define CSWSIGNATURE_OLYMPUS_C1 0x55425355
98 uDWord dCSWTag;
99 uDWord dCSWDataResidue;
100 uByte bCSWStatus;
101#define CSWSTATUS_GOOD 0x0
102#define CSWSTATUS_FAILED 0x1
103#define CSWSTATUS_PHASE 0x2
104} umass_bbb_csw_t;
105#define UMASS_BBB_CSW_SIZE 13
106
107/* CBI features */
108
109#define UR_CBI_ADSC 0x00
110
111typedef unsigned char umass_cbi_cbl_t[16]; /* Command block */
112
113typedef union {
114 struct {
115 uByte type;
116#define IDB_TYPE_CCI 0x00
117 uByte value;
118#define IDB_VALUE_PASS 0x00
119#define IDB_VALUE_FAIL 0x01
120#define IDB_VALUE_PHASE 0x02
121#define IDB_VALUE_PERSISTENT 0x03
122#define IDB_VALUE_STATUS_MASK 0x03
123 } common;
124
125 struct {
126 uByte asc;
127 uByte ascq;
128 } ufi;
129} umass_cbi_sbl_t;

131struct umass_softc; /* see below */
132
133typedef void (*umass_callback)(struct umass_softc *, void *, int, int);
134#define STATUS_CMD_OK 0 /* everything ok */
135#define STATUS_CMD_UNKNOWN 1 /* will have to fetch sense */
136#define STATUS_CMD_FAILED 2 /* transfer was ok, command failed */
137#define STATUS_WIRE_FAILED 3 /* couldn't even get command across */
138#define STATUS_TIMEOUT 4 /* transfer aborted */
139
140typedef void (*umass_wire_xfer)(struct umass_softc *, int, void *, int, void *,
141 int, int, u_int, int, umass_callback, void *);
142typedef void (*umass_wire_reset)(struct umass_softc *, int);
143typedef void (*umass_wire_state)(struct usbd_xfer *, void *,
144 usbd_status);

146struct umass_wire_methods {
147 umass_wire_xfer wire_xfer;
148 umass_wire_reset wire_reset;
149 umass_wire_state wire_state;
150};

156/* the per device structure */
157struct umass_softc {
158 device_t sc_dev; /* base device */
159 struct usbd_device * sc_udev; /* device */
160 struct usbd_interface * sc_iface; /* interface */
161 int sc_ifaceno; /* interface number */
162
163 uint8_t sc_epaddr[UMASS_NEP];
164 struct usbd_pipe * sc_pipe[UMASS_NEP];
165 usb_device_request_t sc_req;
166
167 const struct umass_wire_methods *sc_methods;
168
169 kmutex_t sc_lock;
170 kcondvar_t sc_detach_cv;
171
172 uint8_t sc_wire; /* wire protocol */
173#define UMASS_WPROTO_UNSPEC 0
174#define UMASS_WPROTO_BBB 1
175#define UMASS_WPROTO_CBI 2
176#define UMASS_WPROTO_CBI_I 3
177
178 uint8_t sc_cmd; /* command protocol */
179#define UMASS_CPROTO_UNSPEC 0
180#define UMASS_CPROTO_SCSI 1
181#define UMASS_CPROTO_ATAPI 2
182#define UMASS_CPROTO_UFI 3
183#define UMASS_CPROTO_RBC 4
184#define UMASS_CPROTO_ISD_ATA 5
185
186 uint32_t sc_quirks;
187#define UMASS_QUIRK_WRONG_CSWSIG 0x00000001
188#define UMASS_QUIRK_WRONG_CSWTAG 0x00000002
189#define UMASS_QUIRK_RBC_PAD_TO_12 0x00000004
190#define UMASS_QUIRK_NOGETMAXLUN 0x00000008
191
192#define UMASS_QUIRK_USE_DEFAULTMATCH -1
193
194 uint32_t sc_busquirks;
195
196 /* Bulk specific variables for transfers in progress */
197 umass_bbb_cbw_t cbw; /* command block wrapper */
198 umass_bbb_csw_t csw; /* command status wrapper*/
199 /* CBI specific variables for transfers in progress */
200 umass_cbi_cbl_t cbl; /* command block */
201 umass_cbi_sbl_t sbl; /* status block */
202
203 /* xfer handles
204 * Most of our operations are initiated from interrupt context, so
205 * we need to avoid using the one that is in use. We have to avoid
206 * allocating them in the interrupt context as well.
207 */
208 /* indices into array below */
209#define XFER_BBB_CBW 0 /* Bulk-Only */
210#define XFER_BBB_DATAIN 1
211#define XFER_BBB_DATAOUT 2
212#define XFER_BBB_DCLEAR 3
213#define XFER_BBB_CSW1 4
214#define XFER_BBB_CSW2 5
215#define XFER_BBB_SCLEAR 6
216#define XFER_BBB_RESET1 7
217#define XFER_BBB_RESET2 8
218#define XFER_BBB_RESET3 9
219
220#define XFER_CBI_CB 0 /* CBI */
221#define XFER_CBI_DATAIN 1
222#define XFER_CBI_DATAOUT 2
223#define XFER_CBI_STATUS 3
224#define XFER_CBI_DCLEAR 4
225#define XFER_CBI_SCLEAR 5
226#define XFER_CBI_RESET1 6
227#define XFER_CBI_RESET2 7
228#define XFER_CBI_RESET3 8
229
230#define XFER_NR 10 /* maximum number */
231
232 struct usbd_xfer *transfer_xfer[XFER_NR]; /* for ctrl xfers */
233
234 void *datain_buffer;
235 void *dataout_buffer;
236 void *cmd_buffer;
237 void *s1_buffer;
238 void *s2_buffer;
239
240 int transfer_dir; /* data direction */
241 void *transfer_data; /* data buffer */
242 int transfer_datalen; /* (maximum) length */
243 int transfer_actlen; /* actual length */
244 umass_callback transfer_cb; /* callback */
245 void *transfer_priv; /* for callback */
246 int transfer_status;
247
248 int transfer_state;
249#define TSTATE_IDLE 0
250#define TSTATE_BBB_COMMAND 1 /* CBW transfer */
251#define TSTATE_BBB_DATA 2 /* Data transfer */
252#define TSTATE_BBB_DCLEAR 3 /* clear endpt stall */
253#define TSTATE_BBB_STATUS1 4 /* clear endpt stall */
254#define TSTATE_BBB_SCLEAR 5 /* clear endpt stall */
255#define TSTATE_BBB_STATUS2 6 /* CSW transfer */
256#define TSTATE_BBB_RESET1 7 /* reset command */
257#define TSTATE_BBB_RESET2 8 /* in clear stall */
258#define TSTATE_BBB_RESET3 9 /* out clear stall */
259#define TSTATE_CBI_COMMAND 10 /* command transfer */
260#define TSTATE_CBI_DATA 11 /* data transfer */
261#define TSTATE_CBI_STATUS 12 /* status transfer */
262#define TSTATE_CBI_DCLEAR 13 /* clear ep stall */
263#define TSTATE_CBI_SCLEAR 14 /* clear ep stall */
264#define TSTATE_CBI_RESET1 15 /* reset command */
265#define TSTATE_CBI_RESET2 16 /* in clear stall */
266#define TSTATE_CBI_RESET3 17 /* out clear stall */
267#define TSTATE_STATES 18 /* # of states above */
268
269
270 int timeout; /* in msecs */
271
272 uint8_t maxlun; /* max lun supported */
273
274#ifdef UMASS_DEBUG
275 struct timeval tv;
276#endif

278 char sc_dying;
279 int sc_refcnt;
280 int sc_sense;
281
282 struct umassbus_softc *bus; /* bus dependent data */
283};
