91static int
92umodem_match(device_t parent, cfdata_t match, void *aux)
93{
94 struct usbif_attach_arg *uiaa = aux;
95 usb_interface_descriptor_t *id;
96 int cm, acm;
97
98 if (uiaa->uiaa_class != UICLASS_CDC ||
99 uiaa->uiaa_subclass != UISUBCLASS_ABSTRACT_CONTROL_MODEL ||
100 !(uiaa->uiaa_proto == UIPROTO_CDC_NOCLASS || uiaa->uiaa_proto == UIPROTO_CDC_AT))
101 return UMATCH_NONE;
102
103 id = usbd_get_interface_descriptor(uiaa->uiaa_iface);
104 if (umodem_get_caps(uiaa->uiaa_device, &cm, &acm, id) == -1)
105 return UMATCH_NONE;
106
107 return UMATCH_IFACECLASS_IFACESUBCLASS_IFACEPROTO;
108}
109
110static void
111umodem_attach(device_t parent, device_t self, void *aux)
112{
113 struct umodem_softc *sc = device_private(self);
114 struct usbif_attach_arg *uiaa = aux;
115 struct ucom_attach_args ucaa;
116
117 ucaa.ucaa_portno = UCOM_UNK_PORTNO;
118 ucaa.ucaa_methods = &umodem_methods;
119 ucaa.ucaa_info = NULL;
120
121 if (!pmf_device_register(self, NULL, NULL))
122 aprint_error_dev(self, "couldn't establish power handler");
123
124 if (umodem_common_attach(self, sc, uiaa, &ucaa))
125 return;
126 return;
127}
128
129static int
130umodem_detach(device_t self, int flags)
131{
132 struct umodem_softc *sc = device_private(self);
133
134 pmf_device_deregister(self);
135
136 return umodem_common_detach(sc, flags);
137}
