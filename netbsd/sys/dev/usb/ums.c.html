106int
107ums_match(device_t parent, cfdata_t match, void *aux)
108{
109 struct uhidev_attach_arg *uha = aux;
110 int size;
111 void *desc;
112
113 /*
114 * Some (older) Griffin PowerMate knobs may masquerade as a
115 * mouse, avoid treating them as such, they have only one axis.
116 */
117 if (uha->uiaa->uiaa_vendor == USB_VENDOR_GRIFFIN &&
118 uha->uiaa->uiaa_product == USB_PRODUCT_GRIFFIN_POWERMATE)
119 return UMATCH_NONE;
120
121 uhidev_get_report_desc(uha->parent, &desc, &size);
122 if (!hid_is_collection(desc, size, uha->reportid,
123 HID_USAGE2(HUP_GENERIC_DESKTOP, HUG_MOUSE)) &&
124 !hid_is_collection(desc, size, uha->reportid,
125 HID_USAGE2(HUP_GENERIC_DESKTOP, HUG_POINTER)) &&
126 !hid_is_collection(desc, size, uha->reportid,
127 HID_USAGE2(HUP_DIGITIZERS, 0x0002)))
128 return UMATCH_NONE;
129
130 return UMATCH_IFACECLASS;
131}

133void
134ums_attach(device_t parent, device_t self, void *aux)
135{
136 struct ums_softc *sc = device_private(self);
137 struct uhidev_attach_arg *uha = aux;
138 int size;
139 void *desc;
140 uint32_t quirks;
141
142 aprint_naive("\n");
143
144 sc->sc_hdev.sc_dev = self;
145 sc->sc_hdev.sc_intr = ums_intr;
146 sc->sc_hdev.sc_parent = uha->parent;
147 sc->sc_hdev.sc_report_id = uha->reportid;
148
149 quirks = usbd_get_quirks(uha->parent->sc_udev)->uq_flags;
150 if (quirks & UQ_MS_REVZ)
151 sc->sc_ms.flags |= HIDMS_REVZ;
152 if (quirks & UQ_SPUR_BUT_UP)
153 sc->sc_ms.flags |= HIDMS_SPUR_BUT_UP;
154
155 if (!pmf_device_register(self, NULL, NULL))
156 aprint_error_dev(self, "couldn't establish power handler\n");
157
158 uhidev_get_report_desc(uha->parent, &desc, &size);
159
160 if (!hidms_setup(self, &sc->sc_ms, uha->reportid, desc, size))
161 return;
162
163 if (uha->uiaa->uiaa_vendor == USB_VENDOR_MICROSOFT) {
164 int fixpos;
165 /*
166 * The Microsoft Wireless Laser Mouse 6000 v2.0 and the
167 * Microsoft Comfort Mouse 2.0 report a bad position for
168 * the wheel and wheel tilt controls -- should be in bytes
169 * 3 & 4 of the report. Fix this if necessary.
170 */
171 switch (uha->uiaa->uiaa_product) {
172 case USB_PRODUCT_MICROSOFT_24GHZ_XCVR10:
173 case USB_PRODUCT_MICROSOFT_24GHZ_XCVR20:
174 case USB_PRODUCT_MICROSOFT_NATURAL_6000:
175 fixpos = 24;
176 break;
177 case USB_PRODUCT_MICROSOFT_CM6000:
178 fixpos = 40;
179 break;
180 default:
181 fixpos = 0;
182 break;
183 }
184 if (fixpos) {
185 if ((sc->sc_ms.flags & HIDMS_Z) &&
186 sc->sc_ms.hidms_loc_z.pos == 0)
187 sc->sc_ms.hidms_loc_z.pos = fixpos;
188 if ((sc->sc_ms.flags & HIDMS_W) &&
189 sc->sc_ms.hidms_loc_w.pos == 0)
190 sc->sc_ms.hidms_loc_w.pos =
191 sc->sc_ms.hidms_loc_z.pos + 8;
192 }
193 }
194
195 hidms_attach(self, &sc->sc_ms, &ums_accessops);
196}

198int
199ums_activate(device_t self, enum devact act)
200{
201 struct ums_softc *sc = device_private(self);
202
203 switch (act) {
204 case DVACT_DEACTIVATE:
205 sc->sc_dying = 1;
206 return 0;
207 default:
208 return EOPNOTSUPP;
209 }
210}

212void
213ums_childdet(device_t self, device_t child)
214{
215 struct ums_softc *sc = device_private(self);
216
217 KASSERT(sc->sc_ms.hidms_wsmousedev == child);
218 sc->sc_ms.hidms_wsmousedev = NULL;
219}
220
221int
222ums_detach(device_t self, int flags)
223{
224 struct ums_softc *sc = device_private(self);
225 int rv = 0;
226
227 DPRINTF(("ums_detach: sc=%p flags=%d\n", sc, flags));
228
229 /* No need to do reference counting of ums, wsmouse has all the goo. */
230 if (sc->sc_ms.hidms_wsmousedev != NULL)
231 rv = config_detach(sc->sc_ms.hidms_wsmousedev, flags);
232
233 pmf_device_deregister(self);
234
235 return rv;
236}

238void
239ums_intr(struct uhidev *addr, void *ibuf, u_int len)
240{
241 struct ums_softc *sc = (struct ums_softc *)addr;
242 hidms_intr(&sc->sc_ms, ibuf, len);
243}

245Static int
246ums_enable(void *v)
247{
248 struct ums_softc *sc = v;
249 int error;
250
251 DPRINTFN(1,("ums_enable: sc=%p\n", sc));
252
253 if (sc->sc_dying)
254 return EIO;
255
256 if (sc->sc_enabled)
257 return EBUSY;
258
259 sc->sc_enabled = 1;
260 sc->sc_ms.hidms_buttons = 0;
261
262 error = uhidev_open(&sc->sc_hdev);
263 if (error)
264 sc->sc_enabled = 0;
265
266 return error;
267}
268
269Static void
270ums_disable(void *v)
271{
272 struct ums_softc *sc = v;
273
274 DPRINTFN(1,("ums_disable: sc=%p\n", sc));
275#ifdef DIAGNOSTIC
276 if (!sc->sc_enabled) {
277 printf("ums_disable: not enabled\n");
278 return;
279 }
280#endif
281
282 if (sc->sc_enabled) {
283 sc->sc_enabled = 0;
284 uhidev_close(&sc->sc_hdev);
285 }
286}
287
288Static int
289ums_ioctl(void *v, u_long cmd, void *data, int flag,
290 struct lwp * p)
291
292{
293 struct ums_softc *sc = v;
294
295 switch (cmd) {
296 case WSMOUSEIO_GTYPE:
297 if (sc->sc_ms.flags & HIDMS_ABS)
298 *(u_int *)data = WSMOUSE_TYPE_TPANEL;
299 else
300 *(u_int *)data = WSMOUSE_TYPE_USB;
301 return 0;
302 }
303
304 return EPASSTHROUGH;
305}
