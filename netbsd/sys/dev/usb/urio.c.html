105struct urio_softc {
106 device_t sc_dev;
107 struct usbd_device * sc_udev;
108 struct usbd_interface * sc_iface;
109
110 int sc_in_addr;
111 struct usbd_pipe * sc_in_pipe;

139int
140urio_match(device_t parent, cfdata_t match, void *aux)
141{
142 struct usb_attach_arg *uaa = aux;
143
144 DPRINTFN(50,("urio_match\n"));
145
146 return urio_lookup(uaa->uaa_vendor, uaa->uaa_product) != NULL ?
147 UMATCH_VENDOR_PRODUCT : UMATCH_NONE;
148}

150void
151urio_attach(device_t parent, device_t self, void *aux)
152{
153 struct urio_softc *sc = device_private(self);
154 struct usb_attach_arg *uaa = aux;
155 struct usbd_device * dev = uaa->uaa_device;
156 struct usbd_interface * iface;
157 char *devinfop;
158 usbd_status err;
159 usb_endpoint_descriptor_t *ed;
160 uint8_t epcount;
161 int i;
162
163 DPRINTFN(10,("urio_attach: sc=%p\n", sc));
164
165 sc->sc_dev = self;
166
167 aprint_naive("\n");
168 aprint_normal("\n");
169
170 devinfop = usbd_devinfo_alloc(dev, 0);
171 aprint_normal_dev(self, "%s\n", devinfop);
172 usbd_devinfo_free(devinfop);
173
174 err = usbd_set_config_no(dev, URIO_CONFIG_NO, 1);
175 if (err) {
176 aprint_error_dev(self, "failed to set configuration"
177 ", err=%s\n", usbd_errstr(err));
178 return;
179 }
180
181 err = usbd_device2interface_handle(dev, URIO_IFACE_IDX, &iface);
182 if (err) {
183 aprint_error_dev(self, "getting interface handle failed\n");
184 return;
185 }
186
187 sc->sc_udev = dev;
188 sc->sc_iface = iface;
189
190 epcount = 0;
191 (void)usbd_endpoint_count(iface, &epcount);
192
193 sc->sc_in_addr = -1;
194 sc->sc_out_addr = -1;
195 for (i = 0; i < epcount; i++) {
196 ed = usbd_interface2endpoint_descriptor(iface, i);
197 if (ed == NULL) {
198 aprint_error_dev(self, "couldn't get ep %d\n", i);
199 return;
200 }
201 if (UE_GET_DIR(ed->bEndpointAddress) == UE_DIR_IN &&
202 UE_GET_XFERTYPE(ed->bmAttributes) == UE_BULK) {
203 sc->sc_in_addr = ed->bEndpointAddress;
204 } else if (UE_GET_DIR(ed->bEndpointAddress) == UE_DIR_OUT &&
205 UE_GET_XFERTYPE(ed->bmAttributes) == UE_BULK) {
206 sc->sc_out_addr = ed->bEndpointAddress;
207 }
208 }
209 if (sc->sc_in_addr == -1 || sc->sc_out_addr == -1) {
210 aprint_error_dev(self, "missing endpoint\n");
211 return;
212 }
213
214 DPRINTFN(10, ("urio_attach: %p\n", sc->sc_udev));
215
216 usbd_add_drv_event(USB_EVENT_DRIVER_ATTACH, sc->sc_udev, sc->sc_dev);
217
218 return;
219}

221int
222urio_detach(device_t self, int flags)
223{
224 struct urio_softc *sc = device_private(self);
225 int s;
226 int maj, mn;
227
228 DPRINTF(("urio_detach: sc=%p flags=%d\n", sc, flags));
229
230 sc->sc_dying = 1;
231 /* Abort all pipes. Causes processes waiting for transfer to wake. */
232 if (sc->sc_in_pipe != NULL) {
233 usbd_abort_pipe(sc->sc_in_pipe);
234 usbd_close_pipe(sc->sc_in_pipe);
235 sc->sc_in_pipe = NULL;
236 }
237 if (sc->sc_out_pipe != NULL) {
238 usbd_abort_pipe(sc->sc_out_pipe);
239 usbd_close_pipe(sc->sc_out_pipe);
240 sc->sc_out_pipe = NULL;
241 }
242
243 s = splusb();
244 if (--sc->sc_refcnt >= 0) {
245 /* Wait for processes to go away. */
246 usb_detach_waitold(sc->sc_dev);
247 }
248 splx(s);
249
250 /* locate the major number */
251 maj = cdevsw_lookup_major(&urio_cdevsw);
252
253 /* Nuke the vnodes for any open instances (calls close). */
254 mn = device_unit(self);
255 vdevgone(maj, mn, mn, VCHR);
256
257 usbd_add_drv_event(USB_EVENT_DRIVER_DETACH, sc->sc_udev, sc->sc_dev);
258
259 return 0;
260}

262int
263urio_activate(device_t self, enum devact act)
264{
265 struct urio_softc *sc = device_private(self);
266
267 switch (act) {
268 case DVACT_DEACTIVATE:
269 sc->sc_dying = 1;
270 return 0;
271 default:
272 return EOPNOTSUPP;
273 }
274}

276int
277urioopen(dev_t dev, int flag, int mode, struct lwp *l)
278{
279 struct urio_softc *sc;
280 usbd_status err;
281
282 sc = device_lookup_private(&urio_cd, URIOUNIT(dev));
283 if (sc == NULL)
284 return ENXIO;
285
286 DPRINTFN(5, ("urioopen: flag=%d, mode=%d, unit=%d\n",
287 flag, mode, URIOUNIT(dev)));
288
289 if (sc->sc_dying)
290 return EIO;
291
292 if (sc->sc_in_pipe != NULL)
293 return EBUSY;
294
295 if ((flag & (FWRITE|FREAD)) != (FWRITE|FREAD))
296 return EACCES;
297
298 err = usbd_open_pipe(sc->sc_iface, sc->sc_in_addr, 0, &sc->sc_in_pipe);
299 if (err)
300 return EIO;
301 err = usbd_open_pipe(sc->sc_iface, sc->sc_out_addr,0,&sc->sc_out_pipe);
302 if (err) {
303 usbd_close_pipe(sc->sc_in_pipe);
304 sc->sc_in_pipe = NULL;
305 return EIO;
306 }
307
308 return 0;
309}

311int
312urioclose(dev_t dev, int flag, int mode,
313 struct lwp *l)
314{
315 struct urio_softc *sc;
316 sc = device_lookup_private(&urio_cd, URIOUNIT(dev));
317
318 DPRINTFN(5, ("urioclose: flag=%d, mode=%d, unit=%d\n",
319 flag, mode, URIOUNIT(dev)));
320
321 if (sc->sc_in_pipe != NULL) {
322 usbd_abort_pipe(sc->sc_in_pipe);
323 usbd_close_pipe(sc->sc_in_pipe);
324 sc->sc_in_pipe = NULL;
325 }
326 if (sc->sc_out_pipe != NULL) {
327 usbd_abort_pipe(sc->sc_out_pipe);
328 usbd_close_pipe(sc->sc_out_pipe);
329 sc->sc_out_pipe = NULL;
330 }
331
332 return 0;
333}

335int
336urioread(dev_t dev, struct uio *uio, int flag)
337{
338 struct urio_softc *sc;
339 struct usbd_xfer *xfer;
340 usbd_status err;
341 void *bufp;
342 uint32_t n, tn;
343 int error = 0;
344
345 sc = device_lookup_private(&urio_cd, URIOUNIT(dev));
346
347 DPRINTFN(5, ("urioread: %d\n", URIOUNIT(dev)));
348
349 if (sc->sc_dying)
350 return EIO;
351
352 error = usbd_create_xfer(sc->sc_in_pipe, URIO_BSIZE, 0, 0, &xfer);
353 if (error) {
354 return error;
355 }
356 bufp = usbd_get_buffer(xfer);
357
358 sc->sc_refcnt++;
359
360 while ((n = uimin(URIO_BSIZE, uio->uio_resid)) != 0) {
361 DPRINTFN(1, ("urioread: start transfer %d bytes\n", n));
362 tn = n;
363 err = usbd_bulk_transfer(xfer, sc->sc_in_pipe, 0,
364 URIO_RW_TIMEOUT, bufp, &tn);
365 if (err) {
366 if (err == USBD_INTERRUPTED)
367 error = EINTR;
368 else if (err == USBD_TIMEOUT)
369 error = ETIMEDOUT;
370 else
371 error = EIO;
372 break;
373 }
374
375 DPRINTFN(1, ("urioread: got %d bytes\n", tn));
376
377 error = uiomove(bufp, tn, uio);
378 if (error || tn < n)
379 break;
380 }
381 usbd_destroy_xfer(xfer);
382
383 if (--sc->sc_refcnt < 0)
384 usb_detach_wakeupold(sc->sc_dev);
385
386 return error;
387}

389int
390uriowrite(dev_t dev, struct uio *uio, int flag)
391{
392 struct urio_softc *sc;
393 struct usbd_xfer *xfer;
394 usbd_status err;
395 void *bufp;
396 uint32_t n;
397 int error = 0;
398
399 sc = device_lookup_private(&urio_cd, URIOUNIT(dev));
400
401 DPRINTFN(5, ("uriowrite: unit=%d, len=%ld\n", URIOUNIT(dev),
402 (long)uio->uio_resid));
403
404 if (sc->sc_dying)
405 return EIO;
406
407 error = usbd_create_xfer(sc->sc_out_pipe, URIO_BSIZE, 0, 0, &xfer);
408 if (error) {
409 return error;
410 }
411 bufp = usbd_get_buffer(xfer);
412 sc->sc_refcnt++;
413
414 while ((n = uimin(URIO_BSIZE, uio->uio_resid)) != 0) {
415 error = uiomove(bufp, n, uio);
416 if (error)
417 break;
418
419 DPRINTFN(1, ("uriowrite: transfer %d bytes\n", n));
420
421 err = usbd_bulk_transfer(xfer, sc->sc_out_pipe, 0,
422 URIO_RW_TIMEOUT, bufp, &n);
423 DPRINTFN(2, ("uriowrite: err=%d\n", err));
424 if (err) {
425 if (err == USBD_INTERRUPTED)
426 error = EINTR;
427 else if (err == USBD_TIMEOUT)
428 error = ETIMEDOUT;
429 else
430 error = EIO;
431 break;
432 }
433 }
434
435 usbd_destroy_xfer(xfer);
436
437 if (--sc->sc_refcnt < 0)
438 usb_detach_wakeupold(sc->sc_dev);
439
440 DPRINTFN(5, ("uriowrite: done unit=%d, error=%d\n", URIOUNIT(dev),
441 error));
442
443 return error;
444}

447int
448urioioctl(dev_t dev, u_long cmd, void *addr, int flag, struct lwp *l)
449{
450 struct urio_softc * sc;
451 int unit = URIOUNIT(dev);
452 struct urio_command *rcmd;
453 int requesttype, len;
454 struct iovec iov;
455 struct uio uio;
456 usb_device_request_t req;
457 usbd_status err;
458 int req_flags = 0;
459 uint32_t req_actlen = 0;
460 void *ptr = NULL;
461 int error = 0;
462
463 sc = device_lookup_private(&urio_cd, unit);
464
465 if (sc->sc_dying)
466 return EIO;
467
468 rcmd = (struct urio_command *)addr;
469
470 switch (cmd) {
471 case URIO_RECV_COMMAND:
472 requesttype = rcmd->requesttype | UT_READ_VENDOR_DEVICE;
473 break;
474
475 case URIO_SEND_COMMAND:
476 requesttype = rcmd->requesttype | UT_WRITE_VENDOR_DEVICE;
477 break;
478
479 default:
480 return EINVAL;
481 break;
482 }
483
484 if (!(flag & FWRITE))
485 return EPERM;
486 len = rcmd->length;
487
488 DPRINTFN(1,("urio_ioctl: cmd=0x%08lx reqtype=0x%0x req=0x%0x "
489 "value=0x%0x index=0x%0x len=0x%0x\n",
490 cmd, requesttype, rcmd->request, rcmd->value,
491 rcmd->index, len));
492
493 /* Send rio control message */
494 req.bmRequestType = requesttype;
495 req.bRequest = rcmd->request;
496 USETW(req.wValue, rcmd->value);
497 USETW(req.wIndex, rcmd->index);
498 USETW(req.wLength, len);
499
500 if (len < 0 || len > 32767)
501 return EINVAL;
502 if (len != 0) {
503 iov.iov_base = (void *)rcmd->buffer;
504 iov.iov_len = len;
505 uio.uio_iov = &iov;
506 uio.uio_iovcnt = 1;
507 uio.uio_resid = len;
508 uio.uio_offset = 0;
509 uio.uio_rw = req.bmRequestType & UT_READ ?
510 UIO_READ : UIO_WRITE;
511 uio.uio_vmspace = l->l_proc->p_vmspace;
512 ptr = kmem_alloc(len, KM_SLEEP);
513 if (uio.uio_rw == UIO_WRITE) {
514 error = uiomove(ptr, len, &uio);
515 if (error)
516 goto ret;
517 }
518 }
519
520 sc->sc_refcnt++;
521
522 err = usbd_do_request_flags(sc->sc_udev, &req, ptr, req_flags,
523 &req_actlen, USBD_DEFAULT_TIMEOUT);
524
525 if (--sc->sc_refcnt < 0)
526 usb_detach_wakeupold(sc->sc_dev);
527
528 if (err) {
529 error = EIO;
530 } else {
531 if (len != 0 && uio.uio_rw == UIO_READ)
532 error = uiomove(ptr, len, &uio);
533 }
534
535ret:
536 if (ptr != NULL)
537 kmem_free(ptr, len);
538 return error;
539}
