34typedef struct usb_dma_block {
35 bus_dma_tag_t tag;
36 bus_dmamap_t map;
37 void *kaddr;
38 bus_dma_segment_t *segs;
39 int nsegs;
40 int nsegs_alloc;
41 size_t size;
42 size_t align;
43 int flags;
44#define USB_DMA_FULLBLOCK 0x0001
45 LIST_ENTRY(usb_dma_block) next;
46} usb_dma_block_t;
