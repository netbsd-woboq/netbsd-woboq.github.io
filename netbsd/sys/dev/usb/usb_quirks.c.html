117usbd_status
118usbd_get_desc_fake(struct usbd_device *dev, int type, int index,
119 int len, void *desc)
120{
121 USBHIST_FUNC(); USBHIST_CALLED(usbdebug);
122#ifdef USB_DEBUG
123 const usb_device_descriptor_t *dd = usbd_get_device_descriptor(dev);
124#endif
125 const usb_descriptor_t *ub;
126 int i = 0;
127 int j = 0;
128 usbd_status err = USBD_INVAL;
129
130 if (dev->ud_quirks == NULL || dev->ud_quirks->desc == NULL) {
131 DPRINTF("%04x/%04x: no fake descriptors",
132 UGETW(dd->idVendor), UGETW(dd->idProduct), 0, 0);
133 goto out;
134 }
135
136 for (j = 0; dev->ud_quirks->desc[j]; j++) {
137 ub = dev->ud_quirks->desc[j];
138 if (ub->bDescriptorType == type && i++ == index)
139 break;
140 }
141
142 if (dev->ud_quirks->desc[j] == NULL) {
143 DPRINTF("%04x/%04x: no fake descriptor type = %d, len = %d",
144 UGETW(dd->idVendor), UGETW(dd->idProduct), type, len);
145 goto out;
146 }
147
148 do {
149 ub = dev->ud_quirks->desc[j];
150
151 if (ub->bLength > len) {
152 DPRINTF("%04x/%04x: short buf len = %d, bLength = %d",
153 UGETW(dd->idVendor), UGETW(dd->idProduct),
154 type, ub->bLength);
155 goto out;
156 }
157
158 memcpy(desc, ub, ub->bLength);
159 DPRINTF("%04x/%04x: Use fake descriptor type %d",
160 UGETW(dd->idVendor), UGETW(dd->idProduct),
161 type, 0);
162
163 desc = (char *)desc + ub->bLength;
164 len -= ub->bLength;
165 j++;
166 } while (len && dev->ud_quirks->desc[j] &&
167 dev->ud_quirks->desc[j]->bDescriptorType != type);
168
169 err = USBD_NORMAL_COMPLETION;
170
171 DPRINTF("%04x/%04x: Using fake USB descriptors\n",
172 UGETW(dd->idVendor), UGETW(dd->idProduct), 0, 0);
173out:
174 DPRINTF("return err = %d", err, 0, 0, 0);
175 return err;
176}

317 /*
318 * NXP PN533 bugs
319 *
320 * 1. It corrupts its USB descriptors. The quirk is to provide hardcoded
321 * descriptors instead of getting them from the device.
322 * 2. It mishandles the USB toggle bit. This causes some replies to be
323 * filered out by the USB host controller and be reported as timed out.
324 * NFC tool's libnfc workaround this bug by sending a dummy frame to
325 * resync the toggle bit, but in order to succeed, that operation must
326 * not be reported as failed. The quirk is therefore to pretend to
327 * userland that output timeouts are successes.
328 */

340const struct usbd_quirks *
341usbd_find_quirk(usb_device_descriptor_t *d)
342{
343 const struct usbd_quirk_entry *t;
344 uint16_t vendor = UGETW(d->idVendor);
345 uint16_t product = UGETW(d->idProduct);
346 uint16_t revision = UGETW(d->bcdDevice);
347
348 for (t = usb_quirks; t->idVendor != 0; t++) {
349 if (t->idVendor == vendor &&
350 (t->idProduct == ANY || t->idProduct == product) &&
351 (t->bcdDevice == ANY || t->bcdDevice == revision))
352 break;
353 }
354#ifdef USB_DEBUG
355 if (usbdebug && t->quirks.uq_flags)
356 printf("usbd_find_quirk 0x%04x/0x%04x/%x: %d\n",
357 UGETW(d->idVendor), UGETW(d->idProduct),
358 UGETW(d->bcdDevice), t->quirks.uq_flags);
359#endif
360 return &t->quirks;
361}
