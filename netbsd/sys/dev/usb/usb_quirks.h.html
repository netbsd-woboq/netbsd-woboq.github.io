34struct usbd_quirks {
35 uint32_t uq_flags; /* Device problems: */
36#define UQ_NO_SET_PROTO 0x0001 /* cannot handle SET PROTOCOL. */
37#define UQ_SWAP_UNICODE 0x0002 /* has some Unicode strings swapped. */
38#define UQ_MS_REVZ 0x0004 /* mouse has Z-axis reversed */
39#define UQ_NO_STRINGS 0x0008 /* string descriptors are broken. */
40#define UQ_BAD_ADC 0x0010 /* bad audio spec version number. */
41#define UQ_BAD_AUDIO 0x0040 /* device claims audio class, but isn't */
42#define UQ_SPUR_BUT_UP 0x0080 /* spurious mouse button up events */
43#define UQ_AU_NO_XU 0x0100 /* audio device has broken extension unit */
44#define UQ_AU_NO_FRAC 0x0400 /* don't adjust for fractional samples */
45#define UQ_AU_INP_ASYNC 0x0800 /* input is async despite claim of adaptive */
46#define UQ_ASSUME_CM_OVER_DATA 0x1000 /* modem device breaks on cm over data */
47#define UQ_BROKEN_BIDIR 0x2000 /* printer has broken bidir mode */
48#define UQ_HID_IGNORE 0x4000 /* device should be ignored by hid class */
49#define UQ_NO_UNION_NRM 0x8000 /* has no normal UNION descriptor */
50#define UQ_LOST_CS_DESC 0x10000 /* look everywhere for the CS descriptors */
51#define UQ_APPLE_ISO 0x20000 /* force ISO layout on Apple keyboards */
52#define UQ_DESC_CORRUPT 0x40000 /* may corrupt its config descriptors */
53#define UQ_MISS_OUT_ACK 0x80000 /* may fail to ack output */
54 const usb_descriptor_t **desc; /* Replacement for UQ_DESC_CORRUPT */
55};
