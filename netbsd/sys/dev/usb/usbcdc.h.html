34#ifndef _USBCDC_H_
35#define _USBCDC_H_

37#define UDESCSUB_CDC_HEADER 0
38#define UDESCSUB_CDC_CM 1 /* Call Management */
39#define UDESCSUB_CDC_ACM 2 /* Abstract Control Model */
40#define UDESCSUB_CDC_DLM 3 /* Direct Line Management */
41#define UDESCSUB_CDC_TRF 4 /* Telephone Ringer */
42#define UDESCSUB_CDC_TCLSR 5 /* Telephone Call ... */
43#define UDESCSUB_CDC_UNION 6
44#define UDESCSUB_CDC_CS 7 /* Country Selection */
45#define UDESCSUB_CDC_TOM 8 /* Telephone Operational Modes */
46#define UDESCSUB_CDC_USBT 9 /* USB Terminal */
47#define UDESCSUB_CDC_NCT 10
48#define UDESCSUB_CDC_PUF 11
49#define UDESCSUB_CDC_EUF 12
50#define UDESCSUB_CDC_MCMF 13
51#define UDESCSUB_CDC_CCMF 14
52#define UDESCSUB_CDC_ENF 15
53#define UDESCSUB_CDC_ANF 16
54
55typedef struct {
56 uByte bLength;
57 uByte bDescriptorType;
58 uByte bDescriptorSubtype;
59 uWord bcdCDC;
60} UPACKED usb_cdc_header_descriptor_t;
61
62typedef struct {
63 uByte bLength;
64 uByte bDescriptorType;
65 uByte bDescriptorSubtype;
66 uByte bmCapabilities;
67#define USB_CDC_CM_DOES_CM 0x01
68#define USB_CDC_CM_OVER_DATA 0x02
69 uByte bDataInterface;
70} UPACKED usb_cdc_cm_descriptor_t;
71
72typedef struct {
73 uByte bLength;
74 uByte bDescriptorType;
75 uByte bDescriptorSubtype;
76 uByte bmCapabilities;
77#define USB_CDC_ACM_HAS_FEATURE 0x01
78#define USB_CDC_ACM_HAS_LINE 0x02
79#define USB_CDC_ACM_HAS_BREAK 0x04
80#define USB_CDC_ACM_HAS_NETWORK_CONN 0x08
81} UPACKED usb_cdc_acm_descriptor_t;
82
83typedef struct {
84 uByte bLength;
85 uByte bDescriptorType;
86 uByte bDescriptorSubtype;
87 uByte bMasterInterface;
88 uByte bSlaveInterface[1];
89} UPACKED usb_cdc_union_descriptor_t;
90
91typedef struct {
92 uByte bLength;
93 uByte bDescriptorType;
94 uByte bDescriptorSubtype;
95 uByte iMacAddress;
96 uDWord bmEthernetStatistics;
97 uWord wMaxSegmentSize;
98 uWord wNumberMCFikters;
99 uByte bNumberPowerFilters;
100} UPACKED usb_cdc_ethernet_descriptor_t;
101
102#define UCDC_SEND_ENCAPSULATED_COMMAND 0x00
103#define UCDC_GET_ENCAPSULATED_RESPONSE 0x01
104#define UCDC_SET_COMM_FEATURE 0x02
105#define UCDC_GET_COMM_FEATURE 0x03
106#define UCDC_ABSTRACT_STATE 0x01
107#define UCDC_COUNTRY_SETTING 0x02
108#define UCDC_CLEAR_COMM_FEATURE 0x04
109#define UCDC_SET_LINE_CODING 0x20
110#define UCDC_GET_LINE_CODING 0x21
111#define UCDC_SET_CONTROL_LINE_STATE 0x22
112#define UCDC_LINE_DTR 0x0001
113#define UCDC_LINE_RTS 0x0002
114#define UCDC_SEND_BREAK 0x23
115#define UCDC_BREAK_ON 0xffff
116#define UCDC_BREAK_OFF 0x0000
117
118typedef struct {
119 uWord wState;
120#define UCDC_IDLE_SETTING 0x0001
121#define UCDC_DATA_MULTIPLEXED 0x0002
122} UPACKED usb_cdc_abstract_state_t;
123#define UCDC_ABSTRACT_STATE_LENGTH 2
124
125typedef struct {
126 uDWord dwDTERate;
127 uByte bCharFormat;
128#define UCDC_STOP_BIT_1 0
129#define UCDC_STOP_BIT_1_5 1
130#define UCDC_STOP_BIT_2 2
131 uByte bParityType;
132#define UCDC_PARITY_NONE 0
133#define UCDC_PARITY_ODD 1
134#define UCDC_PARITY_EVEN 2
135#define UCDC_PARITY_MARK 3
136#define UCDC_PARITY_SPACE 4
137 uByte bDataBits;
138} UPACKED usb_cdc_line_state_t;
139#define UCDC_LINE_STATE_LENGTH 7
140
141typedef struct {
142 uByte bmRequestType;
143#define UCDC_NOTIFICATION 0xa1
144 uByte bNotification;
145#define UCDC_N_NETWORK_CONNECTION 0x00
146#define UCDC_N_RESPONSE_AVAILABLE 0x01
147#define UCDC_N_AUX_JACK_HOOK_STATE 0x08
148#define UCDC_N_RING_DETECT 0x09
149#define UCDC_N_SERIAL_STATE 0x20
150#define UCDC_N_CALL_STATE_CHANGED 0x28
151#define UCDC_N_LINE_STATE_CHANGED 0x29
152#define UCDC_N_CONNECTION_SPEED_CHANGE 0x2a
153 uWord wValue;
154 uWord wIndex;
155 uWord wLength;
156 uByte data[16];
157} UPACKED usb_cdc_notification_t;
158#define UCDC_NOTIFICATION_LENGTH 8
159
160/*
161 * Bits set in the SERIAL STATE notifcation (first byte of data)

164#define UCDC_N_SERIAL_OVERRUN 0x40
165#define UCDC_N_SERIAL_PARITY 0x20
166#define UCDC_N_SERIAL_FRAMING 0x10
167#define UCDC_N_SERIAL_RI 0x08
168#define UCDC_N_SERIAL_BREAK 0x04
169#define UCDC_N_SERIAL_DSR 0x02
170#define UCDC_N_SERIAL_DCD 0x01

172#endif /* _USBCDC_H_ */
