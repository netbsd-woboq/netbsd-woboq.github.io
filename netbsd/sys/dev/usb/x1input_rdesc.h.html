30#define USBIF_IS_X1INPUT(uiaa) \
31 ((uiaa)->uiaa_class == UICLASS_VENDOR && \
32 (uiaa)->uiaa_subclass == 0x47 && \
33 (uiaa)->uiaa_proto == 0xd0)
