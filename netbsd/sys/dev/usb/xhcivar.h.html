29#ifndef _DEV_USB_XHCIVAR_H_
30#define _DEV_USB_XHCIVAR_H_
31
32#include <sys/pool.h>
33
34#define XHCI_XFER_NTRB 20
35
36struct xhci_soft_trb {
37 uint64_t trb_0;
38 uint32_t trb_2;
39 uint32_t trb_3;
40};
41
42struct xhci_xfer {
43 struct usbd_xfer xx_xfer;
44 struct xhci_soft_trb xx_trb[XHCI_XFER_NTRB];
45};
46
47#define XHCI_BUS2SC(bus) ((bus)->ub_hcpriv)
48#define XHCI_PIPE2SC(pipe) XHCI_BUS2SC((pipe)->up_dev->ud_bus)
49#define XHCI_XFER2SC(xfer) XHCI_BUS2SC((xfer)->ux_bus)
50#define XHCI_XFER2BUS(xfer) ((xfer)->ux_bus)
51#define XHCI_XPIPE2SC(d) XHCI_BUS2SC((d)->xp_pipe.up_dev->ud_bus)
52
53#define XHCI_XFER2XXFER(xfer) ((struct xhci_xfer *)(xfer))
54
55struct xhci_ring {
56 usb_dma_t xr_dma;
57 kmutex_t xr_lock;
58 struct xhci_trb * xr_trb;
59 void **xr_cookies;
60 u_int xr_ntrb; /* number of elements for above */
61 u_int xr_ep; /* enqueue pointer */
62 u_int xr_cs; /* cycle state */
63 bool is_halted;
64};
65
66struct xhci_endpoint {
67 struct xhci_ring xe_tr; /* transfer ring */
68};
69
70struct xhci_slot {
71 usb_dma_t xs_dc_dma; /* device context page */
72 usb_dma_t xs_ic_dma; /* input context page */
73 struct xhci_endpoint xs_ep[32]; /* endpoints */
74 u_int xs_idx; /* slot index */
75};
76
77struct xhci_softc {
78 device_t sc_dev;
79 device_t sc_child;
80 device_t sc_child2;
81 bus_size_t sc_ios;
82 bus_space_tag_t sc_iot;
83 bus_space_handle_t sc_ioh; /* Base */
84 bus_space_handle_t sc_cbh; /* Capability Base */
85 bus_space_handle_t sc_obh; /* Operational Base */
86 bus_space_handle_t sc_rbh; /* Runtime Base */
87 bus_space_handle_t sc_dbh; /* Doorbell Registers */
88 struct usbd_bus sc_bus; /* USB 3 bus */
89 struct usbd_bus sc_bus2; /* USB 2 bus */
90
91 kmutex_t sc_lock;
92 kmutex_t sc_intr_lock;
93
94 pool_cache_t sc_xferpool;
95
96 bus_size_t sc_pgsz; /* xHCI page size */
97 uint32_t sc_ctxsz;
98 int sc_maxslots;
99 int sc_maxintrs;
100 int sc_maxspbuf;
101
102 /*
103 * Port routing and root hub - xHCI 4.19.7
104 */
105 int sc_maxports; /* number of controller ports */
106
107 uint8_t *sc_ctlrportbus; /* a bus bit per port */
108
109 int *sc_ctlrportmap;
110 int *sc_rhportmap[2];
111 int sc_rhportcount[2];
112 struct usbd_xfer *sc_intrxfer[2];
113
114
115 struct xhci_slot * sc_slots;
116
117 struct xhci_ring sc_cr; /* command ring */
118 struct xhci_ring sc_er; /* event ring */
119
120 usb_dma_t sc_eventst_dma;
121 usb_dma_t sc_dcbaa_dma;
122 usb_dma_t sc_spbufarray_dma;
123 usb_dma_t *sc_spbuf_dma;
124
125 kcondvar_t sc_cmdbusy_cv;
126 kcondvar_t sc_command_cv;
127 bus_addr_t sc_command_addr;
128 struct xhci_soft_trb sc_result_trb;
129 bool sc_resultpending;
130
131 bool sc_ac64;
132 bool sc_dying;
133
134 void (*sc_vendor_init)(struct xhci_softc *);
135 int (*sc_vendor_port_status)(struct xhci_softc *, uint32_t, int);
136
137 int sc_quirks;
138#define XHCI_QUIRK_INTEL __BIT(0) /* Intel xhci chip */
139#define XHCI_DEFERRED_START __BIT(1)
140};
141
142int xhci_init(struct xhci_softc *);
143void xhci_start(struct xhci_softc *);
144int xhci_intr(void *);
145int xhci_detach(struct xhci_softc *, int);
146int xhci_activate(device_t, enum devact);
147void xhci_childdet(device_t, device_t);
148bool xhci_suspend(device_t, const pmf_qual_t *);
149bool xhci_resume(device_t, const pmf_qual_t *);
150bool xhci_shutdown(device_t, int);
151
152#define XHCI_TRANSFER_RING_TRBS 256
153
154#endif /* _DEV_USB_XHCIVAR_H_ */
