86static const char *
87edid_findvendor(const char *vendor)
88{
89#ifdef EDIDVERBOSE
90 int n;
91
92 for (n = 0; n < edid_nvendors; n++)
93 if (memcmp(edid_vendors[n].vendor, vendor, 3) == 0)
94 return edid_vendors[n].name;
95#endif
96 return NULL;
97}

99static const char *
100edid_findproduct(const char *vendor, uint16_t product)
101{
102#ifdef EDIDVERBOSE
103 int n;
104
105 for (n = 0; n < edid_nproducts; n++)
106 if (edid_products[n].product == product &&
107 memcmp(edid_products[n].vendor, vendor, 3) == 0)
108 return edid_products[n].name;
109#endif /* EDIDVERBOSE */
110 return NULL;
111
112}

114static void
115edid_strchomp(char *ptr)
116{
117 for (;;) {
118 switch (*ptr) {
119 case '\0':
120 return;
121 case '\r':
122 case '\n':
123 *ptr = '\0';
124 return;
125 }
126 ptr++;
127 }

130int
131edid_is_valid(uint8_t *d)
132{
133 int sum = 0, i;
134 uint8_t sig[8] = EDID_SIGNATURE;
135
136 if (memcmp(d, sig, 8) != 0)
137 return EINVAL;
138
139 for (i = 0; i < 128; i++)
140 sum += d[i];
141 if ((sum & 0xff) != 0)
142 return EINVAL;
143
144 return 0;
145}

293static const struct videomode *
294edid_mode_lookup_list(const char *name)
295{
296 int i;
297
298 for (i = 0; i < videomode_count; i++)
299 if (strcmp(name, videomode_list[i].name) == 0)
300 return &videomode_list[i];
301 return NULL;
302}

304static struct videomode *
305edid_search_mode(struct edid_info *edid, const struct videomode *mode)
306{
307 int refresh, i;
308
309 refresh = DIVIDE(DIVIDE(mode->dot_clock * 1000,
310 mode->htotal), mode->vtotal);
311 for (i = 0; i < edid->edid_nmodes; i++) {
312 if (mode->hdisplay == edid->edid_modes[i].hdisplay &&
313 mode->vdisplay == edid->edid_modes[i].vdisplay &&
314 refresh == DIVIDE(DIVIDE(
315 edid->edid_modes[i].dot_clock * 1000,
316 edid->edid_modes[i].htotal), edid->edid_modes[i].vtotal)) {
317 return &edid->edid_modes[i];
318 }
319 }
320 return NULL;
321}

323static int
324edid_std_timing(uint8_t *data, struct videomode *vmp)
325{
326 unsigned x, y, f;
327 const struct videomode *lookup;
328 char name[80];
329
330 if ((data[0] == 1 && data[1] == 1) ||
331 (data[0] == 0 && data[1] == 0) ||
332 (data[0] == 0x20 && data[1] == 0x20))
333 return 0;
334
335 x = EDID_STD_TIMING_HRES(data);
336 switch (EDID_STD_TIMING_RATIO(data)) {
337 case EDID_STD_TIMING_RATIO_16_10:
338 y = x * 10 / 16;
339 break;
340 case EDID_STD_TIMING_RATIO_4_3:
341 y = x * 3 / 4;
342 break;
343 case EDID_STD_TIMING_RATIO_5_4:
344 y = x * 4 / 5;
345 break;
346 case EDID_STD_TIMING_RATIO_16_9:
347 default:
348 y = x * 9 / 16;
349 break;
350 }
351 f = EDID_STD_TIMING_VFREQ(data);
352
353 /* first try to lookup the mode as a DMT timing */
354 snprintf(name, sizeof(name), "%dx%dx%d", x, y, f);
355 if ((lookup = edid_mode_lookup_list(name)) != NULL) {
356 *vmp = *lookup;
357 } else {
358 /* failing that, calculate it using gtf */
359 /*
360 * Hmm. I'm not using alternate GTF timings, which
361 * could, in theory, be present.
362 */
363 vesagtf_mode(x, y, f, vmp);
364 }
365 return 1;
366}

368static int
369edid_det_timing(uint8_t *data, struct videomode *vmp)
370{
371 unsigned hactive, hblank, hsyncwid, hsyncoff;
372 unsigned vactive, vblank, vsyncwid, vsyncoff;
373 uint8_t flags;
374
375 flags = EDID_DET_TIMING_FLAGS(data);
376
377 /* we don't support stereo modes (for now) */
378 if (flags & (EDID_DET_TIMING_FLAG_STEREO |
379 EDID_DET_TIMING_FLAG_STEREO_MODE))
380 return 0;
381
382 vmp->dot_clock = EDID_DET_TIMING_DOT_CLOCK(data) / 1000;
383
384 hactive = EDID_DET_TIMING_HACTIVE(data);
385 hblank = EDID_DET_TIMING_HBLANK(data);
386 hsyncwid = EDID_DET_TIMING_HSYNC_WIDTH(data);
387 hsyncoff = EDID_DET_TIMING_HSYNC_OFFSET(data);
388
389 vactive = EDID_DET_TIMING_VACTIVE(data);
390 vblank = EDID_DET_TIMING_VBLANK(data);
391 vsyncwid = EDID_DET_TIMING_VSYNC_WIDTH(data);
392 vsyncoff = EDID_DET_TIMING_VSYNC_OFFSET(data);
393
394 /* Borders are contained within the blank areas. */
395
396 vmp->hdisplay = hactive;
397 vmp->htotal = hactive + hblank;
398 vmp->hsync_start = hactive + hsyncoff;
399 vmp->hsync_end = vmp->hsync_start + hsyncwid;
400
401 vmp->vdisplay = vactive;
402 vmp->vtotal = vactive + vblank;
403 vmp->vsync_start = vactive + vsyncoff;
404 vmp->vsync_end = vmp->vsync_start + vsyncwid;
405
406 vmp->flags = 0;
407
408 if (flags & EDID_DET_TIMING_FLAG_INTERLACE)
409 vmp->flags |= VID_INTERLACE;
410 if (flags & EDID_DET_TIMING_FLAG_HSYNC_POSITIVE)
411 vmp->flags |= VID_PHSYNC;
412 else
413 vmp->flags |= VID_NHSYNC;
414
415 if (flags & EDID_DET_TIMING_FLAG_VSYNC_POSITIVE)
416 vmp->flags |= VID_PVSYNC;
417 else
418 vmp->flags |= VID_NVSYNC;
419
420 return 1;
421}

423static void bump_preferred_mode(struct edid_info *edid, struct videomode *m)
424{
425 /*
426 * XXX
427 * Iiyama 4800 series monitors may have their native resolution in the
428 * 2nd detailed timing descriptor instead of the 1st. Try to detect
429 * that here and pick the native mode anyway.
430 */
431 if (edid->edid_preferred_mode == NULL) {
432 edid->edid_preferred_mode = m;
433 } else if ((strncmp(edid->edid_vendor, "IVM", 3) == 0) &&
434 (edid->edid_product == 0x4800) &&
435 (edid->edid_preferred_mode->dot_clock < m->dot_clock))
436 edid->edid_preferred_mode = m;
437}

439static void
440edid_block(struct edid_info *edid, uint8_t *data)
441{
442 int i;
443 struct videomode mode, *exist_mode;
444
445 if (EDID_BLOCK_IS_DET_TIMING(data)) {
446 if (!edid_det_timing(data, &mode))
447 return;
448 /* Does this mode already exist? */
449 exist_mode = edid_search_mode(edid, &mode);
450 if (exist_mode != NULL) {
451 *exist_mode = mode;
452 bump_preferred_mode(edid, exist_mode);
453 } else {
454 edid->edid_modes[edid->edid_nmodes] = mode;
455 bump_preferred_mode(edid,
456 &edid->edid_modes[edid->edid_nmodes]);
457 edid->edid_nmodes++;
458 }
459 return;
460 }
461
462 switch (EDID_BLOCK_TYPE(data)) {
463 case EDID_DESC_BLOCK_TYPE_SERIAL:
464 memcpy(edid->edid_serial, data + EDID_DESC_ASCII_DATA_OFFSET,
465 EDID_DESC_ASCII_DATA_LEN);
466 edid->edid_serial[sizeof(edid->edid_serial) - 1] = 0;
467 break;
468
469 case EDID_DESC_BLOCK_TYPE_ASCII:
470 memcpy(edid->edid_comment, data + EDID_DESC_ASCII_DATA_OFFSET,
471 EDID_DESC_ASCII_DATA_LEN);
472 edid->edid_comment[sizeof(edid->edid_comment) - 1] = 0;
473 break;
474
475 case EDID_DESC_BLOCK_TYPE_RANGE:
476 edid->edid_have_range = 1;
477 edid->edid_range.er_min_vfreq = EDID_DESC_RANGE_MIN_VFREQ(data);
478 edid->edid_range.er_max_vfreq = EDID_DESC_RANGE_MAX_VFREQ(data);
479 edid->edid_range.er_min_hfreq = EDID_DESC_RANGE_MIN_HFREQ(data);
480 edid->edid_range.er_max_hfreq = EDID_DESC_RANGE_MAX_HFREQ(data);
481 edid->edid_range.er_max_clock = EDID_DESC_RANGE_MAX_CLOCK(data);
482 if (!EDID_DESC_RANGE_HAVE_GTF2(data))
483 break;
484 edid->edid_range.er_have_gtf2 = 1;
485 edid->edid_range.er_gtf2_hfreq =
486 EDID_DESC_RANGE_GTF2_HFREQ(data);
487 edid->edid_range.er_gtf2_c = EDID_DESC_RANGE_GTF2_C(data);
488 edid->edid_range.er_gtf2_m = EDID_DESC_RANGE_GTF2_M(data);
489 edid->edid_range.er_gtf2_j = EDID_DESC_RANGE_GTF2_J(data);
490 edid->edid_range.er_gtf2_k = EDID_DESC_RANGE_GTF2_K(data);
491 break;
492
493 case EDID_DESC_BLOCK_TYPE_NAME:
494 /* copy the product name into place */
495 memcpy(edid->edid_productname,
496 data + EDID_DESC_ASCII_DATA_OFFSET,
497 EDID_DESC_ASCII_DATA_LEN);
498 break;
499
500 case EDID_DESC_BLOCK_TYPE_STD_TIMING:
501 data += EDID_DESC_STD_TIMING_START;
502 for (i = 0; i < EDID_DESC_STD_TIMING_COUNT; i++) {
503 if (edid_std_timing(data, &mode)) {
504 /* Does this mode already exist? */
505 exist_mode = edid_search_mode(edid, &mode);
506 if (exist_mode == NULL) {
507 edid->edid_modes[edid->edid_nmodes] =
508 mode;
509 edid->edid_nmodes++;
510 }
511 }
512 data += 2;
513 }
514 break;
515
516 case EDID_DESC_BLOCK_TYPE_COLOR_POINT:
517 /* XXX: not implemented yet */
518 break;
519 }
520}

522/*
523 * Gets EDID version in BCD, e.g. EDID v1.3 returned as 0x0103
524 */
525int
526edid_parse(uint8_t *data, struct edid_info *edid)
527{
528 uint16_t manfid, estmodes;
529 const struct videomode *vmp;
530 int i;
531 const char *name;
532 int max_dotclock = 0;
533 int mhz;
534
535 if (edid_is_valid(data) != 0)
536 return -1;
537
538 /* get product identification */
539 manfid = EDID_VENDOR_ID(data);
540 edid->edid_vendor[0] = EDID_MANFID_0(manfid);
541 edid->edid_vendor[1] = EDID_MANFID_1(manfid);
542 edid->edid_vendor[2] = EDID_MANFID_2(manfid);
543 edid->edid_vendor[3] = 0; /* null terminate for convenience */
544
545 edid->edid_product = data[EDID_OFFSET_PRODUCT_ID] +
546 (data[EDID_OFFSET_PRODUCT_ID + 1] << 8);
547
548 name = edid_findvendor(edid->edid_vendor);
549 if (name != NULL)
550 strlcpy(edid->edid_vendorname, name,
551 sizeof(edid->edid_vendorname));
552 else
553 edid->edid_vendorname[0] = '\0';
554
555 name = edid_findproduct(edid->edid_vendor, edid->edid_product);
556 if (name != NULL)
557 strlcpy(edid->edid_productname, name,
558 sizeof(edid->edid_productname));
559 else
560 edid->edid_productname[0] = '\0';
561
562 snprintf(edid->edid_serial, sizeof(edid->edid_serial), "%08x",
563 EDID_SERIAL_NUMBER(data));
564
565 edid->edid_week = EDID_WEEK(data);
566 edid->edid_year = EDID_YEAR(data);
567
568 /* get edid revision */
569 edid->edid_version = EDID_VERSION(data);
570 edid->edid_revision = EDID_REVISION(data);
571
572 edid->edid_video_input = EDID_VIDEO_INPUT(data);
573 edid->edid_max_hsize = EDID_MAX_HSIZE(data);
574 edid->edid_max_vsize = EDID_MAX_VSIZE(data);
575
576 edid->edid_gamma = EDID_GAMMA(data);
577 edid->edid_features = EDID_FEATURES(data);
578
579 edid->edid_chroma.ec_redx = EDID_CHROMA_REDX(data);
580 edid->edid_chroma.ec_redy = EDID_CHROMA_REDX(data);
581 edid->edid_chroma.ec_greenx = EDID_CHROMA_GREENX(data);
582 edid->edid_chroma.ec_greeny = EDID_CHROMA_GREENY(data);
583 edid->edid_chroma.ec_bluex = EDID_CHROMA_BLUEX(data);
584 edid->edid_chroma.ec_bluey = EDID_CHROMA_BLUEY(data);
585 edid->edid_chroma.ec_whitex = EDID_CHROMA_WHITEX(data);
586 edid->edid_chroma.ec_whitey = EDID_CHROMA_WHITEY(data);
587
588 edid->edid_ext_block_count = EDID_EXT_BLOCK_COUNT(data);
589
590 /* lookup established modes */
591 edid->edid_nmodes = 0;
592 edid->edid_preferred_mode = NULL;
593 estmodes = EDID_EST_TIMING(data);
594 /* Iterate in esztablished timing order */
595 for (i = 15; i >= 0; i--) {
596 if (estmodes & (1 << i)) {
597 vmp = edid_mode_lookup_list(_edid_modes[i]);
598 if (vmp != NULL) {
599 edid->edid_modes[edid->edid_nmodes] = *vmp;
600 edid->edid_nmodes++;
601 }
602#ifdef DIAGNOSTIC
603 else
604 printf("no data for est. mode %s\n",
605 _edid_modes[i]);
606#endif
607 }
608 }
609
610 /* do standard timing section */
611 for (i = 0; i < EDID_STD_TIMING_COUNT; i++) {
612 struct videomode mode, *exist_mode;
613 if (edid_std_timing(data + EDID_OFFSET_STD_TIMING + i * 2,
614 &mode)) {
615 /* Does this mode already exist? */
616 exist_mode = edid_search_mode(edid, &mode);
617 if (exist_mode == NULL) {
618 edid->edid_modes[edid->edid_nmodes] = mode;
619 edid->edid_nmodes++;
620 }
621 }
622 }
623
624 /* do detailed timings and descriptors */
625 for (i = 0; i < EDID_BLOCK_COUNT; i++) {
626 edid_block(edid, data + EDID_OFFSET_DESC_BLOCK +
627 i * EDID_BLOCK_SIZE);
628 }
629
630 edid_strchomp(edid->edid_vendorname);
631 edid_strchomp(edid->edid_productname);
632 edid_strchomp(edid->edid_serial);
633 edid_strchomp(edid->edid_comment);
634
635 /*
636 * XXX
637 * some monitors lie about their maximum supported dot clock
638 * by claiming to support modes which need a higher dot clock
639 * than the stated maximum.
640 * For sanity's sake we bump it to the highest dot clock we find
641 * in the list of supported modes
642 */
643 for (i = 0; i < edid->edid_nmodes; i++)
644 if (edid->edid_modes[i].dot_clock > max_dotclock)
645 max_dotclock = edid->edid_modes[i].dot_clock;
646
647 aprint_debug("max_dotclock according to supported modes: %d\n",
648 max_dotclock);
649
650 mhz = (max_dotclock + 999) / 1000;
651
652 if (edid->edid_have_range) {
653 if (mhz > edid->edid_range.er_max_clock)
654 edid->edid_range.er_max_clock = mhz;
655 } else
656 edid->edid_range.er_max_clock = mhz;
657
658 return 0;
659}
