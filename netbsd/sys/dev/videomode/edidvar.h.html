34#ifndef _DEV_VIDEOMODE_EDIDVAR_H
35#define _DEV_VIDEOMODE_EDIDVAR_H
36
37struct edid_chroma {
38 uint16_t ec_redx;
39 uint16_t ec_redy;
40 uint16_t ec_greenx;
41 uint16_t ec_greeny;
42 uint16_t ec_bluex;
43 uint16_t ec_bluey;
44 uint16_t ec_whitex;
45 uint16_t ec_whitey;
46};
47
48struct edid_range {
49 uint16_t er_min_vfreq; /* Hz */
50 uint16_t er_max_vfreq; /* Hz */
51 uint16_t er_min_hfreq; /* kHz */
52 uint16_t er_max_hfreq; /* kHz */
53 uint16_t er_max_clock; /* MHz */
54 int er_have_gtf2;
55 uint16_t er_gtf2_hfreq;
56 uint16_t er_gtf2_c;
57 uint16_t er_gtf2_m;
58 uint16_t er_gtf2_k;
59 uint16_t er_gtf2_j;
60};
61
62struct edid_info {
63 uint8_t edid_vendor[4];
64 char edid_vendorname[16];
65 char edid_productname[16];
66 char edid_comment[16];
67 char edid_serial[16];
68 uint16_t edid_product;
69 uint8_t edid_version;
70 uint8_t edid_revision;
71 int edid_year;
72 int edid_week;
73 uint8_t edid_video_input; /* see edidregs.h */
74 uint8_t edid_max_hsize; /* in cm */
75 uint8_t edid_max_vsize; /* in cm */
76 uint8_t edid_gamma;
77 uint8_t edid_features;
78 uint8_t edid_ext_block_count;
79
80 int edid_have_range;
81 struct edid_range edid_range;
82
83 struct edid_chroma edid_chroma;
84
85 /* parsed modes */
86 struct videomode *edid_preferred_mode;
87 int edid_nmodes;
88 struct videomode edid_modes[64];
89};
90
91int edid_is_valid(uint8_t *);
92int edid_parse(uint8_t *, struct edid_info *);
93void edid_print(struct edid_info *);
94
95#endif /* _DEV_VIDEOMODE_EDIDVAR_H */
