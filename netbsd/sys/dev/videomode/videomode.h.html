30#ifndef _DEV_VIDEOMODE_H
31#define _DEV_VIDEOMODE_H
32
33struct videomode {
34 int dot_clock; /* Dot clock frequency in kHz. */
35 int hdisplay;
36 int hsync_start;
37 int hsync_end;
38 int htotal;
39 int vdisplay;
40 int vsync_start;
41 int vsync_end;
42 int vtotal;
43 int flags; /* Video mode flags; see below. */
44 const char *name;
45};
46
47/*
48 * Video mode flags.
49 */
50
51#define VID_PHSYNC 0x0001
52#define VID_NHSYNC 0x0002
53#define VID_PVSYNC 0x0004
54#define VID_NVSYNC 0x0008
55#define VID_INTERLACE 0x0010
56#define VID_DBLSCAN 0x0020
57#define VID_CSYNC 0x0040
58#define VID_PCSYNC 0x0080
59#define VID_NCSYNC 0x0100
60#define VID_HSKEW 0x0200
61#define VID_BCAST 0x0400
62#define VID_PIXMUX 0x1000
63#define VID_DBLCLK 0x2000
64#define VID_CLKDIV2 0x4000
65
66extern const struct videomode videomode_list[];
67extern const int videomode_count;
68
69const struct videomode *pick_mode_by_dotclock(int, int, int);
70const struct videomode *pick_mode_by_ref(int, int, int);
71void sort_modes(struct videomode *, struct videomode **, int);
72
73#endif /* _DEV_VIDEOMODE_H */
