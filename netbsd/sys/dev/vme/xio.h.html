38/*
39 * xylogic ioctl interface
40 */
41
42#include <sys/ioccom.h>
43
44
45struct xd_iocmd {
46 u_char cmd; /* in: command number */
47 u_char subfn; /* in: subfunction number */
48 u_char errnum; /* out: error number */
49 u_char tries; /* out: number of tries */
50 u_short sectcnt;/* in,out: sector count (hw_spt on read drive param) */
51 u_short dlen; /* in: length of data buffer (good sanity check) */
52 u_int block; /* in: block number */
53 void *dptr; /* in: data buffer to do I/O from */
54};
55
56#ifndef DIOSXDCMD
57#define DIOSXDCMD _IOWR('x', 101, struct xd_iocmd) /* do xd command */
58#endif
59
60#define XD_IOCMD_MAXS 16 /* max number of sectors you can do */
61#define XD_IOCMD_HSZ 4 /* size of one header */
62#define XD_IOCMD_DMSZ 24 /* defect map size */
