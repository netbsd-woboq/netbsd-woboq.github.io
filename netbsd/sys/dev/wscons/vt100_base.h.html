29#define VT100_EMUL_NARGS 10 /* max # of args to a command */
30
31struct vt100base_data {
32 const struct wsdisplay_emulops *emulops;
33 void *emulcookie;
34 int scrcapabilities;
35 u_int nrows, ncols, crow, ccol;
36 struct wsdisplay_msgattrs msgattrs;
37 long defattr; /* default attribute */
38
39 void *cbcookie;
40
41 int flags;
42#define VTFL_LASTCHAR 0x001 /* printed last char on line (below cursor) */
43#define VTFL_INSERTMODE 0x002
44#define VTFL_APPLKEYPAD 0x004
45#define VTFL_APPLCURSOR 0x008
46#define VTFL_DECOM 0x010 /* origin mode */
47#define VTFL_DECAWM 0x020 /* auto wrap */
48#define VTFL_CURSORON 0x040
49#define VTFL_NATCHARSET 0x080 /* national replacement charset mode */
50#define VTFL_SAVEDCURS 0x100 /* we have a saved cursor state */
51 long curattr, bkgdattr; /* currently used attribute */
52 int attrflags, fgcol, bgcol; /* properties of curattr */
53 u_int scrreg_startrow;
54 u_int scrreg_nrows;
55 char *tabs;
56 char *dblwid;
57 int dw;
58
59 int nargs;
60 u_int args[VT100_EMUL_NARGS]; /* numeric command args (CSI/DCS) */
61
62 char modif1; /* {>?} in VT100_EMUL_STATE_CSI */
63 char modif2; /* {!"$&} in VT100_EMUL_STATE_CSI */
64
65 int dcstype; /* substate in VT100_EMUL_STATE_STRING */
66 char *dcsarg;
67 int dcspos;
68#define DCS_MAXLEN 256 /* ??? */
69#define DCSTYPE_TABRESTORE 1 /* DCS2$t */
70};
71
72/* some useful utility macros */
73#define ARG(d, n) ((d)->args[(n)])
74#define DEF1_ARG(d, n) (ARG(d, n) ? ARG(d, n) : 1)
75#define DEFx_ARG(d, n, x) (ARG(d, n) ? ARG(d, n) : (x))
76/* the following two can be negative if we are outside the scrolling region */
77#define ROWS_ABOVE(d) ((int)(d)->crow - (int)(d)->scrreg_startrow)
78#define ROWS_BELOW(d) ((int)((d)->scrreg_startrow + (d)->scrreg_nrows) \
79 - (int)(d)->crow - 1)
80#define CHECK_DW(d) do { \
81 if ((d)->dblwid && (d)->dblwid[(d)->crow]) { \
82 (d)->dw = 1; \
83 if ((d)->ccol > ((d)->ncols >> 1) - 1) \
84 (d)->ccol = ((d)->ncols >> 1) - 1; \
85 } else \
86 (d)->dw = 0; \
87} while (0)
88#define NCOLS(d) ((d)->ncols >> (d)->dw)
89#define COLS_LEFT(d) (NCOLS(d) - (d)->ccol - 1)
90#define COPYCOLS(d, f, t, n) (*(d)->emulops->copycols)((d)->emulcookie, \
91 (d)->crow, (f) << (d)->dw, (t) << (d)->dw, (n) << (d)->dw)
92#define ERASECOLS(d, f, n, a) (*(d)->emulops->erasecols)((d)->emulcookie, \
93 (d)->crow, (f) << (d)->dw, (n) << (d)->dw, a)
94
95/*
96 * response to primary DA request
97 * operating level: 61 = VT100, 62 = VT200, 63 = VT300
98 * extensions: 1 = 132 cols, 2 = printer port, 6 = selective erase,
99 * 7 = soft charset, 8 = UDKs, 9 = NRC sets
100 * VT100 = "033[?1;2c"
101 */
102#define WSEMUL_VT_ID1 "\033[?62;6c"
103/*
104 * response to secondary DA request
105 * ident code: 24 = VT320
106 * firmware version
107 * hardware options: 0 = no options
108 */
109#define WSEMUL_VT_ID2 "\033[>24;20;0c"
