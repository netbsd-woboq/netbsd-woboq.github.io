49void
50vcons_replay_msgbuf(struct vcons_screen *scr)
51{
52 int status = scr->scr_status;
53 int rptr = msgbufp->msg_bufr;
54
55#ifdef WSDISPLAY_MULTICONS
56 wsdisplay_multicons_suspend(true);
57#endif
58
59 scr->scr_status &= ~VCONS_IS_VISIBLE;
60 while (rptr != msgbufp->msg_bufx) {
61 cnputc(msgbufp->msg_bufc[rptr]);
62 rptr++;
63 if (rptr >= msgbufp->msg_bufs)
64 rptr = 0;
65 }
66 scr->scr_status = status;
67 vcons_update_screen(scr);
68
69#ifdef WSDISPLAY_MULTICONS
70 wsdisplay_multicons_suspend(false);
71#endif
72}
