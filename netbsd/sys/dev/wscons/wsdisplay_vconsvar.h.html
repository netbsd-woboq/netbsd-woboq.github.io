39struct vcons_screen {
40 struct rasops_info scr_ri;
41 LIST_ENTRY(vcons_screen) next;
42 void *scr_cookie;
43 struct vcons_data *scr_vd;
44 struct vcons_data *scr_origvd;
45 struct wsscreen_descr *scr_type;
46 uint32_t *scr_chars;
47 long *scr_attrs;
48 void (*putchar)(void *, int, int, u_int, long);
49 long scr_defattr;
50 /* static flags set by the driver */
51 uint32_t scr_flags;
52#define VCONS_NO_REDRAW 1 /* don't readraw in switch_screen */
53#define VCONS_SCREEN_IS_STATIC 2 /* don't free() this vcons_screen */
54#define VCONS_SWITCH_NEEDS_POLLING 4 /* rasops can overlap so we need to
55 * poll the busy flag when switching
56 * - for drivers that use software
57 * drawing */
58#define VCONS_DONT_DRAW 8 /* don't draw on this screen at all */
59/*
60 * the following flags are for drivers which either can't accelerate (all) copy
61 * operations or where drawing characters is faster than the blitter
62 * for example, Sun's Creator boards can't accelerate copycols()
63 */
64#define VCONS_NO_COPYCOLS 0x10 /* use putchar() based copycols() */
65#define VCONS_NO_COPYROWS 0x20 /* use putchar() based copyrows() */
66#define VCONS_DONT_READ 0x30 /* avoid framebuffer reads */
67#define VCONS_LOADFONT 0x40 /* driver can load_font() */
68 /* status flags used by vcons */
69 uint32_t scr_status;
70#define VCONS_IS_VISIBLE 1 /* this screen is currently visible */
71 /* non zero when some rasops operation is in progress */
72 int scr_busy;
73#ifdef WSDISPLAY_SCROLLSUPPORT
74 int scr_lines_in_buffer;
75 int scr_current_line;
76 int scr_line_wanted;
77 int scr_offset_to_zero;
78 int scr_current_offset;
79#endif
80#ifdef VCONS_DRAW_INTR
81 unsigned int scr_dirty;
82#endif
83};
84
85#define SCREEN_IS_VISIBLE(scr) (((scr)->scr_status & VCONS_IS_VISIBLE) != 0)
86#define SCREEN_IS_BUSY(scr) ((scr)->scr_busy != 0)
87#define SCREEN_CAN_DRAW(scr) (((scr)->scr_flags & VCONS_DONT_DRAW) == 0)
88#define SCREEN_BUSY(scr) ((scr)->scr_busy = 1)
89#define SCREEN_IDLE(scr) ((scr)->scr_busy = 0)
90#define SCREEN_VISIBLE(scr) ((scr)->scr_status |= VCONS_IS_VISIBLE)
91#define SCREEN_INVISIBLE(scr) ((scr)->scr_status &= ~VCONS_IS_VISIBLE)
92#define SCREEN_DISABLE_DRAWING(scr) ((scr)->scr_flags |= VCONS_DONT_DRAW)
93#define SCREEN_ENABLE_DRAWING(scr) ((scr)->scr_flags &= ~VCONS_DONT_DRAW)
94
95#define DEFATTR ((WS_DEFAULT_FG << 24) || (WS_DEFAULT_BG << 16))
96
97struct vcons_data {
98 /* usually the drivers softc */
99 void *cookie;
100
101 /*
102 * setup the rasops part of the passed vcons_screen, like
103 * geometry, framebuffer address, font, characters, acceleration.
104 * we pass the cookie as 1st parameter
105 */
106 void (*init_screen)(void *, struct vcons_screen *, int,
107 long *);
108
109 /* accessops */
110 int (*ioctl)(void *, void *, u_long, void *, int, struct lwp *);
111
112 /* rasops */
113 void (*copycols)(void *, int, int, int, int);
114 void (*erasecols)(void *, int, int, int, long);
115 void (*copyrows)(void *, int, int, int);
116 void (*eraserows)(void *, int, int, long);
117 void (*cursor)(void *, int, int, int);
118 /* called before vcons_redraw_screen */
119 void *show_screen_cookie;
120 void (*show_screen_cb)(struct vcons_screen *, void *);
121 /* virtual screen management stuff */
122 void (*switch_cb)(void *, int, int);
123 void *switch_cb_arg;
124 struct callout switch_callout;
125 uint32_t switch_pending;
126 LIST_HEAD(, vcons_screen) screens;
127 struct vcons_screen *active, *wanted;
128 const struct wsscreen_descr *currenttype;
129 struct wsscreen_descr *defaulttype;
130 int switch_poll_count;
131#ifdef VCONS_DRAW_INTR
132 int cells;
133 long *attrs;
134 uint32_t *chars;
135 int cursor_offset;
136 callout_t intr;
137 int intr_valid;
138 void *intr_softint;
139 int use_intr; /* use intr drawing when non-zero */
140#endif
141};
142
143int vcons_init(struct vcons_data *, void *cookie, struct wsscreen_descr *,
144 struct wsdisplay_accessops *);
145
146int vcons_init_screen(struct vcons_data *, struct vcons_screen *, int,
147 long *);
148
149/* completely redraw the screen, clear it if RI_FULLCLEAR is set */
150void vcons_redraw_screen(struct vcons_screen *);
151
152#ifdef VCONS_DRAW_INTR
153/* redraw all dirty character cells */
154void vcons_update_screen(struct vcons_screen *);
155void vcons_invalidate_cache(struct vcons_data *);
156#else
157#define vcons_update_screen vcons_redraw_screen
158#endif
159
160void vcons_replay_msgbuf(struct vcons_screen *);
161
162void vcons_enable_polling(struct vcons_data *);
163void vcons_disable_polling(struct vcons_data *);
164void vcons_hard_switch(struct vcons_screen *);
165
166#endif /* _WSDISPLAY_VCONS_H_ */
