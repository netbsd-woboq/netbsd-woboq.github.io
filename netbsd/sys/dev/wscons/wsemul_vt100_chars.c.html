79void
80vt100_initchartables(struct wsemul_vt100_emuldata *edp)
81{
82 int i;
83 struct vt100base_data *vd = &edp->bd;
84
85 for (i = 0; i < 128; i++)
86 (*vd->emulops->mapchar)(vd->emulcookie, 128 + i,
87 &edp->isolatin1tab[i]);
88 for (i = 0; i < 128; i++)
89 (*vd->emulops->mapchar)(vd->emulcookie, decspcgr2uni[i],
90 &edp->decgraphtab[i]);
91 for (i = 0; i < 128; i++)
92 (*vd->emulops->mapchar)(vd->emulcookie, dectech2uni[i],
93 &edp->dectechtab[i]);
94 vt100_setnrc(edp, 0);
95}

142void
143vt100_setnrc(struct wsemul_vt100_emuldata *edp, int nrc)
144{
145 int i;
146 struct vt100base_data *vd = &edp->bd;
147
148 KASSERT(nrc < __arraycount(nrctable));
149
150 for (i = 0; i < 128; i++)
151 (*vd->emulops->mapchar)(vd->emulcookie, i, &edp->nrctab[i]);
152 for (i = 0; i < 12; i++)
153 (*vd->emulops->mapchar)(vd->emulcookie, nrctable[nrc].c[i],
154 &edp->nrctab[nrcovlpos[i]]);
155}
