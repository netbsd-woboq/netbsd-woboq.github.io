46struct wsemulentry {
47 const struct wsemul_ops *ops;
48 int usecnt;
49 LIST_ENTRY(wsemulentry) next;
50};
51static LIST_HEAD(, wsemulentry) wsemuls = LIST_HEAD_INITIALIZER(&wsemuls);
52
53static const struct wsemul_ops *wsemul_conf[] = {
54#ifdef WSEMUL_SUN
55 &wsemul_sun_ops,
56#endif
57#ifdef WSEMUL_VT100
58 &wsemul_vt100_ops,
59#endif
60#ifndef WSEMUL_NO_DUMB
61 &wsemul_dumb_ops,
62#endif
63 NULL
64};
65
66const struct wsemul_ops *
67wsemul_pick(const char *name)
68{
69 const struct wsemul_ops **ops;
70 struct wsemulentry *wep;
71
72 if (name == NULL) {
73 /* default */
74#ifdef WSEMUL_DEFAULT
75 name = WSEMUL_DEFAULT;
76#else
77 return (wsemul_conf[0]);
78#endif
79 }
80
81 LIST_FOREACH(wep, &wsemuls, next)
82 if (!strcmp(name, wep->ops->name)) {
83 wep->usecnt++;
84 return wep->ops;
85 }
86
87 for (ops = &wsemul_conf[0]; *ops != NULL; ops++)
88 if (strcmp(name, (*ops)->name) == 0)
89 break;
90
91 return (*ops);
92}
93
94void
95wsemul_drop(const struct wsemul_ops *ops)
96{
97 struct wsemulentry *wep;
98
99 LIST_FOREACH(wep, &wsemuls, next)
100 if (ops == wep->ops) {
101 wep->usecnt--;
102 return;
103 }
104}
105
106int
107wsemul_add(const struct wsemul_ops *ops)
108{
109 struct wsemulentry *wep;
110
111 wep = malloc(sizeof (struct wsemulentry), M_DEVBUF, M_WAITOK);
112 wep->ops = ops;
113 wep->usecnt = 0;
114 LIST_INSERT_HEAD(&wsemuls, wep, next);
115 return 0;
116}
117
118int
119wsemul_remove(const struct wsemul_ops *ops)
120{
121 struct wsemulentry *wep;
122
123 LIST_FOREACH(wep, &wsemuls, next) {
124 if (ops == wep->ops) {
125 if (wep->usecnt)
126 return EBUSY;
127 LIST_REMOVE(wep, next);
128 return 0;
129 }
130 }
131 return ENOENT;
132}
