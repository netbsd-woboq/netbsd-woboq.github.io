32#ifndef _WSFONT_H_
33#define _WSFONT_H_ 1
34
35#define WSFONT_FLAGS_MASK 0x7f000000
36#define WSFONT_FLAG_OPT 0x01000000 /* use alternate font */
37#define WSFONT_GLYPH(c, font) ((uint8_t *)font->data + \
38 ((c) - font->firstchar) * font->stride * font->fontheight)
39#define FONT_IS_ALPHA(f) ((f)->fontwidth <= (f)->stride)
40
41/*
42 * Example:
43 *
44 * struct wsdisplay_font *font;
45 * int cookie;
46 *
47 * cookie = wsfont_find(NULL, 8, 16, 0, WSDISPLAY_FONTORDER_L2R,
48 * WSDISPLAY_FONTORDER_R2L);
49 * if (cookie <= 0)
50 * panic("unable to get 8x16 font");
51 *
52 * if (wsfont_lock(cookie, &font))
53 * panic("unable to lock font");
54 *
55 * ... do stuff ...
56 *
57 * wsfont_unlock(cookie);
58 */
