53static int
54wsfontopen(dev_t dev, int flag, int mode,
55 struct lwp *l)
56{
57
58 if (wsfont_isopen)
59 return (EBUSY);
60 wsfont_isopen = 1;
61 return (0);
62}
63
64static int
65wsfontclose(dev_t dev, int flag, int mode,
66 struct lwp *l)
67{
68
69 wsfont_isopen = 0;
70 return (0);
71}

73static int
74wsfontioctl(dev_t dev, u_long cmd, void *data, int flag,
75 struct lwp *l)
76{
77 char nbuf[64];
78 void *buf;
79 int res;
80
81 switch (cmd) {
82 case WSDISPLAYIO_LDFONT:
83#define d ((struct wsdisplay_font *)data)
84 if (d->name) {
85 res = copyinstr(d->name, nbuf, sizeof(nbuf), 0);
86 if (res)
87 return (res);
88 d->name = nbuf;
89 } else
90 d->name = "loaded"; /* ??? */
91 buf = malloc(d->fontheight * d->stride * d->numchars,
92 M_DEVBUF, M_WAITOK);
93 res = copyin(d->data, buf,
94 d->fontheight * d->stride * d->numchars);
95 if (res) {
96 free(buf, M_DEVBUF);
97 return (res);
98 }
99 d->data = buf;
100 res = wsfont_add(d, 1);
101 free(buf, M_DEVBUF);
102#undef d
103 return (res);
104 default:
105 return (EINVAL);
106 }
107}
