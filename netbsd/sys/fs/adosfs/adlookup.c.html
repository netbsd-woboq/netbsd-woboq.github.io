46#ifdef ADOSFS_EXACTMATCH
47#define strmatch(s1, l1, s2, l2, i) \
48 ((l1) == (l2) && memcmp((s1), (s2), (l1)) == 0)
49#else
50#define strmatch(s1, l1, s2, l2, i) \
51 ((l1) == (l2) && adoscaseequ((s1), (s2), (l1), (i)))
52#endif

54/*
55 * adosfs lookup. enters with:
56 * pvp (parent vnode) referenced and locked.
57 * exit with:
58 * target vp referenced and locked.
59 * parent pvp locked.
60 * special cases:
61 * pvp == vp, just ref pvp, pvp already holds a ref and lock from
62 * caller, this will not occur with RENAME or CREATE.
63 */
64int
65adosfs_lookup(void *v)
66{
67 struct vop_lookup_v2_args /* {
68 struct vnode *a_dvp;
69 struct vnode **a_vpp;
70 struct componentname *a_cnp;
71 } */ *sp = v;
72 int nameiop, last, flags, error, nocache, i;
73 struct componentname *cnp;
74 struct vnode **vpp; /* place to store result */
75 struct anode *ap; /* anode to find */
76 struct vnode *vdp; /* vnode of search dir */
77 struct anode *adp; /* anode of search dir */
78 kauth_cred_t ucp; /* lookup credentials */
79 u_long bn, plen, hval;
80 const u_char *pelt;
81
82#ifdef ADOSFS_DIAGNOSTIC
83 advopprint(sp);
84#endif
85 cnp = sp->a_cnp;
86 vdp = sp->a_dvp;
87 adp = VTOA(vdp);
88 vpp = sp->a_vpp;
89 *vpp = NULL;
90 ucp = cnp->cn_cred;
91 nameiop = cnp->cn_nameiop;
92 flags = cnp->cn_flags;
93 last = flags & ISLASTCN;
94 pelt = (const u_char *)cnp->cn_nameptr;
95 plen = cnp->cn_namelen;
96 nocache = 0;
97
98 /*
99 * Check accessiblity of directory.
100 */
101 if ((error = VOP_ACCESS(vdp, VEXEC, ucp)) != 0)
102 return (error);
103
104 if ((flags & ISLASTCN) && (vdp->v_mount->mnt_flag & MNT_RDONLY) &&
105 (cnp->cn_nameiop == DELETE || cnp->cn_nameiop == RENAME))
106 return (EROFS);
107
108 /*
109 * Before tediously performing a linear scan of the directory,
110 * check the name cache to see if the directory/name pair
111 * we are looking for is known already.
112 */
113 if (cache_lookup(vdp, cnp->cn_nameptr, cnp->cn_namelen,
114 cnp->cn_nameiop, cnp->cn_flags, NULL, vpp)) {
115 return *vpp == NULLVP ? ENOENT : 0;
116 }
117
118 /*
119 * fake a '.'
120 */
121 if (plen == 1 && pelt[0] == '.') {
122 /* see special cases in prologue. */
123 *vpp = vdp;
124 goto found;
125 }
126 /*
127 * fake a ".."
128 */
129 if (flags & ISDOTDOT) {
130 if (vdp->v_type == VDIR && (vdp->v_vflag & VV_ROOT))
131 panic("adosfs .. attempted through root");
132 /*
133 * cannot get `..' while `vdp' is locked
134 * e.g. procA holds lock on `..' and waits for `vdp'
135 * we wait for `..' and hold lock on `vdp'. deadlock.
136 * because `vdp' may have been achieved through symlink
137 * fancy detection code that decreases the race
138 * window size is not reasonably possible.
139 *
140 * basically unlock the parent, try and lock the child (..)
141 * if that fails relock the parent (ignoring error) and
142 * fail. Otherwise we have the child (..), attempt to
143 * relock the parent. If that fails unlock the child (..)
144 * and fail. Otherwise we have succeded.
145 *
146 */
147 VOP_UNLOCK(vdp); /* race */
148 error = VFS_VGET(vdp->v_mount, (ino_t)adp->pblock, vpp);
149 vn_lock(vdp, LK_EXCLUSIVE | LK_RETRY);
150 if (error) {
151 *vpp = NULL;
152 return (error);
153 }
154 goto found_lockdone;
155 }
156
157 /*
158 * hash the name and grab the first block in chain
159 * then walk the chain. if chain has not been fully
160 * walked before, track the count in `tabi'
161 */
162 hval = adoshash(pelt, plen, adp->ntabent, IS_INTER(adp->amp));
163 bn = adp->tab[hval];
164 i = uimin(adp->tabi[hval], 0);
165 while (bn != 0) {
166 if ((error = VFS_VGET(vdp->v_mount, (ino_t)bn, vpp
167 )) != 0) {
168#ifdef ADOSFS_DIAGNOSTIC
169 printf("[aget] %d)", error);
170#endif
171 return(error);
172 }
173 ap = VTOA(*vpp);
174 if (i <= 0) {
175 if (--i < adp->tabi[hval])
176 adp->tabi[hval] = i;
177 /*
178 * last header in chain lock count down by
179 * negating it to positive
180 */
181 if (ap->hashf == 0) {
182#ifdef DEBUG
183 if (i != adp->tabi[hval])
184 panic("adlookup: wrong chain count");
185#endif
186 adp->tabi[hval] = -adp->tabi[hval];
187 }
188 }
189 if (strmatch(pelt, plen, ap->name, strlen(ap->name),
190 IS_INTER(adp->amp)))
191 goto found;
192 bn = ap->hashf;
193 vput(*vpp);
194 }
195 *vpp = NULL;
196 /*
197 * not found
198 */
199 if ((nameiop == CREATE || nameiop == RENAME) && last) {
200
201 if (vdp->v_mount->mnt_flag & MNT_RDONLY)
202 return (EROFS);
203
204 if ((error = VOP_ACCESS(vdp, VWRITE, ucp)) != 0) {
205#ifdef ADOSFS_DIAGNOSTIC
206 printf("[VOP_ACCESS] %d)", error);
207#endif
208 return (error);
209 }
210#ifdef ADOSFS_DIAGNOSTIC
211 printf("EJUSTRETURN)");
212#endif
213 return(EJUSTRETURN);
214 }
215 if (nameiop != CREATE)
216 cache_enter(vdp, NULL, cnp->cn_nameptr, cnp->cn_namelen,
217 cnp->cn_flags);
218#ifdef ADOSFS_DIAGNOSTIC
219 printf("ENOENT)");
220#endif
221 return(ENOENT);
222
223found:
224 if (nameiop == DELETE && last) {
225 if ((error = VOP_ACCESS(vdp, VWRITE, ucp)) != 0) {
226 if (vdp != *vpp)
227 vput(*vpp);
228 *vpp = NULL;
229 return (error);
230 }
231 nocache = 1;
232 }
233 if (nameiop == RENAME && last) {
234 if (vdp == *vpp)
235 return(EISDIR);
236 if ((error = VOP_ACCESS(vdp, VWRITE, ucp)) != 0) {
237 vput(*vpp);
238 *vpp = NULL;
239 return (error);
240 }
241 nocache = 1;
242 }
243 if (vdp == *vpp)
244 vref(vdp);
245found_lockdone:
246 if (*vpp != vdp)
247 VOP_UNLOCK(*vpp);
248 if (nocache == 0)
249 cache_enter(vdp, *vpp, cnp->cn_nameptr, cnp->cn_namelen,
250 cnp->cn_flags);
251
252#ifdef ADOSFS_DIAGNOSTIC
253 printf("0)\n");
254#endif
255 return(0);
256}
