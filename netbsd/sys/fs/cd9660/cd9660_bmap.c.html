53/*
54 * Bmap converts a the logical block number of a file to its physical block
55 * number on the disk. The conversion is done by using the logical block
56 * number to index into the data block (extent) for the file.
57 */
58int
59cd9660_bmap(void *v)
60{
61 struct vop_bmap_args /* {
62 struct vnode *a_vp;
63 daddr_t a_bn;
64 struct vnode **a_vpp;
65 daddr_t *a_bnp;
66 int *a_runp;
67 } */ *ap = v;
68 struct iso_node *ip = VTOI(ap->a_vp);
69 daddr_t lblkno = ap->a_bn;
70 int bshift;
71
72 /*
73 * Check for underlying vnode requests and ensure that logical
74 * to physical mapping is requested.
75 */
76 if (ap->a_vpp != NULL)
77 *ap->a_vpp = ip->i_mnt->im_devvp;
78 if (ap->a_bnp == NULL)
79 return (0);
80
81 /*
82 * Compute the requested block number
83 */
84 bshift = ip->i_mnt->im_bshift;
85 *ap->a_bnp = (ip->iso_start + lblkno) << (bshift - DEV_BSHIFT);
86
87 /*
88 * Determine maximum number of readahead blocks following the
89 * requested block.
90 */
91 if (ap->a_runp) {
92 int nblk;
93
94 nblk = (ip->i_size >> bshift) - (lblkno + 1);
95 if (nblk <= 0)
96 *ap->a_runp = 0;
97 else if (nblk >= (MAXBSIZE >> bshift))
98 *ap->a_runp = (MAXBSIZE >> bshift) - 1;
99 else
100 *ap->a_runp = nblk;
101 }
102
103 return (0);
104}
