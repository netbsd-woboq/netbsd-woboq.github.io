39/*
40 * Definitions used in the kernel for cd9660 file system support.
41 */
42#ifndef _ISOFS_CD9660_CD9660_EXTERN_H_
43#define _ISOFS_CD9660_CD9660_EXTERN_H_
44
45/*
46 * Sysctl values for the cd9660 filesystem.
47 */
48#define CD9660_UTF8_JOLIET 1 /* UTF-8 encode Joliet file names */
49
50/* CD-ROM Format type */
51enum ISO_FTYPE { ISO_FTYPE_DEFAULT, ISO_FTYPE_9660, ISO_FTYPE_RRIP, ISO_FTYPE_ECMA };
52
53#ifndef ISOFSMNT_ROOT
54#define ISOFSMNT_ROOT 0
55#endif
56
57struct iso_mnt {
58 int im_flags;
59 int im_joliet_level;
60
61 struct mount *im_mountp;
62 dev_t im_dev;
63 struct vnode *im_devvp;
64
65 int logical_block_size;
66 int im_bshift;
67 int im_bmask;
68
69 int volume_space_size;
70
71 char root[ISODCL (157, 190)];
72 int root_extent;
73 int root_size;
74 enum ISO_FTYPE iso_ftype;
75
76 int rr_skip;
77 int rr_skip0;
78};
79
80#define VFSTOISOFS(mp) ((struct iso_mnt *)((mp)->mnt_data))
81
82#define cd9660_blkoff(imp, loc) ((loc) & (imp)->im_bmask)
83#define cd9660_lblktosize(imp, blk) ((blk) << (imp)->im_bshift)
84#define cd9660_lblkno(imp, loc) ((loc) >> (imp)->im_bshift)
85#define cd9660_blksize(imp, ip, lbn) ((imp)->logical_block_size)
