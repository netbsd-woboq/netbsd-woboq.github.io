79struct filecore_mnt {
80 struct mount *fc_mountp;
81 dev_t fc_dev;
82 struct vnode *fc_devvp;
83
84 u_int32_t blksize;
85 u_int32_t log2bsize;
86 u_int32_t map;
87 u_int32_t idspz;
88 u_int32_t mask;
89 u_int64_t nblks;
90
91 uid_t fc_uid; /* uid to set as owner of the files */
92 gid_t fc_gid; /* gid to set as owner of the files */
93 int fc_mntflags;
94 struct filecore_disc_record drec;
95};
96
97#define VFSTOFILECORE(mp) ((struct filecore_mnt *)((mp)->mnt_data))

99#define filecore_blkoff(fcp, loc) ((loc) & ((fcp)->blksize-1))
100#define filecore_lblktosize(fcp, blk) ((blk) << (fcp)->log2bsize)
101#define filecore_lblkno(fcp, loc) ((loc) >> (fcp)->log2bsize)
102#define filecore_blksize(fcp, ip, lbn) ((fcp)->blksize)
