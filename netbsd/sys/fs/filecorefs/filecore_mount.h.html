67#ifndef _FILECOREFS_FILECOREFS_MOUNT_H_
68#define _FILECOREFS_FILECOREFS_MOUNT_H_
69
70/*
71 * Arguments to mount Acorn Filecore filesystems.
72 */
73struct filecore_args {
74 char *fspec; /* block special device to mount */
75 struct export_args30 _pad1; /* compat with old userland tools */
76 uid_t uid; /* uid that owns filecore files */
77 gid_t gid; /* gid that owns filecore files */
78 int flags; /* mounting flags, see below */
79};
80
81#define FILECOREMNT_ROOT 0
82#define FILECOREMNT_OWNACCESS 1 /* Only user has Owner access */
83#define FILECOREMNT_ALLACCESS 2 /* World has Owner access */
84#define FILECOREMNT_OWNREAD 4 /* All files have Owner read access */
85#define FILECOREMNT_USEUID 8 /* Use uid of mount process */
86#define FILECOREMNT_FILETYPE 16 /* Include filetype in filename */
87
88#define FILECOREMNT_BITS "\177\20" \
89 "b\00root\0b\01ownaccess\0b\02allaccess\0b\03ownread\0" \
90 "b\04useuid\0b\05filetype\0"
91
92#endif /* _FILECOREFS_FILECOREFS_MOUNT_H_ */
