94static int
95filecore_check_possible(struct vnode *vp, struct filecore_node *ip,
96 mode_t mode)
97{
98
99 /*
100 * Disallow write attempts unless the file is a socket,
101 * fifo, or a block or character device resident on the
102 * file system.
103 */
104 if (mode & VWRITE) {
105 switch (vp->v_type) {
106 case VDIR:
107 case VLNK:
108 case VREG:
109 return (EROFS);
110 default:
111 break;
112 }
113 }
114
115 return 0;
116}

118/*
119 * Check mode permission on inode pointer. Mode is READ, WRITE or EXEC.
120 * The mode is shifted to select the owner/group/other fields. The
121 * super user is granted all permissions.
122 */
123static int
124filecore_check_permitted(struct vnode *vp, struct filecore_node *ip,
125 mode_t mode, kauth_cred_t cred)
126{
127 struct filecore_mnt *fcmp = ip->i_mnt;
128
129 return kauth_authorize_vnode(cred, KAUTH_ACCESS_ACTION(mode,
130 vp->v_type, filecore_mode(ip)), vp, NULL,
131 genfs_can_access(vp->v_type, filecore_mode(ip), fcmp->fc_uid,
132 fcmp->fc_gid, mode, cred));
133}

135int
136filecore_access(void *v)
137{
138 struct vop_access_args /* {
139 struct vnode *a_vp;
140 int a_mode;
141 kauth_cred_t a_cred;
142 } */ *ap = v;
143 struct vnode *vp = ap->a_vp;
144 struct filecore_node *ip = VTOI(vp);
145 int error;
146
147 error = filecore_check_possible(vp, ip, ap->a_mode);
148 if (error)
149 return error;
150
151 error = filecore_check_permitted(vp, ip, ap->a_mode, ap->a_cred);
152
153 return error;
154}

156int
157filecore_getattr(void *v)
158{
159 struct vop_getattr_args /* {
160 struct vnode *a_vp;
161 struct vattr *a_vap;
162 kauth_cred_t a_cred;
163 } */ *ap = v;
164 struct vnode *vp = ap->a_vp;
165 struct filecore_node *ip = VTOI(vp);
166 struct vattr *vap = ap->a_vap;
167 struct filecore_mnt *fcmp = ip->i_mnt;
168
169 vap->va_fsid = ip->i_dev;
170 vap->va_fileid = ip->i_number;
171
172 vap->va_mode = filecore_mode(ip);
173 vap->va_nlink = 1;
174 vap->va_uid = fcmp->fc_uid;
175 vap->va_gid = fcmp->fc_gid;
176 vap->va_atime = filecore_time(ip);
177 vap->va_mtime = filecore_time(ip);
178 vap->va_ctime = filecore_time(ip);
179 vap->va_rdev = 0; /* We don't support specials */
180
181 vap->va_size = (u_quad_t) ip->i_size;
182 vap->va_flags = 0;
183 vap->va_gen = 1;
184 vap->va_blocksize = fcmp->blksize;
185 vap->va_bytes = vap->va_size;
186 vap->va_type = vp->v_type;
187 return (0);
188}

190/*
191 * Vnode op for reading.
192 */
193int
194filecore_read(void *v)
195{
196 struct vop_read_args /* {
197 struct vnode *a_vp;
198 struct uio *a_uio;
199 int a_ioflag;
200 kauth_cred_t a_cred;
201 } */ *ap = v;
202 struct vnode *vp = ap->a_vp;
203 struct uio *uio = ap->a_uio;
204 struct filecore_node *ip = VTOI(vp);
205 struct filecore_mnt *fcmp;
206 struct buf *bp;
207 daddr_t lbn;
208 off_t diff;
209 int error = 0;
210 long size, n, on;
211
212 if (uio->uio_resid == 0)
213 return (0);
214 if (uio->uio_offset < 0)
215 return (EINVAL);
216 if (uio->uio_offset >= ip->i_size)
217 return (0);
218 ip->i_flag |= IN_ACCESS;
219 fcmp = ip->i_mnt;
220
221 if (vp->v_type == VREG) {
222 const int advice = IO_ADV_DECODE(ap->a_ioflag);
223 error = 0;
224
225 while (uio->uio_resid > 0) {
226 vsize_t bytelen = MIN(ip->i_size - uio->uio_offset,
227 uio->uio_resid);
228
229 if (bytelen == 0) {
230 break;
231 }
232 error = ubc_uiomove(&vp->v_uobj, uio, bytelen, advice,
233 UBC_READ | UBC_PARTIALOK | UBC_UNMAP_FLAG(vp));
234 if (error) {
235 break;
236 }
237 }
238 goto out;
239 }
240
241 do {
242 lbn = filecore_lblkno(fcmp, uio->uio_offset);
243 on = filecore_blkoff(fcmp, uio->uio_offset);
244 n = MIN(filecore_blksize(fcmp, ip, lbn) - on, uio->uio_resid);
245 diff = (off_t)ip->i_size - uio->uio_offset;
246 if (diff <= 0)
247 return (0);
248 if (diff < n)
249 n = diff;
250 size = filecore_blksize(fcmp, ip, lbn);
251 if (ip->i_dirent.attr & FILECORE_ATTR_DIR) {
252 error = filecore_dbread(ip, &bp);
253 on = uio->uio_offset;
254 n = MIN(FILECORE_DIR_SIZE - on, uio->uio_resid);
255 size = FILECORE_DIR_SIZE;
256 } else {
257 error = bread(vp, lbn, size, 0, &bp);
258#ifdef FILECORE_DEBUG_BR
259 printf("bread(%p, %llx, %ld, CRED, %p)=%d\n",
260 vp, (long long)lbn, size, bp, error);
261#endif
262 }
263 if (error) {
264 return (error);
265 }
266 n = MIN(n, size - bp->b_resid);
267
268 error = uiomove((char *)(bp->b_data) + on, (int)n, uio);
269#ifdef FILECORE_DEBUG_BR
270 printf("brelse(%p) vn2\n", bp);
271#endif
272 brelse(bp, 0);
273 } while (error == 0 && uio->uio_resid > 0 && n != 0);
274
275out:
276 return (error);
277}

279/*
280 * Vnode op for readdir
281 */
282int
283filecore_readdir(void *v)
284{
285 struct vop_readdir_args /* {
286 struct vnode *a_vp;
287 struct uio *a_uio;
288 kauth_cred_t a_cred;
289 int *a_eofflag;
290 off_t **a_cookies;
291 int *a_ncookies;
292 } */ *ap = v;
293 struct uio *uio = ap->a_uio;
294 struct vnode *vdp = ap->a_vp;
295 struct filecore_node *dp;
296 struct buf *bp = NULL;
297 struct dirent *de;
298 struct filecore_direntry *dep = NULL;
299 int error = 0;
300 off_t *cookies = NULL;
301 int ncookies = 0;
302 int i;
303 off_t uiooff;
304
305 dp = VTOI(vdp);
306
307 if ((dp->i_dirent.attr & FILECORE_ATTR_DIR) == 0)
308 return (ENOTDIR);
309
310 if (uio->uio_offset % FILECORE_DIRENT_SIZE != 0)
311 return (EINVAL);
312 i = uio->uio_offset / FILECORE_DIRENT_SIZE;
313 uiooff = uio->uio_offset;
314
315 *ap->a_eofflag = 0;
316
317 error = filecore_dbread(dp, &bp);
318 if (error) {
319 return error;
320 }
321
322 if (ap->a_ncookies == NULL)
323 cookies = NULL;
324 else {
325 *ap->a_ncookies = 0;
326 ncookies = uio->uio_resid / _DIRENT_MINSIZE((struct dirent *)0);
327 cookies = malloc(ncookies * sizeof(off_t), M_TEMP, M_WAITOK);
328 }
329
330 de = kmem_zalloc(sizeof(struct dirent), KM_SLEEP);
331
332 for (; ; i++) {
333 switch (i) {
334 case 0:
335 /* Fake the '.' entry */
336 de->d_fileno = dp->i_number;
337 de->d_type = DT_DIR;
338 de->d_namlen = 1;
339 strlcpy(de->d_name, ".", sizeof(de->d_name));
340 break;
341 case 1:
342 /* Fake the '..' entry */
343 de->d_fileno = filecore_getparent(dp);
344 de->d_type = DT_DIR;
345 de->d_namlen = 2;
346 strlcpy(de->d_name, "..", sizeof(de->d_name));
347 break;
348 default:
349 de->d_fileno = dp->i_dirent.addr +
350 ((i - 2) << FILECORE_INO_INDEX);
351 dep = fcdirentry(bp->b_data, i - 2);
352 if (dep->attr & FILECORE_ATTR_DIR)
353 de->d_type = DT_DIR;
354 else
355 de->d_type = DT_REG;
356 if (filecore_fn2unix(dep->name, de->d_name,
357/*###346 [cc] warning: passing arg 3 of `filecore_fn2unix' from incompatible pointer type%%%*/
358 &de->d_namlen)) {
359 *ap->a_eofflag = 1;
360 goto out;
361 }
362 break;
363 }
364 de->d_reclen = _DIRENT_SIZE(de);
365 if (uio->uio_resid < de->d_reclen)
366 goto out;
367 error = uiomove(de, de->d_reclen, uio);
368 if (error)
369 goto out;
370 uiooff += FILECORE_DIRENT_SIZE;
371
372 if (cookies) {
373 *cookies++ = i*FILECORE_DIRENT_SIZE;
374 (*ap->a_ncookies)++;
375 if (--ncookies == 0) goto out;
376 }
377 }
378out:
379 if (cookies) {
380 *ap->a_cookies = cookies;
381 if (error) {
382 free(cookies, M_TEMP);
383 *ap->a_ncookies = 0;
384 *ap->a_cookies = NULL;
385 }
386 }
387 uio->uio_offset = uiooff;
388
389#ifdef FILECORE_DEBUG_BR
390 printf("brelse(%p) vn3\n", bp);
391#endif
392 brelse (bp, 0);
393
394 kmem_free(de, sizeof(*de));
395
396 return (error);
397}

447/*
448 * Calculate the logical to physical mapping if not done already,
449 * then call the device strategy routine.
450 */
451int
452filecore_strategy(void *v)
453{
454 struct vop_strategy_args /* {
455 struct vnode *a_vp;
456 struct buf *a_bp;
457 } */ *ap = v;
458 struct buf *bp = ap->a_bp;
459 struct vnode *vp = ap->a_vp;
460 struct filecore_node *ip;
461 int error;
462
463 ip = VTOI(vp);
464 if (bp->b_blkno == bp->b_lblkno) {
465 error = VOP_BMAP(vp, bp->b_lblkno, NULL, &bp->b_blkno, NULL);
466 if (error) {
467 bp->b_error = error;
468 biodone(bp);
469 return (error);
470 }
471 if ((long)bp->b_blkno == -1)
472 clrbuf(bp);
473 }
474 if ((long)bp->b_blkno == -1) {
475 biodone(bp);
476 return (0);
477 }
478 vp = ip->i_devvp;
479 return (VOP_STRATEGY(vp, bp));
480}

494/*
495 * Return POSIX pathconf information applicable to filecore filesystems.
496 */
497int
498filecore_pathconf(void *v)
499{
500 struct vop_pathconf_args /* {
501 struct vnode *a_vp;
502 int a_name;
503 register_t *a_retval;
504 } */ *ap = v;
505 switch (ap->a_name) {
506 case _PC_LINK_MAX:
507 *ap->a_retval = 1;
508 return (0);
509 case _PC_NAME_MAX:
510 *ap->a_retval = ap->a_vp->v_mount->mnt_stat.f_namemax;
511 return (0);
512 case _PC_PATH_MAX:
513 *ap->a_retval = 256;
514 return (0);
515 case _PC_CHOWN_RESTRICTED:
516 *ap->a_retval = 1;
517 return (0);
518 case _PC_NO_TRUNC:
519 *ap->a_retval = 1;
520 return (0);
521 case _PC_SYNC_IO:
522 *ap->a_retval = 1;
523 return (0);
524 case _PC_FILESIZEBITS:
525 *ap->a_retval = 32;
526 return (0);
527 default:
528 return (EINVAL);
529 }
530 /* NOTREACHED */
531}
