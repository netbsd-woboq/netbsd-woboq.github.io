37#ifndef _FS_NILFS_NILFS_MOUNT_H_
38#define _FS_NILFS_NILFS_MOUNT_H_
39
40/*
41 * Arguments to mount NILFS filingsystem.
42 */
43
44#define NILFSMNT_VERSION 1
45struct nilfs_args {
46 uint32_t version; /* version of this structure */
47 char *fspec; /* mount specifier */
48 uint32_t nilfsmflags; /* mount options */
49
50 int32_t gmtoff; /* offset from UTC in seconds */
51 int64_t cpno; /* checkpoint number */
52
53 /* extendable */
54 uint8_t reserved[32];
55};
56
57
58/* nilfs mount options */
59
60#define NILFSMNT_BITS "\20"
61
62#endif /* !_FS_NILFS_NILFS_MOUNT_H_ */
