47/* UTF-8 encoding stuff */
48#include <fs/unicode.h>
49
50/*
51 * Read one wide character off the string, shift the string pointer
52 * and return the character.
53 */
54wchar
55ntfs_utf8_wget(const char **str, size_t *sz)
56{
57 return (wchar) wget_utf8(str, sz);
58}
59
60/*
61 * Encode wide character and write it to the string. 'n' specifies
62 * how much space there is in the string. Returns number of bytes written
63 * to the target string.
64 */
65int
66ntfs_utf8_wput(char *s, size_t n, wchar wc)
67{
68 return wput_utf8(s, n, (u_int16_t) wc);
69}
70
71/*
72 * Compare two wide characters, returning 1, 0, -1 if the first is
73 * bigger, equal or lower than the second.
74 */
75int
76ntfs_utf8_wcmp(wchar wc1, wchar wc2)
77{
78 /* no conversions needed for utf8 */
79
80 if (wc1 == wc2)
81 return 0;
82 else
83 return (int) wc1 - (int) wc2;
84}
