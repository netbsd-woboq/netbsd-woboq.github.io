74int
75smbfs_loadvnode(struct mount *mp, struct vnode *vp,
76 const void *key, size_t key_len, const void **new_key)
77{
78 struct smbnode *np;
79
80 np = pool_get(&smbfs_node_pool, PR_WAITOK);
81 memset(np, 0, sizeof(*np));
82
83 vp->v_tag = VT_SMBFS;
84 vp->v_op = smbfs_vnodeop_p;
85 vp->v_type = VNON;
86 vp->v_data = np;
87 genfs_node_init(vp, &smbfs_genfsops);
88
89 mutex_init(&np->n_lock, MUTEX_DEFAULT, IPL_NONE);
90 np->n_key = kmem_alloc(key_len, KM_SLEEP);
91 memcpy(np->n_key, key, key_len);
92 KASSERT(key_len == SMBFS_KEYSIZE(np->n_nmlen));
93 np->n_vnode = vp;
94 np->n_mount = VFSTOSMBFS(mp);
95
96 if (np->n_parent != NULL && (np->n_parent->v_vflag & VV_ROOT) == 0) {
97 vref(np->n_parent);
98 np->n_flag |= NREFPARENT;
99 }
100
101 *new_key = np->n_key;
102
103 return 0;
104}

106int
107smbfs_nget(struct mount *mp, struct vnode *dvp, const char *name, int nmlen,
108 struct smbfattr *fap, struct vnode **vpp)
109{
110 struct smbkey *key;
111 struct smbmount *smp __diagused;
112 struct smbnode *np;
113 struct vnode *vp;
114 union {
115 struct smbkey u_key;
116 char u_data[64];
117 } small_key;
118 int error;
119 const int key_len = SMBFS_KEYSIZE(nmlen);
120
121 smp = VFSTOSMBFS(mp);
122
123 /* do not allow allocating root vnode twice */
124 KASSERT(dvp != NULL || smp->sm_root == NULL);
125
126 /* do not call with dot */
127 KASSERT(nmlen != 1 || name[0] != '.');
128
129 if (nmlen == 2 && memcmp(name, "..", 2) == 0) {
130 if (dvp == NULL)
131 return EINVAL;
132 vp = VTOSMB(VTOSMB(dvp)->n_parent)->n_vnode;
133 vref(vp);
134 *vpp = vp;
135 return 0;
136 }
137
138#ifdef DIAGNOSTIC
139 struct smbnode *dnp = dvp ? VTOSMB(dvp) : NULL;
140 if (dnp == NULL && dvp != NULL)
141 panic("smbfs_node_alloc: dead parent vnode %p", dvp);
142#endif
143
144 if (key_len > sizeof(small_key))
145 key = kmem_alloc(key_len, KM_SLEEP);
146 else
147 key = &small_key.u_key;
148 key->k_parent = dvp;
149 key->k_nmlen = nmlen;
150 memcpy(key->k_name, name, nmlen);
151
152retry:
153 error = vcache_get(mp, key, key_len, &vp);
154 if (error)
155 goto out;
156 mutex_enter(vp->v_interlock);
157 np = VTOSMB(vp);
158 KASSERT(np != NULL);
159 mutex_enter(&np->n_lock);
160 mutex_exit(vp->v_interlock);
161
162 if (vp->v_type == VNON) {
163 /*
164 * If we don't have node attributes, then it is an
165 * explicit lookup for an existing vnode.
166 */
167 if (fap == NULL) {
168 mutex_exit(&np->n_lock);
169 vrele(vp);
170 error = ENOENT;
171 goto out;
172 }
173 vp->v_type = fap->fa_attr & SMB_FA_DIR ? VDIR : VREG;
174 np->n_ino = fap->fa_ino;
175 np->n_size = fap->fa_size;
176
177 /* new file vnode has to have a parent */
178 KASSERT(vp->v_type != VREG || dvp != NULL);
179
180 uvm_vnp_setsize(vp, np->n_size);
181 } else {
182 struct vattr vattr;
183
184 /* Force cached attributes to be refreshed if stale. */
185 (void)VOP_GETATTR(vp, &vattr, curlwp->l_cred);
186 /*
187 * If the file type on the server is inconsistent with
188 * what it was when we created the vnode, kill the
189 * bogus vnode now and retry to create a new one with
190 * the right type.
191 */
192 if ((vp->v_type == VDIR && (np->n_dosattr & SMB_FA_DIR) == 0) ||
193 (vp->v_type == VREG && (np->n_dosattr & SMB_FA_DIR) != 0)) {
194 mutex_exit(&np->n_lock);
195 vn_lock(vp, LK_EXCLUSIVE | LK_RETRY);
196 smbfs_uncache(vp);
197 goto retry;
198 }
199 }
200 if (fap)
201 smbfs_attr_cacheenter(vp, fap);
202 *vpp = vp;
203 mutex_exit(&np->n_lock);
204
205out:
206 if (key != &small_key.u_key)
207 kmem_free(key, key_len);
208 return error;
209}

211/*
212 * Remove vnode that changed its type on the server from
213 * the vnode cache and the name cache.
214 */
215void
216smbfs_uncache(struct vnode *vp)
217{
218 static uint32_t gen = 0;
219 int error __diagused;
220 char newname[10];
221 struct mount *mp = vp->v_mount;
222 struct smbnode *np = VTOSMB(vp);
223 struct smbkey *key = np->n_key, *oldkey, *newkey;
224 int key_len = SMBFS_KEYSIZE(key->k_nmlen), newkey_len;
225
226 /* Setup old key as current key. */
227 oldkey = kmem_alloc(key_len, KM_SLEEP);
228 memcpy(oldkey, key, key_len);
229
230 /* Setup new key as unique and illegal name with colon. */
231 snprintf(newname, sizeof(newname), ":%08x", atomic_inc_uint_nv(&gen));
232 newkey = kmem_alloc(SMBFS_KEYSIZE(strlen(newname)), KM_SLEEP);
233 newkey->k_parent = NULL;
234 newkey->k_nmlen = strlen(newname);
235 memcpy(newkey->k_name, newname, newkey->k_nmlen);
236 newkey_len = SMBFS_KEYSIZE(newkey->k_nmlen);
237
238 /* Release parent and mark as gone. */
239 if (np->n_parent && (np->n_flag & NREFPARENT)) {
240 vrele(np->n_parent);
241 np->n_flag &= ~NREFPARENT;
242 }
243 np->n_flag |= NGONE;
244
245 /* Rekey the node. */
246 error = vcache_rekey_enter(mp, vp, oldkey, key_len, newkey, newkey_len);
247 KASSERT(error == 0);
248 np->n_key = newkey;
249 vcache_rekey_exit(mp, vp, oldkey, key_len, newkey, newkey_len);
250
251 /* Purge from name cache and cleanup. */
252 cache_purge(vp);
253 kmem_free(key, key_len);
254 kmem_free(oldkey, key_len);
255
256 vput(vp);
257}

259/*
260 * Free smbnode, and give vnode back to system
261 */
262int
263smbfs_reclaim(void *v)
264{
265 struct vop_reclaim_v2_args /* {
266 struct vnode *a_vp;
267 struct thread *a_p;
268 } */ *ap = v;
269 struct vnode *vp = ap->a_vp;
270 struct vnode *dvp;
271 struct smbnode *np = VTOSMB(vp);
272 struct smbmount *smp = VTOSMBFS(vp);
273
274 VOP_UNLOCK(vp);
275
276 SMBVDEBUG("%.*s,%d\n", (int) np->n_nmlen, np->n_name, vp->v_usecount);
277
278 dvp = (np->n_parent && (np->n_flag & NREFPARENT)) ?
279 np->n_parent : NULL;
280
281 if (smp->sm_root == np) {
282 SMBVDEBUG0("root vnode\n");
283 smp->sm_root = NULL;
284 }
285
286 genfs_node_destroy(vp);
287
288 /* To interlock with smbfs_nget(). */
289 mutex_enter(vp->v_interlock);
290 vp->v_data = NULL;
291 mutex_exit(vp->v_interlock);
292
293 mutex_destroy(&np->n_lock);
294 kmem_free(np->n_key, SMBFS_KEYSIZE(np->n_nmlen));
295 pool_put(&smbfs_node_pool, np);
296 if (dvp) {
297 vrele(dvp);
298 /*
299 * Indicate that we released something; see comment
300 * in smbfs_unmount().
301 */
302 smp->sm_didrele = 1;
303 }
304 return 0;
305}

307int
308smbfs_inactive(void *v)
309{
310 struct vop_inactive_v2_args /* {
311 struct vnode *a_vp;
312 bool *a_recycle;
313 } */ *ap = v;
314 struct lwp *l = curlwp;
315 kauth_cred_t cred = l->l_cred;
316 struct vnode *vp = ap->a_vp;
317 struct smbnode *np = VTOSMB(vp);
318 struct smb_cred scred;
319
320 SMBVDEBUG("%.*s: %d\n", (int) np->n_nmlen, np->n_name, vp->v_usecount);
321 if ((np->n_flag & NOPEN) != 0) {
322 struct smb_share *ssp = np->n_mount->sm_share;
323
324 smbfs_vinvalbuf(vp, V_SAVE, cred, l, 1);
325 smb_makescred(&scred, l, cred);
326
327 if (vp->v_type == VDIR && np->n_dirseq) {
328 smbfs_findclose(np->n_dirseq, &scred);
329 np->n_dirseq = NULL;
330 }
331
332 if (vp->v_type != VDIR
333 || SMB_CAPS(SSTOVC(ssp)) & SMB_CAP_NT_SMBS)
334 smbfs_smb_close(ssp, np->n_fid, &np->n_mtime, &scred);
335
336 np->n_flag &= ~NOPEN;
337 smbfs_attr_cacheremove(vp);
338 }
339 *ap->a_recycle = ((vp->v_type == VNON) || (np->n_flag & NGONE) != 0);
340
341 return (0);
342}
343/*
344 * routines to maintain vnode attributes cache
345 * smbfs_attr_cacheenter: unpack np.i to vattr structure
346 */
347void
348smbfs_attr_cacheenter(struct vnode *vp, struct smbfattr *fap)
349{
350 struct smbnode *np = VTOSMB(vp);
351
352 if (vp->v_type == VREG) {
353 if (np->n_size != fap->fa_size) {
354 np->n_size = fap->fa_size;
355 uvm_vnp_setsize(vp, np->n_size);
356 }
357 } else if (vp->v_type == VDIR) {
358 np->n_size = 16384; /* should be a better way ... */
359 } else
360 return;
361
362 np->n_mtime = fap->fa_mtime;
363 np->n_dosattr = fap->fa_attr;
364
365 np->n_attrage = time_uptime;
366}

368int
369smbfs_attr_cachelookup(struct vnode *vp, struct vattr *va)
370{
371 struct smbnode *np = VTOSMB(vp);
372 struct smbmount *smp = VTOSMBFS(vp);
373 time_t diff;
374
375 diff = time_uptime - np->n_attrage;
376 if (diff > SMBFS_ATTRTIMO) /* XXX should be configurable */
377 return ENOENT;
378
379 va->va_type = vp->v_type; /* vnode type (for create) */
380 if (vp->v_type == VREG) {
381 va->va_mode = smp->sm_args.file_mode; /* files access mode and type */
382 } else if (vp->v_type == VDIR) {
383 va->va_mode = smp->sm_args.dir_mode; /* files access mode and type */
384 } else
385 return EINVAL;
386 va->va_size = np->n_size;
387 va->va_nlink = 1; /* number of references to file */
388 va->va_uid = smp->sm_args.uid; /* owner user id */
389 va->va_gid = smp->sm_args.gid; /* owner group id */
390 va->va_fsid = vp->v_mount->mnt_stat.f_fsidx.__fsid_val[0];
391 va->va_fileid = np->n_ino; /* file id */
392 if (va->va_fileid == 0)
393 va->va_fileid = 2;
394 va->va_blocksize = SSTOVC(smp->sm_share)->vc_txmax;
395 va->va_mtime = np->n_mtime;
396 va->va_atime = va->va_ctime = va->va_mtime; /* time file changed */
397 va->va_gen = VNOVAL; /* generation number of file */
398 va->va_flags = 0; /* flags defined for file */
399 va->va_rdev = VNOVAL; /* device the special file represents */
400 va->va_bytes = va->va_size; /* bytes of disk space held by file */
401 va->va_filerev = 0; /* file modification number */
402 va->va_vaflags = 0; /* operations flags */
403 return 0;
404}
