85/*
86 * Search context supplied by server
87 */
88#define SMB_SKEYLEN 21 /* search context */
89#define SMB_DENTRYLEN (SMB_SKEYLEN + 22) /* entire entry */
90
91struct smbfs_fctx {
92 /*
93 * Setable values
94 */
95 int f_flags; /* SMBFS_RDD_ */
96 /*
97 * Return values
98 */
99 struct smbfattr f_attr; /* current attributes */
100 char * f_name; /* current file name */
101 int f_nmlen; /* name len */
102 /*
103 * Internal variables
104 */
105 int f_limit; /* maximum number of entries */
106 int f_attrmask; /* SMB_FA_ */
107 int f_wclen;
108 const char * f_wildcard;
109 struct smbnode* f_dnp;
110 struct smb_cred*f_scred;
111 struct smb_share *f_ssp;
112 union {
113 struct smb_rq * uf_rq;
114 struct smb_t2rq * uf_t2;
115 } f_urq;
116 int f_left; /* entries left */
117 int f_ecnt; /* entries left in the current reponse */
118 int f_eofs; /* entry offset in the parameter block */
119 u_char f_skey[SMB_SKEYLEN]; /* server side search context */
120 char f_fname[8 + 1 + 3 + 1]; /* common case for 8.3 filenames */
121 u_int16_t f_Sid;
122 u_int16_t f_infolevel;
123 int f_rnamelen;
124 char * f_rname; /* resume name/key */
125 int f_rnameofs;
126};
127
128#define f_rq f_urq.uf_rq
129#define f_t2 f_urq.uf_t2
