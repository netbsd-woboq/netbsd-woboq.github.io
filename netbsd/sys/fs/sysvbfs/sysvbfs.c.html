139
140static int
141sysvbfs_modcmd(modcmd_t cmd, void *arg)
142{
143
144 switch (cmd) {
145 case MODULE_CMD_INIT:
146 return vfs_attach(&sysvbfs_vfsops);
147 case MODULE_CMD_FINI:
148 return vfs_detach(&sysvbfs_vfsops);
149 default:
150 return ENOTTY;
151 }
152}
