101int
102tmpfs_fifo_close(void *v)
103{
104 struct vop_close_args /* {
105 struct vnode *a_vp;
106 int a_fflag;
107 kauth_cred_t a_cred;
108 } */ *ap __unused = v;
109
110 return VOCALL(fifo_vnodeop_p, VOFFSET(vop_close), v);
111}
112
113int
114tmpfs_fifo_read(void *v)
115{
116 struct vop_read_args /* {
117 struct vnode *a_vp;
118 struct uio *a_uio;
119 int a_ioflag;
120 kauth_cred_t a_cred;
121 } */ *ap = v;
122 vnode_t *vp = ap->a_vp;
123
124 tmpfs_update(vp, TMPFS_UPDATE_ATIME);
125 return VOCALL(fifo_vnodeop_p, VOFFSET(vop_read), v);
126}
127
128int
129tmpfs_fifo_write(void *v)
130{
131 struct vop_write_args /* {
132 struct vnode *a_vp;
133 struct uio *a_uio;
134 int a_ioflag;
135 kauth_cred_t a_cred;
136 } */ *ap = v;
137 vnode_t *vp = ap->a_vp;
138
139 tmpfs_update(vp, TMPFS_UPDATE_MTIME);
140 return VOCALL(fifo_vnodeop_p, VOFFSET(vop_write), v);
141}
