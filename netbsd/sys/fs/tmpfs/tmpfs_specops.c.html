104int
105tmpfs_spec_close(void *v)
106{
107 struct vop_close_args /* {
108 struct vnode *a_vp;
109 int a_fflag;
110 kauth_cred_t a_cred;
111 } */ *ap __unused = v;
112
113 return VOCALL(spec_vnodeop_p, VOFFSET(vop_close), v);
114}
115
116int
117tmpfs_spec_read(void *v)
118{
119 struct vop_read_args /* {
120 struct vnode *a_vp;
121 struct uio *a_uio;
122 int a_ioflag;
123 kauth_cred_t a_cred;
124 } */ *ap = v;
125 vnode_t *vp = ap->a_vp;
126
127 tmpfs_update(vp, TMPFS_UPDATE_ATIME);
128 return VOCALL(spec_vnodeop_p, VOFFSET(vop_read), v);
129}
130
131int
132tmpfs_spec_write(void *v)
133{
134 struct vop_write_args /* {
135 struct vnode *a_vp;
136 struct uio *a_uio;
137 int a_ioflag;
138 kauth_cred_t a_cred;
139 } */ *ap = v;
140 vnode_t *vp = ap->a_vp;
141
142 tmpfs_update(vp, TMPFS_UPDATE_MTIME);
143 return VOCALL(spec_vnodeop_p, VOFFSET(vop_write), v);
144}
