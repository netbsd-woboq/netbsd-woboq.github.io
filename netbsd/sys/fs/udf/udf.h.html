188/* logical volume open/close actions */
189#define UDF_OPEN_SESSION 0x01 /* if needed writeout VRS + VDS */
190#define UDF_CLOSE_SESSION 0x02 /* close session after writing VAT */
191#define UDF_FINALISE_DISC 0x04 /* close session after writing VAT */
192#define UDF_WRITE_VAT 0x08 /* sequential VAT filesystem */
193#define UDF_WRITE_LVINT 0x10 /* write out open lvint */
194#define UDF_WRITE_PART_BITMAPS 0x20 /* write out partition space bitmaps */
195#define UDF_APPENDONLY_LVINT 0x40 /* no shifting, only appending */
196#define UDF_WRITE_METAPART_NODES 0x80 /* write out metadata partition nodes*/
197#define UDFLOGVOL_BITS "\20\1OPEN_SESSION\2CLOSE_SESSION\3FINALISE_DISC" \
198 "\4WRITE_VAT\5WRITE_LVINT\6WRITE_PART_BITMAPS" \
199 "\7APPENDONLY_LVINT\10WRITE_METAPART_NODES"

267struct udf_mount {
268 struct mount *vfs_mountp;
269 struct vnode *devvp;
270 struct mmc_discinfo discinfo;
271 struct udf_args mount_args;
272
273 /* format descriptors */
274 kmutex_t logvol_mutex;
275 struct anchor_vdp *anchors[UDF_ANCHORS]; /* anchors to VDS */
276 struct pri_vol_desc *primary_vol; /* identification */
277 struct logvol_desc *logical_vol; /* main mapping v->p */
278 struct unalloc_sp_desc *unallocated; /* free UDF space */
279 struct impvol_desc *implementation; /* likely reduntant */
280 struct logvol_int_desc *logvol_integrity; /* current integrity */
281 struct part_desc *partitions[UDF_PARTITIONS]; /* partitions */
282 /* logvol_info is derived; points *into* other structures */
283 struct udf_logvol_info *logvol_info; /* integrity descr. */
284
285 /* fileset and root directories */
286 struct fileset_desc *fileset_desc; /* normally one */
287
288 /* tracing logvol integrity history */
289 struct udf_lvintq lvint_trace[UDF_LVDINT_SEGMENTS];
290 int lvopen; /* logvol actions */
291 int lvclose; /* logvol actions */
292
293 /* logical to physical translations */
294 int vtop[UDF_PMAPS+1]; /* vpartnr trans */
295 int vtop_tp[UDF_PMAPS+1]; /* type of trans */
296
297 /* disc allocation / writing method */
298 kmutex_t allocate_mutex;
299 int lvreadwrite; /* error handling */
300 int vtop_alloc[UDF_PMAPS+1]; /* alloc scheme */
301 int data_part;
302 int node_part;
303 int fids_part;
304
305 /* sequential track info */
306 struct mmc_trackinfo data_track;
307 struct mmc_trackinfo metadata_track;
308
309 /* VAT */
310 uint32_t first_possible_vat_location;
311 uint32_t last_possible_vat_location;
312 uint32_t vat_entries;
313 uint32_t vat_offset; /* offset in table */
314 uint32_t vat_last_free_lb; /* last free lb_num */
315 uint32_t vat_table_len;
316 uint32_t vat_table_alloc_len;
317 uint8_t *vat_table;
318 uint8_t *vat_pages; /* TODO */
319 struct udf_node *vat_node; /* system node */
320
321 /* space bitmaps for physical partitions */
322 struct space_bitmap_desc*part_unalloc_dscr[UDF_PARTITIONS];
323 struct space_bitmap_desc*part_freed_dscr [UDF_PARTITIONS];
324 struct udf_bitmap part_unalloc_bits[UDF_PARTITIONS];
325 struct udf_bitmap part_freed_bits [UDF_PARTITIONS];
326
327 /* sparable */
328 uint32_t sparable_packet_size;
329 uint32_t packet_size;
330 struct udf_sparing_table*sparing_table;
331
332 /* meta */
333 struct udf_node *metadata_node; /* system node */
334 struct udf_node *metadatamirror_node; /* system node */
335 struct udf_node *metadatabitmap_node; /* system node */
336 struct space_bitmap_desc*metadata_unalloc_dscr;
337 struct udf_bitmap metadata_unalloc_bits;
338 uint32_t metadata_alloc_unit_size;
339 uint16_t metadata_alignment_unit_size;
340 uint8_t metadata_flags;
341
342 /* rb tree for lookup icb to udf_node and sorted list for sync */
343 struct rb_tree udf_node_tree;
344
345 /* syncing */
346 int syncing; /* are we syncing? */
347 kmutex_t sync_lock; /* serialize syncing */
348
349 /* late allocation */
350 int32_t uncommitted_lbs[UDF_PARTITIONS];
351 struct long_ad *la_node_ad_cpy; /* issue buf */
352 uint64_t *la_lmapping, *la_pmapping; /* issue buf */

354 /* lists */
355 STAILQ_HEAD(udfmntpts, udf_mount) all_udf_mntpnts;

357 /* device strategy */
358 struct udf_strategy *strategy;
359 void *strategy_private;

362/*
363 * UDF node describing a file/directory.
364 *
365 * BUGALERT claim node_mutex before reading/writing to prevent inconsistencies !
366 */
367struct udf_node {
368 struct genfs_node i_gnode; /* has to be first */
369 struct vnode *vnode; /* vnode associated */
370 struct udf_mount *ump;
371
372 kmutex_t node_mutex;
373 kcondvar_t node_lock; /* sleeping lock */
374 char const *lock_fname;
375 int lock_lineno;
376
377 /* rb_node for fast lookup and fast sequential visiting */
378 struct rb_node rbnode;
379
380 /* one of `fe' or `efe' can be set, not both (UDF file entry dscr.) */
381 struct file_entry *fe;
382 struct extfile_entry *efe;
383 struct alloc_ext_entry *ext[UDF_MAX_ALLOC_EXTENTS];
384 int num_extensions;
385
386 /* location found, recording location & hints */
387 struct long_ad loc; /* FID/hash loc. */
388 struct long_ad write_loc; /* strat 4096 loc */
389 int needs_indirect; /* has missing indr. */
390 struct long_ad ext_loc[UDF_MAX_ALLOC_EXTENTS];
391
392 struct dirhash *dir_hash;
393
394 /* misc */
395 uint32_t i_flags; /* associated flags */
396 struct lockf *lockf; /* lock list */
397 uint32_t outstanding_bufs; /* file data */
398 uint32_t outstanding_nodedscr; /* node dscr */
399 int32_t uncommitted_lbs; /* in UBC */
400
401 /* references to associated nodes */
402 struct udf_node *extattr;
403 struct udf_node *streamdir;
404 struct udf_node *my_parent; /* if extended attr. */
405};
