255static int
256v7fs_modcmd(modcmd_t cmd, void *arg)
257{
258
259 switch (cmd) {
260 case MODULE_CMD_INIT:
261 return vfs_attach(&v7fs_vfsops);
262 case MODULE_CMD_FINI:
263 return vfs_detach(&v7fs_vfsops);
264 default:
265 return ENOTTY;
266 }
267}
