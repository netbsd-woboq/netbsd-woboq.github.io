43#define ELF32_AUXSIZE (ELF_AUX_ENTRIES * sizeof(Aux32Info) \
44 + MAXPATHLEN + ALIGN(1))

52MODULE(MODULE_CLASS_EXEC, exec_elf32, DEP);

54static struct execsw exec_elf32_execsw[] = {
55 {
56 .es_hdrsz = sizeof (Elf32_Ehdr),
57 .es_makecmds = exec_elf32_makecmds,
58 .u = {
59 .elf_probe_func = netbsd_elf32_probe,
60 },
61 .es_emul = &emul_netbsd,
62 .es_prio = EXECSW_PRIO_FIRST,
63 .es_arglen = ELF32_AUXSIZE,
64 .es_copyargs = elf32_copyargs,
65 .es_setregs = NULL,
66 .es_coredump = coredump_elf32,
67 .es_setup_stack = exec_setup_stack,
68 },
69#if EXEC_ELF_NOTELESS
70 {
71 .es_hdrsz = sizeof (Elf32_Ehdr),
72 .es_makecmds = exec_elf32_makecmds,
73 .u {
74 elf_probe_func = NULL,
75 },
76 .es_emul = &emul_netbsd,
77 .es_prio = EXECSW_PRIO_LAST,
78 .es_arglen = ELF32_AUXSIZE,
79 .es_copyargs = elf32_copyargs,
80 .es_setregs = NULL,
81 .es_coredump = coredump_elf32,
82 .es_setup_stack = exec_setup_stack,
83 },
84#endif
85};

87static int
88exec_elf32_modcmd(modcmd_t cmd, void *arg)
89{
90#if ARCH_ELFSIZE == 64
91 /*
92 * If we are on a 64bit system, we don't want the 32bit execsw[] to be
93 * added in the global array, because the exec_elf32 module only works
94 * on 32bit systems.
95 *
96 * However, we need the exec_elf32 module, because it will make the 32bit
97 * functions available for netbsd32 and linux32.
98 *
99 * Therefore, allow this module on 64bit systems, but make it dormant.
100 */
101
102 (void)exec_elf32_execsw; /* unused */
103
104 switch (cmd) {
105 case MODULE_CMD_INIT:
106 case MODULE_CMD_FINI:
107 return 0;
108 default:
109 return ENOTTY;
110 }
111#else /* ARCH_ELFSIZE == 64 */
112 switch (cmd) {
113 case MODULE_CMD_INIT:
114 return exec_add(exec_elf32_execsw,
115 __arraycount(exec_elf32_execsw));
116
117 case MODULE_CMD_FINI:
118 return exec_remove(exec_elf32_execsw,
119 __arraycount(exec_elf32_execsw));
120
121 default:
122 return ENOTTY;
123 }
124#endif /* ARCH_ELFSIZE == 64 */
125}
