52#define VMCMD_EVCNT_DECL(name) \
53static struct evcnt vmcmd_ev_##name = \
54 EVCNT_INITIALIZER(EVCNT_TYPE_MISC, NULL, "vmcmd", #name); \
55EVCNT_ATTACH_STATIC(vmcmd_ev_##name)

57#define VMCMD_EVCNT_INCR(name) \
58 vmcmd_ev_##name.ev_count++

73/*
74 * new_vmcmd():
75 * create a new vmcmd structure and fill in its fields based
76 * on function call arguments. make sure objects ref'd by
77 * the vmcmd are 'held'.
78 */
79
80void
81new_vmcmd(struct exec_vmcmd_set *evsp,
82 int (*proc)(struct lwp * l, struct exec_vmcmd *),
83 vsize_t len, vaddr_t addr, struct vnode *vp, u_long offset,
84 u_int prot, int flags)
85{
86 struct exec_vmcmd *vcp;
87
88 VMCMD_EVCNT_INCR(calls);
89 KASSERT(proc != vmcmd_map_pagedvn || (vp->v_iflag & VI_TEXT));
90 KASSERT(vp == NULL || vp->v_usecount > 0);
91
92 if (evsp->evs_used >= evsp->evs_cnt)
93 vmcmdset_extend(evsp);
94 vcp = &evsp->evs_cmds[evsp->evs_used++];
95 vcp->ev_proc = proc;
96 vcp->ev_len = len;
97 vcp->ev_addr = addr;
98 if ((vcp->ev_vp = vp) != NULL)
99 vref(vp);
100 vcp->ev_offset = offset;
101 vcp->ev_prot = prot;
102 vcp->ev_flags = flags;
103}
104
105void
106vmcmdset_extend(struct exec_vmcmd_set *evsp)
107{
108 struct exec_vmcmd *nvcp;
109 u_int ocnt;
110
111#ifdef DIAGNOSTIC
112 if (evsp->evs_used < evsp->evs_cnt)
113 panic("vmcmdset_extend: not necessary");
114#endif
115
116 /* figure out number of entries in new set */
117 if ((ocnt = evsp->evs_cnt) != 0) {
118 evsp->evs_cnt += ocnt;
119 VMCMD_EVCNT_INCR(extends);
120 } else
121 evsp->evs_cnt = EXEC_DEFAULT_VMCMD_SETSIZE;
122
123 /* allocate it */
124 nvcp = kmem_alloc(evsp->evs_cnt * sizeof(struct exec_vmcmd), KM_SLEEP);
125
126 /* free the old struct, if there was one, and record the new one */
127 if (ocnt) {
128 memcpy(nvcp, evsp->evs_cmds,
129 (ocnt * sizeof(struct exec_vmcmd)));
130 kmem_free(evsp->evs_cmds, ocnt * sizeof(struct exec_vmcmd));
131 }
132 evsp->evs_cmds = nvcp;
133}
134
135void
136kill_vmcmds(struct exec_vmcmd_set *evsp)
137{
138 struct exec_vmcmd *vcp;
139 u_int i;
140
141 VMCMD_EVCNT_INCR(kills);
142
143 if (evsp->evs_cnt == 0)
144 return;
145
146 for (i = 0; i < evsp->evs_used; i++) {
147 vcp = &evsp->evs_cmds[i];
148 if (vcp->ev_vp != NULL)
149 vrele(vcp->ev_vp);
150 }
151 kmem_free(evsp->evs_cmds, evsp->evs_cnt * sizeof(struct exec_vmcmd));
152 evsp->evs_used = evsp->evs_cnt = 0;
153}

155/*
156 * vmcmd_map_pagedvn():
157 * handle vmcmd which specifies that a vnode should be mmap'd.
158 * appropriate for handling demand-paged text and data segments.
159 */
160
161static int
162vmcmd_get_prot(struct lwp *l, const struct exec_vmcmd *cmd, vm_prot_t *prot,
163 vm_prot_t *maxprot)
164{
165
166 *prot = cmd->ev_prot;
167 *maxprot = PAX_MPROTECT_MAXPROTECT(l, *prot, 0, UVM_PROT_ALL);
168
169 if ((*prot & *maxprot) != *prot)
170 return EACCES;
171 return PAX_MPROTECT_VALIDATE(l, *prot);
172}
173
174int
175vmcmd_map_pagedvn(struct lwp *l, struct exec_vmcmd *cmd)
176{
177 struct uvm_object *uobj;
178 struct vnode *vp = cmd->ev_vp;
179 struct proc *p = l->l_proc;
180 int error;
181 vm_prot_t prot, maxprot;
182
183 KASSERT(vp->v_iflag & VI_TEXT);
184
185 /*
186 * map the vnode in using uvm_map.
187 */
188
189 if (cmd->ev_len == 0)
190 return 0;
191 if (cmd->ev_offset & PAGE_MASK)
192 return EINVAL;
193 if (cmd->ev_addr & PAGE_MASK)
194 return EINVAL;
195 if (cmd->ev_len & PAGE_MASK)
196 return EINVAL;
197
198 if ((error = vmcmd_get_prot(l, cmd, &prot, &maxprot)) != 0)
199 return error;
200
201 /*
202 * check the file system's opinion about mmapping the file
203 */
204
205 error = VOP_MMAP(vp, prot, l->l_cred);
206 if (error)
207 return error;
208
209 if ((vp->v_vflag & VV_MAPPED) == 0) {
210 vn_lock(vp, LK_EXCLUSIVE | LK_RETRY);
211 vp->v_vflag |= VV_MAPPED;
212 VOP_UNLOCK(vp);
213 }
214
215 /*
216 * do the map, reference the object for this map entry
217 */
218 uobj = &vp->v_uobj;
219 vref(vp);
220
221 error = uvm_map(&p->p_vmspace->vm_map, &cmd->ev_addr, cmd->ev_len,
222 uobj, cmd->ev_offset, 0,
223 UVM_MAPFLAG(prot, maxprot, UVM_INH_COPY,
224 UVM_ADV_NORMAL, UVM_FLAG_COPYONW|UVM_FLAG_FIXED));
225 if (error) {
226 uobj->pgops->pgo_detach(uobj);
227 }
228 return error;
229}

231/*
232 * vmcmd_map_readvn():
233 * handle vmcmd which specifies that a vnode should be read from.
234 * appropriate for non-demand-paged text/data segments, i.e. impure
235 * objects (a la OMAGIC and NMAGIC).
236 */
237int
238vmcmd_map_readvn(struct lwp *l, struct exec_vmcmd *cmd)
239{
240 struct proc *p = l->l_proc;
241 int error;
242 long diff;
243
244 if (cmd->ev_len == 0)
245 return 0;
246
247 diff = cmd->ev_addr - trunc_page(cmd->ev_addr);
248 cmd->ev_addr -= diff; /* required by uvm_map */
249 cmd->ev_offset -= diff;
250 cmd->ev_len += diff;
251
252 error = uvm_map(&p->p_vmspace->vm_map, &cmd->ev_addr,
253 round_page(cmd->ev_len), NULL, UVM_UNKNOWN_OFFSET, 0,
254 UVM_MAPFLAG(UVM_PROT_ALL, UVM_PROT_ALL, UVM_INH_COPY,
255 UVM_ADV_NORMAL,
256 UVM_FLAG_FIXED|UVM_FLAG_OVERLAY|UVM_FLAG_COPYONW));
257
258 if (error)
259 return error;
260
261 return vmcmd_readvn(l, cmd);
262}
263
264int
265vmcmd_readvn(struct lwp *l, struct exec_vmcmd *cmd)
266{
267 struct proc *p = l->l_proc;
268 int error;
269 vm_prot_t prot, maxprot;
270
271 error = vn_rdwr(UIO_READ, cmd->ev_vp, (void *)cmd->ev_addr,
272 cmd->ev_len, cmd->ev_offset, UIO_USERSPACE, IO_UNIT,
273 l->l_cred, NULL, l);
274 if (error)
275 return error;
276
277 if ((error = vmcmd_get_prot(l, cmd, &prot, &maxprot)) != 0)
278 return error;
279
280#ifdef PMAP_NEED_PROCWR
281 /*
282 * we had to write the process, make sure the pages are synched
283 * with the instruction cache.
284 */
285 if (prot & VM_PROT_EXECUTE)
286 pmap_procwr(p, cmd->ev_addr, cmd->ev_len);
287#endif
288
289 /*
290 * we had to map in the area at PROT_ALL so that vn_rdwr()
291 * could write to it. however, the caller seems to want
292 * it mapped read-only, so now we are going to have to call
293 * uvm_map_protect() to fix up the protection. ICK.
294 */
295 if (maxprot != VM_PROT_ALL) {
296 error = uvm_map_protect(&p->p_vmspace->vm_map,
297 trunc_page(cmd->ev_addr),
298 round_page(cmd->ev_addr + cmd->ev_len),
299 maxprot, true);
300 if (error)
301 return error;
302 }
303
304 if (prot != maxprot) {
305 error = uvm_map_protect(&p->p_vmspace->vm_map,
306 trunc_page(cmd->ev_addr),
307 round_page(cmd->ev_addr + cmd->ev_len),
308 prot, false);
309 if (error)
310 return error;
311 }
312
313 return 0;
314}

316/*
317 * vmcmd_map_zero():
318 * handle vmcmd which specifies a zero-filled address space region. The
319 * address range must be first allocated, then protected appropriately.
320 */
321
322int
323vmcmd_map_zero(struct lwp *l, struct exec_vmcmd *cmd)
324{
325 struct proc *p = l->l_proc;
326 int error;
327 long diff;
328 vm_prot_t prot, maxprot;
329
330 diff = cmd->ev_addr - trunc_page(cmd->ev_addr);
331 cmd->ev_addr -= diff; /* required by uvm_map */
332 cmd->ev_len += diff;
333
334 if ((error = vmcmd_get_prot(l, cmd, &prot, &maxprot)) != 0)
335 return error;
336
337 error = uvm_map(&p->p_vmspace->vm_map, &cmd->ev_addr,
338 round_page(cmd->ev_len), NULL, UVM_UNKNOWN_OFFSET, 0,
339 UVM_MAPFLAG(prot, maxprot, UVM_INH_COPY,
340 UVM_ADV_NORMAL,
341 UVM_FLAG_FIXED|UVM_FLAG_COPYONW));
342 if (cmd->ev_flags & VMCMD_STACK)
343 curproc->p_vmspace->vm_issize += atop(round_page(cmd->ev_len));
344 return error;
345}

347/*
348 * exec_read_from():
349 *
350 * Read from vnode into buffer at offset.
351 */
352int
353exec_read_from(struct lwp *l, struct vnode *vp, u_long off, void *bf,
354 size_t size)
355{
356 int error;
357 size_t resid;
358
359 if ((error = vn_rdwr(UIO_READ, vp, bf, size, off, UIO_SYSSPACE,
360 0, l->l_cred, &resid, NULL)) != 0)
361 return error;
362 /*
363 * See if we got all of it
364 */
365 if (resid != 0)
366 return ENOEXEC;
367 return 0;
368}

370/*
371 * exec_setup_stack(): Set up the stack segment for an elf
372 * executable.
373 *
374 * Note that the ep_ssize parameter must be set to be the current stack
375 * limit; this is adjusted in the body of execve() to yield the
376 * appropriate stack segment usage once the argument length is
377 * calculated.
378 *
379 * This function returns an int for uniformity with other (future) formats'
380 * stack setup functions. They might have errors to return.
381 */
382
383int
384exec_setup_stack(struct lwp *l, struct exec_package *epp)
385{
386 vsize_t max_stack_size;
387 vaddr_t access_linear_min;
388 vsize_t access_size;
389 vaddr_t noaccess_linear_min;
390 vsize_t noaccess_size;
391
392#ifndef USRSTACK32
393#define USRSTACK32 (0x00000000ffffffffL&~PGOFSET)
394#endif
395#ifndef MAXSSIZ32
396#define MAXSSIZ32 (MAXSSIZ >> 2)
397#endif
398
399 if (epp->ep_flags & EXEC_32) {
400 epp->ep_minsaddr = USRSTACK32;
401 max_stack_size = MAXSSIZ32;
402 } else {
403 epp->ep_minsaddr = USRSTACK;
404 max_stack_size = MAXSSIZ;
405 }
406
407 DPRINTF(("ep_minsaddr=%#jx max_stack_size=%#jx\n",
408 (uintmax_t)epp->ep_minsaddr, (uintmax_t)max_stack_size));
409
410 pax_aslr_stack(epp, &max_stack_size);
411
412 DPRINTF(("[RLIMIT_STACK].lim_cur=%#jx max_stack_size=%#jx\n",
413 (uintmax_t)l->l_proc->p_rlimit[RLIMIT_STACK].rlim_cur,
414 (uintmax_t)max_stack_size));
415 epp->ep_ssize = MIN(l->l_proc->p_rlimit[RLIMIT_STACK].rlim_cur,
416 max_stack_size);
417
418 l->l_proc->p_stackbase = epp->ep_minsaddr;
419
420 epp->ep_maxsaddr = (vaddr_t)STACK_GROW(epp->ep_minsaddr,
421 max_stack_size);
422
423 DPRINTF(("ep_ssize=%#jx ep_minsaddr=%#jx ep_maxsaddr=%#jx\n",
424 (uintmax_t)epp->ep_ssize, (uintmax_t)epp->ep_minsaddr,
425 (uintmax_t)epp->ep_maxsaddr));
426
427 /*
428 * set up commands for stack. note that this takes *two*, one to
429 * map the part of the stack which we can access, and one to map
430 * the part which we can't.
431 *
432 * arguably, it could be made into one, but that would require the
433 * addition of another mapping proc, which is unnecessary
434 */
435 access_size = epp->ep_ssize;
436 access_linear_min = (vaddr_t)STACK_ALLOC(epp->ep_minsaddr, access_size);
437 noaccess_size = max_stack_size - access_size;
438 noaccess_linear_min = (vaddr_t)STACK_ALLOC(STACK_GROW(epp->ep_minsaddr,
439 access_size), noaccess_size);
440
441 DPRINTF(("access_size=%#jx, access_linear_min=%#jx, "
442 "noaccess_size=%#jx, noaccess_linear_min=%#jx\n",
443 (uintmax_t)access_size, (uintmax_t)access_linear_min,
444 (uintmax_t)noaccess_size, (uintmax_t)noaccess_linear_min));
445
446 if (user_stack_guard_size > 0) {
447#ifdef __MACHINE_STACK_GROWS_UP
448 vsize_t guard_size = MIN(VM_MAXUSER_ADDRESS - epp->ep_maxsaddr, user_stack_guard_size);
449 if (guard_size > 0)
450 NEW_VMCMD(&epp->ep_vmcmds, vmcmd_map_zero, guard_size,
451 epp->ep_maxsaddr, NULL, 0, VM_PROT_NONE);
452#else
453 NEW_VMCMD(&epp->ep_vmcmds, vmcmd_map_zero, user_stack_guard_size,
454 epp->ep_maxsaddr - user_stack_guard_size, NULL, 0, VM_PROT_NONE);
455#endif
456 }
457 if (noaccess_size > 0 && noaccess_size <= MAXSSIZ) {
458 NEW_VMCMD2(&epp->ep_vmcmds, vmcmd_map_zero, noaccess_size,
459 noaccess_linear_min, NULL, 0, VM_PROT_NONE, VMCMD_STACK);
460 }
461 KASSERT(access_size > 0 && access_size <= MAXSSIZ);
462 NEW_VMCMD2(&epp->ep_vmcmds, vmcmd_map_zero, access_size,
463 access_linear_min, NULL, 0, VM_PROT_READ | VM_PROT_WRITE,
464 VMCMD_STACK);
465
466 return 0;
467}
