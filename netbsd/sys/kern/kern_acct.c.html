96/*
97 * The routines implemented in this file are described in:
98 * Leffler, et al.: The Design and Implementation of the 4.3BSD
99 * UNIX Operating System (Addison Welley, 1989)
100 * on pages 62-63.
101 *
102 * Arguably, to simplify accounting operations, this mechanism should
103 * be replaced by one in which an accounting log file (similar to /dev/klog)
104 * is read by a user process, etc. However, that has its own problems.
105 */
106
107/*
108 * Lock to serialize system calls and kernel threads.
109 */
110krwlock_t acct_lock;
111
112/*
113 * The global accounting state and related data. Gain the mutex before
114 * accessing these variables.
115 */
116static enum {
117 ACCT_STOP,
118 ACCT_ACTIVE,
119 ACCT_SUSPENDED
120} acct_state; /* The current accounting state. */
121static struct vnode *acct_vp; /* Accounting vnode pointer. */
122static kauth_cred_t acct_cred; /* Credential of accounting file
123 owner (i.e root). Used when
124 accounting file i/o. */
125static struct lwp *acct_dkwatcher; /* Free disk space checker. */
126
127/*
128 * Values associated with enabling and disabling accounting
129 */
130int acctsuspend = 2; /* stop accounting when < 2% free space left */
131int acctresume = 4; /* resume when free space risen to > 4% */
132int acctchkfreq = 15; /* frequency (in seconds) to check space */
133
134/*
135 * Encode_comp_t converts from ticks in seconds and microseconds
136 * to ticks in 1/AHZ seconds. The encoding is described in
137 * Leffler, et al., on page 63.
138 */
139
140#define MANTSIZE 13 /* 13 bit mantissa. */
141#define EXPSIZE 3 /* Base 8 (3 bit) exponent. */
142#define MAXFRACT ((1 << MANTSIZE) - 1) /* Maximum fractional value. */
143
144static comp_t
145encode_comp_t(u_long s, u_long us)
146{
147 int exp, rnd;
148
149 exp = 0;
150 rnd = 0;
151 s *= AHZ;
152 s += us / (1000000 / AHZ); /* Maximize precision. */
153
154 while (s > MAXFRACT) {
155 rnd = s & (1 << (EXPSIZE - 1)); /* Round up? */
156 s >>= EXPSIZE; /* Base 8 exponent == 3 bit shift. */
157 exp++;
158 }
159
160 /* If we need to round up, do it (and handle overflow correctly). */
161 if (rnd && (++s > MAXFRACT)) {
162 s >>= EXPSIZE;
163 exp++;
164 }
165
166 /* Clean it up and polish it off. */
167 exp <<= MANTSIZE; /* Shift the exponent into place */
168 exp += s; /* and add on the mantissa. */
169 return (exp);
170}

172static int
173acct_chkfree(void)
174{
175 int error;
176 struct statvfs *sb;
177 fsblkcnt_t bavail;
178
179 sb = kmem_alloc(sizeof(*sb), KM_SLEEP);
180 error = VFS_STATVFS(acct_vp->v_mount, sb);
181 if (error != 0) {
182 kmem_free(sb, sizeof(*sb));
183 return (error);
184 }
185
186 if (sb->f_bfree < sb->f_bresvd) {
187 bavail = 0;
188 } else {
189 bavail = sb->f_bfree - sb->f_bresvd;
190 }
191
192 switch (acct_state) {
193 case ACCT_SUSPENDED:
194 if (bavail > acctresume * sb->f_blocks / 100) {
195 acct_state = ACCT_ACTIVE;
196 log(LOG_NOTICE, "Accounting resumed\n");
197 }
198 break;
199 case ACCT_ACTIVE:
200 if (bavail <= acctsuspend * sb->f_blocks / 100) {
201 acct_state = ACCT_SUSPENDED;
202 log(LOG_NOTICE, "Accounting suspended\n");
203 }
204 break;
205 case ACCT_STOP:
206 break;
207 }
208 kmem_free(sb, sizeof(*sb));
209 return (0);
210}
211
212static void
213acct_stop(void)
214{
215 int error;
216
217 KASSERT(rw_write_held(&acct_lock));
218
219 if (acct_vp != NULLVP && acct_vp->v_type != VBAD) {
220 error = vn_close(acct_vp, FWRITE, acct_cred);
221#ifdef DIAGNOSTIC
222 if (error != 0)
223 printf("acct_stop: failed to close, errno = %d\n",
224 error);
225#else
226 __USE(error);
227#endif
228 acct_vp = NULLVP;
229 }
230 if (acct_cred != NULL) {
231 kauth_cred_free(acct_cred);
232 acct_cred = NULL;
233 }
234 acct_state = ACCT_STOP;
235}

237/*
238 * Periodically check the file system to see if accounting
239 * should be turned on or off. Beware the case where the vnode
240 * has been vgone()'d out from underneath us, e.g. when the file
241 * system containing the accounting file has been forcibly unmounted.
242 */
243static void
244acctwatch(void *arg)
245{
246 int error;
247
248 log(LOG_NOTICE, "Accounting started\n");
249 rw_enter(&acct_lock, RW_WRITER);
250 while (acct_state != ACCT_STOP) {
251 if (acct_vp->v_type == VBAD) {
252 log(LOG_NOTICE, "Accounting terminated\n");
253 acct_stop();
254 continue;
255 }
256
257 error = acct_chkfree();
258#ifdef DIAGNOSTIC
259 if (error != 0)
260 printf("acctwatch: failed to statvfs, error = %d\n",
261 error);
262#else
263 __USE(error);
264#endif
265 rw_exit(&acct_lock);
266 error = kpause("actwat", false, acctchkfreq * hz, NULL);
267 rw_enter(&acct_lock, RW_WRITER);
268#ifdef DIAGNOSTIC
269 if (error != 0 && error != EWOULDBLOCK)
270 printf("acctwatch: sleep error %d\n", error);
271#endif
272 }
273 acct_dkwatcher = NULL;
274 rw_exit(&acct_lock);
275
276 kthread_exit(0);
277}
278
279void
280acct_init(void)
281{
282
283 acct_state = ACCT_STOP;
284 acct_vp = NULLVP;
285 acct_cred = NULL;
286 rw_init(&acct_lock);
287}

289/*
290 * Accounting system call. Written based on the specification and
291 * previous implementation done by Mark Tinguely.
292 */
293int
294sys_acct(struct lwp *l, const struct sys_acct_args *uap, register_t *retval)
295{
296 /* {
297 syscallarg(const char *) path;
298 } */
299 struct pathbuf *pb;
300 struct nameidata nd;
301 int error;
302
303 /* Make sure that the caller is root. */
304 if ((error = kauth_authorize_system(l->l_cred, KAUTH_SYSTEM_ACCOUNTING,
305 0, NULL, NULL, NULL)))
306 return (error);
307
308 /*
309 * If accounting is to be started to a file, open that file for
310 * writing and make sure it's a 'normal'.
311 */
312 if (SCARG(uap, path) != NULL) {
313 struct vattr va;
314 size_t pad;
315
316 error = pathbuf_copyin(SCARG(uap, path), &pb);
317 if (error) {
318 return error;
319 }
320 NDINIT(&nd, LOOKUP, FOLLOW | TRYEMULROOT, pb);
321 if ((error = vn_open(&nd, FWRITE|O_APPEND, 0)) != 0) {
322 pathbuf_destroy(pb);
323 return error;
324 }
325 if (nd.ni_vp->v_type != VREG) {
326 VOP_UNLOCK(nd.ni_vp);
327 error = EACCES;
328 goto bad;
329 }
330 if ((error = VOP_GETATTR(nd.ni_vp, &va, l->l_cred)) != 0) {
331 VOP_UNLOCK(nd.ni_vp);
332 goto bad;
333 }
334
335 if ((pad = (va.va_size % sizeof(struct acct))) != 0) {
336 u_quad_t size = va.va_size - pad;
337#ifdef DIAGNOSTIC
338 printf("Size of accounting file not a multiple of "
339 "%lu - incomplete record truncated\n",
340 (unsigned long)sizeof(struct acct));
341#endif
342 vattr_null(&va);
343 va.va_size = size;
344 error = VOP_SETATTR(nd.ni_vp, &va, l->l_cred);
345 if (error != 0) {
346 VOP_UNLOCK(nd.ni_vp);
347 goto bad;
348 }
349 }
350 VOP_UNLOCK(nd.ni_vp);
351 }
352
353 rw_enter(&acct_lock, RW_WRITER);
354
355 /*
356 * If accounting was previously enabled, kill the old space-watcher,
357 * free credential for accounting file i/o,
358 * ... (and, if no new file was specified, leave).
359 */
360 acct_stop();
361 if (SCARG(uap, path) == NULL)
362 goto out;
363
364 /*
365 * Save the new accounting file vnode and credential,
366 * and schedule the new free space watcher.
367 */
368 acct_state = ACCT_ACTIVE;
369 acct_vp = nd.ni_vp;
370 acct_cred = l->l_cred;
371 kauth_cred_hold(acct_cred);
372
373 pathbuf_destroy(pb);
374
375 error = acct_chkfree(); /* Initial guess. */
376 if (error != 0) {
377 acct_stop();
378 goto out;
379 }
380
381 if (acct_dkwatcher == NULL) {
382 error = kthread_create(PRI_NONE, KTHREAD_MPSAFE, NULL,
383 acctwatch, NULL, &acct_dkwatcher, "acctwatch");
384 if (error != 0)
385 acct_stop();
386 }
387
388 out:
389 rw_exit(&acct_lock);
390 return (error);
391 bad:
392 vn_close(nd.ni_vp, FWRITE, l->l_cred);
393 pathbuf_destroy(pb);
394 return error;
395}

397/*
398 * Write out process accounting information, on process exit.
399 * Data to be written out is specified in Leffler, et al.
400 * and are enumerated below. (They're also noted in the system
401 * "acct.h" header file.)
402 */
403int
404acct_process(struct lwp *l)
405{
406 struct acct acct;
407 struct timeval ut, st, tmp;
408 struct rusage *r;
409 int t, error = 0;
410 struct rlimit orlim;
411 struct proc *p = l->l_proc;
412
413 if (acct_state != ACCT_ACTIVE)
414 return 0;
415
416 memset(&acct, 0, sizeof(acct)); /* to zerofill padded data */
417
418 rw_enter(&acct_lock, RW_READER);
419
420 /* If accounting isn't enabled, don't bother */
421 if (acct_state != ACCT_ACTIVE)
422 goto out;
423
424 /*
425 * Temporarily raise the file limit so that accounting can't
426 * be stopped by the user.
427 *
428 * XXX We should think about the CPU limit, too.
429 */
430 lim_privatise(p);
431 orlim = p->p_rlimit[RLIMIT_FSIZE];
432 /* Set current and max to avoid illegal values */
433 p->p_rlimit[RLIMIT_FSIZE].rlim_cur = RLIM_INFINITY;
434 p->p_rlimit[RLIMIT_FSIZE].rlim_max = RLIM_INFINITY;
435
436 /*
437 * Get process accounting information.
438 */
439
440 /* (1) The name of the command that ran */
441 strncpy(acct.ac_comm, p->p_comm, sizeof(acct.ac_comm));
442
443 /* (2) The amount of user and system time that was used */
444 mutex_enter(p->p_lock);
445 calcru(p, &ut, &st, NULL, NULL);
446 mutex_exit(p->p_lock);
447 acct.ac_utime = encode_comp_t(ut.tv_sec, ut.tv_usec);
448 acct.ac_stime = encode_comp_t(st.tv_sec, st.tv_usec);
449
450 /* (3) The elapsed time the commmand ran (and its starting time) */
451 acct.ac_btime = p->p_stats->p_start.tv_sec;
452 getmicrotime(&tmp);
453 timersub(&tmp, &p->p_stats->p_start, &tmp);
454 acct.ac_etime = encode_comp_t(tmp.tv_sec, tmp.tv_usec);
455
456 /* (4) The average amount of memory used */
457 r = &p->p_stats->p_ru;
458 timeradd(&ut, &st, &tmp);
459 t = tmp.tv_sec * hz + tmp.tv_usec / tick;
460 if (t)
461 acct.ac_mem = (r->ru_ixrss + r->ru_idrss + r->ru_isrss) / t;
462 else
463 acct.ac_mem = 0;
464
465 /* (5) The number of disk I/O operations done */
466 acct.ac_io = encode_comp_t(r->ru_inblock + r->ru_oublock, 0);
467
468 /* (6) The UID and GID of the process */
469 acct.ac_uid = kauth_cred_getuid(l->l_cred);
470 acct.ac_gid = kauth_cred_getgid(l->l_cred);
471
472 /* (7) The terminal from which the process was started */
473 mutex_enter(proc_lock);
474 if ((p->p_lflag & PL_CONTROLT) && p->p_pgrp->pg_session->s_ttyp)
475 acct.ac_tty = p->p_pgrp->pg_session->s_ttyp->t_dev;
476 else
477 acct.ac_tty = NODEV;
478 mutex_exit(proc_lock);
479
480 /* (8) The boolean flags that tell how the process terminated, etc. */
481 acct.ac_flag = p->p_acflag;
482
483 /*
484 * Now, just write the accounting information to the file.
485 */
486 error = vn_rdwr(UIO_WRITE, acct_vp, (void *)&acct,
487 sizeof(acct), (off_t)0, UIO_SYSSPACE, IO_APPEND|IO_UNIT,
488 acct_cred, NULL, NULL);
489 if (error != 0)
490 log(LOG_ERR, "Accounting: write failed %d\n", error);
491
492 /* Restore limit - rather pointless since process is about to exit */
493 p->p_rlimit[RLIMIT_FSIZE] = orlim;
494
495 out:
496 rw_exit(&acct_lock);
497 return (error);
498}
