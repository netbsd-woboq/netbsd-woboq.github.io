129/*
130 * Fill in the appropriate signal information, and signal the parent.
131 */
132/* XXX noclone works around a gcc 4.5 bug on arm */
133static void __noclone
134exit_psignal(struct proc *p, struct proc *pp, ksiginfo_t *ksi)
135{
136
137 KSI_INIT(ksi);
138 if ((ksi->ksi_signo = P_EXITSIG(p)) == SIGCHLD) {
139 if (p->p_xsig) {
140 if (p->p_sflag & PS_COREDUMP)
141 ksi->ksi_code = CLD_DUMPED;
142 else
143 ksi->ksi_code = CLD_KILLED;
144 ksi->ksi_status = p->p_xsig;
145 } else {
146 ksi->ksi_code = CLD_EXITED;
147 ksi->ksi_status = p->p_xexit;
148 }
149 } else {
150 ksi->ksi_code = SI_USER;
151 ksi->ksi_status = p->p_xsig;
152 }
153 /*
154 * We fill those in, even for non-SIGCHLD.
155 * It's safe to access p->p_cred unlocked here.
156 */
157 ksi->ksi_pid = p->p_pid;
158 ksi->ksi_uid = kauth_cred_geteuid(p->p_cred);
159 /* XXX: is this still valid? */
160 ksi->ksi_utime = p->p_stats->p_ru.ru_utime.tv_sec;
161 ksi->ksi_stime = p->p_stats->p_ru.ru_stime.tv_sec;
162}

164/*
165 * exit --
166 * Death of process.
167 */
168int
169sys_exit(struct lwp *l, const struct sys_exit_args *uap, register_t *retval)
170{
171 /* {
172 syscallarg(int) rval;
173 } */
174 struct proc *p = l->l_proc;
175
176 /* Don't call exit1() multiple times in the same process. */
177 mutex_enter(p->p_lock);
178 if (p->p_sflag & PS_WEXIT) {
179 mutex_exit(p->p_lock);
180 lwp_exit(l);
181 }
182
183 /* exit1() will release the mutex. */
184 exit1(l, SCARG(uap, rval), 0);
185 /* NOTREACHED */
186 return (0);
187}
188
189/*
190 * Exit: deallocate address space and other resources, change proc state
191 * to zombie, and unlink proc from allproc and parent's lists. Save exit
192 * status and rusage for wait(). Check for child processes and orphan them.
193 *
194 * Must be called with p->p_lock held. Does not return.
195 */
196void
197exit1(struct lwp *l, int exitcode, int signo)
198{
199 struct proc *p, *child, *next_child, *old_parent, *new_parent;
200 struct pgrp *pgrp;
201 ksiginfo_t ksi;
202 ksiginfoq_t kq;
203 int wakeinit;
204
205 p = l->l_proc;
206
207 KASSERT(mutex_owned(p->p_lock));
208 KASSERT(p->p_vmspace != NULL);
209
210 if (__predict_false(p == initproc)) {
211 panic("init died (signal %d, exit %d)", signo, exitcode);
212 }
213
214 p->p_sflag |= PS_WEXIT;
215
216 /*
217 * Force all other LWPs to exit before we do. Only then can we
218 * begin to tear down the rest of the process state.
219 */
220 if (p->p_nlwps > 1) {
221 exit_lwps(l);
222 }
223
224 ksiginfo_queue_init(&kq);
225
226 /*
227 * If we have been asked to stop on exit, do so now.
228 */
229 if (__predict_false(p->p_sflag & PS_STOPEXIT)) {
230 KERNEL_UNLOCK_ALL(l, &l->l_biglocks);
231 sigclearall(p, &contsigmask, &kq);
232
233 if (!mutex_tryenter(proc_lock)) {
234 mutex_exit(p->p_lock);
235 mutex_enter(proc_lock);
236 mutex_enter(p->p_lock);
237 }
238 p->p_waited = 0;
239 p->p_pptr->p_nstopchild++;
240 p->p_stat = SSTOP;
241 mutex_exit(proc_lock);
242 lwp_lock(l);
243 p->p_nrlwps--;
244 l->l_stat = LSSTOP;
245 lwp_unlock(l);
246 mutex_exit(p->p_lock);
247 lwp_lock(l);
248 mi_switch(l);
249 KERNEL_LOCK(l->l_biglocks, l);
250 mutex_enter(p->p_lock);
251 }
252
253 /*
254 * Bin any remaining signals and mark the process as dying so it will
255 * not be found for, e.g. signals.
256 */
257 sigfillset(&p->p_sigctx.ps_sigignore);
258 sigclearall(p, NULL, &kq);
259 p->p_stat = SDYING;
260 mutex_exit(p->p_lock);
261 ksiginfo_queue_drain(&kq);
262
263 /* Destroy any lwpctl info. */
264 if (p->p_lwpctl != NULL)
265 lwp_ctl_exit();
266
267 /*
268 * Drain all remaining references that procfs, ptrace and others may
269 * have on the process.
270 */
271 rw_enter(&p->p_reflock, RW_WRITER);
272
273 DPRINTF(("%s: %d.%d exiting.\n", __func__, p->p_pid, l->l_lid));
274
275 timers_free(p, TIMERS_ALL);
276#if defined(__HAVE_RAS)
277 ras_purgeall();
278#endif
279
280 /*
281 * Close open files, release open-file table and free signal
282 * actions. This may block!
283 */
284 fd_free();
285 cwdfree(p->p_cwdi);
286 p->p_cwdi = NULL;
287 doexithooks(p);
288 sigactsfree(p->p_sigacts);
289
290 /*
291 * Write out accounting data.
292 */
293 (void)acct_process(l);
294
295#ifdef KTRACE
296 /*
297 * Release trace file.
298 */
299 if (p->p_tracep != NULL) {
300 mutex_enter(&ktrace_lock);
301 ktrderef(p);
302 mutex_exit(&ktrace_lock);
303 }
304#endif
305
306 p->p_xexit = exitcode;
307 p->p_xsig = signo;
308
309 /*
310 * If emulation has process exit hook, call it now.
311 * Set the exit status now so that the exit hook has
312 * an opportunity to tweak it (COMPAT_LINUX requires
313 * this for thread group emulation)
314 */
315 if (p->p_emul->e_proc_exit)
316 (*p->p_emul->e_proc_exit)(p);
317
318 /*
319 * Free the VM resources we're still holding on to.
320 * We must do this from a valid thread because doing
321 * so may block. This frees vmspace, which we don't
322 * need anymore. The only remaining lwp is the one
323 * we run at this moment, nothing runs in userland
324 * anymore.
325 */
326 ruspace(p); /* Update our vm resource use */
327 uvm_proc_exit(p);
328
329 /*
330 * Stop profiling.
331 */
332 if (__predict_false((p->p_stflag & PST_PROFIL) != 0)) {
333 mutex_spin_enter(&p->p_stmutex);
334 stopprofclock(p);
335 mutex_spin_exit(&p->p_stmutex);
336 }
337
338 /*
339 * If parent is waiting for us to exit or exec, PL_PPWAIT is set; we
340 * wake up the parent early to avoid deadlock. We can do this once
341 * the VM resources are released.
342 */
343 mutex_enter(proc_lock);
344 if (p->p_lflag & PL_PPWAIT) {
345 lwp_t *lp;
346
347 l->l_lwpctl = NULL; /* was on loan from blocked parent */
348 p->p_lflag &= ~PL_PPWAIT;
349
350 lp = p->p_vforklwp;
351 p->p_vforklwp = NULL;
352 lp->l_vforkwaiting = false;
353 cv_broadcast(&lp->l_waitcv);
354 }
355
356 if (SESS_LEADER(p)) {
357 struct vnode *vprele = NULL, *vprevoke = NULL;
358 struct session *sp = p->p_session;
359 struct tty *tp;
360
361 if (sp->s_ttyvp) {
362 /*
363 * Controlling process.
364 * Signal foreground pgrp,
365 * drain controlling terminal
366 * and revoke access to controlling terminal.
367 */
368 tp = sp->s_ttyp;
369 mutex_spin_enter(&tty_lock);
370 if (tp->t_session == sp) {
371 /* we can't guarantee the revoke will do this */
372 pgrp = tp->t_pgrp;
373 tp->t_pgrp = NULL;
374 tp->t_session = NULL;
375 mutex_spin_exit(&tty_lock);
376 if (pgrp != NULL) {
377 pgsignal(pgrp, SIGHUP, 1);
378 }
379 mutex_exit(proc_lock);
380 (void) ttywait(tp);
381 mutex_enter(proc_lock);
382
383 /* The tty could have been revoked. */
384 vprevoke = sp->s_ttyvp;
385 } else
386 mutex_spin_exit(&tty_lock);
387 vprele = sp->s_ttyvp;
388 sp->s_ttyvp = NULL;
389 /*
390 * s_ttyp is not zero'd; we use this to indicate
391 * that the session once had a controlling terminal.
392 * (for logging and informational purposes)
393 */
394 }
395 sp->s_leader = NULL;
396
397 if (vprevoke != NULL || vprele != NULL) {
398 if (vprevoke != NULL) {
399 /* Releases proc_lock. */
400 proc_sessrele(sp);
401 VOP_REVOKE(vprevoke, REVOKEALL);
402 } else
403 mutex_exit(proc_lock);
404 if (vprele != NULL)
405 vrele(vprele);
406 mutex_enter(proc_lock);
407 }
408 }
409 fixjobc(p, p->p_pgrp, 0);
410
411 /* Release fstrans private data. */
412 fstrans_lwp_dtor(l);
413
414 /*
415 * Finalize the last LWP's specificdata, as well as the
416 * specificdata for the proc itself.
417 */
418 lwp_finispecific(l);
419 proc_finispecific(p);
420
421 /*
422 * Notify interested parties of our demise.
423 */
424 KNOTE(&p->p_klist, NOTE_EXIT);
425
426 SDT_PROBE(proc, kernel, , exit,
427 ((p->p_sflag & PS_COREDUMP) ? CLD_DUMPED :
428 (p->p_xsig ? CLD_KILLED : CLD_EXITED)),
429 0,0,0,0);
430
431 /*
432 * Reset p_opptr pointer of all former children which got
433 * traced by another process and were reparented. We reset
434 * it to NULL here; the trace detach code then reparents
435 * the child to initproc. We only check allproc list, since
436 * eventual former children on zombproc list won't reference
437 * p_opptr anymore.
438 */
439 if (__predict_false(p->p_slflag & PSL_CHTRACED)) {
440 struct proc *q;
441 PROCLIST_FOREACH(q, &allproc) {
442 if (q->p_opptr == p)
443 q->p_opptr = NULL;
444 }
445 PROCLIST_FOREACH(q, &zombproc) {
446 if (q->p_opptr == p)
447 q->p_opptr = NULL;
448 }
449 }
450
451 /*
452 * Give orphaned children to init(8).
453 */
454 child = LIST_FIRST(&p->p_children);
455 wakeinit = (child != NULL);
456 for (; child != NULL; child = next_child) {
457 next_child = LIST_NEXT(child, p_sibling);
458
459 /*
460 * Traced processes are killed since their existence
461 * means someone is screwing up. Since we reset the
462 * trace flags, the logic in sys_wait4() would not be
463 * triggered to reparent the process to its
464 * original parent, so we must do this here.
465 */
466 if (__predict_false(child->p_slflag & PSL_TRACED)) {
467 mutex_enter(p->p_lock);
468 child->p_slflag &=
469 ~(PSL_TRACED|PSL_SYSCALL);
470 mutex_exit(p->p_lock);
471 if (child->p_opptr != child->p_pptr) {
472 struct proc *t = child->p_opptr;
473 proc_reparent(child, t ? t : initproc);
474 child->p_opptr = NULL;
475 } else
476 proc_reparent(child, initproc);
477 killproc(child, "orphaned traced process");
478 } else
479 proc_reparent(child, initproc);
480 }
481
482 /*
483 * Move proc from allproc to zombproc, it's now nearly ready to be
484 * collected by parent.
485 */
486 LIST_REMOVE(l, l_list);
487 LIST_REMOVE(p, p_list);
488 LIST_INSERT_HEAD(&zombproc, p, p_list);
489
490 /*
491 * Mark the process as dead. We must do this before we signal
492 * the parent.
493 */
494 p->p_stat = SDEAD;
495
496 /* Put in front of parent's sibling list for parent to collect it */
497 old_parent = p->p_pptr;
498 old_parent->p_nstopchild++;
499 if (LIST_FIRST(&old_parent->p_children) != p) {
500 /* Put child where it can be found quickly */
501 LIST_REMOVE(p, p_sibling);
502 LIST_INSERT_HEAD(&old_parent->p_children, p, p_sibling);
503 }
504
505 /*
506 * Notify parent that we're gone. If parent has the P_NOCLDWAIT
507 * flag set, notify init instead (and hope it will handle
508 * this situation).
509 */
510 if (old_parent->p_flag & (PK_NOCLDWAIT|PK_CLDSIGIGN)) {
511 proc_reparent(p, initproc);
512 wakeinit = 1;
513
514 /*
515 * If this was the last child of our parent, notify
516 * parent, so in case he was wait(2)ing, he will
517 * continue.
518 */
519 if (LIST_FIRST(&old_parent->p_children) == NULL)
520 cv_broadcast(&old_parent->p_waitcv);
521 }
522
523 /* Reload parent pointer, since p may have been reparented above */
524 new_parent = p->p_pptr;
525
526 if (__predict_false(p->p_exitsig != 0)) {
527 exit_psignal(p, new_parent, &ksi);
528 kpsignal(new_parent, &ksi, NULL);
529 }
530
531 /* Calculate the final rusage info. */
532 calcru(p, &p->p_stats->p_ru.ru_utime, &p->p_stats->p_ru.ru_stime,
533 NULL, NULL);
534
535 if (wakeinit)
536 cv_broadcast(&initproc->p_waitcv);
537
538 callout_destroy(&l->l_timeout_ch);
539
540 /*
541 * Release any PCU resources before becoming a zombie.
542 */
543 pcu_discard_all(l);
544
545 mutex_enter(p->p_lock);
546 /* Free the linux lwp id */
547 if ((l->l_pflag & LP_PIDLID) != 0 && l->l_lid != p->p_pid)
548 proc_free_pid(l->l_lid);
549 lwp_drainrefs(l);
550 lwp_lock(l);
551 l->l_prflag &= ~LPR_DETACHED;
552 l->l_stat = LSZOMB;
553 lwp_unlock(l);
554 KASSERT(curlwp == l);
555 KASSERT(p->p_nrlwps == 1);
556 KASSERT(p->p_nlwps == 1);
557 p->p_stat = SZOMB;
558 p->p_nrlwps--;
559 p->p_nzlwps++;
560 p->p_ndlwps = 0;
561 mutex_exit(p->p_lock);
562
563 /*
564 * Signal the parent to collect us, and drop the proclist lock.
565 * Drop debugger/procfs lock; no new references can be gained.
566 */
567 cv_broadcast(&p->p_pptr->p_waitcv);
568 rw_exit(&p->p_reflock);
569 mutex_exit(proc_lock);
570
571 /* Verify that we hold no locks other than the kernel lock. */
572 LOCKDEBUG_BARRIER(&kernel_lock, 0);
573
574 /*
575 * NOTE: WE ARE NO LONGER ALLOWED TO SLEEP!
576 */
577
578 /*
579 * Give machine-dependent code a chance to free any MD LWP
580 * resources. This must be done before uvm_lwp_exit(), in
581 * case these resources are in the PCB.
582 */
583 cpu_lwp_free(l, 1);
584
585 pmap_deactivate(l);
586
587 /* This process no longer needs to hold the kernel lock. */
588#ifdef notyet
589 /* XXXSMP hold in lwp_userret() */
590 KERNEL_UNLOCK_LAST(l);
591#else
592 KERNEL_UNLOCK_ALL(l, NULL);
593#endif
594
595 lwp_exit_switchaway(l);
596}

598void
599exit_lwps(struct lwp *l)
600{
601 proc_t *p = l->l_proc;
602 lwp_t *l2;
603 int nlocks;
604
605 KERNEL_UNLOCK_ALL(l, &nlocks);
606retry:
607 KASSERT(mutex_owned(p->p_lock));
608
609 /*
610 * Interrupt LWPs in interruptable sleep, unsuspend suspended
611 * LWPs and then wait for everyone else to finish.
612 */
613 LIST_FOREACH(l2, &p->p_lwps, l_sibling) {
614 if (l2 == l)
615 continue;
616 lwp_lock(l2);
617 l2->l_flag |= LW_WEXIT;
618 if ((l2->l_stat == LSSLEEP && (l2->l_flag & LW_SINTR)) ||
619 l2->l_stat == LSSUSPENDED || l2->l_stat == LSSTOP) {
620 /* setrunnable() will release the lock. */
621 setrunnable(l2);
622 continue;
623 }
624 lwp_unlock(l2);
625 }
626
627 /*
628 * Wait for every LWP to exit. Note: LWPs can get suspended/slept
629 * behind us or there may even be new LWPs created. Therefore, a
630 * full retry is required on error.
631 */
632 while (p->p_nlwps > 1) {
633 if (lwp_wait(l, 0, NULL, true)) {
634 goto retry;
635 }
636 }
637
638 KERNEL_LOCK(nlocks, l);
639 KASSERT(p->p_nlwps == 1);
640}

642int
643do_sys_waitid(idtype_t idtype, id_t id, int *pid, int *status, int options,
644 struct wrusage *wru, siginfo_t *si)
645{
646 proc_t *child;
647 int error;
648
649
650 if (wru != NULL)
651 memset(wru, 0, sizeof(*wru));
652 if (si != NULL)
653 memset(si, 0, sizeof(*si));
654
655 mutex_enter(proc_lock);
656 error = find_stopped_child(curproc, idtype, id, options, &child,
657 wru, si);
658 if (child == NULL) {
659 mutex_exit(proc_lock);
660 *pid = 0;
661 *status = 0;
662 return error;
663 }
664 *pid = child->p_pid;
665
666 if (child->p_stat == SZOMB) {
667 /* Child is exiting */
668 *status = P_WAITSTATUS(child);
669 /* proc_free() will release the proc_lock. */
670 if (options & WNOWAIT) {
671 mutex_exit(proc_lock);
672 } else {
673 proc_free(child, wru);
674 }
675 } else {
676 /* Don't mark SIGCONT if we are being stopped */
677 *status = (child->p_xsig == SIGCONT && child->p_stat != SSTOP) ?
678 W_CONTCODE() : W_STOPCODE(child->p_xsig);
679 mutex_exit(proc_lock);
680 }
681 return 0;
682}

684int
685do_sys_wait(int *pid, int *status, int options, struct rusage *ru)
686{
687 idtype_t idtype;
688 id_t id;
689 int ret;
690 struct wrusage wru;
691
692 /*
693 * Translate the special pid values into the (idtype, pid)
694 * pair for wait6. The WAIT_MYPGRP case is handled by
695 * find_stopped_child() on its own.
696 */
697 if (*pid == WAIT_ANY) {
698 idtype = P_ALL;
699 id = 0;
700 } else if (*pid < 0) {
701 idtype = P_PGID;
702 id = (id_t)-*pid;
703 } else {
704 idtype = P_PID;
705 id = (id_t)*pid;
706 }
707 options |= WEXITED | WTRAPPED;
708 ret = do_sys_waitid(idtype, id, pid, status, options, ru ? &wru : NULL,
709 NULL);
710 if (ru)
711 *ru = wru.wru_self;
712 return ret;
713}

804/*
805 * Find a process that matches the provided criteria, and fill siginfo
806 * and resources if found.
807 * Returns:
808 * -1: Not found, abort early
809 * 0: Not matched
810 * 1: Matched, there might be more matches
811 * 2: This is the only match
812 */
813static int
814match_process(const struct proc *pp, struct proc **q, idtype_t idtype, id_t id,
815 int options, struct wrusage *wrusage, siginfo_t *siginfo)
816{
817 struct rusage *rup;
818 struct proc *p = *q;
819 int rv = 1;
820
821 mutex_enter(p->p_lock);
822 switch (idtype) {
823 case P_ALL:
824 break;
825 case P_PID:
826 if (p->p_pid != (pid_t)id) {
827 mutex_exit(p->p_lock);
828 p = *q = proc_find_raw((pid_t)id);
829 if (p == NULL || p->p_stat == SIDL || p->p_pptr != pp) {
830 *q = NULL;
831 return -1;
832 }
833 mutex_enter(p->p_lock);
834 }
835 rv++;
836 break;
837 case P_PGID:
838 if (p->p_pgid != (pid_t)id)
839 goto out;
840 break;
841 case P_SID:
842 if (p->p_session->s_sid != (pid_t)id)
843 goto out;
844 break;
845 case P_UID:
846 if (kauth_cred_geteuid(p->p_cred) != (uid_t)id)
847 goto out;
848 break;
849 case P_GID:
850 if (kauth_cred_getegid(p->p_cred) != (gid_t)id)
851 goto out;
852 break;
853 case P_CID:
854 case P_PSETID:
855 case P_CPUID:
856 /* XXX: Implement me */
857 default:
858 out:
859 mutex_exit(p->p_lock);
860 return 0;
861 }
862
863 if ((options & WEXITED) == 0 && p->p_stat == SZOMB)
864 goto out;
865
866 if (siginfo != NULL) {
867 siginfo->si_errno = 0;
868
869 /*
870 * SUSv4 requires that the si_signo value is always
871 * SIGCHLD. Obey it despite the rfork(2) interface
872 * allows to request other signal for child exit
873 * notification.
874 */
875 siginfo->si_signo = SIGCHLD;
876
877 /*
878 * This is still a rough estimate. We will fix the
879 * cases TRAPPED, STOPPED, and CONTINUED later.
880 */
881 if (p->p_sflag & PS_COREDUMP) {
882 siginfo->si_code = CLD_DUMPED;
883 siginfo->si_status = p->p_xsig;
884 } else if (p->p_xsig) {
885 siginfo->si_code = CLD_KILLED;
886 siginfo->si_status = p->p_xsig;
887 } else {
888 siginfo->si_code = CLD_EXITED;
889 siginfo->si_status = p->p_xexit;
890 }
891
892 siginfo->si_pid = p->p_pid;
893 siginfo->si_uid = kauth_cred_geteuid(p->p_cred);
894 siginfo->si_utime = p->p_stats->p_ru.ru_utime.tv_sec;
895 siginfo->si_stime = p->p_stats->p_ru.ru_stime.tv_sec;
896 }
897
898 /*
899 * There should be no reason to limit resources usage info to
900 * exited processes only. A snapshot about any resources used
901 * by a stopped process may be exactly what is needed.
902 */
903 if (wrusage != NULL) {
904 rup = &wrusage->wru_self;
905 *rup = p->p_stats->p_ru;
906 calcru(p, &rup->ru_utime, &rup->ru_stime, NULL, NULL);
907
908 rup = &wrusage->wru_children;
909 *rup = p->p_stats->p_cru;
910 calcru(p, &rup->ru_utime, &rup->ru_stime, NULL, NULL);
911 }
912
913 mutex_exit(p->p_lock);
914 return rv;
915}

917/*
918 * Determine if there are existing processes being debugged
919 * that used to be (and sometime later will be again) children
920 * of a specific parent (while matching wait criteria)
921 */
922static bool
923debugged_child_exists(idtype_t idtype, id_t id, int options, siginfo_t *si,
924 const struct proc *parent)
925{
926 struct proc *pp;
927
928 /*
929 * If we are searching for a specific pid, we can optimise a little
930 */
931 if (idtype == P_PID) {
932 /*
933 * Check the specific process to see if its real parent is us
934 */
935 pp = proc_find_raw((pid_t)id);
936 if (pp != NULL && pp->p_stat != SIDL && pp->p_opptr == parent) {
937 /*
938 * using P_ALL here avoids match_process() doing the
939 * same work that we just did, but incorrectly for
940 * this scenario.
941 */
942 if (match_process(parent, &pp, P_ALL, id, options,
943 NULL, si))
944 return true;
945 }
946 return false;
947 }
948
949 /*
950 * For the hard cases, just look everywhere to see if some
951 * stolen (reparented) process is really our lost child.
952 * Then check if that process could satisfy the wait conditions.
953 */
954
955 /*
956 * XXX inefficient, but hopefully fairly rare.
957 * XXX should really use a list of reparented processes.
958 */
959 PROCLIST_FOREACH(pp, &allproc) {
960 if (pp->p_stat == SIDL) /* XXX impossible ?? */
961 continue;
962 if (pp->p_opptr == parent &&
963 match_process(parent, &pp, idtype, id, options, NULL, si))
964 return true;
965 }
966 PROCLIST_FOREACH(pp, &zombproc) {
967 if (pp->p_stat == SIDL) /* XXX impossible ?? */
968 continue;
969 if (pp->p_opptr == parent &&
970 match_process(parent, &pp, idtype, id, options, NULL, si))
971 return true;
972 }
973
974 return false;
975}

977/*
978 * Scan list of child processes for a child process that has stopped or
979 * exited. Used by sys_wait4 and 'compat' equivalents.
980 *
981 * Must be called with the proc_lock held, and may release while waiting.
982 */
983static int
984find_stopped_child(struct proc *parent, idtype_t idtype, id_t id, int options,
985 struct proc **child_p, struct wrusage *wru, siginfo_t *si)
986{
987 struct proc *child, *dead;
988 int error;
989
990 KASSERT(mutex_owned(proc_lock));
991
992 if (options & ~WALLOPTS) {
993 *child_p = NULL;
994 return EINVAL;
995 }
996
997 if ((options & WSELECTOPTS) == 0) {
998 /*
999 * We will be unable to find any matching processes,
1000 * because there are no known events to look for.
1001 * Prefer to return error instead of blocking
1002 * indefinitely.
1003 */
1004 *child_p = NULL;
1005 return EINVAL;
1006 }
1007
1008 if ((pid_t)id == WAIT_MYPGRP && (idtype == P_PID || idtype == P_PGID)) {
1009 mutex_enter(parent->p_lock);
1010 id = (id_t)parent->p_pgid;
1011 mutex_exit(parent->p_lock);
1012 idtype = P_PGID;
1013 }
1014
1015 for (;;) {
1016 error = ECHILD;
1017 dead = NULL;
1018
1019 LIST_FOREACH(child, &parent->p_children, p_sibling) {
1020 int rv = match_process(parent, &child, idtype, id,
1021 options, wru, si);
1022 if (rv == -1)
1023 break;
1024 if (rv == 0)
1025 continue;
1026
1027 /*
1028 * Wait for processes with p_exitsig != SIGCHLD
1029 * processes only if WALTSIG is set; wait for
1030 * processes with p_exitsig == SIGCHLD only
1031 * if WALTSIG is clear.
1032 */
1033 if (((options & WALLSIG) == 0) &&
1034 (options & WALTSIG ? child->p_exitsig == SIGCHLD
1035 : P_EXITSIG(child) != SIGCHLD)){
1036 if (rv == 2) {
1037 child = NULL;
1038 break;
1039 }
1040 continue;
1041 }
1042
1043 error = 0;
1044 if ((options & WNOZOMBIE) == 0) {
1045 if (child->p_stat == SZOMB)
1046 break;
1047 if (child->p_stat == SDEAD) {
1048 /*
1049 * We may occasionally arrive here
1050 * after receiving a signal, but
1051 * immediately before the child
1052 * process is zombified. The wait
1053 * will be short, so avoid returning
1054 * to userspace.
1055 */
1056 dead = child;
1057 }
1058 }
1059
1060 if ((options & WCONTINUED) != 0 &&
1061 child->p_xsig == SIGCONT &&
1062 (child->p_sflag & PS_CONTINUED)) {
1063 if ((options & WNOWAIT) == 0) {
1064 child->p_sflag &= ~PS_CONTINUED;
1065 child->p_waited = 1;
1066 parent->p_nstopchild--;
1067 }
1068 if (si) {
1069 si->si_status = child->p_xsig;
1070 si->si_code = CLD_CONTINUED;
1071 }
1072 break;
1073 }
1074
1075 if ((options & (WTRAPPED|WSTOPPED)) != 0 &&
1076 child->p_stat == SSTOP &&
1077 child->p_waited == 0 &&
1078 ((child->p_slflag & PSL_TRACED) ||
1079 options & (WUNTRACED|WSTOPPED))) {
1080 if ((options & WNOWAIT) == 0) {
1081 child->p_waited = 1;
1082 parent->p_nstopchild--;
1083 }
1084 if (si) {
1085 si->si_status = child->p_xsig;
1086 si->si_code =
1087 (child->p_slflag & PSL_TRACED) ?
1088 CLD_TRAPPED : CLD_STOPPED;
1089 }
1090 break;
1091 }
1092 if (parent->p_nstopchild == 0 || rv == 2) {
1093 child = NULL;
1094 break;
1095 }
1096 }
1097
1098 /*
1099 * If we found nothing, but we are the bereaved parent
1100 * of a stolen child, look and see if that child (or
1101 * one of them) meets our search criteria. If so, then
1102 * we cannot succeed, but we can hang (wait...),
1103 * or if WNOHANG, return 0 instead of ECHILD
1104 */
1105 if (child == NULL && error == ECHILD &&
1106 (parent->p_slflag & PSL_CHTRACED) &&
1107 debugged_child_exists(idtype, id, options, si, parent))
1108 error = 0;
1109
1110 if (child != NULL || error != 0 ||
1111 ((options & WNOHANG) != 0 && dead == NULL)) {
1112 *child_p = child;
1113 return error;
1114 }
1115
1116 /*
1117 * Wait for another child process to stop.
1118 */
1119 error = cv_wait_sig(&parent->p_waitcv, proc_lock);
1120
1121 if (error != 0) {
1122 *child_p = NULL;
1123 return error;
1124 }
1125 }
1126}

1128/*
1129 * Free a process after parent has taken all the state info. Must be called
1130 * with the proclist lock held, and will release before returning.
1131 *
1132 * *ru is returned to the caller, and must be freed by the caller.
1133 */
1134static void
1135proc_free(struct proc *p, struct wrusage *wru)
1136{
1137 struct proc *parent = p->p_pptr;
1138 struct lwp *l;
1139 ksiginfo_t ksi;
1140 kauth_cred_t cred1, cred2;
1141 uid_t uid;
1142
1143 KASSERT(mutex_owned(proc_lock));
1144 KASSERT(p->p_nlwps == 1);
1145 KASSERT(p->p_nzlwps == 1);
1146 KASSERT(p->p_nrlwps == 0);
1147 KASSERT(p->p_stat == SZOMB);
1148
1149 /*
1150 * If we got the child via ptrace(2) or procfs, and
1151 * the parent is different (meaning the process was
1152 * attached, rather than run as a child), then we need
1153 * to give it back to the old parent, and send the
1154 * parent the exit signal. The rest of the cleanup
1155 * will be done when the old parent waits on the child.
1156 */
1157 if ((p->p_slflag & PSL_TRACED) != 0 && p->p_opptr != parent) {
1158 mutex_enter(p->p_lock);
1159 p->p_slflag &= ~(PSL_TRACED|PSL_SYSCALL);
1160 mutex_exit(p->p_lock);
1161 parent = (p->p_opptr == NULL) ? initproc : p->p_opptr;
1162 proc_reparent(p, parent);
1163 p->p_opptr = NULL;
1164 if (p->p_exitsig != 0) {
1165 exit_psignal(p, parent, &ksi);
1166 kpsignal(parent, &ksi, NULL);
1167 }
1168 cv_broadcast(&parent->p_waitcv);
1169 mutex_exit(proc_lock);
1170 return;
1171 }
1172
1173 sched_proc_exit(parent, p);
1174
1175 /*
1176 * Add child times of exiting process onto its own times.
1177 * This cannot be done any earlier else it might get done twice.
1178 */
1179 l = LIST_FIRST(&p->p_lwps);
1180 p->p_stats->p_ru.ru_nvcsw += (l->l_ncsw - l->l_nivcsw);
1181 p->p_stats->p_ru.ru_nivcsw += l->l_nivcsw;
1182 ruadd(&p->p_stats->p_ru, &l->l_ru);
1183 ruadd(&p->p_stats->p_ru, &p->p_stats->p_cru);
1184 ruadd(&parent->p_stats->p_cru, &p->p_stats->p_ru);
1185 if (wru != NULL) {
1186 wru->wru_self = p->p_stats->p_ru;
1187 wru->wru_children = p->p_stats->p_cru;
1188 }
1189 p->p_xsig = 0;
1190 p->p_xexit = 0;
1191
1192 /*
1193 * At this point we are going to start freeing the final resources.
1194 * If anyone tries to access the proc structure after here they will
1195 * get a shock - bits are missing. Attempt to make it hard! We
1196 * don't bother with any further locking past this point.
1197 */
1198 p->p_stat = SIDL; /* not even a zombie any more */
1199 LIST_REMOVE(p, p_list); /* off zombproc */
1200 parent->p_nstopchild--;
1201 LIST_REMOVE(p, p_sibling);
1202
1203 /*
1204 * Let pid be reallocated.
1205 */
1206 proc_free_pid(p->p_pid);
1207
1208 /*
1209 * Unlink process from its process group.
1210 * Releases the proc_lock.
1211 */
1212 proc_leavepgrp(p);
1213
1214 /*
1215 * Delay release until after lwp_free.
1216 */
1217 cred2 = l->l_cred;
1218
1219 /*
1220 * Free the last LWP's resources.
1221 *
1222 * lwp_free ensures the LWP is no longer running on another CPU.
1223 */
1224 lwp_free(l, false, true);
1225
1226 /*
1227 * Now no one except us can reach the process p.
1228 */
1229
1230 /*
1231 * Decrement the count of procs running with this uid.
1232 */
1233 cred1 = p->p_cred;
1234 uid = kauth_cred_getuid(cred1);
1235 (void)chgproccnt(uid, -1);
1236
1237 /*
1238 * Release substructures.
1239 */
1240
1241 lim_free(p->p_limit);
1242 pstatsfree(p->p_stats);
1243 kauth_cred_free(cred1);
1244 kauth_cred_free(cred2);
1245
1246 /*
1247 * Release reference to text vnode
1248 */
1249 if (p->p_textvp)
1250 vrele(p->p_textvp);
1251 kmem_strfree(p->p_path);
1252
1253 mutex_destroy(&p->p_auxlock);
1254 mutex_obj_free(p->p_lock);
1255 mutex_destroy(&p->p_stmutex);
1256 cv_destroy(&p->p_waitcv);
1257 cv_destroy(&p->p_lwpcv);
1258 rw_destroy(&p->p_reflock);
1259
1260 proc_free_mem(p);
1261}

1263/*
1264 * Change the parent of a process for tracing purposes.
1265 */
1266void
1267proc_changeparent(struct proc *t, struct proc *p)
1268{
1269 SET(t->p_slflag, PSL_TRACED);
1270 t->p_opptr = t->p_pptr;
1271 if (t->p_pptr == p)
1272 return;
1273 struct proc *parent = t->p_pptr;
1274
1275 if (parent->p_lock < t->p_lock) {
1276 if (!mutex_tryenter(parent->p_lock)) {
1277 mutex_exit(t->p_lock);
1278 mutex_enter(parent->p_lock);
1279 mutex_enter(t->p_lock);
1280 }
1281 } else if (parent->p_lock > t->p_lock) {
1282 mutex_enter(parent->p_lock);
1283 }
1284 parent->p_slflag |= PSL_CHTRACED;
1285 proc_reparent(t, p);
1286 if (parent->p_lock != t->p_lock)
1287 mutex_exit(parent->p_lock);
1288}

1290/*
1291 * make process 'parent' the new parent of process 'child'.
1292 *
1293 * Must be called with proc_lock held.
1294 */
1295void
1296proc_reparent(struct proc *child, struct proc *parent)
1297{
1298
1299 KASSERT(mutex_owned(proc_lock));
1300
1301 if (child->p_pptr == parent)
1302 return;
1303
1304 if (child->p_stat == SZOMB || child->p_stat == SDEAD ||
1305 (child->p_stat == SSTOP && !child->p_waited)) {
1306 child->p_pptr->p_nstopchild--;
1307 parent->p_nstopchild++;
1308 }
1309 if (parent == initproc) {
1310 child->p_exitsig = SIGCHLD;
1311 child->p_ppid = parent->p_pid;
1312 }
1313
1314 LIST_REMOVE(child, p_sibling);
1315 LIST_INSERT_HEAD(&parent->p_children, child, p_sibling);
1316 child->p_pptr = parent;
1317}
