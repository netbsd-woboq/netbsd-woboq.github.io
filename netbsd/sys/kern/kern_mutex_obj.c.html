32#include <sys/cdefs.h>
33__KERNEL_RCSID(0, "$NetBSD: kern_mutex_obj.c,v 1.6 2018/02/05 04:25:04 ozaki-r Exp $");
34
35#include <sys/param.h>
36#include <sys/atomic.h>
37#include <sys/mutex.h>
38#include <sys/pool.h>
39
40/* Mutex cache */
41#define MUTEX_OBJ_MAGIC 0x5aa3c85d
42struct kmutexobj {
43 kmutex_t mo_lock;
44 u_int mo_magic;
45 u_int mo_refcnt;
46};
47
48static int mutex_obj_ctor(void *, void *, int);
49
50static pool_cache_t mutex_obj_cache __read_mostly;
51
52/*
53 * mutex_obj_init:
54 *
55 * Initialize the mutex object store.
56 */
57void
58mutex_obj_init(void)
59{
60
61 mutex_obj_cache = pool_cache_init(sizeof(struct kmutexobj),
62 coherency_unit, 0, 0, "mutex", NULL, IPL_NONE, mutex_obj_ctor,
63 NULL, NULL);
64}
65
66/*
67 * mutex_obj_ctor:
68 *
69 * Initialize a new lock for the cache.
70 */
71static int
72mutex_obj_ctor(void *arg, void *obj, int flags)
73{
74 struct kmutexobj * mo = obj;
75
76 mo->mo_magic = MUTEX_OBJ_MAGIC;
77
78 return 0;
79}
80
81/*
82 * mutex_obj_alloc:
83 *
84 * Allocate a single lock object.
85 */
86kmutex_t *
87mutex_obj_alloc(kmutex_type_t type, int ipl)
88{
89 struct kmutexobj *mo;
90 extern void _mutex_init(kmutex_t *, kmutex_type_t, int, uintptr_t);
91
92 mo = pool_cache_get(mutex_obj_cache, PR_WAITOK);
93 _mutex_init(&mo->mo_lock, type, ipl,
94 (uintptr_t)__builtin_return_address(0));
95 mo->mo_refcnt = 1;
96
97 return (kmutex_t *)mo;
98}
99
100/*
101 * mutex_obj_hold:
102 *
103 * Add a single reference to a lock object. A reference to the object
104 * must already be held, and must be held across this call.
105 */
106void
107mutex_obj_hold(kmutex_t *lock)
108{
109 struct kmutexobj *mo = (struct kmutexobj *)lock;
110
111 KASSERTMSG(mo->mo_magic == MUTEX_OBJ_MAGIC,
112 "%s: lock %p: mo->mo_magic (%#x) != MUTEX_OBJ_MAGIC (%#x)",
113 __func__, mo, mo->mo_magic, MUTEX_OBJ_MAGIC);
114 KASSERTMSG(mo->mo_refcnt > 0,
115 "%s: lock %p: mo->mo_refcnt (%#x) == 0",
116 __func__, mo, mo->mo_refcnt);
117
118 atomic_inc_uint(&mo->mo_refcnt);
119}
120
121/*
122 * mutex_obj_free:
123 *
124 * Drop a reference from a lock object. If the last reference is being
125 * dropped, free the object and return true. Otherwise, return false.
126 */
127bool
128mutex_obj_free(kmutex_t *lock)
129{
130 struct kmutexobj *mo = (struct kmutexobj *)lock;
131
132 KASSERTMSG(mo->mo_magic == MUTEX_OBJ_MAGIC,
133 "%s: lock %p: mo->mo_magic (%#x) != MUTEX_OBJ_MAGIC (%#x)",
134 __func__, mo, mo->mo_magic, MUTEX_OBJ_MAGIC);
135 KASSERTMSG(mo->mo_refcnt > 0,
136 "%s: lock %p: mo->mo_refcnt (%#x) == 0",
137 __func__, mo, mo->mo_refcnt);
138
139 if (atomic_dec_uint_nv(&mo->mo_refcnt) > 0) {
140 return false;
141 }
142 mutex_destroy(&mo->mo_lock);
143 pool_cache_put(mutex_obj_cache, mo);
144 return true;
145}
