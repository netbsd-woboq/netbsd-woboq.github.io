38#include <sys/param.h>
39#include <sys/systm.h>
40#include <sys/proc.h>
41#include <sys/reboot.h>
42#include <sys/syscall.h>
43#include <sys/syscallargs.h>
44#include <sys/kauth.h>
45
46/* ARGSUSED */
47int
48sys_reboot(struct lwp *l, const struct sys_reboot_args *uap, register_t *retval)
49{
50 /* {
51 syscallarg(int) opt;
52 syscallarg(char *) bootstr;
53 } */
54 int error;
55 char *bootstr, bs[128];
56
57 if ((error = kauth_authorize_system(l->l_cred, KAUTH_SYSTEM_REBOOT,
58 0, NULL, NULL, NULL)) != 0)
59 return (error);
60
61 /*
62 * Only use the boot string if RB_STRING is set.
63 */
64 if ((SCARG(uap, opt) & RB_STRING) &&
65 (error = copyinstr(SCARG(uap, bootstr), bs, sizeof(bs), 0)) == 0)
66 bootstr = bs;
67 else
68 bootstr = NULL;
69 /*
70 * Not all ports use the bootstr currently.
71 */
72 KERNEL_LOCK(1, NULL);
73 cpu_reboot(SCARG(uap, opt), bootstr);
74 KERNEL_UNLOCK_ONE(NULL);
75 return (0);
76}
