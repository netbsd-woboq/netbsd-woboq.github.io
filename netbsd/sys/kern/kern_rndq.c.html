93/*
94 * This is a little bit of state information attached to each device that we
95 * collect entropy from. This is simply a collection buffer, and when it
96 * is full it will be "detached" from the source and added to the entropy
97 * pool after entropy is distilled as much as possible.

191/*
192 * Generate a 32-bit counter.
193 */
194static inline uint32_t
195rnd_counter(void)
196{
197 struct bintime bt;
198 uint32_t ret;
199
200#if defined(__HAVE_CPU_COUNTER)
201 if (cpu_hascounter())
202 return cpu_counter32();
203#endif
204 if (!rnd_ready)
205 /* Too early to call nanotime. */
206 return 0;
207
208 binuptime(&bt);
209 ret = bt.sec;
210 ret ^= bt.sec >> 32;
211 ret ^= bt.frac;
212 ret ^= bt.frac >> 32;
213
214 return ret;
215}

252/*
253 * Tell any sources with "feed me" callbacks that we are hungry.
254 */
255void
256rnd_getmore(size_t byteswanted)
257{
258 krndsource_t *rs, *next;
259
260 /*
261 * Due to buffering in rnd_process_events, even if the entropy
262 * sources provide the requested number of bytes, users may not
263 * be woken because the data may be stuck in unfilled buffers.
264 * So ask for enough data to fill all the buffers.
265 *
266 * XXX Just get rid of this buffering and solve the
267 * /dev/random-as-side-channel-for-keystroke-timings a
268 * different way.
269 */
270 byteswanted = MAX(byteswanted,
271 MAX(RND_POOLBITS/NBBY, sizeof(uint32_t)*RND_SAMPLE_COUNT));
272
273 mutex_spin_enter(&rnd_global.lock);
274 LIST_FOREACH_SAFE(rs, &rnd_global.sources, list, next) {
275 /* Skip if the source is disabled. */
276 if (!RND_ENABLED(rs))
277 continue;
278
279 /* Skip if there's no callback. */
280 if (!ISSET(rs->flags, RND_FLAG_HASCB))
281 continue;
282 KASSERT(rs->get != NULL);
283
284 /* Skip if there are too many users right now. */
285 if (rs->refcnt == UINT_MAX)
286 continue;
287
288 /*
289 * Hold a reference while we release rnd_global.lock to
290 * call the callback. The callback may in turn call
291 * rnd_add_data, which acquires rnd_global.lock.
292 */
293 rs->refcnt++;
294 mutex_spin_exit(&rnd_global.lock);
295 rs->get(byteswanted, rs->getarg);
296 mutex_spin_enter(&rnd_global.lock);
297 if (--rs->refcnt == 0)
298 cv_broadcast(&rnd_global.cv);
299
300 /* Dribble some goo to the console. */
301 rnd_printf_verbose("rnd: entropy estimate %zu bits\n",
302 rndpool_get_entropy_count(&rnd_global.pool));
303 rnd_printf_verbose("rnd: asking source %s for %zu bytes\n",
304 rs->name, byteswanted);
305 }
306 mutex_spin_exit(&rnd_global.lock);
307
308 /*
309 * Check whether we got entropy samples to process. In that
310 * case, we may need to distribute entropy to waiters. Do
311 * that, if we can do it asynchronously.
312 *
313 * - Conditionally because we don't want a softint loop.
314 * - Asynchronously because if we did it synchronously, we may
315 * end up with lock recursion on rndsinks_lock.
316 */
317 if (!SIMPLEQ_EMPTY(&rnd_samples.q) && rnd_process != NULL)
318 rnd_schedule_process();
319}
320
321/*
322 * Use the timing/value of the event to estimate the entropy gathered.
323 * If all the differentials (first, second, and third) are non-zero, return
324 * non-zero. If any of these are zero, return zero.
325 */
326static inline uint32_t
327rnd_delta_estimate(rnd_delta_t *d, uint32_t v, int32_t delta)
328{
329 int32_t delta2, delta3;
330
331 d->insamples++;
332
333 /*
334 * Calculate the second and third order differentials
335 */
336 delta2 = d->dx - delta;
337 if (delta2 < 0)
338 delta2 = -delta2;
339
340 delta3 = d->d2x - delta2;
341 if (delta3 < 0)
342 delta3 = -delta3;
343
344 d->x = v;
345 d->dx = delta;
346 d->d2x = delta2;
347
348 /*
349 * If any delta is 0, we got no entropy. If all are non-zero, we
350 * might have something.
351 */
352 if (delta == 0 || delta2 == 0 || delta3 == 0)
353 return 0;
354
355 d->outbits++;
356 return 1;
357}

359/*
360 * Delta estimator for 32-bit timeestamps. Must handle wrap.
361 */
362static inline uint32_t
363rnd_dt_estimate(krndsource_t *rs, uint32_t t)
364{
365 int32_t delta;
366 uint32_t ret;
367 rnd_delta_t *d = &rs->time_delta;
368
369 if (t < d->x) {
370 delta = UINT32_MAX - d->x + t;
371 } else {
372 delta = d->x - t;
373 }
374
375 if (delta < 0) {
376 delta = -delta;
377 }
378
379 ret = rnd_delta_estimate(d, t, delta);
380
381 KASSERT(d->x == t);
382 KASSERT(d->dx == delta);
383#ifdef RND_VERBOSE
384 if (deltacnt++ % 1151 == 0) {
385 rnd_printf_verbose("rnd_dt_estimate: %s x = %lld, dx = %lld, "
386 "d2x = %lld\n", rs->name,
387 (int)d->x, (int)d->dx, (int)d->d2x);
388 }
389#endif
390 return ret;
391}
392
393/*
394 * Delta estimator for 32 or bit values. "Wrap" isn't.
395 */
396static inline uint32_t
397rnd_dv_estimate(krndsource_t *rs, uint32_t v)
398{
399 int32_t delta;
400 uint32_t ret;
401 rnd_delta_t *d = &rs->value_delta;
402
403 delta = d->x - v;
404
405 if (delta < 0) {
406 delta = -delta;
407 }
408 ret = rnd_delta_estimate(d, v, (uint32_t)delta);
409
410 KASSERT(d->x == v);
411 KASSERT(d->dx == delta);
412#ifdef RND_VERBOSE
413 if (deltacnt++ % 1151 == 0) {
414 rnd_printf_verbose("rnd_dv_estimate: %s x = %lld, dx = %lld, "
415 " d2x = %lld\n", rs->name,
416 (long long int)d->x,
417 (long long int)d->dx,
418 (long long int)d->d2x);
419 }
420#endif
421 return ret;
422}

424#if defined(__HAVE_CPU_RNG) && !defined(_RUMPKERNEL)
425static struct {
426 kmutex_t lock; /* unfortunately, must protect krndsource */
427 krndsource_t source;
428} rnd_cpu __cacheline_aligned;

430static void
431rnd_cpu_get(size_t bytes, void *priv)
432{
433 krndsource_t *cpusrcp = priv;
434 cpu_rng_t buf[2 * RND_ENTROPY_THRESHOLD / sizeof(cpu_rng_t)];
435 cpu_rng_t *bufp;
436 size_t cnt = __arraycount(buf);
437 size_t entropy = 0;
438
439 KASSERT(cpusrcp == &rnd_cpu.source);
440
441 for (bufp = buf; bufp < buf + cnt; bufp++) {
442 entropy += cpu_rng(bufp);
443 }
444 if (__predict_true(entropy)) {
445 mutex_spin_enter(&rnd_cpu.lock);
446 rnd_add_data_sync(cpusrcp, buf, sizeof(buf), entropy);
447 explicit_memset(buf, 0, sizeof(buf));
448 mutex_spin_exit(&rnd_cpu.lock);
449 }
450}

475static void
476rnd_skew_get(size_t bytes, void *priv)
477{
478 krndsource_t *skewsrcp __diagused = priv;
479
480 KASSERT(skewsrcp == &rnd_skew.source);
481
482 /* Measure 100 times */
483 rnd_skew.iter = 100;
484 callout_schedule(&rnd_skew.callout, 1);
485}

487static void
488rnd_skew_intr(void *arg)
489{
490 /*
491 * Even on systems with seemingly stable clocks, the
492 * delta-time entropy estimator seems to think we get 1 bit here
493 * about every 2 calls.
494 *
495 */
496 mutex_spin_enter(&rnd_skew.lock);
497
498 if (RND_ENABLED(&rnd_skew.source)) {
499 int next_ticks = 1;
500 if (rnd_skew.iter & 1) {
501 rnd_add_uint32(&rnd_skew.source, rnd_counter());
502 next_ticks = hz / 10;
503 }
504 if (--rnd_skew.iter > 0) {
505 callout_schedule(&rnd_skew.callout, next_ticks);
506 }
507 }
508 mutex_spin_exit(&rnd_skew.lock);
509}
510#endif

523/*
524 * Entropy was just added to the pool. If we crossed the threshold for
525 * the first time, set rnd_initial_entropy = 1.
526 */
527static void
528rnd_entropy_added(void)
529{
530 uint32_t pool_entropy;
531
532 KASSERT(mutex_owned(&rnd_global.lock));
533
534 if (__predict_true(rnd_initial_entropy))
535 return;
536 pool_entropy = rndpool_get_entropy_count(&rnd_global.pool);
537 if (pool_entropy > RND_ENTROPY_THRESHOLD * NBBY) {
538 rnd_printf_verbose("rnd: have initial entropy (%zu)\n",
539 pool_entropy);
540 rnd_initial_entropy = 1;
541 }
542}

544/*
545 * initialize the global random pool for our use.
546 * rnd_init() must be called very early on in the boot process, so
547 * the pool is ready for other devices to attach as sources.
548 */
549void
550rnd_init(void)
551{
552 uint32_t c;
553
554 if (rnd_ready)
555 return;
556
557 /*
558 * take a counter early, hoping that there's some variance in
559 * the following operations
560 */
561 c = rnd_counter();
562
563 rndsinks_init();
564
565 /* Initialize the sample queue. */
566 mutex_init(&rnd_samples.lock, MUTEX_DEFAULT, IPL_VM);
567 SIMPLEQ_INIT(&rnd_samples.q);
568
569 /* Initialize the global pool and sources list. */
570 mutex_init(&rnd_global.lock, MUTEX_DEFAULT, IPL_VM);
571 rndpool_init(&rnd_global.pool);
572 LIST_INIT(&rnd_global.sources);
573 cv_init(&rnd_global.cv, "rndsrc");
574
575 rnd_mempc = pool_cache_init(sizeof(rnd_sample_t), 0, 0, 0,
576 "rndsample", NULL, IPL_VM,
577 NULL, NULL, NULL);
578
579 /*
580 * Set resource limit. The rnd_process_events() function
581 * is called every tick and process the sample queue.
582 * Without limitation, if a lot of rnd_add_*() are called,
583 * all kernel memory may be eaten up.
584 */
585 pool_cache_sethardlimit(rnd_mempc, RND_POOLBITS, NULL, 0);
586
587 /*
588 * Mix *something*, *anything* into the pool to help it get started.
589 * However, it's not safe for rnd_counter() to call microtime() yet,
590 * so on some platforms we might just end up with zeros anyway.
591 * XXX more things to add would be nice.
592 */
593 if (c) {
594 mutex_spin_enter(&rnd_global.lock);
595 rndpool_add_data(&rnd_global.pool, &c, sizeof(c), 1);
596 c = rnd_counter();
597 rndpool_add_data(&rnd_global.pool, &c, sizeof(c), 1);
598 mutex_spin_exit(&rnd_global.lock);
599 }
600
601 /*
602 * Attach CPU RNG if available.
603 */
604#if defined(__HAVE_CPU_RNG) && !defined(_RUMPKERNEL)
605 if (cpu_rng_init()) {
606 /* IPL_VM because taken while rnd_global.lock is held. */
607 mutex_init(&rnd_cpu.lock, MUTEX_DEFAULT, IPL_VM);
608 rndsource_setcb(&rnd_cpu.source, rnd_cpu_get, &rnd_cpu.source);
609 rnd_attach_source(&rnd_cpu.source, "cpurng",
610 RND_TYPE_RNG, RND_FLAG_COLLECT_VALUE|
611 RND_FLAG_HASCB|RND_FLAG_HASENABLE);
612 rnd_cpu_get(RND_ENTROPY_THRESHOLD, &rnd_cpu.source);
613 }
614#endif
615
616 /*
617 * If we have a cycle counter, take its error with respect
618 * to the callout mechanism as a source of entropy, ala
619 * TrueRand.
620 *
621 */
622#if defined(__HAVE_CPU_COUNTER)
623 /* IPL_VM because taken while rnd_global.lock is held. */
624 mutex_init(&rnd_skew.lock, MUTEX_DEFAULT, IPL_VM);
625 callout_init(&rnd_skew.callout, CALLOUT_MPSAFE);
626 callout_setfunc(&rnd_skew.callout, rnd_skew_intr, NULL);
627 rndsource_setcb(&rnd_skew.source, rnd_skew_get, &rnd_skew.source);
628 rndsource_setenable(&rnd_skew.source, rnd_skew_enable);
629 rnd_attach_source(&rnd_skew.source, "callout", RND_TYPE_SKEW,
630 RND_FLAG_COLLECT_VALUE|RND_FLAG_ESTIMATE_VALUE|
631 RND_FLAG_HASCB|RND_FLAG_HASENABLE);
632 rnd_skew.iter = 100;
633 rnd_skew_intr(NULL);
634#endif
635
636 rnd_printf_verbose("rnd: initialised (%u)%s", RND_POOLBITS,
637 c ? " with counter\n" : "\n");
638 if (boot_rsp != NULL) {
639 mutex_spin_enter(&rnd_global.lock);
640 rndpool_add_data(&rnd_global.pool, boot_rsp->data,
641 sizeof(boot_rsp->data),
642 MIN(boot_rsp->entropy, RND_POOLBITS / 2));
643 rnd_entropy_added();
644 mutex_spin_exit(&rnd_global.lock);
645 rnd_printf("rnd: seeded with %d bits\n",
646 MIN(boot_rsp->entropy, RND_POOLBITS / 2));
647 explicit_memset(boot_rsp, 0, sizeof(*boot_rsp));
648 }
649 rnd_attach_source(&rnd_printf_source, "printf", RND_TYPE_UNKNOWN,
650 RND_FLAG_NO_ESTIMATE);
651 rnd_attach_source(&rnd_autoconf_source, "autoconf",
652 RND_TYPE_UNKNOWN,
653 RND_FLAG_COLLECT_TIME|RND_FLAG_ESTIMATE_TIME);
654 rnd_ready = 1;
655}

700/*
701 * Add a source to our list of sources.
702 */
703void
704rnd_attach_source(krndsource_t *rs, const char *name, uint32_t type,
705 uint32_t flags)
706{
707 uint32_t ts;
708
709 ts = rnd_counter();
710
711 strlcpy(rs->name, name, sizeof(rs->name));
712 memset(&rs->time_delta, 0, sizeof(rs->time_delta));
713 rs->time_delta.x = ts;
714 memset(&rs->value_delta, 0, sizeof(rs->value_delta));
715 rs->total = 0;
716
717 /*
718 * Some source setup, by type
719 */
720 rs->test = NULL;
721 rs->test_cnt = -1;
722
723 if (flags == 0) {
724 flags = RND_FLAG_DEFAULT;
725 }
726
727 switch (type) {
728 case RND_TYPE_NET: /* Don't collect by default */
729 flags |= (RND_FLAG_NO_COLLECT | RND_FLAG_NO_ESTIMATE);
730 break;
731 case RND_TYPE_RNG: /* Space for statistical testing */
732 rs->test = kmem_alloc(sizeof(rngtest_t), KM_NOSLEEP);
733 rs->test_cnt = 0;
734 /* FALLTHRU */
735 case RND_TYPE_VM: /* Process samples in bulk always */
736 flags |= RND_FLAG_FAST;
737 break;
738 default:
739 break;
740 }
741
742 rs->type = type;
743 rs->flags = flags;
744 rs->refcnt = 1;
745
746 rs->state = rnd_sample_allocate(rs);
747
748 mutex_spin_enter(&rnd_global.lock);
749
750#ifdef DIAGNOSTIC
751 krndsource_t *s;
752 LIST_FOREACH(s, &rnd_global.sources, list) {
753 if (s == rs) {
754 panic("%s: source '%s' already attached",
755 __func__, name);
756 /* NOTREACHED */
757 }
758 }
759#endif
760
761 LIST_INSERT_HEAD(&rnd_global.sources, rs, list);
762
763#ifdef RND_VERBOSE
764 rnd_printf_verbose("rnd: %s attached as an entropy source (",
765 rs->name);
766 if (!(flags & RND_FLAG_NO_COLLECT)) {
767 rnd_printf_verbose("collecting");
768 if (flags & RND_FLAG_NO_ESTIMATE)
769 rnd_printf_verbose(" without estimation");
770 } else {
771 rnd_printf_verbose("off");
772 }
773 rnd_printf_verbose(")\n");
774#endif
775
776 /*
777 * Again, put some more initial junk in the pool.
778 * FreeBSD claim to have an analysis that show 4 bits of
779 * entropy per source-attach timestamp. I am skeptical,
780 * but we count 1 bit per source here.
781 */
782 rndpool_add_data(&rnd_global.pool, &ts, sizeof(ts), 1);
783 mutex_spin_exit(&rnd_global.lock);
784}

786/*
787 * Remove a source from our list of sources.
788 */
789void
790rnd_detach_source(krndsource_t *source)
791{
792 rnd_sample_t *sample;
793
794 mutex_spin_enter(&rnd_global.lock);
795 LIST_REMOVE(source, list);
796 if (0 < --source->refcnt) {
797 do {
798 cv_wait(&rnd_global.cv, &rnd_global.lock);
799 } while (0 < source->refcnt);
800 }
801 mutex_spin_exit(&rnd_global.lock);
802
803 /*
804 * If there are samples queued up "remove" them from the sample queue
805 * by setting the source to the no-collect pseudosource.
806 */
807 mutex_spin_enter(&rnd_samples.lock);
808 sample = SIMPLEQ_FIRST(&rnd_samples.q);
809 while (sample != NULL) {
810 if (sample->source == source)
811 sample->source = &rnd_source_no_collect;
812
813 sample = SIMPLEQ_NEXT(sample, next);
814 }
815 mutex_spin_exit(&rnd_samples.lock);
816
817 if (source->state) {
818 rnd_sample_free(source->state);
819 source->state = NULL;
820 }
821
822 if (source->test) {
823 kmem_free(source->test, sizeof(rngtest_t));
824 }
825
826 rnd_printf_verbose("rnd: %s detached as an entropy source\n",
827 source->name);
828}

830static inline uint32_t
831rnd_estimate(krndsource_t *rs, uint32_t ts, uint32_t val)
832{
833 uint32_t entropy = 0, dt_est, dv_est;
834
835 dt_est = rnd_dt_estimate(rs, ts);
836 dv_est = rnd_dv_estimate(rs, val);
837
838 if (!(rs->flags & RND_FLAG_NO_ESTIMATE)) {
839 if (rs->flags & RND_FLAG_ESTIMATE_TIME) {
840 entropy += dt_est;
841 }
842
843 if (rs->flags & RND_FLAG_ESTIMATE_VALUE) {
844 entropy += dv_est;
845 }
846
847 }
848 return entropy;
849}

851/*
852 * Add a 32-bit value to the entropy pool. The rs parameter should point to
853 * the source-specific source structure.
854 */
855void
856_rnd_add_uint32(krndsource_t *rs, uint32_t val)
857{
858 uint32_t ts;
859 uint32_t entropy = 0;
860
861 if (rs->flags & RND_FLAG_NO_COLLECT)
862 return;
863
864 /*
865 * Sample the counter as soon as possible to avoid
866 * entropy overestimation.
867 */
868 ts = rnd_counter();
869
870 /*
871 * Calculate estimates - we may not use them, but if we do
872 * not calculate them, the estimators' history becomes invalid.
873 */
874 entropy = rnd_estimate(rs, ts, val);
875
876 rnd_add_data_ts(rs, &val, sizeof(val), entropy, ts, true);
877}

931static void
932rnd_add_data_ts(krndsource_t *rs, const void *const data, uint32_t len,
933 uint32_t entropy, uint32_t ts, bool schedule)
934{
935 rnd_sample_t *state = NULL;
936 const uint8_t *p = data;
937 uint32_t dint;
938 int todo, done, filled = 0;
939 int sample_count;
940 struct rnd_sampleq tmp_samples = SIMPLEQ_HEAD_INITIALIZER(tmp_samples);
941
942 if (rs &&
943 (rs->flags & RND_FLAG_NO_COLLECT ||
944 __predict_false(!(rs->flags &
945 (RND_FLAG_COLLECT_TIME|RND_FLAG_COLLECT_VALUE))))) {
946 return;
947 }
948 todo = len / sizeof(dint);
949 /*
950 * Let's try to be efficient: if we are warm, and a source
951 * is adding entropy at a rate of at least 1 bit every 10 seconds,
952 * mark it as "fast" and add its samples in bulk.
953 */
954 if (__predict_true(rs->flags & RND_FLAG_FAST) ||
955 (todo >= RND_SAMPLE_COUNT)) {
956 sample_count = RND_SAMPLE_COUNT;
957 } else {
958 if (!(rs->flags & RND_FLAG_HASCB) &&
959 !cold && rnd_initial_entropy) {
960 struct timeval upt;
961
962 getmicrouptime(&upt);
963 if ((upt.tv_sec > 0 && rs->total > upt.tv_sec * 10) ||
964 (upt.tv_sec > 10 && rs->total > upt.tv_sec) ||
965 (upt.tv_sec > 100 &&
966 rs->total > upt.tv_sec / 10)) {
967 rnd_printf_verbose("rnd: source %s is fast"
968 " (%d samples at once,"
969 " %d bits in %lld seconds), "
970 "processing samples in bulk.\n",
971 rs->name, todo, rs->total,
972 (long long int)upt.tv_sec);
973 rs->flags |= RND_FLAG_FAST;
974 }
975 }
976 sample_count = 2;
977 }
978
979 /*
980 * Loop over data packaging it into sample buffers.
981 * If a sample buffer allocation fails, drop all data.
982 */
983 for (done = 0; done < todo ; done++) {
984 state = rs->state;
985 if (state == NULL) {
986 state = rnd_sample_allocate_isr(rs);
987 if (__predict_false(state == NULL)) {
988 break;
989 }
990 rs->state = state;
991 }
992
993 state->ts[state->cursor] = ts;
994 (void)memcpy(&dint, &p[done*4], 4);
995 state->values[state->cursor] = dint;
996 state->cursor++;
997
998 if (state->cursor == sample_count) {
999 SIMPLEQ_INSERT_HEAD(&tmp_samples, state, next);
1000 filled++;
1001 rs->state = NULL;
1002 }
1003 }
1004
1005 if (__predict_false(state == NULL)) {
1006 while ((state = SIMPLEQ_FIRST(&tmp_samples))) {
1007 SIMPLEQ_REMOVE_HEAD(&tmp_samples, next);
1008 rnd_sample_free(state);
1009 }
1010 return;
1011 }
1012
1013 /*
1014 * Claim all the entropy on the last one we send to
1015 * the pool, so we don't rely on it being evenly distributed
1016 * in the supplied data.
1017 *
1018 * XXX The rndpool code must accept samples with more
1019 * XXX claimed entropy than bits for this to work right.
1020 */
1021 state->entropy += entropy;
1022 rs->total += entropy;
1023
1024 /*
1025 * If we didn't finish any sample buffers, we're done.
1026 */
1027 if (!filled) {
1028 return;
1029 }
1030
1031 mutex_spin_enter(&rnd_samples.lock);
1032 while ((state = SIMPLEQ_FIRST(&tmp_samples))) {
1033 SIMPLEQ_REMOVE_HEAD(&tmp_samples, next);
1034 SIMPLEQ_INSERT_HEAD(&rnd_samples.q, state, next);
1035 }
1036 mutex_spin_exit(&rnd_samples.lock);
1037
1038 /* Cause processing of queued samples, if caller wants it. */
1039 if (schedule)
1040 rnd_schedule_process();
1041}

1043static int
1044rnd_hwrng_test(rnd_sample_t *sample)
1045{
1046 krndsource_t *source = sample->source;
1047 size_t cmplen;
1048 uint8_t *v1, *v2;
1049 size_t resid, totest;
1050
1051 KASSERT(source->type == RND_TYPE_RNG);
1052
1053 /*
1054 * Continuous-output test: compare two halves of the
1055 * sample buffer to each other. The sample buffer (64 ints,
1056 * so either 256 or 512 bytes on any modern machine) should be
1057 * much larger than a typical hardware RNG output, so this seems
1058 * a reasonable way to do it without retaining extra data.
1059 */
1060 cmplen = sizeof(sample->values) / 2;
1061 v1 = (uint8_t *)sample->values;
1062 v2 = (uint8_t *)sample->values + cmplen;
1063
1064 if (__predict_false(!memcmp(v1, v2, cmplen))) {
1065 rnd_printf("rnd: source \"%s\""
1066 " failed continuous-output test.\n",
1067 source->name);
1068 return 1;
1069 }
1070
1071 /*
1072 * FIPS 140 statistical RNG test. We must accumulate 20,000 bits.
1073 */
1074 if (__predict_true(source->test_cnt == -1)) {
1075 /* already passed the test */
1076 return 0;
1077 }
1078 resid = FIPS140_RNG_TEST_BYTES - source->test_cnt;
1079 totest = MIN(RND_SAMPLE_COUNT * 4, resid);
1080 memcpy(source->test->rt_b + source->test_cnt, sample->values, totest);
1081 resid -= totest;
1082 source->test_cnt += totest;
1083 if (resid == 0) {
1084 strlcpy(source->test->rt_name, source->name,
1085 sizeof(source->test->rt_name));
1086 if (rngtest(source->test)) {
1087 rnd_printf("rnd: source \"%s\""
1088 " failed statistical test.",
1089 source->name);
1090 return 1;
1091 }
1092 source->test_cnt = -1;
1093 explicit_memset(source->test, 0, sizeof(*source->test));
1094 }
1095 return 0;
1096}

1098/*
1099 * Process the events in the ring buffer. Called by rnd_timeout or
1100 * by the add routines directly if the callout has never fired (that
1101 * is, if we are "cold" -- just booted).
1102 *
1103 */
1104static void
1105rnd_process_events(void)
1106{
1107 rnd_sample_t *sample = NULL;
1108 krndsource_t *source;
1109 static krndsource_t *last_source;
1110 uint32_t entropy;
1111 size_t pool_entropy;
1112 int wake = 0;
1113 struct rnd_sampleq dq_samples = SIMPLEQ_HEAD_INITIALIZER(dq_samples);
1114 struct rnd_sampleq df_samples = SIMPLEQ_HEAD_INITIALIZER(df_samples);
1115
1116 /*
1117 * Drain to the on-stack queue and drop the lock.
1118 */
1119 mutex_spin_enter(&rnd_samples.lock);
1120 while ((sample = SIMPLEQ_FIRST(&rnd_samples.q))) {
1121 SIMPLEQ_REMOVE_HEAD(&rnd_samples.q, next);
1122 /*
1123 * We repeat this check here, since it is possible
1124 * the source was disabled before we were called, but
1125 * after the entry was queued.
1126 */
1127 if (__predict_false(!(sample->source->flags &
1128 (RND_FLAG_COLLECT_TIME|RND_FLAG_COLLECT_VALUE)))) {
1129 SIMPLEQ_INSERT_TAIL(&df_samples, sample, next);
1130 } else {
1131 SIMPLEQ_INSERT_TAIL(&dq_samples, sample, next);
1132 }
1133 }
1134 mutex_spin_exit(&rnd_samples.lock);
1135
1136 /* Don't thrash the rndpool mtx either. Hold, add all samples. */
1137 mutex_spin_enter(&rnd_global.lock);
1138
1139 pool_entropy = rndpool_get_entropy_count(&rnd_global.pool);
1140
1141 while ((sample = SIMPLEQ_FIRST(&dq_samples))) {
1142 int sample_count;
1143
1144 SIMPLEQ_REMOVE_HEAD(&dq_samples, next);
1145 source = sample->source;
1146 entropy = sample->entropy;
1147 sample_count = sample->cursor;
1148
1149 /*
1150 * Don't provide a side channel for timing attacks on
1151 * low-rate sources: require mixing with some other
1152 * source before we schedule a wakeup.
1153 */
1154 if (!wake &&
1155 (source != last_source || source->flags & RND_FLAG_FAST)) {
1156 wake++;
1157 }
1158 last_source = source;
1159
1160 /*
1161 * If the source has been disabled, ignore samples from
1162 * it.
1163 */
1164 if (source->flags & RND_FLAG_NO_COLLECT)
1165 goto skip;
1166
1167 /*
1168 * Hardware generators are great but sometimes they
1169 * have...hardware issues. Don't use any data from
1170 * them unless it passes some tests.
1171 */
1172 if (source->type == RND_TYPE_RNG) {
1173 if (__predict_false(rnd_hwrng_test(sample))) {
1174 source->flags |= RND_FLAG_NO_COLLECT;
1175 rnd_printf("rnd: disabling source \"%s\".\n",
1176 source->name);
1177 goto skip;
1178 }
1179 }
1180
1181 if (source->flags & RND_FLAG_COLLECT_VALUE) {
1182 rndpool_add_data(&rnd_global.pool, sample->values,
1183 sample_count * sizeof(sample->values[1]),
1184 0);
1185 }
1186 if (source->flags & RND_FLAG_COLLECT_TIME) {
1187 rndpool_add_data(&rnd_global.pool, sample->ts,
1188 sample_count * sizeof(sample->ts[1]),
1189 0);
1190 }
1191
1192 pool_entropy += entropy;
1193 source->total += sample->entropy;
1194skip: SIMPLEQ_INSERT_TAIL(&df_samples, sample, next);
1195 }
1196 rndpool_set_entropy_count(&rnd_global.pool, pool_entropy);
1197 rnd_entropy_added();
1198 mutex_spin_exit(&rnd_global.lock);
1199
1200 /*
1201 * If we filled the pool past the threshold, wake anyone
1202 * waiting for entropy.
1203 */
1204 if (pool_entropy > RND_ENTROPY_THRESHOLD * 8) {
1205 wake++;
1206 }
1207
1208 /* Now we hold no locks: clean up. */
1209 while ((sample = SIMPLEQ_FIRST(&df_samples))) {
1210 SIMPLEQ_REMOVE_HEAD(&df_samples, next);
1211 rnd_sample_free(sample);
1212 }
1213
1214 /*
1215 * Wake up any potential readers waiting.
1216 */
1217 if (wake) {
1218 rnd_schedule_wakeup();
1219 }
1220}

1236static uint32_t
1237rnd_extract_data(void *p, uint32_t len, uint32_t flags)
1238{
1239 static int timed_in;
1240 uint32_t retval;
1241
1242 mutex_spin_enter(&rnd_global.lock);
1243 if (__predict_false(!timed_in)) {
1244 if (boottime.tv_sec) {
1245 rndpool_add_data(&rnd_global.pool, &boottime,
1246 sizeof(boottime), 0);
1247 }
1248 timed_in++;
1249 }
1250 if (__predict_false(!rnd_initial_entropy)) {
1251 uint32_t c;
1252
1253 rnd_printf_verbose("rnd: WARNING! initial entropy low (%u).\n",
1254 rndpool_get_entropy_count(&rnd_global.pool));
1255 /* Try once again to put something in the pool */
1256 c = rnd_counter();
1257 rndpool_add_data(&rnd_global.pool, &c, sizeof(c), 1);
1258 }
1259
1260#ifdef DIAGNOSTIC
1261 while (!rnd_tested) {
1262 int entropy_count =
1263 rndpool_get_entropy_count(&rnd_global.pool);
1264 rnd_printf_verbose("rnd: starting statistical RNG test,"
1265 " entropy = %d.\n",
1266 entropy_count);
1267 if (rndpool_extract_data(&rnd_global.pool, rnd_rt.rt_b,
1268 sizeof(rnd_rt.rt_b), RND_EXTRACT_ANY)
1269 != sizeof(rnd_rt.rt_b)) {
1270 panic("rnd: could not get bits for statistical test");
1271 }
1272 /*
1273 * Stash the tested bits so we can put them back in the
1274 * pool, restoring the entropy count. DO NOT rely on
1275 * rngtest to maintain the bits pristine -- we could end
1276 * up adding back non-random data claiming it were pure
1277 * entropy.
1278 */
1279 memcpy(rnd_testbits, rnd_rt.rt_b, sizeof(rnd_rt.rt_b));
1280 strlcpy(rnd_rt.rt_name, "entropy pool",
1281 sizeof(rnd_rt.rt_name));
1282 if (rngtest(&rnd_rt)) {
1283 /*
1284 * The probabiliity of a Type I error is 3/10000,
1285 * but note this can only happen at boot time.
1286 * The relevant standard says to reset the module,
1287 * but developers objected...
1288 */
1289 rnd_printf("rnd: WARNING, ENTROPY POOL FAILED "
1290 "STATISTICAL TEST!\n");
1291 continue;
1292 }
1293 explicit_memset(&rnd_rt, 0, sizeof(rnd_rt));
1294 rndpool_add_data(&rnd_global.pool, rnd_testbits,
1295 sizeof(rnd_testbits), entropy_count);
1296 explicit_memset(rnd_testbits, 0, sizeof(rnd_testbits));
1297 rnd_printf_verbose("rnd: statistical RNG test done,"
1298 " entropy = %d.\n",
1299 rndpool_get_entropy_count(&rnd_global.pool));
1300 rnd_tested++;
1301 }
1302#endif
1303 retval = rndpool_extract_data(&rnd_global.pool, p, len, flags);
1304 mutex_spin_exit(&rnd_global.lock);
1305
1306 return retval;
1307}

1309/*
1310 * Fill the buffer with as much entropy as we can. Return true if it
1311 * has full entropy and false if not.
1312 */
1313bool
1314rnd_extract(void *buffer, size_t bytes)
1315{
1316 const size_t extracted = rnd_extract_data(buffer, bytes,
1317 RND_EXTRACT_GOOD);
1318
1319 if (extracted < bytes) {
1320 rnd_getmore(bytes - extracted);
1321 (void)rnd_extract_data((uint8_t *)buffer + extracted,
1322 bytes - extracted, RND_EXTRACT_ANY);
1323 return false;
1324 }
1325
1326 return true;
1327}
1328
1329/*
1330 * If we have as much entropy as is requested, fill the buffer with it
1331 * and return true. Otherwise, leave the buffer alone and return
1332 * false.
1333 */
1334
1335CTASSERT(RND_ENTROPY_THRESHOLD <= 0xffffffffUL);
1336CTASSERT(RNDSINK_MAX_BYTES <= (0xffffffffUL - RND_ENTROPY_THRESHOLD));
1337CTASSERT((RNDSINK_MAX_BYTES + RND_ENTROPY_THRESHOLD) <=
1338 (0xffffffffUL / NBBY));
1339
1340bool
1341rnd_tryextract(void *buffer, size_t bytes)
1342{
1343 uint32_t bits_needed, bytes_requested;
1344
1345 KASSERT(bytes <= RNDSINK_MAX_BYTES);
1346 bits_needed = ((bytes + RND_ENTROPY_THRESHOLD) * NBBY);
1347
1348 mutex_spin_enter(&rnd_global.lock);
1349 if (bits_needed <= rndpool_get_entropy_count(&rnd_global.pool)) {
1350 const uint32_t extracted __diagused =
1351 rndpool_extract_data(&rnd_global.pool, buffer, bytes,
1352 RND_EXTRACT_GOOD);
1353
1354 KASSERT(extracted == bytes);
1355 bytes_requested = 0;
1356 } else {
1357 /* XXX Figure the threshold into this... */
1358 bytes_requested = howmany((bits_needed -
1359 rndpool_get_entropy_count(&rnd_global.pool)), NBBY);
1360 KASSERT(0 < bytes_requested);
1361 }
1362 mutex_spin_exit(&rnd_global.lock);
1363
1364 if (0 < bytes_requested)
1365 rnd_getmore(bytes_requested);
1366
1367 return bytes_requested == 0;
1368}

1370void
1371rnd_seed(void *base, size_t len)
1372{
1373 SHA1_CTX s;
1374 uint8_t digest[SHA1_DIGEST_LENGTH];
1375
1376 if (len != sizeof(*boot_rsp)) {
1377 rnd_printf("rnd: bad seed length %d\n", (int)len);
1378 return;
1379 }
1380
1381 boot_rsp = (rndsave_t *)base;
1382 SHA1Init(&s);
1383 SHA1Update(&s, (uint8_t *)&boot_rsp->entropy,
1384 sizeof(boot_rsp->entropy));
1385 SHA1Update(&s, boot_rsp->data, sizeof(boot_rsp->data));
1386 SHA1Final(digest, &s);
1387
1388 if (memcmp(digest, boot_rsp->digest, sizeof(digest))) {
1389 rnd_printf("rnd: bad seed checksum\n");
1390 return;
1391 }
1392
1393 /*
1394 * It's not really well-defined whether bootloader-supplied
1395 * modules run before or after rnd_init(). Handle both cases.
1396 */
1397 if (rnd_ready) {
1398 rnd_printf_verbose("rnd: ready,"
1399 " feeding in seed data directly.\n");
1400 mutex_spin_enter(&rnd_global.lock);
1401 rndpool_add_data(&rnd_global.pool, boot_rsp->data,
1402 sizeof(boot_rsp->data),
1403 MIN(boot_rsp->entropy, RND_POOLBITS / 2));
1404 explicit_memset(boot_rsp, 0, sizeof(*boot_rsp));
1405 mutex_spin_exit(&rnd_global.lock);
1406 } else {
1407 rnd_printf_verbose("rnd: not ready, deferring seed feed.\n");
1408 }
1409}

1422static void
1423krndsource_to_rndsource_est(krndsource_t *kr, rndsource_est_t *re)
1424{
1425
1426 memset(re, 0, sizeof(*re));
1427 krndsource_to_rndsource(kr, &re->rt);
1428 re->dt_samples = kr->time_delta.insamples;
1429 re->dt_total = kr->time_delta.outbits;
1430 re->dv_samples = kr->value_delta.insamples;
1431 re->dv_total = kr->value_delta.outbits;
1432}
1433
1434static void
1435krs_setflags(krndsource_t *kr, uint32_t flags, uint32_t mask)
1436{
1437 uint32_t oflags = kr->flags;
1438
1439 kr->flags &= ~mask;
1440 kr->flags |= (flags & mask);
1441
1442 if (oflags & RND_FLAG_HASENABLE &&
1443 ((oflags & RND_FLAG_NO_COLLECT) !=
1444 (flags & RND_FLAG_NO_COLLECT))) {
1445 kr->enable(kr, !(flags & RND_FLAG_NO_COLLECT));
1446 }
1447}

1449int
1450rnd_system_ioctl(struct file *fp, u_long cmd, void *addr)
1451{
1452 krndsource_t *kr;
1453 rndstat_t *rst;
1454 rndstat_name_t *rstnm;
1455 rndstat_est_t *rset;
1456 rndstat_est_name_t *rsetnm;
1457 rndctl_t *rctl;
1458 rnddata_t *rnddata;
1459 uint32_t count, start;
1460 int ret = 0;
1461 int estimate_ok = 0, estimate = 0;
1462
1463 switch (cmd) {
1464 case RNDGETENTCNT:
1465 break;
1466
1467 case RNDGETPOOLSTAT:
1468 case RNDGETSRCNUM:
1469 case RNDGETSRCNAME:
1470 case RNDGETESTNUM:
1471 case RNDGETESTNAME:
1472 ret = kauth_authorize_device(curlwp->l_cred,
1473 KAUTH_DEVICE_RND_GETPRIV, NULL, NULL, NULL, NULL);
1474 if (ret)
1475 return ret;
1476 break;
1477
1478 case RNDCTL:
1479 ret = kauth_authorize_device(curlwp->l_cred,
1480 KAUTH_DEVICE_RND_SETPRIV, NULL, NULL, NULL, NULL);
1481 if (ret)
1482 return ret;
1483 break;
1484
1485 case RNDADDDATA:
1486 ret = kauth_authorize_device(curlwp->l_cred,
1487 KAUTH_DEVICE_RND_ADDDATA, NULL, NULL, NULL, NULL);
1488 if (ret)
1489 return ret;
1490 estimate_ok = !kauth_authorize_device(curlwp->l_cred,
1491 KAUTH_DEVICE_RND_ADDDATA_ESTIMATE, NULL, NULL, NULL, NULL);
1492 break;
1493
1494 default:
1495 MODULE_HOOK_CALL(rnd_ioctl_50_hook, (fp, cmd, addr),
1496 enosys(), ret);
1497#if defined(_LP64)
1498 if (ret == ENOSYS)
1499 MODULE_HOOK_CALL(rnd_ioctl32_50_hook, (fp, cmd, addr),
1500 enosys(), ret);
1501#endif
1502 if (ret == ENOSYS)
1503 ret = ENOTTY;
1504 return ret;
1505 }
1506
1507 switch (cmd) {
1508 case RNDGETENTCNT:
1509 mutex_spin_enter(&rnd_global.lock);
1510 *(uint32_t *)addr =
1511 rndpool_get_entropy_count(&rnd_global.pool);
1512 mutex_spin_exit(&rnd_global.lock);
1513 break;
1514
1515 case RNDGETPOOLSTAT:
1516 mutex_spin_enter(&rnd_global.lock);
1517 rndpool_get_stats(&rnd_global.pool, addr,
1518 sizeof(rndpoolstat_t));
1519 mutex_spin_exit(&rnd_global.lock);
1520 break;
1521
1522 case RNDGETSRCNUM:
1523 rst = (rndstat_t *)addr;
1524
1525 if (rst->count == 0)
1526 break;
1527
1528 if (rst->count > RND_MAXSTATCOUNT)
1529 return EINVAL;
1530
1531 mutex_spin_enter(&rnd_global.lock);
1532 /*
1533 * Find the starting source by running through the
1534 * list of sources.
1535 */
1536 kr = LIST_FIRST(&rnd_global.sources);
1537 start = rst->start;
1538 while (kr != NULL && start >= 1) {
1539 kr = LIST_NEXT(kr, list);
1540 start--;
1541 }
1542
1543 /*
1544 * Return up to as many structures as the user asked
1545 * for. If we run out of sources, a count of zero
1546 * will be returned, without an error.
1547 */
1548 for (count = 0; count < rst->count && kr != NULL; count++) {
1549 krndsource_to_rndsource(kr, &rst->source[count]);
1550 kr = LIST_NEXT(kr, list);
1551 }
1552
1553 rst->count = count;
1554
1555 mutex_spin_exit(&rnd_global.lock);
1556 break;
1557
1558 case RNDGETESTNUM:
1559 rset = (rndstat_est_t *)addr;
1560
1561 if (rset->count == 0)
1562 break;
1563
1564 if (rset->count > RND_MAXSTATCOUNT)
1565 return EINVAL;
1566
1567 mutex_spin_enter(&rnd_global.lock);
1568 /*
1569 * Find the starting source by running through the
1570 * list of sources.
1571 */
1572 kr = LIST_FIRST(&rnd_global.sources);
1573 start = rset->start;
1574 while (kr != NULL && start > 0) {
1575 kr = LIST_NEXT(kr, list);
1576 start--;
1577 }
1578
1579 /*
1580 * Return up to as many structures as the user asked
1581 * for. If we run out of sources, a count of zero
1582 * will be returned, without an error.
1583 */
1584 for (count = 0; count < rset->count && kr != NULL; count++) {
1585 krndsource_to_rndsource_est(kr, &rset->source[count]);
1586 kr = LIST_NEXT(kr, list);
1587 }
1588
1589 rset->count = count;
1590
1591 mutex_spin_exit(&rnd_global.lock);
1592 break;
1593
1594 case RNDGETSRCNAME:
1595 /*
1596 * Scan through the list, trying to find the name.
1597 */
1598 mutex_spin_enter(&rnd_global.lock);
1599 rstnm = (rndstat_name_t *)addr;
1600 kr = LIST_FIRST(&rnd_global.sources);
1601 while (kr != NULL) {
1602 if (strncmp(kr->name, rstnm->name,
1603 MIN(sizeof(kr->name),
1604 sizeof(rstnm->name))) == 0) {
1605 krndsource_to_rndsource(kr, &rstnm->source);
1606 mutex_spin_exit(&rnd_global.lock);
1607 return 0;
1608 }
1609 kr = LIST_NEXT(kr, list);
1610 }
1611 mutex_spin_exit(&rnd_global.lock);
1612
1613 ret = ENOENT; /* name not found */
1614
1615 break;
1616
1617 case RNDGETESTNAME:
1618 /*
1619 * Scan through the list, trying to find the name.
1620 */
1621 mutex_spin_enter(&rnd_global.lock);
1622 rsetnm = (rndstat_est_name_t *)addr;
1623 kr = LIST_FIRST(&rnd_global.sources);
1624 while (kr != NULL) {
1625 if (strncmp(kr->name, rsetnm->name,
1626 MIN(sizeof(kr->name), sizeof(rsetnm->name)))
1627 == 0) {
1628 krndsource_to_rndsource_est(kr,
1629 &rsetnm->source);
1630 mutex_spin_exit(&rnd_global.lock);
1631 return 0;
1632 }
1633 kr = LIST_NEXT(kr, list);
1634 }
1635 mutex_spin_exit(&rnd_global.lock);
1636
1637 ret = ENOENT; /* name not found */
1638
1639 break;
1640
1641 case RNDCTL:
1642 /*
1643 * Set flags to enable/disable entropy counting and/or
1644 * collection.
1645 */
1646 mutex_spin_enter(&rnd_global.lock);
1647 rctl = (rndctl_t *)addr;
1648 kr = LIST_FIRST(&rnd_global.sources);
1649
1650 /*
1651 * Flags set apply to all sources of this type.
1652 */
1653 if (rctl->type != 0xff) {
1654 while (kr != NULL) {
1655 if (kr->type == rctl->type) {
1656 krs_setflags(kr, rctl->flags,
1657 rctl->mask);
1658 }
1659 kr = LIST_NEXT(kr, list);
1660 }
1661 mutex_spin_exit(&rnd_global.lock);
1662 return 0;
1663 }
1664
1665 /*
1666 * scan through the list, trying to find the name
1667 */
1668 while (kr != NULL) {
1669 if (strncmp(kr->name, rctl->name,
1670 MIN(sizeof(kr->name), sizeof(rctl->name)))
1671 == 0) {
1672 krs_setflags(kr, rctl->flags, rctl->mask);
1673 mutex_spin_exit(&rnd_global.lock);
1674 return 0;
1675 }
1676 kr = LIST_NEXT(kr, list);
1677 }
1678
1679 mutex_spin_exit(&rnd_global.lock);
1680 ret = ENOENT; /* name not found */
1681
1682 break;
1683
1684 case RNDADDDATA:
1685 /*
1686 * Don't seed twice if our bootloader has
1687 * seed loading support.
1688 */
1689 if (!boot_rsp) {
1690 rnddata = (rnddata_t *)addr;
1691
1692 if (rnddata->len > sizeof(rnddata->data))
1693 return EINVAL;
1694
1695 if (estimate_ok) {
1696 /*
1697 * Do not accept absurd entropy estimates, and
1698 * do not flood the pool with entropy such that
1699 * new samples are discarded henceforth.
1700 */
1701 estimate = MIN((rnddata->len * NBBY) / 2,
1702 MIN(rnddata->entropy, RND_POOLBITS / 2));
1703 } else {
1704 estimate = 0;
1705 }
1706
1707 mutex_spin_enter(&rnd_global.lock);
1708 rndpool_add_data(&rnd_global.pool, rnddata->data,
1709 rnddata->len, estimate);
1710 rnd_entropy_added();
1711 mutex_spin_exit(&rnd_global.lock);
1712
1713 rndsinks_distribute();
1714 } else {
1715 rnd_printf_verbose("rnd"
1716 ": already seeded by boot loader\n");
1717 }
1718 break;
1719
1720 default:
1721 return ENOTTY;
1722 }
1723
1724 return ret;
1725}
