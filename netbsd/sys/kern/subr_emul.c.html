74void
75emul_find_root(struct lwp *l, struct exec_package *epp)
76{
77 struct vnode *vp;
78 const char *emul_path;
79
80 if (epp->ep_emul_root != NULL)
81 /* We've already found it */
82 return;
83
84 emul_path = epp->ep_esch->es_emul->e_path;
85 if (emul_path == NULL)
86 /* Emulation doesn't have a root */
87 return;
88
89 if (namei_simple_kernel(emul_path, NSM_FOLLOW_NOEMULROOT, &vp) != 0)
90 /* emulation root doesn't exist */
91 return;
92
93 epp->ep_emul_root = vp;
94}

96/*
97 * Search the alternate path for dynamic binary interpreter. If not found
98 * there, check if the interpreter exists in within 'proper' tree.
99 */
100int
101emul_find_interp(struct lwp *l, struct exec_package *epp, const char *itp)
102{
103 int error;
104 struct pathbuf *pb;
105 struct nameidata nd;
106 unsigned int flags;
107
108 pb = pathbuf_create(itp);
109 if (pb == NULL) {
110 return ENOMEM;
111 }
112
113 /* If we haven't found the emulation root already, do so now */
114 /* Maybe we should remember failures somehow ? */
115 if (epp->ep_esch->es_emul->e_path != 0 && epp->ep_emul_root == NULL)
116 emul_find_root(l, epp);
117
118 if (epp->ep_interp != NULL)
119 vrele(epp->ep_interp);
120
121 /* We need to use the emulation root for the new program,
122 * not the one for the current process. */
123 if (epp->ep_emul_root == NULL)
124 flags = FOLLOW;
125 else {
126 nd.ni_erootdir = epp->ep_emul_root;
127 /* hack: Pass in the emulation path for ktrace calls */
128 nd.ni_next = epp->ep_esch->es_emul->e_path;
129 flags = FOLLOW | TRYEMULROOT | EMULROOTSET;
130 }
131
132 NDINIT(&nd, LOOKUP, flags, pb);
133 error = namei(&nd);
134 if (error != 0) {
135 epp->ep_interp = NULL;
136 pathbuf_destroy(pb);
137 return error;
138 }
139
140 /* Save interpreter in case we actually need to load it */
141 epp->ep_interp = nd.ni_vp;
142
143 pathbuf_destroy(pb);
144
145 return 0;
146}
