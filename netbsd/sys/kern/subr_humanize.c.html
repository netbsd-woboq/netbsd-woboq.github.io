38/*
39 * snprintf() `bytes' into `buf', reformatting it so that the number,
40 * plus a possible `x' + suffix extension) fits into len bytes (including
41 * the terminating NUL).
42 * Returns the number of bytes stored in buf, or -1 if there was a problem.
43 * E.g, given a len of 9 and a suffix of `B':
44 * bytes result
45 * ----- ------
46 * 99999 `99999 B'
47 * 100000 `97 kB'
48 * 66715648 `65152 kB'
49 * 252215296 `240 MB'
50 */
51int
52humanize_number(char *buf, size_t len, uint64_t bytes, const char *suffix,
53 int divisor)
54{
55 /* prefixes are: (none), kilo, Mega, Giga, Tera, Peta, Exa */
56 const char *prefixes;
57 int r;
58 uint64_t umax, b10;
59 size_t i, suffixlen;
60
61 if (buf == NULL || suffix == NULL)
62 return (-1);
63 if (len > 0)
64 buf[0] = '\0';
65 suffixlen = strlen(suffix);
66 /* check if enough room for `x y' + suffix + `\0' */
67 if (len < 4 + suffixlen)
68 return (-1);
69
70 if (divisor == 1024) {
71 /*
72 * binary multiplies
73 * XXX IEC 60027-2 recommends Ki, Mi, Gi...
74 */
75 prefixes = " KMGTPE";
76 } else
77 prefixes = " kMGTPE"; /* SI for decimal multiplies */
78
79 umax = 1;
80 b10 = bytes/10;
81 for (i = 0; i < len - suffixlen - 3; i++) {
82 if (umax > b10) {
83 /*
84 * there is space for the unscaled number
85 * but bytes might be ~0 - there is no bigger
86 * value available for umax, so we must skip
87 * the "bytes >= umax" test below
88 */
89 i = 0;
90 goto nodiv;
91 }
92 umax *= 10;
93 }
94 for (i = 0; bytes >= umax && prefixes[i + 1]; i++)
95 bytes /= divisor;
96 nodiv:
97 r = snprintf(buf, len, "%qu%s%c%s", (unsigned long long)bytes,
98 i == 0 ? "" : " ", prefixes[i], suffix);
99
100 return (r);
101}
102
103int
104format_bytes(char *buf, size_t len, uint64_t bytes)
105{
106 int rv;
107 size_t nlen;
108
109 rv = humanize_number(buf, len, bytes, "B", 1024);
110 if (rv != -1) {
111 /* nuke the trailing ` B' if it exists */
112 nlen = strlen(buf) - 2;
113 if (strcmp(&buf[nlen], " B") == 0)
114 buf[nlen] = '\0';
115 }
116 return (rv);
117}
