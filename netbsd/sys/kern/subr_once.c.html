33#include <sys/param.h>
34#include <sys/systm.h>
35#include <sys/condvar.h>
36#include <sys/mutex.h>
37#include <sys/once.h>
38
39static kmutex_t oncemtx;
40static kcondvar_t oncecv;
41
42void
43once_init(void)
44{
45
46 mutex_init(&oncemtx, MUTEX_DEFAULT, IPL_NONE);
47 cv_init(&oncecv, "runonce");
48}
49
50int
51_init_once(once_t *o, int (*fn)(void))
52{
53 /* Fastpath handled by RUN_ONCE() */
54
55 int error;
56
57 mutex_enter(&oncemtx);
58 while (o->o_status == ONCE_RUNNING)
59 cv_wait(&oncecv, &oncemtx);
60
61 if (o->o_refcnt++ == 0) {
62 o->o_status = ONCE_RUNNING;
63 mutex_exit(&oncemtx);
64 o->o_error = fn();
65 mutex_enter(&oncemtx);
66 o->o_status = ONCE_DONE;
67 cv_broadcast(&oncecv);
68 }
69 KASSERT(o->o_refcnt != 0); /* detect overflow */
70
71 while (o->o_status == ONCE_RUNNING)
72 cv_wait(&oncecv, &oncemtx);
73 error = o->o_error;
74 mutex_exit(&oncemtx);
75
76 return error;
77}
78
79void
80_fini_once(once_t *o, void (*fn)(void))
81{
82 mutex_enter(&oncemtx);
83 while (o->o_status == ONCE_RUNNING)
84 cv_wait(&oncecv, &oncemtx);
85
86 KASSERT(o->o_refcnt != 0); /* we need to call _init_once() once */
87 if (--o->o_refcnt == 0) {
88 o->o_status = ONCE_RUNNING;
89 mutex_exit(&oncemtx);
90 fn();
91 mutex_enter(&oncemtx);
92 o->o_status = ONCE_VIRGIN;
93 cv_broadcast(&oncecv);
94 }
95
96 mutex_exit(&oncemtx);
97}
