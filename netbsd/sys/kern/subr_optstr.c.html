40/*
41 * Given an options string of the form 'a=b c=d ... y=z' and a key,
42 * looks for the given key's value in the string and returns it in buf
43 * with a maximum of bufsize bytes. If the key is found, returns true;
44 * otherwise FALSE.
45 */
46bool
47optstr_get(const char *optstr, const char *key, char *buf, size_t bufsize)
48{
49 bool found;
50
51 found = false;
52
53 /* Skip any initial spaces until we find a word. */
54 while (*optstr == ' ')
55 optstr++;
56
57 /* Search for the given key within the option string. */
58 while (!found && *optstr != '\0') {
59 const char *keyp;
60
61 /* Check if the next word matches the key. */
62 keyp = key;
63 while (*optstr == *keyp) {
64 optstr++;
65 keyp++;
66 }
67
68 if (*optstr == '=' && *keyp == '\0')
69 found = true;
70 else {
71 /* Key not found; skip until next space. */
72 while (*optstr != ' ' && *optstr != '\0')
73 optstr++;
74
75 /* And now skip until next word. */
76 while (*optstr == ' ')
77 optstr++;
78 }
79 }
80
81 /* If the key was found; copy its value to the target buffer. */
82 if (found) {
83 const char *lastbuf;
84
85 lastbuf = buf + (bufsize - 1);
86
87 optstr++; /* Skip '='. */
88 while (buf != lastbuf && *optstr != ' ' && *optstr != '\0')
89 *buf++ = *optstr++;
90 *buf = '\0';
91 }
92
93 return found;
94}
