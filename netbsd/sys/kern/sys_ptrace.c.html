148/*
149 * PTRACE methods
150 */
151
152static int
153ptrace_copyin_piod(struct ptrace_io_desc *piod, const void *addr, size_t len)
154{
155 if (len != 0 && sizeof(*piod) != len)
156 return EINVAL;
157
158 return copyin(addr, piod, sizeof(*piod));
159}
160
161static int
162ptrace_copyout_piod(const struct ptrace_io_desc *piod, void *addr, size_t len)
163{
164 if (len != 0 && sizeof(*piod) != len)
165 return EINVAL;
166
167 return copyout(piod, addr, sizeof(*piod));
168}
169
170static int
171ptrace_copyin_siginfo(struct ptrace_siginfo *psi, const void *addr, size_t len)
172{
173 if (sizeof(*psi) != len)
174 return EINVAL;
175
176 return copyin(addr, psi, sizeof(*psi));
177}
178
179static int
180ptrace_copyout_siginfo(const struct ptrace_siginfo *psi, void *addr, size_t len)
181{
182 if (sizeof(*psi) != len)
183 return EINVAL;
184
185 return copyout(psi, addr, sizeof(*psi));
186}
187
188static struct ptrace_methods native_ptm = {
189 .ptm_copyin_piod = ptrace_copyin_piod,
190 .ptm_copyout_piod = ptrace_copyout_piod,
191 .ptm_copyin_siginfo = ptrace_copyin_siginfo,
192 .ptm_copyout_siginfo = ptrace_copyout_siginfo,
193 .ptm_doregs = process_doregs,
194 .ptm_dofpregs = process_dofpregs,
195 .ptm_dodbregs = process_dodbregs,
196};
197
198static const struct syscall_package ptrace_syscalls[] = {
199 { SYS_ptrace, 0, (sy_call_t *)sys_ptrace },
200 { 0, 0, NULL },
201};
202
203/*
204 * Process debugging system call.
205 */
206int
207sys_ptrace(struct lwp *l, const struct sys_ptrace_args *uap, register_t *retval)
208{
209 /* {
210 syscallarg(int) req;
211 syscallarg(pid_t) pid;
212 syscallarg(void *) addr;
213 syscallarg(int) data;
214 } */
215
216 return do_ptrace(&native_ptm, l, SCARG(uap, req), SCARG(uap, pid),
217 SCARG(uap, addr), SCARG(uap, data), retval);
218}
219
220#define DEPS "ptrace_common"
221
222MODULE(MODULE_CLASS_EXEC, ptrace, DEPS);
223
224static int
225ptrace_modcmd(modcmd_t cmd, void *arg)
226{
227 int error;
228
229 switch (cmd) {
230 case MODULE_CMD_INIT:
231 error = syscall_establish(&emul_netbsd, ptrace_syscalls);
232 break;
233 case MODULE_CMD_FINI:
234 error = syscall_disestablish(&emul_netbsd, ptrace_syscalls);
235 break;
236 default:
237 error = ENOTTY;
238 break;
239 }
240 return error;
241}
