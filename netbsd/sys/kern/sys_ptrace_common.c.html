216static int
217ptrace_listener_cb(kauth_cred_t cred, kauth_action_t action, void *cookie,
218 void *arg0, void *arg1, void *arg2, void *arg3)
219{
220 struct proc *p;
221 int result;
222#ifdef PT_SETDBREGS
223 extern int user_set_dbregs;
224#endif
225
226 result = KAUTH_RESULT_DEFER;
227 p = arg0;
228
229#if 0
230 mutex_enter(&ptrace_mtx);
231 ptrace_cbref++;
232 mutex_exit(&ptrace_mtx);
233#endif
234 if (action != KAUTH_PROCESS_PTRACE)
235 goto out;
236
237 switch ((u_long)arg1) {
238#ifdef PT_SETDBREGS
239 case_PT_SETDBREGS
240 if (kauth_cred_getuid(cred) != 0 && user_set_dbregs == 0) {
241 result = KAUTH_RESULT_DENY;
242 break;
243 }
244#endif
245 /* FALLTHROUGH */
246 case PT_TRACE_ME:
247 case PT_ATTACH:
248 case PT_WRITE_I:
249 case PT_WRITE_D:
250 case PT_READ_I:
251 case PT_READ_D:
252 case PT_IO:
253 case_PT_GETREGS
254 case_PT_SETREGS
255 case_PT_GETFPREGS
256 case_PT_SETFPREGS
257 case_PT_GETDBREGS
258 case PT_SET_EVENT_MASK:
259 case PT_GET_EVENT_MASK:
260 case PT_GET_PROCESS_STATE:
261 case PT_SET_SIGINFO:
262 case PT_GET_SIGINFO:
263#ifdef __HAVE_PTRACE_MACHDEP
264 PTRACE_MACHDEP_REQUEST_CASES
265#endif
266 if (kauth_cred_getuid(cred) != kauth_cred_getuid(p->p_cred) ||
267 ISSET(p->p_flag, PK_SUGID)) {
268 break;
269 }
270
271 result = KAUTH_RESULT_ALLOW;
272
273 break;
274
275#ifdef PT_STEP
276 case PT_STEP:
277 case PT_SETSTEP:
278 case PT_CLEARSTEP:
279#endif
280 case PT_CONTINUE:
281 case PT_KILL:
282 case PT_DETACH:
283 case PT_LWPINFO:
284 case PT_SYSCALL:
285 case PT_SYSCALLEMU:
286 case PT_DUMPCORE:
287 case PT_RESUME:
288 case PT_SUSPEND:
289 result = KAUTH_RESULT_ALLOW;
290 break;
291
292 default:
293 break;
294 }
295
296 out:
297#if 0
298 mutex_enter(&ptrace_mtx);
299 if (--ptrace_cbref == 0)
300 cv_broadcast(&ptrace_cv);
301 mutex_exit(&ptrace_mtx);
302#endif
303
304 return result;
305}

341static struct proc *
342ptrace_find(struct lwp *l, int req, pid_t pid)
343{
344 struct proc *t;
345
346 /* "A foolish consistency..." XXX */
347 if (req == PT_TRACE_ME) {
348 t = l->l_proc;
349 mutex_enter(t->p_lock);
350 return t;
351 }
352
353 /* Find the process we're supposed to be operating on. */
354 t = proc_find(pid);
355 if (t == NULL)
356 return NULL;
357
358 /* XXX-elad */
359 mutex_enter(t->p_lock);
360 int error = kauth_authorize_process(l->l_cred, KAUTH_PROCESS_CANSEE,
361 t, KAUTH_ARG(KAUTH_REQ_PROCESS_CANSEE_ENTRY), NULL, NULL);
362 if (error) {
363 mutex_exit(t->p_lock);
364 return NULL;
365 }
366 return t;
367}

369static int
370ptrace_allowed(struct lwp *l, int req, struct proc *t, struct proc *p,
371 bool *locked)
372{
373 *locked = false;
374
375 /*
376 * Grab a reference on the process to prevent it from execing or
377 * exiting.
378 */
379 if (!rw_tryenter(&t->p_reflock, RW_READER))
380 return EBUSY;
381
382 *locked = true;
383
384 /* Make sure we can operate on it. */
385 switch (req) {
386 case PT_TRACE_ME:
387 /*
388 * You can't say to the parent of a process to start tracing if:
389 * (1) the parent is initproc,
390 */
391 if (p->p_pptr == initproc)
392 return EPERM;
393
394 /*
395 * (2) the process is initproc, or
396 */
397 if (p == initproc)
398 return EPERM;
399
400 /*
401 * (3) the child is already traced.
402 */
403 if (ISSET(p->p_slflag, PSL_TRACED))
404 return EBUSY;
405
406 return 0;
407
408 case PT_ATTACH:
409 /*
410 * You can't attach to a process if:
411 * (1) it's the process that's doing the attaching,
412 */
413 if (t == p)
414 return EINVAL;
415
416 /*
417 * (2) it's a system process,
418 */
419 if (t->p_flag & PK_SYSTEM)
420 return EPERM;
421
422 /*
423 * (3) the tracer is initproc,
424 */
425 if (p == initproc)
426 return EPERM;
427
428 /*
429 * (4) it's already being traced,
430 */
431 if (ISSET(t->p_slflag, PSL_TRACED))
432 return EBUSY;
433
434 /*
435 * (5) it's a vfork(2)ed parent of the current process, or
436 */
437 if (ISSET(p->p_lflag, PL_PPWAIT) && p->p_pptr == t)
438 return EPERM;
439
440 /*
441 * (6) the tracer is chrooted, and its root directory is
442 * not at or above the root directory of the tracee
443 */
444 mutex_exit(t->p_lock); /* XXXSMP */
445 int tmp = proc_isunder(t, l);
446 mutex_enter(t->p_lock); /* XXXSMP */
447 if (!tmp)
448 return EPERM;
449 return 0;
450
451 case PT_READ_I:
452 case PT_READ_D:
453 case PT_WRITE_I:
454 case PT_WRITE_D:
455 case PT_IO:
456 case PT_SET_SIGINFO:
457 case PT_GET_SIGINFO:
458 case_PT_GETREGS
459 case_PT_SETREGS
460 case_PT_GETFPREGS
461 case_PT_SETFPREGS
462 case_PT_GETDBREGS
463 case_PT_SETDBREGS
464#ifdef __HAVE_PTRACE_MACHDEP
465 PTRACE_MACHDEP_REQUEST_CASES
466#endif
467 /*
468 * You can't read/write the memory or registers of a process
469 * if the tracer is chrooted, and its root directory is not at
470 * or above the root directory of the tracee.
471 */
472 mutex_exit(t->p_lock); /* XXXSMP */
473 tmp = proc_isunder(t, l);
474 mutex_enter(t->p_lock); /* XXXSMP */
475 if (!tmp)
476 return EPERM;
477 /*FALLTHROUGH*/
478
479 case PT_CONTINUE:
480 case PT_KILL:
481 case PT_DETACH:
482 case PT_LWPINFO:
483 case PT_SYSCALL:
484 case PT_SYSCALLEMU:
485 case PT_DUMPCORE:
486#ifdef PT_STEP
487 case PT_STEP:
488 case PT_SETSTEP:
489 case PT_CLEARSTEP:
490#endif
491 case PT_SET_EVENT_MASK:
492 case PT_GET_EVENT_MASK:
493 case PT_GET_PROCESS_STATE:
494 case PT_RESUME:
495 case PT_SUSPEND:
496 /*
497 * You can't do what you want to the process if:
498 * (1) It's not being traced at all,
499 */
500 if (!ISSET(t->p_slflag, PSL_TRACED))
501 return EPERM;
502
503 /*
504 * (2) it's not being traced by _you_, or
505 */
506 if (t->p_pptr != p) {
507 DPRINTF(("parent %d != %d\n", t->p_pptr->p_pid,
508 p->p_pid));
509 return EBUSY;
510 }
511
512 /*
513 * (3) it's not currently stopped.
514 */
515 if (t->p_stat != SSTOP || !t->p_waited /* XXXSMP */) {
516 DPRINTF(("stat %d flag %d\n", t->p_stat,
517 !t->p_waited));
518 return EBUSY;
519 }
520 return 0;
521
522 default: /* It was not a legal request. */
523 return EINVAL;
524 }
525}

549static int
550ptrace_update_lwp(struct proc *t, struct lwp **lt, lwpid_t lid)
551{
552 if (lid == 0 || lid == (*lt)->l_lid || t->p_nlwps == 1)
553 return 0;
554
555 lwp_delref(*lt);
556
557 mutex_enter(t->p_lock);
558 *lt = lwp_find(t, lid);
559 if (*lt == NULL) {
560 mutex_exit(t->p_lock);
561 return ESRCH;
562 }
563
564 if ((*lt)->l_flag & LW_SYSTEM) {
565 *lt = NULL;
566 return EINVAL;
567 }
568
569 lwp_addref(*lt);
570 mutex_exit(t->p_lock);
571
572 return 0;
573}

575static int
576ptrace_get_siginfo(struct proc *t, struct ptrace_methods *ptm, void *addr,
577 size_t data)
578{
579 struct ptrace_siginfo psi;
580
581 memset(&psi, 0, sizeof(psi));
582 psi.psi_siginfo._info = t->p_sigctx.ps_info;
583 psi.psi_lwpid = t->p_sigctx.ps_lwp;
584 DPRINTF(("%s: lwp=%d signal=%d\n", __func__, psi.psi_lwpid,
585 psi.psi_siginfo.si_signo));
586
587 return ptm->ptm_copyout_siginfo(&psi, addr, data);
588}
589
590static int
591ptrace_set_siginfo(struct proc *t, struct lwp **lt, struct ptrace_methods *ptm,
592 void *addr, size_t data)
593{
594 struct ptrace_siginfo psi;
595
596 int error = ptm->ptm_copyin_siginfo(&psi, addr, data);
597 if (error)
598 return error;
599
600 /* Check that the data is a valid signal number or zero. */
601 if (psi.psi_siginfo.si_signo < 0 || psi.psi_siginfo.si_signo >= NSIG)
602 return EINVAL;
603
604 if ((error = ptrace_update_lwp(t, lt, psi.psi_lwpid)) != 0)
605 return error;
606
607 t->p_sigctx.ps_faked = true;
608 t->p_sigctx.ps_info = psi.psi_siginfo._info;
609 t->p_sigctx.ps_lwp = psi.psi_lwpid;
610 DPRINTF(("%s: lwp=%d signal=%d\n", __func__, psi.psi_lwpid,
611 psi.psi_siginfo.si_signo));
612 return 0;
613}

615static int
616ptrace_get_event_mask(struct proc *t, void *addr, size_t data)
617{
618 struct ptrace_event pe;
619
620 if (data != sizeof(pe)) {
621 DPRINTF(("%s: %zu != %zu\n", __func__, data, sizeof(pe)));
622 return EINVAL;
623 }
624 memset(&pe, 0, sizeof(pe));
625 pe.pe_set_event = ISSET(t->p_slflag, PSL_TRACEFORK) ?
626 PTRACE_FORK : 0;
627 pe.pe_set_event |= ISSET(t->p_slflag, PSL_TRACEVFORK) ?
628 PTRACE_VFORK : 0;
629 pe.pe_set_event |= ISSET(t->p_slflag, PSL_TRACEVFORK_DONE) ?
630 PTRACE_VFORK_DONE : 0;
631 pe.pe_set_event |= ISSET(t->p_slflag, PSL_TRACELWP_CREATE) ?
632 PTRACE_LWP_CREATE : 0;
633 pe.pe_set_event |= ISSET(t->p_slflag, PSL_TRACELWP_EXIT) ?
634 PTRACE_LWP_EXIT : 0;
635 pe.pe_set_event |= ISSET(t->p_slflag, PSL_TRACEPOSIX_SPAWN) ?
636 PTRACE_POSIX_SPAWN : 0;
637 DPRINTF(("%s: lwp=%d event=%#x\n", __func__,
638 t->p_sigctx.ps_lwp, pe.pe_set_event));
639 return copyout(&pe, addr, sizeof(pe));
640}

642static int
643ptrace_set_event_mask(struct proc *t, void *addr, size_t data)
644{
645 struct ptrace_event pe;
646 int error;
647
648 if (data != sizeof(pe)) {
649 DPRINTF(("%s: %zu != %zu\n", __func__, data, sizeof(pe)));
650 return EINVAL;
651 }
652 if ((error = copyin(addr, &pe, sizeof(pe))) != 0)
653 return error;
654
655 DPRINTF(("%s: lwp=%d event=%#x\n", __func__,
656 t->p_sigctx.ps_lwp, pe.pe_set_event));
657 if (pe.pe_set_event & PTRACE_FORK)
658 SET(t->p_slflag, PSL_TRACEFORK);
659 else
660 CLR(t->p_slflag, PSL_TRACEFORK);
661
662 if (pe.pe_set_event & PTRACE_VFORK)
663 SET(t->p_slflag, PSL_TRACEVFORK);
664 else
665 CLR(t->p_slflag, PSL_TRACEVFORK);
666
667 if (pe.pe_set_event & PTRACE_VFORK_DONE)
668 SET(t->p_slflag, PSL_TRACEVFORK_DONE);
669 else
670 CLR(t->p_slflag, PSL_TRACEVFORK_DONE);
671
672 if (pe.pe_set_event & PTRACE_LWP_CREATE)
673 SET(t->p_slflag, PSL_TRACELWP_CREATE);
674 else
675 CLR(t->p_slflag, PSL_TRACELWP_CREATE);
676
677 if (pe.pe_set_event & PTRACE_LWP_EXIT)
678 SET(t->p_slflag, PSL_TRACELWP_EXIT);
679 else
680 CLR(t->p_slflag, PSL_TRACELWP_EXIT);
681
682 if (pe.pe_set_event & PTRACE_POSIX_SPAWN)
683 SET(t->p_slflag, PSL_TRACEPOSIX_SPAWN);
684 else
685 CLR(t->p_slflag, PSL_TRACEPOSIX_SPAWN);
686
687 return 0;
688}

690static int
691ptrace_get_process_state(struct proc *t, void *addr, size_t data)
692{
693 struct ptrace_state ps;
694
695 if (data != sizeof(ps)) {
696 DPRINTF(("%s: %zu != %zu\n", __func__, data, sizeof(ps)));
697 return EINVAL;
698 }
699 memset(&ps, 0, sizeof(ps));
700
701 if (t->p_fpid) {
702 ps.pe_report_event = PTRACE_FORK;
703 ps.pe_other_pid = t->p_fpid;
704 } else if (t->p_vfpid) {
705 ps.pe_report_event = PTRACE_VFORK;
706 ps.pe_other_pid = t->p_vfpid;
707 } else if (t->p_vfpid_done) {
708 ps.pe_report_event = PTRACE_VFORK_DONE;
709 ps.pe_other_pid = t->p_vfpid_done;
710 } else if (t->p_lwp_created) {
711 ps.pe_report_event = PTRACE_LWP_CREATE;
712 ps.pe_lwp = t->p_lwp_created;
713 } else if (t->p_lwp_exited) {
714 ps.pe_report_event = PTRACE_LWP_EXIT;
715 ps.pe_lwp = t->p_lwp_exited;
716 } else if (t->p_pspid) {
717 ps.pe_report_event = PTRACE_POSIX_SPAWN;
718 ps.pe_other_pid = t->p_pspid;
719 }
720 DPRINTF(("%s: lwp=%d event=%#x pid=%d lwp=%d\n", __func__,
721 t->p_sigctx.ps_lwp, ps.pe_report_event,
722 ps.pe_other_pid, ps.pe_lwp));
723 return copyout(&ps, addr, sizeof(ps));
724}

726static int
727ptrace_lwpinfo(struct proc *t, struct lwp **lt, void *addr, size_t data)
728{
729 struct ptrace_lwpinfo pl;
730
731 if (data != sizeof(pl)) {
732 DPRINTF(("%s: %zu != %zu\n", __func__, data, sizeof(pl)));
733 return EINVAL;
734 }
735 int error = copyin(addr, &pl, sizeof(pl));
736 if (error)
737 return error;
738
739 lwpid_t tmp = pl.pl_lwpid;
740 lwp_delref(*lt);
741 mutex_enter(t->p_lock);
742 if (tmp == 0)
743 *lt = lwp_find_first(t);
744 else {
745 *lt = lwp_find(t, tmp);
746 if (*lt == NULL) {
747 mutex_exit(t->p_lock);
748 return ESRCH;
749 }
750 *lt = LIST_NEXT(*lt, l_sibling);
751 }
752
753 while (*lt != NULL && !lwp_alive(*lt))
754 *lt = LIST_NEXT(*lt, l_sibling);
755
756 pl.pl_lwpid = 0;
757 pl.pl_event = 0;
758 if (*lt) {
759 lwp_addref(*lt);
760 pl.pl_lwpid = (*lt)->l_lid;
761
762 if ((*lt)->l_flag & LW_WSUSPEND)
763 pl.pl_event = PL_EVENT_SUSPENDED;
764 /*
765 * If we match the lwp, or it was sent to every lwp,
766 * we set PL_EVENT_SIGNAL.
767 * XXX: ps_lwp == 0 means everyone and noone, so
768 * check ps_signo too.
769 */
770 else if ((*lt)->l_lid == t->p_sigctx.ps_lwp
771 || (t->p_sigctx.ps_lwp == 0 &&
772 t->p_sigctx.ps_info._signo)) {
773 DPRINTF(("%s: lwp=%d siglwp=%d signo %d\n", __func__,
774 pl.pl_lwpid, t->p_sigctx.ps_lwp,
775 t->p_sigctx.ps_info._signo));
776 pl.pl_event = PL_EVENT_SIGNAL;
777 }
778 }
779 mutex_exit(t->p_lock);
780 DPRINTF(("%s: lwp=%d event=%#x\n", __func__,
781 pl.pl_lwpid, pl.pl_event));
782
783 return copyout(&pl, addr, sizeof(pl));
784}

786static int
787ptrace_startstop(struct proc *t, struct lwp **lt, int rq, void *addr,
788 size_t data)
789{
790 int error;
791
792 if ((error = ptrace_update_lwp(t, lt, data)) != 0)
793 return error;
794
795 DPRINTF(("%s: lwp=%d request=%d\n", __func__, (*lt)->l_lid, rq));
796 lwp_lock(*lt);
797 if (rq == PT_SUSPEND)
798 (*lt)->l_flag |= LW_WSUSPEND;
799 else
800 (*lt)->l_flag &= ~LW_WSUSPEND;
801 lwp_unlock(*lt);
802 return 0;
803}

823static int
824ptrace_regs(struct lwp *l, struct lwp **lt, int rq, struct ptrace_methods *ptm,
825 void *addr, size_t data)
826{
827 int error;
828 struct proc *t = (*lt)->l_proc;
829 struct vmspace *vm;
830
831 if ((error = ptrace_update_lwp(t, lt, data)) != 0)
832 return error;
833
834 int dir = ptrace_uio_dir(rq);
835 size_t size;
836 int (*func)(struct lwp *, struct lwp *, struct uio *);
837
838 DPRINTF(("%s: lwp=%d request=%d\n", __func__, l->l_lid, rq));
839
840 switch (rq) {
841#if defined(PT_SETREGS) || defined(PT_GETREGS)
842 case_PT_GETREGS
843 case_PT_SETREGS
844 if (!process_validregs(*lt))
845 return EINVAL;
846 size = PROC_REGSZ(t);
847 func = ptm->ptm_doregs;
848 break;
849#endif
850#if defined(PT_SETFPREGS) || defined(PT_GETFPREGS)
851 case_PT_GETFPREGS
852 case_PT_SETFPREGS
853 if (!process_validfpregs(*lt))
854 return EINVAL;
855 size = PROC_FPREGSZ(t);
856 func = ptm->ptm_dofpregs;
857 break;
858#endif
859#if defined(PT_SETDBREGS) || defined(PT_GETDBREGS)
860 case_PT_GETDBREGS
861 case_PT_SETDBREGS
862 if (!process_validdbregs(*lt))
863 return EINVAL;
864 size = PROC_DBREGSZ(t);
865 func = ptm->ptm_dodbregs;
866 break;
867#endif
868 default:
869 return EINVAL;
870 }
871
872 error = proc_vmspace_getref(l->l_proc, &vm);
873 if (error)
874 return error;
875
876 struct uio uio;
877 struct iovec iov;
878
879 iov.iov_base = addr;
880 iov.iov_len = size;
881 uio.uio_iov = &iov;
882 uio.uio_iovcnt = 1;
883 uio.uio_offset = 0;
884 uio.uio_resid = iov.iov_len;
885 uio.uio_rw = dir;
886 uio.uio_vmspace = vm;
887
888 error = (*func)(l, *lt, &uio);
889 uvmspace_free(vm);
890 return error;
891}

894static int
895ptrace_sendsig(struct proc *t, struct lwp *lt, int signo, int resume_all)
896{
897 ksiginfo_t ksi;
898
899 t->p_fpid = 0;
900 t->p_vfpid = 0;
901 t->p_vfpid_done = 0;
902 t->p_lwp_created = 0;
903 t->p_lwp_exited = 0;
904 t->p_pspid = 0;
905
906 /* Finally, deliver the requested signal (or none). */
907 if (t->p_stat == SSTOP) {
908 /*
909 * Unstop the process. If it needs to take a
910 * signal, make all efforts to ensure that at
911 * an LWP runs to see it.
912 */
913 t->p_xsig = signo;
914
915 /*
916 * signo > 0 check prevents a potential panic, as
917 * sigismember(&...,0) is invalid check and signo
918 * can be equal to 0 as a special case of no-signal.
919 */
920 if (signo > 0 && sigismember(&stopsigmask, signo)) {
921 t->p_waited = 0;
922 child_psignal(t, 0);
923 } else if (resume_all)
924 proc_unstop(t);
925 else
926 lwp_unstop(lt);
927 return 0;
928 }
929
930 KSI_INIT_EMPTY(&ksi);
931 if (t->p_sigctx.ps_faked) {
932 if (signo != t->p_sigctx.ps_info._signo)
933 return EINVAL;
934 t->p_sigctx.ps_faked = false;
935 ksi.ksi_info = t->p_sigctx.ps_info;
936 ksi.ksi_lid = t->p_sigctx.ps_lwp;
937 } else if (signo == 0) {
938 return 0;
939 } else {
940 ksi.ksi_signo = signo;
941 }
942 DPRINTF(("%s: pid=%d.%d signal=%d resume_all=%d\n", __func__, t->p_pid,
943 t->p_sigctx.ps_lwp, signo, resume_all));
944
945 kpsignal2(t, &ksi);
946 return 0;
947}

949static int
950ptrace_dumpcore(struct lwp *lt, char *path, size_t len)
951{
952 int error;
953 if (path != NULL) {
954
955 if (len >= MAXPATHLEN)
956 return EINVAL;
957
958 char *src = path;
959 path = kmem_alloc(len + 1, KM_SLEEP);
960 error = copyin(src, path, len);
961 if (error)
962 goto out;
963 path[len] = '\0';
964 }
965 DPRINTF(("%s: lwp=%d\n", __func__, lt->l_lid));
966 error = (*coredump_vec)(lt, path);
967out:
968 if (path)
969 kmem_free(path, len + 1);
970 return error;
971}

973static int
974ptrace_doio(struct lwp *l, struct proc *t, struct lwp *lt,
975 struct ptrace_io_desc *piod, void *addr, bool sysspace)
976{
977 struct uio uio;
978 struct iovec iov;
979 int error, tmp;
980
981 error = 0;
982 iov.iov_base = piod->piod_addr;
983 iov.iov_len = piod->piod_len;
984 uio.uio_iov = &iov;
985 uio.uio_iovcnt = 1;
986 uio.uio_offset = (off_t)(unsigned long)piod->piod_offs;
987 uio.uio_resid = piod->piod_len;
988
989 DPRINTF(("%s: lwp=%d request=%d\n", __func__, l->l_lid, piod->piod_op));
990
991 switch (piod->piod_op) {
992 case PIOD_READ_D:
993 case PIOD_READ_I:
994 uio.uio_rw = UIO_READ;
995 break;
996 case PIOD_WRITE_D:
997 case PIOD_WRITE_I:
998 /*
999 * Can't write to a RAS
1000 */
1001 if (ras_lookup(t, addr) != (void *)-1) {
1002 return EACCES;
1003 }
1004 uio.uio_rw = UIO_WRITE;
1005 break;
1006 case PIOD_READ_AUXV:
1007 uio.uio_rw = UIO_READ;
1008 tmp = t->p_execsw->es_arglen;
1009 if (uio.uio_offset > tmp)
1010 return EIO;
1011 if (uio.uio_resid > tmp - uio.uio_offset)
1012 uio.uio_resid = tmp - uio.uio_offset;
1013 piod->piod_len = iov.iov_len = uio.uio_resid;
1014 error = process_auxv_offset(t, &uio);
1015 break;
1016 default:
1017 error = EINVAL;
1018 break;
1019 }
1020
1021 if (error)
1022 return error;
1023
1024 if (sysspace) {
1025 uio.uio_vmspace = vmspace_kernel();
1026 } else {
1027 error = proc_vmspace_getref(l->l_proc, &uio.uio_vmspace);
1028 if (error)
1029 return error;
1030 }
1031
1032 error = process_domem(l, lt, &uio);
1033 if (!sysspace)
1034 uvmspace_free(uio.uio_vmspace);
1035 if (error)
1036 return error;
1037 piod->piod_len -= uio.uio_resid;
1038 return 0;
1039}
1040
1041int
1042do_ptrace(struct ptrace_methods *ptm, struct lwp *l, int req, pid_t pid,
1043 void *addr, int data, register_t *retval)
1044{
1045 struct proc *p = l->l_proc;
1046 struct lwp *lt = NULL;
1047 struct lwp *lt2;
1048 struct proc *t; /* target process */
1049 struct ptrace_io_desc piod;
1050 int error, write, tmp, pheld;
1051 int signo = 0;
1052 int resume_all;
1053 bool locked;
1054 error = 0;
1055
1056 /*
1057 * If attaching or detaching, we need to get a write hold on the
1058 * proclist lock so that we can re-parent the target process.
1059 */
1060 mutex_enter(proc_lock);
1061
1062 t = ptrace_find(l, req, pid);
1063 if (t == NULL) {
1064 mutex_exit(proc_lock);
1065 return ESRCH;
1066 }
1067
1068 pheld = 1;
1069 if ((error = ptrace_allowed(l, req, t, p, &locked)) != 0)
1070 goto out;
1071
1072 if ((error = kauth_authorize_process(l->l_cred,
1073 KAUTH_PROCESS_PTRACE, t, KAUTH_ARG(req), NULL, NULL)) != 0)
1074 goto out;
1075
1076 if ((lt = lwp_find_first(t)) == NULL) {
1077 error = ESRCH;
1078 goto out;
1079 }
1080
1081 /* Do single-step fixup if needed. */
1082 FIX_SSTEP(t);
1083 KASSERT(lt != NULL);
1084 lwp_addref(lt);
1085
1086 /*
1087 * Which locks do we need held? XXX Ugly.
1088 */
1089 if ((pheld = ptrace_needs_hold(req)) == 0) {
1090 mutex_exit(t->p_lock);
1091 mutex_exit(proc_lock);
1092 }
1093
1094 /* Now do the operation. */
1095 write = 0;
1096 *retval = 0;
1097 tmp = 0;
1098 resume_all = 1;
1099
1100 switch (req) {
1101 case PT_TRACE_ME:
1102 /* Just set the trace flag. */
1103 SET(t->p_slflag, PSL_TRACED);
1104 t->p_opptr = t->p_pptr;
1105 break;
1106
1107 /*
1108 * The I and D separate address space has been inherited from PDP-11.
1109 * The 16-bit UNIX started with a single address space per program,
1110 * but was extended to two 16-bit (2 x 64kb) address spaces.
1111 *
1112 * We no longer maintain this feature in maintained architectures, but
1113 * we keep the API for backward compatiblity. Currently the I and D
1114 * operations are exactly the same and not distinguished in debuggers.
1115 */
1116 case PT_WRITE_I:
1117 case PT_WRITE_D:
1118 write = 1;
1119 tmp = data;
1120 /* FALLTHROUGH */
1121 case PT_READ_I:
1122 case PT_READ_D:
1123 piod.piod_addr = &tmp;
1124 piod.piod_len = sizeof(tmp);
1125 piod.piod_offs = addr;
1126 piod.piod_op = write ? PIOD_WRITE_D : PIOD_READ_D;
1127 if ((error = ptrace_doio(l, t, lt, &piod, addr, true)) != 0)
1128 break;
1129 /*
1130 * For legacy reasons we treat here two results as success:
1131 * - incomplete transfer piod.piod_len < sizeof(tmp)
1132 * - no transfer piod.piod_len == 0
1133 *
1134 * This means that there is no way to determine whether
1135 * transfer operation was performed in PT_WRITE and PT_READ
1136 * calls.
1137 */
1138 if (!write)
1139 *retval = tmp;
1140 break;
1141
1142 case PT_IO:
1143 if ((error = ptm->ptm_copyin_piod(&piod, addr, data)) != 0)
1144 break;
1145 if (piod.piod_len < 1) {
1146 error = EINVAL;
1147 break;
1148 }
1149 if ((error = ptrace_doio(l, t, lt, &piod, addr, false)) != 0)
1150 break;
1151 /*
1152 * For legacy reasons we treat here two results as success:
1153 * - incomplete transfer piod.piod_len < sizeof(tmp)
1154 * - no transfer piod.piod_len == 0
1155 */
1156 error = ptm->ptm_copyout_piod(&piod, addr, data);
1157 break;
1158
1159 case PT_DUMPCORE:
1160 error = ptrace_dumpcore(lt, addr, data);
1161 break;
1162
1163#ifdef PT_STEP
1164 case PT_STEP:
1165 /*
1166 * From the 4.4BSD PRM:
1167 * "Execution continues as in request PT_CONTINUE; however
1168 * as soon as possible after execution of at least one
1169 * instruction, execution stops again. [ ... ]"
1170 */
1171#endif
1172 case PT_CONTINUE:
1173 case PT_SYSCALL:
1174 case PT_DETACH:
1175 if (req == PT_SYSCALL) {
1176 if (!ISSET(t->p_slflag, PSL_SYSCALL)) {
1177 SET(t->p_slflag, PSL_SYSCALL);
1178#ifdef __HAVE_SYSCALL_INTERN
1179 (*t->p_emul->e_syscall_intern)(t);
1180#endif
1181 }
1182 } else {
1183 if (ISSET(t->p_slflag, PSL_SYSCALL)) {
1184 CLR(t->p_slflag, PSL_SYSCALL);
1185#ifdef __HAVE_SYSCALL_INTERN
1186 (*t->p_emul->e_syscall_intern)(t);
1187#endif
1188 }
1189 }
1190 t->p_trace_enabled = trace_is_enabled(t);
1191
1192 /*
1193 * Pick up the LWPID, if supplied. There are two cases:
1194 * data < 0 : step or continue single thread, lwp = -data
1195 * data > 0 in PT_STEP : step this thread, continue others
1196 * For operations other than PT_STEP, data > 0 means
1197 * data is the signo to deliver to the process.
1198 */
1199 tmp = data;
1200 if (tmp >= 0) {
1201#ifdef PT_STEP
1202 if (req == PT_STEP)
1203 signo = 0;
1204 else
1205#endif
1206 {
1207 signo = tmp;
1208 tmp = 0; /* don't search for LWP */
1209 }
1210 } else
1211 tmp = -tmp;
1212
1213 if (tmp > 0) {
1214 if (req == PT_DETACH) {
1215 error = EINVAL;
1216 break;
1217 }
1218 lwp_delref2 (lt);
1219 lt = lwp_find(t, tmp);
1220 if (lt == NULL) {
1221 error = ESRCH;
1222 break;
1223 }
1224 lwp_addref(lt);
1225 resume_all = 0;
1226 signo = 0;
1227 }
1228
1229 /*
1230 * From the 4.4BSD PRM:
1231 * "The data argument is taken as a signal number and the
1232 * child's execution continues at location addr as if it
1233 * incurred that signal. Normally the signal number will
1234 * be either 0 to indicate that the signal that caused the
1235 * stop should be ignored, or that value fetched out of
1236 * the process's image indicating which signal caused
1237 * the stop. If addr is (int *)1 then execution continues
1238 * from where it stopped."
1239 */
1240
1241 /* Check that the data is a valid signal number or zero. */
1242 if (signo < 0 || signo >= NSIG) {
1243 error = EINVAL;
1244 break;
1245 }
1246
1247 /* Prevent process deadlock */
1248 if (resume_all) {
1249#ifdef PT_STEP
1250 if (req == PT_STEP) {
1251 if (lt->l_flag & LW_WSUSPEND) {
1252 error = EDEADLK;
1253 break;
1254 }
1255 } else
1256#endif
1257 {
1258 error = EDEADLK;
1259 LIST_FOREACH(lt2, &t->p_lwps, l_sibling) {
1260 if ((lt2->l_flag & LW_WSUSPEND) == 0) {
1261 error = 0;
1262 break;
1263 }
1264 }
1265 if (error != 0)
1266 break;
1267 }
1268 } else {
1269 if (lt->l_flag & LW_WSUSPEND) {
1270 error = EDEADLK;
1271 break;
1272 }
1273 }
1274
1275 /*
1276 * Reject setting program cunter to 0x0 if VA0 is disabled.
1277 *
1278 * Not all kernels implement this feature to set Program
1279 * Counter in one go in PT_CONTINUE and similar operations.
1280 * This causes portability issues as passing address 0x0
1281 * on these kernels is no-operation, but can cause failure
1282 * in most cases on NetBSD.
1283 */
1284 if (user_va0_disable && addr == 0) {
1285 error = EINVAL;
1286 break;
1287 }
1288
1289 /* If the address parameter is not (int *)1, set the pc. */
1290 if ((int *)addr != (int *)1) {
1291 error = process_set_pc(lt, addr);
1292 if (error != 0)
1293 break;
1294 }
1295#ifdef PT_STEP
1296 /*
1297 * Arrange for a single-step, if that's requested and possible.
1298 * More precisely, set the single step status as requested for
1299 * the requested thread, and clear it for other threads.
1300 */
1301 LIST_FOREACH(lt2, &t->p_lwps, l_sibling) {
1302 if (ISSET(lt2->l_pflag, LP_SINGLESTEP)) {
1303 lwp_lock(lt2);
1304 process_sstep(lt2, 1);
1305 lwp_unlock(lt2);
1306 } else if (lt != lt2) {
1307 lwp_lock(lt2);
1308 process_sstep(lt2, 0);
1309 lwp_unlock(lt2);
1310 }
1311 }
1312 error = process_sstep(lt,
1313 ISSET(lt->l_pflag, LP_SINGLESTEP) || req == PT_STEP);
1314 if (error)
1315 break;
1316#endif
1317 if (req == PT_DETACH) {
1318 CLR(t->p_slflag, PSL_TRACED|PSL_SYSCALL);
1319
1320 /* give process back to original parent or init */
1321 if (t->p_opptr != t->p_pptr) {
1322 struct proc *pp = t->p_opptr;
1323 proc_reparent(t, pp ? pp : initproc);
1324 }
1325
1326 /* not being traced any more */
1327 t->p_opptr = NULL;
1328
1329 /* clear single step */
1330 LIST_FOREACH(lt2, &t->p_lwps, l_sibling) {
1331 CLR(lt2->l_pflag, LP_SINGLESTEP);
1332 }
1333 CLR(lt->l_pflag, LP_SINGLESTEP);
1334 }
1335 sendsig:
1336 error = ptrace_sendsig(t, lt, signo, resume_all);
1337 break;
1338
1339 case PT_SYSCALLEMU:
1340 if (!ISSET(t->p_slflag, PSL_SYSCALL) || t->p_stat != SSTOP) {
1341 error = EINVAL;
1342 break;
1343 }
1344 SET(t->p_slflag, PSL_SYSCALLEMU);
1345 break;
1346
1347#ifdef PT_STEP
1348 case PT_SETSTEP:
1349 write = 1;
1350
1351 /* FALLTHROUGH */
1352 case PT_CLEARSTEP:
1353 /* write = 0 done above. */
1354 if ((error = ptrace_update_lwp(t, &lt, data)) != 0)
1355 break;
1356
1357 if (write)
1358 SET(lt->l_pflag, LP_SINGLESTEP);
1359 else
1360 CLR(lt->l_pflag, LP_SINGLESTEP);
1361 break;
1362#endif
1363
1364 case PT_KILL:
1365 /* just send the process a KILL signal. */
1366 signo = SIGKILL;
1367 goto sendsig; /* in PT_CONTINUE, above. */
1368
1369 case PT_ATTACH:
1370 /*
1371 * Go ahead and set the trace flag.
1372 * Save the old parent (it's reset in
1373 * _DETACH, and also in kern_exit.c:wait4()
1374 * Reparent the process so that the tracing
1375 * proc gets to see all the action.
1376 * Stop the target.
1377 */
1378 proc_changeparent(t, p);
1379 signo = SIGSTOP;
1380 goto sendsig;
1381
1382 case PT_GET_EVENT_MASK:
1383 error = ptrace_get_event_mask(t, addr, data);
1384 break;
1385
1386 case PT_SET_EVENT_MASK:
1387 error = ptrace_set_event_mask(t, addr, data);
1388 break;
1389
1390 case PT_GET_PROCESS_STATE:
1391 error = ptrace_get_process_state(t, addr, data);
1392 break;
1393
1394 case PT_LWPINFO:
1395 error = ptrace_lwpinfo(t, &lt, addr, data);
1396 break;
1397
1398 case PT_SET_SIGINFO:
1399 error = ptrace_set_siginfo(t, &lt, ptm, addr, data);
1400 break;
1401
1402 case PT_GET_SIGINFO:
1403 error = ptrace_get_siginfo(t, ptm, addr, data);
1404 break;
1405
1406 case PT_RESUME:
1407 case PT_SUSPEND:
1408 error = ptrace_startstop(t, &lt, req, addr, data);
1409 break;
1410
1411#ifdef PT_REGISTERS
1412 case_PT_SETREGS
1413 case_PT_GETREGS
1414 case_PT_SETFPREGS
1415 case_PT_GETFPREGS
1416 case_PT_SETDBREGS
1417 case_PT_GETDBREGS
1418 error = ptrace_regs(l, &lt, req, ptm, addr, data);
1419 break;
1420#endif
1421
1422#ifdef __HAVE_PTRACE_MACHDEP
1423 PTRACE_MACHDEP_REQUEST_CASES
1424 error = ptrace_machdep_dorequest(l, lt, req, addr, data);
1425 break;
1426#endif
1427 }
1428
1429out:
1430 if (pheld) {
1431 mutex_exit(t->p_lock);
1432 mutex_exit(proc_lock);
1433 }
1434 if (lt != NULL)
1435 lwp_delref(lt);
1436 if (locked)
1437 rw_exit(&t->p_reflock);
1438
1439 return error;
1440}

1442typedef int (*regrfunc_t)(struct lwp *, void *, size_t *);
1443typedef int (*regwfunc_t)(struct lwp *, void *, size_t);

1445#ifdef PT_REGISTERS
1446static int
1447proc_regio(struct lwp *l, struct uio *uio, size_t ks, regrfunc_t r,
1448 regwfunc_t w)
1449{
1450 char buf[1024];
1451 int error;
1452 char *kv;
1453 size_t kl;
1454
1455 if (ks > sizeof(buf))
1456 return E2BIG;
1457
1458 if (uio->uio_offset < 0 || uio->uio_offset > (off_t)ks)
1459 return EINVAL;
1460
1461 kv = buf + uio->uio_offset;
1462 kl = ks - uio->uio_offset;
1463
1464 if (kl > uio->uio_resid)
1465 kl = uio->uio_resid;
1466
1467 error = (*r)(l, buf, &ks);
1468 if (error == 0)
1469 error = uiomove(kv, kl, uio);
1470 if (error == 0 && uio->uio_rw == UIO_WRITE) {
1471 if (l->l_stat != LSSTOP)
1472 error = EBUSY;
1473 else
1474 error = (*w)(l, buf, ks);
1475 }
1476
1477 uio->uio_offset = 0;
1478 return error;
1479}

1617static int
1618process_auxv_offset(struct proc *p, struct uio *uio)
1619{
1620 struct ps_strings pss;
1621 int error;
1622 off_t off = (off_t)p->p_psstrp;
1623
1624 if ((error = copyin_psstrings(p, &pss)) != 0)
1625 return error;
1626
1627 if (pss.ps_envstr == NULL)
1628 return EIO;
1629
1630 uio->uio_offset += (off_t)(vaddr_t)(pss.ps_envstr + pss.ps_nenvstr + 1);
1631#ifdef __MACHINE_STACK_GROWS_UP
1632 if (uio->uio_offset < off)
1633 return EIO;
1634#else
1635 if (uio->uio_offset > off)
1636 return EIO;
1637 if ((uio->uio_offset + uio->uio_resid) > off)
1638 uio->uio_resid = off - uio->uio_offset;
1639#endif
1640 return 0;
1641}
