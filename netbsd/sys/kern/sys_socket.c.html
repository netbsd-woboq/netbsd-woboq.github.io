98/* ARGSUSED */
99int
100soo_read(file_t *fp, off_t *offset, struct uio *uio, kauth_cred_t cred,
101 int flags)
102{
103 struct socket *so = fp->f_socket;
104 int error;
105
106 error = (*so->so_receive)(so, NULL, uio, NULL, NULL, NULL);
107
108 return error;
109}

111/* ARGSUSED */
112int
113soo_write(file_t *fp, off_t *offset, struct uio *uio, kauth_cred_t cred,
114 int flags)
115{
116 struct socket *so = fp->f_socket;
117 int error;
118
119 error = (*so->so_send)(so, NULL, uio, NULL, NULL, 0, curlwp);
120
121 return error;
122}

124int
125soo_ioctl(file_t *fp, u_long cmd, void *data)
126{
127 struct socket *so = fp->f_socket;
128 int error = 0;
129
130 switch (cmd) {
131
132 case FIONBIO:
133 solock(so);
134 if (*(int *)data)
135 so->so_state |= SS_NBIO;
136 else
137 so->so_state &= ~SS_NBIO;
138 sounlock(so);
139 break;
140
141 case FIOASYNC:
142 solock(so);
143 if (*(int *)data) {
144 so->so_state |= SS_ASYNC;
145 so->so_rcv.sb_flags |= SB_ASYNC;
146 so->so_snd.sb_flags |= SB_ASYNC;
147 } else {
148 so->so_state &= ~SS_ASYNC;
149 so->so_rcv.sb_flags &= ~SB_ASYNC;
150 so->so_snd.sb_flags &= ~SB_ASYNC;
151 }
152 sounlock(so);
153 break;
154
155 case FIONREAD:
156 *(int *)data = so->so_rcv.sb_cc;
157 break;
158
159 case FIONWRITE:
160 *(int *)data = so->so_snd.sb_cc;
161 break;
162
163 case FIONSPACE:
164 /*
165 * See the comment around sbspace()'s definition
166 * in sys/socketvar.h in face of counts about maximum
167 * to understand the following test. We detect overflow
168 * and return zero.
169 */
170 solock(so);
171 if ((so->so_snd.sb_hiwat < so->so_snd.sb_cc)
172 || (so->so_snd.sb_mbmax < so->so_snd.sb_mbcnt))
173 *(int *)data = 0;
174 else
175 *(int *)data = sbspace(&so->so_snd);
176 sounlock(so);
177 break;
178
179 case SIOCSPGRP:
180 case FIOSETOWN:
181 case TIOCSPGRP:
182 error = fsetown(&so->so_pgid, cmd, data);
183 break;
184
185 case SIOCGPGRP:
186 case FIOGETOWN:
187 case TIOCGPGRP:
188 error = fgetown(so->so_pgid, cmd, data);
189 break;
190
191 case SIOCATMARK:
192 *(int *)data = (so->so_state&SS_RCVATMARK) != 0;
193 break;
194
195 case SIOCPEELOFF:
196 solock(so);
197 error = do_sys_peeloff(so, data);
198 sounlock(so);
199 break;
200
201 default:
202 /*
203 * Interface/routing/protocol specific ioctls:
204 * interface and routing ioctls should have a
205 * different entry since a socket's unnecessary
206 */
207 if (IOCGROUP(cmd) == 'i')
208 /*
209 * KERNEL_LOCK will be held later if if_ioctl() of the
210 * interface isn't MP-safe.
211 */
212 error = ifioctl(so, cmd, data, curlwp);
213 else {
214 KERNEL_LOCK(1, NULL);
215 error = (*so->so_proto->pr_usrreqs->pr_ioctl)(so,
216 cmd, data, NULL);
217 KERNEL_UNLOCK_ONE(NULL);
218 }
219 break;
220 }
221
222
223 return error;
224}

226int
227soo_poll(file_t *fp, int events)
228{
229
230 return sopoll(fp->f_socket, events);
231}
232
233int
234soo_stat(file_t *fp, struct stat *ub)
235{
236 struct socket *so = fp->f_socket;
237 int error;
238
239 memset(ub, 0, sizeof(*ub));
240 ub->st_mode = S_IFSOCK;
241
242 solock(so);
243 error = (*so->so_proto->pr_usrreqs->pr_stat)(so, ub);
244 sounlock(so);
245
246 return error;
247}
248
249/* ARGSUSED */
250int
251soo_close(file_t *fp)
252{
253 int error = 0;
254
255 if (fp->f_socket)
256 error = soclose(fp->f_socket);
257 fp->f_socket = NULL;
258
259 return error;
260}
261
262void
263soo_restart(file_t *fp)
264{
265
266 sorestart(fp->f_socket);
267}
