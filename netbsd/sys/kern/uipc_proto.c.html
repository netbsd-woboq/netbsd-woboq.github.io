48/*
49 * Definitions of protocols supported in the UNIX domain.
50 */
51
52DOMAIN_DEFINE(unixdomain); /* forward define and add to link set */
53
54const struct protosw unixsw[] = {
55 {
56 .pr_type = SOCK_STREAM,
57 .pr_domain = &unixdomain,
58 .pr_flags = PR_CONNREQUIRED|PR_WANTRCVD|PR_RIGHTS|PR_LISTEN,
59 .pr_ctloutput = uipc_ctloutput,
60 .pr_usrreqs = &unp_usrreqs,
61 },
62 {
63 .pr_type = SOCK_DGRAM,
64 .pr_domain = &unixdomain,
65 .pr_flags = PR_ATOMIC|PR_ADDR|PR_RIGHTS,
66 .pr_ctloutput = uipc_ctloutput,
67 .pr_usrreqs = &unp_usrreqs,
68 },
69 {
70 .pr_type = SOCK_SEQPACKET,
71 .pr_domain = &unixdomain,
72 .pr_flags = PR_CONNREQUIRED|PR_WANTRCVD|PR_RIGHTS|PR_LISTEN|
73 PR_ATOMIC,
74 .pr_ctloutput = uipc_ctloutput,
75 .pr_usrreqs = &unp_usrreqs,
76 }
77};
78
79struct domain unixdomain = {
80 .dom_family = AF_LOCAL,
81 .dom_init = uipc_init,
82 .dom_name = "unix",
83 .dom_externalize = unp_externalize,
84 .dom_dispose = unp_dispose,
85 .dom_protosw = unixsw,
86 .dom_protoswNPROTOSW = &unixsw[__arraycount(unixsw)],
87};
