37int
38vfs_quotactl_stat(struct mount *mp, struct quotastat *info)
39{
40 struct quotactl_args args;
41
42 args.qc_op = QUOTACTL_STAT;
43 args.u.stat.qc_info = info;
44 return VFS_QUOTACTL(mp, &args);
45}
