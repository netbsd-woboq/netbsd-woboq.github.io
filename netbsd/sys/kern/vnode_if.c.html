55vop_pre(vnode_t *vp, struct mount **mp, bool *mpsafe, enum fst_op op)
56{
57 int error;
58
59 *mpsafe = (vp->v_vflag & VV_MPSAFE);
60
61 if (!*mpsafe) {
62 KERNEL_LOCK(1, curlwp);
63 }
64
65 if (op == FST_YES || op == FST_TRY) {
66 for (;;) {
67 *mp = vp->v_mount;
68 if (op == FST_TRY) {
69 error = fstrans_start_nowait(*mp);
70 if (error) {
71 if (!*mpsafe) {
72 KERNEL_UNLOCK_ONE(curlwp);
73 }
74 return error;
75 }
76 } else {
77 fstrans_start(*mp);
78 }
79 if (__predict_true(*mp == vp->v_mount))
80 break;
81 fstrans_done(*mp);
82 }
83 } else {
84 *mp = vp->v_mount;
85 }
86
87 return 0;
88}
