71static __inline uint16_t
72crc16_byte(uint16_t ccrc, const uint8_t b)
73{
74 return crc16_lookup[((ccrc) ^ (b)) & 0xffu] ^ ((ccrc) >> 8);
75}
76
77static __inline uint16_t __unused
78crc16(uint16_t crc, const uint8_t *data, size_t len)
79{
80 while (len--)
81 crc = crc16_byte(crc, *data++);
82 return crc;
83}
