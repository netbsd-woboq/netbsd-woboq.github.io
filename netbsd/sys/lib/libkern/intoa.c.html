50/* Similar to inet_ntoa() */
51char *
52intoa(u_int32_t addr)
53{
54 char *cp;
55 u_int byte;
56 int n;
57 static char buf[17]; /* strlen(".255.255.255.255") + 1 */
58
59 addr = ntohl(addr);
60 cp = &buf[sizeof buf];
61 *--cp = '\0';
62
63 n = 4;
64 do {
65 byte = addr & 0xff;
66 *--cp = byte % 10 + '0';
67 byte /= 10;
68 if (byte > 0) {
69 *--cp = byte % 10 + '0';
70 byte /= 10;
71 if (byte > 0)
72 *--cp = byte + '0';
73 }
74 *--cp = '.';
75 addr >>= 8;
76 } while (--n > 0);
77
78 return (cp+1);
79}
