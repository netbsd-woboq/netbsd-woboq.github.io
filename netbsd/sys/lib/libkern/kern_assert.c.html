34#include <sys/types.h>
35#include <sys/systm.h>
36
37#ifdef _STANDALONE
38#include <lib/libkern/libkern.h>
39#endif
40
41/* coverity[+kill] */
42void
43kern_assert(const char *fmt, ...)
44{
45 va_list ap;
46#ifdef _KERNEL
47 if (panicstr != NULL)
48 return;
49#endif
50 va_start(ap, fmt);
51 vpanic(fmt, ap);
52 va_end(ap);
53
54 /*
55 * Force instructions at the return address of vpanic before
56 * the next symbol, which otherwise the compiler may omit
57 * because vpanic is marked noreturn. This prevents seeing
58 * whatever random symbol came after kern_assert in the linked
59 * kernel in stack traces for assertion failures.
60 */
61 asm volatile(".long 0");
62}
