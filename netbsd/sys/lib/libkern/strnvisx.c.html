36#include <lib/libkern/libkern.h>
37
38#define STRVIS_ISWHITE(x) ((x) == ' ' || (x) == '\0' || (x) == '\377')
39
40int
41strnvisx(char *dst, size_t dlen, const char *src, size_t slen, int flags)
42{
43 if (dlen == 0)
44 return -1;
45
46 if (flags & VIS_TRIM) {
47 /* Trim leading and trailing blanks and NULs. */
48 while (slen > 0 && STRVIS_ISWHITE(src[0]))
49 ++src, --slen;
50 while (slen > 0 && STRVIS_ISWHITE(src[slen - 1]))
51 --slen;
52 }
53
54 while (slen > 0) {
55 if ((flags & VIS_SAFE) && (*src < 0x20 || (*src & 0x80))) {
56 /* non-printable characters */
57 if (dlen < 4)
58 goto out;
59 dlen -= 4;
60 *dst++ = '\\';
61 *dst++ = ((*src & 0300) >> 6) + '0';
62 *dst++ = ((*src & 0070) >> 3) + '0';
63 *dst++ = ((*src & 0007) >> 0) + '0';
64 } else if (*src == '\\') {
65 /* quote characters */
66 if (dlen < 2)
67 goto out;
68 dlen -= 2;
69 *dst++ = '\\';
70 *dst++ = '\\';
71 } else {
72 /* normal characters */
73 if (dlen < 1)
74 goto out;
75 *dst++ = *src;
76 }
77 ++src, --slen;
78 }
79out:
80 if (dlen > 0) {
81 *dst++ = '\0';
82 return slen ? -1 : 0;
83 } else {
84 *--dst = '\0';
85 return -1;
86 }
87}
