36#include <sys/disklabel.h>
37#include <sys/bootblock.h>
38
39int
40xlat_mbr_fstype(int mbr_type)
41{
42 static const struct ptn_types {
43 uint8_t mbr_type;
44 uint8_t netbsd_type;
45 } ptn_types[] = {
46 { MBR_PTYPE_386BSD, FS_BSDFFS },
47 { MBR_PTYPE_APPLE_UFS, FS_APPLEUFS },
48 { MBR_PTYPE_FAT12, FS_MSDOS },
49 { MBR_PTYPE_FAT16B, FS_MSDOS },
50 { MBR_PTYPE_FAT16L, FS_MSDOS },
51 { MBR_PTYPE_FAT16S, FS_MSDOS },
52 { MBR_PTYPE_FAT32, FS_MSDOS },
53 { MBR_PTYPE_FAT32L, FS_MSDOS },
54 { MBR_PTYPE_LNXEXT2, FS_EX2FS },
55 { MBR_PTYPE_LNXSWAP, FS_SWAP },
56 { MBR_PTYPE_NETBSD, FS_BSDFFS },
57 { MBR_PTYPE_NTFS, FS_NTFS },
58 { MBR_PTYPE_MINIX_14B, FS_MINIXFS3 },
59 { MBR_PTYPE_OPENBSD, FS_BSDFFS },
60 { 0, FS_OTHER }
61 };
62 const struct ptn_types *pt;
63 for (pt = ptn_types; pt->mbr_type != 0; pt++)
64 if (mbr_type == pt->mbr_type)
65 break;
66 return pt->netbsd_type;
67}
