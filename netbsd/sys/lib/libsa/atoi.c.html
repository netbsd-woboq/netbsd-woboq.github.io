29#include <sys/types.h>
30#include <sys/reboot.h>
31
32#include <lib/libsa/stand.h>
33#include <lib/libkern/libkern.h>
34
35int
36atoi(const char *in)
37{
38 char *c;
39 int ret;
40
41 ret = 0;
42 c = __UNCONST(in);
43 if (*c == '-')
44 c++;
45 for (; isdigit(*c); c++)
46 ret = (ret * 10) + (*c - '0');
47
48 return (*in == '-') ? -ret : ret;
49}
