29#ifndef _BOOTCFG_H
30#define _BOOTCFG_H
31
32#define BOOTCFG_FILENAME "boot.cfg"
33#define BOOTCFG_MAXMENU 20
34#define BOOTCFG_MAXBANNER 12
35
36#define BOOTCFG_CMD_LOAD "load"
37#define BOOTCFG_CMD_USERCONF "userconf"
38
39typedef void (*bootcfg_command)(const char *cmd, char *arg);
40
41struct bootcfg_def {
42 char *banner[BOOTCFG_MAXBANNER]; /* Banner text */
43 char *command[BOOTCFG_MAXMENU]; /* Menu commands per entry*/
44 char *consdev; /* Console device */
45 int def; /* Default menu option */
46 char *desc[BOOTCFG_MAXMENU]; /* Menu text per entry */
47 int nummenu; /* Number of menu items */
48 int timeout; /* Timeout in seconds */
49 int menuformat; /* Letters instead of numbers */
50 int clear; /* Clear the screen? */
51};
52extern struct bootcfg_def bootcfg_info;
53
54int perform_bootcfg(const char *, bootcfg_command, const off_t);
55void bootcfg_do_noop(const char *, char *);
56
57#endif /* !_BOOTCFG_H */
