63#include "stand.h"
64
65int
66#ifndef __INTERNAL_LIBSA_CREAD
67close(int fd)
68#else
69oclose(int fd)
70#endif
71{
72 struct open_file *f = &files[fd];
73 int err1 = 0, err2 = 0;
74
75#if !defined(LIBSA_NO_FD_CHECKING)
76 if ((unsigned int)fd >= SOPEN_MAX || f->f_flags == 0) {
77 errno = EBADF;
78 return -1;
79 }
80#endif
81#if !defined(LIBSA_NO_RAW_ACCESS)
82 if (!(f->f_flags & F_RAW))
83#endif
84#if !defined(LIBSA_SINGLE_FILESYSTEM)
85 if (f->f_ops != NULL)
86#endif
87 err1 = FS_CLOSE(f->f_ops)(f);
88 if (!(f->f_flags & F_NODEV))
89#if !defined(LIBSA_SINGLE_DEVICE)
90 if (f->f_dev != NULL)
91#endif
92 err2 = DEV_CLOSE(f->f_dev)(f);
93 f->f_flags = 0;
94 if (err1) {
95 errno = err1;
96 return -1;
97 }
98 if (err2) {
99 errno = err2;
100 return -1;
101 }
102 return 0;
103}
