34#include <sys/param.h>
35#include <sys/disklabel.h>
36#include "stand.h"
37
38
39#if defined(LIBSA_NO_DISKLABEL_MSGS)
40#define nolabel (char *)1
41#define corruptedlabel (char *)1
42#else
43static char nolabel[] = "no disk label";
44static char corruptedlabel[] = "disk label corrupted";
45#endif
46
47char *
48getdisklabel(const char *buf, struct disklabel *lp)
49{
50 const struct disklabel *dlp, *elp;
51 char *msg = NULL;
52
53 elp = (const void *)(buf + DEV_BSIZE - sizeof(*dlp));
54 for (dlp = (const void *)buf; dlp <= elp;
55 dlp = (const void *)((const char *)dlp + sizeof(long))) {
56 if (dlp->d_magic != DISKMAGIC || dlp->d_magic2 != DISKMAGIC) {
57 if (msg == NULL)
58 msg = nolabel;
59 } else if (dlp->d_npartitions > MAXPARTITIONS ||
60 dkcksum(dlp) != 0) {
61 msg = corruptedlabel;
62 } else {
63 (void)memcpy(lp, dlp, sizeof *lp);
64 msg = NULL;
65 break;
66 }
67 }
68 return msg;
69}
