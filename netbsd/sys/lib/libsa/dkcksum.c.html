34#include <sys/param.h>
35#include <sys/disklabel.h>
36#include "stand.h"
37
38/*
39 * Compute checksum for disk label.
40 */
41int
42dkcksum(const struct disklabel *lp)
43{
44 const u_short *start, *end;
45 u_short sum = 0;
46
47 start = (const void *)lp;
48 end = (const void *)&lp->d_partitions[lp->d_npartitions];
49 while (start < end)
50 sum ^= *start++;
51 return sum;
52}
