59/* Caller must leave room for ethernet header in front!! */
60ssize_t
61sendether(struct iodesc *d, void *pkt, size_t len, u_char *dea, int etype)
62{
63 ssize_t n;
64 struct ether_header *eh;
65
66#ifdef ETHER_DEBUG
67 if (debug)
68 printf("%s: called\n", __func__);
69#endif
70
71 eh = (struct ether_header *)pkt - 1;
72 len += sizeof(*eh);
73
74 MACPY(d->myea, eh->ether_shost); /* by byte */
75 MACPY(dea, eh->ether_dhost); /* by byte */
76 eh->ether_type = htons(etype);
77
78 n = netif_put(d, eh, len);
79 if (n == -1 || (size_t)n < sizeof(*eh))
80 return -1;
81
82 n -= (ssize_t)sizeof(*eh);
83 return n;
84}

86/*
87 * Get a packet of any Ethernet type, with our address or
88 * the broadcast address. Save the Ether type in arg 5.
89 * NOTE: Caller must leave room for the Ether header.
90 */
91ssize_t
92readether(struct iodesc *d, void *pkt, size_t len, saseconds_t tleft,
93 u_int16_t *etype)
94{
95 ssize_t n;
96 struct ether_header *eh;
97
98#ifdef ETHER_DEBUG
99 if (debug)
100 printf("%s: called\n", __func__);
101#endif
102
103 eh = (struct ether_header *)pkt - 1;
104 len += sizeof(*eh);
105
106 n = netif_get(d, eh, len, tleft);
107 if (n == -1 || (size_t)n < sizeof(*eh))
108 return -1;
109
110 /* Validate Ethernet address. */
111 if (memcmp(d->myea, eh->ether_dhost, ETHER_ADDR_LEN) != 0 &&
112 memcmp(bcea, eh->ether_dhost, ETHER_ADDR_LEN) != 0) {
113#ifdef ETHER_DEBUG
114 if (debug)
115 printf("%s: not ours (ea=%s)\n", __func__,
116 ether_sprintf(eh->ether_dhost));
117#endif
118 return -1;
119 }
120 *etype = ntohs(eh->ether_type);
121
122 n -= (ssize_t)sizeof(*eh);
123 return n;
124}
