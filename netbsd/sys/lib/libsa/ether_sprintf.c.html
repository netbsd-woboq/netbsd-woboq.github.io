55/*
56 * Convert Ethernet address to printable (loggable) representation.
57 */
58char *
59ether_sprintf(const u_char *ap)
60{
61 int i;
62 static char etherbuf[3*ETHER_ADDR_LEN];
63 char *cp = etherbuf;
64
65 for (i = 0; i < ETHER_ADDR_LEN; i++) {
66 *cp++ = hexdigits[*ap >> 4];
67 *cp++ = hexdigits[*ap++ & 0xf];
68 *cp++ = ':';
69 }
70 *--cp = 0;
71 return etherbuf;
72}
