61#include "stand.h"
62#include <lib/libkern/libkern.h>
63
64#if defined(LIBSA_ENABLE_LS_OP)
65int
66fnmatch(const char *fname, const char *pattern)
67{
68 char fc, pc;
69
70 do {
71 fc = *fname++;
72 pc = *pattern++;
73 if (!fc && !pc)
74 return 1;
75 if (pc == '?' && fc)
76 pc = fc;
77 } while (fc == pc);
78
79 if (pc != '*')
80 return 0;
81 /*
82 * Too hard (and unnecessary really) too check for "*?name" etc....
83 * "**" will look for a '*' and "*?" a '?'
84 */
85 pc = *pattern++;
86 if (!pc)
87 return 1;
88 while ((fname = strchr(fname, pc)))
89 if (fnmatch(++fname, pattern))
90 return 1;
91 return 0;
92}
93#endif /* LIBSA_ENABLE_LS_OP */
