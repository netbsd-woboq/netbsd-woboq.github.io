70/*
71 * How much room to leave for headers in UDP packets:
72 * 14: struct ether_header
73 * 20: struct ip
74 * 8: struct udphdr
75 * That's 42 but let's pad it out to 48 bytes.
76 */
77#define ETHERNET_HEADER_SIZE 14
78#define IP_HEADER_SIZE 20
79#define UDP_HEADER_SIZE 8
80
81#define UDP_TOTAL_HEADER_SIZE (ETHERNET_HEADER_SIZE + IP_HEADER_SIZE + UDP_HEADER_SIZE)
82
83/*
84 * How much room to leave for headers in TCP packets:
85 * 14: struct ether_header
86 * 20: struct ip
87 * 20: struct tcphdr
88 */
89#define TCP_HEADER_SIZE 20
90
91#define TCP_TOTAL_HEADER_SIZE (ETHERNET_HEADER_SIZE + IP_HEADER_SIZE + TCP_HEADER_SIZE)
