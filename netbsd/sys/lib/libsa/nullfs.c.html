63#include "stand.h"
64
65/*
66 * Null filesystem
67 */
68
69__compactcall int
70null_open(const char *path, struct open_file *f)
71{
72
73 return EIO;
74}
75
76#ifndef LIBSA_NO_FS_CLOSE
77__compactcall int
78null_close(struct open_file *f)
79{
80
81 return 0;
82}
83#endif
84
85__compactcall int
86null_read(struct open_file *f, void *buf, size_t size, size_t *resid)
87{
88
89 return EIO;
90}
91
92#ifndef LIBSA_NO_FS_WRITE
93__compactcall int
94null_write(struct open_file *f, void *buf, size_t size, size_t *resid)
95{
96
97 return EIO;
98}
99#endif
100
101#ifndef LIBSA_NO_FS_SEEK
102__compactcall off_t
103null_seek(struct open_file *f, off_t offset, int where)
104{
105
106 return (off_t)-1;
107}
108#endif
109
110__compactcall int
111null_stat(struct open_file *f, struct stat *sb)
112{
113
114 return EIO;
115}
116
117#if defined(LIBSA_ENABLE_LS_OP)
118#include "ls.h"
119__compactcall void
120null_ls(struct open_file *f, const char *pattern)
121{
122 lsunsup("nullfs");
123}
124#endif
