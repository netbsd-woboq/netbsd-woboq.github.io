65/*
66 * File primitives proper
67 */
68
69#ifdef HELLO_CTAGS
70oopen(){}
71#endif
72
73int
74#ifndef __INTERNAL_LIBSA_CREAD
75open(const char *fname, int mode)
76#else
77oopen(const char *fname, int mode)
78#endif
79{
80 struct open_file *f;
81 int fd, error;
82#if !defined(LIBSA_SINGLE_FILESYSTEM)
83 int i, besterror;
84#endif
85 char *file;
86
87 /* find a free file descriptor */
88 for (fd = 0, f = files; fd < SOPEN_MAX; fd++, f++)
89 if (f->f_flags == 0)
90 goto fnd;
91 errno = EMFILE;
92 return -1;
93fnd:
94 /*
95 * Try to open the device.
96 * Convert open mode (0,1,2) to F_READ, F_WRITE.
97 */
98 f->f_flags = mode + 1;
99#if !defined(LIBSA_SINGLE_DEVICE)
100 f->f_dev = (struct devsw *)0;
101#endif
102#if !defined(LIBSA_SINGLE_FILESYSTEM)
103 f->f_ops = (struct fs_ops *)0;
104#endif
105#if !defined(LIBSA_NO_RAW_ACCESS)
106 f->f_offset = 0;
107#endif
108 file = (char *)0;
109 error = devopen(f, fname, &file);
110 if (error
111#if !defined(LIBSA_SINGLE_DEVICE)
112 || (((f->f_flags & F_NODEV) == 0) &&
113 f->f_dev == (struct devsw *)0)
114#endif
115 )
116 goto err;
117
118#if !defined(LIBSA_NO_RAW_ACCESS)
119 /* see if we opened a raw device; otherwise, 'file' is the file name. */
120 if (file == (char *)0 || *file == '\0') {
121 f->f_flags |= F_RAW;
122 return fd;
123 }
124#endif
125
126 /* pass file name to the different filesystem open routines */
127#if !defined(LIBSA_SINGLE_FILESYSTEM)
128 besterror = ENOENT;
129 for (i = 0; i < nfsys; i++) {
130 error = FS_OPEN(&file_system[i])(file, f);
131 if (error == 0) {
132 f->f_ops = &file_system[i];
133 return fd;
134 }
135 if (error != EINVAL)
136 besterror = error;
137 }
138 error = besterror;
139#else
140 error = FS_OPEN(&file_system[i])(file, f);
141 if (error == 0)
142 return fd;
143 if (error == EINVAL)
144 error = ENOENT;
145#endif
146
147 if ((f->f_flags & F_NODEV) == 0) {
148#if !defined(LIBSA_SINGLE_DEVICE)
149 if (DEV_CLOSE(f->f_dev) != NULL)
150#endif
151 (void)DEV_CLOSE(f->f_dev)(f);
152 }
153err:
154 f->f_flags = 0;
155 errno = error;
156 return -1;
157}
