32#include <sys/types.h>
33#include "saerrno.h"
34#include "stand.h"
35
36static const struct mi {
37 int errno;
38 const char *msg;
39} errlist[] = {
40 { EADAPT, "bad adaptor number" },
41 { ECTLR, "bad controller number" },
42 { EUNIT, "bad drive number" },
43 { EPART, "bad partition" },
44 { ERDLAB, "can't read disk label" },
45 { EUNLAB, "unlabeled" },
46 { ENXIO, "Device not configured" },
47 { EPERM, "Operation not permitted" },
48 { ENOENT, "No such file or directory" },
49 { ESTALE, "Stale NFS file handle" },
50 { EFTYPE, "Inappropriate file type or format" },
51 { ENOEXEC, "Exec format error" },
52 { EIO, "Input/output error" },
53 { EINVAL, "Invalid argument" },
54 { ENOTDIR, "Not a directory" },
55 { EOFFSET, "invalid file offset" },
56 { EACCES, "Permission denied" },
57 { 0, 0 },
58};
59
60char *
61strerror(int err)
62{
63 static char ebuf[36];
64 const struct mi *mi;
65
66 for (mi = errlist; mi->msg; mi++)
67 if (mi->errno == err)
68 return __UNCONST(mi->msg);
69
70 snprintf(ebuf, sizeof ebuf, "Unknown error: code %d", err);
71 return ebuf;
72}
