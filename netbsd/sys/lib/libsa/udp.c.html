65/* Caller must leave room for ethernet, ip and udp headers in front!! */
66ssize_t
67sendudp(struct iodesc *d, void *pkt, size_t len)
68{
69 ssize_t cc;
70 struct udphdr *uh;
71
72#ifdef NET_DEBUG
73 if (debug) {
74 printf("%s: d=%p called.\n", __func__, d);
75 if (d) {
76 printf("saddr: %s:%d",
77 inet_ntoa(d->myip), ntohs(d->myport));
78 printf(" daddr: %s:%d\n",
79 inet_ntoa(d->destip), ntohs(d->destport));
80 }
81 }
82#endif
83
84 uh = (struct udphdr *)pkt - 1;
85 len += sizeof(*uh);
86
87 (void)memset(uh, 0, sizeof(*uh));
88
89 uh->uh_sport = d->myport;
90 uh->uh_dport = d->destport;
91 uh->uh_ulen = htons(len);
92
93 cc = sendip(d, uh, len, IPPROTO_UDP);
94 if (cc == -1)
95 return -1;
96 if ((size_t)cc != len)
97 panic("%s: bad write (%zd != %zu)", __func__, cc, len);
98 return (cc - sizeof(*uh));
99}

101/*
102 * Receive a UDP packet and validate it is for us.
103 * Caller leaves room for the headers (Ether, IP, UDP)
104 */
105ssize_t
106readudp(struct iodesc *d, void *pkt, size_t len, saseconds_t tleft)
107{
108 ssize_t n;
109 struct udphdr *uh;
110
111 uh = (struct udphdr *)pkt - 1;
112 n = readip(d, uh, len + sizeof(*uh), tleft, IPPROTO_UDP);
113 if (n == -1 || (size_t)n < sizeof(*uh))
114 return -1;
115
116 if (uh->uh_dport != d->myport) {
117#ifdef NET_DEBUG
118 if (debug)
119 printf("%s: bad dport %d != %d\n", __func__,
120 ntohs(d->myport), ntohs(uh->uh_dport));
121#endif
122 return -1;
123 }
124
125 if (ntohs(uh->uh_ulen) < sizeof(*uh)) {
126#ifdef NET_DEBUG
127 if (debug)
128 printf("%s: bad udp len %d < %zu\n", __func__,
129 ntohs(uh->uh_ulen), sizeof(*uh));
130#endif
131 return -1;
132 }
133
134 n -= sizeof(*uh);
135 return n;
136}
