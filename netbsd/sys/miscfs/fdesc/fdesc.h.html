39#ifdef _KERNEL
40
41#define FD_ROOT 2
42#define FD_DEVFD 3
43#define FD_STDIN 4
44#define FD_STDOUT 5
45#define FD_STDERR 6
46#define FD_CTTY 7
47#define FD_DESC 8
48#define FD_MAX 12
49
50typedef enum {
51 Froot,
52 Fdevfd,
53 Fdesc,
54 Flink,
55 Fctty
56} fdntype;
57
58struct fdescnode {
59 struct vnode *fd_vnode; /* Back ptr to vnode */
60 fdntype fd_type; /* Type of this node */
61 unsigned fd_fd; /* Fd to be dup'ed */
62 const char *fd_link; /* Link to fd/n */
63 int fd_ix; /* filesystem index */
64};
65
66#define VTOFDESC(vp) ((struct fdescnode *)(vp)->v_data)
67
68#define FDESC_MAXNAMLEN 255
69
70extern dev_t devctty;
71extern void fdesc_init(void);
72extern void fdesc_done(void);
73extern int fdesc_root(struct mount *, struct vnode **);
74extern int fdesc_allocvp(fdntype, int, struct mount *, struct vnode **);
75extern int (**fdesc_vnodeop_p)(void *);
76extern struct vfsops fdesc_vfsops;
77
78#endif /* _KERNEL */
