41int
42genfs_statvfs(struct mount *mp, struct statvfs *sbp)
43{
44
45 sbp->f_bsize = DEV_BSIZE;
46 sbp->f_frsize = DEV_BSIZE;
47 sbp->f_iosize = DEV_BSIZE;
48 sbp->f_blocks = 2; /* 1k to keep df happy */
49 sbp->f_bfree = 0;
50 sbp->f_bavail = 0;
51 sbp->f_bresvd = 0;
52 sbp->f_files = 0;
53 sbp->f_ffree = 0;
54 sbp->f_favail = 0;
55 sbp->f_fresvd = 0;
56 copy_statvfs_info(sbp, mp);
57
58 return 0;
59}

61int
62genfs_renamelock_enter(struct mount *mp)
63{
64 mutex_enter(&mp->mnt_renamelock);

75int
76genfs_suspendctl(struct mount *mp, int cmd)
77{
78 int error;
79 int error2 __diagused;
80
81 switch (cmd) {
82 case SUSPEND_SUSPEND:
83 error = fstrans_setstate(mp, FSTRANS_SUSPENDING);
84 if (error)
85 return error;
86 error = fstrans_setstate(mp, FSTRANS_SUSPENDED);
87 if (error == 0) {
88 if ((mp->mnt_iflag & IMNT_GONE) != 0)
89 error = ENOENT;
90 if (error) {
91 error2 = fstrans_setstate(mp, FSTRANS_NORMAL);
92 KASSERT(error2 == 0);
93 }
94 }
95 return error;
96
97 case SUSPEND_RESUME:
98 error2 = fstrans_setstate(mp, FSTRANS_NORMAL);
99 KASSERT(error2 == 0);
100 return 0;
101
102 default:
103 panic("%s: bogus command %d", __func__, cmd);
104 }
105}
