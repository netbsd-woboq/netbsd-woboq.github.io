71#ifndef _MISCFS_GENFS_LAYER_H_
72#define _MISCFS_GENFS_LAYER_H_
73
74struct layer_args {
75 char *target; /* Target of loopback */
76 struct export_args30 _pad1; /* compat with old userland tools */
77};
78
79#ifdef _KERNEL
80
81struct layer_mount {
82 struct vnode *layerm_rootvp; /* Ref to root layer_node */
83 u_int layerm_flags; /* mount point layer flags */
84 u_int layerm_size; /* size of fs's struct node */
85 enum vtagtype layerm_tag; /* vtag of our vnodes */
86 int /* bypass routine for this mount */
87 (*layerm_bypass)(void *);
88 int (**layerm_vnodeop_p) /* ops for our nodes */
89 (void *);
90};
91
92#define LAYERFS_MFLAGS 0x00000fff /* reserved layer mount flags */
93#define LAYERFS_MBYPASSDEBUG 0x00000001
94
95/*
96 * A cache of vnode references
97 */
98struct layer_node {
99 struct vnode *layer_lowervp; /* VREFed once */
100 struct vnode *layer_vnode; /* Back pointer */
101 unsigned int layer_flags; /* locking, etc. */
102};
103
104#define LAYERFS_RESFLAGS 0x00000fff /* flags reserved for layerfs */
105#define LAYERFS_REMOVED 0x00000001 /* Did a remove on this node */
106
107#define LAYERFS_DO_BYPASS(vp, ap) \
108 (*MOUNTTOLAYERMOUNT((vp)->v_mount)->layerm_bypass)((ap))
109
110struct vnode *layer_checkvp(struct vnode *vp, const char *fil, int lno);
111
112#define MOUNTTOLAYERMOUNT(mp) ((struct layer_mount *)((mp)->mnt_data))
113#define VTOLAYER(vp) ((struct layer_node *)(vp)->v_data)
114#define LAYERTOV(xp) ((xp)->layer_vnode)
115#ifdef LAYERFS_DIAGNOSTIC
116#define LAYERVPTOLOWERVP(vp) layer_checkvp((vp), __FILE__, __LINE__)
117extern int layerfs_debug;
118#else
119#define LAYERVPTOLOWERVP(vp) (VTOLAYER(vp)->layer_lowervp)
120#endif
121
122#endif /* _KERNEL */
123#endif /* _MISCFS_GENFS_LAYER_H_ */
