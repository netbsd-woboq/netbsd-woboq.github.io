41int
42procfs_doauxv(struct lwp *curl, struct proc *p, struct pfsnode *pfs,
43 struct uio *uio)
44{
45 int error;
46 void *buffer;
47 size_t bufsize;
48
49 if (uio->uio_rw != UIO_READ)
50 return EOPNOTSUPP;
51
52 if ((error = proc_getauxv(p, &buffer, &bufsize)) != 0)
53 return error;
54
55 if (uio->uio_offset < bufsize)
56 error = uiomove((char *)buffer + uio->uio_offset,
57 bufsize - uio->uio_offset, uio);
58 else
59 error = 0;
60
61 kmem_free(buffer, bufsize);
62 return error;
63}
64
65int
66procfs_validauxv(struct lwp *l, struct mount *mp)
67{
68 return l != NULL && l->l_proc != NULL && l->l_proc->p_execsw != NULL;
69}
