46int
47procfs_dofd(lwp_t *curl, proc_t *p, struct pfsnode *pfs, struct uio *uio)
48{
49 int error;
50 file_t *fp;
51 off_t offs;
52
53 if ((fp = fd_getfile2(p, pfs->pfs_fd)) == NULL)
54 return EBADF;
55
56 offs = fp->f_offset;
57
58 switch (uio->uio_rw) {
59 case UIO_READ:
60 error = (*fp->f_ops->fo_read)(fp, &offs, uio, curl->l_cred, 0);
61 break;
62 case UIO_WRITE:
63 error = (*fp->f_ops->fo_write)(fp, &offs, uio, curl->l_cred, 0);
64 break;
65 default:
66 panic("bad uio op");
67 }
68
69 closef(fp);
70
71 return (error);
72}
