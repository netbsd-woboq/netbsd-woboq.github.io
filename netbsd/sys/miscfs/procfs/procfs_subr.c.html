122/*
123 * Allocate a pfsnode/vnode pair. The vnode is referenced.
124 * The pid, type, and file descriptor uniquely identify a pfsnode.
125 */
126int
127procfs_allocvp(struct mount *mp, struct vnode **vpp, pid_t pid,
128 pfstype type, int fd)
129{
130 struct pfskey key;
131
132 memset(&key, 0, sizeof(key));
133 key.pk_type = type;
134 key.pk_pid = pid;
135 key.pk_fd = fd;
136
137 return vcache_get(mp, &key, sizeof(key), vpp);
138}

140int
141procfs_rw(void *v)
142{
143 struct vop_read_args *ap = v;
144 struct vnode *vp = ap->a_vp;
145 struct uio *uio = ap->a_uio;
146 struct lwp *curl;
147 struct lwp *l;
148 struct pfsnode *pfs = VTOPFS(vp);
149 struct proc *p;
150 int error;
151
152 if (uio->uio_offset < 0)
153 return EINVAL;
154
155 if ((error = procfs_proc_lock(pfs->pfs_pid, &p, ESRCH)) != 0)
156 return error;
157
158 curl = curlwp;
159
160 /*
161 * Do not allow init to be modified while in secure mode; it
162 * could be duped into changing the security level.
163 */
164#define M2K(m) ((m) == UIO_READ ? KAUTH_REQ_PROCESS_PROCFS_READ : \
165 KAUTH_REQ_PROCESS_PROCFS_WRITE)
166 mutex_enter(p->p_lock);
167 error = kauth_authorize_process(curl->l_cred, KAUTH_PROCESS_PROCFS,
168 p, pfs, KAUTH_ARG(M2K(uio->uio_rw)), NULL);
169 mutex_exit(p->p_lock);
170 if (error) {
171 procfs_proc_unlock(p);
172 return (error);
173 }
174#undef M2K
175
176 mutex_enter(p->p_lock);
177 LIST_FOREACH(l, &p->p_lwps, l_sibling) {
178 if (l->l_stat != LSZOMB)
179 break;
180 }
181 /* Process is exiting if no-LWPS or all LWPs are LSZOMB */
182 if (l == NULL) {
183 mutex_exit(p->p_lock);
184 procfs_proc_unlock(p);
185 return ESRCH;
186 }
187
188 lwp_addref(l);
189 mutex_exit(p->p_lock);
190
191 switch (pfs->pfs_type) {
192 case PFSnote:
193 case PFSnotepg:
194 error = procfs_donote(curl, p, pfs, uio);
195 break;
196
197 case PFSregs:
198 error = procfs_doregs(curl, l, pfs, uio);
199 break;
200
201 case PFSfpregs:
202 error = procfs_dofpregs(curl, l, pfs, uio);
203 break;
204
205 case PFSstatus:
206 error = procfs_dostatus(curl, l, pfs, uio);
207 break;
208
209 case PFSstat:
210 error = procfs_do_pid_stat(curl, l, pfs, uio);
211 break;
212
213 case PFSlimit:
214 error = procfs_dolimit(curl, p, pfs, uio);
215 break;
216
217 case PFSmap:
218 error = procfs_domap(curl, p, pfs, uio, 0);
219 break;
220
221 case PFSmaps:
222 error = procfs_domap(curl, p, pfs, uio, 1);
223 break;
224
225 case PFSmem:
226 error = procfs_domem(curl, l, pfs, uio);
227 break;
228
229 case PFScmdline:
230 error = procfs_doprocargs(curl, p, pfs, uio, KERN_PROC_ARGV);
231 break;
232
233 case PFSenviron:
234 error = procfs_doprocargs(curl, p, pfs, uio, KERN_PROC_ENV);
235 break;
236
237 case PFSmeminfo:
238 error = procfs_domeminfo(curl, p, pfs, uio);
239 break;
240
241 case PFSdevices:
242 error = procfs_dodevices(curl, p, pfs, uio);
243 break;
244
245 case PFScpuinfo:
246 error = procfs_docpuinfo(curl, p, pfs, uio);
247 break;
248
249 case PFScpustat:
250 error = procfs_docpustat(curl, p, pfs, uio);
251 break;
252
253 case PFSloadavg:
254 error = procfs_doloadavg(curl, p, pfs, uio);
255 break;
256
257 case PFSstatm:
258 error = procfs_do_pid_statm(curl, l, pfs, uio);
259 break;
260
261 case PFSfd:
262 error = procfs_dofd(curl, p, pfs, uio);
263 break;
264
265 case PFSuptime:
266 error = procfs_douptime(curl, p, pfs, uio);
267 break;
268
269 case PFSmounts:
270 error = procfs_domounts(curl, p, pfs, uio);
271 break;
272
273 case PFSemul:
274 error = procfs_doemul(curl, p, pfs, uio);
275 break;
276
277 case PFSversion:
278 error = procfs_doversion(curl, p, pfs, uio);
279 break;
280
281 case PFSauxv:
282 error = procfs_doauxv(curl, p, pfs, uio);
283 break;
284
285#ifdef __HAVE_PROCFS_MACHDEP
286 PROCFS_MACHDEP_NODETYPE_CASES
287 error = procfs_machdep_rw(curl, l, pfs, uio);
288 break;
289#endif
290
291 default:
292 error = EOPNOTSUPP;
293 break;
294 }
295
296 /*
297 * Release the references that we acquired earlier.
298 */
299 lwp_delref(l);
300 procfs_proc_unlock(p);
301
302 return (error);
303}

305/*
306 * Get a string from userland into (bf). Strip a trailing
307 * nl character (to allow easy access from the shell).
308 * The buffer should be *buflenp + 1 chars long. vfs_getuserstr
309 * will automatically add a nul char at the end.
310 *
311 * Returns 0 on success or the following errors
312 *
313 * EINVAL: file offset is non-zero.
314 * EMSGSIZE: message is longer than kernel buffer
315 * EFAULT: user i/o buffer is not addressable
316 */
317int
318vfs_getuserstr(struct uio *uio, char *bf, int *buflenp)
319{
320 int xlen;
321 int error;
322
323 if (uio->uio_offset != 0)
324 return (EINVAL);
325
326 xlen = *buflenp;
327
328 /* must be able to read the whole string in one go */
329 if (xlen < uio->uio_resid)
330 return (EMSGSIZE);
331 xlen = uio->uio_resid;
332
333 if ((error = uiomove(bf, xlen, uio)) != 0)
334 return (error);
335
336 /* allow multiple writes without seeks */
337 uio->uio_offset = 0;
338
339 /* cleanup string and remove trailing newline */
340 bf[xlen] = '\0';
341 xlen = strlen(bf);
342 if (xlen > 0 && bf[xlen-1] == '\n')
343 bf[--xlen] = '\0';
344 *buflenp = xlen;
345
346 return (0);
347}

349const vfs_namemap_t *
350vfs_findname(const vfs_namemap_t *nm, const char *bf, int buflen)
351{
352
353 for (; nm->nm_name; nm++)
354 if (memcmp(bf, nm->nm_name, buflen+1) == 0)
355 return (nm);
356
357 return (0);
358}

360static bool
361procfs_revoke_selector(void *arg, struct vnode *vp)
362{
363 struct proc *p = arg;
364 struct pfsnode *pfs;
365
366 KASSERT(mutex_owned(vp->v_interlock));
367
368 pfs = VTOPFS(vp);
369
370 return (pfs != NULL && pfs->pfs_pid == p->p_pid);
371}

373void
374procfs_revoke_vnodes(struct proc *p, void *arg)
375{
376 int error;
377 bool suspended;
378 struct vnode *vp;
379 struct vnode_iterator *marker;
380 struct mount *mp = (struct mount *)arg;
381
382 if (!(p->p_flag & PK_SUGID))
383 return;
384
385 suspended = false;
386 vfs_vnode_iterator_init(mp, &marker);
387
388 while ((vp = vfs_vnode_iterator_next(marker,
389 procfs_revoke_selector, p)) != NULL) {
390 if (vrecycle(vp))
391 continue;
392 /* Vnode is busy, we have to suspend the mount for vgone(). */
393 while (! suspended) {
394 error = vfs_suspend(mp, 0);
395 if (error == 0) {
396 suspended = true;
397 } else if (error != EINTR && error != ERESTART) {
398 KASSERT(error == EOPNOTSUPP);
399 break;
400 }
401 }
402 vgone(vp);
403 }
404
405 if (suspended)
406 vfs_resume(mp);
407
408 vfs_vnode_iterator_destroy(marker);
409}

411int
412procfs_proc_lock(int pid, struct proc **bunghole, int notfound)
413{
414 struct proc *tp;
415 int error = 0;
416
417 mutex_enter(proc_lock);
418
419 if (pid == 0)
420 tp = &proc0;
421 else if ((tp = proc_find(pid)) == NULL)
422 error = notfound;
423 if (tp != NULL && !rw_tryenter(&tp->p_reflock, RW_READER))
424 error = EBUSY;
425
426 mutex_exit(proc_lock);
427
428 *bunghole = tp;
429 return error;
430}
431
432void
433procfs_proc_unlock(struct proc *p)
434{
435
436 rw_exit(&p->p_reflock);
437}

439int
440procfs_doemul(struct lwp *curl, struct proc *p,
441 struct pfsnode *pfs, struct uio *uio)
442{
443 const char *ename = p->p_emul->e_name;
444 return uiomove_frombuf(__UNCONST(ename), strlen(ename), uio);
