32#include <sys/param.h>
33#include <sys/kernel.h>
34#include <sys/module.h>
35
36#include <dev/md.h>
37
38MODULE(MODULE_CLASS_MISC, miniroot, NULL);
39
40static int
41miniroot_modcmd(modcmd_t cmd, void *arg)
42{
43 void *addr;
44 int error;
45 size_t size;
46
47 switch (cmd) {
48 case MODULE_CMD_INIT:
49 error = module_find_section("miniroot", &addr, &size);
50 if (error == 0) {
51 if (size == 0) {
52 error = EINVAL;
53 } else {
54 md_root_setconf(addr, size);
55 }
56 }
57 return error;
58
59 case MODULE_CMD_FINI:
60 return EOPNOTSUPP;
61
62 default:
63 return ENOTTY;
64 }
65}
