37int
38tlv_check(const void *p, size_t size, const struct tlvhdr *tlv,
39 const struct tlv_template *tmpl, bool check_type)
40{
41
42 while (/* CONSTCOND */ 1) {
43 if ((const char *)tlv - (const char *)p + sizeof(*tlv) > size) {
44 return EINVAL;
45 }
46 if ((check_type && tlv->tlv_type != tmpl->tmpl_type) ||
47 tlv->tlv_length != tmpl->tmpl_length) {
48 return EINVAL;
49 }
50 if (tmpl->tmpl_type == 0) {
51 break;
52 }
53 tlv = (const struct tlvhdr *)
54 ((const char *)tlv + tlv->tlv_length);
55 tmpl++;
56 }
57
58 return 0;
59}
