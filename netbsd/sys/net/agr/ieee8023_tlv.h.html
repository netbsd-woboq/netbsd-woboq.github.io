30#define _NET_AGR_IEEE8023_TLV_H_
31
32/*
33 * TLV on-wire structure.
34 */
35
36struct tlvhdr {
37 uint8_t tlv_type;
38 uint8_t tlv_length;
39 /* uint8_t tlv_value[]; */
40} __packed;
41
42/*
43 * ... and our implementation.
44 */
45
46#define TLV_SET(tlv, type, length) \
47 do { \
48 (tlv)->tlv_type = (type); \
49 (tlv)->tlv_length = sizeof(*tlv) + (length); \
50 } while (/*CONSTCOND*/0)
51
52struct tlv_template {
53 uint8_t tmpl_type;
54 uint8_t tmpl_length;
55};
56
57int tlv_check(const void *, size_t, const struct tlvhdr *,
58 const struct tlv_template *, bool);
59
60#endif /* !_NET_AGR_IEEE8023_TLV_H_ */
