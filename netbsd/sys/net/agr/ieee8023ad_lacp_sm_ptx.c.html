47/* periodic transmit machine */
48
49void
50lacp_sm_ptx_update_timeout(struct lacp_port *lp, uint8_t oldpstate)
51{
52
53 if (LACP_STATE_EQ(oldpstate, lp->lp_partner.lip_state,
54 LACP_STATE_TIMEOUT)) {
55 return;
56 }
57
58 LACP_DPRINTF((lp, "partner timeout changed\n"));
59
60 /*
61 * FAST_PERIODIC -> SLOW_PERIODIC
62 * or
63 * SLOW_PERIODIC (-> PERIODIC_TX) -> FAST_PERIODIC
64 *
65 * let lacp_sm_ptx_tx_schedule to update timeout.
66 */
67
68 LACP_TIMER_DISARM(lp, LACP_TIMER_PERIODIC);
69
70 /*
71 * if timeout has been shortened, assert NTT.
72 */
73
74 if ((lp->lp_partner.lip_state & LACP_STATE_TIMEOUT)) {
75 lacp_sm_assert_ntt(lp);
76 }
77}
78
79void
80lacp_sm_ptx_tx_schedule(struct lacp_port *lp)
81{
82 int timeout;
83
84 if (!(lp->lp_state & LACP_STATE_ACTIVITY) &&
85 !(lp->lp_partner.lip_state & LACP_STATE_ACTIVITY)) {
86
87 /*
88 * NO_PERIODIC
89 */
90
91 LACP_TIMER_DISARM(lp, LACP_TIMER_PERIODIC);
92 return;
93 }
94
95 if (LACP_TIMER_ISARMED(lp, LACP_TIMER_PERIODIC)) {
96 return;
97 }
98
99 timeout = (lp->lp_partner.lip_state & LACP_STATE_TIMEOUT) ?
100 LACP_FAST_PERIODIC_TIME : LACP_SLOW_PERIODIC_TIME;
101
102 LACP_TIMER_ARM(lp, LACP_TIMER_PERIODIC, timeout);
103}

105void
106lacp_sm_ptx_timer(struct lacp_port *lp)
107{
108
109 lacp_sm_assert_ntt(lp);
110}
