45typedef void (*lacp_timer_func_t)(struct lacp_port *);
46
47static const lacp_timer_func_t lacp_timer_funcs[LACP_NTIMER] = {
48 [LACP_TIMER_CURRENT_WHILE] = lacp_sm_rx_timer,
49 [LACP_TIMER_PERIODIC] = lacp_sm_ptx_timer,
50 [LACP_TIMER_WAIT_WHILE] = lacp_sm_mux_timer,
51};
52
53void
54lacp_run_timers(struct lacp_port *lp)
55{
56 int i;
57
58 for (i = 0; i < LACP_NTIMER; i++) {
59 KASSERT(lp->lp_timer[i] >= 0);
60 if (lp->lp_timer[i] == 0) {
61 continue;
62 } else if (--lp->lp_timer[i] <= 0) {
63 if (lacp_timer_funcs[i]) {
64 (*lacp_timer_funcs[i])(lp);
65 }
66 }
67 }
68}
