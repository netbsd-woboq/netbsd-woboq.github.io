46static const struct tlv_template marker_info_tlv_template[] = {
47 { MARKER_TYPE_INFO, 16 },
48 { 0, 0 },
49};
50
51static const struct tlv_template marker_response_tlv_template[] = {
52 { MARKER_TYPE_RESPONSE, 16 },
53 { 0, 0 },
54};
55
56int
57ieee8023ad_marker_input(struct ifnet *ifp, struct mbuf *m)
58{
59 struct markerdu *mdu;
60 struct agr_port *port;
61 int error = 0;
62
63 port = ifp->if_agrprivate; /* XXX race with agr_remport. */
64 KASSERT(port);
65 if (__predict_false(port->port_flags & AGRPORT_DETACHING)) {
66 goto bad;
67 }
68
69 if (m->m_pkthdr.len != sizeof(*mdu)) {
70 goto bad;
71 }
72
73 if ((m->m_flags & M_MCAST) == 0) {
74 goto bad;
75 }
76
77 if (m->m_len < sizeof(*mdu)) {
78 m = m_pullup(m, sizeof(*mdu));
79 if (m == NULL) {
80 return ENOMEM;
81 }
82 }
83
84 mdu = mtod(m, struct markerdu *);
85
86 if (memcmp(&mdu->mdu_eh.ether_dhost,
87 &ethermulticastaddr_slowprotocols, ETHER_ADDR_LEN)) {
88 goto bad;
89 }
90
91 KASSERT(mdu->mdu_sph.sph_subtype == SLOWPROTOCOLS_SUBTYPE_MARKER);
92 if (mdu->mdu_sph.sph_version != 1) {
93 goto bad;
94 }
95
96 switch (mdu->mdu_tlv.tlv_type) {
97 case MARKER_TYPE_INFO:
98 if (tlv_check(mdu, sizeof(*mdu), &mdu->mdu_tlv,
99 marker_info_tlv_template, true)) {
100 goto bad;
101 }
102 mdu->mdu_tlv.tlv_type = MARKER_TYPE_RESPONSE;
103 memcpy(&mdu->mdu_eh.ether_dhost,
104 &ethermulticastaddr_slowprotocols, ETHER_ADDR_LEN);
105 memcpy(&mdu->mdu_eh.ether_shost,
106 &port->port_origlladdr, ETHER_ADDR_LEN);
107 error = agr_xmit_frame(ifp, m);
108 break;
109
110 case MARKER_TYPE_RESPONSE:
111 if (tlv_check(mdu, sizeof(*mdu), &mdu->mdu_tlv,
112 marker_response_tlv_template, true)) {
113 goto bad;
114 }
115 /*
116 * we are not interested in responses as
117 * we don't have a marker sender.
118 */
119 /* FALLTHROUGH */
120 default:
121 goto bad;
122 }
123
124 return error;
125
126bad:
127 m_freem(m);
128 return EINVAL;
129}
