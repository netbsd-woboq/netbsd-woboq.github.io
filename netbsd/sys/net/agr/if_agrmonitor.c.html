42void
43agrport_monitor(struct agr_port *port)
44{
45 struct agr_softc *sc = AGR_SC_FROM_PORT(port);
46 u_int media;
47 u_int status;
48
49 media = IFM_ETHER | IFM_NONE;
50 status = IFM_AVALID;
51 if ((~port->port_ifp->if_flags & (IFF_RUNNING | IFF_UP)) == 0) {
52 int error;
53
54 error = agr_port_getmedia(port, &media, &status);
55 if (error) {
56#if defined(DEBUG)
57 printf("getmedia: error %d\n", error);
58#endif /* defined(DEBUG) */
59 }
60 }
61
62 if ((status & (IFM_AVALID | IFM_ACTIVE)) == IFM_AVALID)
63 media = IFM_ETHER | IFM_NONE; /* XXX ether */
64
65 if (media == IFM_NONE) {
66 /* possible eg. when the phy is not configured. */
67#if defined(DEBUG)
68 printf("%s: IFM_NONE\n", __func__);
69#endif /* defined(DEBUG) */
70 media = IFM_ETHER | IFM_NONE; /* XXX ether */
71 }
72
73 if (port->port_media == media)
74 return;
75
76 port->port_media = media;
77 if (sc->sc_iftop->iftop_portstate)
78 (*sc->sc_iftop->iftop_portstate)(port);
79}
