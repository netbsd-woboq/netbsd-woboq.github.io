41/*
42 * these functions are isolated from if_agr.c to avoid inclusion of if_ether.h.
43 */
44
45struct agr_softc *
46agr_alloc_softc(void)
47{
48 struct agr_softc *sc;
49 union {
50 struct ifnet u_if;
51 struct ethercom u_ec;
52 } *u;
53
54 /*
55 * as we don't know our if_type yet..
56 */
57
58 sc = malloc(sizeof(*sc) - sizeof(sc->sc_if) + sizeof(*u),
59 M_DEVBUF, M_WAITOK | M_ZERO);
60
61 return sc;
62}
63
64void
65agr_free_softc(struct agr_softc *sc)
66{
67
68 free(sc, M_DEVBUF);
69}
