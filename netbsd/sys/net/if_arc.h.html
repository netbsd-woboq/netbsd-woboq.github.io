35#ifndef _NET_IF_ARC_H_
36#define _NET_IF_ARC_H_
37
38/*
39 * Arcnet address - 1 octets
40 * don't know who uses this.
41 */
42struct arc_addr {
43 uint8_t arc_addr_octet[1];
44} __packed;
45
46/*
47 * Structure of a 2.5MB/s Arcnet header.
48 * as given to interface code.
49 */
50struct arc_header {
51 uint8_t arc_shost;
52 uint8_t arc_dhost;
53 uint8_t arc_type;
54 /*
55 * only present for newstyle encoding with LL fragmentation.
56 * Don't use sizeof(anything), use ARC_HDR{,NEW}LEN instead.
57 */
58 uint8_t arc_flag;
59 uint16_t arc_seqid;
60
61 /*
62 * only present in exception packets (arc_flag == 0xff)
63 */
64 uint8_t arc_type2; /* same as arc_type */
65 uint8_t arc_flag2; /* real flag value */
66 uint16_t arc_seqid2; /* real seqid value */
67} __packed;
68
69#define ARC_ADDR_LEN 1
70
71#define ARC_HDRLEN 3
72#define ARC_HDRNEWLEN 6
73#define ARC_HDRNEWLEN_EXC 10
74
75/* these lengths are data link layer length - 2*ARC_ADDR_LEN */
76#define ARC_MIN_LEN 1
77#define ARC_MIN_FORBID_LEN 254
78#define ARC_MAX_FORBID_LEN 256
79#define ARC_MAX_LEN 508
80
81
82/* RFC 1051 */
83#define ARCTYPE_IP_OLD 240 /* IP protocol */
84#define ARCTYPE_ARP_OLD 241 /* address resolution protocol */
85
86/* RFC 1201 */
87#define ARCTYPE_IP 212 /* IP protocol */
88#define ARCTYPE_ARP 213 /* address resolution protocol */
89#define ARCTYPE_REVARP 214 /* reverse addr resolution protocol */
90
91#define ARCTYPE_ATALK 221 /* Appletalk */
92#define ARCTYPE_BANIAN 247 /* Banyan Vines */
93#define ARCTYPE_IPX 250 /* Novell IPX */
94
95#define ARCTYPE_INET6 0xc4 /* IPng */
96#define ARCTYPE_DIAGNOSE 0x80 /* as per ANSI/ATA 878.1 */
97
98#define ARCMTU 507
99#define ARCMIN 0
100
101#define ARC_PHDS_MAXMTU 60480
102
103struct arccom {
104 struct ifnet ac_if; /* network-visible interface */
105
106 uint16_t ac_seqid; /* seq. id used by PHDS encap. */
107
108 struct ac_frag {
109 uint8_t af_maxflag; /* from first packet */
110 uint8_t af_lastseen; /* last split flag seen */
111 uint16_t af_seqid;
112 struct mbuf *af_packet;
113 } ac_fragtab[256]; /* indexed by sender ll address */
114
115};

123int arc_isphds(uint8_t);

126#endif /* !_NET_IF_ARC_H_ */
