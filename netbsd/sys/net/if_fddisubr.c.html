180/*
181 * FDDI output routine.
182 * Encapsulate a packet of type family for the local net.
183 * Assumes that ifp is actually pointer to ethercom structure.
184 */
185static int
186fddi_output(struct ifnet *ifp0, struct mbuf *m0, const struct sockaddr *dst,
187 const struct rtentry *rt)
188{
189 uint16_t etype;
190 int error = 0, hdrcmplt = 0;
191 uint8_t esrc[6], edst[6];
192 struct mbuf *m = m0;
193 struct fddi_header *fh;
194 struct mbuf *mcopy = NULL;
195 struct ifnet *ifp = ifp0;
196
197 MCLAIM(m, ifp->if_mowner);
198
199#if NCARP > 0
200 if (ifp->if_type == IFT_CARP) {
201 struct ifaddr *ifa;
202
203 /* loop back if this is going to the carp interface */
204 if (dst != NULL && ifp0->if_link_state == LINK_STATE_UP) {
205 int s = pserialize_read_enter();
206 ifa = ifa_ifwithaddr(dst);
207 if (ifa != NULL &&
208 ifa->ifa_ifp == ifp0) {
209 pserialize_read_exit(s);
210 return (looutput(ifp0, m, dst, rt));
211 }
212 pserialize_read_exit(s);
213 }
214
215 ifp = ifp->if_carpdev;
216 /* ac = (struct arpcom *)ifp; */
217
218 if ((ifp0->if_flags & (IFF_UP|IFF_RUNNING)) !=
219 (IFF_UP|IFF_RUNNING))
220 senderr(ENETDOWN);
221 }
222#endif /* NCARP > 0 */
223 if ((ifp->if_flags & (IFF_UP|IFF_RUNNING)) != (IFF_UP|IFF_RUNNING))
224 senderr(ENETDOWN);
225
226 /*
227 * If the queueing discipline needs packet classification,
228 * do it before prepending link headers.
229 */
230 IFQ_CLASSIFY(&ifp->if_snd, m, dst->sa_family);
231
232 switch (dst->sa_family) {
233
234#ifdef INET
235 case AF_INET: {
236 if (m->m_flags & M_BCAST)
237 memcpy(edst, fddibroadcastaddr, sizeof(edst));
238 else if (m->m_flags & M_MCAST) {
239 ETHER_MAP_IP_MULTICAST(&satocsin(dst)->sin_addr,
240 (char *)edst);
241 } else if ((error = arpresolve(ifp, rt, m, dst, edst,
242 sizeof(edst))) != 0)
243 return error == EWOULDBLOCK ? 0 : error;
244 /* If broadcasting on a simplex interface, loopback a copy */
245 if ((m->m_flags & M_BCAST) && (ifp->if_flags & IFF_SIMPLEX))
246 mcopy = m_copypacket(m, M_DONTWAIT);
247 etype = htons(ETHERTYPE_IP);
248 break;
249 }
250#endif
251#ifdef INET6
252 case AF_INET6:
253 if (m->m_flags & M_BCAST)
254 (void)memcpy(edst, fddibroadcastaddr, sizeof(edst));
255 else if (m->m_flags & M_MCAST) {
256 ETHER_MAP_IPV6_MULTICAST(&satocsin6(dst)->sin6_addr,
257 edst);
258 } else {
259 error = nd6_resolve(ifp, rt, m, dst, edst,
260 sizeof(edst));
261 if (error != 0) {
262 error = error == EWOULDBLOCK ? 0 : error;
263 return error;
264 }
265 }
266 etype = htons(ETHERTYPE_IPV6);
267 break;
268#endif
269#ifdef AF_ARP
270 case AF_ARP: {
271 struct arphdr *ah = mtod(m, struct arphdr *);
272 if (m->m_flags & M_BCAST)
273 memcpy(edst, etherbroadcastaddr, sizeof(edst));
274 else {
275 void *tha = ar_tha(ah);
276 if (tha == NULL) {
277 m_freem(m);
278 return 0;
279 }
280 memcpy(edst, tha, sizeof(edst));
281 }
282
283 ah->ar_hrd = htons(ARPHRD_ETHER);
284
285 switch (ntohs(ah->ar_op)) {
286 case ARPOP_REVREQUEST:
287 case ARPOP_REVREPLY:
288 etype = htons(ETHERTYPE_REVARP);
289 break;
290
291 case ARPOP_REQUEST:
292 case ARPOP_REPLY:
293 default:
294 etype = htons(ETHERTYPE_ARP);
295 }
296
297 break;
298 }
299#endif /* AF_ARP */
300#ifdef NETATALK
301 case AF_APPLETALK: {
302 struct at_ifaddr *aa;
303 struct ifaddr *ifa;
304 int s;
305
306 if (!aarpresolve(ifp, m, (const struct sockaddr_at *)dst, edst)) {
307#ifdef NETATALKDEBUG
308 printf("aarpresolv: failed\n");
309#endif
310 return (0);
311 }
312 /*
313 * ifaddr is the first thing in at_ifaddr
314 */
315 s = pserialize_read_enter();
316 ifa = at_ifawithnet((const struct sockaddr_at *)dst, ifp);
317 if (ifa == NULL) {
318 pserialize_read_exit(s);
319 goto bad;
320 }
321 aa = (struct at_ifaddr *)ifa;
322
323 /*
324 * In the phase 2 case, we need to prepend an mbuf for the llc
325 * header.
326 *
327 * XXX XXX: Do we need to preserve the value of m?
328 */
329 if (aa->aa_flags & AFA_PHASE2) {
330 struct llc llc;
331
332 M_PREPEND(m, sizeof(struct llc), M_NOWAIT);
333 if (m == NULL) {
334 pserialize_read_exit(s);
335 senderr(ENOBUFS);
336 }
337 llc.llc_dsap = llc.llc_ssap = LLC_SNAP_LSAP;
338 llc.llc_control = LLC_UI;
339 memcpy(llc.llc_snap_org_code, at_org_code,
340 sizeof(at_org_code));
341 llc.llc_snap_ether_type = htons(ETHERTYPE_ATALK);
342 memcpy(mtod(m, void *), &llc, sizeof(struct llc));
343 etype = 0;
344 } else {
345 etype = htons(ETHERTYPE_ATALK);
346 }
347 pserialize_read_exit(s);
348 break;
349 }
350#endif /* NETATALK */
351
352 case pseudo_AF_HDRCMPLT:
353 {
354 const struct fddi_header *fh1 =
355 (const struct fddi_header *)dst->sa_data;
356 hdrcmplt = 1;
357 memcpy(esrc, fh1->fddi_shost, sizeof (esrc));
358 /*FALLTHROUGH*/
359 }
360
361 /*FALLTHROUGH*/
362 case AF_LINK:
363 {
364 const struct fddi_header *fh1 =
365 (const struct fddi_header *)dst->sa_data;
366 memcpy(edst, fh1->fddi_dhost, sizeof (edst));
367 if (*edst & 1)
368 m->m_flags |= (M_BCAST|M_MCAST);
369 etype = 0;
370 break;
371 }
372
373 case AF_UNSPEC:
374 {
375 const struct ether_header *eh;
376 eh = (const struct ether_header *)dst->sa_data;
377 memcpy(edst, eh->ether_dhost, sizeof(edst));
378 if (*edst & 1)
379 m->m_flags |= (M_BCAST|M_MCAST);
380 etype = eh->ether_type;
381 break;
382 }
383
384 default:
385 printf("%s: can't handle af%d\n", ifp->if_xname,
386 dst->sa_family);
387 senderr(EAFNOSUPPORT);
388 }
389
390
391 if (mcopy)
392 (void) looutput(ifp, mcopy, dst, rt);
393 if (etype != 0) {
394 struct llc *l;
395 M_PREPEND(m, sizeof (struct llc), M_DONTWAIT);
396 if (m == 0)
397 senderr(ENOBUFS);
398 l = mtod(m, struct llc *);
399 l->llc_control = LLC_UI;
400 l->llc_dsap = l->llc_ssap = LLC_SNAP_LSAP;
401 l->llc_snap.org_code[0] = l->llc_snap.org_code[1] = l->llc_snap.org_code[2] = 0;
402 memcpy(&l->llc_snap.ether_type, &etype, sizeof(uint16_t));
403 }
404 /*
405 * Add local net header. If no space in first mbuf,
406 * allocate another.
407 */
408 M_PREPEND(m, sizeof (struct fddi_header), M_DONTWAIT);
409 if (m == 0)
410 senderr(ENOBUFS);
411 fh = mtod(m, struct fddi_header *);
412 fh->fddi_fc = FDDIFC_LLC_ASYNC|FDDIFC_LLC_PRIO4;
413 memcpy(fh->fddi_dhost, edst, sizeof (edst));
414 if (hdrcmplt)
415 memcpy(fh->fddi_shost, esrc, sizeof(fh->fddi_shost));
416 else
417 memcpy(fh->fddi_shost, CFDDIADDR(ifp), sizeof(fh->fddi_shost));
418
419#if NCARP > 0
420 if (ifp0 != ifp && ifp0->if_type == IFT_CARP) {
421 if_set_sadl(ifp0, fh->fddi_shost, sizeof(fh->fddi_shost),
422 false);
423 }
424
425 if (ifp != ifp0)
426 ifp0->if_obytes += m->m_pkthdr.len;
427#endif /* NCARP > 0 */
428 return ifq_enqueue(ifp, m);
429
430bad:
431 if (m)
432 m_freem(m);
433 return (error);
434}

436/*
437 * Process a received FDDI packet;
438 * the packet is in the mbuf chain m with
439 * the fddi header.
440 */
441static void
442fddi_input(struct ifnet *ifp, struct mbuf *m)
443{
444#if defined(INET) || defined(INET6)
445 pktqueue_t *pktq = NULL;
446#endif
447#if defined(NETATALK)
448 struct ifqueue *inq = NULL;
449 int isr = 0;
450#endif
451
452 struct llc *l;
453 struct fddi_header *fh;
454
455 MCLAIM(m, &((struct ethercom *)ifp)->ec_rx_mowner);
456 if ((ifp->if_flags & IFF_UP) == 0) {
457 m_freem(m);
458 return;
459 }
460
461 fh = mtod(m, struct fddi_header *);
462
463 ifp->if_ibytes += m->m_pkthdr.len;
464 if (fh->fddi_dhost[0] & 1) {
465 if (memcmp(fddibroadcastaddr, fh->fddi_dhost,
466 sizeof(fddibroadcastaddr)) == 0)
467 m->m_flags |= M_BCAST;
468 else
469 m->m_flags |= M_MCAST;
470 ifp->if_imcasts++;
471 } else if ((ifp->if_flags & IFF_PROMISC)
472 && memcmp(CFDDIADDR(ifp), (void *)fh->fddi_dhost,
473 sizeof(fh->fddi_dhost)) != 0) {
474 m_freem(m);
475 return;
476 }
477
478#ifdef M_LINK0
479 /*
480 * If this has a LLC priority of 0, then mark it so upper
481 * layers have a hint that it really came via a FDDI/Ethernet
482 * bridge.
483 */
484 if ((fh->fddi_fc & FDDIFC_LLC_PRIO7) == FDDIFC_LLC_PRIO0)
485 m->m_flags |= M_LINK0;
486#endif
487
488 l = (struct llc *)(fh+1);
489 switch (l->llc_dsap) {
490#if defined(INET) || defined(INET6) || defined(NETATALK)
491 case LLC_SNAP_LSAP:
492 {
493 uint16_t etype;
494 if (l->llc_control != LLC_UI || l->llc_ssap != LLC_SNAP_LSAP)
495 goto dropanyway;
496
497 /* Strip off the FDDI header. */
498 m_adj(m, sizeof(struct fddi_header));
499
500#ifdef NETATALK
501 if (memcmp(&(l->llc_snap_org_code)[0], at_org_code,
502 sizeof(at_org_code)) == 0 &&
503 ntohs(l->llc_snap_ether_type) == ETHERTYPE_ATALK) {
504 inq = &atintrq2;
505 m_adj( m, sizeof( struct llc ));
506 isr = NETISR_ATALK;
507 break;
508 }
509
510 if (memcmp(&(l->llc_snap_org_code)[0], aarp_org_code,
511 sizeof(aarp_org_code)) == 0 &&
512 ntohs(l->llc_snap_ether_type) == ETHERTYPE_AARP) {
513 m_adj( m, sizeof( struct llc ));
514 aarpinput(ifp, m); /* XXX */
515 return;
516 }
517#endif /* NETATALK */
518 if (l->llc_snap.org_code[0] != 0 || l->llc_snap.org_code[1] != 0|| l->llc_snap.org_code[2] != 0)
519 goto dropanyway;
520 etype = ntohs(l->llc_snap.ether_type);
521 m_adj(m, 8);
522#if NCARP > 0
523 if (ifp->if_carp && ifp->if_type != IFT_CARP &&
524 (carp_input(m, (uint8_t *)&fh->fddi_shost,
525 (uint8_t *)&fh->fddi_dhost, l->llc_snap.ether_type) == 0))
526 return;
527#endif
528
529 switch (etype) {
530#ifdef INET
531 case ETHERTYPE_IP:
532#ifdef GATEWAY
533 if (ipflow_fastforward(m))
534 return;
535#endif
536 pktq = ip_pktq;
537 break;
538
539 case ETHERTYPE_ARP:
540#if !defined(__bsdi__) || _BSDI_VERSION >= 199401
541#if defined(NETATALK)
542 isr = NETISR_ARP;
543 inq = &arpintrq;
544#endif
545 break;
546#else
547 arpinput(ifp, m);
548 return;
549#endif
550#endif
551#ifdef INET6
552 case ETHERTYPE_IPV6:
553#ifdef GATEWAY
554 if (ip6flow_fastforward(&m))
555 return;
556#endif
557 pktq = ip6_pktq;
558 break;
559
560#endif
561#ifdef NETATALK
562 case ETHERTYPE_ATALK:
563 isr = NETISR_ATALK;
564 inq = &atintrq1;
565 break;
566 case ETHERTYPE_AARP:
567 /* probably this should be done with a NETISR as well */
568 aarpinput(ifp, m); /* XXX */
569 return;
570#endif /* NETATALK */
571 default:
572 ifp->if_noproto++;
573 goto dropanyway;
574 }
575 break;
576 }
577#endif /* INET || NS */
578
579 default:
580 ifp->if_noproto++;
581#if defined(INET) || defined(INET6) || defined(NETATALK)
582 dropanyway:
583#endif
584 m_freem(m);
585 return;
586 }
587
588#if defined(INET) || defined(INET6)
589 if (__predict_true(pktq)) {
590 if (__predict_false(!pktq_enqueue(pktq, m, 0))) {
591 m_freem(m);
592 }
593 return;
594 }
595#endif
596#if defined(NETATALK)
597 if (inq == NULL) {
598 m_freem(m);
599 return;
600 }
601 IFQ_LOCK(inq);
602 if (IF_QFULL(inq)) {
603 IF_DROP(inq);
604 IFQ_UNLOCK(inq);
605 m_freem(m);
606 } else {
607 IF_ENQUEUE(inq, m);
608 IFQ_UNLOCK(inq);
609 schednetisr(isr);
610 }
611#endif
612}

614/*
615 * Perform common duties while attaching to interface list
616 */
617void
618fddi_ifattach(struct ifnet *ifp, void *lla)
619{
620 struct ethercom *ec = (struct ethercom *)ifp;
621
622 ifp->if_type = IFT_FDDI;
623 ifp->if_hdrlen = 21;
624 ifp->if_dlt = DLT_FDDI;
625 ifp->if_mtu = FDDIMTU;
626 ifp->if_output = fddi_output;
627 ifp->_if_input = fddi_input;
628 ifp->if_baudrate = IF_Mbps(100);
629
630 /*
631 * Update the max_linkhdr
632 */
633 if (ALIGN(ifp->if_hdrlen) > max_linkhdr)
634 max_linkhdr = ALIGN(ifp->if_hdrlen);
635
636 LIST_INIT(&ec->ec_multiaddrs);
637 if_set_sadl(ifp, lla, 6, true);
638
639 ifp->if_broadcastaddr = fddibroadcastaddr;
640 bpf_attach(ifp, DLT_FDDI, sizeof(struct fddi_header));
641#ifdef MBUFTRACE
642 strlcpy(ec->ec_tx_mowner.mo_name, ifp->if_xname,
643 sizeof(ec->ec_tx_mowner.mo_name));
644 strlcpy(ec->ec_tx_mowner.mo_descr, "tx",
645 sizeof(ec->ec_tx_mowner.mo_descr));
646 strlcpy(ec->ec_rx_mowner.mo_name, ifp->if_xname,
647 sizeof(ec->ec_rx_mowner.mo_name));
648 strlcpy(ec->ec_rx_mowner.mo_descr, "rx",
649 sizeof(ec->ec_rx_mowner.mo_descr));
650 MOWNER_ATTACH(&ec->ec_tx_mowner);
651 MOWNER_ATTACH(&ec->ec_rx_mowner);
652 ifp->if_mowner = &ec->ec_tx_mowner;
653#endif
654}
