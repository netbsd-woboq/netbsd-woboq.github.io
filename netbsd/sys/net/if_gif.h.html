92/*
93 * Get gif_variant from gif_softc.
94 *
95 * Never return NULL by contract.
96 * gif_variant itself is protected not to be freed by gv_psref.
97 * Once a reader dereference sc->sc_var by this API, the reader must not
98 * re-dereference form sc->sc_var.
99 */
100static __inline struct gif_variant *
101gif_getref_variant(struct gif_softc *sc, struct psref *psref)
102{
103 struct gif_variant *var;
104 int s;
105
106 s = pserialize_read_enter();
107 var = sc->gif_var;
108 KASSERT(var != NULL);
109 membar_datadep_consumer();
110 psref_acquire(psref, &var->gv_psref, gv_psref_class);
111 pserialize_read_exit(s);
112
113 return var;
114}
115
116static __inline void
117gif_putref_variant(struct gif_variant *var, struct psref *psref)
118{
119
120 KASSERT(var != NULL);
121 psref_release(psref, &var->gv_psref, gv_psref_class);
122}
123
124static __inline bool
125gif_heldref_variant(struct gif_variant *var)
126{
127
128 return psref_held(&var->gv_psref, gv_psref_class);
129}

140/*
141 * Locking notes:
142 * + gif_softc_list is protected by gif_softcs.lock (an adaptive mutex)
143 * gif_softc_list is list of all gif_softcs. It is used by ioctl
144 * context only.
145 * + gif_softc->gif_var is protected by
146 * - gif_softc->gif_lock (an adaptive mutex) for writer
147 * - gif_var->gv_psref for reader
148 * gif_softc->gif_var is used for variant values while the gif tunnel
149 * exists.
150 * + Each CPU's gif_ro.gr_ro of gif_ro_percpu are protected by
151 * percpu'ed gif_ro.gr_lock.
152 *
153 * Locking order:
154 * - encap_lock => gif_softc->gif_lock => gif_softcs.lock
155 */
