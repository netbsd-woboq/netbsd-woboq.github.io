33#ifndef _NET_IF_HIPPI_H_
34#define _NET_IF_HIPPI_H_
35
36#include <net/if_ether.h>
37
38struct hippi_fp {
39 uint8_t fp_ulp;
40 uint8_t fp_flags;
41#define HIPPI_FP_D1_PRESENT 0x80
42#define HIPPI_FP_D2_ON_BURST 0x40
43 uint16_t fp_offsets;
44#define HIPPI_FP_D2_MASK 0x07
45 uint32_t fp_d2_len;
46} __packed;
47
48struct hippi_le {
49 uint32_t le_dest_switch;
50 uint32_t le_src_switch;
51 uint16_t le_reserved;
52 uint8_t le_dest_addr[6];
53 uint16_t le_local_admin;
54 uint8_t le_src_addr[6];
55} __packed;
56
57struct hippi_header {
58 struct hippi_fp hi_fp;
59 struct hippi_le hi_le;
60} __packed;
61
62#define HIPPI_HDRLEN (sizeof(struct hippi_header))
63
64/* Link-layer ULP identifiers: */
65
66#define HIPPI_ULP_802 4
67#define HIPPI_ULP_IPI3_SLAVE 5
68#define HIPPI_ULP_IPI3_MASTER 6
69#define HIPPI_ULP_IPI3_PEER 7
70

78void hippi_ip_input(struct ifnet *, struct mbuf *);

80#endif /* !_NET_IF_HIPPI_H_ */
