77/*
78 * HIPPI output routine.
79 * Encapsulate a packet of type family for the local net.
80 * I don't know anything about the mapping of AppleTalk or OSI
81 * protocols to HIPPI, so I don't include any code for them.
82 */
83
84static int
85hippi_output(struct ifnet *ifp, struct mbuf *m0, const struct sockaddr *dst,
86 const struct rtentry *rt)
87{
88 uint16_t htype;
89 uint32_t ifield = 0;
90 int error = 0;
91 struct mbuf *m = m0;
92 struct hippi_header *hh;
93 uint32_t *cci;
94 uint32_t d2_len;
95
96 if ((ifp->if_flags & (IFF_UP|IFF_RUNNING)) != (IFF_UP|IFF_RUNNING))
97 senderr(ENETDOWN);
98
99 /* HIPPI doesn't really do broadcast or multicast right now */
100 if (m->m_flags & (M_BCAST | M_MCAST))
101 senderr(EOPNOTSUPP); /* XXX: some other error? */
102
103 /*
104 * If the queueing discipline needs packet classification,
105 * do it before prepending link headers.
106 */
107 IFQ_CLASSIFY(&ifp->if_snd, m, dst->sa_family);
108
109 switch (dst->sa_family) {
110#ifdef INET
111 case AF_INET:
112 if (rt) {
113 const struct sockaddr_dl *sdl =
114 satocsdl(rt->rt_gateway);
115 if (sdl->sdl_family == AF_LINK && sdl->sdl_alen != 0)
116 memcpy(&ifield, CLLADDR(sdl), sizeof(ifield));
117 }
118 if (!ifield) /* XXX: bogus check, but helps us get going */
119 senderr(EHOSTUNREACH);
120 htype = htons(ETHERTYPE_IP);
121 break;
122#endif
123
124#ifdef INET6
125 case AF_INET6:
126 if (rt) {
127 const struct sockaddr_dl *sdl =
128 satocsdl(rt->rt_gateway);
129 if (sdl->sdl_family == AF_LINK && sdl->sdl_alen != 0)
130 memcpy(&ifield, CLLADDR(sdl), sizeof(ifield));
131 }
132 if (!ifield) /* XXX: bogus check, but helps us get going */
133 senderr(EHOSTUNREACH);
134 htype = htons(ETHERTYPE_IPV6);
135 break;
136#endif
137
138 default:
139 printf("%s: can't handle af%d\n", ifp->if_xname,
140 dst->sa_family);
141 senderr(EAFNOSUPPORT);
142 }
143
144 if (htype != 0) {
145 struct llc *l;
146 M_PREPEND(m, sizeof (struct llc), M_DONTWAIT);
147 if (m == 0)
148 senderr(ENOBUFS);
149 l = mtod(m, struct llc *);
150 l->llc_control = LLC_UI;
151 l->llc_dsap = l->llc_ssap = LLC_SNAP_LSAP;
152 l->llc_snap.org_code[0] = l->llc_snap.org_code[1] =
153 l->llc_snap.org_code[2] = 0;
154 memcpy((void *) &l->llc_snap.ether_type, (void *) &htype,
155 sizeof(uint16_t));
156 }
157
158 d2_len = m->m_pkthdr.len;
159
160 /*
161 * Add local net header. If no space in first mbuf,
162 * allocate another.
163 */
164
165 M_PREPEND(m, sizeof (struct hippi_header) + 8, M_DONTWAIT);
166 if (m == 0)
167 senderr(ENOBUFS);
168 cci = mtod(m, uint32_t *);
169 memset(cci, 0, sizeof(struct hippi_header) + 8);
170 cci[0] = 0;
171 cci[1] = ifield;
172 hh = (struct hippi_header *) &cci[2];
173 hh->hi_fp.fp_ulp = HIPPI_ULP_802;
174 hh->hi_fp.fp_flags = HIPPI_FP_D1_PRESENT;
175 hh->hi_fp.fp_offsets = htons(sizeof(struct hippi_le));
176 hh->hi_fp.fp_d2_len = htonl(d2_len);
177
178 /* Pad out the D2 area to end on a quadword (64-bit) boundry. */
179
180 if (d2_len % 8 != 0) {
181 static uint32_t buffer[2] = {0, 0};
182 m_copyback(m, m->m_pkthdr.len, 8 - d2_len % 8, (void *) buffer);
183 }
184
185 return ifq_enqueue(ifp, m);
186
187 bad:
188 if (m)
189 m_freem(m);
190 return (error);
191}

193/*
194 * Process a received HIPPI packet;
195 * the packet is in the mbuf chain m with
196 * the HIPPI header.
197 */
198
199static void
200hippi_input(struct ifnet *ifp, struct mbuf *m)
201{
202 pktqueue_t *pktq;
203 struct llc *l;
204 uint16_t htype;
205 struct hippi_header *hh;
206
207 if ((ifp->if_flags & IFF_UP) == 0) {
208 m_freem(m);
209 return;
210 }
211
212 /* XXX: need to check flags and drop if bogus! */
213
214 hh = mtod(m, struct hippi_header *);
215
216 ifp->if_ibytes += m->m_pkthdr.len;
217 if (hh->hi_le.le_dest_addr[0] & 1) {
218 if (memcmp(etherbroadcastaddr, hh->hi_le.le_dest_addr,
219 sizeof(etherbroadcastaddr)) == 0)
220 m->m_flags |= M_BCAST;
221 else
222 m->m_flags |= M_MCAST;
223 }
224 if (m->m_flags & (M_BCAST|M_MCAST))
225 ifp->if_imcasts++;
226
227 /* Skip past the HIPPI header. */
228 m_adj(m, sizeof(struct hippi_header));
229
230 l = mtod(m, struct llc *);
231 if (l->llc_dsap != LLC_SNAP_LSAP) {
232 m_freem(m);
233 return;
234 }
235 htype = ntohs(l->llc_snap.ether_type);
236 m_adj(m, 8);
237 switch (htype) {
238#ifdef INET
239 case ETHERTYPE_IP:
240 pktq = ip_pktq;
241 break;
242#endif
243#ifdef INET6
244 case ETHERTYPE_IPV6:
245 pktq = ip6_pktq;
246 break;
247#endif
248 default:
249 m_freem(m);
250 return;
251 }
252
253 if (__predict_false(!pktq_enqueue(pktq, m, 0))) {
254 m_freem(m);
255 }
256}

258/*
259 * Handle packet from HIPPI that has no MAC header
260 */
261
262#ifdef INET
263void
264hippi_ip_input(struct ifnet *ifp, struct mbuf *m)
265{
266 int s;
267
268 s = splnet();
269 if (__predict_false(!pktq_enqueue(ip_pktq, m, 0))) {
270 m_freem(m);
271 }
272 splx(s);
273}
274#endif

276/*
277 * Perform common duties while attaching to interface list
278 */
279void
280hippi_ifattach(struct ifnet *ifp, void *lla)
281{
282
283 ifp->if_type = IFT_HIPPI;
284 ifp->if_hdrlen = sizeof(struct hippi_header) + 8; /* add CCI */
285 ifp->if_dlt = DLT_HIPPI;
286 ifp->if_mtu = HIPPIMTU;
287 ifp->if_output = hippi_output;
288 ifp->_if_input = hippi_input;
289 ifp->if_baudrate = IF_Mbps(800); /* XXX double-check */
290
291 if_set_sadl(ifp, lla, 6, true);
292
293 bpf_attach(ifp, DLT_HIPPI, sizeof(struct hippi_header));
294}
