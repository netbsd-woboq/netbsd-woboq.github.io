32#ifndef _NET_IF_IEEE1394_H_
33#define _NET_IF_IEEE1394_H_
34
35/* hardware address information for arp / nd */
36struct ieee1394_hwaddr {
37 uint8_t iha_uid[8]; /* node unique ID */
38 uint8_t iha_maxrec; /* max_rec in the config ROM */
39 uint8_t iha_speed; /* min of link/PHY speed */
40 uint8_t iha_offset[6]; /* unicast FIFO address */
41};
42
43/*
44 * BPF wants to see one of these.
45 */
46struct ieee1394_bpfhdr {
47 uint8_t ibh_dhost[8];
48 uint8_t ibh_shost[8];
49 uint16_t ibh_type;
50};
51
52#ifdef _KERNEL
53
54/* pseudo header */
55struct ieee1394_header {
56 uint8_t ih_uid[8]; /* dst/src uid */
57 uint8_t ih_maxrec; /* dst maxrec for tx */
58 uint8_t ih_speed; /* speed */
59 uint8_t ih_offset[6]; /* dst offset */
60};
61
62/* unfragment encapsulation header */
63struct ieee1394_unfraghdr {
64 uint16_t iuh_ft; /* fragment type == 0 */
65 uint16_t iuh_etype; /* ether_type */
66};
67
68/* fragmented encapsulation header */
69struct ieee1394_fraghdr {
70 uint16_t ifh_ft_size; /* fragment type, data size-1 */
71 uint16_t ifh_etype_off; /* etype for first fragment */
72 /* offset for subseq frag */
73 uint16_t ifh_dgl; /* datagram label */
74 uint16_t ifh_reserved;
75};
76
77#define IEEE1394_FT_SUBSEQ 0x8000
78#define IEEE1394_FT_MORE 0x4000
79
80#define IEEE1394MTU 1500
81
82#define IEEE1394_GASP_LEN 8 /* GASP header for Stream */
83#define IEEE1394_ADDR_LEN 8
84#define IEEE1394_CRC_LEN 4
85
86struct ieee1394_reass_pkt {
87 LIST_ENTRY(ieee1394_reass_pkt) rp_next;
88 struct mbuf *rp_m;
89 uint16_t rp_size;
90 uint16_t rp_etype;
91 uint16_t rp_off;
92 uint16_t rp_dgl;
93 uint16_t rp_len;
94 uint16_t rp_ttl;
95};
96
97struct ieee1394_reassq {
98 LIST_ENTRY(ieee1394_reassq) rq_node;
99 LIST_HEAD(, ieee1394_reass_pkt) rq_pkt;
100 uint32_t fr_id;
101};
102
103struct ieee1394com {
104 struct ifnet fc_if;
105 struct ieee1394_hwaddr ic_hwaddr;
106 uint16_t ic_dgl;
107 LIST_HEAD(, ieee1394_reassq) ic_reassq;
108};
