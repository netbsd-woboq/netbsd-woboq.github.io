34#ifndef _NET_IF_LLC_H_
35#define _NET_IF_LLC_H_
36
37/*
38 * IEEE 802.2 Link Level Control headers, for use in conjunction with
39 * 802.{3,4,5} media access control methods.
40 *
41 * Headers here do not use bit fields due to shortcommings in many
42 * compilers.
43 */
44
45struct llc {
46 uint8_t llc_dsap;
47 uint8_t llc_ssap;
48 union {
49 struct {
50 uint8_t control;
51 uint8_t format_id;
52 uint8_t class_u;
53 uint8_t window_x2;
54 } type_u /* XXX __packed ??? */;
55 struct {
56 uint8_t num_snd_x2;
57 uint8_t num_rcv_x2;
58 } type_i /* XXX __packed ??? */;
59 struct {
60 uint8_t control;
61 uint8_t num_rcv_x2;
62 } type_s /* XXX __packed ??? */;
63 struct {
64 uint8_t control;
65 /*
66 * We cannot put the following fields in a structure because
67 * the structure rounding might cause padding.
68 */
69 uint8_t frmr_rej_pdu0;
70 uint8_t frmr_rej_pdu1;
71 uint8_t frmr_control;
72 uint8_t frmr_control_ext;
73 uint8_t frmr_cause;
74 } type_frmr /* XXX __packed ??? */;
75 struct {
76 uint8_t control;
77 uint8_t org_code[3];
78 uint16_t ether_type;
79 } type_snap __packed;
80 struct {
81 uint8_t control;
82 uint8_t control_ext;
83 } type_raw /* XXX __packed ??? */;
84 } llc_un /* XXX __packed ??? */;
85} __packed;
86
87struct frmrinfo {
88 uint8_t frmr_rej_pdu0;
89 uint8_t frmr_rej_pdu1;
90 uint8_t frmr_control;
91 uint8_t frmr_control_ext;
92 uint8_t frmr_cause;
93} __packed;
94
95#define llc_control llc_un.type_u.control
96#define llc_control_ext llc_un.type_raw.control_ext
97#define llc_fid llc_un.type_u.format_id
98#define llc_class llc_un.type_u.class_u
99#define llc_window llc_un.type_u.window_x2
100#define llc_frmrinfo llc_un.type_frmr.frmr_rej_pdu0
101#define llc_frmr_pdu0 llc_un.type_frmr.frmr_rej_pdu0
102#define llc_frmr_pdu1 llc_un.type_frmr.frmr_rej_pdu1
103#define llc_frmr_control llc_un.type_frmr.frmr_control
104#define llc_frmr_control_ext llc_un.type_frmr.frmr_control_ext
105#define llc_frmr_cause llc_un.type_frmr.frmr_cause
106#define llc_snap llc_un.type_snap
107
108/*
109 * Don't use sizeof(struct llc_un) for LLC header sizes
110 */
111#define LLC_ISFRAMELEN 4
112#define LLC_UFRAMELEN 3
113#define LLC_FRMRLEN 7
114#define LLC_SNAPFRAMELEN 8
115
116/*
117 * Unnumbered LLC format commands
118 */
119#define LLC_UI 0x3
120#define LLC_UI_P 0x13
121#define LLC_DISC 0x43
122#define LLC_DISC_P 0x53
123#define LLC_UA 0x63
124#define LLC_UA_P 0x73
125#define LLC_TEST 0xe3
126#define LLC_TEST_P 0xf3
127#define LLC_FRMR 0x87
128#define LLC_FRMR_P 0x97
129#define LLC_DM 0x0f
130#define LLC_DM_P 0x1f
131#define LLC_XID 0xaf
132#define LLC_XID_P 0xbf
133#define LLC_SABME 0x6f
134#define LLC_SABME_P 0x7f
135
136/*
137 * Supervisory LLC commands
138 */
139#define LLC_RR 0x01
140#define LLC_RNR 0x05
141#define LLC_REJ 0x09
142
143/*
144 * Info format - dummy only
145 */
146#define LLC_INFO 0x00
147
148/*
149 * ISO PDTR 10178 contains among others
150 */
151#define LLC_8021D_LSAP 0x42
152#define LLC_X25_LSAP 0x7e
153#define LLC_SNAP_LSAP 0xaa
154#define LLC_ISO_LSAP 0xfe
155
156/*
157 * LLC XID definitions from 802.2, as needed
158 */
159
160#define LLC_XID_FORMAT_BASIC 0x81
161#define LLC_XID_BASIC_MINLEN (LLC_UFRAMELEN + 3)
162
163#define LLC_XID_CLASS_I 0x1
164#define LLC_XID_CLASS_II 0x3
165#define LLC_XID_CLASS_III 0x5
166#define LLC_XID_CLASS_IV 0x7

169#endif /* !_NET_IF_LLC_H_ */
