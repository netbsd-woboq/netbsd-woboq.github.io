32#ifndef _NET_IF_PPPOE_H_
33#define _NET_IF_PPPOE_H_
34
35#include <sys/ioccom.h>
36
37struct pppoediscparms {
38 char ifname[IFNAMSIZ]; /* pppoe interface name */
39 char eth_ifname[IFNAMSIZ]; /* external ethernet interface name */
40 const char *ac_name; /* access concentrator name (or NULL) */
41 size_t ac_name_len; /* on write: length of buffer for ac_name */
42 const char *service_name; /* service name (or NULL) */
43 size_t service_name_len; /* on write: length of buffer for service name */
44};
45
46#define PPPOESETPARMS _IOW('i', 110, struct pppoediscparms)
47#define PPPOEGETPARMS _IOWR('i', 111, struct pppoediscparms)
48
49#define PPPOE_STATE_INITIAL 0
50#define PPPOE_STATE_PADI_SENT 1
51#define PPPOE_STATE_PADR_SENT 2
52#define PPPOE_STATE_SESSION 3
53#define PPPOE_STATE_CLOSING 4
54/* passive */
55#define PPPOE_STATE_PADO_SENT 1
56
57struct pppoeconnectionstate {
58 char ifname[IFNAMSIZ]; /* pppoe interface name */
59 u_int state; /* one of the PPPOE_STATE_ states above */
60 u_int session_id; /* if state == PPPOE_STATE_SESSION */
61 u_int padi_retry_no; /* number of retries already sent */
62 u_int padr_retry_no;
63};
64
65#define PPPOEGETSESSION _IOWR('i', 112, struct pppoeconnectionstate)
66
67#ifdef _KERNEL
68
69void pppoe_input(struct ifnet *, struct mbuf *);
70void pppoedisc_input(struct ifnet *, struct mbuf *);
71#endif /* _KERNEL */
72/*
73 * Locking notes:
74 * + pppoe_softc_list is protected by pppoe_softc_list_lock (an rwlock)
75 * pppoe_softc_list is a list of all pppoe_softc, and it is used to
76 * find pppoe interface by session id or host unique tag.
77 * + pppoe_softc is protected by pppoe_softc->sc_lock (an rwlock)
78 * pppoe_softc holds session id and parameters to establish the id
79 *
80 * Locking order:
81 * - pppoe_softc_list_lock => pppoe_softc->sc_lock
82 */
83#endif /* !_NET_IF_PPPOE_H_ */
