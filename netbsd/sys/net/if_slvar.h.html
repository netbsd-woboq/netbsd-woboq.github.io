34#ifndef _NET_IF_SLVAR_H_
35#define _NET_IF_SLVAR_H_
36
37/*
38 * Definitions for SLIP interface data structures
39 *
40 * (This exists so programs like slstats can get at the definition
41 * of sl_softc.)
42 */
43struct sl_softc {
44 struct ifnet sc_if; /* network-visible interface */
45 int sc_unit; /* XXX unit number */
46 struct ifqueue sc_fastq; /* interactive output queue */
47 struct ifqueue sc_inq; /* input queue */
48 struct tty *sc_ttyp; /* pointer to tty structure */
49 u_char *sc_mp; /* pointer to next available buf char */
50 u_char *sc_ep; /* pointer to last available buf char */
51 u_char *sc_pktstart; /* pointer to beginning of packet */
52 struct mbuf *sc_mbuf; /* input buffer */
53 u_int sc_flags; /* see below */
54 u_int sc_escape; /* =1 if last char input was FRAME_ESCAPE */
55 long sc_lasttime; /* last time a char arrived */
56 long sc_abortcount; /* number of abort esacpe chars */
57 long sc_starttime; /* time of first abort in window */
58 long sc_oqlen; /* previous output queue size */
59 long sc_otimeout; /* number of times output's stalled */
60 void *sc_si; /* softintr handle */
61#ifdef __NetBSD__
62 int sc_oldbufsize; /* previous output buffer size */
63 int sc_oldbufquot; /* previous output buffer quoting */
64#endif
65#ifdef INET /* XXX */
66 struct slcompress sc_comp; /* tcp compression data */
67#endif
68 struct bintime sc_lastpacket; /* for watchdog */
69 LIST_ENTRY(sl_softc) sc_iflist;
70};
71
72/* internal flags */
73#define SC_ERROR 0x0001 /* had an input error */
74
75/* visible flags */
76#define SC_COMPRESS IFF_LINK0 /* compress TCP traffic */
77#define SC_NOICMP IFF_LINK1 /* suppress ICMP traffic */
78#define SC_AUTOCOMP IFF_LINK2 /* auto-enable TCP compression */
79
80#endif /* _NET_IF_SLVAR_H_ */
