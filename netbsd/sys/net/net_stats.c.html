32#include <sys/cdefs.h>
33__KERNEL_RCSID(0, "$NetBSD: net_stats.c,v 1.5 2017/06/01 02:45:14 chs Exp $");
34
35#include <sys/types.h>
36#include <sys/systm.h>
37#include <sys/sysctl.h>
38#include <sys/kmem.h>
39
40#include <net/net_stats.h>
41
42typedef struct {
43 uint64_t *ctx_counters; /* pointer to collated counter array */
44 u_int ctx_ncounters; /* number of counters in array */
45} netstat_sysctl_context;
46
47/*
48 * netstat_convert_to_user_cb --
49 * Internal routine used as a call-back for percpu data enumeration.
50 */
51static void
52netstat_convert_to_user_cb(void *v1, void *v2, struct cpu_info *ci)
53{
54 const uint64_t *local_counters = v1;
55 netstat_sysctl_context *ctx = v2;
56 u_int i;
57
58 for (i = 0; i < ctx->ctx_ncounters; i++)
59 ctx->ctx_counters[i] += local_counters[i];
60}
61
62/*
63 * netstat_sysctl --
64 * Common routine for collating and reporting network statistics
65 * that are gathered per-CPU. Statistics counters are assumed
66 * to be arrays of uint64_t's.
67 */
68int
69netstat_sysctl(percpu_t *stat, u_int ncounters, SYSCTLFN_ARGS)
70{
71 netstat_sysctl_context ctx;
72 struct sysctlnode node;
73 uint64_t *counters;
74 size_t countersize;
75 int rv;
76
77 countersize = sizeof(uint64_t) * ncounters;
78
79 counters = kmem_zalloc(countersize, KM_SLEEP);
80 ctx.ctx_counters = counters;
81 ctx.ctx_ncounters = ncounters;
82
83 percpu_foreach(stat, netstat_convert_to_user_cb, &ctx);
84
85 node = *rnode;
86 node.sysctl_data = counters;
87 node.sysctl_size = countersize;
88 rv = sysctl_lookup(SYSCTLFN_CALL(&node));
89
90 kmem_free(counters, countersize);
91
92 return (rv);
93}
