37/*
38 * The networking code runs off software interrupts.
39 *
40 * You can switch into the network by doing splsoftnet() and return by splx().
41 * The software interrupt level for the network is higher than the software
42 * level for the clock (so you can enter the network in routines called
43 * at timeout time).
44 *
45 * The routine to request a network software interrupt, setsoftnet(),
46 * is defined in the machine-specific include files.
47 */
48

53#include "opt_atalk.h"
54#include "opt_mpls.h"
55#include "opt_can.h"
56#include "arp.h"
57#endif /* defined(_KERNEL_OPT) */

99/*
100 * Each ``pup-level-1'' input queue has a bit in a ``netisr'' status
101 * word which is used to de-multiplex a single software
102 * interrupt used for scheduling the network code to calls
103 * on the lowest level routine of each protocol.
104 */
105#define NETISR_IP 2 /* same as AF_INET */
106#define NETISR_CCITT 10 /* same as AF_CCITT */
107#define NETISR_ATALK 16 /* same as AF_APPLETALK */
108#define NETISR_IPV6 24 /* same as AF_INET6 */
109#define NETISR_ISDN 26 /* same as AF_E164 */
110#define NETISR_NATM 27 /* same as AF_NATM */
111#define NETISR_ARP 28 /* same as AF_ARP */
112#define NETISR_MPLS 33 /* same as AF_MPLS */
113#define NETISR_CAN 35 /* same as AF_CAN */
114#define NETISR_MAX AF_MAX
