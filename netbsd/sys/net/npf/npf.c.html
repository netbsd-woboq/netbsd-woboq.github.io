49static __read_mostly npf_t * npf_kernel_ctx = NULL;
50
51__dso_public int
52npf_sysinit(unsigned nworkers)
53{
54 npf_bpf_sysinit();
55 npf_tableset_sysinit();
56 npf_nat_sysinit();
57 return npf_worker_sysinit(nworkers);
58}
59
60__dso_public void
61npf_sysfini(void)
62{
63 npf_worker_sysfini();
64 npf_nat_sysfini();
65 npf_tableset_sysfini();
66 npf_bpf_sysfini();
67}
68
69__dso_public npf_t *
70npf_create(int flags, const npf_mbufops_t *mbufops, const npf_ifops_t *ifops)
71{
72 npf_t *npf;
73
74 npf = kmem_zalloc(sizeof(npf_t), KM_SLEEP);
75 npf->qsbr = pserialize_create();
76 npf->stats_percpu = percpu_alloc(NPF_STATS_SIZE);
77 npf->mbufops = mbufops;
78
79 npf_ifmap_init(npf, ifops);
80 npf_conn_init(npf, flags);
81 npf_alg_init(npf);
82 npf_ext_init(npf);
83
84 /* Load an empty configuration. */
85 npf_config_init(npf);
86 return npf;
87}
88
89__dso_public void
90npf_destroy(npf_t *npf)
91{
92 /*
93 * Destroy the current configuration. Note: at this point all
94 * handlers must be deactivated; we will drain any processing.
95 */
96 npf_config_fini(npf);
97
98 /* Finally, safe to destroy the subsystems. */
99 npf_ext_fini(npf);
100 npf_alg_fini(npf);
101 npf_conn_fini(npf);
102 npf_ifmap_fini(npf);
103
104 pserialize_destroy(npf->qsbr);
105 percpu_free(npf->stats_percpu, NPF_STATS_SIZE);
106 kmem_free(npf, sizeof(npf_t));
107}

146/*
147 * NPF statistics interface.
148 */
149
150void
151npf_stats_inc(npf_t *npf, npf_stats_t st)
152{
153 uint64_t *stats = percpu_getref(npf->stats_percpu);
154 stats[st]++;
155 percpu_putref(npf->stats_percpu);
156}
157
158void
159npf_stats_dec(npf_t *npf, npf_stats_t st)
160{
161 uint64_t *stats = percpu_getref(npf->stats_percpu);
162 stats[st]--;
163 percpu_putref(npf->stats_percpu);
164}
165
166static void
167npf_stats_collect(void *mem, void *arg, struct cpu_info *ci)
168{
169 uint64_t *percpu_stats = mem, *full_stats = arg;
170
171 for (unsigned i = 0; i < NPF_STATS_COUNT; i++) {
172 full_stats[i] += percpu_stats[i];
173 }
174}
175
176/*
177 * npf_stats: export collected statistics.
178 */
179__dso_public void
180npf_stats(npf_t *npf, uint64_t *buf)
181{
182 memset(buf, 0, NPF_STATS_SIZE);
183 percpu_foreach(npf->stats_percpu, npf_stats_collect, buf);
184}
