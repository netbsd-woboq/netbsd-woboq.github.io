37#include <sys/types.h>
38
39#include "npf_impl.h"
40
41typedef struct npf_connkey npf_connkey_t;
42
43#if defined(__NPF_CONN_PRIVATE)
44
45/*
46 * See npf_conn_conkey() function for the key layout description.
47 */
48#define NPF_CONN_NKEYWORDS (2 + ((sizeof(npf_addr_t) * 2) >> 2))
49#define NPF_CONN_GETALEN(key) ((key)->ck_key[0] & 0xffff)
50#define NPF_CONN_KEYLEN(key) (8 + (2 * NPF_CONN_GETALEN(key)))
51
52struct npf_connkey {
53 /*
54 * Back-pointer to the actual connection and the compacted key.
55 */
56 npf_conn_t * ck_backptr;
57 uint32_t ck_key[NPF_CONN_NKEYWORDS];
58};
59
60/*
61 * The main connection tracking structure.
62 */
63
64struct npf_conn {
65 /*
66 * Connection "forwards" and "backwards" entries, plus the
67 * interface ID (if zero, then the state is global) and flags.
68 */
69 npf_connkey_t c_forw_entry;
70 npf_connkey_t c_back_entry;
71 u_int c_proto;
72 u_int c_ifid;
73 u_int c_flags;
74
75 /*
76 * Entry in the connection database/list. The entry is
77 * protected by npf_t::conn_lock.
78 */
79 union {
80 npf_conn_t * c_next;
81 LIST_ENTRY(npf_conn) c_entry;
82 };
83
84 /* Associated rule procedure or NAT (if any). */
85 npf_rproc_t * c_rproc;
86 npf_nat_t * c_nat;
87
88 /*
89 * The protocol state, reference count and the last activity
90 * time (used to calculate expiration time).
91 */
92 kmutex_t c_lock;
93 npf_state_t c_state;
94 u_int c_refcnt;
95 uint64_t c_atime;
96
97 /*
98 * Save the matching rule ID and flags.
99 */
100 uint64_t c_rid;
101 u_int c_retfl;
102};
103
104#endif
