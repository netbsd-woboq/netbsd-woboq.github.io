50#define NPF_GC_STEP 256
51
52struct npf_conndb {
53 thmap_t * cd_map;
54
55 /*
56 * There are three lists for connections: new, all and G/C.
57 *
58 * New connections are atomically inserted into the "new-list".
59 * The G/C worker will move them to the doubly-linked list of all
60 * active connections.
61 */
62 npf_conn_t * cd_new;
63 LIST_HEAD(, npf_conn) cd_list;
64 LIST_HEAD(, npf_conn) cd_gclist;
65
66 /* The last inspected connection (for circular iteration). */
67 npf_conn_t * cd_marker;
68};
69
70npf_conndb_t *
71npf_conndb_create(void)
72{
73 npf_conndb_t *cd;
74
75 cd = kmem_zalloc(sizeof(npf_conndb_t), KM_SLEEP);
76 cd->cd_map = thmap_create(0, NULL, THMAP_NOCOPY);
77 KASSERT(cd->cd_map != NULL);
78
79 LIST_INIT(&cd->cd_list);
80 LIST_INIT(&cd->cd_gclist);
81 return cd;
82}
83
84void
85npf_conndb_destroy(npf_conndb_t *cd)
86{
87 KASSERT(cd->cd_new == NULL);
88 KASSERT(cd->cd_marker == NULL);
89 KASSERT(LIST_EMPTY(&cd->cd_list));
90 KASSERT(LIST_EMPTY(&cd->cd_gclist));
91
92 thmap_destroy(cd->cd_map);
93 kmem_free(cd, sizeof(npf_conndb_t));
94}

96/*
97 * npf_conndb_lookup: find a connection given the key.
98 */
99npf_conn_t *
100npf_conndb_lookup(npf_conndb_t *cd, const npf_connkey_t *ck, bool *forw)
101{
102 const unsigned keylen = NPF_CONN_KEYLEN(ck);
103 npf_connkey_t *foundkey;
104 npf_conn_t *con;
105
106 /*
107 * Lookup the connection key in the key-value map.
108 */
109 foundkey = thmap_get(cd->cd_map, ck->ck_key, keylen);
110 if (!foundkey) {
111 return NULL;
112 }
113 con = foundkey->ck_backptr;
114 KASSERT(con != NULL);
115
116 /*
117 * Acquire a reference and return the connection.
118 */
119 atomic_inc_uint(&con->c_refcnt);
120 *forw = (foundkey == &con->c_forw_entry);
121 return con;
122}
123
124/*
125 * npf_conndb_insert: insert the key representing the connection.
126 *
127 * => Returns true on success and false on failure.
128 */
129bool
130npf_conndb_insert(npf_conndb_t *cd, npf_connkey_t *ck)
131{
132 const unsigned keylen = NPF_CONN_KEYLEN(ck);
133 return thmap_put(cd->cd_map, ck->ck_key, keylen, ck) == ck;
134}
135
136/*
137 * npf_conndb_remove: find and delete connection key, returning the
138 * connection it represents.
139 */
140npf_conn_t *
141npf_conndb_remove(npf_conndb_t *cd, npf_connkey_t *ck)
142{
143 const unsigned keylen = NPF_CONN_KEYLEN(ck);
144 npf_connkey_t *foundkey;
145 npf_conn_t *con;
146
147 foundkey = thmap_del(cd->cd_map, ck->ck_key, keylen);
148 if (!foundkey) {
149 return NULL;
150 }
151 con = foundkey->ck_backptr;
152 KASSERT(con != NULL);
153 return con;
154}

156/*
157 * npf_conndb_enqueue: atomically insert the connection into the
158 * singly-linked list of the "new" connections.
159 */
160void
161npf_conndb_enqueue(npf_conndb_t *cd, npf_conn_t *con)
162{
163 npf_conn_t *head;
164
165 do {
166 head = cd->cd_new;
167 con->c_next = head;
168 } while (atomic_cas_ptr(&cd->cd_new, head, con) != head);
169}
170
171/*
172 * npf_conndb_update: migrate all new connections to the list of all
173 * connections; this must also be performed on npf_conndb_getlist()
174 * to provide a complete list of connections.
175 */
176static void
177npf_conndb_update(npf_conndb_t *cd)
178{
179 npf_conn_t *con;
180
181 con = atomic_swap_ptr(&cd->cd_new, NULL);
182 while (con) {
183 npf_conn_t *next = con->c_next; // union
184 LIST_INSERT_HEAD(&cd->cd_list, con, c_entry);
185 con = next;
186 }
187}
188
189/*
190 * npf_conndb_getlist: return the list of all connections.
191 */
192npf_conn_t *
193npf_conndb_getlist(npf_conndb_t *cd)
194{
195 npf_conndb_update(cd);
196 return LIST_FIRST(&cd->cd_list);
197}
198
199/*
200 * npf_conndb_getnext: return the next connection, implementing
201 * the circular iteration.
202 */
203npf_conn_t *
204npf_conndb_getnext(npf_conndb_t *cd, npf_conn_t *con)
205{
206 /* Next.. */
207 if (con == NULL || (con = LIST_NEXT(con, c_entry)) == NULL) {
208 con = LIST_FIRST(&cd->cd_list);
209 }
210 return con;
211}

213/*
214 * npf_conndb_gc_incr: incremental G/C of the expired connections.
215 */
216static void
217npf_conndb_gc_incr(npf_conndb_t *cd, const time_t now)
218{
219 unsigned target = NPF_GC_STEP;
220 npf_conn_t *con;
221
222 /*
223 * Second, start from the "last" (marker) connection.
224 * We must initialise the marker if it is not set yet.
225 */
226 if ((con = cd->cd_marker) == NULL) {
227 con = npf_conndb_getnext(cd, NULL);
228 cd->cd_marker = con;
229 }
230
231 /*
232 * Scan the connections:
233 * - Limit the scan to the G/C step size.
234 * - Stop if we scanned all of them.
235 * - Update the marker connection.
236 */
237 while (con && target--) {
238 npf_conn_t *next = npf_conndb_getnext(cd, con);
239
240 /*
241 * Can we G/C this connection?
242 */
243 if (npf_conn_expired(con, now)) {
244 /* Yes: move to the G/C list. */
245 LIST_REMOVE(con, c_entry);
246 LIST_INSERT_HEAD(&cd->cd_gclist, con, c_entry);
247 npf_conn_remove(cd, con);
248
249 /* This connection cannot be a new marker anymore. */
250 if (con == next) {
251 next = NULL;
252 }
253 if (con == cd->cd_marker) {
254 cd->cd_marker = next;
255 con = next;
256 continue;
257 }
258 }
259 con = next;
260
261 /*
262 * Circular iteration: if we returned back to the
263 * marker connection, then stop.
264 */
265 if (con == cd->cd_marker) {
266 break;
267 }
268 }
269 cd->cd_marker = con;
270}

272/*
273 * npf_conndb_gc: garbage collect the expired connections.
274 *
275 * => Must run in a single-threaded manner.
276 * => If 'flush' is true, then destroy all connections.
277 * => If 'sync' is true, then perform passive serialisation.
278 */
279void
280npf_conndb_gc(npf_t *npf, npf_conndb_t *cd, bool flush, bool sync)
281{
282 struct timespec tsnow;
283 npf_conn_t *con;
284 void *gcref;
285
286 getnanouptime(&tsnow);
287
288 /* First, migrate all new connections. */
289 mutex_enter(&npf->conn_lock);
290 npf_conndb_update(cd);
291 if (flush) {
292 /* Just unlink and move all connections to the G/C list. */
293 while ((con = LIST_FIRST(&cd->cd_list)) != NULL) {
294 LIST_REMOVE(con, c_entry);
295 LIST_INSERT_HEAD(&cd->cd_gclist, con, c_entry);
296 npf_conn_remove(cd, con);
297 }
298 cd->cd_marker = NULL;
299 } else {
300 /* Incremental G/C of the expired connections. */
301 npf_conndb_gc_incr(cd, tsnow.tv_sec);
302 }
303 mutex_exit(&npf->conn_lock);
304
305 /*
306 * Ensure it is safe to destroy the connections.
307 * Note: drop the conn_lock (see the lock order).
308 */
309 gcref = thmap_stage_gc(cd->cd_map);
310 if (sync) {
311 npf_config_enter(npf);
312 npf_config_sync(npf);
313 npf_config_exit(npf);
314 }
315 thmap_gc(cd->cd_map, gcref);
316
317 /*
318 * If there is nothing to G/C, then reduce the worker interval.
319 * We do not go below the lower watermark.
320 */
321 if (LIST_EMPTY(&cd->cd_gclist)) {
322 // TODO: cd->next_gc = MAX(cd->next_gc >> 1, NPF_MIN_GC_TIME);
323 return;
324 }
325
326 /*
327 * Garbage collect all expired connections.
328 * May need to wait for the references to drain.
329 */
330 while ((con = LIST_FIRST(&cd->cd_gclist)) != NULL) {
331 /*
332 * Destroy only if removed and no references. Otherwise,
333 * just do it next time, unless we are destroying all.
334 */
335 if (__predict_false(con->c_refcnt)) {
336 if (!flush) {
337 break;
338 }
339 kpause("npfcongc", false, 1, NULL);
340 continue;
341 }
342 LIST_REMOVE(con, c_entry);
343 npf_conn_destroy(npf, con);
344 }
