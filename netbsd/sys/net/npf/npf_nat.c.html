31/*
32 * NPF network address port translation (NAPT) and other forms of NAT.
33 * Described in RFC 2663, RFC 3022, etc.
34 *
35 * Overview
36 *
37 * There are a few mechanisms: NAT policy, port map and translation.
38 * The NAT module has a separate ruleset where rules always have an
39 * associated NAT policy.
40 *
41 * Translation types
42 *
43 * There are two types of translation: outbound (NPF_NATOUT) and
44 * inbound (NPF_NATIN). It should not be confused with connection
45 * direction. See npf_nat_which() for the description of how the
46 * addresses are rewritten. The bi-directional NAT is a combined
47 * outbound and inbound translation, therefore is constructed as
48 * two policies.
49 *
50 * NAT policies and port maps
51 *
52 * The NAT (translation) policy is applied when packet matches the
53 * rule. Apart from the filter criteria, the NAT policy always has
54 * a translation IP address or a table and an associated port map.
55 * The port map is a bitmap used to reserve and use unique TCP/UDP
56 * ports for translation. Port maps are unique to the IP addresses,
57 * therefore multiple NAT policies with the same IP will share the
58 * same port map. However, if NAT policy is using the translation
59 * table, then the port map is shared per table.
60 *
61 * Connections, translation entries and their life-cycle
62 *
63 * NAT relies on the connection tracking module. Each translated
64 * connection has an associated translation entry (npf_nat_t) which
65 * contains information used for backwards stream translation, i.e.
66 * the original IP address with port and translation port, allocated
67 * from the port map. Each NAT entry is associated with the policy,
68 * which contains translation IP address. Allocated port is returned
69 * to the port map and NAT entry is destroyed when connection expires.
70 */

95/*
96 * NPF portmap structure.
97 */
98typedef struct {
99 u_int p_refcnt;
100 uint32_t p_bitmap[0];
101} npf_portmap_t;
102
103/* Portmap range: [ 1024 .. 65535 ] */
104#define PORTMAP_FIRST (1024)
105#define PORTMAP_SIZE ((65536 - PORTMAP_FIRST) / 32)
106#define PORTMAP_FILLED ((uint32_t)~0U)
107#define PORTMAP_MASK (31)
108#define PORTMAP_SHIFT (5)
109
110#define PORTMAP_MEM_SIZE \
111 (sizeof(npf_portmap_t) + (PORTMAP_SIZE * sizeof(uint32_t)))
112
113/*
114 * NAT policy structure.
115 */
116struct npf_natpolicy {
117 npf_t * n_npfctx;
118 kmutex_t n_lock;
119 LIST_HEAD(, npf_nat) n_nat_list;
120 volatile u_int n_refcnt;
121 npf_portmap_t * n_portmap;
122 uint64_t n_id;
123
124 /*
125 * Translation type, flags, address or table and the port.
126 * Additionally, there may be translation algorithm and any
127 * auxiliary data, e.g. NPTv6 adjustment value.
128 *
129 * NPF_NP_CMP_START mark starts here.
130 */
131 int n_type;
132 unsigned n_flags;
133 unsigned n_alen;
134
135 npf_addr_t n_taddr;
136 npf_netmask_t n_tmask;
137 in_port_t n_tport;
138 unsigned n_tid;
139
140 unsigned n_algo;
141 union {
142 unsigned n_rr_idx;
143 uint16_t n_npt66_adj;
144 };
145};
146
147/*
148 * Private flags - must be in the NPF_NAT_PRIVMASK range.
149 */
150#define NPF_NAT_USETABLE (0x01000000 & NPF_NAT_PRIVMASK)
151
152#define NPF_NP_CMP_START offsetof(npf_natpolicy_t, n_type)
153#define NPF_NP_CMP_SIZE (sizeof(npf_natpolicy_t) - NPF_NP_CMP_START)
154
155/*
156 * NAT translation entry for a connection.
157 */
158struct npf_nat {
159 /* Associated NAT policy. */
160 npf_natpolicy_t * nt_natpolicy;
161
162 /*
163 * Translation address as well as the original address which is
164 * used for backwards translation. The same for ports.
165 */
166 npf_addr_t nt_taddr;
167 npf_addr_t nt_oaddr;
168
169 in_port_t nt_oport;
170 in_port_t nt_tport;
171
172 /* ALG (if any) associated with this NAT entry. */
173 npf_alg_t * nt_alg;
174 uintptr_t nt_alg_arg;
175
176 LIST_ENTRY(npf_nat) nt_entry;
177 npf_conn_t * nt_conn;
178};

201/*
202 * npf_nat_newpolicy: create a new NAT policy.
203 *
204 * => Shares portmap if policy is on existing translation address.
205 */
206npf_natpolicy_t *
207npf_nat_newpolicy(npf_t *npf, const nvlist_t *nat, npf_ruleset_t *rset)
208{
209 npf_natpolicy_t *np;
210 npf_portmap_t *pm;
211 const void *addr;
212 size_t len;
213
214 np = kmem_zalloc(sizeof(npf_natpolicy_t), KM_SLEEP);
215 np->n_npfctx = npf;
216
217 /* The translation type, flags and policy ID. */
218 np->n_type = dnvlist_get_number(nat, "type", 0);
219 np->n_flags = dnvlist_get_number(nat, "flags", 0) & ~NPF_NAT_PRIVMASK;
220 np->n_id = dnvlist_get_number(nat, "nat-policy", 0);
221
222 /* Should be exclusively either inbound or outbound NAT. */
223 if (((np->n_type == NPF_NATIN) ^ (np->n_type == NPF_NATOUT)) == 0) {
224 goto err;
225 }
226 mutex_init(&np->n_lock, MUTEX_DEFAULT, IPL_SOFTNET);
227 LIST_INIT(&np->n_nat_list);
228
229 /*
230 * Translation IP, mask and port (if applicable).
231 * Alternatively, there may be a translation table ID.
232 */
233 if (nvlist_exists_number(nat, "nat-table-id")) {
234 if (np->n_flags & NPF_NAT_STATIC) {
235 goto err;
236 }
237 np->n_tid = nvlist_get_number(nat, "nat-table-id");
238 np->n_tmask = NPF_NO_NETMASK;
239 np->n_flags |= NPF_NAT_USETABLE;
240 } else {
241 addr = dnvlist_get_binary(nat, "nat-ip", &len, NULL, 0);
242 if (!addr || len == 0 || len > sizeof(npf_addr_t)) {
243 goto err;
244 }
245 memcpy(&np->n_taddr, addr, len);
246 np->n_alen = len;
247 np->n_tmask = dnvlist_get_number(nat, "nat-mask", 0);
248 }
249 np->n_tport = dnvlist_get_number(nat, "nat-port", 0);
250
251 /*
252 * NAT algorithm.
253 */
254 np->n_algo = dnvlist_get_number(nat, "nat-algo", 0);
255 switch (np->n_algo) {
256 case NPF_ALGO_NPT66:
257 np->n_npt66_adj = dnvlist_get_number(nat, "npt66-adj", 0);
258 break;
259 case NPF_ALGO_NETMAP:
260 break;
261 case NPF_ALGO_IPHASH:
262 case NPF_ALGO_RR:
263 default:
264 if (np->n_tmask != NPF_NO_NETMASK)
265 goto err;
266 break;
267 }
268
269 /* Determine if port map is needed. */
270 np->n_portmap = NULL;
271 if ((np->n_flags & NPF_NAT_PORTMAP) == 0) {
272 /* No port map. */
273 return np;
274 }
275
276 /*
277 * Inspect NAT policies in the ruleset for port map sharing.
278 * Note that npf_ruleset_sharepm() will increase the reference count.
279 */
280 if (!npf_ruleset_sharepm(rset, np)) {
281 /* Allocate a new port map for the NAT policy. */
282 pm = kmem_zalloc(PORTMAP_MEM_SIZE, KM_SLEEP);
283 pm->p_refcnt = 1;
284 KASSERT((uintptr_t)pm->p_bitmap == (uintptr_t)pm + sizeof(*pm));
285 np->n_portmap = pm;
286 } else {
287 KASSERT(np->n_portmap != NULL);
288 KASSERT(np->n_portmap->p_refcnt > 0);
289 }
290 return np;
291err:
292 mutex_destroy(&np->n_lock);
293 kmem_free(np, sizeof(npf_natpolicy_t));
294 return NULL;
295}

297int
298npf_nat_policyexport(const npf_natpolicy_t *np, nvlist_t *nat)
299{
300 nvlist_add_number(nat, "nat-policy", np->n_id);
301 nvlist_add_number(nat, "type", np->n_type);
302 nvlist_add_number(nat, "flags", np->n_flags);
303
304 if (np->n_flags & NPF_NAT_USETABLE) {
305 nvlist_add_number(nat, "nat-table-id", np->n_tid);
306 } else {
307 nvlist_add_binary(nat, "nat-ip", &np->n_taddr, np->n_alen);
308 nvlist_add_number(nat, "nat-mask", np->n_tmask);
309 }
310 nvlist_add_number(nat, "nat-port", np->n_tport);
311 nvlist_add_number(nat, "nat-algo", np->n_algo);
312
313 switch (np->n_algo) {
314 case NPF_ALGO_NPT66:
315 nvlist_add_number(nat, "npt66-adj", np->n_npt66_adj);
316 break;
317 }
318 return 0;
319}

321/*
322 * npf_nat_freepolicy: free NAT policy and, on last reference, free portmap.
323 *
324 * => Called from npf_rule_free() during the reload via npf_ruleset_destroy().
325 */
326void
327npf_nat_freepolicy(npf_natpolicy_t *np)
328{
329 npf_portmap_t *pm = np->n_portmap;
330 npf_conn_t *con;
331 npf_nat_t *nt;
332
333 /*
334 * Disassociate all entries from the policy. At this point,
335 * new entries can no longer be created for this policy.
336 */
337 while (np->n_refcnt) {
338 mutex_enter(&np->n_lock);
339 LIST_FOREACH(nt, &np->n_nat_list, nt_entry) {
340 con = nt->nt_conn;
341 KASSERT(con != NULL);
342 npf_conn_expire(con);
343 }
344 mutex_exit(&np->n_lock);
345
346 /* Kick the worker - all references should be going away. */
347 npf_worker_signal(np->n_npfctx);
348 kpause("npfgcnat", false, 1, NULL);
349 }
350 KASSERT(LIST_EMPTY(&np->n_nat_list));
351 KASSERT(pm == NULL || pm->p_refcnt > 0);
352
353 /* Destroy the port map, on last reference. */
354 if (pm && atomic_dec_uint_nv(&pm->p_refcnt) == 0) {
355 KASSERT((np->n_flags & NPF_NAT_PORTMAP) != 0);
356 kmem_free(pm, PORTMAP_MEM_SIZE);
357 }
358 mutex_destroy(&np->n_lock);
359 kmem_free(np, sizeof(npf_natpolicy_t));
360}

362void
363npf_nat_freealg(npf_natpolicy_t *np, npf_alg_t *alg)
364{
365 npf_nat_t *nt;
366
367 mutex_enter(&np->n_lock);
368 LIST_FOREACH(nt, &np->n_nat_list, nt_entry) {
369 if (nt->nt_alg == alg)
370 nt->nt_alg = NULL;
371 }
372 mutex_exit(&np->n_lock);
373}

375/*
376 * npf_nat_cmppolicy: compare two NAT policies.
377 *
378 * => Return 0 on match, and non-zero otherwise.
379 */
380bool
381npf_nat_cmppolicy(npf_natpolicy_t *np, npf_natpolicy_t *mnp)
382{
383 const void *np_raw, *mnp_raw;
384
385 /*
386 * Compare the relevant NAT policy information (in raw form),
387 * which is enough for matching criterion.
388 */
389 KASSERT(np && mnp && np != mnp);
390 np_raw = (const uint8_t *)np + NPF_NP_CMP_START;
391 mnp_raw = (const uint8_t *)mnp + NPF_NP_CMP_START;
392 return memcmp(np_raw, mnp_raw, NPF_NP_CMP_SIZE) == 0;
393}

395bool
396npf_nat_sharepm(npf_natpolicy_t *np, npf_natpolicy_t *mnp)
397{
398 npf_portmap_t *pm, *mpm;
399
400 KASSERT(np && mnp && np != mnp);
401 KASSERT(LIST_EMPTY(&mnp->n_nat_list));
402 KASSERT(mnp->n_refcnt == 0);
403
404 /* Using port map and having equal translation address? */
405 if ((np->n_flags & mnp->n_flags & NPF_NAT_PORTMAP) == 0) {
406 return false;
407 }
408 if (np->n_alen != mnp->n_alen) {
409 return false;
410 }
411 if ((np->n_flags & NPF_NAT_USETABLE) != 0 && np->n_tid != mnp->n_tid) {
412 return false;
413 }
414 if (np->n_alen && memcmp(&np->n_taddr, &mnp->n_taddr, np->n_alen)) {
415 return false;
416 }
417 mpm = mnp->n_portmap;
418 KASSERT(mpm == NULL || mpm->p_refcnt > 0);
419
420 /*
421 * If NAT policy has an old port map - drop the reference
422 * and destroy the port map if it was the last.
423 */
424 if (mpm && atomic_dec_uint_nv(&mpm->p_refcnt) == 0) {
425 kmem_free(mpm, PORTMAP_MEM_SIZE);
426 }
427
428 /* Share the port map. */
429 pm = np->n_portmap;
430 atomic_inc_uint(&pm->p_refcnt);
431 mnp->n_portmap = pm;
432 return true;
433}

447/*
448 * npf_nat_getport: allocate and return a port in the NAT policy portmap.
449 *
450 * => Returns in network byte-order.
451 * => Zero indicates failure.
452 */
453static in_port_t
454npf_nat_getport(npf_natpolicy_t *np)
455{
456 npf_portmap_t *pm = np->n_portmap;
457 u_int n = PORTMAP_SIZE, idx, bit;
458 uint32_t map, nmap;
459
460 KASSERT((np->n_flags & NPF_NAT_PORTMAP) != 0);
461 KASSERT(pm->p_refcnt > 0);
462
463 idx = cprng_fast32() % PORTMAP_SIZE;
464 for (;;) {
465 KASSERT(idx < PORTMAP_SIZE);
466 map = pm->p_bitmap[idx];
467 if (__predict_false(map == PORTMAP_FILLED)) {
468 if (n-- == 0) {
469 /* No space. */
470 return 0;
471 }
472 /* This bitmap is filled, next. */
473 idx = (idx ? idx : PORTMAP_SIZE) - 1;
474 continue;
475 }
476 bit = ffs32(~map) - 1;
477 nmap = map | (1 << bit);
478 if (atomic_cas_32(&pm->p_bitmap[idx], map, nmap) == map) {
479 /* Success. */
480 break;
481 }
482 }
483 return htons(PORTMAP_FIRST + (idx << PORTMAP_SHIFT) + bit);
484}

486/*
487 * npf_nat_takeport: allocate specific port in the NAT policy portmap.
488 */
489static bool
490npf_nat_takeport(npf_natpolicy_t *np, in_port_t port)
491{
492 npf_portmap_t *pm = np->n_portmap;
493 uint32_t map, nmap;
494 u_int idx, bit;
495
496 KASSERT((np->n_flags & NPF_NAT_PORTMAP) != 0);
497 KASSERT(pm->p_refcnt > 0);
498
499 port = ntohs(port) - PORTMAP_FIRST;
500 idx = port >> PORTMAP_SHIFT;
501 bit = port & PORTMAP_MASK;
502 map = pm->p_bitmap[idx];
503 nmap = map | (1 << bit);
504 if (map == nmap) {
505 /* Already taken. */
506 return false;
507 }
508 return atomic_cas_32(&pm->p_bitmap[idx], map, nmap) == map;
509}

511/*
512 * npf_nat_putport: return port as available in the NAT policy portmap.
513 *
514 * => Port should be in network byte-order.
515 */
516static void
517npf_nat_putport(npf_natpolicy_t *np, in_port_t port)
518{
519 npf_portmap_t *pm = np->n_portmap;
520 uint32_t map, nmap;
521 u_int idx, bit;
522
523 KASSERT((np->n_flags & NPF_NAT_PORTMAP) != 0);
524 KASSERT(pm->p_refcnt > 0);
525
526 port = ntohs(port) - PORTMAP_FIRST;
527 idx = port >> PORTMAP_SHIFT;
528 bit = port & PORTMAP_MASK;
529 do {
530 map = pm->p_bitmap[idx];
531 KASSERT(map | (1 << bit));
532 nmap = map & ~(1 << bit);
533 } while (atomic_cas_32(&pm->p_bitmap[idx], map, nmap) != map);
534}

536/*
537 * npf_nat_which: tell which address (source or destination) should be
538 * rewritten given the combination of the NAT type and flow direction.
539 */
540static inline u_int
541npf_nat_which(const int type, bool forw)
542{
543 /*
544 * Outbound NAT rewrites:
545 * - Source (NPF_SRC) on "forwards" stream.
546 * - Destination (NPF_DST) on "backwards" stream.
547 * Inbound NAT is other way round.
548 */
549 if (type == NPF_NATOUT) {
550 forw = !forw;
551 } else {
552 KASSERT(type == NPF_NATIN);
553 }
554 CTASSERT(NPF_SRC == 0 && NPF_DST == 1);
555 KASSERT(forw == NPF_SRC || forw == NPF_DST);
556 return (u_int)forw;
557}

559/*
560 * npf_nat_inspect: inspect packet against NAT ruleset and return a policy.
561 *
562 * => Acquire a reference on the policy, if found.
563 */
564static npf_natpolicy_t *
565npf_nat_inspect(npf_cache_t *npc, const int di)
566{
567 int slock = npf_config_read_enter();
568 npf_ruleset_t *rlset = npf_config_natset(npc->npc_ctx);
569 npf_natpolicy_t *np;
570 npf_rule_t *rl;
571
572 rl = npf_ruleset_inspect(npc, rlset, di, NPF_LAYER_3);
573 if (rl == NULL) {
574 npf_config_read_exit(slock);
575 return NULL;
576 }
577 np = npf_rule_getnat(rl);
578 atomic_inc_uint(&np->n_refcnt);
579 npf_config_read_exit(slock);
580 return np;
581}

583/*
584 * npf_nat_create: create a new NAT translation entry.
585 */
586static npf_nat_t *
587npf_nat_create(npf_cache_t *npc, npf_natpolicy_t *np, npf_conn_t *con)
588{
589 const int proto = npc->npc_proto;
590 const unsigned alen = npc->npc_alen;
591 npf_addr_t *taddr;
592 npf_nat_t *nt;
593
594 KASSERT(npf_iscached(npc, NPC_IP46));
595 KASSERT(npf_iscached(npc, NPC_LAYER4));
596
597 if (np->n_flags & NPF_NAT_USETABLE) {
598 npf_tableset_t *ts = npf_config_tableset(np->n_npfctx);
599 npf_table_t *t = npf_tableset_getbyid(ts, np->n_tid);
600 unsigned idx;
601
602 /*
603 * Dynamically select the translation IP address.
604 */
605 switch (np->n_algo) {
606 case NPF_ALGO_RR:
607 idx = atomic_inc_uint_nv(&np->n_rr_idx);
608 break;
609 case NPF_ALGO_IPHASH:
610 default:
611 idx = npf_addr_mix(alen,
612 npc->npc_ips[NPF_SRC], npc->npc_ips[NPF_DST]);
613 break;
614 }
615 taddr = npf_table_getsome(t, alen, idx);
616 if (taddr == NULL) {
617 return NULL;
618 }
619 } else {
620 /* Static IP address. */
621 taddr = &np->n_taddr;
622 }
623
624 /* Construct a new NAT entry and associate it with the connection. */
625 nt = pool_cache_get(nat_cache, PR_NOWAIT);
626 if (nt == NULL){
627 return NULL;
628 }
629 npf_stats_inc(npc->npc_ctx, NPF_STAT_NAT_CREATE);
630 nt->nt_natpolicy = np;
631 nt->nt_conn = con;
632 nt->nt_alg = NULL;
633
634 /* Set the translation address. */
635 memcpy(&nt->nt_taddr, taddr, alen);
636
637 /* Save the original address which may be rewritten. */
638 if (np->n_type == NPF_NATOUT) {
639 /* Outbound NAT: source (think internal) address. */
640 memcpy(&nt->nt_oaddr, npc->npc_ips[NPF_SRC], alen);
641 } else {
642 /* Inbound NAT: destination (think external) address. */
643 KASSERT(np->n_type == NPF_NATIN);
644 memcpy(&nt->nt_oaddr, npc->npc_ips[NPF_DST], alen);
645 }
646
647 /*
648 * Port translation, if required, and if it is TCP/UDP.
649 */
650 if ((np->n_flags & NPF_NAT_PORTS) == 0 ||
651 (proto != IPPROTO_TCP && proto != IPPROTO_UDP)) {
652 nt->nt_oport = 0;
653 nt->nt_tport = 0;
654 goto out;
655 }
656
657 /* Save the relevant TCP/UDP port. */
658 if (proto == IPPROTO_TCP) {
659 const struct tcphdr *th = npc->npc_l4.tcp;
660 nt->nt_oport = (np->n_type == NPF_NATOUT) ?
661 th->th_sport : th->th_dport;
662 } else {
663 const struct udphdr *uh = npc->npc_l4.udp;
664 nt->nt_oport = (np->n_type == NPF_NATOUT) ?
665 uh->uh_sport : uh->uh_dport;
666 }
667
668 /* Get a new port for translation. */
669 if ((np->n_flags & NPF_NAT_PORTMAP) != 0) {
670 nt->nt_tport = npf_nat_getport(np);
671 } else {
672 nt->nt_tport = np->n_tport;
673 }
674out:
675 mutex_enter(&np->n_lock);
676 LIST_INSERT_HEAD(&np->n_nat_list, nt, nt_entry);
677 mutex_exit(&np->n_lock);
678 return nt;
679}

681/*
682 * npf_nat_translate: perform translation given the state data.
683 */
684static inline int
685npf_nat_translate(npf_cache_t *npc, npf_nat_t *nt, bool forw)
686{
687 const npf_natpolicy_t *np = nt->nt_natpolicy;
688 const u_int which = npf_nat_which(np->n_type, forw);
689 const npf_addr_t *addr;
690 in_port_t port;
691
692 KASSERT(npf_iscached(npc, NPC_IP46));
693 KASSERT(npf_iscached(npc, NPC_LAYER4));
694
695 if (forw) {
696 /* "Forwards" stream: use translation address/port. */
697 addr = &nt->nt_taddr;
698 port = nt->nt_tport;
699 } else {
700 /* "Backwards" stream: use original address/port. */
701 addr = &nt->nt_oaddr;
702 port = nt->nt_oport;
703 }
704 KASSERT((np->n_flags & NPF_NAT_PORTS) != 0 || port == 0);
705
706 /* Execute ALG translation first. */
707 if ((npc->npc_info & NPC_ALG_EXEC) == 0) {
708 npc->npc_info |= NPC_ALG_EXEC;
709 npf_alg_exec(npc, nt, forw);
710 npf_recache(npc);
711 }
712 KASSERT(!nbuf_flag_p(npc->npc_nbuf, NBUF_DATAREF_RESET));
713
714 /* Finally, perform the translation. */
715 return npf_napt_rwr(npc, which, addr, port);
716}
717
718/*
719 * npf_nat_algo: perform the translation given the algorithm.
720 */
721static inline int
722npf_nat_algo(npf_cache_t *npc, const npf_natpolicy_t *np, bool forw)
723{
724 const u_int which = npf_nat_which(np->n_type, forw);
725 const npf_addr_t *taddr, *orig_addr;
726 npf_addr_t addr;
727
728 KASSERT(np->n_flags & NPF_NAT_STATIC);
729
730 switch (np->n_algo) {
731 case NPF_ALGO_NETMAP:
732 /*
733 * NETMAP:
734 *
735 * addr = net-addr | (orig-addr & ~mask)
736 */
737 orig_addr = npc->npc_ips[which];
738 npf_addr_mask(&np->n_taddr, np->n_tmask, npc->npc_alen, &addr);
739 npf_addr_bitor(orig_addr, np->n_tmask, npc->npc_alen, &addr);
740 taddr = &addr;
741 break;
742 case NPF_ALGO_NPT66:
743 return npf_npt66_rwr(npc, which, &np->n_taddr,
744 np->n_tmask, np->n_npt66_adj);
745 default:
746 taddr = &np->n_taddr;
747 break;
748 }
749 return npf_napt_rwr(npc, which, taddr, np->n_tport);
750}

752/*
753 * npf_do_nat:
754 *
755 * - Inspect packet for a NAT policy, unless a connection with a NAT
756 * association already exists. In such case, determine whether it
757 * is a "forwards" or "backwards" stream.
758 * - Perform translation: rewrite source or destination fields,
759 * depending on translation type and direction.
760 * - Associate a NAT policy with a connection (may establish a new).
761 */
762int
763npf_do_nat(npf_cache_t *npc, npf_conn_t *con, const int di)
764{
765 nbuf_t *nbuf = npc->npc_nbuf;
766 npf_conn_t *ncon = NULL;
767 npf_natpolicy_t *np;
768 npf_nat_t *nt;
769 int error;
770 bool forw;
771
772 /* All relevant data should be already cached. */
773 if (!npf_iscached(npc, NPC_IP46) || !npf_iscached(npc, NPC_LAYER4)) {
774 return 0;
775 }
776 KASSERT(!nbuf_flag_p(nbuf, NBUF_DATAREF_RESET));
777
778 /*
779 * Return the NAT entry associated with the connection, if any.
780 * Determines whether the stream is "forwards" or "backwards".
781 * Note: no need to lock, since reference on connection is held.
782 */
783 if (con && (nt = npf_conn_getnat(con, di, &forw)) != NULL) {
784 np = nt->nt_natpolicy;
785 goto translate;
786 }
787
788 /*
789 * Inspect the packet for a NAT policy, if there is no connection.
790 * Note: acquires a reference if found.
791 */
792 np = npf_nat_inspect(npc, di);
793 if (np == NULL) {
794 /* If packet does not match - done. */
795 return 0;
796 }
797 forw = true;
798
799 /* Static NAT - just perform the translation. */
800 if (np->n_flags & NPF_NAT_STATIC) {
801 if (nbuf_cksum_barrier(nbuf, di)) {
802 npf_recache(npc);
803 }
804 error = npf_nat_algo(npc, np, forw);
805 atomic_dec_uint(&np->n_refcnt);
806 return error;
807 }
808
809 /*
810 * If there is no local connection (no "stateful" rule - unusual,
811 * but possible configuration), establish one before translation.
812 * Note that it is not a "pass" connection, therefore passing of
813 * "backwards" stream depends on other, stateless filtering rules.
814 */
815 if (con == NULL) {
816 ncon = npf_conn_establish(npc, di, true);
817 if (ncon == NULL) {
818 atomic_dec_uint(&np->n_refcnt);
819 return ENOMEM;
820 }
821 con = ncon;
822 }
823
824 /*
825 * Create a new NAT entry and associate with the connection.
826 * We will consume the reference on success (release on error).
827 */
828 nt = npf_nat_create(npc, np, con);
829 if (nt == NULL) {
830 atomic_dec_uint(&np->n_refcnt);
831 error = ENOMEM;
832 goto out;
833 }
834
835 /* Associate the NAT translation entry with the connection. */
836 error = npf_conn_setnat(npc, con, nt, np->n_type);
837 if (error) {
838 /* Will release the reference. */
839 npf_nat_destroy(nt);
840 goto out;
841 }
842
843 /* Determine whether any ALG matches. */
844 if (npf_alg_match(npc, nt, di)) {
845 KASSERT(nt->nt_alg != NULL);
846 }
847
848translate:
849 /* May need to process the delayed checksums first (XXX: NetBSD). */
850 if (nbuf_cksum_barrier(nbuf, di)) {
851 npf_recache(npc);
852 }
853
854 /* Perform the translation. */
855 error = npf_nat_translate(npc, nt, forw);
856out:
857 if (__predict_false(ncon)) {
858 if (error) {
859 /* It created for NAT - just expire. */
860 npf_conn_expire(ncon);
861 }
862 npf_conn_release(ncon);
863 }
864 return error;
865}

897/*
898 * npf_nat_destroy: destroy NAT structure (performed on connection expiration).
899 */
900void
901npf_nat_destroy(npf_nat_t *nt)
902{
903 npf_natpolicy_t *np = nt->nt_natpolicy;
904
905 /* Return any taken port to the portmap. */
906 if ((np->n_flags & NPF_NAT_PORTMAP) != 0 && nt->nt_tport) {
907 npf_nat_putport(np, nt->nt_tport);
908 }
909 npf_stats_inc(np->n_npfctx, NPF_STAT_NAT_DESTROY);
910
911 mutex_enter(&np->n_lock);
912 LIST_REMOVE(nt, nt_entry);
913 KASSERT(np->n_refcnt > 0);
914 atomic_dec_uint(&np->n_refcnt);
915 mutex_exit(&np->n_lock);
916 pool_cache_put(nat_cache, nt);
917}

936/*
937 * npf_nat_import: find the NAT policy and unserialise the NAT entry.
938 */
939npf_nat_t *
940npf_nat_import(npf_t *npf, const nvlist_t *nat,
941 npf_ruleset_t *natlist, npf_conn_t *con)
942{
943 npf_natpolicy_t *np;
944 npf_nat_t *nt;
945 const void *oaddr;
946 uint64_t np_id;
947 size_t len;
948
949 np_id = dnvlist_get_number(nat, "nat-policy", UINT64_MAX);
950 if ((np = npf_ruleset_findnat(natlist, np_id)) == NULL) {
951 return NULL;
952 }
953 nt = pool_cache_get(nat_cache, PR_WAITOK);
954 memset(nt, 0, sizeof(npf_nat_t));
955
956 oaddr = dnvlist_get_binary(nat, "oaddr", &len, NULL, 0);
957 if (!oaddr || len != sizeof(npf_addr_t)) {
958 pool_cache_put(nat_cache, nt);
959 return NULL;
960 }
961 memcpy(&nt->nt_oaddr, oaddr, sizeof(npf_addr_t));
962 nt->nt_oport = dnvlist_get_number(nat, "oport", 0);
963 nt->nt_tport = dnvlist_get_number(nat, "tport", 0);
964
965 /* Take a specific port from port-map. */
966 if ((np->n_flags & NPF_NAT_PORTMAP) != 0 && nt->nt_tport &&
967 !npf_nat_takeport(np, nt->nt_tport)) {
968 pool_cache_put(nat_cache, nt);
969 return NULL;
970 }
971 npf_stats_inc(npf, NPF_STAT_NAT_CREATE);
972
973 /*
974 * Associate, take a reference and insert. Unlocked since
975 * the policy is not yet visible.
976 */
977 nt->nt_natpolicy = np;
978 nt->nt_conn = con;
979 np->n_refcnt++;
980 LIST_INSERT_HEAD(&np->n_nat_list, nt, nt_entry);
981 return nt;
982}
