39#ifndef _NET_PPP_DEFS_H_
40#define _NET_PPP_DEFS_H_
41
42/*
43 * The basic PPP frame.
44 */
45#define PPP_HDRLEN 4 /* octets for standard ppp header */
46#define PPP_FCSLEN 2 /* octets for FCS */
47
48/*
49 * Packet sizes
50 *
51 * Note - lcp shouldn't be allowed to negotiate stuff outside these
52 * limits. See lcp.h in the pppd directory.
53 * (XXX - these constants should simply be shared by lcp.c instead
54 * of living in lcp.h)
55 */
56#define PPP_MTU 1500 /* Default MTU (size of Info field) */
57#define PPP_MAXMTU 65535 - (PPP_HDRLEN + PPP_FCSLEN)
58#define PPP_MINMTU 64
59#define PPP_MRU 1500 /* default MRU = max length of info field */
60#define PPP_MAXMRU 65000 /* Largest MRU we allow */
61#define PPP_MINMRU 128
62
63#define PPP_ADDRESS(p) (((const u_char *)(p))[0])
64#define PPP_CONTROL(p) (((const u_char *)(p))[1])
65#define PPP_PROTOCOL(p) \
66 ((((const u_char *)(p))[2] << 8) + ((const u_char *)(p))[3])
67
68/*
69 * Significant octet values.
70 */
71#define PPP_ALLSTATIONS 0xff /* All-Stations broadcast address */
72#define PPP_UI 0x03 /* Unnumbered Information */
73#define PPP_FLAG 0x7e /* Flag Sequence */
74#define PPP_ESCAPE 0x7d /* Asynchronous Control Escape */
75#define PPP_TRANS 0x20 /* Asynchronous transparency modifier */
76
77/*
78 * Protocol field values.
79 */
80#define PPP_IP 0x0021 /* Internet Protocol */
81#define PPP_ISO 0x0023 /* ISO OSI Protocol */
82#define PPP_XNS 0x0025 /* Xerox NS Protocol */
83#define PPP_AT 0x0029 /* AppleTalk Protocol */
84#define PPP_IPX 0x002b /* IPX protocol */
85#define PPP_VJC_COMP 0x002d /* VJ compressed TCP */
86#define PPP_VJC_UNCOMP 0x002f /* VJ uncompressed TCP */
87#define PPP_IPV6 0x0057 /* Internet Protocol Version 6 */
88#define PPP_COMP 0x00fd /* compressed packet */
89#define PPP_IPCP 0x8021 /* IP Control Protocol */
90#define PPP_ATCP 0x8029 /* AppleTalk Control Protocol */
91#define PPP_IPXCP 0x802b /* IPX Control Protocol */
92#define PPP_IPV6CP 0x8057 /* IPv6 Control Protocol */
93#define PPP_CCP 0x80fd /* Compression Control Protocol */
94#define PPP_ECP 0x8053 /* Encryption Control Protocol */
95#define PPP_LCP 0xc021 /* Link Control Protocol */
96#define PPP_PAP 0xc023 /* Password Authentication Protocol */
97#define PPP_LQR 0xc025 /* Link Quality Report protocol */
98#define PPP_CHAP 0xc223 /* Crypto Handshake Auth. Protocol */
99#define PPP_CBCP 0xc029 /* Callback Control Protocol */
100#define PPP_EAP 0xc227 /* Extensible Authentication Protocol */
101
102/*
103 * Values for FCS calculations.
104 */
105#define PPP_INITFCS 0xffff /* Initial FCS value */
106#define PPP_GOODFCS 0xf0b8 /* Good final FCS value */
107#define PPP_FCS(fcs, c) (((fcs) >> 8) ^ fcstab[((fcs) ^ (c)) & 0xff])
108
109/*
110 * Extended asyncmap - allows any character to be escaped.
111 */
112typedef uint32_t ext_accm[8];
113
114/*
115 * What to do with network protocol (NP) packets.
116 */
117enum NPmode {
118 NPMODE_PASS, /* pass the packet through */
119 NPMODE_DROP, /* silently drop the packet */
120 NPMODE_ERROR, /* return an error */
121 NPMODE_QUEUE /* save it up for later. */
122};
123
124/*
125 * Statistics.
126 */
127struct pppstat {
128 unsigned int ppp_ibytes; /* bytes received */
129 unsigned int ppp_ipackets; /* packets received */
130 unsigned int ppp_ierrors; /* receive errors */
131 unsigned int ppp_obytes; /* bytes sent */
132 unsigned int ppp_opackets; /* packets sent */
133 unsigned int ppp_oerrors; /* transmit errors */
134};
135
136struct vjstat {
137 unsigned int vjs_packets; /* outbound packets */
138 unsigned int vjs_compressed; /* outbound compressed packets */
139 unsigned int vjs_searches; /* searches for connection state */
140 unsigned int vjs_misses; /* times couldn't find conn. state */
141 unsigned int vjs_uncompressedin; /* inbound uncompressed packets */
142 unsigned int vjs_compressedin; /* inbound compressed packets */
143 unsigned int vjs_errorin; /* inbound unknown type packets */
144 unsigned int vjs_tossed; /* inbound packets tossed because of error */
145};
146
147struct ppp_stats {
148 struct pppstat p; /* basic PPP statistics */
149 struct vjstat vj; /* VJ header compression statistics */
150};
151
152struct compstat {
153 unsigned int unc_bytes; /* total uncompressed bytes */
154 unsigned int unc_packets; /* total uncompressed packets */
155 unsigned int comp_bytes; /* compressed bytes */
156 unsigned int comp_packets; /* compressed packets */
157 unsigned int inc_bytes; /* incompressible bytes */
158 unsigned int inc_packets; /* incompressible packets */
159 unsigned int ratio; /* recent compression ratio << 8 */
160};
161
162struct ppp_comp_stats {
163 struct compstat c; /* packet compression statistics */
164 struct compstat d; /* packet decompression statistics */
165};
166
167/*
168 * The following structure records the time in seconds since
169 * the last NP packet was sent or received.
170 */
171struct ppp_idle {
172 time_t xmit_idle; /* time since last NP packet sent */
173 time_t recv_idle; /* time since last NP packet received */
174};
175
176#endif /* !_NET_PPP_DEFS_H_ */
