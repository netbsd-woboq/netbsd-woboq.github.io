38/*
39 * Same as FreeBSD.
40 *
41 * This rss key is assumed for verification suite in many intel Gigabit and
42 * 10 Gigabit Controller specifications.
43 */
44static uint8_t rss_default_key[RSS_KEYSIZE] = {
45 0x6d, 0x5a, 0x56, 0xda, 0x25, 0x5b, 0x0e, 0xc2,
46 0x41, 0x67, 0x25, 0x3d, 0x43, 0xa3, 0x8f, 0xb0,
47 0xd0, 0xca, 0x2b, 0xcb, 0xae, 0x7b, 0x30, 0xb4,
48 0x77, 0xcb, 0x2d, 0xa3, 0x80, 0x30, 0xf2, 0x0c,
49 0x6a, 0x42, 0xb7, 0x3b, 0xbe, 0xac, 0x01, 0xfa,
50};
51
52#ifdef NOTYET
53/*
54 * Same as DragonFlyBSD.
55 *
56 * This rss key make rss hash value symmetric, that is, the hash value
57 * calculated by func("source address", "destination address") equals to
58 * the hash value calculated by func("destination address", "source address").
59 */
60static uint8_t rss_symmetric_key[RSS_KEYSIZE] = {
61 0x6d, 0x5a, 0x6d, 0x5a, 0x6d, 0x5a, 0x6d, 0x5a,
62 0x6d, 0x5a, 0x6d, 0x5a, 0x6d, 0x5a, 0x6d, 0x5a,
63 0x6d, 0x5a, 0x6d, 0x5a, 0x6d, 0x5a, 0x6d, 0x5a,
64 0x6d, 0x5a, 0x6d, 0x5a, 0x6d, 0x5a, 0x6d, 0x5a,
65};
66#endif

68/*
69 * sizeof(key) must be more than or equal to RSS_KEYSIZE.
70 */
71void
72rss_getkey(uint8_t *key)
73{
74
75 memcpy(key, rss_default_key, sizeof(rss_default_key));
76}
