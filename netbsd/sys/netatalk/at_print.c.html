40int
41at_print(char *buf, size_t len, const struct at_addr *aa)
42{
43 return snprintf(buf, len, "%hu.%hhu", ntohs(aa->s_net), aa->s_node);
44}
45
46// XXX: netrange
47int
48sat_print(char *buf, size_t len, const void *v)
49{
50 const struct sockaddr_at *sat = v;
51 const struct at_addr *aa = &sat->sat_addr;
52 char abuf[ATALK_ADDRSTRLEN];
53
54 if (!sat->sat_port)
55 return at_print(buf, len, aa);
56
57 at_print(abuf, sizeof(abuf), aa);
58 return snprintf(buf, len, "%s:%hhu", abuf, sat->sat_port);
59}
