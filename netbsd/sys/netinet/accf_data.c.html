45MODULE(MODULE_CLASS_MISC, accf_dataready, NULL);
46
47/* accept filter that holds a socket until data arrives */
48
49static void sohasdata(struct socket *so, void *arg, int events, int waitflag);
50
51static struct accept_filter accf_data_filter = {
52 .accf_name = "dataready",
53 .accf_callback = sohasdata,
54};
55
56void
57accf_dataattach(int junk)
58{
59
60}
61
62static int
63accf_dataready_modcmd(modcmd_t cmd, void *arg)
64{
65
66 switch (cmd) {
67 case MODULE_CMD_INIT:
68 return accept_filt_add(&accf_data_filter);
69
70 case MODULE_CMD_FINI:
71 return accept_filt_del(&accf_data_filter);
72
73 default:
74 return ENOTTY;
75 }
76}
77
78static void
79sohasdata(struct socket *so, void *arg, int events, int waitflag)
80{
81
82 if (!soreadable(so))
83 return;
84
85 so->so_upcall = NULL;
86 so->so_rcv.sb_flags &= ~SB_UPCALL;
87 soisconnected(so);
88 return;
89}
