74#ifndef _NETINET_IGMP_H_
75#define _NETINET_IGMP_H_
76
77/*
78 * Internet Group Management Protocol (IGMP) definitions.
79 *
80 * MULTICAST 1.3
81 */
82
83/*
84 * IGMP packet format.
85 */
86struct igmp {
87 uint8_t igmp_type; /* version & type of IGMP message */
88 uint8_t igmp_code; /* code for routing sub-messages */
89 uint16_t igmp_cksum; /* IP-style checksum */
90 struct in_addr igmp_group; /* group address being reported */
91} __packed; /* (zero for queries) */
92
93#define IGMP_MINLEN 8
94
95#define IGMP_HOST_MEMBERSHIP_QUERY 0x11 /* membership query */
96#define IGMP_v1_HOST_MEMBERSHIP_REPORT 0x12 /* v1 membership report */
97#define IGMP_DVMRP 0x13 /* DVMRP routing message */
98#define IGMP_PIM 0x14 /* PIM routing message */
99#define IGMP_v2_HOST_MEMBERSHIP_REPORT 0x16 /* v2 membership report */
100#define IGMP_HOST_LEAVE_MESSAGE 0x17 /* leave-group message */
101#define IGMP_MTRACE_REPLY 0x1e /* traceroute reply */
102#define IGMP_MTRACE_QUERY 0x1f /* traceroute query */
103
104#define IGMP_MAX_HOST_REPORT_DELAY 10 /* max delay for response to */
105 /* query (in seconds) */
106
107#define IGMP_TIMER_SCALE 10 /* denominator for igmp_timer */
108
109/*
110 * States for the IGMP v2 state table.
111 */
112#define IGMP_DELAYING_MEMBER 1
113#define IGMP_IDLE_MEMBER 2
114#define IGMP_LAZY_MEMBER 3
115#define IGMP_SLEEPING_MEMBER 4
116#define IGMP_AWAKENING_MEMBER 5
117
118/*
119 * States for IGMP router version cache.
120 */
121#define IGMP_v1_ROUTER 1
122#define IGMP_v2_ROUTER 2
123
124/*
125 * Revert to v2 if we haven't heard from the router in this amount of time.
126 */
127#define IGMP_AGE_THRESHOLD 540
128
129#endif /* !_NETINET_IGMP_H_ */
