41/*
42 * Checksum of the IPv4 pseudo header.
43 *
44 * off is supposed to be the skipped IPv4 header, len is the payload size.
45 */
46#ifdef DIAGNOSTIC
47#define PANIC(a,...) panic(a, __VA_ARGS__)
48#else
49#define PANIC(a,...) do { \
50 printf(a, __VA_ARGS__); \
51 return -1; \
52} while (/*CONSTCOND*/0)
53#endif
54
55int
56in4_cksum(struct mbuf *m, u_int8_t nxt, int off, int len)
57{
58 uint32_t sum;
59 uint16_t *w;
60
61 if (__predict_false(m->m_len < sizeof(struct ip)))
62 PANIC("%s: mbuf %d too short for IP header %zu", __func__,
63 m->m_len, sizeof(struct ip));
64
65 if (nxt == 0)
66 return cpu_in_cksum(m, len, off, 0);
67
68 if (__predict_false(off < sizeof(struct ip)))
69 PANIC("%s: offset %d too short for IP header %zu", __func__,
70 off, sizeof(struct ip));
71
72 /*
73 * Compute the equivalent of:
74 * struct ipovly ip;
75 *
76 * bzero(sizeof(*ip));
77 * ip.ih_pr = nxt;
78 * ip.ip_len = htons(len);
79 * ip.ih_src = mtod(m, struct ip *)->ip_src;
80 * ip.ih_dst = mtod(m, struct ip *)->ip_dst;
81 * sum = one_add(&ip);
82 */
83
84#if BYTE_ORDER == LITTLE_ENDIAN
85 sum = ((len & 0xffff) + nxt) << 8;
86#else
87 sum = (len & 0xffff) + nxt;
88#endif
89 w = (uint16_t *)(mtod(m, char *) + offsetof(struct ip, ip_src));
90 if (__predict_true((uintptr_t)w % 2 == 0)) {
91 sum += w[0];
92 sum += w[1];
93 sum += w[2];
94 sum += w[3];
95 } else {
96 uint32_t partial;
97 w = (void *)((uintptr_t)w - 1);
98#if BYTE_ORDER == LITTLE_ENDIAN
99 partial = w[0] & 0xff00;
100#else
101 partial = w[0] & 0x00ff;
102#endif
103 partial += w[1];
104 partial += w[2];
105 partial += w[3];
106#if BYTE_ORDER == LITTLE_ENDIAN
107 partial += w[4] & 0x00ff;
108#else
109 partial += w[4] & 0xff00;
110#endif
111 sum += partial << 8;
112 }
113
114 return cpu_in_cksum(m, len, off, sum);
115}
