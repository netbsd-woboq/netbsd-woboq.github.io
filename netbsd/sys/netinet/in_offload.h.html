29#ifndef _NETINET_IN_OFFLOAD_H_
30#define _NETINET_IN_OFFLOAD_H_
31
32/*
33 * Subroutines to do software-only equivalent of h/w offloading.
34 */
35struct mbuf *tcp4_segment(struct mbuf *, int);
36int ip_tso_output(struct ifnet *, struct mbuf *, const struct sockaddr *,
37 struct rtentry *);
38void in_undefer_cksum(struct mbuf *, size_t, int);
39void in_undefer_cksum_tcpudp(struct mbuf *);
40
41/*
42 * Offloading-related sysctl variables.
43 *
44 * They are here because it violates protocol layering in unusual way.
45 * ie. while they are TCP/UDP sysctls, they are used by IP layer.
46 */
47extern int tcp_do_loopback_cksum; /* do TCP checksum on loopback? */
48extern int udp_do_loopback_cksum; /* do UDP checksum on loopback? */
49
50#define IN_LOOPBACK_NEED_CHECKSUM(csum_flags) \
51 ((((csum_flags) & M_CSUM_UDPv4) != 0 && udp_do_loopback_cksum) || \
52 (((csum_flags) & M_CSUM_TCPv4) != 0 && tcp_do_loopback_cksum) || \
53 (((csum_flags) & M_CSUM_IPv4) != 0 && ip_do_loopback_cksum))
54
55#define IN_NEED_CHECKSUM(ifp, csum_flags) \
56 (__predict_true(((ifp)->if_flags & IFF_LOOPBACK) == 0 || \
57 IN_LOOPBACK_NEED_CHECKSUM(csum_flags)))
58
59#endif /* !_NETINET_IN_OFFLOAD_H_ */
