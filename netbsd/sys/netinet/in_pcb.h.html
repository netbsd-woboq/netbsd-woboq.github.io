66#include <sys/queue.h>
67#include <net/route.h>

70/*
71 * Common structure pcb for internet protocol implementation.
72 * Here are stored pointers to local and foreign host table
73 * entries, local and foreign socket numbers, and pointers
74 * up (to a socket structure) and down (to a protocol-specific)
75 * control block.
76 */
77struct inpcb {
78 struct inpcb_hdr inp_head;
79#define inp_hash inp_head.inph_hash
80#define inp_queue inp_head.inph_queue
81#define inp_af inp_head.inph_af
82#define inp_ppcb inp_head.inph_ppcb
83#define inp_state inp_head.inph_state
84#define inp_portalgo inp_head.inph_portalgo
85#define inp_socket inp_head.inph_socket
86#define inp_table inp_head.inph_table
87#define inp_sp inp_head.inph_sp
88 struct route inp_route; /* placeholder for routing entry */
89 u_int16_t inp_fport; /* foreign port */
90 u_int16_t inp_lport; /* local port */
91 int inp_flags; /* generic IP/datagram flags */
92 struct ip inp_ip; /* header prototype; should have more */
93 struct mbuf *inp_options; /* IP options */
94 struct ip_moptions *inp_moptions; /* IP multicast options */
95 int inp_errormtu; /* MTU of last xmit status = EMSGSIZE */
96 uint8_t inp_ip_minttl;
97 bool inp_bindportonsend;
98 struct in_addr inp_prefsrcip; /* preferred src IP when wild */
99};
100
101#define inp_faddr inp_ip.ip_dst
102#define inp_laddr inp_ip.ip_src
103
104/* flags in inp_flags: */
105#define INP_RECVOPTS 0x0001 /* receive incoming IP options */
106#define INP_RECVRETOPTS 0x0002 /* receive IP options for reply */
107#define INP_RECVDSTADDR 0x0004 /* receive IP dst address */
108#define INP_HDRINCL 0x0008 /* user supplies entire IP header */
109#define INP_HIGHPORT 0x0010 /* (unused; FreeBSD compat) */
110#define INP_LOWPORT 0x0020 /* user wants "low" port binding */
111#define INP_ANONPORT 0x0040 /* port chosen for user */
112#define INP_RECVIF 0x0080 /* receive incoming interface */
113/* XXX should move to an UDP control block */
114#define INP_ESPINUDP 0x0100 /* ESP over UDP for NAT-T */
115#define INP_ESPINUDP_NON_IKE 0x0200 /* ESP over UDP for NAT-T */
116#define INP_NOHEADER 0x0400 /* Kernel removes IP header
117 * before feeding a packet
118 * to the raw socket user.
119 * The socket user will
120 * not supply an IP header.
121 * Cancels INP_HDRINCL.
122 */
123#define INP_RECVTTL 0x0800 /* receive incoming IP TTL */
124#define INP_RECVPKTINFO 0x1000 /* receive IP dst if/addr */
125#define INP_CONTROLOPTS (INP_RECVOPTS|INP_RECVRETOPTS|INP_RECVDSTADDR|\
126 INP_RECVIF|INP_RECVTTL|INP_RECVPKTINFO)

131#define inp_locked(inp) solocked((inp)->inp_socket)
