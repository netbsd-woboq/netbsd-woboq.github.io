61/*
62 * modify outer ECN (TOS) field on ingress operation (tunnel encapsulation).
63 */
64void
65ip_ecn_ingress(int mode, u_int8_t *outer, const u_int8_t *inner)
66{
67 if (!outer || !inner)
68 panic("NULL pointer passed to ip_ecn_ingress");
69
70 *outer = *inner;
71 switch (mode) {
72 case ECN_ALLOWED: /* ECN allowed */
73 *outer &= ~IPTOS_ECN_CE;
74 break;
75 case ECN_FORBIDDEN: /* ECN forbidden */
76 *outer &= ~(IPTOS_ECN_ECT0 | IPTOS_ECN_CE);
77 break;
78 case ECN_NOCARE: /* no consideration to ECN */
79 break;
80 }
81}
82
83/*
84 * modify inner ECN (TOS) field on egress operation (tunnel decapsulation).
85 */
86void
87ip_ecn_egress(int mode, const u_int8_t *outer, u_int8_t *inner)
88{
89 if (!outer || !inner)
90 panic("NULL pointer passed to ip_ecn_egress");
91
92 switch (mode) {
93 case ECN_ALLOWED:
94 if (*outer & IPTOS_ECN_CE)
95 *inner |= IPTOS_ECN_CE;
96 break;
97 case ECN_FORBIDDEN: /* ECN forbidden */
98 case ECN_NOCARE: /* no consideration to ECN */
99 break;
100 }
101}

103#ifdef INET6
104void
105ip6_ecn_ingress(int mode, u_int32_t *outer, const u_int32_t *inner)
106{
107 u_int8_t outer8, inner8;
108
109 if (!outer || !inner)
110 panic("NULL pointer passed to ip6_ecn_ingress");
111
112 outer8 = (ntohl(*outer) >> 20) & 0xff;
113 inner8 = (ntohl(*inner) >> 20) & 0xff;
114 ip_ecn_ingress(mode, &outer8, &inner8);
115 *outer &= ~htonl(0xff << 20);
116 *outer |= htonl((u_int32_t)outer8 << 20);
117}
118
119void
120ip6_ecn_egress(int mode, const u_int32_t *outer, u_int32_t *inner)
121{
122 u_int8_t outer8, inner8;
123
124 if (!outer || !inner)
125 panic("NULL pointer passed to ip6_ecn_egress");
126
127 outer8 = (ntohl(*outer) >> 20) & 0xff;
128 inner8 = (ntohl(*inner) >> 20) & 0xff;
129 ip_ecn_egress(mode, &outer8, &inner8);
130 *inner &= ~htonl(0xff << 20);
131 *inner |= htonl((u_int32_t)inner8 << 20);
132}
133#endif
