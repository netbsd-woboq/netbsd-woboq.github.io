42typedef uint32_t tcp_seq;
43/*
44 * TCP header.
45 * Per RFC 793, September, 1981.
46 * Updated by RFC 3168, September, 2001.
47 */
48struct tcphdr {
49 uint16_t th_sport; /* source port */
50 uint16_t th_dport; /* destination port */
51 tcp_seq th_seq; /* sequence number */
52 tcp_seq th_ack; /* acknowledgement number */
53#if BYTE_ORDER == LITTLE_ENDIAN
54 /*LINTED non-portable bitfields*/
55 uint8_t th_x2:4, /* (unused) */
56 th_off:4; /* data offset */
57#endif
58#if BYTE_ORDER == BIG_ENDIAN
59 /*LINTED non-portable bitfields*/
60 uint8_t th_off:4, /* data offset */
61 th_x2:4; /* (unused) */
62#endif
63 uint8_t th_flags;
64#define TH_FIN 0x01
65#define TH_SYN 0x02
66#define TH_RST 0x04
67#define TH_PUSH 0x08
68#define TH_ACK 0x10
69#define TH_URG 0x20
70#define TH_ECE 0x40
71#define TH_CWR 0x80
72 uint16_t th_win; /* window */
73 uint16_t th_sum; /* checksum */
74 uint16_t th_urp; /* urgent pointer */
75} __packed;

144/*
145 * The TCP_INFO socket option comes from the Linux 2.6 TCP API, and permits
146 * the caller to query certain information about the state of a TCP
147 * connection. We provide an overlapping set of fields with the Linux
148 * implementation, but since this is a fixed size structure, room has been
149 * left for growth. In order to maximize potential future compatibility with
150 * the Linux API, the same variable names and order have been adopted, and
151 * padding left to make room for omitted fields in case they are added later.
152 *
153 * XXX: This is currently an unstable ABI/API, in that it is expected to
154 * change.
155 */
156struct tcp_info {
157 uint8_t tcpi_state; /* TCP FSM state. */
158 uint8_t __tcpi_ca_state;
159 uint8_t __tcpi_retransmits;
160 uint8_t __tcpi_probes;
161 uint8_t __tcpi_backoff;
162 uint8_t tcpi_options; /* Options enabled on conn. */
163 /*LINTED: non-portable bitfield*/
164 uint8_t tcpi_snd_wscale:4, /* RFC1323 send shift value. */
165 /*LINTED: non-portable bitfield*/
166 tcpi_rcv_wscale:4; /* RFC1323 recv shift value. */
167
168 uint32_t tcpi_rto; /* Retransmission timeout (usec). */
169 uint32_t __tcpi_ato;
170 uint32_t tcpi_snd_mss; /* Max segment size for send. */
171 uint32_t tcpi_rcv_mss; /* Max segment size for receive. */
172
173 uint32_t __tcpi_unacked;
174 uint32_t __tcpi_sacked;
175 uint32_t __tcpi_lost;
176 uint32_t __tcpi_retrans;
177 uint32_t __tcpi_fackets;
178
179 /* Times; measurements in usecs. */
180 uint32_t __tcpi_last_data_sent;
181 uint32_t __tcpi_last_ack_sent; /* Also unimpl. on Linux? */
182 uint32_t tcpi_last_data_recv; /* Time since last recv data. */
183 uint32_t __tcpi_last_ack_recv;
184
185 /* Metrics; variable units. */
186 uint32_t __tcpi_pmtu;
187 uint32_t __tcpi_rcv_ssthresh;
188 uint32_t tcpi_rtt; /* Smoothed RTT in usecs. */
189 uint32_t tcpi_rttvar; /* RTT variance in usecs. */
190 uint32_t tcpi_snd_ssthresh; /* Slow start threshold. */
191 uint32_t tcpi_snd_cwnd; /* Send congestion window. */
192 uint32_t __tcpi_advmss;
193 uint32_t __tcpi_reordering;
194
195 uint32_t __tcpi_rcv_rtt;
196 uint32_t tcpi_rcv_space; /* Advertised recv window. */
197
198 /* FreeBSD/NetBSD extensions to tcp_info. */
199 uint32_t tcpi_snd_wnd; /* Advertised send window. */
200 uint32_t tcpi_snd_bwnd; /* No longer used. */
201 uint32_t tcpi_snd_nxt; /* Next egress seqno */
202 uint32_t tcpi_rcv_nxt; /* Next ingress seqno */
203 uint32_t tcpi_toe_tid; /* HWTID for TOE endpoints */
204 uint32_t tcpi_snd_rexmitpack; /* Retransmitted packets */
205 uint32_t tcpi_rcv_ooopack; /* Out-of-order packets */
206 uint32_t tcpi_snd_zerowin; /* Zero-sized windows sent */
207
208 /* Padding to grow without breaking ABI. */
209 uint32_t __tcpi_pad[26]; /* Padding. */
210};
