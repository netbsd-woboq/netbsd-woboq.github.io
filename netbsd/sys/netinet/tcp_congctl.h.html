32#ifndef _NETINET_TCP_CONGCTL_H
33#define _NETINET_TCP_CONGCTL_H
34
35#define TCPCC_MAXLEN 12
36
37struct tcp_congctlent {
38 TAILQ_ENTRY(tcp_congctlent) congctl_ent;
39 const struct tcp_congctl *congctl_ctl;
40 unsigned int congctl_refcnt;
41 char congctl_name[TCPCC_MAXLEN];
42};
43/*
44 * Congestion control function table.
45 */
46struct tcp_congctl {
47 /*
48 * fast_retransmit: called on tcprexmtthresh'th dup ACKs.
49 * this actually retransmits packets by calling tcp_output()
50 * if appropriate.
51 * returns 0 if entering fast recovery. otherwise returns non-0.
52 */
53 int (*fast_retransmit)(struct tcpcb *, const struct tcphdr *);
54
55 /*
56 * slow_retransmit: called on RTO to adjust parameters like cwnd.
57 */
58 void (*slow_retransmit)(struct tcpcb *);
59
60 /*
61 * fast_retransmit_newack: called when new data is acked.
62 * ie. when advancing SND.UNA
63 * not called if TCP_SACK_ENABLED.
64 */
65 void (*fast_retransmit_newack)(struct tcpcb *, const struct tcphdr *);
66
67 /*
68 * newack: called when new data is acked. ie. when advancing SND.UNA
69 * it's called before updating tp->snd_una.
70 */
71 void (*newack)(struct tcpcb *, const struct tcphdr *);
72
73 /*
74 * cong_exp: called when congestion is detected. eg. by ECN
75 */
76 void (*cong_exp)(struct tcpcb *);
77};
78
79extern const struct tcp_congctl tcp_reno_ctl;
80extern const struct tcp_congctl tcp_newreno_ctl;
81extern const struct tcp_congctl tcp_cubic_ctl;
82
83/* currently selected global congestion control */
84extern char tcp_congctl_global_name[TCPCC_MAXLEN];
85
86/* available global congestion control algorithms */
87extern char tcp_congctl_avail[10 * TCPCC_MAXLEN];
88
89void tcp_congctl_init(void);
90int tcp_congctl_register(const char *, const struct tcp_congctl *);
91int tcp_congctl_unregister(const char *);
92int tcp_congctl_select(struct tcpcb *, const char *);
93void tcp_congctl_release(struct tcpcb *);
94const char *
95 tcp_congctl_bystruct(const struct tcp_congctl *);
96
97#endif /* _NETINET_TCP_CONGCTL_H */
