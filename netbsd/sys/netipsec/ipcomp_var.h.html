34#ifndef _NETIPSEC_IPCOMP_VAR_H_
35#define _NETIPSEC_IPCOMP_VAR_H_
36
37#define IPCOMP_STAT_HDROPS 0 /* packet shorter than header shows */
38#define IPCOMP_STAT_NOPF 1 /* protocol family not supported */
39#define IPCOMP_STAT_NOTDB 2
40#define IPCOMP_STAT_BADKCR 3
41#define IPCOMP_STAT_QFULL 4
42#define IPCOMP_STAT_NOXFORM 5
43#define IPCOMP_STAT_WRAP 6
44#define IPCOMP_STAT_INPUT 7 /* input IPcomp packets */
45#define IPCOMP_STAT_OUTPUT 8 /* output IPcomp packets */
46#define IPCOMP_STAT_INVALID 9 /* trying to use an invalid TDB */
47#define IPCOMP_STAT_IBYTES 10 /* input bytes */
48#define IPCOMP_STAT_OBYTES 11 /* output bytes */
49#define IPCOMP_STAT_TOOBIG 12 /* packet got larger than IP_MAXPACKET */
50#define IPCOMP_STAT_MINLEN 13 /* packet too short to compress */
51#define IPCOMP_STAT_USELESS 14 /* packet compressed > initial packet */
52#define IPCOMP_STAT_PDROPS 15 /* packet blocked due to policy */
53#define IPCOMP_STAT_CRYPTO 16 /* crypto processing failure */
54#define IPCOMP_STAT_HIST 17 /* per-algorithm op count */
55
56/* space for SADB_CALG_STATS_NUM counters */
57#define IPCOMP_ALG_MAX SADB_CALG_STATS_NUM
58#define IPCOMP_ALG_STR SADB_CALG_STATS_STR
59#define IPCOMP_NSTATS (IPCOMP_STAT_HIST + IPCOMP_ALG_MAX)
60
61#ifdef _KERNEL
62extern const uint8_t ipcomp_stats[256];
63extern int ipcomp_enable;
64#endif /* _KERNEL */
65#endif /* !_NETIPSEC_IPCOMP_VAR_H_ */
