74#ifdef NET_MPSAFE
75#define IPSEC_DECLARE_LOCK_VARIABLE
76#define IPSEC_ACQUIRE_GLOBAL_LOCKS() do { } while (0)
77#define IPSEC_RELEASE_GLOBAL_LOCKS() do { } while (0)
78#else
79#include <sys/socketvar.h> /* for softnet_lock */
80
81#define IPSEC_DECLARE_LOCK_VARIABLE int __s
82#define IPSEC_ACQUIRE_GLOBAL_LOCKS() \
83 do { \
84 __s = splsoftnet(); \
85 mutex_enter(softnet_lock); \
86 } while (0)
87#define IPSEC_RELEASE_GLOBAL_LOCKS() \
88 do { \
89 mutex_exit(softnet_lock); \
90 splx(__s); \
91 } while (0)
92#endif
