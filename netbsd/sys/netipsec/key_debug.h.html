34#ifndef _NETIPSEC_KEY_DEBUG_H_
35#define _NETIPSEC_KEY_DEBUG_H_
36
37#ifdef _KERNEL
38/* debug flags */
39#define KEYDEBUG_STAMP 0x00000001 /* path */
40#define KEYDEBUG_DATA 0x00000002 /* data */
41#define KEYDEBUG_DUMP 0x00000004 /* dump */
42#define KEYDEBUG_MATCH 0x00000008 /* match */
43
44#define KEYDEBUG_KEY 0x00000010 /* key processing */
45#define KEYDEBUG_ALG 0x00000020 /* ciph & auth algorithm */
46#define KEYDEBUG_IPSEC 0x00000040 /* ipsec processing */
47
48#define KEYDEBUG_KEY_STAMP (KEYDEBUG_KEY | KEYDEBUG_STAMP)
49#define KEYDEBUG_KEY_DATA (KEYDEBUG_KEY | KEYDEBUG_DATA)
50#define KEYDEBUG_KEY_DUMP (KEYDEBUG_KEY | KEYDEBUG_DUMP)
51#define KEYDEBUG_ALG_STAMP (KEYDEBUG_ALG | KEYDEBUG_STAMP)
52#define KEYDEBUG_ALG_DATA (KEYDEBUG_ALG | KEYDEBUG_DATA)
53#define KEYDEBUG_ALG_DUMP (KEYDEBUG_ALG | KEYDEBUG_DUMP)
54#define KEYDEBUG_IPSEC_STAMP (KEYDEBUG_IPSEC | KEYDEBUG_STAMP)
55#define KEYDEBUG_IPSEC_DATA (KEYDEBUG_IPSEC | KEYDEBUG_DATA)
56#define KEYDEBUG_IPSEC_DUMP (KEYDEBUG_IPSEC | KEYDEBUG_DUMP)
57
58#define KEYDEBUG_ON(lev) ((key_debug_level & (lev)) == (lev))
59
60#define KEYDEBUG_PRINTF(lev, fmt, ...) \
61 do { \
62 if (KEYDEBUG_ON((lev))) \
63 log(LOG_DEBUG, "%s: " fmt, __func__, \
64 __VA_ARGS__); \
65 } while (0)
66
67extern u_int32_t key_debug_level;
68#endif /*_KERNEL*/
