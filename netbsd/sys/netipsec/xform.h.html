42#ifndef _NETIPSEC_XFORM_H_
43#define _NETIPSEC_XFORM_H_

45#include <sys/types.h>
46#include <netinet/in.h>
47#include <opencrypto/xform.h>
48
49/*
50 * Opaque data structure hung off a crypto operation descriptor.
51 */
52struct secasvar;
53struct tdb_crypto {
54 const struct ipsecrequest *tc_isr; /* ipsec request state */
55 u_int32_t tc_spi; /* associated SPI */
56 union sockaddr_union tc_dst; /* dst addr of packet */
57 u_int8_t tc_proto; /* current protocol, e.g. AH */
58 u_int8_t tc_nxt; /* next protocol, e.g. IPV4 */
59 int tc_protoff; /* current protocol offset */
60 int tc_skip; /* data offset */
61 struct secasvar *tc_sav; /* ipsec SA */
62};
63
64struct ipescrequest;
65
66struct xformsw {
67 u_short xf_type;
68#define XF_IP4 1 /* IP inside IP */
69#define XF_AH 2 /* AH */
70#define XF_ESP 3 /* ESP */
71#define XF_TCPSIGNATURE 5 /* TCP MD5 Signature option, RFC 2358 */
72#define XF_IPCOMP 6 /* IPCOMP */
73 u_short xf_flags;
74#define XFT_AUTH 0x0001
75#define XFT_CONF 0x0100
76#define XFT_COMP 0x1000
77 const char *xf_name;
78 int (*xf_init)(struct secasvar *, const struct xformsw *);
79 int (*xf_zeroize)(struct secasvar *);
80 int (*xf_input)(struct mbuf *, struct secasvar *, int, int);
81 int (*xf_output)(struct mbuf *, const struct ipsecrequest *,
82 struct secasvar *, int, int);
83 struct xformsw *xf_next; /* list of registered xforms */
84};
85
86#ifdef _KERNEL
87void xform_register(struct xformsw *);
88int xform_init(struct secasvar *sav, int);
89
90struct cryptoini;
91
92/* XF_IP4 */
93int ipip_output(struct mbuf *, struct secasvar *, struct mbuf **);
94
95/* XF_AH */
96int ah_init0(struct secasvar *, const struct xformsw *, struct cryptoini *);
97int ah_zeroize(struct secasvar *);
98const struct auth_hash *ah_algorithm_lookup(int);
99size_t ah_authsiz(const struct secasvar *);
100size_t ah_hdrsiz(const struct secasvar *);
101
102/* XF_ESP */
103const struct enc_xform *esp_algorithm_lookup(int);
104size_t esp_hdrsiz(const struct secasvar *);
105
106/* XF_COMP */
107const struct comp_algo *ipcomp_algorithm_lookup(int);
108
109#endif /* _KERNEL */
110#endif /* !_NETIPSEC_XFORM_H_ */
