40#include <net/if.h>

43#define MPLS_LABEL_IPV4NULL 0 /* IPv4 Explicit NULL Label */
44#define MPLS_LABEL_RTALERT 1 /* Router Alert Label */
45#define MPLS_LABEL_IPV6NULL 2 /* IPv6 Explicit NULL Label */
46#define MPLS_LABEL_IMPLNULL 3 /* Implicit NULL Label */
47#define MPLS_LABEL_RESMAX 15 /* Maximum reserved Label */
48
49union mpls_shim {
50 uint32_t s_addr; /* the whole shim */
51 struct {
52#if BYTE_ORDER == LITTLE_ENDIAN
53 uint32_t ttl:8;
54 uint32_t bos:1;
55 uint32_t exp:3;
56 uint32_t label:20;
57#else
58 uint32_t label:20;
59 uint32_t exp:3;
60 uint32_t bos:1;
61 uint32_t ttl:8;
62#endif
63 } shim;
64};
65
66struct sockaddr_mpls {
67 uint8_t smpls_len;
68 uint8_t smpls_family;
69 uint8_t smpls_pad[2];
70 union mpls_shim smpls_addr;
71};
72__CTASSERT(sizeof(struct sockaddr_mpls) == 8);
73
74#endif /* !_NETMPLS_MPLS_H_ */
