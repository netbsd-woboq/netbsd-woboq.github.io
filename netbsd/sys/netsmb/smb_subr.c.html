67void
68smb_makescred(struct smb_cred *scred, struct lwp *l, kauth_cred_t cred)
69{
70 if (l) {
71 scred->scr_l = l;
72 scred->scr_cred = cred ? cred : l->l_cred;
73 } else {
74 scred->scr_l = NULL;
75 scred->scr_cred = cred ? cred : NULL;
76 }
77}

79int
80smb_proc_intr(struct lwp *l)
81{
82 struct proc *p;
83 int error;
84
85 if (l == NULL)
86 return 0;
87 p = l->l_proc;
88
89 mutex_enter(p->p_lock);
90 error = sigispending(l, 0);
91 mutex_exit(p->p_lock);
92
93 return (error != 0 ? EINTR : 0);
94}

96char *
97smb_strdup(const char *s)
98{
99 char *p;
100 size_t len;
101
102 len = s ? strlen(s) + 1 : 1;
103 p = malloc(len, M_SMBSTR, M_WAITOK);
104 if (s)
105 memcpy(p, s, len);
106 else
107 *p = 0;
108 return p;
109}

111/*
112 * duplicate string from a user space.
113 */
114char *
115smb_strdupin(char *s, size_t maxlen)
116{
117 char *p;
118 int error;
119
120 p = malloc(maxlen + 1, M_SMBSTR, M_WAITOK);
121 error = copyinstr(s, p, maxlen + 1, NULL);
122 if (error) {
123 free(p, M_SMBSTR);
124 return NULL;
125 }
126 return p;
127}
128
129/*
130 * duplicate memory block from a user space.
131 */
132void *
133smb_memdupin(void *umem, size_t len)
134{
135 char *p;
136
137 if (len > 8 * 1024)
138 return NULL;
139 p = malloc(len, M_SMBSTR, M_WAITOK);
140 if (copyin(umem, p, len) == 0)
141 return p;
142 free(p, M_SMBSTR);
143 return NULL;
144}

193int
194smb_maperror(int eclass, int eno)
195{
196 if (eclass == 0 && eno == 0)
197 return 0;
198 switch (eclass) {
199 case ERRDOS:
200 switch (eno) {
201 case ERRbadfunc:
202 case ERRbadmcb:
203 case ERRbadenv:
204 case ERRbadformat:
205 case ERRrmuns:
206 return EINVAL;
207 case ERRnofiles:
208 case ERRbadfile:
209 case ERRbadpath:
210 case ERRremcd:
211 case ERRnoipc: /* nt returns it when share not available */
212 case ERRnosuchshare: /* observed from nt4sp6 when sharename wrong */
213 return ENOENT;
214 case ERRnofids:
215 return EMFILE;
216 case ERRnoaccess:
217 case ERRbadshare:
218 return EACCES;
219 case ERRbadfid:
220 return EBADF;
221 case ERRnomem:
222 return ENOMEM; /* actually remote no mem... */
223 case ERRbadmem:
224 return EFAULT;
225 case ERRbadaccess:
226 return EACCES;
227 case ERRbaddata:
228 return E2BIG;
229 case ERRbaddrive:
230 case ERRnotready: /* nt */
231 return ENXIO;
232 case ERRdiffdevice:
233 return EXDEV;
234 case ERRlock:
235 return EDEADLK;
236 case ERRfilexists:
237 return EEXIST;
238 case ERRinvalidname: /* dunno what is it, but samba maps as noent */
239 return ENOENT;
240 case ERRdirnempty: /* samba */
241 return ENOTEMPTY;
242 case ERRrename:
243 return EEXIST;
244 case ERRquota:
245 return EDQUOT;
246 case ERRnotlocked:
247 /* it's okay to try to unlock already unlocked file */
248 return 0;
249 case NT_STATUS_NOTIFY_ENUM_DIR:
250 return EMSGSIZE;
251 }
252 break;
253 case ERRSRV:
254 switch (eno) {
255 case ERRerror:
256 return EINVAL;
257 case ERRbadpw:
258 case ERRpasswordExpired:
259 case ERRbaduid:
260 return EAUTH;
261 case ERRaccess:
262 return EACCES;
263 case ERRinvnid:
264 return ENETRESET;
265 case ERRinvnetname:
266 SMBERROR(("NetBIOS name is invalid\n"));
267 return EAUTH;
268 case ERRbadtype: /* reserved and returned */
269 return EIO;
270 case ERRaccountExpired:
271 case ERRbadClient:
272 case ERRbadLogonTime:
273 return EPERM;
274 case ERRnosupport:
275 return EBADRPC;
276 }
277 break;
278 case ERRHRD:
279 switch (eno) {
280 case ERRnowrite:
281 return EROFS;
282 case ERRbadunit:
283 return ENODEV;
284 case ERRnotready:
285 case ERRbadcmd:
286 case ERRdata:
287 return EIO;
288 case ERRbadreq:
289 return EBADRPC;
290 case ERRbadshare:
291 return ETXTBSY;
292 case ERRlock:
293 return EDEADLK;
294 case ERRgeneral:
295 /* returned e.g. for NT CANCEL SMB by Samba */
296 return ECANCELED;
297 }
298 break;
299 }
300 SMBERROR(("Unmapped error %d:%d\n", eclass, eno));
301 return EBADRPC;
302}

304static int
305smb_copy_iconv(struct mbchain *mbp, const char *src, char *dst,
306 size_t *srclen, size_t *dstlen)
307{
308 int error;
309 size_t inlen = *srclen, outlen = *dstlen;
310
311 error = iconv_conv((struct iconv_drv*)mbp->mb_udata, &src, &inlen,
312 &dst, &outlen);
313 if (inlen != *srclen || outlen != *dstlen) {
314 *srclen -= inlen;
315 *dstlen -= outlen;
316 return 0;
317 } else
318 return error;
319}

321int
322smb_put_dmem(struct mbchain *mbp, struct smb_vc *vcp, const char *src,
323 size_t size, int caseopt)
324{
325 struct iconv_drv *dp = vcp->vc_toserver;
326
327 if (size == 0)
328 return 0;
329 if (dp == NULL) {
330 return mb_put_mem(mbp, (const void *)src, size, MB_MSYSTEM);
331 }
332 mbp->mb_copy = smb_copy_iconv;
333 mbp->mb_udata = dp;
334 return mb_put_mem(mbp, (const void *)src, size, MB_MCUSTOM);
335}

337int
338smb_put_dstring(struct mbchain *mbp, struct smb_vc *vcp, const char *src,
339 int caseopt)
340{
341 int error;
342
343 error = smb_put_dmem(mbp, vcp, src, strlen(src), caseopt);
344 if (error)
345 return error;
346 return mb_put_uint8(mbp, 0);
347}

368struct sockaddr *
369dup_sockaddr(struct sockaddr *sa, int canwait)
370{
371 struct sockaddr *sa2;
372
373 sa2 = malloc(sa->sa_len, M_SONAME, canwait ? M_WAITOK : M_NOWAIT);
374 if (sa2)
375 memcpy(sa2, sa, sa->sa_len);
376 return sa2;
377}

379int
380dup_sockaddr_copyin(struct sockaddr **ksap, struct sockaddr *usa,
381 size_t usalen)
382{
383 struct sockaddr *ksa;
384
385 /* Make sure user provided enough data for a generic sockaddr. */
386 if (usalen < sizeof(*ksa))
387 return EINVAL;
388
389 /* Don't let the user overfeed us. */
390 usalen = MIN(usalen, sizeof(struct sockaddr_storage));
391
392 /* Copy the buffer in from userland. */
393 ksa = smb_memdupin(usa, usalen);
394 if (ksa == NULL)
395 return ENOMEM;
396
397 /* Make sure the user's idea of sa_len is reasonable. */
398 if (ksa->sa_len > usalen) {
399 smb_memfree(ksa);
400 return EINVAL;
401 }
402
403 /* Success! */
404 *ksap = ksa;
405 return 0;
406}
