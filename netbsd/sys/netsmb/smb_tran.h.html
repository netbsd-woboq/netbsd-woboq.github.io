37#ifndef _NETSMB_SMB_TRAN_H_
38#define _NETSMB_SMB_TRAN_H_
39
40#ifndef _KERNEL
41#error not supposed to be exposed to userland.
42#endif /* !_KERNEL */
43
44#include <sys/socket.h>
45
46/*
47 * Known transports
48 */
49#define SMBT_NBTCP 1
50
51/*
52 * Transport parameters
53 */
54#define SMBTP_SNDSZ 1 /* R - int */
55#define SMBTP_RCVSZ 2 /* R - int */
56#define SMBTP_TIMEOUT 3 /* RW - struct timespec */
57#define SMBTP_SELECTID 4 /* RW - (void *) */
58
59struct smb_tran_ops;
60
61struct smb_tran_desc {
62 sa_family_t tr_type;
63 int (*tr_create)(struct smb_vc *vcp, struct lwp *l);
64 int (*tr_done)(struct smb_vc *vcp, struct lwp *l);
65 int (*tr_bind)(struct smb_vc *vcp, struct sockaddr *sap, struct lwp *l);
66 int (*tr_connect)(struct smb_vc *vcp, struct sockaddr *sap, struct lwp *l);
67 int (*tr_disconnect)(struct smb_vc *vcp, struct lwp *l);
68 int (*tr_send)(struct smb_vc *vcp, struct mbuf *m0, struct lwp *l);
69 int (*tr_recv)(struct smb_vc *vcp, struct mbuf **mpp, struct lwp *l);
70 void (*tr_timo)(struct smb_vc *vcp);
71 void (*tr_intr)(struct smb_vc *vcp);
72 int (*tr_getparam)(struct smb_vc *vcp, int param, void *data);
73 int (*tr_setparam)(struct smb_vc *vcp, int param, void *data);
74 int (*tr_fatal)(struct smb_vc *vcp, int error);
75#ifdef notyet
76 int (*tr_poll)(struct smb_vc *vcp, struct lwp *l);
77 int (*tr_cmpaddr)(void *addr1, void *addr2);
78#endif
79 LIST_ENTRY(smb_tran_desc) tr_link;
80};
81
82#define SMB_TRAN_CREATE(vcp,p) (vcp)->vc_tdesc->tr_create(vcp,p)
83#define SMB_TRAN_DONE(vcp,p) (vcp)->vc_tdesc->tr_done(vcp,p)
84#define SMB_TRAN_BIND(vcp,sap,p) (vcp)->vc_tdesc->tr_bind(vcp,sap,p)
85#define SMB_TRAN_CONNECT(vcp,sap,p) (vcp)->vc_tdesc->tr_connect(vcp,sap,p)
86#define SMB_TRAN_DISCONNECT(vcp,p) (vcp)->vc_tdesc->tr_disconnect(vcp,p)
87#define SMB_TRAN_SEND(vcp,m0,p) (vcp)->vc_tdesc->tr_send(vcp,m0,p)
88#define SMB_TRAN_RECV(vcp,m,p) (vcp)->vc_tdesc->tr_recv(vcp,m,p)
89#define SMB_TRAN_TIMO(vcp) (vcp)->vc_tdesc->tr_timo(vcp)
90#define SMB_TRAN_INTR(vcp) (vcp)->vc_tdesc->tr_intr(vcp)
91#define SMB_TRAN_GETPARAM(vcp,par,data) (vcp)->vc_tdesc->tr_getparam(vcp, par, data)
92#define SMB_TRAN_SETPARAM(vcp,par,data) (vcp)->vc_tdesc->tr_setparam(vcp, par, data)
93#define SMB_TRAN_FATAL(vcp, error) (vcp)->vc_tdesc->tr_fatal(vcp, error)
94
95#endif /* !_NETSMB_SMB_TRAN_H_ */
