51enum nbstate {
52 NBST_CLOSED,
53 NBST_RQSENT,
54 NBST_SESSION,
55 NBST_RETARGET,
56 NBST_REFUSED
57};

60/*
61 * socket specific data
62 */
63struct nbpcb {
64 struct smb_vc * nbp_vc;
65 struct socket * nbp_tso; /* transport socket */
66 struct sockaddr_nb *nbp_laddr; /* local address */
67 struct sockaddr_nb *nbp_paddr; /* peer address */
68
69 int nbp_flags;
70#define NBF_LOCADDR 0x0001 /* has local addr */
71#define NBF_CONNECTED 0x0002
72#define NBF_RECVLOCK 0x0004
73
74 enum nbstate nbp_state;
75 void * nbp_selectid;
76};

78/*
79 * Nominal space allocated per a NETBIOS socket.
80 */
81#define NB_SNDQ (64 * 1024)
82#define NB_RCVQ (64 * 1024)
83
84/*
85 * Timeouts (s) used for send/receive. XXX Sysctl this?
86 */
87#define NB_SNDTIMEO (5)
88#define NB_RCVTIMEO (5)
89
90/*
91 * TCP slowstart presents a problem in conjunction with large
92 * reads. To ensure a steady stream of ACKs while reading using
93 * large transaction sizes, we call soreceive() with a smaller
94 * buffer size. See nbssn_recv().
95 */
96#define NB_SORECEIVE_CHUNK (8 * 1024)
