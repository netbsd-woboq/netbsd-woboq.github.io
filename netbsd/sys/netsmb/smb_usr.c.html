64static int
65smb_usr_vc2spec(struct smbioc_ossn *dp, struct smb_vcspec *spec)
66{
67 int flags = 0;
68 int error;
69
70 memset(spec, 0, sizeof(*spec));
71 if (dp->ioc_user[0] == 0)
72 return EINVAL;
73 if (dp->ioc_server == NULL)
74 return EINVAL;
75 if (dp->ioc_localcs[0] == 0) {
76 SMBERROR(("no local charset ?\n"));
77 return EINVAL;
78 }
79
80 error = dup_sockaddr_copyin(&spec->sap, dp->ioc_server, dp->ioc_svlen);
81 if (error)
82 return error;
83 if (dp->ioc_local) {
84 error = dup_sockaddr_copyin(&spec->lap, dp->ioc_local,
85 dp->ioc_lolen);
86 if (error) {
87 smb_usr_vcspec_free(spec);
88 return error;
89 }
90 }
91 spec->srvname = dp->ioc_srvname;
92 spec->pass = dp->ioc_password;
93 spec->domain = dp->ioc_workgroup;
94 spec->username = dp->ioc_user;
95 spec->mode = dp->ioc_mode;
96 spec->rights = dp->ioc_rights;
97 spec->owner = dp->ioc_owner;
98 spec->group = dp->ioc_group;
99 spec->localcs = dp->ioc_localcs;
100 spec->servercs = dp->ioc_servercs;
101 if (dp->ioc_opt & SMBVOPT_PRIVATE)
102 flags |= SMBV_PRIVATE;
103 if (dp->ioc_opt & SMBVOPT_SINGLESHARE)
104 flags |= SMBV_PRIVATE | SMBV_SINGLESHARE;
105 spec->flags = flags;
106 return 0;
107}
108
109static void
110smb_usr_vcspec_free(struct smb_vcspec *spec)
111{
112 if (spec->sap)
113 smb_memfree(spec->sap);
114 if (spec->lap)
115 smb_memfree(spec->lap);
116}

118static int
119smb_usr_share2spec(struct smbioc_oshare *dp, struct smb_sharespec *spec)
120{
121 memset(spec, 0, sizeof(*spec));
122 spec->mode = dp->ioc_mode;
123 spec->rights = dp->ioc_rights;
124 spec->owner = dp->ioc_owner;
125 spec->group = dp->ioc_group;
126 spec->name = dp->ioc_share;
127 spec->stype = dp->ioc_stype;
128 spec->pass = dp->ioc_password;
129 return 0;
130}

132int
133smb_usr_lookup(struct smbioc_lookup *dp, struct smb_cred *scred,
134 struct smb_vc **vcpp, struct smb_share **sspp)
135{
136 struct smb_vc *vcp = NULL;
137 struct smb_vcspec vspec;
138 struct smb_sharespec sspec, *sspecp = NULL;
139 int error;
140
141 if (dp->ioc_level < SMBL_VC || dp->ioc_level > SMBL_SHARE)
142 return EINVAL;
143 error = smb_usr_vc2spec(&dp->ioc_ssn, &vspec);
144 if (error)
145 return error;
146 if (dp->ioc_flags & SMBLK_CREATE)
147 vspec.flags |= SMBV_CREATE;
148
149 if (dp->ioc_level >= SMBL_SHARE) {
150 error = smb_usr_share2spec(&dp->ioc_sh, &sspec);
151 if (error)
152 goto out;
153 sspecp = &sspec;
154 }
155 error = smb_sm_lookup(&vspec, sspecp, scred, &vcp);
156 if (error == 0) {
157 *vcpp = vcp;
158 *sspp = vspec.ssp;
159 }
160out:
161 smb_usr_vcspec_free(&vspec);
162 return error;
163}

165/*
166 * Connect to the resource specified by smbioc_ossn structure.
167 * It may either find an existing connection or try to establish a new one.
168 * If no errors occurred smb_vc returned locked and referenced.
169 */
170int
171smb_usr_opensession(struct smbioc_ossn *dp, struct smb_cred *scred,
172 struct smb_vc **vcpp)
173{
174 struct smb_vc *vcp = NULL;
175 struct smb_vcspec vspec;
176 int error;
177
178 error = smb_usr_vc2spec(dp, &vspec);
179 if (error)
180 return error;
181 if (dp->ioc_opt & SMBVOPT_CREATE)
182 vspec.flags |= SMBV_CREATE;
183
184 error = smb_sm_lookup(&vspec, NULL, scred, &vcp);
185 smb_usr_vcspec_free(&vspec);
186 return error;
187}

189int
190smb_usr_openshare(struct smb_vc *vcp, struct smbioc_oshare *dp,
191 struct smb_cred *scred, struct smb_share **sspp)
192{
193 struct smb_share *ssp;
194 struct smb_sharespec shspec;
195 int error;
196
197 error = smb_usr_share2spec(dp, &shspec);
198 if (error)
199 return error;
200 error = smb_vc_lookupshare(vcp, &shspec, scred, &ssp);
201 if (error == 0) {
202 *sspp = ssp;
203 return 0;
204 }
205 if ((dp->ioc_opt & SMBSOPT_CREATE) == 0)
206 return error;
207 error = smb_share_create(vcp, &shspec, scred, &ssp);
208 if (error)
209 return error;
210 error = smb_smb_treeconnect(ssp, scred);
211 if (error) {
212 smb_share_put(ssp, scred);
213 } else
214 *sspp = ssp;
215 return error;
216}

218int
219smb_usr_simplerequest(struct smb_share *ssp, struct smbioc_rq *dp,
220 struct smb_cred *scred)
221{
222 struct smb_rq *rqp;
223 struct mbchain *mbp;
224 struct mdchain *mdp;
225 u_int8_t wc;
226 u_int16_t bc;
227 int error;
228
229 switch (dp->ioc_cmd) {
230 case SMB_COM_TRANSACTION2:
231 case SMB_COM_TRANSACTION2_SECONDARY:
232 case SMB_COM_CLOSE_AND_TREE_DISC:
233 case SMB_COM_TREE_CONNECT:
234 case SMB_COM_TREE_DISCONNECT:
235 case SMB_COM_NEGOTIATE:
236 case SMB_COM_SESSION_SETUP_ANDX:
237 case SMB_COM_LOGOFF_ANDX:
238 case SMB_COM_TREE_CONNECT_ANDX:
239 return EPERM;
240 }
241 error = smb_rq_alloc(SSTOCP(ssp), dp->ioc_cmd, scred, &rqp);
242 if (error)
243 return error;
244 mbp = &rqp->sr_rq;
245 smb_rq_wstart(rqp);
246 error = mb_put_mem(mbp, dp->ioc_twords, dp->ioc_twc * 2, MB_MUSER);
247 if (error)
248 goto bad;
249 smb_rq_wend(rqp);
250 smb_rq_bstart(rqp);
251 error = mb_put_mem(mbp, dp->ioc_tbytes, dp->ioc_tbc, MB_MUSER);
252 if (error)
253 goto bad;
254 smb_rq_bend(rqp);
255 error = smb_rq_simple(rqp);
256 if (error)
257 goto bad;
258 mdp = &rqp->sr_rp;
259 md_get_uint8(mdp, &wc);
260 dp->ioc_rwc = wc;
261 wc *= 2;
262 if (wc > dp->ioc_rpbufsz) {
263 error = EBADRPC;
264 goto bad;
265 }
266 error = md_get_mem(mdp, dp->ioc_rpbuf, wc, MB_MUSER);
267 if (error)
268 goto bad;
269 md_get_uint16le(mdp, &bc);
270 if ((wc + bc) > dp->ioc_rpbufsz) {
271 error = EBADRPC;
272 goto bad;
273 }
274 dp->ioc_rbc = bc;
275 error = md_get_mem(mdp, dp->ioc_rpbuf + wc, bc, MB_MUSER);
276bad:
277 smb_rq_done(rqp);
278 return error;
279
280}

282static int
283smb_cpdatain(struct mbchain *mbp, int len, void *data)
284{
285 int error;
286
287 if (len == 0)
288 return 0;
289 error = mb_init(mbp);
290 if (error)
291 return error;
292 return mb_put_mem(mbp, data, len, MB_MUSER);
293}

295int
296smb_usr_t2request(struct smb_share *ssp, struct smbioc_t2rq *dp,
297 struct smb_cred *scred)
298{
299 struct smb_t2rq *t2p;
300 struct mdchain *mdp;
301 int error, len;
302
303 if (dp->ioc_setupcnt < 0 || dp->ioc_setupcnt > 3)
304 return EINVAL;
305 error = smb_t2_alloc(SSTOCP(ssp), dp->ioc_setup[0], scred, &t2p);
306 if (error)
307 return error;
308 len = t2p->t2_setupcount = dp->ioc_setupcnt;
309 if (len > 1)
310 t2p->t2_setupdata = dp->ioc_setup;
311 if (dp->ioc_name) {
312 t2p->t_name = smb_strdupin(dp->ioc_name, 128);
313 if (t2p->t_name == NULL) {
314 error = ENOMEM;
315 goto bad;
316 }
317 }
318 t2p->t2_maxscount = 0;
319 t2p->t2_maxpcount = dp->ioc_rparamcnt;
320 t2p->t2_maxdcount = dp->ioc_rdatacnt;
321 error = smb_cpdatain(&t2p->t2_tparam, dp->ioc_tparamcnt, dp->ioc_tparam);
322 if (error)
323 goto bad;
324 error = smb_cpdatain(&t2p->t2_tdata, dp->ioc_tdatacnt, dp->ioc_tdata);
325 if (error)
326 goto bad;
327 error = smb_t2_request(t2p);
328 if (error)
329 goto bad;
330 mdp = &t2p->t2_rparam;
331 if (mdp->md_top) {
332 len = m_fixhdr(mdp->md_top);
333 if (len > dp->ioc_rparamcnt) {
334 error = EMSGSIZE;
335 goto bad;
336 }
337 dp->ioc_rparamcnt = len;
338 error = md_get_mem(mdp, dp->ioc_rparam, len, MB_MUSER);
339 if (error)
340 goto bad;
341 } else
342 dp->ioc_rparamcnt = 0;
343 mdp = &t2p->t2_rdata;
344 if (mdp->md_top) {
345 len = m_fixhdr(mdp->md_top);
346 if (len > dp->ioc_rdatacnt) {
347 error = EMSGSIZE;
348 goto bad;
349 }
350 dp->ioc_rdatacnt = len;
351 error = md_get_mem(mdp, dp->ioc_rdata, len, MB_MUSER);
352 } else
353 dp->ioc_rdatacnt = 0;
354bad:
355 if (t2p->t_name)
356 smb_strfree(t2p->t_name);
357 smb_t2_done(t2p);
358 return error;
359}
