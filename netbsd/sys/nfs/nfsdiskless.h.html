36/*
37 * Structure holds parameters needed by nfs_mountroot(),
38 * which are filled in by nfs_boot_init() using either
39 * BOOTP (RFC951, RFC1048) or Sun RPC/bootparams. These
40 * parameters are INET specific because nfs_boot_init()
41 * currently supports only AF_INET protocols.
42 *
43 * NB: All fields are stored in net byte order to avoid hassles
44 * with client/server byte ordering differences.
45 */
46struct nfs_dlmount {
47 struct nfs_args ndm_args;
48 struct sockaddr ndm_saddr; /* Address of file server */
49 char ndm_host[MNAMELEN]; /* server:pathname */
50 u_char ndm_fh[NFSX_V3FHMAX]; /* The file's file handle */
51};
52struct nfs_diskless {
53 /* the interface used */
54 struct ifnet *nd_ifp;
55 /* A collection of IP addresses, for convenience. */
56 struct in_addr nd_myip; /* My IP address */
57 struct in_addr nd_mask; /* My netmask */
58 struct in_addr nd_gwip; /* My gateway */
59 int nd_mtu; /* Interface MTU */
60 /* Information for each mount point we need. */
61 struct nfs_dlmount nd_root; /* Mount info for root */
62#ifdef TFTPROOT
63 char nd_bootfile[MNAMELEN]; /* Boot file */
64 int nd_nomount;
65#endif
66};
