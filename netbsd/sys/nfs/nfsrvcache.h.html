38#ifndef _NFS_NFSRVCACHE_H_
39#define _NFS_NFSRVCACHE_H_
40
41/*
42 * Definitions for the server recent request cache
43 */
44
45#define NFSRVCACHESIZ 64
46
47struct nfsrvcache {
48 kcondvar_t rc_cv;
49 TAILQ_ENTRY(nfsrvcache) rc_lru; /* LRU chain */
50 LIST_ENTRY(nfsrvcache) rc_hash; /* Hash chain */
51 u_int32_t rc_xid; /* rpc id number */
52 union {
53 struct mbuf *ru_repmb; /* Reply mbuf list OR */
54 int ru_repstat; /* Reply status */
55 } rc_un;
56 union nethostaddr rc_haddr; /* Host address */
57 u_int32_t rc_proc; /* rpc proc number */
58 int rc_state; /* Current state of request */
59 int rc_gflags; /* Flag bits */
60 int rc_flags; /* Flag bits */
61};
62
63#define rc_reply rc_un.ru_repmb
64#define rc_status rc_un.ru_repstat
65#define rc_inetaddr rc_haddr.had_inetaddr
66#define rc_nam rc_haddr.had_nam
67
68/* Cache entry states */
69#define RC_UNUSED 0
70#define RC_INPROG 1
71#define RC_DONE 2
72
73/* Return values */
74#define RC_DROPIT 0
75#define RC_REPLY 1
76#define RC_DOIT 2
77#define RC_CHECKIT 3
78
79/* rc_gflags */
80#define RC_G_LOCKED 0x01
81
82/* rc_flags */
83#define RC_REPSTATUS 0x04
84#define RC_REPMBUF 0x08
85#define RC_INETADDR 0x20
86#define RC_NAM 0x40
87
88#endif
