31#include <sys/param.h>
32#include <sys/conf.h>
33#include <sys/device.h>
34#include <sys/mbuf.h>
35#include <sys/stat.h>
36
37#include <net/bpf.h>
38
39#include <rump-sys/kern.h>
40#include <rump-sys/vfs.h>
41
42RUMP_COMPONENT(RUMP_COMPONENT_NET)
43{
44 extern const struct cdevsw bpf_cdevsw;
45 devmajor_t bmaj, cmaj;
46 int error;
47
48 bmaj = cmaj = NODEVMAJOR;
49 if ((error = devsw_attach("bpf", NULL, &bmaj, &bpf_cdevsw, &cmaj)) != 0)
50 panic("bpf devsw attach failed: %d", error);
51 if ((error = rump_vfs_makeonedevnode(S_IFCHR, "/dev/bpf", cmaj, 0)) !=0)
52 panic("cannot create bpf device nodes: %d", error);
53 if ((error = devsw_detach(NULL, &bpf_cdevsw)) != 0)
54 panic("cannot detach bpf devsw: %d", error);
55}
