31#include <sys/param.h>
32#include <sys/conf.h>
33#include <sys/device.h>
34#include <sys/stat.h>
35
36#include <rump-sys/kern.h>
37#include <rump-sys/vfs.h>
38
39RUMP_COMPONENT(RUMP_COMPONENT_DEV)
40{
41 extern const struct bdevsw fss_bdevsw;
42 extern const struct cdevsw fss_cdevsw;
43 extern devmajor_t fss_bmajor, fss_cmajor;
44 int error;
45
46 fss_bmajor = bdevsw_lookup_major(&fss_bdevsw);
47 fss_cmajor = cdevsw_lookup_major(&fss_cdevsw);
48
49 if ((error = devsw_attach("fss", &fss_bdevsw, &fss_bmajor,
50 &fss_cdevsw, &fss_cmajor)) != 0)
51 panic("cannot attach fss: %d", error);
52
53 if ((error = rump_vfs_makedevnodes(S_IFBLK, "/dev/fss", '0',
54 fss_bmajor, 0, 4)) != 0)
55 panic("cannot create cooked fss dev nodes: %d", error);
56 if ((error = rump_vfs_makedevnodes(S_IFCHR, "/dev/rfss", '0',
57 fss_cmajor, 0, 4)) != 0)
58 panic("cannot create raw fss dev nodes: %d", error);
59
60 devsw_detach(&fss_bdevsw, &fss_cdevsw);
61}
