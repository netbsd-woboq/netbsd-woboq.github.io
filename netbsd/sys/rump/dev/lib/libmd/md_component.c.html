43extern void mdattach(int); /* XXX */
44
45RUMP_COMPONENT(RUMP_COMPONENT_DEV)
46{
47 extern const struct bdevsw md_bdevsw;
48 extern const struct cdevsw md_cdevsw;
49 devmajor_t bmaj, cmaj;
50 int error;
51
52 config_init_component(cfdriver_ioconf_md,
53 cfattach_ioconf_md, cfdata_ioconf_md);
54
55 bmaj = cmaj = NODEVMAJOR;
56 if ((error = devsw_attach("md", &md_bdevsw, &bmaj,
57 &md_cdevsw, &cmaj)) != 0)
58 panic("md devsw attach failed: %d", error);
59
60 if ((error = rump_vfs_makedevnodes(S_IFBLK, "/dev/md0", 'a',
61 bmaj, 0, 7)) != 0)
62 panic("cannot create cooked md dev nodes: %d", error);
63 if ((error = rump_vfs_makedevnodes(S_IFCHR, "/dev/rmd0", 'a',
64 cmaj, 0, 7)) != 0)
65 panic("cannot create raw md dev nodes: %d", error);
66
67 rump_pdev_add(mdattach, 0);
68}
