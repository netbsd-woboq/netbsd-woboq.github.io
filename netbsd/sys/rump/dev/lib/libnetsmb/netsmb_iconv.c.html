40int
41iconv_open(const char *to, const char *from, void **handle)
42{
43 if (strcmp(to, "tolower") && strcmp(to, "toupper"))
44 return rumpcomp_netsmb_iconv_open(to, from, handle);
45 return 0;
46}
47
48int
49iconv_close(void *handle)
50{
51 if (handle != NULL)
52 return rumpcomp_netsmb_iconv_close(handle);
53 return 0;
54}
55
56int
57iconv_conv(void *handle, const char **inbuf, size_t *inbytesleft,
58 char **outbuf, size_t *outbytesleft)
59{
60 size_t len;
61
62 if (handle != NULL)
63 return rumpcomp_netsmb_iconv_conv(handle, inbuf, inbytesleft,
64 outbuf, outbytesleft);
65
66 if (inbuf == NULL)
67 return 0;
68
69 if (*inbytesleft > *outbytesleft)
70 return E2BIG;
71
72 len = *inbytesleft;
73 memcpy(*outbuf, *inbuf, len);
74 *inbuf += len;
75 *inbytesleft = 0;
76 *outbuf += len;
77 *outbytesleft -= len;
78 return 0;
79}
