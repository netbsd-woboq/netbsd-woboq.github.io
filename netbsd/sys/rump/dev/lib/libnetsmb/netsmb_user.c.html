44int
45rumpcomp_netsmb_iconv_open(const char *to, const char *from, void **handle)
46{
47#ifdef __NetBSD__
48 iconv_t cd;
49 int rv;
50
51 cd = iconv_open(to, from);
52 if (cd == (iconv_t)-1)
53 rv = errno;
54 else {
55 if (handle != NULL)
56 *handle = (void *)cd;
57 rv = 0;
58 }
59
60 return rumpuser_component_errtrans(rv);
61#else
62 /* fallback to use dumb copy function */
63 return 0;
64#endif
65}
66
67int
68rumpcomp_netsmb_iconv_close(void *handle)
69{
70#ifdef __NetBSD__
71 int rv;
72
73 if (iconv_close((iconv_t)handle) == -1)
74 rv = errno;
75 else
76 rv = 0;
77
78 return rumpuser_component_errtrans(rv);
79#else
80 /* do nothing */
81 return 0;
82#endif
83}
84
85int
86rumpcomp_netsmb_iconv_conv(void *handle, const char **inbuf,
87 size_t *inbytesleft, char **outbuf, size_t *outbytesleft)
88{
89#ifdef __NetBSD__
90 int rv;
91
92 if (iconv((iconv_t)handle, inbuf, inbytesleft, outbuf, outbytesleft)
93 == (size_t)-1)
94 rv = errno;
95 else
96 rv = 0;
97
98 return rumpuser_component_errtrans(rv);
99#else
100 /* do nothing */
101 return 0;
102#endif
103}
104#endif
