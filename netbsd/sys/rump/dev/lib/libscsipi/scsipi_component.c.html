17RUMP_COMPONENT(RUMP_COMPONENT_DEV)
18{
19 extern struct bdevsw sd_bdevsw, cd_bdevsw;
20 extern struct cdevsw sd_cdevsw, cd_cdevsw;
21 devmajor_t bmaj, cmaj;
22
23 config_init_component(cfdriver_ioconf_scsipi,
24 cfattach_ioconf_scsipi, cfdata_ioconf_scsipi);
25
26 bmaj = cmaj = -1;
27 FLAWLESSCALL(devsw_attach("sd", &sd_bdevsw, &bmaj, &sd_cdevsw, &cmaj));
28
29 FLAWLESSCALL(rump_vfs_makedevnodes(S_IFBLK, "/dev/sd0", 'a',
30 bmaj, 0, 8));
31 FLAWLESSCALL(rump_vfs_makedevnodes(S_IFCHR, "/dev/rsd0", 'a',
32 cmaj, 0, 8));
33
34 bmaj = cmaj = -1;
35 FLAWLESSCALL(devsw_attach("cd", &cd_bdevsw, &bmaj, &cd_cdevsw, &cmaj));
36
37 FLAWLESSCALL(rump_vfs_makedevnodes(S_IFBLK, "/dev/cd0", 'a',
38 bmaj, 0, 8));
39 FLAWLESSCALL(rump_vfs_makedevnodes(S_IFCHR, "/dev/rcd0", 'a',
40 cmaj, 0, 8));
41}
