18void tty_init(void);
19
20RUMP_COMPONENT(RUMP_COMPONENT_DEV)
21{
22 extern struct cdevsw ucom_cdevsw;
23 devmajor_t cmaj, bmaj;
24
25 config_init_component(cfdriver_ioconf_ucom,
26 cfattach_ioconf_ucom, cfdata_ioconf_ucom);
27
28 bmaj = cmaj = -1;
29 FLAWLESSCALL(devsw_attach("ucom", NULL, &bmaj, &ucom_cdevsw, &cmaj));
30
31 FLAWLESSCALL(rump_vfs_makedevnodes(S_IFCHR, "/dev/ttyU", '0',
32 cmaj, 0, 2));
33 FLAWLESSCALL(rump_vfs_makedevnodes(S_IFCHR, "/dev/dtyU", '0',
34 cmaj, 0x80000, 2));
35
36 tty_init();
37}
