34#include <sys/types.h>
35#include <sys/ioctl.h>
36
37#include <errno.h>
38#include <unistd.h>
39
40#include <rump/rumpuser_component.h>
41
42#include "ugenhc_user.h"
43
44int
45rumpcomp_ugenhc_ioctl(int fd, u_long cmd, void *data, int *ioctlrv)
46{
47 void *cookie;
48 int rv;
49
50 cookie = rumpuser_component_unschedule();
51 *ioctlrv = ioctl(fd, cmd, data);
52 if (*ioctlrv == -1)
53 rv = errno;
54 else
55 rv = 0;
56 rumpuser_component_schedule(cookie);
57
58 return rumpuser_component_errtrans(rv);
59}
60#endif
