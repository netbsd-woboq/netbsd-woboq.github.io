81/*
82 * If RUMP_USE_CTOR is defined, we use __attribute__((constructor)) to
83 * determine which components are present when rump_init() is called.
84 * Otherwise, we use link sets and the __start/__stop symbols generated
85 * by the toolchain.
86 */
87
88#ifdef RUMP_USE_CTOR
89#define _RUMP_COMPONENT_REGISTER(type) \
90static void rumpcomp_ctor##type(void) __attribute__((constructor)); \
91static void rumpcomp_ctor##type(void) \
92{ \
93 rump_component_load(&rumpcomp##type); \
94} \
95static void rumpcomp_dtor##type(void) __attribute__((destructor)); \
96static void rumpcomp_dtor##type(void) \
97{ \
98 rump_component_unload(&rumpcomp##type); \
99}
100
101#else /* RUMP_USE_CTOR */
102
103#define _RUMP_COMPONENT_REGISTER(type) \
104__link_set_add_rodata(rump_components, rumpcomp##type);
105#endif /* RUMP_USE_CTOR */
106
107#define RUMP_COMPONENT(type) \
108static void rumpcompinit##type(void); \
109static struct rump_component rumpcomp##type = { \
110 .rc_type = type, \
111 .rc_init = rumpcompinit##type, \
112}; \
113_RUMP_COMPONENT_REGISTER(type) \
114static void rumpcompinit##type(void)

117do { \

210/*
211 * sysproxy is an optional component. The interfaces with "hyp"
212 * in the name come into the rump kernel from the client or sysproxy
213 * stub, the rest go out of the rump kernel.
214 */
215struct rump_sysproxy_ops {
216 int (*rspo_copyin)(void *, const void *, void *, size_t);
217 int (*rspo_copyinstr)(void *, const void *, void *, size_t *);
218 int (*rspo_copyout)(void *, const void *, void *, size_t);
219 int (*rspo_copyoutstr)(void *, const void *, void *, size_t *);
220 int (*rspo_anonmmap)(void *, size_t, void **);
221 int (*rspo_raise)(void *, int);
222 void (*rspo_fini)(void *);
223
224 pid_t (*rspo_hyp_getpid)(void);
225 int (*rspo_hyp_syscall)(int, void *, long *);
226 int (*rspo_hyp_rfork)(void *, int, const char *);
227 void (*rspo_hyp_lwpexit)(void);
228 void (*rspo_hyp_execnotify)(const char *);
229};
230extern struct rump_sysproxy_ops rump_sysproxy_ops;

236 rump_sysproxy_ops.rspo_copyout(arg, laddr, raddr, len)

241#define rump_sysproxy_raise(arg, signo) \
242 rump_sysproxy_ops.rspo_raise(arg, signo)
