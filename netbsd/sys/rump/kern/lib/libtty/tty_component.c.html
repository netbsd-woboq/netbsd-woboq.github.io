50RUMP_COMPONENT(RUMP_COMPONENT_KERN_VFS)
51{
52 extern const struct cdevsw ctty_cdevsw, ptc_cdevsw, pts_cdevsw;
53 extern const struct cdevsw ptm_cdevsw;
54 devmajor_t cmaj, bmaj;
55
56 cmaj = bmaj = -1;
57 FLAWLESSCALL(devsw_attach("ctty", NULL, &bmaj, &ctty_cdevsw, &cmaj));
58 FLAWLESSCALL(rump_vfs_makeonedevnode(S_IFCHR, "/dev/tty", cmaj, 0));
59
60 /* create only 10 nodes. should be enough for everyone ... */
61 cmaj = bmaj = -1;
62 FLAWLESSCALL(devsw_attach("ptc", NULL, &bmaj, &ptc_cdevsw, &cmaj));
63 FLAWLESSCALL(rump_vfs_makedevnodes(S_IFCHR, "/dev/ptyp", '0',
64 cmaj, 0, 9));
65
66 cmaj = bmaj = -1;
67 FLAWLESSCALL(devsw_attach("pts", NULL, &bmaj, &pts_cdevsw, &cmaj));
68 FLAWLESSCALL(rump_vfs_makedevnodes(S_IFCHR, "/dev/ttyp", '0',
69 cmaj, 0, 9));
70
71 cmaj = bmaj = -1;
72 FLAWLESSCALL(devsw_attach("ptm", NULL, &bmaj, &ptm_cdevsw, &cmaj));
73 FLAWLESSCALL(rump_vfs_makeonedevnode(S_IFCHR, "/dev/ptmx", cmaj, 0));
74 FLAWLESSCALL(rump_vfs_makeonedevnode(S_IFCHR, "/dev/ptm", cmaj, 1));
75
76 ptyattach(1);
77}
