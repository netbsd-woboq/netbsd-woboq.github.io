41const char *rootspec = "rump0a"; /* usually comes from config */
42
43RUMP_COMPONENT(RUMP__FACTION_DEV)
44{
45 extern int cold;
46
47 pmf_init();
48
49 KERNEL_LOCK(1, curlwp);
50
51 rump_mainbus_init();
52 config_init_mi();
53
54 rump_component_init(RUMP_COMPONENT_DEV);
55
56 rump_pdev_finalize();
57
58 cold = 0;
59
60 /*
61 * XXX: does the "if" make any sense? What if someone wants
62 * to dynamically load a driver later on?
63 */
64 if (rump_component_count(RUMP_COMPONENT_DEV) > 0
65 || rump_component_count(RUMP_COMPONENT_DEV_AFTERMAINBUS) > 0) {
66 rump_mainbus_attach();
67 if (config_rootfound("mainbus", NULL) == NULL)
68 panic("no mainbus");
69
70 rump_component_init(RUMP_COMPONENT_DEV_AFTERMAINBUS);
71 }
72 config_finalize();
73
74 KERNEL_UNLOCK_LAST(curlwp);
75
76 /* if there is a vfs, perform activity deferred until mountroot */
77 if (rump_component_count(RUMP__FACTION_VFS)) {
78 config_create_mountrootthreads();
79 yield();
80 }
81}
