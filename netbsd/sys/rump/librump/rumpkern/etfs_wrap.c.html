31#include <sys/param.h>
32
33#include <rump-sys/kern.h>
34
35rump_etfs_register_withsize_fn rump__etfs_register = (void *)enosys;
36rump_etfs_remove_fn rump__etfs_remove = (void *)enosys;
37
38int
39rump_etfs_register(const char *key, const char *hostpath,
40 enum rump_etfs_type ftype)
41{
42
43 return rump__etfs_register(key, hostpath, ftype,
44 0, RUMP_ETFS_SIZE_ENDOFF);
45}
46
47int
48rump_etfs_register_withsize(const char *key, const char *hostpath,
49 enum rump_etfs_type ftype, uint64_t begin, uint64_t size)
50{
51
52 return rump__etfs_register(key, hostpath, ftype, begin, size);
53}
54
55int
56rump_etfs_remove(const char *key)
57{
58
59 return rump__etfs_remove(key);
60}
