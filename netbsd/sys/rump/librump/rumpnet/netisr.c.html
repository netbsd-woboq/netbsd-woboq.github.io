40static void *netisrs[NETISR_MAX];
41
42void
43schednetisr(int isr)
44{
45 KASSERT(isr != NETISR_IP);
46 KASSERT(isr != NETISR_IPV6);
47
48 /*
49 * Do not schedule a softint that is not registered.
50 * This might cause the inq to fill, but the one calling us
51 * should start dropping packets once the inq is full,
52 * so no big harm done.
53 */
54 if (__predict_true(netisrs[isr]))
55 softint_schedule(netisrs[isr]);
56}
57
58void
59rump_netisr_register(int level, void (*handler)(void))
60{
61 KASSERT(level != NETISR_IP);
62 KASSERT(level != NETISR_IPV6);
63
64 netisrs[level] = softint_establish(SOFTINT_NET | SOFTINT_MPSAFE,
65 (void (*)(void *))handler, NULL);
66}
