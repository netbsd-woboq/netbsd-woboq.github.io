28/*
29 * /dev/null, the infamous bytesink.
30 *
31 * I can't imagine it being very different in the rump kernel than in
32 * the host kernel. But nonetheless it serves as a simple example.
33 */

45static dev_type_open(rump_devnullopen);
46static dev_type_read(rump_devnullrw);
47
48static struct cdevsw null_cdevsw = {
49 .d_open = rump_devnullopen,
50 .d_close = nullclose,
51 .d_read = rump_devnullrw,
52 .d_write = rump_devnullrw,
53 .d_ioctl = noioctl,
54 .d_stop = nostop,
55 .d_tty = notty,
56 .d_poll = nopoll,
57 .d_mmap = nommap,
58 .d_kqfilter = nokqfilter,
59 .d_discard = nodiscard,
60 .d_flag = D_OTHER | D_MPSAFE
61};
62
63int
64rump_devnull_init()
65{
66 devmajor_t null_bmaj, null_cmaj;
67 int error;
68
69 null_bmaj = null_cmaj = NODEVMAJOR;
70 error = devsw_attach("null", NULL, &null_bmaj, &null_cdevsw,&null_cmaj);
71 KASSERT(error || null_cmaj == 2);
72
73 error = rump_vfs_makeonedevnode(S_IFCHR,
74 "/dev/null", null_cmaj, DEV_NULL);
75 if (error)
76 return error;
77 return rump_vfs_makeonedevnode(S_IFCHR,
78 "/dev/zero", null_cmaj, DEV_ZERO);
79}
80
81static int
82rump_devnullopen(dev_t dev, int flag, int mode, struct lwp *l)
83{
84
85 if (minor(dev) == DEV_ZERO || minor(dev) == DEV_NULL)
86 return 0;
87 return ENXIO;
88}
89
90static int
91rump_devnullrw(dev_t dev, struct uio *uio, int flags)
92{
93 char zeros[512];
94 int error;
95
96 switch (minor(dev)) {
97 case DEV_NULL:
98 if (uio->uio_rw == UIO_WRITE)
99 uio->uio_resid = 0;
100 break;
101 case DEV_ZERO:
102 if (uio->uio_rw == UIO_WRITE) {
103 uio->uio_resid = 0;
104 break;
105 }
106 memset(zeros, 0, sizeof(zeros));
107 while (uio->uio_resid > 0) {
108 error = uiomove(zeros,
109 uimin(sizeof(zeros), uio->uio_resid), uio);
110 if (error)
111 return error;
112 }
113 break;
114 default:
115 return ENXIO; /* how? */
116 }
117
118 return 0;
119}
