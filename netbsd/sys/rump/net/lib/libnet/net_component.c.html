44RUMP_COMPONENT(RUMP_COMPONENT_NET)
45{
46 ifinit1();
47 ifinit();
48 lltableinit();
49}
50
51RUMP_COMPONENT(RUMP_COMPONENT_NET_ROUTE)
52{
53 extern struct domain routedomain, linkdomain;
54#ifdef COMPAT_50
55 extern struct domain compat_50_routedomain;
56#endif
57
58 domain_attach(&linkdomain);
59 domain_attach(&routedomain);
60#ifdef COMPAT_50
61 domain_attach(&compat_50_routedomain);
62#endif
63}
64
65RUMP_COMPONENT(RUMP_COMPONENT_NET_IF)
66{
67
68 ifinit_post();
69 loopinit();
70}
