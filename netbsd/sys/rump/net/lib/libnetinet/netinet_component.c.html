66RUMP_COMPONENT(RUMP_COMPONENT_NET_IFCFG)
67{
68 struct ifaliasreq ia;
69 struct sockaddr_in *sin;
70 struct socket *so;
71 int error;
72
73 if (lo0ifp == NULL)
74 panic("lo0 config: rumpnet_net has not been initialized");
75
76 if ((error = socreate(AF_INET, &so, SOCK_DGRAM, 0, curlwp, NULL)) != 0)
77 panic("lo0 config: cannot create socket");
78
79 /* configure 127.0.0.1 for lo0 */
80 memset(&ia, 0, sizeof(ia));
81 strcpy(ia.ifra_name, "lo0");
82 sin = (struct sockaddr_in *)&ia.ifra_addr;
83 sin->sin_family = AF_INET;
84 sin->sin_len = sizeof(struct sockaddr_in);
85 sin->sin_addr.s_addr = inet_addr("127.0.0.1");
86
87 sin = (struct sockaddr_in *)&ia.ifra_mask;
88 sin->sin_family = AF_INET;
89 sin->sin_len = sizeof(struct sockaddr_in);
90 sin->sin_addr.s_addr = inet_addr("255.0.0.0");
91
92 sin = (struct sockaddr_in *)&ia.ifra_broadaddr;
93 sin->sin_family = AF_INET;
94 sin->sin_len = sizeof(struct sockaddr_in);
95 sin->sin_addr.s_addr = inet_addr("127.255.255.255");
96
97 KERNEL_LOCK(1, NULL);
98 IFNET_LOCK(lo0ifp);
99 in_control(so, SIOCAIFADDR, &ia, lo0ifp);
100 IFNET_UNLOCK(lo0ifp);
101 KERNEL_UNLOCK_ONE(NULL);
102 if_up(lo0ifp);
103 soclose(so);
104}
