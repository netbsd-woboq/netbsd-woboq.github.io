28#include <sys/cdefs.h>
29__KERNEL_RCSID(0, "$NetBSD: shmif_component.c,v 1.2 2016/01/26 23:12:19 pooka Exp $");
30
31#include <sys/param.h>
32#include <sys/vmem.h>
33
34#include <net/if.h>
35
36#include <rump-sys/kern.h>
37
38RUMP_COMPONENT(RUMP_COMPONENT_NET_IF)
39{
40 extern struct if_clone shmif_cloner; /* XXX */
41 extern vmem_t *shmif_units; /* XXX */
42
43 shmif_units = vmem_create("shmif", 1, 1<<15, 1, NULL, NULL, NULL, 1,
44 VM_SLEEP, IPL_NONE);
45 if (shmif_units == NULL)
46 panic("shmif vmem_create failed");
47
48 if_clone_attach(&shmif_cloner);
49}
