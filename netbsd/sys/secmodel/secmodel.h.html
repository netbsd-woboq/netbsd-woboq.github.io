29#ifndef _SECMODEL_SECMODEL_H_
30#define _SECMODEL_SECMODEL_H_
31
32#include <prop/proplib.h>
33
34void secmodel_init(void);
35
36/*
37 * Functions used for inter-secmodel communication, allowing evaluation
38 * or setting information.
39 */
40typedef int (*secmodel_eval_t)(const char *, void *, void *);
41typedef int (*secmodel_setinfo_t)(void *); /* XXX TODO */
42
43/*
44 * Secmodel entry.
45 */
46struct secmodel_descr {
47 LIST_ENTRY(secmodel_descr) sm_list;
48 const char *sm_id;
49 const char *sm_name;
50 prop_dictionary_t sm_behavior;
51 secmodel_eval_t sm_eval;
52 secmodel_setinfo_t sm_setinfo;
53};
54typedef struct secmodel_descr *secmodel_t;
55
56int secmodel_register(secmodel_t *, const char *, const char *,
57 prop_dictionary_t, secmodel_eval_t, secmodel_setinfo_t);
58int secmodel_deregister(secmodel_t);
59int secmodel_nsecmodels(void);
60
61int secmodel_eval(const char *, const char *, void *, void *);
62int secmodel_setinfo(const char *, void *, int *); /* XXX TODO */
63#endif /* !_SECMODEL_SECMODEL_H_ */
