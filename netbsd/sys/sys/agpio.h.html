36/*
37 * The AGP gatt uses 4k pages irrespective of the host page size.
38 */
39#define AGP_PAGE_SIZE 4096
40#define AGP_PAGE_SHIFT 12

42/*
43 * Macros to manipulate AGP mode words.
44 */
45#define AGP_MODE_GET_RQ(x) __SHIFTOUT((x), AGP_MODE_RQ)
46#define AGP_MODE_GET_ARQSZ(x) __SHIFTOUT((x), AGP_MODE_ARQSZ)
47#define AGP_MODE_GET_CAL(x) __SHIFTOUT((x), AGP_MODE_CAL)
48#define AGP_MODE_GET_SBA(x) __SHIFTOUT((x), AGP_MODE_SBA)
49#define AGP_MODE_GET_AGP(x) __SHIFTOUT((x), AGP_MODE_AGP)
50#define AGP_MODE_GET_4G(x) __SHIFTOUT((x), AGP_MODE_4G)
51#define AGP_MODE_GET_FW(x) __SHIFTOUT((x), AGP_MODE_FW)
52#define AGP_MODE_GET_MODE_3(x) __SHIFTOUT((x), AGP_MODE_MODE_3)
53#define AGP_MODE_GET_RATE(x) __SHIFTOUT((x), AGP_MODE_RATE)
54#define AGP_MODE_SET_RQ(x, v) (((x) & ~AGP_MODE_RQ) \
55 | __SHIFTIN((v), AGP_MODE_RQ))
56#define AGP_MODE_SET_ARQSZ(x, v) (((x) & ~AGP_MODE_ARQSZ) \
57 | __SHIFTIN((v), AGP_MODE_ARQSZ))
58#define AGP_MODE_SET_CAL(x, v) (((x) & ~AGP_MODE_CAL) \
59 | __SHIFTIN((v), ~AGP_MODE_CAL))
60#define AGP_MODE_SET_SBA(x, v) (((x) & ~AGP_MODE_SBA) \
61 | __SHIFTIN((v), AGP_MODE_SBA))
62#define AGP_MODE_SET_AGP(x, v) (((x) & ~AGP_MODE_AGP) \
63 | __SHIFTIN((v), AGP_MODE_AGP))
64#define AGP_MODE_SET_4G(x, v) (((x) & ~AGP_MODE_4G) \
65 | __SHIFTIN((v), AGP_MODE_4G))
66#define AGP_MODE_SET_FW(x, v) (((x) & ~AGP_MODE_FW) \
67 | __SHIFTIN((v), AGP_MODE_FW))
68#define AGP_MODE_SET_MODE_3(x, v) (((x) & ~AGP_MODE_MODE_3) \
69 | __SHIFTIN((v), AGP_MODE_MODE_3))
70#define AGP_MODE_SET_RATE(x, v) (((x) & ~AGP_MODE_RATE) \
71 | __SHIFTIN((v), AGP_MODE_RATE))
72

92typedef struct _agp_version {
93 uint16_t major;
94 uint16_t minor;
95} agp_version;
96
97typedef struct _agp_info {
98 agp_version version; /* version of the driver */
99 uint32_t bridge_id; /* bridge vendor/device */
100 uint32_t agp_mode; /* mode info of bridge */
101 off_t aper_base; /* base of aperture */
102 size_t aper_size; /* size of aperture */
103 size_t pg_total; /* max pages (swap + system) */
104 size_t pg_system; /* max pages (system) */
105 size_t pg_used; /* current pages used */
106} agp_info;
107
108typedef struct _agp_setup {
109 uint32_t agp_mode; /* mode info of bridge */
110} agp_setup;
111
112#if 0
113/*

115 */
116typedef struct _agp_segment {
117 off_t pg_start; /* starting page to populate */
118 size_t pg_count; /* number of pages */
119 int prot; /* prot flags for mmap */
120} agp_segment;
121
122typedef struct _agp_region {
123 pid_t pid; /* pid of process */
124 size_t seg_count; /* number of segments */
125 struct _agp_segment *seg_list;
126} agp_region;
127#endif
128
129typedef struct _agp_allocate {
130 int key; /* tag of allocation */
131 size_t pg_count; /* number of pages */
132 uint32_t type; /* 0 == normal, other devspec */
133 uint32_t physical; /* device specific (some devices
134 * need a phys address of the
135 * actual page behind the gatt
136 * table) */
137} agp_allocate;
138
139typedef struct _agp_bind {
140 int key; /* tag of allocation */
141 off_t pg_start; /* starting page to populate */
142} agp_bind;
143
144typedef struct _agp_unbind {
145 int key; /* tag of allocation */
146 uint32_t priority; /* priority for paging out */
147} agp_unbind;
