32#ifndef _SYS_ANSI_H_
33#define _SYS_ANSI_H_
34
35#include <machine/ansi.h>
36
37typedef char * __caddr_t; /* core address */
38typedef __uint32_t __gid_t; /* group id */
39typedef __uint32_t __in_addr_t; /* IP(v4) address */
40typedef __uint16_t __in_port_t; /* "Internet" port number */
41typedef __uint32_t __mode_t; /* file permissions */
42typedef __int64_t __off_t; /* file offset */
43typedef __int32_t __pid_t; /* process id */
44typedef __uint8_t __sa_family_t; /* socket address family */
45typedef unsigned int __socklen_t; /* socket-related datum length */
46typedef __uint32_t __uid_t; /* user id */
47typedef __uint64_t __fsblkcnt_t; /* fs block count (statvfs) */
48typedef __uint64_t __fsfilcnt_t; /* fs file count */
49
50struct __tag_wctrans_t;
51typedef struct __tag_wctrans_t *__wctrans_t;
52
53struct __tag_wctype_t;
54typedef struct __tag_wctype_t *__wctype_t;
55
56/*
57 * mbstate_t is an opaque object to keep conversion state, during multibyte
58 * stream conversions. The content must not be referenced by user programs.
59 */
60typedef union {
61 __int64_t __mbstateL; /* for alignment */
62 char __mbstate8[128];
63} __mbstate_t;
64
65#define _BSD_WCTRANS_T_ __wctrans_t /* wctrans_t */
66#define _BSD_WCTYPE_T_ __wctype_t /* wctype_t */
67#define _BSD_MBSTATE_T_ __mbstate_t /* mbstate_t */
68
69#ifdef __lint__
70typedef char *__va_list;
71#else
72typedef __builtin_va_list __va_list;
73#endif
74
75#endif /* !_SYS_ANSI_H_ */
