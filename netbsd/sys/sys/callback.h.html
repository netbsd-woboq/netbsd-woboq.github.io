29#ifndef _SYS_CALLBACK_H_
30#define _SYS_CALLBACK_H_
31
32#include <sys/queue.h>
33#include <sys/mutex.h>
34#include <sys/condvar.h>
35
36struct callback_entry {
37 TAILQ_ENTRY(callback_entry) ce_q;
38 int (*ce_func)(struct callback_entry *, void *, void *);
39 void *ce_obj;
40};
41
42struct callback_head {
43 kmutex_t ch_lock;
44 kcondvar_t ch_cv;
45 TAILQ_HEAD(, callback_entry) ch_q;
46 struct callback_entry *ch_next;
47 int ch_nentries;
48 int ch_running;
49 int ch_flags;
50};
51
52/* return values of ce_func */
53#define CALLBACK_CHAIN_CONTINUE 0
54#define CALLBACK_CHAIN_ABORT 1
55
56int callback_run_roundrobin(struct callback_head *, void *);
57void callback_register(struct callback_head *, struct callback_entry *,
58 void *, int (*)(struct callback_entry *, void *, void *));
59void callback_unregister(struct callback_head *, struct callback_entry *);
60void callback_head_init(struct callback_head *, int);
61void callback_head_destroy(struct callback_head *);
62
63#endif /* !_SYS_CALLBACK_H_ */
