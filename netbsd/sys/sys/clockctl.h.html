35#ifndef _SYS_CLOCKCTL_H_
36#define _SYS_CLOCKCTL_H_
37
38#include <sys/ioccom.h>
39#include <sys/time.h>
40#include <sys/timex.h>
41
42struct clockctl_settimeofday {
43 const struct timeval *tv;
44 const void *tzp;
45};
46
47#define CLOCKCTL_SETTIMEOFDAY _IOW('C', 0x5, struct clockctl_settimeofday)
48
49struct clockctl_adjtime {
50 const struct timeval *delta;
51 struct timeval *olddelta;
52};
53
54#define CLOCKCTL_ADJTIME _IOWR('C', 0x6, struct clockctl_adjtime)
55
56struct clockctl_clock_settime {
57 clockid_t clock_id;
58 const struct timespec *tp;
59};
60
61#define CLOCKCTL_CLOCK_SETTIME _IOW('C', 0x7, struct clockctl_clock_settime)
62
63struct clockctl_ntp_adjtime {
64 struct timex *tp;
65 __register_t retval;
66};
67
68#define CLOCKCTL_NTP_ADJTIME _IOWR('C', 0x8, struct clockctl_ntp_adjtime)
69
70#ifdef _KERNEL
71void clockctlattach(int);
72int clockctlopen(dev_t, int, int, struct lwp *);
73int clockctlclose(dev_t, int, int, struct lwp *);
74int clockctlioctl(dev_t, u_long, void *, int, struct lwp *);
75int clockctl_init(void);
76#endif
77
78#endif /* _SYS_CLOCKCTL_H_ */
