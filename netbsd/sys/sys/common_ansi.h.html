32#ifndef _SYS_COMMON_ANSI_H_
33#define _SYS_COMMON_ANSI_H_
34
35#include <sys/cdefs.h>
36
37#include <machine/int_types.h>
38
39#if !defined(__PTRDIFF_TYPE__)
40#error __PTRDIFF_TYPE__ not present
41#endif
42
43#if !defined(__SIZE_TYPE__)
44#error __SIZE_TYPE__ not present
45#endif
46
47#if !defined(__WCHAR_TYPE__)
48#error __WCHAR_TYPE__ not present
49#endif
50
51#if !defined(__WINT_TYPE__)
52#error __WINT_TYPE__ not present
53#endif
54
55/*
56 * Types which are fundamental to the implementation and may appear in
57 * more than one standard header are defined here. Standard headers
58 * then use:
59 * #ifdef _BSD_SIZE_T_
60 * typedef _BSD_SIZE_T_ size_t;
61 * #undef _BSD_SIZE_T_
62 * #endif
63 */
64#define _BSD_CLOCK_T_ unsigned int /* clock() */
65#define _BSD_PTRDIFF_T_ __PTRDIFF_TYPE__ /* ptr1 - ptr2 */
66#define _BSD_SSIZE_T_ __PTRDIFF_TYPE__ /* byte count or error */
67#define _BSD_SIZE_T_ __SIZE_TYPE__ /* sizeof() */
68#define _BSD_TIME_T_ __int64_t /* time() */
69#define _BSD_CLOCKID_T_ int /* clockid_t */
70#define _BSD_TIMER_T_ int /* timer_t */
71#define _BSD_SUSECONDS_T_ int /* suseconds_t */
72#define _BSD_USECONDS_T_ unsigned int /* useconds_t */
73#define _BSD_WCHAR_T_ __WCHAR_TYPE__ /* wchar_t */
74#define _BSD_WINT_T_ __WINT_TYPE__ /* wint_t */
75
76#endif /* _SYS_COMMON_ANSI_H_ */
