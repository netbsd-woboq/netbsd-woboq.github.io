32#ifndef _SYS_COMMON_INT_LIMITS_H_
33#define _SYS_COMMON_INT_LIMITS_H_
34
35#ifndef __SIG_ATOMIC_MAX__
36#error Your compiler does not provide limit macros.
37#endif
38
39/*
40 * 7.18.2 Limits of specified-width integer types
41 */
42
43/* 7.18.2.1 Limits of exact-width integer types */
44
45/* minimum values of exact-width signed integer types */
46#define INT8_MIN (-__INT8_MAX__-1)
47#define INT16_MIN (-__INT16_MAX__-1)
48#define INT32_MIN (-__INT32_MAX__-1)
49#define INT64_MIN (-__INT64_MAX__-1)
50
51/* maximum values of exact-width signed integer types */
52#define INT8_MAX __INT8_MAX__
53#define INT16_MAX __INT16_MAX__
54#define INT32_MAX __INT32_MAX__
55#define INT64_MAX __INT64_MAX__
56
57/* maximum values of exact-width unsigned integer types */
58#define UINT8_MAX __UINT8_MAX__
59#define UINT16_MAX __UINT16_MAX__
60#define UINT32_MAX __UINT32_MAX__
61#define UINT64_MAX __UINT64_MAX__
62
63/* 7.18.2.2 Limits of minimum-width integer types */
64
65/* minimum values of minimum-width signed integer types */
66#define INT_LEAST8_MIN (-__INT_LEAST8_MAX__-1)
67#define INT_LEAST16_MIN (-__INT_LEAST16_MAX__-1)
68#define INT_LEAST32_MIN (-__INT_LEAST32_MAX__-1)
69#define INT_LEAST64_MIN (-__INT_LEAST64_MAX__-1)
70
71/* maximum values of minimum-width signed integer types */
72#define INT_LEAST8_MAX __INT_LEAST8_MAX__
73#define INT_LEAST16_MAX __INT_LEAST16_MAX__
74#define INT_LEAST32_MAX __INT_LEAST32_MAX__
75#define INT_LEAST64_MAX __INT_LEAST64_MAX__
76
77/* maximum values of minimum-width unsigned integer types */
78#define UINT_LEAST8_MAX __UINT_LEAST8_MAX__
79#define UINT_LEAST16_MAX __UINT_LEAST16_MAX__
80#define UINT_LEAST32_MAX __UINT_LEAST32_MAX__
81#define UINT_LEAST64_MAX __UINT_LEAST64_MAX__
82
83/* 7.18.2.3 Limits of fastest minimum-width integer types */
84
85/* minimum values of fastest minimum-width signed integer types */
86#define INT_FAST8_MIN (-__INT_FAST8_MAX__-1)
87#define INT_FAST16_MIN (-__INT_FAST16_MAX__-1)
88#define INT_FAST32_MIN (-__INT_FAST32_MAX__-1)
89#define INT_FAST64_MIN (-__INT_FAST64_MAX__-1)
90
91/* maximum values of fastest minimum-width signed integer types */
92#define INT_FAST8_MAX __INT_FAST8_MAX__
93#define INT_FAST16_MAX __INT_FAST16_MAX__
94#define INT_FAST32_MAX __INT_FAST32_MAX__
95#define INT_FAST64_MAX __INT_FAST64_MAX__
96
97/* maximum values of fastest minimum-width unsigned integer types */
98#define UINT_FAST8_MAX __UINT_FAST8_MAX__
99#define UINT_FAST16_MAX __UINT_FAST16_MAX__
100#define UINT_FAST32_MAX __UINT_FAST32_MAX__
101#define UINT_FAST64_MAX __UINT_FAST64_MAX__
102
103/* 7.18.2.4 Limits of integer types capable of holding object pointers */
104#define INTPTR_MIN (-__INTPTR_MAX__-1)
105#define INTPTR_MAX __INTPTR_MAX__
106#define UINTPTR_MAX __UINTPTR_MAX__
107
108/* 7.18.2.5 Limits of greatest-width integer types */
109
110#define INTMAX_MIN (-__INTMAX_MAX__-1)
111#define INTMAX_MAX __INTMAX_MAX__
112#define UINTMAX_MAX __UINTMAX_MAX__
113
114
115/*
116 * 7.18.3 Limits of other integer types
117 */
118
119/* limits of ptrdiff_t */
120#define PTRDIFF_MIN (-__PTRDIFF_MAX__-1)
121#define PTRDIFF_MAX __PTRDIFF_MAX__
122
123/* limits of sig_atomic_t */
124#define SIG_ATOMIC_MIN (-__SIG_ATOMIC_MAX__-1)
125#define SIG_ATOMIC_MAX __SIG_ATOMIC_MAX__
126
127/* limit of size_t */
128#define SIZE_MAX __SIZE_MAX__
129
130#endif /* _SYS_COMMON_INT_LIMITS_H_ */
