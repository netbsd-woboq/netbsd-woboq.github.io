32#ifndef _SYS_COMMON_INT_MWGWTYPES_H_
33#define _SYS_COMMON_INT_MWGWTYPES_H_
34
35#ifndef __UINT_FAST64_TYPE__
36#error Your compiler does not provide min/fast width type macros.
37#endif
38
39/*
40 * 7.18.1 Integer types
41 */
42
43/* 7.18.1.2 Minimum-width integer types */
44
45typedef __INT_LEAST8_TYPE__ int_least8_t;
46typedef __UINT_LEAST8_TYPE__ uint_least8_t;
47typedef __INT_LEAST16_TYPE__ int_least16_t;
48typedef __UINT_LEAST16_TYPE__ uint_least16_t;
49typedef __INT_LEAST32_TYPE__ int_least32_t;
50typedef __UINT_LEAST32_TYPE__ uint_least32_t;
51typedef __INT_LEAST64_TYPE__ int_least64_t;
52typedef __UINT_LEAST64_TYPE__ uint_least64_t;
53
54/* 7.18.1.3 Fastest minimum-width integer types */
55typedef __INT_FAST8_TYPE__ int_fast8_t;
56typedef __UINT_FAST8_TYPE__ uint_fast8_t;
57typedef __INT_FAST16_TYPE__ int_fast16_t;
58typedef __UINT_FAST16_TYPE__ uint_fast16_t;
59typedef __INT_FAST32_TYPE__ int_fast32_t;
60typedef __UINT_FAST32_TYPE__ uint_fast32_t;
61typedef __INT_FAST64_TYPE__ int_fast64_t;
62typedef __UINT_FAST64_TYPE__ uint_fast64_t;
63
64/* 7.18.1.5 Greatest-width integer types */
65
66typedef __INTMAX_TYPE__ intmax_t;
67typedef __UINTMAX_TYPE__ uintmax_t;

69#endif /* _SYS_COMMON_INT_MWGWTYPES_H_ */
