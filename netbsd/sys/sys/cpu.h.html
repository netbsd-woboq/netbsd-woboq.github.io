54 * cpu_need_resched() must always be called with the target CPU
55 * locked (via spc_lock() or another route), unless called locally.
56 * If called locally, the caller need only be at IPL_SCHED.

66/*
67 * CPU_INFO_ITERATOR() may be supplied by machine dependent code as it
68 * controls how the cpu_info structures are allocated.
69 *
70 * This macro must always iterate just the boot-CPU when the system has
71 * not attached any cpus via mi_cpu_attach() yet, and the "ncpu" variable
72 * is zero.
73 */
74#ifndef CPU_INFO_ITERATOR
75#define CPU_INFO_ITERATOR int
76#define CPU_INFO_FOREACH(cii, ci) \
77 (void)cii, ci = curcpu(); ci != NULL; ci = NULL
78#endif
