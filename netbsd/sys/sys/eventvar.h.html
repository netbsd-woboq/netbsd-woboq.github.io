31/*
32 * This header is provided for the kqueue implementation and kmem
33 * grovellers, and is not expected to be used elsewhere.
34 */
35
36#ifndef _SYS_EVENTVAR_H_
37#define _SYS_EVENTVAR_H_
38
39#include <sys/mutex.h>
40#include <sys/selinfo.h>
41#include <sys/filedesc.h>
42
43#define KQ_NEVENTS 8 /* minimize copy{in,out} calls */
44#define KQ_EXTENT 256 /* linear growth by this amount */
45#define KFILTER_MAXNAME 256 /* maximum size of a filter name */
46#define KFILTER_EXTENT 8 /* grow user_kfilters by this amt */
47
48struct kqueue {
49 TAILQ_HEAD(kqlist, knote) kq_head; /* list of pending event */
50 kmutex_t kq_lock; /* mutex for queue access */
51 filedesc_t *kq_fdp;
52 struct selinfo kq_sel;
53 kcondvar_t kq_cv;
54 int kq_count; /* number of pending events */
55};
56
57#endif /* !_SYS_EVENTVAR_H_ */
