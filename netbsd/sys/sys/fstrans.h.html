32/*
33 * File system transaction operations.
34 */
35
36#ifndef _SYS_FSTRANS_H_
37#define _SYS_FSTRANS_H_
38
39#include <sys/mount.h>
40
41#define SUSPEND_SUSPEND 0x0001 /* VFS_SUSPENDCTL: suspend */
42#define SUSPEND_RESUME 0x0002 /* VFS_SUSPENDCTL: resume */
43
44enum fstrans_state {
45 FSTRANS_NORMAL,
46 FSTRANS_SUSPENDED,
47 FSTRANS_SUSPENDING
48};
49
50void fstrans_init(void);
51void fstrans_lwp_dtor(lwp_t *);
52void fstrans_start(struct mount *);
53int fstrans_start_nowait(struct mount *);
54void fstrans_start_lazy(struct mount *);
55void fstrans_done(struct mount *);
56int fstrans_is_owner(struct mount *);
57int fstrans_mount(struct mount *);
58void fstrans_unmount(struct mount *);
59
60int fstrans_setstate(struct mount *, enum fstrans_state);
61enum fstrans_state fstrans_getstate(struct mount *);
62
63int fscow_establish(struct mount *, int (*)(void *, struct buf *, bool),
64 void *);
65int fscow_disestablish(struct mount *, int (*)(void *, struct buf *, bool),
66 void *);
67int fscow_run(struct buf *, bool);
68
69int vfs_suspend(struct mount *, int);
70void vfs_resume(struct mount *);
71
72#endif /* _SYS_FSTRANS_H_ */
