29#ifndef _SYS_INTERRUPT_H_
30#define _SYS_INTERRUPT_H_
31
32#include <sys/types.h>
33#include <sys/intr.h>
34#include <sys/kcpuset.h>
35
36typedef char intrid_t[INTRIDBUF];
37struct intrids_handler {
38 int iih_nids;
39 intrid_t iih_intrids[1];
40 /*
41 * The number of the "iih_intrids" array will be overwritten by
42 * "iih_nids" after intr_construct_intrids().
43 */
44};
45
46uint64_t interrupt_get_count(const char *, u_int);
47void interrupt_get_assigned(const char *, kcpuset_t *);
48void interrupt_get_available(kcpuset_t *);
49void interrupt_get_devname(const char *, char *, size_t);
50struct intrids_handler *interrupt_construct_intrids(const kcpuset_t *);
51void interrupt_destruct_intrids(struct intrids_handler *);
52int interrupt_distribute(void *, const kcpuset_t *, kcpuset_t *);
53int interrupt_distribute_handler(const char *, const kcpuset_t *,
54 kcpuset_t *);
55
56#endif /* !_SYS_INTERRUPT_H_ */
