29#ifndef _SYS_INTRIO_H_
30#define _SYS_INTRIO_H_
31
32#include <sys/types.h>
33#include <sys/intr.h>
34#include <sys/sched.h>
35
36#define INTRIO_LIST_VERSION 1
37
38struct intrio_set {
39 char intrid[INTRIDBUF];
40 cpuset_t *cpuset;
41 size_t cpuset_size;
42};
43
44struct intrio_list_line_cpu {
45 bool illc_assigned;
46 uint64_t illc_count;
47};
48
49struct intrio_list_line {
50 char ill_intrid[INTRIDBUF]; /* NUL terminated. */
51 char ill_xname[INTRDEVNAMEBUF]; /* NUL terminated. */
52 struct intrio_list_line_cpu ill_cpu[1]; /*
53 * Array size is overwritten
54 * to ncpu.
55 */
56};
57
58struct intrio_list {
59 int il_version; /* Version number of this struct. */
60 int il_ncpus;
61 int il_nintrs;
62 size_t il_bufsize;
63
64 size_t il_linesize;
65 off_t il_lineoffset;
66/*
67 * struct intrio_list_line il_lines[interrupt_num] must be followed here.
68 */
69};
70
71#endif /* !_SYS_INTRIO_H_ */
