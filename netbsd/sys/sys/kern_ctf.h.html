29#ifndef _SYS_CTF_H_
30#define _SYS_CTF_H_
31
32/*
33 * Modules CTF section
34 */
35typedef struct mod_ctf {
36 uint8_t *ctftab; /* Decompressed CTF data. */
37 int ctfcnt; /* Number of CTF data bytes. */
38 const Elf_Sym *symtab; /* Ptr to the symbol table. */
39 int nsym; /* Number of symbols. */
40 uint32_t *nmap; /* symbol id map */
41 int nmapsize; /* Span of id map */
42 const char *strtab; /* Ptr to the string table. */
43 int strcnt; /* Number of string bytes. */
44 uint32_t *ctfoffp; /* Ptr to array of obj/fnc offsets. */
45 uint32_t *typoffp; /* Ptr to array of type offsets. */
46 long typlen; /* number of type data entries. */
47 int ctfalloc; /* ctftab is alloced */
48 bool fbt_provided;
49 int nenabled;
50} mod_ctf_t;
51
52int
53mod_ctf_get(struct module *, mod_ctf_t **);
54
55#endif
