29#ifndef _SYS_KMEM_H_
30#define _SYS_KMEM_H_
31
32#include <sys/types.h>
33
34typedef unsigned int km_flag_t;
35
36void kmem_init(void);
37size_t kmem_roundup_size(size_t);
38
39void * kmem_alloc(size_t, km_flag_t);
40void * kmem_zalloc(size_t, km_flag_t);
41void kmem_free(void *, size_t);
42
43void * kmem_intr_alloc(size_t, km_flag_t);
44void * kmem_intr_zalloc(size_t, km_flag_t);
45void kmem_intr_free(void *, size_t);
46
47char * kmem_asprintf(const char *, ...) __printflike(1, 2);
48
49char * kmem_strdupsize(const char *, size_t *, km_flag_t);
50#define kmem_strdup(s, f) kmem_strdupsize((s), NULL, (f))
51char * kmem_strndup(const char *, size_t, km_flag_t);
52void kmem_strfree(char *);
53
54/*
55 * km_flag_t values:
56 */
57#define KM_SLEEP 0x00000001 /* can sleep */
58#define KM_NOSLEEP 0x00000002 /* don't sleep */
59
60#endif /* !_SYS_KMEM_H_ */
