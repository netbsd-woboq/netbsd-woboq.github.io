29#ifndef _SYS_KOBJ_H_
30#define _SYS_KOBJ_H_
31
32#if !defined(ELFSIZE) && !defined(_RUMPKERNEL)
33#define ELFSIZE KERN_ELFSIZE
34#endif
35
36#include <sys/exec.h>
37#include <sys/exec_elf.h>
38
39typedef struct kobj *kobj_t;
40
41/* External interface. */
42int kobj_load_vfs(kobj_t *, const char *, const bool);
43int kobj_load_mem(kobj_t *, const char *, void *, ssize_t);
44int kobj_affix(kobj_t, const char *);
45void kobj_unload(kobj_t);
46int kobj_stat(kobj_t, vaddr_t *, size_t *);
47int kobj_find_section(kobj_t, const char *, void **, size_t *);
48
49/* MI-MD interface. */
50int kobj_sym_lookup(kobj_t, uintptr_t, Elf_Addr *);
51int kobj_reloc(kobj_t, uintptr_t, const void *, bool, bool);
52int kobj_machdep(kobj_t, void *, size_t, bool);
53
54/* implementation interface. */
55int kobj_renamespace(Elf_Sym *, size_t, char **, size_t *);
56
57#endif /* !_SYS_KOBJ_H_ */
