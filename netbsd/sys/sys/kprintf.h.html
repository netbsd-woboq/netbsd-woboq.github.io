37#ifndef _SYS_KPRINTF_H_
38#define _SYS_KPRINTF_H_
39
40#include <sys/stdarg.h>
41
42/*
43 * Implementation internals of the kernel printf. Exposing them here
44 * allows other subsystems to implement precisely the printf semantics
45 * they need.
46 */
47
48/* max size buffer kprintf needs to print quad_t [size in base 8 + \0] */
49#define KPRINTF_BUFSIZE (sizeof(quad_t) * NBBY / 3 + 2)
50
51/* flags for kprintf */
52#define TOCONS 0x0001 /* to the console */
53#define TOTTY 0x0002 /* to the process' tty */
54#define TOLOG 0x0004 /* to the kernel message buffer */
55#define TOBUFONLY 0x0008 /* to the buffer (only) [for snprintf] */
56#define TODDB 0x0010 /* to ddb console */
57#define NOLOCK 0x1000 /* don't acquire a tty lock */
58#define NOTSTAMP 0x2000 /* no time stamp on console */
59
60void kprintf_init(void);
61void kprintf_init_callout(void);
62void kprintf_lock(void);
63void kprintf_unlock(void);
64/*
65 * NOTE: the kprintf mutex must be held when these functions are called!
66 */
67int kprintf(const char *, int, void *, char *, va_list);
68void klogpri(int);
69
70#endif /* _SYS_KPRINTF_H_ */
