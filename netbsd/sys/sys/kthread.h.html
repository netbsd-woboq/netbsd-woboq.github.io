33#ifndef _SYS_KTHREAD_H_
34#define _SYS_KTHREAD_H_
35
36#if !defined(_KERNEL)
37#error "not supposed to be exposed to userland"
38#endif
39
40/*
41 * Kernel thread handling.
42 */
43
44#include <sys/proc.h>
45
46#define KTHREAD_IDLE 0x01 /* Do not run on creation */
47#define KTHREAD_MPSAFE 0x02 /* Do not acquire kernel_lock */
48#define KTHREAD_INTR 0x04 /* Software interrupt handler */
49#define KTHREAD_TS 0x08 /* Time-sharing priority range */
50#define KTHREAD_MUSTJOIN 0x10 /* Must join on exit */
51
52void kthread_sysinit(void);
53
54int kthread_create(pri_t, int, struct cpu_info *,
55 void (*)(void *), void *, lwp_t **, const char *, ...) __printflike(7, 8);
56void kthread_exit(int) __dead;
57int kthread_join(lwp_t *);
58
59#endif /* _SYS_KTHREAD_H_ */
