32#ifndef _SYS_LOCALCOUNT_H
33#define _SYS_LOCALCOUNT_H
34
35#ifndef _KERNEL
36#error <sys/localcount.h> is for kernel consumers only.
37#endif
38
39#include <sys/types.h>
40
41struct kcondvar;
42struct kmutex;
43struct percpu;
44
45struct localcount {
46 int64_t *lc_totalp;
47 struct percpu *lc_percpu; /* int64_t */
48 volatile uint32_t lc_refcnt; /* only for debugging */
49};
50
51void localcount_init(struct localcount *);
52void localcount_drain(struct localcount *, struct kcondvar *,
53 struct kmutex *);
54void localcount_fini(struct localcount *);
55void localcount_acquire(struct localcount *);
56void localcount_release(struct localcount *, struct kcondvar *,
57 struct kmutex *);
58
59uint32_t
60 localcount_debug_refcnt(const struct localcount *);
61
62#endif /* _SYS_LOCALCOUNT_H */
