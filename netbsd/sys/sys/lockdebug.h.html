32#ifndef __SYS_LOCKDEBUG_H__
33#define __SYS_LOCKDEBUG_H__
34
35#ifdef _KERNEL_OPT
36#include "opt_lockdebug.h"
37#endif
38
39#if !defined(_KERNEL) && !defined(_KMEMUSER)
40#error "Sorry, nothing of interest to user level programs here."
41#endif
42
43#define LOCKOPS_SLEEP 0
44#define LOCKOPS_SPIN 1
45#define LOCKOPS_CV 2
46
47typedef void (*lockop_printer_t)(const char *, ...) __printflike(1, 2);
48
49typedef struct lockops {
50 const char *lo_name;
51 int lo_type;
52 void (*lo_dump)(const volatile void *, lockop_printer_t);
53} lockops_t;
54
55#define LOCKDEBUG_ABORT(f, ln, l, o, m) \
56 lockdebug_abort(f, ln, l, o, m)
57
58void lockdebug_dismiss(void);
59void lockdebug_abort(const char *, size_t, const volatile void *,
60 lockops_t *, const char *);
61
62void lockdebug_lock_print(void *, void (*)(const char *, ...)
63 __printflike(1, 2));
64void lockdebug_show_all_locks(void (*)(const char *, ...) __printflike(1, 2),
65 const char *);
66void lockdebug_show_lockstats(void (*)(const char *, ...) __printflike(1, 2));
67
68#ifdef LOCKDEBUG
69
70bool lockdebug_alloc(const char *, size_t, volatile void *, lockops_t *,
71 uintptr_t);
72void lockdebug_free(const char *, size_t, volatile void *);
73void lockdebug_wantlock(const char *, size_t, const volatile void *,
74 uintptr_t, int);
75void lockdebug_locked(const char *, size_t, volatile void *, void *,
76 uintptr_t, int);
77void lockdebug_unlocked(const char *, size_t, volatile void *,
78 uintptr_t, int);
79void lockdebug_barrier(const char *, size_t, volatile void *, int);
80void lockdebug_mem_check(const char *, size_t, void *, size_t);
81void lockdebug_wakeup(const char *, size_t, volatile void *, uintptr_t);
82
83#define LOCKDEBUG_ALLOC(lock, ops, addr) \
84 lockdebug_alloc(__func__, __LINE__, lock, ops, addr)
85#define LOCKDEBUG_FREE(dodebug, lock) \
86 if (dodebug) lockdebug_free(__func__, __LINE__, lock)
87#define LOCKDEBUG_WANTLOCK(dodebug, lock, where, s) \
88 if (dodebug) lockdebug_wantlock(__func__, __LINE__, lock, where, s)
89#define LOCKDEBUG_LOCKED(dodebug, lock, al, where, s) \
90 if (dodebug) lockdebug_locked(__func__, __LINE__, lock, al, where, s)
91#define LOCKDEBUG_UNLOCKED(dodebug, lock, where, s) \
92 if (dodebug) lockdebug_unlocked(__func__, __LINE__, lock, where, s)
93#define LOCKDEBUG_BARRIER(lock, slp) \
94 lockdebug_barrier(__func__, __LINE__, lock, slp)
95#define LOCKDEBUG_MEM_CHECK(base, sz) \
96 lockdebug_mem_check(__func__, __LINE__, base, sz)
97#define LOCKDEBUG_WAKEUP(dodebug, lock, where) \
98 if (dodebug) lockdebug_wakeup(__func__, __LINE__, lock, where)
99
100#else /* LOCKDEBUG */
101
102#define LOCKDEBUG_ALLOC(lock, ops, addr) false
103#define LOCKDEBUG_FREE(dodebug, lock) /* nothing */
104#define LOCKDEBUG_WANTLOCK(dodebug, lock, where, s) /* nothing */
105#define LOCKDEBUG_LOCKED(dodebug, lock, al, where, s) /* nothing */
106#define LOCKDEBUG_UNLOCKED(dodebug, lock, where, s) /* nothing */
107#define LOCKDEBUG_BARRIER(lock, slp) /* nothing */
108#define LOCKDEBUG_MEM_CHECK(base, sz) /* nothing */
109#define LOCKDEBUG_WAKEUP(dodebug, lock, where) /* nothing */
110
111#endif /* LOCKDEBUG */
112
113#endif /* __SYS_LOCKDEBUG_H__ */
