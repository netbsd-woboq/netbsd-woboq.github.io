161/*
162 * record/packet header in first mbuf of chain; valid if M_PKTHDR set
163 *
164 * A note about csum_data:
165 *
166 * o For the out-bound direction, the low 16 bits indicates the offset after
167 * the L4 header where the final L4 checksum value is to be stored and the
168 * high 16 bits is the length of the L3 header (the start of the data to
169 * be checksummed).
170 *
171 * o For the in-bound direction, it is only valid if the M_CSUM_DATA flag is
172 * set. In this case, an L4 checksum has been calculated by hardware and
173 * is stored in csum_data, but it is up to software to perform final
174 * verification.
175 *
176 * Note for in-bound TCP/UDP checksums: we expect the csum_data to NOT
177 * be bit-wise inverted (the final step in the calculation of an IP
178 * checksum) -- this is so we can accumulate the checksum for fragmented
179 * packets during reassembly.
180 *
181 * Size ILP32: 40
182 * LP64: 56
183 */
184struct pkthdr {
185 union {
186 void *ctx; /* for M_GETCTX/M_SETCTX */
187 if_index_t index; /* rcv interface index */
188 } _rcvif;
189#define rcvif_index _rcvif.index
190 SLIST_HEAD(packet_tags, m_tag) tags; /* list of packet tags */
191 int len; /* total packet length */
192 int csum_flags; /* checksum flags */
193 uint32_t csum_data; /* checksum data */
194 u_int segsz; /* segment size */
195 uint16_t ether_vtag; /* ethernet 802.1p+q vlan tag */
196 uint16_t pkthdr_flags; /* flags for pkthdr, see blow */
197#define PKTHDR_FLAG_IPSEC_SKIP_PFIL 0x0001 /* skip pfil_run_hooks() after ipsec decrypt */
198
199 /*
200 * Following three fields are open-coded struct altq_pktattr
201 * to rearrange struct pkthdr fields flexibly.
202 */
203 int pattr_af; /* ALTQ: address family */
204 void *pattr_class; /* ALTQ: sched class set by classifier */
205 void *pattr_hdr; /* ALTQ: saved header position in mbuf */
206};

208/* Checksumming flags (csum_flags). */
209#define M_CSUM_TCPv4 0x00000001 /* TCP header/payload */
210#define M_CSUM_UDPv4 0x00000002 /* UDP header/payload */
211#define M_CSUM_TCP_UDP_BAD 0x00000004 /* TCP/UDP checksum bad */
212#define M_CSUM_DATA 0x00000008 /* consult csum_data */
213#define M_CSUM_TCPv6 0x00000010 /* IPv6 TCP header/payload */
214#define M_CSUM_UDPv6 0x00000020 /* IPv6 UDP header/payload */
215#define M_CSUM_IPv4 0x00000040 /* IPv4 header */
216#define M_CSUM_IPv4_BAD 0x00000080 /* IPv4 header checksum bad */
217#define M_CSUM_TSOv4 0x00000100 /* TCPv4 segmentation offload */
218#define M_CSUM_TSOv6 0x00000200 /* TCPv6 segmentation offload */
219
220/* Checksum-assist quirks: keep separate from jump-table bits. */
221#define M_CSUM_NO_PSEUDOHDR 0x80000000 /* Rx csum_data does not include
222 * the UDP/TCP pseudo-hdr, and
223 * is not yet 1s-complemented.
224 */
225
226#define M_CSUM_BITS \
227 "\20\1TCPv4\2UDPv4\3TCP_UDP_BAD\4DATA\5TCPv6\6UDPv6\7IPv4\10IPv4_BAD" \
228 "\11TSOv4\12TSOv6\40NO_PSEUDOHDR"

230/*
231 * Macros for manipulating csum_data on outgoing packets. These are
232 * used to pass information down from the L4/L3 to the L2.
233 *
234 * _IPHL: Length of the IPv{4/6} header, plus the options; in other
235 * words the offset of the UDP/TCP header in the packet.
236 * _OFFSET: Offset of the checksum field in the UDP/TCP header.
237 */
238#define M_CSUM_DATA_IPv4_IPHL(x) ((x) >> 16)
239#define M_CSUM_DATA_IPv4_OFFSET(x) ((x) & 0xffff)
240#define M_CSUM_DATA_IPv6_IPHL(x) ((x) >> 16)
241#define M_CSUM_DATA_IPv6_OFFSET(x) ((x) & 0xffff)
242#define M_CSUM_DATA_IPv6_SET(x, v) (x) = ((x) & 0xffff) | ((v) << 16)

288 struct m_hdr m_hdr; \

292 union { \
293 struct _m_ext MH_ext; \

296 } MH; \

298 } M_dat; \
299 }

462 * Macros for mbuf external storage.
463 *
464 * MCLGET allocates and adds an mbuf cluster to a normal mbuf;
465 * the flag M_EXT is set upon success.
466 *
467 * MEXTMALLOC allocates external storage and adds it to

476#define MEXTMALLOC(m, size, how) \
477do { \
478 (m)->m_ext_storage.ext_buf = malloc((size), 0, (how)); \
479 if ((m)->m_ext_storage.ext_buf != NULL) { \
480 MCLINITREFERENCE(m); \
481 (m)->m_data = (m)->m_ext.ext_buf; \
482 (m)->m_flags = ((m)->m_flags & ~M_EXTCOPYFLAGS) | \

484 (m)->m_ext.ext_size = (size); \
485 (m)->m_ext.ext_free = NULL; \
486 (m)->m_ext.ext_arg = NULL; \

515/*
516 * Determine if an mbuf's data area is read-only. This is true
517 * if external storage is read-only mapped, or not marked as R/W,
518 * or referenced by more than one mbuf.
519 */
520#define M_READONLY(m) \
521 (((m)->m_flags & M_EXT) != 0 && \
522 (((m)->m_flags & (M_EXT_ROMAP|M_EXT_RW)) != M_EXT_RW || \
523 (m)->m_ext.ext_refcnt > 1))
524
525#define M_UNWRITABLE(__m, __len) \
526 ((__m)->m_len < (__len) || M_READONLY((__m)))
527
528/*
529 * Determine if an mbuf's data area is read-only at the MMU.
530 */
531#define M_ROMAP(m) \
532 (((m)->m_flags & (M_EXT|M_EXT_ROMAP)) == (M_EXT|M_EXT_ROMAP))
533
534/*
535 * Compute the amount of space available
536 * before the current start of data in an mbuf.
537 */
538#define _M_LEADINGSPACE(m) \
539 ((m)->m_flags & M_EXT ? (m)->m_data - (m)->m_ext.ext_buf : \
540 (m)->m_flags & M_PKTHDR ? (m)->m_data - (m)->m_pktdat : \
541 (m)->m_data - (m)->m_dat)
542
543#define M_LEADINGSPACE(m) \
544 (M_READONLY((m)) ? 0 : _M_LEADINGSPACE((m)))
545
546/*
547 * Compute the amount of space available
548 * after the end of data in an mbuf.
549 */
550#define _M_TRAILINGSPACE(m) \
551 ((m)->m_flags & M_EXT ? (m)->m_ext.ext_buf + (m)->m_ext.ext_size - \
552 ((m)->m_data + (m)->m_len) : \
553 &(m)->m_dat[MLEN] - ((m)->m_data + (m)->m_len))
554
555#define M_TRAILINGSPACE(m) \
556 (M_READONLY((m)) ? 0 : _M_TRAILINGSPACE((m)))
557
558/*
559 * Compute the address of an mbuf's data area.
560 */
561#define M_BUFADDR(m) \
562 (((m)->m_flags & M_PKTHDR) ? (m)->m_pktdat : (m)->m_dat)
563
564/*
565 * Compute the offset of the beginning of the data buffer of a non-ext
566 * mbuf.
567 */
568#define M_BUFOFFSET(m) \
569 (((m)->m_flags & M_PKTHDR) ? \
570 offsetof(struct mbuf, m_pktdat) : offsetof(struct mbuf, m_dat))

572/*
573 * Arrange to prepend space of size plen to mbuf m.
574 * If a new mbuf must be allocated, how specifies whether to wait.
575 * If how is M_DONTWAIT and allocation fails, the original mbuf chain
576 * is freed and m is set to NULL.
577 */
578#define M_PREPEND(m, plen, how) \
579do { \
580 if (M_LEADINGSPACE(m) >= (plen)) { \
581 (m)->m_data -= (plen); \
582 (m)->m_len += (plen); \
583 } else \
584 (m) = m_prepend((m), (plen), (how)); \
585 if ((m) && (m)->m_flags & M_PKTHDR) \
586 (m)->m_pkthdr.len += (plen); \
587} while (/* CONSTCOND */ 0)

589/* change mbuf to new type */

614/*
615 * M_REGION_GET ensures that the "len"-sized region of type "typ" starting
616 * from "off" within "m" is located in a single mbuf, contiguously.
617 *
618 * The pointer to the region will be returned to pointer variable "val".

620#define M_REGION_GET(val, typ, m, off, len) \

622 struct mbuf *_t; \

631 (val) = (typ)(mtod(_t, char *) + _tmp); \

634 (m) = NULL; \

641/*
642 * Simple mbuf queueing system
643 *
644 * this is basically a SIMPLEQ adapted to mbuf use (ie using
645 * m_nextpkt instead of field.sqe_next).
646 *
647 * m_next is ignored, so queueing chains of mbufs is possible
648 */
649#define MBUFQ_HEAD(name) \
650struct name { \
651 struct mbuf *mq_first; \
652 struct mbuf **mq_last; \
653}
654
655#define MBUFQ_INIT(q) do { \
656 (q)->mq_first = NULL; \
657 (q)->mq_last = &(q)->mq_first; \
658} while (/*CONSTCOND*/0)
659
660#define MBUFQ_ENQUEUE(q, m) do { \
661 (m)->m_nextpkt = NULL; \
662 *(q)->mq_last = (m); \
663 (q)->mq_last = &(m)->m_nextpkt; \
664} while (/*CONSTCOND*/0)
665
666#define MBUFQ_PREPEND(q, m) do { \
667 if (((m)->m_nextpkt = (q)->mq_first) == NULL) \
668 (q)->mq_last = &(m)->m_nextpkt; \
669 (q)->mq_first = (m); \
670} while (/*CONSTCOND*/0)
671
672#define MBUFQ_DEQUEUE(q, m) do { \
673 if (((m) = (q)->mq_first) != NULL) { \
674 if (((q)->mq_first = (m)->m_nextpkt) == NULL) \
675 (q)->mq_last = &(q)->mq_first; \
676 else \
677 (m)->m_nextpkt = NULL; \
678 } \
679} while (/*CONSTCOND*/0)
680
681#define MBUFQ_DRAIN(q) do { \
682 struct mbuf *__m0; \
683 while ((__m0 = (q)->mq_first) != NULL) { \
684 (q)->mq_first = __m0->m_nextpkt; \
685 m_freem(__m0); \
686 } \
687 (q)->mq_last = &(q)->mq_first; \
688} while (/*CONSTCOND*/0)
689
690#define MBUFQ_FIRST(q) ((q)->mq_first)
691#define MBUFQ_NEXT(m) ((m)->m_nextpkt)
692#define MBUFQ_LAST(q) (*(q)->mq_last)

814/*
815 * Return the number of bytes in the mbuf chain, m.
816 */
817static __inline u_int
818m_length(const struct mbuf *m)
819{
820 const struct mbuf *m0;
821 u_int pktlen;
822
823 if ((m->m_flags & M_PKTHDR) != 0)
824 return m->m_pkthdr.len;
825
826 pktlen = 0;
827 for (m0 = m; m0 != NULL; m0 = m0->m_next)
828 pktlen += m0->m_len;
829 return pktlen;
830}

887/*
888 * Get rcvif of a mbuf.
889 *
890 * The caller must call m_put_rcvif after using rcvif if the returned rcvif
891 * isn't NULL. If the returned rcvif is NULL, the caller doesn't need to call
892 * m_put_rcvif (although calling it is safe).
893 *
894 * The caller must not block or sleep while using rcvif. The API ensures a
895 * returned rcvif isn't freed until m_put_rcvif is called.
896 */
897static __inline struct ifnet *
898m_get_rcvif(const struct mbuf *m, int *s)
899{
900 struct ifnet *ifp;
901
902 KASSERT(m->m_flags & M_PKTHDR);
903 *s = pserialize_read_enter();
904 ifp = if_byindex(m->m_pkthdr.rcvif_index);
905 if (__predict_false(ifp == NULL))
906 pserialize_read_exit(*s);
907
908 return ifp;
909}

911static __inline void
912m_put_rcvif(struct ifnet *ifp, int *s)
913{
914
915 if (ifp == NULL)
916 return;
917 pserialize_read_exit(*s);
918}
