34#ifndef _SYS_MSGBUF_H_
35#define _SYS_MSGBUF_H_
36
37struct kern_msgbuf {
38#define MSG_MAGIC 0x063061
39 long msg_magic;
40 long msg_bufx; /* write pointer */
41 long msg_bufr; /* read pointer */
42 long msg_bufs; /* real msg_bufc size (bytes) */
43 char msg_bufc[1]; /* buffer */
44};
45
46#ifdef _KERNEL
47extern int msgbufmapped; /* is the message buffer mapped */
48extern int msgbufenabled; /* is logging to the buffer enabled */
49extern struct kern_msgbuf *msgbufp; /* the mapped buffer, itself. */
50
51void initmsgbuf(void *, size_t);
52void loginit(void);
53void logputchar(int);
54
55static __inline int
56logenabled(const struct kern_msgbuf *mbp)
57{
58 return msgbufenabled && mbp->msg_magic == MSG_MAGIC;
59}
60#endif
61
62#endif /* !_SYS_MSGBUF_H_ */
