56#if defined(PAX_MPROTECT) || defined(PAX_SEGVGUARD) || defined(PAX_ASLR)

66vm_prot_t pax_mprotect_maxprotect(
67#ifdef PAX_MPROTECT_DEBUG
68 const char *, size_t,
69#endif
70 struct lwp *, vm_prot_t, vm_prot_t, vm_prot_t);
71int pax_mprotect_validate(
72#ifdef PAX_MPROTECT_DEBUG
73 const char *, size_t,
74#endif
75 struct lwp *, vm_prot_t);
76
77#ifndef PAX_MPROTECT
78# define PAX_MPROTECT_MAXPROTECT(l, active, extra, max) (max)
79# define PAX_MPROTECT_VALIDATE(l, prot) (0)
80# define pax_mprotect_prot(l) 0
81#else
82# ifdef PAX_MPROTECT_DEBUG
83# define PAX_MPROTECT_MAXPROTECT(l, active, extra, max) \
84 pax_mprotect_maxprotect(__FILE__, __LINE__, (l), (active), (extra), (max))
85# define PAX_MPROTECT_VALIDATE(l, prot) \
86 pax_mprotect_validate(__FILE__, __LINE__, (l), (prot))
87# else
88# define PAX_MPROTECT_MAXPROTECT(l, active, extra, max) \
89 pax_mprotect_maxprotect((l), (active), (extra), (max))
90# define PAX_MPROTECT_VALIDATE(l, prot) \
91 pax_mprotect_validate((l), (prot))
92# endif
93int pax_mprotect_prot(struct lwp *);
94#endif
95int pax_segvguard(struct lwp *, struct vnode *, const char *, bool);

97#define PAX_ASLR_DELTA(delta, lsb, len) \
98 (((delta) & ((1UL << (len)) - 1)) << (lsb))
99
100#ifdef PAX_ASLR
101void pax_aslr_init_vm(struct lwp *, struct vmspace *, struct exec_package *);
102void pax_aslr_stack(struct exec_package *, vsize_t *);
103uint32_t pax_aslr_stack_gap(struct exec_package *);
104vaddr_t pax_aslr_exec_offset(struct exec_package *, vaddr_t);
105voff_t pax_aslr_rtld_offset(struct exec_package *, vaddr_t, int);
106void pax_aslr_mmap(struct lwp *, vaddr_t *, vaddr_t, int);
107#else
108# define pax_aslr_init_vm(l, v, e)
109# define pax_aslr_stack(e, o)
110# define pax_aslr_stack_gap(e) 0
111# define pax_aslr_exec_offset(e, a) MAX(a, PAGE_SIZE)
112# define pax_aslr_rtld_offset(e, a, u) 0
113# define pax_aslr_mmap(l, a, b, c)
114#endif
