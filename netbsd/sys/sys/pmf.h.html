123static __inline const device_suspensor_t *
124pmf_qual_suspension(const pmf_qual_t *pq)
125{
126 return pq->pq_suspensor;
127}
128
129static __inline devact_level_t
130pmf_qual_depth(const pmf_qual_t *pq)
131{
132 return pq->pq_actlvl;
133}
134
135static __inline bool
136pmf_qual_descend_ok(const pmf_qual_t *pq)
137{
138 return pq->pq_actlvl == DEVACT_LEVEL_FULL;
