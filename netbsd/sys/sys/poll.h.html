32#ifndef _SYS_POLL_H_
33#define _SYS_POLL_H_
34
35#include <sys/featuretest.h>
36
37typedef unsigned int nfds_t;
38
39struct pollfd {
40 int fd; /* file descriptor */
41 short events; /* events to look for */
42 short revents; /* events returned */
43};
44
45/*
46 * Testable events (may be specified in events field).
47 */
48#define POLLIN 0x0001
49#define POLLPRI 0x0002
50#define POLLOUT 0x0004
51#define POLLRDNORM 0x0040
52#define POLLWRNORM POLLOUT
53#define POLLRDBAND 0x0080
54#define POLLWRBAND 0x0100
55
56/*
57 * Non-testable events (may not be specified in events field).
58 */
59#define POLLERR 0x0008
60#define POLLHUP 0x0010
61#define POLLNVAL 0x0020
62
63#if defined(_NETBSD_SOURCE)
64/*
65 * Infinite timeout value.
66 */
67#define INFTIM -1
68#endif
69

74struct timespec;

76int pollcommon(register_t *, struct pollfd *, u_int,
77 struct timespec *, sigset_t *);

79#else

99#endif /* _KERNEL */

101#endif /* !_SYS_POLL_H_ */
