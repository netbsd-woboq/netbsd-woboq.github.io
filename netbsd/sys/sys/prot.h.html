35/*
36 * flags that control when do_setres{u,g}id will do anything
37 *
38 * ID_XXX_EQ_YYY means
39 * "allow modifying XXX uid to the given value if the new value of
40 * XXX uid (or gid) equals the current value of YYY uid (or gid)."
41 */
42
43#define ID_E_EQ_E 0x001 /* effective equals effective */
44#define ID_E_EQ_R 0x002 /* effective equals real */
45#define ID_E_EQ_S 0x004 /* effective equals saved */
46#define ID_R_EQ_E 0x010 /* real equals effective */
47#define ID_R_EQ_R 0x020 /* real equals real */
48#define ID_R_EQ_S 0x040 /* real equals saved */
49#define ID_S_EQ_E 0x100 /* saved equals effective */
50#define ID_S_EQ_R 0x200 /* saved equals real */
51#define ID_S_EQ_S 0x400 /* saved equals saved */
52
53int do_setresuid(struct lwp *, uid_t, uid_t, uid_t, u_int);
54int do_setresgid(struct lwp *, gid_t, gid_t, gid_t, u_int);
55
56#endif /* !_SYS_PROT_H_ */
