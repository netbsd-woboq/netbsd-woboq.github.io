31#ifndef _SYS_RADIOIO_H_
32#define _SYS_RADIOIO_H_
33
34#include <sys/param.h>
35#include <sys/ioccom.h>
36
37#define MIN_FM_FREQ 87500
38#define MAX_FM_FREQ 108000
39
40#define IF_FREQ 10700
41
42struct radio_info {
43 int mute;
44 int volume;
45 int stereo;
46 int rfreq; /* reference frequency */
47 int lock; /* locking field strength during an automatic search */
48 uint32_t freq; /* in kHz */
49 uint32_t caps; /* card capabilities */
50#define RADIO_CAPS_DETECT_STEREO (1<<0)
51#define RADIO_CAPS_DETECT_SIGNAL (1<<1)
52#define RADIO_CAPS_SET_MONO (1<<2)
53#define RADIO_CAPS_HW_SEARCH (1<<3)
54#define RADIO_CAPS_HW_AFC (1<<4)
55#define RADIO_CAPS_REFERENCE_FREQ (1<<5)
56#define RADIO_CAPS_LOCK_SENSITIVITY (1<<6)
57#define RADIO_CAPS_RESERVED1 (1<<7)
58#define RADIO_CAPS_RESERVED2 (0xFF<<8)
59#define RADIO_CARD_TYPE (0xFF<<16)
60 uint32_t info;
61#define RADIO_INFO_STEREO (1<<0)
62#define RADIO_INFO_SIGNAL (1<<1)
63};
64
65/* Radio device operations */
66#define RIOCGINFO _IOR('R', 21, struct radio_info) /* get info */
67#define RIOCSINFO _IOWR('R', 22, struct radio_info) /* set info */
68#define RIOCSSRCH _IOW('R', 23, int) /* search up/down */
69
70#endif /* _SYS_RADIOIO_H_ */
