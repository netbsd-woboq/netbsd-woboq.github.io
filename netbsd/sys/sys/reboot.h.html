37/*
38 * Arguments to reboot system call. These are passed to the boot program,
39 * and then on to init.
40 */
41#define RB_AUTOBOOT 0 /* flags for system auto-booting itself */
42
43#define RB_ASKNAME 0x00000001 /* ask for file name to reboot from */
44#define RB_SINGLE 0x00000002 /* reboot to single user only */
45#define RB_NOSYNC 0x00000004 /* dont sync before reboot */
46#define RB_HALT 0x00000008 /* don't reboot, just halt */
47#define RB_INITNAME 0x00000010 /* name given for /etc/init (unused) */
48#define __RB_UNUSED1 0x00000020 /* was RB_DFLTROOT, obsolete */
49#define RB_KDB 0x00000040 /* give control to kernel debugger */
50#define RB_RDONLY 0x00000080 /* mount root fs read-only */
51#define RB_DUMP 0x00000100 /* dump kernel memory before reboot */
52#define RB_MINIROOT 0x00000200 /* mini-root present in memory */
53#define RB_STRING 0x00000400 /* use provided bootstr */
54#define RB_POWERDOWN (RB_HALT|0x800) /* turn power off (or at least halt) */
55#define RB_USERCONF 0x00001000 /* change configured devices */
56
57/*
58 * Extra autoboot flags (passed by boot prog to kernel). See also
59 * macros bootverbose, bootquiet in <sys/systm.h>.
60 */
61#define AB_NORMAL 0x00000000 /* boot normally (default) */
62#define AB_QUIET 0x00010000 /* boot quietly */
63#define AB_VERBOSE 0x00020000 /* boot verbosely */
64#define AB_SILENT 0x00040000 /* boot silently */
65#define AB_DEBUG 0x00080000 /* boot with debug messages */
66
67/*
68 * The top 4 bits are architecture specific and are used to
69 * pass information between the bootblocks and the machine
70 * initialization code.
71 */
72#define RB_MD1 0x10000000
73#define RB_MD2 0x20000000
74#define RB_MD3 0x40000000
75#define RB_MD4 0x80000000
76
77/*
78 * Constants for converting boot-style device number to type,
79 * adaptor (uba, mba, etc), unit number and partition number.
80 * Type (== major device number) is in the low byte
81 * for backward compatibility. Except for that of the "magic
82 * number", each mask applies to the shifted value.
83 * Format:
84 * (4) (4) (4) (4) (8) (8)
85 * --------------------------------
86 * |MA | AD| CT| UN| PART | TYPE |
87 * --------------------------------
88 */
89#define B_ADAPTORSHIFT 24
90#define B_ADAPTORMASK 0x0f
91#define B_ADAPTOR(val) (((val) >> B_ADAPTORSHIFT) & B_ADAPTORMASK)
92#define B_CONTROLLERSHIFT 20
93#define B_CONTROLLERMASK 0xf
94#define B_CONTROLLER(val) (((val)>>B_CONTROLLERSHIFT) & B_CONTROLLERMASK)
95#define B_UNITSHIFT 16
96#define B_UNITMASK 0xf
97#define B_UNIT(val) (((val) >> B_UNITSHIFT) & B_UNITMASK)
98#define B_PARTITIONSHIFT 8
99#define B_PARTITIONMASK 0xff
100#define B_PARTITION(val) (((val) >> B_PARTITIONSHIFT) & B_PARTITIONMASK)
101#define B_TYPESHIFT 0
102#define B_TYPEMASK 0xff
103#define B_TYPE(val) (((val) >> B_TYPESHIFT) & B_TYPEMASK)
104
105#define B_MAGICMASK 0xf0000000
106#define B_DEVMAGIC 0xa0000000
107
108#define MAKEBOOTDEV(type, adaptor, controller, unit, partition) \
109 (((type) << B_TYPESHIFT) | ((adaptor) << B_ADAPTORSHIFT) | \
110 ((controller) << B_CONTROLLERSHIFT) | ((unit) << B_UNITSHIFT) | \
111 ((partition) << B_PARTITIONSHIFT) | B_DEVMAGIC)
