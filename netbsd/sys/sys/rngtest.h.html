32#define _RNGTEST_H
33
34#include <sys/types.h>
35
36#define FIPS140_RNG_TEST_BITS 20000
37#define FIPS140_RNG_TEST_BYTES (FIPS140_RNG_TEST_BITS / NBBY)
38
39typedef struct {
40 uint8_t rt_b[FIPS140_RNG_TEST_BYTES];
41 int rt_poker[16];
42 int rt_runs[2][7];
43 int rt_nerrs;
44 char rt_name[16];
45} rngtest_t;
46
47int rngtest(rngtest_t *const);

49#endif
