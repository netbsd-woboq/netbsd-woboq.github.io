85struct sockbuf {
86 struct selinfo sb_sel; /* process selecting read/write */
87 struct mowner *sb_mowner; /* who owns data for this sockbuf */
88 struct socket *sb_so; /* back pointer to socket */
89 kcondvar_t sb_cv; /* notifier */
90 /* When re-zeroing this struct, we zero from sb_startzero to the end */
91#define sb_startzero sb_cc
92 u_long sb_cc; /* actual chars in buffer */
93 u_long sb_hiwat; /* max actual char count */
94 u_long sb_mbcnt; /* chars of mbufs used */
95 u_long sb_mbmax; /* max chars of mbufs to use */
96 u_long sb_lowat; /* low water mark */
97 struct mbuf *sb_mb; /* the mbuf chain */
98 struct mbuf *sb_mbtail; /* the last mbuf in the chain */
99 struct mbuf *sb_lastrecord; /* first mbuf of last record in
100 socket buffer */
101 int sb_flags; /* flags, see below */
102 int sb_timeo; /* timeout for read/write */
103 u_long sb_overflowed; /* # of drops due to full buffer */
104};

132struct socket {
133 kmutex_t * volatile so_lock; /* pointer to lock on structure */
134 kcondvar_t so_cv; /* notifier */
135 short so_type; /* generic type, see socket.h */
136 short so_options; /* from socket call, see socket.h */
137 u_short so_linger; /* time to linger while closing */
138 short so_state; /* internal state flags SS_*, below */
139 int so_unused; /* used to be so_nbio */
140 void *so_pcb; /* protocol control block */
141 const struct protosw *so_proto; /* protocol handle */
142/*
143 * Variables for connection queueing.
144 * Socket where accepts occur is so_head in all subsidiary sockets.
145 * If so_head is 0, socket is not related to an accept.
146 * For head socket so_q0 queues partially completed connections,
147 * while so_q is a queue of connections ready to be accepted.
148 * If a connection is aborted and it has so_head set, then
149 * it has to be pulled out of either so_q0 or so_q.
150 * We allow connections to queue up based on current queue lengths
151 * and limit on number of queued connections for this socket.
152 */
153 struct socket *so_head; /* back pointer to accept socket */
154 struct soqhead *so_onq; /* queue (q or q0) that we're on */
155 struct soqhead so_q0; /* queue of partial connections */
156 struct soqhead so_q; /* queue of incoming connections */
157 TAILQ_ENTRY(socket) so_qe; /* our queue entry (q or q0) */
158 short so_q0len; /* partials on so_q0 */
159 short so_qlen; /* number of connections on so_q */
160 short so_qlimit; /* max number queued connections */
161 short so_timeo; /* connection timeout */
162 u_short so_error; /* error affecting connection */
163 u_short so_rerror; /* error affecting receiving */
164 u_short so_aborting; /* references from soabort() */
165 pid_t so_pgid; /* pgid for signals */
166 u_long so_oobmark; /* chars to oob mark */
167 struct sockbuf so_snd; /* send buffer */
168 struct sockbuf so_rcv; /* receive buffer */
169
170 void *so_internal; /* Space for svr4 stream data */
171 void (*so_upcall) (struct socket *, void *, int, int);
172 void * so_upcallarg; /* Arg for above */
173 int (*so_send) (struct socket *, struct sockaddr *,
174 struct uio *, struct mbuf *,
175 struct mbuf *, int, struct lwp *);
176 int (*so_receive) (struct socket *,
177 struct mbuf **,
178 struct uio *, struct mbuf **,
179 struct mbuf **, int *);
180 struct mowner *so_mowner; /* who owns mbufs for this socket */
181 struct uidinfo *so_uidinfo; /* who opened the socket */
182 gid_t so_egid; /* creator effective gid */
183 pid_t so_cpid; /* creator pid */
184 struct so_accf *so_accf;
185 kauth_cred_t so_cred; /* socket credentials */
186};

188/*
189 * Socket state bits.
190 */
191#define SS_NOFDREF 0x001 /* no file table ref any more */
192#define SS_ISCONNECTED 0x002 /* socket connected to a peer */
193#define SS_ISCONNECTING 0x004 /* in process of connecting to peer */
194#define SS_ISDISCONNECTING 0x008 /* in process of disconnecting */
195#define SS_CANTSENDMORE 0x010 /* can't send more data to peer */
196#define SS_CANTRCVMORE 0x020 /* can't receive more data from peer */
197#define SS_RCVATMARK 0x040 /* at mark on input */
198#define SS_ISABORTING 0x080 /* aborting fd references - close() */
199#define SS_RESTARTSYS 0x100 /* restart blocked system calls */
200#define SS_ISDISCONNECTED 0x800 /* socket disconnected from peer */
201
202#define SS_ASYNC 0x100 /* async i/o notify */
203#define SS_MORETOCOME 0x400 /*
204 * hint from sosend to lower layer;
205 * more data coming
206 */
207#define SS_ISAPIPE 0x1000 /* socket is implementing a pipe */
208#define SS_NBIO 0x2000 /* socket is in non blocking I/O */

387/*
388 * Do we need to notify the other side when I/O is possible?
389 */
390static __inline int
391sb_notify(struct sockbuf *sb)
392{
393
394 KASSERT(solocked(sb->sb_so));
395
396 return sb->sb_flags & (SB_NOTIFY | SB_ASYNC | SB_UPCALL | SB_KNOTE);
397}

399/*
400 * How much space is there in a socket buffer (so->so_snd or so->so_rcv)?
401 * Since the fields are unsigned, detect overflow and return 0.
402 */
403static __inline u_long
404sbspace(const struct sockbuf *sb)
405{
406
407 KASSERT(solocked(sb->sb_so));
408 if (sb->sb_hiwat <= sb->sb_cc || sb->sb_mbmax <= sb->sb_mbcnt)
409 return 0;
410 return lmin(sb->sb_hiwat - sb->sb_cc, sb->sb_mbmax - sb->sb_mbcnt);
411}
412
413static __inline u_long
414sbspace_oob(const struct sockbuf *sb)
415{
416 u_long hiwat = sb->sb_hiwat;
417
418 if (hiwat < ULONG_MAX - 1024)
419 hiwat += 1024;
420
421 KASSERT(solocked(sb->sb_so));
422
423 if (hiwat <= sb->sb_cc || sb->sb_mbmax <= sb->sb_mbcnt)
424 return 0;
425 return lmin(hiwat - sb->sb_cc, sb->sb_mbmax - sb->sb_mbcnt);
426}

436/* can we read something from so? */
437static __inline int
438soreadable(const struct socket *so)
439{
440
441 KASSERT(solocked(so));
442
443 return so->so_rcv.sb_cc >= so->so_rcv.sb_lowat ||
444 (so->so_state & SS_CANTRCVMORE) != 0 ||
445 so->so_qlen != 0 || so->so_error != 0;
446}
447
448/* can we write something to so? */
449static __inline int
450sowritable(const struct socket *so)
451{
452
453 KASSERT(solocked(so));
454
455 return (sbspace(&so->so_snd) >= so->so_snd.sb_lowat &&
456 ((so->so_state & SS_ISCONNECTED) != 0 ||
457 (so->so_proto->pr_flags & PR_CONNREQUIRED) == 0)) ||
458 (so->so_state & SS_CANTSENDMORE) != 0 ||
459 so->so_error != 0;
460}

462/* adjust counters in sb reflecting allocation of m */
463static __inline void
464sballoc(struct sockbuf *sb, struct mbuf *m)
465{
466
467 KASSERT(solocked(sb->sb_so));
468
469 sb->sb_cc += m->m_len;
470 sb->sb_mbcnt += MSIZE;
471 if (m->m_flags & M_EXT)
472 sb->sb_mbcnt += m->m_ext.ext_size;
473}
474
475/* adjust counters in sb reflecting freeing of m */
476static __inline void
477sbfree(struct sockbuf *sb, struct mbuf *m)
478{
479
480 KASSERT(solocked(sb->sb_so));
481
482 sb->sb_cc -= m->m_len;
483 sb->sb_mbcnt -= MSIZE;
484 if (m->m_flags & M_EXT)
485 sb->sb_mbcnt -= m->m_ext.ext_size;
486}

489sorwakeup(struct socket *so)
490{
491
492 KASSERT(solocked(so));
493
494 if (sb_notify(&so->so_rcv))

506}

508static __inline void
509solock(struct socket *so)
510{
511 kmutex_t *lock;
512
513 lock = so->so_lock;
514 mutex_enter(lock);
515 if (__predict_false(lock != so->so_lock))
516 solockretry(so, lock);
517}

520sounlock(struct socket *so)

522
523 mutex_exit(so->so_lock);

551/*
552 * Values for socket-buffer-append priority argument to sbappendaddrchain().
553 * The following flags are reserved for future implementation:
554 *
555 * SB_PRIO_NONE: honour normal socket-buffer limits.
556 *
557 * SB_PRIO_ONESHOT_OVERFLOW: if the socket has any space,
558 * deliver the entire chain. Intended for large requests
559 * that should be delivered in their entirety, or not at all.
560 *
561 * SB_PRIO_OVERDRAFT: allow a small (2*MLEN) overflow, over and
562 * aboce normal socket limits. Intended messages indicating
563 * buffer overflow in earlier normal/lower-priority messages .
564 *
565 * SB_PRIO_BESTEFFORT: Ignore limits entirely. Intended only for
566 * kernel-generated messages to specially-marked scokets which
567 * require "reliable" delivery, nd where the source socket/protocol
568 * message generator enforce some hard limit (but possibly well
569 * above kern.sbmax). It is entirely up to the in-kernel source to
570 * avoid complete mbuf exhaustion or DoS scenarios.
571 */
572#define SB_PRIO_NONE 0
573#define SB_PRIO_ONESHOT_OVERFLOW 1
574#define SB_PRIO_OVERDRAFT 2
575#define SB_PRIO_BESTEFFORT 3
