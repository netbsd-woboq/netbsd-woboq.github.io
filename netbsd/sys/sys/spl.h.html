29/*
30 * this header is intended to be included by MD header.
31 *
32 * an assumption: makeiplcookie() is reasonably fast.
33 * if it isn't the case for your port, it's better to have MD optimized
34 * splxxx() functions, rather than using this header.
35 */
36
37#if !defined(_KERNEL) && !defined(_KMEMUSER)
38#error not supposed to be exposed to userland.
39#endif /* !defined(_KERNEL) && !defined(_KMEMUSER) */
40
41#define _SPL_DECL(x, X) \
42 static __inline int \
43 spl##x(void) \
44 { return splraiseipl(makeiplcookie(IPL_##X)); }
45
46#if defined(IPL_SOFTCLOCK)
47_SPL_DECL(softclock, SOFTCLOCK)
48#endif /* defined(IPL_SOFTCLOCK) */
49#if defined(IPL_SOFTNET)
50_SPL_DECL(softnet, SOFTNET)
51#endif /* defined(IPL_SOFTNET) */
52#if defined(IPL_SOFTSERIAL)
53_SPL_DECL(softserial, SOFTSERIAL)
54#endif /* defined(IPL_SOFTSERIAL) */
55
56_SPL_DECL(vm, VM)
57_SPL_DECL(sched, SCHED)
58_SPL_DECL(high, HIGH)
59
60#undef _SPL_DECL
