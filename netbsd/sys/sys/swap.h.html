28#ifndef _SYS_SWAP_H_
29#define _SYS_SWAP_H_
30
31#include <sys/syslimits.h>
32
33/* Thise structure is used to return swap information for userland */
34
35struct swapent {
36 dev_t se_dev; /* device id */
37 int se_flags; /* flags */
38 int se_nblks; /* total blocks */
39 int se_inuse; /* blocks in use */
40 int se_priority; /* priority of this device */
41 char se_path[PATH_MAX+1]; /* path name */
42};
43
44#define SWAP_ON 1 /* begin swapping on device */
45#define SWAP_OFF 2 /* stop swapping on device */
46#define SWAP_NSWAP 3 /* how many swap devices ? */
47#define SWAP_STATS13 4 /* old SWAP_STATS, no se_path */
48#define SWAP_CTL 5 /* change priority on device */
49#define SWAP_STATS50 6 /* old SWAP_STATS, 32 bit dev_t */
50#define SWAP_DUMPDEV 7 /* use this device as dump device */
51#define SWAP_GETDUMPDEV 8 /* use this device as dump device */
52#define SWAP_DUMPOFF 9 /* stop using the dump device */
53#define SWAP_STATS 10 /* get device info */
54
55#define SWF_INUSE 0x00000001 /* in use: we have swapped here */
56#define SWF_ENABLE 0x00000002 /* enabled: we can swap here */
57#define SWF_BUSY 0x00000004 /* busy: I/O happening here */
58#define SWF_FAKE 0x00000008 /* fake: still being built */
59
60#endif /* _SYS_SWAP_H_ */
