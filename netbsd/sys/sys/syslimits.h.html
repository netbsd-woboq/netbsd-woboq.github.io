34#ifndef _SYS_SYSLIMITS_H_
35#define _SYS_SYSLIMITS_H_
36
37#include <sys/featuretest.h>
38
39#if defined(_KERNEL_OPT)
40#include "opt_syslimits.h"
41#endif
42
43#if defined(_POSIX_C_SOURCE) || defined(_XOPEN_SOURCE) || \
44 defined(_NETBSD_SOURCE)
45#define ARG_MAX (256 * 1024) /* max bytes for an exec function */
46#ifndef CHILD_MAX
47#define CHILD_MAX 160 /* max simultaneous processes */
48#endif
49#define GID_MAX 2147483647U /* max value for a gid_t (2^31-2) */
50#define LINK_MAX 32767 /* max file link count */
51#define MAX_CANON 255 /* max bytes in term canon input line */
52#define MAX_INPUT 255 /* max bytes in terminal input */
53#define NAME_MAX 511 /* max bytes in a file name, must be */
54 /* kept in sync with MAXNAMLEN */
55#define NGROUPS_MAX 16 /* max supplemental group id's */
56#define UID_MAX 2147483647U /* max value for a uid_t (2^31-2) */
57#ifndef OPEN_MAX
58#define OPEN_MAX 128 /* max open files per process */
59#endif
60#define PATH_MAX 1024 /* max bytes in pathname */
61#define PIPE_BUF 512 /* max bytes for atomic pipe writes */
62
63#define BC_BASE_MAX INT_MAX /* max ibase/obase values in bc(1) */
64#define BC_DIM_MAX 65535 /* max array elements in bc(1) */
65#define BC_SCALE_MAX INT_MAX /* max scale value in bc(1) */
66#define BC_STRING_MAX INT_MAX /* max const string length in bc(1) */
67#define COLL_WEIGHTS_MAX 2 /* max weights for order keyword */
68#define EXPR_NEST_MAX 32 /* max expressions nested in expr(1) */
69#define LINE_MAX 2048 /* max bytes in an input line */
70#define RE_DUP_MAX 255 /* max RE's in interval notation */
71
72/*
73 * IEEE Std 1003.1c-95, adopted in X/Open CAE Specification Issue 5 Version 2
74 */
75#if (_POSIX_C_SOURCE - 0) >= 199506L || (_XOPEN_SOURCE - 0) >= 500 || \
76 defined(_NETBSD_SOURCE)
77#define LOGIN_NAME_MAX 17 /* max login name length incl. NUL */
78#endif
79
80/*
81 * X/Open CAE Specification Issue 5 Version 2
82 */
83#if defined(_XOPEN_SOURCE) || defined(_NETBSD_SOURCE)
84#define IOV_MAX 1024 /* max # of iovec's for readv(2) etc. */
85#define NZERO 20 /* default "nice" */
86#endif /* _XOPEN_SOURCE || _NETBSD_SOURCE */
87
88#endif /* !_ANSI_SOURCE */
89
90#endif /* !_SYS_SYSLIMITS_H_ */
