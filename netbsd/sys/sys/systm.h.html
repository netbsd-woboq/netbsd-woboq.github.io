539/* Override db_console() in MD headers */
540#ifndef cn_trap
541#define cn_trap() console_debugger()
542#endif
543#ifndef cn_isconsole
544#define cn_isconsole(d) (cn_tab != NULL && (d) == cn_tab->cn_dev)

547void cn_init_magic(cnm_state_t *);
548void cn_destroy_magic(cnm_state_t *);
549int cn_set_magic(const char *);
550int cn_get_magic(char *, size_t);
551/* This should be called for each byte read */
552#ifndef cn_check_magic
553#define cn_check_magic(d, k, s) \
554 do { \
555 if (cn_isconsole(d)) { \
556 int _v = (s).cnm_magic[(s).cnm_state]; \
557 if ((k) == CNS_MAGIC_VAL(_v)) { \
558 (s).cnm_state = CNS_MAGIC_NEXT(_v); \
559 if ((s).cnm_state == CNS_TERM) { \
560 cn_trap(); \
561 (s).cnm_state = 0; \
562 } \
563 } else { \
564 (s).cnm_state = 0; \
565 } \
566 } \
567 } while (/* CONSTCOND */ 0)
568#endif
