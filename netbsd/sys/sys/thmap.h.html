27#ifndef _SYS_THMAP_H_
28#define _SYS_THMAP_H_
29
30#if !defined(_KERNEL)
31#error "not supposed to be exposed to userland"
32#endif
33
34__BEGIN_DECLS
35
36struct thmap;
37typedef struct thmap thmap_t;
38
39#define THMAP_NOCOPY 0x01
40#define THMAP_SETROOT 0x02
41
42typedef struct {
43 uintptr_t (*alloc)(size_t);
44 void (*free)(uintptr_t, size_t);
45} thmap_ops_t;
46
47thmap_t * thmap_create(uintptr_t, const thmap_ops_t *, unsigned);
48void thmap_destroy(thmap_t *);
49
50void * thmap_get(thmap_t *, const void *, size_t);
51void * thmap_put(thmap_t *, const void *, size_t, void *);
52void * thmap_del(thmap_t *, const void *, size_t);
53
54void * thmap_stage_gc(thmap_t *);
55void thmap_gc(thmap_t *, void *);
56
57int thmap_setroot(thmap_t *, uintptr_t);
58uintptr_t thmap_getroot(const thmap_t *);
59
60__END_DECLS
61
62#endif
