32#ifndef _SYS_THREADPOOL_H_
33#define _SYS_THREADPOOL_H_
34
35#if !defined(_KERNEL)
36#error "not supposed to be exposed to userland"
37#endif
38
39#include <sys/types.h>
40#include <sys/param.h>
41#include <sys/mutex.h>
42#include <sys/queue.h>
43
44struct threadpool;
45struct threadpool_job;
46struct threadpool_percpu;
47struct threadpool_thread;
48
49typedef void threadpool_job_fn_t(struct threadpool_job *);
50
51struct threadpool_job {
52 kmutex_t *job_lock;
53 struct threadpool_thread *job_thread;
54 TAILQ_ENTRY(threadpool_job) job_entry;
55 volatile unsigned int job_refcnt;
56 kcondvar_t job_cv;
57 threadpool_job_fn_t *job_fn;
58 char job_name[MAXCOMLEN];
59};
60
61void threadpools_init(void);
62
63int threadpool_get(struct threadpool **, pri_t);
64void threadpool_put(struct threadpool *, pri_t);
65
66int threadpool_percpu_get(struct threadpool_percpu **, pri_t);
67void threadpool_percpu_put(struct threadpool_percpu *, pri_t);
68struct threadpool *
69 threadpool_percpu_ref(struct threadpool_percpu *);
70struct threadpool *
71 threadpool_percpu_ref_remote(struct threadpool_percpu *,
72 struct cpu_info *);
73
74void threadpool_job_init(struct threadpool_job *, threadpool_job_fn_t,
75 kmutex_t *, const char *, ...) __printflike(4,5);
76void threadpool_job_destroy(struct threadpool_job *);
77void threadpool_job_done(struct threadpool_job *);
78
79void threadpool_schedule_job(struct threadpool *, struct threadpool_job *);
80void threadpool_cancel_job(struct threadpool *, struct threadpool_job *);
81bool threadpool_cancel_job_async(struct threadpool *,
82 struct threadpool_job *);
83
84#endif /* _SYS_THREADPOOL_H_ */
