32#ifndef _SYS_UIDINFO_H_
33#define _SYS_UIDINFO_H_
34
35#include <sys/types.h>
36#include <sys/queue.h>
37#include <sys/mutex.h>
38
39/*
40 * Structure associated with user caching.
41 */
42struct uidinfo {
43 SLIST_ENTRY(uidinfo) ui_hash;
44 uid_t ui_uid;
45 u_long ui_proccnt; /* Number of processes */
46 u_long ui_lwpcnt; /* Number of lwps */
47 u_long ui_lockcnt; /* Number of locks */
48 u_long ui_semcnt; /* Number of semaphores */
49 u_long ui_sbsize; /* Socket buffer size */
50};
51
52int chgproccnt(uid_t, int);
53int chglwpcnt(uid_t, int);
54int chgsemcnt(uid_t, int);
55int chgsbsize(struct uidinfo *, u_long *, u_long, rlim_t);
56struct uidinfo *uid_find(uid_t);
57void uid_init(void);
58
59#endif /* _SYS_UIDINFO_H_ */
