65#ifndef _SYS_USERRET_H_
66#define _SYS_USERRET_H_
67
68#include <sys/lockdebug.h>
69#include <sys/intr.h>
70#include <sys/psref.h>
71
72/*
73 * Define the MI code needed before returning to user mode, for
74 * trap and syscall.
75 */
76static __inline void
77mi_userret(struct lwp *l)
78{
79#ifndef __HAVE_PREEMPTION
80 struct cpu_info *ci;
81#endif
82
83 KASSERT(l->l_blcnt == 0);
84#ifndef __HAVE_PREEMPTION
85 KASSERT(curcpu()->ci_biglock_count == 0);
86#endif
87
88 /*
89 * Handle "exceptional" events: pending signals, stop/exit actions,
90 * etc. Note that the event must be flagged BEFORE any AST is
91 * posted as we are reading unlocked.
92 */
93#ifdef __HAVE_PREEMPTION
94 if (__predict_false(l->l_flag & LW_USERRET)) {
95 lwp_userret(l);
96 }
97 l->l_kpriority = false;
98 /*
99 * cpu_set_curpri(prio) is a MD optimized version of:
100 *
101 * kpreempt_disable();
102 * curcpu()->ci_schedstate.spc_curpriority = prio;
103 * kpreempt_enable();
104 */
105 cpu_set_curpri(l->l_priority); /* XXX this needs to die */
106#else
107 ci = l->l_cpu;
108 if (((l->l_flag & LW_USERRET) | ci->ci_data.cpu_softints) != 0) {
109 lwp_userret(l);
110 ci = l->l_cpu;
111 }
112 l->l_kpriority = false;
113 ci->ci_schedstate.spc_curpriority = l->l_priority;
114#endif
115
116 LOCKDEBUG_BARRIER(NULL, 0);
117 KASSERT(l->l_nopreempt == 0);
118 PSREF_DEBUG_BARRIER();
119 KASSERT(l->l_psrefs == 0);
120}
121
122#endif /* !_SYS_USERRET_H_ */
