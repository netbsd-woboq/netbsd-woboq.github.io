29#ifndef _SYS_VMEM_H_
30#define _SYS_VMEM_H_
31
32#include <sys/types.h>
33
34#if defined(_KERNEL)
35#else /* defined(_KERNEL) */
36#include <stdbool.h>
37#endif /* defined(_KERNEL) */
38
39typedef struct vmem vmem_t;
40
41typedef unsigned int vm_flag_t;
42
43typedef uintptr_t vmem_addr_t;
44typedef size_t vmem_size_t;
45#define VMEM_ADDR_MIN 0
46#define VMEM_ADDR_MAX (~(vmem_addr_t)0)
47
48typedef int (vmem_import_t)(vmem_t *, vmem_size_t, vm_flag_t, vmem_addr_t *);
49typedef void (vmem_release_t)(vmem_t *, vmem_addr_t, vmem_size_t);
50
51typedef int (vmem_ximport_t)(vmem_t *, vmem_size_t, vmem_size_t *,
52 vm_flag_t, vmem_addr_t *);
53
54extern vmem_t *kmem_arena;
55extern vmem_t *kmem_meta_arena;
56extern vmem_t *kmem_va_arena;
57
58void vmem_subsystem_init(vmem_t *vm);
59
60vmem_t *vmem_create(const char *, vmem_addr_t, vmem_size_t, vmem_size_t,
61 vmem_import_t *, vmem_release_t *, vmem_t *, vmem_size_t,
62 vm_flag_t, int);
63vmem_t *vmem_xcreate(const char *, vmem_addr_t, vmem_size_t, vmem_size_t,
64 vmem_ximport_t *, vmem_release_t *, vmem_t *, vmem_size_t,
65 vm_flag_t, int);
66vmem_t *vmem_init(vmem_t *, const char *, vmem_addr_t, vmem_size_t, vmem_size_t,
67 vmem_import_t *, vmem_release_t *, vmem_t *, vmem_size_t,
68 vm_flag_t, int);
69void vmem_destroy(vmem_t *);
70int vmem_alloc(vmem_t *, vmem_size_t, vm_flag_t, vmem_addr_t *);
71void vmem_free(vmem_t *, vmem_addr_t, vmem_size_t);
72int vmem_xalloc(vmem_t *, vmem_size_t, vmem_size_t, vmem_size_t,
73 vmem_size_t, vmem_addr_t, vmem_addr_t, vm_flag_t, vmem_addr_t *);
74void vmem_xfree(vmem_t *, vmem_addr_t, vmem_size_t);
75int vmem_add(vmem_t *, vmem_addr_t, vmem_size_t, vm_flag_t);
76vmem_size_t vmem_roundup_size(vmem_t *, vmem_size_t);
77vmem_size_t vmem_size(vmem_t *, int typemask);
78void vmem_rehash_start(void);
79void vmem_whatis(uintptr_t, void (*)(const char *, ...) __printflike(1, 2));
80void vmem_print(uintptr_t, const char *, void (*)(const char *, ...)
81 __printflike(1, 2));
82void vmem_printall(const char *, void (*)(const char *, ...)
83 __printflike(1, 2));
84
85/* vm_flag_t */
86#define VM_SLEEP 0x00000001
87#define VM_NOSLEEP 0x00000002
88#define VM_INSTANTFIT 0x00001000
89#define VM_BESTFIT 0x00002000
90#define VM_BOOTSTRAP 0x00010000
91#define VM_POPULATING 0x00040000
92#define VM_LARGEIMPORT 0x00080000
93#define VM_XIMPORT 0x00100000
94
95/* vmem_size typemask */
96#define VMEM_ALLOC 0x01
97#define VMEM_FREE 0x02
