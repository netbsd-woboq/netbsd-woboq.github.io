35#include <sys/vnode.h>

37struct namecache;
38
39enum vnode_state {
40 VS_ACTIVE, /* Assert only, fs node attached and usecount > 0. */
41 VS_MARKER, /* Stable, used as marker. Will not change. */
42 VS_LOADING, /* Intermediate, initialising the fs node. */
43 VS_LOADED, /* Stable, valid fs node attached. */
44 VS_BLOCKED, /* Intermediate, active, no new references allowed. */
45 VS_RECLAIMING, /* Intermediate, detaching the fs node. */
46 VS_RECLAIMED /* Stable, no fs node attached. */
47};
48
49TAILQ_HEAD(vnodelst, vnode_impl);
50typedef struct vnodelst vnodelst_t;

52struct vcache_key {
53 struct mount *vk_mount;
54 const void *vk_key;
55 size_t vk_key_len;
56};
57
58/*
59 * Reading or writing any of these items requires holding the appropriate
60 * lock. Field markings and the corresponding locks:
61 *
62 * c vcache_lock
63 * d vdrain_lock
64 * i v_interlock
65 * m mntvnode_lock
66 * n namecache_lock
67 * s syncer_data_lock
68 */
69struct vnode_impl {
70 struct vnode vi_vnode;
71 enum vnode_state vi_state; /* i: current state */
72 struct vnodelst *vi_lrulisthd; /* d: current lru list head */
73 TAILQ_ENTRY(vnode_impl) vi_lrulist; /* d: lru list */
74 LIST_HEAD(, namecache) vi_dnclist; /* n: namecaches (children) */
75 LIST_HEAD(, namecache) vi_nclist; /* n: namecaches (parent) */
76 int vi_synclist_slot; /* s: synclist slot index */
77 TAILQ_ENTRY(vnode_impl) vi_synclist; /* s: vnodes with dirty bufs */
78 TAILQ_ENTRY(vnode_impl) vi_mntvnodes; /* m: vnodes for mount point */
79 SLIST_ENTRY(vnode_impl) vi_hash; /* c: vnode cache list */
80 krwlock_t vi_lock; /* -: lock for this vnode */
81 struct vcache_key vi_key; /* c: vnode cache key */
82};
83typedef struct vnode_impl vnode_impl_t;
84
85#define VIMPL_TO_VNODE(vip) (&(vip)->vi_vnode)
