29#ifndef _SYS_WORKQUEUE_H_
30#define _SYS_WORKQUEUE_H_
31
32struct cpu_info;
33
34/*
35 * a simple "do it in thread context" framework.
36 *
37 * this framework is designed to make struct work small as
38 * far as possible, so that it can be embedded into other structures
39 * without too much cost.
40 */
41
42struct work {
43 void *wk_dummy;
44};
45
46struct workqueue;
47
48#define WQ_MPSAFE 0x01
49#define WQ_PERCPU 0x02
50
51int workqueue_create(struct workqueue **, const char *,
52 void (*)(struct work *, void *), void *, pri_t, int, int);
53void workqueue_destroy(struct workqueue *);
54void workqueue_wait(struct workqueue *, struct work *);
55
56void workqueue_enqueue(struct workqueue *, struct work *, struct cpu_info *);
57
58#endif /* _SYS_WORKQUEUE_H_ */
