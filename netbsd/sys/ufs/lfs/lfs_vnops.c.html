355/*
356 * Allocate a new inode.
357 */
358static int
359lfs_makeinode(struct vattr *vap, struct vnode *dvp,
360 const struct ulfs_lookup_results *ulr,
361 struct vnode **vpp, struct componentname *cnp)
362{
363 struct inode *ip;
364 struct vnode *tvp;
365 int error;
366
367 error = vcache_new(dvp->v_mount, dvp, vap, cnp->cn_cred, NULL, &tvp);
368 if (error)
369 return error;
370 error = vn_lock(tvp, LK_EXCLUSIVE);
371 if (error) {
372 vrele(tvp);
373 return error;
374 }
375 MARK_VNODE(tvp);
376 *vpp = tvp;
377 ip = VTOI(tvp);
378 ip->i_state |= IN_ACCESS | IN_CHANGE | IN_UPDATE;
379 ip->i_nlink = 1;
380 DIP_ASSIGN(ip, nlink, 1);
381
382 /* Authorize setting SGID if needed. */
383 if (ip->i_mode & ISGID) {
384 error = kauth_authorize_vnode(cnp->cn_cred, KAUTH_VNODE_WRITE_SECURITY,
385 tvp, NULL, genfs_can_chmod(tvp->v_type, cnp->cn_cred, ip->i_uid,
386 ip->i_gid, MAKEIMODE(vap->va_type, vap->va_mode)));
387 if (error) {
388 ip->i_mode &= ~ISGID;
389 DIP_ASSIGN(ip, mode, ip->i_mode);
390 }
391 }
392
393 if (cnp->cn_flags & ISWHITEOUT) {
394 ip->i_flags |= UF_OPAQUE;
395 DIP_ASSIGN(ip, flags, ip->i_flags);
396 }
397
398 /*
399 * Make sure inode goes to disk before directory entry.
400 */
401 if ((error = lfs_update(tvp, NULL, NULL, UPDATE_DIROP)) != 0)
402 goto bad;
403 error = ulfs_direnter(dvp, ulr, tvp,
404 cnp, ip->i_number, LFS_IFTODT(ip->i_mode), NULL);
405 if (error)
406 goto bad;
407 *vpp = tvp;
408 KASSERT(VOP_ISLOCKED(*vpp) == LK_EXCLUSIVE);
409 return (0);
410
411 bad:
412 /*
413 * Write error occurred trying to update the inode
414 * or the directory so must deallocate the inode.
415 */
416 ip->i_nlink = 0;
417 DIP_ASSIGN(ip, nlink, 0);
418 ip->i_state |= IN_CHANGE;
419 /* If IN_ADIROP, account for it */
420 UNMARK_VNODE(tvp);
421 vput(tvp);
422 return (error);
423}

425/*
426 * Synch an open file.
427 */
428/* ARGSUSED */
429int
430lfs_fsync(void *v)
431{
432 struct vop_fsync_args /* {
433 struct vnode *a_vp;
434 kauth_cred_t a_cred;
435 int a_flags;
436 off_t offlo;
437 off_t offhi;
438 } */ *ap = v;
439 struct vnode *vp = ap->a_vp;
440 int wait;
441 struct inode *ip = VTOI(vp);
442 struct lfs *fs = ip->i_lfs;
443 int error = 0;
444
445 KASSERT(VOP_ISLOCKED(vp) == LK_EXCLUSIVE);
446
447 /* If we're mounted read-only, don't try to sync. */
448 if (fs->lfs_ronly)
449 goto out;
450
451 /* If a removed vnode is being cleaned, no need to sync here. */
452 if ((ap->a_flags & FSYNC_RECLAIM) != 0 && ip->i_mode == 0)
453 goto out;
454
455 /*
456 * Trickle sync simply adds this vnode to the pager list, as if
457 * the pagedaemon had requested a pageout.
458 */
459 if (ap->a_flags & FSYNC_LAZY) {
460 if (lfs_ignore_lazy_sync == 0) {
461 mutex_enter(&lfs_lock);
462 if (!(ip->i_state & IN_PAGING)) {
463 ip->i_state |= IN_PAGING;
464 TAILQ_INSERT_TAIL(&fs->lfs_pchainhd, ip,
465 i_lfs_pchain);
466 }
467 cv_broadcast(&lfs_writerd_cv);
468 mutex_exit(&lfs_lock);
469 }
470 goto out;
471 }
472
473 /*
474 * If a vnode is being cleaned, flush it out before we try to
475 * reuse it. This prevents the cleaner from writing files twice
476 * in the same partial segment, causing an accounting underflow.
477 */
478 if (ap->a_flags & FSYNC_RECLAIM && ip->i_state & IN_CLEANING) {
479 lfs_vflush(vp);
480 }
481
482 wait = (ap->a_flags & FSYNC_WAIT);
483 do {
484 mutex_enter(vp->v_interlock);
485 error = VOP_PUTPAGES(vp, trunc_page(ap->a_offlo),
486 round_page(ap->a_offhi),
487 PGO_CLEANIT | (wait ? PGO_SYNCIO : 0));
488 if (error == EAGAIN) {
489 mutex_enter(&lfs_lock);
490 mtsleep(&fs->lfs_availsleep, PCATCH | PUSER,
491 "lfs_fsync", hz / 100 + 1, &lfs_lock);
492 mutex_exit(&lfs_lock);
493 }
494 } while (error == EAGAIN);
495 if (error)
496 goto out;
497
498 if ((ap->a_flags & FSYNC_DATAONLY) == 0)
499 error = lfs_update(vp, NULL, NULL, wait ? UPDATE_WAIT : 0);
500
501 if (error == 0 && ap->a_flags & FSYNC_CACHE) {
502 int l = 0;
503 error = VOP_IOCTL(ip->i_devvp, DIOCCACHESYNC, &l, FWRITE,
504 curlwp->l_cred);
505 }
506 if (wait && !VPISEMPTY(vp))
507 LFS_SET_UINO(ip, IN_MODIFIED);
508
509out:

555int
556lfs_set_dirop(struct vnode *dvp, struct vnode *vp)
557{
558 struct lfs *fs;
559 int error;
560
561 KASSERT(VOP_ISLOCKED(dvp) == LK_EXCLUSIVE);
562 KASSERT(vp == NULL || VOP_ISLOCKED(vp) == LK_EXCLUSIVE);
563
564 fs = VTOI(dvp)->i_lfs;
565
566 ASSERT_NO_SEGLOCK(fs);
567 /*
568 * LFS_NRESERVE calculates direct and indirect blocks as well
569 * as an inode block; an overestimate in most cases.
570 */
571 if ((error = lfs_reserve(fs, dvp, vp, LFS_NRESERVE(fs))) != 0)
572 return (error);
573
574 restart:
575 mutex_enter(&lfs_lock);
576 if (fs->lfs_dirops == 0) {
577 mutex_exit(&lfs_lock);
578 lfs_check(dvp, LFS_UNUSED_LBN, 0);
579 mutex_enter(&lfs_lock);
580 }
581 while (fs->lfs_writer) {
582 error = cv_wait_sig(&fs->lfs_diropscv, &lfs_lock);
583 if (error == EINTR) {
584 mutex_exit(&lfs_lock);
585 goto unreserve;
586 }
587 }
588 if (lfs_dirvcount > LFS_MAX_DIROP && fs->lfs_dirops == 0) {
589 cv_broadcast(&lfs_writerd_cv);
590 mutex_exit(&lfs_lock);
591 preempt();
592 goto restart;
593 }
594
595 if (lfs_dirvcount > LFS_MAX_DIROP) {
596 DLOG((DLOG_DIROP, "lfs_set_dirop: sleeping with dirops=%d, "
597 "dirvcount=%d\n", fs->lfs_dirops, lfs_dirvcount));
598 if ((error = mtsleep(&lfs_dirvcount,
599 PCATCH | PUSER | PNORELOCK, "lfs_maxdirop", 0,
600 &lfs_lock)) != 0) {
601 mutex_exit(&lfs_lock);
602 goto unreserve;
603 }
604 mutex_exit(&lfs_lock);
605 goto restart;
606 }
607
608 ++fs->lfs_dirops;
609 /* fs->lfs_doifile = 1; */ /* XXX why? --ks */
610 mutex_exit(&lfs_lock);
611
612 /* Hold a reference so SET_ENDOP will be happy */
613 vref(dvp);
614 if (vp) {
615 vref(vp);
616 MARK_VNODE(vp);
617 }
618
619 MARK_VNODE(dvp);
620 return 0;
621
622 unreserve:
623 lfs_reserve(fs, dvp, vp, -LFS_NRESERVE(fs));
624 return error;
625}

627/*
628 * Opposite of lfs_set_dirop... mostly. For now at least must call
629 * UNMARK_VNODE(dvp) explicitly first. (XXX: clean that up)
630 */
631void
632lfs_unset_dirop(struct lfs *fs, struct vnode *dvp, const char *str)
633{
634 mutex_enter(&lfs_lock);
635 --fs->lfs_dirops;
636 if (!fs->lfs_dirops) {
637 if (fs->lfs_nadirop) {
638 panic("lfs_unset_dirop: %s: no dirops but "
639 " nadirop=%d", str,
640 fs->lfs_nadirop);
641 }
642 wakeup(&fs->lfs_writer);
643 mutex_exit(&lfs_lock);
644 lfs_check(dvp, LFS_UNUSED_LBN, 0);
645 } else {
646 mutex_exit(&lfs_lock);
647 }
648 lfs_reserve(fs, dvp, NULL, -LFS_NRESERVE(fs));
649}

651void
652lfs_mark_vnode(struct vnode *vp)
653{
654 struct inode *ip = VTOI(vp);
655 struct lfs *fs = ip->i_lfs;
656
657 mutex_enter(&lfs_lock);
658 if (!(ip->i_state & IN_ADIROP)) {
659 if (!(vp->v_uflag & VU_DIROP)) {
660 mutex_exit(&lfs_lock);
661 vref(vp);
662 mutex_enter(&lfs_lock);
663 ++lfs_dirvcount;
664 ++fs->lfs_dirvcount;
665 TAILQ_INSERT_TAIL(&fs->lfs_dchainhd, ip, i_lfs_dchain);
666 vp->v_uflag |= VU_DIROP;
667 }
668 ++fs->lfs_nadirop;
669 ip->i_state &= ~IN_CDIROP;
670 ip->i_state |= IN_ADIROP;
671 } else
672 KASSERT(vp->v_uflag & VU_DIROP);
673 mutex_exit(&lfs_lock);
674}

684 --ip->i_lfs->lfs_nadirop;
685 ip->i_state &= ~IN_ADIROP;
686 }
687 mutex_exit(&lfs_lock);

758 }

762 vrele(*vpp);

765 UNMARK_VNODE(dvp);
766 /* XXX: is it even possible for the symlink to get MARK'd? */

769 *vpp = NULL;
770 }

917lfs_mkdir(void *v)
918{
919 struct vop_mkdir_v3_args /* {
920 struct vnode *a_dvp;

928 struct componentname *cnp;
929 struct vattr *vap;
930 struct ulfs_lookup_results *ulr;
931 struct buf *bp;
932 LFS_DIRHEADER *dirp;
933 int dirblksiz;
934 int error;
935
936 dvp = ap->a_dvp;
937 tvp = NULL;
938 vpp = ap->a_vpp;
939 cnp = ap->a_cnp;
940 vap = ap->a_vap;
941
942 KASSERT(VOP_ISLOCKED(dvp) == LK_EXCLUSIVE);
943
944 dp = VTOI(dvp);
945 ip = NULL;
946
947 KASSERT(vap->va_type == VDIR);
948 KASSERT(vpp != NULL);
949 KASSERT(*vpp == NULL);
950
951 /* XXX should handle this material another way */
952 ulr = &dp->i_crap;
953 ULFS_CHECK_CRAPCOUNTER(dp);
954
955 fs = VFSTOULFS(dvp->v_mount)->um_lfs;
956 ASSERT_NO_SEGLOCK(fs);
957 if (fs->lfs_ronly) {
958 return EROFS;
959 }
960
961 if ((nlink_t)dp->i_nlink >= LINK_MAX) {
962 return EMLINK;
963 }
964
965 dirblksiz = fs->um_dirblksiz;
966 /* XXX dholland 20150911 I believe this to be true, but... */
967 //KASSERT(dirblksiz == LFS_DIRBLKSIZ);
968
969 error = lfs_set_dirop(dvp, NULL);
970 if (error)
971 return error;
972
973 /*
974 * Must simulate part of lfs_makeinode here to acquire the inode,
975 * but not have it entered in the parent directory. The entry is
976 * made later after writing "." and ".." entries.
977 */
978 error = vcache_new(dvp->v_mount, dvp, vap, cnp->cn_cred, NULL,
979 ap->a_vpp);
980 if (error)
981 goto out;
982
983 error = vn_lock(*ap->a_vpp, LK_EXCLUSIVE);
984 if (error) {
985 vrele(*ap->a_vpp);
986 *ap->a_vpp = NULL;
987 goto out;
988 }
989
990 tvp = *ap->a_vpp;
991 MARK_VNODE(tvp);
992 ip = VTOI(tvp);
993 ip->i_state |= IN_ACCESS | IN_CHANGE | IN_UPDATE;
994 ip->i_nlink = 2;
995 DIP_ASSIGN(ip, nlink, 2);
996 if (cnp->cn_flags & ISWHITEOUT) {
997 ip->i_flags |= UF_OPAQUE;
998 DIP_ASSIGN(ip, flags, ip->i_flags);
999 }
1000
1001 /*
1002 * Bump link count in parent directory to reflect work done below.
1003 */
1004 dp->i_nlink++;
1005 DIP_ASSIGN(dp, nlink, dp->i_nlink);
1006 dp->i_state |= IN_CHANGE;
1007 if ((error = lfs_update(dvp, NULL, NULL, UPDATE_DIROP)) != 0)
1008 goto bad;
1009
1010 /*
1011 * Initialize directory with "." and "..". This used to use a
1012 * static template but that adds moving parts for very little
1013 * benefit.
1014 */
1015 if ((error = lfs_balloc(tvp, (off_t)0, dirblksiz, cnp->cn_cred,
1016 B_CLRBUF, &bp)) != 0)
1017 goto bad;
1018 ip->i_size = dirblksiz;
1019 DIP_ASSIGN(ip, size, dirblksiz);
1020 ip->i_state |= IN_ACCESS | IN_CHANGE | IN_UPDATE;
1021 uvm_vnp_setsize(tvp, ip->i_size);
1022 dirp = bp->b_data;
1023
1024 /* . */
1025 lfs_dir_setino(fs, dirp, ip->i_number);
1026 lfs_dir_setreclen(fs, dirp, LFS_DIRECTSIZ(fs, 1));
1027 lfs_dir_settype(fs, dirp, LFS_DT_DIR);
1028 lfs_dir_setnamlen(fs, dirp, 1);
1029 lfs_copydirname(fs, lfs_dir_nameptr(fs, dirp), ".", 1,
1030 LFS_DIRECTSIZ(fs, 1));
1031 dirp = LFS_NEXTDIR(fs, dirp);
1032 /* .. */
1033 lfs_dir_setino(fs, dirp, dp->i_number);
1034 lfs_dir_setreclen(fs, dirp, dirblksiz - LFS_DIRECTSIZ(fs, 1));
1035 lfs_dir_settype(fs, dirp, LFS_DT_DIR);
1036 lfs_dir_setnamlen(fs, dirp, 2);
1037 lfs_copydirname(fs, lfs_dir_nameptr(fs, dirp), "..", 2,
1038 dirblksiz - LFS_DIRECTSIZ(fs, 1));
1039
1040 /*
1041 * Directory set up; now install its entry in the parent directory.
1042 */
1043 if ((error = VOP_BWRITE(bp->b_vp, bp)) != 0)
1044 goto bad;
1045 if ((error = lfs_update(tvp, NULL, NULL, UPDATE_DIROP)) != 0) {
1046 goto bad;
1047 }
1048 error = ulfs_direnter(dvp, ulr, tvp,
1049 cnp, ip->i_number, LFS_IFTODT(ip->i_mode), bp);
1050 bad:
1051 if (error == 0) {
1052 VN_KNOTE(dvp, NOTE_WRITE | NOTE_LINK);
1053 VOP_UNLOCK(tvp);
1054 } else {
1055 dp->i_nlink--;
1056 DIP_ASSIGN(dp, nlink, dp->i_nlink);
1057 dp->i_state |= IN_CHANGE;
1058 /*
1059 * No need to do an explicit lfs_truncate here, vrele will
1060 * do this for us because we set the link count to 0.
1061 */
1062 ip->i_nlink = 0;
1063 DIP_ASSIGN(ip, nlink, 0);
1064 ip->i_state |= IN_CHANGE;
1065 /* If IN_ADIROP, account for it */
1066 UNMARK_VNODE(tvp);
1067 vput(tvp);
1068 }
1069
1070out:
1071 UNMARK_VNODE(dvp);
1072 UNMARK_VNODE(*vpp);
1073 if (error) {
1074 *vpp = NULL;
1075 }
1076 lfs_unset_dirop(fs, dvp, "mkdir");
1077
1078 vrele(dvp);
1079 return (error);
1080}

1408/*
1409 * Reclaim an inode so that it can be used for other purposes.
1410 */
1411
1412int
1413lfs_reclaim(void *v)
1414{

1416 struct vnode *a_vp;
1417 } */ *ap = v;
1418 struct vnode *vp = ap->a_vp;
1419 struct inode *ip;
1420 struct lfs *fs;
1421 int error;
1422
1423 VOP_UNLOCK(vp);
1424
1425 ip = VTOI(vp);
1426 fs = ip->i_lfs;
1427
1428 /*
1429 * The inode must be freed and updated before being removed
1430 * from its hash chain. Other threads trying to gain a hold
1431 * or lock on the inode will be stalled.
1432 */
1433 if (ip->i_nlink <= 0 && (vp->v_mount->mnt_flag & MNT_RDONLY) == 0)
1434 lfs_vfree(vp, ip->i_number, ip->i_omode);

1437 LFS_CLR_UINO(ip, IN_ALLMOD);
1438 mutex_exit(&lfs_lock);
1439 if ((error = ulfs_reclaim(vp)))
1440 return (error);
1441
1442 /*
1443 * Take us off the paging and/or dirop queues if we were on them.
1444 * We shouldn't be on them.
1445 */
1446 mutex_enter(&lfs_lock);
1447 if (ip->i_state & IN_PAGING) {
1448 log(LOG_WARNING, "%s: reclaimed vnode is IN_PAGING\n",

1450 ip->i_state &= ~IN_PAGING;
1451 TAILQ_REMOVE(&fs->lfs_pchainhd, ip, i_lfs_pchain);

1463 vp->v_data = NULL;

1467/*
1468 * Read a block from a storage device.
1469 *
1470 * Calculate the logical to physical mapping if not done already,
1471 * then call the device strategy routine.
1472 *
1473 * In order to avoid reading blocks that are in the process of being
1474 * written by the cleaner---and hence are not mutexed by the normal
1475 * buffer cache / page cache mechanisms---check for collisions before
1476 * reading.
1477 *
1478 * We inline ulfs_strategy to make sure that the VOP_BMAP occurs *before*
1479 * the active cleaner test.
1480 *
1481 * XXX This code assumes that lfs_markv makes synchronous checkpoints.
1482 */
1483int
1484lfs_strategy(void *v)
1485{
1486 struct vop_strategy_args /* {
1487 struct vnode *a_vp;
1488 struct buf *a_bp;
1489 } */ *ap = v;
1490 struct buf *bp;
1491 struct lfs *fs;
1492 struct vnode *vp;
1493 struct inode *ip;
1494 daddr_t tbn;
1495#define MAXLOOP 25
1496 int i, sn, error, slept, loopcount;
1497
1498 bp = ap->a_bp;
1499 vp = ap->a_vp;
1500 ip = VTOI(vp);
1501 fs = ip->i_lfs;
1502
1503 /* lfs uses its strategy routine only for read */
1504 KASSERT(bp->b_flags & B_READ);
1505
1506 if (vp->v_type == VBLK || vp->v_type == VCHR)
1507 panic("lfs_strategy: spec");
1508 KASSERT(bp->b_bcount != 0);
1509 if (bp->b_blkno == bp->b_lblkno) {
1510 error = VOP_BMAP(vp, bp->b_lblkno, NULL, &bp->b_blkno,
1511 NULL);
1512 if (error) {
1513 bp->b_error = error;
1514 bp->b_resid = bp->b_bcount;
1515 biodone(bp);
1516 return (error);
1517 }
1518 if ((long)bp->b_blkno == -1) /* no valid data */
1519 clrbuf(bp);
1520 }
1521 if ((long)bp->b_blkno < 0) { /* block is not on disk */
1522 bp->b_resid = bp->b_bcount;
1523 biodone(bp);
1524 return (0);
1525 }
1526
1527 slept = 1;
1528 loopcount = 0;
1529 mutex_enter(&lfs_lock);
1530 while (slept && fs->lfs_seglock) {
1531 mutex_exit(&lfs_lock);
1532 /*
1533 * Look through list of intervals.
1534 * There will only be intervals to look through
1535 * if the cleaner holds the seglock.
1536 * Since the cleaner is synchronous, we can trust
1537 * the list of intervals to be current.
1538 */
1539 tbn = LFS_DBTOFSB(fs, bp->b_blkno);
1540 sn = lfs_dtosn(fs, tbn);
1541 slept = 0;
1542 for (i = 0; i < fs->lfs_cleanind; i++) {
1543 if (sn == lfs_dtosn(fs, fs->lfs_cleanint[i]) &&
1544 tbn >= fs->lfs_cleanint[i]) {
1545 DLOG((DLOG_CLEAN,
1546 "lfs_strategy: ino %llu lbn %" PRId64
1547 " ind %d sn %d fsb %" PRIx64
1548 " given sn %d fsb %" PRIx64 "\n",
1549 (unsigned long long) ip->i_number,
1550 bp->b_lblkno, i,
1551 lfs_dtosn(fs, fs->lfs_cleanint[i]),
1552 fs->lfs_cleanint[i], sn, tbn));
1553 DLOG((DLOG_CLEAN,
1554 "lfs_strategy: sleeping on ino %llu lbn %"
1555 PRId64 "\n",
1556 (unsigned long long) ip->i_number,
1557 bp->b_lblkno));
1558 mutex_enter(&lfs_lock);
1559 if (LFS_SEGLOCK_HELD(fs) && fs->lfs_iocount) {
1560 /*
1561 * Cleaner can't wait for itself.
1562 * Instead, wait for the blocks
1563 * to be written to disk.
1564 * XXX we need pribio in the test
1565 * XXX here.
1566 */
1567 mtsleep(&fs->lfs_iocount,
1568 (PRIBIO + 1) | PNORELOCK,
1569 "clean2", hz/10 + 1,
1570 &lfs_lock);
1571 slept = 1;
1572 ++loopcount;
1573 break;
1574 } else if (fs->lfs_seglock) {
1575 mtsleep(&fs->lfs_seglock,
1576 (PRIBIO + 1) | PNORELOCK,
1577 "clean1", 0,
1578 &lfs_lock);
1579 slept = 1;
1580 break;
1581 }
1582 mutex_exit(&lfs_lock);
1583 }
1584 }
1585 mutex_enter(&lfs_lock);
1586 if (loopcount > MAXLOOP) {
1587 printf("lfs_strategy: breaking out of clean2 loop\n");
1588 break;
1589 }
1590 }
1591 mutex_exit(&lfs_lock);
1592
1593 vp = ip->i_devvp;
1594 return VOP_STRATEGY(vp, bp);
1595}

1597/*
1598 * Inline lfs_segwrite/lfs_writevnodes, but just for dirops.
1599 * Technically this is a checkpoint (the on-disk state is valid)
1600 * even though we are leaving out all the file data.
1601 */
1602int
1603lfs_flush_dirops(struct lfs *fs)
1604{
1605 struct inode *ip, *nip;
1606 struct vnode *vp;
1607 extern int lfs_dostats; /* XXX this does not belong here */
1608 struct segment *sp;
1609 SEGSUM *ssp;
1610 int flags = 0;
1611 int error = 0;
1612
1613 ASSERT_MAYBE_SEGLOCK(fs);
1614 KASSERT(fs->lfs_nadirop == 0);
1615
1616 if (fs->lfs_ronly)
1617 return EROFS;
1618
1619 mutex_enter(&lfs_lock);
1620 if (TAILQ_FIRST(&fs->lfs_dchainhd) == NULL) {
1621 mutex_exit(&lfs_lock);
1622 return 0;
1623 } else
1624 mutex_exit(&lfs_lock);
1625
1626 if (lfs_dostats)
1627 ++lfs_stats.flush_invoked;
1628
1629 lfs_imtime(fs);
1630 lfs_seglock(fs, flags);
1631 sp = fs->lfs_sp;
1632
1633 /*
1634 * lfs_writevnodes, optimized to get dirops out of the way.
1635 * Only write dirops, and don't flush files' pages, only
1636 * blocks from the directories.
1637 *
1638 * We don't need to vref these files because they are
1639 * dirops and so hold an extra reference until the
1640 * segunlock clears them of that status.
1641 *
1642 * We don't need to check for IN_ADIROP because we know that
1643 * no dirops are active.
1644 *
1645 */
1646 mutex_enter(&lfs_lock);
1647 for (ip = TAILQ_FIRST(&fs->lfs_dchainhd); ip != NULL; ip = nip) {
1648 nip = TAILQ_NEXT(ip, i_lfs_dchain);
1649 mutex_exit(&lfs_lock);
1650 vp = ITOV(ip);
1651 mutex_enter(vp->v_interlock);
1652
1653 KASSERT((ip->i_state & IN_ADIROP) == 0);
1654 KASSERT(vp->v_uflag & VU_DIROP);
1655 KASSERT(vdead_check(vp, VDEAD_NOWAIT) == 0);
1656
1657 /*
1658 * All writes to directories come from dirops; all
1659 * writes to files' direct blocks go through the page
1660 * cache, which we're not touching. Reads to files
1661 * and/or directories will not be affected by writing
1662 * directory blocks inodes and file inodes. So we don't
1663 * really need to lock.
1664 */
1665 if (vdead_check(vp, VDEAD_NOWAIT) != 0) {
1666 mutex_exit(vp->v_interlock);
1667 mutex_enter(&lfs_lock);
1668 continue;
1669 }
1670 mutex_exit(vp->v_interlock);
1671 /* XXX see below
1672 * waslocked = VOP_ISLOCKED(vp);
1673 */
1674 if (vp->v_type != VREG &&
1675 ((ip->i_state & IN_ALLMOD) || !VPISEMPTY(vp))) {
1676 error = lfs_writefile(fs, sp, vp);
1677 if (!VPISEMPTY(vp) && !WRITEINPROG(vp) &&
1678 !(ip->i_state & IN_ALLMOD)) {
1679 mutex_enter(&lfs_lock);
1680 LFS_SET_UINO(ip, IN_MODIFIED);
1681 mutex_exit(&lfs_lock);
1682 }
1683 if (error && (sp->seg_flags & SEGM_SINGLE)) {
1684 mutex_enter(&lfs_lock);
1685 error = EAGAIN;
1686 break;
1687 }
1688 }
1689 KDASSERT(ip->i_number != LFS_IFILE_INUM);
1690 error = lfs_writeinode(fs, sp, ip);
1691 mutex_enter(&lfs_lock);
1692 if (error && (sp->seg_flags & SEGM_SINGLE)) {
1693 error = EAGAIN;
1694 break;
1695 }
1696
1697 /*
1698 * We might need to update these inodes again,
1699 * for example, if they have data blocks to write.
1700 * Make sure that after this flush, they are still
1701 * marked IN_MODIFIED so that we don't forget to
1702 * write them.
1703 */
1704 /* XXX only for non-directories? --KS */
1705 LFS_SET_UINO(ip, IN_MODIFIED);
1706 }
1707 mutex_exit(&lfs_lock);
1708 /* We've written all the dirops there are */
1709 ssp = (SEGSUM *)sp->segsum;
1710 lfs_ss_setflags(fs, ssp, lfs_ss_getflags(fs, ssp) & ~(SS_CONT));
1711 lfs_finalize_fs_seguse(fs);
1712 (void) lfs_writeseg(fs, sp);
1713 lfs_segunlock(fs);
1714
1715 return error;
1716}

1718/*
1719 * Flush all vnodes for which the pagedaemon has requested pageouts.
1720 * Skip over any files that are marked VU_DIROP (since lfs_flush_dirop()
1721 * has just run, this would be an error). If we have to skip a vnode
1722 * for any reason, just skip it; if we have to wait for the cleaner,
1723 * abort. The writer daemon will call us again later.
1724 */
1725int
1726lfs_flush_pchain(struct lfs *fs)
1727{
1728 struct inode *ip, *nip;
1729 struct vnode *vp;
1730 extern int lfs_dostats;
1731 struct segment *sp;
1732 int error, error2;
1733
1734 ASSERT_NO_SEGLOCK(fs);
1735
1736 if (fs->lfs_ronly)
1737 return EROFS;
1738
1739 mutex_enter(&lfs_lock);
1740 if (TAILQ_FIRST(&fs->lfs_pchainhd) == NULL) {
1741 mutex_exit(&lfs_lock);
1742 return 0;
1743 } else
1744 mutex_exit(&lfs_lock);
1745
1746 /* Get dirops out of the way */
1747 if ((error = lfs_flush_dirops(fs)) != 0)
1748 return error;
1749
1750 if (lfs_dostats)
1751 ++lfs_stats.flush_invoked;
1752
1753 /*
1754 * Inline lfs_segwrite/lfs_writevnodes, but just for pageouts.
1755 */
1756 lfs_imtime(fs);
1757 lfs_seglock(fs, 0);
1758 sp = fs->lfs_sp;
1759
1760 /*
1761 * lfs_writevnodes, optimized to clear pageout requests.
1762 * Only write non-dirop files that are in the pageout queue.
1763 * We're very conservative about what we write; we want to be
1764 * fast and async.
1765 */
1766 mutex_enter(&lfs_lock);
1767 top:
1768 for (ip = TAILQ_FIRST(&fs->lfs_pchainhd); ip != NULL; ip = nip) {
1769 struct mount *mp = ITOV(ip)->v_mount;
1770 ino_t ino = ip->i_number;
1771
1772 nip = TAILQ_NEXT(ip, i_lfs_pchain);
1773
1774 if (!(ip->i_state & IN_PAGING))
1775 goto top;
1776
1777 mutex_exit(&lfs_lock);
1778 if (vcache_get(mp, &ino, sizeof(ino), &vp) != 0) {
1779 mutex_enter(&lfs_lock);
1780 continue;
1781 };
1782 if (vn_lock(vp, LK_EXCLUSIVE | LK_NOWAIT) != 0) {
1783 vrele(vp);
1784 mutex_enter(&lfs_lock);
1785 continue;
1786 }
1787 ip = VTOI(vp);
1788 mutex_enter(&lfs_lock);
1789 if ((vp->v_uflag & VU_DIROP) != 0 || vp->v_type != VREG ||
1790 !(ip->i_state & IN_PAGING)) {
1791 mutex_exit(&lfs_lock);
1792 vput(vp);
1793 mutex_enter(&lfs_lock);
1794 goto top;
1795 }
1796 mutex_exit(&lfs_lock);
1797
1798 error = lfs_writefile(fs, sp, vp);
1799 if (!VPISEMPTY(vp) && !WRITEINPROG(vp) &&
1800 !(ip->i_state & IN_ALLMOD)) {
1801 mutex_enter(&lfs_lock);
1802 LFS_SET_UINO(ip, IN_MODIFIED);
1803 mutex_exit(&lfs_lock);
1804 }
1805 KDASSERT(ip->i_number != LFS_IFILE_INUM);
1806 error2 = lfs_writeinode(fs, sp, ip);
1807
1808 VOP_UNLOCK(vp);
1809 vrele(vp);
1810
1811 if (error == EAGAIN || error2 == EAGAIN) {
1812 lfs_writeseg(fs, sp);
1813 mutex_enter(&lfs_lock);
1814 break;
1815 }
1816 mutex_enter(&lfs_lock);
1817 }
1818 mutex_exit(&lfs_lock);
1819 (void) lfs_writeseg(fs, sp);
1820 lfs_segunlock(fs);
1821
1822 return 0;
1823}

1858 struct vop_fcntl_args /* {
1859 struct vnode *a_vp;
1860 u_int a_command;
1861 void * a_data;
1862 int a_fflag;
1863 kauth_cred_t a_cred;
1864 } */ *ap = v;
1865 struct timeval tv;
1866 struct timeval *tvp;
1867 BLOCK_INFO *blkiov;
1868 BLOCK_INFO_70 *blkiov70;
1869 CLEANERINFO *cip;
1870 SEGUSE *sup;
1871 int blkcnt, i, error;
1872 size_t fh_size;
1873 struct lfs_fcntl_markv blkvp;
1874 struct lfs_fcntl_markv_70 blkvp70;
1875 struct lwp *l;
1876 fsid_t *fsidp;
1877 struct lfs *fs;
1878 struct buf *bp;
1879 fhandle_t *fhp;
1880 daddr_t off;
1881 int oclean;
1882
1883 /* Only respect LFS fcntls on fs root or Ifile */
1884 if (VTOI(ap->a_vp)->i_number != ULFS_ROOTINO &&
1885 VTOI(ap->a_vp)->i_number != LFS_IFILE_INUM) {
1886 return ulfs_fcntl(v);
1887 }
1888
1889 /* Avoid locking a draining lock */
1890 if (ap->a_vp->v_mount->mnt_iflag & IMNT_UNMOUNT) {
1891 return ESHUTDOWN;
1892 }
1893
1894 /* LFS control and monitoring fcntls are available only to root */
1895 l = curlwp;
1896 if (((ap->a_command & 0xff00) >> 8) == 'L' &&
1897 (error = kauth_authorize_system(l->l_cred, KAUTH_SYSTEM_LFS,
1898 KAUTH_REQ_SYSTEM_LFS_FCNTL, NULL, NULL, NULL)) != 0)
1899 return (error);
1900
1901 fs = VTOI(ap->a_vp)->i_lfs;
1902 fsidp = &ap->a_vp->v_mount->mnt_stat.f_fsidx;
1903
1904 error = 0;
1905 switch ((int)ap->a_command) {
1906 case LFCNSEGWAITALL_COMPAT_50:
1907 case LFCNSEGWAITALL_COMPAT:
1908 fsidp = NULL;
1909 /* FALLTHROUGH */
1910 case LFCNSEGWAIT_COMPAT_50:
1911 case LFCNSEGWAIT_COMPAT:
1912 {
1913 struct timeval50 *tvp50
1914 = (struct timeval50 *)ap->a_data;
1915 timeval50_to_timeval(tvp50, &tv);
1916 tvp = &tv;
1917 }
1918 goto segwait_common;
1919 case LFCNSEGWAITALL:
1920 fsidp = NULL;
1921 /* FALLTHROUGH */
1922 case LFCNSEGWAIT:
1923 tvp = (struct timeval *)ap->a_data;
1924segwait_common:
1925 mutex_enter(&lfs_lock);
1926 ++fs->lfs_sleepers;
1927 mutex_exit(&lfs_lock);
1928
1929 error = lfs_segwait(fsidp, tvp);
1930
1931 mutex_enter(&lfs_lock);
1932 if (--fs->lfs_sleepers == 0)
1933 cv_broadcast(&fs->lfs_sleeperscv);
1934 mutex_exit(&lfs_lock);
1935 return error;
1936
1937 case LFCNBMAPV_COMPAT_70:
1938 case LFCNMARKV_COMPAT_70:
1939 blkvp70 = *(struct lfs_fcntl_markv_70 *)ap->a_data;
1940
1941 blkcnt = blkvp70.blkcnt;
1942 if ((u_int) blkcnt > LFS_MARKV_MAXBLKCNT)
1943 return (EINVAL);
1944 blkiov = lfs_malloc(fs, blkcnt * sizeof(BLOCK_INFO), LFS_NB_BLKIOV);
1945 blkiov70 = lfs_malloc(fs, sizeof(BLOCK_INFO_70), LFS_NB_BLKIOV);
1946 for (i = 0; i < blkcnt; i++) {
1947 error = copyin(&blkvp70.blkiov[i], blkiov70,
1948 sizeof(*blkiov70));
1949 if (error) {
1950 lfs_free(fs, blkiov70, LFS_NB_BLKIOV);
1951 lfs_free(fs, blkiov, LFS_NB_BLKIOV);
1952 return error;
1953 }
1954 block_info_from_70(&blkiov[i], blkiov70);
1955 }
1956
1957 mutex_enter(&lfs_lock);
1958 ++fs->lfs_sleepers;
1959 mutex_exit(&lfs_lock);
1960 if (ap->a_command == LFCNBMAPV)
1961 error = lfs_bmapv(l, fsidp, blkiov, blkcnt);
1962 else /* LFCNMARKV */
1963 error = lfs_markv(l, fsidp, blkiov, blkcnt);
1964 if (error == 0) {
1965 for (i = 0; i < blkcnt; i++) {
1966 block_info_to_70(blkiov70, &blkiov[i]);
1967 error = copyout(blkiov70, &blkvp70.blkiov[i],
1968 sizeof(*blkiov70));
1969 if (error) {
1970 break;
1971 }
1972 }
1973 }
1974 mutex_enter(&lfs_lock);
1975 if (--fs->lfs_sleepers == 0)
1976 cv_broadcast(&fs->lfs_sleeperscv);
1977 mutex_exit(&lfs_lock);
1978 lfs_free(fs, blkiov, LFS_NB_BLKIOV);
1979 return error;
1980
1981 case LFCNBMAPV:
1982 case LFCNMARKV:
1983 blkvp = *(struct lfs_fcntl_markv *)ap->a_data;
1984
1985 blkcnt = blkvp.blkcnt;
1986 if ((u_int) blkcnt > LFS_MARKV_MAXBLKCNT)
1987 return (EINVAL);
1988 blkiov = lfs_malloc(fs, blkcnt * sizeof(BLOCK_INFO), LFS_NB_BLKIOV);
1989 if ((error = copyin(blkvp.blkiov, blkiov,
1990 blkcnt * sizeof(BLOCK_INFO))) != 0) {
1991 lfs_free(fs, blkiov, LFS_NB_BLKIOV);
1992 return error;
1993 }
1994
1995 mutex_enter(&lfs_lock);
1996 ++fs->lfs_sleepers;
1997 mutex_exit(&lfs_lock);
1998 if (ap->a_command == LFCNBMAPV)
1999 error = lfs_bmapv(l, fsidp, blkiov, blkcnt);
2000 else /* LFCNMARKV */
2001 error = lfs_markv(l, fsidp, blkiov, blkcnt);
2002 if (error == 0)
2003 error = copyout(blkiov, blkvp.blkiov,
2004 blkcnt * sizeof(BLOCK_INFO));
2005 mutex_enter(&lfs_lock);
2006 if (--fs->lfs_sleepers == 0)
2007 cv_broadcast(&fs->lfs_sleeperscv);
2008 mutex_exit(&lfs_lock);
2009 lfs_free(fs, blkiov, LFS_NB_BLKIOV);
2010 return error;
2011
2012 case LFCNRECLAIM:
2013 /*
2014 * Flush dirops and write Ifile, allowing empty segments
2015 * to be immediately reclaimed.
2016 */
2017 lfs_writer_enter(fs, "pndirop");
2018 off = lfs_sb_getoffset(fs);
2019 lfs_seglock(fs, SEGM_FORCE_CKP | SEGM_CKP);
2020 lfs_flush_dirops(fs);
2021 LFS_CLEANERINFO(cip, fs, bp);
2022 oclean = lfs_ci_getclean(fs, cip);
2023 LFS_SYNC_CLEANERINFO(cip, fs, bp, 1);
2024 lfs_segwrite(ap->a_vp->v_mount, SEGM_FORCE_CKP);
2025 fs->lfs_sp->seg_flags |= SEGM_PROT;
2026 lfs_segunlock(fs);
2027 lfs_writer_leave(fs);
2028
2029#ifdef DEBUG
2030 LFS_CLEANERINFO(cip, fs, bp);
2031 DLOG((DLOG_CLEAN, "lfs_fcntl: reclaim wrote %" PRId64
2032 " blocks, cleaned %" PRId32 " segments (activesb %d)\n",
2033 lfs_sb_getoffset(fs) - off,
2034 lfs_ci_getclean(fs, cip) - oclean,
2035 fs->lfs_activesb));
2036 LFS_SYNC_CLEANERINFO(cip, fs, bp, 0);
2037#else
2038 __USE(oclean);
2039 __USE(off);
2040#endif
2041
2042 return 0;
2043
2044 case LFCNIFILEFH_COMPAT:
2045 /* Return the filehandle of the Ifile */
2046 if ((error = kauth_authorize_system(l->l_cred,
2047 KAUTH_SYSTEM_FILEHANDLE, 0, NULL, NULL, NULL)) != 0)
2048 return (error);
2049 fhp = (struct fhandle *)ap->a_data;
2050 fhp->fh_fsid = *fsidp;
2051 fh_size = 16; /* former VFS_MAXFIDSIZ */
2052 return lfs_vptofh(fs->lfs_ivnode, &(fhp->fh_fid), &fh_size);
2053
2054 case LFCNIFILEFH_COMPAT2:
2055 case LFCNIFILEFH:
2056 /* Return the filehandle of the Ifile */
2057 fhp = (struct fhandle *)ap->a_data;
2058 fhp->fh_fsid = *fsidp;
2059 fh_size = sizeof(struct lfs_fhandle) -
2060 offsetof(fhandle_t, fh_fid);
2061 return lfs_vptofh(fs->lfs_ivnode, &(fhp->fh_fid), &fh_size);
2062
2063 case LFCNREWIND:
2064 /* Move lfs_offset to the lowest-numbered segment */
2065 return lfs_rewind(fs, *(int *)ap->a_data);
2066
2067 case LFCNINVAL:
2068 /* Mark a segment SEGUSE_INVAL */
2069 LFS_SEGENTRY(sup, fs, *(int *)ap->a_data, bp);
2070 if (sup->su_nbytes > 0) {
2071 brelse(bp, 0);
2072 lfs_unset_inval_all(fs);
2073 return EBUSY;
2074 }
2075 sup->su_flags |= SEGUSE_INVAL;
2076 VOP_BWRITE(bp->b_vp, bp);
2077 return 0;
2078
2079 case LFCNRESIZE:
2080 /* Resize the filesystem */
2081 return lfs_resize_fs(fs, *(int *)ap->a_data);
2082
2083 case LFCNWRAPSTOP:
2084 case LFCNWRAPSTOP_COMPAT:
2085 /*
2086 * Hold lfs_newseg at segment 0; if requested, sleep until
2087 * the filesystem wraps around. To support external agents
2088 * (dump, fsck-based regression test) that need to look at
2089 * a snapshot of the filesystem, without necessarily
2090 * requiring that all fs activity stops.
2091 */
2092 if (fs->lfs_stoplwp == curlwp)
2093 return EALREADY;
2094
2095 mutex_enter(&lfs_lock);
2096 while (fs->lfs_stoplwp != NULL)
2097 cv_wait(&fs->lfs_stopcv, &lfs_lock);
2098 fs->lfs_stoplwp = curlwp;
2099 if (fs->lfs_nowrap == 0)
2100 log(LOG_NOTICE, "%s: disabled log wrap\n",
2101 lfs_sb_getfsmnt(fs));
2102 ++fs->lfs_nowrap;
2103 if (*(int *)ap->a_data == 1
2104 || ap->a_command == LFCNWRAPSTOP_COMPAT) {
2105 log(LOG_NOTICE, "LFCNSTOPWRAP waiting for log wrap\n");
2106 error = mtsleep(&fs->lfs_nowrap, PCATCH | PUSER,
2107 "segwrap", 0, &lfs_lock);
2108 log(LOG_NOTICE, "LFCNSTOPWRAP done waiting\n");
2109 if (error) {
2110 lfs_wrapgo(fs, VTOI(ap->a_vp), 0);
2111 }
2112 }
2113 mutex_exit(&lfs_lock);
2114 return 0;
2115
2116 case LFCNWRAPGO:
2117 case LFCNWRAPGO_COMPAT:
2118 /*
2119 * Having done its work, the agent wakes up the writer.
2120 * If the argument is 1, it sleeps until a new segment
2121 * is selected.
2122 */
2123 mutex_enter(&lfs_lock);
2124 error = lfs_wrapgo(fs, VTOI(ap->a_vp),
2125 ap->a_command == LFCNWRAPGO_COMPAT ? 1 :
2126 *((int *)ap->a_data));
2127 mutex_exit(&lfs_lock);
2128 return error;
2129
2130 case LFCNWRAPPASS:
2131 if ((VTOI(ap->a_vp)->i_lfs_iflags & LFSI_WRAPWAIT))
2132 return EALREADY;
2133 mutex_enter(&lfs_lock);
2134 if (fs->lfs_stoplwp != curlwp) {
2135 mutex_exit(&lfs_lock);
2136 return EALREADY;
2137 }
2138 if (fs->lfs_nowrap == 0) {
2139 mutex_exit(&lfs_lock);
2140 return EBUSY;
2141 }
2142 fs->lfs_wrappass = 1;
2143 wakeup(&fs->lfs_wrappass);
2144 /* Wait for the log to wrap, if asked */
2145 if (*(int *)ap->a_data) {
2146 vref(ap->a_vp);
2147 VTOI(ap->a_vp)->i_lfs_iflags |= LFSI_WRAPWAIT;
2148 log(LOG_NOTICE, "LFCNPASS waiting for log wrap\n");
2149 error = mtsleep(&fs->lfs_nowrap, PCATCH | PUSER,
2150 "segwrap", 0, &lfs_lock);
2151 log(LOG_NOTICE, "LFCNPASS done waiting\n");
2152 VTOI(ap->a_vp)->i_lfs_iflags &= ~LFSI_WRAPWAIT;
2153 vrele(ap->a_vp);
2154 }
2155 mutex_exit(&lfs_lock);
2156 return error;
2157
2158 case LFCNWRAPSTATUS:
2159 mutex_enter(&lfs_lock);
2160 *(int *)ap->a_data = fs->lfs_wrapstatus;
2161 mutex_exit(&lfs_lock);
2162 return 0;
2163
2164 default:
2165 return ulfs_fcntl(v);
2166 }
2167 return 0;
2168}

2171 * Return the last logical file offset that should be written for this file
2172 * if we're doing a write that ends at "size". If writing, we need to know
