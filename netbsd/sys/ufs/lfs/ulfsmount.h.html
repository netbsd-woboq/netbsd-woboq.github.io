58/* This structure describes the ULFS specific mount structure data. */
59struct ulfsmount {
60 /* Abstract vfs-level filesystem structure. */
61 struct mount *um_mountp;
62
63 /* type of fs; currently always ULFS1, theoretically also ULFS2 */
64 u_long um_fstype;
65
66 /* pointer to the filesystem-specific filesystem structure */
67 struct lfs *um_lfs;
68
69 /* Extended attribute information. */
70 struct ulfs_extattr_per_mount um_extattr;
71
72 /* Quota-related material. */
73 struct vnode *um_quotas[ULFS_MAXQUOTAS]; /* quota files */
74 kauth_cred_t um_cred[ULFS_MAXQUOTAS]; /* quota file access cred */
75 union {
76 struct um_q1 {
77 time_t q1_btime[ULFS_MAXQUOTAS]; /* block quota time limit */
78 time_t q1_itime[ULFS_MAXQUOTAS]; /* inode quota time limit */
79 char q1_qflags[ULFS_MAXQUOTAS]; /* quota flags */
80 } um_q1;
81 struct um_q2 {
82 uint64_t q2_bsize; /* block size of quota file */
83 uint64_t q2_bmask; /* mask for above */
84 } um_q2;
85 } um_q;
86#define umq1_btime um_q.um_q1.q1_btime
87#define umq1_itime um_q.um_q1.q1_itime
88#define umq1_qflags um_q.um_q1.q1_qflags
89#define umq2_bsize um_q.um_q2.q2_bsize
90#define umq2_bmask um_q.um_q2.q2_bmask
91};
92
93/* ULFS-specific flags for um_flags */
94#define ULFS_NEEDSWAP 0x01 /* filesystem metadata need byte-swapping */
95/* unused 0x02 */
96#define ULFS_QUOTA 0x04 /* filesystem has QUOTA (v1) */
97#define ULFS_QUOTA2 0x08 /* filesystem has QUOTA2 */
98
99/*
100 * Filesystem types
101 */
102#define ULFS1 1
103#define ULFS2 2
104
105
106/*
107 * Flags describing the state of quotas.
108 */
109#define QTF_OPENING 0x01 /* Q_QUOTAON in progress */
110#define QTF_CLOSING 0x02 /* Q_QUOTAOFF in progress */
111
112/* Convert mount ptr to ulfsmount ptr. */
113#define VFSTOULFS(mp) ((struct ulfsmount *)((mp)->mnt_data))
114
115/*
116 * Macros to access file system parameters (no longer) in the
117 * ulfsmount structure.
118 * Used by ulfs_bmap.
119 */
120#define MNINDIR(lfs) ((lfs)->um_nindir)
121#define blkptrtodb(lfs, b) ((b) << (lfs)->um_bptrtodb)
122
123/*
124 * Predicate for byte-swapping support.
125 */
126#define FSFMT(vp) (((vp)->v_mount->mnt_iflag & IMNT_DTYPE) == 0)
