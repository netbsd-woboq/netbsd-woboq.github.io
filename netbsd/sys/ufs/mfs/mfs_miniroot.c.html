37#include <sys/param.h>

39#include <ufs/mfs/mfs_extern.h>
40#include <ufs/ffs/fs.h>
41
42void * mfs_rootbase; /* address of mini-root in kernel virtual memory */
43u_long mfs_rootsize; /* size of mini-root in bytes */
44
45/*
46 * This is called early in boot to set the base address and size
47 * of the mini-root.
48 */
49int
50mfs_initminiroot(void *base)
51{
52 struct fs *fs = (struct fs *)((char *)base + SBLOCK_UFS1);
53 static bool inited = false;
54
55 if (inited)
56 panic("mfs_initminiroot() called more than once");
57 inited = true;
58
59 /* check for valid super block */
60 if (fs->fs_magic != FS_UFS1_MAGIC || fs->fs_bsize > MAXBSIZE ||
61 fs->fs_bsize < sizeof(struct fs))
62 return (0);
63 rootfstype = MOUNT_MFS;
64 mfs_rootbase = base;
65 mfs_rootsize = fs->fs_fsize * fs->fs_size;
66 rootdev = makedev(255, 0);
67 return (mfs_rootsize);
68}
