58static bool
59ufs_issequential(const struct ufsmount *ump, daddr_t daddr0, daddr_t daddr1)
60{
61
62 /* for ufs, blocks in a hole is not 'contiguous'. */
63 if (daddr0 == 0)
64 return false;
65
66 return (daddr0 + ump->um_seqinc == daddr1);
67}

69/*
70 * Bmap converts the logical block number of a file to its physical block
71 * number on the disk. The conversion is done by using the logical block
72 * number to index into the array of block pointers described by the dinode.
73 */
74int
75ufs_bmap(void *v)
76{
77 struct vop_bmap_args /* {
78 struct vnode *a_vp;
79 daddr_t a_bn;
80 struct vnode **a_vpp;
81 daddr_t *a_bnp;
82 int *a_runp;
83 } */ *ap = v;
84 int error;
85
86 /*
87 * Check for underlying vnode requests and ensure that logical
88 * to physical mapping is requested.
89 */
90 if (ap->a_vpp != NULL)
91 *ap->a_vpp = VTOI(ap->a_vp)->i_devvp;
92 if (ap->a_bnp == NULL)
93 return (0);
94
95 error = ufs_bmaparray(ap->a_vp, ap->a_bn, ap->a_bnp, NULL, NULL,
96 ap->a_runp, ufs_issequential);
97 return error;
98}

100/*
101 * Indirect blocks are now on the vnode for the file. They are given negative
102 * logical block numbers. Indirect blocks are addressed by the negative
103 * address of the first data block to which they point. Double indirect blocks
104 * are addressed by one less than the address of the first indirect block to
105 * which they point. Triple indirect blocks are addressed by one less than
106 * the address of the first double indirect block to which they point.
107 *
108 * ufs_bmaparray does the bmap conversion, and if requested returns the
109 * array of logical blocks which must be traversed to get to a block.
110 * Each entry contains the offset into that block that gets you to the
111 * next block and the disk address of the block (if it is assigned).
112 */
113
114int
115ufs_bmaparray(struct vnode *vp, daddr_t bn, daddr_t *bnp, struct indir *ap,
116 int *nump, int *runp, ufs_issequential_callback_t is_sequential)
117{
118 struct inode *ip;
119 struct buf *bp, *cbp;
120 struct ufsmount *ump;
121 struct mount *mp;
122 struct indir a[UFS_NIADDR + 1], *xap;
123 daddr_t daddr;
124 daddr_t metalbn;
125 int error, maxrun = 0, num;
126
127 ip = VTOI(vp);
128 mp = vp->v_mount;
129 ump = ip->i_ump;
130 KASSERTMSG(((ap == NULL) == (nump == NULL)),
131 "ufs_bmaparray: invalid arguments: ap = %p, nump = %p", ap, nump);
132
133 if (runp) {
134 /*
135 * XXX
136 * If MAXBSIZE is the largest transfer the disks can handle,
137 * we probably want maxrun to be 1 block less so that we
138 * don't create a block larger than the device can handle.
139 */
140 *runp = 0;
141 maxrun = MAXPHYS / mp->mnt_stat.f_iosize - 1;
142 }
143
144 if (bn >= 0 && bn < UFS_NDADDR) {
145 if (nump != NULL)
146 *nump = 0;
147 if (ump->um_fstype == UFS1)
148 daddr = ufs_rw32(ip->i_ffs1_db[bn],
149 UFS_MPNEEDSWAP(ump));
150 else
151 daddr = ufs_rw64(ip->i_ffs2_db[bn],
152 UFS_MPNEEDSWAP(ump));
153 *bnp = blkptrtodb(ump, daddr);
154 /*
155 * Since this is FFS independent code, we are out of
156 * scope for the definitions of BLK_NOCOPY and
157 * BLK_SNAP, but we do know that they will fall in
158 * the range 1..um_seqinc, so we use that test and
159 * return a request for a zeroed out buffer if attempts
160 * are made to read a BLK_NOCOPY or BLK_SNAP block.
161 */
162 if ((ip->i_flags & (SF_SNAPSHOT | SF_SNAPINVAL)) == SF_SNAPSHOT
163 && daddr > 0 &&
164 daddr < ump->um_seqinc) {
165 *bnp = -1;
166 } else if (*bnp == 0) {
167 if ((ip->i_flags & (SF_SNAPSHOT | SF_SNAPINVAL))
168 == SF_SNAPSHOT) {
169 *bnp = blkptrtodb(ump, bn * ump->um_seqinc);
170 } else {
171 *bnp = -1;
172 }
173 } else if (runp) {
174 if (ump->um_fstype == UFS1) {
175 for (++bn; bn < UFS_NDADDR && *runp < maxrun &&
176 is_sequential(ump,
177 ufs_rw32(ip->i_ffs1_db[bn - 1],
178 UFS_MPNEEDSWAP(ump)),
179 ufs_rw32(ip->i_ffs1_db[bn],
180 UFS_MPNEEDSWAP(ump)));
181 ++bn, ++*runp);
182 } else {
183 for (++bn; bn < UFS_NDADDR && *runp < maxrun &&
184 is_sequential(ump,
185 ufs_rw64(ip->i_ffs2_db[bn - 1],
186 UFS_MPNEEDSWAP(ump)),
187 ufs_rw64(ip->i_ffs2_db[bn],
188 UFS_MPNEEDSWAP(ump)));
189 ++bn, ++*runp);
190 }
191 }
192 return (0);
193 }
194
195 xap = ap == NULL ? a : ap;
196 if (!nump)
197 nump = &num;
198 if ((error = ufs_getlbns(vp, bn, xap, nump)) != 0)
199 return (error);
200
201 num = *nump;
202
203 /* Get disk address out of indirect block array */
204 if (ump->um_fstype == UFS1)
205 daddr = ufs_rw32(ip->i_ffs1_ib[xap->in_off],
206 UFS_MPNEEDSWAP(ump));
207 else
208 daddr = ufs_rw64(ip->i_ffs2_ib[xap->in_off],
209 UFS_MPNEEDSWAP(ump));
210
211 for (bp = NULL, ++xap; --num; ++xap) {
212 /*
213 * Exit the loop if there is no disk address assigned yet and
214 * the indirect block isn't in the cache, or if we were
215 * looking for an indirect block and we've found it.
216 */
217
218 metalbn = xap->in_lbn;
219 if (metalbn == bn)
220 break;
221 if (daddr == 0) {
222 mutex_enter(&bufcache_lock);
223 cbp = incore(vp, metalbn);
224 mutex_exit(&bufcache_lock);
225 if (cbp == NULL)
226 break;
227 }
228
229 /*
230 * If we get here, we've either got the block in the cache
231 * or we have a disk address for it, go fetch it.
232 */
233 if (bp)
234 brelse(bp, 0);
235
236 xap->in_exists = 1;
237 bp = getblk(vp, metalbn, mp->mnt_stat.f_iosize, 0, 0);
238 if (bp == NULL) {
239
240 /*
241 * getblk() above returns NULL only iff we are
242 * pagedaemon. See the implementation of getblk
243 * for detail.
244 */
245
246 return (ENOMEM);
247 }
248 if (bp->b_oflags & (BO_DONE | BO_DELWRI)) {
249 trace(TR_BREADHIT, pack(vp, size), metalbn);
250 } else {
251 KASSERTMSG((daddr != 0),
252 "ufs_bmaparray: indirect block not in cache");
253 trace(TR_BREADMISS, pack(vp, size), metalbn);
254 bp->b_blkno = blkptrtodb(ump, daddr);
255 bp->b_flags |= B_READ;
256 BIO_SETPRIO(bp, BPRIO_TIMECRITICAL);
257 VOP_STRATEGY(vp, bp);
258 curlwp->l_ru.ru_inblock++; /* XXX */
259 if ((error = biowait(bp)) != 0) {
260 brelse(bp, 0);
261 return (error);
262 }
263 }
264 if (ump->um_fstype == UFS1) {
265 daddr = ufs_rw32(((u_int32_t *)bp->b_data)[xap->in_off],
266 UFS_MPNEEDSWAP(ump));
267 if (num == 1 && daddr && runp) {
268 for (bn = xap->in_off + 1;
269 bn < MNINDIR(ump) && *runp < maxrun &&
270 is_sequential(ump,
271 ufs_rw32(((int32_t *)bp->b_data)[bn-1],
272 UFS_MPNEEDSWAP(ump)),
273 ufs_rw32(((int32_t *)bp->b_data)[bn],
274 UFS_MPNEEDSWAP(ump)));
275 ++bn, ++*runp);
276 }
277 } else {
278 daddr = ufs_rw64(((u_int64_t *)bp->b_data)[xap->in_off],
279 UFS_MPNEEDSWAP(ump));
280 if (num == 1 && daddr && runp) {
281 for (bn = xap->in_off + 1;
282 bn < MNINDIR(ump) && *runp < maxrun &&
283 is_sequential(ump,
284 ufs_rw64(((int64_t *)bp->b_data)[bn-1],
285 UFS_MPNEEDSWAP(ump)),
286 ufs_rw64(((int64_t *)bp->b_data)[bn],
287 UFS_MPNEEDSWAP(ump)));
288 ++bn, ++*runp);
289 }
290 }
291 }
292 if (bp)
293 brelse(bp, 0);
294
295 /*
296 * Since this is FFS independent code, we are out of scope for the
297 * definitions of BLK_NOCOPY and BLK_SNAP, but we do know that they
298 * will fall in the range 1..um_seqinc, so we use that test and
299 * return a request for a zeroed out buffer if attempts are made
300 * to read a BLK_NOCOPY or BLK_SNAP block.
301 */
302 if ((ip->i_flags & (SF_SNAPSHOT | SF_SNAPINVAL)) == SF_SNAPSHOT
303 && daddr > 0 && daddr < ump->um_seqinc) {
304 *bnp = -1;
305 return (0);
306 }
307 *bnp = blkptrtodb(ump, daddr);
308 if (*bnp == 0) {
309 if ((ip->i_flags & (SF_SNAPSHOT | SF_SNAPINVAL))
310 == SF_SNAPSHOT) {
311 *bnp = blkptrtodb(ump, bn * ump->um_seqinc);
312 } else {
313 *bnp = -1;
314 }
315 }
316 return (0);
317}

319/*
320 * Create an array of logical block number/offset pairs which represent the
321 * path of indirect blocks required to access a data block. The first "pair"
322 * contains the logical block number of the appropriate single, double or
323 * triple indirect block and the offset into the inode indirect block array.
324 * Note, the logical block number of the inode single/double/triple indirect
325 * block appears twice in the array, once with the offset into the i_ffs1_ib and
326 * once with the offset into the page itself.
327 */
328int
329ufs_getlbns(struct vnode *vp, daddr_t bn, struct indir *ap, int *nump)
330{
331 daddr_t metalbn, realbn;
332 struct ufsmount *ump;
333 int64_t blockcnt;
334 int lbc;
335 int i, numlevels, off;
336
337 ump = VFSTOUFS(vp->v_mount);
338 if (nump)
339 *nump = 0;
340 numlevels = 0;
341 realbn = bn;
342 if (bn < 0)
343 bn = -bn;
344 KASSERT(bn >= UFS_NDADDR);
345
346 /*
347 * Determine the number of levels of indirection. After this loop
348 * is done, blockcnt indicates the number of data blocks possible
349 * at the given level of indirection, and UFS_NIADDR - i is the number
350 * of levels of indirection needed to locate the requested block.
351 */
352
353 bn -= UFS_NDADDR;
354 for (lbc = 0, i = UFS_NIADDR;; i--, bn -= blockcnt) {
355 if (i == 0)
356 return (EFBIG);
357
358 lbc += ump->um_lognindir;
359 blockcnt = (int64_t)1 << lbc;
360
361 if (bn < blockcnt)
362 break;
363 }
364
365 /* Calculate the address of the first meta-block. */
366 metalbn = -((realbn >= 0 ? realbn : -realbn) - bn + UFS_NIADDR - i);
367
368 /*
369 * At each iteration, off is the offset into the bap array which is
370 * an array of disk addresses at the current level of indirection.
371 * The logical block number and the offset in that block are stored
372 * into the argument array.
373 */
374 ap->in_lbn = metalbn;
375 ap->in_off = off = UFS_NIADDR - i;
376 ap->in_exists = 0;
377 ap++;
378 for (++numlevels; i <= UFS_NIADDR; i++) {
379 /* If searching for a meta-data block, quit when found. */
380 if (metalbn == realbn)
381 break;
382
383 lbc -= ump->um_lognindir;
384 off = (bn >> lbc) & (MNINDIR(ump) - 1);
385
386 ++numlevels;
387 ap->in_lbn = metalbn;
388 ap->in_off = off;
389 ap->in_exists = 0;
390 ++ap;
391
392 metalbn -= -1 + ((int64_t)off << lbc);
393 }
394 if (nump)
395 *nump = numlevels;
396 return (0);
397}
