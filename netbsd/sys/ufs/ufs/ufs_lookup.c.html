87static void
88ufs_dirswap(struct direct *dirp)
89{
90 uint8_t tmp = dirp->d_namlen;
91 dirp->d_namlen = dirp->d_type;
92 dirp->d_type = tmp;
93}

95struct slotinfo {
96 enum {
97 NONE, /* need to search a slot for our new entry */
98 COMPACT, /* a compaction can make a slot in the current
99 DIRBLKSIZ block */
100 FOUND, /* found a slot (or no need to search) */
101 } status;
102 doff_t offset; /* offset of area with free space.
103 a special value -1 for invalid */
104 int size; /* size of area at slotoffset */
105 int freespace; /* accumulated amount of space free in
106 the current DIRBLKSIZ block */
107 int needed; /* size of the entry we're seeking */
108};
109
110static void
111calc_count(struct ufs_lookup_results *results, int dirblksiz, doff_t prevoff)
112{
113 if ((results->ulr_offset & (dirblksiz - 1)) == 0)
114 results->ulr_count = 0;
115 else
116 results->ulr_count = results->ulr_offset - prevoff;
117}
118
119static void
120slot_init(struct slotinfo *slot)
121{
122 slot->status = FOUND;
123 slot->offset = -1;
124 slot->freespace = slot->size = slot->needed = 0;
125}
126
127#ifdef UFS_DIRHASH
128static doff_t
129slot_findfree(struct slotinfo *slot, struct inode *dp)
130{
131 if (slot->status == FOUND)
132 return dp->i_size;
133
134 slot->offset = ufsdirhash_findfree(dp, slot->needed, &slot->size);
135 if (slot->offset < 0)
136 return dp->i_size;
137
138 slot->status = COMPACT;
139 doff_t enduseful = ufsdirhash_enduseful(dp);
140 if (enduseful < 0)
141 return dp->i_size;
142 return enduseful;
143}
144#endif

146static void
147slot_white(struct slotinfo *slot, uint16_t reclen,
148 struct ufs_lookup_results *results)
149{
150 slot->status = FOUND;
151 slot->offset = results->ulr_offset;
152 slot->size = reclen;
153 results->ulr_reclen = slot->size;
154}
155
156static void
157slot_update(struct slotinfo *slot, int size, uint16_t reclen, doff_t offset)
158{
159 if (size >= slot->needed) {
160 slot->status = FOUND;
161 slot->offset = offset;
162 slot->size = reclen;
163 } else if (slot->status == NONE) {
164 slot->freespace += size;
165 if (slot->offset == -1)
166 slot->offset = offset;
167 if (slot->freespace >= slot->needed) {
168 slot->status = COMPACT;
169 slot->size = offset + reclen - slot->offset;
170 }
171 }
172}

174/*
175 * Return an indication of where the new directory entry should be put.
176 * If we didn't find a slot, then set results->ulr_count to 0 indicating
177 * that the new slot belongs at the end of the directory. If we found a slot,
178 * then the new entry can be put in the range from results->ulr_offset to
179 * results->ulr_offset + results->ulr_count.
180 */
181static int
182slot_estimate(const struct slotinfo *slot, int dirblksiz, int nameiop,
183 doff_t prevoff, doff_t enduseful, const struct inode *ip,
184 struct ufs_lookup_results *results)
185{
186 if (slot->status == NONE) {
187 results->ulr_offset = roundup(ip->i_size, dirblksiz);
188 results->ulr_count = 0;
189 enduseful = results->ulr_offset;
190 } else if (nameiop == DELETE) {
191 results->ulr_offset = slot->offset;
192 calc_count(results, dirblksiz, prevoff);
193 } else {
194 results->ulr_offset = slot->offset;
195 results->ulr_count = slot->size;
196 if (enduseful < slot->offset + slot->size)
197 enduseful = slot->offset + slot->size;
198 }
199 results->ulr_endoff = roundup(enduseful, dirblksiz);
200#if 0 /* commented out by dbj. none of the on disk fields changed */
201 ip->i_flag |= IN_CHANGE | IN_UPDATE;
202#endif
203 return EJUSTRETURN;
204}

206/*
207 * Check if we can delete inode tdp in directory vdp with inode ip and creds.
208 */
209static int
210ufs_can_delete(struct vnode *tdp, struct vnode *vdp, struct inode *ip,
211 kauth_cred_t cred)
212{
213 int error;
214 /*
215 * Write access to directory required to delete files.
216 */
217 error = VOP_ACCESS(vdp, VWRITE, cred);
218 if (error)
219 goto out;
220
221 if (!(ip->i_mode & ISVTX))
222 return 0;
223
224 /*
225 * If directory is "sticky", then user must own
226 * the directory, or the file in it, else she
227 * may not delete it (unless she's root). This
228 * implements append-only directories.
229 */
230 error = kauth_authorize_vnode(cred, KAUTH_VNODE_DELETE, tdp, vdp,
231 genfs_can_sticky(cred, ip->i_uid, VTOI(tdp)->i_uid));
232 if (error) {
233 error = EPERM; // Why override?
234 goto out;
235 }
236 return 0;
237out:
238 vrele(tdp);
239 return error;
240}

242static int
243ufs_getino(struct vnode *vdp, struct inode *ip, ino_t foundino,
244 struct vnode **tdp, bool same)
245{
246 if (ip->i_number == foundino) {
247 if (same)
248 return EISDIR;
249 vref(vdp);
250 *tdp = vdp;
251 return 0;
252 }
253 return vcache_get(vdp->v_mount, &foundino, sizeof(foundino), tdp);
254}

257/*
258 * Convert a component of a pathname into a pointer to a locked inode.
259 * This is a very central and rather complicated routine.
260 * If the file system is not maintained in a strict tree hierarchy,
261 * this can result in a deadlock situation (see comments in code below).
262 *
263 * The cnp->cn_nameiop argument is LOOKUP, CREATE, RENAME, or DELETE depending
264 * on whether the name is to be looked up, created, renamed, or deleted.
265 * When CREATE, RENAME, or DELETE is specified, information usable in
266 * creating, renaming, or deleting a directory entry may be calculated.
267 * If flag has LOCKPARENT or'ed into it and the target of the pathname
268 * exists, lookup returns both the target and its parent directory locked.
269 * When creating or renaming and LOCKPARENT is specified, the target may
270 * not be ".". When deleting and LOCKPARENT is specified, the target may
271 * be "."., but the caller must check to ensure it does an vrele and vput
272 * instead of two vputs.
273 *
274 * Overall outline of ufs_lookup:
275 *
276 * check accessibility of directory
277 * look for name in cache, if found, then if at end of path
278 * and deleting or creating, drop it, else return name
279 * search for name in directory, to found or notfound
280 * notfound:
281 * if creating, return locked directory, leaving info on available slots
282 * else return error
283 * found:
284 * if at end of path and deleting, return information to allow delete
285 * if at end of path and rewriting (RENAME and LOCKPARENT), lock target
286 * inode and return info to allow rewrite
287 * if not at end, add name to cache; if at end and neither creating
288 * nor deleting, add name to cache
289 */
290int
291ufs_lookup(void *v)
292{
293 struct vop_lookup_v2_args /* {
294 struct vnode *a_dvp;
295 struct vnode **a_vpp;
296 struct componentname *a_cnp;
297 } */ *ap = v;
298 struct vnode *vdp = ap->a_dvp; /* vnode for directory being searched */
299 struct inode *dp = VTOI(vdp); /* inode for directory being searched */
300 struct buf *bp; /* a buffer of directory entries */
301 struct direct *ep; /* the current directory entry */
302 int entryoffsetinblock; /* offset of ep in bp's buffer */
303 struct slotinfo slot;
304 int numdirpasses; /* strategy for directory search */
305 doff_t endsearch; /* offset to end directory search */
306 doff_t prevoff; /* previous value of ulr_offset */
307 struct vnode *tdp; /* returned by vcache_get */
308 doff_t enduseful; /* pointer past last used dir slot.
309 used for directory truncation. */
310 u_long bmask; /* block offset mask */
311 int error;
312 struct vnode **vpp = ap->a_vpp;
313 struct componentname *cnp = ap->a_cnp;
314 kauth_cred_t cred = cnp->cn_cred;
315 int flags;
316 int nameiop = cnp->cn_nameiop;
317 struct ufsmount *ump = dp->i_ump;
318 const int needswap = UFS_MPNEEDSWAP(ump);
319 int dirblksiz = ump->um_dirblksiz;
320 ino_t foundino;
321 struct ufs_lookup_results *results;
322 int iswhiteout; /* temp result from cache_lookup() */
323 const int fsfmt = FSFMT(vdp);
324 uint16_t reclen;
325
326 flags = cnp->cn_flags;
327
328 bp = NULL;
329 *vpp = NULL;
330 endsearch = 0; /* silence compiler warning */
331
332 /*
333 * Produce the auxiliary lookup results into i_crap. Increment
334 * its serial number so elsewhere we can tell if we're using
335 * stale results. This should not be done this way. XXX.
336 */
337 results = &dp->i_crap;
338 dp->i_crapcounter++;
339
340 /*
341 * Check accessiblity of directory.
342 */
343 if ((error = VOP_ACCESS(vdp, VEXEC, cred)) != 0)
344 return (error);
345
346 if ((flags & ISLASTCN) && (vdp->v_mount->mnt_flag & MNT_RDONLY) &&
347 (nameiop == DELETE || nameiop == RENAME))
348 return (EROFS);
349
350 /*
351 * We now have a segment name to search for, and a directory to search.
352 *
353 * Before tediously performing a linear scan of the directory,
354 * check the name cache to see if the directory/name pair
355 * we are looking for is known already.
356 */
357 if (cache_lookup(vdp, cnp->cn_nameptr, cnp->cn_namelen,
358 cnp->cn_nameiop, cnp->cn_flags, &iswhiteout, vpp)) {
359 if (iswhiteout) {
360 cnp->cn_flags |= ISWHITEOUT;
361 }
362 return *vpp == NULLVP ? ENOENT : 0;
363 }
364 if (iswhiteout) {
365 /*
366 * The namecache set iswhiteout without finding a
367 * cache entry. As of this writing (20121014), this
368 * can happen if there was a whiteout entry that has
369 * been invalidated by the lookup. It is not clear if
370 * it is correct to set ISWHITEOUT in this case or
371 * not; however, doing so retains the prior behavior,
372 * so we'll go with that until some clearer answer
373 * appears. XXX
374 */
375 cnp->cn_flags |= ISWHITEOUT;
376 }
377
378 /*
379 * Suppress search for slots unless creating
380 * file and at end of pathname, in which case
381 * we watch for a place to put the new file in
382 * case it doesn't already exist.
383 */
384 slot_init(&slot);
385
386 if ((nameiop == CREATE || nameiop == RENAME) && (flags & ISLASTCN)) {
387 slot.status = NONE;
388 slot.needed = UFS_DIRECTSIZ(cnp->cn_namelen);
389 }
390
391 /*
392 * If there is cached information on a previous search of
393 * this directory, pick up where we last left off.
394 * We cache only lookups as these are the most common
395 * and have the greatest payoff. Caching CREATE has little
396 * benefit as it usually must search the entire directory
397 * to determine that the entry does not exist. Caching the
398 * location of the last DELETE or RENAME has not reduced
399 * profiling time and hence has been removed in the interest
400 * of simplicity.
401 */
402 bmask = vdp->v_mount->mnt_stat.f_iosize - 1;
403
404#ifdef UFS_DIRHASH
405 /*
406 * Use dirhash for fast operations on large directories. The logic
407 * to determine whether to hash the directory is contained within
408 * ufsdirhash_build(); a zero return means that it decided to hash
409 * this directory and it successfully built up the hash table.
410 */
411 if (ufsdirhash_build(dp) == 0) {
412 /* Look for a free slot if needed. */
413 enduseful = slot_findfree(&slot, dp);
414 /* Look up the component. */
415 numdirpasses = 1;
416 entryoffsetinblock = 0; /* silence compiler warning */
417 switch (ufsdirhash_lookup(dp, cnp->cn_nameptr, cnp->cn_namelen,
418 &results->ulr_offset, &bp,
419 nameiop == DELETE ? &prevoff : NULL)) {
420 case 0:
421 ep = (void *)((char *)bp->b_data +
422 (results->ulr_offset & bmask));
423 reclen = ufs_rw16(ep->d_reclen, needswap);
424 goto foundentry;
425 case ENOENT:
426 results->ulr_offset = roundup(dp->i_size, dirblksiz);
427 goto notfound;
428 default:
429 /* Something failed; just do a linear search. */
430 break;
431 }
432 }
433#endif /* UFS_DIRHASH */
434
435 if (nameiop != LOOKUP || results->ulr_diroff == 0 ||
436 results->ulr_diroff >= dp->i_size) {
437 entryoffsetinblock = 0;
438 results->ulr_offset = 0;
439 numdirpasses = 1;
440 } else {
441 results->ulr_offset = results->ulr_diroff;
442 entryoffsetinblock = results->ulr_offset & bmask;
443 if (entryoffsetinblock != 0 &&
444 (error = ufs_blkatoff(vdp, (off_t)results->ulr_offset,
445 NULL, &bp, false)))
446 goto out;
447 numdirpasses = 2;
448 namecache_count_2passes();
449 }
450 prevoff = results->ulr_offset;
451 endsearch = roundup(dp->i_size, dirblksiz);
452 enduseful = 0;
453
454searchloop:
455 while (results->ulr_offset < endsearch) {
456 if (curcpu()->ci_schedstate.spc_flags & SPCF_SHOULDYIELD)
457 preempt();
458 /*
459 * If necessary, get the next directory block.
460 */
461 if ((results->ulr_offset & bmask) == 0) {
462 if (bp != NULL)
463 brelse(bp, 0);
464 error = ufs_blkatoff(vdp, (off_t)results->ulr_offset,
465 NULL, &bp, false);
466 if (error)
467 goto out;
468 entryoffsetinblock = 0;
469 }
470 /*
471 * If still looking for a slot, and at a DIRBLKSIZ
472 * boundary, have to start looking for free space again.
473 */
474 if (slot.status == NONE &&
475 (entryoffsetinblock & (dirblksiz - 1)) == 0) {
476 slot.offset = -1;
477 slot.freespace = 0;
478 }
479 /*
480 * Get pointer to next entry.
481 * Full validation checks are slow, so we only check
482 * enough to insure forward progress through the
483 * directory. Complete checks can be run by patching
484 * "dirchk" to be true.
485 */
486 KASSERT(bp != NULL);
487 ep = (void *)((char *)bp->b_data + entryoffsetinblock);
488 const char *msg;
489 reclen = ufs_rw16(ep->d_reclen, needswap);
490 if ((reclen == 0 && (msg = "null entry")) || (dirchk &&
491 (msg = ufs_dirbadentry(vdp, ep, entryoffsetinblock)))) {
492 ufs_dirbad(dp, results->ulr_offset, msg);
493 reclen = dirblksiz -
494 (entryoffsetinblock & (dirblksiz - 1));
495 goto next;
496 }
497
498 /*
499 * If an appropriate sized slot has not yet been found,
500 * check to see if one is available. Also accumulate space
501 * in the current block so that we can determine if
502 * compaction is viable.
503 */
504 if (slot.status != FOUND) {
505 int size = reclen;
506 if (ep->d_ino != 0)
507 size -= UFS_DIRSIZ(fsfmt, ep, needswap);
508 if (size > 0)
509 slot_update(&slot, size, reclen,
510 results->ulr_offset);
511 }
512
513 if (ep->d_ino == 0)
514 goto next;
515
516 /*
517 * Check for a name match.
518 */
519 const uint16_t namlen = NAMLEN(fsfmt, needswap, ep);
520 if (namlen != cnp->cn_namelen ||
521 memcmp(cnp->cn_nameptr, ep->d_name, (size_t)namlen))
522 goto next;
523
524#ifdef UFS_DIRHASH
525foundentry:
526#endif
527 /*
528 * Save directory entry's inode number and
529 * reclen, and release directory buffer.
530 */
531 if (!fsfmt && ep->d_type == DT_WHT) {
532 slot_white(&slot, reclen, results);
533 /*
534 * This is used to set results->ulr_endoff, which may
535 * be used by ufs_direnter() as a length to truncate
536 * the directory to. Therefore, it must point past the
537 * end of the last non-empty directory entry. We don't
538 * know where that is in this case, so we effectively
539 * disable shrinking by using the existing size of the
540 * directory.
541 *
542 * Note that we wouldn't expect to shrink the
543 * directory while rewriting an existing entry anyway.
544 */
545 enduseful = endsearch;
546 cnp->cn_flags |= ISWHITEOUT;
547 numdirpasses--;
548 goto notfound;
549 }
550 foundino = ufs_rw32(ep->d_ino, needswap);
551 results->ulr_reclen = reclen;
552 goto found;
553next:
554 prevoff = results->ulr_offset;
555 results->ulr_offset += reclen;
556 entryoffsetinblock += reclen;
557 if (ep->d_ino)
558 enduseful = results->ulr_offset;
559 }
560notfound:
561 /*
562 * If we started in the middle of the directory and failed
563 * to find our target, we must check the beginning as well.
564 */
565 if (numdirpasses == 2) {
566 numdirpasses--;
567 results->ulr_offset = 0;
568 endsearch = results->ulr_diroff;
569 goto searchloop;
570 }
571 if (bp != NULL)
572 brelse(bp, 0);
573 /*
574 * If creating, and at end of pathname and current
575 * directory has not been removed, then can consider
576 * allowing file to be created.
577 */
578 if ((nameiop == CREATE || nameiop == RENAME ||
579 (nameiop == DELETE &&
580 (cnp->cn_flags & DOWHITEOUT) &&
581 (cnp->cn_flags & ISWHITEOUT))) &&
582 (flags & ISLASTCN) && dp->i_nlink != 0) {
583 /*
584 * Access for write is interpreted as allowing
585 * creation of files in the directory.
586 */
587 error = VOP_ACCESS(vdp, VWRITE, cred);
588 if (error)
589 goto out;
590 error = slot_estimate(&slot, dirblksiz, nameiop,
591 prevoff, enduseful, dp, results);
592 /*
593 * We return with the directory locked, so that
594 * the parameters we set up above will still be
595 * valid if we actually decide to do a direnter().
596 * We return ni_vp == NULL to indicate that the entry
597 * does not currently exist; we leave a pointer to
598 * the (locked) directory inode in ndp->ni_dvp.
599 *
600 * NB - if the directory is unlocked, then this
601 * information cannot be used.
602 */
603 goto out;
604 }
605 /*
606 * Insert name into cache (as non-existent) if appropriate.
607 */
608 if (nameiop != CREATE) {
609 cache_enter(vdp, *vpp, cnp->cn_nameptr, cnp->cn_namelen,
610 cnp->cn_flags);
611 }
612 error = ENOENT;
613 goto out;
614
615found:
616 if (numdirpasses == 2)
617 namecache_count_pass2();
618 /*
619 * Check that directory length properly reflects presence
620 * of this entry.
621 */
622 const uint64_t newisize =
623 results->ulr_offset + UFS_DIRSIZ(fsfmt, ep, needswap);
624 if (newisize > dp->i_size) {
625 ufs_dirbad(dp, results->ulr_offset, "i_size too small");
626 dp->i_size = newisize;
627 DIP_ASSIGN(dp, size, dp->i_size);
628 dp->i_flag |= IN_CHANGE | IN_UPDATE;
629 UFS_WAPBL_UPDATE(vdp, NULL, NULL, UPDATE_DIROP);
630 }
631 brelse(bp, 0);
632
633 /*
634 * Found component in pathname.
635 * If the final component of path name, save information
636 * in the cache as to where the entry was found.
637 */
638 if ((flags & ISLASTCN) && nameiop == LOOKUP)
639 results->ulr_diroff = results->ulr_offset & ~(dirblksiz - 1);
640
641 /*
642 * If deleting, and at end of pathname, return
643 * parameters which can be used to remove file.
644 * Lock the inode, being careful with ".".
645 */
646 if (nameiop == DELETE && (flags & ISLASTCN)) {
647 /*
648 * Return pointer to current entry in results->ulr_offset,
649 * and distance past previous entry (if there
650 * is a previous entry in this block) in results->ulr_count.
651 * Save directory inode pointer in ndp->ni_dvp for dirremove().
652 */
653 calc_count(results, dirblksiz, prevoff);
654
655 if ((error = ufs_getino(vdp, dp, foundino, &tdp, false)) != 0)
656 goto out;
657
658 if ((error = ufs_can_delete(tdp, vdp, dp, cred)) != 0)
659 goto out;
660
661 *vpp = tdp;
662 goto out;
663 }
664
665 /*
666 * If rewriting (RENAME), return the inode and the
667 * information required to rewrite the present directory
668 * Must get inode of directory entry to verify it's a
669 * regular file, or empty directory.
670 */
671 if (nameiop == RENAME && (flags & ISLASTCN)) {
672 if ((error = VOP_ACCESS(vdp, VWRITE, cred)) != 0)
673 goto out;
674 /*
675 * Careful about locking second inode.
676 * This can only occur if the target is ".".
677 */
678 if ((error = ufs_getino(vdp, dp, foundino, &tdp, true)) != 0)
679 goto out;
680 *vpp = tdp;
681 goto out;
682 }
683
684 if ((error = ufs_getino(vdp, dp, foundino, &tdp, false)) != 0)
685 goto out;
686
687 *vpp = tdp;
688 /*
689 * Insert name into cache if appropriate.
690 */
691 cache_enter(vdp, *vpp, cnp->cn_nameptr, cnp->cn_namelen, cnp->cn_flags);
692 error = 0;
693
694out:
695 return error;
696}

698void
699ufs_dirbad(struct inode *ip, doff_t offset, const char *how)
700{
701 struct mount *mp = ITOV(ip)->v_mount;
702 void (*p)(const char *, ...) __printflike(1, 2) =
703 (mp->mnt_flag & MNT_RDONLY) == 0 ? panic : printf;
704
705 (*p)("%s: bad dir ino %ju at offset %d: %s\n",
706 mp->mnt_stat.f_mntonname, (uintmax_t)ip->i_number,
707 offset, how);
708}

710/*
711 * Do consistency checking on a directory entry:
712 * record length must be multiple of 4
713 * entry must fit in rest of its DIRBLKSIZ block
714 * record must be large enough to contain entry
715 * name is not longer than FFS_MAXNAMLEN
716 * name must be as long as advertised, and null terminated
717 */
718const char *
719ufs_dirbadentry(const struct vnode *dp, const struct direct *ep,
720 int entryoffsetinblock)
721{
722 const struct ufsmount *ump = VFSTOUFS(dp->v_mount);
723 const int needswap = UFS_MPNEEDSWAP(ump);
724 const int dirblksiz = ump->um_dirblksiz;
725 const int maxsize = dirblksiz - (entryoffsetinblock & (dirblksiz - 1));
726 const int fsfmt = FSFMT(dp);
727 const uint8_t namlen = NAMLEN(fsfmt, needswap, ep);
728 const uint16_t reclen = ufs_rw16(ep->d_reclen, needswap);
729 const int dirsiz = (int)UFS_DIRSIZ(fsfmt, ep, needswap);
730 const char *name = ep->d_name;
731 const char *str;
732#ifdef DIAGNOSTIC
733 static char buf[512];
734#endif
735
736 if ((reclen & 0x3) != 0)
737 str = "not rounded";
738 else if (reclen > maxsize)
739 str = "too big";
740 else if (reclen < dirsiz)
741 str = "too small";
742#if FFS_MAXNAMLEN < 255
743 else if (namlen > FFS_MAXNAMLEN)
744 str = "long name";
745#endif
746 else
747 str = NULL;
748
749 if (str) {
750#ifdef DIAGNOSTIC
751 snprintf(buf, sizeof(buf), "Bad dir (%s), reclen=%#x, "
752 "namlen=%d, dirsiz=%d <= reclen=%d <= maxsize=%d, "
753 "flags=%#x, entryoffsetinblock=%d, dirblksiz=%d",
754 str, reclen, namlen, dirsiz, reclen, maxsize,
755 dp->v_mount->mnt_flag, entryoffsetinblock, dirblksiz);
756 str = buf;
757#endif
758 return str;
759 }
760
761 if (ep->d_ino == 0)
762 return NULL;
763
764 for (uint8_t i = 0; i < namlen; i++)
765 if (name[i] == '\0') {
766 str = "NUL in name";
767#ifdef DIAGNOSTIC
768 snprintf(buf, sizeof(buf), "%s [%s] i=%d, namlen=%d",
769 str, name, i, namlen);
770 str = buf;
771#endif
772 return str;
773 }
774
775 if (name[namlen]) {
776 str = "missing NUL in name";
777#ifdef DIAGNOSTIC
778 snprintf(buf, sizeof(buf), "%s [%*.*s] namlen=%d", str,
779 namlen, namlen, name, namlen);
780 str = buf;
781#endif
782 return str;
783 }
784 return NULL;
785}

787/*
788 * Construct a new directory entry after a call to namei, using the
789 * name in the componentname argument cnp. The argument ip is the
790 * inode to which the new directory entry will refer.
791 */
792void
793ufs_makedirentry(struct inode *ip, struct componentname *cnp,
794 struct direct *newdirp)
795{
796 size_t namelen = cnp->cn_namelen;
797
798 newdirp->d_ino = ip->i_number;
799 newdirp->d_namlen = namelen;
800 memcpy(newdirp->d_name, cnp->cn_nameptr, namelen);
801
802 /* NUL terminate and zero out padding */
803 memset(&newdirp->d_name[namelen], 0, UFS_NAMEPAD(namelen));
804
805 if (FSFMT(ITOV(ip)))
806 newdirp->d_type = 0;
807 else
808 newdirp->d_type = IFTODT(ip->i_mode);
809}

812static int
813ufs_dirgrow(struct vnode *dvp, const struct ufs_lookup_results *ulr,
814 struct vnode *tvp, struct direct *dirp,
815 struct componentname *cnp, struct buf *newdirbp)
816{
817 const kauth_cred_t cr = cnp->cn_cred;
818 const struct ufsmount *ump = VFSTOUFS(dvp->v_mount);
819 const int needswap = UFS_MPNEEDSWAP(ump);
820 const int dirblksiz = ump->um_dirblksiz;
821 const int fsfmt = FSFMT(dvp);
822 const u_int newentrysize = UFS_DIRSIZ(0, dirp, 0);
823 struct inode *dp = VTOI(dvp);
824 int error, ret, blkoff;
825 struct timespec ts;
826 struct buf *bp;
827
828 /*
829 * If ulr_count is 0, then namei could find no
830 * space in the directory. Here, ulr_offset will
831 * be on a directory block boundary and we will write the
832 * new entry into a fresh block.
833 */
834 if (ulr->ulr_offset & (dirblksiz - 1))
835 panic("%s: newblk", __func__);
836 if ((error = UFS_BALLOC(dvp, (off_t)ulr->ulr_offset, dirblksiz,
837 cr, B_CLRBUF | B_SYNC, &bp)) != 0) {
838 return error;
839 }
840
841 dp->i_size = ulr->ulr_offset + dirblksiz;
842 DIP_ASSIGN(dp, size, dp->i_size);
843 dp->i_flag |= IN_CHANGE | IN_UPDATE;
844 uvm_vnp_setsize(dvp, dp->i_size);
845 dirp->d_reclen = ufs_rw16(dirblksiz, needswap);
846 dirp->d_ino = ufs_rw32(dirp->d_ino, needswap);
847 if (fsfmt && ENDIANSWAP(needswap))
848 ufs_dirswap(dirp);
849 blkoff = ulr->ulr_offset & (ump->um_mountp->mnt_stat.f_iosize - 1);
850 memcpy((char *)bp->b_data + blkoff, dirp, newentrysize);
851#ifdef UFS_DIRHASH
852 if (dp->i_dirhash != NULL) {
853 ufsdirhash_newblk(dp, ulr->ulr_offset);
854 ufsdirhash_add(dp, dirp, ulr->ulr_offset);
855 ufsdirhash_checkblock(dp, (char *)bp->b_data + blkoff,
856 ulr->ulr_offset);
857 }
858#endif
859 error = VOP_BWRITE(bp->b_vp, bp);
860 vfs_timestamp(&ts);
861 ret = UFS_UPDATE(dvp, &ts, &ts, UPDATE_DIROP);
862 if (error == 0)
863 return ret;
864 return error;
865}

867static int
868#if __GNUC_PREREQ__(5, 3)
869/* This gets miscompiled by gcc 5.3 PR/51094 */
870__attribute__((__optimize__("no-tree-vrp")))
871#endif
872ufs_dircompact(struct vnode *dvp, const struct ufs_lookup_results *ulr,
873 struct vnode *tvp, struct direct *dirp,
874 struct componentname *cnp, struct buf *newdirbp)
875{
876 const struct ufsmount *ump = VFSTOUFS(dvp->v_mount);
877 const int needswap = UFS_MPNEEDSWAP(ump);
878 const int fsfmt = FSFMT(dvp);
879 const u_int newentrysize = UFS_DIRSIZ(0, dirp, 0);
880 struct inode *dp = VTOI(dvp);
881 struct buf *bp;
882 u_int dsize;
883 struct direct *ep, *nep;
884 int error, loc, spacefree;
885 char *dirbuf;
886 uint16_t reclen;
887
888 UFS_WAPBL_JLOCK_ASSERT(dvp->v_mount);
889
890 /*
891 * If ulr_count is non-zero, then namei found space for the new
892 * entry in the range ulr_offset to ulr_offset + ulr_count
893 * in the directory. To use this space, we may have to compact
894 * the entries located there, by copying them together towards the
895 * beginning of the block, leaving the free space in one usable
896 * chunk at the end.
897 */
898
899 /*
900 * Increase size of directory if entry eats into new space.
901 * This should never push the size past a new multiple of
902 * DIRBLKSIZ.
903 *
904 * N.B. - THIS IS AN ARTIFACT OF 4.2 AND SHOULD NEVER HAPPEN.
905 */
906 if (ulr->ulr_offset + ulr->ulr_count > dp->i_size) {
907#ifdef DIAGNOSTIC
908 printf("%s: reached 4.2-only block, not supposed to happen\n",
909 __func__);
910#endif
911 dp->i_size = ulr->ulr_offset + ulr->ulr_count;
912 DIP_ASSIGN(dp, size, dp->i_size);
913 dp->i_flag |= IN_CHANGE | IN_UPDATE;
914 UFS_WAPBL_UPDATE(dvp, NULL, NULL, UPDATE_DIROP);
915 }
916 /*
917 * Get the block containing the space for the new directory entry.
918 */
919 error = ufs_blkatoff(dvp, (off_t)ulr->ulr_offset, &dirbuf, &bp, true);
920 if (error)
921 return error;
922
923 /*
924 * Find space for the new entry. In the simple case, the entry at
925 * offset base will have the space. If it does not, then namei
926 * arranged that compacting the region ulr_offset to
927 * ulr_offset + ulr_count would yield the space.
928 */
929 ep = (void *)dirbuf;
930 dsize = (ep->d_ino != 0) ? UFS_DIRSIZ(fsfmt, ep, needswap) : 0;
931 reclen = ufs_rw16(ep->d_reclen, needswap);
932 spacefree = reclen - dsize;
933 for (loc = reclen; loc < ulr->ulr_count; ) {
934 nep = (void *)(dirbuf + loc);
935
936 /* Trim the existing slot (NB: dsize may be zero). */
937 ep->d_reclen = ufs_rw16(dsize, needswap);
938 ep = (void *)((char *)ep + dsize);
939
940 reclen = ufs_rw16(nep->d_reclen, needswap);
941 loc += reclen;
942 if (nep->d_ino == 0) {
943 /*
944 * A mid-block unused entry. Such entries are
945 * never created by the kernel, but fsck_ffs
946 * can create them (and it doesn't fix them).
947 *
948 * Add up the free space, and initialise the
949 * relocated entry since we don't memcpy it.
950 */
951 spacefree += reclen;
952 ep->d_ino = 0;
953 dsize = 0;
954 continue;
955 }
956 dsize = UFS_DIRSIZ(fsfmt, nep, needswap);
957 spacefree += reclen - dsize;
958#ifdef UFS_DIRHASH
959 if (dp->i_dirhash != NULL)
960 ufsdirhash_move(dp, nep,
961 ulr->ulr_offset + ((char *)nep - dirbuf),
962 ulr->ulr_offset + ((char *)ep - dirbuf));
963#endif
964 memcpy(ep, nep, dsize);
965 }
966 /*
967 * Here, `ep' points to a directory entry containing `dsize' in-use
968 * bytes followed by `spacefree' unused bytes. If ep->d_ino == 0,
969 * then the entry is completely unused (dsize == 0). The value
970 * of ep->d_reclen is always indeterminate.
971 *
972 * Update the pointer fields in the previous entry (if any),
973 * copy in the new entry, and write out the block.
974 */
975 if (ep->d_ino == 0 ||
976 (ufs_rw32(ep->d_ino, needswap) == UFS_WINO &&
977 memcmp(ep->d_name, dirp->d_name, dirp->d_namlen) == 0)) {
978 if (spacefree + dsize < newentrysize)
979 panic("%s: too big", __func__);
980 dirp->d_reclen = spacefree + dsize;
981 } else {
982 if (spacefree < newentrysize)
983 panic("%s: nospace", __func__);
984 dirp->d_reclen = spacefree;
985 ep->d_reclen = ufs_rw16(dsize, needswap);
986 ep = (void *)((char *)ep + dsize);
987 }
988
989 dirp->d_reclen = ufs_rw16(dirp->d_reclen, needswap);
990 dirp->d_ino = ufs_rw32(dirp->d_ino, needswap);
991 if (fsfmt && ENDIANSWAP(needswap))
992 ufs_dirswap(dirp);
993#ifdef UFS_DIRHASH
994 if (dp->i_dirhash != NULL && (ep->d_ino == 0 ||
995 dirp->d_reclen == spacefree))
996 ufsdirhash_add(dp, dirp, ulr->ulr_offset + ((char *)ep - dirbuf));
997#endif
998 memcpy(ep, dirp, newentrysize);
999#ifdef UFS_DIRHASH
1000 if (dp->i_dirhash != NULL) {
1001 const int dirblkmsk = ump->um_dirblksiz - 1;
1002 ufsdirhash_checkblock(dp, dirbuf -
1003 (ulr->ulr_offset & dirblkmsk),
1004 ulr->ulr_offset & ~dirblkmsk);
1005 }
1006#endif
1007 error = VOP_BWRITE(bp->b_vp, bp);
1008 dp->i_flag |= IN_CHANGE | IN_UPDATE;
1009 /*
1010 * If all went well, and the directory can be shortened, proceed
1011 * with the truncation. Note that we have to unlock the inode for
1012 * the entry that we just entered, as the truncation may need to
1013 * lock other inodes which can lead to deadlock if we also hold a
1014 * lock on the newly entered node.
1015 */
1016 if (error == 0 && ulr->ulr_endoff && ulr->ulr_endoff < dp->i_size) {
1017 const kauth_cred_t cr = cnp->cn_cred;
1018#ifdef UFS_DIRHASH
1019 if (dp->i_dirhash != NULL)
1020 ufsdirhash_dirtrunc(dp, ulr->ulr_endoff);
1021#endif
1022 (void) UFS_TRUNCATE(dvp, (off_t)ulr->ulr_endoff, IO_SYNC, cr);
1023 }
1024 UFS_WAPBL_UPDATE(dvp, NULL, NULL, UPDATE_DIROP);
1025 return error;
1026}

1028/*
1029 * Write a directory entry after a call to namei, using the parameters
1030 * that ufs_lookup left in nameidata and in the ufs_lookup_results.
1031 *
1032 * DVP is the directory to be updated. It must be locked.
1033 * ULR is the ufs_lookup_results structure from the final lookup step.
1034 * TVP is not used. (XXX: why is it here? remove it)
1035 * DIRP is the new directory entry contents.
1036 * CNP is the componentname from the final lookup step.
1037 * NEWDIRBP is not used and (XXX) should be removed. The previous
1038 * comment here said it was used by the now-removed softupdates code.
1039 *
1040 * The link count of the target inode is *not* incremented; the
1041 * caller does that.
1042 *
1043 * If ulr->ulr_count is 0, ufs_lookup did not find space to insert the
1044 * directory entry. ulr_offset, which is the place to put the entry,
1045 * should be on a block boundary (and should be at the end of the
1046 * directory AFAIK) and a fresh block is allocated to put the new
1047 * directory entry in.
1048 *
1049 * If ulr->ulr_count is not zero, ufs_lookup found a slot to insert
1050 * the entry into. This slot ranges from ulr_offset to ulr_offset +
1051 * ulr_count. However, this slot may already be partially populated
1052 * requiring compaction. See notes below.
1053 *
1054 * Furthermore, if ulr_count is not zero and ulr_endoff is not the
1055 * same as i_size, the directory is truncated to size ulr_endoff.
1056 */
1057int
1058ufs_direnter(struct vnode *dvp, const struct ufs_lookup_results *ulr,
1059 struct vnode *tvp, struct direct *dirp,
1060 struct componentname *cnp, struct buf *newdirbp)
1061{
1062 if (ulr->ulr_count == 0)
1063 return ufs_dirgrow(dvp, ulr, tvp, dirp, cnp, newdirbp);
1064 else
1065 return ufs_dircompact(dvp, ulr, tvp, dirp, cnp, newdirbp);
1066}

1068/*
1069 * Remove a directory entry after a call to namei, using the
1070 * parameters that ufs_lookup left in nameidata and in the
1071 * ufs_lookup_results.
1072 *
1073 * DVP is the directory to be updated. It must be locked.
1074 * ULR is the ufs_lookup_results structure from the final lookup step.
1075 * IP, if not null, is the inode being unlinked.
1076 * FLAGS may contain DOWHITEOUT.
1077 * ISRMDIR is not used and (XXX) should be removed.
1078 *
1079 * If FLAGS contains DOWHITEOUT the entry is replaced with a whiteout
1080 * instead of being cleared.
1081 *
1082 * ulr->ulr_offset contains the position of the directory entry
1083 * to be removed.
1084 *
1085 * ulr->ulr_reclen contains the size of the directory entry to be
1086 * removed.
1087 *
1088 * ulr->ulr_count contains the size of the *previous* directory
1089 * entry. This allows finding it, for free space management. If
1090 * ulr_count is 0, the target entry is at the beginning of the
1091 * directory. (Does this ever happen? The first entry should be ".",
1092 * which should only be removed at rmdir time. Does rmdir come here
1093 * to clear out the "." and ".." entries? Perhaps, but I doubt it.)
1094 *
1095 * The space is marked free by adding it to the record length (not
1096 * name length) of the preceding entry. If the first entry becomes
1097 * free, it is marked free by setting the inode number to 0.
1098 *
1099 * The link count of IP is decremented. Note that this is not the
1100 * inverse behavior of ufs_direnter, which does not adjust link
1101 * counts. Sigh.
1102 */
1103int
1104ufs_dirremove(struct vnode *dvp, const struct ufs_lookup_results *ulr,
1105 struct inode *ip, int flags, int isrmdir)
1106{
1107 struct inode *dp = VTOI(dvp);
1108 struct direct *ep;
1109 struct buf *bp;
1110 int error;
1111 const int needswap = UFS_MPNEEDSWAP(dp->i_ump);
1112 uint16_t reclen;
1113
1114 UFS_WAPBL_JLOCK_ASSERT(dvp->v_mount);
1115
1116 if (flags & DOWHITEOUT) {
1117 /*
1118 * Whiteout entry: set d_ino to UFS_WINO.
1119 */
1120 error = ufs_blkatoff(dvp, (off_t)ulr->ulr_offset, &ep,
1121 &bp, true);
1122 if (error)
1123 return (error);
1124 ep->d_ino = ufs_rw32(UFS_WINO, needswap);
1125 ep->d_type = DT_WHT;
1126 goto out;
1127 }
1128
1129 if ((error = ufs_blkatoff(dvp,
1130 (off_t)(ulr->ulr_offset - ulr->ulr_count), &ep, &bp, true)) != 0)
1131 return (error);
1132
1133 reclen = ufs_rw16(ep->d_reclen, needswap);
1134#ifdef UFS_DIRHASH
1135 /*
1136 * Remove the dirhash entry. This is complicated by the fact
1137 * that `ep' is the previous entry when ulr_count != 0.
1138 */
1139 if (dp->i_dirhash != NULL)
1140 ufsdirhash_remove(dp, (ulr->ulr_count == 0) ? ep :
1141 (void *)((char *)ep + reclen), ulr->ulr_offset);
1142#endif
1143
1144 if (ulr->ulr_count == 0) {
1145 /*
1146 * First entry in block: set d_ino to zero.
1147 */
1148 ep->d_ino = 0;
1149 } else {
1150 /*
1151 * Collapse new free space into previous entry.
1152 */
1153 ep->d_reclen = ufs_rw16(reclen + ulr->ulr_reclen, needswap);
1154 }
1155
1156#ifdef UFS_DIRHASH
1157 if (dp->i_dirhash != NULL) {
1158 int dirblksiz = ip->i_ump->um_dirblksiz;
1159 ufsdirhash_checkblock(dp, (char *)ep -
1160 ((ulr->ulr_offset - ulr->ulr_count) & (dirblksiz - 1)),
1161 ulr->ulr_offset & ~(dirblksiz - 1));
1162 }
1163#endif
1164
1165out:
1166 if (ip) {
1167 ip->i_nlink--;
1168 DIP_ASSIGN(ip, nlink, ip->i_nlink);
1169 ip->i_flag |= IN_CHANGE;
1170 UFS_WAPBL_UPDATE(ITOV(ip), NULL, NULL, 0);
1171 }
1172 /*
1173 * XXX did it ever occur to anyone that it might be a good
1174 * idea to restore ip->i_nlink if this fails? Or something?
1175 * Currently on error return from this function the state of
1176 * ip->i_nlink depends on what happened, and callers
1177 * definitely do not take this into account.
1178 */
1179 error = VOP_BWRITE(bp->b_vp, bp);
1180 dp->i_flag |= IN_CHANGE | IN_UPDATE;
1181 /*
1182 * If the last named reference to a snapshot goes away,
1183 * drop its snapshot reference so that it will be reclaimed
1184 * when last open reference goes away.
1185 */
1186 if (ip != 0 && (ip->i_flags & SF_SNAPSHOT) != 0 &&
1187 ip->i_nlink == 0)
1188 UFS_SNAPGONE(ITOV(ip));
1189 UFS_WAPBL_UPDATE(dvp, NULL, NULL, 0);
1190 return (error);
1191}

1193/*
1194 * Rewrite an existing directory entry to point at the inode supplied.
1195 *
1196 * DP is the directory to update.
1197 * OFFSET is the position of the entry in question. It may come
1198 * from ulr_offset of a ufs_lookup_results.
1199 * OIP is the old inode the directory previously pointed to.
1200 * NEWINUM is the number of the new inode.
1201 * NEWTYPE is the new value for the type field of the directory entry.
1202 * (This is ignored if the fs doesn't support that.)
1203 * ISRMDIR is not used and (XXX) should be removed.
1204 * IFLAGS are added to DP's inode flags.
1205 *
1206 * The link count of OIP is decremented. Note that the link count of
1207 * the new inode is *not* incremented. Yay for symmetry.
1208 */
1209int
1210ufs_dirrewrite(struct inode *dp, off_t offset,
1211 struct inode *oip, ino_t newinum, int newtype,
1212 int isrmdir, int iflags)
1213{
1214 struct buf *bp;
1215 struct direct *ep;
1216 struct vnode *vdp = ITOV(dp);
1217 int error;
1218
1219 error = ufs_blkatoff(vdp, offset, &ep, &bp, true);
1220 if (error)
1221 return (error);
1222 ep->d_ino = ufs_rw32(newinum, UFS_MPNEEDSWAP(dp->i_ump));
1223 if (!FSFMT(vdp))
1224 ep->d_type = newtype;
1225 oip->i_nlink--;
1226 DIP_ASSIGN(oip, nlink, oip->i_nlink);
1227 oip->i_flag |= IN_CHANGE;
1228 UFS_WAPBL_UPDATE(ITOV(oip), NULL, NULL, UPDATE_DIROP);
1229 error = VOP_BWRITE(bp->b_vp, bp);
1230 dp->i_flag |= iflags;
1231 /*
1232 * If the last named reference to a snapshot goes away,
1233 * drop its snapshot reference so that it will be reclaimed
1234 * when last open reference goes away.
1235 */
1236 if ((oip->i_flags & SF_SNAPSHOT) != 0 && oip->i_nlink == 0)
1237 UFS_SNAPGONE(ITOV(oip));
1238 UFS_WAPBL_UPDATE(vdp, NULL, NULL, UPDATE_DIROP);
1239 return (error);
1240}

1242/*
1243 * Check if a directory is empty or not.
1244 * Inode supplied must be locked.
1245 *
1246 * Using a struct dirtemplate here is not precisely
1247 * what we want, but better than using a struct direct.
1248 *
1249 * NB: does not handle corrupted directories.
1250 */
1251int
1252ufs_dirempty(struct inode *ip, ino_t parentino, kauth_cred_t cred)
1253{
1254 doff_t off;
1255 struct dirtemplate dbuf;
1256 struct direct *dp = (void *)&dbuf;
1257 int error;
1258 size_t count;
1259 const int needswap = UFS_IPNEEDSWAP(ip);
1260 const int fsfmt = FSFMT(ITOV(ip));
1261#define MINDIRSIZ (sizeof (struct dirtemplate) / 2)
1262
1263 for (off = 0; off < ip->i_size;
1264 off += ufs_rw16(dp->d_reclen, needswap)) {
1265 error = ufs_bufio(UIO_READ, ITOV(ip), dp, MINDIRSIZ,
1266 off, IO_NODELOCKED, cred, &count, NULL);
1267 /*
1268 * Since we read MINDIRSIZ, residual must
1269 * be 0 unless we're at end of file.
1270 */
1271 if (error || count != 0)
1272 return (0);
1273 /* avoid infinite loops */
1274 if (dp->d_reclen == 0)
1275 return (0);
1276 /* skip empty entries */
1277 ino_t ino = ufs_rw32(dp->d_ino, needswap);
1278 if (ino == 0 || ino == UFS_WINO)
1279 continue;
1280 /* accept only "." and ".." */
1281 const uint8_t namlen = NAMLEN(fsfmt, needswap, dp);
1282 if (namlen > 2)
1283 return (0);
1284 if (dp->d_name[0] != '.')
1285 return (0);
1286 /*
1287 * At this point namlen must be 1 or 2.
1288 * 1 implies ".", 2 implies ".." if second
1289 * char is also "."
1290 */
1291 if (namlen == 1 && ino == ip->i_number)
1292 continue;
1293 if (dp->d_name[1] == '.' && ino == parentino)
1294 continue;
1295 return (0);
1296 }
1297 return (1);
1298}
1299
1300#define UFS_DIRRABLKS 0
1301int ufs_dirrablks = UFS_DIRRABLKS;
1302
1303/*
1304 * ufs_blkatoff: Return buffer with the contents of block "offset" from
1305 * the beginning of directory "vp". If "res" is non-NULL, fill it in with
1306 * a pointer to the remaining space in the directory. If the caller intends
1307 * to modify the buffer returned, "modify" must be true.
1308 */
1309
1310int
1311ufs_blkatoff(struct vnode *vp, off_t offset, void *v, struct buf **bpp,
1312 bool modify)
1313{
1314 char **res = v;
1315 struct inode *ip __diagused;
1316 struct buf *bp;
1317 daddr_t lbn;
1318 const int dirrablks = ufs_dirrablks;
1319 daddr_t *blks;
1320 int *blksizes;
1321 int run, error;
1322 struct mount *mp = vp->v_mount;
1323 const int bshift = mp->mnt_fs_bshift;
1324 const int bsize = 1 << bshift;
1325 off_t eof;
1326
1327 blks = kmem_alloc((1 + dirrablks) * sizeof(daddr_t), KM_SLEEP);
1328 blksizes = kmem_alloc((1 + dirrablks) * sizeof(int), KM_SLEEP);
1329 ip = VTOI(vp);
1330 KASSERT(vp->v_size == ip->i_size);
1331 GOP_SIZE(vp, vp->v_size, &eof, 0);
1332 lbn = offset >> bshift;
1333
1334 for (run = 0; run <= dirrablks;) {
1335 const off_t curoff = lbn << bshift;
1336 const int size = MIN(eof - curoff, bsize);
1337
1338 if (size == 0) {
1339 break;
1340 }
1341 KASSERT(curoff < eof);
1342 blks[run] = lbn;
1343 blksizes[run] = size;
1344 lbn++;
1345 run++;
1346 if (size != bsize) {
1347 break;
1348 }
1349 }
1350 KASSERT(run >= 1);
1351 error = breadn(vp, blks[0], blksizes[0], &blks[1], &blksizes[1],
1352 run - 1, (modify ? B_MODIFY : 0), &bp);
1353 if (error != 0) {
1354 *bpp = NULL;
1355 goto out;
1356 }
1357 if (res) {
1358 *res = (char *)bp->b_data + (offset & (bsize - 1));
1359 }
1360 *bpp = bp;
1361
1362 out:
1363 kmem_free(blks, (1 + dirrablks) * sizeof(daddr_t));
1364 kmem_free(blksizes, (1 + dirrablks) * sizeof(int));
1365 return error;
1366}
