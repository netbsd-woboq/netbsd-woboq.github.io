132/*
133 * Create a regular file
134 */
135int
136ufs_create(void *v)
137{
138 struct vop_create_v3_args /* {
139 struct vnode *a_dvp;
140 struct vnode **a_vpp;
141 struct componentname *a_cnp;
142 struct vattr *a_vap;
143 } */ *ap = v;
144 int error;
145 struct vnode *dvp = ap->a_dvp;
146 struct ufs_lookup_results *ulr;
147
148 /* XXX should handle this material another way */
149 ulr = &VTOI(dvp)->i_crap;
150 UFS_CHECK_CRAPCOUNTER(VTOI(dvp));
151
152 /*
153 * UFS_WAPBL_BEGIN(dvp->v_mount) performed by successful
154 * ufs_makeinode
155 */
156 error = ufs_makeinode(ap->a_vap, dvp, ulr, ap->a_vpp, ap->a_cnp);
157 if (error) {
158 return (error);
159 }
160 UFS_WAPBL_END(dvp->v_mount);
161 VN_KNOTE(dvp, NOTE_WRITE);
162 VOP_UNLOCK(*ap->a_vpp);
163 return (0);
164}

167 * Mknod vnode call
168 */
169/* ARGSUSED */
170int
171ufs_mknod(void *v)
172{
173 struct vop_mknod_v3_args /* {
174 struct vnode *a_dvp;
175 struct vnode **a_vpp;
176 struct componentname *a_cnp;
177 struct vattr *a_vap;
178 } */ *ap = v;
179 struct vattr *vap;
180 struct vnode **vpp;
181 struct inode *ip;
182 int error;
183 struct ufs_lookup_results *ulr;
184
185 vap = ap->a_vap;
186 vpp = ap->a_vpp;
187
188 /* XXX should handle this material another way */
189 ulr = &VTOI(ap->a_dvp)->i_crap;
190 UFS_CHECK_CRAPCOUNTER(VTOI(ap->a_dvp));
191
192 /*
193 * UFS_WAPBL_BEGIN(dvp->v_mount) performed by successful
194 * ufs_makeinode
195 */
196 if ((error = ufs_makeinode(vap, ap->a_dvp, ulr, vpp, ap->a_cnp)) != 0)
197 goto out;
198 VN_KNOTE(ap->a_dvp, NOTE_WRITE);
199 ip = VTOI(*vpp);
200 ip->i_flag |= IN_ACCESS | IN_CHANGE | IN_UPDATE;
201 UFS_WAPBL_UPDATE(*vpp, NULL, NULL, 0);
202 UFS_WAPBL_END(ap->a_dvp->v_mount);
203 VOP_UNLOCK(*vpp);
204out:
205 if (error != 0) {
206 *vpp = NULL;
207 return (error);
208 }
209 return (0);
210}

258static int
259ufs_check_possible(struct vnode *vp, struct inode *ip, mode_t mode,
260 kauth_cred_t cred)
261{
262#if defined(QUOTA) || defined(QUOTA2)
263 int error;
264#endif
265
266 /*
267 * Disallow write attempts on read-only file systems;
268 * unless the file is a socket, fifo, or a block or
269 * character device resident on the file system.
270 */
271 if (mode & VWRITE) {
272 switch (vp->v_type) {
273 case VDIR:
274 case VLNK:
275 case VREG:
276 if (vp->v_mount->mnt_flag & MNT_RDONLY)
277 return (EROFS);
278#if defined(QUOTA) || defined(QUOTA2)
279 error = chkdq(ip, 0, cred, 0);
280 if (error != 0)
281 return error;
282#endif
283 break;
284 case VBAD:
285 case VBLK:
286 case VCHR:
287 case VSOCK:
288 case VFIFO:
289 case VNON:
290 default:
291 break;
292 }
293 }
294
295 /* If it is a snapshot, nobody gets access to it. */
296 if ((ip->i_flags & SF_SNAPSHOT))
297 return (EPERM);
298 /* If immutable bit set, nobody gets to write it. */
299 if ((mode & VWRITE) && (ip->i_flags & IMMUTABLE))
300 return (EPERM);
301
302 return 0;
303}

305static int
306ufs_check_permitted(struct vnode *vp, struct inode *ip, mode_t mode,
307 kauth_cred_t cred)
308{
309
310 return kauth_authorize_vnode(cred, KAUTH_ACCESS_ACTION(mode, vp->v_type,
311 ip->i_mode & ALLPERMS), vp, NULL, genfs_can_access(vp->v_type,
312 ip->i_mode & ALLPERMS, ip->i_uid, ip->i_gid, mode, cred));
313}

315int
316ufs_access(void *v)
317{
318 struct vop_access_args /* {
319 struct vnode *a_vp;
320 int a_mode;
321 kauth_cred_t a_cred;
322 } */ *ap = v;
323 struct vnode *vp;
324 struct inode *ip;
325 mode_t mode;
326 int error;
327
328 vp = ap->a_vp;
329 ip = VTOI(vp);
330 mode = ap->a_mode;
331
332 error = ufs_check_possible(vp, ip, mode, ap->a_cred);
333 if (error)
334 return error;
335
336 error = ufs_check_permitted(vp, ip, mode, ap->a_cred);
337
338 return error;
339}

341/* ARGSUSED */
342int
343ufs_getattr(void *v)
344{
345 struct vop_getattr_args /* {
346 struct vnode *a_vp;
347 struct vattr *a_vap;
348 kauth_cred_t a_cred;
349 } */ *ap = v;
350 struct vnode *vp;
351 struct inode *ip;
352 struct vattr *vap;
353
354 vp = ap->a_vp;
355 ip = VTOI(vp);
356 vap = ap->a_vap;
357 UFS_ITIMES(vp, NULL, NULL, NULL);
358
359 /*
360 * Copy from inode table
361 */
362 vap->va_fsid = ip->i_dev;
363 vap->va_fileid = ip->i_number;
364 vap->va_mode = ip->i_mode & ALLPERMS;
365 vap->va_nlink = ip->i_nlink;
366 vap->va_uid = ip->i_uid;
367 vap->va_gid = ip->i_gid;
368 vap->va_size = vp->v_size;
369 if (ip->i_ump->um_fstype == UFS1) {
370 switch (vp->v_type) {
371 case VBLK:
372 case VCHR:
373 vap->va_rdev = (dev_t)ufs_rw32(ip->i_ffs1_rdev,
374 UFS_MPNEEDSWAP(ip->i_ump));
375 break;
376 default:
377 vap->va_rdev = NODEV;
378 break;
379 }
380 vap->va_atime.tv_sec = ip->i_ffs1_atime;
381 vap->va_atime.tv_nsec = ip->i_ffs1_atimensec;
382 vap->va_mtime.tv_sec = ip->i_ffs1_mtime;
383 vap->va_mtime.tv_nsec = ip->i_ffs1_mtimensec;
384 vap->va_ctime.tv_sec = ip->i_ffs1_ctime;
385 vap->va_ctime.tv_nsec = ip->i_ffs1_ctimensec;
386 vap->va_birthtime.tv_sec = 0;
387 vap->va_birthtime.tv_nsec = 0;
388 vap->va_bytes = dbtob((u_quad_t)ip->i_ffs1_blocks);
389 } else {
390 switch (vp->v_type) {
391 case VBLK:
392 case VCHR:
393 vap->va_rdev = (dev_t)ufs_rw64(ip->i_ffs2_rdev,
394 UFS_MPNEEDSWAP(ip->i_ump));
395 break;
396 default:
397 vap->va_rdev = NODEV;
398 break;
399 }
400 vap->va_atime.tv_sec = ip->i_ffs2_atime;
401 vap->va_atime.tv_nsec = ip->i_ffs2_atimensec;
402 vap->va_mtime.tv_sec = ip->i_ffs2_mtime;
403 vap->va_mtime.tv_nsec = ip->i_ffs2_mtimensec;
404 vap->va_ctime.tv_sec = ip->i_ffs2_ctime;
405 vap->va_ctime.tv_nsec = ip->i_ffs2_ctimensec;
406 vap->va_birthtime.tv_sec = ip->i_ffs2_birthtime;
407 vap->va_birthtime.tv_nsec = ip->i_ffs2_birthnsec;
408 vap->va_bytes = dbtob(ip->i_ffs2_blocks);
409 }
410 vap->va_gen = ip->i_gen;
411 vap->va_flags = ip->i_flags;
412
413 /* this doesn't belong here */
414 if (vp->v_type == VBLK)
415 vap->va_blocksize = BLKDEV_IOSIZE;
416 else if (vp->v_type == VCHR)
417 vap->va_blocksize = MAXBSIZE;
418 else
419 vap->va_blocksize = vp->v_mount->mnt_stat.f_iosize;
420 vap->va_type = vp->v_type;
421 vap->va_filerev = ip->i_modrev;
422 return (0);
423}

425/*
426 * Set attribute vnode op. called from several syscalls
427 */
428int
429ufs_setattr(void *v)
430{
431 struct vop_setattr_args /* {
432 struct vnode *a_vp;
433 struct vattr *a_vap;
434 kauth_cred_t a_cred;
435 } */ *ap = v;
436 struct vattr *vap;
437 struct vnode *vp;
438 struct inode *ip;
439 kauth_cred_t cred;
440 struct lwp *l;
441 int error;
442 kauth_action_t action;
443 bool changing_sysflags;
444
445 vap = ap->a_vap;
446 vp = ap->a_vp;
447 ip = VTOI(vp);
448 cred = ap->a_cred;
449 l = curlwp;
450 action = KAUTH_VNODE_WRITE_FLAGS;
451 changing_sysflags = false;
452
453 /*
454 * Check for unsettable attributes.
455 */
456 if ((vap->va_type != VNON) || (vap->va_nlink != VNOVAL) ||
457 (vap->va_fsid != VNOVAL) || (vap->va_fileid != VNOVAL) ||
458 (vap->va_blocksize != VNOVAL) || (vap->va_rdev != VNOVAL) ||
459 ((int)vap->va_bytes != VNOVAL) || (vap->va_gen != VNOVAL)) {
460 return (EINVAL);
461 }
462
463 UFS_WAPBL_JUNLOCK_ASSERT(vp->v_mount);
464
465 if (vap->va_flags != VNOVAL) {
466 if (vp->v_mount->mnt_flag & MNT_RDONLY) {
467 error = EROFS;
468 goto out;
469 }
470
471 /* Snapshot flag cannot be set or cleared */
472 if ((vap->va_flags & (SF_SNAPSHOT | SF_SNAPINVAL)) !=
473 (ip->i_flags & (SF_SNAPSHOT | SF_SNAPINVAL))) {
474 error = EPERM;
475 goto out;
476 }
477
478 if (ip->i_flags & (SF_IMMUTABLE | SF_APPEND)) {
479 action |= KAUTH_VNODE_HAS_SYSFLAGS;
480 }
481
482 if ((vap->va_flags & SF_SETTABLE) !=
483 (ip->i_flags & SF_SETTABLE)) {
484 action |= KAUTH_VNODE_WRITE_SYSFLAGS;
485 changing_sysflags = true;
486 }
487
488 error = kauth_authorize_vnode(cred, action, vp, NULL,
489 genfs_can_chflags(cred, vp->v_type, ip->i_uid,
490 changing_sysflags));
491 if (error)
492 goto out;
493
494 if (changing_sysflags) {
495 error = UFS_WAPBL_BEGIN(vp->v_mount);
496 if (error)
497 goto out;
498 ip->i_flags = vap->va_flags;
499 DIP_ASSIGN(ip, flags, ip->i_flags);
500 } else {
501 error = UFS_WAPBL_BEGIN(vp->v_mount);
502 if (error)
503 goto out;
504 ip->i_flags &= SF_SETTABLE;
505 ip->i_flags |= (vap->va_flags & UF_SETTABLE);
506 DIP_ASSIGN(ip, flags, ip->i_flags);
507 }
508 ip->i_flag |= IN_CHANGE;
509 UFS_WAPBL_UPDATE(vp, NULL, NULL, 0);
510 UFS_WAPBL_END(vp->v_mount);
511 if (vap->va_flags & (IMMUTABLE | APPEND)) {
512 error = 0;
513 goto out;
514 }
515 }
516 if (ip->i_flags & (IMMUTABLE | APPEND)) {
517 error = EPERM;
518 goto out;
519 }
520 /*
521 * Go through the fields and update iff not VNOVAL.
522 */
523 if (vap->va_uid != (uid_t)VNOVAL || vap->va_gid != (gid_t)VNOVAL) {
524 if (vp->v_mount->mnt_flag & MNT_RDONLY) {
525 error = EROFS;
526 goto out;
527 }
528 error = UFS_WAPBL_BEGIN(vp->v_mount);
529 if (error)
530 goto out;
531 error = ufs_chown(vp, vap->va_uid, vap->va_gid, cred, l);
532 UFS_WAPBL_END(vp->v_mount);
533 if (error)
534 goto out;
535 }
536 if (vap->va_size != VNOVAL) {
537 /*
538 * Disallow write attempts on read-only file systems;
539 * unless the file is a socket, fifo, or a block or
540 * character device resident on the file system.
541 */
542 switch (vp->v_type) {
543 case VDIR:
544 error = EISDIR;
545 goto out;
546 case VCHR:
547 case VBLK:
548 case VFIFO:
549 break;
550 case VREG:
551 if (vp->v_mount->mnt_flag & MNT_RDONLY) {
552 error = EROFS;
553 goto out;
554 }
555 if ((ip->i_flags & SF_SNAPSHOT) != 0) {
556 error = EPERM;
557 goto out;
558 }
559 error = ufs_truncate_retry(vp, vap->va_size, cred);
560 if (error)
561 goto out;
562 break;
563 default:
564 error = EOPNOTSUPP;
565 goto out;
566 }
567 }
568 ip = VTOI(vp);
569 if (vap->va_atime.tv_sec != VNOVAL || vap->va_mtime.tv_sec != VNOVAL ||
570 vap->va_birthtime.tv_sec != VNOVAL) {
571 if (vp->v_mount->mnt_flag & MNT_RDONLY) {
572 error = EROFS;
573 goto out;
574 }
575 if ((ip->i_flags & SF_SNAPSHOT) != 0) {
576 error = EPERM;
577 goto out;
578 }
579 error = kauth_authorize_vnode(cred, KAUTH_VNODE_WRITE_TIMES, vp,
580 NULL, genfs_can_chtimes(vp, vap->va_vaflags, ip->i_uid, cred));
581 if (error)
582 goto out;
583 error = UFS_WAPBL_BEGIN(vp->v_mount);
584 if (error)
585 goto out;
586 if (vap->va_atime.tv_sec != VNOVAL)
587 if (!(vp->v_mount->mnt_flag & MNT_NOATIME))
588 ip->i_flag |= IN_ACCESS;
589 if (vap->va_mtime.tv_sec != VNOVAL) {
590 ip->i_flag |= IN_CHANGE | IN_UPDATE;
591 if (vp->v_mount->mnt_flag & MNT_RELATIME)
592 ip->i_flag |= IN_ACCESS;
593 }
594 if (vap->va_birthtime.tv_sec != VNOVAL &&
595 ip->i_ump->um_fstype == UFS2) {
596 ip->i_ffs2_birthtime = vap->va_birthtime.tv_sec;
597 ip->i_ffs2_birthnsec = vap->va_birthtime.tv_nsec;
598 }
599 error = UFS_UPDATE(vp, &vap->va_atime, &vap->va_mtime, 0);
600 UFS_WAPBL_END(vp->v_mount);
601 if (error)
602 goto out;
603 }
604 error = 0;
605 if (vap->va_mode != (mode_t)VNOVAL) {
606 if (vp->v_mount->mnt_flag & MNT_RDONLY) {
607 error = EROFS;
608 goto out;
609 }
610 if ((ip->i_flags & SF_SNAPSHOT) != 0 &&
611 (vap->va_mode & (S_IXUSR | S_IWUSR | S_IXGRP | S_IWGRP |
612 S_IXOTH | S_IWOTH))) {
613 error = EPERM;
614 goto out;
615 }
616 error = UFS_WAPBL_BEGIN(vp->v_mount);
617 if (error)
618 goto out;
619 error = ufs_chmod(vp, (int)vap->va_mode, cred, l);
620 UFS_WAPBL_END(vp->v_mount);
621 }
622 VN_KNOTE(vp, NOTE_ATTRIB);
623out:
624 return (error);
625}

627/*
628 * Change the mode on a file.
629 * Inode must be locked before calling.
630 */
631static int
632ufs_chmod(struct vnode *vp, int mode, kauth_cred_t cred, struct lwp *l)
633{
634 struct inode *ip;
635 int error;
636
637 UFS_WAPBL_JLOCK_ASSERT(vp->v_mount);
638
639 ip = VTOI(vp);
640
641 error = kauth_authorize_vnode(cred, KAUTH_VNODE_WRITE_SECURITY, vp,
642 NULL, genfs_can_chmod(vp->v_type, cred, ip->i_uid, ip->i_gid, mode));
643 if (error)
644 return (error);
645
646 ip->i_mode &= ~ALLPERMS;
647 ip->i_mode |= (mode & ALLPERMS);
648 ip->i_flag |= IN_CHANGE;
649 DIP_ASSIGN(ip, mode, ip->i_mode);
650 UFS_WAPBL_UPDATE(vp, NULL, NULL, 0);
651 return (0);
652}

654/*
655 * Perform chown operation on inode ip;
656 * inode must be locked prior to call.
657 */
658static int
659ufs_chown(struct vnode *vp, uid_t uid, gid_t gid, kauth_cred_t cred,
660 struct lwp *l)
661{
662 struct inode *ip;
663 int error = 0;
664#if defined(QUOTA) || defined(QUOTA2)
665 uid_t ouid;
666 gid_t ogid;
667 int64_t change;
668#endif
669 ip = VTOI(vp);
670 error = 0;
671
672 if (uid == (uid_t)VNOVAL)
673 uid = ip->i_uid;
674 if (gid == (gid_t)VNOVAL)
675 gid = ip->i_gid;
676
677 error = kauth_authorize_vnode(cred, KAUTH_VNODE_CHANGE_OWNERSHIP, vp,
678 NULL, genfs_can_chown(cred, ip->i_uid, ip->i_gid, uid, gid));
679 if (error)
680 return (error);
681
682#if defined(QUOTA) || defined(QUOTA2)
683 ogid = ip->i_gid;
684 ouid = ip->i_uid;
685 change = DIP(ip, blocks);
686 (void) chkdq(ip, -change, cred, 0);
687 (void) chkiq(ip, -1, cred, 0);
688#endif
689 ip->i_gid = gid;
690 DIP_ASSIGN(ip, gid, gid);
691 ip->i_uid = uid;
692 DIP_ASSIGN(ip, uid, uid);
693#if defined(QUOTA) || defined(QUOTA2)
694 if ((error = chkdq(ip, change, cred, 0)) == 0) {
695 if ((error = chkiq(ip, 1, cred, 0)) == 0)
696 goto good;
697 else
698 (void) chkdq(ip, -change, cred, FORCE);
699 }
700 ip->i_gid = ogid;
701 DIP_ASSIGN(ip, gid, ogid);
702 ip->i_uid = ouid;
703 DIP_ASSIGN(ip, uid, ouid);
704 (void) chkdq(ip, change, cred, FORCE);
705 (void) chkiq(ip, 1, cred, FORCE);
706 return (error);
707 good:
708#endif /* QUOTA || QUOTA2 */
709 ip->i_flag |= IN_CHANGE;
710 UFS_WAPBL_UPDATE(vp, NULL, NULL, 0);
711 return (0);
712}

902int
903ufs_mkdir(void *v)
904{
905 struct vop_mkdir_v3_args /* {
906 struct vnode *a_dvp;
907 struct vnode **a_vpp;
908 struct componentname *a_cnp;
909 struct vattr *a_vap;
910 } */ *ap = v;
911 struct vnode *dvp = ap->a_dvp, *tvp;
912 struct vattr *vap = ap->a_vap;

914 struct inode *ip, *dp = VTOI(dvp);
915 struct buf *bp;
916 struct dirtemplate dirtemplate;
917 struct direct *newdir;
918 int error;
919 struct ufsmount *ump = dp->i_ump;
920 int dirblksiz = ump->um_dirblksiz;
921 struct ufs_lookup_results *ulr;
922
923 /* XXX should handle this material another way */
924 ulr = &dp->i_crap;
925 UFS_CHECK_CRAPCOUNTER(dp);
926
927 KASSERT(vap->va_type == VDIR);
928
929 if ((nlink_t)dp->i_nlink >= LINK_MAX) {
930 error = EMLINK;
931 goto out;
932 }
933 /*
934 * Must simulate part of ufs_makeinode here to acquire the inode,
935 * but not have it entered in the parent directory. The entry is
936 * made later after writing "." and ".." entries.
937 */
938 error = vcache_new(dvp->v_mount, dvp, vap, cnp->cn_cred, NULL,
939 ap->a_vpp);
940 if (error)
941 goto out;
942 error = vn_lock(*ap->a_vpp, LK_EXCLUSIVE);
943 if (error) {
944 vrele(*ap->a_vpp);
945 *ap->a_vpp = NULL;
946 goto out;
947 }
948 error = UFS_WAPBL_BEGIN(ap->a_dvp->v_mount);
949 if (error) {
950 vput(*ap->a_vpp);
951 goto out;
952 }

954 tvp = *ap->a_vpp;
955 ip = VTOI(tvp);
956 ip->i_flag |= IN_ACCESS | IN_CHANGE | IN_UPDATE;
957 ip->i_nlink = 2;
958 DIP_ASSIGN(ip, nlink, 2);
959 if (cnp->cn_flags & ISWHITEOUT) {
960 ip->i_flags |= UF_OPAQUE;
961 DIP_ASSIGN(ip, flags, ip->i_flags);
962 }
963
964 /*
965 * Bump link count in parent directory to reflect work done below.
966 * Should be done before reference is created so cleanup is
967 * possible if we crash.
968 */
969 dp->i_nlink++;
970 DIP_ASSIGN(dp, nlink, dp->i_nlink);
971 dp->i_flag |= IN_CHANGE;
972 if ((error = UFS_UPDATE(dvp, NULL, NULL, UPDATE_DIROP)) != 0)
973 goto bad;
974
975 /*
976 * Initialize directory with "." and ".." from static template.
977 */
978 dirtemplate = mastertemplate;
979 dirtemplate.dotdot_reclen = dirblksiz - dirtemplate.dot_reclen;
980 dirtemplate.dot_ino = ufs_rw32(ip->i_number, UFS_MPNEEDSWAP(ump));
981 dirtemplate.dotdot_ino = ufs_rw32(dp->i_number, UFS_MPNEEDSWAP(ump));
982 dirtemplate.dot_reclen = ufs_rw16(dirtemplate.dot_reclen,
983 UFS_MPNEEDSWAP(ump));
984 dirtemplate.dotdot_reclen = ufs_rw16(dirtemplate.dotdot_reclen,
985 UFS_MPNEEDSWAP(ump));
986 if (ump->um_maxsymlinklen <= 0) {
987#if BYTE_ORDER == LITTLE_ENDIAN
988 if (UFS_MPNEEDSWAP(ump) == 0)
989#else
990 if (UFS_MPNEEDSWAP(ump) != 0)
991#endif
992 {
993 dirtemplate.dot_type = dirtemplate.dot_namlen;
994 dirtemplate.dotdot_type = dirtemplate.dotdot_namlen;
995 dirtemplate.dot_namlen = dirtemplate.dotdot_namlen = 0;
996 } else
997 dirtemplate.dot_type = dirtemplate.dotdot_type = 0;
998 }
999 if ((error = UFS_BALLOC(tvp, (off_t)0, dirblksiz, cnp->cn_cred,
1000 B_CLRBUF, &bp)) != 0)
1001 goto bad;
1002 ip->i_size = dirblksiz;
1003 DIP_ASSIGN(ip, size, dirblksiz);
1004 ip->i_flag |= IN_ACCESS | IN_CHANGE | IN_UPDATE;
1005 uvm_vnp_setsize(tvp, ip->i_size);
1006 memcpy((void *)bp->b_data, (void *)&dirtemplate, sizeof dirtemplate);
1007
1008 /*
1009 * Directory set up, now install its entry in the parent directory.
1010 * We must write out the buffer containing the new directory body
1011 * before entering the new name in the parent.
1012 */
1013 if ((error = VOP_BWRITE(bp->b_vp, bp)) != 0)
1014 goto bad;
1015 if ((error = UFS_UPDATE(tvp, NULL, NULL, UPDATE_DIROP)) != 0) {
1016 goto bad;
1017 }
1018 newdir = pool_cache_get(ufs_direct_cache, PR_WAITOK);
1019 ufs_makedirentry(ip, cnp, newdir);
1020 error = ufs_direnter(dvp, ulr, tvp, newdir, cnp, bp);
1021 pool_cache_put(ufs_direct_cache, newdir);
1022 bad:
1023 if (error == 0) {
1024 VN_KNOTE(dvp, NOTE_WRITE | NOTE_LINK);
1025 VOP_UNLOCK(tvp);
1026 UFS_WAPBL_END(dvp->v_mount);
1027 } else {
1028 dp->i_nlink--;
1029 DIP_ASSIGN(dp, nlink, dp->i_nlink);
1030 dp->i_flag |= IN_CHANGE;
1031 UFS_WAPBL_UPDATE(dvp, NULL, NULL, UPDATE_DIROP);
1032 /*
1033 * No need to do an explicit UFS_TRUNCATE here, vrele will
1034 * do this for us because we set the link count to 0.
1035 */
1036 ip->i_nlink = 0;
1037 DIP_ASSIGN(ip, nlink, 0);
1038 ip->i_flag |= IN_CHANGE;
1039 UFS_WAPBL_UPDATE(tvp, NULL, NULL, UPDATE_DIROP);
1040 UFS_WAPBL_END(dvp->v_mount);
1041 vput(tvp);
1042 }
1043 out:
1044 return (error);
1045}

1086 * directory and thus be non-empty.)
1087 */

1104 * inode. If we crash in between, the directory

1106 */

1118 */

1120 DIP_ASSIGN(dp, nlink, dp->i_nlink);
1121 dp->i_flag |= IN_CHANGE;
1122 UFS_WAPBL_UPDATE(dvp, NULL, NULL, UPDATE_DIROP);
1123 ip->i_nlink--;
1124 DIP_ASSIGN(ip, nlink, ip->i_nlink);
1125 ip->i_flag |= IN_CHANGE;

1129 * Unlock the log while we still have reference to unlinked
1130 * directory vp so that it will not get locked for recycling

1133#ifdef UFS_DIRHASH
1134 if (ip->i_dirhash != NULL)
1135 ufsdirhash_free(ip);
1136#endif
1137 out:
1138 VN_KNOTE(vp, NOTE_DELETE);
1139 vput(vp);

1207/*
1208 * Vnode op for reading directories.
1209 *
1210 * This routine handles converting from the on-disk directory format
1211 * "struct direct" to the in-memory format "struct dirent" as well as
1212 * byte swapping the entries if necessary.
1213 */
1214int
1215ufs_readdir(void *v)
1216{
1217 struct vop_readdir_args /* {
1218 struct vnode *a_vp;
1219 struct uio *a_uio;
1220 kauth_cred_t a_cred;
1221 int *a_eofflag;
1222 off_t **a_cookies;
1223 int *a_ncookies;
1224 } */ *ap = v;
1225
1226 /* vnode and fs */
1227 struct vnode *vp = ap->a_vp;
1228 struct ufsmount *ump = VFSTOUFS(vp->v_mount);
1229 int nswap = UFS_MPNEEDSWAP(ump);
1230#if BYTE_ORDER == LITTLE_ENDIAN
1231 int needswap = ump->um_maxsymlinklen <= 0 && nswap == 0;
1232#else
1233 int needswap = ump->um_maxsymlinklen <= 0 && nswap != 0;
1234#endif
1235 /* caller's buffer */
1236 struct uio *calleruio = ap->a_uio;
1237 off_t startoffset, endoffset;
1238 size_t callerbytes;
1239 off_t curoffset;
1240 /* dirent production buffer */
1241 char *direntbuf;
1242 size_t direntbufmax;
1243 struct dirent *dirent, *stopdirent;
1244 /* output cookies array */
1245 off_t *cookies;
1246 size_t numcookies, maxcookies;
1247 /* disk buffer */
1248 off_t physstart, physend;
1249 size_t skipstart, dropend;
1250 char *rawbuf;
1251 size_t rawbufmax, rawbytes;
1252 struct uio rawuio;
1253 struct iovec rawiov;
1254 struct direct *rawdp, *stoprawdp;
1255 /* general */
1256 int error;
1257
1258 KASSERT(VOP_ISLOCKED(vp));
1259
1260 /*
1261 * Figure out where the user wants us to read and how much.
1262 *
1263 * XXX: there should probably be an upper bound on callerbytes
1264 * to avoid silliness trying to do large kernel allocations.
1265 */
1266 callerbytes = calleruio->uio_resid;
1267 startoffset = calleruio->uio_offset;
1268 endoffset = startoffset + callerbytes;
1269
1270 if (callerbytes < _DIRENT_MINSIZE(dirent)) {
1271 /* no room for even one struct dirent */
1272 return EINVAL;
1273 }
1274
1275 /*
1276 * Now figure out where to actually start reading. Round the
1277 * start down to a block boundary: we need to start at the
1278 * beginning of a block in order to read the directory
1279 * correctly.
1280 *
1281 * We also want to always read a whole number of blocks so
1282 * that the copying code below doesn't have to worry about
1283 * partial entries. (It used to try at one point, and was a
1284 * horrible mess.)
1285 *
1286 * Furthermore, since blocks have to be scanned from the
1287 * beginning, if we go partially into another block now we'll
1288 * just have to rescan it on the next readdir call, which
1289 * doesn't really serve any useful purpose.
1290 *
1291 * So, round down the end as well. It's ok to underpopulate
1292 * the transfer buffer, as long as we send back at least one
1293 * dirent so as to avoid giving a bogus EOF indication.
1294 *
1295 * Note that because dirents are larger than ffs struct
1296 * directs, despite the rounding down we may not be able to
1297 * send all the entries in the blocks we read and may have to
1298 * rescan some of them on the next call anyway. Alternatively
1299 * if there's empty space on disk we might have actually been
1300 * able to fit the next block in, and so forth. None of this
1301 * actually matters that much in practice.
1302 *
1303 * XXX: what does ffs do if a directory block becomes
1304 * completely empty, and what happens if all the blocks we
1305 * read are completely empty even though we aren't at EOF? As
1306 * of this writing I (dholland) can't remember the details.
1307 */
1308 physstart = rounddown2(startoffset, ump->um_dirblksiz);
1309 physend = rounddown2(endoffset, ump->um_dirblksiz);
1310
1311 if (physstart >= physend) {
1312 /* Need at least one block */
1313 return EINVAL;
1314 }
1315
1316 /*
1317 * skipstart is the number of bytes we need to read in
1318 * (because we need to start at the beginning of a block) but
1319 * not transfer to the user.
1320 *
1321 * dropend is the number of bytes to ignore at the end of the
1322 * user's buffer.
1323 */
1324 skipstart = startoffset - physstart;
1325 dropend = endoffset - physend;
1326
1327 /*
1328 * Make a transfer buffer.
1329 *
1330 * Note: rawbufmax = physend - physstart. Proof:
1331 *
1332 * physend - physstart = physend - physstart
1333 * = physend - physstart + startoffset - startoffset
1334 * = physend + (startoffset - physstart) - startoffset
1335 * = physend + skipstart - startoffset
1336 * = physend + skipstart - startoffset + endoffset - endoffset
1337 * = skipstart - startoffset + endoffset - (endoffset - physend)
1338 * = skipstart - startoffset + endoffset - dropend
1339 * = skipstart - startoffset + (startoffset + callerbytes) - dropend
1340 * = skipstart + callerbytes - dropend
1341 * = rawbufmax
1342 * Qed.
1343 *
1344 * XXX: this should just use physend - physstart.
1345 *
1346 * XXX: this should be rewritten to read the directs straight
1347 * out of bufferio buffers instead of copying twice. This would
1348 * also let us adapt better to the user's buffer size.
1349 */
1350
1351 /* Base buffer space for CALLERBYTES of new data */
1352 rawbufmax = callerbytes + skipstart;
1353 if (rawbufmax < callerbytes)
1354 return EINVAL;
1355 rawbufmax -= dropend;
1356
1357 if (rawbufmax < _DIRENT_MINSIZE(rawdp)) {
1358 /* no room for even one struct direct */
1359 return EINVAL;
1360 }
1361
1362 /* read it */
1363 rawbuf = kmem_alloc(rawbufmax, KM_SLEEP);
1364 rawiov.iov_base = rawbuf;
1365 rawiov.iov_len = rawbufmax;
1366 rawuio.uio_iov = &rawiov;
1367 rawuio.uio_iovcnt = 1;
1368 rawuio.uio_offset = physstart;
1369 rawuio.uio_resid = rawbufmax;
1370 UIO_SETUP_SYSSPACE(&rawuio);
1371 rawuio.uio_rw = UIO_READ;
1372 error = UFS_BUFRD(vp, &rawuio, 0, ap->a_cred);
1373 if (error != 0) {
1374 kmem_free(rawbuf, rawbufmax);
1375 return error;
1376 }
1377 rawbytes = rawbufmax - rawuio.uio_resid;
1378
1379 /* the raw entries to iterate over */
1380 rawdp = (struct direct *)(void *)rawbuf;
1381 stoprawdp = (struct direct *)(void *)&rawbuf[rawbytes];
1382
1383 /* allocate space to produce dirents into */
1384 direntbufmax = callerbytes;
1385 direntbuf = kmem_alloc(direntbufmax, KM_SLEEP);
1386
1387 /* the dirents to iterate over */
1388 dirent = (struct dirent *)(void *)direntbuf;
1389 stopdirent = (struct dirent *)(void *)&direntbuf[direntbufmax];
1390
1391 /* the output "cookies" (seek positions of directory entries) */
1392 if (ap->a_cookies) {
1393 numcookies = 0;
1394 maxcookies = rawbytes / _DIRENT_RECLEN(rawdp, 1);
1395 cookies = malloc(maxcookies * sizeof(*cookies),
1396 M_TEMP, M_WAITOK);
1397 } else {
1398 /* XXX: GCC */
1399 maxcookies = 0;
1400 cookies = NULL;
1401 }
1402
1403 /* now produce the dirents */
1404 curoffset = calleruio->uio_offset;
1405 while (rawdp < stoprawdp) {
1406 rawdp->d_reclen = ufs_rw16(rawdp->d_reclen, nswap);
1407 if (skipstart > 0) {
1408 /* drain skipstart */
1409 if (rawdp->d_reclen <= skipstart) {
1410 skipstart -= rawdp->d_reclen;
1411 rawdp = _DIRENT_NEXT(rawdp);
1412 continue;
1413 }
1414 /* caller's start position wasn't on an entry */
1415 error = EINVAL;
1416 goto out;
1417 }
1418 if (rawdp->d_reclen == 0) {
1419 struct dirent *save = dirent;
1420 dirent->d_reclen = _DIRENT_MINSIZE(dirent);
1421 dirent = _DIRENT_NEXT(dirent);
1422 save->d_reclen = 0;
1423 rawdp = stoprawdp;
1424 break;
1425 }
1426
1427 /* copy the header */
1428 if (needswap) {
1429 dirent->d_type = rawdp->d_namlen;
1430 dirent->d_namlen = rawdp->d_type;
1431 } else {
1432 dirent->d_type = rawdp->d_type;
1433 dirent->d_namlen = rawdp->d_namlen;
1434 }
1435 dirent->d_reclen = _DIRENT_RECLEN(dirent, dirent->d_namlen);
1436
1437 /* stop if there isn't room for the name AND another header */
1438 if ((char *)(void *)dirent + dirent->d_reclen +
1439 _DIRENT_MINSIZE(dirent) > (char *)(void *)stopdirent)
1440 break;
1441
1442 /* copy the name (and inode (XXX: why after the test?)) */
1443 dirent->d_fileno = ufs_rw32(rawdp->d_ino, nswap);
1444 (void)memcpy(dirent->d_name, rawdp->d_name, dirent->d_namlen);
1445 memset(&dirent->d_name[dirent->d_namlen], 0,
1446 dirent->d_reclen - _DIRENT_NAMEOFF(dirent)
1447 - dirent->d_namlen);
1448
1449 /* onward */
1450 curoffset += rawdp->d_reclen;
1451 if (ap->a_cookies) {
1452 KASSERT(numcookies < maxcookies);
1453 cookies[numcookies++] = curoffset;
1454 }
1455 dirent = _DIRENT_NEXT(dirent);
1456 rawdp = _DIRENT_NEXT(rawdp);
1457 }
1458
1459 /* transfer the dirents to the caller's buffer */
1460 callerbytes = ((char *)(void *)dirent - direntbuf);
1461 error = uiomove(direntbuf, callerbytes, calleruio);
1462
1463out:
1464 calleruio->uio_offset = curoffset;
1465 if (ap->a_cookies) {
1466 if (error) {
1467 free(cookies, M_TEMP);
1468 *ap->a_cookies = NULL;
1469 *ap->a_ncookies = 0;
1470 } else {
1471 *ap->a_cookies = cookies;
1472 *ap->a_ncookies = numcookies;
1473 }
1474 }
1475 kmem_free(direntbuf, direntbufmax);
1476 kmem_free(rawbuf, rawbufmax);
1477 *ap->a_eofflag = VTOI(vp)->i_size <= calleruio->uio_offset;
1478 return error;
1479}

1513 * Calculate the logical to physical mapping if not done already,
1514 * then call the device strategy routine.
1515 */
1516int
1517ufs_strategy(void *v)
1518{
1519 struct vop_strategy_args /* {
1520 struct vnode *a_vp;
1521 struct buf *a_bp;
1522 } */ *ap = v;
1523 struct buf *bp;
1524 struct vnode *vp;
1525 struct inode *ip;
1526 struct mount *mp;

1529 bp = ap->a_bp;
1530 vp = ap->a_vp;
1531 ip = VTOI(vp);
1532 if (vp->v_type == VBLK || vp->v_type == VCHR)
1533 panic("ufs_strategy: spec");
1534 KASSERT(bp->b_bcount != 0);
1535 if (bp->b_blkno == bp->b_lblkno) {

1540 biodone(bp);
1541 return (error);
1542 }
1543 if (bp->b_blkno == -1) /* no valid data */
1544 clrbuf(bp);
1545 }
1546 if (bp->b_blkno < 0) { /* block is not on disk */
1547 biodone(bp);
1548 return (0);
1549 }
1550 vp = ip->i_devvp;
1551
1552 error = VOP_STRATEGY(vp, bp);
1553 if (error)
1554 return error;
1555
1556 if (!BUF_ISREAD(bp))
1557 return 0;
1558
1559 mp = wapbl_vptomp(vp);
1560 if (mp == NULL || mp->mnt_wapbl_replay == NULL ||
1561 !WAPBL_REPLAY_ISOPEN(mp) ||

1563 return 0;
1564
1565 error = biowait(bp);
1566 if (error)
1567 return error;
1568
1569 error = WAPBL_REPLAY_READ(mp, bp->b_data, bp->b_blkno, bp->b_bcount);
1570 if (error) {
1571 mutex_enter(&bufcache_lock);
1572 SET(bp->b_cflags, BC_INVAL);
1573 mutex_exit(&bufcache_lock);
1574 }
1575 return error;
1576}

1850/*
1851 * Allocate a new inode.
1852 */
1853static int
1854ufs_makeinode(struct vattr *vap, struct vnode *dvp,
1855 const struct ufs_lookup_results *ulr,
1856 struct vnode **vpp, struct componentname *cnp)
1857{
1858 struct inode *ip;
1859 struct direct *newdir;
1860 struct vnode *tvp;
1861 int error;
1862
1863 UFS_WAPBL_JUNLOCK_ASSERT(dvp->v_mount);
1864
1865 error = vcache_new(dvp->v_mount, dvp, vap, cnp->cn_cred, NULL, &tvp);
1866 if (error)
1867 return error;
1868 error = vn_lock(tvp, LK_EXCLUSIVE);
1869 if (error) {
1870 vrele(tvp);
1871 return error;
1872 }
1873 *vpp = tvp;
1874 ip = VTOI(tvp);
1875 error = UFS_WAPBL_BEGIN(dvp->v_mount);
1876 if (error) {
1877 vput(tvp);
1878 return (error);
1879 }
1880 ip->i_flag |= IN_ACCESS | IN_CHANGE | IN_UPDATE;
1881 ip->i_nlink = 1;
1882 DIP_ASSIGN(ip, nlink, 1);
1883
1884 /* Authorize setting SGID if needed. */
1885 if (ip->i_mode & ISGID) {
1886 error = kauth_authorize_vnode(cnp->cn_cred, KAUTH_VNODE_WRITE_SECURITY,
1887 tvp, NULL, genfs_can_chmod(tvp->v_type, cnp->cn_cred, ip->i_uid,
1888 ip->i_gid, MAKEIMODE(vap->va_type, vap->va_mode)));
1889 if (error) {
1890 ip->i_mode &= ~ISGID;
1891 DIP_ASSIGN(ip, mode, ip->i_mode);
1892 }
1893 }
1894
1895 if (cnp->cn_flags & ISWHITEOUT) {
1896 ip->i_flags |= UF_OPAQUE;
1897 DIP_ASSIGN(ip, flags, ip->i_flags);
1898 }
1899
1900 /*
1901 * Make sure inode goes to disk before directory entry.
1902 */
1903 if ((error = UFS_UPDATE(tvp, NULL, NULL, UPDATE_DIROP)) != 0)
1904 goto bad;
1905 newdir = pool_cache_get(ufs_direct_cache, PR_WAITOK);
1906 ufs_makedirentry(ip, cnp, newdir);
1907 error = ufs_direnter(dvp, ulr, tvp, newdir, cnp, NULL);
1908 pool_cache_put(ufs_direct_cache, newdir);
1909 if (error)
1910 goto bad;
1911 *vpp = tvp;
1912 return (0);
1913
1914 bad:
1915 /*
1916 * Write error occurred trying to update the inode
1917 * or the directory so must deallocate the inode.
1918 */
1919 ip->i_nlink = 0;
1920 DIP_ASSIGN(ip, nlink, 0);
1921 ip->i_flag |= IN_CHANGE;
1922 UFS_WAPBL_UPDATE(tvp, NULL, NULL, 0);
1923 UFS_WAPBL_END(dvp->v_mount);
1924 vput(tvp);
1925 return (error);
1926}

1928/*
1929 * Allocate len bytes at offset off.
1930 */
1931int
1932ufs_gop_alloc(struct vnode *vp, off_t off, off_t len, int flags,
1933 kauth_cred_t cred)
1934{
1935 struct inode *ip = VTOI(vp);
1936 int error, delta, bshift, bsize;
1937 UVMHIST_FUNC("ufs_gop_alloc"); UVMHIST_CALLED(ubchist);
1938
1939 error = 0;
1940 bshift = vp->v_mount->mnt_fs_bshift;
1941 bsize = 1 << bshift;
1942
1943 delta = off & (bsize - 1);
1944 off -= delta;
1945 len += delta;
1946
1947 while (len > 0) {
1948 bsize = MIN(bsize, len);
1949
1950 error = UFS_BALLOC(vp, off, bsize, cred, flags, NULL);
1951 if (error) {
1952 goto out;
1953 }
1954
1955 /*
1956 * increase file size now, UFS_BALLOC() requires that
1957 * EOF be up-to-date before each call.
1958 */
1959
1960 if (ip->i_size < off + bsize) {
1961 UVMHIST_LOG(ubchist, "vp %#jx old 0x%jx new 0x%x",
1962 (uintptr_t)vp, ip->i_size, off + bsize, 0);
1963 ip->i_size = off + bsize;
1964 DIP_ASSIGN(ip, size, ip->i_size);
1965 }
1966
1967 off += bsize;
1968 len -= bsize;
1969 }
1970
1971out:
1972 UFS_WAPBL_UPDATE(vp, NULL, NULL, 0);
1973 return error;
1974}

1998int
1999ufs_bufio(enum uio_rw rw, struct vnode *vp, void *buf, size_t len, off_t off,
2000 int ioflg, kauth_cred_t cred, size_t *aresid, struct lwp *l)
2001{
2002 struct iovec iov;
2003 struct uio uio;
2004 int error;
2005
2006 KASSERT(ISSET(ioflg, IO_NODELOCKED));
2007 KASSERT(VOP_ISLOCKED(vp));
2008 KASSERT(rw != UIO_WRITE || VOP_ISLOCKED(vp) == LK_EXCLUSIVE);
2009 KASSERT(rw != UIO_WRITE || vp->v_mount->mnt_wapbl == NULL ||
2010 ISSET(ioflg, IO_JOURNALLOCKED));
