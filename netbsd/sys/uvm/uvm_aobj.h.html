39#ifndef _UVM_UVM_AOBJ_H_
40#define _UVM_UVM_AOBJ_H_
41
42/*
43 * Flags for uao_create: UAO_FLAG_KERNOBJ and UAO_FLAG_KERNSWAP are
44 * used only once, to initialise UVM.
45 */
46#define UAO_FLAG_KERNOBJ 0x1 /* create kernel object */
47#define UAO_FLAG_KERNSWAP 0x2 /* enable kernel swap */
48#define UAO_FLAG_NOSWAP 0x8 /* aobj may not swap */
49
50#ifdef _KERNEL
51#if defined(_KERNEL_OPT)
52#include "opt_vmswap.h"
53#endif
54
55void uao_init(void);
56int uao_set_swslot(struct uvm_object *, int, int);
57
58#if defined(VMSWAP)
59int uao_find_swslot(struct uvm_object *, int);
60void uao_dropswap(struct uvm_object *, int);
61bool uao_swap_off(int, int);
62void uao_dropswap_range(struct uvm_object *, voff_t, voff_t);
63#else
64#define uao_find_swslot(obj, off) (__USE(off), 0)
65#define uao_dropswap(obj, off) /* nothing */
66#define uao_dropswap_range(obj, lo, hi) /* nothing */
67#endif
68
69#endif /* _KERNEL */
70
71#endif /* _UVM_UVM_AOBJ_H_ */
