30#ifndef _UVM_UVM_FAULT_H_
31#define _UVM_UVM_FAULT_H_
32
33/*
34 * fault data structures
35 */
36
37/*
38 * uvm_faultinfo: to load one of these fill in all orig_* fields and
39 * then call uvmfault_lookup on it.
40 */
41
42
43struct uvm_faultinfo {
44 struct vm_map *orig_map; /* IN: original map */
45 vaddr_t orig_rvaddr; /* IN: original rounded VA */
46 vsize_t orig_size; /* IN: original size of interest */
47 struct vm_map *map; /* map (could be a submap) */
48 unsigned int mapv; /* map's version number */
49 struct vm_map_entry *entry; /* map entry (from 'map') */
50 vsize_t size; /* size of interest */
51};
52
53#ifdef _KERNEL
54

56 * fault prototypes
57 */

63void uvm_fault_unwire(struct vm_map *, vaddr_t, vaddr_t);
64void uvm_fault_unwire_locked(struct vm_map *, vaddr_t, vaddr_t);
65
66#endif /* _KERNEL */
