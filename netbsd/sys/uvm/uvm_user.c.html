31 * uvm_user.c: high level uvm_allocate/uvm_deallocate interface into vm.
32 */

34#include <sys/cdefs.h>

36
37#include <sys/param.h>
38#include <sys/systm.h>
39#include <sys/proc.h>
40
41#include <uvm/uvm.h>
42
43/*
44 * uvm_deallocate: deallocate memory (unmap)
45 */
46
47void
48uvm_deallocate(struct vm_map *map, vaddr_t start, vsize_t size)
49{
50
51 if (size == 0)
52 return;
53
54 uvm_unmap(map, trunc_page(start), round_page(start + size));
55}
