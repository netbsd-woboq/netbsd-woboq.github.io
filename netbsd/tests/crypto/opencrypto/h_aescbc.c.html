120int
121main(void)
122{
123 int fd, res;
124 size_t i;
125 struct session_op cs;
126 struct crypt_op co;
127 unsigned char buf[64];
128
129 for (i = 0; i < __arraycount(tests); i++) {
130 fd = open("/dev/crypto", O_RDWR, 0);
131 if (fd < 0)
132 err(1, "open %zu", i);
133 memset(&cs, 0, sizeof(cs));
134 cs.cipher = CRYPTO_AES_CBC;
135 cs.keylen = tests[i].key_len;
136 cs.key = __UNCONST(&tests[i].key);
137 res = ioctl(fd, CIOCGSESSION, &cs);
138 if (res < 0)
139 err(1, "CIOCGSESSION %zu", i);
140
141 memset(&co, 0, sizeof(co));
142 co.ses = cs.ses;
143 co.op = COP_ENCRYPT;
144 co.len = tests[i].len;
145 co.src = __UNCONST(&tests[i].plaintx);
146 co.dst = buf;
147 co.dst_len = sizeof(buf);
148 co.iv = __UNCONST(&tests[i].iv);
149 res = ioctl(fd, CIOCCRYPT, &co);
150 if (res < 0)
151 err(1, "CIOCCRYPT %zu", i);
152
153 if (memcmp(co.dst, tests[i].ciphertx, tests[i].len)) {
154 size_t j;
155 for (j = 0; j < tests[i].len; j++)
156 printf("0x%2zu: 0x%2x 0x%2x\n", j,
157 buf[j], tests[i].ciphertx[j]);
158 errx(1, "verification failed %zu", i);
159 }
160 close(fd);
161 }
162 return 0;
163}
