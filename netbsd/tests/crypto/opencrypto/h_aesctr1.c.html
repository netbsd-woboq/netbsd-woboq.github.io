206int
207main(void)
208{
209 int fd, res;
210 size_t i;
211 struct session_op cs;
212 struct crypt_op co;
213 unsigned char buf[36];
214
215 for (i = 0; i < __arraycount(tests); i++) {
216 fd = open("/dev/crypto", O_RDWR, 0);
217 if (fd < 0)
218 err(1, "open %zu", i);
219 memset(&cs, 0, sizeof(cs));
220 cs.cipher = CRYPTO_AES_CTR;
221 cs.keylen = tests[i].key_len;
222 cs.key = __UNCONST(&tests[i].key);
223 res = ioctl(fd, CIOCGSESSION, &cs);
224 if (res < 0)
225 err(1, "CIOCGSESSION %zu", i);
226
227 memset(&co, 0, sizeof(co));
228 co.ses = cs.ses;
229 co.op = COP_ENCRYPT;
230 co.len = tests[i].len;
231 co.src = __UNCONST(&tests[i].plaintx);
232 co.dst = buf;
233 co.dst_len = sizeof(buf);
234 co.iv = __UNCONST(&tests[i].iv);
235 res = ioctl(fd, CIOCCRYPT, &co);
236 if (res < 0)
237 err(1, "CIOCCRYPT %zu", i);
238
239 if (memcmp(co.dst, tests[i].ciphertx, tests[i].len)) {
240 size_t j;
241 printf(" Loc Actual Golden\n");
242 for (j = 0; j < tests[i].len; j++)
243 printf("0x%2zu: 0x%2x 0x%2x\n", j,
244 buf[j], tests[i].ciphertx[j]);
245 warnx("verification failed %zu", i);
246 }
247 close(fd);
248 }
249 return 0;
250}
