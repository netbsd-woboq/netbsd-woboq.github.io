29#include <err.h>
30#include <fcntl.h>
31#include <stdio.h>
32#include <string.h>
33
34#include <sys/ioctl.h>
35#include <sys/time.h>
36
37#include <crypto/cryptodev.h>
38
39unsigned char key[] = "abcdefgh";
40unsigned char iv[8] = {0};
41char plaintx[16] = "1234567890123456";
42const unsigned char ciphertx[16] = {
43 0x21, 0xc6, 0x0d, 0xa5, 0x34, 0x24, 0x8b, 0xce,
44 0x95, 0x86, 0x64, 0xb3, 0x66, 0x77, 0x9b, 0x4c
45};
46
47int
48main(void)
49{
50 int fd, res;
51 struct session_op cs;
52 struct crypt_op co;
53 unsigned char buf[16];
54
55 fd = open("/dev/crypto", O_RDWR, 0);
56 if (fd < 0)
57 err(1, "open");
58 memset(&cs, 0, sizeof(cs));
59 cs.cipher = CRYPTO_DES_CBC;
60 cs.keylen = 8;
61 cs.key = key;
62 res = ioctl(fd, CIOCGSESSION, &cs);
63 if (res < 0)
64 err(1, "CIOCGSESSION");
65
66 memset(&co, 0, sizeof(co));
67 co.ses = cs.ses;
68 co.op = COP_ENCRYPT;
69 co.len = sizeof(plaintx);
70 co.src = plaintx;
71 co.dst = buf;
72 co.dst_len = sizeof(buf);
73 co.iv = iv;
74 res = ioctl(fd, CIOCCRYPT, &co);
75 if (res < 0)
76 err(1, "CIOCCRYPT");
77
78 if (memcmp(co.dst, ciphertx, sizeof(ciphertx)))
79 errx(1, "verification failed");
80
81 return 0;
82}
