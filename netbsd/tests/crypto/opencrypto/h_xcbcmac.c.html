40/* test vectors from RFC3566 */
41unsigned char key[16] = {
42 0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07,
43 0x08, 0x09, 0x0a, 0x0b, 0x0c, 0x0d, 0x0e, 0x0f
44};
45char plaintx[1000] = {
46 0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07,
47 0x08, 0x09, 0x0a, 0x0b, 0x0c, 0x0d, 0x0e, 0x0f,
48 0x10, 0x11, 0x12, 0x13, 0x14, 0x15, 0x16, 0x17,
49 0x18, 0x19, 0x1a, 0x1b, 0x1c, 0x1d, 0x1e, 0x1f,
50 0x20, 0x21
51};
52const struct {
53 size_t len;
54 unsigned char mac[12];
55} tests[] = {
56 { 0, { 0x75, 0xf0, 0x25, 0x1d, 0x52, 0x8a,
57 0xc0, 0x1c, 0x45, 0x73, 0xdf, 0xd5 } },
58 { 3, { 0x5b, 0x37, 0x65, 0x80, 0xae, 0x2f,
59 0x19, 0xaf, 0xe7, 0x21, 0x9c, 0xee } },
60 { 16, { 0xd2, 0xa2, 0x46, 0xfa, 0x34, 0x9b,
61 0x68, 0xa7, 0x99, 0x98, 0xa4, 0x39 } },
62 { 20, { 0x47, 0xf5, 0x1b, 0x45, 0x64, 0x96,
63 0x62, 0x15, 0xb8, 0x98, 0x5c, 0x63 } },
64 { 32, { 0xf5, 0x4f, 0x0e, 0xc8, 0xd2, 0xb9,
65 0xf3, 0xd3, 0x68, 0x07, 0x73, 0x4b } },
66 { 34, { 0xbe, 0xcb, 0xb3, 0xbc, 0xcd, 0xb5,
67 0x18, 0xa3, 0x06, 0x77, 0xd5, 0x48 } },
68 { 1000, { 0xf0, 0xda, 0xfe, 0xe8, 0x95, 0xdb,
69 0x30, 0x25, 0x37, 0x61, 0x10, 0x3b } },
70};
71
72int
73main(void)
74{
75 int fd, res;
76 size_t i;
77 struct session_op cs;
78 struct crypt_op co;
79 unsigned char buf[16];
80
81 fd = open("/dev/crypto", O_RDWR, 0);
82 if (fd < 0)
83 err(1, "open");
84 memset(&cs, 0, sizeof(cs));
85 cs.mac = CRYPTO_AES_XCBC_MAC_96;
86 cs.mackeylen = sizeof(key);
87 cs.mackey = key;
88 res = ioctl(fd, CIOCGSESSION, &cs);
89 if (res < 0)
90 err(1, "CIOCGSESSION");
91
92 for (i = 0; i < __arraycount(tests); i++) {
93 memset(&co, 0, sizeof(co));
94 memset(buf, 0, sizeof(buf));
95 if (tests[i].len == sizeof(plaintx))
96 memset(&plaintx, 0, sizeof(plaintx));
97 co.ses = cs.ses;
98 co.op = COP_ENCRYPT;
99 co.len = tests[i].len;
100 co.src = plaintx;
101 co.mac = buf;
102 res = ioctl(fd, CIOCCRYPT, &co);
103 if (res < 0)
104 err(1, "CIOCCRYPT test %zu", i);
105 if (memcmp(buf, &tests[i].mac, sizeof(tests[i].mac)))
106 errx(1, "verification failed test %zu", i);
107 }
108 return 0;
109}
