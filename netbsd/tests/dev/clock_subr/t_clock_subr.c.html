260ATF_TC_BODY(ymdhms_to_secs, tc)
261{
262 time_t secs;
263 size_t i;
264
265 for (i = 0; i < __arraycount(clock_tests); i++) {
266 secs = clock_ymdhms_to_secs(__UNCONST(&clock_tests[i].clock));
267 ATF_CHECK_EQ_MSG(clock_tests[i].time, secs, "%jd != %jd",
268 (intmax_t)clock_tests[i].time, (intmax_t)secs);
269 }
270}

272ATF_TC(secs_to_ymdhms);
273ATF_TC_HEAD(secs_to_ymdhms, tc)
274{
275
276 atf_tc_set_md_var(tc, "descr", "check clock_secs_to_ymdhms");
277}
278ATF_TC_BODY(secs_to_ymdhms, tc)
279{
280 struct clock_ymdhms ymdhms;
281 size_t i;
282
283#define CHECK_FIELD(f) \
284 ATF_CHECK_EQ_MSG(ymdhms.dt_##f, clock_tests[i].clock.dt_##f, \
285 "%jd != %jd for %jd", (intmax_t)ymdhms.dt_##f, \
286 (intmax_t)clock_tests[i].clock.dt_##f, \
287 (intmax_t)clock_tests[i].time)
288
289 for (i = 0; i < __arraycount(clock_tests); i++) {
290 clock_secs_to_ymdhms(clock_tests[i].time, &ymdhms);
291 CHECK_FIELD(year);
292 CHECK_FIELD(mon);
293 CHECK_FIELD(day);
294 CHECK_FIELD(wday);
295 CHECK_FIELD(hour);
296 CHECK_FIELD(min);
297 CHECK_FIELD(sec);
298 }
299#undef CHECK_FIELD
300}
