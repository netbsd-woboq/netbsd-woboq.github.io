44ATF_TC(noisyeject);
45ATF_TC_HEAD(noisyeject, tc)
46{
47
48 atf_tc_set_md_var(tc, "descr", "test for CD eject noisyness "
49 "(PR kern/43785)");
50}
51
52ATF_TC_BODY(noisyeject, tc)
53{
54 static char fname[] = "/dev/rcd0_";
55 int part, fd, arg = 0;
56
57 RL(part = getrawpartition());
58 fname[strlen(fname)-1] = 'a' + part;
59 rump_init();
60 /*
61 * Rump CD emulation has been fixed, so no longer a problem.
62 *
63 atf_tc_expect_signal(SIGSEGV, "PR kern/47646: Broken test or "
64 "a real problem in rump or the driver");
65 */
66 RL(fd = rump_sys_open(fname, O_RDWR));
67 RL(rump_sys_ioctl(fd, DIOCEJECT, &arg));
68
69 ATF_REQUIRE_EQ(rump_scsitest_err[RUMP_SCSITEST_NOISYSYNC], 0);
70 RL(rump_sys_close(fd));
71 // atf_tc_expect_fail("PR kern/43785");
72 ATF_REQUIRE_EQ(rump_scsitest_err[RUMP_SCSITEST_NOISYSYNC], 0);
73}
74
75ATF_TP_ADD_TCS(tp)
76{
77
78 ATF_TP_ADD_TC(tp, noisyeject);
79
80 return atf_no_error();
81}
