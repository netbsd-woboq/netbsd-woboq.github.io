48struct ext2fstestargs {
49 struct ufs_args ta_uargs;
50 char ta_devpath[MAXPATHLEN];
51 char ta_imgpath[MAXPATHLEN];
52};
53
54int
55ext2fs_fstest_newfs(const atf_tc_t *tc, void **buf, const char *image,
56 off_t size, void *fspriv)
57{
58 char cmd[1024];
59 int res;
60 static unsigned int num = 0;
61 struct ext2fstestargs *args;
62
63 size /= 512;
64 snprintf(cmd, 1024, "newfs_ext2fs -F -s %"PRId64" %s >/dev/null",
65 size, image);
66 res = system(cmd);
67 if (res != 0)
68 return res;
69
70 res = rump_init();
71 if (res != 0)
72 return res;
73
74 args = calloc(1, sizeof(*args));
75 if (args == NULL)
76 return -1;
77
78 snprintf(args->ta_devpath, MAXPATHLEN, "/dev/device%d.ext2fs", num);
79 snprintf(args->ta_imgpath, MAXPATHLEN, "%s", image);
80 args->ta_uargs.fspec = args->ta_devpath;
81
82 res = rump_pub_etfs_register(args->ta_devpath, image, RUMP_ETFS_BLK);
83 if (res != 0) {
84 free(args);
85 return res;
86 }
87
88 *buf = args;
89 num++;
90
91 return res;
92}
93
94int
95ext2fs_fstest_delfs(const atf_tc_t *tc, void *buf)
96{
97 int res;
98 struct ext2fstestargs *args = buf;
99
100 res = rump_pub_etfs_remove(args->ta_devpath);
101 if (res != 0)
102 return res;
103
104 res = unlink(args->ta_imgpath);
105 if (res != 0)
106 return res;
107
108 free(args);
109
110 return 0;
111}
112
113int
114ext2fs_fstest_mount(const atf_tc_t *tc, void *buf, const char *path, int flags)
115{
116 int res;
117 struct ext2fstestargs *args = buf;
118
119 res = rump_sys_mkdir(path, 0777);
120 if (res == -1)
121 return res;
122
123 res = rump_sys_mount(MOUNT_EXT2FS, path, flags, &args->ta_uargs,
124 sizeof(args->ta_uargs));
125 return res;
126}
127
128int
129ext2fs_fstest_unmount(const atf_tc_t *tc, const char *path, int flags)
130{
131 int res;
132
133 res = rump_sys_unmount(path, flags);
134 if (res == -1)
135 return res;
136
137 res = rump_sys_rmdir(path);
138 return res;
139}
