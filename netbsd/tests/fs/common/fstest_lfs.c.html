51struct lfstestargs {
52 struct ufs_args ta_uargs;
53 pthread_t ta_cleanerthread;
54 sem_t ta_cleanerloop;
55 char ta_devpath[MAXPATHLEN];
56 char ta_imgpath[MAXPATHLEN];
57 char ta_mntpath[MAXPATHLEN];
58 char ta_hostpath[MAXPATHLEN];
59};
60
61int
62lfs_fstest_newfs(const atf_tc_t *tc, void **buf, const char *image, off_t size,
63 void *fspriv)
64{
65 char cmd[1024];
66 int res;
67 static unsigned int num = 0;
68 struct lfstestargs *args;
69
70 size /= 512;
71 snprintf(cmd, 1024, "newfs_lfs -D -F -s %"PRId64" ./%s >/dev/null",
72 size, image);
73 res = system(cmd);
74 if (res != 0)
75 return res;
76
77 res = rump_init();
78 if (res != 0)
79 return res;
80
81 args = calloc(1, sizeof(*args));
82 if (args == NULL)
83 return -1;
84
85 strcpy(args->ta_hostpath, image);
86 snprintf(args->ta_devpath, MAXPATHLEN, "/dev/device%d.lfs", num);
87 snprintf(args->ta_imgpath, MAXPATHLEN, "%s", image);
88 args->ta_uargs.fspec = args->ta_devpath;
89 sem_init(&args->ta_cleanerloop, 0, 0);
90
91 res = rump_pub_etfs_register(args->ta_devpath, image, RUMP_ETFS_BLK);
92 if (res != 0) {
93 free(args);
94 return res;
95 }
96
97 *buf = args;
98 num++;
99
100 return 0;
101}

103int
104lfs_fstest_delfs(const atf_tc_t *tc, void *buf)
105{
106 int res;
107 struct lfstestargs *args = buf;
108
109 res = rump_pub_etfs_remove(args->ta_devpath);
110 if (res != 0)
111 return res;
112
113 res = unlink(args->ta_imgpath);
114 if (res != 0)
115 return res;
116
117 pthread_join(args->ta_cleanerthread, NULL);
118 free(args);
119
120 return 0;
121}

123static void *
124cleaner(void *arg)
125{
126 char thepath[MAXPATHLEN];
127 struct lfstestargs *args = arg;
128 const char *the_argv[7];
129 char buf[64];
130
131 rump_pub_lwproc_newlwp(rump_sys_getpid());
132
133 /* this inspired by the cleaner code. fixme */
134 sprintf(thepath, "/dev/r%s", args->ta_devpath+5);
135 rump_pub_etfs_register(thepath, args->ta_hostpath, RUMP_ETFS_CHR);
136 sprintf(buf, "%p", &args->ta_cleanerloop);
137
138 the_argv[0] = "megamaid";
139 the_argv[1] = "-D"; /* don't fork() & detach */
140 the_argv[2] = "-S";
141 the_argv[3] = buf;
142 the_argv[4] = args->ta_mntpath;
143 the_argv[5] = NULL;
144
145 /* xxxatf */
146 optind = 1;
147 opterr = 1;
148
149 lfs_cleaner_main(5, __UNCONST(the_argv));
150
151 rump_pub_lwproc_releaselwp();
152
153 return NULL;
154}

156int
157lfs_fstest_mount(const atf_tc_t *tc, void *buf, const char *path, int flags)
158{
159 struct lfstestargs *args = buf;
160 int res;
161
162 res = rump_sys_mkdir(path, 0777);
163 if (res == -1)
164 return res;
165
166 res = rump_sys_mount(MOUNT_LFS, path, flags, &args->ta_uargs,
167 sizeof(args->ta_uargs));
168 if (res == -1)
169 return res;
170
171 strcpy(args->ta_mntpath, path);
172 res = pthread_create(&args->ta_cleanerthread, NULL, cleaner, args);
173 if (res)
174 return res;
175
176 /* wait for cleaner to initialize */
177 sem_wait(&args->ta_cleanerloop);
178
179 return 0;
180}

182int
183lfs_fstest_unmount(const atf_tc_t *tc, const char *path, int flags)
184{
185 int res;
186
187 res = rump_sys_unmount(path, flags);
188 if (res == -1) {
189 return res;
190 }
191
192 res = rump_sys_rmdir(path);
193 return res;
194}
