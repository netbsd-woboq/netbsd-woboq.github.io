43#include <rump/rump.h>
44#include <rump/rump_syscalls.h>
45
46#include "h_fsmacros.h"
47
48struct tmpfstestargs {
49 struct tmpfs_args ta_uargs;
50};
51
52int
53tmpfs_fstest_newfs(const atf_tc_t *tc, void **buf, const char *image,
54 off_t size, void *fspriv)
55{
56 int res;
57 struct tmpfstestargs *args;
58
59 res = rump_init();
60 if (res != 0)
61 return res;
62
63 args = calloc(1, sizeof(*args));
64 if (args == NULL)
65 return -1;
66
67 args->ta_uargs.ta_version = TMPFS_ARGS_VERSION;
68 args->ta_uargs.ta_root_mode = 0777;
69 args->ta_uargs.ta_size_max = size;
70
71 *buf = args;
72
73 return 0;
74}
75
76int
77tmpfs_fstest_delfs(const atf_tc_t *tc, void *buf)
78{
79 struct tmpfstestargs *args = buf;
80
81 free(args);
82
83 return 0;
84}
85
86int
87tmpfs_fstest_mount(const atf_tc_t *tc, void *buf, const char *path, int flags)
88{
89 int res;
90 struct tmpfstestargs *args = buf;
91
92 res = rump_sys_mkdir(path, 0777);
93 if (res == -1)
94 return res;
95
96 res = rump_sys_mount(MOUNT_TMPFS, path, flags, &args->ta_uargs,
97 sizeof(args->ta_uargs));
98 return res;
99}
100
101int
102tmpfs_fstest_unmount(const atf_tc_t *tc, const char *path, int flags)
103{
104 int res;
105
106 res = rump_sys_unmount(path, flags);
107 if (res == -1)
108 return res;
109
110 res = rump_sys_rmdir(path);
111 return res;
112}
