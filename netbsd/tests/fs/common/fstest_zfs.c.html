46#define SRVPATH "zfssurvo"
47#define SRVURL "unix://" SRVPATH
48#define ZFSDEV "/zfsdev"
49
50int
51zfs_fstest_newfs(const atf_tc_t *tc, void **buf, const char *image,
52 off_t size, void *fspriv)
53{
54 int res;
55 int fd;
56
57 /* XXX: hardcoded zfs minimum size */
58 size = MAX(64*1024*1024, size);
59
60 res = rump_init();
61 if (res != 0) {
62 errno = res;
63 return -1;
64 }
65
66 /* create backing image, sparse file is enough */
67 if ((fd = open(image, O_RDWR | O_CREAT, 0777)) == -1)
68 return -1;
69 if (ftruncate(fd, size) == -1) {
70 close(fd);
71 return -1;
72 }
73 close(fd);
74
75 res = rump_pub_etfs_register(ZFSDEV, image, RUMP_ETFS_BLK);
76 if (res != 0) {
77 errno = res;
78 return -1;
79 }
80
81 res = rump_init_server(SRVURL);
82 if (res != 0) {
83 errno = res;
84 return -1;
85 }
86 *buf = NULL;
87
88 return 0;
89}

99int
100zfs_fstest_mount(const atf_tc_t *tc, void *buf, const char *path, int flags)
101{
102 char tmpbuf[128];
103 int error;
104
105 /* set up the hijack env for running zpool */
106 setenv("RUMP_SERVER", SRVURL, 1);
107 snprintf(tmpbuf, sizeof(tmpbuf)-1,
108 "blanket=/dev/zfs:%s:%s,sysctl=yes,modctl=yes", ZFSDEV, path);
109 setenv("RUMPHIJACK", tmpbuf, 1);
110 setenv("LD_PRELOAD", "/usr/lib/librumphijack.so", 1);
111
112 while (*path == '/')
113 path++;
114
115 /* run zpool create */
116 snprintf(tmpbuf, sizeof(tmpbuf)-1, "zpool create %s %s",
117 path, ZFSDEV);
118 if ((error = system(tmpbuf)) != 0) {
119 errno = error;
120 return -1;
121 }
122
123 return 0;
124}

126int
127zfs_fstest_unmount(const atf_tc_t *tc, const char *path, int flags)
128{
129
130 unmount(path, flags);
131 unlink(SRVPATH);

133 return 0;
