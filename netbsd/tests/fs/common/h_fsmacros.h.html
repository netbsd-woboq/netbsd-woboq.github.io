68#ifndef FSTEST_IMGNAME
69#define FSTEST_IMGNAME "image.fs"
70#endif
71#ifndef FSTEST_IMGSIZE
72#define FSTEST_IMGSIZE (10000 * 512)
73#endif
74#ifndef FSTEST_MNTNAME
75#define FSTEST_MNTNAME "/mnt"
76#endif
77
78#define FSTEST_CONSTRUCTOR(_tc_, _fs_, _args_) \
79do { \
80 if (_fs_##_fstest_newfs(_tc_, &_args_, \
81 FSTEST_IMGNAME, FSTEST_IMGSIZE, NULL) != 0) \
82 atf_tc_fail_errno("newfs failed"); \
83 if (_fs_##_fstest_mount(_tc_, _args_, FSTEST_MNTNAME, 0) != 0) \
84 atf_tc_fail_errno("mount failed"); \
85} while (/*CONSTCOND*/0)
86
87#define FSTEST_CONSTRUCTOR_FSPRIV(_tc_, _fs_, _args_, _privargs_) \
88do { \
89 if (_fs_##_fstest_newfs(_tc_, &_args_, \
90 FSTEST_IMGNAME, FSTEST_IMGSIZE, _privargs_) != 0) \
91 atf_tc_fail_errno("newfs failed"); \
92 if (_fs_##_fstest_mount(_tc_, _args_, FSTEST_MNTNAME, 0) != 0) \
93 atf_tc_fail_errno("mount failed"); \
94} while (/*CONSTCOND*/0)
95
96#define FSTEST_DESTRUCTOR(_tc_, _fs_, _args_) \
97do { \
98 if (_fs_##_fstest_unmount(_tc_, FSTEST_MNTNAME, 0) != 0) { \
99 rump_pub_vfs_mount_print(FSTEST_MNTNAME, 1); \
100 atf_tc_fail_errno("unmount failed"); \
101 } \
102 if (_fs_##_fstest_delfs(_tc_, _args_) != 0) \
103 atf_tc_fail_errno("delfs failed"); \
104} while (/*CONSTCOND*/0)

106#define ATF_TC_FSADD(fs,type,func,desc) \
107 ATF_TC(fs##_##func); \
108 ATF_TC_HEAD(fs##_##func,tc) \
109 { \
110 atf_tc_set_md_var(tc, "descr", type " test for " desc); \
111 atf_tc_set_md_var(tc, "X-fs.type", #fs); \
112 atf_tc_set_md_var(tc, "X-fs.mntname", type); \
113 } \
114 void *fs##func##tmp; \
115 \
116 ATF_TC_BODY(fs##_##func,tc) \
117 { \
118 if (!atf_check_fstype(tc, #fs)) \
119 atf_tc_skip("filesystem not selected"); \
120 FSTEST_CONSTRUCTOR(tc,fs,fs##func##tmp); \
121 func(tc,FSTEST_MNTNAME); \
122 if (fs##_fstest_unmount(tc, FSTEST_MNTNAME, 0) != 0) { \
123 rump_pub_vfs_mount_print(FSTEST_MNTNAME, 1); \
124 atf_tc_fail_errno("unmount failed"); \
125 } \
126 }
127
128#define ATF_TC_FSADD_RO(_fs_,_type_,_func_,_desc_,_gen_) \
129 ATF_TC(_fs_##_##_func_); \
130 ATF_TC_HEAD(_fs_##_##_func_,tc) \
131 { \
132 atf_tc_set_md_var(tc, "descr",_type_" test for "_desc_);\
133 atf_tc_set_md_var(tc, "X-fs.type", #_fs_); \
134 atf_tc_set_md_var(tc, "X-fs.mntname", _type_); \
135 } \
136 void *_fs_##_func_##tmp; \
137 \
138 ATF_TC_BODY(_fs_##_##_func_,tc) \
139 { \
140 if (!atf_check_fstype(tc, #_fs_)) \
141 atf_tc_skip("filesystem not selected"); \
142 FSTEST_CONSTRUCTOR(tc,_fs_,_fs_##_func_##tmp); \
143 _gen_(tc,FSTEST_MNTNAME); \
144 if (_fs_##_fstest_unmount(tc, FSTEST_MNTNAME, 0) != 0) \
145 atf_tc_fail_errno("unmount r/w failed"); \
146 if (_fs_##_fstest_mount(tc, _fs_##_func_##tmp, \
147 FSTEST_MNTNAME, MNT_RDONLY) != 0) \
148 atf_tc_fail_errno("mount ro failed"); \
149 _func_(tc,FSTEST_MNTNAME); \
150 if (_fs_##_fstest_unmount(tc, FSTEST_MNTNAME, 0) != 0) {\
151 rump_pub_vfs_mount_print(FSTEST_MNTNAME, 1); \
152 atf_tc_fail_errno("unmount failed"); \
153 } \
154 }

248static __inline bool
249atf_check_fstype(const atf_tc_t *tc, const char *fs)
250{
251 const char *fstype;
252
253 if (!atf_tc_has_config_var(tc, "fstype"))
254 return true;
255
256 fstype = atf_tc_get_config_var(tc, "fstype");
257 if (strcmp(fstype, fs) == 0)
258 return true;
259 return false;
260}

263 (strcmp(atf_tc_get_md_var(tc, "X-fs.type"), "ext2fs") == 0)
264#define FSTYPE_FFS(tc)\
265 (strcmp(atf_tc_get_md_var(tc, "X-fs.type"), "ffs") == 0)
