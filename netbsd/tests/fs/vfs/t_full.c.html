44/*
45 * Write this much over the image size. This is to force an NFS commit,
46 * since we might just stuff data into the cache and miss the problem.
47 */
48#define NFSBONUS (1<<16)
49
50static void
51fillfs(const atf_tc_t *tc, const char *mp)
52{
53 char buf[8192];
54 size_t written;
55 ssize_t n = 0; /* xxxgcc */
56 size_t bonus;
57 int fd, i = 0;
58
59 if (FSTYPE_P2K_FFS(tc) || FSTYPE_PUFFS(tc) || FSTYPE_RUMPFS(tc) ||
60 FSTYPE_ZFS(tc)) {
61 atf_tc_skip("fs does not support explicit block allocation "
62 "(GOP_ALLOC)");
63 }
64
65 bonus = 0;
66 if (FSTYPE_NFS(tc))
67 bonus = NFSBONUS;
68
69 if (rump_sys_chdir(mp) == -1)
70 atf_tc_fail_errno("chdir mountpoint");
71 fd = rump_sys_open("afile", O_CREAT | O_RDWR, 0600);
72 if (fd == -1)
73 atf_tc_fail_errno("create file");
74
75 for (written = 0; written < FSTEST_IMGSIZE + bonus; written +=n) {
76 memset(buf, i++, sizeof(buf)); /* known garbage */
77 n = rump_sys_write(fd, buf, sizeof(buf));
78 if (n == -1)
79 break;
80 }
81 if (n == -1) {
82 if (errno != ENOSPC)
83 atf_tc_fail_errno("write");
84 } else {
85 atf_tc_fail("filled file system over size limit");
86 }
87
88 rump_sys_close(fd);
89 rump_sys_chdir("/");
90}
91
92ATF_TC_FSAPPLY(fillfs, "fills file system, expects ENOSPC");
93
94ATF_TP_ADD_TCS(tp)
95{
96
97 ATF_TP_FSAPPLY(fillfs);
98
99 return atf_no_error();
100}
