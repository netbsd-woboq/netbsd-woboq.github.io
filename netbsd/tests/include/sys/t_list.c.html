39/*
40 * XXX This is a limited test to make sure the operations behave as
41 * described on a sequential machine. It does nothing to test the
42 * pserialize-safety of any operations.
43 */
44
45ATF_TC(list_move);
46ATF_TC_HEAD(list_move, tc)
47{
48 atf_tc_set_md_var(tc, "descr", "LIST_MOVE verification");
49}
50ATF_TC_BODY(list_move, tc)
51{
52 LIST_HEAD(listhead, entry) old_head, new_head, old_copy;
53 struct entry {
54 LIST_ENTRY(entry) entries;
55 uint64_t value;
56 } *n1, *n2, *n3;
57
58 LIST_INIT(&old_head);
59
60 n1 = malloc(sizeof(struct entry));
61 n1->value = 1;
62 LIST_INSERT_HEAD(&old_head, n1, entries);
63
64 n2 = malloc(sizeof(struct entry));
65 n2->value = 2;
66 LIST_INSERT_HEAD(&old_head, n2, entries);
67
68 LIST_MOVE(&old_head, &new_head, entries);
69
70 memcpy(&old_copy, &old_head, sizeof(old_head));
71
72 n3 = LIST_FIRST(&new_head);
73 ATF_CHECK_MSG(n3->value = 2, "Unexpected value for LIST_FIRST");
74
75 LIST_REMOVE(n3, entries);
76 ATF_CHECK_MSG(memcmp(&old_copy, &old_head, sizeof(old_head)) == 0,
77 "Unexpected modification of old_head during LIST_REMOVE");
78
79 LIST_REMOVE(LIST_FIRST(&new_head), entries);
80 ATF_CHECK_MSG(LIST_EMPTY(&new_head), "New list not empty!");
81}
82
83ATF_TP_ADD_TCS(tp)
84{
85
86 ATF_TP_ADD_TC(tp, list_move);
87
88 return atf_no_error();
89}
