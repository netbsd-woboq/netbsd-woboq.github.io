31#include <sys/cdefs.h>
32__RCSID("$NetBSD: t_glob.c,v 1.1 2011/04/10 08:35:48 jruoho Exp $");
33
34#include <atf-c.h>
35#include <glob.h>
36#include <string.h>
37
38ATF_TC(glob_types);
39ATF_TC_HEAD(glob_types, tc)
40{
41 atf_tc_set_md_var(tc, "descr", "Test glob(3) types");
42}
43
44ATF_TC_BODY(glob_types, tc)
45{
46 glob_t g;
47
48 /*
49 * IEEE Std 1003.1-2008:
50 *
51 * "The <glob.h> header shall define the glob_t structure type,
52 * which shall include at least the following members:
53 *
54 * size_t gl_pathc Count of paths matched by pattern.
55 * char **gl_pathv Pointer to a list of matched pathnames.
56 * size_t gl_offs Slots to reserve at the beginning of gl_pathv."
57 *
58 * Verify that gl_pathc and gl_offs are unsigned; PR standards/21401.
59 */
60 (void)memset(&g, 0, sizeof(glob_t));
61
62 g.gl_offs = g.gl_offs - 1;
63 g.gl_pathc = g.gl_pathc - 1;
64
65 ATF_REQUIRE(g.gl_pathc > 0);
66 ATF_REQUIRE(g.gl_offs > 0);
67}
68
69ATF_TP_ADD_TCS(tp)
70{
71
72 ATF_TP_ADD_TC(tp, glob_types);
73
74 return atf_no_error();
75}
