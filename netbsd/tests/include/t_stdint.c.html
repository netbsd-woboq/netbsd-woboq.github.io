34#include <stdint.h>

36#include <atf-c.h>
37
38ATF_TC_WITHOUT_HEAD(int8);
39ATF_TC_BODY(int8, tc)
40{
41 ATF_CHECK(INT8_MIN < UINT8_MAX);
42}
43
44ATF_TC_WITHOUT_HEAD(int16);
45ATF_TC_BODY(int16, tc)
46{
47 ATF_CHECK(INT16_MIN < UINT16_MAX);
48}
49
50ATF_TP_ADD_TCS(tp)
51{
52
53 ATF_TP_ADD_TC(tp, int8);
54 ATF_TP_ADD_TC(tp, int16);
55
56 return atf_no_error();
57}
