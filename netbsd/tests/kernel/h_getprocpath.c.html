30#include <stdio.h>
31#include <stdlib.h>
32#include <err.h>
33#include <sys/sysctl.h>
34
35static int
36getprocpath(char *path, size_t len, pid_t pid)
37{
38 const int name[] = {
39 CTL_KERN, KERN_PROC_ARGS, pid, KERN_PROC_PATHNAME,
40 };
41 return sysctl(name, __arraycount(name), path, &len, NULL, 0);
42}
43
44int
45main(int argc, char *argv[])
46{
47 if (argc != 2) {
48 fprintf(stderr, "Usage: %s <pid>\n", getprogname());
49 return EXIT_FAILURE;
50 }
51
52 char buf[MAXPATHLEN];
53 pid_t pid = atoi(argv[1]);
54
55 if (getprocpath(buf, sizeof(buf), pid) == -1)
56 err(EXIT_FAILURE, "sysctl: %d", pid);
57 printf("%s\n", buf);
58 return EXIT_SUCCESS;
59}
