34#include <sys/types.h>
35#include <sys/exec.h>
36#include <stdlib.h>
37#include <unistd.h>
38
39extern struct ps_strings *__ps_strings;
40
41int
42main(int argc, char **argv, char **environ)
43{
44 int ret = 0;
45 int nenv;
46
47 if (__ps_strings->ps_nargvstr != argc) {
48 static const char nargv_err[] = "Wrong argc in ps_strings";
49 write(STDOUT_FILENO, nargv_err, sizeof(nargv_err));
50 ret = 1;
51 }
52
53 if (__ps_strings->ps_argvstr != argv) {
54 static const char argv_err[] = "Wrong argv in ps_strings";
55 write(STDOUT_FILENO, argv_err, sizeof(argv_err));
56 ret = 1;
57 }
58
59 if (__ps_strings->ps_envstr != environ) {
60 static const char env_err[] = "Wrong env in ps_strings";
61 write(STDOUT_FILENO, env_err, sizeof(env_err));
62 ret = 1;
63 }
64 nenv = 0;
65 while (environ[nenv])
66 ++nenv;
67 if (__ps_strings->ps_nenvstr != nenv) {
68 static const char nenv_err[] = "Wrong nenv in ps_strings";
69 write(STDOUT_FILENO, nenv_err, sizeof(nenv_err));
70 ret = 1;
71 }
72
73 return ret;
74}
