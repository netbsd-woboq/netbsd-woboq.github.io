34#include <sys/types.h>
35#include <sys/exec.h>
36#include <err.h>
37#include <limits.h>
38#include <stdio.h>
39#include <stdlib.h>
40#include <string.h>
41#include <unistd.h>
42
43#define LEN 16384
44
45extern struct ps_strings *__ps_strings;
46
47int
48main(void)
49{
50 size_t i;
51 char buf[16];
52 char **argv;
53
54 if ((argv = calloc(LEN, sizeof(*argv))) == NULL)
55 errx(1, "calloc failed");
56 for (i = 0; i < LEN; ++i) {
57 snprintf(buf, sizeof(buf), "arg%04zx", i);
58 if ((argv[i] = strdup(buf)) == NULL)
59 errx(1, "strdup failed");
60 }
61 __ps_strings->ps_argvstr = argv;
62 __ps_strings->ps_nargvstr = LEN;
63
64 printf("Sleeping forever...\n");
65 do {
66 sleep(UINT_MAX);
67 } while /* CONSTCOND */ (1);
68 return 0;
69}
