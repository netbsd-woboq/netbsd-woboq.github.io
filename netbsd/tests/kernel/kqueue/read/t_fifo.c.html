49#include "h_macros.h"

51#define FIFONAME "fifo"
52
53ATF_TC(fifo);
54ATF_TC_HEAD(fifo, tc)
55{
56 atf_tc_set_md_var(tc, "descr", "Checks EVFILT_READ on fifo");
57}
58ATF_TC_BODY(fifo, tc)
59{
60 int kq, n, fd;
61 struct kevent event[1];
62 char buffer[128];
63
64 RL(mkfifo(FIFONAME, 0644));
65 RL(fd = open(FIFONAME, O_RDWR, 0644));
66
67 RL(kq = kqueue());
68
69 EV_SET(&event[0], fd, EVFILT_READ, EV_ADD|EV_ENABLE, 0, 0, 0);
70 RL(kevent(kq, event, 1, NULL, 0, NULL));
71
72 /* make sure there is something in the fifo */
73 RL(write(fd, "foo", 3));
74 (void)printf("fifo: wrote 'foo'\n");
75
76 (void)memset(event, 0, sizeof(event));
77
78 RL(n = kevent(kq, NULL, 0, event, 1, NULL));
79
80 (void)printf("kevent num %d filt %d flags: %#x, fflags: %#x, "
81 "data: %" PRId64 "\n", n, event[0].filter, event[0].flags,
82 event[0].fflags, event[0].data);
83
84 ATF_REQUIRE_EQ(event[0].filter, EVFILT_READ);
85
86 RL(n = read(fd, buffer, event[0].data));
87 buffer[n] = '\0';
88 (void)printf("fifo: read '%s'\n", buffer);
89
90 RL(close(fd));
91}
92
93ATF_TP_ADD_TCS(tp)
94{
95 ATF_TP_ADD_TC(tp, fifo);
96
97 return atf_no_error();
98}
