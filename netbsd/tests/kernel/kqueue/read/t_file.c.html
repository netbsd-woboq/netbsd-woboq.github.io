55#define FILENAME "file"
56#define NLINES 5
57
58static void
59child(void)
60{
61 int i, n, fd;
62
63 (void)sleep(1);
64
65 for (i = 0; i < NLINES; ++i) {
66 fd = open(FILENAME, O_WRONLY|O_APPEND, 0644);
67 if (fd < 0)
68 err(EXIT_FAILURE, "open()");
69
70 n = write(fd, "foo\n", 4);
71 if (n < 0)
72 err(EXIT_FAILURE, "write()");
73
74 (void)close(fd);
75 (void)sleep(1);
76 }
77 _exit(0);
78}

80ATF_TC(file);
81ATF_TC_HEAD(file, tc)
82{
83 atf_tc_set_md_var(tc, "descr", "Checks EVFILT_READ on regular file");
84}
85ATF_TC_BODY(file, tc)
86{
87 char buffer[128];
88 struct kevent event[1];
89 pid_t pid;
90 int fd, kq, n, num, status;
91
92 RL(pid = fork());
93 if (pid == 0) {
94 child();
95 /* NOTREACHED */
96 }
97
98 RL(fd = open(FILENAME, O_RDONLY|O_CREAT, 0644));
99
100#if 1 /* XXX: why was this disabled? */
101 RL(lseek(fd, 0, SEEK_END));
102#endif
103
104 RL(kq = kqueue());
105
106 EV_SET(&event[0], fd, EVFILT_READ, EV_ADD|EV_ENABLE, 0, 0, 0);
107 RL(kevent(kq, event, 1, NULL, 0, NULL));
108
109 for (num = 0; num < NLINES;) {
110 RL(n = kevent(kq, NULL, 0, event, 1, NULL));
111 num += n;
112
113 (void)printf("kevent num %d flags: %#x, fflags: %#x, data: "
114 "%" PRId64 "\n", n, event[0].flags, event[0].fflags,
115 event[0].data);
116
117 if (event[0].data < 0)
118#if 1 /* XXXLUKEM */
119 RL(lseek(fd, 0, SEEK_END));
120#else
121 RL(lseek(fd, event[0].data, SEEK_END));
122#endif
123
124 RL(n = read(fd, buffer, 128));
125 buffer[n] = '\0';
126 (void)printf("file(%d): %s", num, buffer);
127 }
128
129 (void)waitpid(pid, &status, 0);
130
131 (void)printf("read: successful end\n");
132}
