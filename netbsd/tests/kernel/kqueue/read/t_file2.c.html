46ATF_TC(file2);
47ATF_TC_HEAD(file2, tc)
48{
49 atf_tc_set_md_var(tc, "descr",
50 "Checks EVFILT_READ for regular files. This test used to "
51 "trigger deadlock caused by problem fixed in revision 1.79.2.10 "
52 "of sys/kern/kern_descrip.c");
53}
54ATF_TC_BODY(file2, tc)
55{
56 int fd1, fd2, kq;
57 struct kevent event[1];
58
59 RL(fd1 = open("afile", O_RDONLY|O_CREAT, 0644));
60 RL(fd2 = open("bfile", O_RDONLY|O_CREAT, 0644));
61
62#if 1 /* XXX: why was this disabled? */
63 RL(lseek(fd1, 0, SEEK_END));
64#endif
65
66 RL(kq = kqueue());
67
68 EV_SET(&event[0], fd1, EVFILT_READ, EV_ADD|EV_ENABLE, 0, 0, 0);
69 RL(kevent(kq, event, 1, NULL, 0, NULL));
70
71 RL(dup2(fd2, fd1));
72}
73
74ATF_TP_ADD_TCS(tp)
75{
76 ATF_TP_ADD_TC(tp, file2);
77
78 return atf_no_error();
79}
