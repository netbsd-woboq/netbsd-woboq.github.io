46ATF_TC(pipe);
47ATF_TC_HEAD(pipe, tc)
48{
49 atf_tc_set_md_var(tc, "descr", "Checks EVFILT_READ for pipes");
50}
51ATF_TC_BODY(pipe, tc)
52{
53 struct kevent event[1];
54 char buffer[128];
55 int fds[2];
56 int kq, n;
57
58 RL(pipe(fds));
59 RL(kq = kqueue());
60
61 EV_SET(&event[0], fds[0], EVFILT_READ, EV_ADD|EV_ENABLE, 0, 0, 0);
62 RL(kevent(kq, event, 1, NULL, 0, NULL));
63
64 /* make sure there is something in the pipe */
65 RL(write(fds[1], "foo", 3));
66 (void)printf("pipe: wrote 'foo' to pipe\n");
67
68 RL(n = kevent(kq, NULL, 0, event, 1, NULL));
69 (void)printf("kevent num %d flags: %#x, fflags: %#x, data: "
70 "%" PRId64 "\n", n, event[0].flags, event[0].fflags, event[0].data);
71
72 RL(n = read(fds[0], buffer, event[0].data));
73 buffer[n] = '\0';
74
75 (void)printf("pipe: read '%s'\n", buffer);
76 (void)printf("pipe: successful end\n");
77}
78
79ATF_TP_ADD_TCS(tp)
80{
81 ATF_TP_ADD_TC(tp, pipe);
82
83 return atf_no_error();
84}
