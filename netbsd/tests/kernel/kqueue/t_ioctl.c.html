47ATF_TC(kfilter_byfilter);
48ATF_TC_HEAD(kfilter_byfilter, tc)
49{
50 atf_tc_set_md_var(tc, "descr", "Checks KFILTER_BYFILTER ioctl");
51}
52ATF_TC_BODY(kfilter_byfilter, tc)
53{
54 char buf[32];
55 struct kfilter_mapping km;
56 int kq;
57 uint32_t i;
58
59 RL(kq = kqueue());
60
61 km.name = buf;
62 km.len = sizeof(buf) - 1;
63
64 for (i = 0; i < 8; ++i) {
65 km.filter = i;
66 RL(ioctl(kq, KFILTER_BYFILTER, &km));
67 (void)printf(" map %d -> %s\n", km.filter, km.name);
68 }
69
70 km.filter = 8;
71 ATF_REQUIRE_EQ(ioctl(kq, KFILTER_BYFILTER, &km), -1);
72}
73
74ATF_TC(kfilter_byname);
75ATF_TC_HEAD(kfilter_byname, tc)
76{
77 atf_tc_set_md_var(tc, "descr", "Checks KFILTER_BYNAME ioctl");
78}
79ATF_TC_BODY(kfilter_byname, tc)
80{
81 const char *tests[] = {
82 "EVFILT_READ",
83 "EVFILT_WRITE",
84 "EVFILT_AIO",
85 "EVFILT_VNODE",
86 "EVFILT_PROC",
87 "EVFILT_SIGNAL",
88 "EVFILT_TIMER",
89 "EVFILT_FS",
90 NULL
91 };
92 char buf[32];
93 struct kfilter_mapping km;
94 const char **test;
95 int kq;
96
97 RL(kq = kqueue());
98
99 km.name = buf;
100
101 for (test = &tests[0]; *test != NULL; ++test) {
102 (void)strlcpy(buf, *test, sizeof(buf));
103 RL(ioctl(kq, KFILTER_BYNAME, &km));
104 (void)printf(" map %s -> %d\n", km.name, km.filter);
105 }
106
107 (void)strlcpy(buf, "NOTREG_FILTER", sizeof(buf));
108 ATF_REQUIRE_EQ(ioctl(kq, KFILTER_BYNAME, &km), -1);
109}
110
111ATF_TP_ADD_TCS(tp)
112{
113 ATF_TP_ADD_TC(tp, kfilter_byfilter);
114 ATF_TP_ADD_TC(tp, kfilter_byname);
115
116 return atf_no_error();
117}
