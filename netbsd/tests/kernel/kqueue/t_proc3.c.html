50ATF_TC(proc3);
51ATF_TC_HEAD(proc3, tc)
52{
53 atf_tc_set_md_var(tc, "descr",
54 "Checks EVFILT_PROC for NOTE_TRACK on self bug ");
55}
56
57ATF_TC_BODY(proc3, tc)
58{
59 pid_t pid = 0;
60 int kq, status;
61 struct kevent ke;
62 struct timespec timeout;
63
64 RL(kq = kqueue());
65
66 EV_SET(&ke, (uintptr_t)getpid(), EVFILT_PROC, EV_ADD, NOTE_TRACK, 0, 0);
67
68 RL(kevent(kq, &ke, 1, NULL, 0, NULL));
69
70 RL(pid = fork());
71 if (pid == 0) {
72 _exit(EXIT_SUCCESS);
73 /* NOTREACHED */
74 }
75
76 RL(waitpid(pid, &status, 0));
77 ATF_REQUIRE(WIFEXITED(status));
78 ATF_REQUIRE_EQ(WEXITSTATUS(status), EXIT_SUCCESS);
79
80 timeout.tv_sec = 0;
81 timeout.tv_nsec = 0;
82 ke.ident = 0;
83 ke.fflags = 0;
84 ke.flags = EV_ENABLE;
85
86 RL(kevent(kq, NULL, 0, &ke, 1, &timeout));
87 RL(close(kq));
88
89 ATF_REQUIRE(ke.fflags & NOTE_CHILD);
90 ATF_REQUIRE((ke.fflags & NOTE_TRACKERR) == 0);
91 ATF_REQUIRE_EQ((pid_t)ke.ident, pid);
92}
93
94ATF_TP_ADD_TCS(tp)
95{
96 ATF_TP_ADD_TC(tp, proc3);
97
98 return atf_no_error();
99}
