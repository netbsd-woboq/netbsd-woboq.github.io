49ATF_TC(pipe1);
50ATF_TC_HEAD(pipe1, tc)
51{
52 atf_tc_set_md_var(tc, "descr",
53 "Checks EVFILT_WRITE for pipes. This test used to trigger "
54 "problem fixed in rev. 1.5.2.7 of sys/kern/sys_pipe.c");
55}
56ATF_TC_BODY(pipe1, tc)
57{
58 struct kevent event[1];
59 int fds[2];
60 int kq, n;
61
62 RL(pipe(fds));
63 RL(kq = kqueue());
64 RL(close(fds[0]));
65
66 EV_SET(&event[0], fds[1], EVFILT_WRITE, EV_ADD|EV_ENABLE, 0, 0, 0);
67 ATF_REQUIRE_EQ_MSG((n = kevent(kq, event, 1, NULL, 0, NULL)),
68 -1, "got: %d", n);
69 ATF_REQUIRE_EQ_MSG(errno, EBADF, "got: %s", strerror(errno));
70}

72ATF_TC(pipe2);
73ATF_TC_HEAD(pipe2, tc)
74{
75 atf_tc_set_md_var(tc, "descr",
76 "Checks EVFILT_WRITE for pipes. This test used to trigger problem "
77 "fixed in rev. 1.5.2.9 of sys/kern/sys_pipe.c");
78}
79ATF_TC_BODY(pipe2, tc)
80{
81 struct kevent event[1];
82 char buffer[128];
83 int fds[2];
84 int kq, n;
85 int status;
86 pid_t child;
87
88 RL(pipe(fds));
89 RL(kq = kqueue());
90
91 EV_SET(&event[0], fds[1], EVFILT_WRITE, EV_ADD|EV_ENABLE, 0, 0, 0);
92 RL(kevent(kq, event, 1, NULL, 0, NULL));
93
94 /* spawn child reader */
95 RL(child = fork());
96 if (child == 0) {
97 int sz = read(fds[0], buffer, 128);
98 if (sz > 0)
99 (void)printf("pipe: child read '%.*s'\n", sz, buffer);
100 exit(sz <= 0);
101 }
102
103 RL(n = kevent(kq, NULL, 0, event, 1, NULL));
104
105 (void)printf("kevent num %d flags: %#x, fflags: %#x, data: "
106 "%" PRId64 "\n", n, event[0].flags, event[0].fflags, event[0].data);
107
108 RL(n = write(fds[1], "foo", 3));
109 RL(close(fds[1]));
110
111 (void)waitpid(child, &status, 0);
112}

114ATF_TC(pipe3);
115ATF_TC_HEAD(pipe3, tc)
116{
117 atf_tc_set_md_var(tc, "descr",
118 "Checks EVFILT_WRITE for pipes. This test used to trigger problem "
119 "fixed in rev. 1.5.2.10 of sys/kern/sys_pipe.c");
120}
121ATF_TC_BODY(pipe3, tc)
122{
123 struct kevent event[1];
124 int fds[2];
125 int kq;
126
127 RL(pipe(fds));
128 RL(kq = kqueue());
129
130 EV_SET(&event[0], fds[1], EVFILT_WRITE, EV_ADD|EV_ENABLE, 0, 0, 0);
131 RL(kevent(kq, event, 1, NULL, 0, NULL));
132
133 /* close 'read' end first, then 'write' */
134
135 RL(close(fds[0]));
136 RL(close(fds[1]));
137}
