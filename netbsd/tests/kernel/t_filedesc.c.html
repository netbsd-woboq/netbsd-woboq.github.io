45ATF_TC(getfilerace);
46ATF_TC_HEAD(getfilerace, tc)
47{
48
49 atf_tc_set_md_var(tc, "descr", "race between multithreaded proc. "
50 "fd_getfile() and fd_close() (PR kern/43694)");
51}
52
53static int fd;
54static volatile bool quit;
55
56static void *
57wrkwrk(void *arg)
58{
59
60 /* just something to cause fd_getfile() to be called */
61 while (!quit)
62 rump_sys_write(fd, &fd, sizeof(fd));
63
64 return NULL;
65}
66
67/* for me, 1000 triggers extremely seldom, 10k sometimes, 100k almost always */
68#define DEFAULT_ITERATIONS 10000
69
70ATF_TC_BODY(getfilerace, tc)
71{
72 pthread_t pt;
73 int fd_wrk;
74 int i, iters;
75
76 /*
77 * Want a multiprocessor virtual kernel. A multiprocessor host
78 * probably helps too, but that's harder to do in software...
79 */
80 setenv("RUMP_NCPU", "2", 1);
81 rump_init();
82
83 fd = fd_wrk = rump_sys_open("/dev/null", O_RDWR, 0);
84 if (fd == -1)
85 atf_tc_fail_errno("cannot open /dev/null");
86
87 if (atf_tc_has_config_var(tc, "iters"))
88 iters = atoi(atf_tc_get_config_var(tc, "iters"));
89 else
90 iters = DEFAULT_ITERATIONS;
91
92 pthread_create(&pt, NULL, wrkwrk, NULL);
93 for (i = 0; i < iters; i++) {
94 rump_sys_close(fd_wrk);
95 fd_wrk = rump_sys_open("/dev/null", O_RDWR, 0);
96 assert(fd == fd_wrk);
97 }
98
99 quit = true;
100 pthread_join(pt, NULL);
101}
102
103ATF_TP_ADD_TCS(tp)
104{
105 ATF_TP_ADD_TC(tp, getfilerace);
106
107 return atf_no_error();
108}
