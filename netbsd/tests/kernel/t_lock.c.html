34#include <sys/time.h>
35
36#include <machine/lock.h>
37#include <signal.h>
38
39#include <atf-c.h>
40
41#include "h_macros.h"
42
43__cpu_simple_lock_t lk;
44volatile int handled = 0;
45
46static void
47handler(int sig)
48{
49 handled = 1;
50 __cpu_simple_unlock(&lk);
51}
52
53ATF_TC(lock);
54ATF_TC_HEAD(lock, tc)
55{
56 atf_tc_set_md_var(tc, "timeout", "3");
57 atf_tc_set_md_var(tc, "descr",
58 "Checks __cpu_simple_lock()/__cpu_simple_unlock()");
59}
60ATF_TC_BODY(lock, tc)
61{
62 struct itimerval itv;
63
64 __cpu_simple_lock_init(&lk);
65
66 REQUIRE_LIBC(signal(SIGVTALRM, handler), SIG_ERR);
67
68 itv.it_interval.tv_sec = 0;
69 itv.it_interval.tv_usec = 0;
70 itv.it_value.tv_sec = 1;
71 itv.it_value.tv_usec = 0;
72 RL(setitimer(ITIMER_VIRTUAL, &itv, NULL));
73
74 __cpu_simple_lock(&lk);
75 __cpu_simple_lock(&lk);
76
77 ATF_REQUIRE(handled);
78
79 __cpu_simple_unlock(&lk);
80}
81
82ATF_TP_ADD_TCS(tp)
83{
84 ATF_TP_ADD_TC(tp, lock);
85
86 return atf_no_error();
87}
