34#include <sys/sysctl.h>
35#include <errno.h>
36#include <memory.h>
37
38#include <atf-c.h>
39
40ATF_TC(bufsize);
41ATF_TC_HEAD(bufsize, tc)
42{
43 atf_tc_set_md_var(tc, "descr",
44 "Test sysctl integer reads with different buffer sizes");
45}
46ATF_TC_BODY(bufsize, tc)
47{
48 union {
49 int int_val;
50 unsigned char space[256];
51 } buf;
52 size_t len;
53 for (len = 0; len < sizeof(buf); len++) {
54 size_t oldlen = len;
55 int r;
56 memset(&buf, 0xFF, sizeof(buf));
57 r = sysctlbyname("kern.job_control", &buf, &oldlen, 0, (size_t) 0);
58 if (len < sizeof(int)) {
59 ATF_REQUIRE_EQ(r, -1);
60 ATF_REQUIRE_EQ(errno, ENOMEM);
61 } else {
62 ATF_REQUIRE_EQ(r, 0);
63 ATF_REQUIRE_EQ(buf.int_val, 1);
64 ATF_REQUIRE_EQ(oldlen, sizeof(int));
65 }
66 }
67}
68
69ATF_TP_ADD_TCS(tp)
70{
71 ATF_TP_ADD_TC(tp, bufsize);
72
73 return atf_no_error();
74}
