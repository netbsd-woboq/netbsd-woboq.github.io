32#if defined(__arm__) || defined(__i386__) || defined(__x86_64__) || defined(__powerpc__) || defined(__sparc__)
33#include <stdlib.h>
34#include <string.h>
35
36static long long
37ifunc1(void)
38{
39 return 0xdeadbeefll;
40}
41
42static long long
43ifunc2(void)
44{
45 return 0xbeefdeadll;
46}
47
48static __attribute__((used))
49long long (*resolve_ifunc(void))(void)
50{
51 const char *e = getenv("USE_IFUNC2");
52 return e && strcmp(e, "1") == 0 ? ifunc2 : ifunc1;
53}
54
55__ifunc(ifunc, resolve_ifunc);
56extern long long ifunc(void);
57
58int
59main(int argc, char **argv)
60{
61
62 if (argc != 2)
63 return 1;
64 return atoll(argv[1]) != ifunc();
65}
66#else
67int
68main(int argc, char **argv)
69{
70 return 127;
71}
72#endif
