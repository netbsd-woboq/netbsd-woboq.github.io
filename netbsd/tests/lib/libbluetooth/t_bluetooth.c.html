32#include <atf-c.h>
33
34#include <bluetooth.h>
35#include <string.h>
36
37ATF_TC(check_bt_aton);
38
39ATF_TC_HEAD(check_bt_aton, tc)
40{
41
42 atf_tc_set_md_var(tc, "descr", "Test bt_aton results");
43}
44
45ATF_TC_BODY(check_bt_aton, tc)
46{
47 bdaddr_t bdaddr;
48
49 ATF_CHECK_EQ(bt_aton("0a:0b:0c:0d:0e", &bdaddr), 0);
50 ATF_CHECK_EQ(bt_aton("0a:0b:0c:0d0:0e:0f", &bdaddr), 0);
51 ATF_CHECK_EQ(bt_aton("0a:0b:0c:0d:0e:0f:00", &bdaddr), 0);
52 ATF_CHECK_EQ(bt_aton("0a:0b:0c:0d:0e:0f\n", &bdaddr), 0);
53 ATF_CHECK_EQ(bt_aton(" 0a:0b:0c:0d:0e:0f", &bdaddr), 0);
54 ATF_CHECK_EQ(bt_aton("0a:0b:0x:0d:0e:0f", &bdaddr), 0);
55
56 ATF_REQUIRE(bt_aton("0a:0b:0c:0d:0e:0f", &bdaddr));
57 ATF_CHECK_EQ(bdaddr.b[0], 0x0f);
58 ATF_CHECK_EQ(bdaddr.b[1], 0x0e);
59 ATF_CHECK_EQ(bdaddr.b[2], 0x0d);
60 ATF_CHECK_EQ(bdaddr.b[3], 0x0c);
61 ATF_CHECK_EQ(bdaddr.b[4], 0x0b);
62 ATF_CHECK_EQ(bdaddr.b[5], 0x0a);
63}
64
65ATF_TC(check_bt_ntoa);
66
67ATF_TC_HEAD(check_bt_ntoa, tc)
68{
69
70 atf_tc_set_md_var(tc, "descr", "Test bt_ntoa results");
71}
72
73ATF_TC_BODY(check_bt_ntoa, tc)
74{
75 bdaddr_t bdaddr = { { 0x00, 0x11, 0x22, 0x33, 0x44, 0x55 } };
76
77 ATF_CHECK_STREQ(bt_ntoa(&bdaddr, NULL), "55:44:33:22:11:00");
78}
79
80ATF_TP_ADD_TCS(tp)
81{
82
83 ATF_TP_ADD_TC(tp, check_bt_aton);
84 ATF_TP_ADD_TC(tp, check_bt_ntoa);
85
86 return atf_no_error();
87}
