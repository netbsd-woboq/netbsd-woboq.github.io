32#include <atf-c.h>
33
34#include <sdp.h>
35
36ATF_TC(check_sdp_data_type);
37
38ATF_TC_HEAD(check_sdp_data_type, tc)
39{
40
41 atf_tc_set_md_var(tc, "descr", "Test sdp_data_type results");
42}
43
44ATF_TC_BODY(check_sdp_data_type, tc)
45{
46 uint8_t data[] = {
47 0x00, // nil
48 0x08, 0x00, // uint8 0x00
49 };
50 sdp_data_t test = { data, data + sizeof(data) };
51 sdp_data_t value;
52
53 ATF_REQUIRE(sdp_get_data(&test, &value));
54 ATF_CHECK_EQ(sdp_data_type(&value), SDP_DATA_NIL);
55
56 ATF_REQUIRE(sdp_get_data(&test, &value));
57 ATF_CHECK_EQ(sdp_data_type(&value), SDP_DATA_UINT8);
58
59 ATF_CHECK_EQ(test.next, test.end);
60}
61
62ATF_TC(check_sdp_data_size);
63
64ATF_TC_HEAD(check_sdp_data_size, tc)
65{
66
67 atf_tc_set_md_var(tc, "descr", "Test sdp_data_size results");
68}
69
70ATF_TC_BODY(check_sdp_data_size, tc)
71{
72 uint8_t data[] = {
73 0x00, // nil
74 0x08, 0x00, // uint8
75 0x11, 0x00, 0x00, // int16
76 0x1a, 0x00, 0x00, 0x00, // uuid32
77 0x00,
78 0x28, 0x00, // bool
79 0x25, 0x00, // str8(0)
80 0x25, 0x02, 0x00, 0x00, // str8(2)
81 0x36, 0x00, 0x00, // seq16(0)
82 0x3e, 0x00, 0x05, 0x00, // alt16(5)
83 0x00, 0x00, 0x00, 0x00,
84 0x37, 0x00, 0x00, 0x00, // seq32(0)
85 0x00,
86 0x47, 0x00, 0x00, 0x00, // url32(7)
87 0x07, 0x00, 0x00, 0x00,
88 0x00, 0x00, 0x00, 0x00,
89 };
90 sdp_data_t test = { data, data + sizeof(data) };
91 sdp_data_t value;
92
93 ATF_REQUIRE(sdp_get_data(&test, &value));
94 ATF_CHECK_EQ(sdp_data_size(&value), 1);
95
96 ATF_REQUIRE(sdp_get_data(&test, &value));
97 ATF_CHECK_EQ(sdp_data_size(&value), 2);
98
99 ATF_REQUIRE(sdp_get_data(&test, &value));
100 ATF_CHECK_EQ(sdp_data_size(&value), 3);
101
102 ATF_REQUIRE(sdp_get_data(&test, &value));
103 ATF_CHECK_EQ(sdp_data_size(&value), 5);
104
105 ATF_REQUIRE(sdp_get_data(&test, &value));
106 ATF_CHECK_EQ(sdp_data_size(&value), 2);
107
108 ATF_REQUIRE(sdp_get_data(&test, &value));
109 ATF_CHECK_EQ(sdp_data_size(&value), 2);
110
111 ATF_REQUIRE(sdp_get_data(&test, &value));
112 ATF_CHECK_EQ(sdp_data_size(&value), 4);
113
114 ATF_REQUIRE(sdp_get_data(&test, &value));
115 ATF_CHECK_EQ(sdp_data_size(&value), 3);
116
117 ATF_REQUIRE(sdp_get_data(&test, &value));
118 ATF_CHECK_EQ(sdp_data_size(&value), 8);
119
120 ATF_REQUIRE(sdp_get_data(&test, &value));
121 ATF_CHECK_EQ(sdp_data_size(&value), 5);
122
123 ATF_REQUIRE(sdp_get_data(&test, &value));
124 ATF_CHECK_EQ(sdp_data_size(&value), 12);
125
126 ATF_CHECK_EQ(test.next, test.end);
127}

134
135 return atf_no_error();
136}
