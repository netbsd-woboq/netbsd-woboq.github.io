32#include <atf-c.h>
33
34#include <sdp.h>
35
36ATF_TC(check_sdp_match_uuid16);
37
38ATF_TC_HEAD(check_sdp_match_uuid16, tc)
39{
40
41 atf_tc_set_md_var(tc, "descr", "Test sdp_match_uuid16 results");
42}
43
44ATF_TC_BODY(check_sdp_match_uuid16, tc)
45{
46 uint8_t data[] = {
47 0x19, 0x11, 0x11, // uuid16 0x1111
48 0x00, // nil
49 0x19, 0x12, 0x34, // uuid16 0x1234
50 0x1a, 0x00, 0x00, 0x34, // uuid32 0x00003456
51 0x56,
52 0x1c, 0x00, 0x00, 0x43, // uuid128 00004321-0000-1000-8000-00805f9b34fb
53 0x21, 0x00, 0x00, 0x10,
54 0x00, 0x80, 0x00, 0x00,
55 0x80, 0x5f, 0x9b, 0x34,
56 0xfb,
57 };
58 sdp_data_t test = { data, data + sizeof(data) };
59 sdp_data_t nil;
60
61 /*
62 * sdp_match_uuid16 advances if the UUID matches the 16-bit short alias given
63 */
64
65 ATF_REQUIRE_EQ(sdp_match_uuid16(&test, 0x1100), false); /* mismatch */
66 ATF_REQUIRE(sdp_match_uuid16(&test, 0x1111));
67
68 ATF_REQUIRE_EQ(sdp_match_uuid16(&test, 0x1234), false); /* not uuid */
69 ATF_REQUIRE(sdp_get_data(&test, &nil)); /* (skip) */
70 ATF_CHECK_EQ(sdp_data_type(&nil), SDP_DATA_NIL);
71 ATF_REQUIRE(sdp_match_uuid16(&test, 0x1234));
72
73 ATF_REQUIRE(sdp_match_uuid16(&test, 0x3456));
74
75 ATF_REQUIRE_EQ(sdp_match_uuid16(&test, 0x1234), false); /* mismatch */
76 ATF_REQUIRE(sdp_match_uuid16(&test, 0x4321));
77
78 ATF_CHECK_EQ(test.next, test.end);
79}
80
81ATF_TP_ADD_TCS(tp)
82{
83
84 ATF_TP_ADD_TC(tp, check_sdp_match_uuid16);
85
86 return atf_no_error();
87}
