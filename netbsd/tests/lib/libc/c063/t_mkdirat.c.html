31#include <sys/cdefs.h>
32__RCSID("$NetBSD: t_mkdirat.c,v 1.2 2013/03/17 04:46:06 jmmv Exp $");
33
34#include <atf-c.h>
35#include <errno.h>
36#include <fcntl.h>
37#include <limits.h>
38#include <paths.h>
39#include <stdio.h>
40#include <string.h>
41#include <unistd.h>
42#include <sys/param.h>
43#include <sys/stat.h>
44
45#define DIR "dir"
46#define SDIR "dir/openat"
47#define BASESDIR "openat"
48#define SDIRERR "dir/openaterr"
49
50ATF_TC(mkdirat_fd);
51ATF_TC_HEAD(mkdirat_fd, tc)
52{
53 atf_tc_set_md_var(tc, "descr", "See that mkdirat works with fd");
54}
55ATF_TC_BODY(mkdirat_fd, tc)
56{
57 int dfd;
58 mode_t mode = 0755;
59
60 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
61 ATF_REQUIRE((dfd = open(DIR, O_RDONLY, 0)) != -1);
62 ATF_REQUIRE(mkdirat(dfd, BASESDIR, mode) != -1);
63 ATF_REQUIRE(close(dfd) == 0);
64 ATF_REQUIRE(access(SDIR, F_OK) == 0);
65}
66
67ATF_TC(mkdirat_fdcwd);
68ATF_TC_HEAD(mkdirat_fdcwd, tc)
69{
70 atf_tc_set_md_var(tc, "descr",
71 "See that mkdirat works with fd as AT_FDCWD");
72}
73ATF_TC_BODY(mkdirat_fdcwd, tc)
74{
75 mode_t mode = 0755;
76
77 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
78 ATF_REQUIRE(mkdirat(AT_FDCWD, SDIR, mode) != -1);
79 ATF_REQUIRE(access(SDIR, F_OK) == 0);
80}
81
82ATF_TC(mkdirat_fdcwderr);
83ATF_TC_HEAD(mkdirat_fdcwderr, tc)
84{
85 atf_tc_set_md_var(tc, "descr",
86 "See that mkdirat fails with fd as AT_FDCWD and bad path");
87}
88ATF_TC_BODY(mkdirat_fdcwderr, tc)
89{
90 mode_t mode = 0755;
91
92 ATF_REQUIRE(mkdirat(AT_FDCWD, SDIRERR, mode) == -1);
93}
94
95ATF_TC(mkdirat_fderr);
96ATF_TC_HEAD(mkdirat_fderr, tc)
97{
98 atf_tc_set_md_var(tc, "descr", "See that mkdirat fails with fd as -1");
99}
100ATF_TC_BODY(mkdirat_fderr, tc)
101{
102 int fd;
103 mode_t mode = 0755;
104
105 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
106 ATF_REQUIRE((fd = open(SDIR, O_CREAT|O_RDWR, 0644)) != -1);
107 ATF_REQUIRE(close(fd) == 0);
108 ATF_REQUIRE(mkdirat(-1, SDIR, mode) == -1);
109}
110
111ATF_TP_ADD_TCS(tp)
112{
113
114 ATF_TP_ADD_TC(tp, mkdirat_fd);
115 ATF_TP_ADD_TC(tp, mkdirat_fdcwd);
116 ATF_TP_ADD_TC(tp, mkdirat_fdcwderr);
117 ATF_TP_ADD_TC(tp, mkdirat_fderr);
118
119 return atf_no_error();
120}
