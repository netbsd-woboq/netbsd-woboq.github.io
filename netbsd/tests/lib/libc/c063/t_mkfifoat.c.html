34#include <atf-c.h>
35#include <errno.h>
36#include <fcntl.h>
37#include <limits.h>

39#include <stdio.h>
40#include <string.h>
41#include <unistd.h>
42#include <sys/param.h>
43#include <sys/stat.h>
44
45#define DIR "dir"
46#define FIFO "dir/openat"
47#define BASEFIFO "openat"
48#define FIFOERR "dir/openaterr"
49
50ATF_TC(mkfifoat_fd);
51ATF_TC_HEAD(mkfifoat_fd, tc)
52{
53 atf_tc_set_md_var(tc, "descr", "See that mkfifoat works with fd");
54}
55ATF_TC_BODY(mkfifoat_fd, tc)
56{
57 int dfd;
58 mode_t mode = 0600;
59
60 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
61 ATF_REQUIRE((dfd = open(DIR, O_RDONLY, 0)) != -1);
62 ATF_REQUIRE(mkfifoat(dfd, BASEFIFO, mode) != -1);
63 ATF_REQUIRE(access(FIFO, F_OK) == 0);
64 (void)close(dfd);
65}
66
67ATF_TC(mkfifoat_fdcwd);
68ATF_TC_HEAD(mkfifoat_fdcwd, tc)
69{
70 atf_tc_set_md_var(tc, "descr",
71 "See that mkfifoat works with fd as AT_FDCWD");
72}
73ATF_TC_BODY(mkfifoat_fdcwd, tc)
74{
75 mode_t mode = 0600;
76
77 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
78 ATF_REQUIRE(mkfifoat(AT_FDCWD, FIFO, mode) != -1);
79 ATF_REQUIRE(access(FIFO, F_OK) == 0);
80}
81
82ATF_TC(mkfifoat_fdcwderr);
83ATF_TC_HEAD(mkfifoat_fdcwderr, tc)
84{
85 atf_tc_set_md_var(tc, "descr",
86 "See that mkfifoat fails with fd as AT_FDCWD and bad path");
87}
88ATF_TC_BODY(mkfifoat_fdcwderr, tc)
89{
90 mode_t mode = 0600;
91
92 ATF_REQUIRE(mkfifoat(AT_FDCWD, FIFOERR, mode) == -1);
93}
94
95ATF_TC(mkfifoat_fderr);
96ATF_TC_HEAD(mkfifoat_fderr, tc)
97{
98 atf_tc_set_md_var(tc, "descr", "See that mkfifoat fails with fd as -1");
99}
100ATF_TC_BODY(mkfifoat_fderr, tc)
101{
102 int fd;
103 mode_t mode = 0600;
104
105 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
106 ATF_REQUIRE((fd = open(FIFO, O_CREAT|O_RDWR, 0644)) != -1);
107 ATF_REQUIRE(close(fd) == 0);
108 ATF_REQUIRE(mkfifoat(-1, FIFO, mode) == -1);
109}
110
111ATF_TC(mknodat_s_ififo);
112ATF_TC_HEAD(mknodat_s_ififo, tc)
113{
114 atf_tc_set_md_var(tc, "descr", "Test mknodat(2) with S_IFIFO");
115}
116
117ATF_TC_BODY(mknodat_s_ififo, tc)
118{
119 struct stat st;
120 int dfd;
121 mode_t mode = S_IFIFO | 0600;
122
123 (void)memset(&st, 0, sizeof(struct stat));
124
125 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
126 ATF_REQUIRE((dfd = open(DIR, O_RDONLY, 0)) != -1);
127 ATF_REQUIRE(mknodat(dfd, BASEFIFO, mode, 0) != -1);
128 ATF_REQUIRE(access(FIFO, F_OK) == 0);
129 ATF_REQUIRE(stat(FIFO, &st) == 0);
130
131 if (S_ISFIFO(st.st_mode) == 0)
132 atf_tc_fail("invalid mode from mknodat(2) with S_IFIFO");
133
134 (void)close(dfd);
135}
136
137ATF_TP_ADD_TCS(tp)
138{
139
140 ATF_TP_ADD_TC(tp, mkfifoat_fd);
141 ATF_TP_ADD_TC(tp, mkfifoat_fdcwd);
142 ATF_TP_ADD_TC(tp, mkfifoat_fdcwderr);
143 ATF_TP_ADD_TC(tp, mkfifoat_fderr);
144 ATF_TP_ADD_TC(tp, mknodat_s_ififo);
145
146 return atf_no_error();
147}
