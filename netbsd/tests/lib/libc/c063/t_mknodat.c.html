45#define DIR "dir"
46#define FILE "dir/openat"
47#define BASEFILE "openat"
48#define FILEERR "dir/openaterr"
49
50static dev_t get_devnull(void);
51
52static dev_t
53get_devnull(void)
54{
55 struct stat st;
56
57 if (stat(_PATH_DEVNULL, &st) != 0)
58 return NODEV;
59
60 return st.st_rdev;
61}

63
64ATF_TC(mknodat_fd);
65ATF_TC_HEAD(mknodat_fd, tc)
66{
67 atf_tc_set_md_var(tc, "descr", "See that mknodat works with fd");
68 atf_tc_set_md_var(tc, "require.user", "root");
69}
70ATF_TC_BODY(mknodat_fd, tc)
71{
72 int dfd;
73 int fd;
74 dev_t dev;
75 mode_t mode = S_IFCHR|0600;
76
77 ATF_REQUIRE((dev = get_devnull()) != NODEV);
78 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
79 ATF_REQUIRE((dfd = open(DIR, O_RDONLY, 0)) != -1);
80 ATF_REQUIRE((fd = mknodat(dfd, BASEFILE, mode, dev)) != -1);
81 ATF_REQUIRE(close(fd) == 0);
82 ATF_REQUIRE(access(FILE, F_OK) == 0);
83 (void)close(dfd);
84}
85
86ATF_TC(mknodat_fdcwd);
87ATF_TC_HEAD(mknodat_fdcwd, tc)
88{
89 atf_tc_set_md_var(tc, "descr",
90 "See that mknodat works with fd as AT_FDCWD");
91 atf_tc_set_md_var(tc, "require.user", "root");
92}
93ATF_TC_BODY(mknodat_fdcwd, tc)
94{
95 int fd;
96 dev_t dev;
97 mode_t mode = S_IFCHR|0600;
98
99 ATF_REQUIRE((dev = get_devnull()) != NODEV);
100 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
101 ATF_REQUIRE((fd = mknodat(AT_FDCWD, FILE, mode, dev)) != -1);
102 ATF_REQUIRE(close(fd) == 0);
103 ATF_REQUIRE(access(FILE, F_OK) == 0);
104}
105
106ATF_TC(mknodat_fdcwderr);
107ATF_TC_HEAD(mknodat_fdcwderr, tc)
108{
109 atf_tc_set_md_var(tc, "descr",
110 "See that mknodat fails with fd as AT_FDCWD and bad path");
111 atf_tc_set_md_var(tc, "require.user", "root");
112}
113ATF_TC_BODY(mknodat_fdcwderr, tc)
114{
115 int fd;
116 dev_t dev;
117 mode_t mode = S_IFCHR|0600;
118
119 ATF_REQUIRE((dev = get_devnull()) != NODEV);
120 ATF_REQUIRE((fd = mknodat(AT_FDCWD, FILEERR, mode, dev)) == -1);
121}
122
123ATF_TC(mknodat_fderr);
124ATF_TC_HEAD(mknodat_fderr, tc)
125{
126 atf_tc_set_md_var(tc, "descr", "See that mknodat fails with fd as -1");
127 atf_tc_set_md_var(tc, "require.user", "root");
128}
129ATF_TC_BODY(mknodat_fderr, tc)
130{
131 int fd;
132 dev_t dev;
133 mode_t mode = S_IFCHR|0600;
134
135 ATF_REQUIRE((dev = get_devnull()) != NODEV);
136 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
137 ATF_REQUIRE((fd = open(FILE, O_CREAT|O_RDWR, 0644)) != -1);
138 ATF_REQUIRE(close(fd) == 0);
139 ATF_REQUIRE((fd = mknodat(-1, FILE, mode, dev)) == -1);
140}

142ATF_TP_ADD_TCS(tp)
143{
144
145 ATF_TP_ADD_TC(tp, mknodat_fd);
146 ATF_TP_ADD_TC(tp, mknodat_fdcwd);
147 ATF_TP_ADD_TC(tp, mknodat_fdcwderr);
148 ATF_TP_ADD_TC(tp, mknodat_fderr);
149
150 return atf_no_error();
151}
