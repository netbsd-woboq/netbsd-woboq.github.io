45#define DIR "dir"
46#define FILE "dir/readlinkat"
47#define BASEFILE "readlinkat"
48#define LINK "dir/symlink"
49#define BASELINK "symlink"
50#define FILEERR "dir/readlinkaterr"
51
52ATF_TC(readlinkat_fd);
53ATF_TC_HEAD(readlinkat_fd, tc)
54{
55 atf_tc_set_md_var(tc, "descr", "See that readlinkat works with fd");
56}
57ATF_TC_BODY(readlinkat_fd, tc)
58{
59 int dfd;
60 int fd;
61 ssize_t len;
62 char buf[MAXPATHLEN];
63
64 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
65 ATF_REQUIRE((fd = open(FILE, O_CREAT|O_RDWR, 0644)) != -1);
66 ATF_REQUIRE(close(fd) == 0);
67 ATF_REQUIRE(symlink(FILE, LINK) == 0);
68
69 ATF_REQUIRE((dfd = open(DIR, O_RDONLY, 0)) != -1);
70 len = readlinkat(dfd, BASELINK, buf, sizeof(buf)-1);
71 ATF_REQUIRE(len != -1);
72 buf[len] = 0;
73 ATF_REQUIRE(close(dfd) == 0);
74
75 ATF_REQUIRE(strcmp(buf, FILE) == 0);
76}

78ATF_TC(readlinkat_fdcwd);
79ATF_TC_HEAD(readlinkat_fdcwd, tc)
80{
81 atf_tc_set_md_var(tc, "descr",
82 "See that readlinkat works with fd as AT_FDCWD");
83}
84ATF_TC_BODY(readlinkat_fdcwd, tc)
85{
86 int fd;
87 ssize_t len;
88 char buf[MAXPATHLEN];
89
90 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
91 ATF_REQUIRE((fd = open(FILE, O_CREAT|O_RDWR, 0644)) != -1);
92 ATF_REQUIRE(close(fd) == 0);
93 ATF_REQUIRE(symlink(FILE, LINK) == 0);
94
95 len = readlinkat(AT_FDCWD, LINK, buf, sizeof(buf)-1);
96 ATF_REQUIRE(len != -1);
97 buf[len] = 0;
98
99 ATF_REQUIRE(strcmp(buf, FILE) == 0);
100}

102ATF_TC(readlinkat_fdcwderr);
103ATF_TC_HEAD(readlinkat_fdcwderr, tc)
104{
105 atf_tc_set_md_var(tc, "descr",
106 "See that readlinkat fails with fd as AT_FDCWD and bad path");
107}
108ATF_TC_BODY(readlinkat_fdcwderr, tc)
109{
110 char buf[MAXPATHLEN];
111
112 ATF_REQUIRE(readlinkat(AT_FDCWD, LINK, buf, sizeof(buf)) == -1);
113}
114
115ATF_TC(readlinkat_fderr1);
116ATF_TC_HEAD(readlinkat_fderr1, tc)
117{
118 atf_tc_set_md_var(tc, "descr", "See that readlinkat fail with bad path");
119}
120ATF_TC_BODY(readlinkat_fderr1, tc)
121{
122 int dfd;
123
124 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
125 ATF_REQUIRE((dfd = open(DIR, O_RDONLY, 0)) != -1);
126 ATF_REQUIRE(readlinkat(dfd, FILEERR, F_OK, 0) == -1);
127 ATF_REQUIRE(close(dfd) == 0);
128}
129
130ATF_TC(readlinkat_fderr2);
131ATF_TC_HEAD(readlinkat_fderr2, tc)
132{
133 atf_tc_set_md_var(tc, "descr", "See that readlinkat fails with fd as -1");
134}
135ATF_TC_BODY(readlinkat_fderr2, tc)
136{
137 int fd;
138 char buf[MAXPATHLEN];
139
140 ATF_REQUIRE(mkdir(DIR, 0755) == 0);
141 ATF_REQUIRE((fd = open(FILE, O_CREAT|O_RDWR, 0644)) != -1);
142 ATF_REQUIRE(close(fd) == 0);
143 ATF_REQUIRE(symlink(FILE, LINK) == 0);
144
145 ATF_REQUIRE(readlinkat(-1, LINK, buf, sizeof(buf)) == -1);
146}
