32#ifndef _TESTS_EXEC_PROT_H_
33#define _TESTS_EXEC_PROT_H_
34
35/*
36 * Prototype definitions of external helper functions for executable
37 * mapping tests.
38 */
39
40/*
41 * Trivial MD shellcode that justs returns 1.
42 */
43int return_one(void); /* begin marker -- shellcode entry */
44int return_one_end(void); /* end marker */
45
46/*
47 * MD callback to verify whether host offers executable space protection.
48 * Returns execute protection level.
49 */
50int exec_prot_support(void);
51
52/* execute protection level */
53enum {
54 NOTIMPL = -1, /* callback not implemented */
55 NO_XP, /* no execute protection */
56 PERPAGE_XP, /* per-page execute protection */
57 PARTIAL_XP /* partial execute protection. Depending on where the
58 page is located in virtual memory, executable space
59 protection may be enforced or not. */
60};
61#endif
