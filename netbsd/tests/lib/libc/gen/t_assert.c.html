46static void
47disable_corefile(void)
48{
49 struct rlimit limits;
50
51 limits.rlim_cur = 0;
52 limits.rlim_max = 0;
53
54 ATF_REQUIRE(setrlimit(RLIMIT_CORE, &limits) == 0);
55}
56
57static void handler(int);
58
59static void
60handler(int signo)
61{
62 /* Nothing. */
63}
64
65ATF_TC(assert_false);
66ATF_TC_HEAD(assert_false, tc)
67{
68 atf_tc_set_md_var(tc, "descr", "Test that assert(3) works, #1");
69}
70
71ATF_TC_BODY(assert_false, tc)
72{
73 struct sigaction sa;
74 pid_t pid;
75 int sta;
76
77 pid = fork();
78 ATF_REQUIRE(pid >= 0);
79
80 if (pid == 0) {
81
82 disable_corefile();
83 (void)closefrom(0);
84 (void)memset(&sa, 0, sizeof(struct sigaction));
85
86 sa.sa_flags = 0;
87 sa.sa_handler = handler;
88
89 (void)sigemptyset(&sa.sa_mask);
90 (void)sigaction(SIGABRT, &sa, 0);
91
92 assert(1 == 1);
93
94 _exit(EXIT_SUCCESS);
95 }
96
97 (void)wait(&sta);
98
99 if (WIFSIGNALED(sta) != 0 || WIFEXITED(sta) == 0)
100 atf_tc_fail("assert(3) fired haphazardly");
101}
102
103ATF_TC(assert_true);
104ATF_TC_HEAD(assert_true, tc)
105{
106 atf_tc_set_md_var(tc, "descr", "Test that assert(3) works, #2");
107}
108
109ATF_TC_BODY(assert_true, tc)
110{
111 struct sigaction sa;
112 pid_t pid;
113 int sta;
114
115 pid = fork();
116 ATF_REQUIRE(pid >= 0);
117
118 if (pid == 0) {
119
120 disable_corefile();
121 (void)closefrom(0);
122 (void)memset(&sa, 0, sizeof(struct sigaction));
123
124 sa.sa_flags = 0;
125 sa.sa_handler = handler;
126
127 (void)sigemptyset(&sa.sa_mask);
128 (void)sigaction(SIGABRT, &sa, 0);
129
130 assert(1 == 2);
131
132 _exit(EXIT_SUCCESS);
133 }
134
135 (void)wait(&sta);
136
137 if (WIFSIGNALED(sta) == 0 || WTERMSIG(sta) != SIGABRT)
138 atf_tc_fail("assert(3) did not fire");
139}
140
141ATF_TP_ADD_TCS(tp)
142{
143
144 ATF_TP_ADD_TC(tp, assert_false);
145 ATF_TP_ADD_TC(tp, assert_true);
146
147 return atf_no_error();
148}
