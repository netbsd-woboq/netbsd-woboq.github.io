31#include <sys/cdefs.h>
32__RCSID("$NetBSD: t_cpuset.c,v 1.1 2011/11/08 05:47:00 jruoho Exp $");
33
34#include <atf-c.h>
35#include <limits.h>
36#include <stdio.h>
37#include <sched.h>
38
39ATF_TC(cpuset_err);
40ATF_TC_HEAD(cpuset_err, tc)
41{
42 atf_tc_set_md_var(tc, "descr", "Test errors from cpuset(3)");
43}
44
45ATF_TC_BODY(cpuset_err, tc)
46{
47 cpuset_t *set;
48
49 set = cpuset_create();
50 ATF_REQUIRE(set != NULL);
51
52 ATF_CHECK(cpuset_set(-1, set) == -1);
53 ATF_CHECK(cpuset_clr(-1, set) == -1);
54 ATF_CHECK(cpuset_isset(-1, set) == -1);
55
56 ATF_CHECK(cpuset_set(INT_MAX, set) == -1);
57 ATF_CHECK(cpuset_clr(INT_MAX, set) == -1);
58 ATF_CHECK(cpuset_isset(INT_MAX, set) == -1);
59
60 cpuset_destroy(set);
61}
62
63ATF_TC(cpuset_set);
64ATF_TC_HEAD(cpuset_set, tc)
65{
66 atf_tc_set_md_var(tc, "descr", "Test cpuset_set(3)");
67}
68
69ATF_TC_BODY(cpuset_set, tc)
70{
71 cpuset_t *set;
72
73 set = cpuset_create();
74 ATF_REQUIRE(set != NULL);
75
76 ATF_REQUIRE(cpuset_set(0, set) == 0);
77 ATF_REQUIRE(cpuset_isset(0, set) > 0);
78 ATF_REQUIRE(cpuset_clr(0, set) == 0);
79 ATF_REQUIRE(cpuset_isset(0, set) == 0);
80
81 cpuset_destroy(set);
82}
83
84ATF_TC(cpuset_size);
85ATF_TC_HEAD(cpuset_size, tc)
86{
87 atf_tc_set_md_var(tc, "descr", "Test puset_size(3)");
88}
89
90ATF_TC_BODY(cpuset_size, tc)
91{
92 cpuset_t *set;
93 size_t size;
94
95 set = cpuset_create();
96 ATF_REQUIRE(set != NULL);
97
98 size = cpuset_size(set);
99
100 ATF_CHECK(cpuset_set((size * 8) - 1, set) == 0);
101 ATF_CHECK(cpuset_set((size * 8) + 1, set) == -1);
102
103 cpuset_destroy(set);
104}
105
106ATF_TP_ADD_TCS(tp)
107{
108
109 ATF_TP_ADD_TC(tp, cpuset_err);
110 ATF_TP_ADD_TC(tp, cpuset_set);
111 ATF_TP_ADD_TC(tp, cpuset_size);
112
113 return atf_no_error();
114}
