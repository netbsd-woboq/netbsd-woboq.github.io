108ATF_TC(floatunditf);
109ATF_TC_HEAD(floatunditf, tc)
110{
111 atf_tc_set_md_var(tc, "descr",
112 "Verify that uint64 -> long double conversion works");
113}
114
115ATF_TC_BODY(floatunditf, tc)
116{
117#ifndef __HAVE_LONG_DOUBLE
118 atf_tc_skip("Requires long double support");
119#else
120 size_t i;
121
122 for (i = 0; i < __arraycount(testcases); ++i)
123 ATF_CHECK_MSG(
124 testcases[i].ld == (long double)testcases[i].u64,
125 "#%zu: expected %.20Lf, got %.20Lf\n", i,
126 testcases[i].ld,
127 (long double)testcases[i].u64);
128#endif
129}
130
131ATF_TP_ADD_TCS(tp)
132{
133 ATF_TP_ADD_TC(tp, floatunditf);
134 return atf_no_error();
135}
