31#include <atf-c.h>
32
33#include <stdio.h>
34#include <stdlib.h>
35
36const char *fmtcheck(const char *f1, const char *f2)
37 __attribute__((__format_arg__(2)));
38
39#include <err.h>
40
41struct test_fmt {
42 const char *fmt1;
43 const char *fmt2;
44 int correct;
45} test_fmts[] = {
46 { "%d", "%d", 1 },
47 { "%2d", "%2.2d", 1 },
48 { "%x", "%d", 1 },
49 { "%u", "%d", 1 },
50 { "%03d", "%d", 1 },
51 { "%-2d", "%d", 1 },
52 { "%d", "%-12.1d", 1 },
53 { "%d", "%-01.3d", 1 },
54 { "%X", "%-01.3d", 1 },
55 { "%D", "%ld", 1 },
56 { "%s", "%s", 1 },
57 { "%s", "This is a %s test", 1 },
58 { "Hi, there. This is a %s test", "%s", 1 },
59 { "%d", "%s", 2 },
60 { "%e", "%s", 2 },
61 { "%r", "%d", 2 },
62 { "%*.2d", "%*d", 1 },
63 { "%2.*d", "%*d", 2 },
64 { "%*d", "%*d", 1 },
65 { "%-3", "%d", 2 },
66 { "%d %s", "%d", 2 },
67 { "%*.*.*d", "%*.*.*d", 2 },
68 { "%d", "%d %s", 1 },
69 { "%40s", "%20s", 1 },
70 { "%x %x %x", "%o %u %d", 1 },
71 { "%o %u %d", "%x %x %X", 1 },
72 { "%#o %u %#-d", "%x %#x %X", 1 },
73 { "%qd", "%llx", 1 },
74 { "%%", "%llx", 1 },
75 { "%ld %30s %#llx %-10.*e", "This number %lu%% and string %s has %qd numbers and %.*g floats", 1 },
76 { "%o", "%lx", 2 },
77 { "%p", "%lu", 2 },
78};
79
80ATF_TC(fmtcheck_basic);
81ATF_TC_HEAD(fmtcheck_basic, tc)
82{
83
84 atf_tc_set_md_var(tc, "descr", "Test fmtcheck(3)");
85}
86
87ATF_TC_BODY(fmtcheck_basic, tc)
88{
89 unsigned int i, r;
90 const char *f, *cf, *f1, *f2;
91
92 r = 0;
93 for (i = 0 ; i < __arraycount(test_fmts); i++) {
94 f1 = test_fmts[i].fmt1;
95 f2 = test_fmts[i].fmt2;
96 f = fmtcheck(f1, f2);
97 if (test_fmts[i].correct == 1) {
98 cf = f1;
99 } else {
100 cf = f2;
101 }
102 if (f != cf) {
103 r++;
104 atf_tc_fail_nonfatal("Test %d: (%s) vs. (%s) failed "
105 "(should have returned %s)", i, f1, f2,
106 (test_fmts[i].correct == 1) ? "1st" : "2nd");
107 }
108 }
109}
110
111ATF_TP_ADD_TCS(tp)
112{
113
114 ATF_TP_ADD_TC(tp, fmtcheck_basic);
115
116 return atf_no_error();
117}
