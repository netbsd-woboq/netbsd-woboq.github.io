31#include <sys/cdefs.h>
32__RCSID("$NetBSD: t_ftok.c,v 1.3 2019/07/16 17:29:18 martin Exp $");
33
34#include <sys/types.h>
35#include <sys/ipc.h>
36
37#include <atf-c.h>
38#include <fcntl.h>
39#include <unistd.h>
40
41static const char *path = "ftok";
42static const char *hlnk = "hlnk";
43static const char *slnk = "slnk";
44static const int key = 123456789;
45
46ATF_TC(ftok_err);
47ATF_TC_HEAD(ftok_err, tc)
48{
49 atf_tc_set_md_var(tc, "descr", "Test errors from ftok(3)");
50}
51
52ATF_TC_BODY(ftok_err, tc)
53{
54 ATF_REQUIRE(ftok("/a/b/c/d/e/f/g/h/i", key) == -1);
55}
56
57ATF_TC_WITH_CLEANUP(ftok_link);
58ATF_TC_HEAD(ftok_link, tc)
59{
60 atf_tc_set_md_var(tc, "descr", "Test that links return the same key");
61}
62
63ATF_TC_BODY(ftok_link, tc)
64{
65 key_t k1, k2, k3;
66 int fd;
67
68 fd = open(path, O_RDONLY | O_CREAT, 0600);
69
70 ATF_REQUIRE(fd >= 0);
71 (void)close(fd);
72 ATF_REQUIRE(link(path, hlnk) == 0);
73 ATF_REQUIRE(symlink(path, slnk) == 0);
74
75 k1 = ftok(path, key);
76 k2 = ftok(hlnk, key);
77 k3 = ftok(slnk, key);
78
79 ATF_REQUIRE(k1 != -1);
80 ATF_REQUIRE(k2 != -1);
81 ATF_REQUIRE(k3 != -1);
82
83 if (k1 != k2)
84 atf_tc_fail("ftok(3) gave different key for a hard link");
85
86 if (k1 != k3)
87 atf_tc_fail("ftok(3) gave different key for a symbolic link");
88
89 ATF_REQUIRE(unlink(path) == 0);
90 ATF_REQUIRE(unlink(hlnk) == 0);
91 ATF_REQUIRE(unlink(slnk) == 0);
92}
93
94ATF_TC_CLEANUP(ftok_link, tc)
95{
96 (void)unlink(path);
97 (void)unlink(hlnk);
98 (void)unlink(slnk);
99}
100
101ATF_TP_ADD_TCS(tp)
102{
103
104 ATF_TP_ADD_TC(tp, ftok_err);
105 ATF_TP_ADD_TC(tp, ftok_link);
106
107 return atf_no_error();
108}
