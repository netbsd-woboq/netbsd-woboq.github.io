34#include <sys/resource.h>
35#include <sys/wait.h>

37#include <atf-c.h>
38#include <errno.h>
39#include <limits.h>
40#include <pthread.h>
41#include <stdlib.h>
42#include <unistd.h>
43
44static void *threadfunc(void *);
45
46static void *
47threadfunc(void *arg)
48{
49 int pri, val;
50
51 val = *(int *)arg;
52
53 errno = 0;
54 pri = getpriority(PRIO_PROCESS, 0);
55 ATF_REQUIRE(errno == 0);
56
57 if (pri != val)
58 atf_tc_fail("nice(3) value was not propagated to threads");
59
60 return NULL;
61}

63ATF_TC(nice_err);
64ATF_TC_HEAD(nice_err, tc)
65{
66 atf_tc_set_md_var(tc, "descr",
67 "Test nice(3) for invalid parameters (PR lib/42587)");
68 atf_tc_set_md_var(tc, "require.user", "unprivileged");
69}
70
71ATF_TC_BODY(nice_err, tc)
72{
73 int i;
74
75 /*
76 * The call should fail with EPERM if the
77 * supplied parameter is negative and the
78 * caller does not have privileges.
79 */
80 for (i = -20; i < 0; i++) {
81
82 errno = 0;
83
84 ATF_REQUIRE_ERRNO(EPERM, nice(i) == -1);
85 }
86}

88ATF_TC(nice_priority);
89ATF_TC_HEAD(nice_priority, tc)
90{
91 atf_tc_set_md_var(tc, "descr", "Test nice(3) vs. getpriority(2)");
92}
93
94ATF_TC_BODY(nice_priority, tc)
95{
96 int i, pri, nic;
97 pid_t pid;
98 int sta;
99
100 for (i = 0; i <= 20; i++) {
101
102 nic = nice(i);
103 ATF_REQUIRE(nic != -1);
104
105 errno = 0;
106 pri = getpriority(PRIO_PROCESS, 0);
107 ATF_REQUIRE(errno == 0);
108
109 if (nic != pri)
110 atf_tc_fail("nice(3) and getpriority(2) conflict");
111
112 /*
113 * Also verify that the nice(3) values
114 * are inherited by child processes.
115 */
116 pid = fork();
117 ATF_REQUIRE(pid >= 0);
118
119 if (pid == 0) {
120
121 errno = 0;
122 pri = getpriority(PRIO_PROCESS, 0);
123 ATF_REQUIRE(errno == 0);
124
125 if (nic != pri)
126 _exit(EXIT_FAILURE);
127
128 _exit(EXIT_SUCCESS);
129 }
130
131 (void)wait(&sta);
132
133 if (WIFEXITED(sta) == 0 || WEXITSTATUS(sta) != EXIT_SUCCESS)
134 atf_tc_fail("nice(3) value was not inherited");
135 }
136}

138ATF_TC(nice_root);
139ATF_TC_HEAD(nice_root, tc)
140{
141 atf_tc_set_md_var(tc, "descr", "Test that nice(3) works");
142 atf_tc_set_md_var(tc, "require.user", "root");
143}
144
145ATF_TC_BODY(nice_root, tc)
146{
147 int i;
148
149 for (i = -20; i <= 20; i++) {
150
151 ATF_REQUIRE(nice(i) != -1);
152 }
153}
154
155ATF_TC(nice_thread);
156ATF_TC_HEAD(nice_thread, tc)
157{
158 atf_tc_set_md_var(tc, "descr", "Test nice(3) with threads");
159}
160
161ATF_TC_BODY(nice_thread, tc)
162{
163 pthread_t tid[5];
164 int rv, val;
165 size_t i;
166
167 /*
168 * Test that the scheduling priority is
169 * propagated to all system scope threads.
170 */
171 for (i = 0; i < __arraycount(tid); i++) {
172
173 val = nice(i);
174 ATF_REQUIRE(val != -1);
175
176 rv = pthread_create(&tid[i], NULL, threadfunc, &val);
177 ATF_REQUIRE(rv == 0);
178
179 rv = pthread_join(tid[i], NULL);
180 ATF_REQUIRE(rv == 0);
181 }
182}

184ATF_TP_ADD_TCS(tp)
185{
186
187 ATF_TP_ADD_TC(tp, nice_err);
188 ATF_TP_ADD_TC(tp, nice_priority);
189 ATF_TP_ADD_TC(tp, nice_root);
190 ATF_TP_ADD_TC(tp, nice_thread);
191
192 return atf_no_error();
193}
