147static void
148digest2string(const uint8_t *digest, char *string, size_t len)
149{
150 while (len--) {
151 if (*digest / 16 < 10)
152 *string++ = '0' + *digest / 16;
153 else
154 *string++ = 'a' + *digest / 16 - 10;
155 if (*digest % 16 < 10)
156 *string++ = '0' + *digest % 16;
157 else
158 *string++ = 'a' + *digest % 16 - 10;
159 ++digest;
160 }
161 *string = '\0';
162}

164ATF_TC_HEAD(t_sha256, tc)
165{
166 atf_tc_set_md_var(tc, "descr",
167 "Test SHA256 functions for consistent results");
168}
169
170ATF_TC_BODY(t_sha256, tc)
171{
172 size_t i, j, len;
173 SHA256_CTX ctx;
174 unsigned char buf[256];
175 unsigned char digest[8 + SHA256_DIGEST_LENGTH];
176 char output[SHA256_DIGEST_STRING_LENGTH];
177
178 for (i = 0; i < sizeof(test256) / sizeof(test256[0]); ++i) {
179 len = strlen(test256[i].vector);
180 for (j = 0; j < 8; ++j) {
181 SHA256_Init(&ctx);
182 memcpy(buf + j, test256[i].vector, len);
183 SHA256_Update(&ctx, buf + j, len);
184 SHA256_Final(digest + j, &ctx);
185 digest2string(digest + j, output, SHA256_DIGEST_LENGTH);
186 ATF_CHECK_STREQ(test256[i].hash, output);
187 }
188 }
189}
