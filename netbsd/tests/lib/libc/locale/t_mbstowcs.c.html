151ATF_TC(mbstowcs_basic);
152ATF_TC_HEAD(mbstowcs_basic, tc)
153{
154 atf_tc_set_md_var(tc, "descr",
155 "Checks wide character functions with different locales");
156}
157ATF_TC_BODY(mbstowcs_basic, tc)
158{
159 struct test *t;
160
161 for (t = &tests[0]; t->data != NULL; ++t) {
162 wchar_t wbuf[SIZE];
163 char buf[SIZE];
164 char visbuf[SIZE];
165 char *str;
166 int i;
167
168 ATF_REQUIRE_STREQ(setlocale(LC_ALL, "C"), "C");
169 ATF_REQUIRE(setlocale(LC_CTYPE, t->locale) != NULL);
170
171 (void)strvis(visbuf, t->data, VIS_WHITE | VIS_OCTAL);
172 (void)printf("Checking string: \"%s\"\n", visbuf);
173
174 ATF_REQUIRE((str = setlocale(LC_ALL, NULL)) != NULL);
175 (void)printf("Using locale: %s\n", str);
176
177 REQUIRE_ERRNO((ssize_t)mbstowcs(wbuf, t->data, SIZE-1), -1);
178 REQUIRE_ERRNO((ssize_t)wcstombs(buf, wbuf, SIZE-1), -1);
179
180 if (strcmp(buf, t->data) != 0) {
181 (void)strvis(visbuf, buf, VIS_WHITE | VIS_OCTAL);
182 (void)printf("Conversion to wcs and back failed: "
183 "\"%s\"\n", visbuf);
184 atf_tc_fail("Test failed");
185 }
186
187 /* The output here is implementation-dependent. */
188
189 for (i = 0; wbuf[i] != 0; ++i) {
190 if (wbuf[i] == t->wchars[i] &&
191 wcwidth(wbuf[i]) == t->widths[i])
192 continue;
193
194 (void)printf("At position %d:\n", i);
195 (void)printf(" expected: 0x%04X (%d)\n",
196 t->wchars[i], t->widths[i]);
197 (void)printf(" got : 0x%04X (%d)\n", wbuf[i],
198 wcwidth(wbuf[i]));
199 atf_tc_fail("Test failed");
200 }
201
202 if (wcswidth(wbuf, SIZE-1) != t->width) {
203 (void)printf("Incorrect wcswidth:\n");
204 (void)printf(" expected: %d\n", t->width);
205 (void)printf(" got : %d\n", wcswidth(wbuf, SIZE-1));
206 atf_tc_fail("Test failed");
207 }
208
209 (void)printf("Ok.\n");
210 }
211}

216
217 return atf_no_error();
