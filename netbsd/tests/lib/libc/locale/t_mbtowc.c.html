61#include <errno.h>
62#include <locale.h>
63#include <stdio.h>
64#include <stdlib.h>
65#include <string.h>
66#include <vis.h>
67#include <wchar.h>

69#include <atf-c.h>

71static void
72h_mbtowc(const char *locale, const char *illegal, const char *legal, size_t stateful)
73{
74 char buf[64];
75 size_t ret;
76 char *str;
77
78 ATF_REQUIRE_STREQ(setlocale(LC_ALL, "C"), "C");
79 (void)printf("Trying locale: %s\n", locale);
80 ATF_REQUIRE(setlocale(LC_CTYPE, locale) != NULL);
81
82 ATF_REQUIRE((str = setlocale(LC_ALL, NULL)) != NULL);
83 (void)printf("Using locale: %s\n", str);
84 (void)printf("Locale is state-%sdependent\n",
85 !stateful ? "in" : "");
86
87 /* initialize internal state */
88 ret = mbtowc(NULL, NULL, 0);
89 ATF_REQUIRE(stateful ? ret : !ret);
90
91 (void)strvis(buf, illegal, VIS_WHITE | VIS_OCTAL);
92 (void)printf("Checking illegal sequence: \"%s\"\n", buf);
93
94 ret = mbtowc(NULL, illegal, strlen(illegal));
95 (void)printf("mbtowc() returned: %zd\n", ret);
96 ATF_REQUIRE_EQ(ret, (size_t)-1);
97 (void)printf("errno: %s\n", strerror(errno));
98 ATF_REQUIRE_EQ(errno, EILSEQ);
99
100 /* if this is stateless encoding, this re-initialization is not required. */
101 if (stateful) {
102 /* re-initialize internal state */
103 mbtowc(NULL, NULL, 0);
104 }
105
106 /* valid multibyte sequence case */
107 (void)strvis(buf, legal, VIS_WHITE | VIS_OCTAL);
108 (void)printf("Checking legal sequence: \"%s\"\n", buf);
109
110 errno = 0;
111 ret = mbtowc(NULL, legal, strlen(legal));
112 (void)printf("mbtowc() returned: %zd\n", ret);
113 ATF_REQUIRE(ret != (size_t)-1);
114 (void)printf("errno: %s\n", strerror(errno));
115 ATF_REQUIRE_EQ(errno, 0);
116
117 (void)printf("Ok.\n");
118}
119
120ATF_TC(mbtowc);
121ATF_TC_HEAD(mbtowc, tc)
122{
123 atf_tc_set_md_var(tc, "descr", "Checks mbtowc(3)");
124}
125ATF_TC_BODY(mbtowc, tc)
126{
127 h_mbtowc("en_US.UTF-8", "\240", "\302\240", 0);
128 h_mbtowc("ja_JP.ISO2022-JP", "\033$B", "\033$B$\"\033(B", 1);
129 h_mbtowc("ja_JP.SJIS", "\202", "\202\240", 0);
130 h_mbtowc("ja_JP.eucJP", "\244", "\244\242", 0);
131 h_mbtowc("zh_CN.GB18030", "\241", "\241\241", 0);
132 h_mbtowc("zh_TW.Big5", "\241", "\241@", 0);
133 h_mbtowc("zh_TW.eucTW", "\241", "\241\241", 0);
134}
135
136ATF_TP_ADD_TCS(tp)
137{
138 ATF_TP_ADD_TC(tp, mbtowc);
139
140 return atf_no_error();
141}
