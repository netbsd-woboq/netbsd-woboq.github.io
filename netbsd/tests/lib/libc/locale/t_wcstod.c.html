388ATF_TC(wcstombs);
389ATF_TC_HEAD(wcstombs, tc)
390{
391 atf_tc_set_md_var(tc, "descr", "Checks wcstombs(3) on the inputs to the wcstod test");
392}
393ATF_TC_BODY(wcstombs, tc)
394{
395#if !defined(__vax__)
396 struct test *t;
397 size_t n;
398 char *buf;
399
400 /*
401 * Previously this was part of the wcstod test.
402 */
403 for (t = &tests[0]; t->wcs != NULL; ++t) {
404 printf("wcslen(\"%S\") = %zu\n", t->wcs, wcslen(t->wcs));
405 n = wcstombs(NULL, t->wcs, 0);
406 printf("wcstombs(NULL, \"%S\", 0) = %d\n", t->wcs, (int)n);
407 ATF_REQUIRE((buf = (void *)calloc(1, n + 1)) != NULL);
408 (void)wcstombs(buf, t->wcs, n + 1);
409 printf("wcstombs(buf, \"%S\", %d) = \"%s\"\n", t->wcs,
410 (int)(n + 1), buf);
411 ATF_REQUIRE_EQ(strlen(buf), wcslen(t->wcs));
412
413 free(buf);
414 }
415#endif
416}
417
418ATF_TC(wcstod);
419ATF_TC_HEAD(wcstod, tc)
420{
421 atf_tc_set_md_var(tc, "descr", "Checks wcstod(3)");
422}
423ATF_TC_BODY(wcstod, tc)
424{
425#if defined(__vax__)
426#else
427 struct test *t;
428#endif
429
430#if !defined(__vax__)
431 for (t = &tests[0]; t->wcs != NULL; ++t) {
432 size_t n;
433 double d;
434 wchar_t *tail;
435
436 (void)printf("Checking wcstod(\"%S\", &tail):\n", t->wcs);
437
438 errno = 0;
439 d = wcstod(t->wcs, &tail);
440 (void)printf("[errno]\n");
441 (void)printf(" got : %s\n", strerror(errno));
442 (void)printf(" expected: %s\n", strerror(t->err));
443 ATF_REQUIRE_EQ(errno, t->err);
444
445 n = (size_t)(tail - t->wcs);
446 (void)printf("[endptr - nptr]\n");
447 (void)printf(" got : %zd\n", n);
448 (void)printf(" expected: %zd\n", t->len);
449 ATF_REQUIRE_EQ(n, t->len);
450
451 (void)printf("[result]\n");
452 (void)printf(" real: %F\n", d);
453 if (t->val == ALT_HUGE_VAL) {
454 (void)printf(" expected: %F\n", HUGE_VAL);
455 ATF_REQUIRE(isinf(d));
456 ATF_REQUIRE_EQ(d, HUGE_VAL);
457 } else if (t->val == ALT_MINUS_HUGE_VAL) {
458 (void)printf(" expected: %F\n", -HUGE_VAL);
459 ATF_REQUIRE(isinf(d));
460 ATF_REQUIRE_EQ(d, -HUGE_VAL);
461 } else if (t->val == ALT_NAN) {
462 (void)printf(" expected: %F\n", NAN);
463 ATF_REQUIRE(isnan(d));
464 } else {
465 (void)printf(" expected: %F\n", t->val);
466 ATF_REQUIRE_EQ(d, t->val);
467 }
468
469 (void)printf("\n");
470 }
471#else /* !defined(__vax__) */
472 atf_tc_skip("Test is unavailable on vax.");
473#endif /* !defined(__vax__) */
474}
