70#define TC_WCTOMB 0
71#define TC_WCRTOMB 1
72#define TC_WCRTOMB_ST 2
73
74static struct test {
75 const char *locale;
76 const char *data;
77 size_t wclen;
78 size_t mblen[16];
79 size_t stateful;
80} tests[] = {
81{
82 "ja_JP.ISO2022-JP",
83 "\x1b$B" /* JIS X 0208-1983 */
84 "\x46\x7c\x4b\x5c\x38\x6c" /* "nihongo" */
85 "\x1b(B" /* ISO 646 */
86 "ABC"
87 "\x1b(I" /* JIS X 0201 katakana */
88 "\xb1\xb2\xb3" /* "aiu" */
89 "\x1b(B", /* ISO 646 */
90 3 + 3 + 3,
91 { 3+2, 2, 2, 3+1, 1, 1, 3+1, 1, 1, 3+1 },
92 1,
93}, {
94 "C",
95 "ABC",
96 3,
97 { 1, 1, 1, 1 },
98 0,
99}, { NULL, NULL, 0, { }, 0 }
100};
101
102static void
103h_wctomb(const struct test *t, char tc)
104{
105 wchar_t wcs[16 + 2];
106 char buf[128];
107 char cs[MB_LEN_MAX];
108 const char *pcs;
109 char *str;
110 mbstate_t st;
111 mbstate_t *stp = NULL;
112 size_t sz, ret, i;
113
114 ATF_REQUIRE_STREQ(setlocale(LC_ALL, "C"), "C");
115 (void)printf("Trying locale: %s\n", t->locale);
116 ATF_REQUIRE(setlocale(LC_CTYPE, t->locale) != NULL);
117
118 if (tc == TC_WCRTOMB_ST) {
119 (void)memset(&st, 0, sizeof(st));
120 stp = &st;
121 } else {
122 (void)printf("Checking correct reporting of statefulness\n");
123 ret = wctomb(NULL, 0);
124 ATF_REQUIRE_EQ(t->stateful, !!ret);
125 }
126
127 (void)strvis(buf, t->data, VIS_WHITE | VIS_OCTAL);
128 (void)printf("Checking sequence: \"%s\"\n", buf);
129
130 ATF_REQUIRE((str = setlocale(LC_ALL, NULL)) != NULL);
131 (void)printf("Using locale: %s\n", str);
132
133 wcs[t->wclen] = L'X'; /* poison */
134 pcs = t->data;
135 sz = mbsrtowcs(wcs, &pcs, t->wclen + 2, NULL);
136 ATF_REQUIRE_EQ_MSG(sz, t->wclen, "mbsrtowcs() returned: "
137 "%zu, expected: %zu", sz, t->wclen);
138 ATF_REQUIRE_EQ(wcs[t->wclen], 0);
139
140 for (i = 0; i < t->wclen + 1; i++) {
141 if (tc == TC_WCTOMB)
142 ret = wctomb(cs, wcs[i]);
143 else
144 ret = wcrtomb(cs, wcs[i], stp);
145
146 if (ret == t->mblen[i])
147 continue;
148
149 (void)printf("At position %zd:\n", i);
150 (void)printf(" expected: %zd\n", t->mblen[i]);
151 (void)printf(" got : %zd\n", ret);
152 (void)strvis(buf, cs, VIS_WHITE | VIS_OCTAL);
153 (void)printf(" sequence: \"%s\"\n", buf);
154 atf_tc_fail("Test failed");
155 /* NOTREACHED */
156 }
157
158 (void)printf("Ok.\n");
159}

161ATF_TC(wctomb);
162ATF_TC_HEAD(wctomb, tc)
163{
164 atf_tc_set_md_var(tc, "descr", "Checks wctomb(3)");
165}
166ATF_TC_BODY(wctomb, tc)
167{
168 struct test *t;
169
170 (void)printf("Checking wctomb()\n");
171
172 for (t = &tests[0]; t->data != NULL; ++t)
173 h_wctomb(t, TC_WCTOMB);
174}

192ATF_TC(wcrtomb);
193ATF_TC_HEAD(wcrtomb, tc)
194{
195 atf_tc_set_md_var(tc, "descr",
196 "Checks wcrtomb(3) (using internal state)");
197}
198ATF_TC_BODY(wcrtomb, tc)
199{
200 struct test *t;
201
202 (void)printf("Checking wcrtomb() (using internal state)\n");

205 h_wctomb(t, TC_WCRTOMB);
206}
