39#include <assert.h>
40#include <nsswitch.h>
41#include <pthread.h>
42#include <stdarg.h>
43#include <stdio.h>
44#include <stdlib.h>
45
46static const ns_src testsrc[] = {
47 { "test", NS_SUCCESS },
48 { NULL, 0 }
49};
50
51static int
52func3(void *rv, void *cb_data, va_list ap)
53{
54 (void)printf("func3: enter\n");
55 (void)printf("func3: exit\n");
56
57 return NS_SUCCESS;
58}
59
60static int
61func2(void *rv, void *cb_data, va_list ap)
62{
63 static const ns_dtab dtab[] = {
64 { "test", func3, NULL },
65 { NULL, NULL, NULL }
66 };
67 int r;
68
69 (void)printf("func2: enter\n");
70 r = nsdispatch(NULL, dtab, "test", "test", testsrc);
71 (void)printf("func2: exit\n");
72
73 return r;
74}
75
76static int
77func1(void)
78{
79 static const ns_dtab dtab[] = {
80 { "test", func2, NULL },
81 { NULL, NULL, NULL }
82 };
83 int r;
84
85 (void)printf("func1: enter\n");
86 r = nsdispatch(NULL, dtab, "test", "test", testsrc);
87 (void)printf("func1: exit\n");
88
89 return r;
90}
91
92static void *
93thrfunc(void *arg)
94{
95 pthread_exit(NULL);
96}
97
98int
99main(int argc, char *argv[])
100{
101 pthread_t thr;
102 void *threval;
103
104 assert(pthread_create(&thr, NULL, thrfunc, NULL) == 0);
105 assert(func1() == NS_SUCCESS);
106 assert(pthread_join(thr, &threval) == 0);
107}
