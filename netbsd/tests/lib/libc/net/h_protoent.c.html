29#include <netdb.h>
30#include <stdint.h>
31#include <stdlib.h>
32#include <unistd.h>
33#include <netinet/in.h>
34#include <sys/types.h>
35#include <stdio.h>
36
37static void
38pserv(const struct protoent *prp)
39{
40 char **pp;
41
42 printf("name=%s, proto=%d, aliases=",
43 prp->p_name, prp->p_proto);
44 for (pp = prp->p_aliases; *pp; pp++)
45 printf("%s ", *pp);
46 printf("\n");
47}
48
49static void
50usage(void)
51{
52 (void)fprintf(stderr, "Usage: %s\n"
53 "\t%s -p <proto>\n"
54 "\t%s -n <name>\n", getprogname(), getprogname(),
55 getprogname());
56 exit(1);
57}
58
59int
60main(int argc, char *argv[])
61{
62 struct protoent *prp;
63 const char *proto = NULL, *name = NULL;
64 int c;
65
66 while ((c = getopt(argc, argv, "p:n:")) != -1) {
67 switch (c) {
68 case 'n':
69 name = optarg;
70 break;
71 case 'p':
72 proto = optarg;
73 break;
74 default:
75 usage();
76 }
77 }
78
79 if (proto && name)
80 usage();
81 if (proto) {
82 if ((prp = getprotobynumber(atoi(proto))) != NULL)
83 pserv(prp);
84 return 0;
85 }
86 if (name) {
87 if ((prp = getprotobyname(name)) != NULL)
88 pserv(prp);
89 return 0;
90 }
91
92 setprotoent(0);
93 while ((prp = getprotoent()) != NULL)
94 pserv(prp);
95 endprotoent();
96 return 0;
97}
