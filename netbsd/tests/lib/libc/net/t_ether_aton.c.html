41#include <atf-c.h>

49#define ETHER_ADDR_LEN 6
50
51int ether_aton_r(u_char *dest, size_t len, const char *str);
52
53static const struct {
54 u_char res[ETHER_ADDR_LEN];
55 const char *str;
56 int error;
57} tests[] = {
58 { { 0x01, 0x23, 0x45, 0x67, 0x89, 0xab }, "01:23:45:67:89:ab", 0 },
59 { { 0x00, 0x01, 0x22, 0x03, 0x14, 0x05 }, "0:1:22-3:14:05", 0 },
60 { { 0x00, 0x01, 0x22, 0x03, 0x14, 0x05 }, "000122031405", 0 },
61 { { 0x0a, 0x0B, 0xcc, 0xdD, 0xEE, 0x0f }, "0a0BccdDEE0f", 0 },
62#define ZERO { 0, 0, 0, 0, 0, 0 }
63 { ZERO, "0:1:2-3:04:05:06", ENAMETOOLONG },
64 { ZERO, "0:1:2-3:04:", ENOBUFS },
65 { ZERO, "0:1:2-3:04:x7", EINVAL },
66 { ZERO, "1:x-3:04:05:7", EINVAL },
67 { ZERO, NULL, 0 },
68};
69
70ATF_TC(tc_ether_aton);
71ATF_TC_HEAD(tc_ether_aton, tc)
72{
73 atf_tc_set_md_var(tc, "descr", "Check that ether_aton(3) works");
74}
75
76ATF_TC_BODY(tc_ether_aton, tc)
77{
78 u_char dest[ETHER_ADDR_LEN];
79 size_t t;
80 int e, r;
81 const char *s;
82
83 for (t = 0; tests[t].str; t++) {
84 s = tests[t].str;
85 if ((e = tests[t].error) == 0) {
86 if (ether_aton_r(dest, sizeof(dest), s) != e)
87 atf_tc_fail("failed on `%s'", s);
88 if (memcmp(dest, tests[t].res, sizeof(dest)) != 0)
89 atf_tc_fail("unexpected result on `%s'", s);
90 } else {
91 if ((r = ether_aton_r(dest, sizeof(dest), s)) != e)
92 atf_tc_fail("unexpectedly succeeded on `%s' "
93 "(%d != %d)", s, r, e);
94 }
95 }
96}
97
98ATF_TP_ADD_TCS(tp)
99{
100
101 ATF_TP_ADD_TC(tp, tc_ether_aton);
102
103 return atf_no_error();
104}
