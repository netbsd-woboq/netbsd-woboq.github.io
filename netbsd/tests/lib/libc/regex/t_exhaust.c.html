50#ifndef REGEX_MAXSIZE
51#define REGEX_MAXSIZE 9999

54static char *
55mkstr(const char *str, size_t len)
56{
57 size_t slen = strlen(str);
58 char *p = malloc(slen * len + 1);
59 ATF_REQUIRE_MSG(p != NULL, "slen=%zu, len=%zu", slen, len);
60 for (size_t i = 0; i < len; i++)
61 strcpy(&p[i * slen], str);
62 return p;
63}
64
65static char *
66concat(const char *d, const char *s)
67{
68 size_t dlen = strlen(d);
69 size_t slen = strlen(s);
70 char *p = malloc(dlen + slen + 1);
71
72 ATF_REQUIRE(p != NULL);
73 strcpy(p, d);
74 strcpy(p + dlen, s);
75 return p;
76}

78static char *
79p0(size_t len)
80{
81 char *d, *s1, *s2;
82 s1 = mkstr("\\(", len);
83 s2 = concat(s1, ")");
84 free(s1);
85 d = concat("(", s2);
86 free(s2);
87 return d;
88}
89
90static char *
91p1(size_t len)
92{
93 char *d, *s1, *s2, *s3;
94 s1 = mkstr("\\(", 60);
95 s2 = mkstr("(.*)", len);
96 s3 = concat(s1, s2);
97 free(s2);
98 free(s1);
99 s1 = concat(s3, ")");
100 free(s3);
101 d = concat("(", s1);
102 free(s1);
103 return d;
104}

106static char *
107ps(const char *m, const char *s, size_t len)
108{
109 char *d, *s1, *s2, *s3;
110 s1 = mkstr(m, len);
111 s2 = mkstr(s, len);
112 s3 = concat(s1, s2);
113 free(s2);
114 free(s1);
115 d = concat("(.?)", s3);
116 free(s3);
117 return d;
118}
119
120static char *
121p2(size_t len)
122{
123 return ps("((.*){0,255}", ")", len);
124}
125
126static char *
127p3(size_t len)
128{
129 return ps("(.\\{0,}", ")", len);
130}
131
132static char *
133p4(size_t len)
134{
135 return ps("((.*){1,255}", ")", len);
136}
137
138static char *
139p5(size_t len)
140{
141 return ps("(", "){1,100}", len);
142}
143
144static char *
145p6(size_t len)
146{
147 char *d, *s1, *s2;
148 s1 = mkstr("(?:(.*)|", len);
149 s2 = concat(s1, "(.*)");
150 free(s1);
151 s1 = mkstr(")", len);
152 d = concat(s2, s1);
153 free(s1);
154 free(s2);
155 return d;
156}

158static const struct {
159 char *(*pattern)(size_t);
160 int type;
161} tests[] = {
162 { p0, REG_EXTENDED },
163 { p1, REG_EXTENDED },
164 { p2, REG_EXTENDED },
165 { p3, REG_EXTENDED },
166 { p4, REG_EXTENDED },
167 { p5, REG_EXTENDED },
168 { p6, REG_BASIC },
169};
170
171ATF_TC(regcomp_too_big);
172
173ATF_TC_HEAD(regcomp_too_big, tc)
174{
175
176 atf_tc_set_md_var(tc, "descr", "Check that large patterns don't"
177 " crash, but return a proper error code");
178 // libtre needs it.
179 atf_tc_set_md_var(tc, "timeout", "600");
180 atf_tc_set_md_var(tc, "require.memory", "256M");
181}
182
183ATF_TC_BODY(regcomp_too_big, tc)
184{
185 regex_t re;
186 int e;
187 struct rlimit limit;
188
189 limit.rlim_cur = limit.rlim_max = 256 * 1024 * 1024;
190 ATF_REQUIRE(setrlimit(RLIMIT_VMEM, &limit) != -1);
191
192 for (size_t i = 0; i < __arraycount(tests); i++) {
193 char *d = (*tests[i].pattern)(REGEX_MAXSIZE);
194 e = regcomp(&re, d, tests[i].type);
195 if (e) {
196 char ebuf[1024];
197 (void)regerror(e, &re, ebuf, sizeof(ebuf));
198 ATF_REQUIRE_MSG(e == REG_ESPACE,
199 "regcomp returned %d (%s) for pattern %zu [%s]", e, ebuf,
200 i, d);
201 free(d);
202 continue;
203 }
204 free(d);
205 (void)regexec(&re, "aaaaaaaaaaa", 0, NULL, 0);
206 regfree(&re);
207 }
208}
209
210ATF_TP_ADD_TCS(tp)
211{
212
213 ATF_TP_ADD_TC(tp, regcomp_too_big);
214 return atf_no_error();
215}
