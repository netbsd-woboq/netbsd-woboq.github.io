38int
39main(int argc, char *argv[])
40{
41 char b[10];
42 char *q = stpcpy(b, argv[1]);
43
44 if ((size_t)(q - b) != strlen(argv[1]))
45 abort();
46
47 (void)printf("%s\n", b);
48 return 0;
49}
