34#include <stdio.h>
35#include <stdlib.h>
36#include <stdarg.h>
37
38static void wrap(size_t, const char *, ...) __printflike(2, 3);
39
40void
41wrap(size_t len, const char *fmt, ...)
42{
43 char b[10];
44 va_list ap;
45 va_start(ap, fmt);
46 (void)vsnprintf(b, len, fmt, ap);
47 (void)printf("%s\n", b);
48 va_end(ap);
49}
50
51int
52main(int argc, char *argv[])
53{
54 size_t len = atoi(argv[1]);
55 wrap(len, "%s", "012345678901234567890");
56 return 0;
57}
