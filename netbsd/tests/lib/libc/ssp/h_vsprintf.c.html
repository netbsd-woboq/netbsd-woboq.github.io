34#include <stdio.h>
35#include <stdarg.h>
36
37static void wrap(const char *, ...) __printflike(1, 2);
38
39static void
40wrap(const char *fmt, ...)
41{
42 char b[10];
43 va_list ap;
44 va_start(ap, fmt);
45 (void)vsprintf(b, fmt, ap);
46 (void)printf("%s\n", b);
47 va_end(ap);
48}
49
50int
51main(int argc, char *argv[])
52{
53 wrap("%s", argv[1]);
54 return 0;
55}
