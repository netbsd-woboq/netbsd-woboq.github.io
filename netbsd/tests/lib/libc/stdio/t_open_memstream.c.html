28ATF_TC(test_open_memstream);
29ATF_TC_HEAD(test_open_memstream, tc)
30{
31 atf_tc_set_md_var(tc, "descr", "Test open_memstream functionality");
32}
33
34#define OFFSET 16384
35
36const char start[] = "start";
37const char hello[] = "hello";
38
39ATF_TC_BODY(test_open_memstream, tc)
40{
41 FILE *fp;
42 char *buf = (char *)0xff;
43 size_t size = 0;
44 off_t off;
45 int i;
46
47 fp = open_memstream(&buf, &size);
48 ATF_REQUIRE(fp != NULL);
49
50 off = ftello(fp);
51 ATF_CHECK(off == 0);
52
53 ATF_CHECK(fflush(fp) == 0);
54 ATF_CHECK(size == 0);
55 ATF_CHECK(buf != (char *)0xff);
56 ATF_CHECK(fseek(fp, -6, SEEK_SET) == -1);
57 ATF_CHECK(fseek(fp, OFFSET, SEEK_SET) == 0);
58 ATF_CHECK(fprintf(fp, hello) != EOF);
59 ATF_CHECK(fflush(fp) != EOF);
60 ATF_CHECK(size == OFFSET + sizeof(hello)-1);
61 ATF_CHECK(fseek(fp, 0, SEEK_SET) == 0);
62 ATF_CHECK(fprintf(fp, start) != EOF);
63 ATF_CHECK(fflush(fp) != EOF);
64 ATF_CHECK(size == sizeof(start)-1);
65
66 /* Needed for sparse files */
67 ATF_CHECK(strncmp(buf, start, sizeof(start)-1) == 0);
68 for (i = sizeof(start)-1; i < OFFSET; i++)
69 ATF_CHECK(buf[i] == '\0');
70
71 ATF_CHECK(memcmp(buf + OFFSET, hello, sizeof(hello)-1) == 0);
72
73 /* verify that simply seeking past the end doesn't increase the size */
74 ATF_CHECK(fseek(fp, 100, SEEK_END) == 0);
75 ATF_CHECK(fflush(fp) != EOF);
76 ATF_CHECK(size == OFFSET + sizeof(hello)-1);
77 ATF_CHECK(fseek(fp, 8, SEEK_SET) == 0);
78 ATF_CHECK(ftell(fp) == 8);
79
80 /* Try to seek backward */
81 ATF_CHECK(fseek(fp, -1, SEEK_CUR) == 0);
82 ATF_CHECK(ftell(fp) == 7);
83 ATF_CHECK(fseek(fp, 5, SEEK_CUR) == 0);
84 ATF_CHECK(fclose(fp) != EOF);
85 ATF_CHECK(size == 12);
86
87 free(buf);
88}

90ATF_TP_ADD_TCS(tp)
91{
92
93 ATF_TP_ADD_TC(tp, test_open_memstream);
94
95 return atf_no_error();
96}
