29#include <atf-c.h>
30#include <math.h>
31#include <stdio.h>
32#include <string.h>
33
34#define NUM -0x1234
35#define STRNUM ___STRING(NUM)
36
37ATF_TC(sscanf_neghex);
38ATF_TC_HEAD(sscanf_neghex, tc)
39{
40 atf_tc_set_md_var(tc, "descr",
41 "PR lib/21691: %%i and %%x fail with negative hex numbers");
42}
43
44ATF_TC_BODY(sscanf_neghex, tc)
45{
46 int i;
47
48 sscanf(STRNUM, "%i", &i);
49 ATF_REQUIRE(i == NUM);
50
51 sscanf(STRNUM, "%x", &i);
52 ATF_REQUIRE(i == NUM);
53}
54
55ATF_TC(sscanf_whitespace);
56ATF_TC_HEAD(sscanf_whitespace, tc)
57{
58
59 atf_tc_set_md_var(tc, "descr", "verify sscanf skips all whitespace");
60}
61
62ATF_TC_BODY(sscanf_whitespace, tc)
63{
64 const char str[] = "\f\n\r\t\v%z";
65 char c;
66
67 /* set of "white space" symbols from isspace(3) */
68 c = 0;
69 (void)sscanf(str, "%%%c", &c);
70 ATF_REQUIRE(c == 'z');
71}
72
73
74ATF_TP_ADD_TCS(tp)
75{
76
77 ATF_TP_ADD_TC(tp, sscanf_neghex);
78 ATF_TP_ADD_TC(tp, sscanf_whitespace);
79
80 return atf_no_error();
81}
