39ATF_TC(abs_basic);
40ATF_TC_HEAD(abs_basic, tc)
41{
42 atf_tc_set_md_var(tc, "descr", "Test that abs(3) works");
43}
44
45ATF_TC_BODY(abs_basic, tc)
46{
47 static const struct {
48 int val;
49 int res;
50 } table[] = {
51 { 0, 0 },
52 { +0, 0 },
53 { -0, 0 },
54 { -0x1010, 0x1010 },
55 { INT_MAX, INT_MAX },
56 { -INT_MAX, INT_MAX },
57 };
58
59 for (size_t i = 0; i < __arraycount(table); i++)
60 ATF_CHECK(abs(table[i].val) == table[i].res);
61}

123{

127 } table[] = {

131 { -1, 1 },

135 { -LONG_MAX, LONG_MAX },
136 { LLONG_MAX, LLONG_MAX },
