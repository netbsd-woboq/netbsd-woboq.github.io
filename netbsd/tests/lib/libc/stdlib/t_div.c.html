29#include <atf-c.h>
30
31#include <stdio.h>
32#include <stdlib.h>
33
34#define NUM 1999236
35#define DENOM 1000000
36#define QUOT 1
37#define REM 999236
38
39ATF_TC(div_basic);
40ATF_TC_HEAD(div_basic, tc)
41{
42
43 atf_tc_set_md_var(tc, "descr", "Test div(3) for correctness");
44}
45
46ATF_TC_BODY(div_basic, tc)
47{
48 div_t d;
49
50 d = div(NUM, DENOM);
51
52 ATF_CHECK(d.quot == QUOT);
53 ATF_CHECK(d.rem == REM);
54}
55

89
90ATF_TP_ADD_TCS(tp)
91{
92
93 ATF_TP_ADD_TC(tp, div_basic);
94 ATF_TP_ADD_TC(tp, ldiv_basic);
95 ATF_TP_ADD_TC(tp, lldiv_basic);
96
97 return atf_no_error();
98}
