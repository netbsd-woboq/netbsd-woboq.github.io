42ATF_TC(random_same);
43ATF_TC_HEAD(random_same, tc)
44{
45 atf_tc_set_md_var(tc, "descr",
46 "Test that random(3) does not always return the same "
47 "value when the seed is initialized to zero");
48}
49
50#define MAX_ITER 10
51
52ATF_TC_BODY(random_same, tc)
53{
54 long buf[MAX_ITER];
55 size_t i, j;
56
57 /*
58 * See CVE-2012-1577.
59 */
60 srandom(0);
61
62 for (i = 0; i < __arraycount(buf); i++) {
63
64 buf[i] = random();
65
66 for (j = 0; j < i; j++) {
67
68 (void)fprintf(stderr, "i = %zu, j = %zu: "
69 "%ld vs. %ld\n", i, j, buf[i], buf[j]);
70
71 ATF_CHECK(buf[i] != buf[j]);
72 }
73 }
74}
75
76ATF_TP_ADD_TCS(tp)
77{
78
79 ATF_TP_ADD_TC(tp, random_same);
80
81 return atf_no_error();
82}
