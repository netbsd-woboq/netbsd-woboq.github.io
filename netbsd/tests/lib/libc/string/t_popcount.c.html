40static unsigned int byte_count[256];
41
42static void
43popcount_init(const char *cfg_var)
44{
45 unsigned int i, j;
46
47 if (strcasecmp(cfg_var, "YES") == 0 ||
48 strcasecmp(cfg_var, "Y") == 0 ||
49 strcasecmp(cfg_var, "1") == 0 ||
50 strcasecmp(cfg_var, "T") == 0 ||
51 strcasecmp(cfg_var, "TRUE") == 0) {
52 for (i = 0; i < 256; ++i) {
53 byte_count[i] = 0;
54 for (j = i; j != 0; j >>= 1) {
55 if (j & 1)
56 ++byte_count[i];
57 }
58 }
59 return;
60 }
61
62 atf_tc_skip("config variable \"run_popcount\" not set to YES/TRUE");
63}

132ATF_TC(popcount_basic);
133ATF_TC_HEAD(popcount_basic, tc)
134{
135
136 atf_tc_set_md_var(tc, "descr", "Test popcount results");
137 atf_tc_set_md_var(tc, "timeout", "0");
138}
139
140ATF_TC_BODY(popcount_basic, tc)
141{
142 unsigned int i, r;
143
144 popcount_init(atf_tc_get_config_var_wd(tc, "run_popcount", "NO"));
145
146 for (i = 0; i < 0xffffffff; ++i) {
147 r = byte_count[i & 255] + byte_count[(i >> 8) & 255]
148 + byte_count[(i >> 16) & 255]
149 + byte_count[(i >> 24) & 255];
150
151 ATF_CHECK_EQ(r, popcount(i));
152 }
153 ATF_CHECK_EQ(popcount(0xffffffff), 32);
154}

156ATF_TC(popcountll_basic);
157ATF_TC_HEAD(popcountll_basic, tc)
158{
159
160 atf_tc_set_md_var(tc, "descr", "Test popcountll results");
161 atf_tc_set_md_var(tc, "timeout", "0");
162}
163
164ATF_TC_BODY(popcountll_basic, tc)
165{
166 unsigned int i, j, r, r2, p;
167 unsigned long long v;
168
169 popcount_init(atf_tc_get_config_var_wd(tc, "run_popcount", "NO"));
170
171 for (j = 0; j < 256; ++j) {
172 p = test_parts[j];
173 r2 = byte_count[p & 255] + byte_count[(p >> 8) & 255]
174 + byte_count[(p >> 16) & 255]
175 + byte_count[(p >> 24) & 255];
176
177 for (i = 0; i < 0xffffffff; ++i) {
178 r = byte_count[i & 255] + byte_count[(i >> 8) & 255]
179 + byte_count[(i >> 16) & 255]
180 + byte_count[(i >> 24) & 255] + r2;
181
182 v = (((unsigned long long)i) << 32) + p;
183 ATF_CHECK_EQ(r, popcountll(v));
184 v = (((unsigned long long)p) << 32) + i;
185 ATF_CHECK_EQ(r, popcountll(v));
186 }
187 }
188
189 ATF_CHECK_EQ(popcountll(0xffffffffffffffffULL), 64);
190}
