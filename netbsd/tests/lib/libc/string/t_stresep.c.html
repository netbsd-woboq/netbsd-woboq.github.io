32#include <atf-c.h>
33
34#include <stdio.h>
35#include <stdlib.h>
36#include <string.h>
37
38#define expect(a) \
39 if ((p = stresep(&q, " ", '\\')) == NULL || strcmp(p, a)) { \
40 fprintf(stderr, "failed on line %d: %s != %s\n", \
41 __LINE__, p, a); \
42 atf_tc_fail("Check stderr for test id/line"); \
43 }
44
45ATF_TC(stresep_basic);
46ATF_TC_HEAD(stresep_basic, tc)
47{
48
49 atf_tc_set_md_var(tc, "descr", "Test stresep results");
50}
51
52ATF_TC_BODY(stresep_basic, tc)
53{
54 char brkstr[] = "foo\\ \\ bar baz bar\\ foo\\ bar\\ \\ foo \\ \\ \\ "
55 "baz bar\\ \\ ";
56 char *p, *q = brkstr;
57
58 expect("foo bar");
59 expect("baz");
60 expect("bar foo ");
61 expect("bar foo");
62 expect(" baz");
63 expect("bar ");
64
65 char brkstr2[] = "aa bb cc\\ \\ \\ \\ dd-";
66 q = brkstr2;
67 expect("aa");
68 expect("bb");
69 expect("cc dd-");
70}
71
72ATF_TP_ADD_TCS(tp)
73{
74
75 ATF_TP_ADD_TC(tp, stresep_basic);
76
77 return atf_no_error();
78}
