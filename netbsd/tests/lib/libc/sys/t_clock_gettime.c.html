79#define MINPOSDIFF 15000000 /* 15 ms for now */
80#define TIMEOUT 5
81
82#define TC_HARDWARE "kern.timecounter.hardware"
83#define TC_CHOICE "kern.timecounter.choice"
84
85static void
86check_timecounter(void)
87{
88 struct timespec tsa, tsb, tsl, res;
89 long long mindiff = INTMAX_MAX;
90 time_t endlimit;
91
92#define CL(x) \
93 do { \
94 if ((x) != -1) \
95 break; \
96 atf_tc_fail_nonfatal("%s: %s", #x, strerror(errno)); \
97 return; \
98 } while (0)
99
100 CL(clock_gettime(CLOCK_REALTIME, &tsa));
101 tsl = tsa;
102
103 CL(time(&endlimit));
104 endlimit += TIMEOUT + 1;
105
106 while ((time_t)tsa.tv_sec < endlimit) {
107 long long diff;
108
109 CL(clock_gettime(CLOCK_REALTIME, &tsb));
110 diff = 1000000000LL * (tsb.tv_sec - tsa.tv_sec)
111 + tsb.tv_nsec - tsa.tv_nsec;
112
113 if (diff > 0 && mindiff > diff)
114 mindiff = diff;
115
116 if (diff < 0 || diff > MINPOSDIFF) {
117 long long elapsed;
118 (void)printf("%stime TSA: 0x%jx.%08jx, TSB: 0x%jx.%08jx, "
119 "diff = %lld nsec, ", (diff < 0) ? "BAD " : "",
120 (uintmax_t)tsa.tv_sec, (uintmax_t)tsa.tv_nsec,
121 (uintmax_t)tsb.tv_sec, (uintmax_t)tsb.tv_nsec, diff);
122
123 elapsed = 1000000000LL * (tsb.tv_sec - tsl.tv_sec)
124 + tsb.tv_nsec - tsl.tv_nsec;
125
126
127 (void)printf("%lld nsec\n", elapsed);
128 tsl = tsb;
129
130 ATF_CHECK(diff >= 0);
131 if (diff < 0)
132 return;
133 }
134
135 tsa.tv_sec = tsb.tv_sec;
136 tsa.tv_nsec = tsb.tv_nsec;
137 }
138
139 if (clock_getres(CLOCK_REALTIME, &res) == 0) {
140 long long r = res.tv_sec * 1000000000 + res.tv_nsec;
141
142 (void)printf("Claimed resolution: %lld nsec (%f Hz) or "
143 "better\n", r, 1.0 / r * 1e9);
144 (void)printf("Observed minimum non zero delta: %lld "
145 "nsec\n", mindiff);
146 }
147
148#undef CL
149}

151ATF_TC(clock_gettime_real);
152ATF_TC_HEAD(clock_gettime_real, tc)
153{
154 atf_tc_set_md_var(tc, "require.user", "root");
155 atf_tc_set_md_var(tc, "descr",
156 "Checks the monotonicity of the CLOCK_REALTIME implementation");
157 atf_tc_set_md_var(tc, "timeout", "300");
158}
159
160ATF_TC_BODY(clock_gettime_real, tc)
161{
162 char name[128], cbuf[512], ctrbuf[10240];
163 size_t cbufsiz = sizeof(cbuf);
164 size_t ctrbufsiz = sizeof(ctrbuf);
165 const char *p;
166 char *save;
167 int quality, n;
168
169 if (sysctlbyname(TC_HARDWARE, cbuf, &cbufsiz, NULL, 0) != 0) {
170 (void)printf("\nChecking legacy time implementation "
171 "for %d seconds\n", TIMEOUT);
172 check_timecounter();
173 return;
174 /* NOTREACHED */
175 }
176 (void)printf("%s = %s\n", TC_HARDWARE, cbuf);
177 REQUIRE_LIBC(save = strdup(cbuf), NULL);
178
179 RL(sysctlbyname(TC_CHOICE, ctrbuf, &ctrbufsiz, NULL, 0));
180 (void)printf("%s = %s\n", TC_CHOICE, ctrbuf);
181
182 for (p = ctrbuf, n = 0; sscanf(p, "%127[^(](q=%d, f=%*u Hz)%*[ ]%n",
183 name, &quality, &n) == 2; p += n) {
184 struct timespec ts;
185 int ret;
186
187 if (quality < 0)
188 continue;
189
190 (void)printf("\nChecking %s for %d seconds\n", name, TIMEOUT);
191 CHECK_LIBC(ret = sysctlbyname(TC_HARDWARE, NULL, 0,
192 name, strlen(name)), -1);
193 if (ret == -1)
194 continue;
195
196 /* wait a bit to select new counter in clockinterrupt */
197 ts.tv_sec = 0;
198 ts.tv_nsec = 100000000;
199 (void)nanosleep(&ts, NULL);
200
201 check_timecounter();
202 }
203
204 RL(sysctlbyname(TC_HARDWARE, NULL, 0, save, strlen(save)));
205}
