49static char path[] = "dup";
50static void check_mode(bool, bool, bool);
51
52static void
53check_mode(bool _dup, bool _dup2, bool _dup3)
54{
55 int mode[3] = { O_RDONLY, O_WRONLY, O_RDWR };
56 int perm[5] = { 0700, 0400, 0600, 0444, 0666 };
57 struct stat st, st1;
58 int fd, fd1, fd2;
59 size_t i, j;
60
61 /*
62 * Check that a duplicated descriptor
63 * retains the mode of the original file.
64 */
65 for (i = 0; i < __arraycount(mode); i++) {
66
67 for (j = 0; j < __arraycount(perm); j++) {
68
69 fd1 = open(path, mode[i] | O_CREAT, perm[j]);
70 fd2 = open("/etc/passwd", O_RDONLY);
71
72 ATF_REQUIRE(fd1 >= 0);
73 ATF_REQUIRE(fd2 >= 0);
74
75 if (_dup != false)
76 fd = dup(fd1);
77 else if (_dup2 != false)
78 fd = dup2(fd1, fd2);
79 else if (_dup3 != false)
80 fd = dup3(fd1, fd2, O_CLOEXEC);
81 else {
82 fd = -1;
83 }
84
85 ATF_REQUIRE(fd >= 0);
86
87 (void)memset(&st, 0, sizeof(struct stat));
88 (void)memset(&st1, 0, sizeof(struct stat));
89
90 ATF_REQUIRE(fstat(fd, &st) == 0);
91 ATF_REQUIRE(fstat(fd1, &st1) == 0);
92
93 if (st.st_mode != st1.st_mode)
94 atf_tc_fail("invalid mode");
95
96 (void)close(fd);
97 (void)close(fd1);
98 (void)close(fd2);
99 (void)unlink(path);
100 }
101 }
102}

104ATF_TC(dup2_basic);
105ATF_TC_HEAD(dup2_basic, tc)
106{
107 atf_tc_set_md_var(tc, "descr", "A basic test of dup2(2)");
108}
109
110ATF_TC_BODY(dup2_basic, tc)
111{
112 int fd, fd1, fd2;
113
114 fd1 = open("/etc/passwd", O_RDONLY);
115 fd2 = open("/etc/passwd", O_RDONLY);
116
117 ATF_REQUIRE(fd1 >= 0);
118 ATF_REQUIRE(fd2 >= 0);
119
120 fd = dup2(fd1, fd2);
121 ATF_REQUIRE(fd >= 0);
122
123 if (fd != fd2)
124 atf_tc_fail("invalid descriptor");
125
126 (void)close(fd);
127 (void)close(fd1);
128
129 ATF_REQUIRE(close(fd2) != 0);
130}

132ATF_TC(dup2_err);
133ATF_TC_HEAD(dup2_err, tc)
134{
135 atf_tc_set_md_var(tc, "descr", "Test error conditions of dup2(2)");
136}
137
138ATF_TC_BODY(dup2_err, tc)
139{
140 int fd;
141
142 fd = open("/etc/passwd", O_RDONLY);
143 ATF_REQUIRE(fd >= 0);
144
145 errno = 0;
146 ATF_REQUIRE_ERRNO(EBADF, dup2(-1, -1) == -1);
147
148 errno = 0;
149 ATF_REQUIRE_ERRNO(EBADF, dup2(fd, -1) == -1);
150
151 errno = 0;
152 ATF_REQUIRE_ERRNO(EBADF, dup2(-1, fd) == -1);
153
154 /*
155 * Note that this should not fail with EINVAL.
156 */
157 ATF_REQUIRE(dup2(fd, fd) != -1);
158
159 (void)close(fd);
160}

196ATF_TC(dup3_err);
197ATF_TC_HEAD(dup3_err, tc)
198{
199 atf_tc_set_md_var(tc, "descr",
200 "Test error conditions of dup3(2) (PR lib/45148)");
201}
202
203ATF_TC_BODY(dup3_err, tc)
204{
205 int fd;
206
207 fd = open("/etc/passwd", O_RDONLY);
208 ATF_REQUIRE(fd >= 0);
209
210 errno = 0;
211 ATF_REQUIRE(dup3(fd, fd, O_CLOEXEC) != -1);
212
213 errno = 0;
214 ATF_REQUIRE_ERRNO(EBADF, dup3(-1, -1, O_CLOEXEC) == -1);
215
216 errno = 0;
217 ATF_REQUIRE_ERRNO(EBADF, dup3(fd, -1, O_CLOEXEC) == -1);
218
219 errno = 0;
220 ATF_REQUIRE_ERRNO(EBADF, dup3(-1, fd, O_CLOEXEC) == -1);
221
222 errno = 0;
223 ATF_REQUIRE_ERRNO(EINVAL, dup3(fd, 1, O_NOFOLLOW) == -1);
224
225 (void)close(fd);
226}

275ATF_TC_WITH_CLEANUP(dup_max);
276ATF_TC_HEAD(dup_max, tc)
277{
278 atf_tc_set_md_var(tc, "descr", "Test dup(2) against limits");
279}
280
281ATF_TC_BODY(dup_max, tc)
282{
283 struct rlimit res;
284 int *buf, fd, sta;
285 size_t i, n;
286 pid_t pid;
287
288 pid = fork();
289 ATF_REQUIRE(pid >= 0);
290
291 if (pid == 0) {
292
293 /*
294 * Open a temporary file until the
295 * maximum number of open files is
296 * reached. Ater that dup(2) family
297 * should fail with EMFILE.
298 */
299 (void)closefrom(0);
300 (void)memset(&res, 0, sizeof(struct rlimit));
301
302 n = 10;
303 res.rlim_cur = res.rlim_max = n;
304 if (setrlimit(RLIMIT_NOFILE, &res) != 0)
305 _exit(EX_OSERR);
306
307 buf = calloc(n, sizeof(int));
308
309 if (buf == NULL)
310 _exit(EX_OSERR);
311
312 buf[0] = mkstemp(path);
313
314 if (buf[0] < 0)
315 _exit(EX_OSERR);
316
317 for (i = 1; i < n; i++) {
318
319 buf[i] = open(path, O_RDONLY);
320
321 if (buf[i] < 0)
322 _exit(EX_OSERR);
323 }
324
325 errno = 0;
326 fd = dup(buf[0]);
327
328 if (fd != -1 || errno != EMFILE)
329 _exit(EX_DATAERR);
330
331 _exit(EXIT_SUCCESS);
332 }
333
334 (void)wait(&sta);
335
336 if (WIFEXITED(sta) == 0 || WEXITSTATUS(sta) != EXIT_SUCCESS) {
337
338 if (WEXITSTATUS(sta) == EX_OSERR)
339 atf_tc_fail("system call error");
340
341 if (WEXITSTATUS(sta) == EX_DATAERR)
342 atf_tc_fail("dup(2) dupped more than RLIMIT_NOFILE");
343
344 atf_tc_fail("unknown error");
345 }
346
347 (void)unlink(path);
348}
