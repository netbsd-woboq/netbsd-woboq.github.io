43ATF_TC(fsync_err);
44ATF_TC_HEAD(fsync_err, tc)
45{
46 atf_tc_set_md_var(tc, "descr",
47 "Test error conditions of fsync(2) (PR kern/30)");
48}
49
50ATF_TC_BODY(fsync_err, tc)
51{
52 int i, fd[2];
53
54 /*
55 * The fsync(2) call should fail with EBADF
56 * when the 'fd' is not a valid descriptor.
57 */
58 for (i = 1; i < 1024; i = i + 128) {
59
60 errno = 0;
61
62 ATF_REQUIRE(fsync(-i) == -1);
63 ATF_REQUIRE(errno == EBADF);
64 }
65
66 /*
67 * On the other hand, EINVAL should follow
68 * if the operation is not possible with
69 * the file descriptor.
70 */
71 ATF_REQUIRE(pipe(fd) == 0);
72
73 errno = 0;
74
75 ATF_REQUIRE(fsync(fd[0]) == -1);
76 ATF_REQUIRE(errno == EINVAL);
77
78 errno = 0;
79
80 ATF_REQUIRE(fsync(fd[1]) == -1);
81 ATF_REQUIRE(errno == EINVAL);
82
83 ATF_REQUIRE(close(fd[0]) == 0);
84 ATF_REQUIRE(close(fd[1]) == 0);
85}
86
87ATF_TC(fsync_sync);
88ATF_TC_HEAD(fsync_sync, tc)
89{
90 atf_tc_set_md_var(tc, "descr", "A basic test of fsync(2)");
91}
92
93ATF_TC_BODY(fsync_sync, tc)
94{
95 char buf[128];
96 int fd, i;
97
98 for (i = 0; i < 10; i++) {
99
100 (void)snprintf(buf, sizeof(buf), "t_fsync-%d", i);
101
102 fd = mkstemp(buf);
103
104 ATF_REQUIRE(fd != -1);
105 ATF_REQUIRE(write(fd, "0", 1) == 1);
106 ATF_REQUIRE(fsync(fd) == 0);
107
108 ATF_REQUIRE(unlink(buf) == 0);
109 ATF_REQUIRE(close(fd) == 0);
110 }
111}
112
113ATF_TP_ADD_TCS(tp)
114{
115
116 ATF_TP_ADD_TC(tp, fsync_err);
117 ATF_TP_ADD_TC(tp, fsync_sync);
118
119 return atf_no_error();
120}
