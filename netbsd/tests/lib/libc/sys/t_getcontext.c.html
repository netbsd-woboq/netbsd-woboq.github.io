34#include <atf-c.h>
35#include <errno.h>
36#include <stdarg.h>
37#include <stdlib.h>
38#include <ucontext.h>
39
40#define STACKSZ (10*1024)
41#define DEPTH 3
42
43static int calls;
44
45static void
46run(int n, ...)
47{
48 va_list va;
49 int i, ia;
50
51 ATF_REQUIRE_EQ(n, DEPTH - calls - 1);
52
53 va_start(va, n);
54 for (i = 0; i < 9; i++) {
55 ia = va_arg(va, int);
56 ATF_REQUIRE_EQ(i, ia);
57 }
58 va_end(va);
59
60 calls++;
61}

63ATF_TC(getcontext_err);
64ATF_TC_HEAD(getcontext_err, tc)
65{
66 atf_tc_set_md_var(tc, "descr", "Test errors from getcontext(2)");
67}
68
69ATF_TC_BODY(getcontext_err, tc)
70{
71
72 errno = 0;
73 ATF_REQUIRE_ERRNO(EFAULT, getcontext((void *)-1) == -1);
74}

76ATF_TC(setcontext_err);

79 atf_tc_set_md_var(tc, "descr", "Test errors from setcontext(2)");
80}
81
82ATF_TC_BODY(setcontext_err, tc)
83{
84
85 errno = 0;
86 ATF_REQUIRE_ERRNO(EFAULT, setcontext((void *)-1) == -1);
87}
88
89ATF_TC(setcontext_link);
90ATF_TC_HEAD(setcontext_link, tc)
91{
92
93 atf_tc_set_md_var(tc, "descr",
94 "Checks get/make/setcontext(), context linking via uc_link(), "
95 "and argument passing to the new context");
96}
97
98ATF_TC_BODY(setcontext_link, tc)
99{
100 ucontext_t uc[DEPTH];
101 ucontext_t save;
102 volatile int i = 0; /* avoid longjmp clobbering */
103
104 for (i = 0; i < DEPTH; ++i) {
105 ATF_REQUIRE_EQ(getcontext(&uc[i]), 0);
106
107 uc[i].uc_stack.ss_sp = malloc(STACKSZ);
108 uc[i].uc_stack.ss_size = STACKSZ;
109 uc[i].uc_link = (i > 0) ? &uc[i - 1] : &save;
110
111 makecontext(&uc[i], (void *)run, 10, i,
112 0, 1, 2, 3, 4, 5, 6, 7, 8);
113 }
114
115 ATF_REQUIRE_EQ(getcontext(&save), 0);
116
117 if (calls == 0)
118 ATF_REQUIRE_EQ(setcontext(&uc[DEPTH-1]), 0);
119}
120
121ATF_TP_ADD_TCS(tp)
122{
123
124 ATF_TP_ADD_TC(tp, getcontext_err);
125 ATF_TP_ADD_TC(tp, setcontext_err);
126 ATF_TP_ADD_TC(tp, setcontext_link);
127
128 return atf_no_error();
129}
