31#include <sys/cdefs.h>
32__RCSID("$NetBSD: t_getpid.c,v 1.1 2011/07/07 06:57:53 jruoho Exp $");

34#include <sys/wait.h>
35
36#include <stdlib.h>
37#include <pthread.h>
38#include <unistd.h>
39
40#include <atf-c.h>
41
42static int maxiter = 10;
43static void *threadfunc(void *);
44
45static void *
46threadfunc(void *arg)
47{
48 *(pid_t *)arg = getpid();
49
50 return NULL;
51}
52
53ATF_TC(getpid_process);
54ATF_TC_HEAD(getpid_process, tc)
55{
56 atf_tc_set_md_var(tc, "descr", "Test getpid(2) with processes");
57}
58
59ATF_TC_BODY(getpid_process, tc)
60{
61 pid_t ppid, fpid, cpid, tpid, wpid;
62 int i, sta;
63
64 for (i = 0; i < maxiter; i++) {
65
66 tpid = getpid();
67 fpid = fork();
68
69 ATF_REQUIRE(fpid >= 0);
70
71 if (fpid == 0) {
72
73 cpid = getpid();
74 ppid = getppid();
75
76 if (tpid != ppid)
77 _exit(EXIT_FAILURE);
78
79 if (cpid == ppid)
80 _exit(EXIT_FAILURE);
81
82 if (tpid == fpid)
83 _exit(EXIT_FAILURE);
84
85 _exit(EXIT_SUCCESS);
86 }
87
88 wpid = wait(&sta);
89
90 if (wpid != fpid)
91 atf_tc_fail("PID mismatch");
92
93 ATF_REQUIRE(WIFEXITED(sta) != 0);
94
95 if (WEXITSTATUS(sta) != EXIT_SUCCESS)
96 atf_tc_fail("PID mismatch");
97 }
98}

100ATF_TC(getpid_thread);
101ATF_TC_HEAD(getpid_thread, tc)
102{
103 atf_tc_set_md_var(tc, "descr", "Test getpid(2) with threads");
104}
105
106ATF_TC_BODY(getpid_thread, tc)
107{
108 pid_t pid, tpid;
109 pthread_t tid;
110 int i, rv;
111
112 for (i = 0; i < maxiter; i++) {
113
114 pid = getpid();
115
116 rv = pthread_create(&tid, NULL, threadfunc, &tpid);
117 ATF_REQUIRE(rv == 0);
118
119 rv = pthread_join(tid, NULL);
120 ATF_REQUIRE(rv == 0);
121
122 if (pid != tpid)
123 atf_tc_fail("Unequal PIDs");
124 }
125}
126
127ATF_TP_ADD_TCS(tp)
128{
129
130 ATF_TP_ADD_TC(tp, getpid_process);
131 ATF_TP_ADD_TC(tp, getpid_thread);
132
133 return atf_no_error();
134}
