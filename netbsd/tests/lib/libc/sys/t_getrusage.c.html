49static void work(void);
50static void sighandler(int);
51
52static const size_t maxiter = 2000;
53
54static void
55sighandler(int signo __unused)
56{
57 /* Nothing. */
58}
59
60static void
61work(void)
62{
63 size_t n = UINT16_MAX * 10;
64
65 while (n > 0) {
66#ifdef __or1k__
67 asm volatile("l.nop"); /* Do something. */
68#elif defined(__ia64__)
69 asm volatile("nop 0"); /* Do something. */
70#else
71 asm volatile("nop"); /* Do something. */
72#endif
73 n--;
74 }
75}

77ATF_TC(getrusage_err);
78ATF_TC_HEAD(getrusage_err, tc)
79{
80 atf_tc_set_md_var(tc, "descr", "Test error conditions");
81}
82
83ATF_TC_BODY(getrusage_err, tc)
84{
85 struct rusage ru;
86
87 errno = 0;
88
89 ATF_REQUIRE(getrusage(INT_MAX, &ru) != 0);
90 ATF_REQUIRE(errno == EINVAL);
91
92 errno = 0;
93
94 ATF_REQUIRE(getrusage(RUSAGE_SELF, (void *)0) != 0);
95 ATF_REQUIRE(errno == EFAULT);
96}

98ATF_TC(getrusage_sig);
99ATF_TC_HEAD(getrusage_sig, tc)
100{
101 atf_tc_set_md_var(tc, "descr", "Test signal count with getrusage(2)");
102}
103
104ATF_TC_BODY(getrusage_sig, tc)
105{
106 struct rusage ru;
107 const long n = 5;
108 int i;
109
110 /*
111 * Test that signals are recorded.
112 */
113 ATF_REQUIRE(signal(SIGUSR1, sighandler) != SIG_ERR);
114
115 for (i = 0; i < n; i++)
116 ATF_REQUIRE(raise(SIGUSR1) == 0);
117
118 (void)memset(&ru, 0, sizeof(struct rusage));
119 ATF_REQUIRE(getrusage(RUSAGE_SELF, &ru) == 0);
120
121 if (n != ru.ru_nsignals)
122 atf_tc_fail("getrusage(2) did not record signals");
123}

125ATF_TC(getrusage_maxrss);
126ATF_TC_HEAD(getrusage_maxrss, tc)
127{
128 atf_tc_set_md_var(tc, "descr", "Test maxrss growing with getrusage(2)");
129}
130
131ATF_TC_BODY(getrusage_maxrss, tc)
132{
133 struct rusage ru;
134 long maxrss;
135 int i, fd;
136
137#define DUMP_FILE "dump"
138
139 fd = open(DUMP_FILE, O_WRONLY|O_CREAT|O_TRUNC, 0222);
140 ATF_REQUIRE(fd != -1);
141
142 (void)memset(&ru, 0, sizeof(struct rusage));
143 ATF_REQUIRE(getrusage(RUSAGE_SELF, &ru) == 0);
144 maxrss = ru.ru_maxrss;
145
146#define CHUNK (1024 * 1024)
147 for (i = 0; i < 40; i++) {
148 void *p = malloc(CHUNK);
149 memset(p, 0, CHUNK);
150 write(fd, p, CHUNK);
151 }
152 close(fd);
153 unlink(DUMP_FILE);
154
155 ATF_REQUIRE(getrusage(RUSAGE_SELF, &ru) == 0);
156 ATF_REQUIRE_MSG(maxrss < ru.ru_maxrss,
157 "maxrss: %ld, ru.ru_maxrss: %ld", maxrss, ru.ru_maxrss);
158}

160ATF_TC(getrusage_msgsnd);
161ATF_TC_HEAD(getrusage_msgsnd, tc)
162{
163 atf_tc_set_md_var(tc, "descr", "Test send growing with getrusage(2)");
164}
165
166ATF_TC_BODY(getrusage_msgsnd, tc)
167{
168 struct rusage ru;
169 long msgsnd;
170 int s, i;
171 struct sockaddr_in sin;
172
173 ATF_REQUIRE(getrusage(RUSAGE_SELF, &ru) == 0);
174 msgsnd = ru.ru_msgsnd;
175
176 s = socket(AF_INET, SOCK_DGRAM, 0);
177 ATF_REQUIRE(s >= 0);
178 memset(&sin, 0, sizeof(sin));
179 sin.sin_family = AF_INET;
180 sin.sin_len = sizeof(sin);
181 sin.sin_addr.s_addr = ntohl(INADDR_LOOPBACK);
182 sin.sin_port = htons(3333);
183
184 for (i = 0; i < 10; i++)
185 ATF_REQUIRE(sendto(s, &sin, sizeof(sin), 0, (void *)&sin,
186 (socklen_t)sizeof(sin)) != -1);
187
188 ATF_REQUIRE(getrusage(RUSAGE_SELF, &ru) == 0);
189 ATF_REQUIRE(msgsnd + 10 == ru.ru_msgsnd);
190 close(s);
191}

193ATF_TC(getrusage_utime_back);
194ATF_TC_HEAD(getrusage_utime_back, tc)
195{
196 atf_tc_set_md_var(tc, "descr", "Test bogus values from getrusage(2)");
197}
198
199ATF_TC_BODY(getrusage_utime_back, tc)
200{
201 struct rusage ru1, ru2;
202 size_t i;
203
204 /*
205 * Test that two consecutive calls are sane.
206 */
207 atf_tc_expect_fail("PR kern/30115");
208
209 for (i = 0; i < maxiter; i++) {
210
211 (void)memset(&ru1, 0, sizeof(struct rusage));
212 (void)memset(&ru2, 0, sizeof(struct rusage));
213
214 work();
215
216 ATF_REQUIRE(getrusage(RUSAGE_SELF, &ru1) == 0);
217
218 work();
219
220 ATF_REQUIRE(getrusage(RUSAGE_SELF, &ru2) == 0);
221
222 if (timercmp(&ru2.ru_utime, &ru1.ru_utime, <) != 0)
223 atf_tc_fail("user time went backwards");
224 }
225
226 atf_tc_fail("anticipated error did not occur");
227}

229ATF_TC(getrusage_utime_zero);
230ATF_TC_HEAD(getrusage_utime_zero, tc)
231{
232 atf_tc_set_md_var(tc, "descr", "Test zero utime from getrusage(2)");
233}
234
235ATF_TC_BODY(getrusage_utime_zero, tc)
236{
237 struct rusage ru;
238 size_t i;
239
240 /*
241 * Test that getrusage(2) does not return
242 * zero user time for the calling process.
243 *
244 * See also (duplicate) PR port-amd64/41734.
245 */
246 atf_tc_expect_fail("PR kern/30115");
247
248 for (i = 0; i < maxiter; i++) {
249
250 work();
251
252 (void)memset(&ru, 0, sizeof(struct rusage));
253
254 ATF_REQUIRE(getrusage(RUSAGE_SELF, &ru) == 0);
255
256 if (ru.ru_utime.tv_sec == 0 && ru.ru_utime.tv_usec == 0)
257 atf_tc_fail("zero user time from getrusage(2)");
258 }
259
260 atf_tc_fail("anticipated error did not occur");
261}
