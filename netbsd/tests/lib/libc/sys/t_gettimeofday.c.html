34#include <sys/time.h>
35
36#include <atf-c.h>
37#include <errno.h>
38#include <string.h>
39
40ATF_TC(gettimeofday_err);
41ATF_TC_HEAD(gettimeofday_err, tc)
42{
43 atf_tc_set_md_var(tc, "descr", "Test errors from gettimeofday(2)");
44}
45
46ATF_TC_BODY(gettimeofday_err, tc)
47{
48
49 errno = 0;
50
51 ATF_REQUIRE_ERRNO(EFAULT, gettimeofday((void *)-1, NULL) != 0);
52}
53
54ATF_TC(gettimeofday_mono);
55ATF_TC_HEAD(gettimeofday_mono, tc)
56{
57 atf_tc_set_md_var(tc, "descr", "Test monotonicity of gettimeofday(2)");
58}
59
60ATF_TC_BODY(gettimeofday_mono, tc)
61{
62 static const size_t maxiter = 100;
63 struct timeval tv1, tv2;
64 size_t i;
65
66 for (i = 0; i < maxiter; i++) {
67
68 (void)memset(&tv1, 0, sizeof(struct timeval));
69 (void)memset(&tv2, 0, sizeof(struct timeval));
70
71 ATF_REQUIRE(gettimeofday(&tv1, NULL) == 0);
72 ATF_REQUIRE(gettimeofday(&tv2, NULL) == 0);
73
74 if (timercmp(&tv2, &tv1, <) != 0)
75 atf_tc_fail("time went backwards");
76 }
77}
78
79ATF_TP_ADD_TCS(tp)
80{
81
82 ATF_TP_ADD_TC(tp, gettimeofday_err);
83 ATF_TP_ADD_TC(tp, gettimeofday_mono);
84
85 return atf_no_error();
86}
