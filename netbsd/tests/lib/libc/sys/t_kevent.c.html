52ATF_TC(kevent_zerotimer);
53ATF_TC_HEAD(kevent_zerotimer, tc)
54{
55 atf_tc_set_md_var(tc, "descr", "Checks that kevent with a 0 timer "
56 "does not crash the system (PR lib/45618)");
57}
58
59ATF_TC_BODY(kevent_zerotimer, tc)
60{
61 struct kevent ev;
62 int kq;
63
64 ATF_REQUIRE((kq = kqueue()) != -1);
65 EV_SET(&ev, 1, EVFILT_TIMER, EV_ADD|EV_ENABLE, 0, 1, 0);
66 ATF_REQUIRE(kevent(kq, &ev, 1, NULL, 0, NULL) != -1);
67 ATF_REQUIRE(kevent(kq, NULL, 0, &ev, 1, NULL) == 1);
68}

70ATF_TC(kqueue_desc_passing);
71ATF_TC_HEAD(kqueue_desc_passing, tc)
72{
73 atf_tc_set_md_var(tc, "descr", "Checks that passing a kqueue to "
74 "another process does not crash the kernel (PR 46463)");
75}
76
77ATF_TC_BODY(kqueue_desc_passing, tc)
78{
79 pid_t child;
80 int s[2], storage, status, kq;
81 struct cmsghdr *msg;
82 struct iovec iov;
83 struct msghdr m;
84 struct kevent ev;
85
86 ATF_REQUIRE((kq = kqueue()) != -1);
87
88 // atf_tc_skip("crashes kernel (PR 46463)");
89
90 ATF_REQUIRE(socketpair(AF_LOCAL, SOCK_STREAM, 0, s) != -1);
91 msg = malloc(CMSG_SPACE(sizeof(int)));
92 m.msg_iov = &iov;
93 m.msg_iovlen = 1;
94 m.msg_name = NULL;
95 m.msg_namelen = 0;
96 m.msg_control = msg;
97 m.msg_controllen = CMSG_SPACE(sizeof(int));
98
99 child = fork();
100 if (child == 0) {
101 close(s[0]);
102
103 iov.iov_base = &storage;
104 iov.iov_len = sizeof(int);
105 m.msg_iov = &iov;
106 m.msg_iovlen = 1;
107
108 if (recvmsg(s[1], &m, 0) == -1)
109 err(1, "child: could not recvmsg");
110
111 kq = *(int *)CMSG_DATA(msg);
112 printf("child (pid %d): received kq fd %d\n", getpid(), kq);
113 exit(0);
114 }
115
116 close(s[1]);
117
118 iov.iov_base = &storage;
119 iov.iov_len = sizeof(int);
120
121 msg->cmsg_level = SOL_SOCKET;
122 msg->cmsg_type = SCM_RIGHTS;
123 msg->cmsg_len = CMSG_LEN(sizeof(int));
124
125 *(int *)CMSG_DATA(msg) = kq;
126
127 EV_SET(&ev, 1, EVFILT_TIMER, EV_ADD|EV_ENABLE, 0, 1, 0);
128 ATF_CHECK(kevent(kq, &ev, 1, NULL, 0, NULL) != -1);
129
130 printf("parent (pid %d): sending kq fd %d\n", getpid(), kq);
131 if (sendmsg(s[0], &m, 0) == -1) {
132 ATF_REQUIRE_EQ_MSG(errno, EBADF, "errno is %d", errno);
133 atf_tc_skip("PR kern/46523");
134 }
135
136 close(kq);
137
138 waitpid(child, &status, 0);
139 ATF_CHECK(WIFEXITED(status) && WEXITSTATUS(status)==0);
140}

142ATF_TC(kqueue_unsupported_fd);
143ATF_TC_HEAD(kqueue_unsupported_fd, tc)
144{
145 atf_tc_set_md_var(tc, "descr", "Checks that watching an fd whose"
146 " type is not supported does not crash the kernel");
147}
148
149ATF_TC_BODY(kqueue_unsupported_fd, tc)
150{
151 /* mqueue and semaphore use fnullop_kqueue also */
152 int fd, kq;
153 struct kevent ev;
154
155 fd = open(DRVCTLDEV, O_RDONLY);
156 if (fd == -1) {
157 switch (errno) {
158 case ENOENT:
159 case ENXIO:
160 atf_tc_skip("no " DRVCTLDEV " available for testing");
161 break;
162 }
163 }
164 ATF_REQUIRE(fd != -1);
165 ATF_REQUIRE((kq = kqueue()) != -1);
166
167 EV_SET(&ev, fd, EVFILT_VNODE, EV_ADD | EV_ENABLE | EV_CLEAR,
168 NOTE_DELETE|NOTE_WRITE|NOTE_EXTEND|NOTE_ATTRIB|NOTE_LINK|
169 NOTE_RENAME|NOTE_REVOKE, 0, 0);
170
171 ATF_REQUIRE(kevent(kq, &ev, 1, NULL, 0, NULL) == -1);
172 ATF_REQUIRE_ERRNO(EOPNOTSUPP, true);
173
174 (void)close(fd);
175}
