34#include <sys/lwpctl.h>

36#include <atf-c.h>
37#include <lwp.h>
38#include <stdio.h>
39#include <time.h>
40
41ATF_TC(lwpctl_counter);
42ATF_TC_HEAD(lwpctl_counter, tc)
43{
44 atf_tc_set_md_var(tc, "descr", "Checks lwpctl preemption counter");
45}
46
47ATF_TC_BODY(lwpctl_counter, tc)
48{
49 lwpctl_t *lc;
50 struct timespec ts;
51 int ctr1, ctr2;
52
53 ATF_REQUIRE(_lwp_ctl(LWPCTL_FEATURE_PCTR, &lc) == 0);
54
55 /* Ensure that preemption is reported. */
56 ctr1 = lc->lc_pctr;
57 (void)printf("pctr = %d\n", ctr1);
58 ts.tv_nsec = 10*1000000;
59 ts.tv_sec = 0;
60
61 ATF_REQUIRE(nanosleep(&ts, 0) != -1);
62
63 ctr2 = lc->lc_pctr;
64 (void)printf("pctr = %d\n", ctr2);
65
66 ATF_REQUIRE_MSG(ctr1 != ctr2, "counters match");
67}
68
69ATF_TP_ADD_TCS(tp)
70{
71 ATF_TP_ADD_TC(tp, lwpctl_counter);
72
73 return atf_no_error();
74}
