48static void
49run(int flags)
50{
51 int fd[2], i;
52
53 while ((i = open("/", O_RDONLY)) < 3)
54 ATF_REQUIRE(i != -1);
55
56 ATF_REQUIRE_MSG(closefrom(3) != -1, "closefrom failed: %s",
57 strerror(errno));
58
59 ATF_REQUIRE(pipe2(fd, flags) == 0);
60
61 ATF_REQUIRE(fd[0] == 3);
62 ATF_REQUIRE(fd[1] == 4);
63
64 if (flags & O_CLOEXEC) {
65 ATF_REQUIRE((fcntl(fd[0], F_GETFD) & FD_CLOEXEC) != 0);
66 ATF_REQUIRE((fcntl(fd[1], F_GETFD) & FD_CLOEXEC) != 0);
67 } else {
68 ATF_REQUIRE((fcntl(fd[0], F_GETFD) & FD_CLOEXEC) == 0);
69 ATF_REQUIRE((fcntl(fd[1], F_GETFD) & FD_CLOEXEC) == 0);
70 }
71
72 if (flags & O_NONBLOCK) {
73 ATF_REQUIRE((fcntl(fd[0], F_GETFL) & O_NONBLOCK) != 0);
74 ATF_REQUIRE((fcntl(fd[1], F_GETFL) & O_NONBLOCK) != 0);
75 } else {
76 ATF_REQUIRE((fcntl(fd[0], F_GETFL) & O_NONBLOCK) == 0);
77 ATF_REQUIRE((fcntl(fd[1], F_GETFL) & O_NONBLOCK) == 0);
78 }
79
80 if (flags & O_NOSIGPIPE) {
81 ATF_REQUIRE(fcntl(fd[0], F_GETNOSIGPIPE) != 0);
82 ATF_REQUIRE(fcntl(fd[1], F_GETNOSIGPIPE) != 0);
83 } else {
84 ATF_REQUIRE(fcntl(fd[0], F_GETNOSIGPIPE) == 0);
85 ATF_REQUIRE(fcntl(fd[1], F_GETNOSIGPIPE) == 0);
86 }
87
88 ATF_REQUIRE(close(fd[0]) != -1);
89 ATF_REQUIRE(close(fd[1]) != -1);
90}

103ATF_TC(pipe2_consume);
104ATF_TC_HEAD(pipe2_consume, tc)
105{
106 atf_tc_set_md_var(tc, "descr", "Test that consuming file descriptors "
107 "with pipe2(2) does not crash the system (PR kern/46457)");
108}
109
110ATF_TC_BODY(pipe2_consume, tc)
111{
112 struct rlimit rl;
113 int err, filedes[2];
114 int old;
115
116 ATF_REQUIRE_MSG(closefrom(4) != -1, "closefrom failed: %s",
117 strerror(errno));
118
119 err = getrlimit(RLIMIT_NOFILE, &rl);
120 ATF_REQUIRE(err == 0);
121 /*
122 * The heart of this test is to run against the number of open
123 * file descriptor limit in the middle of a pipe2() call - i.e.
124 * before the call only a single descriptor may be openend.
125 */
126 old = rl.rlim_cur;
127 rl.rlim_cur = 4;
128 err = setrlimit(RLIMIT_NOFILE, &rl);
129 ATF_REQUIRE(err == 0);
130
131 err = pipe2(filedes, O_CLOEXEC);
132 ATF_REQUIRE(err == -1);
133 rl.rlim_cur = old;
134 err = setrlimit(RLIMIT_NOFILE, &rl);
135}

170ATF_TC(pipe2_einval);
171ATF_TC_HEAD(pipe2_einval, tc)
172{
173 atf_tc_set_md_var(tc, "descr", "A error check of pipe2(2)");
174}
175
176ATF_TC_BODY(pipe2_einval, tc)
177{
178 int fd[2];
179 ATF_REQUIRE_ERRNO(EINVAL, pipe2(fd, O_ASYNC) == -1);
180}
