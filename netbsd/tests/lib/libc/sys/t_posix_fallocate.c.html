32#include <sys/cdefs.h>
33__RCSID("$NetBSD: t_posix_fallocate.c,v 1.1 2015/01/31 23:06:57 christos Exp $");
34
35#include <atf-c.h>
36#include <errno.h>
37#include <fcntl.h>
38#include <unistd.h>
39
40ATF_TC_WITHOUT_HEAD(ebadf);
41ATF_TC_BODY(ebadf, tc)
42{
43 int rc, saved;
44
45 errno = 1111;
46 rc = posix_fallocate(-1, 0, 4096);
47 saved = errno;
48 if (rc == -1 && saved != 1111)
49 atf_tc_fail("Should return error %s without setting errno.",
50 strerror(saved));
51 if (rc != EBADF)
52 atf_tc_fail("returned %s but expected %s.",
53 strerror(saved), strerror(EBADF));
54 if (saved != 1111)
55 atf_tc_fail("errno should be %d but got changed to %d.",
56 1111, saved);
57}
58
59ATF_TP_ADD_TCS(tp)
60{
61 ATF_TP_ADD_TC(tp, ebadf);
62 return atf_no_error();
63}
