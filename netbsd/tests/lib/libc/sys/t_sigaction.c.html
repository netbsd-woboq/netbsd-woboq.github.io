46static bool handler_called = false;
47
48static void
49handler(int signo __unused)
50{
51 handler_called = true;
52}
53
54static void
55sa_resethand_child(const int flags)
56{
57 struct sigaction sa;
58
59 sa.sa_flags = flags;
60 sa.sa_handler = &handler;
61 sigemptyset(&sa.sa_mask);
62
63 sigaction(SIGUSR1, &sa, NULL);
64 kill(getpid(), SIGUSR1);
65 exit(handler_called ? EXIT_SUCCESS : EXIT_FAILURE);
66}
67
68static void
69wait_and_check_child(const pid_t pid, const char *fail_message)
70{
71 int status;
72
73 (void)waitpid(pid, &status, 0);
74
75 if (WIFEXITED(status))
76 ATF_CHECK_EQ(EXIT_SUCCESS, WEXITSTATUS(status));
77 else
78 atf_tc_fail("%s; raw exit status was %d", fail_message, status);
79}
80
81static void
82catch(int sig __unused)
83{
84 return;
85}
86
87ATF_TC(sigaction_basic);
88ATF_TC_HEAD(sigaction_basic, tc)
89{
90
91 atf_tc_set_md_var(tc, "descr", "Checks for correct I&D cache"
92 "synchronization after copying out the trampoline code.");
93}
94
95ATF_TC_BODY(sigaction_basic, tc)
96{
97 static struct sigaction sa;
98
99 sa.sa_handler = catch;
100
101 sigaction(SIGUSR1, &sa, 0);
102 kill(getpid(), SIGUSR1);
103 atf_tc_pass();
104}
105
106ATF_TC(sigaction_noflags);
107ATF_TC_HEAD(sigaction_noflags, tc)
108{
109 atf_tc_set_md_var(tc, "descr", "Checks programming a signal with "
110 "sigaction(2) but without any flags");
111}
112
113ATF_TC_BODY(sigaction_noflags, tc)
114{
115 const pid_t pid = fork();
116 if (pid == -1)
117 atf_tc_fail_errno("fork(2) failed");
118 else if (pid == 0)
119 sa_resethand_child(0);
120 else
121 wait_and_check_child(pid, "Child process did not exit cleanly;"
122 " it failed to process the signal");
123}
124
125ATF_TC(sigaction_resethand);
126ATF_TC_HEAD(sigaction_resethand, tc)
127{
128 atf_tc_set_md_var(tc, "descr", "Checks that SA_RESETHAND works");
129}
130
131ATF_TC_BODY(sigaction_resethand, tc)
132{
133 const pid_t pid = fork();
134 if (pid == -1)
135 atf_tc_fail_errno("fork(2) failed");
136 else if (pid == 0)
137 sa_resethand_child(SA_RESETHAND);
138 else {
139 wait_and_check_child(pid, "Child process did not exit cleanly;"
140 " it either failed to process the signal or SA_RESETHAND"
141 " is broken");
142 }
143}
144
145ATF_TP_ADD_TCS(tp)
146{
147
148 ATF_TP_ADD_TC(tp, sigaction_basic);
149 ATF_TP_ADD_TC(tp, sigaction_noflags);
150 ATF_TP_ADD_TC(tp, sigaction_resethand);
151
152 return atf_no_error();
153}
