39ATF_TC(ucontext_basic);
40ATF_TC_HEAD(ucontext_basic, tc)
41{
42 atf_tc_set_md_var(tc, "descr", "Checks {get,set}context(2)");
43}
44
45ATF_TC_BODY(ucontext_basic, tc)
46{
47 ucontext_t u, v, w;
48 volatile int x, y;
49
50 x = 0;
51 y = 0;
52
53 printf("Start\n");
54
55 getcontext(&u);
56 y++;
57
58 printf("x == %d\n", x);
59
60 getcontext(&v);
61
62 if ( x < 20 ) {
63 x++;
64 getcontext(&w);
65 setcontext(&u);
66 }
67
68 printf("End, y = %d\n", y);
69 ATF_REQUIRE_EQ(y, 21);
70}
