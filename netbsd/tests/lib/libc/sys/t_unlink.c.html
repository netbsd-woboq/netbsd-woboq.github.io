43static char path[] = "unlink";
44
45ATF_TC_WITH_CLEANUP(unlink_basic);
46ATF_TC_HEAD(unlink_basic, tc)
47{
48 atf_tc_set_md_var(tc, "descr", "A basic test of unlink(2)");
49}
50
51ATF_TC_BODY(unlink_basic, tc)
52{
53 const size_t n = 512;
54 size_t i;
55 int fd;
56
57 for (i = 0; i < n; i++) {
58
59 fd = open(path, O_RDWR | O_CREAT, 0666);
60
61 ATF_REQUIRE(fd != -1);
62 ATF_REQUIRE(close(fd) == 0);
63 ATF_REQUIRE(unlink(path) == 0);
64
65 errno = 0;
66 ATF_REQUIRE_ERRNO(ENOENT, open(path, O_RDONLY) == -1);
67 }
68}
69
70ATF_TC_CLEANUP(unlink_basic, tc)
71{
72 (void)unlink(path);
73}
74
75ATF_TC_WITH_CLEANUP(unlink_err);
76ATF_TC_HEAD(unlink_err, tc)
77{
78 atf_tc_set_md_var(tc, "descr", "Test error conditions of unlink(2)");
79}
80
81ATF_TC_BODY(unlink_err, tc)
82{
83 char buf[PATH_MAX + 1];
84
85 (void)memset(buf, 'x', sizeof(buf));
86
87 errno = 0;
88 ATF_REQUIRE_ERRNO(EBUSY, unlink("/") == -1);
89
90 errno = 0;
91 ATF_REQUIRE_ERRNO(ENAMETOOLONG, unlink(buf) == -1);
92
93 errno = 0;
94 ATF_REQUIRE_ERRNO(ENOENT, unlink("/a/b/c/d/e/f/g/h/i/j/k/l/m") == -1);
95}
96
97ATF_TC_CLEANUP(unlink_err, tc)
98{
99 (void)unlink(path);
100}
101
102ATF_TC_WITH_CLEANUP(unlink_fifo);
103ATF_TC_HEAD(unlink_fifo, tc)
104{
105 atf_tc_set_md_var(tc, "descr", "Test unlink(2) for a FIFO");
106}
107
108ATF_TC_BODY(unlink_fifo, tc)
109{
110
111 ATF_REQUIRE(mkfifo(path, 0666) == 0);
112 ATF_REQUIRE(unlink(path) == 0);
113
114 errno = 0;
115 ATF_REQUIRE_ERRNO(ENOENT, open(path, O_RDONLY) == -1);
116}
117
118ATF_TC_CLEANUP(unlink_fifo, tc)
119{
120 (void)unlink(path);
121}
122
123ATF_TC_WITH_CLEANUP(unlink_perm);
124ATF_TC_HEAD(unlink_perm, tc)
125{
126 atf_tc_set_md_var(tc, "descr", "Test permissions with unlink(2)");
127 atf_tc_set_md_var(tc, "require.user", "unprivileged");
128}
129
130ATF_TC_BODY(unlink_perm, tc)
131{
132 int rv;
133
134 errno = 0;
135 rv = unlink("/etc");
136 ATF_REQUIRE_MSG(rv == -1 && (errno == EACCES || errno == EPERM),
137 "unlinking a directory did not fail with EPERM or EACCESS; "
138 "unlink() returned %d, errno %d", rv, errno);
139
140 errno = 0;
141 ATF_REQUIRE_ERRNO(EACCES, unlink("/root/.profile") == -1);
142}
143
144ATF_TC_CLEANUP(unlink_perm, tc)
145{
146 (void)unlink(path);
147}
148
149ATF_TP_ADD_TCS(tp)
150{
151
152 ATF_TP_ADD_TC(tp, unlink_basic);
153 ATF_TP_ADD_TC(tp, unlink_err);
154 ATF_TP_ADD_TC(tp, unlink_fifo);
155 ATF_TP_ADD_TC(tp, unlink_perm);
156
157 return atf_no_error();
158}
