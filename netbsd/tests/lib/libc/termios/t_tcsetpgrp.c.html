34#include <sys/wait.h>
35
36#include <atf-c.h>
37#include <errno.h>
38#include <stdlib.h>
39#include <unistd.h>

41ATF_TC(tcsetpgrp_err);
42ATF_TC_HEAD(tcsetpgrp_err, tc)
43{
44 atf_tc_set_md_var(tc, "descr", "Test errors from tcsetpgrp(3)"
45 " (PR lib/41673)");
46}
47
48ATF_TC_BODY(tcsetpgrp_err, tc)
49{
50 int rv, sta;
51 pid_t pid;
52
53 if (isatty(STDIN_FILENO) == 0)
54 return;
55
56 pid = fork();
57 ATF_REQUIRE(pid >= 0);
58
59 if (pid == 0) {
60
61 /*
62 * The child process ID doesn't match any active
63 * process group ID, so the following call should
64 * fail with EPERM (and not EINVAL).
65 */
66 errno = 0;
67 rv = tcsetpgrp(STDIN_FILENO, getpid());
68
69 if (rv == 0 || errno != EPERM)
70 _exit(EXIT_FAILURE);
71
72 _exit(EXIT_SUCCESS);
73 }
74
75 (void)wait(&sta);
76
77 if (WIFEXITED(sta) == 0 || WEXITSTATUS(sta) != EXIT_SUCCESS)
78 atf_tc_fail("wrong errno");
79}
80
81ATF_TP_ADD_TCS(tp)
82{
83
84 ATF_TP_ADD_TC(tp, tcsetpgrp_err);
85
86 return atf_no_error();
87}
