34#include <sys/cdefs.h>
35__RCSID("$NetBSD: t_tls_dynamic.c,v 1.3 2012/01/17 20:34:57 joerg Exp $");
36
37#include <atf-c.h>
38#include <pthread.h>
39#include <unistd.h>
40
41#include <sys/tls.h>
42
43#ifdef __HAVE_NO___THREAD
44#define __thread
45#endif
46
47ATF_TC(t_tls_dynamic);
48
49ATF_TC_HEAD(t_tls_dynamic, tc)
50{
51 atf_tc_set_md_var(tc, "descr",
52 "Test (un)initialized TLS variables in dynamic binaries");
53}
54
55void testf_dso_helper(int, int);
56
57extern __thread int var1;
58extern __thread int var2;
59extern __thread pid_t (*dso_var1)(void);
60
61__thread int *var3 = &optind;
62int var4_helper;
63__thread int *var4 = &var4_helper;
64
65static void *
66testf(void *dummy)
67{
68 ATF_CHECK_EQ(var1, 1);
69 ATF_CHECK_EQ(var2, 0);
70 testf_dso_helper(2, 2);
71 ATF_CHECK_EQ(var1, 2);
72 ATF_CHECK_EQ(var2, 2);
73 testf_dso_helper(3, 3);
74 ATF_CHECK_EQ(var1, 3);
75 ATF_CHECK_EQ(var2, 3);
76 ATF_CHECK_EQ(var3, &optind);
77 ATF_CHECK_EQ(var4, &var4_helper);
78 ATF_CHECK_EQ(dso_var1, getpid);
79
80 return NULL;
81}
82
83ATF_TC_BODY(t_tls_dynamic, tc)
84{
85 pthread_t t;
86
87#ifdef __HAVE_NO___THREAD
88 atf_tc_skip("no TLS support on this platform");
89#endif
90
91 testf(NULL);
92
93 pthread_create(&t, 0, testf, 0);
94 pthread_join(t, NULL);
95
96 pthread_create(&t, 0, testf, 0);
97 pthread_join(t, NULL);
98}
99
100ATF_TP_ADD_TCS(tp)
101{
102 ATF_TP_ADD_TC(tp, t_tls_dynamic);
103
104 return atf_no_error();
105}
