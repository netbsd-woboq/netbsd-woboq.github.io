34#include <sys/cdefs.h>
35__RCSID("$NetBSD: t_tls_static.c,v 1.4 2017/10/28 19:25:31 christos Exp $");
36
37#include <atf-c.h>
38#include <pthread.h>
39
40#include <sys/tls.h>
41
42#ifdef __HAVE_NO___THREAD
43#define __thread
44#endif
45
46ATF_TC(t_tls_static);
47
48ATF_TC_HEAD(t_tls_static, tc)
49{
50 atf_tc_set_md_var(tc, "descr",
51 "Test (un)initialized TLS variables in static binaries");
52}
53
54void testf_helper(void);
55
56__thread int var1 = 1;
57__thread int var2;
58
59static void *
60testf(void *dummy)
61{
62#define CHECK(a, b) \
63 ATF_CHECK_EQ_MSG(var ## a, b, "var%d[%d] != %d", a, var ## a, b)
64 CHECK(1, 1);
65 CHECK(2, 0);
66 testf_helper();
67 CHECK(1, -1);
68 CHECK(2, -1);
69
70 return NULL;
71}
72
73ATF_TC_BODY(t_tls_static, tc)
74{
75 pthread_t t;
76
77#ifdef __HAVE_NO___THREAD
78 atf_tc_skip("no TLS support on this platform");
79#endif
80
81 testf(NULL);
82
83 pthread_create(&t, 0, testf, 0);
84 pthread_join(t, NULL);
85
86 pthread_create(&t, 0, testf, 0);
87 pthread_join(t, NULL);
88}
89
90ATF_TP_ADD_TCS(tp)
91{
92 ATF_TP_ADD_TC(tp, t_tls_static);
93
94 return atf_no_error();
95}
