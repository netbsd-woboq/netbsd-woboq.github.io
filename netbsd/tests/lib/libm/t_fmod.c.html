32#include <atf-c.h>
33#include <float.h>
34#include <math.h>
35
36#include "isqemu.h"
37
38ATF_TC(fmod);
39ATF_TC_HEAD(fmod, tc)
40{
41 atf_tc_set_md_var(tc, "descr","Check fmod family");
42}
43
44ATF_TC_BODY(fmod, tc)
45{
46 if (isQEMU())
47 atf_tc_expect_fail("PR misc/44767");
48
49 ATF_CHECK(fmodf(2.0, 1.0) == 0);
50 ATF_CHECK(fmod(2.0, 1.0) == 0);
51 ATF_CHECK(fmodl(2.0, 1.0) == 0);
52
53 ATF_CHECK(fmodf(2.0, 0.5) == 0);
54 ATF_CHECK(fmod(2.0, 0.5) == 0);
55 ATF_CHECK(fmodl(2.0, 0.5) == 0);
56
57 ATF_CHECK(fabsf(fmodf(1.0, 0.1) - 0.1f) <= 55 * FLT_EPSILON);
58 ATF_CHECK(fabs(fmod(1.0, 0.1) - 0.1) <= 55 * DBL_EPSILON);
59 ATF_CHECK(fabsl(fmodl(1.0, 0.1L) - 0.1L) <= 55 * LDBL_EPSILON);
60}
61
62ATF_TP_ADD_TCS(tp)
63{
64
65 ATF_TP_ADD_TC(tp, fmod);
66
67 return atf_no_error();
68}
