29#include <atf-c.h>
30#include <math.h>
31
32ATF_TC(hypot_integer);
33ATF_TC_HEAD(hypot_integer, tc)
34{
35 atf_tc_set_md_var(tc, "descr", "Test hypot with integer args");
36}
37
38ATF_TC_BODY(hypot_integer, tc)
39{
40 /* volatile so hypotf() won't be evaluated at compile time */
41 volatile double a = 5;
42 volatile double b = 12;
43 ATF_CHECK(hypot(a, b) == 13.0);
44}
45
46ATF_TC(hypotf_integer);
47ATF_TC_HEAD(hypotf_integer, tc)
48{
49 atf_tc_set_md_var(tc, "descr", "Test hypotf with integer args");
50}
51
52ATF_TC_BODY(hypotf_integer, tc)
53{
54 volatile float a = 5;
55 volatile float b = 12;
56 ATF_CHECK(hypotf(a, b) == 13.0f);
57}
58
59ATF_TC(pr50698);
60ATF_TC_HEAD(pr50698, tc)
61{
62 atf_tc_set_md_var(tc, "descr", "Check for the bug of PR 50698");
63}
64
65ATF_TC_BODY(pr50698, tc)
66{
67 volatile float a = 1e-18f;
68 float val = hypotf(a, a);
69 ATF_CHECK(!isinf(val));
70 ATF_CHECK(!isnan(val));
71}
72
73ATF_TP_ADD_TCS(tp)
74{
75
76 ATF_TP_ADD_TC(tp, hypot_integer);
77 ATF_TP_ADD_TC(tp, hypotf_integer);
78 ATF_TP_ADD_TC(tp, pr50698);
79
80 return atf_no_error();
81}
