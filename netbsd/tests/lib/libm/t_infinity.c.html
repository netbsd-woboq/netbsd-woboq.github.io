31#include <sys/cdefs.h>
32__RCSID("$NetBSD: t_infinity.c,v 1.6 2012/09/26 07:24:38 jruoho Exp $");
33
34#include <atf-c.h>
35#include <math.h>
36#include <float.h>
37#include <stdlib.h>
38
39ATF_TC(infinity_float);
40ATF_TC_HEAD(infinity_float, tc)
41{
42 atf_tc_set_md_var(tc, "descr",
43 "check FPU generated infinite float values");
44}
45
46ATF_TC_BODY(infinity_float, tc)
47{
48 float v;
49
50 v = FLT_MAX;
51 v *= v;
52 ATF_REQUIRE(isinf(v));
53 ATF_REQUIRE(fpclassify(v) == FP_INFINITE);
54
55 v = -FLT_MAX;
56 v *= v;
57 ATF_REQUIRE(isinf(v));
58 ATF_REQUIRE(fpclassify(v) == FP_INFINITE);
59}

83ATF_TC(infinity_long_double);
84ATF_TC_HEAD(infinity_long_double, tc)
85{
86 atf_tc_set_md_var(tc, "descr",
87 "check FPU generated infinite long double values");
88}
89
90ATF_TC_BODY(infinity_long_double, tc)
91{
92
93#ifndef LDBL_MAX
94 atf_tc_skip("no long double support on this architecture");
95 return;
96#else
97 long double v;
98
99 v = LDBL_MAX;
100 v *= v;
101 ATF_REQUIRE(isinf(v));
102 ATF_REQUIRE(fpclassify(v) == FP_INFINITE);
103
104 v = -LDBL_MAX;
105 v *= v;
106 ATF_REQUIRE(isinf(v));
107 ATF_REQUIRE(fpclassify(v) == FP_INFINITE);
108#endif
109}
110
111ATF_TP_ADD_TCS(tp)
112{
113
114 ATF_TP_ADD_TC(tp, infinity_float);
115 ATF_TP_ADD_TC(tp, infinity_double);
116 ATF_TP_ADD_TC(tp, infinity_long_double);
117
118 return atf_no_error();
119}
