32#include <atf-c.h>
33#include <float.h>
34#include <math.h>
35
36ATF_TC(modf);
37ATF_TC_HEAD(modf, tc)
38{
39 atf_tc_set_md_var(tc, "descr","Check modf family");
40}
41
42ATF_TC_BODY(modf, tc)
43{
44 float basef;
45 double base;
46 long double basel;
47 ATF_CHECK(modff(1.0, &basef) == 0.0);
48 ATF_CHECK(basef == 1.0);
49 ATF_CHECK(modf(1.0, &base) == 0.0);
50 ATF_CHECK(base == 1.0);
51 ATF_CHECK(modfl(1.0, &basel) == 0.0);
52 ATF_CHECK(basel == 1.0);
53
54 ATF_CHECK(modff(-1 - FLT_EPSILON, &basef) == -FLT_EPSILON);
55 ATF_CHECK(basef == -1.0);
56 ATF_CHECK(modf(-1 - DBL_EPSILON, &base) == -DBL_EPSILON);
57 ATF_CHECK(base == -1.0);
58 ATF_CHECK(modfl(-1 - LDBL_EPSILON, &basel) == -LDBL_EPSILON);
59 ATF_CHECK(basel == -1.0);
60}
61
62ATF_TP_ADD_TCS(tp)
63{
64
65 ATF_TP_ADD_TC(tp, modf);
66
67 return atf_no_error();
68}
