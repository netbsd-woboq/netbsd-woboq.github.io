37#if 0
38#include <atf-c.h>

40#include <assert.h>
41#define ATF_REQUIRE(a) assert(a)
42#endif
43#include <unistd.h>
44#include <pthread.h>
45
46int testf_dso_null(void);
47int testf_dso_mutex_lock(pthread_mutex_t *);
48int testf_dso_mutex_unlock(pthread_mutex_t *);
49int testf_dso_pthread_create(pthread_t *, const pthread_attr_t *,
50 void *(*)(void *), void *);
51
52int
53testf_dso_null(void)
54{
55 return 0xcafe;
56}
57
58int
59testf_dso_mutex_lock(pthread_mutex_t *mtx)
60{
61 ATF_REQUIRE(mtx != NULL);
62 ATF_REQUIRE(pthread_mutex_lock(mtx) == 0);
63
64 return 0xcafe;
65}
66
67int
68testf_dso_mutex_unlock(pthread_mutex_t *mtx)
69{
70 ATF_REQUIRE(mtx != NULL);
71 ATF_REQUIRE(pthread_mutex_unlock(mtx) == 0);
72
73 return 0xcafe;
74}
75
76int
77testf_dso_pthread_create(pthread_t *thread, const pthread_attr_t *attr,
78 void *(*routine)(void *), void *arg)
79{
80 int ret;
81
82 ret = pthread_create(thread, attr, routine, arg);
83 ATF_REQUIRE(ret == 0);
84
85 return 0;
86}
