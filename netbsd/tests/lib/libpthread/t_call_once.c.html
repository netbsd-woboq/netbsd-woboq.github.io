37#include <threads.h>
38
39#include <atf-c.h>
40
41ATF_TC(call_once);
42ATF_TC_HEAD(call_once, tc)
43{
44 atf_tc_set_md_var(tc, "descr", "Test C11 call_once(3)");
45}
46
47#define CO_THREADS 8
48
49static int co_val;
50static once_flag oflag = ONCE_FLAG_INIT;
51
52static void
53called_once(void)
54{
55
56 ++co_val;
57}
58
59static int
60co_func(void *arg __unused)
61{
62
63 call_once(&oflag, called_once);
64
65 return 0;
66}
67
68ATF_TC_BODY(call_once, tc)
69{
70 thrd_t t[CO_THREADS];
71 size_t i;
72
73 for (i = 0; i < CO_THREADS; i++) {
74 ATF_REQUIRE_EQ(thrd_create(&t[i], co_func, NULL), thrd_success);
75 }
76
77 for (i = 0; i < CO_THREADS; i++) {
78 ATF_REQUIRE_EQ(thrd_join(t[i], NULL), thrd_success);
79 }
80
81 ATF_REQUIRE_EQ(co_val, 1);
82}
83
84ATF_TP_ADD_TCS(tp)
85{
86 ATF_TP_ADD_TC(tp, call_once);
87
88 return atf_no_error();
89}
