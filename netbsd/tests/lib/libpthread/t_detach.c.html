32__RCSID("$NetBSD: t_detach.c,v 1.2 2017/01/16 16:29:54 christos Exp $");
33
34#include <errno.h>
35#include <pthread.h>
36#include <time.h>
37
38#include <atf-c.h>
39
40#include "h_common.h"
41
42static void *func(void *);
43
44static void *
45func(void *arg)
46{
47 sleep(2);
48 return NULL;
49}
50
51ATF_TC(pthread_detach);
52ATF_TC_HEAD(pthread_detach, tc)
53{
54 atf_tc_set_md_var(tc, "descr", "A test of pthread_detach(3)");
55}
56
57ATF_TC_BODY(pthread_detach, tc)
58{
59 const int state = PTHREAD_CREATE_JOINABLE;
60 pthread_attr_t attr;
61 pthread_t t;
62 int rv;
63
64 /*
65 * Create a joinable thread.
66 */
67 PTHREAD_REQUIRE(pthread_attr_init(&attr));
68 PTHREAD_REQUIRE(pthread_attr_setdetachstate(&attr, state));
69 PTHREAD_REQUIRE(pthread_create(&t, &attr, func, NULL));
70
71 /*
72 * Detach the thread and try to
73 * join it; EINVAL should follow.
74 */
75 PTHREAD_REQUIRE(pthread_detach(t));
76
77 sleep(1);
78 rv = pthread_join(t, NULL);
79 ATF_REQUIRE(rv == EINVAL);
80
81 sleep(3);
82
83 /*
84 * As usual, ESRCH should follow if
85 * we try to detach an invalid thread.
86 */
87 rv = pthread_cancel(t);
88 ATF_REQUIRE(rv == ESRCH);
89}
90
91ATF_TP_ADD_TCS(tp)
92{
93 ATF_TP_ADD_TC(tp, pthread_detach);
94
95 return atf_no_error();
96}
