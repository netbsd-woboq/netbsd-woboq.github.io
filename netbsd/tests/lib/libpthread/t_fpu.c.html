65#define N_RECURSE 10
66
67static void recurse(void);
68
69int recursion_depth = 0;
70pthread_mutex_t recursion_depth_lock;
71
72static void *
73stir(void *p)
74{
75 double *q = (double *)p;
76 double x = *q++;
77 double y = *q++;
78 double z = *q++;
79
80 for (;;) {
81 x = sin ((y = cos (x + y + .4)) - (z = cos (x + z + .6)));
82 ATF_REQUIRE_MSG(sched_yield() == 0,
83 "sched_yield failed: %s", strerror(errno));
84 }
85}
86
87static double
88mul3(double x, double y, double z)
89{
90 ATF_REQUIRE_MSG(sched_yield() == 0,
91 "sched_yield failed: %s", strerror(errno));
92
93 return x * y * z;
94}
95
96static void *
97bar(void *p)
98{
99 double d;
100 int rc;
101
102 d = mul3(mul3(2., 3., 5.), mul3(7., 11., 13.), mul3(17., 19., 23.));
103 ATF_REQUIRE_EQ(d, 223092870.);
104
105 PTHREAD_REQUIRE(pthread_mutex_lock(&recursion_depth_lock));
106 rc = recursion_depth++;
107 PTHREAD_REQUIRE(pthread_mutex_unlock(&recursion_depth_lock));
108
109 if (rc < N_RECURSE)
110 recurse();
111 else
112 atf_tc_pass();
113
114 /* NOTREACHED */
115 return NULL;
116}
117
118static void
119recurse(void) {
120 pthread_t s2;
121 PTHREAD_REQUIRE(pthread_create(&s2, 0, bar, 0));
122 sleep(20); /* XXX must be long enough for our slowest machine */
123}

125ATF_TC(fpu);
126ATF_TC_HEAD(fpu, tc)
127{
128 atf_tc_set_md_var(tc, "descr",
129 "Checks that thread context switches will leave the "
130 "floating point computations unharmed");
131}
132ATF_TC_BODY(fpu, tc)
133{
134 double stirseed[] = { 1.7, 3.2, 2.4 };
135 pthread_t s5;
136
137 printf("Testing threaded floating point computations...\n");
138
139 PTHREAD_REQUIRE(pthread_mutex_init(&recursion_depth_lock, 0));
140
141 PTHREAD_REQUIRE(pthread_create(&s5, 0, stir, stirseed));
142 recurse();
143
144 atf_tc_fail("exiting from main");
145}

147ATF_TP_ADD_TCS(tp)
148{
149 ATF_TP_ADD_TC(tp, fpu);
150
151 return atf_no_error();
152}
