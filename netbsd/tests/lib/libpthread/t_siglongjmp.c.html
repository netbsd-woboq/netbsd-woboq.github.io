34/*
35 * Regression test for siglongjmp out of a signal handler back into
36 * its thread.
37 *
38 * Written by Christian Limpach <cl@NetBSD.org>, December 2003.
39 * Public domain.
40 */
41
42#include <sys/resource.h>
43
44#include <pthread.h>
45#include <setjmp.h>
46#include <signal.h>
47#include <unistd.h>
48
49#include <atf-c.h>
50
51#include "h_common.h"
52
53static sigjmp_buf env;
54
55static void *
56thread(void *arg)
57{
58 return NULL;
59}
60
61static void
62handler(int sig, siginfo_t *info, void *ctx)
63{
64 siglongjmp(env, 1);
65}
66
67ATF_TC(siglongjmp1);
68ATF_TC_HEAD(siglongjmp1, tc)
69{
70 atf_tc_set_md_var(tc, "descr",
71 "Checks siglongjmp() out of a signal handler back into its thread");
72}
73ATF_TC_BODY(siglongjmp1, tc)
74{
75 pthread_t t;
76 sigset_t nset;
77 struct rlimit rlim;
78 struct sigaction act;
79
80 rlim.rlim_cur = rlim.rlim_max = 0;
81 (void)setrlimit(RLIMIT_CORE, &rlim);
82
83 PTHREAD_REQUIRE(pthread_create(&t, NULL, thread, NULL));
84
85 sigemptyset(&nset);
86 sigaddset(&nset, SIGUSR1);
87 PTHREAD_REQUIRE(pthread_sigmask(SIG_SETMASK, &nset, NULL));
88
89 act.sa_sigaction = handler;
90 sigemptyset(&act.sa_mask);
91 sigaddset(&act.sa_mask, SIGUSR2);
92 act.sa_flags = 0;
93 sigaction(SIGSEGV, &act, NULL);
94
95 ATF_REQUIRE_EQ(sigsetjmp(env, 1), 0);
96
97 PTHREAD_REQUIRE(pthread_sigmask(0, NULL, &nset));
98
99 ATF_REQUIRE_EQ_MSG(sigismember(&nset, SIGSEGV), 0, "SIGSEGV set");
100 ATF_REQUIRE_EQ_MSG(sigismember(&nset, SIGUSR2), 0, "SIGUSR2 set");
101 ATF_REQUIRE_MSG(sigismember(&nset, SIGUSR1), "SIGUSR1 not set");
102}
103
104ATF_TP_ADD_TCS(tp)
105{
106 ATF_TP_ADD_TC(tp, siglongjmp1);
107
108 return atf_no_error();
109}
