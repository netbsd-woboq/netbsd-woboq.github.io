41#include <atf-c.h>

43#include "h_common.h"
44
45#define LONGTIME 2000000000
46
47static void *
48threadfunc(void *arg)
49{
50 sleep(LONGTIME);
51
52 return NULL;
53}
54
55static void
56handler(int sig)
57{
58 /*
59 * Nothing to do; invoking the handler is enough to interrupt
60 * the sleep.
61 */
62}
63
64ATF_TC(sleep1);
65ATF_TC_HEAD(sleep1, tc)
66{
67 atf_tc_set_md_var(tc, "descr", "Checks sleeping past the time when "
68 "time_t wraps");
69}
70ATF_TC_BODY(sleep1, tc)
71{
72 pthread_t thread;
73 struct itimerval it;
74 struct sigaction act;
75 sigset_t mtsm;
76
77 printf("Testing sleeps unreasonably far into the future.\n");
78
79 act.sa_handler = handler;
80 sigemptyset(&act.sa_mask);
81 act.sa_flags = 0;
82 sigaction(SIGALRM, &act, NULL);
83
84 PTHREAD_REQUIRE(pthread_create(&thread, NULL, threadfunc, NULL));
85
86 /* make sure the signal is delivered to the child thread */
87 sigemptyset(&mtsm);
88 sigaddset(&mtsm, SIGALRM);
89 PTHREAD_REQUIRE(pthread_sigmask(SIG_BLOCK, &mtsm, 0));
90
91 timerclear(&it.it_interval);
92 timerclear(&it.it_value);
93 it.it_value.tv_sec = 1;
94 setitimer(ITIMER_REAL, &it, NULL);
95
96 PTHREAD_REQUIRE(pthread_join(thread, NULL));
97}
98
99ATF_TP_ADD_TCS(tp)
100{
101 ATF_TP_ADD_TC(tp, sleep1);
102
103 return atf_no_error();
104}
