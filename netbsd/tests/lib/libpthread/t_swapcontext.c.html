41#include "h_common.h"

43#define STACKSIZE 65536
44
45char stack[STACKSIZE];
46ucontext_t nctx;
47ucontext_t octx;
48void *oself;
49void *nself;
50int val1, val2;
51
52static void
53swapfunc(void)
54{
55 /*
56 * If the test fails, we are very likely to crash
57 * without the opportunity to report
58 */
59 nself = (void *)pthread_self();
60 printf("after swapcontext self = %p\n", nself);
61
62 ATF_REQUIRE_EQ(oself, nself);
63 printf("Test succeeded\n");
64}
65
66/* ARGSUSED0 */
67static void *
68threadfunc(void *arg)
69{
70 nctx.uc_stack.ss_sp = stack;
71 nctx.uc_stack.ss_size = sizeof(stack);
72 nctx.uc_link = &octx;
73
74 makecontext(&nctx, swapfunc, 0);
75
76 oself = (void *)pthread_self();
77 printf("before swapcontext self = %p\n", oself);
78 ATF_REQUIRE_MSG(swapcontext(&octx, &nctx) != -1, "swapcontext failed: %s",
79 strerror(errno));
80
81 return NULL;
82}
83
84
85ATF_TC(swapcontext1);
86ATF_TC_HEAD(swapcontext1, tc)
87{
88 atf_tc_set_md_var(tc, "descr", "Testing if swapcontext() "
89 "alters pthread_self()");
90}
91ATF_TC_BODY(swapcontext1, tc)
92{
93 pthread_t thread;
94
95 oself = (void *)&val1;
96 nself = (void *)&val2;
97
98 printf("Testing if swapcontext() alters pthread_self()\n");
99
100 ATF_REQUIRE_MSG(getcontext(&nctx) != -1, "getcontext failed: %s",
101 strerror(errno));
102 PTHREAD_REQUIRE(pthread_create(&thread, NULL, threadfunc, NULL));
103 PTHREAD_REQUIRE(pthread_join(thread, NULL));
104}

106ATF_TP_ADD_TCS(tp)
107{
108 ATF_TP_ADD_TC(tp, swapcontext1);
109
110 return atf_no_error();
111}
