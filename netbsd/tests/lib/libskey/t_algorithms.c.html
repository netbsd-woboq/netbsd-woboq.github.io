40#define H_REQUIRE(x, y) \
41 ATF_REQUIRE_MSG(strcasecmp((x), (y)) == 0, "\"%s\" != \"%s\"", (x), (y))
42
43static void
44h_check(const char *pass, const char *seed,
45 const char *algo, const char *zero,
46 const char *one, const char *nine)
47{
48 char prn[64];
49 char data[16];
50 int i;
51
52 skey_set_algorithm(algo);
53
54 keycrunch(data, seed, pass);
55 btoa8(prn, data);
56 H_REQUIRE(prn, zero);
57
58 f(data);
59 btoa8(prn, data);
60 H_REQUIRE(prn, one);
61
62 for(i = 1; i < 99; ++i)
63 f(data);
64 btoa8(prn, data);
65 H_REQUIRE(prn, nine);
66}

68ATF_TC(md4);
69ATF_TC_HEAD(md4, tc)
70{
71 atf_tc_set_md_var(tc, "descr", "Checks MD4 algorithm");
72}
73ATF_TC_BODY(md4, tc)
74{
75 h_check("This is a test.", "TeSt", "md4", "D1854218EBBB0B51",
76 "63473EF01CD0B444", "C5E612776E6C237A");
77 h_check("AbCdEfGhIjK", "alpha1", "md4", "50076F47EB1ADE4E",
78 "65D20D1949B5F7AB", "D150C82CCE6F62D1");
79 h_check("OTP's are good", "correct", "md4", "849C79D4F6F55388",
80 "8C0992FB250847B1", "3F3BF4B4145FD74B");
81}
