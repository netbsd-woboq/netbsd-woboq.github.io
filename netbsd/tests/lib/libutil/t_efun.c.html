41static bool fail;
42static void handler(int, const char *, ...);
43
44static void
45handler(int ef, const char *fmt, ...)
46{
47 fail = false;
48}
49
50ATF_TC(ecalloc);
51ATF_TC_HEAD(ecalloc, tc)
52{
53 atf_tc_set_md_var(tc, "descr", "A basic test of ecalloc(3)");
54}
55
56ATF_TC_BODY(ecalloc, tc)
57{
58 char *x;
59
60 fail = true;
61 x = ecalloc(-1, 1);
62
63 ATF_REQUIRE(x == NULL);
64 ATF_REQUIRE(fail != true);
65
66 fail = true;
67 x = ecalloc(SIZE_MAX, 2);
68
69 ATF_REQUIRE(x == NULL);
70 ATF_REQUIRE(fail != true);
71}

73ATF_TC(efopen);
74ATF_TC_HEAD(efopen, tc)
75{
76 atf_tc_set_md_var(tc, "descr", "A basic test of efopen(3)");
77}
78
79ATF_TC_BODY(efopen, tc)
80{
81 FILE *f;
82
83 fail = true;
84 f = efopen("XXX", "XXX");
85
86 ATF_REQUIRE(f == NULL);

107ATF_TC(erealloc);
108ATF_TC_HEAD(erealloc, tc)
109{
110 atf_tc_set_md_var(tc, "descr", "A basic test of erealloc(3)");
111}
112
113ATF_TC_BODY(erealloc, tc)
114{
115 char *x;
116
117 fail = true;
118 x = erealloc(NULL, -1);
119
120 ATF_REQUIRE(x == NULL);
121 ATF_REQUIRE(fail != true);
122}
123
124ATF_TP_ADD_TCS(tp)
125{
126
127 ATF_REQUIRE(esetfunc(handler) != NULL);
128
129 ATF_TP_ADD_TC(tp, ecalloc);
130 ATF_TP_ADD_TC(tp, efopen);
131 ATF_TP_ADD_TC(tp, emalloc);
132 ATF_TP_ADD_TC(tp, erealloc);
133
134 return atf_no_error();
135}
