41static void
42h_snprintb(const char *fmt, uint64_t val, const char *res)
43{
44 char buf[1024];
45 int len, slen;
46
47 len = snprintb(buf, sizeof(buf), fmt, val);
48 slen = (int) strlen(res);
49
50 ATF_REQUIRE_STREQ(res, buf);
51 ATF_REQUIRE_EQ(len, slen);
52}

102static void
103h_snprintb_m(const char *fmt, uint64_t val, int line_max, const char *res,
104 int res_len)
105{
106 char buf[1024];
107 int len;
108
109 len = snprintb_m(buf, sizeof(buf), fmt, val, line_max);
110
111 ATF_REQUIRE_EQ(len, res_len);
112 ATF_REQUIRE_EQ(0, memcmp(res, buf, res_len + 1));
113}
