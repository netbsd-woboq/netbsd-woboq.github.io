30#include <poll.h>
31#include <stddef.h>
32
33extern int sleep_init;
34extern int sleep_fini;
35extern int dlopen_cookie;
36
37static void __attribute__((__constructor__))
38init2(void)
39{
40 dlopen_cookie = 2;
41 if (!sleep_init)
42 return;
43 for (;;)
44 poll(NULL, 0, -1);
45}
46
47extern int dlclose_cookie;
48
49static void __attribute__((__destructor__))
50fini2(void)
51{
52 dlclose_cookie = 2;
53 if (!sleep_fini)
54 return;
55 for (;;)
56 poll(NULL, 0, -1);
57}
