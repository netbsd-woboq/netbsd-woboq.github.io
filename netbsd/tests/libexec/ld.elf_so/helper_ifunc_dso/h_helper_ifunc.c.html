30#include <stdlib.h>
31#include <string.h>
32
33static long long
34ifunc1(void)
35{
36 return 0xdeadbeefll;
37}
38
39static long long
40ifunc2(void)
41{
42 return 0xbeefdeadll;
43}
44
45static __attribute__((used))
46long long (*resolve_ifunc(void))(void)
47{
48 const char *e = getenv("USE_IFUNC2");
49 return e && strcmp(e, "1") == 0 ? ifunc2 : ifunc1;
50}
51
52static __attribute__((used))
53long long (*resolve_ifunc2(void))(void)
54{
55 const char *e = getenv("USE_IFUNC2");
56 return e && strcmp(e, "1") == 0 ? ifunc1 : ifunc2;
57}
58
59__ifunc(ifunc, resolve_ifunc);
60__hidden_ifunc(ifunc_hidden, resolve_ifunc2);
61
62long long ifunc_hidden(void);
63long long ifunc_plt(void);
64
65long long ifunc_plt(void)
66{
67 return ifunc_hidden();
68}
69
70long long (*ifunc_indirect(void))(void);
71
72long long (*ifunc_indirect(void))(void)
73{
74 return ifunc_hidden;
75}
