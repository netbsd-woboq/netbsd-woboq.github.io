30#include <sys/types.h>
31
32#include <atf-c.h>
33#include <dlfcn.h>
34#include <link_elf.h>
35
36#include "h_macros.h"
37
38ATF_TC(rtld_dlerror_false);
39ATF_TC_HEAD(rtld_dlerror_false, tc)
40{
41 atf_tc_set_md_var(tc, "descr",
42 "error set by dlopen persists past a successful dlopen call");
43}
44
45ATF_TC_BODY(rtld_dlerror_false, tc)
46{
47 void *handle, *sym;
48 char *error;
49
50 /*
51 *
52 * Test for dlerror() being set by a successful library open.
53 * Requires that the rpath be set to something that does not
54 * include libm.so.
55 */
56
57 handle = dlopen("libm.so", RTLD_LAZY);
58 error = dlerror();
59 ATF_CHECK(error == NULL);
60 ATF_CHECK(handle != NULL);
61
62 sym = dlsym(handle, "sin");
63 error = dlerror();
64 ATF_CHECK(sym != NULL);
65 ATF_CHECK(error == NULL);
66
67 dlclose(handle);
68 error = dlerror();
69
70 ATF_CHECK(error == NULL);
71
72}
73
74ATF_TP_ADD_TCS(tp)
75{
76 ATF_TP_ADD_TC(tp, rtld_dlerror_false);
77
78 return atf_no_error();
79}
