45static int load(const char *, ...);
46
47/*
48 * A program that loads a module and returns the errno(2) from modctl(8).
49 */
50int
51main(int argc, char *argv[])
52{
53 assert(argc == 3);
54
55 return load(argv[1], argv[2]);
56}
57
58static __printflike(1, 2) int
59load(const char *fmt, ...)
60{
61 char filename[MAXPATHLEN], *propsstr, *shortname, *dot;
62 prop_dictionary_t props;
63 modctl_load_t ml;
64 int serrno, rv;
65 va_list ap;
66
67 props = prop_dictionary_create();
68 propsstr = prop_dictionary_externalize(props);
69 assert(propsstr != NULL);
70 prop_object_release(props);
71
72 va_start(ap, fmt);
73 (void)vsnprintf(filename, sizeof(filename), fmt, ap);
74 va_end(ap);
75
76 ml.ml_filename = filename;
77 ml.ml_flags = 0;
78 ml.ml_props = propsstr;
79 ml.ml_propslen = strlen(propsstr);
80
81 printf("Loading module %s\n", filename);
82 errno = serrno = 0;
83
84 rv = modctl(MODCTL_LOAD, &ml);
85
86 if (rv != 0)
87 serrno = errno;
88
89 shortname = strrchr(filename, '/');
90 if (shortname != NULL)
91 shortname++;
92 else
93 shortname = filename;
94 dot = strrchr(shortname, '.');
95 if (dot)
96 *dot = 0;
97 (void)modctl(MODCTL_UNLOAD, shortname);
98
99 free(propsstr);
100
101 return serrno;
102}
