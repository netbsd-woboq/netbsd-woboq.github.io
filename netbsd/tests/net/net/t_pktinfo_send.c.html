67static void
68sock_in_init(struct sockaddr_in *sin, const char *addr, in_port_t port)
69{
70 memset(sin, 0, sizeof(struct sockaddr_in));
71
72 sin->sin_family = AF_INET;
73 sin->sin_port = htons(port);
74 inet_pton(AF_INET, addr, &sin->sin_addr);
75}

77static int
78sock_bind(int sock, const char *addr, in_port_t port)
79{
80 struct sockaddr_in bindaddr;
81
82 sock_in_init(&bindaddr, addr, port);
83 return rump_sys_bind(sock,
84 (struct sockaddr *)&bindaddr, sizeof(bindaddr));
85}

87static int
88udp_server(const char *addr, in_port_t port)
89{
90 int s, rv;
91
92 RL(s = rump_sys_socket(AF_INET, SOCK_DGRAM, 0));

98static int
99addrcmp(struct sockaddr_in *sin, const char *addr)
100{
101 struct in_addr inaddr;
102 inet_pton(AF_INET, addr, &inaddr);
103 return memcmp(&inaddr, &sin->sin_addr, sizeof(inaddr));
104}

107static ssize_t
108sendto_pktinfo(int s, const void *buf, size_t len, int flags,
109 const char *src, const char *dst, in_port_t dstport)
110{
111 /* for sendmsg */
112 struct sockaddr_in to;
113 struct msghdr msg;
114 char cmsgbuf[CMSG_SPACE(sizeof(struct in_pktinfo)) * 2];
115 struct iovec vec;
116
117 /* for store to cmsghdr */
118 struct cmsghdr *cmsg;
119
120 /* for pktinfo */
121 struct in_pktinfo *pi;
122 struct in_addr addr;
123
124 /* setup msghdr */
125 sock_in_init(&to, dst, dstport);
126
127 vec.iov_base = __UNCONST(buf);
128 vec.iov_len = len;
129 memset(&msg, 0, sizeof(msg));
130 msg.msg_iov = &vec;
131 msg.msg_iovlen = 1;
132 msg.msg_name = (caddr_t)&to;
133 msg.msg_namelen = sizeof(to);
134 msg.msg_control = cmsgbuf;
135 msg.msg_controllen = 0;
136 cmsg = (struct cmsghdr *)cmsgbuf;
137
138 /* setup ip_pktinfo */
139 msg.msg_controllen += CMSG_SPACE(sizeof(struct in_pktinfo));
140 cmsg->cmsg_len = CMSG_LEN(sizeof(struct in_pktinfo));
141 cmsg->cmsg_level = IPPROTO_IP;
142 cmsg->cmsg_type = IP_PKTINFO;
143
144 pi = (struct in_pktinfo *)CMSG_DATA(cmsg);
145 memset(pi, 0, sizeof(*pi));
146
147 /* treat 0.x.x.x/8 as interface index (like RFC1724 ss.3.3) */
148 inet_pton(AF_INET, src, &addr);
149 if (ntohl(addr.s_addr) >> 24 == 0)
150 pi->ipi_ifindex = ntohl(addr.s_addr) & 0xffffff;
151 else {
152 pi->ipi_addr = addr;
153 }
154
155 if (msg.msg_controllen == 0)
156 msg.msg_control = NULL;
157
158 return rump_sys_sendmsg(s, &msg, flags);
159}

161static void
162try_sendmsg_pktinfo(int client, int server, const char *data, size_t datalen,
163 const char *src, const char *dst, in_port_t dstport)
164{
165 struct sockaddr_in from;
166 socklen_t fromlen;
167 int rv;
168 char buf[sizeof(message) * 2];
169
170 RL(rv = sendto_pktinfo(client, data, datalen, 0,
171 src, dst, dstport));
172
173 fromlen = sizeof(from);
174 RL(rv = rump_sys_recvfrom(server, buf, sizeof(buf), 0,
175 (struct sockaddr *)&from, &fromlen));
176
177 ATF_REQUIRE_MSG(addrcmp(&from, src) == 0,
178 "source address of received packet is %s, must be %s",
179 inet_ntoa(from.sin_addr), src);
180}

182static void
183do_send_pktinfo_tests(int client, int server, const char *data, size_t datalen)
184{
185 struct sockaddr_in sa_before, sa_after;
186 socklen_t sa_beforelen, sa_afterlen;
187 int rv;
188 char ipbuf1[sizeof("255.255.255.255")];
189 char ipbuf2[sizeof("255.255.255.255")];
190
191 /* get sockaddr before sendmsg w/IP_PKTINFO */
192 sa_beforelen = sizeof(sa_before);
193 RL(rv = rump_sys_getsockname(client,
194 (struct sockaddr *)&sa_before, &sa_beforelen));
195
196 /*
197 * sendmsg with IP_PKTINFO: 127.0.0.[2345] -> 127.0.0.1:54321, and
198 * check received packet is from 127.0.0.[2345]
199 */
200 try_sendmsg_pktinfo(client, server, data, datalen,
201 "127.0.0.2", "127.0.0.1", SERVERPORT);
202 try_sendmsg_pktinfo(client, server, data, datalen,
203 "127.0.0.3", "127.0.0.1", SERVERPORT);
204 try_sendmsg_pktinfo(client, server, data, datalen,
205 "127.0.0.4", "127.0.0.1", SERVERPORT);
206 try_sendmsg_pktinfo(client, server, data, datalen,
207 "127.0.0.5", "127.0.0.1", SERVERPORT);
208
209 /* get sockaddr after sendmsg w/IP_PKTINFO */
210 sa_afterlen = sizeof(sa_after);
211 RL(rv = rump_sys_getsockname(client,
212 (struct sockaddr *)&sa_after, &sa_afterlen));
213
214 /* confirm sockaddr is not changed */
215 inet_ntop(AF_INET, &sa_before.sin_addr, ipbuf1, sizeof(ipbuf1));
216 inet_ntop(AF_INET, &sa_after.sin_addr, ipbuf2, sizeof(ipbuf2));
217 ATF_REQUIRE_MSG(sa_before.sin_addr.s_addr == sa_after.sin_addr.s_addr,
218 "sockaddr is different from before send. before=%s, after=%s",
219 ipbuf1, ipbuf2);
220}

468static u_int16_t
469in_cksum(uint16_t *p, int len)
470{
471 u_int32_t sum;
472
473 for (sum = 0; len >= 2; len -= 2)
474 sum += *p++;
475 if (len == 1)
476 sum += ntohs(*(uint8_t *)p * 256);
477 sum = (sum >> 16) + (sum & 0xffff);
478 sum = (sum >> 16) + (sum & 0xffff);
479 return ~sum;
480}

488ATF_TC_BODY(pktinfo_send_rawip, tc)
489{
490 struct icmp icmp;
491 size_t icmplen;
492 int client, server;
493
494 setup_test_environment();
495
496 RL(server = rump_sys_socket(AF_INET, SOCK_RAW, IPPROTO_ICMP));
497 RL(client = rump_sys_socket(AF_INET, SOCK_RAW, IPPROTO_ICMP));
498
499 memset(&icmp, 0, sizeof(icmp));
500 icmp.icmp_type = ICMP_ECHOREPLY; /* against confuse REQ with REPLY */
501 icmp.icmp_id = htons(getpid());
502 icmp.icmp_cksum = in_cksum((uint16_t *)&icmp, sizeof(icmp));
503 icmplen = sizeof(icmp);
504
505 /* sendmsg w/IP_PKTINFO from 127.0.0.2 */
506 try_sendmsg_pktinfo(client, server, (const char *)&icmp, icmplen,
507 "127.0.0.2", "127.0.0.1", 0);
508
509 rump_sys_close(client);
510 rump_sys_close(server);
511}

589ATF_TC(pktinfo_send_ifindex);
590ATF_TC_HEAD(pktinfo_send_ifindex, tc)
591{
592 atf_tc_set_md_var(tc, "descr",
593 "sendmsg with IP_PKTINFO to specified interface");
594}
595ATF_TC_BODY(pktinfo_send_ifindex, tc)
596{
597 pid_t child;
598 int channel[2], i;
599 char ifname2[IFNAMSIZ];
600 char ifname3[IFNAMSIZ];
601 char ethername[MAXPATHLEN];
602 char token;
603
604 snprintf(ethername, sizeof(ethername),
605 "t_pktinfo_send.link.%u", getpid());
606
607 RL(pipe(channel));
608
609 child = fork();
610
611 RZ(rump_init()); /* XXX: lo0 is ifindex 1 */
612 netcfg_rump_makeshmif(ethername, ifname2); /* XXX: ifindex=2 */
613 netcfg_rump_makeshmif(ethername, ifname3); /* XXX: ifindex=3 */
614
615 switch (child) {
616 case -1:
617 atf_tc_fail_errno("fork failed");
618 case 0:
619 {
620 int client, rv;
621
622 netcfg_rump_if(ifname2, "192.168.2.1", "255.255.255.0");
623 netcfg_rump_if(ifname3, "192.168.0.1", "255.255.0.0");
624
625 RL(client = rump_sys_socket(AF_INET, SOCK_DGRAM, 0));
626
627 /* wait to ready */
628 close(channel[1]);
629 ATF_CHECK(read(channel[0], &token, 1) == 1 &&
630 token == 'U');
631 close(channel[0]);
632
633 /* few packets would be discarded while resolving arp */
634 for (i = 0; i < 3; i++) {
635 /* send from ifindex 3 = 192.168.0.1 */
636 snprintf(message, sizeof(message), "Hello PKTINFO %d", i);
637
638 RL(rv = sendto_pktinfo(client, message, strlen(message),
639 0, "0.0.0.3", "192.168.2.2", SERVERPORT));
640 }
641
642 rump_sys_close(client);
643 pause();
644 }
645 break;
646 default:
647 {
648 struct sockaddr_in from;
649 socklen_t fromlen;
650 int server, rv;
651 char buf[sizeof(message)];
652
653 netcfg_rump_if(ifname2, "192.168.2.2", "255.255.255.0");
654 netcfg_rump_if(ifname3, "192.168.0.2", "255.255.255.0");
655
656 server = udp_server("0.0.0.0", SERVERPORT);
657
658 /* notify to child */
659 close(channel[0]);
660 ATF_CHECK(write(channel[1], "U", 1) == 1);
661 close(channel[1]);
662
663 memset(buf, 0, sizeof(buf));
664 fromlen = sizeof(from);
665 RL(rv = rump_sys_recvfrom(server, buf, sizeof(buf), 0,
666 (struct sockaddr *)&from, &fromlen));
667 printf("%s: received \"%s\" from %s:%u\n", __func__,
668 buf,
669 inet_ntoa(from.sin_addr), ntohs(from.sin_port));
670
671 ATF_REQUIRE_MSG(addrcmp(&from, "192.168.0.1") == 0,
672 "source address of received packet is %s,"
673 " must be %s",
674 inet_ntoa(from.sin_addr), "192.168.0.1");
675
676 rump_sys_close(server);
677 kill(child, SIGKILL);
678 }
679 break;
680 }
681}

684ATF_TC(pktinfo_send_multicast);
685ATF_TC_HEAD(pktinfo_send_multicast, tc)
686{
687 atf_tc_set_md_var(tc, "descr",
688 "sendmsg with IP_PKTINFO to multicast address"
689 " and specified interface");
690}
691ATF_TC_BODY(pktinfo_send_multicast, tc)
692{
693 pid_t child;
694 int channel[2];
695 char ifname2[IFNAMSIZ];
696 char ifname3[IFNAMSIZ];
697 char ethername[MAXPATHLEN];
698 char token;
699
700 snprintf(ethername, sizeof(ethername),
701 "t_pktinfo_send.link.%u", getpid());
702
703 RL(pipe(channel));
704
705 child = fork();
706
707 RZ(rump_init()); /* XXX: lo0 is ifindex 1 */
708 netcfg_rump_makeshmif(ethername, ifname2); /* XXX: ifindex=2 */
709 netcfg_rump_makeshmif(ethername, ifname3); /* XXX: ifindex=3 */
710
711 switch (child) {
712 case -1:
713 atf_tc_fail_errno("fork failed");
714 case 0:
715 {
716 int client, rv;
717
718 netcfg_rump_if(ifname2, "192.168.2.1", "255.255.255.0");
719 netcfg_rump_if(ifname3, "192.168.0.1", "255.255.0.0");
720
721 RL(client = rump_sys_socket(AF_INET, SOCK_DGRAM, 0));
722
723 /* wait to ready */
724 close(channel[1]);
725 ATF_CHECK(read(channel[0], &token, 1) == 1 &&
726 token == 'U');
727 close(channel[0]);
728
729 /* send from ifindex 2 = 192.168.2.1 */
730 RL(rv = sendto_pktinfo(client, message, strlen(message),
731 0, "0.0.0.2", "224.0.0.1", SERVERPORT));
732
733 /* send from ifindex 3 = 192.168.0.1 */
734 RL(rv = sendto_pktinfo(client, message, strlen(message),
735 0, "0.0.0.3", "224.0.0.1", SERVERPORT));
736
737 rump_sys_close(client);
738 pause();
739 }
740 break;
741 default:
742 {
743 struct sockaddr_in from;
744 socklen_t fromlen;
745 int server, i, rv;
746 char buf[sizeof(message) * 2];
747
748 netcfg_rump_if(ifname2, "192.168.2.2", "255.255.255.0");
749 netcfg_rump_if(ifname3, "192.168.0.2", "255.255.255.0");
750
751 server = udp_server("0.0.0.0", SERVERPORT);
752
753 /* notify to child */
754 close(channel[0]);
755 ATF_CHECK(write(channel[1], "U", 1) == 1);
756 close(channel[1]);
757
758 for (i = 0; i < 2; i++) {
759 fromlen = sizeof(from);
760 RL(rv = rump_sys_recvfrom(server, buf,
761 sizeof(buf), 0,
762 (struct sockaddr *)&from, &fromlen));
763 printf("%s: received from %s:%u\n", __func__,
764 inet_ntoa(from.sin_addr),
765 ntohs(from.sin_port));
766 }
767
768 rump_sys_close(server);
769 kill(child, SIGKILL);
770 }
771 break;
772 }
