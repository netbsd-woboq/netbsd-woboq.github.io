68static void
69print(const char *msg, struct sockaddr_un *addr, socklen_t len)
70{
71 size_t i;
72
73 printf("%s: client socket length: %zu\n", msg, (size_t)len);
74 printf("%s: client family %d\n", msg, addr->sun_family);
75#ifdef BSD4_4
76 printf("%s: client len %d\n", msg, addr->sun_len);
77#endif
78 printf("%s: socket name: ", msg);
79 for (i = 0; i < len - OF; i++) {
80 int ch = addr->sun_path[i];
81 if (ch < ' ' || '~' < ch)
82 printf("\\x%02x", ch);
83 else
84 printf("%c", ch);
85 }
86 printf("\n");
87}

89static int
90acc(int s)
91{
92 char guard1;
93 struct sockaddr_un sun;
94 char guard2;
95 socklen_t len;
96
97 guard1 = guard2 = 's';
98
99 memset(&sun, 0, sizeof(sun));
100 len = sizeof(sun);
101 if ((s = accept(s, (struct sockaddr *)&sun, &len)) == -1)
102 FAIL("accept");
103 if (guard1 != 's')
104 FAIL("guard1 = '%c'", guard1);
105 if (guard2 != 's')
106 FAIL("guard2 = '%c'", guard2);
107#ifdef DEBUG
108 print("accept", &sun, len);
109#endif
110 if (len != 2)
111 FAIL("len %d != 2", len);
112 if (sun.sun_family != AF_UNIX)
113 FAIL("sun->sun_family %d != AF_UNIX", sun.sun_family);
114#ifdef BSD4_4
115 if (sun.sun_len != 2)
116 FAIL("sun->sun_len %d != 2", sun.sun_len);
117#endif
118 for (size_t i = 0; i < sizeof(sun.sun_path); i++)
119 if (sun.sun_path[i])
120 FAIL("sun.sun_path[%zu] %d != NULL", i,
121 sun.sun_path[i]);
122 return s;
123fail:
124 if (s != -1)
125 close(s);
126 return -1;
127}

129static int
130peercred(int s, uid_t *euid, gid_t *egid, pid_t *pid)
131{
132#ifdef LOCAL_PEEREID
133 struct unpcbid cred;
134 socklen_t crl;
135 crl = sizeof(cred);
136 if (getsockopt(s, 0, LOCAL_PEEREID, &cred, &crl) == -1)
137 return -1;
138 *euid = cred.unp_euid;
139 *egid = cred.unp_egid;
140 *pid = cred.unp_pid;
141 return 0;
142#else
143 *pid = -1;
144 return getpeereid(s, euid, egid);
145#endif
146}

148static int
149test(bool forkit, bool closeit, size_t len)
150{
151 size_t slen;
152 socklen_t sl;
153 int srvr = -1, clnt = -1, acpt = -1;
154 uid_t euid;
155 gid_t egid;
156 pid_t srvrpid, clntpid;
157 struct sockaddr_un *sock_addr = NULL, *sun = NULL;
158 socklen_t sock_addrlen;
159 socklen_t peer_addrlen;
160 struct sockaddr_un peer_addr;
161
162 srvrpid = clntpid = getpid();
163 srvr = socket(AF_UNIX, SOCK_STREAM, 0);
164 if (srvr == -1)
165 FAIL("socket(srvrer)");
166
167 slen = len + OF + 1;
168
169 if ((sun = calloc(1, slen)) == NULL)
170 FAIL("calloc");
171
172 srvr = socket(AF_UNIX, SOCK_STREAM, 0);
173 if (srvr == -1)
174 FAIL("socket");
175
176 memset(sun->sun_path, 'a', len);
177 sun->sun_path[len] = '\0';
178 (void)unlink(sun->sun_path);
179
180 sl = SUN_LEN(sun);
181#ifdef BSD4_4
182 sun->sun_len = sl;
183#endif
184 sun->sun_family = AF_UNIX;
185
186 if (bind(srvr, (struct sockaddr *)sun, sl) == -1) {
187 if (errno == EINVAL && sl >= 256) {
188 close(srvr);
189 return -1;
190 }
191 FAIL("bind");
192 }
193
194 if (listen(srvr, SOMAXCONN) == -1)
195 FAIL("listen");
196
197 if (forkit) {
198 switch (clntpid = fork()) {
199 case 0: /* child */
200 srvrpid = getppid();
201 clntpid = getpid();
202 break;
203 case -1:
204 FAIL("fork");
205 default:
206 break;
207 }
208 }
209
210 if (clntpid == getpid()) {
211 pid_t pid = srvrpid;
212 clnt = socket(AF_UNIX, SOCK_STREAM, 0);
213 if (clnt == -1)
214 FAIL("socket(client)");
215
216 if (connect(clnt, (const struct sockaddr *)sun, sl) == -1)
217 FAIL("connect");
218
219 if (peercred(clnt, &euid, &egid, &pid) == -1)
220 FAIL("peercred (client)");
221 printf("peercred(client) euid=%jd egid=%jd pid=%jd\n",
222 (intmax_t)euid, (intmax_t)egid, (intmax_t)pid);
223 CHECK_EQUAL(euid, geteuid(), "client");
224 CHECK_EQUAL(egid, getegid(), "client");
225 CHECK_EQUAL(pid, srvrpid, "client");
226 }
227
228 if (srvrpid == getpid()) {
229 acpt = acc(srvr);
230
231 peer_addrlen = sizeof(peer_addr);
232 memset(&peer_addr, 0, sizeof(peer_addr));
233 if (getpeername(acpt, (struct sockaddr *)&peer_addr,
234 &peer_addrlen) == -1)
235 FAIL("getpeername");
236 print("peer", &peer_addr, peer_addrlen);
237 }
238
239 if (clntpid == getpid()) {
240 if (closeit) {
241 if (close(clnt) == -1)
242 FAIL("close");
243 clnt = -1;
244 }
245 }
246
247 if (srvrpid == getpid()) {
248 pid_t pid = clntpid;
249 if (peercred(acpt, &euid, &egid, &pid) == -1)
250 FAIL("peercred (server)");
251 printf("peercred(server) euid=%jd egid=%jd pid=%jd\n",
252 (intmax_t)euid, (intmax_t)egid, (intmax_t)pid);
253 CHECK_EQUAL(euid, geteuid(), "server");
254 CHECK_EQUAL(egid, getegid(), "server");
255 CHECK_EQUAL(pid, clntpid, "server");
256
257 if ((sock_addr = calloc(1, slen)) == NULL)
258 FAIL("calloc");
259 sock_addrlen = slen;
260 if (getsockname(srvr, (struct sockaddr *)sock_addr,
261 &sock_addrlen) == -1)
262 FAIL("getsockname");
263 print("sock", sock_addr, sock_addrlen);
264
265 if (sock_addr->sun_family != AF_UNIX)
266 FAIL("sock_addr->sun_family %d != AF_UNIX",
267 sock_addr->sun_family);
268
269 len += OF;
270 if (sock_addrlen LX != len)
271 FAIL("sock_addr_len %zu != %zu", (size_t)sock_addrlen,
272 len);
273#ifdef BSD4_4
274 if (sock_addr->sun_len != sl)
275 FAIL("sock_addr.sun_len %d != %zu", sock_addr->sun_len,
276 (size_t)sl);
277#endif
278 for (size_t i = 0; i < slen - OF; i++)
279 if (sock_addr->sun_path[i] != sun->sun_path[i])
280 FAIL("sock_addr.sun_path[%zu] %d != "
281 "sun->sun_path[%zu] %d\n", i,
282 sock_addr->sun_path[i], i,
283 sun->sun_path[i]);
284
285 if (acpt != -1)
286 (void)close(acpt);
287 if (srvr != -1)
288 (void)close(srvr);
289 free(sock_addr);
290 sock_addr = NULL;
291 if (forkit && waitpid(clntpid, NULL, 0) == -1)
292 FAIL("waitpid");
293 }
294 if (clnt != -1 && !closeit)
295 (void)close(clnt);
296
297 free(sock_addr);
298 free(sun);
299 return 0;
300fail:
301 if (srvrpid == getpid()) {
302 if (acpt != -1)
303 (void)close(acpt);
304 if (srvr != -1)
305 (void)close(srvr);
306 }
307 if (clntpid == getpid()) {
308 if (clnt != -1 && !closeit)
309 (void)close(clnt);
310 }
311 free(sock_addr);
312 free(sun);
313 return -1;
314}

318ATF_TC(sockaddr_un_len_exceed);
319ATF_TC_HEAD(sockaddr_un_len_exceed, tc)
320{
321
322 atf_tc_set_md_var(tc, "descr", "Check that exceeding the size of "
323 "unix domain sockets does not trash memory or kernel when "
324 "exceeding the size of the fixed sun_path");
325}
326
327ATF_TC_BODY(sockaddr_un_len_exceed, tc)
328{
329 ATF_REQUIRE_MSG(test(false, false, 254) == -1,
330 "test(false, false, 254): %s", strerror(errno));
331}
