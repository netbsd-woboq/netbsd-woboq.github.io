43#include "kernspace.h"
44
45static struct uvm_object *uobj;
46static struct vm_page *testpg;
47static kcondvar_t tcv;
48
49static bool threadrun = false;
50
51static void
52thread(void *arg)
53{
54
55 mutex_enter(uobj->vmobjlock);
56 threadrun = true;
57 cv_signal(&tcv);
58 testpg->flags |= PG_WANTED;
59 UVM_UNLOCK_AND_WAIT(testpg, uobj->vmobjlock, false, "tw", 0);
60 kthread_exit(0);
61}
62
63void
64rumptest_busypage()
65{
66 struct lwp *newl;
67 int rv;
68
69 cv_init(&tcv, "napina");
70
71 uobj = uao_create(1, 0);
72 mutex_enter(uobj->vmobjlock);
73 testpg = uvm_pagealloc(uobj, 0, NULL, 0);
74 mutex_exit(uobj->vmobjlock);
75 if (testpg == NULL)
76 panic("couldn't create vm page");
77
78 rv = kthread_create(PRI_NONE, KTHREAD_MUSTJOIN | KTHREAD_MPSAFE, NULL,
79 thread, NULL, &newl, "jointest");
80 if (rv)
81 panic("thread creation failed: %d", rv);
82
83 mutex_enter(uobj->vmobjlock);
84 while (!threadrun)
85 cv_wait(&tcv, uobj->vmobjlock);
86
87 uvm_page_unbusy(&testpg, 1);
88 mutex_exit(uobj->vmobjlock);
89
90 rv = kthread_join(newl);
91 if (rv)
92 panic("thread join failed: %d", rv);
93
94}
