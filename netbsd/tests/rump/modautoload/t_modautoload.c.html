25ATF_TC(modautoload);
26ATF_TC_HEAD(modautoload, tc)
27{
28
29 atf_tc_set_md_var(tc, "descr", "tests that kernel module "
30 "autoload works in rump");
31}
32
33static void
34mountkernfs(void)
35{
36 bool old_autoload, new_autoload;
37 size_t old_len, new_len;
38 int error;
39
40 if (!rump_nativeabi_p())
41 atf_tc_skip("host kernel modules not supported");
42
43 rump_init();
44
45 if (rump_sys_mkdir("/kern", 0777) == -1)
46 atf_tc_fail_errno("mkdir /kern");
47
48 new_autoload = true;
49 old_len = sizeof(old_autoload);
50 new_len = sizeof(new_autoload);
51 error = sysctlbyname("kern.module.autoload",
52 &old_autoload, &old_len,
53 &new_autoload, new_len);
54 if (error != 0)
55 atf_tc_fail_errno("could not enable module autoload");
56
57 if (rump_sys_mount(MOUNT_KERNFS, "/kern", 0, NULL, 0) == -1)
58 atf_tc_fail_errno("could not mount kernfs");
59}
60
61/*
62 * Why use kernfs here? It talks to plenty of other parts with the
63 * kernel (e.g. vfs_attach() in modcmd), but is still easy to verify
64 * it's working correctly.
65 */
66
67#define MAGICNUM 1323
68ATF_TC_BODY(modautoload, tc)
69{
70 extern int rumpns_hz;
71 char buf[64];
72 int fd;
73
74 mountkernfs();
75 rumpns_hz = MAGICNUM;
76 if ((fd = rump_sys_open("/kern/hz", O_RDONLY)) == -1)
77 atf_tc_fail_errno("open /kern/hz");
78 if (rump_sys_read(fd, buf, sizeof(buf)) <= 0)
79 atf_tc_fail_errno("read");
80 ATF_REQUIRE(atoi(buf) == MAGICNUM);
81}
82
83ATF_TP_ADD_TCS(tp)
84{
85 ATF_TP_ADD_TC(tp, modautoload);
86
87 return atf_no_error();
88}
