57{
58 atf_tc_set_md_var(tc, "descr", "Checks that loading and unloading "
59 "a module (vfs/tmpfs) is possible");

62static int
63disable_autoload(void)
64{
65 struct sysctlnode q, ans[256];
66 int mib[3];
67 size_t alen;
68 unsigned i;
69 bool no;
70
71 mib[0] = CTL_KERN;
72 mib[1] = CTL_QUERY;
73 alen = sizeof(ans);
74
75 memset(&q, 0, sizeof(q));
76 q.sysctl_flags = SYSCTL_VERSION;
77
78 if (rump_sys___sysctl(mib, 2, ans, &alen, &q, sizeof(q)) == -1)
79 return -1;
80
81 for (i = 0; i < __arraycount(ans); i++)
82 if (strcmp("module", ans[i].sysctl_name) == 0)
83 break;
84 if (i == __arraycount(ans)) {
85 errno = ENOENT;
86 return -1;
87 }
88
89 mib[1] = ans[i].sysctl_num;
90 mib[2] = CTL_QUERY;
91
92 if (rump_sys___sysctl(mib, 3, ans, &alen, &q, sizeof(q)) == -1)
93 return errno;
94
95 for (i = 0; i < __arraycount(ans); i++)
96 if (strcmp("autoload", ans[i].sysctl_name) == 0)
97 break;
98 if (i == __arraycount(ans)) {
99 errno = ENOENT;
100 return -1;
101 }
102
103 mib[2] = ans[i].sysctl_num;
104
105 no = false;
106 alen = 0;
107 if (rump_sys___sysctl(mib, 3, NULL, &alen, &no, sizeof(no)) == -1)
108 return errno;
109
110 return 0;
111
112}

114#define TMPFSMODULE "librumpfs_tmpfs.so"
115ATF_TC_BODY(cmsg_modcmd, tc)
116{
117 struct tmpfs_args args;
118 const struct modinfo *const *mi_start, *const *mi_end;
119 void *handle;
120 int i, rv, loop = 0;
121
122 rump_init();
123
124 if (disable_autoload() == -1)
125 atf_tc_fail_errno("count not disable module autoload");
126
127 memset(&args, 0, sizeof(args));
128 args.ta_version = TMPFS_ARGS_VERSION;
129 args.ta_root_mode = 0777;
130
131 if (rump_sys_mkdir("/mp", 0777) == -1)
132 atf_tc_fail_errno("mkdir mountpoint");
133 if (rump_sys_mount(MOUNT_TMPFS, "/mp", 0, &args, sizeof(args)) != -1)
134 atf_tc_fail("mount unexpectedly succeeded");
135
136 handle = dlopen(TMPFSMODULE, RTLD_GLOBAL);
137 if (handle == NULL) {
138 const char *dlmsg = dlerror();
139 atf_tc_fail("cannot open %s: %s", TMPFSMODULE, dlmsg);
140 }
141
142 again:
143 mi_start = dlsym(handle, "__start_link_set_modules");
144 mi_end = dlsym(handle, "__stop_link_set_modules");
145 if (mi_start == NULL || mi_end == NULL)
146 atf_tc_fail("cannot find module info");
147 if ((rv = rump_pub_module_init(mi_start, (size_t)(mi_end-mi_start)))!=0)
148 atf_tc_fail("module init failed: %d (%s)", rv, strerror(rv));
149 if ((rv = rump_pub_module_init(mi_start, (size_t)(mi_end-mi_start)))==0)
150 atf_tc_fail("module double init succeeded");
151
152 if (rump_sys_mount(MOUNT_TMPFS, "/mp", 0, &args, sizeof(args)) == -1)
153 atf_tc_fail_errno("still cannot mount");
154 if (rump_sys_unmount("/mp", 0) == -1)
155 atf_tc_fail("cannot unmount");
156 for (i = 0; i < (int)(mi_end-mi_start); i++) {
157 if ((rv = rump_pub_module_fini(mi_start[i])) != 0)
158 atf_tc_fail("module fini failed: %d (%s)",
159 rv, strerror(rv));
160 }
161 for (i = 0; i < (int)(mi_end-mi_start); i++) {
162 if ((rv = rump_pub_module_fini(mi_start[i])) == 0)
163 atf_tc_fail("module double fini succeeded");
164 }
165 if (loop++ == 0)
166 goto again;
167
168 if (dlclose(handle)) {
169 const char *dlmsg = dlerror();
170 atf_tc_fail("cannot close %s: %s", TMPFSMODULE, dlmsg);
171 }
172
173 if (rump_sys_mount(MOUNT_TMPFS, "/mp", 0, &args, sizeof(args)) != -1)
174 atf_tc_fail("mount unexpectedly succeeded");
175}
