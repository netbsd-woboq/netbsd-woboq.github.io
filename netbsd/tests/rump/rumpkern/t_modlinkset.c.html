44ATF_TC(modlinkset);
45ATF_TC_HEAD(modlinkset, tc)
46{
47 atf_tc_set_md_var(tc, "descr", "Check that module linkset bootstrap "
48 "works");
49}
50
51/*
52 * We link against cd9660 and msdosfs (both chosed because the names
53 * are unlikely to ever be a substring of a another file system).
54 * Without proper linkset handling at most one will be reported.
55 */
56ATF_TC_BODY(modlinkset, tc)
57{
58 char buf[1024];
59
60 rump_init();
61 if (ukfs_vfstypes(buf, sizeof(buf)) == -1)
62 atf_tc_fail_errno("ukfs_vfstypes");
63
64 ATF_CHECK((strstr(buf, "msdos") != NULL));
65 ATF_CHECK((strstr(buf, "cd9660") != NULL));
66}
67
68ATF_TP_ADD_TCS(tp)
69{
70 ATF_TP_ADD_TC(tp, modlinkset);
71
72 return atf_no_error();
73}
