30#include <sys/types.h>
31#include <sys/mount.h>
32#include <sys/sysctl.h>
33
34#include <rump/rump.h>
35
36#include <atf-c.h>
37
38#include "h_macros.h"
39#include "../kernspace/kernspace.h"
40
41ATF_TC(tsleep);
42ATF_TC_HEAD(tsleep, tc)
43{
44
45 atf_tc_set_md_var(tc, "descr", "Check tsleep variants");
46}
47
48ATF_TC_BODY(tsleep, tc)
49{
50
51 rump_init();
52
53 rump_schedule();
54 rumptest_tsleep(); /* panics if fails */
55 rump_unschedule();
56}
57
58ATF_TP_ADD_TCS(tp)
59{
60 ATF_TP_ADD_TC(tp, tsleep);
61
62 return atf_no_error();
63}
