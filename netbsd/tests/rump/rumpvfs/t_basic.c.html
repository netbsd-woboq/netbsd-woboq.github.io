45ATF_TC(lseekrv);
46ATF_TC_HEAD(lseekrv, tc)
47{
48
49 atf_tc_set_md_var(tc, "descr", "Test lseek return values");
50}
51
52#define TESTFILE "testi"
53
54#define FIVE_MEGS (5*1024*1024)
55#define FIVE_GIGS (5*1024*1024*1024LL)
56
57ATF_TC_BODY(lseekrv, tc)
58{
59 off_t rv;
60 int fd;
61
62 RZ(rump_init());
63 RL(fd = rump_sys_open(TESTFILE, O_RDWR | O_CREAT, 0777));
64
65 rv = rump_sys_lseek(37, FIVE_MEGS, SEEK_SET);
66 ATF_REQUIRE_ERRNO(EBADF, rv == -1);
67
68 rv = rump_sys_lseek(fd, FIVE_MEGS, SEEK_SET);
69 ATF_REQUIRE_EQ(rv, FIVE_MEGS);
70
71 rv = rump_sys_lseek(fd, FIVE_GIGS, SEEK_SET);
72 ATF_REQUIRE_EQ(rv, FIVE_GIGS);
73}
74
75ATF_TP_ADD_TCS(tp)
76{
77
78 ATF_TP_ADD_TC(tp, lseekrv);
79
80 return atf_no_error();
81}
