46ATF_TC(makecn);
47ATF_TC_HEAD(makecn, tc)
48{
49
50 atf_tc_set_md_var(tc, "descr", "Tests makecn/LOOKUP/freecn");
51}
52
53#define TESTFILE "testi"
54
55ATF_TC_BODY(makecn, tc)
56{
57 struct componentname *cn;
58 char pathstr[MAXPATHLEN] = TESTFILE;
59 struct vnode *vp;
60 extern struct vnode *rumpns_rootvnode;
61
62 rump_init();
63
64 /*
65 * Strategy is to create a componentname, edit the passed
66 * string, and then do a lookup with the componentname.
67 */
68 RL(rump_sys_mkdir("/" TESTFILE, 0777));
69
70 /* need stable lwp for componentname */
71 RZ(rump_pub_lwproc_rfork(RUMP_RFCFDG));
72
73 /* try it once with the right path */
74 cn = rump_pub_makecn(RUMP_NAMEI_LOOKUP, 0, pathstr, strlen(pathstr),
75 rump_pub_cred_create(0, 0, 0, NULL), rump_pub_lwproc_curlwp());
76 RZ(RUMP_VOP_LOOKUP(rumpns_rootvnode, &vp, cn));
77 rump_pub_freecn(cn, RUMPCN_FREECRED);
78
79 /* and then with modification-in-the-middle */
80 cn = rump_pub_makecn(RUMP_NAMEI_LOOKUP, 0, pathstr, strlen(pathstr),
81 rump_pub_cred_create(0, 0, 0, NULL), rump_pub_lwproc_curlwp());
82 strcpy(pathstr, "/muuta");
83 RZ(RUMP_VOP_LOOKUP(rumpns_rootvnode, &vp, cn));
84 rump_pub_freecn(cn, RUMPCN_FREECRED);
85}
86
87ATF_TP_ADD_TCS(tp)
88{
89
90 ATF_TP_ADD_TC(tp, makecn);
91
92 return atf_no_error();
93}
