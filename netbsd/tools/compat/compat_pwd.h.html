32#ifndef _COMPAT_PWD_H_
33#define _COMPAT_PWD_H_
34
35/* A very special version of <pwd.h> for pwd_mkdb(8) and __nbcompat_pwscan(3). */
36
37#include "../../include/pwd.h"
38
39#define passwd __nbcompat_passwd
40#define pw_scan __nbcompat_pw_scan
41#ifdef LOGIN_NAME_MAX
42#undef LOGIN_NAME_MAX
43#endif
44/* Taken from syslimits.h. Need the NetBSD def, not the local system def */
45#define LOGIN_NAME_MAX 17
46
47/* All elements exactly sized: */
48struct passwd {
49 char *pw_name;
50 char *pw_passwd;
51 int32_t pw_uid;
52 int32_t pw_gid;
53 int64_t pw_change;
54 char *pw_class;
55 char *pw_gecos;
56 char *pw_dir;
57 char *pw_shell;
58 int64_t pw_expire;
59};
60
61int pw_scan(char *, struct passwd *, int *);
62
63#endif /* !_PWD_H_ */
