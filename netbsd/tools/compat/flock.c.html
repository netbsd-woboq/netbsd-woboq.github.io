32/*
33 * Emulate flock() with fcntl(), where available.
34 * Otherwise, don't do locking; just pretend success.
35 */
36
37#include "nbtool_config.h"
38
39#if !HAVE_FLOCK
40#include <errno.h>
41#include <fcntl.h>
42
43int flock(int fd, int op) {
44 int rc = 0;
45
46#if defined(F_SETLK) && defined(F_SETLKW)
47 struct flock fl = {0};
48
49 switch (op & (LOCK_EX|LOCK_SH|LOCK_UN)) {
50 case LOCK_EX:
51 fl.l_type = F_WRLCK;
52 break;
53
54 case LOCK_SH:
55 fl.l_type = F_RDLCK;
56 break;
57
58 case LOCK_UN:
59 fl.l_type = F_UNLCK;
60 break;
61
62 default:
63 errno = EINVAL;
64 return -1;
65 }
66
67 fl.l_whence = SEEK_SET;
68 rc = fcntl(fd, op & LOCK_NB ? F_SETLK : F_SETLKW, &fl);
69
70 if (rc && (errno == EAGAIN))
71 errno = EWOULDBLOCK;
72#endif
73
74 return rc;
75}
76#endif
