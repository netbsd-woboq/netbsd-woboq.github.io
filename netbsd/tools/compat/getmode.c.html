32#include "nbtool_config.h"
33#include <stdlib.h>
34#include <unistd.h>
35
36void *
37setmode(const char *str)
38{
39 mode_t *mp = malloc(sizeof(mode_t));
40
41 *mp = strtoul(str, NULL, 8);
42
43 return mp;
44}
45
46mode_t
47getmode(const void *mp, mode_t mode)
48{
49 mode_t m;
50
51 m = *((const mode_t *)mp);
52
53 mode &= ~ALLPERMS; /* input mode less RWX permissions */
54 m &= ALLPERMS; /* new RWX permissions */
55
56 return m | mode;
57}
