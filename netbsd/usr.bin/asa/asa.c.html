33#include <sys/cdefs.h>
34#ifndef lint
35__RCSID("$NetBSD: asa.c,v 1.17 2016/09/05 00:40:28 sevan Exp $");
36#endif
37
38#include <err.h>
39#include <stdio.h>
40#include <stdlib.h>
41
42static void asa(FILE *);
43
44int
45/*ARGSUSED*/
46main (int argc, char *argv[])
47{
48 FILE *fp;
49
50 /* skip progname */
51 argv++;
52
53 if (*argv == NULL)
54 asa(stdin);
55 else
56 do {
57 if ((fp = fopen(*argv, "r")) == NULL) {
58 warn("%s", *argv);
59 continue;
60 }
61 asa(fp);
62 (void)fclose(fp);
63 } while (*++argv != NULL);
64
65 return 0;
66}
67
68static void
69asa(FILE *f)
70{
71 char *buf;
72 size_t len;
73
74 if ((buf = fgetln(f, &len)) != NULL) {
75 if (len > 0 && buf[len - 1] == '\n')
76 buf[--len] = '\0';
77 /* special case the first line */
78 switch (buf[0]) {
79 case '0':
80 (void)putchar('\n');
81 break;
82 case '1':
83 (void)putchar('\f');
84 break;
85 }
86
87 if (len > 1 && buf[0] && buf[1])
88 (void)fwrite(buf + 1, 1, len - 1, stdout);
89
90 while ((buf = fgetln(f, &len)) != NULL) {
91 if (len > 0 && buf[len - 1] == '\n')
92 buf[--len] = '\0';
93 switch (buf[0]) {
94 default:
95 case ' ':
96 (void)putchar('\n');
97 break;
98 case '0':
99 (void)putchar('\n');
100 (void)putchar('\n');
101 break;
102 case '1':
103 (void)putchar('\f');
104 break;
105 case '+':
106 (void)putchar('\r');
107 break;
108 }
109
110 if (len > 1 && buf[0] && buf[1])
111 (void)fwrite(buf + 1, 1, len - 1, stdout);
112 }
113
114 (void)putchar('\n');
115 }
116}
