53static inline void
54change_sign8(u_char *p, int cc)
55{
56 while (--cc >= 0) {
57 *p ^= 0x80;
58 ++p;
59 }
60}
61
62static inline void
63change_sign16_le(u_char *p, int cc)
64{
65 while ((cc -= 2) >= 0) {
66 p[1] ^= 0x80;
67 p += 2;
68 }
69}
70
71static inline void
72change_sign16_be(u_char *p, int cc)
73{
74 while ((cc -= 2) >= 0) {
75 p[0] ^= 0x80;
76 p += 2;
77 }
78}

98static inline void
99swap_bytes(u_char *p, int cc)
100{
101 u_char t;
102
103 while ((cc -= 2) >= 0) {
104 t = p[0];
105 p[0] = p[1];
106 p[1] = t;
107 p += 2;
108 }
109}
110
111static inline void
112swap_bytes32(u_char *p, int cc)
113{
114 u_char t;
115
116 while ((cc -= 4) >= 0) {
117 t = p[0];
118 p[0] = p[3];
119 p[3] = t;
120 t = p[1];
121 p[1] = p[2];
122 p[2] = t;
123 p += 4;
124 }
125}
