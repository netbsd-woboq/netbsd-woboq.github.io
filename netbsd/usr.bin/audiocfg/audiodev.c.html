51static int
52audiodev_getinfo(struct audiodev *adev)
53{
54 struct stat st;
55 struct audiofmt *f;
56 audio_format_query_t query;
57 int i;
58
59 if (stat(adev->ctlpath, &st) == -1)
60 return -1;
61 adev->dev = st.st_rdev;
62
63 if (stat(_PATH_AUDIOCTL, &st) != -1 && st.st_rdev == adev->dev)
64 adev->defaultdev = true;
65
66 adev->fd = open(adev->ctlpath, O_RDONLY);
67 if (adev->fd == -1) {
68 return -1;
69 }
70 if (ioctl(adev->fd, AUDIO_GETDEV, &adev->audio_device) == -1) {
71 close(adev->fd);
72 return -1;
73 }
74
75 for (i = 0; ;i++) {
76 memset(&query, 0, sizeof(query));
77 query.index = i;
78 if (ioctl(adev->fd, AUDIO_QUERYFORMAT, &query) == -1) {
79 if (errno == ENODEV) {
80 /* QUERYFORMAT not supported. */
81 break;
82 }
83 if (errno == EINVAL)
84 break;
85 close(adev->fd);
86 return -1;
87 }
88
89 f = calloc(1, sizeof(*f));
90 f->fmt = query.fmt;
91 TAILQ_INSERT_TAIL(&adev->formats, f, next);
92 }
93
94 if (ioctl(adev->fd, AUDIO_GETFORMAT, &adev->info) == -1) {
95 close(adev->fd);
96 return -1;
97 }
98
99 return 0;
100}

102static int
103audiodev_add(const char *pdev, const char *dev, unsigned int unit)
104{
105 struct audiodev *adev;
106
107 adev = calloc(1, sizeof(*adev));
108 if (adev == NULL)
109 return -1;
110
111 strlcpy(adev->pxname, pdev, sizeof(adev->pxname));
112 strlcpy(adev->xname, dev, sizeof(adev->xname));
113 snprintf(adev->path, sizeof(adev->path), "/dev/%s", dev);
114 snprintf(adev->ctlpath, sizeof(adev->ctlpath), "/dev/audioctl%d", unit);
115 adev->unit = unit;
116 TAILQ_INIT(&adev->formats);
117
118 if (audiodev_getinfo(adev) == -1) {
119 free(adev);
120 return -1;
121 }
122
123#ifdef DEBUG
124 printf("DEBUG: [%c] %s(%s): %s\n", adev->defaultdev ? '*' : ' ',
125 adev->path, adev->ctlpath, adev->audio_device.name);
126 struct audiofmt *f;
127 TAILQ_FOREACH(f, &adev->formats, next) {
128 printf("DEBUG: enc%d, %d/%d, %dch\n",
129 f->fmt.encoding,
130 f->fmt.validbits,
131 f->fmt.precision,
132 f->fmt.channels);
133 }
134#endif
135
136 TAILQ_INSERT_TAIL(&audiodevlist, adev, next);
137
138 return 0;
139}

141static void
142audiodev_cb(void *args, const char *pdev, const char *dev, unsigned int unit)
143{
144 audiodev_add(pdev, dev, unit);
145}

147int
148audiodev_refresh(void)
149{
150 struct audiodev *adev;
151 int fd, error;
152
153 fd = open(DRVCTLDEV, O_RDONLY);
154 if (fd == -1) {
155 perror("open " DRVCTLDEV);
156 return -1;
157 }
158
159 while (!TAILQ_EMPTY(&audiodevlist)) {
160 adev = TAILQ_FIRST(&audiodevlist);
161 if (adev->fd != -1)
162 close(adev->fd);
163 TAILQ_REMOVE(&audiodevlist, adev, next);
164 free(adev);
165 }
166
167 error = drvctl_foreach(fd, "audio", audiodev_cb, NULL);
168 if (error == -1) {
169 perror("drvctl");
170 return -1;
171 }
172
173 close(fd);
174
175 return 0;
176}

178unsigned int
179audiodev_count(void)
180{
181 struct audiodev *adev;
182 unsigned int n;
183
184 n = 0;
185 TAILQ_FOREACH(adev, &audiodevlist, next)
186 ++n;
187
188 return n;
189}

191struct audiodev *
192audiodev_get(unsigned int i)
193{
194 struct audiodev *adev;
195 unsigned int n;
196
197 n = 0;
198 TAILQ_FOREACH(adev, &audiodevlist, next) {
199 if (n == i)
200 return adev;
201 ++n;
202 }
203
204 return NULL;
205}

207int
208audiodev_set_default(struct audiodev *adev)
209{
210 char audiopath[PATH_MAX+1];
211 char soundpath[PATH_MAX+1];
212 char audioctlpath[PATH_MAX+1];
213 char mixerpath[PATH_MAX+1];
214
215 snprintf(audiopath, sizeof(audiopath),
216 _PATH_AUDIO "%u", adev->unit);
217 snprintf(soundpath, sizeof(soundpath),
218 _PATH_SOUND "%u", adev->unit);
219 snprintf(audioctlpath, sizeof(audioctlpath),
220 _PATH_AUDIOCTL "%u", adev->unit);
221 snprintf(mixerpath, sizeof(mixerpath),
222 _PATH_MIXER "%u", adev->unit);
223
224 unlink(_PATH_AUDIO);
225 unlink(_PATH_SOUND);
226 unlink(_PATH_AUDIOCTL);
227 unlink(_PATH_MIXER);
228
229 if (symlink(audiopath, _PATH_AUDIO) == -1) {
230 perror("symlink " _PATH_AUDIO);
231 return -1;
232 }
233 if (symlink(soundpath, _PATH_SOUND) == -1) {
234 perror("symlink " _PATH_SOUND);
235 return -1;
236 }
237 if (symlink(audioctlpath, _PATH_AUDIOCTL) == -1) {
238 perror("symlink " _PATH_AUDIOCTL);
239 return -1;
240 }
241 if (symlink(mixerpath, _PATH_MIXER) == -1) {
242 perror("symlink " _PATH_MIXER);
243 return -1;
244 }
245
246 return 0;
247}

249int
250audiodev_set_param(struct audiodev *adev, int mode,
251 const char *encname, unsigned int prec, unsigned int ch, unsigned int freq)
252{
253 struct audio_info ai;
254 int setmode;
255 u_int enc;
256
257 setmode = 0;
258 ai = adev->info;
259
260 for (enc = 0; enc < encoding_max; enc++) {
261 if (strcmp(encname, encoding_names[enc]) == 0)
262 break;
263 }
264 if (enc >= encoding_max) {
265 fprintf(stderr, "unknown encoding name: %s\n", encname);
266 errno = EINVAL;
267 return -1;
268 }
269
270 if ((ai.mode & mode & AUMODE_PLAY)) {
271 setmode |= AUMODE_PLAY;
272 ai.play.encoding = enc;
273 ai.play.precision = prec;
274 ai.play.channels = ch;
275 ai.play.sample_rate = freq;
276 }
277 if ((ai.mode & mode & AUMODE_RECORD)) {
278 setmode |= AUMODE_RECORD;
279 ai.record.encoding = enc;
280 ai.record.precision = prec;
281 ai.record.channels = ch;
282 ai.record.sample_rate = freq;
283 }
284
285 if (setmode == 0) {
286 errno = EINVAL;
287 return -1;
288 }
289
290 ai.mode = setmode;
291 printf("setting %s to %s:%u, %uch, %uHz\n",
292 adev->xname, encname, prec, ch, freq);
293 if (ioctl(adev->fd, AUDIO_SETFORMAT, &ai) == -1) {
294 perror("ioctl AUDIO_SETFORMAT");
295 return -1;
296 }
297 return 0;
298}

300int
301audiodev_test(struct audiodev *adev, unsigned int chanmask)
302{
303 audio_info_t info;
304 int16_t *buf;
305 size_t buflen;
306 off_t off;
307 int fd;
308 int rv = -1;
309
310 fd = open(adev->path, O_WRONLY);
311 if (fd == -1) {
312 perror("open");
313 return -1;
314 }
315
316 AUDIO_INITINFO(&info);
317 info.play.sample_rate = AUDIODEV_SAMPLE_RATE;
318 info.play.channels = adev->info.play.channels;
319 info.play.precision = 16;
320 info.play.encoding = AUDIO_ENCODING_SLINEAR_LE;
321 info.mode = AUMODE_PLAY;
322 if (ioctl(fd, AUDIO_SETINFO, &info) == -1) {
323 perror("ioctl AUDIO_SETINFO");
324 goto abort;
325 }
326 if (ioctl(fd, AUDIO_GETINFO, &info) == -1) {
327 perror("ioctl AUDIO_GETINFO");
328 goto abort;
329 }
330
331 dtmf_new(&buf, &buflen, info.play.sample_rate, 2,
332 adev->info.play.channels, chanmask, 350.0, 440.0);
333 if (buf == NULL) {
334 goto abort;
335 }
336
337 off = 0;
338 while (buflen > 0) {
339 size_t wlen;
340 ssize_t ret;
341
342 wlen = info.play.buffer_size;
343 if (wlen > buflen)
344 wlen = buflen;
345 ret = write(fd, (char *)buf + off, wlen);
346 if (ret == -1) {
347 perror("write");
348 goto done;
349 }
350 wlen = ret;
351 off += wlen;
352 buflen -= wlen;
353 }
354
355 if (ioctl(fd, AUDIO_DRAIN) == -1) {
356 perror("ioctl AUDIO_DRAIN");
357 goto done;
358 }
359
360 rv = 0;
361done:
362 free(buf);
363abort:
364 close(fd);
365
366 return rv;
367}
