38struct audiofmt {
39 struct audio_format fmt;
40 TAILQ_ENTRY(audiofmt) next;
41};

43struct audiodev {
44 char pxname[16]; /* hw (parent) device */
45 char xname[16]; /* audio(4) device */
46 uint16_t unit;
47 char path[PATH_MAX+1];
48 char ctlpath[PATH_MAX+1];
49
50 int fd;
51 dev_t dev;
52 bool defaultdev;
53
54 audio_device_t audio_device;
55 TAILQ_HEAD(, audiofmt) formats;
56 struct audio_info info;
57
58 TAILQ_ENTRY(audiodev) next;
59};
