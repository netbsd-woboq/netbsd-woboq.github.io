55char *
56ttoa(char *buf, size_t len, time_t tval)
57{
58
59 if (tval) {
60 struct tm *tp = localtime(&tval);
61
62 (void) strftime(buf, len, "%B %d, %Y", tp);
63 buf[len - 1] = '\0';
64 }
65 else if (len > 0)
66 *buf = '\0';
67 return (buf);
68}

70int
71atot(const char *p, time_t *store)
72{
73 static struct tm *lt;
74 struct tm tm;
75 char *t;
76 time_t tval;
77
78 if (!*p) {
79 *store = 0;
80 return (0);
81 }
82 if (!lt) {
83 unsetenv("TZ");
84 (void)time(&tval);
85 lt = localtime(&tval);
86 }
87 (void) memset(&tm, 0, sizeof(tm));
88 while ((t = strchr(p, ',')) != NULL)
89 *t = ' ';
90 t = strptime(p, "%B %d %Y", &tm);
91 if (t == NULL || (*t != '\0' && *t != '\n'))
92 return 1;
93 if ((*store = mktime(&tm)) == (time_t) -1)
94 return 1;
95 return (0);
96}

98const char *
99ok_shell(const char *name)
100{
101 char *p;
102 const char *sh;
103
104 setusershell();
105 while ((sh = getusershell()) != NULL) {
106 if (!strcmp(name, sh))
107 return (name);
108 /* allow just shell name, but use "real" path */
109 if ((p = strrchr(sh, '/')) && strcmp(name, p + 1) == 0)
110 return (sh);
111 }
112 return (NULL);
113}
