50int
51csum2(int fd, u_int32_t *cval, off_t *clen)
52{
53 u_int32_t thecrc;
54 off_t total;
55 int nr;
56 u_char *p;
57 u_char buf[8192];
58
59 /*
60 * Draft 8 POSIX 1003.2:
61 *
62 * s = sum of all bytes
63 * r = s % 2^16 + (s % 2^32) / 2^16
64 * thecrc = (r % 2^16) + r / 2^16
65 */
66 thecrc = 0;
67 total = 0;
68 while ((nr = read(fd, buf, sizeof(buf))) > 0)
69 for (total += nr, p = buf; nr--; ++p)
70 thecrc += *p;
71 if (nr < 0)
72 return 1;
73
74 thecrc = (thecrc & 0xffff) + (thecrc >> 16);
75 thecrc = (thecrc & 0xffff) + (thecrc >> 16);
76
77 *cval = thecrc;
78 *clen = total;
79 return 0;
80}
