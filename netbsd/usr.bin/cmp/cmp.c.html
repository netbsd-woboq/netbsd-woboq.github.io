60int lflag, sflag;
61
62__dead static void usage(void);
63
64int
65main(int argc, char *argv[])
66{
67 struct stat sb1, sb2;
68 off_t skip1 = 0, skip2 = 0;
69 int ch, fd1, fd2, special;
70 const char *file1, *file2;
71
72 setlocale(LC_ALL, "");
73
74 special = 0;
75 while ((ch = getopt(argc, argv, "cls")) != -1)
76 switch (ch) {
77 case 'c':
78 special = 1; /* careful! Don't use mmap() */
79 break;
80 case 'l': /* print all differences */
81 lflag = 1;
82 break;
83 case 's': /* silent run */
84 sflag = 1;
85 break;
86 case '?':
87 default:
88 usage();
89 }
90 argv += optind;
91 argc -= optind;
92
93 if (lflag && sflag)
94 errx(ERR_EXIT, "only one of -l and -s may be specified");
95
96 if (argc < 2 || argc > 4)
97 usage();
98
99 /* Backward compatibility -- handle "-" meaning stdin. */
100 if (strcmp(file1 = argv[0], "-") == 0) {
101 special = 1;
102 fd1 = 0;
103 file1 = "stdin";
104 }
105 else if ((fd1 = open(file1, O_RDONLY, 0)) < 0) {
106 if (!sflag)
107 warn("%s", file1);
108 exit(ERR_EXIT);
109 }
110 if (strcmp(file2 = argv[1], "-") == 0) {
111 if (special)
112 errx(ERR_EXIT,
113 "standard input may only be specified once");
114 special = 1;
115 fd2 = 0;
116 file2 = "stdin";
117 }
118 else if ((fd2 = open(file2, O_RDONLY, 0)) < 0) {
119 if (!sflag)
120 warn("%s", file2);
121 exit(ERR_EXIT);
122 }
123
124 if (argc > 2) {
125 char *ep;
126
127 errno = 0;
128 skip1 = strtoq(argv[2], &ep, 0);
129 if (errno || ep == argv[2])
130 usage();
131
132 if (argc == 4) {
133 skip2 = strtoq(argv[3], &ep, 0);
134 if (errno || ep == argv[3])
135 usage();
136 }
137 }
138
139 if (!special) {
140 if (fstat(fd1, &sb1))
141 err(ERR_EXIT, "%s", file1);
142 if (!S_ISREG(sb1.st_mode))
143 special = 1;
144 else {
145 if (fstat(fd2, &sb2))
146 err(ERR_EXIT, "%s", file2);
147 if (!S_ISREG(sb2.st_mode))
148 special = 1;
149 }
150 }
151
152 if (special)
153 c_special(fd1, file1, skip1, fd2, file2, skip2);
154 else
155 c_regular(fd1, file1, skip1, sb1.st_size,
156 fd2, file2, skip2, sb2.st_size);
157 exit(0);
158}
159
160static void
161usage(void)
162{
163
164 (void)fprintf(stderr,
165 "Usage: %s [-c] [-l | -s] file1 file2 [skip1 [skip2]]\n",
166 getprogname());
167 exit(ERR_EXIT);
168}
