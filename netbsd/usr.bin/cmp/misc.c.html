49void
50errmsg(const char *file, off_t byte, off_t line)
51{
52 if (lflag)
53 err(ERR_EXIT, "%s: char %lld, line %lld", file,
54 (long long)byte, (long long)line);
55 else
56 err(ERR_EXIT, "%s", file);
57}
58
59void
60eofmsg(const char *file, off_t byte, off_t line)
61{
62 if (!sflag) {
63 if (!lflag)
64 warnx("EOF on %s", file);
65 else {
66 if (line > 0)
67 warnx("EOF on %s: char %lld, line %lld",
68 file, (long long)byte, (long long)line);
69 else
70 warnx("EOF on %s: char %lld",
71 file, (long long)byte);
72 }
73 }
74 exit(DIFF_EXIT);
75}
76
77void
78diffmsg(const char *file1, const char *file2, off_t byte, off_t line)
79{
80 if (!sflag)
81 (void)printf("%s %s differ: char %lld, line %lld\n",
82 file1, file2, (long long)byte, (long long)line);
83 exit(DIFF_EXIT);
84}
