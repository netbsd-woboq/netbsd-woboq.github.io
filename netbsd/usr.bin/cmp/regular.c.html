53void
54c_regular(int fd1, const char *file1, off_t skip1, off_t len1,
55 int fd2, const char *file2, off_t skip2, off_t len2)
56{
57 u_char ch, *p1, *p2;
58 off_t byte, length, line;
59 int dfound;
60 size_t blk_sz, blk_cnt;
61
62 if (sflag && len1 != len2)
63 exit(1);
64
65 if (skip1 > len1)
66 eofmsg(file1, len1 + 1, 0);
67 len1 -= skip1;
68 if (skip2 > len2)
69 eofmsg(file2, len2 + 1, 0);
70 len2 -= skip2;
71
72 byte = line = 1;
73 dfound = 0;
74 length = MIN(len1, len2);
75 for (blk_sz = 1024 * 1024; length != 0; length -= blk_sz) {
76 if ((uintmax_t)blk_sz > (uintmax_t)length)
77 blk_sz = length;
78 p1 = mmap(NULL, blk_sz, PROT_READ, MAP_FILE|MAP_SHARED,
79 fd1, skip1);
80 if (p1 == MAP_FAILED)
81 goto mmap_failed;
82
83 p2 = mmap(NULL, blk_sz, PROT_READ, MAP_FILE|MAP_SHARED,
84 fd2, skip2);
85 if (p2 == MAP_FAILED) {
86 munmap(p1, blk_sz);
87 goto mmap_failed;
88 }
89
90 blk_cnt = blk_sz;
91 for (; blk_cnt--; ++p1, ++p2, ++byte) {
92 if ((ch = *p1) != *p2) {
93 if (!lflag) {
94 diffmsg(file1, file2, byte, line);
95 /* NOTREACHED */
96 }
97 dfound = 1;
98 (void)printf("%6lld %3o %3o\n",
99 (long long)byte, ch, *p2);
100 }
101 if (ch == '\n')
102 ++line;
103 }
104 munmap(p1 - blk_sz, blk_sz);
105 munmap(p2 - blk_sz, blk_sz);
106 skip1 += blk_sz;
107 skip2 += blk_sz;
108 }
109
110 if (len1 != len2)
111 eofmsg(len1 > len2 ? file2 : file1, byte, line);
112 if (dfound)
113 exit(DIFF_EXIT);
114 return;
115
116mmap_failed:
117 c_special(fd1, file1, skip1, fd2, file2, skip2);
118}
