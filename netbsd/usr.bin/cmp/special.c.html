49void
50c_special(int fd1, const char *file1, off_t skip1, int fd2, const char *file2, off_t skip2)
51{
52 int ch1, ch2;
53 off_t byte, line;
54 FILE *fp1, *fp2;
55 int dfound;
56
57 dfound = 0;
58 if ((fp1 = fdopen(fd1, "r")) == NULL)
59 err(ERR_EXIT, "%s", file1);
60 if ((fp2 = fdopen(fd2, "r")) == NULL)
61 err(ERR_EXIT, "%s", file2);
62
63 for (byte = line = 1; skip1--; byte++) {
64 ch1 = getc(fp1);
65 if (ch1 == EOF)
66 goto eof;
67 if (ch1 == '\n')
68 line++;
69 }
70 for (byte = line = 1; skip2--; byte++) {
71 ch2 = getc(fp2);
72 if (ch2 == EOF)
73 goto eof;
74 if (ch2 == '\n')
75 line++;
76 }
77 dfound = 0;
78 for (byte = line = 1;; ++byte) {
79 ch1 = getc(fp1);
80 ch2 = getc(fp2);
81 if (ch1 == EOF || ch2 == EOF)
82 break;
83 if (ch1 != ch2) {
84 if (lflag) {
85 dfound = 1;
86 (void)printf("%6lld %3o %3o\n", (long long)byte,
87 ch1, ch2);
88 } else
89 diffmsg(file1, file2, byte, line);
90 /* NOTREACHED */
91 }
92 if (ch1 == '\n')
93 ++line;
94 }
95
96 eof:
97 if (ferror(fp1))
98 errmsg(file1, byte, line);
99 if (ferror(fp2))
100 errmsg(file2, byte, line);
101 if (feof(fp1)) {
102 if (!feof(fp2))
103 eofmsg(file1, byte, line);
104 } else
105 if (feof(fp2))
106 eofmsg(file2, byte, line);
107 (void)fclose(fp1);
108 (void)fclose(fp2);
109 if (dfound)
110 exit(DIFF_EXIT);
111}
