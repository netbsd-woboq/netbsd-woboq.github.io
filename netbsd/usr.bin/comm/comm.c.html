56#define MAXLINELEN (LINE_MAX + 1)
57
58static const char *tabs[] = { "", "\t", "\t\t" };
59
60static FILE *file(const char *);
61static void show(FILE *, const char *, char *);
62__dead static void usage(void);
63static char *getnextln(char *buf, FILE *);
64
65int
66main(int argc, char **argv)
67{
68 int comp, file1done, file2done, read1, read2;
69 int ch, flag1, flag2, flag3;
70 FILE *fp1, *fp2;
71 const char *col1, *col2, *col3, **p;
72 char line1[MAXLINELEN], line2[MAXLINELEN];
73 int (*compare)(const char*,const char*);
74
75 (void)setlocale(LC_ALL, "");
76
77 file1done = file2done = 0;
78 flag1 = flag2 = flag3 = 1;
79 compare = strcoll;
80 while ((ch = getopt(argc, argv, "123f")) != -1)
81 switch(ch) {
82 case '1':
83 flag1 = 0;
84 break;
85 case '2':
86 flag2 = 0;
87 break;
88 case '3':
89 flag3 = 0;
90 break;
91 case 'f':
92 compare = strcasecmp;
93 break;
94 case '?':
95 default:
96 usage();
97 }
98 argc -= optind;
99 argv += optind;
100
101 if (argc != 2)
102 usage();
103
104 fp1 = file(argv[0]);
105 fp2 = file(argv[1]);
106
107 /* for each column printed, add another tab offset */
108 p = tabs;
109 col1 = col2 = col3 = NULL;
110 if (flag1)
111 col1 = *p++;
112 if (flag2)
113 col2 = *p++;
114 if (flag3)
115 col3 = *p;
116
117 for (read1 = read2 = 1;;) {
118 /* read next line, check for EOF */
119 if (read1)
120 file1done = !getnextln(line1, fp1);
121 if (read2)
122 file2done = !getnextln(line2, fp2);
123
124 /* if one file done, display the rest of the other file */
125 if (file1done) {
126 if (!file2done && col2)
127 show(fp2, col2, line2);
128 break;
129 }
130 if (file2done) {
131 if (!file1done && col1)
132 show(fp1, col1, line1);
133 break;
134 }
135
136 /* lines are the same */
137 if (!(comp = compare(line1, line2))) {
138 read1 = read2 = 1;
139 if (col3)
140 if (printf("%s%s\n", col3, line1) < 0)
141 break;
142 continue;
143 }
144
145 /* lines are different */
146 if (comp < 0) {
147 read1 = 1;
148 read2 = 0;
149 if (col1)
150 if (printf("%s%s\n", col1, line1) < 0)
151 break;
152 } else {
153 read1 = 0;
154 read2 = 1;
155 if (col2)
156 if (printf("%s%s\n", col2, line2) < 0)
157 break;
158 }
159 }
160
161 if (ferror (stdout) || fclose (stdout) == EOF)
162 err(1, "stdout");
163
164 exit(0);
165}

167static void
168show(FILE *fp, const char *offset, char *buf)
169{
170 while (printf("%s%s\n", offset, buf) >= 0 && getnextln(buf, fp))
171 ;
172}

174static FILE *
175file(const char *name)
176{
177 FILE *fp;
178
179 if (!strcmp(name, "-"))
180 return (stdin);
181 if ((fp = fopen(name, "r")) == NULL)
182 err(1, "%s", name);
183 return (fp);
184}

187usage(void)

189

192}

194static char *
195getnextln(char *buf, FILE *fp)
196{
197 size_t i = 0;
198 int c;
199
200 while ((c = getc(fp)) != '\n' && c != EOF) {
201 buf[i++] = c;
202
203 if (i >= MAXLINELEN)
204 i--; /* consumes extra characters till newline */
205 }
206
207 if (c == EOF && !i)
208 return NULL;
209
210 buf[i] = 0;
211 return buf;
212}
