58/*
59 * ctags: create a tags file
60 */
61
62NODE *head; /* head of the sorted binary tree */
63
64 /* boolean "func" (see init()) */
65bool _wht[256], _etk[256], _itk[256], _btk[256], _gd[256];
66
67FILE *inf; /* ioptr for current input file */
68FILE *outf; /* ioptr for tags file */
69
70long lineftell; /* ftell after getc( inf ) == '\n' */
71
72int lineno; /* line number of current line */
73int dflag; /* -d: non-macro defines */
74int tflag; /* -t: create tags for typedefs */
75int vflag; /* -v: vgrind style index output */
76int wflag; /* -w: suppress warnings */
77int xflag; /* -x: cxref style output */
78
79char *curfile; /* current input file name */
80char searchar = '/'; /* use /.../ searches by default */
81char lbuf[LINE_MAX];
82
83void init(void);
84void find_entries(char *);
85
86int
87main(int argc, char **argv)
88{
89 static const char *outfile = "tags"; /* output file */
90 int aflag; /* -a: append to tags */
91 int uflag; /* -u: update tags */
92 int exit_val; /* exit value */
93 int step; /* step through args */
94 int ch; /* getopts char */
95 char cmd[100]; /* too ugly to explain */
96
97 aflag = uflag = NO;
98 while ((ch = getopt(argc, argv, "BFadf:tuwvx")) != -1)
99 switch(ch) {
100 case 'B':
101 searchar = '?';
102 break;
103 case 'F':
104 searchar = '/';
105 break;
106 case 'a':
107 aflag++;
108 break;
109 case 'd':
110 dflag++;
111 break;
112 case 'f':
113 outfile = optarg;
114 break;
115 case 't':
116 tflag++;
117 break;
118 case 'u':
119 uflag++;
120 break;
121 case 'w':
122 wflag++;
123 break;
124 case 'v':
125 vflag++;
126 /* FALLTHROUGH */
127 case 'x':
128 xflag++;
129 break;
130 case '?':
131 default:
132 goto usage;
133 }
134 argv += optind;
135 argc -= optind;
136 if (!argc) {
137usage: (void)fprintf(stderr,
138 "usage: ctags [-BFadtuwvx] [-f tagsfile] file ...\n");
139 exit(1);
140 }
141
142 init();
143
144 for (exit_val = step = 0; step < argc; ++step)
145 if (!(inf = fopen(argv[step], "r"))) {
146 warn("%s", argv[step]);
147 exit_val = 1;
148 }
149 else {
150 curfile = argv[step];
151 find_entries(argv[step]);
152 (void)fclose(inf);
153 }
154
155 if (head) {
156 if (xflag)
157 put_entries(head);
158 else {
159 if (uflag) {
160 for (step = 0; step < argc; step++) {
161 (void)snprintf(cmd, sizeof(cmd),
162 "mv %s OTAGS; fgrep -v '\t%s\t' OTAGS >%s; rm OTAGS",
163 outfile, argv[step],
164 outfile);
165 system(cmd);
166 }
167 ++aflag;
168 }
169 if (!(outf = fopen(outfile, aflag ? "a" : "w")))
170 err(exit_val, "%s", outfile);
171 put_entries(head);
172 (void)fclose(outf);
173 if (uflag) {
174 (void)snprintf(cmd, sizeof(cmd),
175 "sort -o %s %s", outfile, outfile);
176 system(cmd);
177 }
178 }
179 }
180 exit(exit_val);
181}

183/*
184 * init --
185 * this routine sets up the boolean psuedo-functions which work by
186 * setting boolean flags dependent upon the corresponding character.
187 * Every char which is NOT in that string is false with respect to
188 * the pseudo-function. Therefore, all of the array "_wht" is NO
189 * by default and then the elements subscripted by the chars in
190 * CWHITE are set to YES. Thus, "_wht" of a char is YES if it is in
191 * the string CWHITE, else NO.
192 */
193void
194init(void)
195{
196 int i;
197 unsigned const char *sp;
198
199 for (i = 0; i < 256; i++) {
200 _wht[i] = _etk[i] = _itk[i] = _btk[i] = NO;
201 _gd[i] = YES;
202 }
203#define CWHITE " \f\t\n"
204 for (sp = CWHITE; *sp; sp++) /* white space chars */
205 _wht[*sp] = YES;
206#define CTOKEN " \t\n\"'#()[]{}=-+%*/&|^~!<>;,.:?"
207 for (sp = CTOKEN; *sp; sp++) /* token ending chars */
208 _etk[*sp] = YES;
209#define CINTOK "ABCDEFGHIJKLMNOPQRSTUVWXYZ_abcdefghijklmnopqrstuvwxyz0123456789"
210 for (sp = CINTOK; *sp; sp++) /* valid in-token chars */
211 _itk[*sp] = YES;
212#define CBEGIN "ABCDEFGHIJKLMNOPQRSTUVWXYZ_abcdefghijklmnopqrstuvwxyz"
213 for (sp = CBEGIN; *sp; sp++) /* token starting chars */
214 _btk[*sp] = YES;
215#define CNOTGD ",;"
216 for (sp = CNOTGD; *sp; sp++) /* invalid after-function chars */
217 _gd[*sp] = NO;
218}

220/*
221 * find_entries --
222 * this routine opens the specified file and calls the function
223 * which searches the file.
224 */
225void
226find_entries(char *file)
227{
228 char *cp;
229
230 lineno = 0; /* should be 1 ?? KB */
231 if ((cp = strrchr(file, '.')) != NULL) {
232 if (cp[1] == 'l' && !cp[2]) {
233 int c;
234
235 for (;;) {
236 if (GETC(==, EOF))
237 return;
238 if (!iswhite(c)) {
239 rewind(inf);
240 break;
241 }
242 }
243#define LISPCHR ";(["
244/* lisp */ if (strchr(LISPCHR, c)) {
245 l_entries();
246 return;
247 }
248/* lex */ else {
249 /*
250 * we search all 3 parts of a lex file
251 * for C references. This may be wrong.
252 */
253 toss_yysec();
254 (void)strlcpy(lbuf, "%%$", sizeof(lbuf));
255 pfnote("yylex", lineno);
256 rewind(inf);
257 }
258 }
259/* yacc */ else if (cp[1] == 'y' && !cp[2]) {
260 /*
261 * we search only the 3rd part of a yacc file
262 * for C references. This may be wrong.
263 */
264 toss_yysec();
265 (void)strlcpy(lbuf, "%%$", sizeof(lbuf));
266 pfnote("yyparse", lineno);
267 y_entries();
268 }
269/* fortran */ else if ((cp[1] != 'c' && cp[1] != 'h') && !cp[2]) {
270 if (PF_funcs())
271 return;
272 rewind(inf);
273 }
274 }
275/* C */ c_entries();
276}
