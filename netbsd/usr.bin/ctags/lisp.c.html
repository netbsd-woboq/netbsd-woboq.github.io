52/*
53 * lisp tag functions
54 * just look for (def or (DEF
55 */
56void
57l_entries(void)
58{
59 int special;
60 char *cp;
61 char savedc;
62 char tok[MAXTOKEN];
63
64 for (;;) {
65 lineftell = ftell(inf);
66 if (!fgets(lbuf, sizeof(lbuf), inf))
67 return;
68 ++lineno;
69 lbp = lbuf;
70 if (!cicmp("(def"))
71 continue;
72 special = NO;
73 switch(*lbp | ' ') {
74 case 'm':
75 if (cicmp("method"))
76 special = YES;
77 break;
78 case 'w':
79 if (cicmp("wrapper") || cicmp("whopper"))
80 special = YES;
81 }
82 for (; !isspace((unsigned char)*lbp); ++lbp)
83 continue;
84 for (; isspace((unsigned char)*lbp); ++lbp)
85 continue;
86 for (cp = lbp; *cp && *cp != '\n'; ++cp)
87 continue;
88 *cp = EOS;
89 if (special) {
90 if (!(cp = strchr(lbp, ')')))
91 continue;
92 for (; cp >= lbp && *cp != ':'; --cp)
93 continue;
94 if (cp < lbp)
95 continue;
96 lbp = cp;
97 for (; *cp && *cp != ')' && *cp != ' '; ++cp)
98 continue;
99 }
100 else
101 for (cp = lbp + 1;
102 *cp && *cp != '(' && *cp != ' '; ++cp)
103 continue;
104 savedc = *cp;
105 *cp = EOS;
106 (void)strlcpy(tok, lbp, sizeof(tok));
107 *cp = savedc;
108 get_line();
109 pfnote(tok, lineno);
110 }
111 /*NOTREACHED*/
112}
