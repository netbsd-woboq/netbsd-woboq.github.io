53/*
54 * get_line --
55 * get the line the token of interest occurred on,
56 * prepare it for printing.
57 */
58void
59get_line(void)
60{
61 long saveftell;
62 int c;
63 int cnt;
64 char *cp;
65
66 saveftell = ftell(inf);
67 (void)fseek(inf, lineftell, SEEK_SET);
68 if (xflag)
69 for (cp = lbuf; GETC(!=, '\n'); *cp++ = c)
70 continue;
71 /*
72 * do all processing here, so we don't step through the
73 * line more than once; means you don't call this routine
74 * unless you're sure you've got a keeper.
75 */
76 else for (cnt = 0, cp = lbuf; GETC(!=, EOF) && cnt < ENDLINE; ++cnt) {
77 if (c == '\\') { /* backslashes */
78 if (cnt > ENDLINE - 2)
79 break;
80 *cp++ = '\\'; *cp++ = '\\';
81 ++cnt;
82 }
83 else if (c == (int)searchar) { /* search character */
84 if (cnt > ENDLINE - 2)
85 break;
86 *cp++ = '\\'; *cp++ = c;
87 ++cnt;
88 }
89 else if (c == '\n') { /* end of keep */
90 *cp++ = '$'; /* can find whole line */
91 break;
92 }
93 else
94 *cp++ = c;
95 }
96 *cp = EOS;
97 (void)fseek(inf, saveftell, SEEK_SET);
98}

100/*
101 * put_entries --
102 * write out the tags
103 */
104void
105put_entries(NODE *node)
106{
107
108 if (node->left)
109 put_entries(node->left);
110 if (vflag)
111 printf("%s %s %d\n",
112 node->entry, node->file, (node->lno + 63) / 64);
113 else if (xflag)
114 printf("%-16s%4d %-16s %s\n",
115 node->entry, node->lno, node->file, node->pat);
116 else
117 fprintf(outf, "%s\t%s\t%c^%s%c\n",
118 node->entry, node->file, searchar, node->pat, searchar);
119 if (node->right)
120 put_entries(node->right);
121}
