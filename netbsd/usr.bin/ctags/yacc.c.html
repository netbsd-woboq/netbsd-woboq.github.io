52/*
53 * y_entries:
54 * find the yacc tags and put them in.
55 */
56void
57y_entries(void)
58{
59 int c;
60 char *sp;
61 bool in_rule;
62 char tok[MAXTOKEN];
63
64 in_rule = NO;
65
66 while (GETC(!=, EOF))
67 switch (c) {
68 case '\n':
69 SETLINE;
70 /* FALLTHROUGH */
71 case ' ':
72 case '\f':
73 case '\r':
74 case '\t':
75 break;
76 case '{':
77 if (skip_key('}'))
78 in_rule = NO;
79 break;
80 case '\'':
81 case '"':
82 if (skip_key(c))
83 in_rule = NO;
84 break;
85 case '%':
86 if (GETC(==, '%'))
87 return;
88 (void)ungetc(c, inf);
89 break;
90 case '/':
91 if (GETC(==, '*'))
92 skip_comment('*');
93 else
94 (void)ungetc(c, inf);
95 break;
96 case '|':
97 case ';':
98 in_rule = NO;
99 break;
100 default:
101 if (in_rule || (!isalpha(c) && c != '.' && c != '_'))
102 break;
103 sp = tok;
104 *sp++ = c;
105 while (GETC(!=, EOF) && (intoken(c) || c == '.'))
106 *sp++ = c;
107 *sp = EOS;
108 get_line(); /* may change before ':' */
109 if (c == EOF)
110 return;
111 while (iswhite(c)) {
112 if (c == '\n')
113 SETLINE;
114 if (GETC(==, EOF))
115 return;
116 }
117 if (c == ':') {
118 pfnote(tok, lineno);
119 in_rule = YES;
120 }
121 else
122 (void)ungetc(c, inf);
123 }
124}

126/*
127 * toss_yysec --
128 * throw away lines up to the next "\n%%\n"
129 */
130void
131toss_yysec(void)
132{
133 int c; /* read character */
134 int state;
135
136 /*
137 * state == 0 : waiting
138 * state == 1 : received a newline
139 * state == 2 : received first %
140 * state == 3 : received second %
141 */
142 lineftell = ftell(inf);
143 for (state = 0; GETC(!=, EOF);)
144 switch (c) {
145 case '\n':
146 ++lineno;
147 lineftell = ftell(inf);
148 if (state == 3) /* done! */
149 return;
150 state = 1; /* start over */
151 break;
152 case '%':
153 if (state) /* if 1 or 2 */
154 ++state; /* goto 3 */
155 break;
156 default:
157 state = 0; /* reset */
158 break;
159 }
160}
