29/* Type of a Cell */
30typedef int64_t Cell;
31
32/* Token from the scanner. */
33struct tok {
34 int type;
35 char *text;
36};
37
38#define TOKEN struct tok
39#define YY_DECL TOKEN* yylex(void)
40
41#define FCODE 0x000FC0DE
42#define MACRO 0x0000F00D
43
44/* Defined fcode and string. */
45struct fcode {
46 const char *name;
47 long num;
48 int type;
49 struct fcode *l;
50 struct fcode *r;
51};
52
53/* macro instruction as separate words */
54struct macro {
55 const char *name;
56 const char *equiv;
57 int type;
58 struct macro *l;
59 struct macro *r;
60};
61
62/*
63 * FCode header -- assumes big-endian machine,
64 * otherwise the bits need twiddling.
65 */
66struct fcode_header {
67 char header;
68 char format;
69 short checksum;
70 int length;
71};
