63void
64printlong(char *name, /* filename to print */
65 char *accpath, /* current valid path to filename */
66 struct stat *sb) /* stat buffer */
67{
68 char modep[15];
69
70 (void)printf("%7lu %6lld ", (u_long)sb->st_ino,
71 (long long)sb->st_blocks);
72 (void)strmode(sb->st_mode, modep);
73 (void)printf("%s %3lu %-*s %-*s ", modep, (unsigned long)sb->st_nlink,
74 LOGIN_NAME_MAX, user_from_uid(sb->st_uid, 0), LOGIN_NAME_MAX,
75 group_from_gid(sb->st_gid, 0));
76
77 if (S_ISCHR(sb->st_mode) || S_ISBLK(sb->st_mode))
78 (void)printf("%3llu,%5llu ",
79 (unsigned long long)major(sb->st_rdev),
80 (unsigned long long)minor(sb->st_rdev));
81 else
82 (void)printf("%9lld ", (long long)sb->st_size);
83 printtime(sb->st_mtime);
84 (void)printf("%s", name);
85 if (S_ISLNK(sb->st_mode))
86 printlink(accpath);
87 (void)putchar('\n');
88}

90static void
91printtime(time_t ftime)
92{
93 int i;
94 char *longstring;
95
96 longstring = ctime(&ftime);
97 for (i = 4; i < 11; ++i)
98 (void)putchar(longstring[i]);
99
100#define SIXMONTHS ((DAYSPERNYEAR / 2) * SECSPERDAY)
101 if (ftime + SIXMONTHS > time(NULL))
102 for (i = 11; i < 16; ++i)
103 (void)putchar(longstring[i]);
104 else {
105 (void)putchar(' ');
106 for (i = 20; i < 24; ++i)
107 (void)putchar(longstring[i]);
108 }
109 (void)putchar(' ');
110}

112static void
113printlink(char *name)
114{
115 int lnklen;
116 char path[MAXPATHLEN + 1];
117
118 if ((lnklen = readlink(name, path, sizeof(path) - 1)) == -1) {
119 warn("%s", name);
120 return;
121 }
122 path[lnklen] = '\0';
123 (void)printf(" -> %s", path);
124}
