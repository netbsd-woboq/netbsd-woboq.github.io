57/*
58 * brace_subst --
59 * Replace occurrences of {} in orig with path, and place it in a malloced
60 * area of memory set in store.
61 */
62void
63brace_subst(char *orig, char **store, char *path, int *len)
64{
65 int nlen, plen, rest;
66 char ch, *p, *ostore;
67
68 plen = strlen(path);
69 for (p = *store; (ch = *orig) != '\0'; ++orig)
70 if (ch == '{' && orig[1] == '}') {
71 /* Length of string after the {}. */
72 rest = strlen(&orig[2]);
73
74 nlen = *len;
75 while ((p - *store) + plen + rest + 1 > nlen)
76 nlen *= 2;
77
78 if (nlen > *len) {
79 ostore = *store;
80 if ((*store = realloc(ostore, nlen)) == NULL)
81 err(1, "realloc");
82 *len = nlen;
83 p += *store - ostore; /* Relocate. */
84 }
85 memmove(p, path, plen);
86 p += plen;
87 ++orig;
88 } else
89 *p++ = ch;
90 *p = '\0';
91}

93/*
94 * queryuser --
95 * print a message to standard error and then read input from standard
96 * input. If the input is 'y' then 1 is returned.
97 */
98int
99queryuser(char **argv)
100{
101 int ch, first, nl;
102
103 (void)fprintf(stderr, "\"%s", *argv);
104 while (*++argv)
105 (void)fprintf(stderr, " %s", *argv);
106 (void)fprintf(stderr, "\"? ");
107 (void)fflush(stderr);
108
109 first = ch = getchar();
110 for (nl = 0;;) {
111 if (ch == '\n') {
112 nl = 1;
113 break;
114 }
115 if (ch == EOF)
116 break;
117 ch = getchar();
118 }
119
120 if (!nl) {
121 (void)fprintf(stderr, "\n");
122 (void)fflush(stderr);
123 }
124 return (first == 'y');
125}

127/*
128 * show_path --
129 * called on SIGINFO
130 */
131/* ARGSUSED */
132void
133show_path(int sig)
134{
135 extern FTSENT *g_entry;
136 int errno_bak;
137
138 if (g_entry == NULL) {
139 /*
140 * not initialized yet.
141 * assumption: pointer assignment is atomic.
142 */
143 return;
144 }
145
146 errno_bak = errno;
147 write(STDERR_FILENO, "find path: ", 11);
148 write(STDERR_FILENO, g_entry->fts_path, g_entry->fts_pathlen);
149 write(STDERR_FILENO, "\n", 1);
150 errno = errno_bak;
151}
