148/*
149 * find_create --
150 * create a node corresponding to a command line argument.
151 *
152 * TODO:
153 * add create/process function pointers to node, so we can skip
154 * this switch stuff.
155 */
156PLAN *
157find_create(char ***argvp)
158{
159 OPTION *p;
160 PLAN *new;
161 char **argv;
162 char *opt;
163
164 argv = *argvp;
165 opt = *argv;
166
167 if ((p = option(opt)) == NULL)
168 errx(1, "%s: unknown option", opt);
169 ++argv;
170 if (p->arg && !*argv)
171 errx(1, "%s: requires additional arguments", opt);
172
173 new = (p->create)(&argv, p->token == N_OK, opt);
174
175 *argvp = argv;
176 return (new);
177}
178
179static OPTION *
180option(char *name)
181{
182 OPTION tmp;
183
184 tmp.name = name;
185 return ((OPTION *)bsearch(&tmp, options,
186 sizeof(options)/sizeof(OPTION), sizeof(OPTION), typecompare));
187}
188
189int
190typecompare(const void *a, const void *b)
191{
192
193 return (strcmp(((const OPTION *)a)->name, ((const OPTION *)b)->name));
194}
