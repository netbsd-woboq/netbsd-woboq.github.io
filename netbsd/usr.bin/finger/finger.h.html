38/*
39 * All unique persons are linked in a list headed by "head" and linkd
40 * by the "next" field, as well as kept in a hash table.
41 */
42
43typedef struct person {
44 uid_t uid; /* user id */
45 char *dir; /* user's home directory */
46 char *homephone; /* pointer to home phone no. */
47 char *name; /* login name */
48 char *office; /* pointer to office name */
49 char *officephone; /* pointer to office phone no. */
50 char *realname; /* pointer to full name */
51 char *shell; /* user's shell */
52 time_t mailread; /* last time mail was read */
53 time_t mailrecv; /* last time mail was read */
54 struct where *whead, *wtail; /* list of where user is or has been */
55} PERSON;
56
57enum status { LASTLOG, LOGGEDIN };
58
59typedef struct where {
60 struct where *next; /* next place user is or has been */
61 enum status info; /* type/status of request */
62 short writable; /* tty is writable */
63 time_t loginat; /* time of (last) login */
64 time_t idletime; /* how long idle (if logged in) */
65 char *tty; /* tty line */
66 char *host; /* remote host name */
67} WHERE;
