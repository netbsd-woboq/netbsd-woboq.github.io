66void
67netfinger(char *name)
68{
69 FILE *fp;
70 int c, lastc;
71 int s;
72 char *host;
73 struct addrinfo hints, *res, *res0;
74 int error;
75 const char *emsg = NULL;
76
77 lastc = 0;
78 if (!(host = strrchr(name, '@')))
79 return;
80 *host++ = '\0';
81 memset(&hints, 0, sizeof(hints));
82 hints.ai_family = PF_UNSPEC;
83 hints.ai_socktype = SOCK_STREAM;
84 hints.ai_flags = AI_CANONNAME;
85 error = getaddrinfo(host, "finger", &hints, &res0);
86 if (error) {
87 warnx("%s: %s", gai_strerror(error), host);
88 return;
89 }
90
91 s = -1;
92 for (res = res0; res; res = res->ai_next) {
93 s = socket(res->ai_family, res->ai_socktype, res->ai_protocol);
94 if (s < 0) {
95 emsg = "socket";
96 continue;
97 }
98
99 if (connect(s, res->ai_addr, res->ai_addrlen)) {
100 close(s);
101 s = -1;
102 emsg = "connect";
103 continue;
104 }
105
106 break;
107 }
108 if (s < 0) {
109 if (emsg != NULL)
110 warn("%s", emsg);
111 return;
112 }
113
114 /* have network connection; identify the host connected with */
115 (void)printf("[%s]\n", res0->ai_canonname ? res0->ai_canonname : host);
116
117 /* -l flag for remote fingerd */
118 if (lflag)
119 write(s, "/W ", 3);
120 /* send the name followed by <CR><LF> */
121 (void)write(s, name, strlen(name));
122 (void)write(s, "\r\n", 2);
123
124 /*
125 * Read from the remote system; once we're connected, we assume some
126 * data. If none arrives, we hang until the user interrupts.
127 *
128 * If we see a <CR> followed by a newline character, only output
129 * one newline.
130 *
131 * If a character isn't printable and it isn't a space, we strip the
132 * 8th bit and set the 7th bit. Every ASCII character with bit 7 set
133 * is printable.
134 */
135 if ((fp = fdopen(s, "r")) != NULL)
136 while ((c = getc(fp)) != EOF) {
137 if (c == '\r') {
138 if (lastc == '\r') /* ^M^M - skip dupes */
139 continue;
140 c = '\n';
141 lastc = '\r';
142 } else {
143 if (!(eightflag || isprint(c) || isspace(c))) {
144 c &= 0x7f;
145 c |= 0x40;
146 }
147 if (lastc != '\r' || c != '\n')
148 lastc = c;
149 else {
150 lastc = '\n';
151 continue;
152 }
153 }
154 putchar(c);
155 }
156 if (lastc != '\n')
157 putchar('\n');
158 (void)fclose(fp);
159}
