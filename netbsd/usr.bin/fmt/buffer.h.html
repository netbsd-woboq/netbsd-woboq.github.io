32#include <stddef.h>
33#include <stdio.h>
34#include <stdlib.h>
35#include <wchar.h>
36#include <err.h>
37
38#define BUF_SIZE BUFSIZ
39struct buffer {
40 wchar_t *ptr;
41 wchar_t *bptr;
42 wchar_t *eptr;
43};
44
45static void
46buf_init(struct buffer *buf)
47{
48 buf->ptr = buf->bptr = calloc(BUF_SIZE, sizeof(*buf->ptr));
49 if (buf->ptr == NULL)
50 err(1, "Cannot allocate buffer");
51 buf->eptr = buf->ptr + BUF_SIZE;
52}
53
54static void
55buf_end(struct buffer *buf)
56{
57 free(buf->bptr);
58}
59
60static void
61buf_grow(struct buffer *buf, size_t minsize)
62{
63 ptrdiff_t diff;
64 size_t len = (buf->eptr - buf->bptr) +
65 (minsize > BUF_SIZE ? minsize : BUF_SIZE);
66 wchar_t *nptr = realloc(buf->bptr, len * sizeof(*buf->ptr));
67
68 if (nptr == NULL)
69 err(1, "Cannot grow buffer");
70
71 if (nptr == buf->bptr) {
72 buf->eptr = buf->bptr + len;
73 return;
74 }
75
76 diff = nptr - buf->bptr;
77 buf->bptr += diff;
78 buf->eptr = buf->bptr + len;
79 buf->ptr += diff;
80}
81
82static inline void
83buf_putc(struct buffer *buf, wchar_t c)
84{
85 if (buf->ptr >= buf->eptr)
86 buf_grow(buf, 1);
87 *buf->ptr++ = c;
88}
89
90static inline void
91buf_reset(struct buffer *buf)
92{
93 buf->ptr = buf->bptr;
94}
95
96static inline wchar_t
97buf_unputc(struct buffer *buf)
98{
99 return buf->ptr > buf->bptr ? *--buf->ptr : '\0';
100}
