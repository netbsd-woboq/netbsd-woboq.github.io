58int
59main(int argc, char **argv)
60{
61 struct passwd *pwd;
62 int ch, newline;
63 char *file, *sender, *p;
64#if MAXPATHLEN > BUFSIZ
65 char buf[MAXPATHLEN];
66#else
67 char buf[BUFSIZ];
68#endif
69
70 file = sender = NULL;
71 while ((ch = getopt(argc, argv, "f:s:")) != -1)
72 switch((char)ch) {
73 case 'f':
74 file = optarg;
75 break;
76 case 's':
77 sender = optarg;
78 for (p = sender; *p; ++p)
79 *p = tolower((unsigned char)*p);
80 break;
81 default:
82 fprintf(stderr, "usage: from [-f file] [-s sender] [user]\n");
83 exit(1);
84 }
85 argv += optind;
86
87 /*
88 * We find the mailbox by:
89 * 1 -f flag
90 * 2 user
91 * 2 MAIL environment variable
92 * 3 _PATH_MAILDIR/file
93 */
94 if (!file) {
95 if (!(file = *argv)) {
96 if (!(file = getenv("MAIL"))) {
97 if (!(pwd = getpwuid(getuid())))
98 errx(1, "no password file entry for you");
99 if ((file = getenv("LOGNAME")) != NULL ||
100 (file = getenv("USER")) != NULL) {
101 (void)snprintf(buf, sizeof(buf),
102 "%s/%s", _PATH_MAILDIR, file);
103 file = buf;
104 } else
105 (void)snprintf(file = buf, sizeof(buf),
106 "%s/%s",
107 _PATH_MAILDIR, pwd->pw_name);
108 }
109 } else {
110 (void)snprintf(buf, sizeof(buf), "%s/%s",
111 _PATH_MAILDIR, file);
112 file = buf;
113 }
114 }
115 if (!freopen(file, "r", stdin))
116 err(1, "can't read %s", file);
117
118 for (newline = 1; fgets(buf, sizeof(buf), stdin);) {
119 if (*buf == '\n') {
120 newline = 1;
121 continue;
122 }
123 if (newline && !strncmp(buf, "From ", 5) &&
124 (!sender || match(buf + 5, sender)))
125 printf("%s", buf);
126 newline = 0;
127 }
128 exit(0);
129}

131int
132match(const char *line, const char *sender)
133{
134 char ch, pch, first;
135 const char *p, *t;
136
137 for (first = *sender++;;) {
138 if (isspace((unsigned char)(ch = *line)))
139 return(0);
140 ++line;
141 ch = tolower((unsigned char)ch);
142 if (ch != first)
143 continue;
144 for (p = sender, t = line;;) {
145 if (!(pch = *p++))
146 return(1);
147 ch = tolower((unsigned char)*t++);
148 if (ch != pch)
149 break;
150 }
151 }
152 /* NOTREACHED */
153}
