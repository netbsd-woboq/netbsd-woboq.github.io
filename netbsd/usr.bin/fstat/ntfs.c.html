50int
51ntfs_filestat(struct vnode *vp, struct filestat *fsp)
52{
53 struct ntnode ntnode;
54 struct fnode fn;
55 struct ntfsmount ntm;
56
57 /* to get the ntnode, we have to go in two steps - firstly
58 * to read appropriate struct fnode and then getting the address
59 * of ntnode and reading its contents */
60 if (!KVM_READ(VTOF(vp), &fn, sizeof (fn))) {
61 dprintf("can't read fnode at %p for pid %d", VTOF(vp), Pid);
62 return 0;
63 }
64 if (!KVM_READ(FTONT(&fn), &ntnode, sizeof (ntnode))) {
65 dprintf("can't read ntnode at %p for pid %d", FTONT(&fn), Pid);
66 return 0;
67 }
68 if (!KVM_READ(ntnode.i_mp, &ntm, sizeof (ntm))) {
69 dprintf("can't read ntfsmount at %p for pid %d",
70 FTONT(&fn), Pid);
71 return 0;
72 }
73
74 fsp->fsid = ntnode.i_dev & 0xffff;
75 fsp->fileid = ntnode.i_number;
76 fsp->mode = (mode_t)ntm.ntm_mode | getftype(vp->v_type);
77 fsp->size = fn.f_size;
78 fsp->rdev = 0; /* XXX */
79 return 1;
80}
