47int
48tmpfs_filestat(struct vnode *vp, struct filestat *fsp)
49{
50 struct tmpfs_node tn;
51 struct mount mt;
52
53 if (!KVM_READ(VP_TO_TMPFS_NODE(vp), &tn, sizeof(tn))) {
54 dprintf("can't read tmpfs_node at %p for pid %d",
55 VP_TO_TMPFS_NODE(vp), Pid);
56 return 0;
57 }
58 if (!KVM_READ(vp->v_mount, &mt, sizeof(mt))) {
59 dprintf("can't read mount at %p for pid %d",
60 vp->v_mount, Pid);
61 return 0;
62 }
63
64 fsp->fsid = mt.mnt_stat.f_fsidx.__fsid_val[0];
65 fsp->fileid = tn.tn_id;
66 fsp->mode = tn.tn_mode | getftype(vp->v_type);
67 fsp->size = tn.tn_size;
68 switch (tn.tn_type) {
69 case VBLK:
70 case VCHR:
71 fsp->rdev = tn.tn_spec.tn_dev.tn_rdev;
72 break;
73 default:
74 fsp->rdev = 0;
75 break;
76 }
77
78 return 1;
79}
