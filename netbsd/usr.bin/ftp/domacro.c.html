47void
48domacro(int argc, char *argv[])
49{
50 int i, j, count = 2, loopflg = 0;
51 char *cp1, *cp2, line2[FTPBUFLEN];
52 struct cmd *c;
53 char cmdbuf[MAX_C_NAME];
54
55 if ((argc == 0 && argv != NULL) ||
56 (argc < 2 && !another(&argc, &argv, "macro name"))) {
57 UPRINTF("usage: %s macro_name [args]\n", argv[0]);
58 code = -1;
59 return;
60 }
61 for (i = 0; i < macnum; ++i) {
62 if (!strncmp(argv[1], macros[i].mac_name, 9))
63 break;
64 }
65 if (i == macnum) {
66 fprintf(ttyout, "'%s' macro not found.\n", argv[1]);
67 code = -1;
68 return;
69 }
70 (void)strlcpy(line2, line, sizeof(line2));
71 TOP:
72 cp1 = macros[i].mac_start;
73 while (cp1 != macros[i].mac_end) {
74 while (isspace((unsigned char)*cp1))
75 cp1++;
76 cp2 = line;
77 while (*cp1 != '\0') {
78 switch(*cp1) {
79 case '\\':
80 *cp2++ = *++cp1;
81 break;
82 case '$':
83 if (isdigit((unsigned char)*(cp1+1))) {
84 j = 0;
85 while (isdigit((unsigned char)*++cp1))
86 j = 10*j + *cp1 - '0';
87 cp1--;
88 if (argc - 2 >= j) {
89 (void)strlcpy(cp2, argv[j+1],
90 sizeof(line) - (cp2 - line));
91 cp2 += strlen(argv[j+1]);
92 }
93 break;
94 }
95 if (*(cp1+1) == 'i') {
96 loopflg = 1;
97 cp1++;
98 if (count < argc) {
99 (void)strlcpy(cp2, argv[count],
100 sizeof(line) - (cp2 - line));
101 cp2 += strlen(argv[count]);
102 }
103 break;
104 }
105 /* FALLTHROUGH */
106 default:
107 *cp2++ = *cp1;
108 break;
109 }
110 if (*cp1 != '\0')
111 cp1++;
112 }
113 *cp2 = '\0';
114 makeargv();
115 c = getcmd(margv[0]);
116 if (c == (struct cmd *)-1) {
117 fputs("?Ambiguous command.\n", ttyout);
118 code = -1;
119 } else if (c == 0) {
120 fputs("?Invalid command.\n", ttyout);
121 code = -1;
122 } else if (c->c_conn && !connected) {
123 fputs("Not connected.\n", ttyout);
124 code = -1;
125 } else {
126 if (verbose) {
127 fputs(line, ttyout);
128 putc('\n', ttyout);
129 }
130 (void)strlcpy(cmdbuf, c->c_name, sizeof(cmdbuf));
131 margv[0] = cmdbuf;
132 (*c->c_handler)(margc, margv);
133 if (bell && c->c_bell)
134 (void)putc('\007', ttyout);
135 (void)strlcpy(line, line2, sizeof(line));
136 makeargv();
137 argc = margc;
138 argv = margv;
139 }
140 if (cp1 != macros[i].mac_end)
141 cp1++;
142 }
143 if (loopflg && ++count < argc)
144 goto TOP;
145}
