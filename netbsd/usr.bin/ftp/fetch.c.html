141/*
142 * Determine if token is the next word in buf (case insensitive).
143 * If so, advance buf past the token and any trailing LWS, and
144 * return a pointer to the token (in buf). Otherwise, return NULL.
145 * token may be preceded by LWS.
146 * token must be followed by LWS or NUL. (I.e, don't partial match).
147 */
148static const char *
149match_token(const char **buf, const char *token)
150{
151 const char *p, *orig;
152 size_t tlen;
153
154 tlen = strlen(token);
155 p = *buf;
156 SKIPLWS(p);
157 orig = p;
158 if (strncasecmp(p, token, tlen) != 0)
159 return NULL;
160 p += tlen;
161 if (*p != '\0' && !ISLWS(*p))
162 return NULL;
163 SKIPLWS(p);
164 orig = *buf;
165 *buf = p;
166 return orig;
167}

218/*
219 * Generate authorization response based on given authentication challenge.
220 * Returns -1 if an error occurred, otherwise 0.
221 * Sets response to a malloc(3)ed string; caller should free.
222 */
223static int
224auth_url(const char *challenge, char **response, const struct authinfo *auth)
225{
226 const char *cp, *scheme, *errormsg;
227 char *ep, *clear, *realm;
228 char uuser[BUFSIZ], *gotpass;
229 const char *upass;
230 int rval;
231 size_t len, clen, rlen;
232
233 *response = NULL;
234 clear = realm = NULL;
235 rval = -1;
236 cp = challenge;
237 scheme = "Basic"; /* only support Basic authentication */
238 gotpass = NULL;
239
240 DPRINTF("auth_url: challenge `%s'\n", challenge);
241
242 if (! match_token(&cp, scheme)) {
243 warnx("Unsupported authentication challenge `%s'",
244 challenge);
245 goto cleanup_auth_url;
246 }
247
248#define REALM "realm=\""
249 if (STRNEQUAL(cp, REALM))
250 cp += sizeof(REALM) - 1;
251 else {
252 warnx("Unsupported authentication challenge `%s'",
253 challenge);
254 goto cleanup_auth_url;
255 }
256/* XXX: need to improve quoted-string parsing to support \ quoting, etc. */
257 if ((ep = strchr(cp, '\"')) != NULL) {
258 len = ep - cp;
259 realm = (char *)ftp_malloc(len + 1);
260 (void)strlcpy(realm, cp, len + 1);
261 } else {
262 warnx("Unsupported authentication challenge `%s'",
263 challenge);
264 goto cleanup_auth_url;
265 }
266
267 fprintf(ttyout, "Username for `%s': ", realm);
268 if (auth->user != NULL) {
269 (void)strlcpy(uuser, auth->user, sizeof(uuser));
270 fprintf(ttyout, "%s\n", uuser);
271 } else {
272 (void)fflush(ttyout);
273 if (get_line(stdin, uuser, sizeof(uuser), &errormsg) < 0) {
274 warnx("%s; can't authenticate", errormsg);
275 goto cleanup_auth_url;
276 }
277 }
278 if (auth->pass != NULL)
279 upass = auth->pass;
280 else {
281 gotpass = getpass("Password: ");
282 if (gotpass == NULL) {
283 warnx("Can't read password");
284 goto cleanup_auth_url;
285 }
286 upass = gotpass;
287 }
288
289 clen = strlen(uuser) + strlen(upass) + 2; /* user + ":" + pass + "\0" */
290 clear = (char *)ftp_malloc(clen);
291 (void)strlcpy(clear, uuser, clen);
292 (void)strlcat(clear, ":", clen);
293 (void)strlcat(clear, upass, clen);
294 if (gotpass)
295 memset(gotpass, 0, strlen(gotpass));
296
297 /* scheme + " " + enc + "\0" */
298 rlen = strlen(scheme) + 1 + (clen + 2) * 4 / 3 + 1;
299 *response = ftp_malloc(rlen);
300 (void)strlcpy(*response, scheme, rlen);
301 len = strlcat(*response, " ", rlen);
302 /* use `clen - 1' to not encode the trailing NUL */
303 base64_encode((unsigned char *)clear, clen - 1,
304 (unsigned char *)*response + len);
305 memset(clear, 0, clen);
306 rval = 0;
307
308 cleanup_auth_url:
309 FREEPTR(clear);
310 FREEPTR(realm);
311 return (rval);
312}

314/*
315 * Encode len bytes starting at clear using base64 encoding into encoded,
316 * which should be at least ((len + 2) * 4 / 3 + 1) in size.
317 */
318static void
319base64_encode(const unsigned char *clear, size_t len, unsigned char *encoded)
320{
321 static const unsigned char enc[] =
322 "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
323 unsigned char *cp;
324 size_t i;
325
326 cp = encoded;
327 for (i = 0; i < len; i += 3) {
328 *(cp++) = enc[((clear[i + 0] >> 2))];
329 *(cp++) = enc[((clear[i + 0] << 4) & 0x30)
330 | ((clear[i + 1] >> 4) & 0x0f)];
331 *(cp++) = enc[((clear[i + 1] << 2) & 0x3c)
332 | ((clear[i + 2] >> 6) & 0x03)];
333 *(cp++) = enc[((clear[i + 2] ) & 0x3f)];
334 }
335 *cp = '\0';
336 while (i-- > len)
337 *(--cp) = '=';
338}

341/*
342 * Decode %xx escapes in given string, `in-place'.
343 */
344static void
345url_decode(char *url)
346{
347 unsigned char *p, *q;
348
349 if (EMPTYSTRING(url))
350 return;
351 p = q = (unsigned char *)url;
352
353#define HEXTOINT(x) (x - (isdigit(x) ? '0' : (islower(x) ? 'a' : 'A') - 10))
354 while (*p) {
355 if (p[0] == '%'
356 && p[1] && isxdigit((unsigned char)p[1])
357 && p[2] && isxdigit((unsigned char)p[2])) {
358 *q++ = HEXTOINT(p[1]) * 16 + HEXTOINT(p[2]);
359 p+=3;
360 } else
361 *q++ = *p++;
362 }
363 *q = '\0';
364}

367/*
368 * Parse URL of form (per RFC 3986):
369 * <type>://[<user>[:<password>]@]<host>[:<port>][/<path>]
370 * Returns -1 if a parse error occurred, otherwise 0.
371 * It's the caller's responsibility to url_decode() the returned
372 * user, pass and path.
373 *
374 * Sets type to url_t, each of the given char ** pointers to a
375 * malloc(3)ed strings of the relevant section, and port to
376 * the number given, or ftpport if ftp://, or httpport if http://.
377 *
378 * XXX: this is not totally RFC 3986 compliant; <path> will have the
379 * leading `/' unless it's an ftp:// URL, as this makes things easier
380 * for file:// and http:// URLs. ftp:// URLs have the `/' between the
381 * host and the URL-path removed, but any additional leading slashes
382 * in the URL-path are retained (because they imply that we should
383 * later do "CWD" with a null argument).
384 *
385 * Examples:
386 * input URL output path
387 * --------- -----------
388 * "http://host" "/"
389 * "http://host/" "/"
390 * "http://host/path" "/path"
391 * "file://host/dir/file" "dir/file"
392 * "ftp://host" ""
393 * "ftp://host/" ""
394 * "ftp://host//" "/"
395 * "ftp://host/dir/file" "dir/file"
396 * "ftp://host//dir/file" "/dir/file"
397 */
398
399static int
400parse_url(const char *url, const char *desc, struct urlinfo *ui,
401 struct authinfo *auth)
402{
403 const char *origurl, *tport;
404 char *cp, *ep, *thost;
405 size_t len;
406
407 if (url == NULL || desc == NULL || ui == NULL || auth == NULL)
408 errx(1, "parse_url: invoked with NULL argument!");
409 DPRINTF("parse_url: %s `%s'\n", desc, url);
410
411 origurl = url;
412 tport = NULL;
413
414 if (STRNEQUAL(url, HTTP_URL)) {
415 url += sizeof(HTTP_URL) - 1;
416 ui->utype = HTTP_URL_T;
417 ui->portnum = HTTP_PORT;
418 tport = httpport;
419 } else if (STRNEQUAL(url, FTP_URL)) {
420 url += sizeof(FTP_URL) - 1;
421 ui->utype = FTP_URL_T;
422 ui->portnum = FTP_PORT;
423 tport = ftpport;
424 } else if (STRNEQUAL(url, FILE_URL)) {
425 url += sizeof(FILE_URL) - 1;
426 ui->utype = FILE_URL_T;
427 tport = "";
428#ifdef WITH_SSL
429 } else if (STRNEQUAL(url, HTTPS_URL)) {
430 url += sizeof(HTTPS_URL) - 1;
431 ui->utype = HTTPS_URL_T;
432 ui->portnum = HTTPS_PORT;
433 tport = httpsport;
434#endif
435 } else {
436 warnx("Invalid %s `%s'", desc, url);
437 cleanup_parse_url:
438 freeauthinfo(auth);
439 freeurlinfo(ui);
440 return (-1);
441 }
442
443 if (*url == '\0')
444 return (0);
445
446 /* find [user[:pass]@]host[:port] */
447 ep = strchr(url, '/');
448 if (ep == NULL)
449 thost = ftp_strdup(url);
450 else {
451 len = ep - url;
452 thost = (char *)ftp_malloc(len + 1);
453 (void)strlcpy(thost, url, len + 1);
454 if (ui->utype == FTP_URL_T) /* skip first / for ftp URLs */
455 ep++;
456 ui->path = ftp_strdup(ep);
457 }
458
459 cp = strchr(thost, '@'); /* look for user[:pass]@ in URLs */
460 if (cp != NULL) {
461 if (ui->utype == FTP_URL_T)
462 anonftp = 0; /* disable anonftp */
463 auth->user = thost;
464 *cp = '\0';
465 thost = ftp_strdup(cp + 1);
466 cp = strchr(auth->user, ':');
467 if (cp != NULL) {
468 *cp = '\0';
469 auth->pass = ftp_strdup(cp + 1);
470 }
471 url_decode(auth->user);
472 if (auth->pass)
473 url_decode(auth->pass);
474 }
475
476#ifdef INET6
477 /*
478 * Check if thost is an encoded IPv6 address, as per
479 * RFC 3986:
480 * `[' ipv6-address ']'
481 */
482 if (*thost == '[') {
483 cp = thost + 1;
484 if ((ep = strchr(cp, ']')) == NULL ||
485 (ep[1] != '\0' && ep[1] != ':')) {
486 warnx("Invalid address `%s' in %s `%s'",
487 thost, desc, origurl);
488 goto cleanup_parse_url;
489 }
490 len = ep - cp; /* change `[xyz]' -> `xyz' */
491 memmove(thost, thost + 1, len);
492 thost[len] = '\0';
493 if (! isipv6addr(thost)) {
494 warnx("Invalid IPv6 address `%s' in %s `%s'",
495 thost, desc, origurl);
496 goto cleanup_parse_url;
497 }
498 cp = ep + 1;
499 if (*cp == ':')
500 cp++;
501 else
502 cp = NULL;
503 } else
504#endif /* INET6 */
505 if ((cp = strchr(thost, ':')) != NULL)
506 *cp++ = '\0';
507 ui->host = thost;
508
509 /* look for [:port] */
510 if (cp != NULL) {
511 unsigned long nport;
512
513 nport = strtoul(cp, &ep, 10);
514 if (*cp == '\0' || *ep != '\0' ||
515 nport < 1 || nport > MAX_IN_PORT_T) {
516 warnx("Unknown port `%s' in %s `%s'",
517 cp, desc, origurl);
518 goto cleanup_parse_url;
519 }
520 ui->portnum = nport;
521 tport = cp;
522 }
523
524 if (tport != NULL)
525 ui->port = ftp_strdup(tport);
526 if (ui->path == NULL) {
527 const char *emptypath = "/";
528 if (ui->utype == FTP_URL_T) /* skip first / for ftp URLs */
529 emptypath++;
530 ui->path = ftp_strdup(emptypath);
531 }
532
533 DPRINTF("parse_url: user `%s' pass `%s' host %s port %s(%d) "
534 "path `%s'\n",
535 STRorNULL(auth->user), STRorNULL(auth->pass),
536 STRorNULL(ui->host), STRorNULL(ui->port),
537 ui->portnum ? ui->portnum : -1, STRorNULL(ui->path));
538
539 return (0);
540}

544static int
545ftp_socket(const struct urlinfo *ui, void **ssl)
546{
547 struct addrinfo hints, *res, *res0 = NULL;
548 int error;
549 int s;
550 const char *host = ui->host;
551 const char *port = ui->port;
552
553 if (ui->utype != HTTPS_URL_T)
554 ssl = NULL;
555
556 memset(&hints, 0, sizeof(hints));
557 hints.ai_flags = 0;
558 hints.ai_family = family;
559 hints.ai_socktype = SOCK_STREAM;
560 hints.ai_protocol = 0;
561
562 error = getaddrinfo(host, port, &hints, &res0);
563 if (error) {
564 warnx("Can't LOOKUP `%s:%s': %s", host, port,
565 (error == EAI_SYSTEM) ? strerror(errno)
566 : gai_strerror(error));
567 return -1;
568 }
569
570 if (res0->ai_canonname)
571 host = res0->ai_canonname;
572
573 s = -1;
574 if (ssl)
575 *ssl = NULL;
576 for (res = res0; res; res = res->ai_next) {
577 char hname[NI_MAXHOST], sname[NI_MAXSERV];
578
579 ai_unmapped(res);
580 if (getnameinfo(res->ai_addr, res->ai_addrlen,
581 hname, sizeof(hname), sname, sizeof(sname),
582 NI_NUMERICHOST | NI_NUMERICSERV) != 0) {
583 strlcpy(hname, "?", sizeof(hname));
584 strlcpy(sname, "?", sizeof(sname));
585 }
586
587 if (verbose && res0->ai_next) {
588#ifdef INET6
589 if(res->ai_family == AF_INET6) {
590 fprintf(ttyout, "Trying [%s]:%s ...\n",
591 hname, sname);
592 } else {
593#endif
594 fprintf(ttyout, "Trying %s:%s ...\n",
595 hname, sname);
596#ifdef INET6
597 }
598#endif
599 }
600
601 s = socket(res->ai_family, SOCK_STREAM, res->ai_protocol);
602 if (s < 0) {
603 warn(
604 "Can't create socket for connection to "
605 "`%s:%s'", hname, sname);
606 continue;
607 }
608
609 if (ftp_connect(s, res->ai_addr, res->ai_addrlen,
610 verbose || !res->ai_next) < 0) {
611 close(s);
612 s = -1;
613 continue;
614 }
615
616#ifdef WITH_SSL
617 if (ssl) {
618 if ((*ssl = fetch_start_ssl(s, host)) == NULL) {
619 close(s);
620 s = -1;
621 continue;
622 }
623 }
624#endif
625 break;
626 }
627 if (res0)
628 freeaddrinfo(res0);
629 return s;
630}

632static int
633handle_noproxy(const char *host, in_port_t portnum)
634{
635
636 char *cp, *ep, *np, *np_copy, *np_iter, *no_proxy;
637 unsigned long np_port;
638 size_t hlen, plen;
639 int isproxy = 1;
640
641 /* check URL against list of no_proxied sites */
642 no_proxy = getoptionvalue("no_proxy");
643 if (EMPTYSTRING(no_proxy))
644 return isproxy;
645
646 np_iter = np_copy = ftp_strdup(no_proxy);
647 hlen = strlen(host);
648 while ((cp = strsep(&np_iter, " ,")) != NULL) {
649 if (*cp == '\0')
650 continue;
651 if ((np = strrchr(cp, ':')) != NULL) {
652 *np++ = '\0';
653 np_port = strtoul(np, &ep, 10);
654 if (*np == '\0' || *ep != '\0')
655 continue;
656 if (np_port != portnum)
657 continue;
658 }
659 plen = strlen(cp);
660 if (hlen < plen)
661 continue;
662 if (strncasecmp(host + hlen - plen, cp, plen) == 0) {
663 isproxy = 0;
664 break;
665 }
666 }
667 FREEPTR(np_copy);
668 return isproxy;
669}

671static int
672handle_proxy(const char *url, const char *penv, struct urlinfo *ui,
673 struct authinfo *pauth)
674{
675 struct urlinfo pui;
676
677 if (isipv6addr(ui->host) && strchr(ui->host, '%') != NULL) {
678 warnx("Scoped address notation `%s' disallowed via web proxy",
679 ui->host);
680 return -1;
681 }
682
683 initurlinfo(&pui);
684 if (parse_url(penv, "proxy URL", &pui, pauth) == -1)
685 return -1;
686
687 if ((!IS_HTTP_TYPE(pui.utype) && pui.utype != FTP_URL_T) ||
688 EMPTYSTRING(pui.host) ||
689 (! EMPTYSTRING(pui.path) && strcmp(pui.path, "/") != 0)) {
690 warnx("Malformed proxy URL `%s'", penv);
691 freeurlinfo(&pui);
692 return -1;
693 }
694
695 FREEPTR(pui.path);
696 pui.path = ftp_strdup(url);
697
698 freeurlinfo(ui);
699 *ui = pui;
700
701 return 0;
702}

704static void
705print_host(FETCH *fin, const struct urlinfo *ui)
706{
707 char *h, *p;
708
709 if (strchr(ui->host, ':') == NULL) {
710 fetch_printf(fin, "Host: %s", ui->host);

712 /*
713 * strip off IPv6 scope identifier, since it is
714 * local to the node

716 h = ftp_strdup(ui->host);
717 if (isipv6addr(h) && (p = strchr(h, '%')) != NULL)
718 *p = '\0';
719
720 fetch_printf(fin, "Host: [%s]", h);
721 free(h);
722 }
723
724 if ((ui->utype == HTTP_URL_T && ui->portnum != HTTP_PORT) ||
725 (ui->utype == HTTPS_URL_T && ui->portnum != HTTPS_PORT))
726 fetch_printf(fin, ":%u", ui->portnum);
727 fetch_printf(fin, "\r\n");
728}

859static int
860getresponseline(FETCH *fin, char *buf, size_t buflen, int *len)
861{
862 const char *errormsg;
863
864 alarmtimer(quit_time ? quit_time : 60);
865 *len = fetch_getline(fin, buf, buflen, &errormsg);
866 alarmtimer(0);
867 if (*len < 0) {
868 if (*errormsg == '\n')
869 errormsg++;
870 warnx("Receiving HTTP reply: %s", errormsg);
871 return C_CLEANUP;
872 }
873 while (*len > 0 && (ISLWS(buf[*len-1])))
874 buf[--*len] = '\0';
875
876 if (*len)
877 DPRINTF("%s: received `%s'\n", __func__, buf);
878 return C_OK;

881static int
882getresponse(FETCH *fin, char **cp, size_t buflen, int *hcode)
883{
884 int len, rv;
885 char *ep, *buf = *cp;
886
887 *hcode = 0;
888 if ((rv = getresponseline(fin, buf, buflen, &len)) != C_OK)
889 return rv;
890
891 /* Determine HTTP response code */
892 *cp = strchr(buf, ' ');
893 if (*cp == NULL)
894 return C_IMPROPER;
895
896 (*cp)++;
897
898 *hcode = strtol(*cp, &ep, 10);
899 if (*ep != '\0' && !isspace((unsigned char)*ep))
900 return C_IMPROPER;
901
902 return C_OK;
903}

905static int
906parse_posinfo(const char **cp, struct posinfo *pi)
907{
908 char *ep;
909 if (!match_token(cp, "bytes"))
910 return -1;
911
912 if (**cp == '*')
913 (*cp)++;
914 else {
915 pi->rangestart = STRTOLL(*cp, &ep, 10);
916 if (pi->rangestart < 0 || *ep != '-')
917 return -1;
918 *cp = ep + 1;
919 pi->rangeend = STRTOLL(*cp, &ep, 10);
920 if (pi->rangeend < 0 || pi->rangeend < pi->rangestart)
921 return -1;
922 *cp = ep;
923 }
924 if (**cp != '/')
925 return -1;
926 (*cp)++;
927 if (**cp == '*')
928 (*cp)++;
929 else {
930 pi->entitylen = STRTOLL(*cp, &ep, 10);
931 if (pi->entitylen < 0)
932 return -1;
933 *cp = ep;
934 }
935 if (**cp != '\0')
936 return -1;
937
938#ifndef NO_DEBUG
939 if (ftp_debug) {
940 fprintf(ttyout, "parsed range as: ");
941 if (pi->rangestart == -1)
942 fprintf(ttyout, "*");
943 else
944 fprintf(ttyout, LLF "-" LLF, (LLT)pi->rangestart,
945 (LLT)pi->rangeend);
946 fprintf(ttyout, "/" LLF "\n", (LLT)pi->entitylen);
947 }
948#endif
949 return 0;
950}

994 }

998static int
999negotiate_connection(FETCH *fin, const char *url, const char *penv,
1000 struct posinfo *pi, time_t *mtime, struct authinfo *wauth,
1001 struct authinfo *pauth, volatile int *rval, volatile int *ischunked,
1002 char **auth)
1003{
1004 int len, hcode, rv;
1005 char buf[FTPBUFLEN], *ep;
1006 const char *cp, *token;
1007 char *location, *message;
1008
1009 *auth = message = location = NULL;
1010
1011 /* Read the response */
1012 ep = buf;
1013 switch (getresponse(fin, &ep, sizeof(buf), &hcode)) {
1014 case C_CLEANUP:
1015 goto cleanup_fetch_url;
1016 case C_IMPROPER:
1017 goto improper;
1018 case C_OK:
1019 message = ftp_strdup(ep);
1020 break;
1021 }
1022
1023 /* Read the rest of the header. */
1024
1025 for (;;) {
1026 if ((rv = getresponseline(fin, buf, sizeof(buf), &len)) != C_OK)
1027 goto cleanup_fetch_url;
1028 if (len == 0)
1029 break;
1030
1031 /*
1032 * Look for some headers
1033 */
1034
1035 cp = buf;
1036
1037 if (match_token(&cp, "Content-Length:")) {
1038 filesize = STRTOLL(cp, &ep, 10);
1039 if (filesize < 0 || *ep != '\0')
1040 goto improper;
1041 DPRINTF("%s: parsed len as: " LLF "\n",
1042 __func__, (LLT)filesize);
1043
1044 } else if (match_token(&cp, "Content-Range:")) {
1045 if (parse_posinfo(&cp, pi) == -1)
1046 goto improper;
1047 if (! restart_point) {
1048 warnx(
1049 "Received unexpected Content-Range header");
1050 goto cleanup_fetch_url;
1051 }
1052
1053 } else if (match_token(&cp, "Last-Modified:")) {
1054 getmtime(cp, mtime);
1055
1056 } else if (match_token(&cp, "Location:")) {
1057 location = ftp_strdup(cp);
1058 DPRINTF("%s: parsed location as `%s'\n",
1059 __func__, cp);
1060
1061 } else if (match_token(&cp, "Transfer-Encoding:")) {
1062 if (match_token(&cp, "binary")) {
1063 warnx(
1064 "Bogus transfer encoding `binary' (fetching anyway)");
1065 continue;
1066 }
1067 if (! (token = match_token(&cp, "chunked"))) {
1068 warnx(
1069 "Unsupported transfer encoding `%s'",
1070 token);
1071 goto cleanup_fetch_url;
1072 }
1073 (*ischunked)++;
1074 DPRINTF("%s: using chunked encoding\n",
1075 __func__);
1076
1077 } else if (match_token(&cp, "Proxy-Authenticate:")
1078 || match_token(&cp, "WWW-Authenticate:")) {
1079 if (! (token = match_token(&cp, "Basic"))) {
1080 DPRINTF("%s: skipping unknown auth "
1081 "scheme `%s'\n", __func__, token);
1082 continue;
1083 }
1084 FREEPTR(*auth);
1085 *auth = ftp_strdup(token);
1086 DPRINTF("%s: parsed auth as `%s'\n",
1087 __func__, cp);
1088 }
1089
1090 }
1091 /* finished parsing header */
1092
1093 switch (hcode) {
1094 case 200:
1095 break;
1096 case 206:
1097 if (! restart_point) {
1098 warnx("Not expecting partial content header");
1099 goto cleanup_fetch_url;
1100 }
1101 break;
1102 case 300:
1103 case 301:
1104 case 302:
1105 case 303:
1106 case 305:
1107 case 307:
1108 if (EMPTYSTRING(location)) {
1109 warnx(
1110 "No redirection Location provided by server");
1111 goto cleanup_fetch_url;
1112 }
1113 if (redirect_loop++ > 5) {
1114 warnx("Too many redirections requested");
1115 goto cleanup_fetch_url;
1116 }
1117 if (hcode == 305) {
1118 if (verbose)
1119 fprintf(ttyout, "Redirected via %s\n",
1120 location);
1121 *rval = fetch_url(url, location,
1122 pauth->auth, wauth->auth);
1123 } else {
1124 if (verbose)
1125 fprintf(ttyout, "Redirected to %s\n",
1126 location);
1127 *rval = go_fetch(location);
1128 }
1129 goto cleanup_fetch_url;
1130#ifndef NO_AUTH
1131 case 401:
1132 case 407:
1133 do_auth(hcode, url, penv, wauth, pauth, auth, message, rval);
1134 goto cleanup_fetch_url;
1135#endif
1136 default:
1137 if (message)
1138 warnx("Error retrieving file `%s'", message);
1139 else
1140 warnx("Unknown error retrieving file");
1141 goto cleanup_fetch_url;
1142 }
1143 rv = C_OK;
1144 goto out;
1145
1146cleanup_fetch_url:
1147 rv = C_CLEANUP;
1148 goto out;
1149improper:
1150 rv = C_IMPROPER;
1151 goto out;
1152out:
1153 FREEPTR(message);
1154 FREEPTR(location);
1155 return rv;
1156} /* end of ftp:// or http:// specific setup */

1256/*
1257 * Retrieve URL, via a proxy if necessary, using HTTP.
1258 * If proxyenv is set, use that for the proxy, otherwise try ftp_proxy or
1259 * http_proxy/https_proxy as appropriate.
1260 * Supports HTTP redirects.
1261 * Returns 1 on failure, 0 on completed xfer, -1 if ftp connection
1262 * is still open (e.g, ftp xfer with trailing /)
1263 */
1264static int
1265fetch_url(const char *url, const char *proxyenv, char *proxyauth, char *wwwauth)
1266{
1267 sigfunc volatile oldint;
1268 sigfunc volatile oldpipe;
1269 sigfunc volatile oldalrm;
1270 sigfunc volatile oldquit;
1271 int volatile s;
1272 struct stat sb;
1273 int volatile isproxy;
1274 int volatile rval, ischunked;
1275 size_t flen;
1276 static size_t bufsize;
1277 static char *xferbuf;
1278 const char *cp;
1279 char *ep;
1280 char *volatile auth;
1281 char *volatile savefile;
1282 char *volatile location;
1283 char *volatile message;
1284 char *volatile decodedpath;
1285 struct authinfo wauth, pauth;
1286 struct posinfo pi;
1287 off_t hashbytes;
1288 int (*volatile closefunc)(FILE *);
1289 FETCH *volatile fin;
1290 FILE *volatile fout;
1291 const char *volatile penv = proxyenv;
1292 struct urlinfo ui, oui;
1293 time_t mtime;
1294 void *ssl = NULL;
1295
1296 DPRINTF("%s: `%s' proxyenv `%s'\n", __func__, url, STRorNULL(penv));
1297
1298 oldquit = oldalrm = oldint = oldpipe = NULL;
1299 closefunc = NULL;
1300 fin = NULL;
1301 fout = NULL;
1302 s = -1;
1303 savefile = NULL;
1304 auth = location = message = NULL;
1305 ischunked = isproxy = 0;
1306 rval = 1;
1307
1308 initurlinfo(&ui);
1309 initurlinfo(&oui);
1310 initauthinfo(&wauth, wwwauth);
1311 initauthinfo(&pauth, proxyauth);
1312
1313 decodedpath = NULL;
1314
1315 if (sigsetjmp(httpabort, 1))
1316 goto cleanup_fetch_url;
1317
1318 if (parse_url(url, "URL", &ui, &wauth) == -1)
1319 goto cleanup_fetch_url;
1320
1321 copyurlinfo(&oui, &ui);
1322
1323 if (ui.utype == FILE_URL_T && ! EMPTYSTRING(ui.host)
1324 && strcasecmp(ui.host, "localhost") != 0) {
1325 warnx("No support for non local file URL `%s'", url);
1326 goto cleanup_fetch_url;
1327 }
1328
1329 if (EMPTYSTRING(ui.path)) {
1330 if (ui.utype == FTP_URL_T) {
1331 rval = fetch_ftp(url);
1332 goto cleanup_fetch_url;
1333 }
1334 if (!IS_HTTP_TYPE(ui.utype) || outfile == NULL) {
1335 warnx("Invalid URL (no file after host) `%s'", url);
1336 goto cleanup_fetch_url;
1337 }
1338 }
1339
1340 decodedpath = ftp_strdup(ui.path);
1341 url_decode(decodedpath);
1342
1343 if (outfile)
1344 savefile = outfile;
1345 else {
1346 cp = strrchr(decodedpath, '/'); /* find savefile */
1347 if (cp != NULL)
1348 savefile = ftp_strdup(cp + 1);
1349 else
1350 savefile = ftp_strdup(decodedpath);
1351 /*
1352 * Use the first URL we requested not the name after a
1353 * possible redirect, but careful to save it because our
1354 * "safety" check is the match to outfile.
1355 */
1356 outfile = ftp_strdup(savefile);
1357 }
1358 DPRINTF("%s: savefile `%s'\n", __func__, savefile);
1359 if (EMPTYSTRING(savefile)) {
1360 if (ui.utype == FTP_URL_T) {
1361 rval = fetch_ftp(url);
1362 goto cleanup_fetch_url;
1363 }
1364 warnx("No file after directory (you must specify an "
1365 "output file) `%s'", url);
1366 goto cleanup_fetch_url;
1367 }
1368
1369 restart_point = 0;
1370 filesize = -1;
1371 initposinfo(&pi);
1372 mtime = -1;
1373 if (restartautofetch) {
1374 if (stat(savefile, &sb) == 0)
1375 restart_point = sb.st_size;
1376 }
1377 if (ui.utype == FILE_URL_T) { /* file:// URLs */
1378 direction = "copied";
1379 fin = fetch_open(decodedpath, "r");
1380 if (fin == NULL) {
1381 warn("Can't open `%s'", decodedpath);
1382 goto cleanup_fetch_url;
1383 }
1384 if (fstat(fetch_fileno(fin), &sb) == 0) {
1385 mtime = sb.st_mtime;
1386 filesize = sb.st_size;
1387 }
1388 if (restart_point) {
1389 if (lseek(fetch_fileno(fin), restart_point, SEEK_SET) < 0) {
1390 warn("Can't seek to restart `%s'",
1391 decodedpath);
1392 goto cleanup_fetch_url;
1393 }
1394 }
1395 if (verbose) {
1396 fprintf(ttyout, "Copying %s", decodedpath);
1397 if (restart_point)
1398 fprintf(ttyout, " (restarting at " LLF ")",
1399 (LLT)restart_point);
1400 fputs("\n", ttyout);
1401 }
1402 if (0 == rcvbuf_size) {
1403 rcvbuf_size = 8 * 1024; /* XXX */
1404 }
1405 } else { /* ftp:// or http:// URLs */
1406 int hasleading;
1407
1408 if (penv == NULL) {
1409#ifdef WITH_SSL
1410 if (ui.utype == HTTPS_URL_T)
1411 penv = getoptionvalue("https_proxy");
1412#endif
1413 if (penv == NULL && IS_HTTP_TYPE(ui.utype))
1414 penv = getoptionvalue("http_proxy");
1415 else if (ui.utype == FTP_URL_T)
1416 penv = getoptionvalue("ftp_proxy");
1417 }
1418 direction = "retrieved";
1419 if (! EMPTYSTRING(penv)) { /* use proxy */
1420
1421 isproxy = handle_noproxy(ui.host, ui.portnum);
1422
1423 if (isproxy == 0 && ui.utype == FTP_URL_T) {
1424 rval = fetch_ftp(url);
1425 goto cleanup_fetch_url;
1426 }
1427
1428 if (isproxy) {
1429 if (restart_point) {
1430 warnx(
1431 "Can't restart via proxy URL `%s'",
1432 penv);
1433 goto cleanup_fetch_url;
1434 }
1435 if (handle_proxy(url, penv, &ui, &pauth) < 0)
1436 goto cleanup_fetch_url;
1437 }
1438 } /* ! EMPTYSTRING(penv) */
1439
1440 s = ftp_socket(&ui, &ssl);
1441 if (s < 0) {
1442 warnx("Can't connect to `%s:%s'", ui.host, ui.port);
1443 goto cleanup_fetch_url;
1444 }
1445
1446 oldalrm = xsignal(SIGALRM, timeouthttp);
1447 alarmtimer(quit_time ? quit_time : 60);
1448 fin = fetch_fdopen(s, "r+");
1449 fetch_set_ssl(fin, ssl);
1450 alarmtimer(0);
1451
1452 alarmtimer(quit_time ? quit_time : 60);
1453 /*
1454 * Construct and send the request.
1455 */
1456 if (verbose)
1457 fprintf(ttyout, "Requesting %s\n", url);
1458
1459 hasleading = 0;
1460#ifdef WITH_SSL
1461 if (isproxy && oui.utype == HTTPS_URL_T) {
1462 switch (connectmethod(fin, url, penv, &oui, &ui,
1463 &wauth, &pauth, __UNVOLATILE(&auth), &hasleading,
1464 &rval)) {
1465 case C_CLEANUP:
1466 goto cleanup_fetch_url;
1467 case C_IMPROPER:
1468 goto improper;
1469 case C_OK:
1470 break;
1471 default:
1472 abort();
1473 }
1474 }
1475#endif
1476
1477 hasleading = print_get(fin, hasleading, isproxy, &oui, &ui);
1478
1479 if (flushcache)
1480 print_cache(fin, isproxy);
1481
1482 print_agent(fin);
1483 hasleading = print_proxy(fin, hasleading, wauth.auth,
1484 auth ? NULL : pauth.auth);
1485 if (hasleading) {
1486 hasleading = 0;
1487 if (verbose)
1488 fputs(")\n", ttyout);
1489 }
1490
1491 fetch_printf(fin, "\r\n");
1492 if (fetch_flush(fin) == EOF) {
1493 warn("Writing HTTP request");
1494 alarmtimer(0);
1495 goto cleanup_fetch_url;
1496 }
1497 alarmtimer(0);
1498
1499 switch (negotiate_connection(fin, url, penv, &pi,
1500 &mtime, &wauth, &pauth, &rval, &ischunked,
1501 __UNVOLATILE(&auth))) {
1502 case C_OK:
1503 break;
1504 case C_CLEANUP:
1505 goto cleanup_fetch_url;
1506 case C_IMPROPER:
1507 goto improper;
1508 default:
1509 abort();
1510 }
1511 }
1512
1513 /* Open the output file. */
1514
1515 /*
1516 * Only trust filenames with special meaning if they came from
1517 * the command line
1518 */
1519 if (outfile == savefile) {
1520 if (strcmp(savefile, "-") == 0) {
1521 fout = stdout;
1522 } else if (*savefile == '|') {
1523 oldpipe = xsignal(SIGPIPE, SIG_IGN);
1524 fout = popen(savefile + 1, "w");
1525 if (fout == NULL) {
1526 warn("Can't execute `%s'", savefile + 1);
1527 goto cleanup_fetch_url;
1528 }
1529 closefunc = pclose;
1530 }
1531 }
1532 if (fout == NULL) {
1533 if ((pi.rangeend != -1 && pi.rangeend <= restart_point) ||
1534 (pi.rangestart == -1 &&
1535 filesize != -1 && filesize <= restart_point)) {
1536 /* already done */
1537 if (verbose)
1538 fprintf(ttyout, "already done\n");
1539 rval = 0;
1540 goto cleanup_fetch_url;
1541 }
1542 if (restart_point && pi.rangestart != -1) {
1543 if (pi.entitylen != -1)
1544 filesize = pi.entitylen;
1545 if (pi.rangestart != restart_point) {
1546 warnx(
1547 "Size of `%s' differs from save file `%s'",
1548 url, savefile);
1549 goto cleanup_fetch_url;
1550 }
1551 fout = fopen(savefile, "a");
1552 } else
1553 fout = fopen(savefile, "w");
1554 if (fout == NULL) {
1555 warn("Can't open `%s'", savefile);
1556 goto cleanup_fetch_url;
1557 }
1558 closefunc = fclose;
1559 }
1560
1561 /* Trap signals */
1562 oldquit = xsignal(SIGQUIT, psummary);
1563 oldint = xsignal(SIGINT, aborthttp);
1564
1565 assert(rcvbuf_size > 0);
1566 if ((size_t)rcvbuf_size > bufsize) {
1567 if (xferbuf)
1568 (void)free(xferbuf);
1569 bufsize = rcvbuf_size;
1570 xferbuf = ftp_malloc(bufsize);
1571 }
1572
1573 bytes = 0;
1574 hashbytes = mark;
1575 if (oldalrm) {
1576 (void)xsignal(SIGALRM, oldalrm);
1577 oldalrm = NULL;
1578 }
1579 progressmeter(-1);
1580
1581 /* Finally, suck down the file. */
1582 do {
1583 long chunksize;
1584 short lastchunk;
1585
1586 chunksize = 0;
1587 lastchunk = 0;
1588 /* read chunk-size */
1589 if (ischunked) {
1590 if (fetch_getln(xferbuf, bufsize, fin) == NULL) {
1591 warnx("Unexpected EOF reading chunk-size");
1592 goto cleanup_fetch_url;
1593 }
1594 errno = 0;
1595 chunksize = strtol(xferbuf, &ep, 16);
1596 if (ep == xferbuf) {
1597 warnx("Invalid chunk-size");
1598 goto cleanup_fetch_url;
1599 }
1600 if (errno == ERANGE || chunksize < 0) {
1601 errno = ERANGE;
1602 warn("Chunk-size `%.*s'",
1603 (int)(ep-xferbuf), xferbuf);
1604 goto cleanup_fetch_url;
1605 }
1606
1607 /*
1608 * XXX: Work around bug in Apache 1.3.9 and
1609 * 1.3.11, which incorrectly put trailing
1610 * space after the chunk-size.
1611 */
1612 while (*ep == ' ')
1613 ep++;
1614
1615 /* skip [ chunk-ext ] */
1616 if (*ep == ';') {
1617 while (*ep && *ep != '\r')
1618 ep++;
1619 }
1620
1621 if (strcmp(ep, "\r\n") != 0) {
1622 warnx("Unexpected data following chunk-size");
1623 goto cleanup_fetch_url;
1624 }
1625 DPRINTF("%s: got chunk-size of " LLF "\n", __func__,
1626 (LLT)chunksize);
1627 if (chunksize == 0) {
1628 lastchunk = 1;
1629 goto chunkdone;
1630 }
1631 }
1632 /* transfer file or chunk */
1633 while (1) {
1634 struct timeval then, now, td;
1635 volatile off_t bufrem;
1636
1637 if (rate_get)
1638 (void)gettimeofday(&then, NULL);
1639 bufrem = rate_get ? rate_get : (off_t)bufsize;
1640 if (ischunked)
1641 bufrem = MIN(chunksize, bufrem);
1642 while (bufrem > 0) {
1643 size_t nr = MIN((off_t)bufsize, bufrem);
1644 flen = fetch_read(xferbuf, sizeof(char),
1645 nr, fin);
1646 if (flen == 0) {
1647 if (fetch_error(fin))
1648 goto chunkerror;
1649 goto chunkdone;
1650 }
1651 bytes += flen;
1652 bufrem -= flen;
1653 if (fwrite(xferbuf, sizeof(char), flen, fout)
1654 != flen) {
1655 warn("Writing `%s'", savefile);
1656 goto cleanup_fetch_url;
1657 }
1658 if (hash && !progress) {
1659 while (bytes >= hashbytes) {
1660 (void)putc('#', ttyout);
1661 hashbytes += mark;
1662 }
1663 (void)fflush(ttyout);
1664 }
1665 if (ischunked) {
1666 chunksize -= flen;
1667 if (chunksize <= 0)
1668 break;
1669 }
1670 }
1671 if (rate_get) {
1672 while (1) {
1673 (void)gettimeofday(&now, NULL);
1674 timersub(&now, &then, &td);
1675 if (td.tv_sec > 0)
1676 break;
1677 usleep(1000000 - td.tv_usec);
1678 }
1679 }
1680 if (ischunked && chunksize <= 0)
1681 break;
1682 }
1683 /* read CRLF after chunk*/
1684 chunkdone:
1685 if (ischunked) {
1686 if (fetch_getln(xferbuf, bufsize, fin) == NULL) {
1687 alarmtimer(0);
1688 warnx("Unexpected EOF reading chunk CRLF");
1689 goto cleanup_fetch_url;
1690 }
1691 if (strcmp(xferbuf, "\r\n") != 0) {
1692 warnx("Unexpected data following chunk");
1693 goto cleanup_fetch_url;
1694 }
1695 if (lastchunk)
1696 break;
1697 }
1698 } while (ischunked);
1699
1700/* XXX: deal with optional trailer & CRLF here? */
1701chunkerror:
1702 if (hash && !progress && bytes > 0) {
1703 if (bytes < mark)
1704 (void)putc('#', ttyout);
1705 (void)putc('\n', ttyout);
1706 }
1707 if (fetch_error(fin)) {
1708 warn("Reading file");
1709 goto cleanup_fetch_url;
1710 }
1711 progressmeter(1);
1712 (void)fflush(fout);
1713 if (closefunc == fclose && mtime != -1) {
1714 struct timeval tval[2];
1715
1716 (void)gettimeofday(&tval[0], NULL);
1717 tval[1].tv_sec = mtime;
1718 tval[1].tv_usec = 0;
1719 (*closefunc)(fout);
1720 fout = NULL;
1721
1722 if (utimes(savefile, tval) == -1) {
1723 fprintf(ttyout,
1724 "Can't change modification time to %s",
1725 rfc2822time(localtime(&mtime)));
1726 }
1727 }
1728 if (bytes > 0)
1729 ptransfer(0);
1730 bytes = 0;
1731
1732 rval = 0;
1733 goto cleanup_fetch_url;
1734
1735 improper:
1736 warnx("Improper response from `%s:%s'", ui.host, ui.port);
1737
1738 cleanup_fetch_url:
1739 if (oldint)
1740 (void)xsignal(SIGINT, oldint);
1741 if (oldpipe)
1742 (void)xsignal(SIGPIPE, oldpipe);
1743 if (oldalrm)
1744 (void)xsignal(SIGALRM, oldalrm);
1745 if (oldquit)
1746 (void)xsignal(SIGQUIT, oldpipe);
1747 if (fin != NULL)
1748 fetch_close(fin);
1749 else if (s != -1)
1750 close(s);
1751 if (closefunc != NULL && fout != NULL)
1752 (*closefunc)(fout);
1753 if (savefile != outfile)
1754 FREEPTR(savefile);
1755 freeurlinfo(&ui);
1756 freeurlinfo(&oui);
1757 freeauthinfo(&wauth);
1758 freeauthinfo(&pauth);
1759 FREEPTR(decodedpath);
1760 FREEPTR(auth);
1761 FREEPTR(location);
1762 FREEPTR(message);
1763 return (rval);
1764}

1802/*
1803 * Retrieve ftp URL or classic ftp argument using FTP.
1804 * Returns 1 on failure, 0 on completed xfer, -1 if ftp connection
1805 * is still open (e.g, ftp xfer with trailing /)
1806 */
1807static int
1808fetch_ftp(const char *url)
1809{
1810 char *cp, *xargv[5], rempath[MAXPATHLEN];
1811 char *dir, *file;
1812 char cmdbuf[MAXPATHLEN];
1813 char dirbuf[4];
1814 int dirhasglob, filehasglob, rval, transtype, xargc;
1815 int oanonftp, oautologin;
1816 struct authinfo auth;
1817 struct urlinfo ui;
1818
1819 DPRINTF("fetch_ftp: `%s'\n", url);
1820 dir = file = NULL;
1821 rval = 1;
1822 transtype = TYPE_I;
1823
1824 initurlinfo(&ui);
1825 initauthinfo(&auth, NULL);
1826
1827 if (STRNEQUAL(url, FTP_URL)) {
1828 if ((parse_url(url, "URL", &ui, &auth) == -1) ||
1829 (auth.user != NULL && *auth.user == '\0') ||
1830 EMPTYSTRING(ui.host)) {
1831 warnx("Invalid URL `%s'", url);
1832 goto cleanup_fetch_ftp;
1833 }
1834 /*
1835 * Note: Don't url_decode(path) here. We need to keep the
1836 * distinction between "/" and "%2F" until later.
1837 */
1838
1839 /* check for trailing ';type=[aid]' */
1840 if (! EMPTYSTRING(ui.path) && (cp = strrchr(ui.path, ';')) != NULL) {
1841 if (strcasecmp(cp, ";type=a") == 0)
1842 transtype = TYPE_A;
1843 else if (strcasecmp(cp, ";type=i") == 0)
1844 transtype = TYPE_I;
1845 else if (strcasecmp(cp, ";type=d") == 0) {
1846 warnx(
1847 "Directory listing via a URL is not supported");
1848 goto cleanup_fetch_ftp;
1849 } else {
1850 warnx("Invalid suffix `%s' in URL `%s'", cp,
1851 url);
1852 goto cleanup_fetch_ftp;
1853 }
1854 *cp = 0;
1855 }
1856 } else { /* classic style `[user@]host:[file]' */
1857 ui.utype = CLASSIC_URL_T;
1858 ui.host = ftp_strdup(url);
1859 cp = strchr(ui.host, '@');
1860 if (cp != NULL) {
1861 *cp = '\0';
1862 auth.user = ui.host;
1863 anonftp = 0; /* disable anonftp */
1864 ui.host = ftp_strdup(cp + 1);
1865 }
1866 cp = strchr(ui.host, ':');
1867 if (cp != NULL) {
1868 *cp = '\0';
1869 ui.path = ftp_strdup(cp + 1);
1870 }
1871 }
1872 if (EMPTYSTRING(ui.host))
1873 goto cleanup_fetch_ftp;
1874
1875 /* Extract the file and (if present) directory name. */
1876 dir = ui.path;
1877 if (! EMPTYSTRING(dir)) {
1878 /*
1879 * If we are dealing with classic `[user@]host:[path]' syntax,
1880 * then a path of the form `/file' (resulting from input of the
1881 * form `host:/file') means that we should do "CWD /" before
1882 * retrieving the file. So we set dir="/" and file="file".
1883 *
1884 * But if we are dealing with URLs like `ftp://host/path' then
1885 * a path of the form `/file' (resulting from a URL of the form
1886 * `ftp://host//file') means that we should do `CWD ' (with an
1887 * empty argument) before retrieving the file. So we set
1888 * dir="" and file="file".
1889 *
1890 * If the path does not contain / at all, we set dir=NULL.
1891 * (We get a path without any slashes if we are dealing with
1892 * classic `[user@]host:[file]' or URL `ftp://host/file'.)
1893 *
1894 * In all other cases, we set dir to a string that does not
1895 * include the final '/' that separates the dir part from the
1896 * file part of the path. (This will be the empty string if
1897 * and only if we are dealing with a path of the form `/file'
1898 * resulting from an URL of the form `ftp://host//file'.)
1899 */
1900 cp = strrchr(dir, '/');
1901 if (cp == dir && ui.utype == CLASSIC_URL_T) {
1902 file = cp + 1;
1903 (void)strlcpy(dirbuf, "/", sizeof(dirbuf));
1904 dir = dirbuf;
1905 } else if (cp != NULL) {
1906 *cp++ = '\0';
1907 file = cp;
1908 } else {
1909 file = dir;
1910 dir = NULL;
1911 }
1912 } else
1913 dir = NULL;
1914 if (ui.utype == FTP_URL_T && file != NULL) {
1915 url_decode(file);
1916 /* but still don't url_decode(dir) */
1917 }
1918 DPRINTF("fetch_ftp: user `%s' pass `%s' host %s port %s "
1919 "path `%s' dir `%s' file `%s'\n",
1920 STRorNULL(auth.user), STRorNULL(auth.pass),
1921 STRorNULL(ui.host), STRorNULL(ui.port),
1922 STRorNULL(ui.path), STRorNULL(dir), STRorNULL(file));
1923
1924 dirhasglob = filehasglob = 0;
1925 if (doglob &&
1926 (ui.utype == CLASSIC_URL_T || ui.utype == FTP_URL_T)) {
1927 if (! EMPTYSTRING(dir) && strpbrk(dir, "*?[]{}") != NULL)
1928 dirhasglob = 1;
1929 if (! EMPTYSTRING(file) && strpbrk(file, "*?[]{}") != NULL)
1930 filehasglob = 1;
1931 }
1932
1933 /* Set up the connection */
1934 oanonftp = anonftp;
1935 if (connected)
1936 disconnect(0, NULL);
1937 anonftp = oanonftp;
1938 (void)strlcpy(cmdbuf, getprogname(), sizeof(cmdbuf));
1939 xargv[0] = cmdbuf;
1940 xargv[1] = ui.host;
1941 xargv[2] = NULL;
1942 xargc = 2;
1943 if (ui.port) {
1944 xargv[2] = ui.port;
1945 xargv[3] = NULL;
1946 xargc = 3;
1947 }
1948 oautologin = autologin;
1949 /* don't autologin in setpeer(), use ftp_login() below */
1950 autologin = 0;
1951 setpeer(xargc, xargv);
1952 autologin = oautologin;
1953 if ((connected == 0) ||
1954 (connected == 1 && !ftp_login(ui.host, auth.user, auth.pass))) {
1955 warnx("Can't connect or login to host `%s:%s'",
1956 ui.host, ui.port ? ui.port : "?");
1957 goto cleanup_fetch_ftp;
1958 }
1959
1960 switch (transtype) {
1961 case TYPE_A:
1962 setascii(1, xargv);
1963 break;
1964 case TYPE_I:
1965 setbinary(1, xargv);
1966 break;
1967 default:
1968 errx(1, "fetch_ftp: unknown transfer type %d", transtype);
1969 }
1970
1971 /*
1972 * Change directories, if necessary.
1973 *
1974 * Note: don't use EMPTYSTRING(dir) below, because
1975 * dir=="" means something different from dir==NULL.
1976 */
1977 if (dir != NULL && !dirhasglob) {
1978 char *nextpart;
1979
1980 /*
1981 * If we are dealing with a classic `[user@]host:[path]'
1982 * (urltype is CLASSIC_URL_T) then we have a raw directory
1983 * name (not encoded in any way) and we can change
1984 * directories in one step.
1985 *
1986 * If we are dealing with an `ftp://host/path' URL
1987 * (urltype is FTP_URL_T), then RFC 3986 says we need to
1988 * send a separate CWD command for each unescaped "/"
1989 * in the path, and we have to interpret %hex escaping
1990 * *after* we find the slashes. It's possible to get
1991 * empty components here, (from multiple adjacent
1992 * slashes in the path) and RFC 3986 says that we should
1993 * still do `CWD ' (with a null argument) in such cases.
1994 *
1995 * Many ftp servers don't support `CWD ', so if there's an
1996 * error performing that command, bail out with a descriptive
1997 * message.
1998 *
1999 * Examples:
2000 *
2001 * host: dir="", urltype=CLASSIC_URL_T
2002 * logged in (to default directory)
2003 * host:file dir=NULL, urltype=CLASSIC_URL_T
2004 * "RETR file"
2005 * host:dir/ dir="dir", urltype=CLASSIC_URL_T
2006 * "CWD dir", logged in
2007 * ftp://host/ dir="", urltype=FTP_URL_T
2008 * logged in (to default directory)
2009 * ftp://host/dir/ dir="dir", urltype=FTP_URL_T
2010 * "CWD dir", logged in
2011 * ftp://host/file dir=NULL, urltype=FTP_URL_T
2012 * "RETR file"
2013 * ftp://host//file dir="", urltype=FTP_URL_T
2014 * "CWD ", "RETR file"
2015 * host:/file dir="/", urltype=CLASSIC_URL_T
2016 * "CWD /", "RETR file"
2017 * ftp://host///file dir="/", urltype=FTP_URL_T
2018 * "CWD ", "CWD ", "RETR file"
2019 * ftp://host/%2F/file dir="%2F", urltype=FTP_URL_T
2020 * "CWD /", "RETR file"
2021 * ftp://host/foo/file dir="foo", urltype=FTP_URL_T
2022 * "CWD foo", "RETR file"
2023 * ftp://host/foo/bar/file dir="foo/bar"
2024 * "CWD foo", "CWD bar", "RETR file"
2025 * ftp://host//foo/bar/file dir="/foo/bar"
2026 * "CWD ", "CWD foo", "CWD bar", "RETR file"
2027 * ftp://host/foo//bar/file dir="foo//bar"
2028 * "CWD foo", "CWD ", "CWD bar", "RETR file"
2029 * ftp://host/%2F/foo/bar/file dir="%2F/foo/bar"
2030 * "CWD /", "CWD foo", "CWD bar", "RETR file"
2031 * ftp://host/%2Ffoo/bar/file dir="%2Ffoo/bar"
2032 * "CWD /foo", "CWD bar", "RETR file"
2033 * ftp://host/%2Ffoo%2Fbar/file dir="%2Ffoo%2Fbar"
2034 * "CWD /foo/bar", "RETR file"
2035 * ftp://host/%2Ffoo%2Fbar%2Ffile dir=NULL
2036 * "RETR /foo/bar/file"
2037 *
2038 * Note that we don't need `dir' after this point.
2039 */
2040 do {
2041 if (ui.utype == FTP_URL_T) {
2042 nextpart = strchr(dir, '/');
2043 if (nextpart) {
2044 *nextpart = '\0';
2045 nextpart++;
2046 }
2047 url_decode(dir);
2048 } else
2049 nextpart = NULL;
2050 DPRINTF("fetch_ftp: dir `%s', nextpart `%s'\n",
2051 STRorNULL(dir), STRorNULL(nextpart));
2052 if (ui.utype == FTP_URL_T || *dir != '\0') {
2053 (void)strlcpy(cmdbuf, "cd", sizeof(cmdbuf));
2054 xargv[0] = cmdbuf;
2055 xargv[1] = dir;
2056 xargv[2] = NULL;
2057 dirchange = 0;
2058 cd(2, xargv);
2059 if (! dirchange) {
2060 if (*dir == '\0' && code == 500)
2061 fprintf(stderr,
2062"\n"
2063"ftp: The `CWD ' command (without a directory), which is required by\n"
2064" RFC 3986 to support the empty directory in the URL pathname (`//'),\n"
2065" conflicts with the server's conformance to RFC 959.\n"
2066" Try the same URL without the `//' in the URL pathname.\n"
2067"\n");
2068 goto cleanup_fetch_ftp;
2069 }
2070 }
2071 dir = nextpart;
2072 } while (dir != NULL);
2073 }
2074
2075 if (EMPTYSTRING(file)) {
2076 rval = -1;
2077 goto cleanup_fetch_ftp;
2078 }
2079
2080 if (dirhasglob) {
2081 (void)strlcpy(rempath, dir, sizeof(rempath));
2082 (void)strlcat(rempath, "/", sizeof(rempath));
2083 (void)strlcat(rempath, file, sizeof(rempath));
2084 file = rempath;
2085 }
2086
2087 /* Fetch the file(s). */
2088 xargc = 2;
2089 (void)strlcpy(cmdbuf, "get", sizeof(cmdbuf));
2090 xargv[0] = cmdbuf;
2091 xargv[1] = file;
2092 xargv[2] = NULL;
2093 if (dirhasglob || filehasglob) {
2094 int ointeractive;
2095
2096 ointeractive = interactive;
2097 interactive = 0;
2098 if (restartautofetch)
2099 (void)strlcpy(cmdbuf, "mreget", sizeof(cmdbuf));
2100 else
2101 (void)strlcpy(cmdbuf, "mget", sizeof(cmdbuf));
2102 xargv[0] = cmdbuf;
2103 mget(xargc, xargv);
2104 interactive = ointeractive;
2105 } else {
2106 char *destfile = outfile;
2107 if (destfile == NULL) {
2108 cp = strrchr(file, '/'); /* find savefile */
2109 if (cp != NULL)
2110 destfile = cp + 1;
2111 else
2112 destfile = file;
2113 }
2114 xargv[2] = (char *)destfile;
2115 xargv[3] = NULL;
2116 xargc++;
2117 if (restartautofetch)
2118 reget(xargc, xargv);
2119 else
2120 get(xargc, xargv);
2121 }
2122
2123 if ((code / 100) == COMPLETE)
2124 rval = 0;
2125
2126 cleanup_fetch_ftp:
2127 freeurlinfo(&ui);
2128 freeauthinfo(&auth);
2129 return (rval);
2130}

2169 } else if (strcasecmp(url, "version") == 0) {

2178 } else {

2180 }
2181 fputs("\n", ttyout);
2182 return (0);
2183 }
2184#endif
2185
2186 /*
2187 * Check for file:// and http:// URLs.
2188 */
2189 if (STRNEQUAL(url, HTTP_URL)
2190#ifdef WITH_SSL
2191 || STRNEQUAL(url, HTTPS_URL)
2192#endif
2193 || STRNEQUAL(url, FILE_URL))
2194 return (fetch_url(url, NULL, NULL, NULL));
2195
2196 /*
2197 * If it contains "://" but does not begin with ftp://
2198 * or something that was already handled, then it's
2199 * unsupported.
2200 *
2201 * If it contains ":" but not "://" then we assume the
2202 * part before the colon is a host name, not an URL scheme,

2206 errx(1, "Unsupported URL scheme `%.*s'", (int)(p - url), url);

2208 /*
2209 * Try FTP URL-style and host:file arguments next.
2210 * If ftpproxy is set with an FTP URL, use fetch_url()
2211 * Othewise, use fetch_ftp().
2212 */

2215 return (fetch_url(url, NULL, NULL, NULL));
2216
2217 return (fetch_ftp(url));
2218}
