189static int a_flag = 0; /* list all variables */
190
191int
192main(int argc, char **argv)
193{
194 int ch;
195 const struct conf_variable *cp;
196 const char *varname, *pathname;
197 int found;
198
199 setprogname(argv[0]);
200 (void)setlocale(LC_ALL, "");
201
202 while ((ch = getopt(argc, argv, "a")) != -1) {
203 switch (ch) {
204 case 'a':
205 a_flag = 1;
206 break;
207 case '?':
208 default:
209 usage();
210 }
211 }
212 argc -= optind;
213 argv += optind;
214
215 if (!a_flag) {
216 if (argc == 0)
217 usage();
218 varname = argv[0];
219 argc--;
220 argv++;
221 } else
222 varname = NULL;
223
224 if (argc > 1)
225 usage();
226 pathname = argv[0]; /* may be NULL */
227
228 found = 0;
229 for (cp = conf_table; cp->name != NULL; cp++) {
230 if (a_flag || strcmp(varname, cp->name) == 0) {
231 /*LINTED weird expression*/
232 if ((cp->type == PATHCONF) == (pathname != NULL)) {
233 printvar(cp, pathname);
234 found = 1;
235 } else if (!a_flag)
236 errx(EXIT_FAILURE,
237 "%s: invalid variable type", cp->name);
238 }
239 }
240
241 if (!a_flag && !found)
242 errx(EXIT_FAILURE, "%s: unknown variable", varname);
243
244 (void)fflush(stdout);
245 return ferror(stdout) ? EXIT_FAILURE : EXIT_SUCCESS;
246}

248static void
249print_longvar(const char *name, long value)
250{
251 if (a_flag)
252 (void)printf("%s = %ld\n", name, value);
253 else
254 (void)printf("%ld\n", value);
255}
256
257static void
258print_strvar(const char *name, const char *sval)
259{
260 if (a_flag)
261 (void)printf("%s = %s\n", name, sval);
262 else
263 (void)printf("%s\n", sval);
264}
265
266static void
267printvar(const struct conf_variable *cp, const char *pathname)
268{
269 size_t slen;
270 char *sval;
271 long val;
272
273 switch (cp->type) {
274 case CONSTANT:
275 print_longvar(cp->name, cp->value);
276 break;
277
278 case CONFSTR:
279 errno = 0;
280 slen = confstr((int)cp->value, NULL, 0);
281 if (slen == 0) {
282 if (errno != 0)
283
284out: err(EXIT_FAILURE, "confstr(%ld)", cp->value);
285 else
286 print_strvar(cp->name, "undefined");
287 }
288
289 if ((sval = malloc(slen)) == NULL)
290 err(EXIT_FAILURE, "Can't allocate %zu bytes", slen);
291
292 errno = 0;
293 if (confstr((int)cp->value, sval, slen) == 0) {
294 if (errno != 0)
295 goto out;
296 else
297 print_strvar(cp->name, "undefined");
298 } else
299 print_strvar(cp->name, sval);
300
301 free(sval);
302 break;
303
304 case SYSCONF:
305 errno = 0;
306 if ((val = sysconf((int)cp->value)) == -1) {
307 if (errno != 0)
308 err(EXIT_FAILURE, "sysconf(%ld)", cp->value);
309 print_strvar(cp->name, "undefined");
310 } else
311 print_longvar(cp->name, val);
312 break;
313
314 case PATHCONF:
315 errno = 0;
316 if ((val = pathconf(pathname, (int)cp->value)) == -1) {
317 if (errno != 0) {
318 if (a_flag && errno == EINVAL) {
319 /* Just skip invalid variables */
320 return;
321 }
322 err(EXIT_FAILURE, "pathconf(%s, %ld)",
323 pathname, cp->value);
324 /* NOTREACHED */
325 }
326
327 print_strvar(cp->name, "undefined");
328 } else
329 print_longvar(cp->name, val);
330 break;
331 }
332}

335static void
336usage(void)
337{
338 const char *p = getprogname();
339 (void)fprintf(stderr, "Usage: %s system_var\n\t%s -a\n"
340 "\t%s path_var pathname\n\t%s -a pathname\n", p, p, p, p);
341 exit(EXIT_FAILURE);
342}
