52void
53conv_c(PR *pr, u_char *p)
54{
55 char buf[10];
56 char const *str;
57
58 switch(*p) {
59 case '\0':
60 str = "\\0";
61 goto strpr;
62 /* case '\a': */
63 case '\007':
64 if (odmode) /* od doesn't know about \a */
65 break;
66 str = "\\a";
67 goto strpr;
68 case '\b':
69 str = "\\b";
70 goto strpr;
71 case '\f':
72 str = "\\f";
73 goto strpr;
74 case '\n':
75 str = "\\n";
76 goto strpr;
77 case '\r':
78 str = "\\r";
79 goto strpr;
80 case '\t':
81 str = "\\t";
82 goto strpr;
83 case '\v':
84 if (odmode)
85 break;
86 str = "\\v";
87 goto strpr;
88 default:
89 break;
90 }
91 if (isprint(*p)) {
92 *pr->cchar = 'c';
93 (void)printf(pr->fmt, *p);
94 } else {
95 (void)sprintf(buf, "%03o", (int)*p);
96 str = buf;
97strpr: *pr->cchar = 's';
98 (void)printf(pr->fmt, str);
99 }
100}

102void
103conv_u(PR *pr, u_char *p)
104{
105 static const char *list[] = {
106 "nul", "soh", "stx", "etx", "eot", "enq", "ack", "bel",
107 "bs", "ht", "lf", "vt", "ff", "cr", "so", "si",
108 "dle", "dc1", "dc2", "dc3", "dc4", "nak", "syn", "etb",
109 "can", "em", "sub", "esc", "fs", "gs", "rs", "us",
110 };
111
112 /* od used nl, not lf */
113 if (*p <= 0x1f) {
114 *pr->cchar = 's';
115 if (odmode && *p == 0x0a)
116 (void)printf(pr->fmt, "nl");
117 else
118 (void)printf(pr->fmt, list[*p]);
119 } else if (*p == 0x7f) {
120 *pr->cchar = 's';
121 (void)printf(pr->fmt, "del");
122 } else if (odmode && *p == 0x20) { /* od replaces space with sp */
123 *pr->cchar = 's';
124 (void)printf(pr->fmt, " sp");
125 } else if (isprint(*p)) {
126 *pr->cchar = 'c';
127 (void)printf(pr->fmt, *p);
128 } else {
129 *pr->cchar = 'x';
130 (void)printf(pr->fmt, (int)*p);
131 }
132}
