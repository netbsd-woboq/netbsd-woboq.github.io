57FS *fshead; /* head of format strings */
58int blocksize; /* data block size */
59int exitval; /* final exit value */
60int length = -1; /* max bytes to read */
61static int isod = 0;
62
63int
64main(int argc, char *argv[])
65{
66 FS *tfs;
67 char *p;
68
69 setlocale(LC_ALL, "");
70
71 isod = 0;
72 p = strrchr(argv[0], 'o');
73 if (p != NULL && strcmp(p, "od") == 0)
74 isod = 1;
75
76 if (isod)
77 odsyntax(argc, &argv);
78 else
79 hexsyntax(argc, &argv);
80
81 /* figure out the data block size */
82 for (blocksize = 0, tfs = fshead; tfs; tfs = tfs->nextfs) {
83 tfs->bcnt = size(tfs);
84 if (blocksize < tfs->bcnt)
85 blocksize = tfs->bcnt;
86 }
87 /* rewrite the rules, do syntax checking */
88 for (tfs = fshead; tfs; tfs = tfs->nextfs)
89 rewrite(tfs);
90
91 stashargv(argv);
92 display();
93 exit(exitval);
94}
95
96void
97usage(void)
98{
99 const char *pname = getprogname();
100
101 (void)fprintf(stderr, "usage: %s ", pname);
102 if (isod)
103 (void)fprintf(stderr, "[-aBbcDdeFfHhIiLlOovXx] [-A base] "
104 "[-j skip] [-N length] [-t type_string] [[+]offset[.][Bb]] "
105 "[file ...]\n");
106 else
107 (void)fprintf(stderr, "[-bCcdovx] [-e format_string] [-f format_file] "
108 "[-n length] [-s skip] [file ...]\n");
109 exit(1);
110}
