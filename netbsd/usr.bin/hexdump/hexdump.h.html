34typedef struct _pr {
35 struct _pr *nextpr; /* next print unit */
36#define F_ADDRESS 0x001 /* print offset */
37#define F_BPAD 0x002 /* blank pad */
38#define F_C 0x004 /* %_c */
39#define F_CHAR 0x008 /* %c */
40#define F_DBL 0x010 /* %[EefGf] */
41#define F_INT 0x020 /* %[di] */
42#define F_P 0x040 /* %_p */
43#define F_STR 0x080 /* %s */
44#define F_U 0x100 /* %_u */
45#define F_UINT 0x200 /* %[ouXx] */
46#define F_TEXT 0x400 /* no conversions */
47 u_int flags; /* flag values */
48 int bcnt; /* byte count */
49 char *cchar; /* conversion character */
50 char *fmt; /* printf format */
51 char *nospace; /* no whitespace version */
52} PR;
53
54typedef struct _fu {
55 struct _fu *nextfu; /* next format unit */
56 struct _pr *nextpr; /* next print unit */
57#define F_IGNORE 0x01 /* %_A */
58#define F_SETREP 0x02 /* rep count set, not default */
59 u_int flags; /* flag values */
60 int reps; /* repetition count */
61 int bcnt; /* byte count */
62 char *fmt; /* format string */
63} FU;
64
65typedef struct _fs { /* format strings */
66 struct _fs *nextfs; /* linked list of format strings */
67 struct _fu *nextfu; /* linked list of format units */
68 int bcnt;
69} FS;
70
71enum _vflag { ALL, DUP, FIRST, WAIT }; /* -v values */
72
73extern int blocksize; /* data block size */
74extern int odmode; /* od compatibility */
75extern FU *endfu; /* format at end-of-data */
76extern int exitval; /* final exit value */
77extern FS *fshead; /* head of format strings list */
78extern int length; /* max bytes to read */
79extern off_t skip; /* bytes to skip */
80extern enum _vflag vflag;
