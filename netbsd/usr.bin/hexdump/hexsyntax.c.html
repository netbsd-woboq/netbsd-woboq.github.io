55off_t skip; /* bytes to skip */
56
57void
58hexsyntax(int argc, char ***argvp)
59{
60 int ch;
61 char *p, **argv;
62
63 argv = *argvp;
64 while ((ch = getopt(argc, argv, "bcCde:f:n:os:vx")) != -1)
65 switch (ch) {
66 case 'b':
67 add("\"%07.7_Ax\n\"");
68 add("\"%07.7_ax \" 16/1 \"%03o \" \"\\n\"");
69 break;
70 case 'c':
71 add("\"%07.7_Ax\n\"");
72 add("\"%07.7_ax \" 16/1 \"%3_c \" \"\\n\"");
73 break;
74 case 'C':
75 add("\"%08.8_Ax\n\"");
76 add("\"%08.8_ax \" 8/1 \"%02x \" \" \" 8/1 \"%02x \" ");
77 add("\" |\" 16/1 \"%_p\" \"|\\n\"");
78 break;
79 case 'd':
80 add("\"%07.7_Ax\n\"");
81 add("\"%07.7_ax \" 8/2 \" %05u \" \"\\n\"");
82 break;
83 case 'e':
84 add(optarg);
85 break;
86 case 'f':
87 addfile(optarg);
88 break;
89 case 'n':
90 if ((length = atoi(optarg)) < 0)
91 errx(1, "%s: bad length value", optarg);
92 break;
93 case 'o':
94 add("\"%07.7_Ax\n\"");
95 add("\"%07.7_ax \" 8/2 \" %06o \" \"\\n\"");
96 break;
97 case 's':
98 if ((skip = strtol(optarg, &p, 0)) < 0)
99 errx(1, "%s: bad skip value", optarg);
100 switch(*p) {
101 case 'b':
102 skip *= 512;
103 break;
104 case 'k':
105 skip *= 1024;
106 break;
107 case 'm':
108 skip *= 1048576;
109 break;
110 }
111 break;
112 case 'v':
113 vflag = ALL;
114 break;
115 case 'x':
116 add("\"%07.7_Ax\n\"");
117 add("\"%07.7_ax \" 8/2 \" %04x \" \"\\n\"");
118 break;
119 case '?':
120 usage();
121 }
122
123 if (!fshead) {
124 add("\"%07.7_Ax\n\"");
125 add("\"%07.7_ax \" 8/2 \"%04x \" \"\\n\"");
126 }
127
128 *argvp += optind;
129}
