63#include <sys/cdefs.h>
64
65struct emulation {
66 const char *name; /* Emulation name */
67 const char * const *sysnames; /* Array of system call names */
68 int nsysnames; /* Number of */
69 const int *errnomap; /* Array of error number mapping */
70 int nerrnomap; /* number of elements in array */
71 const int *signalmap; /* Array of signal number mapping */
72 int nsignalmap; /* number of elements in array */
73 u_int32_t flags; /* Special flags for this emulation */
74#define EMUL_FLAG_NETBSD32 0x00000001 /* this emulation is
75 * based on netbsd32 */
76};
77
78extern const struct emulation *cur_emul;
79extern const struct emulation *prev_emul;
80
