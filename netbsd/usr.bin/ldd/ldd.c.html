113static void
114usage(void)
115{
116 fprintf(stderr, "Usage: %s [-f <format 1>] [-f <format 2>] <filename>"
117 " ...\n", getprogname());
118 exit(1);
119}
120
121int
122main(int argc, char **argv)
123{
124 const char *fmt1 = NULL, *fmt2 = NULL;
125 int c, exit_status = EXIT_SUCCESS;
126
127#ifdef DEBUG
128 debug = 1;
129#endif
130 while ((c = getopt(argc, argv, "f:o")) != -1) {
131 switch (c) {
132 case 'f':
133 if (fmt1) {
134 if (fmt2)
135 errx(1, "Too many formats");
136 fmt2 = optarg;
137 } else
138 fmt1 = optarg;
139 break;
140 case 'o':
141 if (fmt1 || fmt2)
142 errx(1, "Cannot use -o and -f together");
143 fmt1 = "%a:-l%o.%m => %p\n";
144 break;
145 default:
146 usage();
147 /*NOTREACHED*/
148 }
149 }
150 argc -= optind;
151 argv += optind;
152
153 if (argc <= 0) {
154 usage();
155 /*NOTREACHED*/
156 }
157
158 for (; argc != 0; argc--, argv++) {
159 int fd;
160
161 fd = open(*argv, O_RDONLY);
162 if (fd == -1) {
163 exit_status = EXIT_FAILURE;
164 warn("%s", *argv);
165 continue;
166 }
167 if (elf_ldd(fd, *argv, fmt1, fmt2) == -1
168 /* Alpha never had 32 bit support. */
169#if (defined(_LP64) && !defined(ELF64_ONLY)) || defined(MIPS_N32)
170 && elf32_ldd(fd, *argv, fmt1, fmt2) == -1
171#if defined(__mips__) && 0 /* XXX this is still hosed for some reason */
172 && elf32_ldd_compat(fd, *argv, fmt1, fmt2) == -1
173#endif
174#endif
175 ) {
176 exit_status = EXIT_FAILURE;
177 warnx("%s", error_message);
178 }
179 close(fd);
180 }
181
182 return exit_status;
183}

185/*
186 * Error reporting function. Use it like printf. If formats the message
187 * into a buffer, and sets things up so that the next call to dlerror()
188 * will return the message.
189 */
190void
191_rtld_error(const char *fmt, ...)
192{
193 static char buf[512];
194 va_list ap;
195 va_start(ap, fmt);
196 xvsnprintf(buf, sizeof buf, fmt, ap);
197 error_message = buf;
198 va_end(ap);
199}
200
201char *
202dlerror()
203{
204 char *msg = error_message;
205 error_message = NULL;
206 return msg;
207}
208
209void
210_rtld_die(void)
211{
212 const char *msg = dlerror();
213
214 if (msg == NULL)
215 msg = "Fatal error";
216 xerrx(1, "%s", msg);
217}
