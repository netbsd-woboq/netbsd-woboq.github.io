55#define SECOND 1
56#define MINUTE (SECOND * 60)
57#define HOUR (MINUTE * 60)
58
59/*
60 * leave [[+]hhmm]
61 *
62 * Reminds you when you have to leave.
63 * Leave prompts for input and goes away if you hit return.
64 * It nags you like a mother hen.
65 */
66
67__dead static void doalarm(u_int);
68__dead static void usage(void);
69
70int
71main(int argc, char **argv)
72{
73 u_int secs;
74 int hours, minutes;
75 char c, *cp;
76 struct tm *t = NULL;
77 time_t now;
78 int plusnow;
79 char buf[50];
80
81 if (setvbuf(stdout, NULL, _IONBF, 0) != 0)
82 errx(1, "Cannot set stdout to unbuffered.");
83
84 if (argc < 2) {
85 (void)puts("When do you have to leave? ");
86 cp = fgets(buf, sizeof(buf), stdin);
87 if (cp == NULL || *cp == '\n')
88 exit(0);
89 } else
90 cp = argv[1];
91
92 if (*cp == '+') {
93 plusnow = 1;
94 ++cp;
95 } else {
96 plusnow = 0;
97 (void)time(&now);
98 t = localtime(&now);
99 }
100
101 for (hours = 0; (c = *cp) && c != '\n'; ++cp) {
102 if (!isdigit((unsigned char)c))
103 usage();
104 hours = hours * 10 + (c - '0');
105 }
106 minutes = hours % 100;
107 hours /= 100;
108
109 if (minutes < 0 || minutes > 59)
110 usage();
111 if (plusnow)
112 secs = (hours * HOUR) + (minutes * MINUTE);
113 else {
114 if (hours > 23)
115 usage();
116 if (t->tm_hour >= 12)
117 t->tm_hour -= 12;
118 if (hours >= 12)
119 hours -= 12;
120 if (t->tm_hour > hours ||
121 (t->tm_hour == hours && minutes <= t->tm_min))
122 hours += 12;
123 secs = (hours - t->tm_hour) * HOUR;
124 secs += (minutes - t->tm_min) * MINUTE;
125 }
126 doalarm(secs);
127 exit(0);
128}
129
130static void
131doalarm(u_int secs)
132{
133 int bother;
134 time_t daytime;
135
136 switch (fork()) {
137 case 0:
138 break;
139 case -1:
140 err(1, "Fork failed");
141 /*NOTREACHED*/
142 default:
143 exit(0);
144 }
145
146 (void)time(&daytime);
147 daytime += secs;
148 printf("Alarm set for %.16s. (pid %u)\n",
149 ctime(&daytime), (unsigned)getpid());
150
151 /*
152 * if write fails, we've lost the terminal through someone else
153 * causing a vhangup by logging in.
154 */
155#define FIVEMIN (5 * MINUTE)
156 if (secs >= FIVEMIN) {
157 sleep(secs - FIVEMIN);
158 if (puts("\07\07You have to leave in 5 minutes.\n") == EOF)
159 exit(0);
160 secs = FIVEMIN;
161 }
162
163#define ONEMIN (MINUTE)
164 if (secs >= ONEMIN) {
165 sleep(secs - ONEMIN);
166 if (puts("\07\07Just one more minute!\n") == EOF)
167 exit(0);
168 }
169
170 for (bother = 10; bother--;) {
171 sleep((u_int)ONEMIN);
172 if (puts("\07\07Time to leave!\n") == EOF)
173 exit(0);
174 }
175
176 (void)puts("\07\07That was the last time I'll tell you. Bye.\n");
177 exit(0);
178}

181usage(void)
182{
183 (void)fprintf(stderr, "usage: %s [[+]hhmm]\n", getprogname());
184 exit(1);
185}
