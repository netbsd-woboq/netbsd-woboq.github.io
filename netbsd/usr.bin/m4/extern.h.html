125/* speeded-up versions of chrsave/pushback */
126#define PUSHBACK(c) \
127 do { \
128 if (bp >= endpbb) \
129 enlarge_bufspace(); \
130 *bp++ = (c); \
131 } while(0)
132
133#define CHRSAVE(c) \
134 do { \
135 if (ep >= endest) \
136 enlarge_strspace(); \
137 *ep++ = (c); \
138 } while(0)
