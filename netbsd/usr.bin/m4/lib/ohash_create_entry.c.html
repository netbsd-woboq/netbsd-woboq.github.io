22/* This handles the common case of variable length keys, where the
23 * key is stored at the end of the record.
24 */
25void *
26ohash_create_entry(struct ohash_info *i, const char *start, const char **end)
27{
28 char *p;
29
30 if (!*end)
31 *end = start + strlen(start);
32 p = (i->alloc)(i->key_offset + (*end - start) + 1, i->data);
33 if (p) {
34 memcpy(p+i->key_offset, start, *end-start);
35 p[i->key_offset + (*end - start)] = '\0';
36 }
37 return (void *)p;
38}
