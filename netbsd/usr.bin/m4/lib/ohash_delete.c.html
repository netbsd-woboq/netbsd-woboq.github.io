20#include "ohash_int.h"
21/* hash_delete only frees the hash structure. Use hash_first/hash_next
22 * to free entries as well. */
23void
24ohash_delete(struct ohash *h)
25{
26 (h->info.hfree)(h->t, sizeof(struct _ohash_record) * h->size,
27 h->info.data);
28#ifndef NDEBUG
29 h->t = NULL;
30#endif
31}
