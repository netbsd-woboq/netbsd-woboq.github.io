20#include "ohash_int.h"
21
22void *
23ohash_first(struct ohash *h, unsigned int *pos)
24{
25 *pos = 0;
26 return ohash_next(h, pos);
27}
28
29void *
30ohash_next(struct ohash *h, unsigned int *pos)
31{
32 for (; *pos < h->size; (*pos)++)
33 if (h->t[*pos].p != DELETED && h->t[*pos].p != NULL)
34 return __UNCONST(h->t[(*pos)++].p);
35 return NULL;
36}
