20#include "ohash_int.h"
21
22void
23ohash_init(struct ohash *h, unsigned int size, struct ohash_info *info)
24{
25 h->size = 1UL << size;
26 if (h->size < MINSIZE)
27 h->size = MINSIZE;
28#ifdef STATS_HASH
29 STAT_HASH_CREATION++;
30 STAT_HASH_SIZE += h->size;
31#endif
32 /* Copy info so that caller may free it. */
33 h->info.key_offset = info->key_offset;
34 h->info.halloc = info->halloc;
35 h->info.hfree = info->hfree;
36 h->info.alloc = info->alloc;
37 h->info.data = info->data;
38 h->t = (h->info.halloc)(sizeof(struct _ohash_record) * h->size,
39 h->info.data);
40 h->total = h->deleted = 0;
41}
