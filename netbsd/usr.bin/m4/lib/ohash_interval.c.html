21
22uint32_t
23ohash_interval(const char *s, const char **e)
24{
25 uint32_t k;
26
27 if (!*e)
28 *e = s + strlen(s);
29 if (s == *e)
30 k = 0;
31 else
32 k = *s++;
33 while (s != *e)
34 k = ((k << 2) | (k >> 30)) ^ *s++;
35 return k;
36}
