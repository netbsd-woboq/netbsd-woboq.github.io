44/* Getname / getuserid for those with hashed passwd data base). */
45
46/*
47 * Search the passwd file for a uid. Return name on success, NULL on failure
48 */
49PUBLIC const char *
50getname(uid_t uid)
51{
52 struct passwd *pw;
53
54 if ((pw = getpwuid(uid)) == NULL)
55 return NULL;
56 return pw->pw_name;
57}
58
59/*
60 * Convert the passed name to a user id and return it. Return -1
61 * on error.
62 */
63PUBLIC int
64getuserid(char name[])
65{
66 struct passwd *pw;
67
68 if ((pw = getpwnam(name)) == NULL)
69 return -1;
70 return pw->pw_uid;
71}
