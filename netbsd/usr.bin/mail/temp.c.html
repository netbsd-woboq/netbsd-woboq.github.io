45/*
46 * Mail -- a mail program
47 *
48 * Give names to all the temporary files that we will need.
49 */
50
51PUBLIC void
52tinit(void)
53{
54 char pathbuf[MAXPATHLEN];
55 const char *cp;
56 char *p;
57
58 /*
59 * It's okay to call savestr in here because main will
60 * do a spreserve() after us.
61 */
62 if ((cp = getenv("TMPDIR")) == NULL || *cp == '\0')
63 cp = _PATH_TMP;
64
65 tmpdir = savestr(cp);
66
67 /* Remove trailing slashes. */
68 p = tmpdir + strlen(tmpdir) - 1;
69 while (p > tmpdir && *p == '/') {
70 *p = '\0';
71 p--;
72 }
73
74 if (myname != NULL) {
75 if (getuserid(myname) < 0)
76 errx(EXIT_FAILURE, "`%s' is not a user of this system", myname);
77 }
78 else {
79 if ((cp = username()) == NULL) {
80 myname = savestr("nobody");
81 if (mailmode == mm_receiving)
82 errx(EXIT_FAILURE, "who am I receiving for?");
83 } else
84 myname = savestr(cp);
85 }
86 if ((cp = getenv("HOME")) == NULL)
87 cp = ".";
88 homedir = savestr(cp);
89
90 if (getcwd(pathbuf, sizeof(pathbuf)) != NULL)
91 origdir = savestr(pathbuf);
92 else {
93 warn("getcwd");
94 origdir = savestr(".");
95 }
96
97 if (debug)
98 (void)printf("user = %s, homedir = %s, origdir = %s\n",
99 myname, homedir, origdir);
100}
