75/*-
76 * buf.h --
77 * Header for users of the buf library.
78 */
79
80#ifndef MAKE_BUF_H
81#define MAKE_BUF_H
82
83typedef char Byte;
84
85typedef struct Buffer {
86 int size; /* Current size of the buffer */
87 int count; /* Number of bytes in buffer */
88 Byte *buffer; /* The buffer itself (zero terminated) */
89} Buffer;
90
91/* If we aren't on netbsd, __predict_false() might not be defined. */
92#ifndef __predict_false
93#define __predict_false(x) (x)
94#endif
95
96/* Buf_AddByte adds a single byte to a buffer. */
97#define Buf_AddByte(bp, byte) do { \
98 int _count = ++(bp)->count; \
99 char *_ptr; \
100 if (__predict_false(_count >= (bp)->size)) \
101 Buf_Expand_1(bp); \
102 _ptr = (bp)->buffer + _count; \
103 _ptr[-1] = (byte); \
104 _ptr[0] = 0; \
105 } while (0)
106
107#define BUF_ERROR 256
108
109#define Buf_Size(bp) ((bp)->count)
110
111void Buf_Expand_1(Buffer *);
112void Buf_AddBytes(Buffer *, int, const Byte *);
113Byte *Buf_GetAll(Buffer *, int *);
114void Buf_Empty(Buffer *);
115void Buf_Init(Buffer *, int);
116Byte *Buf_Destroy(Buffer *, Boolean);
117Byte *Buf_DestroyCompact(Buffer *);
118
119#endif /* MAKE_BUF_H */
