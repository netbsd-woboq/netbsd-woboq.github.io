48/*-
49 * listDupl.c --
50 * Duplicate a list. This includes duplicating the individual
51 * elements.
52 */
53
54#include "lstInt.h"
55
56/*-
57 *-----------------------------------------------------------------------
58 * Lst_Duplicate --
59 * Duplicate an entire list. If a function to copy a void *is
60 * given, the individual client elements will be duplicated as well.
61 *
62 * Input:
63 * l the list to duplicate
64 * copyProc A function to duplicate each void *
65 *
66 * Results:
67 * The new Lst structure or NULL if failure.
68 *
69 * Side Effects:
70 * A new list is created.
71 *-----------------------------------------------------------------------
72 */
73Lst
74Lst_Duplicate(Lst l, DuplicateProc *copyProc)
75{
76 Lst nl;
77 ListNode ln;
78 List list = l;
79
80 if (!LstValid (l)) {
81 return NULL;
82 }
83
84 nl = Lst_Init(list->isCirc);
85 if (nl == NULL) {
86 return NULL;
87 }
88
89 ln = list->firstPtr;
90 while (ln != NULL) {
91 if (copyProc != NULL) {
92 if (Lst_AtEnd(nl, copyProc(ln->datum)) == FAILURE) {
93 return NULL;
94 }
95 } else if (Lst_AtEnd(nl, ln->datum) == FAILURE) {
96 return NULL;
97 }
98
99 if (list->isCirc && ln == list->lastPtr) {
100 ln = NULL;
101 } else {
102 ln = ln->nextPtr;
103 }
104 }
105
106 return (nl);
107}
