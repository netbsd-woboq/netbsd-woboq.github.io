48/*-
49 * LstForeach.c --
50 * Perform a given function on all elements of a list.
51 */
52
53#include "lstInt.h"
54
55/*-
56 *-----------------------------------------------------------------------
57 * Lst_ForEach --
58 * Apply the given function to each element of the given list. The
59 * function should return 0 if Lst_ForEach should continue and non-
60 * zero if it should abort.
61 *
62 * Results:
63 * None.
64 *
65 * Side Effects:
66 * Only those created by the passed-in function.
67 *
68 *-----------------------------------------------------------------------
69 */
70/*VARARGS2*/
71int
72Lst_ForEach(Lst l, int (*proc)(void *, void *), void *d)
73{
74 return Lst_ForEachFrom(l, Lst_First(l), proc, d);
75}
