137/*-
138 * The structure for an individual graph node. Each node has several
139 * pieces of data associated with it.
140 * 1) the name of the target it describes
141 * 2) the location of the target file in the file system.
142 * 3) the type of operator used to define its sources (qv. parse.c)
143 * 4) whether it is involved in this invocation of make
144 * 5) whether the target has been remade
145 * 6) whether any of its children has been remade
146 * 7) the number of its children that are, as yet, unmade
147 * 8) its modification time
148 * 9) the modification time of its youngest child (qv. make.c)
149 * 10) a list of nodes for which this is a source (parents)
150 * 11) a list of nodes on which this depends (children)
151 * 12) a list of nodes that depend on this, as gleaned from the
152 * transformation rules (iParents)
153 * 13) a list of ancestor nodes, which includes parents, iParents,
154 * and recursive parents of parents
155 * 14) a list of nodes of the same name created by the :: operator
156 * 15) a list of nodes that must be made (if they're made) before
157 * this node can be, but that do not enter into the datedness of
158 * this node.
159 * 16) a list of nodes that must be made (if they're made) before
160 * this node or any child of this node can be, but that do not
161 * enter into the datedness of this node.
162 * 17) a list of nodes that must be made (if they're made) after
163 * this node is, but that do not depend on this node, in the
164 * normal sense.
165 * 18) a Lst of ``local'' variables that are specific to this target
166 * and this target only (qv. var.c [$@ $< $?, etc.])
167 * 19) a Lst of strings that are commands to be given to a shell
168 * to create this target.
169 */
170typedef struct GNode {
171 char *name; /* The target's name */
172 char *uname; /* The unexpanded name of a .USE node */
173 char *path; /* The full pathname of the file */
174 int type; /* Its type (see the OP flags, below) */
175
176 int flags;
177#define REMAKE 0x1 /* this target needs to be (re)made */
178#define CHILDMADE 0x2 /* children of this target were made */
179#define FORCE 0x4 /* children don't exist, and we pretend made */
180#define DONE_WAIT 0x8 /* Set by Make_ProcessWait() */
181#define DONE_ORDER 0x10 /* Build requested by .ORDER processing */
182#define FROM_DEPEND 0x20 /* Node created from .depend */
183#define DONE_ALLSRC 0x40 /* We do it once only */
184#define CYCLE 0x1000 /* Used by MakePrintStatus */
185#define DONECYCLE 0x2000 /* Used by MakePrintStatus */
186#define INTERNAL 0x4000 /* Internal use only */
187 enum enum_made {
188 UNMADE, DEFERRED, REQUESTED, BEINGMADE,
189 MADE, UPTODATE, ERROR, ABORTED
190 } made; /* Set to reflect the state of processing
191 * on this node:
192 * UNMADE - Not examined yet
193 * DEFERRED - Examined once (building child)
194 * REQUESTED - on toBeMade list
195 * BEINGMADE - Target is already being made.
196 * Indicates a cycle in the graph.
197 * MADE - Was out-of-date and has been made
198 * UPTODATE - Was already up-to-date
199 * ERROR - An error occurred while it was being
200 * made (used only in compat mode)
201 * ABORTED - The target was aborted due to
202 * an error making an inferior (compat).
203 */
204 int unmade; /* The number of unmade children */
205
206 time_t mtime; /* Its modification time */
207 struct GNode *cmgn; /* The youngest child */
208
209 Lst iParents; /* Links to parents for which this is an
210 * implied source, if any */
211 Lst cohorts; /* Other nodes for the :: operator */
212 Lst parents; /* Nodes that depend on this one */
213 Lst children; /* Nodes on which this one depends */
214 Lst order_pred; /* .ORDER nodes we need made */
215 Lst order_succ; /* .ORDER nodes who need us */
216
217 char cohort_num[8]; /* #n for this cohort */
218 int unmade_cohorts;/* # of unmade instances on the
219 cohorts list */
220 struct GNode *centurion; /* Pointer to the first instance of a ::
221 node; only set when on a cohorts list */
222 unsigned int checked; /* Last time we tried to makle this node */
223
224 Hash_Table context; /* The local variables */
225 Lst commands; /* Creation commands */
226
227 struct _Suff *suffix; /* Suffix for the node (determined by
228 * Suff_FindDeps and opaque to everyone
229 * but the Suff module) */
230 const char *fname; /* filename where the GNode got defined */
231 int lineno; /* line number where the GNode got defined */
232} GNode;

234/*
235 * The OP_ constants are used when parsing a dependency line as a way of
236 * communicating to other parts of the program the way in which a target
237 * should be made. These constants are bitwise-OR'ed together and
238 * placed in the 'type' field of each node. Any node that has
239 * a 'type' field which satisfies the OP_NOP function was never never on
240 * the lefthand side of an operator, though it may have been on the
241 * righthand side...
242 */
243#define OP_DEPENDS 0x00000001 /* Execution of commands depends on
244 * kids (:) */
245#define OP_FORCE 0x00000002 /* Always execute commands (!) */
246#define OP_DOUBLEDEP 0x00000004 /* Execution of commands depends on kids
247 * per line (::) */
248#define OP_OPMASK (OP_DEPENDS|OP_FORCE|OP_DOUBLEDEP)
249
250#define OP_OPTIONAL 0x00000008 /* Don't care if the target doesn't
251 * exist and can't be created */
252#define OP_USE 0x00000010 /* Use associated commands for parents */
253#define OP_EXEC 0x00000020 /* Target is never out of date, but always
254 * execute commands anyway. Its time
255 * doesn't matter, so it has none...sort
256 * of */
257#define OP_IGNORE 0x00000040 /* Ignore errors when creating the node */
258#define OP_PRECIOUS 0x00000080 /* Don't remove the target when
259 * interrupted */
260#define OP_SILENT 0x00000100 /* Don't echo commands when executed */
261#define OP_MAKE 0x00000200 /* Target is a recursive make so its
262 * commands should always be executed when
263 * it is out of date, regardless of the
264 * state of the -n or -t flags */
265#define OP_JOIN 0x00000400 /* Target is out-of-date only if any of its
266 * children was out-of-date */
267#define OP_MADE 0x00000800 /* Assume the children of the node have
268 * been already made */
269#define OP_SPECIAL 0x00001000 /* Special .BEGIN, .END, .INTERRUPT */
270#define OP_USEBEFORE 0x00002000 /* Like .USE, only prepend commands */
271#define OP_INVISIBLE 0x00004000 /* The node is invisible to its parents.
272 * I.e. it doesn't show up in the parents's
273 * local variables. */
274#define OP_NOTMAIN 0x00008000 /* The node is exempt from normal 'main
275 * target' processing in parse.c */
276#define OP_PHONY 0x00010000 /* Not a file target; run always */
277#define OP_NOPATH 0x00020000 /* Don't search for file in the path */
278#define OP_WAIT 0x00040000 /* .WAIT phony node */
279#define OP_NOMETA 0x00080000 /* .NOMETA do not create a .meta file */
280#define OP_META 0x00100000 /* .META we _do_ want a .meta file */
281#define OP_NOMETA_CMP 0x00200000 /* Do not compare commands in .meta file */
282#define OP_SUBMAKE 0x00400000 /* Possibly a submake node */
283/* Attributes applied by PMake */
284#define OP_TRANSFORM 0x80000000 /* The node is a transformation rule */
285#define OP_MEMBER 0x40000000 /* Target is a member of an archive */
286#define OP_LIB 0x20000000 /* Target is a library */
287#define OP_ARCHV 0x10000000 /* Target is an archive construct */
288#define OP_HAS_COMMANDS 0x08000000 /* Target has all the commands it should.
289 * Used when parsing to catch multiple
290 * commands for a target */
291#define OP_SAVE_CMDS 0x04000000 /* Saving commands on .END (Compat) */
292#define OP_DEPS_FOUND 0x02000000 /* Already processed by Suff_FindDeps */
293#define OP_MARK 0x01000000 /* Node found while expanding .ALLSRC */
294
295#define NoExecute(gn) ((gn->type & OP_MAKE) ? noRecursiveExecute : noExecute)
296/*
297 * OP_NOP will return TRUE if the node with the given type was not the
298 * object of a dependency operator
299 */
300#define OP_NOP(t) (((t) & OP_OPMASK) == 0x00000000)
301
302#define OP_NOTARGET (OP_NOTMAIN|OP_USE|OP_EXEC|OP_TRANSFORM)

304/*
305 * The TARG_ constants are used when calling the Targ_FindNode and
306 * Targ_FindList functions in targ.c. They simply tell the functions what to
307 * do if the desired node(s) is (are) not found. If the TARG_CREATE constant
308 * is given, a new, empty node will be created for the target, placed in the
309 * table of all targets and its address returned. If TARG_NOCREATE is given,
310 * a NULL pointer will be returned.
311 */
312#define TARG_NOCREATE 0x00 /* don't create it */
313#define TARG_CREATE 0x01 /* create node if not found */
314#define TARG_NOHASH 0x02 /* don't look in/add to hash table */
315
316/*
317 * These constants are all used by the Str_Concat function to decide how the
318 * final string should look. If STR_ADDSPACE is given, a space will be
319 * placed between the two strings. If STR_ADDSLASH is given, a '/' will
320 * be used instead of a space. If neither is given, no intervening characters
321 * will be placed between the two strings in the final output. If the
322 * STR_DOFREE bit is set, the two input strings will be freed before
323 * Str_Concat returns.
324 */
325#define STR_ADDSPACE 0x01 /* add a space when Str_Concat'ing */
326#define STR_ADDSLASH 0x02 /* add a slash when Str_Concat'ing */
