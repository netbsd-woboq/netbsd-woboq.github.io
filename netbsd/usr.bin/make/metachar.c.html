45/*
46 * The following array is used to make a fast determination of which
47 * characters are interpreted specially by the shell. If a command
48 * contains any of these characters, it is executed by the shell, not
49 * directly by us.
50 *
51 * perhaps move it to ctype?
52 */
53
54unsigned char _metachar[128] = {
55// nul soh stx etx eot enq ack bel
56 1, 0, 0, 0, 0, 0, 0, 0,
57// bs ht nl vt np cr so si
58 0, 0, 1, 0, 0, 0, 0, 0,
59// dle dc1 dc2 dc3 dc4 nak syn etb
60 0, 0, 0, 0, 0, 0, 0, 0,
61// can em sub esc fs gs rs us
62 0, 0, 0, 0, 0, 0, 0, 0,
63// sp ! " # $ % & '
64 0, 1, 1, 1, 1, 0, 1, 1,
65// ( ) * + , - . /
66 1, 1, 1, 0, 0, 0, 0, 0,
67// 0 1 2 3 4 5 6 7
68 0, 0, 0, 0, 0, 0, 0, 0,
69// 8 9 : ; < = > ?
70 0, 0, 0, 1, 1, 0, 1, 1,
71// @ A B C D E F G
72 0, 0, 0, 0, 0, 0, 0, 0,
73// H I J K L M N O
74 0, 0, 0, 0, 0, 0, 0, 0,
75// P Q R S T U V W
76 0, 0, 0, 0, 0, 0, 0, 0,
77// X Y Z [ \ ] ^ _
78 0, 0, 0, 1, 1, 1, 1, 0,
79// ` a b c d e f g
80 1, 0, 0, 0, 0, 0, 0, 0,
81// h i j k l m n o
82 0, 0, 0, 0, 0, 0, 0, 0,
83// p q r s t u v w
84 0, 0, 0, 0, 0, 0, 0, 0,
85// x y z { | } ~ del
86 0, 0, 0, 1, 1, 1, 1, 0,
87};
