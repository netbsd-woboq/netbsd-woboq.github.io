31#ifndef _METACHAR_H
32#define _METACHAR_H
33
34#include <ctype.h>
35
36extern unsigned char _metachar[];
37
38#define ismeta(c) _metachar[(c) & 0x7f]
39
40static inline int
41hasmeta(const char *cmd)
42{
43 while (!ismeta(*cmd))
44 cmd++;
45
46 return *cmd != '\0';
47}
48
49static inline int
50needshell(const char *cmd, int white)
51{
52 while (!ismeta(*cmd) && *cmd != ':' && *cmd != '=') {
53 if (white && isspace((unsigned char)*cmd))
54 break;
55 cmd++;
56 }
57
58 return *cmd != '\0';
59}
60
61#endif /* _METACHAR_H */
