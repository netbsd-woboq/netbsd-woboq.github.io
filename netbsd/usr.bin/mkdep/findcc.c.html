51char *
52findcc(const char *progname)
53{
54 char *path, *dir, *next;
55 char buffer[MAXPATHLEN];
56
57 if ((next = strchr(progname, ' ')) != NULL) {
58 *next = '\0';
59 }
60
61 if (strchr(progname, '/') != NULL)
62 return access(progname, X_OK) ? NULL : strdup(progname);
63
64 if (((path = getenv("PATH")) == NULL) ||
65 ((path = strdup(path)) == NULL))
66 return NULL;
67
68 dir = path;
69 while (dir != NULL) {
70 if ((next = strchr(dir, ':')) != NULL)
71 *next++ = '\0';
72
73 if (snprintf(buffer, sizeof(buffer),
74 "%s/%s", dir, progname) < (int)sizeof(buffer)) {
75 if (!access(buffer, X_OK)) {
76 free(path);
77 return strdup(buffer);
78 }
79 }
80 dir = next;
81 }
82
83 free(path);
84 return NULL;
85}
