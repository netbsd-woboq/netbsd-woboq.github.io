52/*
53 * mkstr - create a string error message file by massaging C source
54 *
55 * Bill Joy UCB August 1977
56 *
57 * Modified March 1978 to hash old messages to be able to recompile
58 * without addding messages to the message file (usually)
59 *
60 * Based on an earlier program conceived by Bill Joy and Chuck Haley
61 *
62 * Program to create a string error message file
63 * from a group of C programs. Arguments are the name
64 * of the file where the strings are to be placed, the
65 * prefix of the new files where the processed source text
66 * is to be placed, and the files to be processed.
67 *
68 * The program looks for 'error("' in the source stream.
69 * Whenever it finds this, the following characters from the '"'
70 * to a '"' are replaced by 'seekpt' where seekpt is a
71 * pointer into the error message file.
72 * If the '(' is not immediately followed by a '"' no change occurs.
73 *
74 * The optional '-' causes strings to be added at the end of the
75 * existing error message file for recompilation of single routines.
76 */

92int
93main(int argc, char *argv[])
94{
95 char addon = 0;
96
97 argc--, progname = *argv++;
98 if (argc > 1 && argv[0][0] == '-')
99 addon++, argc--, argv++;
100 if (argc < 3)
101 fprintf(stderr, usagestr, progname), exit(1);
102 mesgwrite = fopen(argv[0], addon ? "a" : "w");
103 if (mesgwrite == NULL)
104 perror(argv[0]), exit(1);
105 mesgread = fopen(argv[0], "r");
106 if (mesgread == NULL)
107 perror(argv[0]), exit(1);
108 inithash();
109 argc--, argv++;
110 strlcpy(name, argv[0], sizeof(name));
111 np = name + strlen(name);
112 argc--, argv++;
113 do {
114 strlcpy(np, argv[0], sizeof(name) - (np - name));
115 if (freopen(name, "w", stdout) == NULL)
116 perror(name), exit(1);
117 if (freopen(argv[0], "r", stdin) == NULL)
118 perror(argv[0]), exit(1);
119 process();
120 argc--, argv++;
121 } while (argc > 0);
122 exit(0);
123}

125void
126process(void)
127{
128 int c;
129
130 for (;;) {
131 c = getchar();
132 if (c == EOF)
133 return;
134 if (c != 'e') {
135 putchar(c);
136 continue;
137 }
138 if (match("error(")) {
139 printf("error(");
140 c = getchar();
141 if (c != '"')
142 putchar(c);
143 else
144 copystr();
145 }
146 }
147}

149int
150match(const char *ocp)
151{
152 const char *cp;
153 int c;
154
155 for (cp = ocp + 1; *cp; cp++) {
156 c = getchar();
157 if (c != *cp) {
158 while (ocp < cp)
159 putchar(*ocp++);
160 ungetchar(c);
161 return (0);
162 }
163 }
164 return (1);
165}

167void
168copystr(void)
169{
170 int c, ch;
171 char buf[512];
172 char *cp = buf;
173
174 for (;;) {
175 c = getchar();
176 if (c == EOF)
177 break;
178 switch (c) {
179
180 case '"':
181 *cp++ = 0;
182 goto out;
183 case '\\':
184 c = getchar();
185 switch (c) {
186
187 case 'b':
188 c = '\b';
189 break;
190 case 't':
191 c = '\t';
192 break;
193 case 'r':
194 c = '\r';
195 break;
196 case 'n':
197 c = '\n';
198 break;
199 case '\n':
200 continue;
201 case 'f':
202 c = '\f';
203 break;
204 case '0':
205 c = 0;
206 break;
207 case '\\':
208 break;
209 default:
210 if (!octdigit(c))
211 break;
212 c -= '0';
213 ch = getchar();
214 if (!octdigit(ch))
215 break;
216 c <<= 7, c += ch - '0';
217 ch = getchar();
218 if (!octdigit(ch))
219 break;
220 c <<= 3, c+= ch - '0', ch = -1;
221 break;
222 }
223 }
224 *cp++ = c;
225 }
226out:
227 *cp = 0;
228 printf("%ld", hashit(buf, 1, 0));
229}

231int
232octdigit(char c)
233{
234
235 return (c >= '0' && c <= '7');
236}

238void
239inithash(void)
240{
241 char buf[512];
242 long mesgpt = 0;
243
244 rewind(mesgread);
245 while (fgetNUL(buf, sizeof buf, mesgread) != 0) {
246 hashit(buf, 0, mesgpt);
247 mesgpt += strlen(buf) + 2;
248 }
249}

251#define NBUCKETS 511
252
253struct hash {
254 long hval;
255 long hpt;
256 struct hash *hnext;
257} *bucket[NBUCKETS];
258
259long
260hashit(const char *str, char really, long fakept)
261{
262 int i;
263 struct hash *hp;
264 char buf[512];
265 long hashval = 0;
266 const char *cp;
267
268#ifdef __GNUC__
269 hp = NULL; /* XXX gcc */
270#endif
271 if (really)
272 fflush(mesgwrite);
273 for (cp = str; *cp;)
274 hashval = (hashval << 1) + *cp++;
275 i = hashval % NBUCKETS;
276 if (i < 0)
277 i += NBUCKETS;
278 if (really != 0)
279 for (hp = bucket[i]; hp != 0; hp = hp->hnext)
280 if (hp->hval == hashval) {
281 fseek(mesgread, hp->hpt, 0);
282 fgetNUL(buf, sizeof buf, mesgread);
283/*
284 fprintf(stderr, "Got (from %ld) %s\n", hp->hpt, buf);
285*/
286 if (strcmp(buf, str) == 0)
287 break;
288 }
289 if (!really || hp == 0) {
290 hp = (struct hash *) calloc(1, sizeof *hp);
291 hp->hnext = bucket[i];
292 hp->hval = hashval;
293 hp->hpt = really ? ftell(mesgwrite) : fakept;
294 if (really) {
295 fwrite(str, sizeof (char), strlen(str) + 1, mesgwrite);
296 fwrite("\n", sizeof (char), 1, mesgwrite);
297 }
298 bucket[i] = hp;
299 }
300/*
301 fprintf(stderr, "%s hashed to %ld at %ld\n", str, hp->hval, hp->hpt);
302*/
303 return (hp->hpt);
304}

309int
310fgetNUL(char *obuf, int rmdr, FILE *file)
311{
312 int c;
313 char *buf = obuf;
314
315 while (--rmdr > 0 && (c = getc(file)) != 0 && c != EOF)
316 *buf++ = c;
317 *buf++ = 0;
318 getc(file);
319 return ((feof(file) || ferror(file)) ? 0 : 1);
320}
