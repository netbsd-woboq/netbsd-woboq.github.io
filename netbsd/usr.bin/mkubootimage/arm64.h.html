28#ifndef _HAVE_ARM64_H
29#define _HAVE_ARM64_H
30
31/*
32 * AArch64 Linux kernel image header, as specified in
33 * https://www.kernel.org/doc/Documentation/arm64/booting.txt
34 */
35
36/* 64-byte kernel image header */
37struct arm64_image_header {
38 uint32_t code0; /* Executable code */
39 uint32_t code1; /* Executable code */
40 uint64_t text_offset; /* Image load offset */
41 uint64_t image_size; /* Effective image size */
42 uint64_t flags; /* kernel flags */
43 uint64_t res2; /* reserved */
44 uint64_t res3; /* reserved */
45 uint64_t res4; /* reserved */
46 uint32_t magic; /* Magic number ("ARM\x64") */
47 uint32_t res5; /* reserved (used for PE COFF offset) */
48};
49
50/* Kernel flags */
51#define ARM64_FLAGS_ENDIAN_BE (1 << 0)
52#define ARM64_FLAGS_PAGE_SIZE (3 << 1)
53#define ARM64_FLAGS_PAGE_SIZE_UNSPEC (0 << 1)
54#define ARM64_FLAGS_PAGE_SIZE_4K (1 << 1)
55#define ARM64_FLAGS_PAGE_SIZE_16K (2 << 1)
56#define ARM64_FLAGS_PAGE_SIZE_64K (3 << 1)
57#define ARM64_FLAGS_PHYS_PLACEMENT (1 << 3)
58#define ARM64_FLAGS_PHYS_PLACEMENT_DRAM_BASE (0 << 3)
59#define ARM64_FLAGS_PHYS_PLACEMENT_ANY (1 << 3)
60
61/* Magic */
62#define ARM64_MAGIC 0x644d5241
63
64/* Executable code. Program relative branch forward 64 bytes. */
65#define ARM64_CODE0 0x14000010
66
67#endif /* !_HAVE_ARM64_H */
