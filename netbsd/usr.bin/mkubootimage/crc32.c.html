91uint32_t
92crc32v(const struct iovec *iov, int cnt)
93{
94 const uint8_t *p;
95 uint32_t crc;
96 int i, len;
97
98 crc = ~0U;
99
100 for (i = 0; i < cnt; i++) {
101 p = iov[i].iov_base;
102 len = iov[i].iov_len;
103 while (len--)
104 crc = crc32_tab[(crc ^ *p++) & 0xFF] ^ (crc >> 8);
105 }
106
107 return crc ^ ~0U;
108}

110uint32_t
111crc32(const void *buf, size_t size)
112{
113 struct iovec iov[1];
114
115 iov[0].iov_base = __UNCONST(buf);
116 iov[0].iov_len = size;
117
118 return crc32v(iov, 1);
119}
