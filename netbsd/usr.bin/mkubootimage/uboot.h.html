28#ifndef _HAVE_UBOOT_H
29#define _HAVE_UBOOT_H
30
31enum uboot_image_os {
32 IH_OS_UNKNOWN = 0,
33 IH_OS_OPENBSD = 1,
34 IH_OS_NETBSD = 2,
35 IH_OS_FREEBSD = 3,
36 IH_OS_LINUX = 5
37};
38
39enum uboot_image_arch {
40 IH_ARCH_UNKNOWN = 0,
41 IH_ARCH_ARM = 2,
42 IH_ARCH_I386 = 3,
43 IH_ARCH_MIPS = 5,
44 IH_ARCH_MIPS64 = 6,
45 IH_ARCH_PPC = 7,
46 IH_ARCH_SH = 9,
47 IH_ARCH_OPENRISC = 21,
48 IH_ARCH_ARM64 = 22
49};
50
51enum uboot_image_type {
52 IH_TYPE_UNKNOWN = 0,
53 IH_TYPE_STANDALONE = 1,
54 IH_TYPE_KERNEL = 2,
55 IH_TYPE_RAMDISK = 3,
56 IH_TYPE_SCRIPT = 6,
57 IH_TYPE_FILESYSTEM = 7,
58 IH_TYPE_KERNEL_NOLOAD = 14,
59};
60
61enum uboot_image_comp {
62 IH_COMP_NONE = 0,
63 IH_COMP_GZIP = 1,
64 IH_COMP_BZIP2 = 2,
65 IH_COMP_LZMA = 3,
66 IH_COMP_LZO = 4,
67};
68
69#define IH_MAGIC 0x27051956
70#define IH_NMLEN 32
71
72struct uboot_image_header {
73 uint32_t ih_magic;
74 uint32_t ih_hcrc;
75 uint32_t ih_time;
76 uint32_t ih_size;
77 uint32_t ih_load;
78 uint32_t ih_ep;
79 uint32_t ih_dcrc;
80 uint8_t ih_os;
81 uint8_t ih_arch;
82 uint8_t ih_type;
83 uint8_t ih_comp;
84 uint8_t ih_name[IH_NMLEN];
85};
86
87#endif /* !_HAVE_UBOOT_H */
