29#ifndef _QFILE_H_
30#define _QFILE_H_
31
32#include <stdio.h>
33#include <stdlib.h>
34#include <openssl/bn.h>
35
36/* need line long enough for largest moduli plus headers */
37#define QLINESIZE (100+8192)
38
39/*
40 * Type: decimal. Specifies the internal structure of the prime modulus.
41 */
42#define QTYPE_UNKNOWN (0)
43#define QTYPE_UNSTRUCTURED (1)
44#define QTYPE_SAFE (2)
45#define QTYPE_SCHNOOR (3)
46#define QTYPE_SOPHIE_GERMAINE (4)
47#define QTYPE_STRONG (5)
48
49/*
50 * Tests: decimal (bit field). Specifies the methods used in checking for
51 * primality. Usually, more than one test is used.
52 */
53#define QTEST_UNTESTED (0x00)
54#define QTEST_COMPOSITE (0x01)
55#define QTEST_SIEVE (0x02)
56#define QTEST_MILLER_RABIN (0x04)
57#define QTEST_JACOBI (0x08)
58#define QTEST_ELLIPTIC (0x10)
59
60/*
61 * Size: decimal. Specifies the number of the most significant bit (0 to M). *
62 * WARNING: internally, usually 1 to N.
63 */
64#define QSIZE_MINIMUM (511)
65
66
67int
68qfileout(FILE *, uint32_t, uint32_t, uint32_t, uint32_t, uint32_t,
69 BIGNUM *);
70
71#endif /* !_QFILE_H_ */
