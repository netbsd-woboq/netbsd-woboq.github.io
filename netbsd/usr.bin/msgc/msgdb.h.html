38#ifndef MSGDB_H
39#define MSGDB_H
40
41/* Definition for a list of messages. */
42
43typedef struct id_rec {
44 char *id;
45
46 /* For the balanced tree. */
47 short balance;
48 struct id_rec *left, *right;
49
50 /* For the list of messages. */
51 struct id_rec *next;
52 char *msg;
53 int msg_no;
54} id_rec;

56#endif
