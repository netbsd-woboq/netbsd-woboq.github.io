58int
59main (int argc, char **argv)
60{
61 int ch;
62
63 prog_name = argv[0];
64
65 /* Process the arguments. */
66 while ( (ch = getopt (argc, argv, "o:")) != -1 ) {
67 switch (ch) {
68 case 'o': /* output file name */
69 out_name = optarg;
70 break;
71 default:
72 usage (prog_name);
73 }
74 }
75
76 if (optind != argc-1)
77 usage (prog_name);
78
79 src_name = argv[optind];
80
81 yyin = fopen (src_name, "r");
82 if (yyin == NULL) {
83 (void) fprintf (stderr, "%s: could not open %s.\n",
84 prog_name, src_name);
85 exit (1);
86 }
87
88 /* Do the parse */
89 (void) yyparse ();
90
91 if (had_errors)
92 return 1;
93
94 write_msg_file();
95
96 return 0;
97}

104 exit (1);
