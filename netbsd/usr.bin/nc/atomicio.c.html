31#include <errno.h>
32#include <poll.h>
33#include <unistd.h>

35#include "atomicio.h"

37/*
38 * ensure all of data on socket comes through. f==read || f==vwrite
39 */
40size_t
41atomicio(ssize_t (*f) (int, void *, size_t), int fd, void *_s, size_t n)
42{
43 char *s = _s;
44 size_t pos = 0;
45 ssize_t res;
46 struct pollfd pfd;
47
48 pfd.fd = fd;
49 pfd.events = f == read ? POLLIN : POLLOUT;
50 while (n > pos) {
51 res = (f) (fd, s + pos, n - pos);
52 switch (res) {
53 case -1:
54 if (errno == EINTR)
55 continue;
56 if ((errno == EAGAIN) || (errno == ENOBUFS)) {
57 (void)poll(&pfd, 1, -1);
58 continue;
59 }
60 return 0;
61 case 0:
62 errno = EPIPE;
63 return pos;
64 default:
65 pos += (size_t)res;
66 }
67 }
68 return (pos);
69}
