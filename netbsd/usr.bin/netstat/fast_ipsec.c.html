66/*
67 * Table-driven mapping from SADB algorithm codes to string names.
68 */
69struct alg {
70 int a;
71 const char *name;
72};
73
74static const char *ahalgs[] = { AH_ALG_STR };
75static const char *espalgs[] = { ESP_ALG_STR };
76static const char *ipcompalgs[] = { IPCOMP_ALG_STR };
77
78static const char *
79algname(size_t a, const char *algs[], size_t nalgs)
80{
81 static char buf[80];
82
83 if (a < nalgs)
84 return algs[a];
85 snprintf(buf, sizeof(buf), "alg#%zu", a);
86 return buf;
87}

89/*
90 * Print the fast_ipsec statistics.
91 * Since NetBSD's netstat(1) seems not to find us for "netstat -s",
92 * but does(?) find KAME, be prepared to be called explicitly from
93 * netstat's main program for "netstat -s"; but silently do nothing
94 * if that happens when we are running on KAME IPsec.
95 */
96void
97fast_ipsec_stats(u_long off, const char *name)
98{
99 uint64_t ipsecstats[IPSEC_NSTATS];
100 uint64_t ahstats[AH_NSTATS];
101 uint64_t espstats[ESP_NSTATS];
102 uint64_t ipcs[IPCOMP_NSTATS];
103 uint64_t ipips[IPIP_NSTATS];
104 int status;
105 size_t slen, i;
106
107 if (! use_sysctl) {
108 warnx("IPsec stats not available via KVM.");
109 return;
110 }
111
112 memset(ipsecstats, 0, sizeof(ipsecstats));
113 memset(ahstats, 0, sizeof(ahstats));
114 memset(espstats, 0, sizeof(espstats));
115 memset(ipcs, 0, sizeof(ipcs));
116 memset(ipips, 0, sizeof(ipips));
117
118 slen = sizeof(ipsecstats);
119 status = sysctlbyname("net.inet.ipsec.ipsecstats", ipsecstats, &slen,
120 NULL, 0);
121 if (status < 0) {
122 if (errno == ENOENT)
123 return;
124 if (errno != ENOMEM)
125 err(1, "net.inet.ipsec.ipsecstats");
126 }
127
128 slen = sizeof (ahstats);
129 status = sysctlbyname("net.inet.ah.ah_stats", ahstats, &slen, NULL, 0);
130 if (status < 0 && errno != ENOMEM)
131 err(1, "net.inet.ah.ah_stats");
132
133 slen = sizeof (espstats);
134 status = sysctlbyname("net.inet.esp.esp_stats", espstats, &slen, NULL, 0);
135 if (status < 0 && errno != ENOMEM)
136 err(1, "net.inet.esp.esp_stats");
137
138 slen = sizeof(ipcs);
139 status = sysctlbyname("net.inet.ipcomp.ipcomp_stats", ipcs, &slen, NULL, 0);
140 if (status < 0 && errno != ENOMEM)
141 err(1, "net.inet.ipcomp.ipcomp_stats");
142
143 slen = sizeof(ipips);
144 status = sysctlbyname("net.inet.ipip.ipip_stats", ipips, &slen, NULL, 0);
145 if (status < 0 && errno != ENOMEM)
146 err(1, "net.inet.ipip.ipip_stats");
147
148 printf("%s:\n", name);
149
150#define STAT(x,fmt) if ((x) || sflag <= 1) printf("\t%"PRIu64" " fmt "\n", x)
151 if (ipsecstats[IPSEC_STAT_IN_POLVIO]+ipsecstats[IPSEC_STAT_OUT_POLVIO])
152 printf("\t%"PRIu64" policy violations: %"PRIu64" input %"PRIu64" output\n",
153 ipsecstats[IPSEC_STAT_IN_POLVIO] + ipsecstats[IPSEC_STAT_OUT_POLVIO],
154 ipsecstats[IPSEC_STAT_IN_POLVIO], ipsecstats[IPSEC_STAT_OUT_POLVIO]);
155 STAT(ipsecstats[IPSEC_STAT_OUT_NOSA], "no SA found (output)");
156 STAT(ipsecstats[IPSEC_STAT_OUT_NOMEM], "no memory available (output)");
157 STAT(ipsecstats[IPSEC_STAT_OUT_NOROUTE], "no route available (output)");
158 STAT(ipsecstats[IPSEC_STAT_OUT_INVAL], "generic errors (output)");
159 STAT(ipsecstats[IPSEC_STAT_OUT_BUNDLESA], "bundled SA processed (output)");
160 STAT(ipsecstats[IPSEC_STAT_SPDCACHELOOKUP], "SPD cache lookups");
161 STAT(ipsecstats[IPSEC_STAT_SPDCACHEMISS], "SPD cache misses");
162#undef STAT
163
164 printf("\tah:\n");
165#define AHSTAT(x,fmt) if ((x) || sflag <= 1) printf("\t\t%"PRIu64" ah " fmt "\n", x)
166 AHSTAT(ahstats[AH_STAT_INPUT], "input packets processed");
167 AHSTAT(ahstats[AH_STAT_OUTPUT], "output packets processed");
168 AHSTAT(ahstats[AH_STAT_HDROPS], "headers too short");
169 AHSTAT(ahstats[AH_STAT_NOPF], "headers for unsupported address family");
170 AHSTAT(ahstats[AH_STAT_NOTDB], "packets with no SA");
171 AHSTAT(ahstats[AH_STAT_BADKCR], "packets dropped by crypto returning NULL mbuf");
172 AHSTAT(ahstats[AH_STAT_BADAUTH], "packets with bad authentication");
173 AHSTAT(ahstats[AH_STAT_NOXFORM], "packets with no xform");
174 AHSTAT(ahstats[AH_STAT_QFULL], "packets dropped due to queue full");
175 AHSTAT(ahstats[AH_STAT_WRAP], "packets dropped for replay counter wrap");
176 AHSTAT(ahstats[AH_STAT_REPLAY], "packets dropped for possible replay");
177 AHSTAT(ahstats[AH_STAT_BADAUTHL],"packets dropped for bad authenticator length");
178 AHSTAT(ahstats[AH_STAT_INVALID], "packets with an invalid SA");
179 AHSTAT(ahstats[AH_STAT_TOOBIG], "packets too big");
180 AHSTAT(ahstats[AH_STAT_PDROPS], "packets blocked due to policy");
181 AHSTAT(ahstats[AH_STAT_CRYPTO], "failed crypto requests");
182 AHSTAT(ahstats[AH_STAT_TUNNEL], "tunnel sanity check failures");
183
184 printf("\tah histogram:\n");
185 for (i = 0; i < AH_ALG_MAX; i++)
186 if (ahstats[AH_STAT_HIST + i])
187 printf("\t\tah packets with %s: %"PRIu64"\n"
188 , algname(i, ahalgs, __arraycount(ahalgs))
189 , ahstats[AH_STAT_HIST + i]
190 );
191 AHSTAT(ahstats[AH_STAT_IBYTES], "bytes received");
192 AHSTAT(ahstats[AH_STAT_OBYTES], "bytes transmitted");
193#undef AHSTAT
194
195 printf("\tesp:\n");
196#define ESPSTAT(x,fmt) if ((x) || sflag <= 1) printf("\t\t%"PRIu64" esp " fmt "\n", x)
197 ESPSTAT(espstats[ESP_STAT_INPUT],"input packets processed");
198 ESPSTAT(espstats[ESP_STAT_OUTPUT],"output packets processed");
199 ESPSTAT(espstats[ESP_STAT_HDROPS],"headers too short");
200 ESPSTAT(espstats[ESP_STAT_NOPF], "headers for unsupported address family");
201 ESPSTAT(espstats[ESP_STAT_NOTDB],"packets with no SA");
202 ESPSTAT(espstats[ESP_STAT_BADKCR],"packets dropped by crypto returning NULL mbuf");
203 ESPSTAT(espstats[ESP_STAT_QFULL],"packets dropped due to queue full");
204 ESPSTAT(espstats[ESP_STAT_NOXFORM],"packets with no xform");
205 ESPSTAT(espstats[ESP_STAT_BADILEN],"packets with bad ilen");
206 ESPSTAT(espstats[ESP_STAT_BADENC],"packets with bad encryption");
207 ESPSTAT(espstats[ESP_STAT_BADAUTH],"packets with bad authentication");
208 ESPSTAT(espstats[ESP_STAT_WRAP], "packets dropped for replay counter wrap");
209 ESPSTAT(espstats[ESP_STAT_REPLAY],"packets dropped for possible replay");
210 ESPSTAT(espstats[ESP_STAT_INVALID],"packets with an invalid SA");
211 ESPSTAT(espstats[ESP_STAT_TOOBIG],"packets too big");
212 ESPSTAT(espstats[ESP_STAT_PDROPS],"packets blocked due to policy");
213 ESPSTAT(espstats[ESP_STAT_CRYPTO],"failed crypto requests");
214 ESPSTAT(espstats[ESP_STAT_TUNNEL],"tunnel sanity check failures");
215 printf("\tesp histogram:\n");
216 for (i = 0; i < ESP_ALG_MAX; i++)
217 if (espstats[ESP_STAT_HIST + i])
218 printf("\t\tesp packets with %s: %"PRIu64"\n"
219 , algname(i, espalgs, __arraycount(espalgs))
220 , espstats[ESP_STAT_HIST + i]
221 );
222 ESPSTAT(espstats[ESP_STAT_IBYTES], "bytes received");
223 ESPSTAT(espstats[ESP_STAT_OBYTES], "bytes transmitted");
224#undef ESPSTAT
225 printf("\tipip:\n");
226
227#define IPIPSTAT(x,fmt) \
228 if ((x) || sflag <= 1) printf("\t\t%"PRIu64" ipip " fmt "\n", x)
229 IPIPSTAT(ipips[IPIP_STAT_IPACKETS],"total input packets");
230 IPIPSTAT(ipips[IPIP_STAT_OPACKETS],"total output packets");
231 IPIPSTAT(ipips[IPIP_STAT_HDROPS],"packets too short for header length");
232 IPIPSTAT(ipips[IPIP_STAT_QFULL],"packets dropped due to queue full");
233 IPIPSTAT(ipips[IPIP_STAT_PDROPS],"packets blocked due to policy");
234 IPIPSTAT(ipips[IPIP_STAT_SPOOF],"IP spoofing attempts");
235 IPIPSTAT(ipips[IPIP_STAT_FAMILY],"protocol family mismatched");
236 IPIPSTAT(ipips[IPIP_STAT_UNSPEC],"missing tunnel-endpoint address");
237 IPIPSTAT(ipips[IPIP_STAT_IBYTES],"input bytes received");
238 IPIPSTAT(ipips[IPIP_STAT_OBYTES],"output bytes processed");
239#undef IPIPSTAT
240
241 printf("\tipcomp:\n");
242#define IPCOMP(x,fmt) \
243 if ((x) || sflag <= 1) printf("\t\t%"PRIu64" ipcomp " fmt "\n", x)
244
245 IPCOMP(ipcs[IPCOMP_STAT_HDROPS],"packets too short for header length");
246 IPCOMP(ipcs[IPCOMP_STAT_NOPF], "protocol family not supported");
247 IPCOMP(ipcs[IPCOMP_STAT_NOTDB], "packets with no SA");
248 IPCOMP(ipcs[IPCOMP_STAT_BADKCR],"packets dropped by crypto returning NULL mbuf");
249 IPCOMP(ipcs[IPCOMP_STAT_QFULL], "queue full");
250 IPCOMP(ipcs[IPCOMP_STAT_NOXFORM],"no support for transform");
251 IPCOMP(ipcs[IPCOMP_STAT_WRAP], "packets dropped for replay counter wrap");
252 IPCOMP(ipcs[IPCOMP_STAT_INPUT], "input IPcomp packets");
253 IPCOMP(ipcs[IPCOMP_STAT_OUTPUT],"output IPcomp packets");
254 IPCOMP(ipcs[IPCOMP_STAT_INVALID],"packets with an invalid SA");
255 IPCOMP(ipcs[IPCOMP_STAT_TOOBIG],"packets decompressed as too big");
256 IPCOMP(ipcs[IPCOMP_STAT_MINLEN], "packets too short to be compressed");
257 IPCOMP(ipcs[IPCOMP_STAT_USELESS],"packet for which compression was useless");
258 IPCOMP(ipcs[IPCOMP_STAT_PDROPS],"packets blocked due to policy");
259 IPCOMP(ipcs[IPCOMP_STAT_CRYPTO],"failed crypto requests");
260
261 printf("\tipcomp histogram:\n");
262 for (i = 0; i < IPCOMP_ALG_MAX; i++)
263 if (ipcs[IPCOMP_STAT_HIST + i])
264 printf("\t\tIPcomp packets with %s: %"PRIu64"\n"
265 , algname(i, ipcompalgs, __arraycount(ipcompalgs))
266 , ipcs[IPCOMP_STAT_HIST + i]
267 );
268 IPCOMP(ipcs[IPCOMP_STAT_IBYTES],"input bytes");
269 IPCOMP(ipcs[IPCOMP_STAT_OBYTES],"output bytes");
270#undef IPCOMP
271}
