58#define DEFNICE 10
59
60static void usage(void) __dead;
61
62int
63main(int argc, char **argv)
64{
65 char *ep;
66 int niceness = DEFNICE;
67 int c;
68 long tmp;
69
70 setprogname(argv[0]);
71 (void)setlocale(LC_ALL, "");
72
73 /* handle obsolete -number syntax */
74 if (argc > 1 && argv[1][0] == '-' &&
75 isdigit((unsigned char)argv[1][1])) {
76 niceness = atoi (argv[1] + 1);
77 argc--; argv++;
78 }
79
80 while ((c = getopt (argc, argv, "n:")) != -1) {
81 switch (c) {
82 case 'n':
83 errno = 0;
84 tmp = strtol(optarg, &ep, 10);
85 if (*ep != '\0' || tmp < INT_MIN || tmp > INT_MAX)
86 errx(EXIT_FAILURE, "invalid argument: `%s'",
87 optarg);
88 niceness = (int)tmp;
89 break;
90 default:
91 usage();
92 break;
93 }
94 }
95 argc -= optind;
96 argv += optind;
97
98 if (argc == 0)
99 usage();
100
101 errno = 0;
102 niceness += getpriority(PRIO_PROCESS, 0);
103 if (errno) {
104 err(EXIT_FAILURE, "getpriority");
105 /* NOTREACHED */
106 }
107 if (setpriority(PRIO_PROCESS, 0, niceness) == -1) {
108 warn("setpriority");
109 }
110
111 (void)execvp(argv[0], &argv[0]);
112 err((errno == ENOENT || errno == ENOTDIR) ? 127 : 126, "%s", argv[0]);
113 /* NOTREACHED */
114}

116static void
117usage(void)
118{
119 (void)fprintf(stderr,
120 "Usage: %s [ -n increment ] utility [ argument ...]\n",
121 getprogname());
122 exit(EXIT_FAILURE);
123}
