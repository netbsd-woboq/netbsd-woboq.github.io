58static pam_handle_t *pamh;
59static struct pam_conv pamc = {
60 openpam_ttyconv,
61 NULL
62};
63
64#define pam_check(msg) \
65do { \
66 if (pam_err != PAM_SUCCESS) { \
67 warnx("%s: %s", (msg), pam_strerror(pamh, pam_err)); \
68 goto end; \
69 } \
70} while (/*CONSTCOND*/0)

72void
73pwpam_process(const char *username, int argc, char **argv)
74{
75 int ch, pam_err;
76 char hostname[MAXHOSTNAMELEN + 1];
77
78 /* details about the invoking user for logging */
79 const uid_t i_uid = getuid();
80 const struct passwd *const i_pwd = getpwuid(i_uid);
81 const char *const i_username = (i_pwd && i_pwd->pw_name)
82 ? i_pwd->pw_name : "(null)";
83
84 while ((ch = getopt(argc, argv, "")) != -1) {
85 switch (ch) {
86 default:
87 usage();
88 /* NOTREACHED */
89 }
90 }
91
92 argc -= optind;
93 argv += optind;
94
95 switch (argc) {
96 case 0:
97 /* username already provided */
98 break;
99 case 1:
100 username = argv[0];
101 break;
102 default:
103 usage();
104 /* NOTREACHED */
105 }
106
107 (void)printf("Changing password for %s.\n", username);
108
109 /* initialize PAM -- always use the program name "passwd" */
110 pam_err = pam_start("passwd", username, &pamc, &pamh);
111 if (pam_err != PAM_SUCCESS)
112 errx(1, "unable to start PAM session: %s",
113 pam_strerror(NULL, pam_err));
114
115 pam_err = pam_set_item(pamh, PAM_TTY, ttyname(STDERR_FILENO));
116 pam_check("unable to set TTY");
117
118 (void)gethostname(hostname, sizeof hostname);
119 pam_err = pam_set_item(pamh, PAM_RHOST, hostname);
120 pam_check("unable to set RHOST");
121
122 pam_err = pam_set_item(pamh, PAM_RUSER, getlogin());
123 pam_check("unable to set RUSER");
124
125 /* set new password */
126 pam_err = pam_chauthtok(pamh, 0);
127 if (pam_err != PAM_SUCCESS) {
128 if (pam_err == PAM_PERM_DENIED) {
129 syslog(LOG_AUTH | LOG_NOTICE,
130 "user %s (UID %lu) failed to change the "
131 "PAM authentication token of user %s: %s",
132 i_username, (unsigned long)i_uid, username,
133 pam_strerror(pamh, pam_err));
134 }
135 printf("Unable to change auth token: %s\n",
136 pam_strerror(pamh, pam_err));
137 } else {
138 syslog(LOG_AUTH | LOG_INFO,
139 "user %s (UID %lu) successfully changed the "
140 "PAM authentication token of user %s",
141 i_username, (unsigned long)i_uid, username);
142 }
143
144 end:
145 pam_end(pamh, pam_err);
146 if (pam_err == PAM_SUCCESS)
147 return;
148 exit(1);
149}
