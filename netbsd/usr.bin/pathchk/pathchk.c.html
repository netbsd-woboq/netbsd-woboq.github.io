49__dead static void usage(void);

51static int pflag; /* Perform portability checks */
52
53int
54main(int argc, char *argv[])
55{
56 int ch, rval;
57 const char *arg;
58
59 while ((ch = getopt(argc, argv, "p")) > 0) {
60 switch (ch) {
61 case 'p':
62 pflag = 1;
63 break;
64 default:
65 usage();
66 /*NOTREACHED*/
67 }
68 }
69 argc -= optind;
70 argv += optind;
71
72 if (argc == 0)
73 usage();
74
75 rval = 0;
76 while ((arg = *argv++) != NULL)
77 rval |= check(arg);
78
79 exit(rval);
80}

82static void
83usage(void)
84{
85
86 fprintf(stderr, "usage: pathchk [-p] pathname...\n");
87 exit(1);
88}

90static int
91check(const char *path)
92{
93 struct stat sb;
94 long complen, namemax, pathmax, svnamemax;
95 int badch, last;
96 char *end, *p, *pathd;
97
98 if ((pathd = strdup(path)) == NULL)
99 err(1, "strdup");
100
101 p = pathd;
102
103 if (!pflag) {
104 errno = 0;
105 namemax = pathconf(*p == '/' ? "/" : ".", _PC_NAME_MAX);
106 if (namemax == -1 && errno != 0)
107 namemax = NAME_MAX;
108 } else
109 namemax = _POSIX_NAME_MAX;
110
111 for (;;) {
112 p += strspn(p, "/");
113 complen = (long)strcspn(p, "/");
114 end = p + complen;
115 last = *end == '\0';
116 *end = '\0';
117
118 if (namemax != -1 && complen > namemax) {
119 warnx("%s: %s: component too long (limit %ld)", path,
120 p, namemax);
121 goto bad;
122 }
123
124 if (!pflag && stat(pathd, &sb) == -1 && errno != ENOENT) {
125 warn("%s: %.*s", path, (int)(strlen(pathd) -
126 complen - 1), pathd);
127 goto bad;
128 }
129
130 if (pflag && (badch = portable(p)) >= 0) {
131 warnx("%s: %s: component contains non-portable "
132 "character `%c'", path, p, badch);
133 goto bad;
134 }
135
136 if (last)
137 break;
138
139 if (!pflag) {
140 errno = 0;
141 svnamemax = namemax;
142 namemax = pathconf(pathd, _PC_NAME_MAX);
143 if (namemax == -1 && errno != 0)
144 namemax = svnamemax;
145 }
146
147 *end = '/';
148 p = end + 1;
149 }
150
151 if (!pflag) {
152 errno = 0;
153 pathmax = pathconf(path, _PC_PATH_MAX);
154 if (pathmax == -1 && errno != 0)
155 pathmax = PATH_MAX;
156 } else
157 pathmax = _POSIX_PATH_MAX;
158 if (pathmax != -1 && strlen(path) >= (size_t)pathmax) {
159 warnx("%s: path too long (limit %ld)", path, pathmax - 1);
160 goto bad;
161 }
162
163 free(pathd);
164 return (0);
165
166bad: free(pathd);
167 return (1);
168}

170/*
171 * Check whether a path component contains only portable characters. Return
172 * the first non-portable character found.
173 */
174static int
175portable(const char *path)
176{
177 static const char charset[] =
178 "abcdefghijklmnopqrstuvwxyz"
179 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
180 "0123456789._-";
181 long s;
182
183 if (*path == '-')
184 return (*path);
185
186 s = strspn(path, charset);
187 if (path[s] != '\0')
188 return (path[s]);
189
190 return (-1);
191}
