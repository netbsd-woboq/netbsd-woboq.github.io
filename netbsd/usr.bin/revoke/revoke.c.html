31#include <sys/types.h>
32#include <stdio.h>
33#include <stdlib.h>
34#include <unistd.h>
35#include <err.h>
36
37int
38main(int argc, char *argv[])
39{
40 if (argc != 2)
41 errx(EXIT_FAILURE, "usage: %s <file>", getprogname());
42
43 if (revoke(argv[1]) != 0)
44 err(EXIT_FAILURE, "revoke(%s)", argv[1]);
45
46 return EXIT_SUCCESS;
47}
