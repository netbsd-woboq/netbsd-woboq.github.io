129int
130main(int argc, char *argv[])
131{
132 struct passwd *pw;
133 struct servent *sp;
134 struct termios tty;
135 sigset_t smask;
136 uid_t uid;
137 int argoff, ch, dflag, nflag, one;
138 int i, len, len2;
139 int family = AF_UNSPEC;
140 char *host, *p, *user, *name, term[1024] = "network";
141 speed_t ospeed;
142 struct sigaction sa;
143 char *service = NULL;
144 struct rlimit rlim;
145
146 argoff = dflag = nflag = 0;
147 one = 1;
148 host = user = NULL;
149 sp = NULL;
150
151 if (strcmp(getprogname(), "rlogin") != 0) {
152 host = strdup(getprogname());
153 if (host == NULL)
154 err(1, NULL);
155 }
156
157 /* handle "rlogin host flags" */
158 if (!host && argc > 2 && argv[1][0] != '-') {
159 host = argv[1];
160 argoff = 1;
161 }
162
163#define OPTIONS "468dEe:l:np:"
164 while ((ch = getopt(argc - argoff, argv + argoff, OPTIONS)) != -1)
165 switch(ch) {
166 case '4':
167 family = AF_INET;
168 break;
169 case '6':
170 family = AF_INET6;
171 break;
172 case '8':
173 eight = 1;
174 break;
175 case 'd':
176 dflag = 1;
177 break;
178 case 'E':
179 noescape = 1;
180 break;
181 case 'e':
182 noescape = 0;
183 escapechar = getescape(optarg);
184 break;
185 case 'l':
186 user = optarg;
187 break;
188 case 'n':
189 nflag = 1;
190 break;
191 case 'p':
192 sp = getport(service = optarg, "tcp");
193 break;
194 case '?':
195 default:
196 usage();
197 }
198 optind += argoff;
199 argc -= optind;
200 argv += optind;
201
202 /* if haven't gotten a host yet, do so */
203 if (!host && !(host = *argv++))
204 usage();
205
206 if (*argv)
207 usage();
208
209 if (!(pw = getpwuid(uid = getuid())))
210 errx(1, "unknown user id.");
211 /* Accept user1@host format, though "-l user2" overrides user1 */
212 p = strchr(host, '@');
213 if (p) {
214 *p = '\0';
215 if (!user && p > host)
216 user = host;
217 host = p + 1;
218 if (*host == '\0')
219 usage();
220 }
221 if ((name = strdup(pw->pw_name)) == NULL)
222 err(1, "malloc");
223 if (!user)
224 user = name;
225
226 if (sp == NULL)
227 sp = getservbyname("login", "tcp");
228 if (sp == NULL)
229 errx(1, "login/tcp: unknown service.");
230
231 if ((p = getenv("TERM")) != NULL)
232 (void)strlcpy(term, p, sizeof(term));
233 len = strlen(term);
234 if (len < (int)(sizeof(term) - 1) && tcgetattr(0, &tty) == 0) {
235 /* start at 2 to include the / */
236 for (ospeed = i = cfgetospeed(&tty), len2 = 2; i > 9; len2++)
237 i /= 10;
238
239 if (len + len2 < (int)sizeof(term))
240 (void)snprintf(term + len, len2 + 1, "/%d", ospeed);
241 }
242
243 (void)get_window_size(0, &winsize);
244
245 sigemptyset(&sa.sa_mask);
246 sa.sa_flags = SA_RESTART;
247 sa.sa_handler = lostpeer;
248 (void)sigaction(SIGPIPE, &sa, (struct sigaction *)0);
249 /* will use SIGUSR1 for window size hack, so hold it off */
250 sigemptyset(&smask);
251 sigaddset(&smask, SIGURG);
252 sigaddset(&smask, SIGUSR1);
253 (void)sigprocmask(SIG_SETMASK, &smask, &smask);
254 /*
255 * We set SIGURG and SIGUSR1 below so that an
256 * incoming signal will be held pending rather than being
257 * discarded. Note that these routines will be ready to get
258 * a signal by the time that they are unblocked below.
259 */
260 sa.sa_handler = copytochild;
261 (void)sigaction(SIGURG, &sa, (struct sigaction *) 0);
262 sa.sa_handler = writeroob;
263 (void)sigaction(SIGUSR1, &sa, (struct sigaction *) 0);
264
265 /* don't dump core */
266 rlim.rlim_cur = rlim.rlim_max = 0;
267 if (setrlimit(RLIMIT_CORE, &rlim) < 0)
268 warn("setrlimit");
269
270 rem = rcmd_af(&host, sp->s_port, name, user, term, 0, family);
271 if (rem < 0)
272 exit(1);
273
274 if (dflag &&
275 setsockopt(rem, SOL_SOCKET, SO_DEBUG, &one, sizeof(one)) < 0)
276 warn("setsockopt DEBUG (ignored)");
277 if (nflag &&
278 setsockopt(rem, IPPROTO_TCP, TCP_NODELAY, &one, sizeof(one)) < 0)
279 warn("setsockopt NODELAY (ignored)");
280
281 {
282 struct sockaddr_storage ss;
283 socklen_t sslen = sizeof(ss);
284 if (getsockname(rem, (struct sockaddr *)&ss, &sslen) == 0
285 && ((struct sockaddr *)&ss)->sa_family == AF_INET) {
286 one = IPTOS_LOWDELAY;
287 if (setsockopt(rem, IPPROTO_IP, IP_TOS, (char *)&one,
288 sizeof(int)) < 0)
289 warn("setsockopt TOS (ignored)");
290 }
291 }
292
293 (void)setuid(uid);
294 doit(&smask);
295 /*NOTREACHED*/
296 return (0);
297}

299static pid_t child;
300
301static void
302doit(sigset_t *smask)
303{
304 struct sigaction sa;
305
306 sigemptyset(&sa.sa_mask);
307 sa.sa_flags = SA_RESTART;
308 sa.sa_handler = SIG_IGN;
309 (void)sigaction(SIGINT, &sa, (struct sigaction *) 0);
310 setsignal(SIGHUP);
311 setsignal(SIGQUIT);
312 mode(1);
313 child = fork();
314 if (child == -1) {
315 warn("fork");
316 done(1);
317 }
318 if (child == 0) {
319 mode(1);
320 if (reader(smask) == 0) {
321 msg("connection closed.");
322 exit(0);
323 }
324 sleep(1);
325 msg("\aconnection closed.");
326 exit(1);
327 }
328
329 /*
330 * We may still own the socket, and may have a pending SIGURG (or might
331 * receive one soon) that we really want to send to the reader. When
332 * one of these comes in, the trap copytochild simply copies such
333 * signals to the child. We can now unblock SIGURG and SIGUSR1
334 * that were set above.
335 */
336 (void)sigprocmask(SIG_SETMASK, smask, (sigset_t *) 0);
337 sa.sa_handler = catch_child;
338 (void)sigaction(SIGCHLD, &sa, (struct sigaction *) 0);
339 writer();
340 msg("closed connection.");
341 done(0);
342}

344/* trap a signal, unless it is being ignored. */
345static void
346setsignal(int sig)
347{
348 struct sigaction sa;
349 sigset_t sigs;
350
351 sigemptyset(&sigs);
352 sigaddset(&sigs, sig);
353 sigprocmask(SIG_BLOCK, &sigs, &sigs);
354
355 sigemptyset(&sa.sa_mask);
356 sa.sa_handler = exit;
357 sa.sa_flags = SA_RESTART;
358 (void)sigaction(sig, &sa, &sa);
359 if (sa.sa_handler == SIG_IGN)
360 (void)sigaction(sig, &sa, (struct sigaction *) 0);
361
362 (void)sigprocmask(SIG_SETMASK, &sigs, (sigset_t *) 0);
363}

365static void
366done(int status)
367{
368 pid_t w;
369 int wstatus;
370 struct sigaction sa;
371
372 mode(0);
373 if (child > 0) {
374 /* make sure catch_child does not snap it up */
375 sigemptyset(&sa.sa_mask);
376 sa.sa_handler = SIG_DFL;
377 sa.sa_flags = 0;
378 (void)sigaction(SIGCHLD, &sa, (struct sigaction *) 0);
379 if (kill(child, SIGKILL) >= 0)
380 while ((w = wait(&wstatus)) > 0 && w != child)
381 continue;
382 }
383 exit(status);
384}

388/*
389 * This is called when the reader process gets the out-of-band (urgent)
390 * request to turn on the window-changing protocol.
391 */
392static void
393writeroob(int signo)
394{
395 struct sigaction sa;
396
397 if (dosigwinch == 0) {
398 sendwindow();
399 sigemptyset(&sa.sa_mask);
400 sa.sa_handler = sigwinch;
401 sa.sa_flags = SA_RESTART;
402 (void)sigaction(SIGWINCH, &sa, (struct sigaction *) 0);
403 }
404 dosigwinch = 1;
405}

407static void
408catch_child(int signo)
409{
410 int status;
411 pid_t pid;
412
413 for (;;) {
414 pid = waitpid(-1, &status, WNOHANG|WUNTRACED);
415 if (pid == 0)
416 return;
417 /* if the child (reader) dies, just quit */
418 if (pid < 0 || (pid == child && !WIFSTOPPED(status)))
419 done(WEXITSTATUS(status) | WTERMSIG(status));
420 }
421 /* NOTREACHED */
422}

424/*
425 * writer: write to remote: 0 -> line.
426 * ~. terminate
427 * ~^Z suspend rlogin process.
428 * ~<delayed-suspend char> suspend rlogin process, but leave reader alone.
429 */
430static void
431writer(void)
432{
433 int bol, local, n;
434 char c;
435
436 bol = 1; /* beginning of line */
437 local = 0;
438 for (;;) {
439 n = read(STDIN_FILENO, &c, 1);
440 if (n <= 0) {
441 if (n < 0 && errno == EINTR)
442 continue;
443 break;
444 }
445 /*
446 * If we're at the beginning of the line and recognize a
447 * command character, then we echo locally. Otherwise,
448 * characters are echo'd remotely. If the command character
449 * is doubled, this acts as a force and local echo is
450 * suppressed.
451 */
452 if (bol) {
453 bol = 0;
454 if (!noescape && c == escapechar) {
455 local = 1;
456 continue;
457 }
458 } else if (local) {
459 local = 0;
460 if (c == '.' || CCEQ(deftty.c_cc[VEOF], c)) {
461 echo((int)c);
462 break;
463 }
464 if (CCEQ(deftty.c_cc[VSUSP], c)) {
465 bol = 1;
466 echo((int)c);
467 stop(1);
468 continue;
469 }
470 if (CCEQ(deftty.c_cc[VDSUSP], c)) {
471 bol = 1;
472 echo((int)c);
473 stop(0);
474 continue;
475 }
476 if (c != escapechar) {
477 (void)write(rem, &escapechar, 1);
478 }
479 }
480
481 if (write(rem, &c, 1) == 0) {
482 msg("line gone");
483 break;
484 }
485
486 bol = CCEQ(deftty.c_cc[VKILL], c) ||
487 CCEQ(deftty.c_cc[VEOF], c) ||
488 CCEQ(deftty.c_cc[VINTR], c) ||
489 CCEQ(deftty.c_cc[VSUSP], c) ||
490 c == '\r' || c == '\n';
491 }
492}

494static void
495echo(int i)
496{
497 char c = (char)i;
498 char *p;
499 char buf[8];
500
501 p = buf;
502 c &= 0177;
503 *p++ = escapechar;
504 if (c < ' ') {
505 *p++ = '^';
506 *p++ = c + '@';
507 } else if (c == 0177) {
508 *p++ = '^';
509 *p++ = '?';
510 } else
511 *p++ = c;
512 *p++ = '\r';
513 *p++ = '\n';
514 (void)write(STDOUT_FILENO, buf, p - buf);
515}

517static void
518stop(int all)
519{
520 struct sigaction sa;
521
522 mode(0);
523 sigemptyset(&sa.sa_mask);
524 sa.sa_handler = SIG_IGN;
525 sa.sa_flags = SA_RESTART;
526 (void)sigaction(SIGCHLD, &sa, (struct sigaction *) 0);
527 (void)kill(all ? 0 : getpid(), SIGTSTP);
528 sa.sa_handler = catch_child;
529 (void)sigaction(SIGCHLD, &sa, (struct sigaction *) 0);
530 mode(1);
531 sigwinch(0); /* check for size changes */
532}

546/*
547 * Send the window size to the server via the magic escape
548 */
549static void
550sendwindow(void)
551{
552 struct winsize *wp;
553 char obuf[4 + sizeof (struct winsize)];
554
555 wp = (struct winsize *)(obuf+4);
556 obuf[0] = 0377;
557 obuf[1] = 0377;
558 obuf[2] = 's';
559 obuf[3] = 's';
560 wp->ws_row = htons(winsize.ws_row);
561 wp->ws_col = htons(winsize.ws_col);
562 wp->ws_xpixel = htons(winsize.ws_xpixel);
563 wp->ws_ypixel = htons(winsize.ws_ypixel);
564
565 (void)write(rem, obuf, sizeof(obuf));
566}

568/*
569 * reader: read from remote: line -> 1
570 */
571#define READING 1
572#define WRITING 2
573
574static jmp_buf rcvtop;
575static pid_t ppid;
576static int rcvcnt, rcvstate;
577static char rcvbuf[8 * 1024];
578
579static void
580oob(int signo)
581{
582 struct termios tty;
583 int atmark, n, rcvd;
584 char waste[BUFSIZ], mark;
585
586 rcvd = 0;
587 while (recv(rem, &mark, 1, MSG_OOB) < 0) {
588 switch (errno) {
589 case EWOULDBLOCK:
590 /*
591 * Urgent data not here yet. It may not be possible
592 * to send it yet if we are blocked for output and
593 * our input buffer is full.
594 */
595 if (rcvcnt < (int)sizeof(rcvbuf)) {
596 n = read(rem, rcvbuf + rcvcnt,
597 sizeof(rcvbuf) - rcvcnt);
598 if (n <= 0)
599 return;
600 rcvd += n;
601 } else {
602 n = read(rem, waste, sizeof(waste));
603 if (n <= 0)
604 return;
605 }
606 continue;
607 default:
608 return;
609 }
610 }
611 if (mark & TIOCPKT_WINDOW) {
612 /* Let server know about window size changes */
613 (void)kill(ppid, SIGUSR1);
614 }
615 if (!eight && (mark & TIOCPKT_NOSTOP)) {
616 (void)tcgetattr(0, &tty);
617 tty.c_iflag &= ~IXON;
618 (void)tcsetattr(0, TCSANOW, &tty);
619 }
620 if (!eight && (mark & TIOCPKT_DOSTOP)) {
621 (void)tcgetattr(0, &tty);
622 tty.c_iflag |= (deftty.c_iflag & IXON);
623 (void)tcsetattr(0, TCSANOW, &tty);
624 }
625 if (mark & TIOCPKT_FLUSHWRITE) {
626 (void)tcflush(1, TCIOFLUSH);
627 for (;;) {
628 if (ioctl(rem, SIOCATMARK, &atmark) < 0) {
629 warn("ioctl SIOCATMARK (ignored)");
630 break;
631 }
632 if (atmark)
633 break;
634 n = read(rem, waste, sizeof (waste));
635 if (n <= 0)
636 break;
637 }
638 /*
639 * Don't want any pending data to be output, so clear the recv
640 * buffer. If we were hanging on a write when interrupted,
641 * don't want it to restart. If we were reading, restart
642 * anyway.
643 */
644 rcvcnt = 0;
645 longjmp(rcvtop, 1);
646 }
647
648 /* oob does not do FLUSHREAD (alas!) */
649
650 /*
651 * If we filled the receive buffer while a read was pending, longjmp
652 * to the top to restart appropriately. Don't abort a pending write,
653 * however, or we won't know how much was written.
654 */
655 if (rcvd && rcvstate == READING)
656 longjmp(rcvtop, 1);
657}

659/* reader: read from remote: line -> 1 */
660static int
661reader(sigset_t *smask)
662{
663 pid_t pid;
664 int n, remaining;
665 char *bufp;
666 struct sigaction sa;
667
668 pid = getpid(); /* modern systems use positives for pid */
669 sigemptyset(&sa.sa_mask);
670 sa.sa_flags = SA_RESTART;
671 sa.sa_handler = SIG_IGN;
672 (void)sigaction(SIGTTOU, &sa, (struct sigaction *) 0);
673 sa.sa_handler = oob;
674 (void)sigaction(SIGURG, &sa, (struct sigaction *) 0);
675 ppid = getppid();
676 (void)fcntl(rem, F_SETOWN, pid);
677 (void)setjmp(rcvtop);
678 (void)sigprocmask(SIG_SETMASK, smask, (sigset_t *) 0);
679 bufp = rcvbuf;
680 for (;;) {
681 while ((remaining = rcvcnt - (bufp - rcvbuf)) > 0) {
682 rcvstate = WRITING;
683 n = write(STDOUT_FILENO, bufp, remaining);
684 if (n < 0) {
685 if (errno != EINTR)
686 return (-1);
687 continue;
688 }
689 bufp += n;
690 }
691 bufp = rcvbuf;
692 rcvcnt = 0;
693 rcvstate = READING;
694
695 rcvcnt = read(rem, rcvbuf, sizeof (rcvbuf));
696 if (rcvcnt == 0)
697 return (0);
698 if (rcvcnt < 0) {
699 if (errno == EINTR)
700 continue;
701 warn("read");
702 return (-1);
703 }
704 }
705}

707static void
708mode(int f)
709{
710 struct termios tty;
711
712 switch (f) {
713 case 0:
714 (void)tcsetattr(0, TCSANOW, &deftty);
715 break;
716 case 1:
717 (void)tcgetattr(0, &deftty);
718 tty = deftty;
719 /* This is loosely derived from sys/compat/tty_compat.c. */
720 tty.c_lflag &= ~(ECHO|ICANON|ISIG|IEXTEN);
721 tty.c_iflag &= ~ICRNL;
722 tty.c_oflag &= ~OPOST;
723 tty.c_cc[VMIN] = 1;
724 tty.c_cc[VTIME] = 0;
725 if (eight) {
726 tty.c_iflag &= IXOFF;
727 tty.c_cflag &= ~(CSIZE|PARENB);
728 tty.c_cflag |= CS8;
729 }
730 (void)tcsetattr(0, TCSANOW, &tty);
731 break;
732
733 default:
734 return;
735 }
736}

795static u_int
796getescape(char *p)
797{
798 long val;
799 int len;
800
801 if ((len = strlen(p)) == 1) /* use any single char, including '\' */
802 return ((u_int)*p);
803 /* otherwise, \nnn */
804 if (*p == '\\' && len >= 2 && len <= 4) {
805 val = strtol(++p, NULL, 8);
806 for (;;) {
807 if (!*++p)
808 return ((u_int)val);
809 if (*p < '0' || *p > '8')
810 break;
811 }
812 }
813 msg("illegal option value -- e");
814 usage();
815 /* NOTREACHED */
816 return (0);
817}
