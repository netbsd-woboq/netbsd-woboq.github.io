32#include <sys/cdefs.h>
33__RCSID("$NetBSD: getport.c,v 1.2 2008/04/28 20:24:14 martin Exp $");
34
35#include <stdlib.h>
36#include <err.h>
37#include <netdb.h>
38#include <limits.h>
39#include <errno.h>
40#include <netinet/in.h>
41
42#include "getport.h"
43
44struct servent *
45getport(const char *service, const char *protocol)
46{
47 long port;
48 char *ep;
49 struct servent *sp = getservbyname(service, protocol);
50 if (sp != NULL)
51 return sp;
52
53 if ((sp = calloc(1, sizeof(*sp))) == NULL)
54 err(1, "malloc");
55 sp->s_name = __UNCONST(service);
56 sp->s_proto = __UNCONST(protocol);
57 port = strtol(service, &ep, 0);
58 if ((service[0] != '\0' && *ep != '\0') ||
59 (errno == ERANGE &&
60 (port == LONG_MAX || port == LONG_MIN)) ||
61 (port <= 0 || port >= IPPORT_ANONMAX))
62 errx(1,"port must be between 1 and %d",
63 IPPORT_ANONMAX);
64 sp->s_port = htons((uint16_t)port);
65 return sp;
66}
