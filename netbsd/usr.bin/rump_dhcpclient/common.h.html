36#define timeval_to_double(tv) ((tv)->tv_sec * 1.0 + (tv)->tv_usec * 1.0e-6)
37#define timernorm(tvp) \
38 do { \
39 while ((tvp)->tv_usec >= 1000000) { \
40 (tvp)->tv_sec++; \
41 (tvp)->tv_usec -= 1000000; \
42 } \
43 } while (0 /* CONSTCOND */);
