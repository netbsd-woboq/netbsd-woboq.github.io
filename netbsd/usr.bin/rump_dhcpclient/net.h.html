62/* Work out if we have a private address or not
63 * 10/8
64 * 172.16/12
65 * 192.168/16
66 */
67#ifndef IN_PRIVATE
68# define IN_PRIVATE(addr) (((addr & IN_CLASSA_NET) == 0x0a000000) || \
69 ((addr & 0xfff00000) == 0xac100000) || \
70 ((addr & IN_CLASSB_NET) == 0xc0a80000))
71#endif
72
73#define LINKLOCAL_ADDR 0xa9fe0000
74#define LINKLOCAL_MASK IN_CLASSB_NET
75#define LINKLOCAL_BRDC (LINKLOCAL_ADDR | ~LINKLOCAL_MASK)
76
77#ifndef IN_LINKLOCAL
78# define IN_LINKLOCAL(addr) ((addr & IN_CLASSB_NET) == LINKLOCAL_ADDR)
79#endif
