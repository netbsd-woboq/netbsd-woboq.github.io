62/*
63 * malloc with result test
64 */
65void *
66xmalloc(size_t size)
67{
68 void *p;
69
70 if ((p = malloc(size)) == NULL)
71 err(1, "malloc(%zu)", size);
72 return p;
73}
74
75/*
76 * realloc with result test
77 */
78void *
79xrealloc(void *p, size_t size)
80{
81 if (p == NULL) /* Compatibility hack. */
82 return (xmalloc(size));
83
84 if ((p = realloc(p, size)) == NULL)
85 err(1, "realloc(%zu)", size);
86 return p;
87}
88
89/*
90 * realloc with result test
91 */
92void *
93xcalloc(size_t c, size_t n)
94{
95 void *p;
96
97 if ((p = calloc(c, n)) == NULL)
98 err(1, "calloc(%zu, %zu)", c, n);
99 return p;
100}
101/*
102 * Return a string for a regular expression error passed. This is overkill,
103 * because of the silly semantics of regerror (we can never know the size of
104 * the buffer).
105 */
106char *
107strregerror(int errcode, regex_t *preg)
108{
109 char buf[1];
110 static char *oe;
111 size_t s;
112
113 if (oe != NULL)
114 free(oe);
115 s = regerror(errcode, preg, buf, 0);
116 oe = xmalloc(s);
117 (void)regerror(errcode, preg, oe, s);
118 return (oe);
119}
