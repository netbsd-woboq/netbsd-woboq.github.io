32#include <sys/cdefs.h>
33#ifndef lint
34__RCSID("$NetBSD: skeyinfo.c,v 1.7 2009/09/05 06:15:24 dholland Exp $");
35#endif
36
37#include <err.h>
38#include <errno.h>
39#include <pwd.h>
40#include <stdio.h>
41#include <string.h>
42#include <unistd.h>
43
44#include <skey.h> /* requires stdio.h */
45
46int
47main(int argc, char *argv[])
48{
49 struct skey skey;
50 char name[100], prompt[1024];
51 int uid;
52 struct passwd *pw = NULL;
53
54 argc--;
55 argv++;
56
57 if (geteuid())
58 errx(1, "Must be root to read /etc/skeykeys");
59
60 uid = getuid();
61
62 if (!argc)
63 pw = getpwuid(uid);
64 else if (!uid)
65 pw = getpwnam(argv[0]);
66 else {
67 errno = EPERM;
68 err(1, "%s", argv[0]);
69 }
70
71 if (!pw) {
72 if (argc)
73 errx(1, "%s: no such user", argv[0]);
74 else
75 errx(1, "Who are you?");
76 }
77
78 (void) strlcpy(name, pw->pw_name, sizeof(name));
79
80 if (getskeyprompt(&skey, name, prompt) == -1) {
81 printf("%s %s no s/key\n",
82 argc ? name : "You",
83 argc ? "has" : "have");
84 }
85 else {
86 if (argc)
87 printf("%s's ", pw->pw_name);
88 else
89 printf("Your ");
90 printf("next %s", prompt);
91 }
92 return 0;
93}
