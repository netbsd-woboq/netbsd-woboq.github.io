70/*
71 * copy sorted lines to output
72 * Ignore duplicates (marked with -ve keylen)
73 */
74void
75append(RECHEADER **keylist, int nelem, FILE *fp, put_func_t put)
76{
77 RECHEADER **cpos, **lastkey;
78 RECHEADER *crec;
79
80 lastkey = keylist + nelem;
81 if (REVERSE) {
82 for (cpos = lastkey; cpos-- > keylist;) {
83 crec = *cpos;
84 if (crec->keylen >= 0)
85 put(crec, fp);
86 }
87 } else {
88 for (cpos = keylist; cpos < lastkey; cpos++) {
89 crec = *cpos;
90 if (crec->keylen >= 0)
91 put(crec, fp);
92 }
93 }
94}
