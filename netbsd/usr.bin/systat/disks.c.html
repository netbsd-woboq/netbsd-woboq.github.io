48static void drvselect(char *args, int truefalse, int selections[]);
49
50void
51disks_add(char *args)
52{
53
54 if (args)
55 drvselect(args, 1, drv_select);
56}
57
58void
59disks_remove(char *args)
60{
61
62 if (args)
63 drvselect(args, 0, drv_select);
64}
65
66void
67disks_drives(char *args)
68{
69 size_t i;
70
71 if (args) {
72 for (i = 0; i < ndrive; i++)
73 drv_select[i] = 0;
74 disks_add(args);
75 } else {
76 move(CMDLINE, 0);
77 clrtoeol();
78 for (i = 0; i < ndrive; i++)
79 printw("%s ", dr_name[i]);
80 }
81}
82
83static void
84drvselect(char *args, int truefalse, int selections[])
85{
86 char *cp;
87 size_t i;
88
89 cp = strchr(args, '\n');
90 if (cp)
91 *cp = '\0';
92 for (;;) {
93 for (cp = args; *cp && isspace((unsigned char)*cp); cp++)
94 ;
95 args = cp;
96 for (; *cp && !isspace((unsigned char)*cp); cp++)
97 ;
98 if (*cp)
99 *cp++ = '\0';
100 if (cp - args == 0)
101 break;
102 for (i = 0; i < ndrive; i++)
103 if (fnmatch(args, dr_name[i], 0) == 0) {
104 selections[i] = truefalse;
105 break;
106 }
107 if (i >= ndrive)
108 error("%s: unknown drive", args);
109 args = cp;
110 }
111 labels();
112 display(0);
113}
