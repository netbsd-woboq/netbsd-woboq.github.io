48int curscale = SC_AUTO;
49char *matchline = NULL;
50int showpps = 0;
51int needsort = 0;
52
53int
54ifcmd(const char *cmd, const char *args)
55{
56 int scale;
57
58 if (prefix(cmd, "scale")) {
59 if ((scale = get_scale(args)) != -1)
60 curscale = scale;
61 else {
62 move(CMDLINE, 0);
63 clrtoeol();
64 addstr("what scale? ");
65 addstr(get_helplist());
66 }
67 } else if (prefix(cmd, "match")) {
68 if (args != NULL && *args != '\0' && memcmp(args, "*", 2) != 0) {
69 /* We got a valid match line */
70 if (matchline != NULL)
71 free(matchline);
72 needsort = 1;
73 matchline = strdup(args);
74 } else {
75 /* Empty or * pattern, turn filtering off */
76 if (matchline != NULL)
77 free(matchline);
78 needsort = 1;
79 matchline = NULL;
80 }
81 } else if (prefix(cmd, "pps"))
82 showpps = !showpps;
83
84 return (1);
85}
