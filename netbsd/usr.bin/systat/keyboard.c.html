52extern sig_atomic_t needsredraw;
53
54void
55keyboard(void)
56{
57 int ch, rch, col;
58 char *line;
59 int i, linesz;
60 static char help[] = "help";
61 static char quit[] = "quit";
62
63 ch = 0; /* XXX gcc */
64 linesz = COLS - 2; /* XXX does not get updated on SIGWINCH */
65 if ((line = malloc(linesz)) == NULL) {
66 error("malloc failed");
67 die(0);
68 }
69
70 for (;;) {
71 col = 0;
72 move(CMDLINE, 0);
73
74 while (col == 0 || (ch != '\r' && ch != '\n')) {
75 refresh();
76 for (;;) {
77 ch = getch();
78 if (!needsredraw)
79 break;
80 redraw();
81 }
82 if (ch == ERR) {
83 display(SIGALRM);
84 continue;
85 }
86 if (ch == KEY_RESIZE) {
87 redraw();
88 continue;
89 }
90 ch &= 0177;
91 rch = ch;
92 if (col == 0) {
93 switch(ch) {
94 case '\n':
95 case '\r':
96 case ' ':
97 clearerror();
98 display(0);
99 break;
100 case CTRL('l'):
101 clearerror();
102 wrefresh(curscr);
103 break;
104 case CTRL('g'):
105 status();
106 break;
107 case '?':
108 case 'H':
109 case 'h':
110 command(help);
111 move(CMDLINE, 0);
112 break;
113 case 'Q':
114 case 'q':
115 command(quit);
116 break;
117 case ':':
118 move(CMDLINE, 0);
119 clrtoeol();
120 addch(':');
121 col++;
122 break;
123 }
124 continue;
125 }
126 if (ch == '\b' || ch == '\177' || ch == erasechar()) {
127 if (col > 0)
128 col--;
129 goto doerase;
130 }
131 if (ch == CTRL('w') && col > 0) {
132 while (--col >= 0 &&
133 isspace((unsigned char)line[col]))
134 continue;
135 col++;
136 while (--col >= 0 &&
137 !isspace((unsigned char)line[col]))
138 if (col == 0)
139 break;
140 col++;
141 goto doerase;
142 }
143 if (ch == killchar() && col > 0) {
144 col = 1;
145 doerase:
146 move(CMDLINE, col);
147 clrtoeol();
148 continue;
149 }
150 if (isprint(rch) || rch == ' ') {
151 if (col < linesz) {
152 line[col] = rch;
153 mvaddch(CMDLINE, col, rch);
154 col++;
155 }
156 }
157 }
158 line[col] = '\0';
159 /* pass commands as lowercase */
160 for (i = 1; i < col ; i++)
161 line[i] = tolower((unsigned char)line[i]);
162 command(line + 1);
163 }
164 /* NOTREACHED */
165}
