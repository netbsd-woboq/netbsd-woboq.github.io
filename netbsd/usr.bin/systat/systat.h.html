36struct mode {
37 const char *c_name; /* command name */
38 void (*c_refresh)(void); /* display refresh */
39 void (*c_fetch)(void); /* sets up data structures */
40 void (*c_label)(void); /* label display */
41 int (*c_init)(void); /* initialize namelist, etc. */
42 WINDOW *(*c_open)(void); /* open display */
43 void (*c_close)(WINDOW *); /* close display */
44 struct command *c_commands; /* commands for mode */
45 char c_flags; /* see below */
46};
47
48struct command {
49 const char *c_name;
50 void (*c_cmd)(char *args);
51 const char *helptext;
52};
