48void
49get_addrs(const char *my_machine_name, const char *his_machine_name)
50{
51 struct hostent *hp;
52 struct servent *sp;
53
54 msg.pid = htonl(getpid());
55 /*
56 * If the callee is on-machine, just use loopback
57 * otherwise do a lookup...
58 */
59 if (strcmp(his_machine_name, my_machine_name) != 0) {
60 /* look up the address of the local host */
61 hp = gethostbyname(my_machine_name);
62 if (hp == NULL)
63 errx(EXIT_FAILURE, "%s: %s", my_machine_name,
64 hstrerror(h_errno));
65 memcpy(&my_machine_addr, hp->h_addr, sizeof(my_machine_addr));
66 hp = gethostbyname(his_machine_name);
67 if (hp == NULL)
68 errx(EXIT_FAILURE, "%s: %s", his_machine_name,
69 hstrerror(h_errno));
70 memcpy(&his_machine_addr, hp->h_addr, sizeof(his_machine_addr));
71 } else
72 his_machine_addr.s_addr = my_machine_addr.s_addr =
73 htonl(INADDR_LOOPBACK);
74
75 /* find the server's port */
76 sp = getservbyname("ntalk", "udp");
77 if (sp == 0)
78 errx(EXIT_FAILURE, "%s/%s: service is not registered.",
79 "ntalk", "udp");
80 daemon_port = sp->s_port;
81}
