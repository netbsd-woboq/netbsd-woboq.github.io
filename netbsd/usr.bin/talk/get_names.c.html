50/*
51 * Determine the local and remote user, tty, and machines
52 */
53void
54get_names(int argc, char *argv[])
55{
56 char hostname[MAXHOSTNAMELEN + 1];
57 const char *his_name, *my_name;
58 const char *my_machine_name, *his_machine_name;
59 const char *his_tty;
60 char *cp;
61 char *names;
62
63 if (argc < 2 ) {
64 fprintf(stderr, "Usage: %s user [ttyname]\n", getprogname());
65 exit(1);
66 }
67 if (!isatty(0))
68 errx(EXIT_FAILURE, "Standard input must be a tty, "
69 "not a pipe or a file");
70 if ((my_name = getlogin()) == NULL) {
71 struct passwd *pw;
72
73 if ((pw = getpwuid(getuid())) == NULL)
74 errx(EXIT_FAILURE, "You don't exist. Go away.");
75 my_name = pw->pw_name;
76 }
77 if ((cp = getenv("TALKHOST")) != NULL)
78 (void)estrlcpy(hostname, cp, sizeof(hostname));
79 else {
80 if (gethostname(hostname, sizeof(hostname)) == -1)
81 err(EXIT_FAILURE, "gethostname");
82 hostname[sizeof(hostname) - 1] = '\0';
83 }
84 my_machine_name = hostname;
85 /* check for, and strip out, the machine name of the target */
86 names = strdup(argv[1]);
87 for (cp = names; *cp && !strchr("@:!.", *cp); cp++)
88 ;
89 if (*cp == '\0') {
90 /* this is a local to local talk */
91 his_name = names;
92 his_machine_name = my_machine_name;
93 } else {
94 if (*cp++ == '@') {
95 /* user@host */
96 his_name = names;
97 his_machine_name = cp;
98 } else {
99 /* host.user or host!user or host:user */
100 his_name = cp;
101 his_machine_name = names;
102 }
103 *--cp = '\0';
104 }
105 if (argc > 2)
106 his_tty = argv[2]; /* tty name is arg 2 */
107 else
108 his_tty = "";
109 get_addrs(my_machine_name, his_machine_name);
110 /*
111 * Initialize the message template.
112 */
113 msg.vers = TALK_VERSION;
114 msg.addr.sa_family = htons(AF_INET);
115 msg.ctl_addr.sa_family = htons(AF_INET);
116 msg.id_num = htonl(0);
117 strncpy(msg.l_name, my_name, NAME_SIZE);
118 msg.l_name[NAME_SIZE - 1] = '\0';
119 strncpy(msg.r_name, his_name, NAME_SIZE);
120 msg.r_name[NAME_SIZE - 1] = '\0';
121 strncpy(msg.r_tty, his_tty, TTY_SIZE);
122 msg.r_tty[TTY_SIZE - 1] = '\0';
123 free(names);
124}
