45/*
46 * See if the local daemon has an invitation for us.
47 */
48int
49check_local(void)
50{
51 CTL_RESPONSE response;
52 CTL_RESPONSE *rp = &response;
53 struct sockaddr addr;
54
55 /* the rest of msg was set up in get_names */
56#ifdef MSG_EOR
57 /* copy new style sockaddr to old, swap family (short in old) */
58 msg.ctl_addr = *(struct talkd_sockaddr *)(void *)&ctl_addr;
59 msg.ctl_addr.sa_family = htons(ctl_addr.sin_family);
60#else
61 msg.ctl_addr = *(struct sockaddr *)&ctl_addr;
62#endif
63 /* must be initiating a talk */
64 if (!look_for_invite(rp))
65 return (0);
66 /*
67 * There was an invitation waiting for us,
68 * so connect with the other (hopefully waiting) party
69 */
70 current_state = "Waiting to connect with caller";
71 do {
72 if (rp->addr.sa_family != AF_INET)
73 p_error("Response uses invalid network address");
74
75 (void)memcpy(&addr, &rp->addr.sa_family, sizeof(addr));
76 addr.sa_family = rp->addr.sa_family;
77 addr.sa_len = sizeof(addr);
78
79 errno = 0;
80 if (connect(sockt, &addr, sizeof(addr)) != -1)
81 return (1);
82 } while (errno == EINTR);
83 if (errno == ECONNREFUSED) {
84 /*
85 * The caller gave up, but his invitation somehow
86 * was not cleared. Clear it and initiate an
87 * invitation. (We know there are no newer invitations,
88 * the talkd works LIFO.)
89 */
90 ctl_transact(his_machine_addr, msg, DELETE, rp);
91 close(sockt);
92 open_sockt();
93 return (0);
94 }
95 p_error("Unable to connect with initiator");
96 /*NOTREACHED*/
97 return (0);
98}
99
100/*
101 * Look for an invitation on 'machine'
102 */
103int
104look_for_invite(CTL_RESPONSE *rp)
105{
106 current_state = "Checking for invitation on caller's machine";
107 ctl_transact(his_machine_addr, msg, LOOK_UP, rp);
108 /* the switch is for later options, such as multiple invitations */
109 switch (rp->answer) {
110
111 case SUCCESS:
112 msg.id_num = htonl(rp->id_num);
113 return (1);
114
115 default:
116 /* there wasn't an invitation waiting for us */
117 return (0);
118 }
119}
