40/*
41 * A package to display what is happening every MSG_INTERVAL seconds
42 * if we are slow connecting.
43 */
44
45#include <sys/time.h>
46#include <signal.h>
47#include <stdio.h>
48#include "talk.h"
49
50#define MSG_INTERVAL 4
51
52const char *current_state;
53int current_line = 0;
54
55void
56disp_msg(int dummy)
57{
58 message(current_state);
59}
60
61void
62start_msgs(void)
63{
64 struct itimerval itimer;
65
66 message(current_state);
67 signal(SIGALRM, disp_msg);
68 itimer.it_value.tv_sec = itimer.it_interval.tv_sec = MSG_INTERVAL;
69 itimer.it_value.tv_usec = itimer.it_interval.tv_usec = 0;
70 setitimer(ITIMER_REAL, &itimer, (struct itimerval *)0);
71}
72
73void
74end_msgs(void)
75{
76 struct itimerval itimer;
77
78 timerclear(&itimer.it_value);
79 timerclear(&itimer.it_interval);
80 setitimer(ITIMER_REAL, &itimer, (struct itimerval *)0);
81 signal(SIGALRM, SIG_DFL);
82}
