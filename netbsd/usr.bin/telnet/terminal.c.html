63void
64init_terminal(void)
65{
66 if (ring_init(&ttyoring, ttyobuf, sizeof ttyobuf) != 1) {
67 exit(1);
68 }
69 if (ring_init(&ttyiring, ttyibuf, sizeof ttyibuf) != 1) {
70 exit(1);
71 }
72 autoflush = TerminalAutoFlush();
73}

76/*
77 * Send as much data as possible to the terminal, else exits if
78 * it encounters a permanent failure when writing to the tty.
79 *
80 * Return value:
81 * -1: No useful work done, data waiting to go out.
82 * 0: No data was waiting, so nothing was done.
83 * 1: All waiting data was written out.
84 * n: All data - n was written out.
85 */
86
87
88int
89ttyflush(int drop)
90{
91 int n, n0, n1;
92
93 n0 = ring_full_count(&ttyoring);
94 if ((n1 = n = ring_full_consecutive(&ttyoring)) > 0) {
95 if (drop) {
96 TerminalFlushOutput();
97 /* we leave 'n' alone! */
98 } else {
99 n = TerminalWrite(ttyoring.consume, n);
100 }
101 }
102 if (n > 0) {
103 if (termdata && n) {
104 Dump('>', ttyoring.consume, n);
105 }
106 /*
107 * If we wrote everything, and the full count is
108 * larger than what we wrote, then write the
109 * rest of the buffer.
110 */
111 if (n1 == n && n0 > n) {
112 n1 = n0 - n;
113 if (!drop)
114 n1 = TerminalWrite(ttyoring.bottom, n1);
115 if (n1 > 0)
116 n += n1;
117 }
118 ring_consumed(&ttyoring, n);
119 }
120 if (n < 0) {
121 if (errno == EAGAIN || errno == EINTR) {
122 return -1;
123 } else {
124 ring_consumed(&ttyoring, ring_full_count(&ttyoring));
125 setconnmode(0);
126 setcommandmode();
127 NetClose(net);
128 fprintf(stderr, "Connection closed by foreign host.\n");
129 exit(1);
130 }
131 }
132 if (n == n0) {
133 if (n0)
134 return -1;
135 return 0;
136 }
137 return n0 - n + 1;
138}

142 * These routines decides on what the mode should be (based on the values
143 * of various global variables).
144 */

146
147int
148getconnmode(void)
149{
150 extern int linemode;
151 int mode = 0;
152#ifdef KLUDGELINEMODE
153 extern int kludgelinemode;
154#endif
155
156 if (my_want_state_is_dont(TELOPT_ECHO))
157 mode |= MODE_ECHO;
158
159 if (localflow)
160 mode |= MODE_FLOW;
161
162 if (my_want_state_is_will(TELOPT_BINARY))
163 mode |= MODE_INBIN;
164
165 if (his_want_state_is_will(TELOPT_BINARY))
166 mode |= MODE_OUTBIN;
167
168#ifdef KLUDGELINEMODE
169 if (kludgelinemode) {
170 if (my_want_state_is_dont(TELOPT_SGA)) {
171 mode |= (MODE_TRAPSIG|MODE_EDIT);
172 if (dontlecho && (clocks.echotoggle > clocks.modenegotiated)) {
173 mode &= ~MODE_ECHO;
174 }
175 }
176 return(mode);
177 }
178#endif
179 if (my_want_state_is_will(TELOPT_LINEMODE))
180 mode |= linemode;
181 return(mode);
182}
183
184void
185setconnmode(int force)
186{
187#ifdef ENCRYPTION
188 static int enc_passwd = 0;
189#endif
190 int newmode;
191
192 newmode = getconnmode()|(force?MODE_FORCE:0);
193
194 TerminalNewMode(newmode);
195
196#ifdef ENCRYPTION
197 if ((newmode & (MODE_ECHO|MODE_EDIT)) == MODE_EDIT) {
198 if (my_want_state_is_will(TELOPT_ENCRYPT)
199 && (enc_passwd == 0) && !encrypt_output) {
200 encrypt_request_start(0, 0);
201 enc_passwd = 1;
202 }
203 } else {
204 if (enc_passwd) {
205 encrypt_request_end();
206 enc_passwd = 0;
207 }
208 }
209#endif /* ENCRYPTION */
210}

213void
214setcommandmode(void)
215{
216 TerminalNewMode(-1);
217}
