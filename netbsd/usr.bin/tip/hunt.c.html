42static jmp_buf deadline;
43static int deadfl;
44
45__dead static void dead(int);
46
47static void
48/*ARGSUSED*/
49dead(int dummy __unused)
50{
51 deadfl = 1;
52 longjmp(deadline, 1);
53}
54
55int
56hunt(char *name)
57{
58 char *cp;
59 sig_t f;
60
61 f = signal(SIGALRM, dead);
62 while ((cp = getremote(name)) != NULL) {
63 deadfl = 0;
64 /*
65 * Straight through call units, such as the BIZCOMP,
66 * VADIC and the DF, must indicate they're hardwired in
67 * order to get an open file descriptor placed in FD.
68 * Otherwise, as for a DN-11, the open will have to
69 * be done in the "open" routine.
70 */
71 if (!HW)
72 break;
73 if (setjmp(deadline) == 0) {
74 (void)alarm(10);
75 FD = open(cp, (O_RDWR | (DC ? O_NONBLOCK : 0)));
76 }
77 (void)alarm(0);
78 if (FD < 0) {
79 warn("%s", cp);
80 deadfl = 1;
81 } else if (!deadfl) {
82 struct termios cntrl;
83
84 if (flock(FD, (LOCK_EX|LOCK_NB)) != 0) {
85 (void)close(FD);
86 FD = -1;
87 continue;
88 }
89
90 (void)tcgetattr(FD, &cntrl);
91 if (!DC)
92 cntrl.c_cflag |= HUPCL;
93 (void)tcsetattr(FD, TCSAFLUSH, &cntrl);
94 (void)ioctl(FD, TIOCEXCL, 0);
95 (void)signal(SIGALRM, SIG_DFL);
96 return (cp != NULL);
97 }
98 }
99 (void)signal(SIGALRM, f);
100 return (deadfl ? -1 : cp != NULL);
101}
