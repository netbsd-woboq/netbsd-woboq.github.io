51static void process(FILE *, const char *, int);
52
53int
54main(int argc, char *argv[])
55{
56 FILE *fp;
57 int ch, eflags = 0;
58
59 setprogname(argv[0]);
60 while ((ch = getopt(argc, argv, "eHhm")) != -1)
61 switch((char)ch) {
62 case 'e':
63 eflags |= VIS_NOESCAPE;
64 break;
65 case 'H':
66 eflags |= VIS_HTTP1866;
67 break;
68 case 'h':
69 eflags |= VIS_HTTP1808;
70 break;
71 case 'm':
72 eflags |= VIS_MIMESTYLE;
73 break;
74 case '?':
75 default:
76 (void)fprintf(stderr,
77 "Usage: %s [-e] [-Hh | -m] [file...]\n",
78 getprogname());
79 return EXIT_FAILURE;
80 }
81 argc -= optind;
82 argv += optind;
83
84 switch (eflags & (VIS_HTTP1808|VIS_HTTP1866|VIS_MIMESTYLE)) {
85 case VIS_HTTP1808|VIS_MIMESTYLE:
86 case VIS_HTTP1866|VIS_MIMESTYLE:
87 case VIS_HTTP1808|VIS_HTTP1866|VIS_MIMESTYLE:
88 errx(EXIT_FAILURE, "Can't mix -m with -h and/or -H");
89 /*NOTREACHED*/
90 default:
91 break;
92 }
93
94 if (*argv)
95 while (*argv) {
96 if ((fp = fopen(*argv, "r")) != NULL)
97 process(fp, *argv, eflags);
98 else
99 warn("%s", *argv);
100 argv++;
101 }
102 else
103 process(stdin, "<stdin>", eflags);
104 return EXIT_SUCCESS;
105}
106
107static void
108process(FILE *fp, const char *filename, int eflags)
109{
110 int offset = 0, c, ret;
111 int state = 0;
112 char outc;
113
114 while ((c = getc(fp)) != EOF) {
115 offset++;
116 again:
117 switch(ret = unvis(&outc, (char)c, &state, eflags)) {
118 case UNVIS_VALID:
119 (void)putchar(outc);
120 break;
121 case UNVIS_VALIDPUSH:
122 (void)putchar(outc);
123 goto again;
124 case UNVIS_SYNBAD:
125 warnx("%s: offset: %d: can't decode", filename, offset);
126 state = 0;
127 break;
128 case 0:
129 case UNVIS_NOCHAR:
130 break;
131 default:
132 errx(1, "bad return value (%d), can't happen", ret);
133 /* NOTREACHED */
134 }
135 }
136 if (unvis(&outc, (char)0, &state, eflags | UNVIS_END) == UNVIS_VALID)
137 (void)putchar(outc);
138}
