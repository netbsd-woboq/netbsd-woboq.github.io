39/* poseur disk entry to hold the information we're interested in. */
40struct _drive {
41 int *select; /* Display stats for selected disks. */
42 char **name; /* Disk names (sd0, wd1, etc). */
43 u_int64_t *rxfer; /* # of read transfers. */
44 u_int64_t *wxfer; /* # of write transfers. */
45 u_int64_t *seek; /* # of seeks (currently unused). */
46 u_int64_t *rbytes; /* # of bytes read. */
47 u_int64_t *wbytes; /* # of bytes written. */
48 int *scale; /* Sticky scale for bytes */
49 struct timeval *time; /* Time spent in disk i/o. */
50 struct timeval *wait; /* Time spent in queue waiting. */
51 struct timeval *busysum; /* Time busy * queue length */
52 struct timeval *waitsum; /* Time waiting * queue length */
53 struct timeval *timestamp; /* Disk sample time */
54 u_int64_t tk_nin; /* TTY Chars in. */
55 u_int64_t tk_nout; /* TTY Chars out. */
56 u_int64_t cp_time[CPUSTATES]; /* System timer ticks. */
57 int cp_ncpu; /* Number of cpu's */
58 double cp_etime; /* Elapsed time */
59};

62extern char **dr_name;
63extern int *drv_select;
64extern size_t ndrive;
