51/*
52 * pr_attime --
53 * Print the time since the user logged in.
54 *
55 * Note: SCCS forces the bizarre string manipulation, things like
56 * %I% get replaced in the source code.
57 */
58void
59pr_attime(time_t *started, time_t *now)
60{
61 static char buf[256];
62 int tnow_yday;
63 struct tm *tp;
64 time_t diff;
65
66 tnow_yday = localtime(now)->tm_yday;
67 tp = localtime(started);
68 diff = *now - *started;
69
70 if (diff > SECSPERDAY * DAYSPERWEEK) {
71 /* If more than a week, use day-month-year. */
72 (void)strftime(buf, sizeof(buf), "%d%b%y", tp);
73 } else if (tp->tm_yday != tnow_yday) {
74 /* If not today, use day-hour-am/pm. Damn SCCS */
75 (void)strftime(buf, sizeof(buf), "%a%" "I%p", tp);
76 } else {
77 /* Default is hh:mm{am,pm}. Damn SCCS */
78 (void)strftime(buf, sizeof(buf), "%l:%" "M%p", tp);
79 }
80
81 buf[sizeof(buf) - 1] = '\0';
82 (void)fputs(buf, stdout);
83}

85/*
86 * pr_idle --
87 * Display the idle time.
88 */
89void
90pr_idle(time_t idle)
91{
92 int days;
93
94 if (idle == (time_t)-1) {
95 (void)printf(" ? ");
96 return;
97 }
98
99 days = idle / SECSPERDAY;
100
101 /* If idle more than 36 hours, print as a number of days. */
102 if (idle >= 48 * SECSPERHOUR)
103 printf(" %ddays ", days);
104 else if (idle >= 36 * SECSPERHOUR)
105 printf(" 1day ");
106
107 /* If idle more than an hour, print as HH:MM. */
108 else if (idle >= SECSPERHOUR)
109 (void)printf(" %2d:%02d ",
110 (int)(idle / SECSPERHOUR),
111 (int)((idle % SECSPERHOUR) / SECSPERMIN));
112
113 /* Else print the minutes idle. */
114 else
115 (void)printf(" %2d ", (int)(idle / SECSPERMIN));
116}
