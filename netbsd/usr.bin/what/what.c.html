50static void search(void);
51__dead static void usage(void);
52
53static int matches;
54static int sflag;
55
56/*
57 * what
58 */
59int
60main(int argc, char **argv)
61{
62 int c;
63
64 (void)setlocale(LC_ALL, "");
65
66 matches = sflag = 0;
67 while ((c = getopt(argc, argv, "s")) != -1) {
68 switch (c) {
69 case 's':
70 sflag = 1;
71 break;
72 case '?':
73 default:
74 usage();
75 }
76 }
77 argc -= optind;
78 argv += optind;
79
80 if (argc < 1) {
81 usage();
82 } else do {
83 if (freopen(*argv, "r", stdin) == NULL) {
84 perror(*argv);
85 exit(matches ? EXIT_SUCCESS : 1);
86 }
87 printf("%s\n", *argv);
88 search();
89 } while (*++argv != NULL);
90
91 /* Note: the standard explicitly specifies an exit status of 1. */
92 exit(matches ? EXIT_SUCCESS : 1);
93 /* NOTREACHED */
94}
95
96static void
97search(void)
98{
99 int c;
100
101 while ((c = getchar()) != EOF) {
102loop: if (c != '@')
103 continue;
104 if ((c = getchar()) != '(')
105 goto loop;
106 if ((c = getchar()) != '#')
107 goto loop;
108 if ((c = getchar()) != ')')
109 goto loop;
110 putchar('\t');
111 while ((c = getchar()) != EOF && c != '\0' && c != '"' &&
112 c != '>' && c != '\n' && c != '\\')
113 putchar(c);
114 putchar('\n');
115 matches++;
116 if (sflag)
117 break;
118 }
119}

121static void
122usage(void)
123{
124
125 (void)fprintf(stderr, "usage: what [-s] file ...\n");
126 exit(1);
127}
