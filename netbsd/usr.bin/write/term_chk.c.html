54/*
55 * term_chk - check that a terminal exists, and get the message bit
56 * and the access time
57 */
58int
59term_chk(uid_t uid, const char *tty, int *msgsokP, time_t *atimeP, int ismytty,
60 gid_t saved_egid)
61{
62 char path[MAXPATHLEN];
63 struct stat s;
64 int i, fd, serrno;
65
66 if (strstr(tty, "../") != NULL) {
67 errno = EINVAL;
68 return -1;
69 }
70 i = snprintf(path, sizeof path, _PATH_DEV "%s", tty);
71 if (i < 0 || i >= (int)sizeof(path)) {
72 errno = ENOMEM;
73 return -1;
74 }
75
76 (void)setegid(saved_egid);
77 fd = open(path, O_WRONLY, 0);
78 serrno = errno;
79 (void)setegid(getgid());
80 errno = serrno;
81
82 if (fd == -1)
83 return(-1);
84 if (fstat(fd, &s) == -1)
85 goto error;
86 if (!isatty(fd))
87 goto error;
88 if (s.st_uid != uid && uid != 0) {
89 errno = EPERM;
90 goto error;
91 }
92 if (msgsokP)
93 *msgsokP = (s.st_mode & S_IWGRP) != 0; /* group write bit */
94 if (atimeP)
95 *atimeP = s.st_atime;
96 if (ismytty)
97 (void)close(fd);
98 return ismytty ? 0 : fd;
99error:
100 if (fd != -1) {
101 serrno = errno;
102 (void)close(fd);
103 errno = serrno;
104 }
105 return -1;
106}

108char *
109check_sender(time_t *atime, uid_t myuid, gid_t saved_egid)
110{
111 int myttyfd;
112 int msgsok;
113 char *mytty;
114
115 /* check that sender has write enabled */
116 if (isatty(fileno(stdin)))
117 myttyfd = fileno(stdin);
118 else if (isatty(fileno(stdout)))
119 myttyfd = fileno(stdout);
120 else if (isatty(fileno(stderr)))
121 myttyfd = fileno(stderr);
122 else if (atime == NULL)
123 return NULL;
124 else
125 errx(1, "Cannot find your tty");
126 if ((mytty = ttyname(myttyfd)) == NULL)
127 err(1, "Cannot find the name of your tty");
128 if (strncmp(mytty, _PATH_DEV, sizeof(_PATH_DEV) - 1) == 0)
129 mytty += sizeof(_PATH_DEV) - 1;
130 if (term_chk(myuid, mytty, &msgsok, atime, 1, saved_egid) == -1)
131 err(1, "%s%s", _PATH_DEV, mytty);
132 if (!msgsok) {
133 warnx(
134 "You have write permission turned off; no reply possible");
135 }
136 return mytty;
137}
