40/*
41 * Should be set to 1 if the difference of two pointers is of type long
42 * or the value of sizeof is of type unsigned long. Note this MUST be
43 * kept in sync with the compiler!
44 */
45
46#define PTRDIFF_IS_LONG 1
47#define SIZEOF_IS_ULONG 1
48#define INTPTR_IS_LONG 1
49
50#define FLOAT_SIZE (4 * CHAR_BIT)
51#define DOUBLE_SIZE (8 * CHAR_BIT)
52#define LDOUBLE_SIZE (16 * CHAR_BIT)
53
54#define ENUM_SIZE (4 * CHAR_BIT)
