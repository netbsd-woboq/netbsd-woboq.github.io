49/* name and handle of output file */
50static const char *loname;
51static FILE *lout;
52
53/* output buffer data */
54ob_t ob;
55
56static void outxbuf(void);
57
58
59/*
60 * initialize output
61 */
62void
63outopen(const char *name)
64{
65
66 loname = name;
67
68 /* Open output file */
69 if ((lout = fopen(name, "w")) == NULL)
70 err(1, "cannot open '%s'", name);
71
72 /* Create output buffer */
73 ob.o_len = 1024;
74 ob.o_end = (ob.o_buf = ob.o_nxt = xmalloc(ob.o_len)) + ob.o_len;
75}
76
77/*
78 * flush output buffer and close file
79 */
80void
81outclose(void)
82{
83
84 outclr();
85 if (fclose(lout) == EOF)
86 err(1, "cannot close '%s'", loname);
87}
88
89/*
90 * resize output buffer
91 */
92static void
93outxbuf(void)
94{
95 ptrdiff_t coffs;
96
97 coffs = ob.o_nxt - ob.o_buf;
98 ob.o_len *= 2;
99 ob.o_end = (ob.o_buf = xrealloc(ob.o_buf, ob.o_len)) + ob.o_len;
100 ob.o_nxt = ob.o_buf + coffs;
101}
102
103/*
104 * reset output buffer
105 * if it is not empty, it is flushed
106 */
107void
108outclr(void)
109{
110 size_t sz;
111
112 if (ob.o_buf != ob.o_nxt) {
113 outchar('\n');
114 sz = ob.o_nxt - ob.o_buf;
115 if (sz > ob.o_len)
116 errx(1, "internal error: outclr() 1");
117 if (fwrite(ob.o_buf, sz, 1, lout) != 1)
118 err(1, "cannot write to %s", loname);
119 ob.o_nxt = ob.o_buf;
120 }
121}
122
123/*
124 * write a character to the output buffer
125 */
126void
127outchar(int c)
128{
129
130 if (ob.o_nxt == ob.o_end)
131 outxbuf();
132 *ob.o_nxt++ = (char)c;
133}
134
135/*
136 * write a character to the output buffer, qouted if necessary
137 */
138void
139outqchar(int c)
140{
141
142 if (isprint(c) && c != '\\' && c != '"' && c != '\'') {
143 outchar(c);
144 } else {
145 outchar('\\');
146 switch (c) {
147 case '\\':
148 outchar('\\');
149 break;
150 case '"':
151 outchar('"');
152 break;
153 case '\'':
154 outchar('\'');
155 break;
156 case '\b':
157 outchar('b');
158 break;
159 case '\t':
160 outchar('t');
161 break;
162 case '\n':
163 outchar('n');
164 break;
165 case '\f':
166 outchar('f');
167 break;
168 case '\r':
169 outchar('r');
170 break;
171 case '\v':
172 outchar('v');
173 break;
174 case '\a':
175 outchar('a');
176 break;
177 default:
178 outchar((((u_int)c >> 6) & 07) + '0');
179 outchar((((u_int)c >> 3) & 07) + '0');
180 outchar((c & 07) + '0');
181 break;
182 }
183 }
184}

186/*
187 * write a strint to the output buffer
188 * the string must not contain any characters which
189 * should be quoted
190 */
191void
192outstrg(const char *s)
193{
194
195 while (*s != '\0') {
196 if (ob.o_nxt == ob.o_end)
197 outxbuf();
198 *ob.o_nxt++ = *s++;
199 }
200}
201
202/*
203 * write an integer value to toe output buffer
204 */
205void
206outint(int i)
207{
208
209 if ((size_t)(ob.o_end - ob.o_nxt) < 3 * sizeof (int))
210 outxbuf();
211 ob.o_nxt += sprintf(ob.o_nxt, "%d", i);
212}

214/*
215 * write the name of a symbol to the output buffer
216 * the name is preceded by its length
217 */
218void
219outname1(const char *file, size_t line, const char *name)
220{
221
222 if (name == NULL)
223 errx(1, "%s, %zu: internal error: outname(NULL)", file, line);
224 outint((int)strlen(name));
225 outstrg(name);
226}
227
228/*
229 * write the name of the .c source
230 */
231void
232outsrc(const char *name)
233{
234
235 outclr();
236 outchar('S');
237 outstrg(name);
238}
