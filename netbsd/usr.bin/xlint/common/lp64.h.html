38/*
39 * Type sizes for LP64 platforms (long, pointer: 64-bit)
40 */
41
42#define CHAR_SIZE (CHAR_BIT)
43#define SHORT_SIZE (2 * CHAR_BIT)
44#define INT_SIZE (4 * CHAR_BIT)
45#define LONG_SIZE (8 * CHAR_BIT)
46#define QUAD_SIZE (8 * CHAR_BIT)
47#define PTR_SIZE (8 * CHAR_BIT)
48#ifdef _LP64
49#define INT128_SIZE (16 * CHAR_BIT)
50#endif
51
52#define TARG_SCHAR_MAX ((signed char) (((unsigned char) -1) >> 1))
53#define TARG_SCHAR_MIN ((-TARG_CHAR_MAX) - 1)
54#define TARG_UCHAR_MAX ((unsigned char) -1)
55
56#define TARG_SHRT_MAX ((int16_t) (((uint16_t) -1) >> 1))
57#define TARG_SHRT_MIN ((-TARG_SHRT_MAX) - 1)
58#define TARG_USHRT_MAX ((uint16_t) -1)
59
60#define TARG_INT_MAX ((int32_t) (((uint32_t) -1) >> 1))
61#define TARG_INT_MIN ((-TARG_INT_MAX) - 1)
62#define TARG_UINT_MAX ((uint32_t) -1)
63
64#define TARG_LONG_MAX TARG_QUAD_MAX
65#define TARG_LONG_MIN TARG_QUAD_MIN
66#define TARG_ULONG_MAX TARG_UQUAD_MAX
67
68#define TARG_QUAD_MAX ((int64_t) (((uint64_t) -1) >> 1))
69#define TARG_QUAD_MIN ((-TARG_QUAD_MAX) - 1)
70#define TARG_UQUAD_MAX ((uint64_t) -1)
71
72#ifdef _LP64
73/* XXX on a 32 build for a 64 build host we skip these */
74#define TARG_INT128_MAX ((__int128_t) (((__uint128_t) -1) >> 1))
75#define TARG_INT128_MIN ((-TARG_INT128_MAX) - 1)
76#define TARG_UINT128_MAX ((__uint128_t) -1)
77#endif
