406/*
407 * If Fflag is not set lbasename() returns a pointer to the last
408 * component of the path, otherwise it returns the argument.
409 */
410static const char *
411lbasename(const char *path)
412{
413 const char *cp, *cp1, *cp2;
414
415 if (Fflag)
416 return (path);
417
418 cp = cp1 = cp2 = path;
419 while (*cp != '\0') {
420 if (*cp++ == '/') {
421 cp2 = cp1;
422 cp1 = cp;
423 }
424 }
425 return (*cp1 == '\0' ? cp2 : cp1);
426}

428static void
429verror( int n, va_list ap)
430{
431 const char *fn;
432
433 if (ERR_ISSET(n, &msgset))
434 return;
435
436 fn = lbasename(curr_pos.p_file);
437 (void)printf("%s(%d): ", fn, curr_pos.p_line);
438 (void)vprintf(msgs[n], ap);
439 (void)printf(" [%d]\n", n);
440 nerr++;
441}
442
443static void
444vwarning(int n, va_list ap)
445{
446 const char *fn;
447
448 if (ERR_ISSET(n, &msgset))
449 return;
450
451#ifdef DEBUG
452 printf("%s: lwarn=%d n=%d\n", __func__, lwarn, n);
453#endif
454 if (lwarn == LWARN_NONE || lwarn == n)
455 /* this warning is suppressed by a LINTED comment */
456 return;
457
458 fn = lbasename(curr_pos.p_file);
459 (void)printf("%s(%d): warning: ", fn, curr_pos.p_line);
460 (void)vprintf(msgs[n], ap);
461 (void)printf(" [%d]\n", n);
462 if (wflag)
463 nerr++;
464}

476void
477lerror(const char *file, int line, const char *msg, ...)
478{
479 va_list ap;
480 const char *fn;
481
482 va_start(ap, msg);
483 fn = lbasename(curr_pos.p_file);
484 (void)fprintf(stderr, "%s(%d): lint error: %s, %d: ",
485 fn, curr_pos.p_line, file, line);
486 (void)vfprintf(stderr, msg, ap);
487 (void)fprintf(stderr, "\n");
488 va_end(ap);
489 abort();
490}

502void
503message(int n, ...)
504{
505 va_list ap;
506 const char *fn;
507
508 if (ERR_ISSET(n, &msgset))
509 return;
510
511 va_start(ap, n);
512 fn = lbasename(curr_pos.p_file);
513 (void)printf("%s(%d): ", fn, curr_pos.p_line);
514 (void)vprintf(msgs[n], ap);
515 (void)printf(" [%d]\n", n);
516 va_end(ap);
517}

519/*
520 * XXX I think the logic is possibly somewhat screwed up here. The
521 * question is, how do we want to interpret the -s and -S flags going
522 * forward? We need to answer that and then we can fix this to be
523 * "right"... [perry, 2 Nov 2002]
524*/
525int
526c99ism(int n, ...)
527{
528 va_list ap;
529 int msg;
530
531 va_start(ap, n);
532 if (sflag && !(Sflag || gflag)) {
533 verror(n, ap);
534 msg = 1;
535 } else if (!sflag && (Sflag || gflag)) {
536 msg = 0;
537 } else {
538 vwarning(n, ap);
539 msg = 1;
540 }
541 va_end(ap);
542
543 return (msg);
544}

546int
547gnuism(int n, ...)
548{
549 va_list ap;
550 int msg;
551
552 va_start(ap, n);
553 if (sflag && !gflag) {
554 verror(n, ap);
555 msg = 1;
556 } else if (!sflag && gflag) {
557 msg = 0;
558 } else {
559 vwarning(n, ap);
560 msg = 1;
561 }
562 va_end(ap);
563
564 return (msg);
565}
