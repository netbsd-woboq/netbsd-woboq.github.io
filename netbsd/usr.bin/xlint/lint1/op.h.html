34/*
35 * Various information about operators
36 */
37typedef struct {
38 u_int m_binary : 1; /* binary op. */
39 u_int m_logop : 1; /* logical op., result is int */
40 u_int m_rqint : 1; /* operands must have integer type */
41 u_int m_rqsclt : 1; /* operands must have scalar type */
42 u_int m_rqatyp : 1; /* operands must have arithmetic type */
43 u_int m_fold : 1; /* operands should be folded */
44 u_int m_vctx : 1; /* value context for left operand */
45 u_int m_tctx : 1; /* test context for left operand */
46 u_int m_balance : 1; /* op. requires balancing */
47 u_int m_sideeff : 1; /* op. has side effect */
48 u_int m_tlansiu : 1; /* warning if left op. is unsign. in ANSI C */
49 u_int m_transiu : 1; /* warning if right op. is unsign. in ANSI C */
50 u_int m_tpconf : 1; /* test possible precedence confusion */
51 u_int m_comp : 1; /* op. performs comparison */
52 u_int m_enumop : 1; /* valid operation on enums */
53 u_int m_badeop : 1; /* dubious operation on enums */
54 u_int m_eqwarn : 1; /* warning if on operand stems from == */
55 u_int m_rqintcomp : 1;/* operands must be integer or complex */
56 const char *m_name; /* name of op. */
57} mod_t;
