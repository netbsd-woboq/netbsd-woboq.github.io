37/*
38 * Types are described by structures of type type_t.
39 */
40struct type {
41 tspec_t t_tspec; /* type specifier */
42 u_int t_const : 1; /* constant */
43 u_int t_volatile : 1; /* volatile */
44 u_int t_vararg : 1; /* function has variable number of arguments */
45 u_int t_isenum : 1; /* enum type */
46 u_int t_proto : 1; /* this is a prototype */
47 u_int t_istag : 1; /* tag with _t_tag valid */
48 u_int t_istynam : 1; /* tag with _t_tynam valid */
49 u_int t_isuniqpos : 1; /* tag with _t_uniqpos valid */
50 union {
51 int _t_dim; /* if the type is an ARRAY than this
52 is the dimension of the array. */
53 struct hte *_t_tag; /* hash table entry of tag if
54 t_isenum, STRUCT or UNION */
55 struct hte *_t_tynam; /* hash table entry of typename if
56 t_isenum, STRUCT or UNION */
57 struct {
58 int p_line;
59 short p_file;
60 int p_uniq;
61 } _t_uniqpos; /* unique position, for untagged
62 untyped STRUCTs, UNIONS, and ENUMs,
63 if t_isuniqpos */
64 struct type **_t_args; /* list of argument types if this
65 is a prototype */
66 } t_u;
67 struct type *t_subt; /* indirected type (array element, pointed to
68 type, type of return value) */
69};
70
71#define t_dim t_u._t_dim
72#define t_tag t_u._t_tag
73#define t_tynam t_u._t_tynam
74#define t_uniqpos t_u._t_uniqpos
75#define t_args t_u._t_args

77/*
78 * argument information
79 *
80 * Such a structure is created for each argument of a function call
81 * which is an integer constant or a constant string.
82 */
83typedef struct arginf {
84 int a_num; /* # of argument (1..) */
85 u_int a_zero : 1; /* argument is 0 */
86 u_int a_pcon : 1; /* msb of argument is not set */
87 u_int a_ncon : 1; /* msb of argument is set */
88 u_int a_fmt : 1; /* a_fstrg points to format string */
89 char *a_fstrg; /* format string */
90 struct arginf *a_nxt; /* information for next const. argument */
91} arginf_t;
92
93/*
94 * Keeps information about position in source file.
95 */
96typedef struct {
97 u_short p_src; /* index of name of translation unit
98 (the name which was specified at the
99 command line) */
100 u_short p_line; /* line number in p_src */
101 u_short p_isrc; /* index of (included) file */
102 u_short p_iline; /* line number in p_iline */
103} pos_t;
104
105/*
106 * Used for definitions and declarations
107 *
108 * To save memory, variable sized structures are used. If
109 * all s_va, s_prfl and s_scfl are not set, the memory allocated
110 * for a symbol is only large enough to keep the first member of
111 * struct sym, s_s.
112 */
113typedef struct sym {
114 struct {
115 pos_t s_pos; /* pos of def./decl. */
116#ifndef lint
117 u_int s_def : 3; /* DECL, TDEF or DEF */
118#else
119 def_t s_def;
120#endif
121 u_int s_rval : 1; /* function has return value */
122 u_int s_inline : 1; /* function is inline */
123 u_int s_osdef : 1; /* old style function definition */
124 u_int s_static : 1; /* symbol is static */
125 u_int s_va : 1; /* check only first s_nva arguments */
126 u_int s_prfl : 1; /* printflike */
127 u_int s_scfl : 1; /* scanflike */
128 u_short s_type; /* type */
129 struct sym *s_nxt; /* next symbol with same name */
130 } s_s;
131 short s_nva;
132 short s_nprfl;
133 short s_nscfl;
134} sym_t;
135
136#define s_pos s_s.s_pos
137#define s_rval s_s.s_rval
138#define s_osdef s_s.s_osdef
139#define s_inline s_s.s_inline
140#define s_static s_s.s_static
141#define s_def s_s.s_def
142#define s_va s_s.s_va
143#define s_prfl s_s.s_prfl
144#define s_scfl s_s.s_scfl
145#define s_type s_s.s_type
146#define s_nxt s_s.s_nxt
147
148/*
149 * Used to store informations about function calls.
150 */
151typedef struct fcall {
152 pos_t f_pos; /* position of call */
153 u_int f_rused : 1; /* return value used */
154 u_int f_rdisc : 1; /* return value discarded (casted to void) */
155 u_short f_type; /* types of expected return value and args */
156 arginf_t *f_args; /* information about constant arguments */
157 struct fcall *f_nxt; /* next call of same function */
158} fcall_t;
159
160/*
161 * Used to store information about usage of symbols other
162 * than for function calls.
163 */
164typedef struct usym {
165 pos_t u_pos; /* position */
166 struct usym *u_nxt; /* next usage */
167} usym_t;
168
169/*
170 * hash table entry
171 */
172typedef struct hte {
173 const char *h_name; /* name */
174 u_int h_used : 1; /* symbol is used */
175 u_int h_def : 1; /* symbol is defined */
176 u_int h_static : 1; /* static symbol */
177 sym_t *h_syms; /* declarations and definitions */
178 sym_t **h_lsym; /* points to s_nxt of last decl./def. */
179 fcall_t *h_calls; /* function calls */
180 fcall_t **h_lcall; /* points to f_nxt of last call */
181 usym_t *h_usyms; /* usage info */
182 usym_t **h_lusym; /* points to u_nxt of last usage info */
183 struct hte *h_link; /* next hte with same hash function */
184 struct hte *h_hte; /* pointer to other htes (for renames */
185} hte_t;
