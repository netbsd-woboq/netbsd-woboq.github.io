50/* length of new allocated memory blocks */
51static size_t mblklen;
52
53/* offset of next free byte in mbuf */
54static size_t nxtfree;
55
56/* current buffer to server memory requests from */
57static void *mbuf;
58
59void
60initmem(void)
61{
62 int pgsz;
63
64 pgsz = getpagesize();
65 mblklen = ((MBLKSIZ + pgsz - 1) / pgsz) * pgsz;
66
67 nxtfree = mblklen;
68}
69
70/*
71 * Allocate memory in large chunks to avoid space and time overhead of
72 * malloc(). This is possible because memory allocated by xalloc()
73 * need never to be freed.
74 */
75void *
76xalloc(size_t sz)
77{
78 void *ptr;
79
80 /* Align to at least 8 bytes. */
81 sz = (sz + 7) & ~7L;
82 if (nxtfree + sz > mblklen) {
83 /* use mmap() instead of malloc() to avoid malloc overhead. */
84 mbuf = xmapalloc(mblklen);
85 (void)memset(mbuf, 0, mblklen);
86 nxtfree = 0;
87 }
88
89 ptr = (char *)mbuf + nxtfree;
90 nxtfree += sz;
91
92 return (ptr);
93}
