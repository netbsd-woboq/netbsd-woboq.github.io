71static const char *lbasename(const char *);
72
73void
74msg(int n, ...)
75{
76 va_list ap;
77
78 va_start(ap, n);
79
80 (void)vprintf(msgs[n], ap);
81 (void)printf("\n");
82
83 va_end(ap);
84}

86/*
87 * Return a pointer to the last component of a path.
88 */
89static const char *
90lbasename(const char *path)
91{
92 const char *cp, *cp1, *cp2;
93
94 if (Fflag)
95 return (path);
96
97 cp = cp1 = cp2 = path;
98 while (*cp != '\0') {
99 if (*cp++ == '/') {
100 cp2 = cp1;
101 cp1 = cp;
102 }
103 }
104 return (*cp1 == '\0' ? cp2 : cp1);
105}

107/*
108 * Create a string which describes a position in a source file.
109 */
110const char *
111mkpos(pos_t *posp)
112{
113 size_t len;
114 const char *fn;
115 static char *buf;
116 static size_t blen = 0;
117 int qm, src, line;
118
119 if (Hflag && posp->p_src != posp->p_isrc) {
120 src = posp->p_isrc;
121 line = posp->p_iline;
122 } else {
123 src = posp->p_src;
124 line = posp->p_line;
125 }
126 qm = !Hflag && posp->p_src != posp->p_isrc;
127
128 len = strlen(fn = lbasename(fnames[src]));
129 len += 3 * sizeof (u_short) + 4;
130
131 if (len > blen)
132 buf = xrealloc(buf, blen = len);
133 if (line != 0) {
134 (void)sprintf(buf, "%s%s(%d)",
135 fn, qm ? "?" : "", line);
136 } else {
137 (void)sprintf(buf, "%s", fn);
138 }
139
140 return (buf);
141}
