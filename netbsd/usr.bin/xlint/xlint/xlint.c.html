148/*
149 * Some functions to deal with lists of strings.
150 * Take care that we get no surprises in case of asynchronous signals.
151 */
152static void
153appstrg(char ***lstp, char *s)
154{
155 char **lst, **olst;
156 int i;
157
158 olst = *lstp;
159 for (i = 0; olst[i] != NULL; i++)
160 continue;
161 lst = xrealloc(olst, (i + 2) * sizeof (char *));
162 lst[i] = s;
163 lst[i + 1] = NULL;
164 *lstp = lst;
165}
166
167static void
168appcstrg(char ***lstp, const char *s)
169{
170
171 appstrg(lstp, xstrdup(s));
172}

174static void
175applst(char ***destp, char *const *src)
176{
177 int i, k;
178 char **dest, **odest;
179
180 odest = *destp;
181 for (i = 0; odest[i] != NULL; i++)
182 continue;
183 for (k = 0; src[k] != NULL; k++)
184 continue;
185 dest = xrealloc(odest, (i + k + 1) * sizeof (char *));
186 for (k = 0; src[k] != NULL; k++)
187 dest[i + k] = xstrdup(src[k]);
188 dest[i + k] = NULL;
189 *destp = dest;
190}

192static void
193freelst(char ***lstp)
194{
195 char *s;
196 int i;
197
198 for (i = 0; (*lstp)[i] != NULL; i++)
199 continue;
200 while (i-- > 0) {
201 s = (*lstp)[i];
202 (*lstp)[i] = NULL;
203 free(s);
204 }
205}

207static char *
208concat2(const char *s1, const char *s2)
209{
210 char *s;
211
212 s = xmalloc(strlen(s1) + strlen(s2) + 1);
213 (void)strcpy(s, s1);
214 (void)strcat(s, s2);
215
216 return (s);
217}

219static char *
220concat3(const char *s1, const char *s2, const char *s3)
221{
222 char *s;
223
224 s = xmalloc(strlen(s1) + strlen(s2) + strlen(s3) + 1);
225 (void)strcpy(s, s1);
226 (void)strcat(s, s2);
227 (void)strcat(s, s3);
228
229 return (s);
230}

232/*
233 * Clean up after a signal.
234 */
235static void
236terminate(int signo)
237{
238 int i;
239
240 if (cppoutfd != -1)
241 (void)close(cppoutfd);
242 if (cppout != NULL)
243 (void)remove(cppout);
244
245 if (p1out != NULL) {
246 for (i = 0; p1out[i] != NULL; i++)
247 (void)remove(p1out[i]);
248 }
249
250 if (p2out != NULL)
251 (void)remove(p2out);
252
253 if (currfn != NULL)
254 (void)remove(currfn);
255
256 if (signo != 0)
257 (void)raise_default_signal(signo);
258 exit(signo != 0 ? 1 : 0);
259}

261/*
262 * Returns a pointer to the last component of strg after delim.
263 * Returns strg if the string does not contain delim.
264 */
265static const char *
266lbasename(const char *strg, int delim)
267{
268 const char *cp, *cp1, *cp2;
269
270 cp = cp1 = cp2 = strg;
271 while (*cp != '\0') {
272 if (*cp++ == delim) {
273 cp2 = cp1;
274 cp1 = cp;
275 }
276 }
277 return (*cp1 == '\0' ? cp2 : cp1);
278}

286}

307int
308main(int argc, char *argv[])
309{
310 int c;
311 char flgbuf[3], *tmp;
312 size_t len;
313 const char *ks;
314
315 setprogname(argv[0]);
316
317 if ((tmp = getenv("TMPDIR")) == NULL || (len = strlen(tmp)) == 0) {
318 tmpdir = xstrdup(_PATH_TMP);
319 } else {
320 char *p = xmalloc(len + 2);
321 (void)sprintf(p, "%s%s", tmp, tmp[len - 1] == '/' ? "" : "/");
322 tmpdir = p;
323 }
324
325 cppout = xmalloc(strlen(tmpdir) + sizeof ("lint0.XXXXXX"));
326 (void)sprintf(cppout, "%slint0.XXXXXX", tmpdir);
327 cppoutfd = mkstemp(cppout);
328 if (cppoutfd == -1) {
329 warn("can't make temp");
330 terminate(-1);
331 }
332
333 p1out = xcalloc(1, sizeof (char *));
334 p2in = xcalloc(1, sizeof (char *));
335 cflags = xcalloc(1, sizeof (char *));
336 lcflags = xcalloc(1, sizeof (char *));
337 l1flags = xcalloc(1, sizeof (char *));
338 l2flags = xcalloc(1, sizeof (char *));
339 l2libs = xcalloc(1, sizeof (char *));
340 deflibs = xcalloc(1, sizeof (char *));
341 libs = xcalloc(1, sizeof (char *));
342 libsrchpath = xcalloc(1, sizeof (char *));
343
344 appcstrg(&cflags, "-E");
345 appcstrg(&cflags, "-x");
346 appcstrg(&cflags, "c");
347#if 0
348 appcstrg(&cflags, "-D__attribute__(x)=");
349 appcstrg(&cflags, "-D__extension__(x)=/*NOSTRICT*/0");
350#else
351 appcstrg(&cflags, "-U__GNUC__");
352 appcstrg(&cflags, "-U__PCC__");
353 appcstrg(&cflags, "-U__SSE__");
354 appcstrg(&cflags, "-U__SSE4_1__");
355#endif
356#if 0
357 appcstrg(&cflags, "-Wp,-$");
358#endif
359 appcstrg(&cflags, "-Wp,-CC");
360 appcstrg(&cflags, "-Wcomment");
361 appcstrg(&cflags, "-D__LINT__");
362 appcstrg(&cflags, "-Dlint"); /* XXX don't def. with -s */
363
364 appdef(&cflags, "lint");
365
366 appcstrg(&deflibs, "c");
367
368 if (signal(SIGHUP, terminate) == SIG_IGN)
369 (void)signal(SIGHUP, SIG_IGN);
370 (void)signal(SIGINT, terminate);
371 (void)signal(SIGQUIT, terminate);
372 (void)signal(SIGTERM, terminate);
373 while ((c = getopt(argc, argv, "abcd:eghil:no:prstuvwxzB:C:D:FHI:L:M:PR:SU:VX:")) != -1) {
374 switch (c) {
375
376 case 'a':
377 case 'b':
378 case 'c':
379 case 'e':
380 case 'g':
381 case 'r':
382 case 'v':
383 case 'w':
384 case 'z':
385 (void)sprintf(flgbuf, "-%c", c);
386 appcstrg(&l1flags, flgbuf);
387 break;
388
389 case 'F':
390 Fflag = 1;
391 /* FALLTHROUGH */
392 case 'u':
393 case 'h':
394 (void)sprintf(flgbuf, "-%c", c);
395 appcstrg(&l1flags, flgbuf);
396 appcstrg(&l2flags, flgbuf);
397 break;
398
399 case 'X':
400 (void)sprintf(flgbuf, "-%c", c);
401 appcstrg(&l1flags, flgbuf);
402 appcstrg(&l1flags, optarg);
403 break;
404
405 case 'i':
406 if (Cflag)
407 usage();
408 iflag = 1;
409 break;
410
411 case 'n':
412 freelst(&deflibs);
413 break;
414
415 case 'p':
416 appcstrg(&l1flags, "-p");
417 appcstrg(&l2flags, "-p");
418 if (*deflibs != NULL) {
419 freelst(&deflibs);
420 appcstrg(&deflibs, "c");
421 }
422 break;
423
424 case 'P':
425 appcstrg(&l1flags, "-P");
426 break;
427
428 case 'R':
429 appcstrg(&l1flags, concat2("-R", optarg));
430 break;
431
432 case 's':
433 if (tflag)
434 usage();
435 freelst(&lcflags);
436 appcstrg(&lcflags, "-trigraphs");
437 appcstrg(&lcflags, "-Wtrigraphs");
438 appcstrg(&lcflags, "-pedantic");
439 appcstrg(&lcflags, "-D__STRICT_ANSI__");
440 appcstrg(&l1flags, "-s");
441 appcstrg(&l2flags, "-s");
442 sflag = 1;
443 break;
444
445 case 'S':
446 if (tflag)
447 usage();
448 appcstrg(&l1flags, "-S");
449 Sflag = 1;
450 break;
451
452#if ! HAVE_NBTOOL_CONFIG_H
453 case 't':
454 if (sflag)
455 usage();
456 freelst(&lcflags);
457 appcstrg(&lcflags, "-traditional");
458 appcstrg(&lcflags, "-Wtraditional");
459 appstrg(&lcflags, concat2("-D", MACHINE));
460 appstrg(&lcflags, concat2("-D", MACHINE_ARCH));
461 appcstrg(&l1flags, "-t");
462 appcstrg(&l2flags, "-t");
463 tflag = 1;
464 break;
465#endif
466
467 case 'x':
468 appcstrg(&l2flags, "-x");
469 break;
470
471 case 'C':
472 if (Cflag || oflag || iflag)
473 usage();
474 Cflag = 1;
475 appstrg(&l2flags, concat2("-C", optarg));
476 p2out = xmalloc(sizeof ("llib-l.ln") + strlen(optarg));
477 (void)sprintf(p2out, "llib-l%s.ln", optarg);
478 freelst(&deflibs);
479 break;
480
481 case 'd':
482 if (dflag)
483 usage();
484 dflag = 1;
485 appcstrg(&cflags, "-nostdinc");
486 appcstrg(&cflags, "-isystem");
487 appcstrg(&cflags, optarg);
488 break;
489
490 case 'D':
491 case 'I':
492 case 'M':
493 case 'U':
494 (void)sprintf(flgbuf, "-%c", c);
495 appstrg(&cflags, concat2(flgbuf, optarg));
496 break;
497
498 case 'l':
499 appcstrg(&libs, optarg);
500 break;
501
502 case 'o':
503 if (Cflag || oflag)
504 usage();
505 oflag = 1;
506 outputfn = xstrdup(optarg);
507 break;
508
509 case 'L':
510 appcstrg(&libsrchpath, optarg);
511 break;
512
513 case 'H':
514 appcstrg(&l2flags, "-H");
515 break;
516
517 case 'B':
518 Bflag = 1;
519 libexec_path = xstrdup(optarg);
520 break;
521
522 case 'V':
523 Vflag = 1;
524 break;
525
526 default:
527 usage();
528 /* NOTREACHED */
529 }
530 }
531 argc -= optind;
532 argv += optind;
533
534 /*
535 * To avoid modifying getopt(3)'s state engine midstream, we
536 * explicitly accept just a few options after the first source file.
537 *
538 * In particular, only -l<lib> and -L<libdir> (and these with a space
539 * after -l or -L) are allowed.
540 */
541 while (argc > 0) {
542 const char *arg = argv[0];
543
544 if (arg[0] == '-') {
545 char ***list;
546
547 list = NULL; /* XXXGCC -Wuninitialized */
548
549 /* option */
550 switch (arg[1]) {
551 case 'l':
552 list = &libs;
553 break;
554
555 case 'L':
556 list = &libsrchpath;
557 break;
558
559 default:
560 usage();
561 /* NOTREACHED */
562 }
563 if (arg[2])
564 appcstrg(list, arg + 2);
565 else if (argc > 1) {
566 argc--;
567 appcstrg(list, *++argv);
568 } else
569 usage();
570 } else {
571 /* filename */
572 fname(arg);
573 first = 0;
574 }
575 argc--;
576 argv++;
577 }
578
579 if (first)
580 usage();
581
582 if (iflag)
583 terminate(0);
584
585 if (!oflag) {
586 if ((ks = getenv("LIBDIR")) == NULL || strlen(ks) == 0)
587 ks = PATH_LINTLIB;
588 appcstrg(&libsrchpath, ks);
589 findlibs(libs);
590 findlibs(deflibs);
591 }
592
593 (void)printf("Lint pass2:\n");
594 lint2();
595
596 if (oflag)
597 cat(p2in, outputfn);
598
599 if (Cflag)
600 p2out = NULL;
601
602 terminate(0);
603 /* NOTREACHED */
604}

606/*
607 * Read a file name from the command line
608 * and pass it through lint1 if it is a C source.
609 */
610static void
611fname(const char *name)
612{
613 const char *bn, *suff;
614 char **args, *ofn, *pathname;
615 const char *CC;
616 size_t len;
617 int is_stdin;
618 int fd;
619
620 is_stdin = (strcmp(name, "-") == 0);
621 bn = lbasename(name, '/');
622 suff = lbasename(bn, '.');
623
624 if (strcmp(suff, "ln") == 0) {
625 /* only for lint2 */
626 if (!iflag)
627 appcstrg(&p2in, name);
628 return;
629 }
630
631 if (!is_stdin && strcmp(suff, "c") != 0 &&
632 (strncmp(bn, "llib-l", 6) != 0 || bn != suff)) {
633 warnx("unknown file type: %s", name);
634 return;
635 }
636
637 if (!iflag || !first)
638 (void)printf("%s:\n",
639 is_stdin ? "{standard input}" : Fflag ? name : bn);
640
641 /* build the name of the output file of lint1 */
642 if (oflag) {
643 ofn = outputfn;
644 outputfn = NULL;
645 oflag = 0;
646 } else if (iflag) {
647 if (is_stdin) {
648 warnx("-i not supported without -o for standard input");
649 return;
650 }
651 ofn = xmalloc(strlen(bn) + (bn == suff ? 4 : 2));
652 len = bn == suff ? strlen(bn) : (size_t)((suff - 1) - bn);
653 (void)sprintf(ofn, "%.*s", (int)len, bn);
654 (void)strcat(ofn, ".ln");
655 } else {
656 ofn = xmalloc(strlen(tmpdir) + sizeof ("lint1.XXXXXX"));
657 (void)sprintf(ofn, "%slint1.XXXXXX", tmpdir);
658 fd = mkstemp(ofn);
659 if (fd == -1) {
660 warn("can't make temp");
661 terminate(-1);
662 }
663 close(fd);
664 }
665 if (!iflag)
666 appcstrg(&p1out, ofn);
667
668 args = xcalloc(1, sizeof (char *));
669
670 /* run cc */
671 if ((CC = getenv("CC")) == NULL)
672 CC = DEFAULT_CC;
673 if ((pathname = findcc(CC)) == NULL)
674 if (!setenv("PATH", DEFAULT_PATH, 1))
675 pathname = findcc(CC);
676 if (pathname == NULL) {
677 (void)fprintf(stderr, "%s: %s: not found\n", getprogname(), CC);
678 exit(EXIT_FAILURE);
679 }
680
681 appcstrg(&args, pathname);
682 applst(&args, cflags);
683 applst(&args, lcflags);
684 appcstrg(&args, name);
685
686 /* we reuse the same tmp file for cpp output, so rewind and truncate */
687 if (lseek(cppoutfd, (off_t)0, SEEK_SET) != 0) {
688 warn("lseek");
689 terminate(-1);
690 }
691 if (ftruncate(cppoutfd, (off_t)0) != 0) {
692 warn("ftruncate");
693 terminate(-1);
694 }
695
696 runchild(pathname, args, cppout, cppoutfd);
697 free(pathname);
698 freelst(&args);
699
700 /* run lint1 */
701
702 if (!Bflag) {
703 pathname = xmalloc(strlen(PATH_LIBEXEC) + sizeof ("/lint1") +
704 strlen(target_prefix));
705 (void)sprintf(pathname, "%s/%slint1", PATH_LIBEXEC,
706 target_prefix);
707 } else {
708 /*
709 * XXX Unclear whether we should be using target_prefix
710 * XXX here. --thorpej@wasabisystems.com
711 */
712 pathname = xmalloc(strlen(libexec_path) + sizeof ("/lint1"));
713 (void)sprintf(pathname, "%s/lint1", libexec_path);
714 }
715
716 appcstrg(&args, pathname);
717 applst(&args, l1flags);
718 appcstrg(&args, cppout);
719 appcstrg(&args, ofn);
720
721 runchild(pathname, args, ofn, -1);
722 free(pathname);
723 freelst(&args);
724
725 appcstrg(&p2in, ofn);
726 free(ofn);
727
728 free(args);
729}

731static void
732runchild(const char *path, char *const *args, const char *crfn, int fdout)
733{
734 int status, rv, signo, i;
735
736 if (Vflag) {
737 for (i = 0; args[i] != NULL; i++)
738 (void)printf("%s ", args[i]);
739 (void)printf("\n");
740 }
741
742 currfn = crfn;
743
744 (void)fflush(stdout);
745
746 switch (vfork()) {
747 case -1:
748 warn("cannot fork");
749 terminate(-1);
750 /* NOTREACHED */
751 default:
752 /* parent */
753 break;
754 case 0:
755 /* child */
756
757 /* setup the standard output if necessary */
758 if (fdout != -1) {
759 dup2(fdout, STDOUT_FILENO);
760 close(fdout);
761 }
762 (void)execvp(path, args);
763 warn("cannot exec %s", path);
764 _exit(1);
765 /* NOTREACHED */
766 }
767
768 while ((rv = wait(&status)) == -1 && errno == EINTR) ;
769 if (rv == -1) {
770 warn("wait");
771 terminate(-1);
772 }
773 if (WIFSIGNALED(status)) {
774 signo = WTERMSIG(status);
775#if HAVE_DECL_SYS_SIGNAME
776 warnx("%s got SIG%s", path, sys_signame[signo]);
777#else
778 warnx("%s got signal %d", path, signo);
779#endif
780 terminate(-1);
781 }
782 if (WEXITSTATUS(status) != 0)
783 terminate(-1);
784 currfn = NULL;
785}

787static void
788findlibs(char *const *liblst)
789{
790 int i, k;
791 const char *lib, *path;
792 char *lfn;
793 size_t len;
794
795 lfn = NULL;
796
797 for (i = 0; (lib = liblst[i]) != NULL; i++) {
798 for (k = 0; (path = libsrchpath[k]) != NULL; k++) {
799 len = strlen(path) + strlen(lib);
800 lfn = xrealloc(lfn, len + sizeof ("/llib-l.ln"));
801 (void)sprintf(lfn, "%s/llib-l%s.ln", path, lib);
802 if (rdok(lfn))
803 break;
804 lfn = xrealloc(lfn, len + sizeof ("/lint/llib-l.ln"));
805 (void)sprintf(lfn, "%s/lint/llib-l%s.ln", path, lib);
806 if (rdok(lfn))
807 break;
808 }
809 if (path != NULL) {
810 appstrg(&l2libs, concat2("-l", lfn));
811 } else {
812 warnx("cannot find llib-l%s.ln", lib);
813 }
814 }
815
816 free(lfn);
817}

819static int
820rdok(const char *path)
821{
822 struct stat sbuf;
823
824 if (stat(path, &sbuf) == -1)
825 return (0);
826 if (!S_ISREG(sbuf.st_mode))
827 return (0);
828 if (access(path, R_OK) == -1)
829 return (0);
830 return (1);
831}

865static void
866cat(char *const *srcs, const char *dest)
867{
868 int ifd, ofd, i;
869 char *src, *buf;
870 ssize_t rlen;
871
872 if ((ofd = open(dest, O_WRONLY | O_CREAT | O_TRUNC, 0666)) == -1) {
873 warn("cannot open %s", dest);
874 terminate(-1);
875 }
876
877 buf = xmalloc(MBLKSIZ);
878
879 for (i = 0; (src = srcs[i]) != NULL; i++) {
880 if ((ifd = open(src, O_RDONLY)) == -1) {
881 free(buf);
882 warn("cannot open %s", src);
883 terminate(-1);
884 }
885 do {
886 if ((rlen = read(ifd, buf, MBLKSIZ)) == -1) {
887 free(buf);
888 warn("read error on %s", src);
889 terminate(-1);
890 }
891 if (write(ofd, buf, (size_t)rlen) == -1) {
892 free(buf);
893 warn("write error on %s", dest);
894 terminate(-1);
895 }
896 } while (rlen == MBLKSIZ);
897 (void)close(ifd);
898 }
899 (void)close(ofd);
900 free(buf);
901}
