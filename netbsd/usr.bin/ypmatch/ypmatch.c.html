51static void usage(void) __attribute__((__noreturn__));
52
53int
54main(int argc, char *argv[])
55{
56 char *domainname, *b_retry_cnt;
57 char *inkey, *outbuf;
58 const char *inmap;
59 int outbuflen, key, null, notrans;
60 int c, r, len;
61 size_t i;
62 int rval;
63 const struct ypalias *ypaliases;
64
65 setprogname(*argv);
66 domainname = b_retry_cnt = NULL;
67 notrans = key = null = 0;
68 ypaliases = ypalias_init();
69 while ((c = getopt(argc, argv, "bd:ktxz")) != -1) {
70 switch (c) {
71 case 'x':
72 for(i = 0; ypaliases[i].alias; i++)
73 printf("Use \"%s\" for \"%s\"\n",
74 ypaliases[i].alias,
75 ypaliases[i].name);
76 return 0;
77
78 case 'b':
79 b_retry_cnt = optarg;
80 break;
81
82 case 'd':
83 domainname = optarg;
84 break;
85
86 case 't':
87 notrans++;
88 break;
89
90 case 'k':
91 key++;
92 break;
93
94 case 'z':
95 null++;
96 break;
97
98 default:
99 usage();
100 }
101 }
102
103 argc -= optind;
104 argv += optind;
105
106 if (argc < 2)
107 usage();
108
109 if (b_retry_cnt != NULL) {
110 char *s;
111 unsigned long l;
112
113 l = strtoul(b_retry_cnt, &s, 10);
114 if (*s != '\0' || l > 0xffff) usage();
115 yp_setbindtries((int)l);
116 }
117
118 if (domainname == NULL)
119 yp_get_default_domain(&domainname);
120
121 inmap = argv[argc - 1];
122 if (notrans == 0) {
123 for (i = 0; ypaliases[i].alias; i++)
124 if (strcmp(inmap, ypaliases[i].alias) == 0)
125 inmap = ypaliases[i].name;
126 }
127
128 rval = 0;
129 for(c = 0; c < (argc - 1); c++) {
130 inkey = argv[c];
131
132 len = strlen(inkey);
133 if (null)
134 len++;
135 r = yp_match(domainname, inmap, inkey, len,
136 &outbuf, &outbuflen);
137 switch (r) {
138 case 0:
139 if (key)
140 printf("%s: ", inkey);
141 fwrite(outbuf, outbuflen, 1, stdout);
142 putc('\n', stdout);
143 break;
144
145 case YPERR_YPBIND:
146 errx(1, "not running ypbind");
147
148 default:
149 warnx("can't match key %s in map %s. Reason: %s",
150 inkey, inmap, yperr_string(r));
151 rval = 1;
152 break;
153 }
154 }
155
156 return rval;
157}
158
159static void
160usage(void)
161{
162
163 (void)fprintf(stderr, "Usage: %s [-ktz] [-b <num-retry>] "
164 "[-d <domainname>] <key> ... <mapname>\n", getprogname());
165 (void)fprintf(stderr, " %s -x\n", getprogname());
166 exit(1);
167}
