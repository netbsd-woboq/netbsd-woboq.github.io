48void
49fifoq_stat_loop(int fd, const char *ifname, int count, int interval)
50{
51 struct fifoq_getstats get_stats;
52 struct timeval cur_time, last_time;
53 u_int64_t last_bytes;
54 double sec;
55 int cnt = count;
56 sigset_t omask;
57
58 strlcpy(get_stats.iface.fifoq_ifname, ifname,
59 sizeof(get_stats.iface.fifoq_ifname));
60
61 gettimeofday(&last_time, NULL);
62 last_time.tv_sec -= interval;
63 last_bytes = 0;
64
65 for (;;) {
66 if (ioctl(fd, FIFOQ_GETSTATS, &get_stats) < 0)
67 err(1, "ioctl FIFOQ_GETSTATS");
68
69 gettimeofday(&cur_time, NULL);
70 sec = calc_interval(&cur_time, &last_time);
71
72 printf(" q_len:%d q_limit:%d period:%u\n",
73 get_stats.q_len, get_stats.q_limit, get_stats.period);
74 printf(" xmit:%llu pkts (%llu bytes) drop:%llu pkts (%llu bytes)\n",
75 (ull)get_stats.xmit_cnt.packets,
76 (ull)get_stats.xmit_cnt.bytes,
77 (ull)get_stats.drop_cnt.packets,
78 (ull)get_stats.drop_cnt.bytes);
79 printf(" throughput: %sbps\n",
80 rate2str(calc_rate(get_stats.xmit_cnt.bytes,
81 last_bytes, sec)));
82
83 last_bytes = get_stats.xmit_cnt.bytes;
84 last_time = cur_time;
85
86 if (count != 0 && --cnt == 0)
87 break;
88
89 /* wait for alarm signal */
90 if (sigprocmask(SIG_BLOCK, NULL, &omask) == 0)
91 sigsuspend(&omask);
92 }
93}
