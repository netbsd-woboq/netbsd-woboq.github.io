48static int avg_scale = 4096; /* default fixed-point scale */
49
50void
51red_stat_loop(int fd, const char *ifname, int count, int interval)
52{
53 struct red_stats red_stats;
54 struct timeval cur_time, last_time;
55 u_int64_t last_bytes;
56 double sec;
57 int cnt = count;
58 sigset_t omask;
59
60 strlcpy(red_stats.iface.red_ifname, ifname,
61 sizeof(red_stats.iface.red_ifname));
62
63 gettimeofday(&last_time, NULL);
64 last_time.tv_sec -= interval;
65 last_bytes = 0;
66
67 for (;;) {
68 if (ioctl(fd, RED_GETSTATS, &red_stats) < 0)
69 err(1, "ioctl RED_GETSTATS");
70
71 gettimeofday(&cur_time, NULL);
72 sec = calc_interval(&cur_time, &last_time);
73
74 printf(" weight:%d inv_pmax:%d qthresh:(%d,%d)\n",
75 red_stats.weight, red_stats.inv_pmax,
76 red_stats.th_min, red_stats.th_max);
77 printf(" q_len:%d (avg: %.2f), q_limit:%d\n",
78 red_stats.q_len,
79 ((double)red_stats.q_avg)/(double)avg_scale,
80 red_stats.q_limit);
81 printf(" xmit:%llu pkts, drop:%llu pkts (forced: %u, early: %u)\n",
82 (ull)red_stats.xmit_cnt.packets,
83 (ull)red_stats.drop_cnt.packets,
84 red_stats.drop_forced, red_stats.drop_unforced);
85 if (red_stats.marked_packets != 0)
86 printf(" marked: %u\n", red_stats.marked_packets);
87 printf(" throughput: %sbps\n",
88 rate2str(calc_rate(red_stats.xmit_cnt.bytes,
89 last_bytes, sec)));
90 if (red_stats.fv_alloc > 0) {
91 printf(" flowvalve: alloc:%u flows:%u\n",
92 red_stats.fv_alloc, red_stats.fv_flows);
93 printf(" predrop:%u pass:%u escape:%u\n",
94 red_stats.fv_predrop, red_stats.fv_pass,
95 red_stats.fv_escape);
96 }
97 printf("\n");
98
99 last_bytes = red_stats.xmit_cnt.bytes;
100 last_time = cur_time;
101
102 if (count != 0 && --cnt == 0)
103 break;
104
105 /* wait for alarm signal */
106 if (sigprocmask(SIG_BLOCK, NULL, &omask) == 0)
107 sigsuspend(&omask);
108 }
109}
110
111int
112print_redstats(struct redstats *rp)
113{
114 printf(" RED q_avg:%.2f xmit:%llu (forced:%u early:%u marked:%u)\n",
115 ((double)rp->q_avg)/(double)avg_scale,
116 (ull)rp->xmit_cnt.packets,
117 rp->drop_forced,
118 rp->drop_unforced,
119 rp->marked_packets);
120 return 0;
121}
