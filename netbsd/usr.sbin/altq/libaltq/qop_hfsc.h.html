31/*
32 * generalized service curve used for admission control
33 */
34struct segment {
35 LIST_ENTRY(segment) _next;
36 double x, y, d, m;
37};
38
39typedef LIST_HEAD(gen_sc, segment) gsc_head_t;
40
41/*
42 * hfsc private ifinfo structure
43 */
44struct hfsc_ifinfo {
45 struct classinfo *root_class; /* root class */
46 struct classinfo *default_class; /* default class */
47};
48
49/*
50 * hfsc private classinfo structure
51 */
52struct hfsc_classinfo {
53 struct service_curve rsc; /* real-time service curve */
54 struct service_curve fsc; /* fair service curve */
55 struct service_curve usc; /* upper limit service curve */
56 gsc_head_t gen_rsc; /* generalized real-time sc */
57 gsc_head_t gen_fsc; /* generalized fsc */
58 int qlimit;
59 int flags;
60};
