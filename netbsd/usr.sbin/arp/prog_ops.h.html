29#ifndef _PROG_OPS_H_
30#define _PROG_OPS_H_
31
32#include <sys/types.h>
33
34#ifndef CRUNCHOPS
35/* XXX: Keep same order with netstat! */
36struct prog_ops {
37 int (*op_init)(void);
38
39 int (*op_sysctl)(const int *, u_int, void *, size_t *,
40 const void *, size_t);
41
42 int (*op_socket)(int, int, int);
43 int (*op_open)(const char *, int, ...);
44 pid_t (*op_getpid)(void);
45
46 ssize_t (*op_read)(int, void *, size_t);
47 ssize_t (*op_write)(int, const void *, size_t);
48
49 int (*op_close)(int);
50};
51extern const struct prog_ops prog_ops;
52
53#define prog_init prog_ops.op_init
54#define prog_socket prog_ops.op_socket
55#define prog_open prog_ops.op_open
56#define prog_getpid prog_ops.op_getpid
57#define prog_read prog_ops.op_read
58#define prog_write prog_ops.op_write
59#define prog_close prog_ops.op_close
60#define prog_sysctl prog_ops.op_sysctl
61#else
62#define prog_init ((int (*)(void))NULL)
63#define prog_socket socket
64#define prog_open open
65#define prog_getpid getpid
66#define prog_read read
67#define prog_write write
68#define prog_close close
69#define prog_sysctl sysctl
70#endif
71
72#endif /* _PROG_OPS_H_ */
