34#include <sys/types.h>
35#include <stdbool.h>
36/* Our endpoint. */
37#define INTSEP 8
38
39/* AVDTP signals. */
40
41#define AVDTP_DISCOVER 0x01
42#define AVDTP_GET_CAPABILITIES 0x02
43#define AVDTP_SET_CONFIGURATION 0x03
44#define AVDTP_GET_CONFIGURATION 0x04
45#define AVDTP_RECONFIGURE 0x05
46#define AVDTP_OPEN 0x06
47#define AVDTP_START 0x07
48#define AVDTP_CLOSE 0x08
49#define AVDTP_SUSPEND 0x09
50#define AVDTP_ABORT 0x0a
51#define AVDTP_SECUURITY_CONTROL 0x0b
52
53/* Signal Command & Response Header Masks. */
54
55#define TRANSACTIONLABEL 0xf0
56#define TRANSACTIONLABEL_S 4
57#define SIGNALID_MASK 0x3f
58#define PACKETTYPE 0x0c
59#define PACKETTYPE_S 0x02
60#define MESSAGETYPE 0x03
61#define SIGNALIDENTIFIER 0x3f
62#define DISCOVER_SEP_IN_USE 0x02
63
64/* Packet Types */
65#define singlePacket 0x0
66#define startPacket 0x1
67#define continuePacket 0x2
68#define endPacket 0x3
69
70/* Message Types */
71#define COMMAND 0x0
72#define RESPONSEACCEPT 0x2
73#define RESPONSEREJECT 0x3
74
75/* Response general error/success lengths */
76#define AVDTP_LEN_SUCCESS 2
77#define AVDTP_LEN_ERROR 3
78
79/* Error codes */
80#define BAD_HEADER_FORMAT 0x01
81#define BAD_LENGTH 0x11
82#define BAD_ACP_SEID 0x12
83#define SEP_IN_USE 0x13
84#define SEP_NOT_IN_USE 0x14
85#define BAD_SERV_CATAGORY 0x17
86#define BAD_PAYLOAD_FORMAT 0x18
87#define NOT_SUPPORTED_COMMAND 0x19
88#define INVALID_CAPABILITIES 0x1a
89
90#define BAD_RECOVERY_TYPE 0x22
91#define BAD_MEDIA_TRANSPORT_FORMAT 0x23
92#define BAD_RECOVERY_FORMAT 0x25
93#define BAD_ROHC_FORMAT 0x26
94#define BAD_CP_FORMAT 0x27
95#define BAD_MULTIPLEXING_FORMAT 0x28
96#define UNSUPPORTED_CONFIGURATION 0x29
97#define BAD_STATE 0x31
98
99/* Service Capabilities Field. */
100#define mediaTransport 0x1
101#define reporting 0x2
102#define recovery 0x3
103#define contentProtection 0x4
104#define headerCompression 0x5
105#define multiplexing 0x6
106#define mediaCodec 0x7
107
108/* Media Codec Capabilities */
109#define mediaType 0xf0
110
111struct avdtp_sepInfo {
112 uint8_t sep;
113 uint8_t media_Type;
114};

116int avdtpSendCommand(int fd, uint8_t command, uint8_t type,
117 uint8_t *data, size_t datasize);
118int avdtpCheckResponse(int recvfd, bool *isCommand, uint8_t *trans, uint8_t
119 *signalId, uint8_t *pkt_type, uint8_t *data, size_t *datasize, uint8_t
120 *sep);
121int avdtpDiscover(uint8_t *buffer, size_t recvsize, struct avdtp_sepInfo
122 *sepInfo, bool sink);
123void avdtpGetCapabilities(int fd, int recvfd, uint8_t sep);
124int avdtpAutoConfigSBC(int fd, int recvfd, uint8_t * capabilities, size_t
125 caplen, uint8_t sep, u_int8_t *freq, u_int8_t *mode, u_int8_t *alloc_method,
126 u_int8_t *bitpool, u_int8_t* bands, u_int8_t *blocks, u_int8_t srcsep);
127int avdtpSetConfiguration(int fd, int recvfd, uint8_t sep, uint8_t *data,
128 size_t datasize, int srcsep);
129void avdtpOpen(int fd, int recvfd, uint8_t sep);
130void avdtpStart(int fd, int recvfd, uint8_t sep);
131void avdtpClose(int fd, int recvfd, uint8_t sep);
132void avdtpSuspend(int fd, int recvfd, uint8_t sep);
133void avdtpAbort(int fd, int recvfd, uint8_t sep);
134int avdtpSendDiscResponseAudio(int, int, uint8_t, uint8_t, bool);
135int avdtpSendCapabilitiesResponseSBC(int, int, int, uint8_t, uint8_t,
136 uint8_t, uint8_t, uint8_t, uint8_t, uint8_t);
137int avdtpSendAccept(int, int, uint8_t, uint8_t);
