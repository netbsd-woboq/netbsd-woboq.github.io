258int
259main(int ac, char *av[])
260{
261 int enc, i, n, m, l, j, k, o, ch, freqnum, blocksnum;
262 u_int tmpbitpool;
263 bdaddr_copy(&info.laddr, BDADDR_ANY);
264
265 sc = hc = -1;
266 verbose = asSpeaker = test_mode = initDiscover = false;
267 n = m = l = i = j = o = 0;
268 freqs[0] = frequency;
269 channel_config[0] = channel_mode;
270 blocks_config[0] = blocks;
271 bands_config[0] = bands;
272 alloc_config[0] = alloc_method;
273 channel_config[0] = channel_mode;
274
275 while ((ch = getopt(ac, av, "A:a:B:b:d:e:f:IKM:m:p:r:tV:v")) != EOF) {
276 switch (ch) {
277 case 'A':
278 for (k = 0; k < (int)strlen(optarg); k++) {
279 if (o > 1)
280 break;
281 switch (optarg[k]) {
282 case '0':
283 alloc_config[o++] = ALLOC_ANY;
284 break;
285 case 'S':
286 alloc_config[o++] = ALLOC_SNR;
287 break;
288 case 'L':
289 alloc_config[o++] = ALLOC_LOUDNESS;
290 break;
291 default:
292 usage();
293 }
294 }
295 break;
296 case 'a':
297 if (!bt_aton(optarg, &info.raddr)) {
298 struct hostent *he = NULL;
299
300 if ((he = bt_gethostbyname(optarg)) == NULL)
301 usage();
302
303 bdaddr_copy(&info.raddr,
304 (bdaddr_t *)he->h_addr);
305 }
306 break;
307 case 'B': /* Max bitpool value */
308 bitpool = (uint8_t)atoi(optarg);
309 break;
310 case 'b':
311 if (m > 3)
312 break;
313 blocksnum = atoi(optarg);
314 switch (blocksnum) {
315 case 0:
316 blocks_config[m++] = BLOCKS_ANY;
317 break;
318 case 4:
319 blocks_config[m++] = BLOCKS_4;
320 break;
321 case 8:
322 blocks_config[m++] = BLOCKS_8;
323 break;
324 case 12:
325 blocks_config[m++] = BLOCKS_12;
326 break;
327 case 16:
328 blocks_config[m++] = BLOCKS_16;
329 break;
330 default:
331 usage();
332 }
333 break;
334 case 'd': /* local device address */
335 if (!bt_devaddr(optarg, &info.laddr))
336 usage();
337 break;
338 case 'e':
339 if (l > 1)
340 break;
341 enc = atoi(optarg);
342 switch (enc) {
343 case 0:
344 bands_config[l++] = BANDS_ANY;
345 break;
346 case 4:
347 bands_config[l++] = BANDS_4;
348 break;
349 case 8:
350 bands_config[l++] = BANDS_8;
351 break;
352 default:
353 usage();
354 }
355 break;
356 case 'f':
357 for (k = 0; k < (int)strlen(optarg); k++) {
358 if (n > 3)
359 break;
360 switch (optarg[k]) {
361 case '0':
362 channel_config[n++] = MODE_ANY;
363 break;
364 case '2':
365 channel_config[n++] = MODE_DUAL;
366 break;
367 case 'j':
368 channel_config[n++] = MODE_JOINT;
369 break;
370 case 'm':
371 channel_config[n++] = MODE_MONO;
372 break;
373 case 's':
374 channel_config[n++] = MODE_STEREO;
375 break;
376 default:
377 usage();
378 }
379 }
380 break;
381 case 'I':
382 initDiscover = true;
383 break;
384 case 'K':
385 asSpeaker = true;
386 break;
387 case 'M':
388 userset_mtu = (uint16_t)atoi(optarg);
389 break;
390 case 'm': /* link mode */
391 if (strcasecmp(optarg, "auth") == 0)
392 l2cap_mode = L2CAP_LM_AUTH;
393 else if (strcasecmp(optarg, "encrypt") == 0)
394 l2cap_mode = L2CAP_LM_ENCRYPT;
395 else if (strcasecmp(optarg, "secure") == 0)
396 l2cap_mode = L2CAP_LM_SECURE;
397 else if (strcasecmp(optarg, "none"))
398 errx(EXIT_FAILURE, "%s: unknown mode", optarg);
399
400 break;
401 case 'p':
402 l2cap_psm = (uint16_t)atoi(optarg);
403 break;
404 case 'r':
405 if (i > 3)
406 break;
407 freqnum = atoi(optarg);
408 switch (freqnum) {
409 case 0:
410 freqs[i++] = FREQ_ANY;
411 break;
412 case 16000:
413 freqs[i++] = FREQ_16K;
414 break;
415 case 32000:
416 freqs[i++] = FREQ_32K;
417 break;
418 case 44100:
419 freqs[i++] = FREQ_44_1K;
420 break;
421 case 48000:
422 freqs[i++] = FREQ_48K;
423 break;
424 default:
425 usage();
426 }
427 break;
428 case 't':
429 test_mode = true;
430 break;
431 case 'V': /* Volume Multiplier */
432 volume = atoi(optarg);
433 break;
434 case 'v':
435 verbose = true;
436 break;
437 default:
438 usage();
439 }
440 }
441 av += optind;
442 ac -= optind;
443
444 numfiles = ac;
445
446 for (i = 0; i < numfiles; i++)
447 files2open[i] = av[i];
448
449 if (bdaddr_any(&info.raddr) && (!asSpeaker && !test_mode))
450 usage();
451
452 if (volume < 0 || volume > 2)
453 usage();
454
455 if (!asSpeaker && l2cap_psm)
456 usage();
457
458 if (asSpeaker) {
459 if (l == 0)
460 bands = BANDS_ANY;
461 if (m == 0)
462 blocks = BLOCKS_ANY;
463 if (n == 0)
464 channel_mode = MODE_ANY;
465 if (o == 0)
466 alloc_method = ALLOC_ANY;
467 }
468
469 if (i) {
470 frequency = 0;
471 for (j = 0; j < i; j++)
472 frequency |= freqs[j];
473 }
474 if (m) {
475 blocks = 0;
476 for (j = 0; j < m; j++)
477 blocks |= blocks_config[j];
478 }
479 if (l) {
480 bands = 0;
481 for (j = 0; j < l; j++)
482 bands |= bands_config[j];
483 }
484 if (n) {
485 channel_mode = 0;
486 for (j = 0; j < n; j++)
487 channel_mode |= channel_config[j];
488 }
489 if (o) {
490 alloc_method = 0;
491 for (j = 0; j < o; j++)
492 alloc_method |= alloc_config[j];
493 }
494
495 if (channel_mode == MODE_MONO || channel_mode == MODE_DUAL)
496 tmpbitpool = 16;
497 else
498 tmpbitpool = 32;
499
500 if (bands == BANDS_8)
501 tmpbitpool *= 8;
502 else
503 tmpbitpool *= 4;
504
505 if (tmpbitpool > DEFAULT_MAXBPOOL)
506 tmpbitpool = DEFAULT_MAXBPOOL;
507
508 if (bitpool == 0 || tmpbitpool < bitpool)
509 bitpool = (uint8_t)tmpbitpool;
510
511again:
512 base = event_init();
513
514 if (asSpeaker == 0) {
515 if (init_client(&info) < 0)
516 err(EXIT_FAILURE, "init client");
517 } else {
518 if (init_server(&info) < 0)
519 err(EXIT_FAILURE, "init server");
520 }
521
522 if (verbose) {
523 fprintf(stderr, "A2DP:\n");
524 fprintf(stderr, "\tladdr: %s\n", bt_ntoa(&info.laddr, NULL));
525 fprintf(stderr, "\traddr: %s\n", bt_ntoa(&info.raddr, NULL));
526 }
527
528 event_base_loop(base, 0);
529 event_base_free(base);
530
531 sdp_close(ss);
532
533 if (audfile != -1 && audfile != STDIN_FILENO && audfile !=
534 STDOUT_FILENO) {
535 close(audfile);
536 audfile = -1;
537 }
538
539 if (asSpeaker)
540 goto again;
541
542 return EXIT_SUCCESS;
543}

607static void
608do_ctlreq(int fd, short ev, void *arg)
609{
610 bool isCommand;
611 uint8_t sep, signal;
612 static struct sockaddr_bt addr;
613 size_t bufflen;
614 socklen_t len, mtusize;
615 static uint8_t trans;
616 uint8_t buff[1024];
617 static size_t offset = 0;
618 int result;
619
620 if(avdtpCheckResponse(fd, &isCommand, &trans, &signal, NULL, buff,
621 &bufflen, &sep) == ENOMEM) {
622 event_del(&ctl_ev);
623 close(fd);
624
625 if (asSpeaker) {
626 close(orighc);
627 orighc = -1;
628 event_del(&recv_ev);
629 } else
630 event_del(&interrupt_ev);
631
632 if (sc != -1)
633 close(sc);
634 sc = hc = -1;
635 state = 0;
636
637 return;
638 } else if (isCommand) {
639 switch (signal) {
640 case AVDTP_ABORT:
641 case AVDTP_CLOSE:
642 avdtpSendAccept(fd, fd, trans, signal);
643 if (asSpeaker) {
644 event_del(&recv_ev);
645 } else
646 event_del(&interrupt_ev);
647
648 if (sc != -1)
649 close(sc);
650 sc = -1;
651 state = 0;
652 break;
653 case AVDTP_DISCOVER:
654 avdtpSendDiscResponseAudio(fd, fd, trans,
655 (asSpeaker ? SNK_SEP : SRC_SEP), asSpeaker);
656 state = 1;
657 break;
658 case AVDTP_GET_CAPABILITIES:
659 avdtpSendCapabilitiesResponseSBC(fd, fd, trans,
660 (asSpeaker ? SNK_SEP : SRC_SEP), bitpool,
661 frequency , channel_mode, bands, blocks,
662 alloc_method);
663 state = 2;
664 break;
665 case AVDTP_SET_CONFIGURATION:
666 avdtpSendAccept(fd, fd, trans, signal);
667 state = 3;
668 break;
669 case AVDTP_OPEN:
670 avdtpSendAccept(fd, fd, trans, signal);
671 if (state < 5)
672 state = 5;
673 break;
674 case AVDTP_SUSPEND:
675 case AVDTP_START:
676 avdtpSendAccept(fd, fd, trans, signal);
677 if (state < 6)
678 state = 6;
679 break;
680 default:
681 avdtpSendReject(fd, fd, trans, signal);
682 }
683 if (verbose)
684 fprintf(stderr, "Received command %d\n",signal);
685 } else {
686 switch (signal) {
687 case AVDTP_DISCOVER:
688 if (offset >= bufflen)
689 offset = 0;
690 avdtpDiscover(buff + offset, bufflen - offset,
691 &mySepInfo, !asSpeaker);
692 avdtpGetCapabilities(fd, fd, mySepInfo.sep);
693 break;
694 case AVDTP_GET_CAPABILITIES:
695 result = avdtpAutoConfigSBC(fd, fd, buff, bufflen,
696 mySepInfo.sep, &frequency, &channel_mode,
697 &alloc_method, &bitpool, &bands, &blocks,
698 (asSpeaker ? SNK_SEP : SRC_SEP));
699
700 if (result == EINVAL) {
701 offset += 2;
702 avdtpSendCommand(hc, AVDTP_DISCOVER, 0,
703 NULL, 0);
704 }
705
706 if (!result && verbose)
707 fprintf(stderr, "Bitpool value = %d\n",bitpool);
708 state = 3;
709 break;
710 case AVDTP_SET_CONFIGURATION:
711 if (state == 3 && !asSpeaker)
712 avdtpOpen(fd, fd, mySepInfo.sep);
713 state = 4;
714 break;
715 case AVDTP_OPEN:
716 if (state == 4)
717 state = 5;
718 break;
719 case AVDTP_SUSPEND:
720 case AVDTP_START:
721 if (state < 6)
722 state = 6;
723 break;
724 default:
725 avdtpSendReject(fd, fd, trans, signal);
726 }
727 if (verbose)
728 fprintf(stderr, "Responded to command %d\n",signal);
729 }
730
731
732 if (state < 5 || state > 7)
733 return;
734
735 if (asSpeaker && state == 6) {
736 len = sizeof(addr);
737 if ((sc = accept(orighc,(struct sockaddr*)&addr, &len)) < 0)
738 err(EXIT_FAILURE, "stream accept");
739
740 }
741 if (state == 6)
742 goto opened_connection;
743
744 memset(&addr, 0, sizeof(addr));
745
746 addr.bt_len = sizeof(addr);
747 addr.bt_family = AF_BLUETOOTH;
748 bdaddr_copy(&addr.bt_bdaddr, &info.laddr);
749 addr.bt_psm = l2cap_psm;
750
751 sc = socket(PF_BLUETOOTH, SOCK_SEQPACKET, BTPROTO_L2CAP);
752 if (sc < 0)
753 return;
754
755 if (bind(sc, (struct sockaddr *)&addr, sizeof(addr)) < 0)
756 return;
757
758 if (setsockopt(sc, BTPROTO_L2CAP, SO_L2CAP_LM,
759 &l2cap_mode, sizeof(l2cap_mode)) == -1) {
760 log_err("Could not set link mode (0x%4.4x)", l2cap_mode);
761 exit(EXIT_FAILURE);
762 }
763
764 bdaddr_copy(&addr.bt_bdaddr, &info.raddr);
765 if (connect(sc,(struct sockaddr*)&addr, sizeof(addr)) < 0)
766 return;
767
768 if (!asSpeaker)
769 avdtpStart(fd, fd, mySepInfo.sep);
770
771 return;
772
773opened_connection:
774 if (asSpeaker) {
775 event_set(&recv_ev, sc, EV_READ | EV_PERSIST, do_recv, NULL);
776 if (event_add(&recv_ev, NULL) < 0)
777 err(EXIT_FAILURE, "recv_ev");
778 state = 7;
779 } else {
780 event_set(&interrupt_ev, audfile, EV_READ | EV_PERSIST,
781 do_interrupt, NULL);
782 if (event_add(&interrupt_ev, NULL) < 0)
783 err(EXIT_FAILURE, "interrupt_ev");
784 state = 7;
785 }
786
787 mtusize = sizeof(uint16_t);
788 getsockopt(sc, BTPROTO_L2CAP, SO_L2CAP_OMTU, &mtu, &mtusize);
789 if (userset_mtu != 0 && userset_mtu > 100 && userset_mtu < mtu)
790 mtu = userset_mtu;
791 else if (userset_mtu == 0 && mtu >= 500)
792 mtu /= 2;
793}

795static void
796do_recv(int fd, short ev, void *arg)
797{
798 ssize_t len;
799
800 len = recvstream(fd, audfile);
801
802 if (verbose)
803 fprintf(stderr, "Recving %zd bytes\n",len);
804
805 if (len < 0) {
806 event_del(&recv_ev);
807 close(fd);
808 fd = -1;
809 exit(1);
810 }
811}

814static void
815do_interrupt(int fd, short ev, void *arg)
816{
817 static int currentFileInd = 0;
818 ssize_t len;
819
820 if (currentFileInd == 0)
821 currentFileInd = startFileInd;
822
823 len = stream(fd, sc, channel_mode, frequency, bands, blocks,
824 alloc_method, bitpool, mtu, volume);
825
826 if (len == -1 && currentFileInd >= numfiles -1) {
827 event_del(&interrupt_ev);
828 close(fd);
829 fd = -1;
830 exit(1);
831 } else if (len == -1) {
832 close(fd);
833next_file:
834 currentFileInd++;
835 audfile = open(files2open[currentFileInd], O_RDONLY);
836 if (audfile < 0) {
837 warn("error opening file %s",
838 files2open[currentFileInd]);
839 goto next_file;
840 }
841
842 event_del(&interrupt_ev);
843 event_set(&interrupt_ev, audfile, EV_READ |
844 EV_PERSIST, do_interrupt, NULL);
845 if (event_add(&interrupt_ev, NULL) < 0)
846 err(EXIT_FAILURE, "interrupt_ev");
847 }
848
849 if (verbose)
850 fprintf(stderr, "Streaming %zd bytes\n",len);
851}

854 * Initialise as an audio sink

856static int
857init_server(struct l2cap_info *myInfo)
858{
859 struct sockaddr_bt addr;
860 socklen_t len;
861 int flags = O_WRONLY | O_CREAT;
862 static bool first_time = true;
863
864 if (atexit(exit_func))
865 err(EXIT_FAILURE,"atexit failed to initialize");
866
867 if (numfiles == 0)
868 audfile = STDOUT_FILENO;
869 else if (numfiles > 1)
870 usage();
871 else {
872 if (first_time)
873 flags |= O_TRUNC;
874 else
875 flags |= O_APPEND;
876
877 first_time = false;
878
879 audfile = open(files2open[0], flags, 0600);
880 if (audfile < 0) {
881 err(EXIT_FAILURE, "error opening file %s",
882 files2open[0]);
883 }
884 }
885 if (test_mode)
886 goto just_decode;
887
888 if (l2cap_psm == 0)
889 l2cap_psm = L2CAP_PSM_AVDTP;
890
891 sdp_set_uint(&sink_psm, l2cap_psm);
892
893 ss = sdp_open_local(NULL);
894 if (ss == NULL)
895 return -1;
896
897 if (!sdp_record_insert(ss, &myInfo->laddr, NULL, &sink_record)) {
898 sdp_close(ss);
899 return -1;
900 }
901
902 orighc = socket(PF_BLUETOOTH, SOCK_SEQPACKET, BTPROTO_L2CAP);
903 if (orighc < 0)
904 return -1;
905
906 memset(&addr, 0, sizeof(addr));
907 addr.bt_len = sizeof(addr);
908 addr.bt_family = AF_BLUETOOTH;
909 addr.bt_psm = l2cap_psm;
910 bdaddr_copy(&addr.bt_bdaddr, &myInfo->laddr);
911
912 if (bind(orighc, (struct sockaddr *)&addr, sizeof(addr)) < 0)
913 return -1;
914
915 if (setsockopt(orighc, BTPROTO_L2CAP, SO_L2CAP_LM,
916 &l2cap_mode, sizeof(l2cap_mode)) == -1) {
917 log_err("Could not set link mode (0x%4.4x)", l2cap_mode);
918 exit(EXIT_FAILURE);
919 }
920
921 bdaddr_copy(&addr.bt_bdaddr, &myInfo->raddr);
922 if (listen(orighc,0) < 0)
923 return -1;
924
925 len = sizeof(addr);
926 if ((hc = accept(orighc,(struct sockaddr*)&addr, &len)) < 0)
927 return -1;
928
929 if (initDiscover)
930 avdtpSendCommand(hc, AVDTP_DISCOVER, 0, NULL, 0);
931
932 event_set(&ctl_ev, hc, EV_READ | EV_PERSIST, do_ctlreq, NULL);
933 if (event_add(&ctl_ev, NULL) < 0)
934 err(EXIT_FAILURE, "ctl_ev");
935
936just_decode:
937 if (test_mode) {
938 if (userset_mtu)
939 mtu = userset_mtu;
940 sc = STDIN_FILENO;
941
942 event_set(&recv_ev, sc, EV_READ | EV_PERSIST, do_recv, NULL);
943 if (event_add(&recv_ev, NULL) < 0)
944 err(EXIT_FAILURE, "recv_ev");
945 }
946
947 return 0;
948}

1042static void
1043client_query(void)
1044{
1045 uint8_t buf[12]; /* enough for SSP and AIL both */
1046 sdp_session_t myss;
1047 sdp_data_t ssp, ail, rsp, rec, value, pdl, seq;
1048 uintmax_t psm, ver;
1049 uint16_t attr;
1050 bool rv;
1051
1052 myss = sdp_open(&info.laddr, &info.raddr);
1053 if (myss == NULL) {
1054 log_err("%s: Could not open sdp session", service_type);
1055 exit(EXIT_FAILURE);
1056 }
1057
1058 log_info("Searching for %s service at %s",
1059 service_type, bt_ntoa(&info.raddr, NULL));
1060
1061 seq.next = buf;
1062 seq.end = buf + sizeof(buf);
1063
1064 /*
1065 * build ServiceSearchPattern (9 bytes)
1066 *
1067 * uuid16 "service_class"
1068 * uuid16 L2CAP
1069 * uuid16 AVDTP
1070 */
1071 ssp.next = seq.next;
1072 sdp_put_uuid16(&seq, service_class);
1073 sdp_put_uuid16(&seq, SDP_UUID_PROTOCOL_L2CAP);
1074 sdp_put_uuid16(&seq, SDP_UUID_PROTOCOL_AVDTP);
1075 ssp.end = seq.next;
1076
1077 /*
1078 * build AttributeIDList (3 bytes)
1079 *
1080 * uint16 ProtocolDescriptorList
1081 */
1082 ail.next = seq.next;
1083 sdp_put_uint16(&seq, SDP_ATTR_PROTOCOL_DESCRIPTOR_LIST);
1084 ail.end = seq.next;
1085
1086 rv = sdp_service_search_attribute(myss, &ssp, &ail, &rsp);
1087 if (!rv) {
1088 log_err("%s: Required sdp record not found", service_type);
1089 exit(EXIT_FAILURE);
1090 }
1091
1092 /*
1093 * we expect the response to contain a list of records
1094 * containing a ProtocolDescriptorList. Find the first
1095 * one containing L2CAP and AVDTP >= 1.0, and extract
1096 * the PSM.
1097 */
1098 rv = false;
1099 while (!rv && sdp_get_seq(&rsp, &rec)) {
1100 if (!sdp_get_attr(&rec, &attr, &value)
1101 || attr != SDP_ATTR_PROTOCOL_DESCRIPTOR_LIST)
1102 continue;
1103
1104 sdp_get_alt(&value, &value); /* drop any alt header */
1105 while (!rv && sdp_get_seq(&value, &pdl)) {
1106 if (sdp_get_seq(&pdl, &seq) &&
1107 sdp_match_uuid16(&seq, SDP_UUID_PROTOCOL_L2CAP) &&
1108 sdp_get_uint(&seq, &psm) &&
1109 sdp_get_seq(&pdl, &seq) &&
1110 sdp_match_uuid16(&seq, SDP_UUID_PROTOCOL_AVDTP) &&
1111 sdp_get_uint(&seq, &ver) && ver >= 0x0100)
1112 rv = true;
1113 }
1114 }
1115
1116 sdp_close(myss);
1117
1118 if (!rv) {
1119 log_err("%s query failed", service_type);
1120 exit(EXIT_FAILURE);
1121 }
1122
1123 l2cap_psm = (uint16_t)psm;
1124 log_info("Found PSM %u for service %s", l2cap_psm, service_type);
1125}
