142u_int
143FLS(uint8_t x)
144{
145 u_int numset = 0;
146 while (x) {
147 if (x & 1)
148 break;
149 x >>= 1;
150 numset++;
151 }
152 return numset;
153}

155uint8_t
156calc_scalefactors_joint(int32_t sb_sample[16][2][8])
157{
158 int64_t sb_j[16][2];
159 uint32_t lz, x, y;
160 int32_t ax;
161 int block, sb;
162 unsigned int joint;
163
164 joint = 0;
165 for (sb = 0;sb < global_bands - 1;sb++) {
166 for (block = 0; block < global_blocks; block++) {
167 sb_j[block][0] = (sb_sample[block][0][sb]) +
168 (sb_sample[block][1][sb]);
169 sb_j[block][1] = (sb_sample[block][0][sb]) -
170 (sb_sample[block][1][sb]);
171 }
172
173 x = 1 << 15;
174 y = 1 << 15;
175 for (block = 0; block < global_blocks; block++) {
176 ax = abs((int32_t)(sb_j[block][0] / 2));
177 if (ax)
178 x |= (uint32_t)ax;
179 ax = abs((int32_t)(sb_j[block][1] / 2));
180 if (ax)
181 y |= (uint32_t)ax;
182 }
183
184 lz = 1;
185 while (!(x & __BIT(30))) {
186 lz++;
187 x <<= 1;
188 }
189 x = 16 - lz;
190
191 lz = 1;
192 while (!(y & __BIT(30))) {
193 lz++;
194 y <<= 1;
195 }
196 y = 16 - lz;
197
198 if ((scalefactor[0][sb] + scalefactor[1][sb]) > x + y) {
199 joint |= (unsigned int)(1 << (global_bands - sb - 1));
200 scalefactor[0][sb] = x;
201 scalefactor[1][sb] = y;
202 for (block = 0; block < global_blocks; block++) {
203 sb_sample[block][0][sb] = (int32_t)
204 (sb_j[block][0] / 2);
205 sb_sample[block][1][sb] = (int32_t)
206 (sb_j[block][1] / 2);
207 }
208 }
209 }
210
211 return (uint8_t)joint;
212}

214void
215calc_scalefactors(int32_t samples[16][2][8])
216{
217 uint32_t lz, x;
218 int32_t ax;
219 int ch, sb, block;
220
221 for (ch = 0; ch < global_chan; ch++) {
222 for (sb = 0; sb < global_bands; sb++) {
223 x = 1 << 16;
224 for (block = 0; block < global_blocks; block++) {
225 ax = abs((int32_t)samples[block][ch][sb]);
226 if (ax)
227 x |= (uint32_t)ax;
228 }
229
230 lz = 1;
231 while (!(x & __BIT(30))) {
232 lz++;
233 x <<= 1;
234 }
235 scalefactor[ch][sb] = 16 - lz;
236 }
237 }
238}

240void
241calc_bitneed(void)
242{
243 int32_t bitneed[2][8];
244 int32_t max_bitneed, bitcount;
245 int32_t slicecount, bitslice;
246 int32_t loudness;
247 int ch, sb,start_chan = 0;
248
249 if (global_mode == MODE_DUAL)
250 global_chan = 1;
251next_chan:
252 max_bitneed=0;
253 bitcount=0;
254 slicecount=0;
255
256 if (global_alloc == ALLOC_SNR) {
257 for (ch = start_chan;ch < global_chan;ch++) {
258 for (sb = 0;sb < global_bands;sb++) {
259 bitneed[ch][sb] = (int32_t)scalefactor[ch][sb];
260
261 if (bitneed[ch][sb] > max_bitneed)
262 max_bitneed = bitneed[ch][sb];
263 }
264 }
265 } else {
266 for (ch = start_chan;ch < global_chan;ch++) {
267 for (sb = 0;sb < global_bands;sb++) {
268 if (scalefactor[ch][sb] == 0)
269 bitneed[ch][sb] = -5;
270 else {
271 if (global_bands == 8) {
272 loudness = (int32_t)
273 ((int)scalefactor[ch][sb]
274 - loudnessoffset8
275 [3 - FLS(global_freq)][sb]);
276 } else {
277 loudness = (int32_t)
278 ((int)scalefactor[ch][sb]
279 - loudnessoffset4
280 [3 - FLS(global_freq)][sb]);
281 }
282 if (loudness > 0)
283 bitneed[ch][sb] = loudness / 2;
284 else
285 bitneed[ch][sb] = loudness;
286 }
287 if (bitneed[ch][sb] > max_bitneed)
288 max_bitneed = bitneed[ch][sb];
289 }
290 }
291 }
292
293 slicecount = bitcount = 0;
294 bitslice = max_bitneed+1;
295 do {
296 bitslice--;
297 bitcount += slicecount;
298 slicecount = 0;
299 for (ch = start_chan;ch < global_chan;ch++) {
300 for (sb = 0;sb < global_bands;sb++) {
301 if((bitneed[ch][sb] > bitslice + 1)&&
302 (bitneed[ch][sb] < bitslice + 16))
303 slicecount++;
304 else if(bitneed[ch][sb] == bitslice + 1)
305 slicecount += 2;
306 }
307 }
308 } while (bitcount + slicecount < global_bitpool);
309 if (bitcount + slicecount == global_bitpool) {
310 bitcount += slicecount;
311 bitslice--;
312 }
313
314 for (ch = start_chan;ch < global_chan;ch++) {
315 for (sb = 0;sb < global_bands;sb++) {
316 if (bitneed[ch][sb] < bitslice + 2)
317 bits[ch][sb] = 0;
318 else {
319 bits[ch][sb] = bitneed[ch][sb] - bitslice;
320 if (bits[ch][sb] > 16)
321 bits[ch][sb] = 16;
322 }
323 }
324 }
325
326 if (global_mode == MODE_DUAL)
327 ch = start_chan;
328 else
329 ch = 0;
330 sb = 0;
331 while (bitcount < global_bitpool && sb < global_bands) {
332 if ((bits[ch][sb] >= 2) && (bits[ch][sb] < 16)) {
333 bits[ch][sb]++;
334 bitcount++;
335 } else if ((bitneed[ch][sb] == bitslice + 1) &&
336 (global_bitpool > bitcount + 1)) {
337 bits[ch][sb] = 2;
338 bitcount += 2;
339 }
340 if (global_chan == 1 || start_chan == 1)
341 sb++;
342 else if (ch == 1) {
343 ch = 0;
344 sb++;
345 } else
346 ch = 1;
347 }
348
349 if (global_mode == MODE_DUAL)
350 ch = start_chan;
351 else
352 ch = 0;
353 sb = 0;
354 while (bitcount < global_bitpool && sb < global_bands) {
355 if (bits[ch][sb] < 16) {
356 bits[ch][sb]++;
357 bitcount++;
358 }
359 if (global_chan == 1 || start_chan == 1)
360 sb++;
361 else if (ch == 1) {
362 ch = 0;
363 sb++;
364 } else
365 ch = 1;
366 }
367
368 if (global_mode == MODE_DUAL && start_chan == 0) {
369 start_chan = 1;
370 global_chan = 2;
371 goto next_chan;
372 }
373}

375ssize_t
376get_bits(uint8_t *data, int numbits, uint32_t *sample)
377{
378 static uint64_t cache = 0;
379 static int cache_pos = 0;
380 uint64_t tmp_cache;
381 ssize_t written = 0;
382
383 while (cache_pos < numbits) {
384 cache <<= 8;
385 cache |= *data & 0xff;
386 data++;
387 written++;
388 cache_pos += 8;
389 }
390
391 if (numbits == 0) {
392 if (cache_pos >= 8) {
393 cache_pos -= 8;
394 tmp_cache = cache >> cache_pos;
395 *sample = (uint32_t)tmp_cache;
396 written--;
397 }
398 if (cache_pos) {
399 *sample = (uint32_t)cache;
400 written--;
401 }
402 cache = 0;
403 cache_pos = 0;
404 } else {
405 cache_pos -= numbits;
406 tmp_cache = cache & __BITS((uintmax_t)(numbits + cache_pos),
407 (uintmax_t)cache_pos);
408 cache &= ~tmp_cache;
409 tmp_cache >>= cache_pos;
410 *sample = (uint32_t)tmp_cache;
411 }
412 return written;
413}

415ssize_t
416move_bits(uint8_t *data, int numbits, uint32_t sample)
417{
418 static uint64_t cache = 0;
419 static int cache_pos = 0;
420 uint8_t tmp_cache;
421 ssize_t written = 0;
422
423 if (numbits == 0) {
424 while (cache_pos >= 8) {
425 cache_pos -= 8;
426 tmp_cache = (uint8_t)(cache >> cache_pos);
427 *data++ = tmp_cache;
428 written++;
429 }
430 if (cache_pos)
431 *data = (uint8_t)cache;
432 cache = 0;
433 cache_pos = 0;
434 } else {
435 cache_pos += numbits;
436 cache <<= numbits;
437 cache |= sample & __BITS((uintmax_t)numbits, 0);
438 while (cache_pos >= 8) {
439 cache_pos -= 8;
440 tmp_cache = (uint8_t)(cache >> cache_pos);
441 *data++ = tmp_cache;
442 written++;
443 }
444 }
445 return written;
446}

448ssize_t
449move_bits_crc(uint8_t *data, int numbits, uint32_t sample)
450{
451 static uint64_t cache = 0;
452 static int cache_pos = 0;
453 uint8_t tmp_cache;
454 ssize_t written = 0;
455
456 if (numbits > 8 || numbits < 0)
457 return 0;
458
459 if (numbits == 0) {
460 while (cache_pos >= 8) {
461 cache_pos -= 8;
462 tmp_cache = (uint8_t)(cache >> cache_pos);
463 *data++ = tmp_cache;
464 written++;
465 }
466 if (cache_pos)
467 *data = (uint8_t)cache;
468 cache = 0;
469 cache_pos = 0;
470 } else {
471 cache_pos += numbits;
472 cache <<= numbits;
473 cache |= sample & __BITS((uintmax_t)numbits, 0);
474 if (cache_pos >= 8) {
475 cache_pos -= 8;
476 tmp_cache = (uint8_t)(cache >> cache_pos);
477 *data = tmp_cache;
478 written++;
479 }
480 }
481 return written;
482}

484size_t
485sbc_encode(int16_t *input, int32_t *samples)
486{
487 int64_t delta[2][8], levels[2][8], S[80];
488 static int32_t L[80], R[80];
489 int32_t *X, Z[80], Y[80];
490 int32_t output[16][2][8];
491 int32_t audioout;
492 int16_t left[8], right[8], *data;
493 size_t numsamples;
494 int i, k, block, chan, sb;
495
496 for (block = 0;block < global_blocks; block++) {
497
498 k = 0;
499 for (i = 0;i < global_bands;i++) {
500 left[i] = input[k++];
501 if (global_chan == 2)
502 right[i] = input[k++];
503 }
504 input += k;
505
506 for (chan = 0; chan < global_chan; chan++) {
507 if (chan == 0) {
508 X = L;
509 data = left;
510 } else {
511 X = R;
512 data = right;
513 }
514
515 for (i = (global_bands * 10) - 1;i > global_bands -1;
516 i--)
517 X[i] = X[i - global_bands];
518 k = 0;
519 for (i = global_bands - 1; i >= 0; i--)
520 X[i] = (int16_t)le16toh(data[k++]);
521 for (i = 0; i < global_bands * 10; i++) {
522 if (global_bands == 8) {
523 Z[i] = (sbc_coeffs8[i] * (X[i] <<
524 global_volume));
525 } else {
526 Z[i] = (sbc_coeffs4[i] * (X[i] <<
527 global_volume));
528 }
529 }
530 for (i = 0; i < global_bands * 2; i++) {
531 Y[i] = 0;
532 for (k = 0;k < 5;k++)
533 Y[i] += Z[i + k * global_bands * 2];
534 }
535 for (i = 0; i < global_bands; i++) {
536 S[i] = 0;
537 for (k = 0; k < global_bands * 2; k++) {
538 if (global_bands == 8) {
539 S[i] += (int64_t)cosdata8[i][k]
540 * (int64_t)Y[k];
541 } else {
542 S[i] += (int64_t)cosdata4[i][k]
543 * (int64_t)Y[k];
544 }
545 }
546 output[block][chan][i] = (int32_t)(S[i] /
547 SIMULTI);
548 }
549 }
550 }
551
552 calc_scalefactors(output);
553 if (global_mode == MODE_JOINT)
554 join = calc_scalefactors_joint(output);
555
556 calc_bitneed();
557
558 for(chan = 0;chan < global_chan;chan++) {
559 for (sb = 0; sb < global_bands;sb++) {
560 levels[chan][sb] = ((1 << bits[chan][sb]) - 1) <<
561 (15 - scalefactor[chan][sb]);
562 delta[chan][sb] = 1 << (scalefactor[chan][sb] + 16);
563 }
564 }
565
566 numsamples = 0;
567 for (block = 0; block < global_blocks; block++) {
568 for (chan = 0; chan < global_chan; chan++) {
569 for (sb = 0; sb < global_bands; sb++) {
570 if (bits[chan][sb] == 0)
571 continue;
572
573 audioout = (int32_t)((levels[chan][sb] *
574 (delta[chan][sb] + (int32_t)output[block]
575 [chan][sb])) >> 32);
576
577 samples[numsamples++] = audioout;
578 }
579 }
580 }
581 return numsamples;
582}

584size_t
585sbc_decode(int32_t *samples, int16_t *pcm)
586{
587 static int64_t levels[2][8], delta[2][8];
588 static int64_t S[8], V[2][160], L[160], R[160];
589 int64_t *X;
590 static int64_t U[2][160], W[2][160];
591 int64_t audioout;
592 int chan, block, sb, position, i, k;
593 size_t numsamples;
594
595 for(chan = 0;chan < global_chan;chan++) {
596 for (sb = 0; sb < global_bands;sb++) {
597 levels[chan][sb] = (1 << bits[chan][sb]) - 1;
598 delta[chan][sb] = 1 << (scalefactor[chan][sb] + 1);
599 }
600 }
601
602 numsamples = 0;
603 for (block = 0; block < global_blocks; block++) {
604 for (chan = 0; chan < global_chan; chan++) {
605 for (sb = 0; sb < global_bands; sb++) {
606 if (bits[chan][sb] == 0)
607 audioout = 0;
608 else {
609 audioout = ((((samples[numsamples]
610 * 2) + 1) * delta[chan][sb]) /
611 levels[chan][sb]) -
612 delta[chan][sb];
613 }
614 samples[numsamples] = (int32_t)audioout;
615 numsamples++;
616 }
617 }
618 }
619
620 if (global_mode == MODE_JOINT) {
621 k = 0;
622 while (k < (global_blocks * global_bands * global_chan)) {
623 for (sb = 0; sb < global_bands;sb++) {
624 if (join & 1 << (global_bands - sb - 1)) {
625 audioout = samples[k];
626 samples[k] = (2 * samples[k]) + (2 *
627 samples[k + global_bands]);
628 samples[k + global_bands] =
629 (int32_t)(2 * audioout) - (2 *
630 samples[k + global_bands]);
631 samples[k] /= 2;
632 samples[k + global_bands] /= 2;
633 }
634 k++;
635 }
636 k += global_bands;
637 }
638 }
639
640
641 position = 0;
642 for (block = 0;block < global_blocks; block++) {
643 for (chan = 0; chan < global_chan; chan++) {
644 if (chan == 0)
645 X = L;
646 else
647 X = R;
648
649 for (i = 0; i < global_bands; i++)
650 S[i] = samples[position++];
651 for (i = ((global_bands * 20) - 1); i >= (global_bands
652 * 2); i--)
653 V[chan][i] = V[chan][i - (global_bands * 2)];
654
655 for (k = 0; k < global_bands * 2; k++) {
656 V[chan][k] = 0;
657 for (i = 0; i < global_bands; i++) {
658 if (global_bands == 8) {
659 V[chan][k] += cosdecdata8[i][k]
660 * S[i];
661 } else {
662 V[chan][k] += cosdecdata4[i][k]
663 * S[i];
664 }
665 }
666 V[chan][k] /= SIMULTI;
667 }
668
669 for (i = 0; i <= 4; i++) {
670 for (k = 0; k < global_bands; k++) {
671 U[chan][(i * global_bands * 2) + k] =
672 V[chan][(i * global_bands * 4) + k];
673 U[chan][(i * global_bands
674 * 2) + global_bands + k] =
675 V[chan][(i * global_bands * 4) +
676 (global_bands * 3) + k];
677 }
678 }
679
680 for (i = 0; i < global_bands * 10; i++) {
681 if (global_bands == 4) {
682 W[chan][i] = U[chan][i] *
683 (sbc_coeffs4[i] * -4);
684 } else if (global_bands == 8) {
685 W[chan][i] = U[chan][i] *
686 (sbc_coeffs8[i] * -8);
687 }
688 }
689
690 for (k = 0; k < global_bands; k++) {
691 int offset = k + (block * global_bands);
692 X[offset] = 0;
693 for (i = 0; i < 10; i++) {
694 X[offset] += W[chan][k + (i *
695 global_bands)];
696 }
697 X[offset] /= COEFFSMULTI;
698 }
699 }
700 }
701
702 k = 0;
703 i = 0;
704 while (k < (global_blocks * global_bands)) {
705 pcm[i++] = (int16_t)L[k];
706 if (global_chan == 2)
707 pcm[i++] = (int16_t)R[k];
708 k++;
709 }
710return numsamples;
711}

713uint8_t
714Crc8(uint8_t inCrc, uint8_t *inData, size_t numbits, ssize_t inBytes)
715{
716 uint8_t data;
717 int i;
718
719 for (i = 0; i < (int)inBytes; i++) {
720 data = inCrc ^ inData[i];
721
722 if (numbits == 8)
723 data = sbc_crc8[data];
724 else if (numbits == 4)
725 data = sbc_crc4[data];
726
727 inCrc = data;
728 }
729 return inCrc;
730}

732uint8_t
733make_crc(uint8_t config)
734{
735 uint8_t crc, data[11];
736 int i, j;
737 uint8_t *dataStart = data;
738 uint8_t *crcData = data;
739
740
741 crcData += move_bits_crc(crcData, 8, config);
742 crcData += move_bits_crc(crcData, 8, global_bitpool);
743 if (global_mode == MODE_JOINT) {
744 if (global_bands == 8)
745 crcData += move_bits_crc(crcData, 8, join);
746 else
747 crcData += move_bits_crc(crcData, 4, join);
748 }
749
750 for(i = 0; i < global_chan; i++) {
751 for (j = 0; j < global_bands; j++)
752 crcData += move_bits_crc(crcData, 4, scalefactor[i][j]);
753 }
754
755 crc = Crc8(0xf, data, 8, (crcData - dataStart));
756
757 if (global_mode == MODE_JOINT && global_bands == 4) {
758 move_bits_crc(crcData, 0, 0);
759 crc = Crc8(crc, crcData, 4, 1);
760 }
761
762 return crc;
763}

765ssize_t
766make_frame(uint8_t *frame, int16_t *input)
767{
768 static int32_t samples[256 * 2];
769 uint8_t config, crc;
770 int block, chan, sb, j, i;
771
772 uint8_t *frameStart = frame;
773
774 config = (uint8_t)(((3 - FLS(global_freq)) << 6) |
775 ((3 - FLS(global_block_config)) << 4) | ((3 - FLS(global_mode))
776 << 2) | ((FLS(global_alloc)) << 1) |
777 (1 - FLS(global_bands_config)));
778
779 sbc_encode(input,samples);
780
781 crc = make_crc(config);
782
783 frame += move_bits(frame, 8, SYNCWORD);
784 frame += move_bits(frame, 8, config);
785 frame += move_bits(frame, 8, global_bitpool);
786 frame += move_bits(frame, 8, crc);
787
788 if (global_mode == MODE_JOINT && global_bands == 8)
789 frame += move_bits(frame, 8, join);
790 else if (global_mode == MODE_JOINT && global_bands == 4)
791 frame += move_bits(frame, 4, join);
792
793 for(i = 0; i < global_chan; i++) {
794 for (j = 0; j < global_bands; j++)
795 frame += move_bits(frame, 4, scalefactor[i][j]);
796 }
797
798 i = 0;
799 for (block = 0; block < global_blocks; block++) {
800 for (chan = 0; chan < global_chan; chan++) {
801 for (sb = 0; sb < global_bands; sb++) {
802 if (bits[chan][sb] == 0)
803 continue;
804
805 frame += move_bits(frame, bits[chan][sb],
806 (uint32_t)samples[i++]);
807 }
808 }
809 }
810 frame += move_bits(frame, 0, 0);
811
812 return frame - frameStart;
813}

815ssize_t
816stream(int in, int outfd, uint8_t mode, uint8_t freq, uint8_t bands, uint8_t
817 blocks, uint8_t alloc_method, uint8_t bitpool, size_t mtu, int volume)
818{
819 struct rtpHeader myHeader;
820 uint8_t *whole, *frameData;
821 int16_t music[2048];
822 ssize_t len, mySize[16], offset, next_pkt;
823 ssize_t pkt_len;
824 size_t readsize, totalSize;
825 size_t frequency;
826 static size_t ts = 0;
827 static uint16_t seqnumber = 0;
828 int numpkts, tries;
829
830 global_mode = mode;
831 global_bitpool = bitpool;
832 global_alloc = alloc_method;
833 global_freq = freq;
834 global_volume = volume;
835
836 global_bands_config = bands;
837 if (bands == BANDS_8)
838 global_bands = 8;
839 else
840 global_bands = 4;
841
842 if (blocks == BLOCKS_4)
843 global_blocks = 4;
844 else if (blocks == BLOCKS_8)
845 global_blocks = 8;
846 else if (blocks == BLOCKS_12)
847 global_blocks = 12;
848 else {
849 blocks = BLOCKS_16;
850 global_blocks = 16;
851 }
852
853 global_block_config = blocks;
854
855 global_chan = 2;
856 if (global_mode == MODE_MONO)
857 global_chan = 1;
858
859 if (global_freq == FREQ_16K)
860 frequency = 16000;
861 else if (global_freq == FREQ_32K)
862 frequency = 32000;
863 else if (global_freq == FREQ_48K)
864 frequency = 48000;
865 else
866 frequency = 44100;
867
868 memset(&myHeader, 0, sizeof(myHeader));
869 myHeader.id = 0x80; /* RTP v2 */
870 myHeader.id2 = 0x60; /* payload type 96. */
871 myHeader.seqnumMSB = (uint8_t)(seqnumber >> 8);
872 myHeader.seqnumLSB = (uint8_t)seqnumber;
873 myHeader.ts3 = (uint8_t)(ts >> 24);
874 myHeader.ts2 = (uint8_t)(ts >> 16);
875 myHeader.ts1 = (uint8_t)(ts >> 8);
876 myHeader.ts0 = (uint8_t)ts;
877 myHeader.reserved0 = 0x01;
878
879 totalSize = sizeof(myHeader);
880
881 frameData = malloc(mtu);
882 if (frameData == NULL)
883 return -1;
884
885 readsize = (size_t)((global_blocks * global_bands * global_chan) * 2);
886
887 numpkts = 0;
888 next_pkt = 0;
889 len = 0;
890 pkt_len = 80;
891 while (totalSize + ((size_t)pkt_len * 2) <= mtu) {
892
893 len = read(in, music, readsize);
894 if (len < (int)readsize)
895 break;
896
897 pkt_len = make_frame(frameData + next_pkt, music);
898
899 mySize[numpkts] = pkt_len;
900 next_pkt += pkt_len;
901 totalSize += (size_t)mySize[numpkts];
902 numpkts++;
903
904 if (numpkts > 13)
905 break;
906 }
907
908 if (len < (int)readsize) {
909 free(frameData);
910 return -1;
911 }
912
913 myHeader.numFrames = (uint8_t)numpkts;
914 whole = malloc(totalSize);
915 if (whole == NULL)
916 return -1;
917
918 memcpy(whole, &myHeader, sizeof(myHeader));
919 offset = sizeof(myHeader);
920
921 memcpy(whole + offset, frameData, (size_t)next_pkt);
922 free(frameData);
923
924 tries = 1;
925send_again:
926 len = write(outfd, whole, totalSize);
927
928 if (len == -1 && errno == EAGAIN) {
929 tries --;
930 if (tries >= 0) {
931 usleep(1);
932 goto send_again;
933 } else
934 len = (ssize_t)totalSize;
935 } else if (len == -1 && (errno == EINPROGRESS ||
936 errno == EWOULDBLOCK)) {
937 usleep(1);
938 len = (ssize_t)totalSize;
939 }
940
941 seqnumber++;
942 ts += (1000000 * (size_t)(global_blocks * global_bands)
943 / frequency) * (size_t)numpkts;
944
945 free(whole);
946
947 return len;
948}

950ssize_t
951recvstream(int in, int outfd)
952{
953 struct rtpHeader myHeader;
954 static uint8_t frameData[8192];
955 uint8_t *myFrame;
956 static int16_t music[4096];
957 size_t decsize, totalSize, offset;
958 int numpkts, cur_pkt;
959 ssize_t len, res, pkt_len, next_pkt;
960 static ssize_t readlen = 0;
961 totalSize = 0;
962
963 len = read(in, frameData + readlen, (size_t)(1000 - readlen));
964 readlen += len;
965 if (readlen <= 0) {
966 readlen = 0;
967 return -1;
968 }
969
970 if (readlen < (int)sizeof(myHeader)) {
971 readlen = 0;
972 return -1;
973 }
974
975 memcpy(&myHeader, frameData, sizeof(myHeader));
976 if (myHeader.id != 0x80) {
977 return -1;
978 }
979
980 numpkts = myHeader.numFrames;
981 if (numpkts < 1 || numpkts > 13) {
982 return -1;
983 }
984
985
986 myFrame = frameData + sizeof(myHeader);
987 next_pkt = 0;
988 pkt_len = 0;
989 cur_pkt = 0;
990 offset = 0;
991 while (cur_pkt < numpkts) {
992 pkt_len = parseFrame(myFrame + next_pkt, &music[offset]);
993 decsize = (size_t)(global_blocks * global_bands * global_chan);
994
995 next_pkt += pkt_len;
996 totalSize += 2 * decsize;
997 offset += decsize;
998 cur_pkt++;
999
1000 if (cur_pkt > 13)
1001 break;
1002 }
1003
1004 res = (ssize_t)(sizeof(myHeader)) + next_pkt;
1005 readlen -= res;
1006 if (readlen > 0)
1007 memcpy(frameData, frameData + res, (size_t)readlen);
1008
1009send_again:
1010 len = write(outfd, music, totalSize);
1011
1012 if (len == -1 && errno == EAGAIN)
1013 goto send_again;
1014
1015 return len;
1016}

1018ssize_t
1019parseFrame(uint8_t *myFrame, int16_t *pcm)
1020{
1021 uint8_t mode, bitpool, alloc_method, freq, bands, config, myCrc, blkCrc;
1022 int sb, chan, block, blocks, i, j;
1023 int32_t samples[768];
1024
1025 uint8_t *myFrame_start = myFrame;
1026 if (*myFrame++ != SYNCWORD)
1027 return -1;
1028
1029 config = *myFrame++;
1030 bitpool = *myFrame++;
1031 myCrc = *myFrame++;
1032
1033 freq = (uint8_t)(1 << (3 - ((config & 0xc0) >> 6)));
1034 blocks = 1 << (3 - ((config & 0x30) >> 4));
1035 mode = (uint8_t)(1 << (3 - ((config & 0x0c) >> 2)));
1036 alloc_method = (uint8_t)(1 << (((config & 0x02) >> 1)));
1037 bands = (uint8_t)(1 << (1 - (config & 0x01)));
1038
1039 global_mode = mode;
1040 global_bitpool = bitpool;
1041 global_alloc = alloc_method;
1042 global_freq = freq;
1043
1044 global_bands_config = bands;
1045 if (bands == BANDS_8)
1046 global_bands = 8;
1047 else
1048 global_bands = 4;
1049
1050 if (blocks == BLOCKS_4)
1051 global_blocks = 4;
1052 else if (blocks == BLOCKS_8)
1053 global_blocks = 8;
1054 else if (blocks == BLOCKS_12)
1055 global_blocks = 12;
1056 else {
1057 blocks = BLOCKS_16;
1058 global_blocks = 16;
1059 }
1060
1061 global_block_config = (uint8_t)blocks;
1062
1063 global_chan = 2;
1064 if (global_mode == MODE_MONO)
1065 global_chan = 1;
1066
1067 if (global_mode == MODE_JOINT && global_bands == 8)
1068 myFrame += get_bits(myFrame, 8, (uint32_t *)&join);
1069 else if (global_mode == MODE_JOINT && global_bands == 4)
1070 myFrame += get_bits(myFrame, 4, (uint32_t *)&join);
1071 else
1072 join = 0;
1073
1074 for(i = 0; i < global_chan; i++) {
1075 for (j = 0; j < global_bands; j++) {
1076 myFrame += get_bits(myFrame, 4,
1077 (uint32_t *)&scalefactor[i][j]);
1078 }
1079 }
1080
1081 blkCrc = make_crc(config);
1082 if (blkCrc != myCrc)
1083 return 0;
1084
1085 calc_bitneed();
1086 i = 0;
1087 for (block = 0; block < global_blocks; block++) {
1088 for (chan = 0; chan < global_chan; chan++) {
1089 for (sb = 0; sb < global_bands; sb++) {
1090 samples[i] = 0;
1091 if (bits[chan][sb] == 0) {
1092 i++;
1093 continue;
1094 }
1095
1096 myFrame += get_bits(myFrame, bits[chan][sb],
1097 (uint32_t *)&samples[i++]);
1098 }
1099 }
1100 }
1101 myFrame += get_bits(myFrame, 0, (uint32_t *)&samples[i]);
1102 sbc_decode(samples, pcm);
1103
1104 return myFrame - myFrame_start;
1105}
