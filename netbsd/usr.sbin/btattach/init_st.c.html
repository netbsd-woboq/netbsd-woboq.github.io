44#define HCI_CMD_ST_SET_UART_BAUD_RATE \
45 HCI_OPCODE(HCI_OGF_VENDOR, 0x046)
46
47void
48init_st(int fd, unsigned int speed)
49{
50 uint8_t rate;
51
52 switch(speed) {
53 case B9600: rate = 0x09; break;
54 case B19200: rate = 0x0b; break;
55 case B38400: rate = 0x0d; break;
56 case B57600: rate = 0x0e; break;
57 case B115200: rate = 0x10; break;
58 case B230400: rate = 0x12; break;
59 case B460800: rate = 0x13; break;
60 case B921600: rate = 0x14; break;
61 default:
62 errx(EXIT_FAILURE, "invalid speed for st: %u", speed);
63 }
64
65 uart_send_cmd(fd, HCI_CMD_ST_SET_UART_BAUD_RATE, &rate, sizeof(rate));
66 /* wait for response? */
67 /* assume it succeeded? */
68}
