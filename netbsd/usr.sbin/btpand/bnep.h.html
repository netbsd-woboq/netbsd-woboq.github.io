28/*
29 * Constants defined in the Bluetooth Network Encapsulation
30 * Protocol (BNEP) specification v1.0
31 */
32
33#define BNEP_MTU_MIN 1691
34
35#define BNEP_EXT 0x80
36#define BNEP_TYPE(x) ((x) & 0x7f)
37#define BNEP_TYPE_EXT(x) (((x) & BNEP_EXT) == BNEP_EXT)
38
39#define BNEP_MAX_NET_TYPE_FILTERS 421
40#define BNEP_MAX_MULTI_ADDR_FILTERS 120
41
42/* BNEP packet types */
43#define BNEP_GENERAL_ETHERNET 0x00
44#define BNEP_CONTROL 0x01
45#define BNEP_COMPRESSED_ETHERNET 0x02
46#define BNEP_COMPRESSED_ETHERNET_SRC_ONLY 0x03
47#define BNEP_COMPRESSED_ETHERNET_DST_ONLY 0x04
48
49/* BNEP extension header types */
50#define BNEP_EXTENSION_CONTROL 0x00
51
52/* BNEP control types */
53#define BNEP_CONTROL_COMMAND_NOT_UNDERSTOOD 0x00
54#define BNEP_SETUP_CONNECTION_REQUEST 0x01
55#define BNEP_SETUP_CONNECTION_RESPONSE 0x02
56#define BNEP_FILTER_NET_TYPE_SET 0x03
57#define BNEP_FILTER_NET_TYPE_RESPONSE 0x04
58#define BNEP_FILTER_MULTI_ADDR_SET 0x05
59#define BNEP_FILTER_MULTI_ADDR_RESPONSE 0x06
60
61/* BNEP setup response codes */
62#define BNEP_SETUP_SUCCESS 0x0000
63#define BNEP_SETUP_INVALID_SRC_UUID 0x0001
64#define BNEP_SETUP_INVALID_DST_UUID 0x0002
65#define BNEP_SETUP_INVALID_UUID_SIZE 0x0003
66#define BNEP_SETUP_NOT_ALLOWED 0x0004
67
68/* BNEP filter return codes */
69#define BNEP_FILTER_SUCCESS 0x0000
70#define BNEP_FILTER_UNSUPPORTED_REQUEST 0x0001
71#define BNEP_FILTER_INVALID_RANGE 0x0002
72#define BNEP_FILTER_TOO_MANY_FILTERS 0x0003
73#define BNEP_FILTER_SECURITY_FAILURE 0x0004
74
