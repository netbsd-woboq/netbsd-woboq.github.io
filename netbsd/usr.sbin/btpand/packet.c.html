33packet_t *
34packet_alloc(channel_t *chan)
35{
36 packet_t *pkt;
37
38 pkt = malloc(sizeof(packet_t) + chan->mru);
39 if (pkt == NULL) {
40 log_err("%s() failed: %m", __func__);
41 return NULL;
42 }
43
44 memset(pkt, 0, sizeof(packet_t));
45 STAILQ_INIT(&pkt->extlist);
46 pkt->ptr = pkt->buf;
47
48 pkt->chan = chan;
49 chan->refcnt++;
50
51 return pkt;
52}
53
54void
55packet_free(packet_t *pkt)
56{
57 exthdr_t *eh;
58
59 if (pkt->refcnt-- > 0)
60 return;
61
62 while ((eh = STAILQ_FIRST(&pkt->extlist)) != NULL) {
63 STAILQ_REMOVE_HEAD(&pkt->extlist, next);
64 free(eh);
65 }
66
67 pkt->chan->refcnt--;
68 if (pkt->chan->refcnt == 0)
69 channel_free(pkt->chan);
70
71 free(pkt);
72}
73
74void
75packet_adj(packet_t *pkt, size_t size)
76{
77
78 assert(pkt->refcnt == 0);
79 assert(pkt->len >= size);
80
81 pkt->ptr += size;
82 pkt->len -= size;
83}
84
85pkthdr_t *
86pkthdr_alloc(packet_t *pkt)
87{
88 pkthdr_t *ph;
89
90 ph = malloc(sizeof(pkthdr_t));
91 if (ph == NULL) {
92 log_err("%s() failed: %m", __func__);
93 return NULL;
94 }
95
96 ph->data = pkt;
97 pkt->refcnt++;
98
99 return ph;
100}
101
102void
103pkthdr_free(pkthdr_t *ph)
104{
105
106 packet_free(ph->data);
107 free(ph);
108}
