31/* From sys/arch/x86/include/cpufunc.h */
32#define OPTERON_MSR_PASSCODE 0x9c5a203aU
33
34 .text
35
36ENTRY(x86_cpuid2)
37 pushq %rbx
38 movq %rdi, %rax
39 movq %rsi, %rcx
40 movq %rdx, %rsi
41 cpuid
42 movl %eax, 0(%rsi)
43 movl %ebx, 4(%rsi)
44 movl %ecx, 8(%rsi)
45 movl %edx, 12(%rsi)
46 popq %rbx
47 ret
48
49ENTRY(x86_xgetbv)
50 xorl %ecx, %ecx
51 xgetbv
52 ret
53
54ENTRY(x86_identify)
55 movl $-1,%eax
56 ret
