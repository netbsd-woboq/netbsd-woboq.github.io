50#ifdef VM_MIN_KERNEL_ADDRESS_DEFAULT
51vaddr_t vm_min_kernel_address = VM_MIN_KERNEL_ADDRESS_DEFAULT;
52#endif
53
54static struct nlist nl[] = {
55 { .n_name = "_dumppcb" },
56#ifdef VM_MIN_KERNEL_ADDRESS_DEFAULT
57 { .n_name = "vm_min_kernel_address" },
58#endif
59 { .n_name = NULL },
60};
61
62struct pcb pcb;
63
64void
65db_mach_init(kvm_t *kd)
66{
67
68 if (kvm_nlist(kd, nl) == -1) {
69 errx(EXIT_FAILURE, "kvm_nlist: %s", kvm_geterr(kd));
70 }
71 if ((size_t)kvm_read(kd, nl[0].n_value, &pcb, sizeof(pcb)) !=
72 sizeof(pcb)) {
73 errx(EXIT_FAILURE, "cannot read dumppcb: %s", kvm_geterr(kd));
74 }
75#ifdef VM_MIN_KERNEL_ADDRESS_DEFAULT
76 if ((size_t)kvm_read(kd, nl[1].n_value, &vm_min_kernel_address,
77 sizeof(vm_min_kernel_address)) != sizeof(vm_min_kernel_address)) {
78 errx(EXIT_FAILURE, "cannot read vm_min_kernel_address: %s",
79 kvm_geterr(kd));
80 }
81#endif
82 ddb_regs.tf_sp = pcb.pcb_sp;
83 ddb_regs.tf_bp = pcb.pcb_bp;
84 if (ddb_regs.tf_bp != 0 && ddb_regs.tf_sp != 0) {
85 printf("Backtrace from time of crash is available.\n");
86 }
87}
