128int
129main(int argc, char *argv[])
130{
131 int ch, eval;
132
133 while ((ch = getopt(argc, argv, "acijmsvF")) != -1)
134 switch(ch) {
135 case 'a': /* alternate superblocks */
136 case 'c': /* cylinder group info */
137 case 'i': /* actual inodes */
138 case 'j': /* journal */
139 case 'm': /* cylinder group summary */
140 case 's': /* superblock */
141 case 'v': /* more verbose */
142 opt_flags |= OPT_FLAG(ch);
143 break;
144 case 'F': /* File (not device) */
145 Fflag = 1;
146 break;
147 case '?':
148 default:
149 usage();
150 }
151 argc -= optind;
152 argv += optind;
153
154 if ((opt_flags & DFLT_CHECK) == 0)
155 opt_flags |= DFLT_OPTS;
156
157 if (argc < 1)
158 usage();
159
160 for (eval = 0; *argv; ++argv) {
161 eval |= dumpfs(*argv);
162 }
163
164 exit(eval);
165}

168static int
169dumpfs(const char *name)
170{
171 static const off_t sblock_try[] = SBLOCKSEARCH;
172 char device[MAXPATHLEN];
173 int fd, i;
174 int rval = 1;
175
176 if (Fflag)
177 fd = open(name, O_RDONLY);
178 else {
179 fd = openpartition(name, O_RDONLY, device, sizeof(device));
180 name = device;
181 }
182 if (fd == -1)
183 goto err;
184
185 for (i = 0; ; i++) {
186 if (sblock_try[i] == -1) {
187 warnx("%s: could not find superblock, skipped", name);
188 return 1;
189 }
190 if (lseek(fd, sblock_try[i], SEEK_SET) == (off_t)-1)
191 continue;
192 if (read(fd, &afs, SBLOCKSIZE) != SBLOCKSIZE)
193 continue;
194 switch(afs.fs_magic) {
195 case FS_UFS2_MAGIC:
196 is_ufs2 = 1;
197 break;
198 case FS_UFS1_MAGIC:
199 break;
200 case FS_UFS2_MAGIC_SWAPPED:
201 is_ufs2 = 1;
202 needswap = 1;
203 break;
204 case FS_UFS1_MAGIC_SWAPPED:
205 needswap = 1;
206 break;
207 default:
208 continue;
209 }
210 fix_superblock(&afs, opostblsave);
211 if (printold) {
212 if (sblock_try[i] == SBLOCK_UFS2)
213 /* This might be an alternate superblock */
214 continue;
215 } else {
216 if (sblock_try[i] != afs.fs_sblockloc)
217 /* This must be an alternate superblock */
218 continue;
219 }
220 break;
221 }
222
223
224 dev_bsize = afs.fs_fsize / FFS_FSBTODB(&afs, 1);
225
226 rval = 0;
227 printf("file system: %s\n", name);
228
229 if (ISOPT(opt_superblock))
230 rval = print_superblock(&afs, opostblsave, name, fd, sblock_try[i]);
231 if (rval == 0 && ISOPT(opt_cg_summary))
232 rval = print_cgsum(name, fd);
233 if (rval == 0 && ISOPT(opt_alt_super))
234 rval = print_alt_super(name, fd);
235 if (rval == 0 && ISOPT(opt_journal))
236 rval = print_journal(name, fd);
237 if (rval == 0 && ISOPT(opt_cg_info))
238 rval = print_cginfo(name, fd);
239 else if (rval == 0 && ISOPT(opt_inodes))
240 rval = print_inodes(name, fd, 0, afs.fs_ncg);
241
242 err:
243 if (fd != -1)
244 (void)close(fd);
245 if (rval)
246 warn("%s", name);
247 return rval;
248}

250static void
251fix_superblock(struct fs *fs, uint16_t *opostbl)
252{
253 if (needswap &&
254 (((int32_t)bswap32(fs->fs_old_postblformat) == FS_42POSTBLFMT) ||
255 (bswap32(fs->fs_old_postbloff) == offsetof(struct fs, fs_old_postbl_start)))) {
256 int i;
257 memcpy(opostbl, &fs->fs_old_postbl_start, 512);
258 for(i=0;i<256;i++)
259 opostbl[i] = bswap16(opostbl[i]);
260 } else if (!needswap &&
261 ((fs->fs_old_postblformat == FS_42POSTBLFMT) ||
262 (fs->fs_old_postbloff == offsetof(struct fs, fs_old_postbl_start)))) {
263 memcpy(opostbl, &fs->fs_old_postbl_start, 512);
264 }
265
266 if (needswap)
267 ffs_sb_swap(fs, fs);
268
269 printold = (fs->fs_magic == FS_UFS1_MAGIC &&
270 (fs->fs_old_flags & FS_FLAGS_UPDATED) == 0);
271 if (printold) {
272 fs->fs_sblockloc = SBLOCK_UFS1;
273 fs->fs_flags = fs->fs_old_flags;
274 fs->fs_maxbsize = fs->fs_bsize;
275 fs->fs_time = fs->fs_old_time;
276 fs->fs_size = fs->fs_old_size;
277 fs->fs_dsize = fs->fs_old_dsize;
278 fs->fs_csaddr = fs->fs_old_csaddr;
279 fs->fs_cstotal.cs_ndir = fs->fs_old_cstotal.cs_ndir;
280 fs->fs_cstotal.cs_nbfree = fs->fs_old_cstotal.cs_nbfree;
281 fs->fs_cstotal.cs_nifree = fs->fs_old_cstotal.cs_nifree;
282 fs->fs_cstotal.cs_nffree = fs->fs_old_cstotal.cs_nffree;
283 }
284
285 if (printold && fs->fs_old_postblformat == FS_42POSTBLFMT)
286 fs->fs_old_nrpos = 8;
287}

289static int
290print_superblock(struct fs *fs, uint16_t *opostbl,
291 const char *name, int fd, off_t sblock)
292{
293 int i, size;
294 time_t t;
295 int32_t fsflags;
296
297 printf("format\tFFSv%d\n", is_ufs2+1);
298#if BYTE_ORDER == LITTLE_ENDIAN
299 if (needswap)
300#else
301 if (!needswap)
302#endif
303 printf("endian\tbig-endian\n");
304 else
305 printf("endian\tlittle-endian\n");
306 t = fs->fs_time;
307 if ((sblock != SBLOCK_UFS1) || ISOPT(opt_alt_super))
308 printf("location %lld\t(-b %lld)\n",
309 (long long)sblock, (long long)(sblock/dev_bsize));
310 printf("magic\t%-8x\ttime\t%s",
311 fs->fs_magic, ctime(&t));
312
313 if (is_ufs2)
314 i = 5;
315 else {
316 i = 0;
317 if (fs->fs_old_postblformat != FS_42POSTBLFMT) {
318 i++;
319 if (fs->fs_old_inodefmt >= FS_44INODEFMT) {
320 int max;
321
322 i++;
323 max = fs->fs_maxcontig;
324 size = fs->fs_contigsumsize;
325 if ((max < 2 && size == 0)
326 || (max > 1 && size >= MIN(max, FS_MAXCONTIG)))
327 i++;
328 }
329 if (fs->fs_old_flags & FS_FLAGS_UPDATED) {
330 i = 4;
331 }
332 }
333 }
334 if (!printold || fs->fs_sblockloc != SBLOCK_UFS1 ||
335 fs->fs_id[0] || fs->fs_id[1])
336 printf("superblock location\t%jd\tid\t[ %x %x ]\n",
337 (intmax_t)fs->fs_sblockloc, fs->fs_id[0], fs->fs_id[1]);
338 printf("cylgrp\t%s\tinodes\t%s\tsblock\t%s\tfslevel %d\n",
339 i < 1 ? "static" : "dynamic",
340 i < 2 ? "4.2/4.3BSD" : i < 5 ? "4.4BSD" : "FFSv2",
341 i < 4 ? "FFSv1" : "FFSv2", i);
342 printf("nbfree\t%lld\tndir\t%lld\tnifree\t%lld\tnffree\t%lld\n",
343 (long long)fs->fs_cstotal.cs_nbfree,
344 (long long)fs->fs_cstotal.cs_ndir,
345 (long long)fs->fs_cstotal.cs_nifree,
346 (long long)fs->fs_cstotal.cs_nffree);
347 if (printold)
348 printf("ncg\t%d\tncyl\t%d\tsize\t%lld\tblocks\t%lld\n",
349 fs->fs_ncg, fs->fs_old_ncyl, (long long)fs->fs_size, (long long)fs->fs_dsize);
350 else
351 printf("ncg\t%d\tsize\t%lld\tblocks\t%lld\n",
352 fs->fs_ncg, (long long)fs->fs_size, (long long)fs->fs_dsize);
353 printf("bsize\t%d\tshift\t%d\tmask\t0x%08x\n",
354 fs->fs_bsize, fs->fs_bshift, fs->fs_bmask);
355 printf("fsize\t%d\tshift\t%d\tmask\t0x%08x\n",
356 fs->fs_fsize, fs->fs_fshift, fs->fs_fmask);
357 printf("frag\t%d\tshift\t%d\tfsbtodb\t%d\n",
358 fs->fs_frag, fs->fs_fragshift, fs->fs_fsbtodb);
359 if (printold)
360 printf("cpg\t%d\t", fs->fs_old_cpg);
361 printf("bpg\t%d\tfpg\t%d\tipg\t%d\n",
362 fs->fs_fpg / fs->fs_frag, fs->fs_fpg, fs->fs_ipg);
363 printf("minfree\t%d%%\toptim\t%s\tmaxcontig %d\tmaxbpg\t%d\n",
364 fs->fs_minfree, fs->fs_optim == FS_OPTSPACE ? "space" : "time",
365 fs->fs_maxcontig, fs->fs_maxbpg);
366 if (printold) {
367 printf("rotdelay %dms\theadswitch %dus\ttrackseek %dus\trps\t%d\n",
368 fs->fs_old_rotdelay, fs->fs_old_headswitch,
369 fs->fs_old_trkseek, fs->fs_old_rps);
370 printf("ntrak\t%d\tnsect\t%d\tnpsect\t%d\tspc\t%d\n",
371 fs->fs_spare2, fs->fs_old_nsect, fs->fs_old_npsect,
372 fs->fs_old_spc);
373 }
374 printf("symlinklen %d\t", fs->fs_maxsymlinklen);
375 if (printold)
376 printf("trackskew %d\tinterleave %d\t",
377 fs->fs_old_trackskew, fs->fs_old_interleave);
378 printf("contigsumsize %d\n", fs->fs_contigsumsize);
379 printf("maxfilesize 0x%016llx\n",
380 (unsigned long long)fs->fs_maxfilesize);
381 if (printold)
382 printf("nindir\t%d\tinopb\t%d\tnspf\t%d\n", fs->fs_nindir,
383 fs->fs_inopb, fs->fs_old_nspf);
384 else
385 printf("nindir\t%d\tinopb\t%d\n", fs->fs_nindir, fs->fs_inopb);
386 if (!printold || (fs->fs_avgfilesize > 0) || (fs->fs_avgfpdir > 0))
387 printf("avgfilesize %d\tavgfpdir %d\n",
388 fs->fs_avgfilesize, fs->fs_avgfpdir);
389 printf("sblkno\t%d\tcblkno\t%d\tiblkno\t%d\tdblkno\t%d\n",
390 fs->fs_sblkno, fs->fs_cblkno, fs->fs_iblkno, fs->fs_dblkno);
391 printf("sbsize\t%d\tcgsize\t%d", fs->fs_sbsize, fs->fs_cgsize);
392 if (printold)
393 printf("\toffset\t%d\tmask\t0x%08x",
394 fs->fs_old_cgoffset, fs->fs_old_cgmask);
395 printf("\ncsaddr\t%lld\tcssize\t%d",
396 (long long)fs->fs_csaddr, fs->fs_cssize);
397 if (printold)
398 printf("\tshift\t%d\tmask\t0x%08x",
399 fs->fs_old_csshift, fs->fs_old_csmask);
400 printf("\ncgrotor\t%d\tfmod\t%d\tronly\t%d\tclean\t0x%02x\n",
401 fs->fs_cgrotor, fs->fs_fmod, fs->fs_ronly, fs->fs_clean);
402 printf("wapbl version 0x%x\tlocation %u\tflags 0x%x\n",
403 fs->fs_journal_version, fs->fs_journal_location,
404 fs->fs_journal_flags);
405 printf("wapbl loc0 %" PRIu64 "\tloc1 %" PRIu64,
406 fs->fs_journallocs[0], fs->fs_journallocs[1]);
407 printf("\tloc2 %" PRIu64 "\tloc3 %" PRIu64 "\n",
408 fs->fs_journallocs[2], fs->fs_journallocs[3]);
409 printf("usrquota %" PRIu64 "\tgrpquota %" PRIu64 "\n",
410 fs->fs_quotafile[USRQUOTA], fs->fs_quotafile[GRPQUOTA]);
411 printf("flags\t");
412 if (fs->fs_flags == 0)
413 printf("none");
414 if (fs->fs_flags & FS_UNCLEAN)
415 printf("unclean ");
416 if (fs->fs_flags & FS_DOSOFTDEP)
417 printf("soft-updates ");
418 if (fs->fs_flags & FS_NEEDSFSCK)
419 printf("needs fsck run ");
420 if (fs->fs_flags & FS_SUJ)
421 printf("journaled soft-updates ");
422 if (fs->fs_flags & FS_ACLS)
423 printf("acls ");
424 if (fs->fs_flags & FS_MULTILABEL)
425 printf("multilabel ");
426 if (fs->fs_flags & FS_GJOURNAL)
427 printf("gjournal ");
428 if (fs->fs_flags & FS_FLAGS_UPDATED)
429 printf("fs_flags expanded ");
430 if (fs->fs_flags & FS_DOWAPBL)
431 printf("wapbl ");
432 if (fs->fs_flags & FS_DOQUOTA2)
433 printf("quotas ");
434 if (fs->fs_flags & FS_TRIM)
435 printf("trim ");
436 fsflags = fs->fs_flags & ~(FS_UNCLEAN | FS_DOSOFTDEP | FS_NEEDSFSCK |
437 FS_SUJ | FS_ACLS | FS_MULTILABEL | FS_GJOURNAL |
438 FS_FLAGS_UPDATED | FS_DOWAPBL | FS_DOQUOTA2 | FS_TRIM);
439#ifdef FS_INDEXDIRS
440 if (fs->fs_flags & FS_INDEXDIRS)
441 printf("indexed directories ");
442 fsflags &= ~FS_INDEXDIRS
443#endif
444 if (fsflags != 0)
445 printf("unknown flags (%#x)", fsflags);
446 printf("\nfsmnt\t%s\n", fs->fs_fsmnt);
447 if (!printold)
448 printf("volname\t%s\tswuid\t%ju\n",
449 fs->fs_volname, (uintmax_t)fs->fs_swuid);
450 if (printold) {
451 if (fs->fs_old_cpc != 0)
452 printf("blocks available in each of %d rotational "
453 "positions\n", fs->fs_old_nrpos);
454 else
455 printf("(no rotational position table)\n\n");
456 if (ISOPT(opt_verbose)) {
457 int c, j, k;
458 for (c = 0; c < fs->fs_old_cpc; c++) {
459 printf("cylinder number %d:", c);
460 for (i = 0; i < fs->fs_old_nrpos; i++) {
461 if (old_fs_postbl(&afs, c, opostbl)[i] == -1)
462 continue;
463 printf("\n position %d:\t", i);
464 for (j = old_fs_postbl(&afs, c, opostbl)[i], k = 1; ;
465 j += old_fs_rotbl(&afs)[j], k++) {
466 printf("%5d", j);
467 if (k % 12 == 0)
468 printf("\n\t\t");
469 if (old_fs_rotbl(&afs)[j] == 0)
470 break;
471 }
472 }
473 printf("\n");
474 }
475 }
476 }
477
478 return 0;
479}

481static int
482print_cgsum(const char *name, int fd)
483{
484 struct csum *ccsp;
485 int i, j, size;
486
487 afs.fs_csp = calloc(1, afs.fs_cssize);
488 for (i = 0, j = 0; i < afs.fs_cssize; i += afs.fs_bsize, j++) {
489 size = afs.fs_cssize - i < afs.fs_bsize ?
490 afs.fs_cssize - i : afs.fs_bsize;
491 ccsp = (struct csum *)((char *)afs.fs_csp + i);
492 if (lseek(fd,
493 (off_t)(FFS_FSBTODB(&afs, (afs.fs_csaddr + j * afs.fs_frag))) *
494 dev_bsize, SEEK_SET) == (off_t)-1)
495 return 1;
496 if (read(fd, ccsp, size) != size)
497 return 1;
498 if (needswap)
499 ffs_csum_swap(ccsp, ccsp, size);
500 }
501
502 printf("cs[].cs_(nbfree,ndir,nifree,nffree):\n\t");
503 for (i = 0; i < afs.fs_ncg; i++) {
504 struct csum *cs = &afs.fs_cs(&afs, i);
505 if (i && i % 4 == 0)
506 printf("\n\t");
507 printf("(%d,%d,%d,%d) ",
508 cs->cs_nbfree, cs->cs_ndir, cs->cs_nifree, cs->cs_nffree);
509 }
510 printf("\n");
511 if (printold && (afs.fs_old_ncyl % afs.fs_old_cpg)) {
512 printf("cylinders in last group %d\n",
513 i = afs.fs_old_ncyl % afs.fs_old_cpg);
514 printf("blocks in last group %d\n",
515 i * afs.fs_old_spc / (afs.fs_old_nspf << afs.fs_fragshift));
516 }
517 free(afs.fs_csp);
518 afs.fs_csp = NULL;
519
520 return 0;
521}

523static int
524print_alt_super(const char *name, int fd)
525{
526 union fsun alt;
527 int i;
528 off_t loc;
529 uint16_t alt_opostblsave[32*8];
530 int save_printold;
531
532 for (i = 0; i < afs.fs_ncg; i++) {
533 loc = (off_t)(FFS_FSBTODB(&afs, cgsblock(&afs, i))) * dev_bsize;
534 printf("\nalternate %d\n", i);
535 if (pread(fd, &alt, sizeof alt, loc) != sizeof alt) {
536 warnx("%s: error reading alt %d", name, i);
537 return (1);
538 }
539 save_printold = printold;
540 fix_superblock(&alt.fs, alt_opostblsave);
541 if (print_superblock(&alt.fs, alt_opostblsave, name, fd, loc)) {
542 printold = save_printold;
543 return 1;
544 }
545 printold = save_printold;
546 }
547 return 0;
548}

566static int
567print_inodes(const char *name, int fd, int c, int n)
568{
569 void *ino_buf = malloc(afs.fs_bsize);
570 void (*print_inode)(int, int, void *);
571 int i, inum;
572
573 if (ino_buf == 0)
574 return 1;
575
576 print_inode = is_ufs2 ? print_ufs2_inode : print_ufs1_inode;
577
578 for (inum = c * afs.fs_ipg ; inum < (c+n) * afs.fs_ipg; inum += afs.fs_inopb) {
579 if (pread(fd, ino_buf, afs.fs_bsize,
580 ino_to_fsba(&afs, inum) * afs.fs_fsize) != afs.fs_bsize) {
581 free(ino_buf);
582 return 1;
583 }
584 for (i = 0; i < afs.fs_inopb; i++)
585 print_inode(inum + i, i, ino_buf);
586 }

644 cg_clustersum(&acg, 0)[afs.fs_contigsumsize]);
645 printf("clusters free:\t");
646 pbits(0, cg_clustersfree(&acg, 0), acg.cg_nclusterblks);
647 } else
648 printf("\n");
649 printf("iused:\t");
650 pbits(0 * c * afs.fs_ipg, cg_inosused(&acg, 0), afs.fs_ipg);

652 pbits(0, cg_blksfree(&acg, 0), afs.fs_fpg);

661 old_fs_postbl(&afs, i % afs.fs_old_cpc, opostblsave)[j] == -1)
662 continue;
663 printf(" %d", old_cg_blks(&afs, &acg, i, 0)[j]);
664 }
665 printf("\n");
666 }
667 }
668 return (0);

723pbits(int offset, void *vp, int max)
724{
725 int i;

730 if (isset(p, i)) {

734 printf("%d", offset + i);

744static int
745print_journal(const char *name, int fd)
746{
747 daddr_t off;
748 size_t count, blklen, bno, skip;
749 off_t boff, head, tail, len;
750 uint32_t generation;
751
752 if (afs.fs_journal_version != UFS_WAPBL_VERSION)
753 return 0;
754
755 generation = 0;
756 head = tail = 0;
757
758 switch (afs.fs_journal_location) {
759 case UFS_WAPBL_JOURNALLOC_END_PARTITION:
760 case UFS_WAPBL_JOURNALLOC_IN_FILESYSTEM:
761
762 off = afs.fs_journallocs[0];
763 count = afs.fs_journallocs[1];
764 blklen = afs.fs_journallocs[2];
765
766 for (bno=0; bno<count; bno += skip / blklen) {
767
768 skip = blklen;
769
770 boff = bno * blklen;
771 if (bno * blklen >= 2 * blklen &&
772 ((head >= tail && (boff < tail || boff >= head)) ||
773 (head < tail && (boff >= head && boff < tail))))
774 continue;
775
776 printf("journal block %lu offset %lld\n",
777 (unsigned long)bno, (long long) boff);
778
779 if (lseek(fd, (off_t)(off*blklen) + boff, SEEK_SET)
780 == (off_t)-1)
781 return (1);
782 if (read(fd, &jbuf, blklen) != (ssize_t)blklen) {
783 warnx("%s: error reading journal", name);
784 return 1;
785 }
786
787 switch (awh.wc_type) {
788 case 0:
789 break;
790 case WAPBL_WC_HEADER:
791 print_journal_header(name);
792 if (awh.wc_generation > generation) {
793 head = awh.wc_head;
794 tail = awh.wc_tail;
795 }
796 generation = awh.wc_generation;
797 skip = awh.wc_len;
798 break;
799 default:
800 len = print_journal_entries(name, blklen);
801 skip = awh.wc_len;
802 if (len != (off_t)skip)
803 printf(" CORRUPTED RECORD\n");
804 break;
805 }
806
807 if (blklen == 0)
808 break;
809
810 skip = (skip + blklen - 1) / blklen * blklen;
811 if (skip == 0)
812 break;
813
814 }
815 break;
816 }
817
818 return 0;
819}

860static off_t
861print_journal_entries(const char *name, size_t blklen)
862{
863 int i, n;
864 struct wapbl_wc_blocklist *wcb;
865 struct wapbl_wc_inodelist *wci;
866 off_t len = 0;
867 int ph;
868
869 printf(" type %s len %d",
870 wapbl_type_string(awn.wc_type), awn.wc_len);
871
872 switch (awn.wc_type) {
873 case WAPBL_WC_BLOCKS:
874 case WAPBL_WC_REVOCATIONS:
875 wcb = (struct wapbl_wc_blocklist *)&awn;
876 printf(" blkcount %u\n", wcb->wc_blkcount);
877 ph = (blklen - offsetof(struct wapbl_wc_blocklist, wc_blocks))
878 / sizeof(wcb->wc_blocks[0]);
879 n = MIN(wcb->wc_blkcount, ph);
880 for (i=0; i<n; i++) {
881 if (ISOPT(opt_verbose)) {
882 printf(" %3d: daddr %14llu dlen %d\n", i,
883 (unsigned long long)wcb->wc_blocks[i].wc_daddr,
884 wcb->wc_blocks[i].wc_dlen);
885 }
886 len += wcb->wc_blocks[i].wc_dlen;
887 }
888 if (awn.wc_type == WAPBL_WC_BLOCKS) {
889 if (len % blklen)
890 len += blklen - len % blklen;
891 } else
892 len = 0;
893 break;
894 case WAPBL_WC_INODES:
895 wci = (struct wapbl_wc_inodelist *)&awn;
896 printf(" count %u clear %u\n",
897 wci->wc_inocnt, wci->wc_clear);
898 ph = (blklen - offsetof(struct wapbl_wc_inodelist, wc_inodes))
899 / sizeof(wci->wc_inodes[0]);
900 n = MIN(wci->wc_inocnt, ph);
901 for (i=0; i<n; ++i) {
902 if (ISOPT(opt_verbose)) {
903 printf(" %3d: inumber %10u imode %08x\n", i,
904 wci->wc_inodes[i].wc_inumber,
905 wci->wc_inodes[i].wc_imode);
906 }
907 }
908 break;
909 default:
910 printf("\n");
911 break;
912 }
913
914 return len + blklen;
915}

925static int
926openpartition(const char *name, int flags, char *device, size_t devicelen)
927{
928 char rawspec[MAXPATHLEN], xbuf[MAXPATHLEN], *p;
929 struct fstab *fs;
930 int fd, oerrno;
931
932 fs = getfsfile(name);
933 if (fs) {
934 const char *fsspec;
935 fsspec = getfsspecname(xbuf, sizeof(xbuf), fs->fs_spec);
936 if (fsspec == NULL) {
937 warn("%s", xbuf);
938 return -1;
939 }
940 if ((p = strrchr(fsspec, '/')) != NULL) {
941 snprintf(rawspec, sizeof(rawspec), "%.*s/r%s",
942 (int)(p - fsspec), fsspec, p + 1);
943 name = rawspec;
944 } else
945 name = fsspec;
946 }
947 fd = opendisk(name, flags, device, devicelen, 0);
948 if (fd == -1 && errno == ENOENT) {
949 oerrno = errno;
950 strlcpy(device, name, devicelen);
951 errno = oerrno;
952 }
953 return (fd);
954}
