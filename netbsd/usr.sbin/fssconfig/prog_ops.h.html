29#ifndef _PROG_OPS_H_
30#define _PROG_OPS_H_
31
32#include <sys/types.h>
33#include <sys/stat.h>
34#include <sys/statvfs.h>
35
36#include <fcntl.h>
37
38#ifndef CRUNCHOPS
39struct prog_ops {
40 int (*op_init)(void);
41 int (*op_open)(const char *, int, ...);
42 int (*op_ioctl)(int, unsigned long, ...);
43 int (*op_close)(int);
44 ssize_t (*op_pread)(int, void *, size_t, off_t);
45 int (*op_statvfs1)(const char *, struct statvfs *, int);
46 int (*op_stat)(const char *, struct stat *);
47 int (*op_fstat)(int, struct stat *);
48};
49extern const struct prog_ops prog_ops;
50
51#define prog_init prog_ops.op_init
52#define prog_open prog_ops.op_open
53#define prog_ioctl prog_ops.op_ioctl
54#define prog_close prog_ops.op_close
55#define prog_pread prog_ops.op_pread
56#define prog_stat prog_ops.op_stat
57#define prog_statvfs1 prog_ops.op_statvfs1
58#define prog_fstat prog_ops.op_fstat
59#else
60#define prog_init ((int (*)(void))NULL)
61#define prog_open open
62#define prog_ioctl ioctl
63#define prog_close close
64#define prog_pread pread
65#define prog_stat stat
66#define prog_statvfs1 statvfs1
67#define prog_fstat fstat
68#endif
69
70#endif /* _PROG_OPS_H_ */
