31#ifndef _SYS_EUI64_H
32#define _SYS_EUI64_H
33
34/*
35 * Size of the ASCII representation of an EUI-64.
36 */
37#define EUI64_SIZ 24
38
39/*
40 * The number of bytes in an EUI-64.
41 */
42#define EUI64_LEN 8
43
44/*
45 * Structure of an IEEE EUI-64.
46 */
47struct eui64 {
48 u_char octet[EUI64_LEN];
49};
50
51#ifndef _KERNEL
52int eui64_aton(const char *, struct eui64 *);
53int eui64_ntoa(const struct eui64 *, char *, size_t);
54int eui64_ntohost(char *, size_t, const struct eui64 *);
55int eui64_hostton(const char *, struct eui64 *);
56#endif /* !_KERNEL */
57
58#endif /* !_SYS_EUI64_H */
