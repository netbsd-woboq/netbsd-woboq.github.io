94static int
95sparc_clearheader(ib_params *params, struct bbinfo_params *bb_params,
96 uint8_t *bb)
97{
98
99 assert(params != NULL);
100 assert(bb_params != NULL);
101 assert(bb != NULL);
102
103 memset(bb, 0, bb_params->headeroffset);
104 return (1);
105}
106
107static int
108sparc_setheader(ib_params *params, struct bbinfo_params *bb_params, uint8_t *bb)
109{
110
111 assert(params != NULL);
112 assert(bb_params != NULL);
113 assert(bb != NULL);
114
115 /*
116 * sun4c/sun4m PROMs require an a.out(5) format header.
117 * Old-style sun4 PROMs do not expect a header at all.
118 * To deal with this, we construct a header that is also executable
119 * code containing a forward branch that gets us past the 32-byte
120 * header where the actual code begins. In assembly:
121 * .word MAGIC ! a NOP
122 * ba,a start !
123 * .skip 24 ! pad
124 * start:
125 */
126#define SUN_MAGIC 0x01030107
127#define SUN4_BASTART 0x30800007 /* i.e.: ba,a `start' */
128 *((uint32_t *)bb) = htobe32(SUN_MAGIC);
129 *((uint32_t *)bb + 1) = htobe32(SUN4_BASTART);
130
131 return (1);
132}
