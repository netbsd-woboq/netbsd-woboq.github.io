79uint16_t
80compute_sunsum(const uint16_t *bb16)
81{
82 uint16_t i, s;
83
84 assert(bb16 != NULL);
85
86 s = 0;
87 for (i = 0; i < 255; ++i)
88 s ^= bb16[i];
89 return (s);
90}
91
92int
93set_sunsum(ib_params *params, uint16_t *bb16, uint16_t sum)
94{
95
96 assert(params != NULL);
97 assert(bb16 != NULL);
98
99#define SUNSUM_OFFSET 255
100 if (params->flags & IB_VERBOSE) {
101 printf("Old Sun checksum: 0x%04x\n",
102 be16toh(bb16[SUNSUM_OFFSET]));
103 printf("Recalculated Sun checksum: 0x%04x\n", be16toh(sum));
104 }
105 // XXX: does this need to be big endian?
106 bb16[SUNSUM_OFFSET] = sum;
107 return (1);
108}
