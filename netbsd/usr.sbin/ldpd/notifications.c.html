42/* builds a notification TLV. Do not forget to free the result */
43struct notification_tlv *
44build_notification(uint32_t msg, uint32_t n)
45{
46 struct notification_tlv *t;
47 t = malloc(sizeof(*t));
48
49 if (!t) {
50 fatalp("build_notification: malloc problem\n");
51 return NULL;
52 }
53
54 t->type = htons(LDP_NOTIFICATION);
55 t->length = htons(sizeof(struct notification_tlv) - TLV_TYPE_LENGTH);
56 t->status = htons(TLV_STATUS);
57 t->st_len = sizeof(struct notification_tlv) - 2 * TLV_TYPE_LENGTH;
58 t->st_code = htonl(n);
59 t->msg_id = htonl(msg);
60 t->msg_type = 0;
61 return t;
62
63}
64
65int
66send_notification(const struct ldp_peer * p, uint32_t msg, uint32_t code)
67{
68 struct notification_tlv *nt;
69 int rv;
70
71 if (p->state != LDP_PEER_CONNECTED && p->state != LDP_PEER_ESTABLISHED)
72 return -1;
73
74 nt = build_notification(msg, code);
75
76 rv = send_tlv(p, (struct tlv *) nt);
77 free(nt);
78 return rv;
79}
