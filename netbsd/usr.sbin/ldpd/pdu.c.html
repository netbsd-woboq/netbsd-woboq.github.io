45uint
46get_pdu(const unsigned char *s, struct ldp_pdu * p)
47{
48 const struct ldp_pdu *p1 = (const struct ldp_pdu *) s;
49
50 p->version = ntohs(p1->version);
51 p->length = ntohs(p1->length);
52 memcpy(&p->ldp_id, &p1->ldp_id, sizeof(struct in_addr));
53 p->label_space = ntohs(p1->label_space);
54
55 return MIN_PDU_SIZE;
56}

58/* Checks an incoming PDU for size and version */
59int
60check_recv_pdu(const struct ldp_peer * p, const struct ldp_pdu * rpdu, int c)
61{
62 struct notification_tlv *notiftlv;
63
64 /* Avoid underflow */
65 if (c < MIN_PDU_SIZE)
66 return LDP_E_BAD_LENGTH;
67
68 if (p->ldp_id.s_addr != rpdu->ldp_id.s_addr) {
69 fatalp("Invalid LDP ID %s received from ",
70 inet_ntoa(rpdu->ldp_id));
71 fatalp("%s\n", inet_ntoa(p->ldp_id));
72 notiftlv = build_notification(0,
73 NOTIF_FATAL | NOTIF_BAD_LDP_ID);
74 send_tlv(p, (struct tlv *) notiftlv);
75 free(notiftlv);
76 return LDP_E_BAD_ID;
77 }
78
79 /* Check PDU for right LDP version */
80 if (ntohs(rpdu->version) != LDP_VERSION) {
81 fatalp("Invalid PDU version received from %s (%d)\n",
82 satos(p->address), ntohs(rpdu->version));
83 notiftlv = build_notification(0,
84 NOTIF_FATAL | NOTIF_BAD_LDP_VER);
85 send_tlv(p, (struct tlv *) notiftlv);
86 free(notiftlv);
87 return LDP_E_BAD_VERSION;
88 }
89 /* Check PDU for length validity */
90 if (ntohs(rpdu->length) > c - PDU_VER_LENGTH) {
91 fatalp("Invalid PDU length received from %s (announced %d, "
92 "received %d)\n", satos(p->address),
93 ntohs(rpdu->length), (int) (c - PDU_VER_LENGTH));
94 notiftlv = build_notification(0,
95 NOTIF_FATAL | NOTIF_BAD_PDU_LEN);
96 send_tlv(p, (struct tlv *) notiftlv);
97 free(notiftlv);
98 return LDP_E_BAD_LENGTH;
99 }
100 return LDP_E_OK;
101}
