32#ifndef _PDU_H_
33#define _PDU_H_
34
35#include <sys/types.h>
36#include <netinet/in.h>
37
38#include "ldp_peer.h"
39
40#define MIN_PDU_SIZE 10
41#define MAX_PDU_SIZE 1300
42
43#define PDU_VER_LENGTH (sizeof(uint16_t) + sizeof(uint16_t))
44#define PDU_PAYLOAD_LENGTH (sizeof(struct in_addr) + sizeof(uint16_t))
45
46struct ldp_pdu {
47 uint16_t version;
48 uint16_t length;
49 /* draft-ietf-mpls-ldp-ipv6-07 keeps this IPv4 only for now */
50 struct in_addr ldp_id;
51 uint16_t label_space;
52} __packed;
53
54
55uint get_pdu(const unsigned char *, struct ldp_pdu *);
56int check_recv_pdu(const struct ldp_peer *, const struct ldp_pdu *, int);
57
58
59#endif /* !_PDU_H_ */
