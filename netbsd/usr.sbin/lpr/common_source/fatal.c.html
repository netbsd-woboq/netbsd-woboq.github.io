44#include <stdarg.h>
45#include <stdio.h>
46#include <stdlib.h>
47
48#include "lp.h"
49
50void
51fatal(const char *msg, ...)
52{
53 va_list ap;
54
55 va_start(ap, msg);
56 if (from != host)
57 (void)printf("%s: ", host);
58 (void)printf("%s: ", getprogname());
59 if (printer)
60 (void)printf("%s: ", printer);
61 (void)vprintf(msg, ap);
62 va_end(ap);
63 (void)putchar('\n');
64 exit(1);
65}
