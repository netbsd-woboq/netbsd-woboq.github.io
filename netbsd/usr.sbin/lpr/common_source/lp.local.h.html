35 * Possibly, local parameters to the spooling system
36 */

38/*
39 * Defaults for line printer capabilities data base
40 */
41#define DEFLP "lp"
42#define DEFLOCK "lock"
43#define DEFSTAT "status"
44#define DEFMX 1000
45#define DEFMAXCOPIES 0
46#define DEFFF "\f"
47#define DEFWIDTH 132
48#define DEFLENGTH 66
49#define DEFUID 1
50
51/*
52 * When files are created in the spooling area, they are normally
53 * readable only by their owner and the spooling group. If you
54 * want otherwise, change this mode.
55 */
56#define FILMOD 0660
57
58/*
59 * Printer is assumed to support LINELEN (for block chars)
60 * and background character (blank) is a space
61 */
62#define LINELEN 132
63#define BACKGND ' '
64
65#define HEIGHT 9 /* height of characters */
66#define WIDTH 8 /* width of characters */
67#define DROP 3 /* offset to drop characters with descenders */
68
69/*
70 * Define TERMCAP if the terminal capabilites are to be used for lpq.
71 */
72#define TERMCAP
73
74/*
75 * Maximum number of user and job requests for lpq and lprm.
76 */
77#define MAXUSERS 50
78#define MAXREQUESTS 50
