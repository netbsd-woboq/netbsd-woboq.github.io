71static struct key {
72 const char *name; /* name */
73 void (*f)(struct info *); /* function */
74#define F_NEEDARG 0x01 /* needs an argument */
75#define F_OFFOK 0x02 /* can turn off */
76 int flags;
77} const keys[] = {
78 { "cbreak", f_cbreak, F_OFFOK },
79 { "cols", f_columns, F_NEEDARG },
80 { "columns", f_columns, F_NEEDARG },
81 { "cooked", f_sane, 0 },
82 { "dec", f_dec, 0 },
83 { "extproc", f_extproc, F_OFFOK },
84 { "ispeed", f_ispeed, F_NEEDARG },
85 { "new", f_tty, 0 },
86 { "nl", f_nl, F_OFFOK },
87 { "old", f_tty, 0 },
88 { "ospeed", f_ospeed, F_NEEDARG },
89 { "raw", f_raw, F_OFFOK },
90 { "rows", f_rows, F_NEEDARG },
91 { "sane", f_sane, 0 },
92 { "tty", f_tty, 0 },
93};

95static int
96c_key(const void *a, const void *b)
97{
98
99 return (strcmp(((const struct key *)a)->name,
100 ((const struct key *)b)->name));
101}
102
103int
104ksearch(char ***argvp, struct info *ip)
105{
106 char *name;
107 struct key *kp, tmp;
108
109 name = **argvp;
110 if (*name == '-') {
111 ip->off = 1;
112 ++name;
113 } else
114 ip->off = 0;
115
116 tmp.name = name;
117 if (!(kp = (struct key *)bsearch(&tmp, keys,
118 sizeof(keys)/sizeof(struct key), sizeof(struct key), c_key)))
119 return (0);
120 if (!(kp->flags & F_OFFOK) && ip->off) {
121 syslog(LOG_INFO, "%s: illegal option: %s", printer, name);
122 return (1);
123 }
124 if (kp->flags & F_NEEDARG && !(ip->arg = *++*argvp)) {
125 syslog(LOG_INFO, "%s: option requires an argument: %s",
126 printer, name);
127 return (1);
128 }
129 kp->f(ip);
130 return (1);
131}

133void
134f_cbreak(struct info *ip)
135{
136
137 if (ip->off)
138 f_sane(ip);
139 else {
140 ip->t.c_iflag |= BRKINT|IXON|IMAXBEL;
141 ip->t.c_oflag |= OPOST;
142 ip->t.c_lflag |= ISIG|IEXTEN;
143 ip->t.c_lflag &= ~ICANON;
144 ip->set = 1;
145 }
146}

148void
149f_columns(struct info *ip)
150{
151
152 ip->win.ws_col = atoi(ip->arg);
153 ip->wset = 1;
154}
155
156void
157f_dec(struct info *ip)
158{
159
160 ip->t.c_cc[VERASE] = (u_char)0177;
161 ip->t.c_cc[VKILL] = CTRL('u');
162 ip->t.c_cc[VINTR] = CTRL('c');
163 ip->t.c_lflag &= ~ECHOPRT;
164 ip->t.c_lflag |= ECHOE|ECHOKE|ECHOCTL;
165 ip->t.c_iflag &= ~IXANY;
166 ip->set = 1;
167}

169void
170f_extproc(struct info *ip)
171{
172
173 if (ip->set) {
174 int tmp = 1;
175 (void)ioctl(ip->fd, TIOCEXT, &tmp);
176 } else {
177 int tmp = 0;
178 (void)ioctl(ip->fd, TIOCEXT, &tmp);
179 }
180}

183f_ispeed(struct info *ip)

185

190void
191f_nl(struct info *ip)
192{
193
194 if (ip->off) {
195 ip->t.c_iflag |= ICRNL;
196 ip->t.c_oflag |= ONLCR;
197 } else {
198 ip->t.c_iflag &= ~ICRNL;
199 ip->t.c_oflag &= ~ONLCR;
200 }
201 ip->set = 1;
202}

212void
213f_raw(struct info *ip)
214{
215
216 if (ip->off)
217 f_sane(ip);
218 else {
219 cfmakeraw(&ip->t);
220 ip->t.c_cflag &= ~(CSIZE|PARENB);
221 ip->t.c_cflag |= CS8;
222 ip->set = 1;
223 }
224}

226void
227f_rows(struct info *ip)
228{
229
230 ip->win.ws_row = atoi(ip->arg);
231 ip->wset = 1;
232}
233
234void
235f_sane(struct info *ip)
236{
237
238 ip->t.c_cflag = TTYDEF_CFLAG | (ip->t.c_cflag & (CLOCAL|CRTSCTS|CDTRCTS));
239 ip->t.c_iflag = TTYDEF_IFLAG;
240 ip->t.c_iflag |= ICRNL;
241 /* preserve user-preference flags in lflag */
242#define LKEEP (ECHOKE|ECHOE|ECHOK|ECHOPRT|ECHOCTL|ALTWERASE|TOSTOP|NOFLSH)
243 ip->t.c_lflag = TTYDEF_LFLAG | (ip->t.c_lflag & LKEEP);
244 ip->t.c_oflag = TTYDEF_OFLAG;
245 ip->set = 1;
246}

248void
249f_tty(struct info *ip)
250{
251 int tmp;
252
253 tmp = TTYDISC;
254 if (ioctl(0, TIOCSETD, &tmp) < 0)
255 syslog(LOG_ERR, "%s: ioctl(TIOCSETD): %m", printer);
256}
