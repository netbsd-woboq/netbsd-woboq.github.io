30 * cd9660_archimedes.c - support for RISC OS "ARCHIMEDES" extension
31 */
32
33struct ISO_ARCHIMEDES {
34 char magic[10]; /* "ARCHIMEDES" */
35 unsigned char loadaddr[4]; /* Load address, little-endian */
36 unsigned char execaddr[4]; /* Exec address, little-endian */
37 unsigned char ro_attr; /* RISC OS attributes */
38#define RO_ACCESS_UR 0x01 /* Owner read */
39#define RO_ACCESS_UW 0x02 /* Owner write */
40#define RO_ACCESS_L 0x04 /* Locked */
41#define RO_ACCESS_OR 0x10 /* Public read */
42#define RO_ACCESS_OW 0x20 /* Public write */
43 unsigned char cdfs_attr; /* Extra attributes for CDFS */
44#define CDFS_PLING 0x01 /* Filename begins with '!' */
45 char reserved[12];
46};
