38#ifndef _FFS_H
39#define _FFS_H
40
41#include <ufs/ufs/dinode.h>
42#include <ufs/ffs/fs.h>
43
44typedef struct {
45 char label[MAXVOLLEN]; /* volume name/label */
46 int bsize; /* block size */
47 int fsize; /* fragment size */
48 int cpg; /* cylinders per group */
49 int cpgflg; /* cpg was specified by user */
50 int density; /* bytes per inode */
51 int ntracks; /* number of tracks */
52 int nsectors; /* number of sectors */
53 int rpm; /* rpm */
54 int minfree; /* free space threshold */
55 int optimization; /* optimization (space or time) */
56 int maxcontig; /* max contiguous blocks to allocate */
57 int rotdelay; /* rotational delay between blocks */
58 int maxbpg; /* maximum blocks per file in a cyl group */
59 int nrpos; /* # of distinguished rotational positions */
60 int avgfilesize; /* expected average file size */
61 int avgfpdir; /* expected # of files per directory */
62 int version; /* filesystem version (1 = FFS, 2 = UFS2) */
63 int maxbsize; /* maximum extent size */
64 int maxblkspercg; /* max # of blocks per cylinder group */
65 /* XXX: support `old' file systems ? */
66} ffs_opt_t;
67
68#endif /* _FFS_H */
