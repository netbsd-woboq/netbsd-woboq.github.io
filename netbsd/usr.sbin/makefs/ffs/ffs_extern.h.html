37/*
38 * Structure used to pass around logical block paths generated by
39 * ufs_getlbns and used by truncate and bmap code.
40 */
41struct indir {
42 daddr_t in_lbn; /* Logical block number. */
43 int in_off; /* Offset in buffer. */
44 int in_exists; /* Flag if the block exists. */
45};
