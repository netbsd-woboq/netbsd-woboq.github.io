40union dinode {
41 struct ufs1_dinode ffs1_din;
42 struct ufs2_dinode ffs2_din;
43};
44
45struct inode {
46 ino_t i_number; /* The identity of the inode. */
47 struct vnode *i_devvp; /* device vnode for block I/O */
48 struct fs *i_fs; /* File system */
49 union dinode i_din;
50 uint64_t i_size;
51};

93#undef DIP
94#define DIP(ip, field) \
95 (((ip)->i_fs->fs_magic == FS_UFS1_MAGIC) ? \
96 (ip)->i_ffs1_##field : (ip)->i_ffs2_##field)
97
98#define DIP_ASSIGN(ip, field, value) \
99 do { \
100 if ((ip)->i_fs->fs_magic == FS_UFS1_MAGIC) \
101 (ip)->i_ffs1_##field = (value); \
102 else \
103 (ip)->i_ffs2_##field = (value); \
104 } while(0)
105
106#define DIP_ADD(ip, field, value) \
107 do { \
108 if ((ip)->i_fs->fs_magic == FS_UFS1_MAGIC) \
109 (ip)->i_ffs1_##field += (value); \
110 else \
111 (ip)->i_ffs2_##field += (value); \
112 } while(0)
