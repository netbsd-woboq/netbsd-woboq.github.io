29#ifndef _CDIO_MMC_EMU_H_
30#define _CDIO_MMC_EMU_H_
31
32#include <sys/types.h>
33
34/*
35 * MMC device abstraction interface.
36 *
37 * It gathers information from GET_CONFIGURATION, READ_DISCINFO,
38 * READ_TRACKINFO, READ_TOC2, READ_CD_CAPACITY and GET_CONFIGURATION
39 * SCSI/ATAPI calls regardless if its a legacy CD-ROM/DVD-ROM device or a MMC
40 * standard recordable device.
41 */
42struct mmc_discinfo {
43 uint16_t mmc_profile;
44 uint16_t mmc_class;
45
46 uint8_t disc_state;
47 uint8_t last_session_state;
48 uint8_t bg_format_state;
49 uint8_t link_block_penalty; /* in sectors */
50
51 uint64_t mmc_cur; /* current MMC_CAPs */
52 uint64_t mmc_cap; /* possible MMC_CAPs */
53
54 uint32_t disc_flags; /* misc flags */
55
56 uint32_t disc_id;
57 uint64_t disc_barcode;
58 uint8_t application_code; /* 8 bit really */
59
60 uint8_t unused1[3]; /* padding */
61
62 uint32_t last_possible_lba; /* last leadout start adr. */
63 uint32_t sector_size;
64
65 uint16_t num_sessions;
66 uint16_t num_tracks; /* derived */
67
68 uint16_t first_track;
69 uint16_t first_track_last_session;
70 uint16_t last_track_last_session;
71
72 uint16_t unused2; /* padding/misc info resv. */
73
74 uint16_t reserved1[4]; /* MMC-5 track resources */
75 uint32_t reserved2[3]; /* MMC-5 POW resources */
76
77 uint32_t reserved3[8]; /* MMC-5+ */
78};
79#define MMCGETDISCINFO _IOR('c', 28, struct mmc_discinfo)
80
81#define MMC_CLASS_UNKN 0
82#define MMC_CLASS_DISC 1
83#define MMC_CLASS_CD 2
84#define MMC_CLASS_DVD 3
85#define MMC_CLASS_MO 4
86#define MMC_CLASS_BD 5
87#define MMC_CLASS_FILE 0xffff /* emulation mode */
88
89#define MMC_DFLAGS_BARCODEVALID (1 << 0) /* barcode is present and valid */
90#define MMC_DFLAGS_DISCIDVALID (1 << 1) /* discid is present and valid */
91#define MMC_DFLAGS_APPCODEVALID (1 << 2) /* application code valid */
92#define MMC_DFLAGS_UNRESTRICTED (1 << 3) /* restricted, then set app. code */
93
94#define MMC_DFLAGS_FLAGBITS \
95 "\10\1BARCODEVALID\2DISCIDVALID\3APPCODEVALID\4UNRESTRICTED"
96
97#define MMC_CAP_SEQUENTIAL (1 << 0) /* sequential writable only */
98#define MMC_CAP_RECORDABLE (1 << 1) /* record-able; i.e. not static */
99#define MMC_CAP_ERASABLE (1 << 2) /* drive can erase sectors */
100#define MMC_CAP_BLANKABLE (1 << 3) /* media can be blanked */
101#define MMC_CAP_FORMATTABLE (1 << 4) /* media can be formatted */
102#define MMC_CAP_REWRITABLE (1 << 5) /* media can be rewritten */
103#define MMC_CAP_MRW (1 << 6) /* Mount Rainier formatted */
104#define MMC_CAP_PACKET (1 << 7) /* using packet recording */
105#define MMC_CAP_STRICTOVERWRITE (1 << 8) /* only writes a packet at a time */
106#define MMC_CAP_PSEUDOOVERWRITE (1 << 9) /* overwrite through replacement */
107#define MMC_CAP_ZEROLINKBLK (1 << 10) /* zero link block length capable */
108#define MMC_CAP_HW_DEFECTFREE (1 << 11) /* hardware defect management */
109
110#define MMC_CAP_FLAGBITS \
111 "\10\1SEQUENTIAL\2RECORDABLE\3ERASABLE\4BLANKABLE\5FORMATTABLE" \
112 "\6REWRITABLE\7MRW\10PACKET\11STRICTOVERWRITE\12PSEUDOOVERWRITE" \
113 "\13ZEROLINKBLK\14HW_DEFECTFREE"
114
115#define MMC_STATE_EMPTY 0
116#define MMC_STATE_INCOMPLETE 1
117#define MMC_STATE_FULL 2
118#define MMC_STATE_CLOSED 3
119
120#define MMC_BGFSTATE_UNFORM 0
121#define MMC_BGFSTATE_STOPPED 1
122#define MMC_BGFSTATE_RUNNING 2
123#define MMC_BGFSTATE_COMPLETED 3
124

126struct mmc_trackinfo {
127 uint16_t tracknr; /* IN/OUT */
128 uint16_t sessionnr;
129
130 uint8_t track_mode;
131 uint8_t data_mode;
132
133 uint16_t flags;
134
135 uint32_t track_start;
136 uint32_t next_writable;
137 uint32_t free_blocks;
138 uint32_t packet_size;
139 uint32_t track_size;
140 uint32_t last_recorded;
141};
142#define MMCGETTRACKINFO _IOWR('c', 29, struct mmc_trackinfo)
143
144#define MMC_TRACKINFO_COPY (1 << 0)
145#define MMC_TRACKINFO_DAMAGED (1 << 1)
146#define MMC_TRACKINFO_FIXED_PACKET (1 << 2)
147#define MMC_TRACKINFO_INCREMENTAL (1 << 3)
148#define MMC_TRACKINFO_BLANK (1 << 4)
149#define MMC_TRACKINFO_RESERVED (1 << 5)
150#define MMC_TRACKINFO_NWA_VALID (1 << 6)
151#define MMC_TRACKINFO_LRA_VALID (1 << 7)
152#define MMC_TRACKINFO_DATA (1 << 8)
153#define MMC_TRACKINFO_AUDIO (1 << 9)
154#define MMC_TRACKINFO_AUDIO_4CHAN (1 << 10)
155#define MMC_TRACKINFO_PRE_EMPH (1 << 11)
156
157#define MMC_TRACKINFO_FLAGBITS \
158 "\10\1COPY\2DAMAGED\3FIXEDPACKET\4INCREMENTAL\5BLANK" \
159 "\6RESERVED\7NWA_VALID\10LRA_VALID\11DATA\12AUDIO" \
160 "\13AUDIO_4CHAN\14PRE_EMPH"
161
162#endif /* _CDIO_MMC_EMU_H_ */
