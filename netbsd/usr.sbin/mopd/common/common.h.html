47 * structure per interface
48 *
49 */
50
51struct if_info {
52 int fd; /* File Descriptor */
53 int trans; /* Transport type Ethernet/802.3 */
54 u_char eaddr[6]; /* Ethernet addr of this interface */
55 char if_name[IFNAME_SIZE]; /* Interface Name */
56 int (*iopen)(const char *, int, u_short, int);
57 /* Interface Open Routine */
58 int (*write)(int, const u_char *, int, int);
59 /* Interface Write Routine */
60 void (*read)(void); /* Interface Read Routine */
61 struct if_info *next; /* Next Interface */
62};
63
64#define DL_STATUS_FREE 0
65#define DL_STATUS_READ_IMGHDR 1
66#define DL_STATUS_SENT_MLD 2
67#define DL_STATUS_SENT_PLT 3
68
69typedef enum {
70 IMAGE_TYPE_MOP, /* MOP image */
71 IMAGE_TYPE_AOUT, /* a.out image */
72 IMAGE_TYPE_ELF32 /* Elf32 image */
73} mopd_imagetype;
74
75struct dllist {
76 u_char status; /* Status byte */
77 struct if_info *ii; /* interface pointer */
78 u_char eaddr[6]; /* targets ethernet addres */
79 int ldfd; /* filedescriptor for loadfile */
80 u_short dl_bsz; /* Data Link Buffer Size */
81 int timeout; /* Timeout counter */
82 u_char count; /* Packet Counter */
83 u_int32_t loadaddr; /* Load Address */
84 u_int32_t xferaddr; /* Transfer Address */
85 u_int32_t nloadaddr; /* Next Load Address */
86 off_t lseek; /* Seek before last read */
87 mopd_imagetype image_type; /* what type of image is it? */
88
89 /* For Elf32 files */
90 int e_machine; /* Machine ID */
91 int e_nsec; /* number of program sections */
92#define SEC_MAX 4
93 struct {
94 off_t s_foff; /* file offset of section */
95 u_int32_t s_vaddr; /* virtual address of section */
96 u_int32_t s_fsize; /* file size of section */
97 u_int32_t s_msize; /* memory size of section */
98 u_int32_t s_pad; /* padding until next section */
99 u_int32_t s_loff; /* logical offset into image */
100 } e_sections[SEC_MAX]; /* program sections */
101 u_int32_t e_curpos; /* current logical position */
102 int e_cursec; /* current section */
103
104 /* For a.out files */
105 int a_mid; /* Machine ID */
106 u_int32_t a_text; /* Size of text segment */
107 u_int32_t a_text_fill; /* Size of text segment fill */
108 u_int32_t a_data; /* Size of data segment */
109 u_int32_t a_data_fill; /* Size of data segment fill */
110 u_int32_t a_bss; /* Size of bss segment */
111 u_int32_t a_bss_fill; /* Size of bss segment fill */
112 off_t a_lseek; /* Keep track of pos in newfile */
113};
