36u_char
37mopGetChar(const u_char *pkt, int *idx)
38{
39 u_char ret;
40
41 ret = pkt[*idx];
42 *idx = *idx + 1;
43 return(ret);
44}
45
46u_short
47mopGetShort(const u_char *pkt, int *idx)
48{
49 u_short ret;
50
51 ret = pkt[*idx] + pkt[*idx+1]*256;
52 *idx = *idx + 2;
53 return(ret);
54}
55
56u_int32_t
57mopGetLong(const u_char *pkt, int *idx)
58{
59 u_int32_t ret;
60
61 ret = pkt[*idx] +
62 pkt[*idx+1]*0x100 +
63 pkt[*idx+2]*0x10000 +
64 pkt[*idx+3]*0x1000000;
65 *idx = *idx + 4;
66 return(ret);
67}

69void
70mopGetMulti(const u_char *pkt, int *idx, u_char *dest, int size)
71{
72 int i;
73
74 for (i = 0; i < size; i++) {
75 dest[i] = pkt[*idx+i];
76 }
77 *idx = *idx + size;
78
79}

81int
82mopGetTrans(const u_char *pkt, int trans)
83{
84 const u_short *ptype;
85
86 if (trans == 0) {
87 ptype = (const u_short *)(pkt+12);
88 if (ntohs(*ptype) < 1600) {
89 trans = TRANS_8023;
90 } else {
91 trans = TRANS_ETHER;
92 }
93 }
94 return(trans);
95}
96
97void
98mopGetHeader(const u_char *pkt, int *idx, const u_char **dst, const u_char **src,
99 u_short *proto, int *len, int trans)
100{
101 *dst = pkt;
102 *src = pkt + 6;
103 *idx = *idx + 12;
104
105 switch(trans) {
106 case TRANS_ETHER:
107 *proto = (u_short)(pkt[*idx]*256 + pkt[*idx+1]);
108 *idx = *idx + 2;
109 *len = (int)(pkt[*idx+1]*256 + pkt[*idx]);
110 *idx = *idx + 2;
111 break;
112 case TRANS_8023:
113 *len = (int)(pkt[*idx]*256 + pkt[*idx+1]);
114 *idx = *idx + 8;
115 *proto = (u_short)(pkt[*idx]*256 + pkt[*idx+1]);
116 *idx = *idx + 2;
117 break;
118 }
119}
120
121u_short
122mopGetLength(const u_char *pkt, int trans)
123{
124 switch(trans) {
125 case TRANS_ETHER:
126 return(pkt[15]*256 + pkt[14]);
127 break;
128 case TRANS_8023:
129 return(pkt[12]*256 + pkt[13]);
130 break;
131 }
132 return(0);
133}
