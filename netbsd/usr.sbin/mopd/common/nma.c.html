190const char *
191nmaGetShort(int devno)
192{
193 struct commDev *current;
194
195 current = nmaCommDev;
196
197 while (current->sname != NULL) {
198 if (current->val == devno)
199 break;
200 current++;
201 }
202
203 return(current->sname);
204}
205
206const char *
207nmaGetDevice(int devno)
208{
209 struct commDev *current;
210
211 current = nmaCommDev;
212
213 while (current->name != NULL) {
214 if (current->val == devno)
215 break;
216 current++;
217 }
218
219 return(current->name);
220}
