111/*
112 * Compute a POSIX 1003.2 checksum. This routine has been broken out so that
113 * other programs can use it. It takes a file descriptor to read from and
114 * locations to store the crc and the number of bytes read. It returns 0 on
115 * success and 1 on failure. Errno is set on failure.
116 */
117u_int32_t crc_total = ~0; /* The crc over a number of files. */
118
119int
120crc(int fd, u_int32_t *cval, u_int32_t *clen)
121{
122 u_char *p;
123 int nr;
124 u_int32_t thecrc, len;
125 u_int32_t crctot;
126 u_char buf[16 * 1024];
127
128#define COMPUTE(var, ch) (var) = (var) << 8 ^ crctab[(var) >> 24 ^ (ch)]
129
130 thecrc = len = crctot = 0;
131 if (sflag)
132 crctot = ~crc_total;
133 while ((nr = read(fd, buf, sizeof(buf))) > 0)
134 if (sflag) {
135 for (len += nr, p = buf; nr--; ++p) {
136 COMPUTE(thecrc, *p);
137 COMPUTE(crctot, *p);
138 }
139 } else {
140 for (len += nr, p = buf; nr--; ++p)
141 COMPUTE(thecrc, *p);
142 }
143 if (nr < 0)
144 return 1;
145
146 *clen = len;
147
148 /* Include the length of the file. */
149 if (sflag) {
150 for (; len != 0; len >>= 8) {
151 COMPUTE(thecrc, len & 0xff);
152 COMPUTE(crctot, len & 0xff);
153 }
154 } else {
155 for (; len != 0; len >>= 8)
156 COMPUTE(thecrc, len & 0xff);
157 }
158
159 *cval = ~thecrc;
160 if (sflag)
161 crc_total = ~crctot;
162 return 0;
163}
