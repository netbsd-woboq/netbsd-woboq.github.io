100void
101cwalk(FILE *fp)
102{
103 FTS *t;
104 FTSENT *p;
105 time_t clocktime;
106 char host[MAXHOSTNAMELEN + 1];
107 const char *user;
108 char *argv[2];
109 char dot[] = ".";
110 int indent = 0;
111
112 argv[0] = dot;
113 argv[1] = NULL;
114
115 time(&clocktime);
116 gethostname(host, sizeof(host));
117 host[sizeof(host) - 1] = '\0';
118 if ((user = getlogin()) == NULL) {
119 struct passwd *pw;
120 user = (pw = getpwuid(getuid())) != NULL ? pw->pw_name :
121 "<unknown>";
122 }
123
124 if (!nflag)
125 fprintf(fp,
126 "#\t user: %s\n#\tmachine: %s\n#\t tree: %s\n"
127 "#\t date: %s",
128 user, host, fullpath, ctime(&clocktime));
129
130 if ((t = fts_open(argv, ftsoptions, dcmp)) == NULL)
131 mtree_err("fts_open: %s", strerror(errno));
132 while ((p = fts_read(t)) != NULL) {
133 if (jflag)
134 indent = p->fts_level * 4;
135 if (check_excludes(p->fts_name, p->fts_path)) {
136 fts_set(t, p, FTS_SKIP);
137 continue;
138 }
139 if (!find_only(p->fts_path)) {
140 fts_set(t, p, FTS_SKIP);
141 continue;
142 }
143 switch(p->fts_info) {
144 case FTS_D:
145 if (!bflag)
146 fprintf(fp, "\n");
147 if (!nflag)
148 fprintf(fp, "# %s\n", p->fts_path);
149 statd(fp, t, p, &uid, &gid, &mode, &flags);
150 statf(fp, indent, p);
151 break;
152 case FTS_DP:
153 if (p->fts_level > 0)
154 if (!nflag)
155 fprintf(fp, "%*s# %s\n", indent, "",
156 p->fts_path);
157 if (p->fts_level > 0 || flavor == F_FREEBSD9) {
158 fprintf(fp, "%*s..\n", indent, "");
159 if (!bflag)
160 fprintf(fp, "\n");
161 }
162 break;
163 case FTS_DNR:
164 case FTS_ERR:
165 case FTS_NS:
166 mtree_err("%s: %s",
167 p->fts_path, strerror(p->fts_errno));
168 break;
169 default:
170 if (!dflag)
171 statf(fp, indent, p);
172 break;
173
174 }
175 }
176 fts_close(t);
177 if (sflag && keys & F_CKSUM)
178 mtree_err("%s checksum: %u", fullpath, crc_total);
179}

181static void
182dosum(FILE *fp, int indent, FTSENT *p, int *offset, int flag,
183 char * (*func)(const char *, char *), const char *key)
184{
185 char *digestbuf;
186
187 if ((keys & flag) == 0)
188 return;
189
190 digestbuf = (*func)(p->fts_accpath, NULL);
191 if (digestbuf != NULL) {
192 output(fp, indent, offset, "%s=%s", key, digestbuf);
193 free(digestbuf);
194 return;
195 }
196
197 if (qflag) {
198 warn("%s: %s failed", p->fts_path, key);
199 return;
200 }
201
202 mtree_err("%s: %s failed: %s", p->fts_path, key, strerror(errno));
203}

205static char *
206crcFile(const char *fname, char *dummy __unused)
207{
208 char *ptr;
209 uint32_t val, len;
210 int fd, e;
211
212 if ((fd = open(fname, O_RDONLY)) == -1)
213 goto out;
214
215 e = crc(fd, &val, &len);
216 close(fd);
217 if (e)
218 goto out;
219
220 if (asprintf(&ptr, "%u", val) < 0)
221 goto out;
222
223 return ptr;
224out:
225 mtree_err("%s: %s", fname, strerror(errno));
226 return NULL;
227}

229static void
230statf(FILE *fp, int indent, FTSENT *p)
231{
232 int offset;
233 const char *name = NULL;
234
235 offset = fprintf(fp, "%*s%s%s", indent, "",
236 S_ISDIR(p->fts_statp->st_mode) ? "" : " ", vispath(p->fts_name));
237
238 if (offset > (INDENTNAMELEN + indent))
239 offset = MAXLINELEN;
240 else
241 offset += fprintf(fp, "%*s",
242 (INDENTNAMELEN + indent) - offset, "");
243
244 if (!S_ISREG(p->fts_statp->st_mode) && (flavor == F_NETBSD6 || !dflag))
245 output(fp, indent, &offset, "type=%s",
246 inotype(p->fts_statp->st_mode));
247 if (keys & (F_UID | F_UNAME) && p->fts_statp->st_uid != uid) {
248 if (keys & F_UNAME &&
249 (name = user_from_uid(p->fts_statp->st_uid, 1)) != NULL)
250 output(fp, indent, &offset, "uname=%s", name);
251 if (keys & F_UID || (keys & F_UNAME && name == NULL))
252 output(fp, indent, &offset, "uid=%u",
253 p->fts_statp->st_uid);
254 }
255 if (keys & (F_GID | F_GNAME) && p->fts_statp->st_gid != gid) {
256 if (keys & F_GNAME &&
257 (name = group_from_gid(p->fts_statp->st_gid, 1)) != NULL)
258 output(fp, indent, &offset, "gname=%s", name);
259 if (keys & F_GID || (keys & F_GNAME && name == NULL))
260 output(fp, indent, &offset, "gid=%u",
261 p->fts_statp->st_gid);
262 }
263 if (keys & F_MODE && (p->fts_statp->st_mode & MBITS) != mode)
264 output(fp, indent, &offset, "mode=%#o",
265 p->fts_statp->st_mode & MBITS);
266 if (keys & F_DEV &&
267 (S_ISBLK(p->fts_statp->st_mode) || S_ISCHR(p->fts_statp->st_mode)))
268 output(fp, indent, &offset, "device=%#jx",
269 (uintmax_t)p->fts_statp->st_rdev);
270 if (keys & F_NLINK && p->fts_statp->st_nlink != 1)
271 output(fp, indent, &offset, "nlink=%ju",
272 (uintmax_t)p->fts_statp->st_nlink);
273 if (keys & F_SIZE &&
274 (flavor == F_FREEBSD9 || S_ISREG(p->fts_statp->st_mode)))
275 output(fp, indent, &offset, "size=%ju",
276 (uintmax_t)p->fts_statp->st_size);
277 if (keys & F_TIME)
278#if defined(BSD4_4) && !defined(HAVE_NBTOOL_CONFIG_H)
279 output(fp, indent, &offset, "time=%jd.%09ld",
280 (intmax_t)p->fts_statp->st_mtimespec.tv_sec,
281 p->fts_statp->st_mtimespec.tv_nsec);
282#else
283 output(fp, indent, &offset, "time=%jd.%09ld",
284 (intmax_t)p->fts_statp->st_mtime, (long)0);
285#endif
286 if (S_ISREG(p->fts_statp->st_mode)) {
287 dosum(fp, indent, p, &offset, F_CKSUM, crcFile, "cksum");
288#ifndef NO_MD5
289 dosum(fp, indent, p, &offset, F_MD5, MD5File, MD5KEY);
290#endif /* ! NO_MD5 */
291#ifndef NO_RMD160
292 dosum(fp, indent, p, &offset, F_RMD160, RMD160File, RMD160KEY);
293#endif /* ! NO_RMD160 */
294#ifndef NO_SHA1
295 dosum(fp, indent, p, &offset, F_SHA1, SHA1File, SHA1KEY);
296#endif /* ! NO_SHA1 */
297#ifndef NO_SHA2
298 dosum(fp, indent, p, &offset, F_SHA256, SHA256_File, SHA256KEY);
299#ifdef SHA384_BLOCK_LENGTH
300 dosum(fp, indent, p, &offset, F_SHA384, SHA384_File, SHA384KEY);
301#endif
302 dosum(fp, indent, p, &offset, F_SHA512, SHA512_File, SHA512KEY);
303#endif /* ! NO_SHA2 */
304 }
305 if (keys & F_SLINK &&
306 (p->fts_info == FTS_SL || p->fts_info == FTS_SLNONE))
307 output(fp, indent, &offset, "link=%s",
308 vispath(rlink(p->fts_accpath)));
309#if HAVE_STRUCT_STAT_ST_FLAGS
310 if (keys & F_FLAGS && p->fts_statp->st_flags != flags) {
311 char *str = flags_to_string(p->fts_statp->st_flags, "none");
312 output(fp, indent, &offset, "flags=%s", str);
313 free(str);
314 }
315#endif
316 putchar('\n');
317}

319/* XXX
320 * FLAGS2INDEX will fail once the user and system settable bits need more
321 * than one byte, respectively.
322 */
323#define FLAGS2INDEX(x) (((x >> 8) & 0x0000ff00) | (x & 0x000000ff))
324
325#define MTREE_MAXGID 5000
326#define MTREE_MAXUID 5000
327#define MTREE_MAXMODE (MBITS + 1)
328#if HAVE_STRUCT_STAT_ST_FLAGS
329#define MTREE_MAXFLAGS (FLAGS2INDEX(CH_MASK) + 1) /* 1808 */
330#else
331#define MTREE_MAXFLAGS 1
332#endif
333#define MTREE_MAXS 16

335static int
336statd(FILE *fp, FTS *t, FTSENT *parent, uid_t *puid, gid_t *pgid, mode_t *pmode,
337 u_long *pflags)
338{
339 FTSENT *p;
340 gid_t sgid;
341 uid_t suid;
342 mode_t smode;
343 u_long sflags = 0;
344 const char *name = NULL;
345 gid_t savegid;
346 uid_t saveuid;
347 mode_t savemode;
348 u_long saveflags;
349 u_short maxgid, maxuid, maxmode, maxflags;
350 u_short g[MTREE_MAXGID], u[MTREE_MAXUID],
351 m[MTREE_MAXMODE], f[MTREE_MAXFLAGS];
352 static int first = 1;
353
354 savegid = *pgid;
355 saveuid = *puid;
356 savemode = *pmode;
357 saveflags = *pflags;
358 if ((p = fts_children(t, 0)) == NULL) {
359 if (errno)
360 mtree_err("%s: %s", RP(parent), strerror(errno));
361 return (1);
362 }
363
364 memset(g, 0, sizeof(g));
365 memset(u, 0, sizeof(u));
366 memset(m, 0, sizeof(m));
367 memset(f, 0, sizeof(f));
368
369 maxuid = maxgid = maxmode = maxflags = 0;
370 for (; p; p = p->fts_link) {
371 if (flavor == F_NETBSD6 || !dflag ||
372 (dflag && S_ISDIR(p->fts_statp->st_mode))) {
373 smode = p->fts_statp->st_mode & MBITS;
374 if (smode < MTREE_MAXMODE && ++m[smode] > maxmode) {
375 savemode = smode;
376 maxmode = m[smode];
377 }
378 sgid = p->fts_statp->st_gid;
379 if (sgid < MTREE_MAXGID && ++g[sgid] > maxgid) {
380 savegid = sgid;
381 maxgid = g[sgid];
382 }
383 suid = p->fts_statp->st_uid;
384 if (suid < MTREE_MAXUID && ++u[suid] > maxuid) {
385 saveuid = suid;
386 maxuid = u[suid];
387 }
388
389#if HAVE_STRUCT_STAT_ST_FLAGS
390 sflags = FLAGS2INDEX(p->fts_statp->st_flags);
391 if (sflags < MTREE_MAXFLAGS && ++f[sflags] > maxflags) {
392 saveflags = p->fts_statp->st_flags;
393 maxflags = f[sflags];
394 }
395#endif
396 }
397 }
398 /*
399 * If the /set record is the same as the last one we do not need to
400 * output a new one. So first we check to see if anything changed.
401 * Note that we always output a /set record for the first directory.
402 */
403 if (((keys & (F_UNAME | F_UID)) && (*puid != saveuid)) ||
404 ((keys & (F_GNAME | F_GID)) && (*pgid != savegid)) ||
405 ((keys & F_MODE) && (*pmode != savemode)) ||
406 ((keys & F_FLAGS) && (*pflags != saveflags)) ||
407 first) {
408 first = 0;
409 if (flavor != F_NETBSD6 && dflag)
410 fprintf(fp, "/set type=dir");
411 else
412 fprintf(fp, "/set type=file");
413 if (keys & (F_UID | F_UNAME)) {
414 if (keys & F_UNAME &&
415 (name = user_from_uid(saveuid, 1)) != NULL)
416 fprintf(fp, " uname=%s", name);
417 if (keys & F_UID || (keys & F_UNAME && name == NULL))
418 fprintf(fp, " uid=%lu", (u_long)saveuid);
419 }
420 if (keys & (F_GID | F_GNAME)) {
421 if (keys & F_GNAME &&
422 (name = group_from_gid(savegid, 1)) != NULL)
423 fprintf(fp, " gname=%s", name);
424 if (keys & F_GID || (keys & F_GNAME && name == NULL))
425 fprintf(fp, " gid=%lu", (u_long)savegid);
426 }
427 if (keys & F_MODE)
428 fprintf(fp, " mode=%#lo", (u_long)savemode);
429 if (keys & F_NLINK)
430 fprintf(fp, " nlink=1");
431 if (keys & F_FLAGS) {
432 char *str = flags_to_string(saveflags, "none");
433 fprintf(fp, " flags=%s", str);
434 free(str);
435 }
436 fprintf(fp, "\n");
437 *puid = saveuid;
438 *pgid = savegid;
439 *pmode = savemode;
440 *pflags = saveflags;
441 }
442 return (0);
443}

445/*
446 * dcmp --
447 * used as a comparison function passed to fts_open() to control
448 * the order in which fts_read() returns results. We make
449 * directories sort after non-directories, but otherwise sort in
450 * strcmp() order.
451 *
452 * Keep this in sync with nodecmp() in spec.c.
453 */
454static int
455dcmp(const FTSENT *FTS_CONST *a, const FTSENT *FTS_CONST *b)
456{
457
458 if (S_ISDIR((*a)->fts_statp->st_mode)) {
459 if (!S_ISDIR((*b)->fts_statp->st_mode))
460 return (1);
461 } else if (S_ISDIR((*b)->fts_statp->st_mode))
462 return (-1);
463 return (strcmp((*a)->fts_name, (*b)->fts_name));
464}

466void
467output(FILE *fp, int indent, int *offset, const char *fmt, ...)
468{
469 va_list ap;
470 char buf[1024];
471
472 va_start(ap, fmt);
473 vsnprintf(buf, sizeof(buf), fmt, ap);
474 va_end(ap);
475
476 if (*offset + strlen(buf) > MAXLINELEN - 3) {
477 fprintf(fp, " \\\n%*s", INDENTNAMELEN + indent, "");
478 *offset = INDENTNAMELEN + indent;
479 }
480 *offset += fprintf(fp, " %s", buf) + 1;
481}
