60static NODE *root;
61static char path[MAXPATHLEN];
62
63static void miss(NODE *, char *);
64static int vwalk(void);
65
66int
67verify(FILE *fi)
68{
69 int rval;
70
71 root = spec(fi);
72 rval = vwalk();
73 miss(root, path);
74 return (rval);
75}

77static int
78vwalk(void)
79{
80 FTS *t;
81 FTSENT *p;
82 NODE *ep, *level;
83 int specdepth, rval;
84 char *argv[2];
85 char dot[] = ".";
86 argv[0] = dot;
87 argv[1] = NULL;
88
89 if ((t = fts_open(argv, ftsoptions, NULL)) == NULL)
90 mtree_err("fts_open: %s", strerror(errno));
91 level = root;
92 specdepth = rval = 0;
93 while ((p = fts_read(t)) != NULL) {
94 if (check_excludes(p->fts_name, p->fts_path)) {
95 fts_set(t, p, FTS_SKIP);
96 continue;
97 }
98 if (!find_only(p->fts_path)) {
99 fts_set(t, p, FTS_SKIP);
100 continue;
101 }
102 switch(p->fts_info) {
103 case FTS_D:
104 case FTS_SL:
105 break;
106 case FTS_DP:
107 if (specdepth > p->fts_level) {
108 for (level = level->parent; level->prev;
109 level = level->prev)
110 continue;
111 --specdepth;
112 }
113 continue;
114 case FTS_DNR:
115 case FTS_ERR:
116 case FTS_NS:
117 warnx("%s: %s", RP(p), strerror(p->fts_errno));
118 continue;
119 default:
120 if (dflag)
121 continue;
122 }
123
124 if (specdepth != p->fts_level)
125 goto extra;
126 for (ep = level; ep; ep = ep->next)
127 if ((ep->flags & F_MAGIC &&
128 !fnmatch(ep->name, p->fts_name, FNM_PATHNAME)) ||
129 !strcmp(ep->name, p->fts_name)) {
130 ep->flags |= F_VISIT;
131 if ((ep->flags & F_NOCHANGE) == 0 &&
132 compare(ep, p))
133 rval = MISMATCHEXIT;
134 if (!(ep->flags & F_IGN) &&
135 ep->type == F_DIR &&
136 p->fts_info == FTS_D) {
137 if (ep->child) {
138 level = ep->child;
139 ++specdepth;
140 }
141 } else
142 fts_set(t, p, FTS_SKIP);
143 break;
144 }
145
146 if (ep)
147 continue;
148 extra:
149 if (!eflag && !(dflag && p->fts_info == FTS_SL)) {
150 printf("extra: %s", RP(p));
151 if (rflag) {
152#if HAVE_STRUCT_STAT_ST_FLAGS
153 if (rflag > 1 &&
154 lchflags(p->fts_accpath, 0) == -1)
155 printf(" (chflags %s)",
156 strerror(errno));
157#endif
158 if ((S_ISDIR(p->fts_statp->st_mode)
159 ? rmdir : unlink)(p->fts_accpath)) {
160 printf(", not removed: %s",
161 strerror(errno));
162 } else
163 printf(", removed");
164 }
165 putchar('\n');
166 }
167 fts_set(t, p, FTS_SKIP);
168 }
169 fts_close(t);
170 if (sflag)
171 warnx("%s checksum: %u", fullpath, crc_total);
172 return (rval);
173}

175static void
176miss(NODE *p, char *tail)
177{
178 int create;
179 char *tp;
180 const char *type;
181 u_int32_t flags;
182
183 for (; p; p = p->next) {
184 if (p->flags & F_OPT && !(p->flags & F_VISIT))
185 continue;
186 if (p->type != F_DIR && (dflag || p->flags & F_VISIT))
187 continue;
188 strcpy(tail, p->name);
189 if (!(p->flags & F_VISIT)) {
190 /* Don't print missing message if file exists as a
191 symbolic link and the -q flag is set. */
192 struct stat statbuf;
193
194 if (qflag && stat(path, &statbuf) == 0 &&
195 S_ISDIR(statbuf.st_mode))
196 p->flags |= F_VISIT;
197 else
198 (void)printf("%s missing", path);
199 }
200 switch (p->type) {
201 case F_BLOCK:
202 case F_CHAR:
203 type = "device";
204 break;
205 case F_DIR:
206 type = "directory";
207 break;
208 case F_LINK:
209 type = "symlink";
210 break;
211 default:
212 putchar('\n');
213 continue;
214 }
215
216 create = 0;
217 if (!(p->flags & F_VISIT) && uflag) {
218 if (mtree_Wflag || p->type == F_LINK)
219 goto createit;
220 if (!(p->flags & (F_UID | F_UNAME)))
221 printf(
222 " (%s not created: user not specified)", type);
223 else if (!(p->flags & (F_GID | F_GNAME)))
224 printf(
225 " (%s not created: group not specified)", type);
226 else if (!(p->flags & F_MODE))
227 printf(
228 " (%s not created: mode not specified)", type);
229 else
230 createit:
231 switch (p->type) {
232 case F_BLOCK:
233 case F_CHAR:
234 if (mtree_Wflag)
235 continue;
236 if (!(p->flags & F_DEV))
237 printf(
238 " (%s not created: device not specified)",
239 type);
240 else if (mknod(path,
241 p->st_mode | nodetoino(p->type),
242 p->st_rdev) == -1)
243 printf(" (%s not created: %s)\n",
244 type, strerror(errno));
245 else
246 create = 1;
247 break;
248 case F_LINK:
249 if (!(p->flags & F_SLINK))
250 printf(
251 " (%s not created: link not specified)\n",
252 type);
253 else if (symlink(p->slink, path))
254 printf(
255 " (%s not created: %s)\n",
256 type, strerror(errno));
257 else
258 create = 1;
259 break;
260 case F_DIR:
261 if (mkdir(path, S_IRWXU|S_IRWXG|S_IRWXO))
262 printf(" (not created: %s)",
263 strerror(errno));
264 else
265 create = 1;
266 break;
267 default:
268 mtree_err("can't create create %s",
269 nodetype(p->type));
270 }
271 }
272 if (create)
273 printf(" (created)");
274 if (p->type == F_DIR) {
275 if (!(p->flags & F_VISIT))
276 putchar('\n');
277 for (tp = tail; *tp; ++tp)
278 continue;
279 *tp = '/';
280 miss(p->child, tp + 1);
281 *tp = '\0';
282 } else
283 putchar('\n');
284
285 if (!create || mtree_Wflag)
286 continue;
287 if ((p->flags & (F_UID | F_UNAME)) &&
288 (p->flags & (F_GID | F_GNAME)) &&
289 (lchown(path, p->st_uid, p->st_gid))) {
290 printf("%s: user/group/mode not modified: %s\n",
291 path, strerror(errno));
292 printf("%s: warning: file mode %snot set\n", path,
293 (p->flags & F_FLAGS) ? "and file flags " : "");
294 continue;
295 }
296 if (p->flags & F_MODE) {
297 if (lchmod(path, p->st_mode))
298 printf("%s: permissions not set: %s\n",
299 path, strerror(errno));
300 }
301#if HAVE_STRUCT_STAT_ST_FLAGS
302 if ((p->flags & F_FLAGS) && p->st_flags) {
303 if (iflag)
304 flags = p->st_flags;
305 else
306 flags = p->st_flags & ~SP_FLGS;
307 if (lchflags(path, flags))
308 printf("%s: file flags not set: %s\n",
309 path, strerror(errno));
310 }
311#endif /* HAVE_STRUCT_STAT_ST_FLAGS */
312 }
