17#define CHECK_TRUE(x) \
18 if (!(x)) { printf("FAIL: %s line %d\n", __func__, __LINE__); return 0; }

20static unsigned
21count_conns(npf_conndb_t *cd)
22{
23 npf_conn_t *head = npf_conndb_getlist(cd), *conn = head;
24 unsigned n = 0;
25
26 while (conn) {
27 n++;
28 conn = npf_conndb_getnext(cd, conn);
29 if (conn == head) {
30 break;
31 }
32 }
33 return n;
34}

36static struct mbuf *
37fill_packet(unsigned i)
38{
39 struct mbuf *m;
40 struct ip *ip;
41 struct udphdr *uh;
42
43 m = mbuf_construct(IPPROTO_UDP);
44 uh = mbuf_return_hdrs(m, false, &ip);
45 ip->ip_src.s_addr = inet_addr("10.0.0.1") + i;
46 ip->ip_dst.s_addr = inet_addr("172.16.0.1");
47 uh->uh_sport = htons(9000);
48 uh->uh_dport = htons(9000);
49 return m;
50}

52static bool
53enqueue_connection(unsigned i, bool expire)
54{
55 ifnet_t *dummy_ifp = npf_test_addif(IFNAME_TEST, false, false);
56 npf_cache_t npc = { .npc_info = 0, .npc_ctx = npf_getkernctx() };
57 struct mbuf *m = fill_packet(i);
58 npf_conn_t *con;
59 nbuf_t nbuf;
60
61 nbuf_init(npf_getkernctx(), &nbuf, m, dummy_ifp);
62 npc.npc_nbuf = &nbuf;
63 npf_cache_all(&npc);
64
65 con = npf_conn_establish(&npc, PFIL_IN, true);
66 CHECK_TRUE(con != NULL);
67 if (expire) {
68 npf_conn_expire(con);
69 }
70 npf_conn_release(con);
71 return true;
72}

74static bool
75run_conn_gc(unsigned active, unsigned expired, unsigned expected)
76{
77 npf_t *npf = npf_getkernctx();
78 npf_conndb_t *cd = npf_conndb_create();
79 unsigned total, n = 0;
80
81 npf->conn_db = cd;
82
83 /*
84 * Insert the given number of active and expired connections..
85 */
86 total = active + expired;
87
88 while (active || expired) {
89 if (active) {
90 enqueue_connection(n++, false);
91 active--;
92 }
93 if (expired) {
94 enqueue_connection(n++, true);
95 expired--;
96 }
97 }
98
99 /* Verify the count. */
100 n = count_conns(cd);
101 CHECK_TRUE(n == total);
102
103 /*
104 * Run G/C. Check the remaining.
105 */
106 npf_conndb_gc(npf, cd, false, false);
107 n = count_conns(cd);
108 if (lverbose) {
109 printf("in conndb -- %u (expected %u)\n", n, expected);
110 }
111 CHECK_TRUE(n == expected);
112
113 /* Flush and destroy. */
114 npf_conndb_gc(npf, cd, true, false);
115 npf_conndb_destroy(cd);
116 npf->conn_db = NULL;
117 return true;
118}

120static bool
121run_gc_tests(void)
122{
123 bool ok;
124
125 /* Empty => GC => 0 in conndb. */
126 ok = run_conn_gc(0, 0, 0);
127 CHECK_TRUE(ok);
128
129 /* 1 active => GC => 1 in conndb. */
130 ok = run_conn_gc(1, 0, 1);
131 CHECK_TRUE(ok);
132
133 /* 1 expired => GC => 0 in conndb. */
134 ok = run_conn_gc(0, 1, 0);
135 CHECK_TRUE(ok);
136
137 /* 1 active and 1 expired => GC => 1 in conndb. */
138 ok = run_conn_gc(1, 1, 1);
139 CHECK_TRUE(ok);
140
141 /* 2 expired => GC => 0 in conndb. */
142 ok = run_conn_gc(0, 2, 0);
143 CHECK_TRUE(ok);
144
145 /* 128 expired => GC => 0 in conndb. */
146 ok = run_conn_gc(0, 128, 0);
147 CHECK_TRUE(ok);
148
149 /* 512 expired => GC => 256 in conndb. */
150 ok = run_conn_gc(0, 512, 256);
151 CHECK_TRUE(ok);
152
153 return true;
154}

156bool
157npf_conn_test(bool verbose)
158{
159 npf_t *npf = npf_getkernctx();
160 npf_conndb_t *orig_cd = npf->conn_db;
161 bool ok;
162
163 lverbose = verbose;
164
165 npf_config_enter(npf);
166 npf_conn_tracking(npf, true);
167 npf_config_exit(npf);
168
169 ok = run_gc_tests();
170
171 /* We *MUST* restore the valid conndb. */
172 npf->conn_db = orig_cd;
173 return ok;
174}
