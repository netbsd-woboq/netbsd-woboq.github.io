29#ifndef _PROG_OPS_H_
30#define _PROG_OPS_H_
31
32#include <sys/types.h>
33#include <sys/event.h>
34#include <sys/time.h>
35
36struct prog_ops {
37 int (*op_init)(void);
38
39 int (*op_open)(const char *, int, ...);
40 int (*op_close)(int);
41 ssize_t (*op_read)(int, void *, size_t);
42 int (*op_fcntl)(int, int, ...);
43 int (*op_ioctl)(int, unsigned long, ...);
44 int (*op_kqueue)(void);
45 int (*op_kevent)(int, const struct kevent *, size_t, struct kevent *,
46 size_t, const struct timespec *);
47};
48extern const struct prog_ops prog_ops;
49
50#define prog_init prog_ops.op_init
51#define prog_open prog_ops.op_open
52#define prog_close prog_ops.op_close
53#define prog_read prog_ops.op_read
54#define prog_kevent prog_ops.op_kevent
55#define prog_ioctl prog_ops.op_ioctl
56#define prog_fcntl prog_ops.op_fcntl
57#define prog_kqueue prog_ops.op_kqueue
58
59#endif /* _PROG_OPS_H_ */
