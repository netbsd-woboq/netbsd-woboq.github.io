75#define CUROFF(pb) (puffs_framebuf_telloff(pb))
76int
77p9pbuf_read(struct puffs_usermount *pu, struct puffs_framebuf *pb,
78 int fd, int *done)
79{
80 void *win;
81 ssize_t n;
82 size_t howmuch, winlen;
83 int lenstate;
84
85 the_next_level:
86 if ((lenstate = (CUROFF(pb) < 4)))
87 howmuch = 4 - CUROFF(pb);
88 else
89 howmuch = p9pbuf_get_len(pb) - CUROFF(pb);
90
91 if (puffs_framebuf_reserve_space(pb, howmuch) == -1)
92 return errno;
93
94 while (howmuch) {
95 winlen = howmuch;
96 if (puffs_framebuf_getwindow(pb, CUROFF(pb), &win, &winlen)==-1)
97 return errno;
98 n = read(fd, win, winlen);
99 switch (n) {
100 case 0:
101 return ECONNRESET;
102 case -1:
103 if (errno == EAGAIN)
104 return 0;
105 return errno;
106 default:
107 howmuch -= n;
108 puffs_framebuf_seekset(pb, CUROFF(pb) + n);
109 break;
110 }
111 }
112
113 if (!lenstate) {
114 puffs_framebuf_seekset(pb, 7);
115 *done = 1;
116 return 0;
117 } else
118 goto the_next_level;
119}

121int
122p9pbuf_write(struct puffs_usermount *pu, struct puffs_framebuf *pb,
123 int fd, int *done)
124{
125 void *win;
126 ssize_t n;
127 size_t winlen, howmuch;
128
129 if (CUROFF(pb) == 0) {
130 uint32_t len;
131
132 len = htole32(puffs_framebuf_tellsize(pb));
133 puffs_framebuf_putdata_atoff(pb, 0, &len, 4);
134 }
135
136 howmuch = puffs_framebuf_tellsize(pb) - CUROFF(pb);
137 while (howmuch) {
138 winlen = howmuch;
139 if (puffs_framebuf_getwindow(pb, CUROFF(pb), &win, &winlen)==-1)
140 return errno;
141 n = send(fd, win, winlen, MSG_NOSIGNAL);
142 switch (n) {
143 case 0:
144 return ECONNRESET;
145 case -1:
146 if (errno == EAGAIN)
147 return 0;
148 return errno;
149 default:
150 howmuch -= n;
151 puffs_framebuf_seekset(pb, CUROFF(pb) + n);
152 break;
153 }
154 }
155
156 *done = 1;
157 return 0;
158}

162p9pbuf_cmp(struct puffs_usermount *pu,
163 struct puffs_framebuf *c1, struct puffs_framebuf *c2, int *notresp)
164{
165
166 return p9pbuf_get_tag(c1) != p9pbuf_get_tag(c2);

294int
295p9pbuf_get_data(struct puffs_framebuf *pb, uint8_t **dp, uint16_t *dlenp)
296{
297 uint8_t *data;
298 uint16_t len;
299 int rv;
300
301 rv = p9pbuf_get_2(pb, &len);
302 if (rv)
303 return errno;
304
305 if (puffs_framebuf_remaining(pb) < len)
306 return EPROTO;
307
308 if (dp) {
309 data = emalloc(len+1);
310 rv = puffs_framebuf_getdata(pb, data, len);
311 if (rv) {
312 free(data);
313 return errno;
314 }
315 data[len] = '\0';
316 *dp = data;
317 } else
318 puffs_framebuf_seekset(pb, puffs_framebuf_telloff(pb)+len);
319
320 if (dlenp)
321 *dlenp = len;
322
323 return 0;
324}
