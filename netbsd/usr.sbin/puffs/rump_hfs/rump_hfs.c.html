28#include <sys/types.h>
29#include <sys/mount.h>
30
31#include <fs/hfs/hfs.h>
32
33#include <err.h>
34#include <puffs.h>
35#include <stdlib.h>
36#include <unistd.h>
37
38#include <rump/p2k.h>
39#include <rump/ukfs.h>
40
41#include "mount_hfs.h"
42
43int
44main(int argc, char *argv[])
45{
46 struct hfs_args args;
47 char canon_dev[UKFS_DEVICE_MAXPATHLEN], canon_dir[MAXPATHLEN];
48 struct ukfs_part *part;
49 int mntflags;
50 int rv;
51
52 setprogname(argv[0]);
53
54 UKFS_DEVICE_ARGVPROBE(&part);
55 mount_hfs_parseargs(argc, argv, &args, &mntflags, canon_dev, canon_dir);
56 rv = p2k_run_diskfs(MOUNT_HFS, canon_dev, part, canon_dir, mntflags,
57 &args, sizeof(args), 0);
58 ukfs_part_release(part);
59 if (rv)
60 err(1, "mount");
61
62 return 0;
63}
