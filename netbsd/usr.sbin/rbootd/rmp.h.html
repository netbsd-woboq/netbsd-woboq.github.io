45/*
46 * Define MIN/MAX sizes of RMP (ethernet) packet.
47 * For ease of computation, the 4 octet CRC field is not included.
48 *
49 * MCLBYTES is for bpfwrite(); it is adamant about using a cluster.
50 */
51
52#define RMP_MAX_PACKET MIN(1514,MCLBYTES)
53#define RMP_MIN_PACKET 60
54
55/*
56 * Define RMP/Ethernet Multicast address (9:0:9:0:0:4) and its length.
57 */
58#define RMP_ADDR { 0x9, 0x0, 0x9, 0x0, 0x0, 0x4 }
59#define RMP_ADDRLEN 6
60
61/*
62 * Define IEEE802.2 (Logical Link Control) information.
63 */
64#define IEEE_DSAP_HP 0xF8 /* Destination Service Access Point */
65#define IEEE_SSAP_HP 0xF8 /* Source Service Access Point */
66#define IEEE_CNTL_HP 0x0300 /* Type 1 / I format control information */
67
68#define HPEXT_DXSAP 0x608 /* HP Destination Service Access Point */
69#define HPEXT_SXSAP 0x609 /* HP Source Service Access Point */
70
71/*
72 * 802.3-style "Ethernet" header.
73 */
74
75struct hp_hdr {
76 u_int8_t daddr[RMP_ADDRLEN];
77 u_int8_t saddr[RMP_ADDRLEN];
78 u_int16_t len;
79};
80
81/*
82 * HP uses 802.2 LLC with their own local extensions. This struct makes
83 * sense out of this data (encapsulated in the above 802.3 packet).
84 */
85
86struct hp_llc {
87 u_int8_t dsap; /* 802.2 DSAP */
88 u_int8_t ssap; /* 802.2 SSAP */
89 u_int16_t cntrl; /* 802.2 control field */
90 u_int16_t filler; /* HP filler (must be zero) */
91 u_int16_t dxsap; /* HP extended DSAP */
92 u_int16_t sxsap; /* HP extended SSAP */
93};
