58#define DEFAULT_WAIT 5
59
60static int s;
61static int query_wait = DEFAULT_WAIT;
62static struct sockaddr_in6 sin6;
63static struct rip6 *ripbuf;
64
65#define RIPSIZE(n) (sizeof(struct rip6) + (n-1) * sizeof(struct netinfo6))
66
67__dead static void usage(void);
68__dead static void sigalrm_handler(int);
69static const char *sa_n2a(struct sockaddr *);
70static const char *inet6_n2a(struct in6_addr *);
71
72int
73main(int argc, char **argv)
74{
75 struct netinfo6 *np;
76 struct sockaddr_in6 fsock;
77 int i, n, len;
78 socklen_t flen;
79 int c;
80 int ifidx = -1;
81 int error;
82 char pbuf[NI_MAXSERV];
83 struct addrinfo hints, *res;
84
85 while ((c = getopt(argc, argv, "I:w:")) != -1) {
86 switch (c) {
87 case 'I':
88 ifidx = if_nametoindex(optarg);
89 if (ifidx == 0) {
90 errx(1, "invalid interface %s", optarg);
91 /*NOTREACHED*/
92 }
93 break;
94 case 'w':
95 query_wait = atoi(optarg);
96 break;
97 default:
98 usage();
99 /*NOTREACHED*/
100 }
101 }
102 argv += optind;
103 argc -= optind;
104
105 if (argc != 1)
106 usage();
107
108 if ((s = socket(AF_INET6, SOCK_DGRAM, 0)) < 0) {
109 err(1, "socket");
110 /*NOTREACHED*/
111 }
112
113 /* getaddrinfo is preferred for addr%scope syntax */
114 snprintf(pbuf, sizeof(pbuf), "%d", RIP6_PORT);
115 memset(&hints, 0, sizeof(hints));
116 hints.ai_family = AF_INET6;
117 hints.ai_socktype = SOCK_DGRAM;
118 error = getaddrinfo(argv[0], pbuf, &hints, &res);
119 if (error) {
120 errx(1, "%s: %s", argv[0], gai_strerror(error));
121 /*NOTREACHED*/
122 }
123 if (res->ai_next) {
124 errx(1, "%s: %s", argv[0], "resolved to multiple addrs");
125 /*NOTREACHED*/
126 }
127 if (sizeof(sin6) != res->ai_addrlen) {
128 errx(1, "%s: %s", argv[0], "invalid addrlen");
129 /*NOTREACHED*/
130 }
131 memcpy(&sin6, res->ai_addr, res->ai_addrlen);
132 if (ifidx >= 0)
133 sin6.sin6_scope_id = ifidx;
134
135 if ((ripbuf = (struct rip6 *)malloc(BUFSIZ)) == NULL) {
136 err(1, "malloc");
137 /*NOTREACHED*/
138 }
139 ripbuf->rip6_cmd = RIP6_REQUEST;
140 ripbuf->rip6_vers = RIP6_VERSION;
141 ripbuf->rip6_res1[0] = 0;
142 ripbuf->rip6_res1[1] = 0;
143 np = ripbuf->rip6_nets;
144 bzero(&np->rip6_dest, sizeof(struct in6_addr));
145 np->rip6_tag = 0;
146 np->rip6_plen = 0;
147 np->rip6_metric = HOPCNT_INFINITY6;
148 if (sendto(s, ripbuf, RIPSIZE(1), 0, (struct sockaddr *)&sin6,
149 sizeof(struct sockaddr_in6)) < 0) {
150 err(1, "send");
151 /*NOTREACHED*/
152 }
153 signal(SIGALRM, sigalrm_handler);
154 for (;;) {
155 flen = sizeof(fsock);
156 alarm(query_wait);
157 if ((len = recvfrom(s, ripbuf, BUFSIZ, 0,
158 (struct sockaddr *)&fsock, &flen)) < 0) {
159 err(1, "recvfrom");
160 /*NOTREACHED*/
161 }
162 alarm(0);
163 printf("Response from %s len %d\n",
164 sa_n2a((struct sockaddr *)&fsock), len);
165 n = (len - sizeof(struct rip6) + sizeof(struct netinfo6)) /
166 sizeof(struct netinfo6);
167 np = ripbuf->rip6_nets;
168 for (i = 0; i < n; i++, np++) {
169 printf("\t%s/%d [%d]", inet6_n2a(&np->rip6_dest),
170 np->rip6_plen, np->rip6_metric);
171 if (np->rip6_tag)
172 printf(" tag=0x%x", ntohs(np->rip6_tag));
173 printf("\n");
174 }
175 }
176
177 exit(0);
178}

180static void
181usage(void)
182{
183 fprintf(stderr, "usage: rip6query [-I iface] [-w wait] address\n");
184 exit(1);
185}

187/* getnameinfo() is preferred as we may be able to show ifindex as ifname */
188static const char *
189sa_n2a(struct sockaddr *sa)
190{
191 static char buf[NI_MAXHOST];
192
193 if (getnameinfo(sa, sa->sa_len, buf, sizeof(buf),
194 NULL, 0, NI_NUMERICHOST) != 0) {
195 snprintf(buf, sizeof(buf), "%s", "(invalid)");
196 }
197 return buf;
198}
199
200static const char *
201inet6_n2a(struct in6_addr *addr)
202{
203 static char buf[NI_MAXHOST];
204
205 return inet_ntop(AF_INET6, addr, buf, sizeof(buf));
206}

208static void
209sigalrm_handler(int sig)
210{
211
212 _exit(0);
213}
