83static size_t
84encode_domain(char *dst, const char *src)
85{
86 ssize_t len;
87 char *odst, *p;
88
89 odst = dst;
90 while (src && (len = strlen(src)) != 0) {
91 p = strchr(src, '.');
92 *dst++ = len = MIN(63, p == NULL ? len : p - src);
93 memcpy(dst, src, len);
94 dst += len;
95 if (p == NULL)
96 break;
97 src = p + 1;
98 }
99 *dst++ = '\0';
100
101 return dst - odst;
102}

104void
105free_rainfo(struct rainfo *rai)
106{
107 struct soliciter *sol;
108 struct prefix *pfx;
109 struct rtinfo *rti;
110 struct rdnss *rdnss;
111 struct rdnss_addr *rdnsa;
112 struct dnssl *dnssl;
113 struct dnssl_domain *dnsd;
114
115 rtadvd_remove_timer(&rai->timer);
116 rtadvd_remove_timer(&rai->timer_sol);
117
118 while ((sol = TAILQ_FIRST(&rai->soliciter))) {
119 TAILQ_REMOVE(&rai->soliciter, sol, next);
120 free(sol);
121 }
122
123 while ((pfx = TAILQ_FIRST(&rai->prefix))) {
124 TAILQ_REMOVE(&rai->prefix, pfx, next);
125 free(pfx);
126 }
127
128 while ((rti = TAILQ_FIRST(&rai->route))) {
129 TAILQ_REMOVE(&rai->route, rti, next);
130 free(rti);
131 }
132
133 while ((rdnss = TAILQ_FIRST(&rai->rdnss))) {
134 TAILQ_REMOVE(&rai->rdnss, rdnss, next);
135 while ((rdnsa = TAILQ_FIRST(&rdnss->list))) {
136 TAILQ_REMOVE(&rdnss->list, rdnsa, next);
137 free(rdnsa);
138 }
139 free(rdnss);
140 }
141
142 while ((dnssl = TAILQ_FIRST(&rai->dnssl))) {
143 TAILQ_REMOVE(&rai->dnssl, dnssl, next);
144 while ((dnsd = TAILQ_FIRST(&dnssl->list))) {
145 TAILQ_REMOVE(&dnssl->list, dnsd, next);
146 free(dnsd);
147 }
148 free(dnssl);
149 }
150
151 free(rai->sdl);
152 free(rai->ra_data);
153 free(rai);
154}

156void
157getconfig(const char *intface, int exithard)
158{
159 int stat, c, i;
160 char tbuf[BUFSIZ];
161 struct rainfo *tmp, *rai;
162 int32_t val;
163 int64_t val64;
164 char buf[BUFSIZ];
165 char *bp = buf;
166 char *addr, *flagstr, *ap;
167 static int forwarding = -1;
168 char entbuf[256], abuf[256];
169 struct rdnss *rdnss;
170 struct dnssl *dnssl;
171
172#define MUSTHAVE(var, cap) \
173 do { \
174 int64_t t; \
175 if ((t = agetnum(cap)) < 0) { \
176 fprintf(stderr, "rtadvd: need %s for interface %s\n", \
177 cap, intface); \
178 goto errexit; \
179 } \
180 var = t; \
181 } while (0)
182#define MAYHAVE(var, cap, def) \
183 do { \
184 if ((var = agetnum(cap)) < 0) \
185 var = def; \
186 } while (0)
187#define ELM_MALLOC(p) \
188 do { \
189 p = calloc(1, sizeof(*p)); \
190 if (p == NULL) { \
191 logit(LOG_ERR, "<%s> calloc failed: %m", \
192 __func__); \
193 goto errexit; \
194 } \
195 } while(/*CONSTCOND*/0)
196
197 if (if_nametoindex(intface) == 0) {
198 logit(LOG_INFO, "<%s> interface %s not found, ignoring",
199 __func__, intface);
200 return;
201 }
202
203 logit(LOG_DEBUG, "<%s> loading configuration for interface %s",
204 __func__, intface);
205
206 if ((stat = agetent(tbuf, intface)) <= 0) {
207 memset(tbuf, 0, sizeof(tbuf));
208 logit(LOG_INFO,
209 "<%s> %s isn't defined in the configuration file"
210 " or the configuration file doesn't exist."
211 " Treat it as default",
212 __func__, intface);
213 }
214
215 ELM_MALLOC(tmp);
216 TAILQ_INIT(&tmp->soliciter);
217 TAILQ_INIT(&tmp->prefix);
218 TAILQ_INIT(&tmp->route);
219 TAILQ_INIT(&tmp->rdnss);
220 TAILQ_INIT(&tmp->dnssl);
221
222 /* check if we are allowed to forward packets (if not determined) */
223 if (forwarding < 0) {
224 if ((forwarding = getinet6sysctl(IPV6CTL_FORWARDING)) < 0)
225 exit(1);
226 }
227
228 /* get interface information */
229 if (agetflag("nolladdr"))
230 tmp->advlinkopt = 0;
231 else
232 tmp->advlinkopt = 1;
233 if (tmp->advlinkopt) {
234 if ((tmp->sdl = if_nametosdl(intface)) == NULL) {
235 logit(LOG_ERR,
236 "<%s> can't get information of %s",
237 __func__, intface);
238 goto errexit;
239 }
240 tmp->ifindex = tmp->sdl->sdl_index;
241 } else {
242 tmp->ifindex = if_nametoindex(intface);
243 if (tmp->ifindex == 0) {
244 logit(LOG_ERR,
245 "<%s> can't get information of %s",
246 __func__, intface);
247 goto errexit;
248 }
249 }
250 tmp->ifflags = if_getflags(tmp->ifindex, 0);
251 strlcpy(tmp->ifname, intface, sizeof(tmp->ifname));
252 if ((tmp->phymtu = if_getmtu(intface)) == 0) {
253 tmp->phymtu = IPV6_MMTU;
254 logit(LOG_WARNING,
255 "<%s> can't get interface mtu of %s. Treat as %d",
256 __func__, intface, IPV6_MMTU);
257 }
258
259 /*
260 * set router configuration variables.
261 */
262 MAYHAVE(val, "maxinterval", DEF_MAXRTRADVINTERVAL);
263 if (val < MIN_MAXINTERVAL || val > MAX_MAXINTERVAL) {
264 logit(LOG_ERR,
265 "<%s> maxinterval (%d) on %s is invalid "
266 "(must be between %u and %u)", __func__, val,
267 intface, MIN_MAXINTERVAL, MAX_MAXINTERVAL);
268 goto errexit;
269 }
270 tmp->maxinterval = val;
271 MAYHAVE(val, "mininterval", tmp->maxinterval/3);
272 if (val < MIN_MININTERVAL || val > (tmp->maxinterval * 3) / 4) {
273 logit(LOG_ERR,
274 "<%s> mininterval (%d) on %s is invalid "
275 "(must be between %u and %d)",
276 __func__, val, intface, MIN_MININTERVAL,
277 (tmp->maxinterval * 3) / 4);
278 goto errexit;
279 }
280 tmp->mininterval = val;
281
282 MAYHAVE(val, "chlim", DEF_ADVCURHOPLIMIT);
283 tmp->hoplimit = val & 0xff;
284
285 if ((flagstr = (char *)agetstr("raflags", &bp))) {
286 val = 0;
287 if (strchr(flagstr, 'm'))
288 val |= ND_RA_FLAG_MANAGED;
289 if (strchr(flagstr, 'o'))
290 val |= ND_RA_FLAG_OTHER;
291 if (strchr(flagstr, 'h'))
292 val |= ND_RA_FLAG_RTPREF_HIGH;
293 if (strchr(flagstr, 'l')) {
294 if ((val & ND_RA_FLAG_RTPREF_HIGH)) {
295 logit(LOG_ERR, "<%s> the \'h\' and \'l\'"
296 " router flags are exclusive", __func__);
297 goto errexit;
298 }
299 val |= ND_RA_FLAG_RTPREF_LOW;
300 }
301 } else {
302 MAYHAVE(val, "raflags", 0);
303 }
304 tmp->managedflg = val & ND_RA_FLAG_MANAGED;
305 tmp->otherflg = val & ND_RA_FLAG_OTHER;
306#ifndef ND_RA_FLAG_RTPREF_MASK
307#define ND_RA_FLAG_RTPREF_MASK 0x18 /* 00011000 */
308#define ND_RA_FLAG_RTPREF_RSV 0x10 /* 00010000 */
309#endif
310 tmp->rtpref = val & ND_RA_FLAG_RTPREF_MASK;
311 if (tmp->rtpref == ND_RA_FLAG_RTPREF_RSV) {
312 logit(LOG_ERR, "<%s> invalid router preference (%02x) on %s",
313 __func__, tmp->rtpref, intface);
314 goto errexit;
315 }
316
317 MAYHAVE(val, "rltime", DEF_ADVROUTERLIFETIME);
318 if (val && (val < tmp->maxinterval || val > MAXROUTERLIFETIME)) {
319 logit(LOG_ERR,
320 "<%s> router lifetime (%d) on %s is invalid "
321 "(must be 0 or between %d and %d)",
322 __func__, val, intface,
323 tmp->maxinterval, MAXROUTERLIFETIME);
324 goto errexit;
325 }
326 /*
327 * Basically, hosts MUST NOT send Router Advertisement messages at any
328 * time (RFC 2461, Section 6.2.3). However, it would sometimes be
329 * useful to allow hosts to advertise some parameters such as prefix
330 * information and link MTU. Thus, we allow hosts to invoke rtadvd
331 * only when router lifetime (on every advertising interface) is
332 * explicitly set zero. (see also the above section)
333 */
334 if (val && forwarding == 0) {
335 logit(LOG_ERR,
336 "<%s> non zero router lifetime is specified for %s, "
337 "which must not be allowed for hosts. you must "
338 "change router lifetime or enable IPv6 forwarding.",
339 __func__, intface);
340 goto errexit;
341 }
342 tmp->lifetime = val & 0xffff;
343
344 MAYHAVE(val, "rtime", DEF_ADVREACHABLETIME);
345 if (val < 0 || val > MAXREACHABLETIME) {
346 logit(LOG_ERR,
347 "<%s> reachable time (%d) on %s is invalid "
348 "(must be no greater than %d)",
349 __func__, val, intface, MAXREACHABLETIME);
350 goto errexit;
351 }
352 tmp->reachabletime = (uint32_t)val;
353
354 MAYHAVE(val64, "retrans", DEF_ADVRETRANSTIMER);
355 if (val64 < 0 || val64 > 0xffffffff) {
356 logit(LOG_ERR, "<%s> retrans time (%lld) on %s out of range",
357 __func__, (long long)val64, intface);
358 goto errexit;
359 }
360 tmp->retranstimer = (uint32_t)val64;
361
362 if (agetnum("hapref") != -1 || agetnum("hatime") != -1) {
363 logit(LOG_ERR,
364 "<%s> mobile-ip6 configuration not supported",
365 __func__);
366 goto errexit;
367 }
368 /* prefix information */
369
370 /*
371 * This is an implementation specific parameter to consider
372 * link propagation delays and poorly synchronized clocks when
373 * checking consistency of advertised lifetimes.
374 */
375 MAYHAVE(val, "clockskew", 0);
376 tmp->clockskew = val;
377
378 tmp->pfxs = 0;
379 for (i = -1; i < MAXPREFIX; i++) {
380 struct prefix *pfx;
381
382 makeentry(entbuf, sizeof(entbuf), i, "addr");
383 addr = (char *)agetstr(entbuf, &bp);
384 if (addr == NULL)
385 continue;
386
387 /* allocate memory to store prefix information */
388 if ((pfx = calloc(1, sizeof(*pfx))) == NULL) {
389 logit(LOG_ERR,
390 "<%s> can't allocate memory: %m",
391 __func__);
392 goto errexit;
393 }
394
395 TAILQ_INSERT_TAIL(&tmp->prefix, pfx, next);
396 tmp->pfxs++;
397 pfx->rainfo = tmp;
398
399 pfx->origin = PREFIX_FROM_CONFIG;
400
401 if (inet_pton(AF_INET6, addr, &pfx->prefix) != 1) {
402 logit(LOG_ERR,
403 "<%s> inet_pton failed for %s",
404 __func__, addr);
405 goto errexit;
406 }
407 if (IN6_IS_ADDR_MULTICAST(&pfx->prefix)) {
408 logit(LOG_ERR,
409 "<%s> multicast prefix (%s) must "
410 "not be advertised on %s",
411 __func__, addr, intface);
412 goto errexit;
413 }
414 if (IN6_IS_ADDR_LINKLOCAL(&pfx->prefix))
415 logit(LOG_NOTICE,
416 "<%s> link-local prefix (%s) will be"
417 " advertised on %s",
418 __func__, addr, intface);
419
420 makeentry(entbuf, sizeof(entbuf), i, "prefixlen");
421 MAYHAVE(val, entbuf, 64);
422 if (val < 0 || val > 128) {
423 logit(LOG_ERR, "<%s> prefixlen (%d) for %s "
424 "on %s out of range",
425 __func__, val, addr, intface);
426 goto errexit;
427 }
428 pfx->prefixlen = (int)val;
429
430 makeentry(entbuf, sizeof(entbuf), i, "pinfoflags");
431 if ((flagstr = (char *)agetstr(entbuf, &bp))) {
432 val = 0;
433 if (strchr(flagstr, 'l'))
434 val |= ND_OPT_PI_FLAG_ONLINK;
435 if (strchr(flagstr, 'a'))
436 val |= ND_OPT_PI_FLAG_AUTO;
437 } else {
438 MAYHAVE(val, entbuf,
439 (ND_OPT_PI_FLAG_ONLINK|ND_OPT_PI_FLAG_AUTO));
440 }
441 pfx->onlinkflg = val & ND_OPT_PI_FLAG_ONLINK;
442 pfx->autoconfflg = val & ND_OPT_PI_FLAG_AUTO;
443
444 makeentry(entbuf, sizeof(entbuf), i, "vltime");
445 MAYHAVE(val64, entbuf, DEF_ADVVALIDLIFETIME);
446 if (val64 < 0 || val64 > 0xffffffff) {
447 logit(LOG_ERR, "<%s> vltime (%lld) for "
448 "%s/%d on %s is out of range",
449 __func__, (long long)val64,
450 addr, pfx->prefixlen, intface);
451 goto errexit;
452 }
453 pfx->validlifetime = (uint32_t)val64;
454
455 makeentry(entbuf, sizeof(entbuf), i, "vltimedecr");
456 if (agetflag(entbuf)) {
457 struct timespec now;
458 prog_clock_gettime(CLOCK_MONOTONIC, &now);
459 pfx->vltimeexpire =
460 now.tv_sec + pfx->validlifetime;
461 }
462
463 makeentry(entbuf, sizeof(entbuf), i, "pltime");
464 MAYHAVE(val64, entbuf, DEF_ADVPREFERREDLIFETIME);
465 if (val64 < 0 || val64 > 0xffffffff) {
466 logit(LOG_ERR,
467 "<%s> pltime (%lld) for %s/%d on %s "
468 "is out of range",
469 __func__, (long long)val64,
470 addr, pfx->prefixlen, intface);
471 goto errexit;
472 }
473 pfx->preflifetime = (uint32_t)val64;
474
475 makeentry(entbuf, sizeof(entbuf), i, "pltimedecr");
476 if (agetflag(entbuf)) {
477 struct timespec now;
478 prog_clock_gettime(CLOCK_MONOTONIC, &now);
479 pfx->pltimeexpire =
480 now.tv_sec + pfx->preflifetime;
481 }
482 }
483 if (TAILQ_FIRST(&tmp->prefix) == NULL && !agetflag("noifprefix"))
484 get_prefix(tmp);
485
486 MAYHAVE(val64, "mtu", 0);
487 if (val64 < 0 || val64 > 0xffffffff) {
488 logit(LOG_ERR,
489 "<%s> mtu (%" PRIi64 ") on %s out of range",
490 __func__, val64, intface);
491 goto errexit;
492 }
493 tmp->linkmtu = (uint32_t)val64;
494 if (tmp->linkmtu == 0) {
495 char *mtustr;
496
497 if ((mtustr = (char *)agetstr("mtu", &bp)) &&
498 strcmp(mtustr, "auto") == 0)
499 tmp->linkmtu = tmp->phymtu;
500 }
501 else if (tmp->linkmtu < IPV6_MMTU || tmp->linkmtu > tmp->phymtu) {
502 logit(LOG_ERR,
503 "<%s> advertised link mtu (%d) on %s is invalid (must "
504 "be between least MTU (%d) and physical link MTU (%d)",
505 __func__, tmp->linkmtu, intface,
506 IPV6_MMTU, tmp->phymtu);
507 goto errexit;
508 }
509
510#ifdef SIOCSIFINFO_IN6
511 {
512 struct in6_ndireq ndi;
513 int s;
514
515 if ((s = prog_socket(AF_INET6, SOCK_DGRAM, 0)) < 0) {
516 logit(LOG_ERR, "<%s> socket: %m", __func__);
517 goto errexit;
518 }
519 memset(&ndi, 0, sizeof(ndi));
520 strncpy(ndi.ifname, intface, IFNAMSIZ);
521 if (prog_ioctl(s, SIOCGIFINFO_IN6, &ndi) < 0) {
522 logit(LOG_INFO, "<%s> ioctl:SIOCGIFINFO_IN6 at %s: %m",
523 __func__, intface);
524 }
525
526 /* reflect the RA info to the host variables in kernel */
527 ndi.ndi.chlim = tmp->hoplimit;
528 ndi.ndi.retrans = tmp->retranstimer;
529 ndi.ndi.basereachable = tmp->reachabletime;
530 if (prog_ioctl(s, SIOCSIFINFO_IN6, &ndi) < 0) {
531 logit(LOG_INFO, "<%s> ioctl:SIOCSIFINFO_IN6 at %s: %m",
532 __func__, intface);
533 }
534 prog_close(s);
535 }
536#endif
537
538 /* route information */
539 for (i = -1; i < MAXROUTE; i++) {
540 struct rtinfo *rti;
541 char oentbuf[256];
542
543 makeentry(entbuf, sizeof(entbuf), i, "rtprefix");
544 addr = (char *)agetstr(entbuf, &bp);
545 if (addr == NULL) {
546 makeentry(oentbuf, sizeof(oentbuf), i, "rtrprefix");
547 addr = (char *)agetstr(oentbuf, &bp);
548 if (addr) {
549 fprintf(stderr, "%s was obsoleted. Use %s.\n",
550 oentbuf, entbuf);
551 }
552 }
553 if (addr == NULL)
554 continue;
555
556 ELM_MALLOC(rti);
557 memset(rti, 0, sizeof(*rti));
558
559 /* link into chain */
560 TAILQ_INSERT_TAIL(&tmp->route, rti, next);
561
562 if (inet_pton(AF_INET6, addr, &rti->prefix) != 1) {
563 logit(LOG_ERR, "<%s> inet_pton failed for %s",
564 __func__, addr);
565 goto errexit;
566 }
567#if 0
568 /*
569 * XXX: currently there's no restriction in route information
570 * prefix according to
571 * draft-ietf-ipngwg-router-selection-00.txt.
572 * However, I think the similar restriction be necessary.
573 */
574 MAYHAVE(val64, entbuf, DEF_ADVVALIDLIFETIME);
575 if (IN6_IS_ADDR_MULTICAST(&rti->prefix)) {
576 logit(LOG_ERR,
577 "<%s> multicast route (%s) must "
578 "not be advertised on %s",
579 __func__, addr, intface);
580 goto errexit;
581 }
582 if (IN6_IS_ADDR_LINKLOCAL(&rti->prefix)) {
583 logit(LOG_NOTICE,
584 "<%s> link-local route (%s) will "
585 "be advertised on %s",
586 __func__, addr, intface);
587 goto errexit;
588 }
589#endif
590
591 makeentry(entbuf, sizeof(entbuf), i, "rtplen");
592 /* XXX: 256 is a magic number for compatibility check. */
593 MAYHAVE(val, entbuf, 256);
594 if (val == 256) {
595 makeentry(oentbuf, sizeof(oentbuf), i, "rtrplen");
596 MAYHAVE(val, oentbuf, 256);
597 if (val != 256) {
598 fprintf(stderr, "%s was obsoleted. Use %s.\n",
599 oentbuf, entbuf);
600 } else
601 val = 64;
602 }
603 if (val < 0 || val > 128) {
604 logit(LOG_ERR, "<%s> prefixlen (%d) for %s on %s "
605 "out of range",
606 __func__, val, addr, intface);
607 goto errexit;
608 }
609 rti->prefixlen = (int)val;
610
611 makeentry(entbuf, sizeof(entbuf), i, "rtflags");
612 if ((flagstr = (char *)agetstr(entbuf, &bp))) {
613 val = 0;
614 if (strchr(flagstr, 'h'))
615 val |= ND_RA_FLAG_RTPREF_HIGH;
616 if (strchr(flagstr, 'l')) {
617 if ((val & ND_RA_FLAG_RTPREF_HIGH)) {
618 logit(LOG_ERR,
619 "<%s> the \'h\' and \'l\' route"
620 " preferences are exclusive",
621 __func__);
622 goto errexit;
623 }
624 val |= ND_RA_FLAG_RTPREF_LOW;
625 }
626 } else
627 MAYHAVE(val, entbuf, 256); /* XXX */
628 if (val == 256) {
629 makeentry(oentbuf, sizeof(oentbuf), i, "rtrflags");
630 MAYHAVE(val, oentbuf, 256);
631 if (val != 256) {
632 fprintf(stderr, "%s was obsoleted. Use %s.\n",
633 oentbuf, entbuf);
634 } else
635 val = 0;
636 }
637 rti->rtpref = val & ND_RA_FLAG_RTPREF_MASK;
638 if (rti->rtpref == ND_RA_FLAG_RTPREF_RSV) {
639 logit(LOG_ERR, "<%s> invalid route preference (%02x) "
640 "for %s/%d on %s",
641 __func__, rti->rtpref, addr,
642 rti->prefixlen, intface);
643 goto errexit;
644 }
645
646 /*
647 * Since the spec does not a default value, we should make
648 * this entry mandatory. However, FreeBSD 4.4 has shipped
649 * with this field being optional, we use the router lifetime
650 * as an ad-hoc default value with a warning message.
651 */
652 makeentry(entbuf, sizeof(entbuf), i, "rtltime");
653 MAYHAVE(val64, entbuf, -1);
654 if (val64 == -1) {
655 makeentry(oentbuf, sizeof(oentbuf), i, "rtrltime");
656 MAYHAVE(val64, oentbuf, -1);
657 if (val64 != -1) {
658 fprintf(stderr, "%s was obsoleted. Use %s.\n",
659 oentbuf, entbuf);
660 } else {
661 fprintf(stderr, "%s should be specified "
662 "for interface %s.\n",
663 entbuf, intface);
664 val64 = tmp->lifetime;
665 }
666 }
667 if (val64 < 0 || val64 > 0xffffffff) {
668 logit(LOG_ERR, "<%s> route lifetime (%lld) for "
669 "%s/%d on %s out of range", __func__,
670 (long long)val64, addr, rti->prefixlen, intface);
671 goto errexit;
672 }
673 rti->ltime = (uint32_t)val64;
674 }
675
676 /* RDNSS */
677 for (i = -1; i < MAXRDNSS; i++) {
678 struct rdnss_addr *rdnsa;
679
680 makeentry(entbuf, sizeof(entbuf), i, "rdnss");
681 addr = (char *)agetstr(entbuf, &bp);
682 if (addr == NULL)
683 continue;
684
685 ELM_MALLOC(rdnss);
686 TAILQ_INSERT_TAIL(&tmp->rdnss, rdnss, next);
687 TAILQ_INIT(&rdnss->list);
688
689 for (ap = addr; ap - addr < (ssize_t)strlen(addr); ap += c+1) {
690 c = strcspn(ap, ",");
691 strncpy(abuf, ap, c);
692 abuf[c] = '\0';
693 ELM_MALLOC(rdnsa);
694 TAILQ_INSERT_TAIL(&rdnss->list, rdnsa, next);
695 if (inet_pton(AF_INET6, abuf, &rdnsa->addr) != 1) {
696 logit(LOG_ERR, "<%s> inet_pton failed for %s",
697 __func__, addr);
698 goto errexit;
699 }
700 }
701
702 makeentry(entbuf, sizeof(entbuf), i, "rdnssltime");
703 MAYHAVE(val64, entbuf, tmp->maxinterval * 3 / 2);
704 if (val64 < 0 || val64 > 0xffffffff) {
705 logit(LOG_ERR, "<%s> %s (%lld) on %s is invalid",
706 __func__, entbuf, (long long)val64, intface);
707 goto errexit;
708 }
709 rdnss->lifetime = (uint32_t)val64;
710
711 }
712
713 /* DNSSL */
714 TAILQ_INIT(&tmp->dnssl);
715 for (i = -1; i < MAXDNSSL; i++) {
716 struct dnssl_domain *dnsd;
717
718 makeentry(entbuf, sizeof(entbuf), i, "dnssl");
719 addr = (char *)agetstr(entbuf, &bp);
720 if (addr == NULL)
721 continue;
722
723 ELM_MALLOC(dnssl);
724 TAILQ_INSERT_TAIL(&tmp->dnssl, dnssl, next);
725 TAILQ_INIT(&dnssl->list);
726
727 for (ap = addr; ap - addr < (ssize_t)strlen(addr); ap += c+1) {
728 c = strcspn(ap, ",");
729 strncpy(abuf, ap, c);
730 abuf[c] = '\0';
731 ELM_MALLOC(dnsd);
732 TAILQ_INSERT_TAIL(&dnssl->list, dnsd, next);
733 dnsd->len = encode_domain(dnsd->domain, abuf);
734 }
735
736 makeentry(entbuf, sizeof(entbuf), i, "dnsslltime");
737 MAYHAVE(val64, entbuf, tmp->maxinterval * 3 / 2);
738 if (val64 < 0 || val64 > 0xffffffff) {
739 logit(LOG_ERR, "<%s> %s (%lld) on %s is invalid",
740 __func__, entbuf, (long long)val64, intface);
741 goto errexit;
742 }
743 dnssl->lifetime = (uint32_t)val64;
744
745 }
746
747 TAILQ_FOREACH(rai, &ralist, next) {
748 if (rai->ifindex == tmp->ifindex) {
749 TAILQ_REMOVE(&ralist, rai, next);
750 /* If we already have a leaving RA use that
751 * as this config hasn't been advertised */
752 if (rai->leaving) {
753 tmp->leaving = rai->leaving;
754 free_rainfo(rai);
755 rai = tmp->leaving;
756 rai->leaving_for = tmp;
757 break;
758 }
759 rai->lifetime = 0;
760 TAILQ_FOREACH(rdnss, &rai->rdnss, next)
761 rdnss->lifetime = 0;
762 TAILQ_FOREACH(dnssl, &rai->dnssl, next)
763 dnssl->lifetime = 0;
764 rai->leaving_for = tmp;
765 tmp->leaving = rai;
766 rai->initcounter = MAX_INITIAL_RTR_ADVERTISEMENTS;
767 rai->mininterval = MIN_DELAY_BETWEEN_RAS;
768 rai->maxinterval = MIN_DELAY_BETWEEN_RAS;
769 rai->leaving_adv = MAX_FINAL_RTR_ADVERTISEMENTS;
770 if (rai->timer == NULL)
771 rai->timer = rtadvd_add_timer(ra_timeout,
772 ra_timer_update,
773 rai, rai);
774 ra_timer_update((void *)rai, &rai->timer->tm);
775 rtadvd_set_timer(&rai->timer->tm, rai->timer);
776 break;
777 }
778 }
779
780 /* okey */
781 TAILQ_INSERT_TAIL(&ralist, tmp, next);
782
783 /* construct the sending packet */
784 make_packet(tmp);
785
786 /* set timer */
787 if (rai)
788 return;
789 tmp->timer = rtadvd_add_timer(ra_timeout, ra_timer_update,
790 tmp, tmp);
791 ra_timer_set_short_delay(tmp, tmp->timer);
792 tmp->timer_sol = rtadvd_add_timer(ra_timeout_sol, NULL, tmp, NULL);
793
794 return;
795
796errexit:
797 if (exithard)
798 exit(1);
799 free_rainfo(tmp);
800}

802void
803get_prefix(struct rainfo *rai)
804{
805 struct ifaddrs *ifap, *ifa;
806 struct prefix *pp;
807 struct in6_addr *a;
808 unsigned char *p, *ep, *m, *lim;
809 char ntopbuf[INET6_ADDRSTRLEN];
810
811 if (getifaddrs(&ifap) < 0) {
812 logit(LOG_ERR,
813 "<%s> can't get interface addresses",
814 __func__);
815 exit(1);
816 }
817
818 for (ifa = ifap; ifa; ifa = ifa->ifa_next) {
819 int plen;
820
821 if (strcmp(ifa->ifa_name, rai->ifname) != 0)
822 continue;
823 if (ifa->ifa_addr->sa_family != AF_INET6)
824 continue;
825 a = &((struct sockaddr_in6 *)ifa->ifa_addr)->sin6_addr;
826 if (IN6_IS_ADDR_LINKLOCAL(a))
827 continue;
828 /* get prefix length */
829 m = (unsigned char *)&((struct sockaddr_in6 *)ifa->ifa_netmask)->sin6_addr;
830 lim = (unsigned char *)(ifa->ifa_netmask) + ifa->ifa_netmask->sa_len;
831 plen = prefixlen(m, lim);
832 if (plen <= 0 || plen > 128) {
833 logit(LOG_ERR, "<%s> failed to get prefixlen "
834 "or prefix is invalid",
835 __func__);
836 exit(1);
837 }
838 if (plen == 128) /* XXX */
839 continue;
840 if (find_prefix(rai, a, plen)) {
841 /* ignore a duplicated prefix. */
842 continue;
843 }
844
845 /* allocate memory to store prefix info. */
846 if ((pp = calloc(1, sizeof(*pp))) == NULL) {
847 logit(LOG_ERR,
848 "<%s> can't get allocate buffer for prefix",
849 __func__);
850 exit(1);
851 }
852
853 /* set prefix, sweep bits outside of prefixlen */
854 pp->prefixlen = plen;
855 memcpy(&pp->prefix, a, sizeof(*a));
856 if (1)
857 {
858 p = (unsigned char *)&pp->prefix;
859 ep = (unsigned char *)(&pp->prefix + 1);
860 while (m < lim && p < ep)
861 *p++ &= *m++;
862 while (p < ep)
863 *p++ = 0x00;
864 }
865 if (!inet_ntop(AF_INET6, &pp->prefix, ntopbuf,
866 sizeof(ntopbuf))) {
867 logit(LOG_ERR, "<%s> inet_ntop failed", __func__);
868 exit(1);
869 }
870 logit(LOG_DEBUG,
871 "<%s> add %s/%d to prefix list on %s",
872 __func__, ntopbuf, pp->prefixlen, rai->ifname);
873
874 /* set other fields with protocol defaults */
875 pp->validlifetime = DEF_ADVVALIDLIFETIME;
876 pp->preflifetime = DEF_ADVPREFERREDLIFETIME;
877 pp->onlinkflg = 1;
878 pp->autoconfflg = 1;
879 pp->origin = PREFIX_FROM_KERNEL;
880 pp->rainfo = rai;
881
882 /* link into chain */
883 TAILQ_INSERT_TAIL(&rai->prefix, pp, next);
884 rai->pfxs++;
885 }
886
887 freeifaddrs(ifap);
888}

900/*
901 * Add a prefix to the list of specified interface and reconstruct
902 * the outgoing packet.
903 * The prefix must not be in the list.
904 * XXX: other parameters of the prefix(e.g. lifetime) should be
905 * able to be specified.
906 */
907static void
908add_prefix(struct rainfo *rai, struct in6_prefixreq *ipr)
909{
910 struct prefix *prefix;
911 char ntopbuf[INET6_ADDRSTRLEN];
912
913 if ((prefix = calloc(1, sizeof(*prefix))) == NULL) {
914 logit(LOG_ERR, "<%s> memory allocation failed",
915 __func__);
916 return; /* XXX: error or exit? */
917 }
918 prefix->prefix = ipr->ipr_prefix.sin6_addr;
919 prefix->prefixlen = ipr->ipr_plen;
920 prefix->validlifetime = ipr->ipr_vltime;
921 prefix->preflifetime = ipr->ipr_pltime;
922 prefix->onlinkflg = ipr->ipr_raf_onlink;
923 prefix->autoconfflg = ipr->ipr_raf_auto;
924 prefix->origin = PREFIX_FROM_DYNAMIC;
925
926 prefix->rainfo = rai;
927 TAILQ_INSERT_TAIL(&rai->prefix, prefix, next);
928 rai->pfxs++;
929
930 logit(LOG_DEBUG, "<%s> new prefix %s/%d was added on %s",
931 __func__, inet_ntop(AF_INET6, &ipr->ipr_prefix.sin6_addr,
932 ntopbuf, INET6_ADDRSTRLEN),
933 ipr->ipr_plen, rai->ifname);
934
935 /* free the previous packet */
936 free(rai->ra_data);
937 rai->ra_data = NULL;
938
939 /* reconstruct the packet */
940 make_packet(rai);
941}

943/*
944 * Delete a prefix to the list of specified interface and reconstruct
945 * the outgoing packet.
946 * The prefix must be in the list.
947 */
948void
949delete_prefix(struct prefix *prefix)
950{
951 char ntopbuf[INET6_ADDRSTRLEN];
952 struct rainfo *rai = prefix->rainfo;
953
954 TAILQ_REMOVE(&rai->prefix, prefix, next);
955 rai->pfxs--;
956 logit(LOG_DEBUG, "<%s> prefix %s/%d was deleted on %s",
957 __func__, inet_ntop(AF_INET6, &prefix->prefix,
958 ntopbuf, INET6_ADDRSTRLEN),
959 prefix->prefixlen, rai->ifname);
960 rtadvd_remove_timer(&prefix->timer);
961 free(prefix);
962}

964void
965invalidate_prefix(struct prefix *prefix)
966{
967 char ntopbuf[INET6_ADDRSTRLEN];
968 struct timespec timo;
969 struct rainfo *rai = prefix->rainfo;
970
971 if (prefix->timer) { /* sanity check */
972 logit(LOG_ERR,
973 "<%s> assumption failure: timer already exists",
974 __func__);
975 exit(1);
976 }
977
978 logit(LOG_DEBUG, "<%s> prefix %s/%d was invalidated on %s, "
979 "will expire in %ld seconds", __func__,
980 inet_ntop(AF_INET6, &prefix->prefix, ntopbuf, INET6_ADDRSTRLEN),
981 prefix->prefixlen, rai->ifname, (long)prefix_timo);
982
983 /* set the expiration timer */
984 prefix->timer = rtadvd_add_timer(prefix_timeout, NULL, prefix, NULL);
985 if (prefix->timer == NULL) {
986 logit(LOG_ERR, "<%s> failed to add a timer for a prefix. "
987 "remove the prefix", __func__);
988 delete_prefix(prefix);
989 }
990 timo.tv_sec = prefix_timo;
991 timo.tv_nsec = 0;
992 rtadvd_set_timer(&timo, prefix->timer);
993}

1096void
1097make_packet(struct rainfo *rainfo)
1098{
1099 size_t packlen, lladdroptlen = 0;
1100 char *buf;
1101 struct nd_router_advert *ra;
1102 struct nd_opt_prefix_info *ndopt_pi;
1103 struct nd_opt_mtu *ndopt_mtu;
1104 struct prefix *pfx;
1105 struct nd_opt_route_info *ndopt_rti;
1106 struct rtinfo *rti;
1107 struct nd_opt_rdnss *ndopt_rdnss;
1108 struct rdnss *rdns;
1109 struct rdnss_addr *rdnsa;
1110 struct nd_opt_dnssl *ndopt_dnssl;
1111 struct dnssl *dnsl;
1112 struct dnssl_domain *dnsd;
1113 size_t len, plen;
1114
1115 /* calculate total length */
1116 packlen = sizeof(struct nd_router_advert);
1117 if (rainfo->advlinkopt) {
1118 if ((lladdroptlen = lladdropt_length(rainfo->sdl)) == 0) {
1119 logit(LOG_INFO,
1120 "<%s> link-layer address option has"
1121 " null length on %s. Treat as not included.",
1122 __func__, rainfo->ifname);
1123 rainfo->advlinkopt = 0;
1124 }
1125 packlen += lladdroptlen;
1126 }
1127 if (TAILQ_FIRST(&rainfo->prefix) != NULL)
1128 packlen += sizeof(struct nd_opt_prefix_info) * rainfo->pfxs;
1129 if (rainfo->linkmtu)
1130 packlen += sizeof(struct nd_opt_mtu);
1131 TAILQ_FOREACH(rti, &rainfo->route, next)
1132 packlen += sizeof(struct nd_opt_route_info) +
1133 ((rti->prefixlen + 0x3f) >> 6) * 8;
1134
1135 TAILQ_FOREACH(rdns, &rainfo->rdnss, next) {
1136 packlen += sizeof(struct nd_opt_rdnss);
1137 TAILQ_FOREACH(rdnsa, &rdns->list, next)
1138 packlen += sizeof(rdnsa->addr);
1139 }
1140 TAILQ_FOREACH(dnsl, &rainfo->dnssl, next) {
1141 packlen += sizeof(struct nd_opt_dnssl);
1142 len = 0;
1143 TAILQ_FOREACH(dnsd, &dnsl->list, next)
1144 len += dnsd->len;
1145 len += len % 8 ? 8 - len % 8 : 0;
1146 packlen += len;
1147 }
1148
1149 /* allocate memory for the packet */
1150 if ((buf = realloc(rainfo->ra_data, packlen)) == NULL) {
1151 logit(LOG_ERR,
1152 "<%s> can't get enough memory for an RA packet %m",
1153 __func__);
1154 exit(1);
1155 }
1156 rainfo->ra_data = buf;
1157 /* XXX: what if packlen > 576? */
1158 rainfo->ra_datalen = packlen;
1159#define CHECKLEN(size) \
1160 do { \
1161 if (buf + size > rainfo->ra_data + packlen) { \
1162 logit(LOG_ERR, \
1163 "<%s, %d> RA packet does not fit in %zu",\
1164 __func__, __LINE__, packlen); \
1165 exit(1); \
1166 } \
1167 } while (/*CONSTCOND*/0)
1168 /*
1169 * construct the packet
1170 */
1171 CHECKLEN(sizeof(*ra));
1172 ra = (struct nd_router_advert *)buf;
1173 ra->nd_ra_type = ND_ROUTER_ADVERT;
1174 ra->nd_ra_code = 0;
1175 ra->nd_ra_cksum = 0;
1176 ra->nd_ra_curhoplimit = (uint8_t)(0xff & rainfo->hoplimit);
1177 ra->nd_ra_flags_reserved = 0; /* just in case */
1178 /*
1179 * XXX: the router preference field, which is a 2-bit field, should be
1180 * initialized before other fields.
1181 */
1182 ra->nd_ra_flags_reserved = 0xff & rainfo->rtpref;
1183 ra->nd_ra_flags_reserved |=
1184 rainfo->managedflg ? ND_RA_FLAG_MANAGED : 0;
1185 ra->nd_ra_flags_reserved |=
1186 rainfo->otherflg ? ND_RA_FLAG_OTHER : 0;
1187 ra->nd_ra_router_lifetime = htons(rainfo->lifetime);
1188 ra->nd_ra_reachable = htonl(rainfo->reachabletime);
1189 ra->nd_ra_retransmit = htonl(rainfo->retranstimer);
1190 buf += sizeof(*ra);
1191
1192 if (rainfo->advlinkopt) {
1193 CHECKLEN(sizeof(struct nd_opt_hdr));
1194 lladdropt_fill(rainfo->sdl, (struct nd_opt_hdr *)buf);
1195 buf += lladdroptlen;
1196 }
1197
1198 if (rainfo->linkmtu) {
1199 CHECKLEN(sizeof(*ndopt_mtu));
1200 ndopt_mtu = (struct nd_opt_mtu *)buf;
1201 ndopt_mtu->nd_opt_mtu_type = ND_OPT_MTU;
1202 ndopt_mtu->nd_opt_mtu_len = 1;
1203 ndopt_mtu->nd_opt_mtu_reserved = 0;
1204 ndopt_mtu->nd_opt_mtu_mtu = htonl(rainfo->linkmtu);
1205 buf += sizeof(struct nd_opt_mtu);
1206 }
1207
1208 TAILQ_FOREACH(pfx, &rainfo->prefix, next) {
1209 uint32_t vltime, pltime;
1210 struct timespec now;
1211
1212 CHECKLEN(sizeof(*ndopt_pi));
1213 ndopt_pi = (struct nd_opt_prefix_info *)buf;
1214 ndopt_pi->nd_opt_pi_type = ND_OPT_PREFIX_INFORMATION;
1215 ndopt_pi->nd_opt_pi_len = 4;
1216 ndopt_pi->nd_opt_pi_prefix_len = pfx->prefixlen;
1217 ndopt_pi->nd_opt_pi_flags_reserved = 0;
1218 if (pfx->onlinkflg)
1219 ndopt_pi->nd_opt_pi_flags_reserved |=
1220 ND_OPT_PI_FLAG_ONLINK;
1221 if (pfx->autoconfflg)
1222 ndopt_pi->nd_opt_pi_flags_reserved |=
1223 ND_OPT_PI_FLAG_AUTO;
1224 if (pfx->timer)
1225 vltime = 0;
1226 else {
1227 if (pfx->vltimeexpire || pfx->pltimeexpire)
1228 prog_clock_gettime(CLOCK_MONOTONIC, &now);
1229 if (pfx->vltimeexpire == 0)
1230 vltime = pfx->validlifetime;
1231 else
1232 vltime = (pfx->vltimeexpire > now.tv_sec) ?
1233 pfx->vltimeexpire - now.tv_sec : 0;
1234 }
1235 if (pfx->timer)
1236 pltime = 0;
1237 else {
1238 if (pfx->pltimeexpire == 0)
1239 pltime = pfx->preflifetime;
1240 else
1241 pltime = (pfx->pltimeexpire > now.tv_sec) ?
1242 pfx->pltimeexpire - now.tv_sec : 0;
1243 }
1244 if (vltime < pltime) {
1245 /*
1246 * this can happen if vltime is decrement but pltime
1247 * is not.
1248 */
1249 pltime = vltime;
1250 }
1251 ndopt_pi->nd_opt_pi_valid_time = htonl(vltime);
1252 ndopt_pi->nd_opt_pi_preferred_time = htonl(pltime);
1253 ndopt_pi->nd_opt_pi_reserved2 = 0;
1254 ndopt_pi->nd_opt_pi_prefix = pfx->prefix;
1255
1256 buf += sizeof(struct nd_opt_prefix_info);
1257 }
1258
1259 TAILQ_FOREACH(rti, &rainfo->route, next) {
1260 uint8_t psize = (rti->prefixlen + 0x3f) >> 6;
1261
1262 CHECKLEN(sizeof(*ndopt_rti));
1263 ndopt_rti = (struct nd_opt_route_info *)buf;
1264 ndopt_rti->nd_opt_rti_type = ND_OPT_ROUTE_INFO;
1265 ndopt_rti->nd_opt_rti_len = 1 + psize;
1266 ndopt_rti->nd_opt_rti_prefixlen = rti->prefixlen;
1267 ndopt_rti->nd_opt_rti_flags = 0xff & rti->rtpref;
1268 ndopt_rti->nd_opt_rti_lifetime = htonl(rti->ltime);
1269 memcpy(ndopt_rti + 1, &rti->prefix, psize * 8);
1270 buf += sizeof(struct nd_opt_route_info) + psize * 8;
1271 }
1272
1273 TAILQ_FOREACH(rdns, &rainfo->rdnss, next) {
1274 CHECKLEN(sizeof(*ndopt_rdnss));
1275 ndopt_rdnss = (struct nd_opt_rdnss *)buf;
1276 ndopt_rdnss->nd_opt_rdnss_type = ND_OPT_RDNSS;
1277 ndopt_rdnss->nd_opt_rdnss_len = 1;
1278 ndopt_rdnss->nd_opt_rdnss_reserved = 0;
1279 ndopt_rdnss->nd_opt_rdnss_lifetime = htonl(rdns->lifetime);
1280 buf += sizeof(*ndopt_rdnss);
1281
1282 TAILQ_FOREACH(rdnsa, &rdns->list, next) {
1283 CHECKLEN(sizeof(rdnsa->addr));
1284 memcpy(buf, &rdnsa->addr, sizeof(rdnsa->addr));
1285 ndopt_rdnss->nd_opt_rdnss_len += 2;
1286 buf += sizeof(rdnsa->addr);
1287 }
1288 }
1289
1290 TAILQ_FOREACH(dnsl, &rainfo->dnssl, next) {
1291 CHECKLEN(sizeof(*ndopt_dnssl));
1292 ndopt_dnssl = (struct nd_opt_dnssl *)buf;
1293 ndopt_dnssl->nd_opt_dnssl_type = ND_OPT_DNSSL;
1294 ndopt_dnssl->nd_opt_dnssl_len = 0;
1295 ndopt_dnssl->nd_opt_dnssl_reserved = 0;
1296 ndopt_dnssl->nd_opt_dnssl_lifetime = htonl(dnsl->lifetime);
1297 buf += sizeof(*ndopt_dnssl);
1298
1299 TAILQ_FOREACH(dnsd, &dnsl->list, next) {
1300 CHECKLEN(dnsd->len);
1301 memcpy(buf, dnsd->domain, dnsd->len);
1302 buf += dnsd->len;
1303 }
1304 /* Ensure our length is padded correctly */
1305 len = buf - (char *)ndopt_dnssl;
1306 plen = len % 8 ? 8 - len % 8 : 0;
1307 CHECKLEN(plen);
1308 memset(buf, 0, plen);
1309 buf += plen;
1310 ndopt_dnssl->nd_opt_dnssl_len = (len + plen) / 8;
1311 }
1312 memset(buf, 0, packlen - (buf - rainfo->ra_data));
1313}
