59#define RR_ISSET_SEGNUM(segnum_bits, segnum) \
60 ((((segnum_bits)[(segnum) >> 5]) & (1 << ((segnum) & 31))) != 0)
61#define RR_SET_SEGNUM(segnum_bits, segnum) \
62 (((segnum_bits)[(segnum) >> 5]) |= (1 << ((segnum) & 31)))

79/*
80 * Check validity of a Prefix Control Operation(PCO).
81 * Return 0 on success, 1 on failure.
82 */
83static int
84rr_pco_check(int len, struct rr_pco_match *rpm)
85{
86 struct rr_pco_use *rpu, *rpulim;
87 int checklen;
88
89 /* rpm->rpm_len must be (4N * 3) as router-renum-05.txt */
90 if ((rpm->rpm_len - 3) < 0 || /* must be at least 3 */
91 (rpm->rpm_len - 3) & 0x3) { /* must be multiple of 4 */
92 logit(LOG_WARNING, "<%s> rpm_len %d is not 4N * 3",
93 __func__, rpm->rpm_len);
94 return 1;
95 }
96 /* rpm->rpm_code must be valid value */
97 switch (rpm->rpm_code) {
98 case RPM_PCO_ADD:
99 case RPM_PCO_CHANGE:
100 case RPM_PCO_SETGLOBAL:
101 break;
102 default:
103 logit(LOG_WARNING, "<%s> unknown rpm_code %d", __func__,
104 rpm->rpm_code);
105 return 1;
106 }
107 /* rpm->rpm_matchlen must be 0 to 128 inclusive */
108 if (rpm->rpm_matchlen > 128) {
109 logit(LOG_WARNING, "<%s> rpm_matchlen %d is over 128",
110 __func__, rpm->rpm_matchlen);
111 return 1;
112 }
113
114 /*
115 * rpu->rpu_uselen, rpu->rpu_keeplen, and sum of them must be
116 * between 0 and 128 inclusive
117 */
118 for (rpu = (struct rr_pco_use *)(rpm + 1),
119 rpulim = (struct rr_pco_use *)((char *)rpm + len);
120 rpu < rpulim;
121 rpu += 1) {
122 checklen = rpu->rpu_uselen;
123 checklen += rpu->rpu_keeplen;
124 /*
125 * omit these check, because either of rpu_uselen
126 * and rpu_keeplen is unsigned char
127 * (128 > rpu_uselen > 0)
128 * (128 > rpu_keeplen > 0)
129 * (rpu_uselen + rpu_keeplen > 0)
130 */
131 if (checklen > 128) {
132 logit(LOG_WARNING, "<%s> sum of rpu_uselen %d and"
133 " rpu_keeplen %d is %d(over 128)",
134 __func__, rpu->rpu_uselen,
135 rpu->rpu_keeplen,
136 rpu->rpu_uselen + rpu->rpu_keeplen);
137 return 1;
138 }
139 }
140 return 0;
141}

143static void
144do_use_prefix(int len, struct rr_pco_match *rpm,
145 struct in6_rrenumreq *irr, int ifindex)
146{
147 struct rr_pco_use *rpu, *rpulim;
148 struct rainfo *rai;
149 struct prefix *pp;
150 struct in6_addr rpm_prefix, rpu_prefix;
151
152 rpu = (struct rr_pco_use *)(rpm + 1);
153 rpulim = (struct rr_pco_use *)((char *)rpm + len);
154
155 if (rpu == rpulim) { /* no use prefix */
156 if (rpm->rpm_code == RPM_PCO_ADD)
157 return;
158
159 irr->irr_u_uselen = 0;
160 irr->irr_u_keeplen = 0;
161 irr->irr_raf_mask_onlink = 0;
162 irr->irr_raf_mask_auto = 0;
163 irr->irr_vltime = 0;
164 irr->irr_pltime = 0;
165 memset(&irr->irr_flags, 0, sizeof(irr->irr_flags));
166 irr->irr_useprefix.sin6_len = 0; /* let it mean, no addition */
167 irr->irr_useprefix.sin6_family = 0;
168 irr->irr_useprefix.sin6_addr = in6addr_any;
169 if (prog_ioctl(s, rrcmd2pco[rpm->rpm_code], irr) < 0 &&
170 errno != EADDRNOTAVAIL)
171 logit(LOG_ERR, "<%s> ioctl: %m", __func__);
172 return;
173 }
174
175 for (rpu = (struct rr_pco_use *)(rpm + 1),
176 rpulim = (struct rr_pco_use *)((char *)rpm + len);
177 rpu < rpulim;
178 rpu += 1) {
179 /* init in6_rrenumreq fields */
180 irr->irr_u_uselen = rpu->rpu_uselen;
181 irr->irr_u_keeplen = rpu->rpu_keeplen;
182 irr->irr_raf_mask_onlink =
183 !!(rpu->rpu_ramask & ICMP6_RR_PCOUSE_RAFLAGS_ONLINK);
184 irr->irr_raf_mask_auto =
185 !!(rpu->rpu_ramask & ICMP6_RR_PCOUSE_RAFLAGS_AUTO);
186 irr->irr_vltime = ntohl(rpu->rpu_vltime);
187 irr->irr_pltime = ntohl(rpu->rpu_pltime);
188 irr->irr_raf_onlink =
189 (rpu->rpu_raflags & ICMP6_RR_PCOUSE_RAFLAGS_ONLINK) == 0 ? 0 : 1;
190 irr->irr_raf_auto =
191 (rpu->rpu_raflags & ICMP6_RR_PCOUSE_RAFLAGS_AUTO) == 0 ? 0 : 1;
192 irr->irr_rrf_decrvalid =
193 (rpu->rpu_flags & ICMP6_RR_PCOUSE_FLAGS_DECRVLTIME) == 0 ? 0 : 1;
194 irr->irr_rrf_decrprefd =
195 (rpu->rpu_flags & ICMP6_RR_PCOUSE_FLAGS_DECRPLTIME) == 0 ? 0 : 1;
196 irr->irr_useprefix.sin6_len = sizeof(irr->irr_useprefix);
197 irr->irr_useprefix.sin6_family = AF_INET6;
198 irr->irr_useprefix.sin6_addr = rpu->rpu_prefix;
199
200 if (prog_ioctl(s, rrcmd2pco[rpm->rpm_code], irr) < 0 &&
201 errno != EADDRNOTAVAIL)
202 logit(LOG_ERR, "<%s> ioctl: %m", __func__);
203
204 /* very adhoc: should be rewritten */
205 memcpy(&rpm_prefix, &rpm->rpm_prefix, sizeof(rpm_prefix));
206 memcpy(&rpu_prefix, &rpu->rpu_prefix, sizeof(rpu_prefix));
207 if (rpm->rpm_code == RPM_PCO_CHANGE &&
208 IN6_ARE_ADDR_EQUAL(&rpm_prefix, &rpu_prefix) &&
209 rpm->rpm_matchlen == rpu->rpu_uselen &&
210 rpu->rpu_uselen == rpu->rpu_keeplen) {
211 if ((rai = if_indextorainfo(ifindex)) == NULL)
212 continue; /* non-advertising IF */
213
214 TAILQ_FOREACH(pp, &rai->prefix, next) {
215 struct timespec now;
216
217 if (prefix_match(&pp->prefix, pp->prefixlen,
218 &rpm_prefix,
219 rpm->rpm_matchlen)) {
220 /* change parameters */
221 pp->validlifetime = ntohl(rpu->rpu_vltime);
222 pp->preflifetime = ntohl(rpu->rpu_pltime);
223 if (irr->irr_rrf_decrvalid) {
224 prog_clock_gettime(CLOCK_MONOTONIC,
225 &now);
226 pp->vltimeexpire =
227 now.tv_sec + pp->validlifetime;
228 } else
229 pp->vltimeexpire = 0;
230 if (irr->irr_rrf_decrprefd) {
231 prog_clock_gettime(CLOCK_MONOTONIC,
232 &now);
233 pp->pltimeexpire =
234 now.tv_sec + pp->preflifetime;
235 } else
236 pp->pltimeexpire = 0;
237 }
238 }
239 }
240 }
241}

243/*
244 * process a Prefix Control Operation(PCO).
245 * return 0 on success, 1 on failure
246 */
247static int
248do_pco(struct icmp6_router_renum *rr, int len, struct rr_pco_match *rpm)
249{
250 int ifindex = 0;
251 struct in6_rrenumreq irr;
252 struct rainfo *rai;
253
254 if ((rr_pco_check(len, rpm) != 0))
255 return 1;
256
257 if (s == -1 && (s = prog_socket(AF_INET6, SOCK_DGRAM, 0)) < 0) {
258 logit(LOG_ERR, "<%s> socket: %m", __func__);
259 exit(1);
260 }
261
262 memset(&irr, 0, sizeof(irr));
263 irr.irr_origin = PR_ORIG_RR;
264 irr.irr_m_len = rpm->rpm_matchlen;
265 irr.irr_m_minlen = rpm->rpm_minlen;
266 irr.irr_m_maxlen = rpm->rpm_maxlen;
267 irr.irr_matchprefix.sin6_len = sizeof(irr.irr_matchprefix);
268 irr.irr_matchprefix.sin6_family = AF_INET6;
269 irr.irr_matchprefix.sin6_addr = rpm->rpm_prefix;
270
271 /*
272 * if ICMP6_RR_FLAGS_FORCEAPPLY(A flag) is 0 and IFF_UP is off,
273 * the interface is not applied
274 */
275
276 if (rr->rr_flags & ICMP6_RR_FLAGS_FORCEAPPLY) {
277 while (if_indextoname(++ifindex, irr.irr_name)) {
278 rai = if_indextorainfo(ifindex);
279 if (rai && (rai->ifflags & IFF_UP)) {
280 /* TODO: interface scope check */
281 do_use_prefix(len, rpm, &irr, ifindex);
282 }
283 }
284 }
285 if (errno == ENXIO)
286 return 0;
287 else if (errno) {
288 logit(LOG_ERR, "<%s> if_indextoname: %m", __func__);
289 return 1;
290 }
291 return 0;
292}

294/*
295 * call do_pco() for each Prefix Control Operations(PCOs) in a received
296 * Router Renumbering Command packet.
297 * return 0 on success, 1 on failure
298 */
299static int
300do_rr(size_t len, struct icmp6_router_renum *rr)
301{
302 struct rr_pco_match *rpm;
303 char *cp, *lim;
304
305 lim = (char *)rr + len;
306 cp = (char *)(rr + 1);
307 len -= sizeof(struct icmp6_router_renum);
308
309 while (cp < lim) {
310 size_t rpmlen;
311
312 rpm = (struct rr_pco_match *)cp;
313 if (len < sizeof(struct rr_pco_match)) {
314 tooshort:
315 logit(LOG_ERR, "<%s> pkt too short. left len = %zd. "
316 "garbage at end of pkt?", __func__, len);
317 return 1;
318 }
319 rpmlen = rpm->rpm_len << 3;
320 if (len < rpmlen)
321 goto tooshort;
322
323 if (do_pco(rr, rpmlen, rpm)) {
324 logit(LOG_WARNING, "<%s> invalid PCO", __func__);
325 goto next;
326 }
327
328 next:
329 cp += rpmlen;
330 len -= rpmlen;
331 }
332
333 return 0;
334}

336/*
337 * check validity of a router renumbering command packet
338 * return 0 on success, 1 on failure
339 */
340static int
341rr_command_check(size_t len, struct icmp6_router_renum *rr,
342 struct in6_addr *from, struct in6_addr *dst)
343{
344 char ntopbuf[INET6_ADDRSTRLEN];
345
346 /* omit rr minimal length check. hope kernel have done it. */
347 /* rr_command length check */
348 if (len < (sizeof(struct icmp6_router_renum) +
349 sizeof(struct rr_pco_match))) {
350 logit(LOG_ERR, "<%s> rr_command len %zd is too short",
351 __func__, len);
352 return 1;
353 }
354
355 /* destination check. only for multicast. omit unicast check. */
356 if (IN6_IS_ADDR_MULTICAST(dst) && !IN6_IS_ADDR_MC_LINKLOCAL(dst) &&
357 !IN6_IS_ADDR_MC_SITELOCAL(dst)) {
358 logit(LOG_ERR, "<%s> dst mcast addr %s is illegal",
359 __func__,
360 inet_ntop(AF_INET6, dst, ntopbuf, INET6_ADDRSTRLEN));
361 return 1;
362 }
363
364 /* seqnum and segnum check */
365 if (rro.rro_seqnum > rr->rr_seqnum) {
366 logit(LOG_WARNING,
367 "<%s> rcvd old seqnum %d from %s",
368 __func__, (uint32_t)ntohl(rr->rr_seqnum),
369 inet_ntop(AF_INET6, from, ntopbuf, INET6_ADDRSTRLEN));
370 return 1;
371 }
372 if (rro.rro_seqnum == rr->rr_seqnum &&
373 (rr->rr_flags & ICMP6_RR_FLAGS_TEST) == 0 &&
374 RR_ISSET_SEGNUM(rro.rro_segnum_bits, rr->rr_segnum)) {
375 if ((rr->rr_flags & ICMP6_RR_FLAGS_REQRESULT) != 0)
376 logit(LOG_WARNING,
377 "<%s> rcvd duped segnum %d from %s",
378 __func__, rr->rr_segnum,
379 inet_ntop(AF_INET6, from, ntopbuf,
380 INET6_ADDRSTRLEN));
381 return 0;
382 }
383
384 /* update seqnum */
385 if (rro.rro_seqnum != rr->rr_seqnum) {
386 /* then must be "<" */
387
388 /* init rro_segnum_bits */
389 memset(rro.rro_segnum_bits, 0,
390 sizeof(rro.rro_segnum_bits));
391 }
392 rro.rro_seqnum = rr->rr_seqnum;
393
394 return 0;
395}

397static void
398rr_command_input(int len, struct icmp6_router_renum *rr,
399 struct in6_addr *from, struct in6_addr *dst)
400{
401 /* rr_command validity check */
402 if (rr_command_check(len, rr, from, dst))
403 goto failed;
404 if ((rr->rr_flags & (ICMP6_RR_FLAGS_TEST|ICMP6_RR_FLAGS_REQRESULT)) ==
405 ICMP6_RR_FLAGS_TEST)
406 return;
407
408 /* do router renumbering */
409 if (do_rr(len, rr)) {
410 goto failed;
411 }
412
413 /* update segnum */
414 RR_SET_SEGNUM(rro.rro_segnum_bits, rr->rr_segnum);
415
416 return;
417
418 failed:
419 logit(LOG_ERR, "<%s> received RR was invalid", __func__);
420 return;
421}

423void
424rr_input(size_t len, struct icmp6_router_renum *rr, struct in6_pktinfo *pi,
425 struct sockaddr_in6 *from, struct in6_addr *dst)
426{
427 char ntopbuf[2][INET6_ADDRSTRLEN], ifnamebuf[IFNAMSIZ];
428
429 logit(LOG_DEBUG,
430 "<%s> RR received from %s to %s on %s",
431 __func__,
432 inet_ntop(AF_INET6, &from->sin6_addr,
433 ntopbuf[0], INET6_ADDRSTRLEN),
434 inet_ntop(AF_INET6, &dst, ntopbuf[1], INET6_ADDRSTRLEN),
435 if_indextoname(pi->ipi6_ifindex, ifnamebuf));
436
437 /* packet validation based on Section 4.1 of RFC2894 */
438 if (len < sizeof(struct icmp6_router_renum)) {
439 logit(LOG_NOTICE,
440 "<%s>: RR short message (size %zd) from %s to %s on %s",
441 __func__, len,
442 inet_ntop(AF_INET6, &from->sin6_addr,
443 ntopbuf[0], INET6_ADDRSTRLEN),
444 inet_ntop(AF_INET6, &dst, ntopbuf[1], INET6_ADDRSTRLEN),
445 if_indextoname(pi->ipi6_ifindex, ifnamebuf));
446 return;
447 }
448
449 /*
450 * If the IPv6 destination address is neither an All Routers multicast
451 * address [AARCH] nor one of the receiving router's unicast addresses,
452 * the message MUST be discarded and SHOULD be logged to network
453 * management.
454 * We rely on the kernel input routine for unicast addresses, and thus
455 * check multicast destinations only.
456 */
457 if (IN6_IS_ADDR_MULTICAST(&pi->ipi6_addr) &&
458 !IN6_ARE_ADDR_EQUAL(&sin6_sitelocal_allrouters.sin6_addr,
459 &pi->ipi6_addr))
460 {
461 logit(LOG_NOTICE,
462 "<%s>: RR message with invalid destination (%s) "
463 "from %s on %s",
464 __func__,
465 inet_ntop(AF_INET6, &dst, ntopbuf[0], INET6_ADDRSTRLEN),
466 inet_ntop(AF_INET6, &from->sin6_addr,
467 ntopbuf[1], INET6_ADDRSTRLEN),
468 if_indextoname(pi->ipi6_ifindex, ifnamebuf));
469 return;
470 }
471
472 rr_rcvifindex = pi->ipi6_ifindex;
473
474 switch (rr->rr_code) {
475 case ICMP6_ROUTER_RENUMBERING_COMMAND:
476 rr_command_input(len, rr, &from->sin6_addr, dst);
477 /* TODO: send reply msg */
478 break;
479 case ICMP6_ROUTER_RENUMBERING_RESULT:
480 /* RESULT will be processed by rrenumd */
481 break;
482 case ICMP6_ROUTER_RENUMBERING_SEQNUM_RESET:
483 /* TODO: sequence number reset */
484 break;
485 default:
486 logit(LOG_ERR, "<%s> received unknown code %d",
487 __func__, rr->rr_code);
488 break;
489
490 }
491
492 return;
493}
