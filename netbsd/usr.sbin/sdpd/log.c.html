40void
41log_open(char const *prog, bool log2stderr)
42{
43 openlog(prog, LOG_PID|LOG_NDELAY|(log2stderr? LOG_PERROR:0), LOG_USER);
44}
45
46void
47log_close(void)
48{
49 closelog();
50}
51
52void
53log_emerg(char const *message, ...)
54{
55 va_list ap;
56
57 va_start(ap, message);
58 vsyslog(LOG_EMERG, message, ap);
59 va_end(ap);
60}
