37#include <unistd.h>
38#include <stdlib.h>
39#include <stdio.h>
40#include <fcntl.h>
41#include <sys/ioctl.h>
42#include SESINC
43
44int
45main(int a, char *v[])
46{
47 int fd;
48 long val;
49 ses_encstat sestat;
50
51 if (a != 3) {
52 fprintf(stderr, "usage: %s device enclosure_status\n", *v);
53 return (1);
54 }
55 fd = open(v[1], O_RDWR);
56 if (fd < 0) {
57 perror(v[1]);
58 return (1);
59 }
60
61 val = strtol(v[2], NULL, 0);
62 sestat = (ses_encstat) val;
63 if (ioctl(fd, SESIOC_SETENCSTAT, (caddr_t) &sestat) < 0) {
64 perror("SESIOC_SETENCSTAT");
65 }
66 (void) close(fd);
67 return (0);
68}
