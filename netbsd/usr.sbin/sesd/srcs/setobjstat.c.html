37#include <unistd.h>
38#include <stdlib.h>
39#include <stdio.h>
40#include <fcntl.h>
41#include <sys/ioctl.h>
42#include SESINC
43
44int
45main(int a, char *v[])
46{
47 int fd;
48 int i;
49 ses_objstat obj;
50 long cvt;
51 char *x;
52
53 if (a != 7) {
54usage:
55 fprintf(stderr,
56 "usage: %s device objectid stat0 stat1 stat2 stat3\n", *v);
57 return (1);
58 }
59 fd = open(v[1], O_RDWR);
60 if (fd < 0) {
61 perror(v[1]);
62 return (1);
63 }
64 x = v[2];
65 cvt = strtol(v[2], &x, 0);
66 if (x == v[2]) {
67 goto usage;
68 }
69 obj.obj_id = cvt;
70 for (i = 0; i < 4; i++) {
71 x = v[3 + i];
72 cvt = strtol(v[3 + i], &x, 0);
73 if (x == v[3 + i]) {
74 goto usage;
75 }
76 obj.cstat[i] = cvt;
77 }
78 if (ioctl(fd, SESIOC_SETOBJSTAT, (caddr_t) &obj) < 0) {
79 perror("SESIOC_SETOBJSTAT");
80 }
81 (void) close(fd);
82 return (0);
83}
