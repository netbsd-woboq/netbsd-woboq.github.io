13extern const char *__progname;
14
15#define ACT_ERROR 1 /* none of the below */
16#define ACT_QUERYALL 2 /* srtX */
17#define ACT_QUERYONE 3 /* srtX N */
18#define ACT_DEL 4 /* srtX del N */
19#define ACT_ADD 5 /* srtX add srcaddr mask dstif dstaddr */
20#define ACT_SET 6 /* srtX set N srcaddr mask dstif dstaddr */
21#define ACT_FLAGS 7 /* srtX flags */
22#define ACT_SFLAG 8 /* srtX flags [+|-]flag */
23#define ACT_DEBUG 9 /* srtX debug */
24static int action = ACT_ERROR;
25
26static char *txt_dev;
27static char *txt_n;
28static char *txt_flg;
29static char *txt_addr;
30static char *txt_mask;
31static char *txt_dstif;
32static char *txt_dstaddr;
33
34static int devfd;
35
36static struct {
37 const char *name;
38 unsigned int bit;
39 } flagbits[] = { { "mtulock", SSF_MTULOCK },
40 { 0, 0 } };

42static void handleargs(int ac, char **av)
43{
44 txt_dev = av[1];
45 if (ac == 2)
46 { action = ACT_QUERYALL;
47 }
48 else if ((ac == 3) && !strcmp(av[2],"debug"))
49 { action = ACT_DEBUG;
50 }
51 else if ((ac == 3) && !strcmp(av[2],"flags"))
52 { action = ACT_FLAGS;
53 }
54 else if (ac == 3)
55 { action = ACT_QUERYONE;
56 txt_n = av[2];
57 }
58 else if ((ac == 4) && !strcmp(av[2],"del"))
59 { action = ACT_DEL;
60 txt_n = av[3];
61 }
62 else if ((ac == 4) && !strcmp(av[2],"flags"))
63 { action = ACT_SFLAG;
64 txt_flg = av[3];
65 }
66 else if ((ac == 7) && !strcmp(av[2],"add"))
67 { action = ACT_ADD;
68 txt_addr = av[3];
69 txt_mask = av[4];
70 txt_dstif = av[5];
71 txt_dstaddr = av[6];
72 }
73 else if ((ac == 8) && !strcmp(av[2],"set"))
74 { action = ACT_SET;
75 txt_n = av[3];
76 txt_addr = av[4];
77 txt_mask = av[5];
78 txt_dstif = av[6];
79 txt_dstaddr = av[7];
80 }
81 if (action == ACT_ERROR)
82 { fprintf(stderr,"Usage: %s srtX\n",__progname);
83 fprintf(stderr," %s srtX N\n",__progname);
84 fprintf(stderr," %s srtX del N\n",__progname);
85 fprintf(stderr," %s srtX add addr mask dstif dstaddr\n",__progname);
86 fprintf(stderr," %s srtX set N addr mask dstif dstaddr\n",__progname);
87 fprintf(stderr," %s srtX flags {[+|-]flag}\n",__progname);
88 fprintf(stderr," %s srtX debug\n",__progname);
89 exit(1);
90 }
91}

93static void open_dev(int how)
94{
95 if (! index(txt_dev,'/'))
96 { char *tmp;
97 asprintf(&tmp,"/dev/%s",txt_dev);
98 txt_dev = tmp;
99 }
100 devfd = open(txt_dev,how,0);
101 if (devfd < 0)
102 { fprintf(stderr,"%s: can't open %s: %s\n",__progname,txt_dev,strerror(errno));
103 exit(1);
104 }
105}

107static void query_n(int n)
108{
109 struct srt_rt r;
110 char obuf[64];
111
112 r.inx = n;
113 if (ioctl(devfd,SRT_GETRT,&r) < 0)
114 { fprintf(stderr,"%s: can't get rt #%d: %s\n",__progname,n,strerror(errno));
115 return;
116 }
117 printf("%d:",n);
118 printf(" %s",inet_ntop(r.af,&r.srcmatch,&obuf[0],sizeof(obuf)));
119 printf(" /%d",r.srcmask);
120 printf(" %.*s",(int)sizeof(r.u.dstifn),&r.u.dstifn[0]);
121 switch (r.af)
122 { case AF_INET:
123 printf(" %s",inet_ntoa(r.dst.sin.sin_addr));
124 break;
125 case AF_INET6:
126 printf(" %s",inet_ntop(AF_INET6,&r.dst.sin6.sin6_addr,&obuf[0],sizeof(obuf)));
127 break;
128 default:
129 printf(" ?af%d",r.af);
130 break;
131 }
132 printf("\n");
133}
134
135static void do_query(int narg)
136{
137 int i;
138 int n;
139
140 open_dev(O_RDONLY);
141 if (narg >= 0)
142 { query_n(narg);
143 }
144 else
145 { if (ioctl(devfd,SRT_GETNRT,&n) < 0)
146 { fprintf(stderr,"%s: can't get count: %s\n",__progname,strerror(errno));
147 exit(1);
148 }
149 for (i=0;i<n;i++) query_n(i);
150 }
151}

153static void do_del(unsigned int n)
154{
155 open_dev(O_RDWR);
156 if (ioctl(devfd,SRT_DELRT,&n) < 0)
157 { fprintf(stderr,"%s: can't delete #%u: %s\n",__progname,n,strerror(errno));
158 exit(1);
159 }
160}

162static void do_set(int n)
163{
164 struct srt_rt r;
165 int w;
166 int maxw;
167 void *dp;
168
169 open_dev(O_RDWR);
170 if (n < 0)
171 { unsigned int v;
172 if (ioctl(devfd,SRT_GETNRT,&v) < 0)
173 { fprintf(stderr,"%s: can't get count: %s\n",__progname,strerror(errno));
174 exit(1);
175 }
176 n = v;
177 }
178 bzero(&r.dst,sizeof(r.dst));
179 r.inx = n;
180 if (inet_pton(AF_INET,txt_addr,&r.srcmatch.v4) == 1)
181 { r.af = AF_INET;
182 r.dst.sin.sin_family = AF_INET;
183 r.dst.sin.sin_len = sizeof(r.dst.sin);
184 dp = &r.dst.sin.sin_addr;
185 maxw = 32;
186 }
187 else if (inet_pton(AF_INET6,txt_addr,&r.srcmatch.v6) == 1)
188 { r.af = AF_INET6;
189 r.dst.sin6.sin6_family = AF_INET6;
190 r.dst.sin6.sin6_len = sizeof(r.dst.sin6);
191 dp = &r.dst.sin6.sin6_addr;
192 maxw = 128;
193 }
194 else
195 { fprintf(stderr,"%s: %s: invalid match address\n",__progname,txt_addr);
196 exit(1);
197 }
198 if (txt_mask[0] == '/') txt_mask ++;
199 w = atoi(txt_mask);
200 if ((w < 0) || (w > maxw))
201 { fprintf(stderr,"%s: %s: out-of-range CIDR width\n",__progname,txt_mask);
202 exit(1);
203 }
204 r.srcmask = w;
205 if (strlen(txt_dstif) > sizeof(r.u.dstifn)-1)
206 { fprintf(stderr,"%s: %s: too long\n",__progname,txt_dstif);
207 exit(1);
208 }
209 strncpy(&r.u.dstifn[0],txt_dstif,sizeof(r.u.dstifn));
210 if (inet_pton(r.af,txt_dstaddr,dp) != 1)
211 { fprintf(stderr,"%s: %s: invalid destination address\n",__progname,txt_dstaddr);
212 exit(1);
213 }
214 if (ioctl(devfd,SRT_SETRT,&r) < 0)
215 { fprintf(stderr,"%s: can't set route: %s\n",__progname,strerror(errno));
216 exit(1);
217 }
218}

220static void do_flags(void)
221{
222 unsigned int f;
223 int i;
224
225 open_dev(O_RDONLY);
226 if (ioctl(devfd,SRT_GFLAGS,&f) < 0)
227 { fprintf(stderr,"%s: can't get flags: %s\n",__progname,strerror(errno));
228 exit(1);
229 }
230 for (i=0;flagbits[i].name;i++)
231 { printf(" %c%s",(f&flagbits[i].bit)?'+':'-',flagbits[i].name);
232 f &= ~flagbits[i].bit;
233 }
234 if (f) printf(" +0x%x",f);
235 printf("\n");
236}
237
238static void do_sflag(void)
239{
240 unsigned int f;
241 unsigned int b;
242 int i;
243
244 switch (txt_flg[0])
245 { case '+': case '-': break;
246 default:
247 fprintf(stderr,"%s: last argument must be +flag or -flag\n",__progname);
248 exit(1);
249 break;
250 }
251 for (i=0;flagbits[i].name;i++) if (!strcmp(flagbits[i].name,txt_flg+1)) break;
252 if (! flagbits[i].name)
253 { fprintf(stderr,"%s: unrecognized flag bit `%s'\n",__progname,txt_flg+1);
254 exit(1);
255 }
256 b = flagbits[i].bit;
257 open_dev(O_RDWR);
258 if (ioctl(devfd,SRT_GFLAGS,&f) < 0)
259 { fprintf(stderr,"%s: can't get flags: %s\n",__progname,strerror(errno));
260 exit(1);
261 }
262 if (txt_flg[0] == '+') f |= b; else f &= ~b;
263 if (ioctl(devfd,SRT_SFLAGS,&f) < 0)
264 { fprintf(stderr,"%s: can't set flags: %s\n",__progname,strerror(errno));
265 exit(1);
266 }
267}
268
269static void do_debug(void)
270{
271 void *vp;
272
273 open_dev(O_RDWR);
274 vp = 0;
275 if (ioctl(devfd,SRT_DEBUG,&vp) < 0)
276 { fprintf(stderr,"%s: can't SRT_DEBUG: %s\n",__progname,strerror(errno));
277 exit(1);
278 }
279}
280
281int main(int ac, char **av)
282{
283 handleargs(ac,av);
284 switch (action)
285 { case ACT_QUERYALL:
286 do_query(-1);
287 break;
288 case ACT_QUERYONE:
289 do_query(atoi(txt_n));
290 break;
291 case ACT_DEL:
292 do_del(atoi(txt_n));
293 break;
294 case ACT_ADD:
295 do_set(-1);
296 break;
297 case ACT_SET:
298 do_set(atoi(txt_n));
299 break;
300 case ACT_FLAGS:
301 do_flags();
302 break;
303 case ACT_SFLAG:
304 do_sflag();
305 break;
306 case ACT_DEBUG:
307 do_debug();
308 break;
309 default:
310 abort();
311 break;
312 }
313 exit(0);
314}
