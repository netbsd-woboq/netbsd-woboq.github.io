27/*
28 * expand - expand wildcard filename specifications
29 *
30 * Usage:
31 * int expand(spec, buffer, bufsize);
32 * char *spec, **buffer;
33 * int bufsize;
34 *
35 * Expand takes a file specification, and expands it into filenames
36 * by resolving the characters '*', '?', '[', ']', '{', '}' and '~'
37 * in the same manner as the shell. You provide "buffer", which is
38 * an array of char *'s, and you tell how big it is in bufsize.
39 * Expand will compute the corresponding filenames, and will fill up
40 * the entries of buffer with pointers to malloc'd strings.
41 *
42 * The value returned by expand is the number of filenames found. If
43 * this value is -1, then malloc failed to allocate a string. If the
44 * value is bufsize + 1, then too many names were found and you can try
45 * again with a bigger buffer.
46 *
47 * This routine was basically created from the csh sh.glob.c file with
48 * the following intended differences:
49 *
50 * Filenames are not sorted.
51 * All expanded filenames returned exist.
52 *
53 **********************************************************************
54 * HISTORY
55 * 13-Nov-85 Glenn Marcy (gm0w) at Carnegie-Mellon University
56 * Replaced a stat() with lstat() and changed glob() to only call
57 * matchdir() for directories.
58 *
59 * 20-Oct-85 Glenn Marcy (gm0w) at Carnegie-Mellon University
60 * Created from csh glob() function and 4.1 expand() function.
61 *
62 **********************************************************************
63 */

103int
104expand(char *spec, char **buffer, int bufsize)
105{
106 pathp = path = pathbuf;
107 *pathp = 0;
108 maxpathlen = sizeof(pathbuf) - 1;
109 lastpathp = &path[maxpathlen];
110 BUFFER = buffer;
111 BUFSIZE = bufsize;
112 bufcnt = 0;
113 if (setjmp(sjbuf) == 0)
114 glob(spec);
115 return (bufcnt);
116}

118static void
119glob(char *as)
120{
121 char *cs;
122 char *spathp, *oldcs;
123 struct stat stb;
124
125 spathp = pathp;
126 cs = as;
127 if (*cs == '~' && pathp == path) {
128 if (addpath('~'))
129 goto endit;
130 for (cs++; isalnum((unsigned char)*cs) || *cs == '_' || *cs == '-';)
131 if (addpath(*cs++))
132 goto endit;
133 if (!*cs || *cs == '/') {
134 if (pathp != path + 1) {
135 *pathp = 0;
136 if (gethdir(path + 1, maxpathlen))
137 goto endit;
138 strlcpy(path, path + 1, maxpathlen);
139 } else
140 strlcpy(path, getenv("HOME"), maxpathlen);
141 pathp = path + strlen(path);
142 }
143 }
144 while (*cs == 0 || strchr(globchars, *cs) == 0) {
145 if (*cs == 0) {
146 if (lstat(fixit(path), &stb) >= 0)
147 addone(path, "");
148 goto endit;
149 }
150 if (addpath(*cs++))
151 goto endit;
152 }
153 oldcs = cs;
154 while (cs > as && *cs != '/')
155 cs--, pathp--;
156 if (*cs == '/')
157 cs++, pathp++;
158 *pathp = 0;
159 if (*oldcs == '{') {
160 execbrc(cs, NULL);
161 return;
162 }
163 /* this should not be an lstat */
164 if (stat(fixit(path), &stb) >= 0 && S_ISDIR(stb.st_mode))
165 matchdir(cs);
166endit:
167 pathp = spathp;
168 *pathp = 0;
169 return;
170}

172static void
173matchdir(char *pattern)
174{
175#ifdef HAS_POSIX_DIR
176 struct dirent *dp;
177#else
178 struct direct *dp;
179#endif
180 DIR *dirp;
181
182 dirp = opendir(fixit(path));
183 if (dirp == NULL)
184 return;
185 while ((dp = readdir(dirp)) != NULL) {
186#if defined(HAS_POSIX_DIR) && !defined(__SVR4) && !defined(__CYGWIN__)
187 if (dp->d_fileno == 0)
188 continue;
189#else
190 if (dp->d_ino == 0)
191 continue;
192#endif
193 if (match(dp->d_name, pattern))
194 addone(path, dp->d_name);
195 }
196 closedir(dirp);
197 return;
198}

200static int
201execbrc(char *p, char *s)
202{
203 char restbuf[MAXPATHLEN + 1];
204 char *pe, *pm, *pl;
205 int brclev = 0;
206 char *lm, savec, *spathp;
207
208 for (lm = restbuf; *p != '{'; *lm++ = *p++)
209 continue;
210 for (pe = ++p; *pe; pe++)
211 switch (*pe) {
212 case '{':
213 brclev++;
214 continue;
215 case '}':
216 if (brclev == 0)
217 goto pend;
218 brclev--;
219 continue;
220 case '[':
221 for (pe++; *pe && *pe != ']'; pe++)
222 continue;
223 if (!*pe)
224 break;
225 continue;
226 }
227pend:
228 if (brclev || !*pe)
229 return (0);
230 for (pl = pm = p; pm <= pe; pm++)
231 switch (*pm & 0177) {
232 case '{':
233 brclev++;
234 continue;
235 case '}':
236 if (brclev) {
237 brclev--;
238 continue;
239 }
240 goto doit;
241 case ',':
242 if (brclev)
243 continue;
244 doit:
245 savec = *pm;
246 *pm = 0;
247 snprintf(lm, sizeof(restbuf) - (lm - restbuf),
248 "%s%s", pl, pe + 1);
249 *pm = savec;
250 if (s == 0) {
251 spathp = pathp;
252 glob(restbuf);
253 pathp = spathp;
254 *pathp = 0;
255 } else if (amatch(s, restbuf))
256 return (1);
257 pl = pm + 1;
258 continue;
259
260 case '[':
261 for (pm++; *pm && *pm != ']'; pm++)
262 continue;
263 if (!*pm)
264 break;
265 continue;
266 }
267 return (0);
268}

270static int
271match(char *s, char *p)
272{
273 int c;
274 char *sentp;
275
276 if (*s == '.' && *p != '.')
277 return (0);
278 sentp = entp;
279 entp = s;
280 c = amatch(s, p);
281 entp = sentp;
282 return (c);
283}

285static int
286amatch(char *s, char *p)
287{
288 int scc;
289 int ok, lc;
290 char *spathp;
291 struct stat stb;
292 int c, cc;
293
294 for (;;) {
295 scc = *s++ & 0177;
296 switch (c = *p++) {
297 case '{':
298 return (execbrc(p - 1, s - 1));
299 case '[':
300 ok = 0;
301 lc = 077777;
302 while ((cc = *p++) != 0) {
303 if (cc == ']') {
304 if (ok)
305 break;
306 return (0);
307 }
308 if (cc == '-') {
309 if (lc <= scc && scc <= *p++)
310 ok++;
311 } else if (scc == (lc = cc))
312 ok++;
313 }
314 if (cc == 0)
315 return (0);
316 continue;
317 case '*':
318 if (!*p)
319 return (1);
320 if (*p == '/') {
321 p++;
322 goto slash;
323 }
324 for (s--; *s; s++)
325 if (amatch(s, p))
326 return (1);
327 return (0);
328 case 0:
329 return (scc == 0);
330 default:
331 if (c != scc)
332 return (0);
333 continue;
334 case '?':
335 if (scc == 0)
336 return (0);
337 continue;
338 case '/':
339 if (scc)
340 return (0);
341 slash:
342 s = entp;
343 spathp = pathp;
344 while (*s)
345 if (addpath(*s++))
346 goto pathovfl;
347 if (addpath('/'))
348 goto pathovfl;
349 if (stat(fixit(path), &stb) >= 0 &&
350 (stb.st_mode & S_IFMT) == S_IFDIR) {
351 if (*p == 0)
352 addone(path, "");
353 else
354 glob(p);
355 }
356 pathovfl:
357 pathp = spathp;
358 *pathp = 0;
359 return (0);
360 }
361 }
362}

364static void
365addone(char *s1, const char *s2)
366{
367 char *ep;
368
369 if (bufcnt >= BUFSIZE) {
370 bufcnt = BUFSIZE + 1;
371 longjmp(sjbuf, 1);
372 }
373 ep = malloc(strlen(s1) + strlen(s2) + 1);
374 if (ep == 0) {
375 bufcnt = -1;
376 longjmp(sjbuf, 1);
377 }
378 BUFFER[bufcnt++] = ep;
379 while (*s1)
380 *ep++ = *s1++;
381 while ((*ep++ = *s2++) != '\0')
382 continue;
383}
384
385static int
386addpath(char c)
387{
388 if (pathp >= lastpathp)
389 return (1);
390 *pathp++ = c;
391 *pathp = 0;
392 return (0);
393}

395static int
396gethdir(char *home, size_t homelen)
397{
398 struct passwd *pp = getpwnam(home);
399
400 if (pp == 0)
401 return (1);
402 strlcpy(home, pp->pw_dir, homelen);
403 return (0);
404}
