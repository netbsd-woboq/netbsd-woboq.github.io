55static int opened = 0;
56
57void
58logopen(char *program)
59{
60 if (opened)
61 return;
62 openlog(program, LOG_PID, LOG_DAEMON);
63 opened++;
64}
65
66static void
67vfmtbuf(int f, char *buf, size_t bufsiz, const char *fmt, va_list ap)
68{
69 char hostname[MAXHOSTNAMELEN];
70 char xbuf[STRINGLENGTH], ybuf[STRINGLENGTH];
71 gethostname(hostname, sizeof(hostname));
72 snprintf(ybuf, sizeof(ybuf), "SUP@%s%s ", hostname, f ? ":" : "");
73 vsnprintf(xbuf, sizeof(xbuf), fmt, ap);
74 snprintf(buf, bufsiz, "%s%s", ybuf, xbuf);
75 return;
76}

78
79void
80logquit(int retval, const char *fmt, ...)
81{
82 char buf[STRINGLENGTH];
83 va_list ap;
84
85 va_start(ap, fmt);
86 vfmtbuf(1, buf, sizeof(buf), fmt, ap);
87 va_end(ap);
88 if (opened) {
89 syslog(LOG_ERR, "%s", buf);
90 closelog();
91 exit(retval);
92 }
93 quit(retval, "SUP: %s\n", buf);
94}
95
96void
97logerr(const char *fmt, ...)
98{
99 char buf[STRINGLENGTH];
100 va_list ap;
101
102 va_start(ap, fmt);
103 vfmtbuf(1, buf, sizeof(buf), fmt, ap);
104 va_end(ap);
105 if (opened) {
106 syslog(LOG_ERR, "%s", buf);
107 return;
108 }
109 fprintf(stderr, "SUP: %s\n", buf);
110 (void) fflush(stderr);
111}
112
113void
114loginfo(const char *fmt, ...)
115{
116 char buf[STRINGLENGTH];
117 va_list ap;
118
119 va_start(ap, fmt);
120 vfmtbuf(0, buf, sizeof(buf), fmt, ap);
121 va_end(ap);
122 if (opened) {
123 syslog(LOG_INFO, "%s", buf);
124 return;
125 }
126 printf("%s\n", buf);
127 (void) fflush(stdout);
128}
