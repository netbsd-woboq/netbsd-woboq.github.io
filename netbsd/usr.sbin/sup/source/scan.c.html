197static void
198passdelim(char **ptr, char delim)
199{ /* skip over delimiter */
200 *ptr = skipover(*ptr, " \t");
201 if (_argbreak != delim && **ptr == delim) {
202 (*ptr)++;
203 *ptr = skipover(*ptr, " \t");
204 }
205}

207static char *
208parserelease(TREELIST ** tlp, char *relname, char *args)
209{
210 TREELIST *tl;
211 char *arg;
212 OPTION option;
213 int opno;
214 char *nextrel;
215
216 tl = malloc(sizeof(TREELIST));
217 if ((*tlp = tl) == NULL)
218 goaway("Couldn't allocate TREELIST");
219 tl->TLnext = NULL;
220 tl->TLname = estrdup(relname);
221 tl->TLprefix = NULL;
222 tl->TLlist = NULL;
223 tl->TLscan = NULL;
224 tl->TLhost = NULL;
225 nextrel = NULL;
226 args = skipover(args, " \t");
227 while (*(arg = nxtarg(&args, " \t="))) {
228 for (opno = 0; options[opno] != NULL; opno++)
229 if (strcmp(arg, options[opno]) == 0)
230 break;
231 if (options[opno] == NULL)
232 goaway("Invalid release option %s for release %s",
233 arg, relname);
234 option = (OPTION) opno;
235 switch (option) {
236 case ONEXT:
237 passdelim(&args, '=');
238 arg = nxtarg(&args, " \t");
239 if (nextrel)
240 free(nextrel);
241 nextrel = estrdup(arg);
242 break;
243 case OPREFIX:
244 passdelim(&args, '=');
245 arg = nxtarg(&args, " \t");
246 tl->TLprefix = estrdup(arg);
247 break;
248 case OLIST:
249 passdelim(&args, '=');
250 arg = nxtarg(&args, " \t");
251 tl->TLlist = estrdup(arg);
252 break;
253 case OSCAN:
254 passdelim(&args, '=');
255 arg = nxtarg(&args, " \t");
256 tl->TLscan = estrdup(arg);
257 break;
258 case OHOST:
259 passdelim(&args, '=');
260 arg = nxtarg(&args, " \t");
261 tl->TLhost = estrdup(arg);
262 break;
263 }
264 }
265 return (nextrel);
266}

268int
269getrelease(char *release)
270{
271 TREELIST *tl;
272 char buf[STRINGLENGTH];
273 char *p, *q;
274 int rewound;
275 char *frelease = NULL;
276 FILE *f;
277
278 if (release == NULL)
279 frelease = release = estrdup(DEFRELEASE);
280 listTL = NULL;
281
282 (void) sprintf(buf, FILERELEASES, collname);
283 f = fopen(buf, "r");
284 if (f != NULL) {
285 rewound = TRUE;
286 for (;;) {
287 p = fgets(buf, sizeof(buf), f);
288 if (p == NULL) {
289 if (rewound)
290 break;
291 rewind(f);
292 rewound = TRUE;
293 continue;
294 }
295 q = strchr(p, '\n');
296 if (q)
297 *q = 0;
298 if (strchr("#;:", *p))
299 continue;
300 q = nxtarg(&p, " \t");
301 if (strcmp(q, release) != 0)
302 continue;
303 release = parserelease(&tl, release, p);
304 if (tl->TLprefix == NULL)
305 tl->TLprefix = prefix;
306 else if (chdir(tl->TLprefix) < 0) {
307 free(tl);
308 fclose(f);
309 if (frelease)
310 free(frelease);
311 return (FALSE);
312 } else
313 if (chdir(basedir) < 0)
314 goaway("Can't chdir to %s (%s)",
315 basedir, strerror(errno));
316 tl->TLnext = listTL;
317 listTL = tl;
318 if (release == NULL)
319 break;
320 rewound = FALSE;
321 }
322 (void) fclose(f);
323 }
324 if (release == NULL) {
325 if (frelease)
326 free(frelease);
327 return (TRUE);
328 }
329 if (strcmp(release, DEFRELEASE) != 0) {
330 if (frelease)
331 free(frelease);
332 return (FALSE);
333 }
334 (void) parserelease(&tl, release, "");
335 tl->TLprefix = prefix;
336 tl->TLnext = listTL;
337 listTL = tl;
338 if (frelease)
339 free(frelease);
340 return (TRUE);
341}

343void
344makescanlists(void)
345{
346 TREELIST *tl;
347 char buf[STRINGLENGTH];
348 char *p, *q;
349 FILE *f;
350 char *saveprefix = prefix;
351 int count = 0;
352
353 (void) sprintf(buf, FILERELEASES, collname);
354 f = fopen(buf, "r");
355 if (f != NULL) {
356 while ((p = fgets(buf, sizeof(buf), f)) != NULL) {
357 q = strchr(p, '\n');
358 if (q)
359 *q = 0;
360 if (strchr("#;:", *p))
361 continue;
362 q = nxtarg(&p, " \t");
363 (void) parserelease(&tl, q, p);
364 if ((prefix = tl->TLprefix) == NULL)
365 prefix = saveprefix;
366 if (prefix != NULL) {
367 if (chdir(prefix) < 0)
368 goaway("Can't chdir to %s (%s)",
369 prefix, strerror(errno));
370 if (chdir(basedir) < 0)
371 goaway("Can't chdir to %s (%s)",
372 basedir, strerror(errno));
373 }
374 makescan(tl->TLlist, tl->TLscan);
375 free(tl);
376 count++;
377 }
378 (void) fclose(f);
379 }
380 if (count == 0)
381 makescan(NULL, NULL);
382}

384static int
385scanone(TREE * t, void *v __unused)
386{
387 TREE *newt;
388
389 if (newonly && (t->Tflags & FNEW) == 0)
390 return (SCMOK);
391 newt = Tinsert(&listT, t->Tname, FALSE);
392 if (newt == NULL)
393 return (SCMOK);
394 newt->Tmode = t->Tmode;
395 newt->Tflags = t->Tflags;
396 newt->Tmtime = t->Tmtime;
397 return (SCMOK);
398}

400void
401getscanlists(void)
402{
403 TREELIST *tl, *stl;
404
405 stl = listTL;
406 listTL = NULL;
407 while ((tl = stl) != NULL) {
408 prefix = tl->TLprefix;
409 getscan(tl->TLlist, tl->TLscan);
410 tl->TLtree = listT;
411 stl = tl->TLnext;
412 tl->TLnext = listTL;
413 listTL = tl;
414 }
415 listT = NULL;
416 for (tl = listTL; tl != NULL; tl = tl->TLnext)
417 (void) Tprocess(tl->TLtree, scanone, NULL);
418}

440static void
441doscan(char *listfile)
442{
443 char buf[STRINGLENGTH];
444
445 upgT = NULL;
446 flagsT = NULL;
447 omitT = NULL;
448 omanyT = NULL;
449 execT = NULL;
450 symT = NULL;
451 rsymT = NULL;
452 if (listfile == NULL)
453 listfile = FILELISTDEF;
454 (void) sprintf(buf, FILELIST, collname, listfile);

456 (void) Tprocess(upgT, listone, NULL); /* build list of files

460 Tfree(&flagsT);
461 Tfree(&omitT);
462 Tfree(&omanyT);

464 Tfree(&symT);
465 Tfree(&rsymT);

468static void
469readlistfile(char *fname)
470{
471 char buf[STRINGLENGTH + MAXPATHLEN * 4 + 1], *p;
472 char *q, *r;
473 FILE *f;
474 int ltn, n, i, flags;
475 TREE **t = NULL;
476 LISTTYPE lt;
477 char *speclist[SPECNUMBER];
478
479 f = fopen(fname, "r");
480 if (f == NULL)
481 goaway("Can't read list file %s", fname);
482 cdprefix(prefix);
483 while ((p = fgets(buf, sizeof(buf), f)) != NULL) {
484 if ((q = strchr(p, '\n')) != NULL)
485 *q = '\0';
486 if (strchr("#;:", *p))
487 continue;
488 q = nxtarg(&p, " \t");
489 if (*q == '\0')
490 continue;
491 for (ltn = 0; ltname[ltn] && strcmp(q, ltname[ltn]) != 0; ltn++);
492 if (ltname[ltn] == NULL)
493 goaway("Invalid list file keyword %s", q);
494 lt = (LISTTYPE) ltn;
495 flags = 0;
496 switch (lt) {
497 case LUPGRADE:
498 t = &upgT;
499 break;
500 case LBACKUP:
501 t = &flagsT;
502 flags = FBACKUP;
503 break;
504 case LNOACCT:
505 t = &flagsT;
506 flags = FNOACCT;
507 break;
508 case LSYMLINK:
509 t = &symT;
510 break;
511 case LRSYMLINK:
512 t = &rsymT;
513 break;
514 case LOMIT:
515 t = &omitT;
516 break;
517 case LOMITANY:
518 t = &omanyT;
519 break;
520 case LALWAYS:
521 t = &upgT;
522 flags = FALWAYS;
523 break;
524 case LINCLUDE:
525 while (*(q = nxtarg(&p, " \t"))) {
526 cdprefix(NULL);
527 n = expand(q, speclist, SPECNUMBER);
528 for (i = 0; i < n && i < SPECNUMBER; i++) {
529 readlistfile(speclist[i]);
530 cdprefix(NULL);
531 free(speclist[i]);
532 }
533 cdprefix(prefix);
534 }
535 continue;
536 case LEXECUTE:
537 r = p = q = skipover(p, " \t");
538 do {
539 q = p = skipto(p, " \t(");
540 p = skipover(p, " \t");
541 } while (*p != '(' && *p != '\0');
542 if (*p++ == '(') {
543 *q = '\0';
544 do {
545 q = nxtarg(&p, " \t)");
546 if (*q == 0)
547 _argbreak = ')';
548 else
549 expTinsert(q, &execT, 0, r);
550 } while (_argbreak != ')');
551 continue;
552 }
553 /* FALLTHROUGH */
554 default:
555 goaway("Error in handling list file keyword %d", ltn);
556 }
557 while (*(q = nxtarg(&p, " \t"))) {
558 if (lt == LOMITANY)
559 (void) Tinsert(t, q, FALSE);
560 else
561 expTinsert(q, t, flags, NULL);
562 }
563 }
564 (void) fclose(f);
565}

567static void
568expTinsert(char *p, TREE ** t, int flags, char *exec)
569{
570 int n, i;
571 TREE *newt;
572 char *speclist[SPECNUMBER];
573 char buf[STRINGLENGTH];
574
575 n = expand(p, speclist, SPECNUMBER);
576 for (i = 0; i < n && i < SPECNUMBER; i++) {
577 newt = Tinsert(t, speclist[i], TRUE);
578 newt->Tflags |= flags;
579 if (exec) {
580 (void) sprintf(buf, exec, speclist[i]);
581 (void) Tinsert(&newt->Texec, buf, FALSE);
582 }
583 free(speclist[i]);
584 }
585}

594static void
595listentry(char *name, char *fullname, char *updir, int always)
596{
597 struct stat statbuf;
598 int linkcount = 0;
599
600 if (Tlookup(refuseT, fullname))
601 return;
602 if (!always) {
603 if (Tsearch(omitT, fullname))
604 return;
605 if (Tprocess(omanyT, omitanyone, fullname) != SCMOK)
606 return;
607 }
608 if (lstat(name, &statbuf) < 0)
609 return;
610 if (S_ISLNK(statbuf.st_mode)) {
611 if (Tsearch(symT, fullname)) {
612 listname(fullname, &statbuf);
613 return;
614 }
615 if (Tlookup(rsymT, fullname)) {
616 listname(fullname, &statbuf);
617 return;
618 }
619 if (updir)
620 linkcount++;
621 if (stat(name, &statbuf) < 0)
622 return;
623 }
624 if (S_ISDIR(statbuf.st_mode)) {
625 if (access(name, R_OK | X_OK) < 0)
626 return;
627 if (chdir(name) < 0)
628 return;
629 listname(fullname, &statbuf);
630 if (trace) {
631 printf("Scanning directory %s\n", fullname);
632 (void) fflush(stdout);
633 }
634 listdir(fullname, always);
635 if (updir == 0 || linkcount) {
636 if (chdir(basedir) < 0)
637 goaway("Can't chdir to %s (%s)",
638 basedir, strerror(errno));
639 if (prefix)
640 if (chdir(prefix) < 0)
641 goaway("Can't chdir to %s (%s)",
642 prefix, strerror(errno));
643 if (updir && *updir)
644 if (chdir(updir) < 0)
645 goaway("Can't chdir to %s (%s)",
646 updir, strerror(errno));
647 } else
648 if (chdir("..") < 0)
649 goaway("Can't chdir to %s (%s)",
650 "..", strerror(errno));
651 return;
652 }
653 if (access(name, R_OK) < 0)
654 return;
655 listname(fullname, &statbuf);
656}

658static void
659listname(char *name, struct stat * st)
660{
661 TREE *t, *ts;
662 int new;
663 TREELIST *tl;
664
665 new = st->st_ctime > lasttime;
666 if (newonly && !new) {
667 for (tl = listTL; tl != NULL; tl = tl->TLnext)
668 if ((ts = Tsearch(tl->TLtree, name)) != NULL)
669 ts->Tflags &= ~FNEW;
670 return;
671 }
672 t = Tinsert(&listT, name, FALSE);
673 if (t == NULL)
674 return;
675 t->Tmode = st->st_mode;
676 t->Tctime = st->st_ctime;
677 t->Tmtime = st->st_mtime;
678 if (new)
679 t->Tflags |= FNEW;
680 if ((ts = Tsearch(flagsT, name)) != NULL)
681 t->Tflags |= ts->Tflags;
682 if ((ts = Tsearch(execT, name)) != NULL) {
683 t->Texec = ts->Texec;
684 ts->Texec = NULL;
685 }
686}

688static void
689listdir(char *name, int always)
690{ /* expand directory */
691#ifdef HAS_POSIX_DIR
692 struct dirent *dentry;
693#else
694 struct direct *dentry;
695#endif
696 DIR *dirp;
697 char newname[STRINGLENGTH], filename[STRINGLENGTH];
698 char *p, *newp;
699
700 dirp = opendir(".");
701 if (dirp == 0)
702 return; /* unreadable: probably protected */
703
704 p = name; /* punt leading ./ and trailing / */
705 newp = newname;
706 if (p[0] == '.' && p[1] == '/') {
707 p += 2;
708 while (*p == '/')
709 p++;
710 }
711 while ((*newp++ = *p++) != '\0'); /* copy string */
712 --newp; /* trailing null */
713 while (newp > newname && newp[-1] == '/')
714 --newp; /* trailing / */
715 *newp = 0;
716 if (strcmp(newname, ".") == 0)
717 newname[0] = 0; /* "." ==> "" */
718
719 while ((dentry = readdir(dirp)) != NULL) {
720 if (dentry->d_ino == 0)
721 continue;
722 if (strcmp(dentry->d_name, ".") == 0)
723 continue;
724 if (strcmp(dentry->d_name, "..") == 0)
725 continue;
726 if (*newname) {
727 (void)snprintf(filename, sizeof(filename), "%s/%s",
728 newname, dentry->d_name);
729 } else {
730 (void)strncpy(filename, dentry->d_name,
731 sizeof(filename) - 1);
732 filename[sizeof(filename) - 1] = '\0';
733 }
734 listentry(dentry->d_name, filename, newname, always);
735 }
736 closedir(dirp);
737}

748static int
749anyglob(char *pattern, char *match)
750{
751 char *p, *m;
752 char *pb, *pe;
753
754 p = pattern;
755 m = match;
756 while (*m && *p == *m) {
757 p++;
758 m++;
759 }
760 if (*p == '\0' && *m == '\0')
761 return (TRUE);
762 switch (*p++) {
763 case '*':
764 for (;;) {
765 if (*p == '\0')
766 return (TRUE);
767 if (*m == '\0')
768 return (*p == '\0');
769 if (anyglob(p, ++m))
770 return (TRUE);
771 }
772 case '?':
773 return (anyglob(p, ++m));
774 case '[':
775 pb = p;
776 while (*(++p) != ']')
777 if (*p == '\0')
778 return (FALSE);
779 pe = p;
780 for (p = pb + 1; p != pe; p++) {
781 switch (*p) {
782 case '-':
783 if (p == pb && *m == '-') {
784 p = pe + 1;
785 return (anyglob(p, ++m));
786 }
787 if (p == pb)
788 continue;
789 if ((p + 1) == pe)
790 return (FALSE);
791 if (*m > *(p - 1) &&
792 *m <= *(p + 1)) {
793 p = pe + 1;
794 return (anyglob(p, ++m));
795 }
796 continue;
797 default:
798 if (*m == *p) {
799 p = pe + 1;
800 return (anyglob(p, ++m));
801 }
802 }
803 }
804 return (FALSE);
805 default:
806 return (FALSE);
807 }
808}

813static int
814getscanfile(char *scanfile)
815{
816 char buf[STRINGLENGTH];
817#ifdef HAS_VIS
818 char fname[MAXPATHLEN];
819#else
820 char *fname;
821#endif
822 struct stat sbuf;
823 FILE *f;
824 TREE ts;
825 char *p, *q;
826 TREE *tmp, *t = NULL;
827 int notwanted;
828 TREELIST *tl;
829
830 if (scanfile == NULL)
831 scanfile = FILESCANDEF;
832 (void) sprintf(buf, FILESCAN, collname, scanfile);
833 if (stat(buf, &sbuf) < 0)
834 return (FALSE);
835 if ((f = fopen(buf, "r")) == NULL)
836 return (FALSE);
837 if ((p = fgets(buf, sizeof(buf), f)) == NULL) {
838 (void) fclose(f);
839 return (FALSE);
840 }
841 if ((q = strchr(p, '\n')) != NULL)
842 *q = '\0';
843 if (*p++ != 'V') {
844 (void) fclose(f);
845 return (FALSE);
846 }
847 if (atoi(p) != SCANVERSION) {
848 (void) fclose(f);
849 return (FALSE);
850 }
851 scantime = sbuf.st_mtime; /* upgrade time is time of supscan,
852 * i.e. time of creation of scanfile */
853 if (newonly && scantime == lasttime) {
854 (void) fclose(f);
855 return (TRUE);
856 }
857 notwanted = FALSE;
858 while ((p = fgets(buf, sizeof(buf), f)) != NULL) {
859 q = strchr(p, '\n');
860 if (q)
861 *q = 0;
862 ts.Tflags = 0;
863 if (*p == 'X') {
864 if (notwanted)
865 continue;
866 if (t == NULL)
867 goaway("scanfile format inconsistent");
868 (void) Tinsert(&t->Texec, ++p, FALSE);
869 continue;
870 }
871 notwanted = FALSE;
872 if (*p == 'B') {
873 p++;
874 ts.Tflags |= FBACKUP;
875 }
876 if (*p == 'N') {
877 p++;
878 ts.Tflags |= FNOACCT;
879 }
880 if ((q = strchr(p, ' ')) == NULL)
881 goaway("scanfile format inconsistent");
882 *q++ = '\0';
883 ts.Tmode = atoo(p);
884 p = q;
885 if ((q = strchr(p, ' ')) == NULL)
886 goaway("scanfile format inconsistent");
887 *q++ = '\0';
888 ts.Tctime = atoi(p);
889 p = q;
890 if ((q = strchr(p, ' ')) == NULL)
891 goaway("scanfile format inconsistent");
892 *q++ = 0;
893 ts.Tmtime = atoi(p);
894#ifdef HAS_VIS
895 (void) strunvis(fname, q);
896#else
897 fname = q;
898#endif
899 if (ts.Tctime > lasttime)
900 ts.Tflags |= FNEW;
901 else if (newonly) {
902 for (tl = listTL; tl != NULL; tl = tl->TLnext)
903 if ((tmp = Tsearch(tl->TLtree, fname)) != NULL)
904 tmp->Tflags &= ~FNEW;
905 notwanted = TRUE;
906 continue;
907 }
908 if (Tlookup(refuseT, fname)) {
909 notwanted = TRUE;
910 continue;
911 }
912 t = Tinsert(&listT, fname, TRUE);
913 t->Tmode = ts.Tmode;
914 t->Tflags = ts.Tflags;
915 t->Tctime = ts.Tctime;
916 t->Tmtime = ts.Tmtime;
917 }
918 (void) fclose(f);
919 return (TRUE);
920}

943static void
944makescanfile(char *scanfile)
945{
946 char tname[STRINGLENGTH], fname[STRINGLENGTH];
947 struct timeval tbuf[2];
948 FILE *scanF; /* output file for scanned file list */
949
950 if (scanfile == NULL)
951 scanfile = FILESCANDEF;
952 (void) sprintf(fname, FILESCAN, collname, scanfile);
953 (void) sprintf(tname, "%s.temp", fname);
954 scanF = fopen(tname, "w");
955 if (scanF == NULL)
956 goto out;
957 if (fprintf(scanF, "V%d\n", SCANVERSION) < 0)
958 goto closeout;
959 if (Tprocess(listT, recordone, scanF) != SCMOK)
960 goto closeout;
961 if (fclose(scanF) != 0)
962 goto out;
963 if (rename(tname, fname) < 0) {
964 (void)unlink(tname);
965 goaway("Can't change %s to %s", tname, fname);
966 }
967 tbuf[0].tv_sec = time(NULL);
968 tbuf[0].tv_usec = 0;
969 tbuf[1].tv_sec = scantime;
970 tbuf[1].tv_usec = 0;
971 (void) utimes(fname, tbuf);
972 return;
973closeout:
974 (void) fclose(scanF);
975out:
976 goaway("Can't write scan file temp %s for %s", tname, collname);
977}

979static int
980recordone(TREE * t, void *v)
981{
982 FILE *scanF = v;
983#ifdef HAS_VIS
984 char fname[MAXPATHLEN * 4 + 1];
985 strvis(fname, t->Tname, VIS_WHITE);
986#else
987 char *fname = t->Tname;
988#endif
989
990 if (t->Tflags & FBACKUP)
991 if (fprintf(scanF, "B") < 0)
992 return SCMERR;
993 if (t->Tflags & FNOACCT)
994 if (fprintf(scanF, "N") < 0)
995 return SCMERR;
996 if (fprintf(scanF, "%o %d %d %s\n",
997 t->Tmode, t->Tctime, t->Tmtime, fname) < 0)
998 return SCMERR;
999 return Tprocess(t->Texec, recordexec, scanF);
1000}

1017void
1018cdprefix(char *prefix)
1019{
1020 static char *curprefix = NULL;
1021
1022 if (curprefix == NULL) {
1023 if (prefix == NULL)
1024 return;
1025 if (chdir(prefix) < 0)
1026 goaway("Can't chdir to %s (%s)",
1027 prefix, strerror(errno));
1028 curprefix = prefix;
1029 return;
1030 }
1031 if (prefix == NULL) {
1032 if (chdir(basedir) < 0)
1033 goaway("Can't chdir to %s (%s)",
1034 basedir, strerror(errno));
1035 curprefix = NULL;
1036 return;
1037 }
1038 if (prefix == curprefix)
1039 return;
1040 if (strcmp(prefix, curprefix) == 0) {
1041 curprefix = prefix;
1042 return;
1043 }
1044 if (chdir(basedir) < 0)
1045 goaway("Can't chdir to %s (%s)",
1046 basedir, strerror(errno));
1047 if (chdir(prefix) < 0)
1048 goaway("Can't chdir to %s (%s)",
1049 prefix, strerror(errno));
1050 curprefix = prefix;
1051}
