27/*
28 * SUP Communication Module for 4.3 BSD
29 *
30 * SUP COMMUNICATION MODULE SPECIFICATIONS:
31 *
32 * IN THIS MODULE:
33 *
34 * CONNECTION ROUTINES
35 *
36 * FOR SERVER
37 * servicesetup (port) establish TCP port connection
38 * char *port; name of service
39 * service () accept TCP port connection
40 * servicekill () close TCP port in use by another process
41 * serviceprep () close temp ports used to make connection
42 * serviceend () close TCP port
43 *
44 * FOR CLIENT
45 * request (port,hostname,retry) establish TCP port connection
46 * char *port,*hostname; name of service and host
47 * int retry; true if retries should be used
48 * requestend () close TCP port
49 *
50 * HOST NAME CHECKING
51 * p = remotehost () remote host name (if known)
52 * char *p;
53 * i = samehost () whether remote host is also this host
54 * int i;
55 * i = matchhost (name) whether remote host is same as name

57 * char *name;

230int
231servicesetup(char *server, int af)
232{ /* listen for clients */
233 struct addrinfo hints, *res0, *res;
234 char port[NI_MAXSERV];
235 int error;
236 const char *cause = "unknown";
237 int one = 1;
238
239 memset(&hints, 0, sizeof(hints));
240 hints.ai_family = af;
241 hints.ai_socktype = SOCK_STREAM;
242 hints.ai_flags = AI_PASSIVE;
243 error = getaddrinfo(NULL, server, &hints, &res0);
244 if (error) {
245 /* retry with precompiled knowledge */
246 if (strcmp(server, FILEPORT) == 0)
247 snprintf(port, sizeof(port), "%u", FILEPORTNUM);
248 else if (strcmp(server, DEBUGFPORT) == 0)
249 snprintf(port, sizeof(port), "%u", DEBUGFPORTNUM);
250 else
251 port[0] = '\0';
252 if (port[0])
253 error = getaddrinfo(NULL, port, &hints, &res0);
254 if (error)
255 return (scmerr(-1, "%s: %s", server,
256 gai_strerror(error)));
257 }
258 for (res = res0; res; res = res->ai_next) {
259 sock = socket(res->ai_family, res->ai_socktype,
260 res->ai_protocol);
261 if (sock < 0) {
262 cause = "socket";
263 continue;
264 }
265 if (setsockopt(sock, SOL_SOCKET, SO_REUSEADDR,
266 &one, sizeof(int)) < 0) {
267 cause = "setsockopt(SO_REUSEADDR)";
268 close(sock);
269 continue;
270 }
271 if (bind(sock, res->ai_addr, res->ai_addrlen) < 0) {
272 cause = "bind";
273 close(sock);
274 continue;
275 }
276 if (listen(sock, NCONNECTS) < 0) {
277 cause = "listen";
278 close(sock);
279 continue;
280 }
281 freeaddrinfo(res0);
282 return SCMOK;
283 }
284
285 freeaddrinfo(res0);
286 return (scmerr(errno, "%s", cause));
287}

289int
290service(void)
291{
292 struct sockaddr_storage from;
293 int x;
294 socklen_t len;
295
296 remotename = NULL;
297 len = sizeof(from);
298 do {
299 netfile = accept(sock, (struct sockaddr *)(void *)&from, &len);
300 } while (netfile < 0 && errno == EINTR);
301 if (netfile < 0)
302 return (scmerr(errno, "Can't accept connections"));
303 if (len > sizeof(remoteaddr)) {
304 close(netfile);
305 return (scmerr(errno, "Can't accept connections"));
306 }
307 memcpy(&remoteaddr, &from, len);
308 if (read(netfile, &x, sizeof(int)) != sizeof(int))
309 return (scmerr(errno, "Can't transmit data on connection"));
310 if (x == 0x01020304)
311 swapmode = 0;
312 else if (x == 0x04030201)
313 swapmode = 1;
314 else
315 return (scmerr(-1, "Unexpected byteswap mode %x", x));
316 setproctitle("Serving %s", remotehost());
317 return (SCMOK);
318}

362int
363dobackoff(int *t, int *b)
364{
365 struct timeval tt;
366 unsigned s;
367
368 if (*t == 0)
369 return (0);
370 s = *b * 30;
371 if (gettimeofday(&tt, NULL) >= 0)
372 s += ((uint32_t)tt.tv_usec >> 8) % s;
373 if (*b < 32)
374 *b <<= 1;
375 if (*t != -1) {
376 if (s > (unsigned) *t)
377 s = *t;
378 *t -= s;
379 }
380 if (!silent)
381 (void) scmerr(-1, "Will retry in %d seconds", s);
382 sleep(s);
383 return (1);
384}

386int
387request(char *server, char *hostname, int *retry)
388{ /* connect to server */
389 struct addrinfo hints, *res, *res0;
390 int error;
391 char port[NI_MAXSERV];
392 int backoff;
393 int x;
394
395 memset(&hints, 0, sizeof(hints));
396 hints.ai_family = PF_UNSPEC;
397 hints.ai_socktype = SOCK_STREAM;
398 error = getaddrinfo(hostname, server, &hints, &res0);
399 if (error) {
400 /* retry with precompiled knowledge */
401 if (strcmp(server, FILEPORT) == 0)
402 snprintf(port, sizeof(port), "%u", FILEPORTNUM);
403 else if (strcmp(server, DEBUGFPORT) == 0)
404 snprintf(port, sizeof(port), "%u", DEBUGFPORTNUM);
405 else
406 port[0] = '\0';
407 if (port[0])
408 error = getaddrinfo(hostname, port, &hints, &res0);
409 if (error)
410 return (scmerr(-1, "%s: %s", server,
411 gai_strerror(error)));
412 }
413 backoff = 1;
414 for (;;) {
415 netfile = -1;
416 for (res = res0; res; res = res->ai_next) {
417 if (res->ai_addrlen > sizeof(remoteaddr))
418 continue;
419 netfile = socket(res->ai_family, res->ai_socktype,
420 res->ai_protocol);
421 if (netfile < 0)
422 continue;
423 if (connect(netfile, res->ai_addr, res->ai_addrlen) < 0) {
424 close(netfile);
425 netfile = -1;
426 continue;
427 }
428 break;
429 }
430
431 if (netfile < 0) {
432 if (!dobackoff(retry, &backoff)) {
433 freeaddrinfo(res0);
434 return (SCMERR);
435 }
436 continue;
437 } else
438 break;
439 }
440
441 if (res == NULL) {
442 freeaddrinfo(res0);
443 return (SCMERR);
444 }
445 memcpy(&remoteaddr, res->ai_addr, res->ai_addrlen);
446 remotename = estrdup(hostname);
447 x = 0x01020304;
448 if (write(netfile, &x, sizeof(int)) == -1)
449 return (SCMERR);
450 swapmode = 0; /* swap only on server, not client */
451 freeaddrinfo(res0);
452 return (SCMOK);
453}

455int
456requestend(void)
457{ /* end connection to server */
458 (void) readflush();
459 if (netfile >= 0) {
460 (void) close(netfile);
461 netfile = -1;
462 }
463 if (remotename) {
464 free(remotename);
465 remotename = NULL;
466 }
467 return (SCMOK);
468}

473static char *
474myhost(void)
475{ /* find my host name */
476 struct hostent *h;
477 static char name[MAXHOSTNAMELEN + 1];
478
479 if (name[0] == '\0') {
480 if (gethostname(name, sizeof name) < 0)
481 return (NULL);
482 name[sizeof(name) - 1] = '\0';
483 if ((h = gethostbyname(name)) == NULL)
484 return (NULL);
485 (void) strcpy(name, h->h_name);
486 }
487 return (name);
488}

490const char *
491remotehost(void)
492{ /* remote host name (if known) */
493 char h1[NI_MAXHOST];
494
495 if (remotename == NULL) {
496 if (getnameinfo((struct sockaddr *)(void *)&remoteaddr,
497#ifdef BSD4_4
498 (socklen_t)remoteaddr.ss_len,
499#else
500 sizeof(struct sockaddr),
501#endif
502 h1, sizeof(h1), NULL, 0, 0))
503 return ("UNKNOWN");
504 remotename = estrdup(h1);
505 if (remotename == NULL)
506 return ("UNKNOWN");
507 }
508 return (remotename);
509}

511int
512thishost(char *host)
513{
514 struct hostent *h;
515 char *name;
516
517 if ((name = myhost()) == NULL)
518 logquit(1, "Can't find my host entry '%s'", myhost());
519 h = gethostbyname(host);
520 if (h == NULL)
521 return (0);
522 return (strcasecmp(name, h->h_name) == 0);
523}

533static int
534getifaddrs(struct ifaddrs **ifap)
535{
536 struct ifaddrs *ifa;
537 int nint;
538 int n;
539 char buf[10 * 1024];
540 struct ifconf ifc;
541 struct ifreq *ifr;
542 int s;
543
544 if ((s = socket (AF_INET, SOCK_DGRAM, 0)) == -1)
545 return -1;
546
547 ifc.ifc_len = sizeof(buf);
548 ifc.ifc_buf = buf;
549
550 if (ioctl(s, SIOCGIFCONF, &ifc) == -1) {
551 (void)close(s);
552 return -1;
553 }
554
555 (void)close(s);
556
557 if ((nint = ifc.ifc_len / sizeof(struct ifreq)) <= 0)
558 return 0;
559
560 if ((ifa = malloc((unsigned)nint * sizeof(struct ifaddrs))) == NULL)
561 return -1;
562
563 for (ifr = ifc.ifc_req, n = 0; n < nint; n++, ifr++) {
564 ifa[n].ifa_next = &ifa[n + 1];
565 ifa[n].ifa_addr = &ifa[n].ifa_addrspace;
566 (void)memcpy(ifa[n].ifa_addr, &ifr->ifr_addr,
567 sizeof(*ifa[n].ifa_addr));
568 }
569
570 ifa[nint - 1].ifa_next = NULL;
571 *ifap = ifa;
572 return nint;
573}

583int
584samehost(void)
585{ /* is remote host same as local host? */
586 struct ifaddrs *ifap, *ifa;
587 char h1[NI_MAXHOST], h2[NI_MAXHOST];
588 const int niflags = NI_NUMERICHOST;
589
590 if (getnameinfo((struct sockaddr *)(void *)&remoteaddr,
591#ifdef BSD4_4
592 (socklen_t)remoteaddr.ss_len,
593#else
594 sizeof(struct sockaddr),
595#endif
596 h1, sizeof(h1), NULL, 0, niflags))
597 return (0);
598 if (getifaddrs(&ifap) < 0)
599 return (0);
600 for (ifa = ifap; ifa; ifa = ifa->ifa_next) {
601 if (remoteaddr.ss_family != ifa->ifa_addr->sa_family)
602 continue;
603 if (getnameinfo(ifa->ifa_addr,
604#ifdef BSD4_4
605 (socklen_t)ifa->ifa_addr->sa_len,
606#else
607 sizeof(struct sockaddr),
608#endif
609 h2, sizeof(h2), NULL, 0, niflags))
610 continue;
611 if (strcmp(h1, h2) == 0) {
612 freeifaddrs(ifap);
613 return (1);
614 }
615 }
616 freeifaddrs(ifap);
617 return (0);
618}

620int
621matchhost(char *name)
622{ /* is this name of remote host? */
623 char h1[NI_MAXHOST], h2[NI_MAXHOST];
624 const int niflags = NI_NUMERICHOST;
625 struct addrinfo hints, *res0, *res;
626
627 if (getnameinfo((struct sockaddr *)(void *)&remoteaddr,
628#ifdef BSD4_4
629 (socklen_t)remoteaddr.ss_len,
630#else
631 sizeof(struct sockaddr),
632#endif
633 h1, sizeof(h1), NULL, 0, niflags))
634 return (0);
635 memset(&hints, 0, sizeof(hints));
636 hints.ai_family = PF_UNSPEC;
637 hints.ai_socktype = SOCK_DGRAM; /* dummy */
638 if (getaddrinfo(name, "0", &hints, &res0) != 0)
639 return (0);
640 for (res = res0; res; res = res->ai_next) {
641 if (remoteaddr.ss_family != res->ai_family)
642 continue;
643 if (getnameinfo(res->ai_addr, res->ai_addrlen,
644 h2, sizeof(h2), NULL, 0, niflags))
645 continue;
646 if (strcmp(h1, h2) == 0) {
647 freeaddrinfo(res0);
648 return (1);
649 }
650 }
651 freeaddrinfo(res0);
652 return (0);
653}

655int
656scmerr(int error, const char *fmt, ...)
657{
658 va_list ap;
659 char hostname[MAXHOSTNAMELEN];
660 gethostname(hostname, sizeof(hostname));
661
662 va_start(ap, fmt);
663
664 (void) fflush(stdout);
665 if (progpid > 0)
666 fprintf(stderr, "%s@%s %d: ", program, hostname, progpid);
667 else
668 fprintf(stderr, "%s@%s: ", program, hostname);
669
670 vfprintf(stderr, fmt, ap);
671 va_end(ap);
672 if (error >= 0)
673 fprintf(stderr, " (%s)\n", strerror(error));
674 else
675 fprintf(stderr, "\n");
676 (void) fflush(stderr);
677 return (SCMERR);
678}

683union intchar {
684 int ui;
685 char uc[sizeof(int)];
686};
687
688int
689byteswap(int in)
690{
691 union intchar x, y;
692 int ix, iy;
693
694 if (swapmode == 0)
695 return (in);
696 x.ui = in;
697 iy = sizeof(int);
698 for (ix = 0; ix < (int) sizeof(int); ix++) {
699 --iy;
700 y.uc[iy] = x.uc[ix];
701 }
702 return (y.ui);
703}
