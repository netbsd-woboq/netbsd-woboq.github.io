77void
78Tfree(TREE ** t)
79{
80 if (*t == NULL)
81 return;
82 Tfree(&((*t)->Tlink));
83 Tfree(&((*t)->Texec));
84 Tfree(&((*t)->Tlo));
85 Tfree(&((*t)->Thi));
86 if ((*t)->Tname)
87 free((*t)->Tname);
88 if ((*t)->Tuser)
89 free((*t)->Tuser);
90 if ((*t)->Tgroup)
91 free((*t)->Tgroup);
92 free(*(char **) t);
93 *t = NULL;
94}

96static TREE *
97Tmake(const char *p)
98{
99 TREE *t;
100 t = (TREE *) malloc(sizeof(TREE));
101 if (t == NULL)
102 goaway("Cannot allocate memory");
103 t->Tname = (p == NULL) ? NULL : estrdup(p);
104 t->Tflags = 0;
105 t->Tuid = 0;
106 t->Tgid = 0;
107 t->Tuser = NULL;
108 t->Tgroup = NULL;
109 t->Tmode = 0;
110 t->Tctime = 0;
111 t->Tmtime = 0;
112 t->Tlink = NULL;
113 t->Texec = NULL;
114 t->Tbf = 0;
115 t->Tlo = NULL;
116 t->Thi = NULL;
117 return (t);
118}

120static TREE *
121Trotll(TREE * tp, TREE * tl)
122{
123 tp->Tlo = tl->Thi;
124 tl->Thi = tp;
125 tp->Tbf = tl->Tbf = 0;
126 return (tl);
127}
128
129static TREE *
130Trotlh(TREE * tp, TREE * tl)
131{
132 TREE *th;
133
134 th = tl->Thi;
135 tp->Tlo = th->Thi;
136 tl->Thi = th->Tlo;
137 th->Thi = tp;
138 th->Tlo = tl;
139 tp->Tbf = tl->Tbf = 0;
140 if (th->Tbf == 1)
141 tp->Tbf = -1;
142 else if (th->Tbf == -1)
143 tl->Tbf = 1;
144 th->Tbf = 0;
145 return (th);
146}

148static TREE *
149Trothl(TREE * tp, TREE * th)
150{
151 TREE *tl;
152
153 tl = th->Tlo;
154 tp->Thi = tl->Tlo;
155 th->Tlo = tl->Thi;
156 tl->Tlo = tp;
157 tl->Thi = th;
158 tp->Tbf = th->Tbf = 0;
159 if (tl->Tbf == -1)
160 tp->Tbf = 1;
161 else if (tl->Tbf == 1)
162 th->Tbf = -1;
163 tl->Tbf = 0;
164 return (tl);
165}

167static TREE *
168Trothh(TREE * tp, TREE * th)
169{
170 tp->Thi = th->Tlo;
171 th->Tlo = tp;
172 tp->Tbf = th->Tbf = 0;
173 return (th);
174}

176static void
177Tbalance(TREE ** t)
178{
179 if ((*t)->Tbf < 2 && (*t)->Tbf > -2)
180 return;
181 if ((*t)->Tbf > 0) {
182 if ((*t)->Tlo->Tbf > 0)
183 *t = Trotll(*t, (*t)->Tlo);
184 else
185 *t = Trotlh(*t, (*t)->Tlo);
186 } else {
187 if ((*t)->Thi->Tbf > 0)
188 *t = Trothl(*t, (*t)->Thi);
189 else
190 *t = Trothh(*t, (*t)->Thi);
191 }
192}

194static TREE *
195Tinsertavl(TREE ** t, const char *p, int find, int *dh)
196{
197 TREE *newt;
198 int cmp;
199 int deltah;
200
201 if (*t == NULL) {
202 *t = Tmake(p);
203 *dh = 1;
204 return (*t);
205 }
206 if ((cmp = strcmp(p, (*t)->Tname)) == 0) {
207 if (!find)
208 return (NULL); /* node already exists */
209 *dh = 0;
210 return (*t);
211 } else if (cmp < 0) {
212 if ((newt = Tinsertavl(&((*t)->Tlo), p, find, &deltah)) == NULL)
213 return (NULL);
214 (*t)->Tbf += deltah;
215 } else {
216 if ((newt = Tinsertavl(&((*t)->Thi), p, find, &deltah)) == NULL)
217 return (NULL);
218 (*t)->Tbf -= deltah;
219 }
220 Tbalance(t);
221 if ((*t)->Tbf == 0)
222 deltah = 0;
223 *dh = deltah;
224 return (newt);
225}

227TREE *
228Tinsert(TREE ** t, const char *p, int find)
229{
230 int deltah;
231
232 if (p != NULL && p[0] == '.' && p[1] == '/') {
233 p += 2;
234 while (*p == '/')
235 p++;
236 if (*p == 0)
237 p = ".";
238 }
239 return (Tinsertavl(t, p, find, &deltah));
240}

242TREE *
243Tsearch(TREE * t, const char *p)
244{
245 TREE *x;
246 int cmp;
247
248 x = t;
249 while (x) {
250 cmp = strcmp(p, x->Tname);
251 if (cmp == 0)
252 return (x);
253 if (cmp < 0)
254 x = x->Tlo;
255 else
256 x = x->Thi;
257 }
258 return (NULL);
259}

261TREE *
262Tlookup(TREE * t, const char *p)
263{
264 TREE *x;
265 char buf[MAXPATHLEN + 1];
266 char *q;
267
268 if (p == NULL)
269 return (NULL);
270 if (p[0] == '.' && p[1] == '/') {
271 p += 2;
272 while (*p == '/')
273 p++;
274 if (*p == 0)
275 p = ".";
276 }
277 if ((x = Tsearch(t, p)) != NULL)
278 return (x);
279 if (*p != '/' && (x = Tsearch(t, ".")) != NULL)
280 return (x);
281 (void) strncpy(buf, p, sizeof(buf) - 1);
282 buf[MAXPATHLEN] = '\0';
283 while ((q = strrchr(buf, '/')) != NULL) {
284 while (q >= buf && *(q - 1) == '/')
285 q--;
286 if (q == buf)
287 *(q + 1) = '\0';
288 else
289 *q = '\0';
290 if ((x = Tsearch(t, buf)) != NULL)
291 return (x);
292 if (q == buf)
293 break;
294 }
295 return (NULL);
296}

298static int process_level;
299
300static int
301Tsubprocess(TREE * t, int reverse, int (*f) (TREE *, void *), void *argp)
302{
303 int x = SCMOK;
304 process_level++;
305 if (reverse ? t->Thi : t->Tlo)
306 x = Tsubprocess(reverse ? t->Thi : t->Tlo,
307 reverse, f, argp);
308 if (x == SCMOK) {
309 x = (*f) (t, argp);
310 if (x == SCMOK) {
311 if (reverse ? t->Tlo : t->Thi)
312 x = Tsubprocess(reverse ? t->Tlo : t->Thi,
313 reverse, f, argp);
314 }
315 }
316 process_level--;
317 return (x);
318}

323{
324 if (t == NULL)
325 return (SCMOK);
326 process_level = 0;
327 return (Tsubprocess(t, TRUE, f, args));

340static int
341/*ARGSUSED*/
342Tprintone(TREE * t, void *v __unused)
343{
344 int i;
345 for (i = 0; i < (process_level * 2); i++)
346 (void) putchar(' ');
347 printf("Node at %p name '%s' flags %o hi %p lo %p\n", t, t->Tname, t->Tflags, t->Thi, t->Tlo);
348 return (SCMOK);
349}

351void
352Tprint(TREE * t, char *p)
353{ /* print tree -- for debugging */
354 printf("%s\n", p);
355 (void) Tprocess(t, Tprintone, NULL);
356 printf("End of tree\n");
357 (void) fflush(stdout);
358}
