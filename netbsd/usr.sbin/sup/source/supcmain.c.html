27/*
28 * sup -- Software Upgrade Protocol client process
29 *
30 * Usage: sup [ flags ] [ supfile ] [ collection ... ]
31 *
32 * The only required argument to sup is the name of a supfile. It
33 * must either be given explicitly on the command line, or the -s
34 * flag must be specified. If the -s flag is given, the system
35 * supfile will be used and a supfile command argument should not be
36 * specified. The list of collections is optional and if specified
37 * will be the only collections upgraded. The following flags affect
38 * all collections specified.
39 *
40 * -s "system upgrade" flag
41 * As described above.
42 *
43 * -t "upgrade time" flag
44 * When this flag is given, Sup will print the time
45 * that each collection was last upgraded, rather than
46 * performing actual upgrades.
47 *
48 * -u "utimes" flag
49 * When this flags is given sup will not attempt to
50 * restore access and modification files on the collection
51 * files from the server.
52 *
53 * -R "resource pause" flag
54 * Sup will not disable resource pausing and will not
55 * make filesystem space checks.
56 *
57 * -N "debug network" flag
58 * Sup will trace messages sent and received that
59 * implement the Sup network protocol.
60 *
61 * -P "debug ports" flag
62 * Sup will use a set of non-privileged network
63 * ports reserved for debugging purposes.
64 *
65 * -X "crosspatch" flag
66 * Sup is being run remotely with a crosspatch.
67 * Need to be carefull as we may be running as root
68 * instead of collection owner.
69 *
70 * The remaining flags affect all collections unless an explicit list
71 * of collections are given with the flags. Multiple flags may be
72 * specified together that affect the same collections. For the sake
73 * of convience, any flags that always affect all collections can be
74 * specified with flags that affect only some collections. For
75 * example, "sup -sde=coll1,coll2" would perform a system upgrade,
76 * and the first two collections would allow both file deletions and
77 * command executions. Note that this is not the same command as
78 * "sup -sde=coll1 coll2", which would perform a system upgrade of
79 * just the coll2 collection and would ignore the flags given for the
80 * coll1 collection.
81 *
82 * -a "all files" flag
83 * All files in the collection will be copied from
84 * the repository, regardless of their status on the
85 * current machine. Because of this, it is a very
86 * expensive operation and should only be done for
87 * small collections if data corruption is suspected
88 * and been confirmed. In most cases, the -o flag
89 * should be sufficient.
90 *
91 * -b "backup files" flag
92 * If the -b flag if given, or the "backup" supfile
93 * option is specified, the contents of regular files
94 * on the local system will be saved before they are
95 * overwritten with new data. The data will be saved
96 * in a subdirectory called "BACKUP" in the directory
97 * containing the original version of the file, in a
98 * file with the same non-directory part of the file
99 * name. The files to backup are specified by the
100 * list file on the repository.
101 *
102 * -B "don't backup files" flag
103 * The -B flag overrides and disables the -b flag and
104 * the "backup" supfile option.
105 *
106 * -d "delete files" flag
107 * Files that are no longer in the collection on the
108 * repository will be deleted if present on the local
109 * machine. This may also be specified in a supfile
110 * with the "delete" option.
111 *
112 * -D "don't delete files" flag
113 * The -D flag overrides and disables the -d flag and
114 * the "delete" supfile option.
115 *
116 * -e "execute files" flag
117 * Sup will execute commands sent from the repository
118 * that should be run when a file is upgraded. If
119 * the -e flag is omitted, Sup will print a message
120 * that specifies the command to execute. This may
121 * also be specified in a supfile with the "execute"
122 * option.
123 *
124 * -E "don't execute files" flag
125 * The -E flag overrides and disables the -e flag and
126 * the "execute" supfile option.
127 *
128 * -f "file listing" flag
129 * A "list-only" upgrade will be performed. Messages
130 * will be printed that indicate what would happen if
131 * an actual upgrade were done.
132 *
133 * -k "keep newer files" flag
134 * The -k flag, or "keep" supfile option, will cause
135 * Sup to check to see whether there is a newer file on
136 * the local disk before updating files. Only files
137 * which are newer on the repository will be updated.
138 *
139 * -K "don't keep newer files" flag
140 * The -K flag overrides and disables the -k flag and
141 * the "keep" supfile option.
142 *
143 * -l "local upgrade" flag
144 * Normally, Sup will not upgrade a collection if the
145 * repository is on the same machine. This allows
146 * users to run upgrades on all machines without
147 * having to make special checks for the repository
148 * machine. If the -l flag is specified, collections
149 * will be upgraded even if the repository is local.
150 *
151 * -m "mail" flag
152 * Normally, Sup used standard output for messages.
153 * If the -m flag if given, Sup will send mail to the
154 * user running Sup, or a user specified with the
155 * "notify" supfile option, that contains messages
156 * printed by Sup.
157 *
158 * -M "mailto" flag
159 * Like -m, but send mail to the specified user.
160 * -o "old files" flag
161 * Sup will normally only upgrade files that have
162 * changed on the repository since the last time an
163 * upgrade was performed. The -o flag, or the "old"
164 * supfile option, will cause Sup to check all files
165 * in the collection for changes instead of just the
166 * new ones.
167 *
168 * -O "not old files" flag
169 * The -O flag overrides and disables the -o flag and
170 * the "old" supfile option.
171 *
172 * -v "verbose" flag
173 * Normally, Sup will only print messages if there
174 * are problems. This flag causes Sup to also print
175 * messages during normal progress showing what Sup
176 * is doing.
177 *

359int
360main(int argc, char **argv)
361{
362 char *progname, *supfname;
363 int restart;
364 dev_t sfdev = 0;
365 ino_t sfino = 0;
366 time_t sfmtime = 0;
367 struct stat sbuf;
368 struct sigaction ign;
369
370 /* initialize global variables */
371 pgmversion = PGMVERSION;/* export version number */
372 isserver = FALSE; /* export that we're not a server */
373 collname = NULL; /* no current collection yet */
374 dontjump = TRUE; /* clear setjmp buffer */
375 progname = estrdup(argv[0]);
376
377 supfname = init(argc, argv);
378 restart = -1; /* don't make restart checks */
379 if (*progname == '/' && *supfname == '/') {
380 if (stat(supfname, &sbuf) < 0) {
381 logerr("Can't stat supfile %s", supfname);
382 } else {
383 sfdev = sbuf.st_dev;
384 sfino = sbuf.st_ino;
385 sfmtime = sbuf.st_mtime;
386 restart = 0;
387 }
388 }
389 if (timeflag) {
390 for (thisC = firstC; thisC; thisC = thisC->Cnext)
391 prtime();
392 } else {
393 /* ignore network pipe signals */
394 ign.sa_handler = SIG_IGN;
395 ign.sa_flags = 0;
396 sigemptyset(&ign.sa_mask);
397 (void) sigaction(SIGPIPE, &ign, NULL);
398 getnams(); /* find unknown repositories */
399 for (thisC = firstC; thisC; thisC = thisC->Cnext) {
400 getcoll(); /* upgrade each collection */
401 if (restart == 0) {
402 if (stat(supfname, &sbuf) < 0)
403 logerr("Can't stat supfile %s",
404 supfname);
405 else if (sfmtime != sbuf.st_mtime ||
406#ifndef __CYGWIN__ /* Cygwin's inodes are not constant */
407 sfino != sbuf.st_ino ||
408#endif
409 sfdev != sbuf.st_dev) {
410 restart = 1;
411 break;
412 }
413 }
414 }
415 endpwent(); /* close /etc/passwd */
416 (void) endgrent(); /* close /etc/group */
417 if (restart == 1) {
418 int fd;
419 if (!silent)
420 loginfo("SUP Restarting %s with new supfile %s",
421 progname, supfname);
422 for (fd = getdtablesize(); fd > 3; fd--)
423 (void) close(fd);
424 execv(progname, argv);
425 logquit(1, "Restart failed");
426 }
427 }
428 while ((thisC = firstC) != NULL) {
429 firstC = firstC->Cnext;
430 free(thisC->Cname);
431 Tfree(&thisC->Chtree);
432 free(thisC->Cbase);
433 if (thisC->Chbase)
434 free(thisC->Chbase);
435 if (thisC->Cprefix)
436 free(thisC->Cprefix);
437 if (thisC->Crelease)
438 free(thisC->Crelease);
439 if (thisC->Cnotify)
440 free(thisC->Cnotify);
441 if (thisC->Clogin)
442 free(thisC->Clogin);
443 if (thisC->Cpswd)
444 free(thisC->Cpswd);
445 if (thisC->Ccrypt)
446 free(thisC->Ccrypt);
447 free(thisC);
448 }
449 exit(0);
450}

463static void
464doswitch(int *argc, char ***argv, TREE ** collTp, int *oflagsp, int *aflagsp,
465 char *username, size_t ulen)
466{
467 TREE *t;
468 char *coll, *argp;
469 int oflags, aflags;
470 int c;
471
472#define SUPOPTIONS "abBCdDeEifkKlmM:NoOPRsStuvXzZ=:"
473
474 oflags = aflags = 0;
475 while ((c = getopt(*argc, *argv, SUPOPTIONS)) != -1)
476 switch (c) {
477 default:
478 logerr("Invalid flag '%c' ignored", c);
479 break;
480 case '\0':
481 case '=':
482 argp = optarg;
483 do {
484 coll = nxtarg(&argp, ", \t");
485 t = Tinsert(collTp, coll, TRUE);
486 t->Toflags |= oflags;
487 t->Toflags &= ~aflags;
488 t->Taflags |= aflags;
489 t->Taflags &= ~oflags;
490 argp = skipover(argp, ", \t");
491 } while (*argp);
492 break;
493 case 'a':
494 oflags |= CFALL;
495 break;
496 case 'b':
497 oflags |= CFBACKUP;
498 aflags &= ~CFBACKUP;
499 break;
500 case 'B':
501 oflags &= ~CFBACKUP;
502 aflags |= CFBACKUP;
503 break;
504 case 'C':
505 oflags |= CFCANONICALIZE;
506 break;
507 case 'd':
508 oflags |= CFDELETE;
509 aflags &= ~CFDELETE;
510 break;
511 case 'D':
512 oflags &= ~CFDELETE;
513 aflags |= CFDELETE;
514 break;
515 case 'e':
516 oflags |= CFEXECUTE;
517 aflags &= ~CFEXECUTE;
518 break;
519 case 'E':
520 oflags &= ~CFEXECUTE;
521 aflags |= CFEXECUTE;
522 break;
523 case 'f':
524 oflags |= CFLIST;
525 break;
526 case 'i':
527 oflags |= CFIGNCHERR;
528 aflags &= ~CFIGNCHERR;
529 break;
530 case 'k':
531 oflags |= CFKEEP;
532 aflags &= ~CFKEEP;
533 break;
534 case 'K':
535 oflags &= ~CFKEEP;
536 aflags |= CFKEEP;
537 break;
538 case 'l':
539 oflags |= CFLOCAL;
540 break;
541 case 'm':
542 oflags |= CFMAIL;
543 break;
544 case 'M':
545 oflags |= CFMAIL;
546 strncpy(username, optarg, ulen);
547 username[ulen - 1] = '\0';
548 break;
549 case 'N':
550 scmdebug++;
551 break;
552 case 'o':
553 oflags |= CFOLD;
554 aflags &= ~CFOLD;
555 break;
556 case 'O':
557 oflags &= ~CFOLD;
558 aflags |= CFOLD;
559 break;
560 case 'P':
561 portdebug = TRUE;
562 break;
563 case 'R':
564#if MACH
565 rpauseflag = TRUE;
566#endif /* MACH */
567 break;
568 case 's':
569 sysflag = TRUE;
570 break;
571 case 'S':
572 silent = TRUE;
573 break;
574 case 't':
575 timeflag = TRUE;
576 break;
577 case 'u':
578 noutime = TRUE;
579 break;
580 case 'v':
581 oflags |= CFVERBOSE;
582 break;
583 case 'X':
584 xpatchflag = TRUE;
585 break;
586 case 'z':
587 oflags |= CFCOMPRESS;
588 break;
589 case 'Z':
590 oflags &= ~CFCOMPRESS;
591 break;
592 }
593
594 *oflagsp |= oflags;
595 *oflagsp &= ~aflags;
596 *aflagsp |= aflags;
597 *aflagsp &= ~oflags;
598 *argc -= optind;
599 *argv += optind;
600}

602static char *
603init(int argc, char **argv)
604{
605 char buf[STRINGLENGTH], *p;
606 const char *u;
607 char username[STRINGLENGTH];
608 char *supfname, *arg;
609 COLLECTION *c, *lastC;
610 FILE *f;
611 int bogus;
612 struct passwd *pw;
613 TREE *t;
614 TREE *collT; /* collections we are interested in */
615 time_t timenow; /* startup time */
616 int oflags, aflags;
617 int cwant;
618#ifdef MACH
619 void (*oldsigsys) ();
620#endif /* MACH */
621
622 username[0] = '\0';
623 sysflag = FALSE; /* not system upgrade */
624 timeflag = FALSE; /* don't print times */
625#if MACH
626 rpauseflag = FALSE; /* don't disable resource pausing */
627#endif /* MACH */
628 xpatchflag = FALSE; /* not normally crosspatched */
629 scmdebug = 0; /* level zero, no SCM debugging */
630 portdebug = FALSE; /* no debugging ports */
631
632 collT = NULL;
633 oflags = aflags = 0;
634
635 doswitch(&argc, &argv, &collT, &oflags, &aflags, username,
636 sizeof(username));
637
638 if (argc == 0 && !sysflag)
639 logquit(1, "Need either -s or supfile");
640#if MACH
641 oldsigsys = signal(SIGSYS, SIG_IGN);
642 if (rpauseflag != TRUE)
643 if (syscall(SYS_rpause, ENOSPC, RPAUSE_ALL, RPAUSE_DISABLE) < 0)
644 rpauseflag = TRUE;
645 (void) signal(SIGSYS, oldsigsys);
646#endif /* MACH */
647 if (sysflag)
648 (void) snprintf(supfname = buf, sizeof(buf),
649 timeflag ? FILESUPTDEFAULT : FILESUPDEFAULT,
650 DEFDIR);
651 else {
652 supfname = *argv;
653 if (strcmp(supfname, "-") == 0)
654 supfname = "";
655 --argc;
656 argv++;
657 }
658 cwant = argc > 0;
659 while (argc > 0) {
660 t = Tinsert(&collT, *argv, TRUE);
661 t->Twant = TRUE;
662 --argc;
663 argv++;
664 }
665 if (*username == '\0' && ((u = getlogin()) ||
666 ((pw = getpwuid((int) getuid())) && (u = pw->pw_name)))) {
667 (void)strncpy(username, u, sizeof(username));
668 username[sizeof(username) - 1] = '\0';
669 }
670 if (*supfname) {
671 f = fopen(supfname, "r");
672 if (f == NULL)
673 logquit(1, "Can't open supfile %s", supfname);
674 } else
675 f = stdin;
676 firstC = NULL;
677 lastC = NULL;
678 bogus = FALSE;
679 while ((p = read_line(f, NULL, NULL, NULL, 0)) != NULL) {
680 if (strchr("#;:", *p))
681 continue;
682 arg = nxtarg(&p, " \t");
683 if (*arg == '\0') {
684 logerr("Missing collection name in supfile");
685 bogus = TRUE;
686 continue;
687 }
688 if (cwant) {
689 TREE *mytree;
690 if ((mytree = Tsearch(collT, arg)) == NULL)
691 continue;
692 mytree->Tcount++;
693 }
694 c = (COLLECTION *) malloc(sizeof(COLLECTION));
695 if (c == NULL) {
696 logerr("Cannot allocate memory");
697 exit(1);
698 }
699 if (firstC == NULL)
700 firstC = c;
701 if (lastC != NULL)
702 lastC->Cnext = c;
703 lastC = c;
704 if (parsecoll(c, arg, p) < 0) {
705 bogus = TRUE;
706 continue;
707 }
708 c->Cflags |= oflags;
709 c->Cflags &= ~aflags;
710 if ((t = Tsearch(collT, c->Cname)) != NULL) {
711 c->Cflags |= t->Toflags;
712 c->Cflags &= ~t->Taflags;
713 }
714 if ((c->Cflags & CFMAIL) && c->Cnotify == NULL) {
715 if (*username == '\0')
716 logerr("User unknown, notification disabled");
717 else
718 c->Cnotify = estrdup(username);
719 }
720 if (c->Cbase == NULL) {
721 (void) sprintf(buf, FILEBASEDEFAULT, c->Cname);
722 c->Cbase = estrdup(buf);
723 }
724 }
725 if (bogus)
726 logquit(1, "Aborted due to supfile errors");
727 if (f != stdin)
728 (void) fclose(f);
729 if (cwant)
730 (void) Tprocess(collT, checkcoll, NULL);
731 Tfree(&collT);
732 if (firstC == NULL)
733 logquit(1, "No collections to upgrade");
734 timenow = time(NULL);
735 if (*supfname == '\0')
736 p = "standard input";
737 else if (sysflag)
738 p = "system software";
739 else
740 (void) sprintf(p = buf, "file %s", supfname);
741 if (!silent)
742 loginfo("SUP %d.%d (%s) for %s at %s", PROTOVERSION, PGMVERSION,
743 scmversion, p, fmttime(timenow));
744 return (estrdup(supfname));
745}

747static int
748checkcoll(TREE * t, void *dummy __unused)
749{
750 if (!t->Twant)
751 return (SCMOK);
752 if (t->Tcount == 0)
753 logerr("Collection %s not found", t->Tname);
754 if (t->Tcount > 1)
755 logerr("Collection %s found more than once", t->Tname);
756 return (SCMOK);
757}
