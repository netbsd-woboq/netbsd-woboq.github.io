146/* The next two routines define the fsm to support multiple fileservers
147 * per collection.
148 */
149int
150getonehost(TREE * t, void *v)
151{
152 long *state = v;
153 if (t->Tflags != *state)
154 return (SCMOK);
155 if (*state != 0 && t->Tmode == SCMEOF) {
156 t->Tflags = 0;
157 return (SCMOK);
158 }
159 if (*state == 2)
160 t->Tflags--;
161 else
162 t->Tflags++;
163 thisC->Chost = t;
164 return (SCMEOF);
165}

176TREE *
177getcollhost(int *tout, int *backoff, long int *state, int *nhostsp)
178{
179 static long laststate = 0;
180 static int nhosts = 0;
181
182 if (*state != laststate) {
183 *nhostsp = nhosts;
184 laststate = *state;
185 nhosts = 0;
186 }
187 if (Tprocess(thisC->Chtree, getonehost, state) == SCMEOF) {
188 if (*state != 0 && nhosts == 0 && !dobackoff(tout, backoff))
189 return (NULL);
190 nhosts++;
191 return (thisC->Chost);
192 }
193 if (nhosts == 0)
194 return (NULL);
195 if (*state == 2)
196 (*state)--;
197 else
198 (*state)++;
199 return (getcollhost(tout, backoff, state, nhostsp));
200}
201/* Upgrade a collection from the file server on the appropriate
202 * host machine.
203 */
204
205void
206getcoll(void)
207{
208 TREE *t;
209 int x;
210 int tout, backoff, nhosts;
211 long state;
212
213 collname = thisC->Cname;
214 tout = thisC->Ctimeout;
215 lastT = NULL;
216 backoff = 2;
217 state = 0;
218 nhosts = 0;
219 for (;;) {
220 t = getcollhost(&tout, &backoff, &state, &nhosts);
221 if (t == NULL) {
222 finishup(SCMEOF);
223 notify(0, NULL);
224 return;
225 }
226 t->Tmode = SCMEOF;
227 dontjump = FALSE;
228 if (!setjmp(sjbuf) && !signon(t, nhosts, &tout) && !setup(t))
229 break;
230 (void) requestend();
231 }
232 dontjump = FALSE;
233 if (setjmp(sjbuf))
234 x = SCMERR;
235 else {
236 suplogin();
237 listfiles();
238 recvfiles();
239 x = SCMOK;
240 }
241 if (thisC->Clockfd >= 0) {
242 (void) close(thisC->Clockfd);
243 thisC->Clockfd = -1;
244 }
245 finishup(x);
246 notify(0, NULL);
247}

250int
251signon(TREE * t, int nhosts, int *tout)
252{
253 int x;
254 int timeout;
255 time_t tloc;
256
257 if ((thisC->Cflags & CFLOCAL) == 0 && thishost(thisC->Chost->Tname)) {
258 vnotify(1, "Skipping local collection %s", collname);
259 t->Tmode = SCMEOF;
260 return (TRUE);
261 }
262 if (nhosts == 1)
263 timeout = *tout;
264 else
265 timeout = 0;
266 x = request(portdebug ? DEBUGFPORT : FILEPORT,
267 thisC->Chost->Tname, &timeout);
268 if (nhosts == 1)
269 *tout = timeout;
270 if (x != SCMOK) {
271 if (nhosts) {
272 notify(1, "Can't connect to host %s",
273 thisC->Chost->Tname);
274 t->Tmode = SCMEOF;
275 } else
276 t->Tmode = SCMOK;
277 return (TRUE);
278 }
279 xpatch = FALSE;
280 x = msgsignon(); /* signon to fileserver */
281 if (x != SCMOK)
282 goaway("Error sending signon request to fileserver");
283 x = msgsignonack(); /* receive signon ack from fileserver */
284 if (x != SCMOK)
285 goaway("Error reading signon reply from fileserver");
286 tloc = time(NULL);
287 vnotify(0, "Fileserver %d.%d (%s) %d on %s at %.8s",
288 protver, pgmver, scmver, fspid, remotehost(), supctime(&tloc) + 11);
289 free(scmver);
290 scmver = NULL;
291 if (protver < 4) {
292 dontjump = TRUE;
293 goaway("Fileserver sup protocol version is obsolete.");
294 notify(1, "This version of sup can only communicate with a fileserver using at least");
295 notify(1, "version 4 of the sup network protocol. You should either run a newer");
296 notify(1, "version of the sup fileserver or find an older version of sup.");
297 t->Tmode = SCMEOF;
298 return (TRUE);
299 }
300 /* If protocol is > 7 then try compression */
301 if (protver > 7) {
302 cancompress = TRUE;
303 }
304 return (FALSE);
305}

308int
309setup(TREE * t)
310{
311 char relsufix[STRINGLENGTH];
312 int x;
313 struct stat sbuf;
314
315 if (chdir(thisC->Cbase) < 0)
316 goaway("Can't change to base directory %s", thisC->Cbase);
317 if (stat("sup", &sbuf) < 0) {
318 (void) mkdir("sup", 0755);
319 if (stat("sup", &sbuf) < 0)
320 goaway("Can't create directory %s/sup", thisC->Cbase);
321 vnotify(0, "Created directory %s/sup", thisC->Cbase);
322 }
323 if (thisC->Cprefix && chdir(thisC->Cprefix) < 0)
324 goaway("Can't change to %s from base directory %s",
325 thisC->Cprefix, thisC->Cbase);
326 if (stat(".", &sbuf) < 0)
327 goaway("Can't stat %s directory %s",
328 thisC->Cprefix ? "prefix" : "base",
329 thisC->Cprefix ? thisC->Cprefix : thisC->Cbase);
330 if (thisC->Cprefix)
331 if (chdir(thisC->Cbase) < 0)
332 goaway("Can't chdir to %s (%s)", thisC->Cbase,
333 strerror(errno));
334 /* read time of last upgrade from when file */
335
336 if ((thisC->Cflags & CFURELSUF) && thisC->Crelease)
337 (void) snprintf(relsufix, sizeof(relsufix), ".%s",
338 thisC->Crelease);
339 else
340 relsufix[0] = '\0';
341 lasttime = getwhen(collname, relsufix);
342 /* setup for msgsetup */
343 basedir = thisC->Chbase;
344 basedev = sbuf.st_dev;
345 baseino = sbuf.st_ino;
346 listonly = (thisC->Cflags & CFLIST);
347 newonly = ((thisC->Cflags & (CFALL | CFDELETE | CFOLD)) == 0);
348 release = thisC->Crelease;
349 x = msgsetup();
350 if (x != SCMOK)
351 goaway("Error sending setup request to file server");
352 x = msgsetupack();
353 if (x != SCMOK)
354 goaway("Error reading setup reply from file server");
355 if (setupack == FSETUPOK) {
356 /* Test encryption */
357 if (netcrypt(thisC->Ccrypt) != SCMOK)
358 goaway("Running non-crypting sup");
359 crypttest = CRYPTTEST;
360 x = msgcrypt();
361 if (x != SCMOK)
362 goaway("Error sending encryption test request");
363 x = msgcryptok();
364 if (x == SCMEOF)
365 goaway("Data encryption test failed");
366 if (x != SCMOK)
367 goaway("Error reading encryption test reply");
368 return (FALSE);
369 }
370 switch (setupack) {
371 case FSETUPSAME:
372 notify(1, "Attempt to upgrade from same host to same directory");
373 done(FDONESRVERROR, "Overwrite error");
374 break;
375 case FSETUPHOST:
376 notify(1, "This host has no permission to access %s",
377 collname);
378 done(FDONESRVERROR, "Permission denied");
379 break;
380 case FSETUPOLD:
381 notify(1, "This version of SUP is too old for the fileserver");
382 done(FDONESRVERROR, "Obsolete client");
383 break;
384 case FSETUPRELEASE:
385 notify(1, "Invalid release %s for collection %s",
386 release == NULL ? DEFRELEASE : release, collname);
387 done(FDONESRVERROR, "Invalid release");
388 break;
389 case FSETUPBUSY:
390 vnotify(0, "Fileserver is currently busy");
391 t->Tmode = SCMOK;
392 doneack = FDONESRVERROR;
393 donereason = "Fileserver is busy";
394 (void) netcrypt(NULL);
395 (void) msgdone();
396 return (TRUE);
397 default:
398 goaway("Unrecognized file server setup status %d", setupack);
399 }
400 /* NOTREACHED */
401 return FALSE;
402}

405void
406suplogin(void)
407{
408 char buf[STRINGLENGTH];
409 int f, x;
410
411 /* lock collection if desired */
412 (void) snprintf(buf, sizeof(buf), FILELOCK, collname);
413 f = open(buf, O_RDONLY, 0);
414 if (f >= 0) {
415
416#if defined(LOCK_EX)
417#define TESTLOCK(f) flock(f, LOCK_EX|LOCK_NB)
418#define SHARELOCK(f) flock(f, LOCK_SH|LOCK_NB)
419#define WAITLOCK(f) flock(f, LOCK_EX)
420#elif defined(F_LOCK)
421#define TESTLOCK(f) lockf(f, F_TLOCK, 0)
422#define SHARELOCK(f) 1
423#define WAITLOCK(f) lockf(f, F_LOCK, 0)
424#else
425#define TESTLOCK(f) (close(f), f = -1, 1)
426#define SHARELOCK(f) 1
427#define WAITLOCK(f) 1
428#endif
429 if (TESTLOCK(f) < 0) {
430 if (errno != EWOULDBLOCK && errno != EAGAIN) {
431 (void) close(f);
432 goaway("Can't lock collection %s", collname);
433 }
434 if (SHARELOCK(f) < 0) {
435 (void) close(f);
436 if (errno == EWOULDBLOCK && errno != EAGAIN)
437 goaway("Collection %s is locked by another sup", collname);
438 goaway("Can't lock collection %s", collname);
439 }
440 vnotify(0, "Waiting for exclusive access lock");
441 if (WAITLOCK(f) < 0) {
442 (void) close(f);
443 goaway("Can't lock collection %s", collname);
444 }
445 }
446 thisC->Clockfd = f;
447 vnotify(0, "Locked collection %s for exclusive access",
448 collname);
449 }
450 logcrypt = NULL;
451 loguser = thisC->Clogin;
452 logpswd = thisC->Cpswd;
453
454#ifndef CRYPTING /* Define CRYPTING for backwards compatibility
455 * with old supfileservers */
456 if (thisC->Clogin != NULL) /* othewise we only encrypt if
457 * there is a login id */
458#endif /* CRYPTING */
459 {
460 logcrypt = CRYPTTEST;
461 (void) netcrypt(PSWDCRYPT); /* encrypt password data */
462 }
463 x = msglogin();
464#ifndef CRYPTING
465 if (thisC->Clogin != NULL)
466#endif
467 (void) netcrypt(NULL); /* turn off encryption */
468 if (x != SCMOK)
469 goaway("Error sending login request to file server");
470 x = msglogack();
471 if (x != SCMOK)
472 goaway("Error reading login reply from file server");
473 if (logack == FLOGNG) {
474 notify(1, "%s", logerror);
475 free(logerror);
476 logerror = NULL;
477 notify(1, "Improper login to %s account",
478 thisC->Clogin ? thisC->Clogin : "default");
479 done(FDONESRVERROR, "Improper login");
480 }
481 if (netcrypt(thisC->Ccrypt) != SCMOK) /* restore encryption */
482 goaway("Running non-crypting sup");
483}
484/*
485 * send list of files that we are not interested in. receive list of
486 * files that are on the repository that could be upgraded. Find the
487 * ones that we need. Receive the list of files that the server could
488 * not access. Delete any files that have been upgraded in the past
489 * which are no longer on the repository.
490 */
491
492void
493listfiles(void)
494{
495 char buf[STRINGLENGTH];
496 char relsufix[STRINGLENGTH];
497 char *p, *q;
498 FILE *f;
499 int x;
500
501
502 if ((thisC->Cflags & CFURELSUF) && release)
503 (void) snprintf(relsufix, sizeof(relsufix), ".%s", release);
504 else
505 relsufix[0] = '\0';
506 (void) snprintf(buf, sizeof(buf), FILELAST, collname, relsufix);
507 f = fopen(buf, "r");
508 if (f) {
509 while ((p = fgets(buf, STRINGLENGTH, f))) {
510 if ((q = strchr(p, '\n')))
511 *q = '\0';
512 else {
513 p[512] = '\0';
514 goaway("Line too long in LAST: %s", p);
515 }
516 if (strchr("#;:", *p))
517 continue;
518 if (canonicalize(p) != 0)
519 continue;
520 (void) Tinsert(&lastT, p, FALSE);
521 }
522 (void) fclose(f);
523 }
524 refuseT = NULL;
525 (void) snprintf(buf, sizeof(buf), FILEREFUSE, collname);
526 f = fopen(buf, "r");
527 if (f) {
528 while ((p = fgets(buf, STRINGLENGTH, f))) {
529 if ((q = strchr(p, '\n')))
530 *q = '\0';
531 else {
532 p[512] = '\0';
533 goaway("Line too long in REFUSE: %s", p);
534 }
535 if (strchr("#;:", *p))
536 continue;
537 (void) Tinsert(&refuseT, p, FALSE);
538 }
539 (void) fclose(f);
540 }
541 vnotify(0, "Requesting changes since %s", supctime(&lasttime) + 4);
542 x = msgrefuse();
543 if (x != SCMOK)
544 goaway("Error sending refuse list to file server");
545 listT = NULL;
546 x = msglist();
547 if (x != SCMOK)
548 goaway("Error reading file list from file server");
549 if (thisC->Cprefix)
550 if (chdir(thisC->Cprefix) < 0)
551 goaway("Can't chdir to %s (%s)",
552 thisC->Cprefix, strerror(errno));
553 needT = NULL;
554 (void) Tprocess(listT, needone, NULL);
555 Tfree(&listT);
556 x = msgneed();
557 if (x != SCMOK)
558 goaway("Error sending needed files list to file server");
559 Tfree(&needT);
560 denyT = NULL;
561 x = msgdeny();
562 if (x != SCMOK)
563 goaway("Error reading denied files list from file server");
564 if (thisC->Cflags & CFVERBOSE)
565 (void) Tprocess(denyT, denyone, NULL);
566 Tfree(&denyT);
567 if (thisC->Cflags & (CFALL | CFDELETE | CFOLD))
568 (void) Trprocess(lastT, deleteone, NULL);
569 Tfree(&refuseT);
570}

572static int
573needone(TREE * t, void *dummy __unused)
574{
575 TREE *newt;
576 int exists, fetch;
577 struct stat sbuf;
578
579 newt = Tinsert(&lastT, t->Tname, TRUE);
580 newt->Tflags |= FUPDATE;
581 fetch = TRUE;
582 if ((thisC->Cflags & CFALL) == 0) {
583 if ((t->Tflags & FNEW) == 0 && (thisC->Cflags & CFOLD) == 0)
584 return (SCMOK);
585 if (S_ISLNK(t->Tmode))
586 exists = (lstat(t->Tname, &sbuf) == 0);
587 else
588 exists = (stat(t->Tname, &sbuf) == 0);
589 /* This is moderately complicated: If the file is the wrong
590 * type or doesn't exist, we need to fetch the whole file. If
591 * the file is a special file, we rely solely on the server:
592 * if the file changed, we do an update; otherwise nothing. If
593 * the file is a normal file, we check timestamps. If we are
594 * in "keep" mode, we fetch if the file on the server is
595 * newer, and do nothing otherwise. Otherwise, we fetch if the
596 * timestamp is wrong; if the file changed on the server but
597 * the timestamp is right, we do an update. (Update refers to
598 * updating stat information, i.e. timestamp, owner, mode
599 * bits, etc.) */
600 if (exists && (sbuf.st_mode & S_IFMT) == (t->Tmode & S_IFMT)) {
601 if (!S_ISREG(t->Tmode))
602 if (t->Tflags & FNEW)
603 fetch = FALSE;
604 else
605 return (SCMOK);
606 else if ((thisC->Cflags & CFKEEP) &&
607 sbuf.st_mtime > t->Tmtime)
608 return (SCMOK);
609 else if (sbuf.st_mtime == t->Tmtime) {
610 if (t->Tflags & FNEW)
611 fetch = FALSE;
612 else
613 return (SCMOK);
614 }
615 }
616 }
617 /* If we get this far, we're either doing an update or a full fetch. */
618 newt = Tinsert(&needT, t->Tname, TRUE);
619 if (!fetch && S_ISREG(t->Tmode))
620 newt->Tflags |= FUPDATE;
621 return (SCMOK);
622}

631static int
632deleteone(TREE * t, void *v __unused)
633{
634 struct stat sbuf, pbuf;
635 int x;
636 char *name = t->Tname;
637 char pname[MAXPATHLEN];
638
639 if (t->Tflags & FUPDATE)/* in current upgrade list */
640 return (SCMOK);
641 if (lstat(name, &sbuf) < 0) /* doesn't exist */
642 return (SCMOK);
643 /* is it a symbolic link ? */
644 if (S_ISLNK(sbuf.st_mode)) {
645 if (Tlookup(refuseT, name)) {
646 vnotify(0, "Would not delete symbolic link %s",
647 name);
648 return (SCMOK);
649 }
650 if (thisC->Cflags & CFLIST) {
651 vnotify(0, "Would delete symbolic link %s", name);
652 return (SCMOK);
653 }
654 if ((thisC->Cflags & CFDELETE) == 0) {
655 notify(0, "Please delete symbolic link %s", name);
656 t->Tflags |= FUPDATE;
657 return (SCMOK);
658 }
659 x = unlink(name);
660 if (x < 0) {
661 notify(1, "Unable to delete symbolic link %s (%s)",
662 name, strerror(errno));
663 t->Tflags |= FUPDATE;
664 return (SCMOK);
665 }
666 vnotify(0, "Deleted symbolic link %s", name);
667 return (SCMOK);
668 }
669 /* is it a directory ? */
670 if (S_ISDIR(sbuf.st_mode)) {
671 if (Tlookup(refuseT, name)) {
672 vnotify(0, "Would not delete directory %s", name);
673 return (SCMOK);
674 }
675 if (thisC->Cflags & CFLIST) {
676 vnotify(0, "Would delete directory %s", name);
677 return (SCMOK);
678 }
679 if ((thisC->Cflags & CFDELETE) == 0) {
680 notify(0, "Please delete directory %s", name);
681 t->Tflags |= FUPDATE;
682 return (SCMOK);
683 }
684 if (rmdir(name) < 0) {
685 (void) chmod(name, sbuf.st_mode | S_IRWXU);
686 snprintf(pname, sizeof(pname), "%s/..", name);
687 if (stat(pname, &pbuf) == 0)
688 (void) chmod(pname, pbuf.st_mode | S_IRWXU);
689 runp("rm", "rm", "-rf", name, 0);
690 }
691 if (rmdir(name) < 0 && errno != ENOENT) {
692 notify(1, "Unable to delete directory %s (%s)",
693 name, strerror(errno));
694 t->Tflags |= FUPDATE;
695 return (SCMOK);
696 }
697 vnotify(0, "Deleted directory %s", name);
698 return (SCMOK);
699 }
700 /* it is a file */
701 if (Tlookup(refuseT, name)) {
702 vnotify(0, "Would not delete file %s", name);

1177/*
1178 * We know that since "to" is a pathname coming from the server, it must
1179 * not contain any symbolic links after the root, because otherwise the
1180 * server would send us only the symlink above it. So we hunt for the symlink
1181 * above and if found we convert the symlink to a directory, prepare the
1182 * path below the symlink, and keep
1183 */
1184static int
1185canonicalize(const char *to)
1186{
1187 char absto[STRINGLENGTH], cabsto[STRINGLENGTH * 4];
1188 char dir[STRINGLENGTH], file[STRINGLENGTH];
1189 char *a;
1190 char *c;
1191 size_t len;
1192 struct stat st;
1193 const char *pwd = thisC->Cprefix ? thisC->Cprefix : thisC->Cbase;
1194
1195 if ((thisC->Cflags & CFCANONICALIZE) == 0)
1196 return 0;
1197
1198 path(to, dir, file);
1199
1200 len = strlen(pwd);
1201 (void)snprintf(absto, sizeof(absto), "%s/%s", pwd, dir);
1202
1203 len++;
1204 if (realpath(absto, cabsto) == NULL)
1205 return -1;
1206
1207 a = absto + len;
1208 c = cabsto + len;
1209
1210 while (*a && *c && *a == *c)
1211 a++, c++;
1212
1213 if (*a == '\0' && *c == '\0')
1214 return 0;
1215
1216 while (*a && *a != '/')
1217 a++;
1218
1219 *a = '\0';
1220 if (lstat(absto, &st) == -1 || !S_ISLNK(st.st_mode))
1221 return -1;
1222
1223 if (unlink(absto) == -1)
1224 return -1;
1225
1226 strcpy(c, a);
1227 if (estabd(file, cabsto) == -1) {
1228 return -1;
1229 }
1230 return 1;
1231}

1354 goaway("Error in server space check");
1355 logquit(1, "Error in server space check");
1356 }
1357 errno = 0;
1358 if (ioctl(tof, FIOCFSPARAM, (char *) &fsp) < 0 &&
1359 errno != EINVAL) {
1360 if (istemp)
1361 (void) unlink(tname);
1362 lockout(FALSE);
1363 x = readskip();
1364 if (x != SCMOK)
1365 goaway("Can't skip file transfer");
1366 goaway("Error in disk space check");
1367 logquit(1, "Error in disk space check");
1368 }
1369 if (errno == 0) {
1370 fsize = (fsize + 1023) / 1024;
1371 x = fsp.fsp_size * MAX(fsp.fsp_minfree, 1) / 100;
1372 fsp.fsp_free -= x;
1373 if (fsize > MAX(fsp.fsp_free, 0)) {
1374 if (istemp)
1375 (void) unlink(tname);
1376 lockout(FALSE);
1377 x = readskip();
1378 if (x != SCMOK)
1379 goaway("Can't skip file transfer");
1380 goaway("No disk space for file %s", to);
1381 logquit(1, "No disk space for file %s", to);
1382 }
1383 }
1384 }
1385#endif /* MACH */
1386 x = readfile(tof);
1387 (void) close(tof);
1388 if (x != SCMOK) {
1389 if (istemp)
1390 (void) unlink(tname);
1391 lockout(FALSE);
1392 goaway("Error in receiving %s", to);
1393 }
1394 }
1395 if (!istemp) { /* no temp file used */
1396 lockout(FALSE);
1397 return (FALSE);
1398 }
1399 /*
1400 ** If the file is compressed, uncompress it in place. We open the
1401 ** temp file for reading, unlink the file, and then open the same
1402 ** file again for writing. Then we pipe through gzip. When
1403 ** finished the temp file contains the uncompressed version and we
1404 ** can continue as before.
1405 **
1406 ** Since sup prefers to write close to the original file the
1407 ** benefits of atomic updates probably outweigh the cost of the
1408 ** extra filecopy which occurs when the temp file is on a different
1409 ** filesystem from the original.
1410 */
1411 if (docompress) {
1412 char *av[4];
1413 int ac = 0;
1414 int infd = -1;
1415 int outfd = -1;
1416 av[ac++] = "gzip";
1417 av[ac++] = "-d";
1418 av[ac++] = NULL;
1419 if ((infd = open(tname, O_RDONLY, 0)) == -1 ||
1420 unlink(tname) == -1 ||
1421 (outfd = open(tname, O_WRONLY | O_CREAT | O_TRUNC | O_EXCL, 0600)) == -1 ||
1422 runiofd(av, infd, outfd, 2) != 0) {
1423 notify(1, "Error in uncompressing file %s (%s)",
1424 to, tname);
1425 (void) unlink(tname);
1426 if (infd != -1)
1427 (void) close(infd);
1428 if (outfd != -1)
1429 (void) close(outfd);
1430 lockout(FALSE);
1431 return (TRUE);
1432 }
1433 (void) close(infd);
1434 (void) close(outfd);
1435 }
1436 /* move to destination */
1437 if (rename(tname, to) == 0) {
1438 (void) unlink(tname);
1439 lockout(FALSE);
1440 return (FALSE);
1441 }
1442 fromf = open(tname, O_RDONLY, 0);
1443 if (fromf < 0) {
1444 notify(1, "Error in moving temp file to %s (%s)",
1445 to, strerror(errno));
1446 (void) unlink(tname);
1447 lockout(FALSE);
1448 return (TRUE);
1449 }
1450 tof = open(to, (O_WRONLY | O_CREAT | O_TRUNC | O_EXCL), 0600);
1451 if (tof < 0) {
1452 (void) close(fromf);
1453 notify(1, "Can't create %s from temp file (%s)",
1454 to, strerror(errno));
1455 (void) unlink(tname);
1456 lockout(FALSE);
1457 return (TRUE);
1458 }
1459 x = filecopy(fromf, tof);
1460 (void) close(fromf);
1461 (void) close(tof);
1462 (void) unlink(tname);
1463 lockout(FALSE);
1464 if (x < 0) {
1465 notify(1, "Error in storing data in %s", to);
1466 return (TRUE);
1467 }
1468 return (FALSE);
1469}

1474{

1483 (void) snprintf(collrelname, sizeof(collrelname), "%s-%s",
1484 collname, release);
1485 } else {
1486 relsufix[0] = '\0';
1487 (void) strcpy(collrelname, collname);
1488 }
1489 dontjump = TRUE; /* once here, no more longjmp */
1490 (void) netcrypt(NULL);
1491 if (protver < 6) {
1492 /* done with server */
1493 if (x == SCMOK)
1494 goaway(NULL);
1495 (void) requestend();
1496 }
1497 tloc = time(NULL);
1498 if (x != SCMOK) {
1499 notify(1, "Upgrade of %s aborted at %s",
1500 collrelname, supctime(&tloc) + 4);
1501 Tfree(&lastT);
1502 if (protver < 6)
1503 return;
1504 /* if we've not been blown off, make sure he is! */
1505 if (x != SCMEOF)
1506 goaway("Aborted");
1507 (void) requestend();
1508 return;
1509 }
1510 if (thisC->Cnogood) {
1511 notify(1, "Upgrade of %s completed with errors at %s",
1512 collrelname, supctime(&tloc) + 4);
1513 notify(1, "Upgrade time will not be updated");
1514 Tfree(&lastT);
1515 if (protver < 6)
1516 return;
1517 done(FDONEUSRERROR, "Completed with errors");
1518 (void) requestend();
1519 return;
1520 }
1521 if (thisC->Cprefix)
1522 if (chdir(thisC->Cbase) < 0)
1523 goaway("Can't chdir to %s (%s)", thisC->Cbase,
1524 strerror(errno));
1525 vnotify(0, "Upgrade of %s completed at %s",
1526 collrelname, supctime(&tloc) + 4);
1527 if (thisC->Cflags & CFLIST) {
1528 Tfree(&lastT);
1529 if (protver < 6)
1530 return;
1531 done(FDONEDONTLOG, "List only");
1532 (void) requestend();
1533 return;
1534 }
1535 (void) snprintf(fname, sizeof(fname), FILEWHEN, collname, relsufix);
1536 if (establishdir(fname)) {
1537 int oerrno = errno;
1538 Tfree(&lastT);
1539 if (protver < 6)
1540 return;
1541 done(FDONEUSRERROR, "Couldn't create directory `%s' (%s)",
1542 fname, strerror(oerrno));
1543 (void) requestend();
1544 return;
1545 }
1546 if (!putwhen(fname, scantime)) {
1547 int oerrno = errno;
1548 notify(1, "Can't record current time in %s (%s)",
1549 fname, strerror(oerrno));
1550 Tfree(&lastT);
1551 if (protver < 6)
1552 return;
1553 done(FDONEUSRERROR, "Couldn't timestamp `%s' (%s)",
1554 fname, strerror(oerrno));
1555 (void) requestend();
1556 return;
1557 }
1558 if (protver >= 6) {
1559 /* At this point we have let the server go */
1560 /* "I'm sorry, we've had to let you go" */
1561 done(FDONESUCCESS, "Success");
1562 (void) requestend();
1563 }
1564 (void) snprintf(tname, sizeof(tname), FILELASTTEMP, collname, relsufix);
1565 finishfile = fopen(tname, "w");
1566 if (finishfile == NULL) {
1567 notify(1, "Can't record list of all files in %s", tname);
1568 Tfree(&lastT);
1569 return;
1570 }
1571 (void) Tprocess(lastT, finishone, finishfile);
1572 (void) fclose(finishfile);

1575 notify(1, "Can't change %s to %s (%s)", tname, fname,
1576 strerror(errno));
1577 (void) unlink(tname);
1578 Tfree(&lastT);
1579}
