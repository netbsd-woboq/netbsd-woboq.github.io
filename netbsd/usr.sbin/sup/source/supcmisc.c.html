54struct liststruct { /* uid and gid lists */
55 char *Lname; /* name */
56 int Lnumber; /* uid or gid */
57 struct liststruct *Lnext;
58};
59typedef struct liststruct LIST;

61#define HASHBITS 4
62#define HASHSIZE (1<<HASHBITS)
63#define HASHMASK (HASHSIZE-1)
64#define LISTSIZE (HASHSIZE*HASHSIZE)

80void
81prtime(void)
82{
83 char buf[STRINGLENGTH];
84 char relsufix[STRINGLENGTH];
85 time_t twhen;
86
87 if ((thisC->Cflags & CFURELSUF) && thisC->Crelease)
88 (void) sprintf(relsufix, ".%s", thisC->Crelease);
89 else
90 relsufix[0] = '\0';
91 if (chdir(thisC->Cbase) < 0)
92 logerr("Can't change to base directory %s for collection %s",
93 thisC->Cbase, thisC->Cname);
94 twhen = getwhen(thisC->Cname, relsufix);
95 (void) strcpy(buf, ctime(&twhen));
96 buf[strlen(buf) - 1] = '\0';
97 loginfo("Last update occurred at %s for collection %s%s",
98 buf, thisC->Cname, relsufix);
99}

109int
110makedir(char *fname, unsigned int mode, struct stat * statp)
111{
112 int en, rv;
113
114 if (lstat(fname, statp) != -1 && !S_ISDIR(statp->st_mode)) {
115 if (unlink(fname) == -1) {
116 notify(1, "Can't delete %s (%s)", fname,
117 strerror(errno));
118 return -1;
119 }
120 }
121 if (mkdir(fname, mode) == -1)
122 en = errno;
123 else
124 en = -1;
125
126 rv = stat(fname, statp);
127
128 if (en != -1)
129 errno = en;
130
131 return rv;
132}

134int
135estabd(char *fname, char *dname)
136{
137 char dpart[STRINGLENGTH], fpart[STRINGLENGTH];
138 struct stat sbuf;
139 int x;
140
141 if (stat(dname, &sbuf) >= 0)
142 return (FALSE); /* exists */
143 path(dname, dpart, fpart);
144 if (strcmp(fpart, ".") == 0) { /* dname is / or . */
145 notify(1, "Can't create directory %s for %s (Invalid name)",
146 dname, fname);
147 errno = EINVAL;
148 return (TRUE);
149 }
150 x = estabd(fname, dpart);
151 if (x)
152 return (TRUE);
153 if (makedir(dname, 0755, &sbuf) < 0) {
154 int oerrno = errno;
155 notify(1, "Can't create directory %s for %s (%s)", dname,
156 fname, strerror(errno));
157 errno = oerrno;
158 return TRUE;
159 }
160 vnotify(0, "Created directory %s for %s", dname, fname);
161 return (FALSE);
162}

167static int
168Lhash(char *name)
169{
170 /* Hash function is: HASHSIZE * (strlen mod HASHSIZE) +
171 * (char mod HASHSIZE) where "char" is last character of name (if
172 * name is non-null). */
173
174 int len;
175 char c;
176
177 len = strlen(name);
178 if (len > 0)
179 c = name[len - 1];
180 else
181 c = 0;
182 return (((len & HASHMASK) << HASHBITS) | (((int) c) & HASHMASK));
183}
184
185static void
186Linsert(LIST ** table, char *name, int number)
187{
188 LIST *l;
189 int lno;
190 lno = Lhash(name);
191 l = (LIST *) malloc(sizeof(LIST));
192 if (l == NULL)
193 goaway("Cannot allocate memory");
194 l->Lname = name;
195 l->Lnumber = number;
196 l->Lnext = table[lno];
197 table[lno] = l;
198}
199
200static LIST *
201Llookup(LIST ** table, char *name)
202{
203 int lno;
204 LIST *l;
205 lno = Lhash(name);
206 for (l = table[lno]; l && strcmp(l->Lname, name) != 0; l = l->Lnext);
207 return (l);
208}

210void
211ugconvert(char *uname, char *gname, int *uid, int *gid, int *mode)
212{
213 LIST *u, *g;
214 struct passwd *pw;
215 struct group *gr;
216 struct stat sbuf;
217 static int defuid = -1;
218 static int defgid;
219 static int first = TRUE;
220
221 if (first) {
222 bzero(uidL, sizeof(uidL));
223 bzero(gidL, sizeof(gidL));
224 first = FALSE;
225 }
226 pw = NULL;
227 if ((u = Llookup(uidL, uname)) != NULL)
228 *uid = u->Lnumber;
229 else if ((pw = getpwnam(uname)) != NULL) {
230 Linsert(uidL, estrdup(uname), pw->pw_uid);
231 *uid = pw->pw_uid;
232 }
233 if (u || pw) {
234 if ((g = Llookup(gidL, gname)) != NULL) {
235 *gid = g->Lnumber;
236 return;
237 }
238 if ((gr = getgrnam(gname)) != NULL) {
239 Linsert(gidL, estrdup(gname), gr->gr_gid);
240 *gid = gr->gr_gid;
241 return;
242 }
243 if (pw == NULL)
244 pw = getpwnam(uname);
245 *mode &= ~S_ISGID;
246 *gid = pw->pw_gid;
247 return;
248 }
249 *mode &= ~(S_ISUID | S_ISGID);
250 if (defuid >= 0) {
251 *uid = defuid;
252 *gid = defgid;
253 return;
254 }
255 if (stat(".", &sbuf) < 0) {
256 *uid = defuid = getuid();
257 *gid = defgid = getgid();
258 return;
259 }
260 *uid = defuid = sbuf.st_uid;
261 *gid = defgid = sbuf.st_gid;
262}

269void
270notify(int f, const char *fmt, ...)
271{ /* record error message */
272 char buf[STRINGLENGTH];
273 char collrelname[STRINGLENGTH];
274 char hostname[STRINGLENGTH];
275 time_t tloc;
276 static FILE *noteF = NULL; /* mail program on pipe */
277 va_list ap;
278 int shouldMail = (thisC->Cflags & CFMAIL) && thisC->Cnotify;
279 int needFile = shouldMail || silent;
280
281 (void)gethostname(hostname, sizeof(hostname));
282 va_start(ap, fmt);
283
284 if ((thisC->Cflags & CFURELSUF) && thisC->Crelease)
285 (void) sprintf(collrelname, "%s-%s", collname, thisC->Crelease);
286 else
287 (void) strcpy(collrelname, collname);
288
289 if (fmt == NULL) {
290 if (noteF && noteF != stdout && (!silent || thisC->Cnogood)) {
291 int nr;
292 FILE *outF;
293
294 if (shouldMail) {
295 (void) snprintf(buf, sizeof(buf),
296 "mail -s \"SUP Upgrade of %s on %s\" %s >"
297 " /dev/null", collrelname, hostname,
298 thisC->Cnotify);
299 outF = popen(buf, "w");
300 if (outF == NULL) {
301 logerr("Can't send mail to %s for %s",
302 thisC->Cnotify, collrelname);
303 outF = stdout;
304 }
305 } else
306 outF = stdout;
307
308 (void)rewind(noteF);
309 while ((nr = fread(buf, 1, sizeof(buf), noteF)) > 0)
310 (void)fwrite(buf, 1, nr, outF);
311 (void)fflush(outF);
312 if (outF != stdout)
313 (void)pclose(outF);
314 (void)fclose(noteF);
315 }
316 noteF = NULL;
317 va_end(ap);
318 return;
319 }
320
321 if (noteF == NULL) {
322 if (needFile) {
323 char template[] = "/tmp/sup.XXXXXX";
324 int fd = mkstemp(template);
325 if (fd == -1 || (noteF = fdopen(fd, "r+")) == NULL) {
326 logerr("Can't open temporary file for %s",
327 collrelname);
328 silent = 0;
329 noteF = stdout;
330 }
331 (void)unlink(template);
332 } else
333 noteF = stdout;
334 tloc = time(NULL);
335 fprintf(noteF, "SUP@%s Upgrade of %s at %s", hostname,
336 collrelname, ctime(&tloc));
337 (void) fflush(noteF);
338 }
339 fprintf(noteF, "SUP@%s%s ", hostname, f ? ":" : "");
340 vfprintf(noteF, fmt, ap);
341 fprintf(noteF, "\n");
342 va_end(ap);
343 (void) fflush(noteF);
344}

346void
347lockout(int on)
348{ /* lock out interrupts */
349 static sigset_t oset;
350 sigset_t nset;
351
352 if (on) {
353 sigemptyset(&nset);
354 sigaddset(&nset, SIGHUP);
355 sigaddset(&nset, SIGINT);
356 sigaddset(&nset, SIGTERM);
357 sigaddset(&nset, SIGQUIT);
358 (void) sigprocmask(SIG_BLOCK, &nset, &oset);
359 } else {
360 (void) sigprocmask(SIG_SETMASK, &oset, NULL);
361 }
362}
