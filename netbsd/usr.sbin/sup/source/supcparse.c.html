66extern char _argbreak; /* break character from nxtarg */
67
68typedef enum { /* supfile options */
69 OHOST, OBASE, OHOSTBASE, OPREFIX, ORELEASE,
70 ONOTIFY, OLOGIN, OPASSWORD, OCRYPT,
71 OBACKUP, ODELETE, OEXECUTE, OOLD, OTIMEOUT, OKEEP, OURELSUF,
72 OCOMPRESS, OCANONICALIZE, OIGNCHERR,
73} OPTION;
74
75struct option {
76 char *op_name;
77 OPTION op_enum;
78} options[] = {
79 { "host", OHOST },
80 { "base", OBASE },
81 { "hostbase", OHOSTBASE },
82 { "prefix", OPREFIX },
83 { "release", ORELEASE },
84 { "notify", ONOTIFY },
85 { "login", OLOGIN },
86 { "password", OPASSWORD },
87 { "crypt", OCRYPT },
88 { "backup", OBACKUP },
89 { "delete", ODELETE },
90 { "execute", OEXECUTE },
91 { "old", OOLD },
92 { "timeout", OTIMEOUT },
93 { "keep", OKEEP },
94 { "use-rel-suffix", OURELSUF },
95 { "compress", OCOMPRESS },
96 { "canonicalize", OCANONICALIZE },
97 { "igncherr", OIGNCHERR },
98};
99
100static void passdelim(char **, char);
101
102static void
103passdelim(char **ptr, char delim)
104{ /* skip over delimiter */
105 *ptr = skipover(*ptr, " \t");
106 if (_argbreak != delim && **ptr == delim) {
107 (*ptr)++;
108 *ptr = skipover(*ptr, " \t");
109 }
110}

112int
113parsecoll(COLLECTION * c, char *collname, char *args)
114{
115 char *arg, *p;
116 OPTION option;
117 int opno;
118
119 c->Cnext = NULL;
120 c->Cname = estrdup(collname);
121 c->Chost = NULL;
122 c->Chtree = NULL;
123 c->Cbase = NULL;
124 c->Chbase = NULL;
125 c->Cprefix = NULL;
126 c->Crelease = NULL;
127 c->Cnotify = NULL;
128 c->Clogin = NULL;
129 c->Cpswd = NULL;
130 c->Ccrypt = NULL;
131 c->Ctimeout = 3 * 60 * 60; /* default to 3 hours instead of no
132 * timeout */
133 c->Cflags = 0;
134 c->Cnogood = FALSE;
135 c->Clockfd = -1;
136 args = skipover(args, " \t");
137 while (*(arg = nxtarg(&args, " \t="))) {
138 for (opno = 0; opno < sizeofA(options); opno++)
139 if (strcmp(arg, options[opno].op_name) == 0)
140 break;
141 if (opno == sizeofA(options)) {
142 logerr("Invalid supfile option %s for collection %s",
143 arg, c->Cname);
144 return (-1);
145 }
146 option = options[opno].op_enum;
147 switch (option) {
148 case OHOST:
149 passdelim(&args, '=');
150 do {
151 arg = nxtarg(&args, ", \t");
152 (void) Tinsert(&c->Chtree, arg, FALSE);
153 arg = args;
154 p = skipover(args, " \t");
155 if (*p++ == ',')
156 args = p;
157 } while (arg != args);
158 break;
159 case OBASE:
160 passdelim(&args, '=');
161 arg = nxtarg(&args, " \t");
162 c->Cbase = estrdup(arg);
163 break;
164 case OHOSTBASE:
165 passdelim(&args, '=');
166 arg = nxtarg(&args, " \t");
167 c->Chbase = estrdup(arg);
168 break;
169 case OPREFIX:
170 passdelim(&args, '=');
171 arg = nxtarg(&args, " \t");
172 c->Cprefix = estrdup(arg);
173 break;
174 case ORELEASE:
175 passdelim(&args, '=');
176 arg = nxtarg(&args, " \t");
177 c->Crelease = estrdup(arg);
178 break;
179 case ONOTIFY:
180 passdelim(&args, '=');
181 arg = nxtarg(&args, " \t");
182 c->Cnotify = estrdup(arg);
183 break;
184 case OLOGIN:
185 passdelim(&args, '=');
186 arg = nxtarg(&args, " \t");
187 c->Clogin = estrdup(arg);
188 break;
189 case OPASSWORD:
190 passdelim(&args, '=');
191 arg = nxtarg(&args, " \t");
192 c->Cpswd = estrdup(arg);
193 break;
194 case OCRYPT:
195 passdelim(&args, '=');
196 arg = nxtarg(&args, " \t");
197 c->Ccrypt = estrdup(arg);
198 break;
199 case OBACKUP:
200 c->Cflags |= CFBACKUP;
201 break;
202 case ODELETE:
203 c->Cflags |= CFDELETE;
204 break;
205 case OEXECUTE:
206 c->Cflags |= CFEXECUTE;
207 break;
208 case OOLD:
209 c->Cflags |= CFOLD;
210 break;
211 case OKEEP:
212 c->Cflags |= CFKEEP;
213 break;
214 case OURELSUF:
215 c->Cflags |= CFURELSUF;
216 break;
217 case OCOMPRESS:
218 c->Cflags |= CFCOMPRESS;
219 break;
220 case OCANONICALIZE:
221 c->Cflags |= CFCANONICALIZE;
222 break;
223 case OTIMEOUT:
224 passdelim(&args, '=');
225 arg = nxtarg(&args, " \t");
226 c->Ctimeout = atoi(arg);
227 break;
228 case OIGNCHERR:
229 c->Cflags |= CFIGNCHERR;
230 break;
231 }
232 }
233 return (0);
234}

237time_t
238getwhen(char *collection, char *relsuffix)
239{
240 char buf[STRINGLENGTH];
241 char *ep;
242 FILE *fp;
243 time_t tstamp;
244
245 (void) sprintf(buf, FILEWHEN, collection, relsuffix);
246
247 if ((fp = fopen(buf, "r")) == NULL)
248 return 0;
249
250 if (fgets(buf, sizeof(buf), fp) == NULL) {
251 (void) fclose(fp);
252 return 0;
253 }
254 (void) fclose(fp);
255
256 if ((tstamp = strtol(buf, &ep, 0)) == -1 || *ep != '\n')
257 return 0;
258
259 return tstamp;
260}
261
262int
263putwhen(char *fname, time_t tstamp)
264{
265 FILE *fp;
266 if ((fp = fopen(fname, "w")) == NULL)
267 return 0;
268 if (fprintf(fp, "%lu\n", (u_long) tstamp) < 0) {
269 (void) fclose(fp);
270 return 0;
271 }
272 if (fclose(fp) != 0)
273 return 0;
274 return 1;
275}
