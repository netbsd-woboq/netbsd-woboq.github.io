377static void
378addchild(pid_t pid)
379{
380 size_t i;
381 for (i = 0; i < maxchildren; i++)
382 if (hp[i].pid == 0) {
383 hp[i].pid = pid;
384 strcpy(hp[i].name, remotehost());
385 nchildren++;
386 return;
387 }
388 logerr("Out of space adding child %s", remotehost());
389}
390
391static void
392removechild(pid_t pid)
393{
394 size_t i;
395 for (i = 0; i < maxchildren; i++)
396 if (hp[i].pid == pid) {
397 hp[i].pid = 0;
398 nchildren--;
399 return;
400 }
401 logerr("Child with pid %jd not found", (intmax_t)pid);
402}
403
404static int
405checkchild(void)
406{
407 const char *h = remotehost();
408 size_t i;
409 for (i = 0; i < maxchildren; i++)
410 if (hp[i].pid && strcmp(hp[i].name, h) == 0) {
411 logerr("Ignoring connection frm %s", h);
412 return 0;
413 }
414 return 1;
415}

421int
422main(int argc, char **argv)
423{
424 int x, pid;
425 sigset_t nset, oset;
426 struct sigaction chld, ign;
427 time_t tloc;
428#ifdef LIBWRAP
429 struct request_info req;
430#endif
431
432 /* initialize global variables */
433 pgmversion = PGMVERSION;/* export version number */
434 isserver = TRUE; /* export that we're not a server */
435 collname = NULL; /* no current collection yet */
436 maxchildren = MAXCHILDREN; /* defined in sup.h */
437
438 init(argc, argv); /* process arguments */
439
440#ifdef HAS_DAEMON
441 if (!live) /* if not debugging, turn into daemon */
442 if (daemon(0, 0) == -1)
443 goaway("Daemon failed (%s)", strerror(errno));
444#endif
445 hp = malloc(sizeof(*hp) * maxchildren);
446 if (hp == NULL)
447 goaway("Cannot allocate memory");
448
449 logopen("supfile");
450 tloc = time(NULL);
451 loginfo("SUP File Server Version %d.%d (%s) starting at %s",
452 PROTOVERSION, PGMVERSION, scmversion, fmttime(tloc));
453 if (live) {
454 x = service();
455
456 if (x != SCMOK)
457 logquit(1, "Can't connect to network");
458#ifdef LIBWRAP
459 request_init(&req, RQ_DAEMON, "supfilesrv", RQ_FILE, netfile,
460 NULL);
461 fromhost(&req);
462 if (hosts_access(&req) == 0) {
463 logdeny("refused connection from %.500s",
464 eval_client(&req));
465 servicekill();
466 exit(1);
467 }
468 if (sup_clog) {
469 logallow("connection from %.500s", eval_client(&req));
470 }
471#endif
472 answer();
473 (void) serviceend();
474 exit(0);
475 }
476 ign.sa_handler = SIG_IGN;
477 sigemptyset(&ign.sa_mask);
478 ign.sa_flags = 0;
479 (void) sigaction(SIGHUP, &ign, NULL);
480 (void) sigaction(SIGINT, &ign, NULL);
481 (void) sigaction(SIGPIPE, &ign, NULL);
482 chld.sa_handler = chldsig;
483 sigemptyset(&chld.sa_mask);
484 chld.sa_flags = 0;
485 (void) sigaction(SIGCHLD, &chld, NULL);
486 nchildren = 0;
487 for (;;) {
488 x = service();
489 if (x != SCMOK) {
490 logerr("Error in establishing network connection");
491 (void) servicekill();
492 continue;
493 }
494 /*
495 * If we are being bombarded, don't even spend time forking
496 * or conversing
497 */
498 if (nchildren > maxchildren) {
499 (void) servicekill();
500 continue;
501 }
502 sigemptyset(&nset);
503 sigaddset(&nset, SIGCHLD);
504 sigprocmask(SIG_BLOCK, &nset, &oset);
505 if ((pid = fork()) == 0) { /* server process */
506#ifdef LIBWRAP
507 request_init(&req, RQ_DAEMON, "supfilesrv", RQ_FILE,
508 netfile, NULL);
509 fromhost(&req);
510 if (hosts_access(&req) == 0) {
511 logdeny("refused connection from %.500s",
512 eval_client(&req));
513 servicekill();
514 exit(1);
515 }
516 if (sup_clog) {
517 logallow("connection from %.500s",
518 eval_client(&req));
519 }
520#endif
521 (void) serviceprep();
522 answer();
523 (void) serviceend();
524 exit(0);
525 }
526 (void) servicekill(); /* parent */
527 if (pid > 0) {
528 addchild(pid);
529 setproctitle("Master [%d/%d]", nchildren, maxchildren);
530 }
531 (void) sigprocmask(SIG_SETMASK, &oset, NULL);
532 }
533}
534/*
535 * Child status signal handler
536 */
537
538void
539chldsig(int snum __unused)
540{
541 int w;
542 pid_t pid;
543
544 while ((pid = waitpid(-1, &w, WNOHANG)) > 0) {
545 if (kill(pid, 0) == -1)
546 switch (errno) {
547 case ESRCH:
548 if (nchildren == 0) {
549 logerr("no children but pid %jd\n",
550 (intmax_t)pid);
551 break;
552 }
553 removechild(pid);
554 break;
555 default:
556 logerr("killing pid %jd: (%s)\n", (intmax_t)
557 pid, strerror(errno));
558 break;
559 }
560
561 }
562}

577void
578init(int argc, char **argv)
579{
580 int i;
581 int x;
582 char *clienthost, *clientuser;
583 char *p, *q;
584 char buf[STRINGLENGTH];
585 int maxsleep;
586 FILE *f;
587 int af = AF_INET;
588
589#ifdef RCS
590 candorcs = FALSE;
591#endif
592 live = FALSE;
593#ifdef LIBWRAP
594 sup_clog = FALSE;
595#endif
596 dbgportsq = FALSE;
597 scmdebug = 0;
598 clienthost = NULL;
599 clientuser = NULL;
600 maxsleep = 5;
601 if (--argc < 0)
602 usage();
603 argv++;
604 while (clienthost == NULL && argc > 0 && argv[0][0] == '-') {
605 switch (argv[0][1]) {
606 case 'S':
607 silent = TRUE;
608 break;
609#ifdef LIBWRAP
610 case 'l':
611 sup_clog = TRUE;
612 break;
613#endif
614 case 'd':
615 live = TRUE;
616 break;
617 case 'P':
618 dbgportsq = TRUE;
619 break;
620 case 'N':
621 scmdebug++;
622 break;
623 case 'C':
624 if (--argc < 1)
625 quit(1, "Missing arg to -C\n");
626 argv++;
627 maxchildren = atoi(argv[0]);
628 break;
629 case 'H':
630 if (--argc < 3)
631 quit(1, "Missing args to -H\n");
632 argv++;
633 clienthost = argv[0];
634 clientuser = argv[1];
635 cryptkey = argv[2];
636 argc -= 2;
637 argv += 2;
638 break;
639#ifdef RCS
640 case 'R':
641 candorcs = TRUE;
642 break;
643#endif
644 case '4':
645 af = AF_INET;
646 break;
647#ifdef AF_INET6
648 case '6':
649 af = AF_INET6;
650 break;
651#endif
652 default:
653 fprintf(stderr, "Unknown flag %s ignored\n", argv[0]);
654 break;
655 }
656 --argc;
657 argv++;
658 }
659 if (clienthost == NULL) {
660 if (argc != 0)
661 usage();
662 x = servicesetup(dbgportsq ? DEBUGFPORT : FILEPORT, af);
663 if (x != SCMOK)
664 quit(1, "Error in network setup");
665 for (i = 0; i < HASHSIZE; i++)
666 uidH[i] = gidH[i] = inodeH[i] = NULL;
667 return;
668 }
669 isserver = FALSE;
670 if (argc < 1)
671 usage();
672 f = fopen(cryptkey, "r");
673 if (f == NULL)
674 quit(1, "Unable to open cryptfile %s\n", cryptkey);
675 if ((p = fgets(buf, STRINGLENGTH, f)) != NULL) {
676 if ((q = strchr(p, '\n')) != NULL)
677 *q = '\0';
678 if (*p == '\0')
679 quit(1, "No cryptkey found in %s\n", cryptkey);
680 cryptkey = estrdup(buf);
681 }
682 (void) fclose(f);
683 x = request(dbgportsq ? DEBUGFPORT : FILEPORT, clienthost, &maxsleep);
684 if (x != SCMOK)
685 quit(1, "Unable to connect to host %s\n", clienthost);
686 x = msgsignon();
687 if (x != SCMOK)
688 quit(1, "Error sending signon request to fileserver\n");
689 x = msgsignonack();
690 if (x != SCMOK)
691 quit(1, "Error reading signon reply from fileserver\n");
692 printf("SUP Fileserver %d.%d (%s) %d on %s\n",
693 protver, pgmver, scmver, fspid, remotehost());
694 free(scmver);
695 scmver = NULL;
696 if (protver < 7)
697 quit(1, "Remote fileserver does not implement reverse sup\n");
698 xpatch = TRUE;
699 xuser = clientuser;
700 x = msgsetup();
701 if (x != SCMOK)
702 quit(1, "Error sending setup request to fileserver\n");
703 x = msgsetupack();
704 if (x != SCMOK)
705 quit(1, "Error reading setup reply from fileserver\n");
706 switch (setupack) {
707 case FSETUPOK:
708 break;
709 case FSETUPSAME:
710 quit(1, "User %s not found on remote client\n", xuser);
711 case FSETUPHOST:
712 quit(1, "This host has no permission to reverse sup\n");
713 default:
714 quit(1, "Unrecognized file server setup status %d\n", setupack);
715 }
716 if (netcrypt(cryptkey) != SCMOK)
717 quit(1, "Running non-crypting fileserver\n");
718 crypttest = CRYPTTEST;

1891/*
1892 * Determine whether the file referenced by the file descriptor 'handle' can
1893 * be trusted, namely is it a file resident in the local file system.
1894 *
1895 * The main method of operation is to perform operations on the file
1896 * descriptor so that an attempt to spoof the checks should fail, for
1897 * example renamimg the file from underneath us and/or changing where the
1898 * file lives from underneath us.
1899 *
1900 * returns: -1 for error, indicating that we can not tell
1901 * 0 for file is definately not local, or it is an RFS link
1902 * 1 for file is local and can be trusted
1903 *
1904 * Side effect: copies the stat information into the supplied buffer,
1905 * regardless of the type of file system the file resides.
1906 *
1907 * Currently, the cases that we try to distinguish are RFS, AFS, NFS and
1908 * UFS, where the latter is considered a trusted file. We assume that the
1909 * caller has disabled link following and will detect an attempt to access
1910 * a file through an RFS link, except in the case the last component is
1911 * an RFS link. With link following disabled, the last component itself is
1912 * interpreted as a regular file if it is really an RFS link, so we
1913 * disallow the RFS link identified by group "symlink" and mode "IEXEC by
1914 * owner only". An AFS file is
1915 * detected by trying the VIOCIGETCELL ioctl, which is one of the few AFS
1916 * ioctls which operate on a file descriptor. Note, this AFS ioctl is
1917 * implemented in the cache manager, so the decision does not involve a
1918 * query with the AFS file server. An NFS file is detected by looking at
1919 * the major device number and seeing if it matches the known values for
1920 * MACH NSF/Sun OS 3.x or Sun OS 4.x.
1921 *
1922 * Having the fstatvfs() system call would make this routine easier and
1923 * more reliable.
1924 *
1925 * Note, in order to make the checks simpler, the file referenced by the
1926 * file descriptor can not be a BSD style symlink. Even with symlink
1927 * following of the last path component disabled, the attempt to open a
1928 * file which is a symlink will succeed, so we check for the BSD symlink
1929 * file type here. Also, the link following on/off and RFS file types
1930 * are only relevant in a MACH environment.
1931 */
1932#ifdef AFS
1933#include <sys/viceioctl.h>
1934#endif
1935
1936#define SYMLINK_GRP 64
1937
1938int
1939local_file(int handle, struct stat * sinfo)
1940{
1941 struct stat sb;
1942#ifdef VIOCIGETCELL
1943 /*
1944 * dummies for the AFS ioctl
1945 */
1946 struct ViceIoctl vdata;
1947 char cellname[512];
1948#endif /* VIOCIGETCELL */
1949
1950 if (fstat(handle, &sb) < 0)
1951 return (-1);
1952 if (sinfo != NULL)
1953 *sinfo = sb;
1954
1955#if CMUCS
1956 /*
1957 * If the following test succeeds, then the file referenced by
1958 * 'handle' is actually an RFS link, so we will not trust it.
1959 * See <sys/inode.h>.
1960 */
1961 if (sb.st_gid == SYMLINK_GRP
1962 && (sb.st_mode & (S_IFMT | S_IEXEC | (S_IEXEC >> 3) | (S_IEXEC >> 6)))
1963 == (S_IFREG | S_IEXEC))
1964 return (0);
1965#endif /* CMUCS */
1966
1967 /*
1968 * Do not trust BSD style symlinks either.
1969 */
1970 if (S_ISLNK(sb.st_mode))
1971 return (0);
1972
1973#ifdef VIOCIGETCELL
1974 /*
1975 * This is the VIOCIGETCELL ioctl, which takes an fd, not
1976 * a path name. If it succeeds, then the file is in AFS.
1977 *
1978 * On failure, ENOTTY indicates that the file was not in
1979 * AFS; all other errors are pessimistically assumed to be
1980 * a temporary AFS error.
1981 */
1982 vdata.in_size = 0;
1983 vdata.out_size = sizeof(cellname);
1984 vdata.out = cellname;
1985 if (ioctl(handle, VIOCIGETCELL, (char *) &vdata) != -1)
1986 return (0);
1987 if (errno != ENOTTY)
1988 return (-1);
1989#endif /* VIOCIGETCELL */
1990
1991 /*
1992 * Verify the file is not in NFS.
1993 *
1994 * Our current implementation and Sun OS 3.x use major device
1995 * 255 for NFS files; Sun OS 4.x seems to use 130 (I have only
1996 * determined this empirically -- DLC). Without a fstatvfs()
1997 * system call, this will have to do for now.
1998 */
1999#if defined(__SVR4) || __NetBSD_Version__ > 299000900
2000 {
2001 struct statvfs sf;
2002
2003 if (fstatvfs(handle, &sf) == -1)
2004 return (-1);
2005#ifdef __SVR4
2006 return strncmp(sf.f_basetype, "nfs", 3) != 0;
2007#else
2008 return strncmp(sf.f_fstypename, "nfs", 3) != 0;
2009#endif
2010 }
2011#elif defined(__NetBSD__)
2012 {
2013 struct statfs sf;
2014 if (fstatfs(handle, &sf) == -1)
2015 return (-1);
2016 return strncmp(sf.f_fstypename, "nfs", 3) != 0;
2017 }
2018#else
2019 if (major(sb.st_dev) == 255 || major(sb.st_dev) == 130)
2020 return (0);
2021 else
2022 return (1);
2023#endif
2024
2025}

2027 * Companion routine for ensuring that a local file can be trusted. Compare
2028 * various pieces of the stat information to make sure that the file can be
2029 * trusted. Returns true for stat information which meets the criteria
2030 * for being trustworthy. The main paranoia is to prevent a hard link to
2031 * a root owned file. Since the link could be removed after the file is
2032 * opened, a simply fstat() can not be relied upon. The two stat buffers
2033 * for comparison should come from a stat() on the file name and a following
2034 * fstat() on the open file. Some of the following checks are also an
2035 * additional level of paranoia. Also, this test will fail (correctly) if
2036 * either or both of the stat structures have all fields zeroed; typically
2037 * due to a stat() failure.
2038 */

2042stat_info_ok(struct stat * sb1, struct stat * sb2)
2043{
2044 return (sb1->st_ino == sb2->st_ino && /* Still the same file */
2045 sb1->st_dev == sb2->st_dev && /* On the same device */
2046 sb1->st_mode == sb2->st_mode && /* Perms (and type) same */
2047 S_ISREG(sb1->st_mode) && /* Only allow reg files */
2048 (sb1->st_mode & 077) == 0 && /* Owner only perms */
2049 sb1->st_nlink == sb2->st_nlink && /* # hard links same... */
2050 sb1->st_nlink == 1 && /* and only 1 */
2051 sb1->st_uid == sb2->st_uid && /* owner and ... */
2052 sb1->st_gid == sb2->st_gid && /* group unchanged */
2053 sb1->st_mtime == sb2->st_mtime && /* Unmodified between stats */
2054 sb1->st_ctime == sb2->st_ctime); /* Inode unchanged. Hopefully
2055 * a catch-all paranoid test */
2056}
