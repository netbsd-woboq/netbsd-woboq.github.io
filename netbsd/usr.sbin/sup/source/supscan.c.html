27/*
28 * supscan -- SUP Scan File Builder
29 *
30 * Usage: supscan [ -v ] collection [ basedir ]
31 * supscan [ -v ] -f dirfile
32 * supscan [ -v ] -s
33 * -f "file" -- use dirfile instead of system coll.dir
34 * -s "system" -- perform scan for system supfile
35 * -v "verbose" -- print messages as you go
36 * collection -- name of the desired collection if not -s
37 * basedir -- name of the base directory, if not
38 * the default or recorded in coll.dir
39 * dirfile -- name of replacement for system coll.dir.
40 *

169int
170main(int argc, char **argv)
171{
172 SCAN_COLLECTION * volatile c; /* Avoid longjmp clobbering */
173 volatile int errs;
174#ifdef RLIMIT_DATA
175 struct rlimit dlim;
176
177 if (getrlimit(RLIMIT_DATA, &dlim) == -1)
178 goaway("Error getting resource limit (%s)",
179 strerror(errno));
180 if (dlim.rlim_cur != dlim.rlim_max) {
181 dlim.rlim_cur = dlim.rlim_max;
182 if (setrlimit(RLIMIT_DATA, &dlim) == -1)
183 goaway("Error setting resource limit (%s)",
184 strerror(errno));
185 }
186#endif
187
188 errs = init(argc, argv); /* process arguments */
189 if (errs) {
190 fprintf(stderr, "supscan: %d collections had errors", errs);
191 return 1;
192 }
193 for (c = firstC; c; c = c->Cnext) {
194 collname = c->Cname;
195 basedir = c->Cbase;
196 prefix = c->Cprefix;
197 if (chdir(basedir) < 0) {
198 fprintf(stderr, "supscan: Can't chdir to %s (%s)",
199 basedir, strerror(errno));
200 return 1;
201 }
202 scantime = time(NULL);
203 if (!quiet)
204 printf("SUP Scan for %s starting at %s", collname,
205 ctime(&scantime));
206 (void) fflush(stdout);
207 if (!setjmp(sjbuf)) {
208 makescanlists(); /* record names in scan files */
209 scantime = time(NULL);
210 if (!quiet)
211 printf("SUP Scan for %s completed at %s",
212 collname, ctime(&scantime));
213 } else {
214 fprintf(stderr,
215 "SUP: Scan for %s aborted at %s", collname,
216 ctime(&scantime));
217 errs++;
218 }
219 if (!quiet)
220 (void) fflush(stdout);
221 }
222 while ((c = firstC) != NULL) {
223 firstC = firstC->Cnext;
224 free(c->Cname);
225 free(c->Cbase);
226 if (c->Cprefix)
227 free(c->Cprefix);
228 free(c);
229 }
230 return errs ? 1 : 0;
231}

236void
237usage(void)
238{
239 fprintf(stderr, "Usage: supscan [ -vq ] collection [ basedir ]\n");
240 fprintf(stderr, " supscan [ -vq ] -f dirfile\n");
241 fprintf(stderr, " supscan [ -vq ] -s\n");
242 exit(1);
243}

245int
246init(int argc, char **argv)
247{
248 char buf[STRINGLENGTH], fbuf[STRINGLENGTH], *p, *q;
249 FILE *f;
250 SCAN_COLLECTION **c;
251 int fflag, sflag;
252 char *filename = NULL;
253 int errs = 0;
254
255 quiet = FALSE;
256 trace = FALSE;
257 fflag = FALSE;
258 sflag = FALSE;
259 while (argc > 1 && argv[1][0] == '-') {
260 switch (argv[1][1]) {
261 case 'f':
262 fflag = TRUE;
263 if (argc == 2)
264 usage();
265 --argc;
266 argv++;
267 filename = argv[1];
268 break;
269 case 'q':
270 quiet = TRUE;
271 break;
272 case 'v':
273 trace = TRUE;
274 break;
275 case 's':
276 sflag = TRUE;
277 break;
278 default:
279 fprintf(stderr, "supscan: Invalid flag %s ignored\n",
280 argv[1]);
281 (void) fflush(stderr);
282 }
283 --argc;
284 argv++;
285 }
286 if (!fflag) {
287 (void) sprintf(fbuf, FILEDIRS, DEFDIR);
288 filename = fbuf;
289 }
290 if (sflag) {
291 if (argc != 1)
292 usage();
293 firstC = NULL;
294 c = &firstC;
295 (void) sprintf(buf, FILEHOSTS, DEFDIR);
296 if ((f = fopen(buf, "r")) == NULL)
297 quit(1, "supscan: Unable to open %s\n", buf);
298 while ((p = fgets(buf, STRINGLENGTH, f)) != NULL) {
299 q = strchr(p, '\n');
300 if (q)
301 *q = 0;
302 if (strchr("#;:", *p))
303 continue;
304 collname = nxtarg(&p, " \t=");
305 p = skipover(p, " \t=");
306 if (!localhost(p))
307 continue;
308 *c = getscancoll(filename, estrdup(collname), NULL);
309 if (*c)
310 c = &((*c)->Cnext);
311 else
312 errs++;
313 }
314 (void) fclose(f);
315 return errs;
316 }
317 if (argc < 2 && fflag) {
318 firstC = NULL;
319 c = &firstC;
320 if ((f = fopen(filename, "r")) == NULL)
321 quit(1, "supscan: Unable to open %s\n", filename);
322 while ((p = fgets(buf, STRINGLENGTH, f)) != NULL) {
323 q = strchr(p, '\n');
324 if (q)
325 *q = 0;
326 if (strchr("#;:", *p))
327 continue;
328 q = nxtarg(&p, " \t=");
329 p = skipover(p, " \t=");
330 *c = getscancoll(filename, estrdup(q), estrdup(p));
331 if (*c)
332 c = &((*c)->Cnext);
333 else
334 errs++;
335 }
336 (void) fclose(f);
337 return errs;
338 }
339 if (argc < 2 || argc > 3)
340 usage();
341 firstC = getscancoll(filename, estrdup(argv[1]),
342 argc > 2 ? estrdup(argv[2]) : NULL);
343 if (firstC == NULL)
344 errs++;
345 return errs;
346}

348static SCAN_COLLECTION *
349getscancoll(char *filename, char *collname, char *basedir)
350{
351 char buf[STRINGLENGTH], *p, *q;
352 FILE *f;
353 SCAN_COLLECTION *c;
354
355 if (basedir == NULL) {
356 if ((f = fopen(filename, "r")) != NULL) {
357 while ((p = fgets(buf, STRINGLENGTH, f)) != NULL) {
358 q = strchr(p, '\n');
359 if (q)
360 *q = 0;
361 if (strchr("#;:", *p))
362 continue;
363 q = nxtarg(&p, " \t=");
364 if (strcmp(q, collname) == 0) {
365 p = skipover(p, " \t=");
366 basedir = estrdup(p);
367 break;
368 }
369 }
370 (void) fclose(f);
371 }
372 if (basedir == NULL) {
373 (void) sprintf(buf, FILEBASEDEFAULT, collname);
374 basedir = estrdup(buf);
375 }
376 }
377 if (chdir(basedir) < 0) {
378 fprintf(stderr, "supscan: Can't chdir to base directory %s "
379 "for %s (%s)\n", basedir, collname, strerror(errno));
380 return (NULL);
381 }
382 prefix = NULL;
383 (void) sprintf(buf, FILEPREFIX, collname);
384 if ((f = fopen(buf, "r")) != NULL) {
385 while ((p = fgets(buf, STRINGLENGTH, f)) != NULL) {
386 q = strchr(p, '\n');
387 if (q)
388 *q = 0;
389 if (strchr("#;:", *p))
390 continue;
391 prefix = estrdup(p);
392 if (chdir(prefix) < 0) {
393 fprintf(stderr, "supscan: can't chdir to %s "
394 " from base directory %s for %s (%s)\n",
395 prefix, basedir, collname, strerror(errno));
396 fclose(f);
397 free(prefix);
398 return (NULL);
399 }
400 break;
401 }
402 (void) fclose(f);
403 }
404 if ((c = malloc(sizeof(*c))) == NULL)
405 quit(1, "supscan: can't malloc collection structure\n");
406 c->Cname = collname;
407 c->Cbase = basedir;
408 c->Cprefix = prefix;
409 c->Cnext = NULL;
410 return (c);
411}

413void
414goaway(const char *fmt, ...)
415{
416 va_list ap;
417
418 va_start(ap, fmt);
419 vfprintf(stderr, fmt, ap);
420 va_end(ap);
421 (void) putc('\n', stderr);
422 (void) fflush(stderr);
423 longjmp(sjbuf, TRUE);
424}

426int
427localhost(char *host)
428{
429 static char myhost[MAXHOSTNAMELEN + 1];
430 static unsigned int myhostlen;
431 unsigned int hostlen;
432
433 if (*myhost == '\0') {
434 /*
435 * We assume that the host name in the kernel is the
436 * fully qualified form.
437 */
438 if (gethostname(myhost, sizeof(myhost)) < 0) {
439 quit(1, "supscan: can't get kernel host name\n");
440 }
441 myhost[sizeof(myhost) - 1] = '\0';
442 myhostlen = strlen(myhost);
443 }
444 /*
445 * Here, we assume that the 'host' parameter from the
446 * coll.host file is at least a prefix of the fully qualified
447 * host name of some machine. This will be true when modcoll(8)
448 * (and related scripts) maintain the relevant files, but if
449 * a person makes a manual change, problems could result. In
450 * particular, if a nicname, such as "Y" for "GANDALF.CS.CMU.EDU"
451 * is present in the coll.host file, things will not work as
452 * expected.
453 */
454
455 hostlen = strlen(host);
456
457 return (strncasecmp(myhost, host,
458 hostlen < myhostlen ? hostlen : myhostlen) == 0);
459}
