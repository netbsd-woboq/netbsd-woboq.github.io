222static void
223set_pset_exit_str(struct partition_usage_set *pset)
224{
225 char *str, num[25];
226 const char *args[2];
227 bool overrun;
228 daddr_t free_space = pset->cur_free_space;
229
230 /* format exit string */
231 overrun = free_space < 0;
232 if (overrun)
233 free_space = -free_space;
234
235 snprintf(num, sizeof(num), "%" PRIu64, free_space / sizemult);
236 args[0] = num;
237 args[1] = multname;
238 str = str_arg_subst(
239 msg_string(overrun ? MSG_fssizesbad : MSG_fssizesok),
240 2, args);
241 strlcpy(size_menu_exit, str, sizeof(size_menu_exit));
242 free(str);
243}

245static void
246draw_size_menu_header(menudesc *m, void *arg)
247{
248 struct partition_usage_set *pset = arg;
249 size_t i;
250 char col1[70], desc[MENUSTRSIZE];
251 bool need_ext = false, need_existing = false;
252
253 msg_display(MSG_ptnsizes);
254
255 for (i = 0; i < pset->num; i++) {
256 if (pset->infos[i].flags & PUIFLG_IS_OUTER)
257 need_ext = true;
258 else if (pset->infos[i].cur_part_id != NO_PART)
259 need_existing = true;
260 }
261 if (need_ext && need_existing)
262 snprintf(desc, sizeof desc, "%s, %s",
263 msg_string(MSG_ptnsizes_mark_existing),
264 msg_string(MSG_ptnsizes_mark_external));
265 else if (need_existing)
266 strlcpy(desc, msg_string(MSG_ptnsizes_mark_existing),
267 sizeof desc);
268 else if (need_ext)
269 strlcpy(desc, msg_string(MSG_ptnsizes_mark_external),
270 sizeof desc);
271 if (need_ext || need_existing) {
272 msg_printf("\n");
273 msg_display_add_subst(msg_string(MSG_ptnsizes_markers),
274 1, &desc);
275 }
276 msg_printf("\n\n");
277
278 /* update menu title */
279 snprintf(col1, sizeof col1, "%s (%s)", msg_string(MSG_ptnheaders_size),
280 multname);
281 snprintf(size_menu_title, sizeof size_menu_title,
282 " %-37.37s %s\n %s", col1,
283 msg_string(MSG_ptnheaders_filesystem), size_separator);
284}

286static void
287draw_size_menu_line(menudesc *m, int opt, void *arg)
288{
289 struct partition_usage_set *pset = arg;
290 daddr_t size;
291 char psize[38], inc_free[16], flag, swap[40];
292 const char *mount;
293 bool free_mount = false;
294
295 if (opt < 0 || (size_t)opt >= pset->num)
296 return;
297
298 inc_free[0] = 0;
299 if ((pset->infos[opt].flags & PUIFLAG_EXTEND) &&
300 pset->cur_free_space > 0) {
301 size = pset->infos[opt].size + pset->cur_free_space;
302 snprintf(inc_free, sizeof inc_free, " (%" PRIu64 ")",
303 size / sizemult);
304 }
305 size = pset->infos[opt].size;
306 snprintf(psize, sizeof psize, "%" PRIu64 "%s",
307 size / sizemult, inc_free);
308
309 if (pset->infos[opt].type == PT_swap) {
310 snprintf(swap, sizeof swap, "<%s>",
311 msg_string(MSG_swap_display));
312 mount = swap;
313 } else if (pset->infos[opt].flags & PUIFLG_JUST_MOUNTPOINT) {
314 snprintf(swap, sizeof swap, "%s (%s)",
315 pset->infos[opt].mount,
316 getfslabelname(pset->infos[opt].fs_type,
317 pset->infos[opt].fs_version));
318 mount = swap;
319 } else if (pset->infos[opt].mount[0]) {
320 mount = pset->infos[opt].mount;
321 } else {
322 mount = getfslabelname(pset->infos[opt].fs_type,
323 pset->infos[opt].fs_version);
324 mount = str_arg_subst(msg_string(MSG_size_ptn_not_mounted),
325 1, &mount);
326 free_mount = true;
327 }
328 flag = ' ';
329 if (pset->infos[opt].flags & PUIFLAG_EXTEND)
330 flag = '+';
331 else if (pset->infos[opt].flags & PUIFLG_IS_OUTER)
332 flag = '@';
333 else if (pset->infos[opt].cur_part_id != NO_PART)
334 flag = '=';
335 wprintw(m->mw, "%-35.35s %c %s", psize, flag, mount);
336 if (free_mount)
337 free(__UNCONST(mount));
338
339 if (opt == 0)
340 set_pset_exit_str(pset);
341}

343static int
344add_other_ptn_size(menudesc *menu, void *arg)
345{
346 struct partition_usage_set *pset = arg;
347 struct part_usage_info *p;
348 struct menu_ent *m;
349 char new_mp[MOUNTLEN], *err;
350 const char *args;
351
352 for (;;) {
353 msg_prompt_win(partman_go?MSG_askfsmountadv:MSG_askfsmount,
354 -1, 18, 0, 0, NULL, new_mp, sizeof(new_mp));
355 if (new_mp[0] == 0)
356 return 0;
357 if (new_mp[0] != '/') {
358 /* we need absolute mount paths */
359 memmove(new_mp+1, new_mp, sizeof(new_mp)-1);
360 new_mp[0] = '/';
361 /* duplicates? */
362 bool duplicate = false;
363 for (size_t i = 0; i < pset->num; i++) {
364 if (strcmp(pset->infos[i].mount,
365 new_mp) == 0) {
366 args = new_mp;
367 err = str_arg_subst(
368 msg_string(MSG_mp_already_exists),
369 1, &args);
370 err_msg_win(err);
371 free(err);
372 duplicate = true;
373 break;
374 }
375 }
376 if (!duplicate)
377 break;
378 }
379 }
380
381 m = realloc(pset->menu_opts, (pset->num+4)*sizeof(*pset->menu_opts));
382 if (m == NULL)
383 return 0;
384 p = realloc(pset->infos, (pset->num+1)*sizeof(*pset->infos));
385 if (p == NULL)
386 return 0;
387
388 pset->infos = p;
389 pset->menu_opts = m;
390 menu->opts = m;
391 menu->numopts = pset->num+4;
392 m += pset->num;
393 p += pset->num;
394 memset(m, 0, sizeof(*m));
395 memset(p, 0, sizeof(*p));
396 strncpy(p->mount, new_mp, sizeof(p->mount));
397
398 menu->cursel = pset->num;
399 pset->num++;
400 fill_ptn_menu(pset);
401
402 return -1;
403}

405static size_t
406fill_ptn_menu(struct partition_usage_set *pset)
407{
408 struct part_usage_info *p;
409 struct disk_part_info info;
410 menu_ent *m;
411 size_t i;
412 daddr_t free_space;
413
414 memset(pset->menu_opts, 0, (pset->num+3)*sizeof(*pset->menu_opts));
415 for (m = pset->menu_opts, p = pset->infos, i = 0; i < pset->num;
416 m++, p++, i++) {
417 m->opt_action = set_ptn_size;
418 }
419
420 m->opt_name = size_separator;
421 m->opt_flags = OPT_IGNORE|OPT_NOSHORT;
422 m++;
423
424 m->opt_name = MSG_add_another_ptn;
425 m->opt_action = add_other_ptn_size;
426 m++;
427
428 m->opt_name = MSG_askunits;
429 m->opt_menu = MENU_sizechoice;
430 m->opt_flags = OPT_SUB;
431 m++;
432
433 /* calculate free space */
434 free_space = pset->parts->free_space;
435 for (i = 0; i < pset->parts->num_part; i++) {
436 if (!pset->parts->pscheme->get_part_info(pset->parts, i,
437 &info))
438 continue;
439 if (info.flags & (PTI_SEC_CONTAINER|PTI_WHOLE_DISK|
440 PTI_PSCHEME_INTERNAL|PTI_RAW_PART))
441 continue;
442 free_space += info.size;
443 }
444 for (i = 0; i < pset->num; i++) {
445 if (pset->infos[i].flags &
446 (PUIFLG_IS_OUTER|PUIFLG_JUST_MOUNTPOINT))
447 continue;
448 free_space -= pset->infos[i].size;
449 }
450 pset->cur_free_space = free_space;
451 set_pset_exit_str(pset);
452
453 if (pset->menu >= 0)
454 set_menu_numopts(pset->menu, m - pset->menu_opts);
455
456 return m - pset->menu_opts;
457}

462 size_t i;
463 struct disk_part_info info;

465 for (i = 0; i < parts->num_part; i++) {

475int
476set_ptn_size(menudesc *m, void *arg)
477{
478 struct partition_usage_set *pset = arg;
479 struct part_usage_info *p = &pset->infos[m->cursel];
480 char answer[16], dflt[16];
481 const char *err_msg;
482 size_t i, root = ~0U;
483 daddr_t size, old_size, new_size_val, mult;
484 int rv;
485 bool non_zero, extend;
486
487 if (pset->cur_free_space == 0 && p->size == 0 &&
488 !(p->flags & PUIFLG_JUST_MOUNTPOINT))
489 /* Don't allow 'free_parts' to go negative */
490 return 0;
491
492 if (p->cur_part_id != NO_PART) {
493 rv = 0;
494 process_menu(MENU_ptnsize_replace_existing_partition, &rv);
495 if (rv == 0)
496 return 0;
497 if (!pset->parts->pscheme->delete_partition(pset->parts,
498 p->cur_part_id, &err_msg)) {
499 if (err_msg)
500 err_msg_win(err_msg);
501 return 0;
502 }
503 p->cur_part_id = NO_PART;
504 /*
505 * All other part ids are invalid now too - update them!
506 */
507 for (i = 0; i < pset->num; i++) {
508 if (pset->infos[i].cur_part_id == NO_PART)
509 continue;
510 pset->infos[i].cur_part_id =
511 find_part_at(pset->parts, pset->infos[i].cur_start);
512 }
513 }
514
515 size = p->size;
516 old_size = size;
517 if (size == 0)
518 size = p->def_size;
519 size /= sizemult;
520 snprintf(dflt, sizeof dflt, "%" PRIu64 "%s",
521 size, p->flags & PUIFLAG_EXTEND ? "+" : "");
522
523 for (;;) {
524 msg_fmt_prompt_win(MSG_askfssize, -1, 18, 0, 0,
525 dflt, answer, sizeof answer, "%s%s", p->mount, multname);
526
527 /* cp will be checked below */
528 mult = sizemult;
529 new_size_val = parse_disk_pos(answer, &mult, pm->dlcylsize,
530 &extend);
531
532 if (strcmp(answer, dflt) == 0)
533 non_zero = p->def_size > 0;
534 else
535 non_zero = new_size_val > 0;
536
537 /* Some special cases when /usr is first given a size */
538 if (old_size == 0 && non_zero &&
539 strcmp(p->mount, "/usr") == 0) {
540 for (i = 0; i < pset->num; i++) {
541 if (strcmp(pset->infos[i].mount, "/") == 0) {
542 root = i;
543 break;
544 }
545 }
546 /* Remove space for /usr from / */
547 if (root < pset->num && pset->infos[i].cur_part_id ==
548 NO_PART) {
549 pset->infos[root].size -= p->def_size;
550 pset->cur_free_space += p->def_size;
551 }
552 /* hack to add free space to default sized /usr */
553 if (strcmp(answer, dflt) == 0) {
554 size = p->def_size;
555 pset->infos[root].flags &= ~PUIFLAG_EXTEND;
556 p->flags |= PUIFLAG_EXTEND;
557 goto adjust_free;
558 }
559 }
560 if (new_size_val < 0)
561 continue;
562 size = new_size_val;
563 break;
564 }
565
566 daddr_t align = pset->parts->pscheme->get_part_alignment(pset->parts);
567 size = NUMSEC(size, mult, align);
568 if (p->flags & PUIFLAG_EXTEND)
569 p->flags &= ~PUIFLAG_EXTEND;
570 if (extend && (p->limit == 0 || p->limit > p->size)) {
571 p->flags |= PUIFLAG_EXTEND;
572 if (size == 0)
573 size = align;
574 }
575 if (p->limit != 0 && size > p->limit)
576 size = p->limit;
577 adjust_free:
578 if ((p->flags & (PUIFLG_IS_OUTER|PUIFLG_JUST_MOUNTPOINT)) == 0)
579 pset->cur_free_space += p->size - size;
580 p->size = size;
581 set_pset_exit_str(pset);
582
583 return 0;
584}

586/*
587 * User interface to edit a "wanted" partition layout "pset" as first
588 * abstract phase (not concrete partitions).
589 * Make sure to have everything (at least theoretically) fit the
590 * available space.
591 * During editing we keep the part_usage_info and the menu_opts
592 * in pset in sync, that is: we always allocate just enough entries
593 * in pset->infos as we have usage infos in the list (pset->num),
594 * and two additional menu entries ("add a partition" and "select units").
595 * The menu exit string changes depending on content, and implies
596 * abort while the partition set is not valid (does not fit).
597 * Return true when the user wants to continue (by editing the concrete

617 draw_size_menu_header, draw_size_menu_line, NULL,
618 NULL, size_menu_exit);
619
620 if (pset->menu < 0) {
621 free(pset->menu_opts);
622 pset->menu_opts = NULL;
623 return false;
624 }
625
626 pset->ok = true;
627 process_menu(pset->menu, pset);
628
629 free_menu(pset->menu);
630 free(pset->menu_opts);
631 pset->menu = -1;
632 pset->menu_opts = NULL;
633
634 if (!pset->ok)
635 return false;
636

672 /* Claw back space for /usr */
673 cur_ptns.ptn_sizes[PI_USR].size =
674 cur_ptns.ptn_sizes[PI_USR].dflt_size;
675 cur_ptns.ptn_sizes[PI_ROOT].size -=
676 cur_ptns.ptn_sizes[PI_USR].dflt_size;
677 cur_ptns.ptn_sizes[PI_ROOT].changed = 1;
678 /* Give free space to /usr */
679 cur_ptns.pool_part = &cur_ptns.ptn_sizes[PI_USR];
680 }
681 }
682
683 /* Change preset sizes from MB to sectors */
684 cur_ptns.free_space = sectors;
685 for (p = cur_ptns.ptn_sizes; p->mount[0]; p++) {
686 p->size = NUMSEC(p->size, sm, pm->dlcylsize);
687 p->dflt_size = NUMSEC(p->dflt_size, sm, pm->dlcylsize);
688 cur_ptns.free_space -= p->size;
689 }
690
691 /* Steal space from swap to make things fit.. */
692 if (cur_ptns.free_space < 0) {
693 i = roundup(-cur_ptns.free_space, pm->dlcylsize);
694 if (i > cur_ptns.ptn_sizes[PI_SWAP].size)
695 i = cur_ptns.ptn_sizes[PI_SWAP].size;
696 cur_ptns.ptn_sizes[PI_SWAP].size -= i;
697 cur_ptns.free_space += i;
698 }
699
700 /* Add space for 2 system dumps to / (traditional) */
701 i = get_ramsize() * sm;
702 i = roundup(i, pm->dlcylsize);
703 if (cur_ptns.free_space > i * 2)
704 i *= 2;
705 if (cur_ptns.free_space > i) {
706 cur_ptns.ptn_sizes[PI_ROOT].size += i;
707 cur_ptns.free_space -= i;
708 }
709
710 if (root_created && partman_go) {
711 cur_ptns.ptn_sizes[PI_ROOT].size = 0;
712 cur_ptns.pool_part = 0;
713 }
714
715 /* Ensure all of / is readable by the system boot code */
716 i = cur_ptns.ptn_sizes[PI_ROOT].limit;
717 if (i != 0 && (i -= cur_ptns.ptn_sizes[PI_ROOT].size) < 0) {
718 cur_ptns.ptn_sizes[PI_ROOT].size += i;
719 cur_ptns.free_space -= i;
720 }
721
722 /* Count free partition slots */
723 cur_ptns.free_parts = 0;
724#if 0 // XXX
725 for (i = 0; i < maxpart; i++) {
726 if (pm->bsdlabel[i].pi_size == 0)
727 cur_ptns.free_parts++;
728 }
729#endif
730 for (i = 0; i < MAXPARTITIONS; i++) {
731 p = &cur_ptns.ptn_sizes[i];
732 if (i != 0 && p->use == 0)
733 p->use = PART_EXTRA;
734 if (p->size != 0)
735 cur_ptns.free_parts--;
736 }
737
738 cur_ptns.menu_no = new_menu(0, cur_ptns.ptn_menus,
739 __arraycount(cur_ptns.ptn_menus),
740 3, -1, 12, 70,
741 MC_ALWAYS_SCROLL | MC_NOBOX | MC_NOCLEAR,
742 NULL, set_ptn_titles, NULL,
743 "help", cur_ptns.exit_msg);
744
745 if (cur_ptns.menu_no < 0)
746 return;
747 }
748
749 do {
750 set_ptn_menu(&cur_ptns);
751 pm->current_cylsize = pm->dlcylsize;
752 process_menu(cur_ptns.menu_no, &cur_ptns);
753 } while (cur_ptns.free_space < 0 || cur_ptns.free_parts < 0);
754
755 /* Give any cylinder fragment to last partition */
756 if (cur_ptns.pool_part != NULL || cur_ptns.free_space < pm->dlcylsize) {
757 for (p = cur_ptns.ptn_sizes + __arraycount(cur_ptns.ptn_sizes) - 1; ;p--) {
758 if (p->size == 0) {
759 if (p == cur_ptns.ptn_sizes)
760 break;
761 continue;
762 }
763 if (p->use == PART_TMP_RAMDISK)
764 continue;
765 p->size += cur_ptns.free_space % pm->dlcylsize;
766 cur_ptns.free_space -= cur_ptns.free_space % pm->dlcylsize;
767 break;
768 }
769 }
770
771 for (p = cur_ptns.ptn_sizes; p->mount[0]; p++, part_start += size) {
772 size = p->size;
773 if (p == cur_ptns.pool_part) {
774 size += rounddown(cur_ptns.free_space, pm->dlcylsize);
775 if (p->limit != 0 && size > p->limit)
776 size = p->limit;
777 }
778 i = p->use;
779 if (i == PART_TMP_RAMDISK) {
780 tmp_ramdisk_size = size;
781 size = 0;
782 continue;
783 }
784 if (size == 0)
785 continue;
786 if (i == PART_ROOT && size > 0)
787 root_created = 1;
788 if (i == PART_SWAP) {
789 if (size > 0)
790 swap_created = 1;
791 save_ptn(i, part_start, size, FS_SWAP, NULL);
792 continue;
793 }
794 if (!strcmp(p->mount, "raid")) {
795 save_ptn(i, part_start, size, FS_RAID, NULL);
796 continue;
797 } else if (!strcmp(p->mount, "cgd")) {
798 save_ptn(i, part_start, size, FS_CGD, NULL);
799 continue;
800 }
801 save_ptn(i, part_start, size, FS_BSDFFS, p->mount);
802 }
803#endif
804 return true;
805}

904static void
905merge_part_with_wanted(struct disk_partitions *parts, part_id pno,
906 const struct disk_part_info *info, struct partition_usage_set *wanted,
907 bool is_outer)
908{
909 struct part_usage_info *infos;
910
911 /*
912 * does this partition match something in the wanted set?
913 */
914 for (size_t i = 0; i < wanted->num; i++) {
915 if (wanted->infos[i].type != info->nat_type->generic_ptype)
916 continue;
917 if (info->last_mounted != NULL && info->last_mounted[0] != 0 &&
918 strcmp(info->last_mounted, wanted->infos[i].mount) != 0)
919 continue;
920 if (wanted->infos[i].cur_part_id != NO_PART)
921 continue;
922 wanted->infos[i].cur_part_id = pno;
923 wanted->infos[i].parts = parts;
924 wanted->infos[i].size = info->size;
925 wanted->infos[i].cur_start = info->start;
926 wanted->infos[i].flags &= ~PUIFLAG_EXTEND;
927 if (wanted->infos[i].fs_type != FS_UNUSED &&
928 wanted->infos[i].type != PT_swap)
929 wanted->infos[i].instflags |= PUIINST_MOUNT;
930 if (is_outer)
931 wanted->infos[i].flags |= PUIFLG_IS_OUTER;
932 return;
933 }
934
935 /*
936 * no match - if this is fromt the outer scheme, we are done.
937 * otherwise it must be inserted into the wanted set.
938 */
939 if (is_outer)
940 return;
941
942 /*
943 * create a new entry for this
944 */
945 infos = realloc(wanted->infos, sizeof(*infos)*(wanted->num+1));
946 if (infos == NULL)
947 return;
948 wanted->infos = infos;
949 infos += wanted->num;
950 wanted->num++;
951 memset(infos, 0, sizeof(*infos));
952 if (info->last_mounted != NULL && info->last_mounted[0] != 0)
953 strlcpy(infos->mount, info->last_mounted,
954 sizeof(infos->mount));
955 infos->type = info->nat_type->generic_ptype;
956 infos->cur_part_id = pno;
957 infos->parts = parts;
958 infos->size = info->size;
959 infos->cur_start = info->start;
960 infos->fs_type = info->fs_type;
961 infos->fs_version = info->fs_sub_type;
962 if (is_outer)
963 infos->flags |= PUIFLG_IS_OUTER;
964}

979static void
980fill_defaults(struct partition_usage_set *wanted, struct disk_partitions *parts,
981 daddr_t ptstart, daddr_t ptsize)
982{
983 size_t i, root = ~0U, usr = ~0U, swap = ~0U, def_usr = ~0U;
984 daddr_t free_space, dump_space, required;
985#if defined(DEFAULT_UFS2) && !defined(HAVE_UFS2_BOOT)
986 size_t boot = ~0U;
987#endif
988
989 memset(wanted, 0, sizeof(*wanted));
990 wanted->parts = parts;
991 wanted->num = __arraycount(default_parts_init);
992 wanted->infos = calloc(wanted->num, sizeof(*wanted->infos));
993 if (wanted->infos == NULL) {
994 err_msg_win(err_outofmem);
995 return;
996 }
997
998 memcpy(wanted->infos, default_parts_init, sizeof(default_parts_init));
999
1000#ifdef MD_PART_DEFAULTS
1001 MD_PART_DEFAULTS(pm, wanted->infos, wanted->num);
1002#endif
1003
1004 for (i = 0; i < wanted->num; i++) {
1005 wanted->infos[i].parts = parts;
1006 wanted->infos[i].cur_part_id = NO_PART;
1007
1008#if DEFSWAPSIZE == -1
1009 if (wanted->infos[i].type == PT_swap)
1010 wanted->infos[i].size = get_ramsize() * (MEG / 512);
1011#endif
1012 if (wanted->infos[i].type == PT_swap && swap > wanted->num)
1013 swap = i;
1014#if defined(DEFAULT_UFS2) && !defined(HAVE_UFS2_BOOT)
1015 if (wanted->infos[i].instflags & PUIINST_BOOT)
1016 boot = i;
1017#endif
1018 if (wanted->infos[i].type == PT_root) {
1019 if (strcmp(wanted->infos[i].mount, "/") == 0) {
1020 root = i;
1021 } else if (
1022 strcmp(wanted->infos[i].mount, "/usr") == 0) {
1023 if (wanted->infos[i].size > 0)
1024 usr = i;
1025 else
1026 def_usr = i;
1027 }
1028 if (wanted->infos[i].fs_type == FS_UNUSED)
1029 wanted->infos[i].fs_type = FS_BSDFFS;
1030 if (wanted->infos[i].fs_type == FS_BSDFFS) {
1031#ifdef DEFAULT_UFS2
1032#ifndef HAVE_UFS2_BOOT
1033 if (boot < wanted->num || i != root)
1034#endif
1035 wanted->infos[i].fs_version = 2;
1036#endif
1037 }
1038 }
1039 if ((wanted->infos[i].flags & PUIFLG_JUST_MOUNTPOINT) &&
1040 wanted->infos[i].size == 0)
1041 /* default tmpfs to 1/4 RAM */
1042 wanted->infos[i].def_size =
1043 get_ramsize() * (MEG/512/4);
1044 }
1045
1046 /*
1047 * Now we have the defaults as if we were installing to an
1048 * empty disk. Merge the partitions in target range that are already
1049 * there (match with wanted) or are there additionaly.
1050 * The only thing outside of target range that we care for
1051 * is a potential swap partition - we assume one is enough.
1052 */
1053 if (parts->parent) {
1054 for (part_id pno = 0; pno < parts->parent->num_part; pno++) {
1055 struct disk_part_info info;
1056
1057 if (!parts->parent->pscheme->get_part_info(
1058 parts->parent, pno, &info))
1059 continue;
1060 if (info.nat_type->generic_ptype != PT_swap)
1061 continue;
1062 merge_part_with_wanted(parts->parent, pno, &info,
1063 wanted, true);
1064 break;
1065 }
1066 }
1067 for (part_id pno = 0; pno < parts->num_part; pno++) {
1068 struct disk_part_info info;
1069
1070 if (!parts->pscheme->get_part_info(parts, pno, &info))
1071 continue;
1072
1073 if (info.flags & PTI_PSCHEME_INTERNAL)
1074 continue;
1075
1076 if (info.nat_type->generic_ptype != PT_swap &&
1077 (info.start < ptstart ||
1078 (info.start + info.size) > (ptstart+ptsize)))
1079 continue;
1080
1081 merge_part_with_wanted(parts, pno, &info,
1082 wanted, false);
1083 }
1084
1085 daddr_t align = parts->pscheme->get_part_alignment(parts);
1086
1087 if (root < wanted->num && wanted->infos[root].cur_part_id == NO_PART) {
1088 daddr_t max_root_size = parts->disk_start + parts->disk_size;
1089 if (root_limit > 0) {
1090 /* Bah - bios can not read all the disk, limit root */
1091 max_root_size = root_limit - parts->disk_start;
1092 }
1093 wanted->infos[root].limit = max_root_size;
1094 }
1095
1096 if (have_x11_by_default()) {
1097 daddr_t xsize = XNEEDMB * (MEG / 512);
1098 if (usr < wanted->num) {
1099 if (wanted->infos[usr].cur_part_id == NO_PART) {
1100 wanted->infos[usr].size += xsize;
1101 wanted->infos[usr].def_size += xsize;
1102 }
1103 } else if (root < wanted->num &&
1104 wanted->infos[root].cur_part_id == NO_PART &&
1105 (wanted->infos[root].limit == 0 ||
1106 (wanted->infos[root].size + xsize) <=
1107 wanted->infos[root].limit)) {
1108 wanted->infos[root].size += xsize;
1109 }
1110 }
1111 if (wanted->infos[root].limit > 0 &&
1112 wanted->infos[root].size > wanted->infos[root].limit) {
1113 if (usr < wanted->num) {
1114 /* move space from root to usr */
1115 daddr_t spill = wanted->infos[root].size -
1116 wanted->infos[root].limit;
1117 spill = roundup(spill, align);
1118 wanted->infos[root].size =
1119 wanted->infos[root].limit;
1120 wanted->infos[usr].size = spill;
1121 } else {
1122 wanted->infos[root].size =
1123 wanted->infos[root].limit;
1124 }
1125 }
1126
1127 /*
1128 * Preliminary calc additional space to allocate and how much
1129 * we likely will have left over. Use that to do further
1130 * adjustments, so we don't present the user inherently
1131 * impossible defaults.
1132 */
1133 free_space = parts->free_space;
1134 required = 0;
1135 if (root < wanted->num)
1136 required += wanted->infos[root].size;
1137 if (usr < wanted->num)
1138 required += wanted->infos[usr].size;
1139 else if (def_usr < wanted->num)
1140 required += wanted->infos[def_usr].def_size;
1141 free_space -= required;
1142 for (i = 0; i < wanted->num; i++) {
1143 if (i == root || i == usr)
1144 continue; /* already accounted above */
1145 if (wanted->infos[i].cur_part_id != NO_PART)
1146 continue;
1147 if (wanted->infos[i].size == 0)
1148 continue;
1149 if (wanted->infos[i].flags
1150 & (PUIFLG_IS_OUTER|PUIFLG_JUST_MOUNTPOINT))
1151 continue;
1152 free_space -= wanted->infos[i].size;
1153 }
1154 if (free_space < 0 && swap < wanted->num) {
1155 /* steel from swap partition */
1156 daddr_t d = wanted->infos[swap].size;
1157 daddr_t inc = roundup(-free_space, align);
1158 if (inc > d)
1159 inc = d;
1160 free_space += inc;
1161 wanted->infos[swap].size -= inc;
1162 }
1163 if (root < wanted->num) {
1164 /* Add space for 2 system dumps to / (traditional) */
1165 dump_space = get_ramsize() * (MEG/512);
1166 dump_space = roundup(dump_space, align);
1167 if (free_space > dump_space*2)
1168 dump_space *= 2;
1169 if (free_space > dump_space)
1170 wanted->infos[root].size += dump_space;
1171 }
1172}

1174/*
1175 * We sort pset->infos to sync with pset->parts and
1176 * the cur_part_id, to allow using the same index into both
1177 * "array" in later phases. This may include inserting
1178 * dummy entries (when we do not actually want the
1179 * partition, but it is forced upon us, like RAW_PART in
1180 * disklabel).
1181 */
1182static void
1183sort_and_sync_parts(struct partition_usage_set *pset)
1184{
1185 struct part_usage_info *infos;
1186 size_t i, j, no;
1187 part_id pno;
1188
1189 pset->cur_free_space = pset->parts->free_space;
1190
1191 /* count non-empty entries that are not in pset->parts */
1192 no = pset->parts->num_part;
1193 for (i = 0; i < pset->num; i++) {
1194 if (pset->infos[i].size == 0)
1195 continue;
1196 if (pset->infos[i].cur_part_id != NO_PART)
1197 continue;
1198 no++;
1199 }
1200
1201 /* allocate new infos */
1202 infos = calloc(no, sizeof *infos);
1203 if (infos == NULL)
1204 return;
1205
1206 /* pre-initialize the first entires as dummy entries */
1207 for (i = 0; i < pset->parts->num_part; i++) {
1208 infos[i].cur_part_id = NO_PART;
1209 infos[i].cur_flags = PTI_PSCHEME_INTERNAL;
1210 }
1211 /*
1212 * Now copy over eveything from our old entries that points to
1213 * a real partition.
1214 */
1215 for (i = 0; i < pset->num; i++) {
1216 pno = pset->infos[i].cur_part_id;
1217 if (pno == NO_PART)
1218 continue;
1219 if (pset->parts != pset->infos[i].parts)
1220 continue;
1221 if (pset->infos[i].flags & PUIFLG_JUST_MOUNTPOINT)
1222 continue;
1223 if ((pset->infos[i].flags & (PUIFLG_IS_OUTER|PUIFLAG_ADD_INNER))
1224 == PUIFLG_IS_OUTER)
1225 continue;
1226 if (pno >= pset->parts->num_part)
1227 continue;
1228 memcpy(infos+pno, pset->infos+i, sizeof(*infos));
1229 }
1230 /* Fill in the infos for real partitions where we had no data */
1231 for (pno = 0; pno < pset->parts->num_part; pno++) {
1232 struct disk_part_info info;
1233
1234 if (infos[pno].cur_part_id != NO_PART)
1235 continue;
1236
1237 if (!pset->parts->pscheme->get_part_info(pset->parts, pno,
1238 &info))
1239 continue;
1240
1241 infos[pno].parts = pset->parts;
1242 infos[pno].cur_part_id = pno;
1243 infos[pno].cur_flags = info.flags;
1244 infos[pno].size = info.size;
1245 infos[pno].type = info.nat_type->generic_ptype;
1246 infos[pno].cur_start = info.start;
1247 infos[pno].fs_type = info.fs_type;
1248 infos[pno].fs_version = info.fs_sub_type;
1249 }
1250 /* Add the non-partition entires after that */
1251 j = pset->num;
1252 for (i = 0; i < pset->num; i++) {
1253 if (j >= no)
1254 break;
1255 if (pset->infos[i].size == 0)
1256 continue;
1257 if (pset->infos[i].cur_part_id != NO_PART)
1258 continue;
1259 memcpy(infos+j, pset->infos+i, sizeof(*infos));
1260 j++;
1261 }
1262
1263 /* done, replace infos */
1264 free(pset->infos);
1265 pset->num = no;
1266 pset->infos = infos;
1267}

1269static void
1270apply_settings_to_partitions(struct pm_devs *p, struct disk_partitions *parts,
1271 struct partition_usage_set *wanted, daddr_t start, daddr_t size)
1272{
1273 size_t i, exp_ndx = ~0U;
1274 daddr_t planned_space = 0, nsp, from, align;
1275 struct disk_part_info *infos;
1276 struct disk_part_free_space space;
1277 struct disk_partitions *ps = NULL;
1278 part_id pno, new_part_id;
1279
1280 infos = calloc(wanted->num, sizeof(*infos));
1281 if (infos == NULL) {
1282 err_msg_win(err_outofmem);
1283 return;
1284 }
1285
1286 align = wanted->parts->pscheme->get_part_alignment(wanted->parts);
1287
1288 /*
1289 * Pass one: calculate space available for expanding
1290 * the marked partition.
1291 */
1292 for (i = 0; i < wanted->num; i++) {
1293 if ((wanted->infos[i].flags & PUIFLAG_EXTEND) &&
1294 exp_ndx == ~0U)
1295 exp_ndx = i;
1296 if (wanted->infos[i].flags &
1297 (PUIFLG_JUST_MOUNTPOINT|PUIFLG_IS_OUTER))
1298 continue;
1299 nsp = wanted->infos[i].size;
1300 if (wanted->infos[i].cur_part_id != NO_PART) {
1301 ps = wanted->infos[i].flags & PUIFLG_IS_OUTER ?
1302 parts->parent : parts;
1303
1304 if (ps->pscheme->get_part_info(ps,
1305 wanted->infos[i].cur_part_id, &infos[i]))
1306 nsp -= infos[i].size;
1307 }
1308 if (nsp > 0)
1309 planned_space += roundup(nsp, align);
1310 }
1311
1312 /*
1313 * Expand the pool partition (or shrink, if we overran),
1314 */
1315 if (exp_ndx < wanted->num)
1316 wanted->infos[exp_ndx].size +=
1317 parts->free_space - planned_space;
1318
1319 /*
1320 * Now it gets tricky: we want the wanted partitions in order
1321 * as defined, but any already existing partitions should not
1322 * be moved. We allow them to change size though.
1323 * To keep it simple, we just assign in order and skip blocked
1324 * spaces. This may shuffle the order of the resulting partitions
1325 * compared to the wanted list.
1326 */
1327
1328 /* Adjust sizes of existing partitions */
1329 for (i = 0; i < wanted->num; i++) {
1330 ps = wanted->infos[i].flags & PUIFLG_IS_OUTER ?
1331 parts->parent : parts;
1332 const struct part_usage_info *want = &wanted->infos[i];
1333
1334 if (want->cur_part_id == NO_PART)
1335 continue;
1336 if (i == exp_ndx) /* the exp. part. can not exist yet */
1337 continue;
1338 daddr_t free_size = ps->pscheme->max_free_space_at(ps,
1339 infos[i].start);
1340 if (free_size < wanted->infos[i].size)
1341 continue;
1342 infos[i].size = wanted->infos[i].size;
1343 ps->pscheme->set_part_info(ps, want->cur_part_id,
1344 &infos[i], NULL);
1345 }
1346
1347 from = -1;
1348 /*
1349 * First add all outer partitions - we need to align those exactly
1350 * with the inner counterpart later.
1351 */
1352 if (parts->parent) {
1353 ps = parts->parent;
1354 daddr_t outer_align = ps->pscheme->get_part_alignment(ps);
1355
1356 for (i = 0; i < wanted->num; i++) {
1357 struct part_usage_info *want = &wanted->infos[i];
1358
1359 if (want->cur_part_id != NO_PART)
1360 continue;
1361 if (!(want->flags & PUIFLAG_ADD_OUTER))
1362 continue;
1363 if (want->size <= 0)
1364 continue;
1365
1366 size_t cnt = ps->pscheme->get_free_spaces(ps,
1367 &space, 1, want->size-2*outer_align,
1368 outer_align, from, -1);
1369
1370 if (cnt == 0) /* no free space for this partition */
1371 continue;
1372
1373 infos[i].start = space.start;
1374 infos[i].size = min(want->size, space.size);
1375 infos[i].nat_type =
1376 ps->pscheme->get_fs_part_type(
1377 want->fs_type, want->fs_version);
1378 infos[i].last_mounted = want->mount;
1379 infos[i].fs_type = want->fs_type;
1380 infos[i].fs_sub_type = want->fs_version;
1381 new_part_id = ps->pscheme->add_partition(ps,
1382 &infos[i], NULL);
1383 if (new_part_id == NO_PART)
1384 continue; /* failed to add, skip */
1385
1386 ps->pscheme->get_part_info(ps,
1387 new_part_id, &infos[i]);
1388 want->cur_part_id = new_part_id;
1389
1390 want->flags |= PUIFLAG_ADD_INNER|PUIFLG_IS_OUTER;
1391 from = rounddown(infos[i].start +
1392 infos[i].size+outer_align, outer_align);
1393 }
1394 }
1395
1396 /*
1397 * Now add new inner partitions
1398 */
1399 for (i = 0; i < wanted->num && from < wanted->parts->disk_size; i++) {
1400 struct part_usage_info *want = &wanted->infos[i];
1401
1402 if (want->cur_part_id != NO_PART)
1403 continue;
1404 if (want->flags & (PUIFLG_JUST_MOUNTPOINT|PUIFLG_IS_OUTER))
1405 continue;
1406 if (want->size <= 0)
1407 continue;
1408
1409 size_t cnt = wanted->parts->pscheme->get_free_spaces(
1410 wanted->parts, &space, 1, want->size-2*align, align, from,
1411 -1);
1412
1413 if (cnt == 0)
1414 continue; /* no free space for this partition */
1415
1416 infos[i].start = space.start;
1417 infos[i].size = min(want->size, space.size);
1418 infos[i].nat_type =
1419 wanted->parts->pscheme->get_fs_part_type(want->fs_type,
1420 want->fs_version);
1421 infos[i].last_mounted = want->mount;
1422 infos[i].fs_type = want->fs_type;
1423 infos[i].fs_sub_type = want->fs_version;
1424 if (want->fs_type != FS_UNUSED && want->type != PT_swap) {
1425 want->instflags |= PUIINST_NEWFS;
1426 if (want->mount[0] != 0)
1427 want->instflags |= PUIINST_MOUNT;
1428 }
1429 new_part_id = wanted->parts->pscheme->add_partition(
1430 wanted->parts, &infos[i], NULL);
1431 if (new_part_id == NO_PART)
1432 continue; /* failed to add, skip */
1433
1434 wanted->parts->pscheme->get_part_info(
1435 wanted->parts, new_part_id, &infos[i]);
1436 from = rounddown(infos[i].start+infos[i].size+align, align);
1437 }
1438
1439
1440 /*
1441 * If there are any outer partitions that we need as inner ones
1442 * too, add them to the inner partitioning scheme.
1443 */
1444 for (i = 0; i < wanted->num; i++) {
1445 struct part_usage_info *want = &wanted->infos[i];
1446
1447 if (want->cur_part_id != NO_PART)
1448 continue;
1449 if (want->flags & PUIFLG_JUST_MOUNTPOINT)
1450 continue;
1451 if (want->size <= 0)
1452 continue;
1453
1454 if ((want->flags & (PUIFLAG_ADD_INNER|PUIFLG_IS_OUTER)) !=
1455 (PUIFLAG_ADD_INNER|PUIFLG_IS_OUTER))
1456 continue;
1457
1458 infos[i].start = want->cur_start;
1459 infos[i].size = want->size;
1460 infos[i].nat_type = wanted->parts->pscheme->get_fs_part_type(
1461 want->fs_type, want->fs_version);
1462 infos[i].last_mounted = want->mount;
1463 infos[i].fs_type = want->fs_type;
1464 infos[i].fs_sub_type = want->fs_version;
1465
1466 if (wanted->parts->pscheme->add_outer_partition
1467 != NULL)
1468 new_part_id = wanted->parts->pscheme->
1469 add_outer_partition(
1470 wanted->parts, &infos[i], NULL);
1471 else
1472 new_part_id = wanted->parts->pscheme->
1473 add_partition(
1474 wanted->parts, &infos[i], NULL);
1475
1476 if (new_part_id == NO_PART)
1477 continue; /* failed to add, skip */
1478
1479 wanted->parts->pscheme->get_part_info(
1480 wanted->parts, new_part_id, &infos[i]);
1481 }
1482
1483 /*
1484 * Note: all part_ids are invalid now, as we have added things!
1485 */
1486 for (i = 0; i < wanted->num; i++)
1487 wanted->infos[i].cur_part_id = NO_PART;
1488 for (pno = 0; pno < parts->num_part; pno++) {
1489 struct disk_part_info t;
1490
1491 if (!parts->pscheme->get_part_info(parts, pno, &t))
1492 continue;
1493
1494 for (i = 0; i < wanted->num; i++) {
1495 if (wanted->infos[i].cur_part_id != NO_PART)
1496 continue;
1497 if (t.start == infos[i].start) {
1498 wanted->infos[i].cur_part_id = pno;
1499 wanted->infos[i].cur_start = infos[i].start;
1500 wanted->infos[i].cur_flags = infos[i].flags;
1501 break;
1502 }
1503 }
1504 }
1505 free(infos);
1506
1507 /* sort, and sync part ids and wanted->infos[] indices */
1508 sort_and_sync_parts(wanted);
1509}

1540/*
1541 * md back-end code for menu-driven BSD disklabel editor.
1542 * returns 0 on failure, 1 on success.
1543 * fills the install target with a list for newfs/fstab.
1544 */
1545bool
1546make_bsd_partitions(struct install_partition_desc *install)
1547{
1548 struct disk_partitions *parts = pm->parts;
1549 const struct disk_partitioning_scheme *pscheme;
1550 struct partition_usage_set wanted;
1551 enum layout_type layoutkind = LY_SETSIZES;
1552 bool have_existing;
1553
1554 if (pm && pm->no_part && parts == NULL)
1555 return true;
1556
1557 if (parts == NULL) {
1558 pscheme = select_part_scheme(pm, NULL, true, NULL);
1559 if (pscheme == NULL)
1560 return false;
1561 parts = pscheme->create_new_for_disk(pm->diskdev,
1562 0, pm->dlsize, pm->dlsize, true);
1563 if (parts == NULL)
1564 return false;
1565 pm->parts = parts;
1566 } else {
1567 pscheme = parts->pscheme;
1568 }
1569
1570 if (pscheme->secondary_partitions) {
1571 struct disk_partitions *p;
1572
1573 p = pscheme->secondary_partitions(parts, pm->ptstart, false);
1574 if (p) {
1575 parts = p;
1576 pscheme = parts->pscheme;
1577 }
1578 }
1579
1580 have_existing = check_existing_netbsd(parts);
1581
1582 /*
1583 * Initialize global variables that track space used on this disk.
1584 */
1585 if (pm->ptsize == 0)
1586 pm->ptsize = pm->dlsize - pm->ptstart;
1587 if (pm->dlsize == 0)
1588 pm->dlsize = pm->ptstart + pm->ptsize;
1589
1590 if (logfp) fprintf(logfp, "dlsize=%" PRId64 " ptsize=%" PRId64
1591 " ptstart=%" PRId64 "\n",
1592 pm->dlsize, pm->ptsize, pm->ptstart);
1593
1594 if (pm->current_cylsize == 0)
1595 pm->current_cylsize = pm->dlcylsize;
1596
1597 /* Ask for layout type -- standard or special */
1598 if (partman_go == 0) {
1599 char bsd_size[6], min_size[6], x_size[6];
1600
1601 humanize_number(bsd_size, sizeof(bsd_size),
1602 (uint64_t)pm->ptsize*pm->sectorsize,
1603 "", HN_AUTOSCALE, HN_B | HN_NOSPACE | HN_DECIMAL);
1604 humanize_number(min_size, sizeof(min_size),
1605 (uint64_t)(DEFROOTSIZE + DEFSWAPSIZE + DEFUSRSIZE)*MEG,
1606 "", HN_AUTOSCALE, HN_B | HN_NOSPACE | HN_DECIMAL);
1607 humanize_number(x_size, sizeof(x_size),
1608 (uint64_t)(DEFROOTSIZE + DEFSWAPSIZE + DEFUSRSIZE
1609 + XNEEDMB)*MEG,
1610 "", HN_AUTOSCALE, HN_B | HN_NOSPACE | HN_DECIMAL);
1611
1612 msg_display_subst(
1613 have_existing ? MSG_layout_prologue_existing
1614 : MSG_layout_prologue_none, 6, pm->diskdev,
1615 msg_string(parts->pscheme->name),
1616 msg_string(parts->pscheme->short_name),
1617 bsd_size, min_size, x_size);
1618 msg_display_add_subst(MSG_layout_main, 6,
1619 pm->diskdev,
1620 msg_string(parts->pscheme->name),
1621 msg_string(parts->pscheme->short_name),
1622 bsd_size, min_size, x_size);
1623 msg_display_add("\n\n");
1624 layoutkind = ask_layout(parts, have_existing);
1625 }
1626
1627 if (layoutkind == LY_USEDEFAULT) {
1628 replace_by_default(pm, parts, pm->ptstart, pm->ptsize,
1629 &wanted);
1630 } else if (layoutkind == LY_SETSIZES) {
1631 if (!edit_with_defaults(pm, parts, pm->ptstart, pm->ptsize,
1632 &wanted)) {
1633 free_usage_set(&wanted);
1634 return false;
1635 }
1636 } else {
1637 usage_set_from_parts(&wanted, parts);
1638 }
1639
1640 /*
1641 * OK, we have a partition table. Give the user the chance to
1642 * edit it and verify it's OK, or abort altogether.
1643 */
1644 for (;;) {
1645 int rv = edit_and_check_label(pm, &wanted);
1646 if (rv == 0) {
1647 msg_display(MSG_abort_part);
1648 free_usage_set(&wanted);
1649 return false;
1650 }
1651 /* update install infos */
1652 install->num = wanted.num;
1653 install->infos = wanted.infos;
1654 /* and check them */
1655 if (check_partitions(install))
1656 break;
1657 }
1658
1659 /* we moved infos from wanted to install target */
1660 wanted.infos = NULL;
1661 free_usage_set(&wanted);
1662
1663 /* Everything looks OK. */
1664 return true;
1665}

1671/*
1672 * check that there is at least a / somewhere.
1673 */
1674bool
1675check_partitions(struct install_partition_desc *install)
1676{

1680#endif

1688 bootxx = bootxx_name(install);
1689 if (bootxx != NULL) {
1690 rv = access(bootxx, R_OK);
1691 free(bootxx);
1692 } else
1693 rv = -1;

1695 hit_enter_to_continue(NULL, MSG_No_Bootcode);
1696 return false;
1697 }
1698 }
1699#endif
1700#ifndef HAVE_UFS2_BOOT
1701 if (MD_NEED_BOOTBLOCK(install)) {
1702 for (i = 0; i < install->num; i++) {
1703 if (install->infos[i].type != PT_root)
1704 continue;
1705 if (strcmp(install->infos[i].mount, "/") != 0)
1706 continue;
1707 if (install->infos[i].fs_type != FS_BSDFFS)
1708 continue;
1709 if (install->infos[i].fs_version != 2)
1710 continue;
1711 hit_enter_to_continue(NULL, MSG_cannot_ufs2_root);
1712 return false;
