108static void
109config_list_init(void)
110{
111 int i;
112
113 for (i=0; i < CONFIGOPT_LAST; i++) {
114 switch (i) {
115 case CONFIGOPT_TZ:
116 get_tz_default();
117 config_list[CONFIGOPT_TZ].setting = tz_default;
118 break;
119 case CONFIGOPT_ROOTSH:
120 get_rootsh();
121 break;
122 case CONFIGOPT_ROOTPW:
123 if (check_root_password())
124 config_list[i].setting = MSG_password_set;
125 else
126 config_list[i].setting = MSG_empty;
127 break;
128 default:
129 if (config_list[i].rcvar != NULL) {
130 if (check_rcvar(config_list[i].rcvar))
131 config_list[i].setting = MSG_YES;
132 else
133 config_list[i].setting = MSG_NO;
134 }
135 break;
136 }
137 }
138}

140static void
141get_rootsh(void)
142{
143 static char *buf = NULL;
144
145 if (buf != NULL)
146 free(buf);
147
148 if (target_already_root())
149 collect(T_OUTPUT, &buf,
150 "/usr/bin/awk -F: '$1==\"root\" { print $NF; exit }'"
151 " /etc/passwd");
152 else
153 collect(T_OUTPUT, &buf,
154 "chroot %s /usr/bin/awk -F: '$1==\"root\" { print $NF; exit }'"
155 " /etc/passwd",target_prefix());
156
157 config_list[CONFIGOPT_ROOTSH].setting = (const char *)buf;
158}

160static void
161set_config(menudesc *menu, int opt, void *arg)
162{
163 configinfo **configp = arg;
164 configinfo *config = configp[opt];
165 const char *optname, *setting;
166
167 optname = config->optname;
168 setting = msg_string(config->setting);
169
170 wprintw(menu->mw, "%-50s %-10s", msg_string(optname), setting);
171}

173static int
174init_config_menu(configinfo *conf, menu_ent *me, configinfo **ce)
175{
176 int opt;
177 int configopts;
178
179 for (configopts = 0; ; conf++) {
180 opt = conf->opt;
181 if (opt == CONFIGOPT_LAST)
182 break;
183 *ce = conf;
184 memset(me, 0, sizeof(*me));
185 me->opt_action = conf->action;
186 configopts++;
187 ce++;
188 me++;
189 }
190
191 return configopts;
192}

205static int
206set_root_shell(struct menudesc *menu, void *arg)
207{
208 configinfo **confp = arg;
209
210 process_menu(MENU_rootsh, &confp[menu->cursel]->setting);
211 if (run_program(RUN_PROGRESS | RUN_CHROOT,
212 "chpass -s %s root", confp[menu->cursel]->setting) != 0)
213 confp[menu->cursel]->setting = MSG_failed;
214 return 0;
215}

226static int
227check_root_password(void)
228{
229 char *buf;
230 int rval;
231
232 if (target_already_root())
233 collect(T_OUTPUT, &buf, "getent passwd root | cut -d: -f2");
234 else
235 collect(T_OUTPUT, &buf, "chroot %s getent passwd root | "
236 "chroot %s cut -d: -f2",
237 target_prefix(), target_prefix());
238
239 if (logfp)
240 fprintf(logfp,"buf %s strlen(buf) %zu\n", buf, strlen(buf));
241
242 if (strlen(buf) <= 1) /* newline */
243 rval = 0;
244 else
245 rval = 1;
246 free(buf);
247 return rval;
248}

250static int
251add_new_user(struct menudesc *menu, void *arg)
252{
253 char username[STRSIZE] = "";
254 int inwheel=0;
255
256 msg_prompt(MSG_addusername, NULL, username, sizeof username -1);
257 if (strlen(username) == 0)
258 return 0;
259 inwheel = ask_yesno(MSG_addusertowheel);
260 ushell = "/bin/csh";
261 process_menu(MENU_usersh, NULL);
262 if (inwheel)
263 run_program(RUN_PROGRESS | RUN_CHROOT,
264 "/usr/sbin/useradd -m -s %s -G wheel %s",
265 ushell, username);
266 else
267 run_program(RUN_PROGRESS | RUN_CHROOT,
268 "/usr/sbin/useradd -m -s %s %s", ushell, username);
269 run_program(RUN_DISPLAY | RUN_PROGRESS | RUN_CHROOT,
270 "passwd -l %s", username);
271 return 0;
272}

274static int
275change_root_password(struct menudesc *menu, void *arg)
276{
277 configinfo **confp = arg;
278
279 msg_display(MSG_rootpw);
280 if (ask_yesno(NULL)) {
281 if (run_program(RUN_DISPLAY | RUN_PROGRESS | RUN_CHROOT,
282 "passwd -l root") == 0)
283 confp[menu->cursel]->setting = MSG_password_set;
284 else
285 confp[menu->cursel]->setting = MSG_failed;
286 }
287 return 0;
288}

290static int
291set_binpkg(struct menudesc *menu, void *arg)
292{
293 configinfo **confp = arg;
294 char additional_pkgs[STRSIZE] = {0};
295 int allok = 0;
296 arg_rv parm;
297
298 do {
299 parm.rv = -1;
300 parm.arg = additional_pkgs;
301 process_menu(MENU_binpkg, &parm);
302 if (parm.rv == SET_SKIP) {
303 confp[menu->cursel]->setting = MSG_abandoned;
304 return 0;
305 }
306
307 make_url(pkgpath, &pkg, pkg_dir);
308 if (run_program(RUN_DISPLAY | RUN_PROGRESS | RUN_CHROOT,
309 "pkg_add %s/pkgin", pkgpath) == 0) {
310 allok = 1;
311 }
312 } while (allok == 0);
313
314 /* configure pkgin to use $pkgpath as a repository */
315 replace("/usr/pkg/etc/pkgin/repositories.conf", "s,^[^#].*$,%s,",
316 pkgpath);
317
318 run_program(RUN_DISPLAY | RUN_PROGRESS | RUN_CHROOT,
319 "/usr/pkg/bin/pkgin -y update");
320
321 if (strlen(additional_pkgs) > 0)
322 run_program(RUN_DISPLAY | RUN_PROGRESS | RUN_CHROOT,
323 "/usr/pkg/bin/pkgin -y install %s", additional_pkgs);
324
325 hit_enter_to_continue(MSG_binpkg_installed, NULL);
326
327 confp[menu->cursel]->setting = MSG_DONE;
328 return 0;
329}

331static int
332set_pkgsrc(struct menudesc *menu, void *arg)
333{
334 configinfo **confp = arg;
335 distinfo dist;
336
337 dist.name = "pkgsrc";
338 dist.set = SET_PKGSRC;
339 dist.desc = "source for 3rd-party packages";
340 dist.marker_file = NULL;
341
342 int status = SET_RETRY;
343
344 do {
345 status = get_pkgsrc();
346 if (status == SET_OK) {
347 status = extract_file(&dist, 0);
348 continue;
349 } else if (status == SET_SKIP) {
350 confp[menu->cursel]->setting = MSG_abandoned;
351 return 0;
352 }
353 if (!ask_yesno(MSG_retry_pkgsrc_network)) {
354 confp[menu->cursel]->setting = MSG_abandoned;
355 return 1;
356 }
357 }
358 while (status == SET_RETRY);
359
360 confp[menu->cursel]->setting = MSG_DONE;
361 return 0;
362}

364static int
365toggle_rcvar(struct menudesc *menu, void *arg)
366{
367 configinfo **confp = arg;
368 int s;
369 const char *setting, *varname;
370 char pattern[STRSIZE];
371 char buf[STRSIZE];
372 char *cp;
373 int found = 0;
374 FILE *fp;
375
376 varname = confp[menu->cursel]->rcvar;
377
378 s = check_rcvar(varname);
379
380 /* we're toggling, so invert the sense */
381 if (s) {
382 confp[menu->cursel]->setting = MSG_NO;
383 setting = "NO";
384 } else {
385 confp[menu->cursel]->setting = MSG_YES;
386 setting = "YES";
387 }
388
389 if (!(fp = fopen(target_expand("/etc/rc.conf"), "r"))) {
390 msg_fmt_display(MSG_openfail, "%s%s",
391 target_expand("/etc/rc.conf"), strerror(errno));
392 hit_enter_to_continue(NULL, NULL);
393 return 0;
394 }
395
396 while (fgets(buf, sizeof buf, fp) != NULL) {
397 cp = buf + strspn(buf, " \t"); /* Skip initial spaces */
398 if (strncmp(cp, varname, strlen(varname)) == 0) {
399 cp += strlen(varname);
400 if (*cp != '=')
401 continue;
402 buf[strlen(buf) - 1] = 0;
403 snprintf(pattern, sizeof pattern,
404 "s,^%s$,%s=%s,",
405 buf, varname, setting);
406 found = 1;
407 break;
408 }
409 }
410
411 fclose(fp);
412
413 if (!found) {
414 add_rc_conf("%s=%s\n", varname, setting);
415 if (logfp) {
416 fprintf(logfp, "adding %s=%s\n", varname, setting);
417 fflush(logfp);
418 }
419 } else {
420 if (logfp) {
421 fprintf(logfp, "replacement pattern is %s\n", pattern);
422 fflush(logfp);
423 }
424 replace("/etc/rc.conf", "%s", pattern);
425 }
426
427 return 0;
428}

436void
437do_configmenu(struct install_partition_desc *install)
438{
439 int menu_no;
440 int opts;
441 menu_ent me[CONFIGOPT_LAST];
442 configinfo *ce[CONFIGOPT_LAST];
443
444 memset(me, 0, sizeof(me));
445
446 /* if the target isn't mounted already, figure it out. */
447 if (install != NULL && target_mounted() == 0) {
448 partman_go = 0;
449 if (find_disks(msg_string(MSG_configure_prior)) < 0)
450 return;
451
452 if (mount_disks(install) != 0)
453 return;
454 }
455
456 config_list_init();
457 make_url(pkgpath, &pkg, pkg_dir);
458 opts = init_config_menu(config_list, me, ce);
459
460 wrefresh(curscr);
461 wmove(stdscr, 0, 0);
462 wclear(stdscr);
463 wrefresh(stdscr);
464
465 menu_no = new_menu(NULL, me, opts, 0, -4, 0, 70,
466 MC_SCROLL | MC_NOBOX | MC_DFLTEXIT,
467 configmenu_hdr, set_config, NULL, "XXX Help String",
468 MSG_doneconfig);
469
470 process_menu(menu_no, ce);
471 free_menu(menu_no);
472}
