147/*
148 * Notes on the extended partition editor.
149 *
150 * The extended partition structure is actually a singly linked list.
151 * Each of the 'mbr' sectors can only contain 2 items, the first describes
152 * a user partition (relative to that mbr sector), the second describes
153 * the following partition (relative to the start of the extended partition).
154 *
155 * The 'start' sector for the user partition is always the size of one
156 * track - very often 63. The extended partitions themselves should
157 * always start on a cylinder boundary using the BIOS geometry - often
158 * 16065 sectors per cylinder.
159 *
160 * The disk is also always described in increasing sector order.
161 *
162 * During editing we keep the mbr sectors accurate (it might have been
163 * easier to use absolute sector numbers though), and keep a copy of the
164 * entire sector - to preserve any information any other OS has tried
165 * to squirrel away in the (apparently) unused space.
166 *
167 * Typical disk (with some small numbers):
168 *
169 * 0 -> a 63 37 dos
170 * b 100 1000 extended LBA (type 15)
171 *
172 * 100 -> a 63 37 user
173 * b 100 200 extended partiton (type 5)
174 *
175 * 200 -> a 63 37 user
176 * b 200 300 extended partiton (type 5)
177 *
178 * 300 -> a 63 37 user
179 * b 0 0 0 (end of chain)
180 *
181 */
182

267/*
268 * get C/H/S geometry from user via menu interface and
269 * store in globals.
270 */
271void
272set_bios_geom(struct disk_partitions *parts, int *cyl, int *head, int *sec)
273{
274 char res[80];
275 int bsec, bhead, bcyl;
276 daddr_t s;
277
278 if (parts->pscheme->change_disk_geom == NULL)
279 return;
280
281 msg_display_add(MSG_setbiosgeom);
282
283 do {
284 snprintf(res, 80, "%d", *sec);
285 msg_prompt_add(MSG_sectors, res, res, 80);
286 bsec = atoi(res);
287 } while (bsec <= 0 || bsec > MAXSECTOR);
288
289 do {
290 snprintf(res, 80, "%d", *head);
291 msg_prompt_add(MSG_heads, res, res, 80);
292 bhead = atoi(res);
293 } while (bhead <= 0 || bhead > MAXHEAD);
294 s = min(pm->dlsize, mbr_parts.size_limit);
295 bcyl = s / bsec / bhead;
296 if (s != bcyl * bsec * bhead)
297 bcyl++;
298 if (bcyl > MAXCYL)
299 bcyl = MAXCYL;
300 pm->max_chs = (unsigned long)bcyl * bhead * bsec;
301 pm->current_cylsize = bhead * bsec;
302 parts->pscheme->change_disk_geom(parts, bcyl, bhead, bsec);
303 *cyl = bcyl;
304 *head = bhead;
305 *sec = bsec;
306}

308static int
309find_mbr_space(const struct mbr_info_t *mbrs, uint *start, uint *size,
310 uint from, uint end_of_disk, uint ignore_at, bool primary_only)
311{
312 uint sz;
313 int i, j;
314 uint s, e;
315 const mbr_info_t *m, *me;
316 bool is_extended;
317
318 check_again:
319 m = mbrs;
320 sz = end_of_disk-from;
321 if (from >= end_of_disk)
322 return -1;
323
324 for (i = 0; i < MBR_PART_COUNT; i++) {
325 if (m->mbr.mbr_parts[i].mbrp_type == MBR_PTYPE_UNUSED)
326 continue;
327
328 is_extended = MBR_IS_EXTENDED(
329 m->mbr.mbr_parts[i].mbrp_type);
330
331 s = m->mbr.mbr_parts[i].mbrp_start + m->sector;
332 if (s == ignore_at)
333 continue;
334 e = s + m->mbr.mbr_parts[i].mbrp_size;
335 if (s <= from && e > from
336 && (!is_extended || primary_only)) {
337 if (e - 1 >= end_of_disk)
338 break;
339 if (e >= UINT_MAX) {
340 sz = 0;
341 break;
342 }
343 from = e + 1;
344 goto check_again;
345 }
346 if (s <= from && e > from && is_extended) {
347 /*
348 * if we start in the extended partiton,
349 * we must end before its end
350 */
351 sz = e - from;
352 }
353 if (s > from && s - from < sz)
354 sz = s - from;
355
356 if (is_extended) {
357 for (me = m->extended; me != NULL; me = me->extended) {
358 for (j = 0; j < MBR_PART_COUNT; j++) {
359 if (me->mbr.mbr_parts[j].mbrp_type ==
360 MBR_PTYPE_UNUSED)
361 continue;
362
363 is_extended = MBR_IS_EXTENDED(
364 me->mbr.mbr_parts[j].mbrp_type);
365
366 if (is_extended && i > 0)
367 break;
368
369 s = me->mbr.mbr_parts[j].mbrp_start +
370 me->sector;
371 if (s == ignore_at)
372 continue;
373 e = s + me->mbr.mbr_parts[j].mbrp_size;
374 if (me->sector != 0 && me->sector< s)
375 /*
376 * can not allow to overwrite
377 * the ext mbr
378 */
379 s = me->sector;
380 if (s <= from && e > from) {
381 if (e - 1 >= end_of_disk)
382 break;
383 from = e + 1;
384 goto check_again;
385 }
386 if (s > from && s - from < sz)
387 sz = s - from;
388
389 }
390 }
391 }
392 }
393
394 if (sz == 0)
395 return -1;
396 if (start != NULL)
397 *start = from;
398 if (size != NULL)
399 *size = sz;
400 return 0;
401}

403#ifdef BOOTSEL
404static int
405validate_and_set_names(mbr_info_t *mbri, const struct mbr_bootsel *src,
406 uint32_t ext_base)
407{
408 size_t i, l;
409 const unsigned char *p;
410
411 /*
412 * The 16 bit magic used to detect whether mbr_bootsel is valid
413 * or not is pretty week - collisions have been seen in the wild;
414 * but maybe it is just foreign tools corruption reminiscents
415 * of NetBSD MBRs. Anyway, before accepting a boot menu definition,
416 * make sure it is kinda "sane".
417 */
418
419 for (i = 0; i < MBR_PART_COUNT; i++) {
420 /*
421 * Make sure the name does not contain controll chars
422 * (not using iscntrl due to minimalistic locale support
423 * in miniroot environments) and is properly 0-terminated.
424 */
425 for (l = 0, p = (const unsigned char *)&src->mbrbs_nametab[i];
426 *p != 0; l++, p++) {
427 if (l > MBR_BS_PARTNAMESIZE)
428 return 0;
429 if (*p < ' ') /* hacky 'iscntrl' */
430 return 0;
431 }
432 }
433
434 memcpy(&mbri->mbrb, src, sizeof(*src));
435
436 if (ext_base == 0)
437 return mbri->mbrb.mbrbs_defkey - SCAN_1;
438 return 0;
439}

462static int
463read_mbr(const char *disk, mbr_info_t *mbri)
464{
465 struct mbr_partition *mbrp;
466 struct mbr_sector *mbrs = &mbri->mbr;
467 mbr_info_t *ext = NULL;
468 char diskpath[MAXPATHLEN];
469 int fd, i;
470 uint32_t ext_base = 0, next_ext = 0;
471 int rval = -1;
472#ifdef BOOTSEL
473 mbr_info_t *ombri = mbri;
474 int bootkey = 0;
475#endif
476
477 memset(mbri, 0, sizeof *mbri);
478
479 /* Open the disk. */
480 fd = opendisk(disk, O_RDONLY, diskpath, sizeof(diskpath), 0);
481 if (fd < 0)
482 goto bad_mbr;
483
484 for (;;) {
485 if (pread(fd, mbrs, sizeof *mbrs,
486 (ext_base + next_ext) * (off_t)MBR_SECSIZE) - sizeof *mbrs != 0)
487 break;
488
489 if (!valid_mbr(mbrs))
490 break;
491
492 mbrp = &mbrs->mbr_parts[0];
493 if (ext_base != 0) {
494 /* sanity check extended chain */
495 if (MBR_IS_EXTENDED(mbrp[0].mbrp_type))
496 break;
497 if (mbrp[1].mbrp_type != MBR_PTYPE_UNUSED &&
498 !MBR_IS_EXTENDED(mbrp[1].mbrp_type))
499 break;
500 if (mbrp[2].mbrp_type != MBR_PTYPE_UNUSED
501 || mbrp[3].mbrp_type != MBR_PTYPE_UNUSED)
502 break;
503 /* Looks ok, link into extended chain */
504 mbri->extended = ext;
505 ext->extended = NULL;
506 mbri = ext;
507 ext = NULL;
508 }
509#if BOOTSEL
510 if (mbrs->mbr_bootsel_magic == htole16(MBR_MAGIC)) {
511 /* old bootsel, grab bootsel info */
512 bootkey = validate_and_set_names(mbri,
513 (struct mbr_bootsel *)
514 ((uint8_t *)mbrs + MBR_BS_OLD_OFFSET),
515 ext_base);
516 } else if (mbrs->mbr_bootsel_magic == htole16(MBR_BS_MAGIC)) {
517 /* new location */
518 bootkey = validate_and_set_names(mbri,
519 &mbrs->mbr_bootsel, ext_base);
520 }
521 /* Save original flags for mbr code update tests */
522 mbri->oflags = mbri->mbrb.mbrbs_flags;
523#endif
524 mbri->sector = next_ext + ext_base;
525 next_ext = 0;
526 rval = 0;
527 for (i = 0; i < MBR_PART_COUNT; mbrp++, i++) {
528 if (mbrp->mbrp_type == MBR_PTYPE_UNUSED) {
529 /* type is unused, discard scum */
530 memset(mbrp, 0, sizeof *mbrp);
531 continue;
532 }
533 mbrp->mbrp_start = le32toh(mbrp->mbrp_start);
534 mbrp->mbrp_size = le32toh(mbrp->mbrp_size);
535 if (MBR_IS_EXTENDED(mbrp->mbrp_type)) {
536 next_ext = mbrp->mbrp_start;
537 } else {
538 uint flags = 0;
539 if (mbrp->mbrp_type == MBR_PTYPE_NETBSD)
540 flags |= GLM_LIKELY_FFS;
541 else if (mbrp->mbrp_type == MBR_PTYPE_FAT12 ||
542 mbrp->mbrp_type == MBR_PTYPE_FAT16S ||
543 mbrp->mbrp_type == MBR_PTYPE_FAT16B ||
544 mbrp->mbrp_type == MBR_PTYPE_FAT32 ||
545 mbrp->mbrp_type == MBR_PTYPE_FAT32L ||
546 mbrp->mbrp_type == MBR_PTYPE_FAT16L)
547 flags |= GLM_MAYBE_FAT32;
548 else if (mbrp->mbrp_type == MBR_PTYPE_NTFS)
549 flags |= GLM_MAYBE_NTFS;
550 if (flags != 0) {
551 const char *mount = get_last_mounted(
552 fd, mbri->sector + mbrp->mbrp_start,
553 &mbri->fs_type[i],
554 &mbri->fs_sub_type[i],
555 flags);
556 mbri->last_mounted[i] = strdup(mount);
557 }
558 }
559#if BOOTSEL
560 if (mbri->mbrb.mbrbs_nametab[i][0] != 0
561 && bootkey-- == 0)
562 ombri->bootsec = mbri->sector +
563 mbrp->mbrp_start;
564#endif
565 }
566
567 if (next_ext == 0 || ext_base + next_ext <= mbri->sector)
568 break;
569 if (ext_base == 0) {
570 ext_base = next_ext;
571 next_ext = 0;
572 }
573 ext = calloc(sizeof *ext, 1);
574 if (!ext)
575 break;
576 mbrs = &ext->mbr;
577 }
578
579 bad_mbr:
580 free_mbr_info(ext);
581 if (fd >= 0)
582 close(fd);
583 if (rval == -1) {
584 memset(&mbrs->mbr_parts, 0, sizeof mbrs->mbr_parts);
585 mbrs->mbr_magic = htole16(MBR_MAGIC);
586 }
587 dump_mbr(ombri, "read");
588 return rval;
589}

591static int
592write_mbr(const char *disk, mbr_info_t *mbri, int bsec, int bhead, int bcyl)
593{
594 char diskpath[MAXPATHLEN];
595 int fd, i, ret = 0;
596 struct mbr_partition *mbrp;
597 u_int32_t pstart, psize;
598#ifdef BOOTSEL
599 struct mbr_sector *mbrs;
600#endif
601 struct mbr_sector mbrsec;
602 mbr_info_t *ext;
603 uint sector;
604
605 dump_mbr(mbri, "write");
606
607 /* Open the disk. */
608 fd = opendisk(disk, O_WRONLY, diskpath, sizeof(diskpath), 0);
609 if (fd < 0)
610 return -1;
611
612#ifdef BOOTSEL
613 /*
614 * If the main boot code (appears to) contain the netbsd bootcode,
615 * copy in all the menu strings and set the default keycode
616 * to be that for the default partition.
617 * Unfortunately we can't rely on the user having actually updated
618 * to the new mbr code :-(
619 */
620 if (mbri->mbr.mbr_bootsel_magic == htole16(MBR_BS_MAGIC)
621 || mbri->mbr.mbr_bootsel_magic == htole16(MBR_MAGIC)) {
622 int8_t key = SCAN_1;
623 uint offset = MBR_BS_OFFSET;
624 if (mbri->mbr.mbr_bootsel_magic == htole16(MBR_MAGIC))
625 offset = MBR_BS_OLD_OFFSET;
626 mbri->mbrb.mbrbs_defkey = SCAN_ENTER;
627 if (mbri->mbrb.mbrbs_timeo == 0)
628 mbri->mbrb.mbrbs_timeo = 182; /* 10 seconds */
629 for (ext = mbri; ext != NULL; ext = ext->extended) {
630 mbrs = &ext->mbr;
631 mbrp = &mbrs->mbr_parts[0];
632 /* Ensure marker is set in each sector */
633 mbrs->mbr_bootsel_magic = mbri->mbr.mbr_bootsel_magic;
634 /* and copy in bootsel parameters */
635 *(struct mbr_bootsel *)((uint8_t *)mbrs + offset) =
636 ext->mbrb;
637 for (i = 0; i < MBR_PART_COUNT; i++) {
638 if (ext->mbrb.mbrbs_nametab[i][0] == 0)
639 continue;
640 if (ext->sector + mbrp->mbrp_start ==
641 mbri->bootsec)
642 mbri->mbrb.mbrbs_defkey = key;
643 key++;
644 }
645 }
646 /* copy main data (again) since we've put the 'key' in */
647 *(struct mbr_bootsel *)((uint8_t *)&mbri->mbr + offset) =
648 mbri->mbrb;
649 }
650#endif
651
652 for (ext = mbri; ext != NULL; ext = ext->extended) {
653 sector = ext->sector;
654 mbrsec = ext->mbr; /* copy sector */
655 mbrp = &mbrsec.mbr_parts[0];
656
657 if (sector != 0 && ext->extended != NULL
658 && ext->extended->mbr.mbr_parts[0].mbrp_type
659 == MBR_PTYPE_UNUSED) {
660
661 /*
662 * This should never happen nowadays, old code
663 * inserted empty ext sectors in the chain to
664 * help the gui out - we do not do that anymore.
665 */
666 assert(false);
667
668 /* We are followed by an empty slot, collapse out */
669 ext = ext->extended;
670 /* Make us describe the next non-empty partition */
671 mbrp[1] = ext->mbr.mbr_parts[1];
672 }
673
674 for (i = 0; i < MBR_PART_COUNT; i++) {
675 if (mbrp[i].mbrp_start == 0 && mbrp[i].mbrp_size == 0) {
676 mbrp[i].mbrp_scyl = 0;
677 mbrp[i].mbrp_shd = 0;
678 mbrp[i].mbrp_ssect = 0;
679 mbrp[i].mbrp_ecyl = 0;
680 mbrp[i].mbrp_ehd = 0;
681 mbrp[i].mbrp_esect = 0;
682 continue;
683 }
684 pstart = mbrp[i].mbrp_start;
685 psize = mbrp[i].mbrp_size;
686 mbrp[i].mbrp_start = htole32(pstart);
687 mbrp[i].mbrp_size = htole32(psize);
688 if (bsec && bcyl && bhead) {
689 convert_mbr_chs(bcyl, bhead, bsec,
690 &mbrp[i].mbrp_scyl, &mbrp[i].mbrp_shd,
691 &mbrp[i].mbrp_ssect, pstart);
692 convert_mbr_chs(bcyl, bhead, bsec,
693 &mbrp[i].mbrp_ecyl, &mbrp[i].mbrp_ehd,
694 &mbrp[i].mbrp_esect, pstart + psize - 1);
695 }
696 }
697
698 mbrsec.mbr_magic = htole16(MBR_MAGIC);
699 if (pwrite(fd, &mbrsec, sizeof mbrsec,
700 sector * (off_t)MBR_SECSIZE) < 0) {
701 ret = -1;
702 break;
703 }
704 }
705
706 (void)close(fd);
707 return ret;
708}

710static void
711convert_mbr_chs(int cyl, int head, int sec,
712 uint8_t *cylp, uint8_t *headp, uint8_t *secp,
713 uint32_t relsecs)
714{
715 unsigned int tcyl, temp, thead, tsec;
716
717 temp = head * sec;
718 tcyl = relsecs / temp;
719 relsecs -= tcyl * temp;
720
721 thead = relsecs / sec;
722 tsec = relsecs - thead * sec + 1;
723
724 if (tcyl > MAXCYL)
725 tcyl = MAXCYL;
726
727 *cylp = MBR_PUT_LSCYL(tcyl);
728 *headp = thead;
729 *secp = MBR_PUT_MSCYLANDSEC(tcyl, tsec);
730}

732/*
733 * This function is ONLY to be used as a last resort to provide a
734 * hint for the user. Ports should provide a more reliable way
735 * of getting the BIOS geometry. The i386 code, for example,
736 * uses the BIOS geometry as passed on from the bootblocks,
737 * and only uses this as a hint to the user when that information
738 * is not present, or a match could not be made with a NetBSD
739 * device.
740 */
741int
742guess_biosgeom_from_parts(struct disk_partitions *parts,
743 int *cyl, int *head, int *sec)
744{
745
746 /*
747 * The physical parameters may be invalid as bios geometry.
748 * If we cannot determine the actual bios geometry, we are
749 * better off picking a likely 'faked' geometry than leaving
750 * the invalid physical one.
751 */
752
753 int xcylinders = pm->dlcyl;
754 int xheads = pm->dlhead;
755 daddr_t xsectors = pm->dlsec;
756 daddr_t xsize = min(pm->dlsize, mbr_parts.size_limit);
757 if (xcylinders > MAXCYL || xheads > MAXHEAD || xsectors > MAXSECTOR) {
758 xsectors = MAXSECTOR;
759 xheads = MAXHEAD;
760 xcylinders = xsize / (MAXSECTOR * MAXHEAD);
761 if (xcylinders > MAXCYL)
762 xcylinders = MAXCYL;
763 }
764 *cyl = xcylinders;
765 *head = xheads;
766 *sec = xsectors;
767
768 if (parts->pscheme->guess_disk_geom == NULL)
769 return -1;
770
771 return parts->pscheme->guess_disk_geom(parts, cyl, head, sec);
772}

798static void
799mbr_init_default_alignments(struct mbr_disk_partitions *parts, uint track)
800{
801 if (track == 0)
802 track = 16065;
803
804 assert(parts->dp.disk_size > 0);
805 if (parts->dp.disk_size < 0)
806 return;
807
808 /* Use 1MB offset/alignemnt for large (>128GB) disks */
809 if (parts->dp.disk_size > HUGE_DISK_SIZE) {
810 parts->ptn_alignment = 2048;
811 parts->ptn_0_offset = 2048;
812 } else if (parts->dp.disk_size > TINY_DISK_SIZE) {
813 parts->ptn_alignment = 64;
814 parts->ptn_0_offset = parts->geo_sec;
815 } else {
816 parts->ptn_alignment = 1;
817 parts->ptn_0_offset = parts->geo_sec;
818 }
819 parts->ext_ptn_alignment = track;
820}

851static void
852mbr_calc_free_space(struct mbr_disk_partitions *parts)
853{
854 size_t i;
855 mbr_info_t *m;
856
857 parts->dp.free_space = parts->dp.disk_size - 1;
858 parts->dp.num_part = 0;
859 m = &parts->mbr;
860 do {
861 for (i = 0; i < MBR_PART_COUNT; i++) {
862 if (m->mbr.mbr_parts[i].mbrp_type == MBR_PTYPE_UNUSED)
863 continue;
864
865 if (m != &parts->mbr && i > 0 &&
866 MBR_IS_EXTENDED(m->mbr.mbr_parts[i].mbrp_type))
867 break;
868
869 parts->dp.num_part++;
870 if (MBR_IS_EXTENDED(m->mbr.mbr_parts[i].mbrp_type))
871 continue;
872
873 daddr_t psize = m->mbr.mbr_parts[i].mbrp_size;
874 if (m != &parts->mbr)
875 psize += m->mbr.mbr_parts[i].mbrp_start;
876
877 if (psize > parts->dp.free_space)
878 parts->dp.free_space = 0;
879 else
880 parts->dp.free_space -= psize;
881
882 }
883 } while ((m = m->extended));
884}

918static bool
919mbr_write_to_disk(struct disk_partitions *new_state)
920{
921 struct mbr_disk_partitions *parts =
922 (struct mbr_disk_partitions *)new_state;
923 unsigned long bsec, bhead, bcyl;
924 daddr_t t;
925
926 assert(parts->geo_sec != 0);
927 if (parts->geo_sec != 0) {
928 bsec = parts->geo_sec;
929 bhead = parts->ext_ptn_alignment / bsec;
930 } else {
931 bsec = MAXSECTOR;
932 bhead = MAXHEAD;
933 }
934 t = bsec * bhead;
935 assert(t != 0);
936 if ((daddr_t)(1UL<<10) * t <= parts->dp.disk_size)
937 bcyl = (1UL<<10) - 1;
938 else
939 bcyl = (unsigned long)(parts->dp.disk_size / t);
940
941 return write_mbr(parts->dp.disk, &parts->mbr,
942 bsec, bhead, bcyl) == 0;
943}

945static bool
946mbr_change_disk_geom(struct disk_partitions *arg, int ncyl, int nhead,
947 int nsec)
948{
949 struct mbr_disk_partitions *parts = (struct mbr_disk_partitions *)arg;
950 daddr_t ptn_0_base, ptn_0_limit;
951 struct disk_part_info info;
952
953 /* Default to using 'traditional' cylinder alignment */
954 mbr_init_chs(parts, ncyl, nhead, nsec);
955 mbr_init_default_alignments(parts, nhead * nsec);
956
957 if (parts->dp.disk_size <= TINY_DISK_SIZE) {
958 set_default_sizemult(1);
959 return true;
960 }
961
962 if (parts->dp.num_part > 0 &&
963 parts->dp.pscheme->get_part_info(arg, 0, &info)) {
964
965 /* Try to copy offset of first partition */
966 ptn_0_base = info.start;
967 ptn_0_limit = info.start + info.size;
968 if (!(ptn_0_limit & 2047)) {
969 /* Partition ends on a 1MB boundary, align to 1MB */
970 parts->ptn_alignment = 2048;
971 if (ptn_0_base <= 2048
972 && !(ptn_0_base & (ptn_0_base - 1))) {
973 /* ptn_base is a power of 2, use it */
974 parts->ptn_0_offset = ptn_0_base;
975 }
976 }
977 }
978 set_default_sizemult(MEG/512);
979 return true;
980}

1176static bool
1177mbr_part_apply(const struct disk_partitions *arg, part_id id,
1178 bool (*func)(const struct disk_partitions *arg, part_id id,
1179 const mbr_info_t *mb, int i, bool primary,
1180 const struct mbr_partition *mp, void *),
1181 void *cookie)
1182{
1183 const struct mbr_disk_partitions *parts =
1184 (const struct mbr_disk_partitions*)arg;
1185 part_id i, j, no;
1186 const mbr_info_t *m = &parts->mbr, *me;
1187
1188 no = 0;
1189 for (i = 0; i < MBR_PART_COUNT; i++) {
1190 if (m->mbr.mbr_parts[i].mbrp_type == MBR_PTYPE_UNUSED)
1191 continue;
1192
1193 if (no == id) {
1194 return func(arg, id, m, i, true,
1195 &m->mbr.mbr_parts[i], cookie);
1196 }
1197 no++;
1198
1199 if (MBR_IS_EXTENDED(m->mbr.mbr_parts[i].mbrp_type)) {
1200 for (me = m->extended; me != NULL; me = me->extended) {
1201 for (j = 0; j < MBR_PART_COUNT; j++) {
1202 if (me->mbr.mbr_parts[j].mbrp_type ==
1203 MBR_PTYPE_UNUSED)
1204 continue;
1205 if (j > 0 && MBR_IS_EXTENDED(
1206 me->mbr.mbr_parts[j].mbrp_type))
1207 break;
1208 if (no == id) {
1209 return func(arg, id, me, j,
1210 false,
1211 &me->mbr.mbr_parts[j],
1212 cookie);
1213 }
1214 no++;
1215 }
1216 }
1217 }
1218 }
1219
1220
1221 return false;
1222}

1454static bool
1455mbr_set_part_info(struct disk_partitions *arg, part_id id,
1456 const struct disk_part_info *info, const char **err_msg)
1457{
1458 struct mbr_disk_partitions *parts =
1459 (struct mbr_disk_partitions*)arg;
1460 struct disk_part_info data = *info;
1461 part_id i, j, no, ext_ndx, t;
1462 mbr_info_t *m = &parts->mbr, *me;
1463 uint pt = mbr_type_from_gen_desc(info->nat_type);
1464
1465 if (MBR_IS_EXTENDED(pt)) {
1466 /* check for duplicate ext part */
1467 no = 0;
1468 t = ext_ndx = MBR_PART_COUNT;
1469 for (i = 0; i < MBR_PART_COUNT; i++) {
1470 if (m->mbr.mbr_parts[i].mbrp_type == MBR_PTYPE_UNUSED)
1471 continue;
1472 if (MBR_IS_EXTENDED(m->mbr.mbr_parts[i].mbrp_type))
1473 ext_ndx = i;
1474 if (no == id)
1475 t = i;
1476 no++;
1477 }
1478 if (ext_ndx < MBR_PART_COUNT && t != ext_ndx) {
1479 if (err_msg)
1480 *err_msg =
1481 msg_string(MSG_Only_one_extended_ptn);
1482 return false;
1483 }
1484 /* this partition becomes an extended one, apply alignment */
1485 data.start = max(roundup(data.start, parts->ext_ptn_alignment),
1486 parts->ext_ptn_alignment);
1487 }
1488
1489 no = 0;
1490 for (i = 0; i < MBR_PART_COUNT; i++) {
1491 if (m->mbr.mbr_parts[i].mbrp_type == MBR_PTYPE_UNUSED)
1492 continue;
1493
1494 if (no == id)
1495 goto found;
1496 no++;
1497
1498 if (MBR_IS_EXTENDED(m->mbr.mbr_parts[i].mbrp_type)) {
1499 for (me = m->extended; me != NULL; me = me->extended) {
1500 for (j = 0; j < MBR_PART_COUNT; j++) {
1501 if (me->mbr.mbr_parts[j].mbrp_type ==
1502 MBR_PTYPE_UNUSED)
1503 continue;
1504 if (j > 0 && MBR_IS_EXTENDED(
1505 me->mbr.mbr_parts[j].mbrp_type))
1506 break;
1507 if (no == id) {
1508 i = j;
1509 m = me;
1510 goto found;
1511 }
1512 no++;
1513 }
1514 }
1515 }
1516 }
1517
1518 if (err_msg)
1519 *err_msg = INTERNAL_ERROR;
1520 return false;
1521
1522found:
1523 /*
1524 * We assume that m is the mbr we want to update and
1525 * i is the local partition index into it.
1526 */
1527 if (m == &parts->mbr) {
1528 if (MBR_IS_EXTENDED(
1529 m->mbr.mbr_parts[i].mbrp_type) &&
1530 !ext_part_good(&parts->mbr, data.start, data.size)) {
1531 if (err_msg)
1532 *err_msg =
1533 MSG_mbr_ext_nofit;
1534 return false;
1535 }
1536 } else if (!inside_ext_part(&parts->mbr, data.start)) {
1537 if (err_msg)
1538 *err_msg = msg_string(MSG_mbr_inside_ext);
1539 return false;
1540 }
1541 uint start = data.start, size = data.size;
1542 uint oldstart = m->mbr.mbr_parts[i].mbrp_start + m->sector;
1543 if (parts->ptn_0_offset > 0 &&
1544 start < parts->ptn_0_offset)
1545 start = parts->ptn_0_offset;
1546 if (find_mbr_space(m, &start, &size, start, parts->dp.disk_size,
1547 oldstart, false) < 0) {
1548 if (err_msg != NULL)
1549 *err_msg = INTERNAL_ERROR;
1550 return false;
1551 }
1552 data.start = start;
1553 if (size < data.size)
1554 data.size = size;
1555 uint old_start = m->mbr.mbr_parts[i].mbrp_start;
1556 if (!mbr_info_to_partitition(&data,
1557 &m->mbr.mbr_parts[i], m->sector, err_msg))
1558 return false;
1559 if (data.last_mounted && m->last_mounted[i] &&
1560 data.last_mounted != m->last_mounted[i]) {
1561 free(__UNCONST(m->last_mounted[i]));
1562 m->last_mounted[i] = strdup(data.last_mounted);
1563 }
1564 if (data.fs_type != 0)
1565 m->fs_type[i] = data.fs_type;
1566 if (data.fs_sub_type != 0)
1567 m->fs_sub_type[i] = data.fs_sub_type;
1568
1569 if (m == &parts->mbr) {
1570 if (m->mbr.mbr_parts[i].mbrp_start !=
1571 old_start)
1572 mbr_sort_main_mbr(&m->mbr);
1573 } else {
1574 adjust_ext_part(&parts->mbr,
1575 data.start, data.size);
1576 }
1577 mbr_calc_free_space(parts);
1578 return true;
1579}

1656static int
1657get_mapping(struct mbr_partition *parts, int i,
1658 int *cylinder, int *head, int *sector, daddr_t *absolute)
1659{
1660 struct mbr_partition *apart = &parts[i / 2];
1661
1662 if (apart->mbrp_type == MBR_PTYPE_UNUSED)
1663 return -1;
1664 if (i % 2 == 0) {
1665 *cylinder = MBR_PCYL(apart->mbrp_scyl, apart->mbrp_ssect);
1666 *head = apart->mbrp_shd;
1667 *sector = MBR_PSECT(apart->mbrp_ssect) - 1;
1668 *absolute = le32toh(apart->mbrp_start);
1669 } else {
1670 *cylinder = MBR_PCYL(apart->mbrp_ecyl, apart->mbrp_esect);
1671 *head = apart->mbrp_ehd;
1672 *sector = MBR_PSECT(apart->mbrp_esect) - 1;
1673 *absolute = le32toh(apart->mbrp_start)
1674 + le32toh(apart->mbrp_size) - 1;
1675 }
1676 /* Sanity check the data against max values */
1677 if ((((*cylinder * MAXHEAD) + *head) * (uint32_t)MAXSECTOR + *sector) < *absolute)
1678 /* cannot be a CHS mapping */
1679 return -1;
1680
1681 return 0;
1682}

1747static bool
1748mbr_do_delete_part(const struct disk_partitions *arg, part_id id,
1749 const mbr_info_t *mb, int i, bool primary,
1750 const struct mbr_partition *mp, void *cookie)
1751{
1752 struct delete_part_args *marg = cookie;
1753 bool is_ext_part = MBR_IS_EXTENDED(mp->mbrp_type);
1754
1755 /* can not delete non-empty extended partitions */
1756 if (MBR_IS_EXTENDED(mp->mbrp_type)
1757 && marg->parts->mbr.extended != NULL) {
1758 if (marg->err_msg)
1759 *marg->err_msg = msg_string(MSG_mbr_ext_not_empty);
1760 return false;
1761 }
1762
1763 /* return position/size to caller */
1764 marg->start = mb->sector + mp->mbrp_start;
1765 marg->size = mp->mbrp_size;
1766
1767 if (primary) {
1768 /* if deleting the primary extended partition, just kill it */
1769 struct mbr_partition *md = &marg->parts->mbr.mbr.mbr_parts[i];
1770 md->mbrp_size = 0;
1771 md->mbrp_start = 0;
1772 md->mbrp_type = MBR_PTYPE_UNUSED;
1773 if (marg->parts->mbr.last_mounted[i]) {
1774 free(__UNCONST(marg->parts->mbr.last_mounted[i]));
1775 marg->parts->mbr.last_mounted[i] = NULL;
1776 }
1777 if (is_ext_part) {
1778 for (mbr_info_t *m = marg->parts->mbr.extended;
1779 m != NULL; ) {
1780 mbr_info_t *n = m->extended;
1781 free_mbr_info(m);
1782 m = n;
1783 }
1784 marg->parts->mbr.extended = NULL;
1785 }
1786 } else {
1787 /* find the size of the primary extended partition */
1788 uint ext_start = 0, ext_size = 0;
1789 for (i = 0; i < MBR_PART_COUNT; i++) {
1790 if (!MBR_IS_EXTENDED(marg->parts->mbr.mbr.mbr_parts[i]
1791 .mbrp_type))
1792 continue;
1793 ext_start = marg->parts->mbr.mbr.mbr_parts[i]
1794 .mbrp_start;
1795 ext_size = marg->parts->mbr.mbr.mbr_parts[i]
1796 .mbrp_size;
1797 break;
1798 }
1799
1800 /*
1801 * If we are in an extended partition chain, unlink this MBR,
1802 * unless it is the very first one at the start of the extended
1803 * partition (we would have no previos ext mbr to fix up
1804 * the chain in that case)
1805 */
1806 if (marg->parts->mbr.extended == mb) {
1807 struct mbr_partition *part =
1808 &marg->parts->mbr.extended->mbr.mbr_parts[0];
1809 part->mbrp_type = MBR_PTYPE_UNUSED;
1810 part->mbrp_start = 0;
1811 part->mbrp_size = 0;
1812 } else {
1813 mbr_info_t *p, *last;
1814 for (last = NULL, p = &marg->parts->mbr; p != NULL;
1815 last = p, p = p->extended)
1816 if (p == mb)
1817 break;
1818 if (last == NULL) {
1819 if (marg->err_msg != NULL)
1820 *marg->err_msg= INTERNAL_ERROR;
1821 return false;
1822 }
1823 last->extended = p->extended;
1824 free_mbr_info(p);
1825 if (last == &marg->parts->mbr && last->extended &&
1826 last->extended->extended == NULL &&
1827 last->extended->mbr.mbr_parts[0].mbrp_type ==
1828 MBR_PTYPE_UNUSED) {
1829 /*
1830 * we deleted the last extended sector,
1831 * remove the whole chain
1832 */
1833 free_mbr_info(last->extended);
1834 last->extended = NULL;
1835 }
1836 }
1837 mbr_fixup_ext_chain(&marg->parts->mbr, ext_start,
1838 ext_start+ext_size);
1839 }
1840 mbr_calc_free_space(marg->parts);
1841 return true;
1842}

1924static part_id
1925mbr_add_part(struct disk_partitions *arg,
1926 const struct disk_part_info *info, const char **errmsg)
1927{
1928 struct mbr_disk_partitions *parts =
1929 (struct mbr_disk_partitions*)arg;
1930 part_id i, j, no, free_primary = UINT_MAX;
1931 mbr_info_t *m = &parts->mbr, *me, *last, *t;
1932 daddr_t ext_start = 0, ext_size = 0;
1933 uint start, size;
1934 struct disk_part_info data = *info;
1935 struct mbr_partition *newp;
1936
1937 if (errmsg != NULL)
1938 *errmsg = NULL;
1939
1940 assert(info->nat_type != NULL);
1941 if (info->nat_type == NULL) {
1942 if (errmsg != NULL)
1943 *errmsg = INTERNAL_ERROR;
1944 return NO_PART;
1945 }
1946 if (mbr_type_from_gen_desc(info->nat_type) == MBR_PTYPE_UNUSED) {
1947 if (errmsg != NULL)
1948 *errmsg = INTERNAL_ERROR;
1949 return NO_PART;
1950 }
1951
1952 /* do we have free primary slots and/or an extended partition? */
1953 for (i = 0; i < MBR_PART_COUNT; i++) {
1954 if (m->mbr.mbr_parts[i].mbrp_type == MBR_PTYPE_UNUSED
1955 && free_primary > MBR_PART_COUNT)
1956 free_primary = i;
1957 if (MBR_IS_EXTENDED(m->mbr.mbr_parts[i].mbrp_type)) {
1958 ext_start = m->mbr.mbr_parts[i].mbrp_start+m->sector;
1959 ext_size = m->mbr.mbr_parts[i].mbrp_size;
1960 continue;
1961 }
1962 if (m->mbr.mbr_parts[i].mbrp_type == MBR_PTYPE_UNUSED
1963 && m->mbr.mbr_parts[i].mbrp_size == 0)
1964 continue;
1965 }
1966 if (ext_start > 0 && ext_size > 0 &&
1967 MBR_IS_EXTENDED(mbr_type_from_gen_desc(info->nat_type))) {
1968 /*
1969 * Do not allow a second extended partition
1970 */
1971 if (errmsg)
1972 *errmsg = MSG_Only_one_extended_ptn;
1973 return NO_PART;
1974 }
1975
1976 /* should this go into the extended partition? */
1977 if (ext_size > 0 && info->start >= ext_start
1978 && info->start < ext_start + ext_size) {
1979
1980 /* must fit into the extended partition */
1981 if (info->start + info->size > ext_start + ext_size) {
1982 if (errmsg != NULL)
1983 *errmsg = MSG_mbr_ext_nofit;
1984 return NO_PART;
1985 }
1986
1987 /* walk the chain untill we find a proper insert position */
1988 daddr_t e_end, e_start;
1989 for (last = m, m = m->extended; m != NULL;
1990 last = m, m = m->extended) {
1991 e_start = m->mbr.mbr_parts[1].mbrp_start
1992 + ext_start;
1993 e_end = e_start + m->mbr.mbr_parts[1].mbrp_size;
1994 if (data.start <= e_start)
1995 break;
1996 }
1997 if (m == NULL) {
1998 /* add new tail record */
1999 e_end = ext_start + ext_size;
2000 /* new part needs to fit inside primary extended one */
2001 if (data.start + data.size > e_end) {
2002 if (errmsg)
2003 *errmsg = MSG_No_free_space;
2004 return NO_PART;
2005 }
2006 } else if (data.start + data.size > e_start) {
2007 /* new part needs to fit before next extended */
2008 if (errmsg)
2009 *errmsg = MSG_No_free_space;
2010 return NO_PART;
2011 }
2012 /*
2013 * now last points to previous mbr (maybe primary), m
2014 * points to the one that should take the new partition
2015 * or we have to insert a new mbr between the two, or
2016 * m needs to be split and we go into the one after it.
2017 */
2018 if (m && m->mbr.mbr_parts[0].mbrp_type == MBR_PTYPE_UNUSED) {
2019 /* empty slot, we can just use it */
2020 newp = &m->mbr.mbr_parts[0];
2021 mbr_info_to_partitition(&data, &m->mbr.mbr_parts[0],
2022 m->sector, errmsg);
2023 if (data.last_mounted && m->last_mounted[0] &&
2024 data.last_mounted != m->last_mounted[0]) {
2025 free(__UNCONST(m->last_mounted[0]));
2026 m->last_mounted[0] = strdup(data.last_mounted);
2027 }
2028 } else {
2029 mbr_info_t *new_mbr;
2030 if (m == NULL)
2031 m = last;
2032 daddr_t p_start = m->mbr.mbr_parts[0].mbrp_start
2033 + m->sector;
2034 daddr_t p_end = p_start
2035 + m->mbr.mbr_parts[0].mbrp_size;
2036 bool before;
2037 if (m == last || data.start > p_end)
2038 before = false;
2039 else if (data.start + data.size < p_start)
2040 before = true;
2041 else {
2042 if (errmsg)
2043 *errmsg = MSG_No_free_space;
2044 return NO_PART;
2045 }
2046 new_mbr = calloc(1, sizeof *new_mbr);
2047 if (!new_mbr) {
2048 if (errmsg)
2049 *errmsg = err_outofmem;
2050 return NO_PART;
2051 }
2052 new_mbr->mbr.mbr_magic = htole16(MBR_MAGIC);
2053 new_mbr->mbr.mbr_parts[1].mbrp_type = MBR_PTYPE_EXT;
2054 if (before) {
2055 /*
2056 * This is a hypthetical case where
2057 * an extended MBR uses an unusual high
2058 * offset (m->sector to parts[0].mbrp_start)
2059 * and we want to go into that space.
2060 * Should not happen in the real world (tm)
2061 * and is untested....
2062 */
2063
2064 /* make sure the aligned new mbr fits */
2065 uint mbrsec = rounddown(p_start,
2066 parts->ext_ptn_alignment);
2067 if (mbrsec <= data.start + data.size)
2068 data.size = mbrsec-1-data.start;
2069
2070 /* now the new partition data is ready,
2071 * write out to old position */
2072 new_mbr->sector = m->sector;
2073 newp = &new_mbr->mbr.mbr_parts[0];
2074 mbr_info_to_partitition(&data,
2075 &new_mbr->mbr.mbr_parts[0],
2076 new_mbr->sector, errmsg);
2077 if (data.last_mounted && m->last_mounted[0] &&
2078 data.last_mounted != m->last_mounted[0]) {
2079 free(__UNCONST(m->last_mounted[0]));
2080 m->last_mounted[0] =
2081 strdup(data.last_mounted);
2082 }
2083 new_mbr->extended = m;
2084 } else {
2085 new_mbr->sector = max(roundup(data.start,
2086 parts->ext_ptn_alignment),
2087 parts->ext_ptn_alignment);
2088 uint off = new_mbr->sector - data.start;
2089 data.start += parts->ptn_0_offset+off;
2090 if (data.start + data.size > e_end)
2091 data.size = e_end - data.start;
2092 newp = &new_mbr->mbr.mbr_parts[0];
2093 mbr_info_to_partitition(&data,
2094 &new_mbr->mbr.mbr_parts[0],
2095 new_mbr->sector, errmsg);
2096 if (data.last_mounted && m->last_mounted[0] &&
2097 data.last_mounted != m->last_mounted[0]) {
2098 free(__UNCONST(m->last_mounted[0]));
2099 m->last_mounted[0] =
2100 strdup(data.last_mounted);
2101 }
2102 /*
2103 * Special case: if we are creating the
2104 * first extended mbr, but do not start
2105 * at the beginning of the primary
2106 * extended partition, we need to insert
2107 * another extended mbr at the start.
2108 */
2109 if (m == &parts->mbr && m->extended == NULL
2110 && new_mbr->sector > ext_start) {
2111 t = calloc(1, sizeof *new_mbr);
2112 if (!t) {
2113 free_mbr_info(new_mbr);
2114 if (errmsg)
2115 *errmsg = err_outofmem;
2116 return NO_PART;
2117 }
2118 t->sector = ext_start;
2119 t->mbr.mbr_magic = htole16(MBR_MAGIC);
2120 t->mbr.mbr_parts[1].mbrp_type =
2121 MBR_PTYPE_EXT;
2122 m->extended = t;
2123 m = t;
2124 }
2125 new_mbr->extended = m->extended;
2126 m->extended = new_mbr;
2127 }
2128 }
2129 mbr_fixup_ext_chain(&parts->mbr, ext_start, ext_start+ext_size);
2130 dump_mbr(&parts->mbr, "after adding in extended");
2131 goto find_rval;
2132 }
2133
2134 /* this one is for the primary boot block */
2135 if (free_primary > MBR_PART_COUNT) {
2136 if (errmsg != NULL)
2137 *errmsg = ext_size > 0 ?
2138 MSG_mbr_no_free_primary_have_ext
2139 : MSG_mbr_no_free_primary_no_ext;
2140 return NO_PART;
2141 }
2142
2143 start = max(info->start, parts->ptn_0_offset);
2144 size = info->size;
2145 if (find_mbr_space(m, &start, &size, start, parts->dp.disk_size,
2146 start, true) < 0 || size < info->size) {
2147 if (errmsg != NULL)
2148 *errmsg = MSG_No_free_space;
2149 return NO_PART;
2150 }
2151 data.start = start;
2152 if (MBR_IS_EXTENDED(mbr_type_from_gen_desc(info->nat_type))) {
2153 data.start = max(roundup(data.start, parts->ext_ptn_alignment),
2154 parts->ext_ptn_alignment);
2155 }
2156 if (data.start + data.size > start + size)
2157 data.size = start + size - data.start;
2158 mbr_info_to_partitition(&data, &m->mbr.mbr_parts[free_primary],
2159 m->sector, errmsg);
2160 if (data.last_mounted && m->last_mounted[free_primary] &&
2161 data.last_mounted != m->last_mounted[free_primary]) {
2162 free(__UNCONST(m->last_mounted[free_primary]));
2163 m->last_mounted[free_primary] = strdup(data.last_mounted);
2164 }
2165 start = m->mbr.mbr_parts[free_primary].mbrp_start;
2166 mbr_sort_main_mbr(&m->mbr);
2167
2168 /* find the partition again after sorting */
2169 newp = NULL;
2170 for (i = 0; i < MBR_PART_COUNT; i++) {
2171 if (m->mbr.mbr_parts[i].mbrp_type == MBR_PTYPE_UNUSED)
2172 continue;
2173 if (m->mbr.mbr_parts[i].mbrp_start != start)
2174 continue;
2175 newp = &m->mbr.mbr_parts[i];
2176 break;
2177 }
2178
2179 dump_mbr(&parts->mbr, "after adding in primary");
2180
2181find_rval:
2182 mbr_calc_free_space(parts);
2183 if (newp == NULL)
2184 return 0;
2185
2186 /*
2187 * Now newp points to the modified partition entry but we do not know
2188 * a good part_id for it.
2189 * Iterate from start and find it.
2190 */
2191 no = 0;
2192 for (i = 0; i < MBR_PART_COUNT; i++) {
2193 if (m->mbr.mbr_parts[i].mbrp_type == MBR_PTYPE_UNUSED)
2194 continue;
2195
2196 if (newp == &m->mbr.mbr_parts[i])
2197 return no;
2198 no++;
2199
2200 if (MBR_IS_EXTENDED(m->mbr.mbr_parts[i].mbrp_type)) {
2201 for (me = m->extended; me != NULL; me = me->extended) {
2202 for (j = 0; j < MBR_PART_COUNT; j++) {
2203 if (me->mbr.mbr_parts[j].mbrp_type ==
2204 MBR_PTYPE_UNUSED)
2205 continue;
2206 if (j > 0 && MBR_IS_EXTENDED(
2207 me->mbr.mbr_parts[j].mbrp_type))
2208 break;
2209 if (newp == &me->mbr.mbr_parts[j])
2210 return no;
2211 no++;
2212 }
2213 }
2214 }
2215 }
2216 return 0;
2217}

2219static int
2220mbr_guess_geom(struct disk_partitions *arg, int *cyl, int *head, int *sec)
2221{
2222 struct mbr_disk_partitions *myparts = (struct mbr_disk_partitions*)arg;
2223 struct mbr_sector *mbrs = &myparts->mbr.mbr;
2224 struct mbr_partition *parts = &mbrs->mbr_parts[0];
2225 int xcylinders, xheads, i, j;
2226 daddr_t xsectors, xsize;
2227 int c1, h1, s1, c2, h2, s2;
2228 daddr_t a1, a2;
2229 uint64_t num, denom;
2230
2231 xheads = -1;
2232
2233 /* Try to deduce the number of heads from two different mappings. */
2234 for (i = 0; i < MBR_PART_COUNT * 2 - 1; i++) {
2235 if (get_mapping(parts, i, &c1, &h1, &s1, &a1) < 0)
2236 continue;
2237 a1 -= s1;
2238 for (j = i + 1; j < MBR_PART_COUNT * 2; j++) {
2239 if (get_mapping(parts, j, &c2, &h2, &s2, &a2) < 0)
2240 continue;
2241 a2 -= s2;
2242 num = (uint64_t)h1 * a2 - (quad_t)h2 * a1;
2243 denom = (uint64_t)c2 * a1 - (quad_t)c1 * a2;
2244 if (num != 0 && denom != 0 && num % denom == 0) {
2245 xheads = (int)(num / denom);
2246 xsectors = a1 / (c1 * xheads + h1);
2247 break;
2248 }
2249 }
2250 if (xheads != -1)
2251 break;
2252 }
2253
2254 if (xheads == -1)
2255 return -1;
2256
2257 /*
2258 * Estimate the number of cylinders.
2259 * XXX relies on get_disks having been called.
2260 */
2261 xsize = min(pm->dlsize, mbr_parts.size_limit);
2262 xcylinders = xsize / xheads / xsectors;
2263 if (xsize != xcylinders * xheads * xsectors)
2264 xcylinders++;
2265
2266 /*
2267 * Now verify consistency with each of the partition table entries.
2268 * Be willing to shove cylinders up a little bit to make things work,
2269 * but translation mismatches are fatal.
2270 */
2271 for (i = 0; i < MBR_PART_COUNT * 2; i++) {
2272 if (get_mapping(parts, i, &c1, &h1, &s1, &a1) < 0)
2273 continue;
2274 if (c1 >= MAXCYL - 1)
2275 /* Ignore anything that is near the CHS limit */
2276 continue;
2277 if (xsectors * (c1 * xheads + h1) + s1 != a1)
2278 return -1;
2279 }
2280
2281 /*
2282 * Everything checks out. Reset the geometry to use for further
2283 * calculations.
2284 */
2285 *cyl = MIN(xcylinders, MAXCYL);
2286 *head = xheads;
2287 *sec = xsectors;
2288 return 0;
2289}

2325 parts->dp.disk_size, ignore > 0 ? (uint)ignore : UINT_MAX,
