110/*
111 * URL encode unsafe characters. See RFC 1738.
112 *
113 * Copies src string to dst, encoding unsafe or reserved characters
114 * in %hex form as it goes, and returning a pointer to the result.
115 * The result is always a nul-terminated string even if it had to be
116 * truncated to avoid overflowing the available space.
117 *
118 * This url_encode() function does not operate on complete URLs, it
119 * operates on strings that make up parts of URLs. For example, in a
120 * URL like "ftp://username:password@host/path", the username, password,
121 * host and path should each be encoded separately before they are
122 * joined together with the punctuation characters.
123 *
124 * In most ordinary use, the path portion of a URL does not start with
125 * a slash; the slash is a separator between the host portion and the
126 * path portion, and is dealt with by software outside the url_encode()
127 * function. However, it is valid for url_encode() to be passed a
128 * string that does begin with a slash. For example, the string might
129 * represent a password, or a path part of a URL that the user really
130 * does want to begin with a slash.
131 *
132 * len is the length of the destination buffer. The result will be
133 * truncated if necessary to fit in the destination buffer.
134 *
135 * safe_chars is a string of characters that should not be encoded. If
136 * safe_chars is non-NULL, any characters in safe_chars as well as any
137 * alphanumeric characters will be copied from src to dst without
138 * encoding. Some potentially useful settings for this parameter are:
139 *
140 * NULL Everything is encoded (even alphanumerics)
141 * "" Everything except alphanumerics are encoded
142 * "/" Alphanumerics and '/' remain unencoded
143 * "$-_.+!*'()," Consistent with a strict reading of RFC 1738
144 * "$-_.+!*'(),/" As above, except '/' is not encoded
145 * "-_.+!,/" As above, except shell special characters are encoded
146 *
147 * encode_leading_slash is a flag that determines whether or not to
148 * encode a leading slash in a string. If this flag is set, and if the
149 * first character in the src string is '/', then the leading slash will
150 * be encoded (as "%2F"), even if '/' is one of the characters in the
151 * safe_chars string. Note that only the first character of the src
152 * string is affected by this flag, and that leading slashes are never
153 * deleted, but either retained unchanged or encoded.
154 *
155 * Unsafe and reserved characters are defined in RFC 1738 section 2.2.
156 * The most important parts are:
157 *
158 * The characters ";", "/", "?", ":", "@", "=" and "&" are the
159 * characters which may be reserved for special meaning within a
160 * scheme. No other characters may be reserved within a scheme.
161 * [...]
162 *
163 * Thus, only alphanumerics, the special characters "$-_.+!*'(),",
164 * and reserved characters used for their reserved purposes may be
165 * used unencoded within a URL.
166 *
167 */
168
169#define RFC1738_SAFE "$-_.+!*'(),"
170#define RFC1738_SAFE_LESS_SHELL "-_.+!,"
171#define RFC1738_SAFE_LESS_SHELL_PLUS_SLASH "-_.+!,/"
172
173static char *
174url_encode(char *dst, const char *src, const char *ep,
175 const char *safe_chars, int encode_leading_slash)
176{
177 int ch;
178
179 ep--;
180
181 for (; dst < ep; src++) {
182 ch = *src & 0xff;
183 if (ch == 0)
184 break;
185 if (safe_chars != NULL &&
186 (ch != '/' || !encode_leading_slash) &&
187 (isalnum(ch) || strchr(safe_chars, ch))) {
188 *dst++ = ch;
189 } else {
190 /* encode this char */
191 if (ep - dst < 3)
192 break;
193 snprintf(dst, ep - dst, "%%%02X", ch);
194 dst += 3;
195 }
196 encode_leading_slash = 0;
197 }
198 *dst = '\0';
199 return dst;
200}

222static int
223get_ifconfig_info(struct net_desc *devs)
224{
225 char *buf_in;
226 char *buf_tmp;
227 const char **ignore;
228 char *buf;
229 char *tmp;
230 int textsize;
231 int i;
232
233 /* Get ifconfig information */
234 textsize = collect(T_OUTPUT, &buf_in, "/sbin/ifconfig -l 2>/dev/null");
235 if (textsize < 0) {
236 if (logfp)
237 (void)fprintf(logfp,
238 "Aborting: Could not run ifconfig.\n");
239 (void)fprintf(stderr, "Could not run ifconfig.");
240 exit(1);
241 }
242
243 buf = malloc (STRSIZE * sizeof(char));
244 for (i = 0, buf_tmp = buf_in; i < MAX_NETS && strlen(buf_tmp) > 0
245 && buf_tmp < buf_in + strlen(buf_in);) {
246 tmp = stpncpy(buf, buf_tmp, strcspn(buf_tmp," \n"));
247 *tmp='\0';
248 buf_tmp += (strcspn(buf_tmp, " \n") + 1) * sizeof(char);
249
250 /* Skip ignored interfaces */
251 for (ignore = ignored_if_names; *ignore != NULL; ignore++) {
252 size_t len = strlen(*ignore);
253 if (strncmp(buf, *ignore, len) == 0 &&
254 isdigit((unsigned char)buf[len]))
255 break;
256 }
257 if (*ignore != NULL)
258 continue;
259
260 strlcpy (devs[i].if_dev, buf, STRSIZE);
261 i++;
262 }
263 strcpy(devs[i].if_dev, "\0");
264
265 free(buf);
266 free(buf_in);
267 return i;
268}

270static int
271do_ifreq(struct ifreq *ifr, unsigned long cmd)
272{
273 int sock;
274 int rval;
275
276 sock = socket(PF_INET, SOCK_DGRAM, 0);
277 if (sock == -1)
278 return -1;
279
280 memset(ifr, 0, sizeof *ifr);
281 strncpy(ifr->ifr_name, net_dev, sizeof ifr->ifr_name);
282 rval = ioctl(sock, cmd, ifr);
283 close(sock);
284
285 return rval;
286}

306/* Fill in defaults network values for the selected interface */
307static void
308get_ifinterface_info(void)
309{
310 struct ifreq ifr;
311 struct ifmediareq ifmr;
312 struct sockaddr_in *sa_in = (void*)&ifr.ifr_addr;
313 int modew;
314 const char *media_opt;
315 const char *sep;
316
317 if (do_ifreq(&ifr, SIOCGIFADDR) == 0 && sa_in->sin_addr.s_addr != 0)
318 strlcpy(net_ip, inet_ntoa(sa_in->sin_addr), sizeof net_ip);
319
320 if (do_ifreq(&ifr, SIOCGIFNETMASK) == 0 && sa_in->sin_addr.s_addr != 0)
321 strlcpy(net_mask, inet_ntoa(sa_in->sin_addr), sizeof net_mask);
322
323 if (do_ifmreq(&ifmr, SIOCGIFMEDIA) == 0) {
324 /* Get the name of the media word */
325 modew = ifmr.ifm_current;
326 strlcpy(net_media, get_media_subtype_string(modew),
327 sizeof net_media);
328 /* and add any media options */
329 sep = " mediaopt ";
330 while ((media_opt = get_media_option_string(&modew)) != NULL) {
331 strlcat(net_media, sep, sizeof net_media);
332 strlcat(net_media, media_opt, sizeof net_media);
333 sep = ",";
334 }
335 }
336}

341static void
342get_if6interface_info(void)
343{
344 char *textbuf, *t;
345 int textsize;
346
347 textsize = collect(T_OUTPUT, &textbuf,
348 "/sbin/ifconfig %s inet6 2>/dev/null", net_dev);
349 if (textsize >= 0) {
350 char *p;
351
352 (void)strtok(textbuf, "\n"); /* ignore first line */
353 while ((t = strtok(NULL, "\n")) != NULL) {
354 if (strncmp(t, "\tinet6 ", 7) != 0)
355 continue;
356 t += 7;
357 if (strstr(t, "tentative") || strstr(t, "duplicated"))
358 continue;
359 if (strncmp(t, "fe80:", 5) == 0)
360 continue;
361
362 p = t;
363 while (*p && *p != ' ' && *p != '\n')
364 p++;
365 *p = '\0';
366 strlcpy(net_ip6, t, sizeof(net_ip6));
367 break;
368 }
369 }
370 free(textbuf);
371}

374static void
375get_host_info(void)
376{
377 char hostname[MAXHOSTNAMELEN + 1];
378 char *dot;
379
380 /* Check host (and domain?) name */
381 if (gethostname(hostname, sizeof(hostname)) == 0 && hostname[0] != 0) {
382 hostname[sizeof(hostname) - 1] = 0;
383 /* check for a . */
384 dot = strchr(hostname, '.');
385 if (dot == NULL) {
386 /* if not found its just a host, punt on domain */
387 strlcpy(net_host, hostname, sizeof net_host);
388 } else {
389 /* split hostname into host/domain parts */
390 *dot++ = 0;
391 strlcpy(net_host, hostname, sizeof net_host);
392 strlcpy(net_domain, dot, sizeof net_domain);
393 }
394 }
395}

397/*
398 * recombine name parts split in get_host_info and config_network
399 * (common code moved here from write_etc_hosts)
400 */
401static char *
402recombine_host_domain(void)
403{
404 static char recombined[MAXHOSTNAMELEN + 1];
405 int l = strlen(net_host) - strlen(net_domain);
406
407 strlcpy(recombined, net_host, sizeof(recombined));
408
409 if (strlen(net_domain) != 0 && (l <= 0 ||
410 net_host[l - 1] != '.' ||
411 strcasecmp(net_domain, net_host + l) != 0)) {
412 /* net_host isn't an FQDN. */
413 strlcat(recombined, ".", sizeof(recombined));
414 strlcat(recombined, net_domain, sizeof(recombined));
415 }
416 return recombined;
417}

433static int
434handle_license(const char *dev)
435{
436 static struct {
437 const char *dev;
438 const char *lic;
439 } licdev[] = {
440 { "iwi", "/libdata/firmware/if_iwi/LICENSE.ipw2200-fw" },
441 { "ipw", "/libdata/firmware/if_ipw/LICENSE" },
442 };
443
444 size_t i;
445
446 for (i = 0; i < __arraycount(licdev); i++)
447 if (strncmp(dev, licdev[i].dev, 3) == 0) {
448 char buf[64];
449 int val;
450 size_t len = sizeof(int);
451 (void)snprintf(buf, sizeof(buf), "hw.%s.accept_eula",
452 licdev[i].dev);
453 if (sysctlbyname(buf, &val, &len, NULL, 0) != -1
454 && val != 0)
455 return 1;
456 msg_fmt_display(MSG_license, "%s%s",
457 dev, licdev[i].lic);
458 if (ask_yesno(NULL)) {
459 val = 1;
460 if (sysctlbyname(buf, NULL, NULL, &val,
461 0) == -1)
462 return 0;
463 add_sysctl_conf("%s=1", buf);
464 return 1;
465 } else
466 return 0;
467 }
468 return 1;
469}

471/*
472 * Get the information to configure the network, configure it and
473 * make sure both the gateway and the name server are up.
474 */
475int
476config_network(void)
477{
478 char *textbuf;
479 int octet0;
480 int dhcp_config;
481 int nfs_root = 0;
482 int slip = 0;
483 int pid, status;
484 char **ap, *slcmd[10], *in_buf;
485 char buffer[STRSIZE];
486 struct statvfs sb;
487 struct net_desc net_devs[MAX_NETS];
488 menu_ent *net_menu;
489 int menu_no;
490 int num_devs;
491 int selected_net;
492 int i;
493#ifdef INET6
494 int v6config = 1, rv;
495#endif
496
497 FILE *f;
498 time_t now;
499
500 if (network_up)
501 return (1);
502
503 num_devs = get_ifconfig_info(net_devs);
504
505 if (num_devs < 1) {
506 /* No network interfaces found! */
507 hit_enter_to_continue(NULL, MSG_nonet);
508 return -1;
509 }
510
511 net_menu = calloc(num_devs, sizeof(*net_menu));
512 if (net_menu == NULL) {
513 err_msg_win(err_outofmem);
514 return -1;
515 }
516
517 for (i = 0; i < num_devs; i++) {
518 net_menu[i].opt_name = net_devs[i].if_dev;
519 net_menu[i].opt_flags = OPT_EXIT;
520 net_menu[i].opt_action = set_menu_select;
521 }
522
523 menu_no = new_menu(MSG_netdevs,
524 net_menu, num_devs, -1, 4, 0, 0,
525 MC_SCROLL,
526 NULL, NULL, NULL, NULL, NULL);
527again:
528 selected_net = -1;
529 msg_display(MSG_asknetdev);
530 process_menu(menu_no, &selected_net);
531
532 if (selected_net == -1) {
533 free_menu(menu_no);
534 free(net_menu);
535 return 0;
536 }
537
538 network_up = 1;
539 dhcp_config = 0;
540
541 strncpy(net_dev, net_devs[selected_net].if_dev, STRSIZE);
542
543 if (!handle_license(net_dev))
544 goto done;
545
546 slip = net_dev[0] == 's' && net_dev[1] == 'l' &&
547 isdigit((unsigned char)net_dev[2]);
548
549 /* If root is on NFS do not reconfigure the interface. */
550 if (statvfs("/", &sb) == 0 && strcmp(sb.f_fstypename, "nfs") == 0) {
551 nfs_root = 1;
552 get_ifinterface_info();
553 get_if6interface_info();
554 get_host_info();
555 } else if (!slip) {
556 /* Preload any defaults we can find */
557 get_ifinterface_info();
558 get_if6interface_info();
559 get_host_info();
560
561 /* domain and host */
562 msg_display(MSG_netinfo);
563
564 /* ethernet medium */
565 for (;;) {
566 msg_prompt_add(MSG_net_media, net_media, net_media,
567 sizeof net_media);
568
569 /*
570 * ifconfig does not allow media specifiers on
571 * IFM_MANUAL interfaces. Our UI gives no way
572 * to set an option back
573 * to null-string if it gets accidentally set.
574 * Check for plausible alternatives.
575 */
576 if (strcmp(net_media, "<default>") == 0 ||
577 strcmp(net_media, "default") == 0 ||
578 strcmp(net_media, "<manual>") == 0 ||
579 strcmp(net_media, "manual") == 0 ||
580 strcmp(net_media, "<none>") == 0 ||
581 strcmp(net_media, "none") == 0 ||
582 strcmp(net_media, " ") == 0) {
583 *net_media = '\0';
584 }
585
586 if (*net_media == '\0')
587 break;
588 /*
589 * We must set the media type here - to give dhcp
590 * a chance
591 */
592 if (run_program(0, "/sbin/ifconfig %s media %s",
593 net_dev, net_media) == 0)
594 break;
595 /* Failed to set - output the supported values */
596 if (collect(T_OUTPUT, &textbuf, "/sbin/ifconfig -m %s |"
597 "while IFS=; read line;"
598 " do [ \"$line\" = \"${line#*media}\" ] || "
599 "echo $line;"
600 " done", net_dev ) > 0)
601 msg_display(textbuf);
602 free(textbuf);
603 }
604
605 net_dhcpconf = 0;
606 /* try a dhcp configuration */
607 dhcp_config = config_dhcp(net_dev);
608 if (dhcp_config) {
609 char *nline;
610
611 /* Get newly configured data off interface. */
612 get_ifinterface_info();
613 get_if6interface_info();
614 get_host_info();
615
616 net_dhcpconf |= DHCPCONF_IPADDR;
617
618 /*
619 * Extract default route from output of
620 * 'route -n show'
621 */
622 if (collect(T_OUTPUT, &textbuf,
623 "/sbin/route -n show | "
624 "while read dest gateway flags;"
625 " do [ \"$dest\" = default ] && {"
626 " echo \"$gateway\"; break; };"
627 " done" ) > 0)
628 strlcpy(net_defroute, textbuf,
629 sizeof net_defroute);
630 free(textbuf);
631 if ((nline = strchr(net_defroute, '\n')))
632 *nline = '\0';
633
634 /* pull nameserver info out of /etc/resolv.conf */
635 if (collect(T_OUTPUT, &textbuf,
636 "cat /etc/resolv.conf 2>/dev/null |"
637 " while read keyword address rest;"
638 " do [ \"$keyword\" = nameserver ] &&"
639 " { echo \"$address\"; break; };"
640 " done" ) > 0)
641 strlcpy(net_namesvr, textbuf,
642 sizeof net_namesvr);
643 free(textbuf);
644 if ((nline = strchr(net_namesvr, '\n')))
645 *nline = '\0';
646 if (net_namesvr[0] != '\0')
647 net_dhcpconf |= DHCPCONF_NAMESVR;
648
649 /* pull domain info out of /etc/resolv.conf */
650 if (collect(T_OUTPUT, &textbuf,
651 "cat /etc/resolv.conf 2>/dev/null |"
652 " while read keyword domain rest;"
653 " do [ \"$keyword\" = domain ] &&"
654 " { echo \"$domain\"; break; };"
655 " done" ) > 0)
656 strlcpy(net_domain, textbuf,
657 sizeof net_domain);
658 free(textbuf);
659 if (net_domain[0] == '\0') {
660 /* pull domain info out of /etc/resolv.conf */
661 if (collect(T_OUTPUT, &textbuf,
662 "cat /etc/resolv.conf 2>/dev/null |"
663 " while read keyword search rest;"
664 " do [ \"$keyword\" = search ] &&"
665 " { echo \"$search\"; break; };"
666 " done" ) > 0)
667 strlcpy(net_domain, textbuf,
668 sizeof net_domain);
669 free(textbuf);
670 }
671 if ((nline = strchr(net_domain, '\n')))
672 *nline = '\0';
673 if (net_domain[0] != '\0')
674 net_dhcpconf |= DHCPCONF_DOMAIN;
675
676 if (gethostname(net_host, sizeof(net_host)) == 0 &&
677 net_host[0] != 0)
678 net_dhcpconf |= DHCPCONF_HOST;
679 }
680 }
681
682 if (!(net_dhcpconf & DHCPCONF_HOST))
683 msg_prompt_add(MSG_net_host, net_host, net_host,
684 sizeof net_host);
685
686 if (!(net_dhcpconf & DHCPCONF_DOMAIN))
687 msg_prompt_add(MSG_net_domain, net_domain, net_domain,
688 sizeof net_domain);
689
690 if (!dhcp_config) {
691 /* Manually configure IPv4 */
692 if (!nfs_root)
693 msg_prompt_add(MSG_net_ip, net_ip, net_ip,
694 sizeof net_ip);
695 if (slip)
696 msg_prompt_add(MSG_net_srv_ip, net_srv_ip, net_srv_ip,
697 sizeof net_srv_ip);
698 else if (!nfs_root) {
699 /* We don't want netmasks for SLIP */
700 octet0 = atoi(net_ip);
701 if (!net_mask[0]) {
702 if (0 <= octet0 && octet0 <= 127)
703 strlcpy(net_mask, "0xff000000",
704 sizeof(net_mask));
705 else if (128 <= octet0 && octet0 <= 191)
706 strlcpy(net_mask, "0xffff0000",
707 sizeof(net_mask));
708 else if (192 <= octet0 && octet0 <= 223)
709 strlcpy(net_mask, "0xffffff00",
710 sizeof(net_mask));
711 }
712 msg_prompt_add(MSG_net_mask, net_mask, net_mask,
713 sizeof net_mask);
714 }
715 msg_prompt_add(MSG_net_defroute, net_defroute, net_defroute,
716 sizeof net_defroute);
717 }
718
719 if (!(net_dhcpconf & DHCPCONF_NAMESVR)) {
720#ifdef INET6
721 if (v6config) {
722 rv = 0;
723 process_menu(MENU_namesrv6, &rv);
724 if (!rv)
725 msg_prompt_add(MSG_net_namesrv, net_namesvr,
726 net_namesvr, sizeof net_namesvr);
727 } else
728#endif
729 msg_prompt_add(MSG_net_namesrv, net_namesvr, net_namesvr,
730 sizeof net_namesvr);
731 }
732
733 /* confirm the setting */
734 msg_clear();
735 if (slip)
736 msg_fmt_table_add(MSG_netok_slip, "%s%s%s%s%s%s%s%s%s",
737 net_domain,
738 net_host,
739 *net_namesvr == '\0' ? "<none>" : net_namesvr,
740 net_dev,
741 *net_media == '\0' ? "<default>" : net_media,
742 *net_ip == '\0' ? "<none>" : net_ip,
743 *net_srv_ip == '\0' ? "<none>" : net_srv_ip,
744 *net_mask == '\0' ? "<none>" : net_mask,
745 *net_defroute == '\0' ? "<none>" : net_defroute);
746 else
747 msg_fmt_table_add(MSG_netok, "%s%s%s%s%s%s%s%s",
748 net_domain,
749 net_host,
750 *net_namesvr == '\0' ? "<none>" : net_namesvr,
751 net_dev,
752 *net_media == '\0' ? "<default>" : net_media,
753 *net_ip == '\0' ? "<none>" : net_ip,
754 *net_mask == '\0' ? "<none>" : net_mask,
755 *net_defroute == '\0' ? "<none>" : net_defroute);
756#ifdef INET6
757 msg_fmt_table_add(MSG_netokv6, "%s",
758 !is_v6kernel() ? "<not supported>" : net_ip6);
759#endif
760done:
761 if (!ask_yesno(MSG_netok_ok))
762 goto again;
763
764 free_menu(menu_no);
765 free(net_menu);
766
767 run_program(0, "/sbin/ifconfig lo0 127.0.0.1");
768
769 /* dhcpcd will have configured it all for us */
770 if (dhcp_config) {
771 fflush(NULL);
772 network_up = 1;
773 return network_up;
774 }
775
776 /*
777 * we may want to perform checks against inconsistent configuration,
778 * like IPv4 DNS server without IPv4 configuration.
779 */
780
781 /* Create /etc/resolv.conf if a nameserver was given */
782 if (net_namesvr[0] != '\0') {
783 f = fopen("/etc/resolv.conf", "w");
784 if (f == NULL) {
785 if (logfp)
786 (void)fprintf(logfp,
787 "%s", msg_string(MSG_resolv));
788 (void)fprintf(stderr, "%s", msg_string(MSG_resolv));
789 exit(1);
790 }
791 scripting_fprintf(NULL, "cat <<EOF >/etc/resolv.conf\n");
792 time(&now);
793 scripting_fprintf(f, ";\n; BIND data file\n; %s %s;\n",
794 "Created by NetBSD sysinst on", safectime(&now));
795 if (net_domain[0] != '\0')
796 scripting_fprintf(f, "search %s\n", net_domain);
797 if (net_namesvr[0] != '\0')
798 scripting_fprintf(f, "nameserver %s\n", net_namesvr);
799 scripting_fprintf(NULL, "EOF\n");
800 fflush(NULL);
801 fclose(f);
802 }
803
804 if (net_ip[0] != '\0') {
805 if (slip) {
806 /* XXX: needs 'ifconfig sl0 create' much earlier */
807 /* Set SLIP interface UP */
808 run_program(0, "/sbin/ifconfig %s inet %s %s up",
809 net_dev, net_ip, net_srv_ip);
810 strcpy(sl_flags, "-s 115200 -l /dev/tty00");
811 msg_prompt_win(MSG_slattach, -1, 12, 70, 0,
812 sl_flags, sl_flags, sizeof sl_flags);
813
814 /* XXX: wtf isn't run_program() used here? */
815 pid = fork();
816 if (pid == 0) {
817 strcpy(buffer, "/sbin/slattach ");
818 strcat(buffer, sl_flags);
819 in_buf = buffer;
820
821 for (ap = slcmd; (*ap = strsep(&in_buf, " ")) != NULL;)
822 if (**ap != '\0')
823 ++ap;
824
825 execvp(slcmd[0], slcmd);
826 } else
827 wait4(pid, &status, WNOHANG, 0);
828 } else if (!nfs_root) {
829 if (net_mask[0] != '\0') {
830 run_program(0, "/sbin/ifconfig %s inet %s netmask %s",
831 net_dev, net_ip, net_mask);
832 } else {
833 run_program(0, "/sbin/ifconfig %s inet %s",
834 net_dev, net_ip);
835 }
836 }
837 }
838
839 /* Set host name */
840 if (net_host[0] != '\0')
841 sethostname(net_host, strlen(net_host));
842
843 /* Set a default route if one was given */
844 if (!nfs_root && net_defroute[0] != '\0') {
845 run_program(RUN_DISPLAY | RUN_PROGRESS,
846 "/sbin/route -n flush -inet");
847 run_program(RUN_DISPLAY | RUN_PROGRESS,
848 "/sbin/route -n add default %s", net_defroute);
849 }
850
851 /*
852 * wait for addresses to become valid
853 */
854 if (!nfs_root) {
855 msg_display_add(MSG_wait_network);
856 network_up = !run_program(RUN_DISPLAY | RUN_PROGRESS,
857 "/sbin/ifconfig -w 15 -W 5");
858 } else {
859 /* Assume network is up. */
860 network_up = 1;
861 }
862
863 fflush(NULL);
864
865 return network_up;
866}

879void
880make_url(char *urlbuffer, struct ftpinfo *f, const char *dir)
881{
882 char ftp_user_encoded[STRSIZE];
883 char ftp_dir_encoded[STRSIZE];
884 char *cp;
885 const char *dir2;
886
887 /*
888 * f->pass is quite likely to contain unsafe characters
889 * that need to be encoded in the URL (for example,
890 * "@", ":" and "/" need quoting). Let's be
891 * paranoid and also encode f->user and f->dir. (For
892 * example, f->dir could easily contain '~', which is
893 * unsafe by a strict reading of RFC 1738).
894 */
895 if (strcmp("ftp", f->user) == 0 && f->pass[0] == 0) {
896 ftp_user_encoded[0] = 0;
897 } else {
898 cp = url_encode(ftp_user_encoded, f->user,
899 ftp_user_encoded + sizeof ftp_user_encoded - 1,
900 RFC1738_SAFE_LESS_SHELL, 0);
901 *cp++ = ':';
902 cp = url_encode(cp, f->pass,
903 ftp_user_encoded + sizeof ftp_user_encoded - 1,
904 NULL, 0);
905 *cp++ = '@';
906 *cp = 0;
907 }
908 cp = url_encode(ftp_dir_encoded, f->dir,
909 ftp_dir_encoded + sizeof ftp_dir_encoded - 1,
910 RFC1738_SAFE_LESS_SHELL_PLUS_SLASH, 1);
911 if (cp != ftp_dir_encoded && cp[-1] != '/')
912 *cp++ = '/';
913
914 dir2 = dir;
915 while (*dir2 == '/')
916 ++dir2;
917
918 url_encode(cp, dir2,
919 ftp_dir_encoded + sizeof ftp_dir_encoded,
920 RFC1738_SAFE_LESS_SHELL_PLUS_SLASH, 0);
921
922 snprintf(urlbuffer, STRSIZE, "%s://%s%s/%s", url_proto(f->xfer),
923 ftp_user_encoded, f->xfer_host[f->xfer], ftp_dir_encoded);
924}

927/* ftp_fetch() and pkgsrc_fetch() are essentially the same, with a different
928 * ftpinfo var and pkgsrc always using .tgz suffix, while for
929 * regular sets we only use .tgz for source sets on some architectures. */
930static int do_ftp_fetch(const char *, bool, struct ftpinfo *);
931
932static int
933ftp_fetch(const char *set_name)
934{
935 return do_ftp_fetch(set_name, use_tgz_for_set(set_name), &ftp);
936}
937
938static int
939pkgsrc_fetch(const char *set_name)
940{
941 return do_ftp_fetch(set_name, true, &pkgsrc);
942}
943
944static int
945do_ftp_fetch(const char *set_name, bool force_tgz, struct ftpinfo *f)
946{
947 const char *ftp_opt;
948 char url[STRSIZE];
949 int rval;
950
951 /*
952 * Invoke ftp to fetch the file.
953 */
954 if (strcmp("ftp", f->user) == 0 && f->pass[0] == 0) {
955 /* do anon ftp */
956 ftp_opt = "-a ";
957 } else {
958 ftp_opt = "";
959 }
960
961 make_url(url, f, set_dir_for_set(set_name));
962 rval = run_program(RUN_DISPLAY | RUN_PROGRESS | RUN_XFER_DIR,
963 "/usr/bin/ftp %s%s/%s%s",
964 ftp_opt, url, set_name,
965 force_tgz ? dist_tgz_postfix : dist_postfix);
966
967 return rval ? SET_RETRY : SET_OK;
968}

1067/*
1068 * Write the network config info the user entered via menus into the
1069 * config files in the target disk. Be careful not to lose any
1070 * information we don't immediately add back, in case the install
1071 * target is the currently-active root.
1072 */
1073void
1074mnt_net_config(void)
1075{
1076 char ifconfig_fn[STRSIZE];
1077 FILE *ifconf = NULL;
1078
1079 if (!network_up)
1080 return;
1081 if (!ask_yesno(MSG_mntnetconfig))
1082 return;
1083
1084 /* Write hostname to /etc/rc.conf */
1085 if ((net_dhcpconf & DHCPCONF_HOST) == 0)
1086 if (del_rc_conf("hostname") == 0)
1087 add_rc_conf("hostname=%s\n", recombine_host_domain());
1088
1089 /* Copy resolv.conf to target. If DHCP was used to create it,
1090 * it will be replaced on next boot anyway. */
1091 if (net_namesvr[0] != '\0')
1092 dup_file_into_target("/etc/resolv.conf");
1093
1094 /*
1095 * bring the interface up, it will be necessary for IPv6, and
1096 * it won't make trouble with IPv4 case either
1097 */
1098 snprintf(ifconfig_fn, sizeof ifconfig_fn, "/etc/ifconfig.%s", net_dev);
1099 ifconf = target_fopen(ifconfig_fn, "w");
1100 if (ifconf != NULL) {
1101 scripting_fprintf(NULL, "cat <<EOF >>%s%s\n",
1102 target_prefix(), ifconfig_fn);
1103 scripting_fprintf(ifconf, "up\n");
1104 if (*net_media != '\0')
1105 scripting_fprintf(ifconf, "media %s\n", net_media);
1106 scripting_fprintf(NULL, "EOF\n");
1107 }
1108
1109 if ((net_dhcpconf & DHCPCONF_IPADDR) == 0) {
1110 FILE *hosts;
1111
1112 /* Write IPaddr and netmask to /etc/ifconfig.if[0-9] */
1113 if (ifconf != NULL) {
1114 scripting_fprintf(NULL, "cat <<EOF >>%s%s\n",
1115 target_prefix(), ifconfig_fn);
1116 if (*net_media != '\0')
1117 scripting_fprintf(ifconf,
1118 "%s netmask %s media %s\n",
1119 net_ip, net_mask, net_media);
1120 else
1121 scripting_fprintf(ifconf, "%s netmask %s\n",
1122 net_ip, net_mask);
1123 scripting_fprintf(NULL, "EOF\n");
1124 }
1125
1126 /*
1127 * Add IPaddr/hostname to /etc/hosts.
1128 * Be careful not to clobber any existing contents.
1129 * Relies on ordered search of /etc/hosts. XXX YP?
1130 */
1131 hosts = target_fopen("/etc/hosts", "a");
1132 if (hosts != 0) {
1133 scripting_fprintf(NULL, "cat <<EOF >>%s/etc/hosts\n",
1134 target_prefix());
1135 write_etc_hosts(hosts);
1136 (void)fclose(hosts);
1137 scripting_fprintf(NULL, "EOF\n");
1138 }
1139
1140 if (del_rc_conf("defaultroute") == 0)
1141 add_rc_conf("defaultroute=\"%s\"\n", net_defroute);
1142 } else {
1143 /*
1144 * Start dhcpcd quietly and in master mode, but restrict
1145 * it to our interface
1146 */
1147 add_rc_conf("dhcpcd=YES\n");
1148 add_rc_conf("dhcpcd_flags=\"-qM %s\"\n", net_dev);
1149 }
1150
1151 if (ifconf)
1152 fclose(ifconf);
1153
1154 fflush(NULL);
1155}

1157int
1158config_dhcp(char *inter)
1159{
1160 int dhcpautoconf;
1161
1162 /*
1163 * Don't bother checking for an existing instance of dhcpcd, just
1164 * ask it to renew the lease. It will fork and daemonize if there
1165 * wasn't already an instance.
1166 */
1167
1168 if (!file_mode_match(DHCPCD, S_IFREG))
1169 return 0;
1170 if (ask_yesno(MSG_Perform_autoconfiguration)) {
1171 /* spawn off dhcpcd and wait for parent to exit */
1172 dhcpautoconf = run_program(RUN_DISPLAY | RUN_PROGRESS,
1173 "%s -d -n %s", DHCPCD, inter);
1174 return dhcpautoconf ? 0 : 1;
1175 }
1176 return 0;
1177}
