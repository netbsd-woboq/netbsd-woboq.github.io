34/*
35 * A list of partitioning schemes, so we can iterate over everything
36 * supported (e.g. when partitioning a new disk). NULL terminated.
37 */
38const struct disk_partitioning_scheme **available_part_schemes;
39/*
40 * The number of valid entries on above list
41 */
42size_t num_available_part_schemes;
43
44/*
45 * Generic reader - query a disk device and read all partitions from it
46 */
47struct disk_partitions *
48partitions_read_disk(const char *dev, daddr_t disk_size)
49{
50 const struct disk_partitioning_scheme **ps;
51
52 if (!available_part_schemes)
53 return NULL;
54
55 for (ps = available_part_schemes; *ps; ps++) {
56 struct disk_partitions *parts =
57 (*ps)->read_from_disk(dev, 0, disk_size);
58 if (parts)
59 return parts;
60 }
61 return NULL;
62}

64/*************** global init ****************************************/
65/*
66 * Helper structure to fill our global list of available partitioning
67 * schemes.
68 */
69struct part_scheme_desc {
70 bool (*is_available)(void);
71 const struct disk_partitioning_scheme *ps;
72};
73
74#ifdef HAVE_GPT
75bool gpt_parts_check(void);
76extern const struct disk_partitioning_scheme gpt_parts;
77#endif
78#ifdef HAVE_MBR
79extern const struct disk_partitioning_scheme mbr_parts;
80#endif
81#if RAW_PART == 2 /* only available as primary on some architectures */
82const struct disk_partitioning_scheme disklabel_parts;
83#endif
84
85/*
86 * One time initialization
87 */
88void
89partitions_init(void)
90{
91 /*
92 * List of partitioning schemes.
93 * Order is important, the selection menu is created from start
94 * to end. Keep good defaults early. Most architectures will
95 * only offer very few entries.
96 */
97static const struct part_scheme_desc all_descs[] = {
98#if RAW_PART == 2 /* only available as primary on some architectures */
99 { NULL, &disklabel_parts },
100#endif
101#ifdef HAVE_GPT
102 { gpt_parts_check, &gpt_parts },
103#endif
104#ifdef HAVE_MBR
105 { NULL, &mbr_parts },
106#endif
107 };
108
109 size_t i, avail;
110 const struct disk_partitioning_scheme **out;
111 bool *is_available;
112 static const size_t all_cnt = __arraycount(all_descs);
113
114 check_available_binaries();
115
116 is_available = malloc(all_cnt);
117
118 for (avail = i = 0; i < all_cnt; i++) {
119 is_available[i] = all_descs[i].is_available == NULL
120 || all_descs[i].is_available();
121 if (is_available[i])
122 avail++;
123 }
124
125 if (avail == 0)
126 return;
127
128 num_available_part_schemes = avail;
129 available_part_schemes = malloc(sizeof(*available_part_schemes)
130 * (avail+1));
131 if (available_part_schemes == NULL)
132 return;
133
134 for (out = available_part_schemes, i = 0; i < all_cnt; i++) {
135 if (!is_available[i])
136 continue;
137 *out++ = all_descs[i].ps;
138 }
139 *out = NULL;
140
141 free(is_available);
142}
