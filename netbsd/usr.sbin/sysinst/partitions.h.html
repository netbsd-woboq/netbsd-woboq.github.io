58/*
59 * A partition index / handle, identifies a singlepartition within
60 * a struct disk_partitions. This is just an iterator/index - whenever
61 * changes to the set of partitions are done, partitions may get a new
62 * part_id.
63 * We assume that partitioning schemes keep partitions sorted (with
64 * key = start address, some schemes will have overlapping partitions,
65 * like MBR extended partitions).
66 */
67typedef size_t part_id;
68
69/*
70 * An invalid value for a partition index / handle
71 */
72#define NO_PART ((part_id)~0U)
73
74/*
75 * Intended usage for a partition
76 */
77enum part_type {
78 PT_undef, /* invalid value */
79 PT_unknown, /* anything we can not map to one of these */
80 PT_root, /* the NetBSD / partition (bootable) */
81 PT_swap, /* the NetBSD swap partition */
82 PT_FAT, /* boot partition (e.g. for u-boot) */
83 PT_EFI_SYSTEM, /* (U)EFI boot partition */
84};
85
86/*
87 * A generic structure describing partition types for menu/user interface
88 * purposes. The internal details may be richer and the *pointer* value
89 * is the unique token - that is: the partitioning scheme will hand out
90 * pointers to internal data and recognize the exact partition type details
91 * by pointer comparision.
92 */
93struct part_type_desc {
94 enum part_type generic_ptype; /* what this maps to in generic terms */
95 const char *short_desc; /* short type description */
96 const char *description; /* full description */
97};
98
99/* Bits for disk_part_info.flags: */
100#define PTI_SEC_CONTAINER 1 /* this covers our secondary
101 partitions */
102#define PTI_WHOLE_DISK 2 /* all of the NetBSD disk */
103#define PTI_BOOT 4 /* required for booting */
104#define PTI_PSCHEME_INTERNAL 8 /* no user partition, e.g.
105 MBRs extend partition */
106#define PTI_RAW_PART 16 /* total disk */
107
108/* A single partition */
109struct disk_part_info {
110 daddr_t start, size; /* start and size on disk */
111 uint32_t flags; /* active PTI_ flags */
112 const struct part_type_desc *nat_type; /* native partition type */
113 /*
114 * The following will only be available
115 * a) for a small subset of file system types
116 * b) if the partition (in this state) has already been
117 * used before
118 * It is OK to leave all these zeroed / NULL when setting
119 * partition data - or leave them at the last values a get operation
120 * returned. Backends can not rely on them to be valid.
121 */
122 const char *last_mounted; /* last mount point or NULL */
123 unsigned int fs_type, fs_sub_type; /* FS_* type of filesystem
124 * and for some FS a sub
125 * type (e.g. FFSv1 vs. FFSv2)
126 */
127};

129/* An unused area that may be used for new partitions */
130struct disk_part_free_space {
131 daddr_t start, size;
132};

134/*
135 * Some partition schemes define additional data that needs to be edited.
136 * These attributes are described in this structure and referenced by
137 * their index into the fixed list of available attributes.
138 */
139enum custom_attr_type { pet_bool, pet_cardinal, pet_str };
140struct disk_part_custom_attribute {
141 msg label; /* Name, like "active partition" */
142 enum custom_attr_type type; /* bool, long, char* */
143 size_t strlen; /* maximum length if pet_str */
144};

159/*
160 * When querying partition "device" names, we may ask for:
161 */
162enum dev_name_usage {
163 parent_device_only, /* wd0 instead of wd0i, no path */
164 logical_name, /* NAME=my-root instead of dk7 */
165 plain_name, /* e.g. /dev/wd0i or /dev/dk7 */
166 raw_dev_name, /* e.g. /dev/rwd0i or /dev/rdk7 */
167};

169/*
170 * A scheme how to store partitions on-disk, and methods to read/write
171 * them to/from our abstract internal presentation.
172 */
173struct disk_partitioning_scheme {
174 /* name of the on-disk scheme, retrieved via msg_string */
175 msg name, short_name;
176
177 /* prompt shown when creating custom partition types */
178 msg new_type_prompt;
179
180 /* description of scheme specific partition flags */
181 msg part_flag_desc;
182
183 /* size restrictions for this partitioning scheme */
184 daddr_t size_limit; /* 0 if not limited */
185
186 /*
187 * If this scheme allows sub-partitions (i.e. MBR -> disklabel),
188 * this is a pointer to the (potential/optional) secondary
189 * scheme. Depending on partitioning details it may not be
190 * used in the end.
191 * This link is only here for better help messages.
192 * See *secondary_partitions further below for actually accesing
193 * secondary partitions.
194 */
195 const struct disk_partitioning_scheme *secondary_scheme;
196
197 /*
198 * Partition editor colum descriptions for whatever the scheme
199 * needs to display (see format_partition_table_str below).
200 */
201 size_t edit_columns_count;
202 const struct disk_part_edit_column_desc *edit_columns;
203
204 /*
205 * Custom attributes editable by the partitioning scheme (but of
206 * no particular meaning for sysinst)
207 */
208 size_t custom_attribute_count;
209 const struct disk_part_custom_attribute *custom_attributes;
210
211 /*
212 * Partition types supported by this scheme,
213 * first function gets the number, second queries single elements
214 */
215 size_t (*get_part_types_count)(void);
216 const struct part_type_desc * (*get_part_type)(size_t ndx);
217 /*
218 * Get the prefered native representation for a generic partition type
219 */
220 const struct part_type_desc * (*get_generic_part_type)(enum part_type);
221 /*
222 * Get the prefered native partition type for a specific file system
223 * type (FS_*) and subtype (fs specific value)
224 */
225 const struct part_type_desc * (*get_fs_part_type)(unsigned, unsigned);
226 /*
227 * Create a custom partition type. If the type already exists
228 * (or there is a collision), the old existing type will be
229 * returned and no new type created. This is not considered
230 * an error (to keep the user interface simple).
231 * On failure NULL is returned and (if passed != NULL)
232 * *err_msg is set to a message describing the error.
233 */
234 const struct part_type_desc * (*create_custom_part_type)
235 (const char *custom, const char **err_msg);

296 /* Retrieve device and partition names, e.g. for checking
297 * against kern.root_device or invoking newfs.
298 * For disklabel partitions, "part" will be set to the partition
299 * index (a = 0, b = 1, ...), for others it will get set to -1.
300 * If dev_name_usage is parent_device_only, the device name will
301 * not include a partition letter - obviously this only makes a
302 * difference with disklabel partitions.
303 * If dev_name_usage is logical_name instead of a device name
304 * a given name may be returned in NAME= syntax.
305 * If with_path is true (and the returned value is a device
306 * node), include the /dev/ prefix in the result string
307 * (this is ignored when returning NAME= syntax for /etc/fstab).
308 */
309 bool (*get_part_device)(const struct disk_partitions*,
310 part_id, char *devname, size_t max_devname_len, int *part,
311 enum dev_name_usage, bool with_path);
312
313 /*
314 * How big could we resize the given position (start of existing
315 * partition or free space)
316 */
317 daddr_t (*max_free_space_at)(const struct disk_partitions*, daddr_t);
318
319 /*
320 * Provide a list of free spaces usable for further partitioning,
321 * assuming the given partition alignment.
322 * If start is > 0 no space with lower sector numbers will
323 * be found.
324 * If ignore is > 0, any partition starting at that sector will
325 * be considered "free", this is used e.g. when moving an existing
326 * partition around.
327 */
328 size_t (*get_free_spaces)(const struct disk_partitions*,
329 struct disk_part_free_space *result, size_t max_num_result,
330 daddr_t min_space_size, daddr_t align, daddr_t start,
331 daddr_t ignore /* -1 */);
332
333 /*
334 * Translate a partition description from a foreign partitioning
335 * scheme as close as possible to what we can handle in add_partition.
336 * This mostly adjusts flags and partition type pointers (using
337 * more lose matching than add_partition would do).

339 bool (*adapt_foreign_part_info)(const struct disk_partitions*,

352 /*

359 /* Delete all partitions */

361
362 /* Optional: delete any partitions inside the given range */

366 /* Delete the specified partition */
367 bool (*delete_partition)(struct disk_partitions*, part_id,

370 /*
371 * Methods for the whole set of partitions
372 */
373 /*
374 * If this scheme only creates a singly NetBSD partition, which
375 * then is sub-partitioned (usually by disklabel), this returns a
376 * pointer to the secondary partition set.
377 * Otherwise NULL is returned, e.g. when there is no
378 * NetBSD partition defined (so this might change over time).
379 * Schemes that NEVER use a secondary scheme set this
380 * function pointer to NULL.
381 *
382 * If force_empty = true, ignore all on-disk contents and just
383 * create a new disk_partitons structure for the secondary scheme
384 * (this is used after deleting all partitions and setting up
385 * things for "use whole disk").
386 *
387 * The returned pointer is always owned by the primary partitions,
388 * caller MUST never free it, but otherwise can manipulate it
389 * arbitrarily.
390 */
391 struct disk_partitions *
392 (*secondary_partitions)(struct disk_partitions *, daddr_t start,
393 bool force_empty);
394
395 /*
396 * Write the whole set (in new_state) back to disk.
397 */
398 bool (*write_to_disk)(struct disk_partitions *new_state);
399
400 /*
401 * Try to read partitions from a disk, return NULL if this is not
402 * the partitioning scheme in use on that device.
403 * Usually start and len are 0 (and ignored).
404 * If this is about a part of a disk (like only the NetBSD
405 * MBR partition, start and len are the valid part of the
406 * disk.
407 */
408 struct disk_partitions * (*read_from_disk)(const char *,
409 daddr_t start, daddr_t len);
410
411 /*
412 * Set up all internal data for a new disk
413 */
414 struct disk_partitions * (*create_new_for_disk)(const char *,
415 daddr_t start, daddr_t len, daddr_t disk_total_size,
416 bool is_boot_drive);
417
418 /*
419 * Optional: this scheme may be used to boot from the given disk
420 */
421 bool (*have_boot_support)(const char *disk);
422 /*
423 * Optional: try to guess disk geometry from the partition information
424 */
425 int (*guess_disk_geom)(struct disk_partitions *,
426 int *cyl, int *head, int *sec);
427 /*
428 * Optional: change used geometry info and update internal state
429 */
430 bool (*change_disk_geom)(struct disk_partitions *,
431 int cyl, int head, int sec);
432 /*
433 * Optional:
434 * Get or set a name for the whole disk (most partitioning
435 * schemes do not provide this). Used for disklabel "pack names",
436 * which then may be used for aut-discovery of wedges, so it
437 * makes sense for the user to edit them.
438 */
439 bool (*get_disk_pack_name)(const struct disk_partitions *,
440 char *, size_t);
441 bool (*set_disk_pack_name)(struct disk_partitions *, const char *);
442 /*
443 * Optional:
444 * Try to guess install target partition from internal data,
445 * returns true if a safe match was found and sets start/size
446 * to the target partition.
447 */
448 bool (*guess_install_target)(const struct disk_partitions *,
449 daddr_t *start, daddr_t *size);
450 /*
451 * Optional: verify that the whole set of partitions would be bootable,
452 * fix up any issues (with user interaction) where needed.
453 * If "quiet" is true, fix up everything silently if possible
454 * and never return 1.
455 * Returns:
456 * 0: abort install
457 * 1: re-edit partitions
458 * 2: use anyway (continue)
459 */
460 int (*post_edit_verify)(struct disk_partitions *, bool quiet);
461 /*
462 * Optional: called during updates, before mounting the target disk(s),
463 * before md_pre_update() is called. Can be used to fixup
464 * partition info for historic errors (e.g. i386 changing MBR
465 * partition type from 165 to 169), similar to post_edit_verify.
466 * Returns:
467 * true if the partition info has changed (write back required)
468 * false if nothing further needs to be done.
469 */
470 bool (*pre_update_verify)(struct disk_partitions *);
471
472 /* Free all the data */
473 void (*free)(struct disk_partitions*);
474};

476/*
477 * The in-memory representation of all partitions on a concrete disk,
478 * tied to the partitioning scheme in use.
479 *
480 * Concrete schemes will derive from the abstract disk_partitions
481 * structure (by aggregation), but consumers of the API will only
482 * ever see this public part.
483 */
484struct disk_partitions {
485 /* which partitioning scheme is in use */
486 const struct disk_partitioning_scheme *pscheme;
487
488 /* the disk device this came from (or should go to) */
489 const char *disk;
490
491 /* global/public disk data */
492
493 /*
494 * Valid partitions may have IDs in the range 0 .. num_part (excl.)
495 */
496 part_id num_part;
497
498 /*
499 * If this is a sub-partitioning, the start of the "disk" is
500 * some arbitrary partition in the parent. Sometimes we need
501 * to be able to calculate absoluted offsets.
502 */
503 daddr_t disk_start;
504 /*
505 * Total size of the disk (usable for partitioning)
506 */
507 daddr_t disk_size;
508
509 /*
510 * Space not yet allocated
511 */
512 daddr_t free_space;
513
514 /*
515 * If this is the secondary partitioning scheme, pointer to
516 * the outer one. Otherwise NULL.
517 */
518 struct disk_partitions *parent;
519};
