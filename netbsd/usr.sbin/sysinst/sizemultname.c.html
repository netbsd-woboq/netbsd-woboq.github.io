51void
52set_sizemult(uint secs)
53{
54 if (secs == 0)
55 return;
56
57 sizemult = secs;
58
59 if (secs == 1)
60 multname = msg_string(MSG_secname);
61 else if (secs == MEG/512)
62 multname = msg_string(MSG_megname);
63 else if (secs == GIG/512)
64 multname = msg_string(MSG_gigname);
65 else
66 multname = msg_string(MSG_cylname);
67}
68
69/*
70 * Only first call sets it
71 */
72void
73set_default_sizemult(uint secs)
74{
75 static bool default_done;
76
77 if (default_done)
78 return;
79
80 default_done = true;
81 set_sizemult(secs);
82}
