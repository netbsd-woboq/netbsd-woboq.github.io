51/*
52 * Do the system upgrade.
53 */
54void
55do_upgrade(void)
56{
57 struct install_partition_desc install;
58 int retcode = 0;
59 partman_go = 0;
60
61 msg_display(MSG_upgradeusure);
62 if (!ask_noyes(NULL))
63 return;
64
65 get_ramsize();
66
67 if (find_disks(msg_string(MSG_upgrade)) < 0)
68 return;
69
70 if (pm->parts->pscheme->pre_update_verify) {
71 if (pm->parts->pscheme->pre_update_verify(pm->parts))
72 pm->parts->pscheme->write_to_disk(pm->parts);
73 }
74
75 install_desc_from_parts(&install, pm->parts);
76
77 if (set_swap_if_low_ram(&install) < 0)
78 return;
79
80 if (md_pre_update(&install) < 0)
81 goto free_install;
82
83 if (mount_disks(&install) != 0)
84 goto free_install;
85
86
87 /*
88 * Save X symlink, ...
89 */
90 if (save_X("/usr/X11R6"))
91 goto free_install;
92 if (save_X("/usr/X11R7"))
93 goto free_install;
94
95#ifdef AOUT2ELF
96 move_aout_libs();
97#endif
98 /* Do any md updating of the file systems ... e.g. bootblocks,
99 copy file systems ... */
100 /* XXX pass install here too */
101 if (!md_update(&install))
102 goto free_install;
103
104 wrefresh(curscr);
105 wmove(stdscr, 0, 0);
106 wclear(stdscr);
107 wrefresh(stdscr);
108
109 /* Done with disks. Ready to get and unpack tarballs. */
110 process_menu(MENU_distset, &retcode);
111 if (retcode == 0)
112 goto free_install;
113 if (get_and_unpack_sets(1, MSG_disksetupdoneupdate,
114 MSG_upgrcomplete, MSG_abortupgr) != 0)
115 goto free_install;
116
117 if (md_post_extract(&install))
118 goto free_install;
119
120 merge_X("/usr/X11R6");
121 merge_X("/usr/X11R7");
122
123 sanity_check();
124
125free_install:
126 free_install_desc(&install);
127}

129/*
130 * Save X symlink to X.old so it can be recovered later
131 */
132static int
133save_X(const char *xroot)
134{
135 char newx[MAXPATHLEN], oldx[MAXPATHLEN];
136
137 strlcpy(newx, xroot, sizeof(newx));
138 strlcat(newx, "/bin/X", sizeof(newx));
139 strlcpy(oldx, newx, sizeof(oldx));
140 strlcat(oldx, ".old", sizeof(oldx));
141
142 /* Only care for X if it's a symlink */
143 if (target_symlink_exists_p(newx)) {
144 if (target_symlink_exists_p(oldx)) {
145 msg_fmt_display(MSG_X_oldexists,
146 "%s%s%s%s%s%s%s%s%s%s%s",
147 xroot, xroot, xroot, xroot, xroot, xroot, xroot,
148 xroot, xroot, xroot, xroot);
149 hit_enter_to_continue(NULL, NULL);
150 return EEXIST;
151 }
152
153#ifdef DEBUG
154 printf("saving %s as %s ...", newx, oldx);
155#endif
156
157 /* Move target .../X to .../X.old. Abort on error. */
158 mv_within_target_or_die(newx, oldx);
159 }
160
161 return 0;
162}
163
164/*
165 * Merge back saved target X files after unpacking the new
166 * sets has completed.
167 */
168static int
169merge_X(const char *xroot)
170{
171 char newx[MAXPATHLEN], oldx[MAXPATHLEN];
172
173 strlcpy(newx, xroot, sizeof(newx));
174 strlcat(newx, "/bin/X", sizeof(newx));
175 strlcpy(oldx, newx, sizeof(oldx));
176 strlcat(oldx, ".old", sizeof(oldx));
177
178 if (target_symlink_exists_p(oldx)) {
179 /* Only move back X if it's a symlink - we don't want
180 * to restore old binaries */
181 mv_within_target_or_die(oldx, newx);
182 }
183
184 return 0;
185}

187/*
188 * Unpacks sets, clobbering existing contents.
189 */
190void
191do_reinstall_sets(struct install_partition_desc *install)
192{
193 int retcode = 0;
194
195 unwind_mounts();
196 msg_display(MSG_reinstallusure);
197 if (!ask_noyes(NULL))
198 return;
199
200 if (find_disks(msg_string(MSG_reinstall)) < 0)
201 return;
202
203 /* XXX find proper pm pointer and pass it here, make sure we have
204 * read partitions and provide "infos" in there */
205 if (mount_disks(install) != 0)
206 return;
207
208 /* Unpack the distribution. */
209 process_menu(MENU_distset, &retcode);
210 if (retcode == 0)
211 return;
212 if (get_and_unpack_sets(0, NULL, MSG_unpackcomplete, MSG_abortunpack) != 0)
213 return;
214
215 sanity_check();
216}
