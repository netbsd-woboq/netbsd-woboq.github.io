177void
178init_set_status(int flags)
179{
180 static const uint8_t sets_valid[] = {MD_SETS_VALID};
181 static const uint8_t sets_selected_full[] = {MD_SETS_SELECTED};
182 static const uint8_t sets_selected_minimal[] = {MD_SETS_SELECTED_MINIMAL};
183 static const uint8_t sets_selected_nox[] = {MD_SETS_SELECTED_NOX};
184 static const uint8_t *sets_selected;
185 unsigned int nelem_selected;
186 unsigned int i, len;
187 const char *longest;
188
189 if (flags & SFLAG_MINIMAL) {
190 sets_selected = sets_selected_minimal;
191 nelem_selected = __arraycount(sets_selected_minimal);
192 } else if (flags & SFLAG_NOX) {
193 sets_selected = sets_selected_nox;
194 nelem_selected = __arraycount(sets_selected_nox);
195 } else {
196 sets_selected = sets_selected_full;
197 nelem_selected = __arraycount(sets_selected_full);
198 }
199
200 for (i = 0; i < __arraycount(sets_valid); i++)
201 set_status[sets_valid[i]] = SET_VALID;
202 for (i = 0; i < nelem_selected; i++)
203 set_status[sets_selected[i]] |= SET_SELECTED;
204
205 set_status[SET_GROUP] = SET_VALID;
206
207 /* Lookup some strings we need lots of times */
208 msg_yes = msg_string(MSG_Yes);
209 msg_no = msg_string(MSG_No);
210 msg_all = msg_string(MSG_All);
211 msg_some = msg_string(MSG_Some);
212 msg_none = msg_string(MSG_None);
213
214 /* Find longest and use it to determine width of selection menu */
215 len = strlen(msg_no); longest = msg_no;
216 i = strlen(msg_yes); if (i > len) {len = i; longest = msg_yes; }
217 i = strlen(msg_all); if (i > len) {len = i; longest = msg_all; }
218 i = strlen(msg_some); if (i > len) {len = i; longest = msg_some; }
219 i = strlen(msg_none); if (i > len) {len = i; longest = msg_none; }
220 select_menu_width = snprintf(NULL, 0, "%-30s %s", "", longest);
221
222 /* Give the md code a chance to choose the right kernel, etc. */
223 md_init_set_status(flags);
224}

251{
252 uint64_t ramsize;

254 int mib[2] = {CTL_HW, HW_PHYSMEM64};

281/*
282 * Performs in-place replacement of a set of patterns in a file that lives
283 * inside the installed system. The patterns must be separated by a semicolon.
284 * For example:
285 *
286 * replace("/etc/some-file.conf", "s/prop1=NO/prop1=YES/;s/foo/bar/");
287 */
288void
289replace(const char *path, const char *patterns, ...)
290{
291 char *spatterns;
292 va_list ap;
293
294 va_start(ap, patterns);
295 vasprintf(&spatterns, patterns, ap);
296 va_end(ap);
297 if (spatterns == NULL)
298 err(1, "vasprintf(&spatterns, \"%s\", ...)", patterns);
299
300 run_program(RUN_CHROOT, "sed -an -e '%s;H;$!d;g;w %s' %s", spatterns,
301 path, path);
302
303 free(spatterns);
304}

306static int
307floppy_fetch(const char *set_name)
308{
309 char post[4];
310 msg errmsg;
311 int menu;
312 int status;
313 const char *write_mode = ">";
314
315 strcpy(post, "aa");
316
317 errmsg = "";
318 menu = MENU_fdok;
319 for (;;) {
320 umount_mnt2();
321 msg_display(errmsg);
322 msg_fmt_display_add(MSG_fdmount, "%s%s", set_name, post);
323 process_menu(menu, &status);
324 if (status != SET_CONTINUE)
325 return status;
326 menu = MENU_fdremount;
327 errmsg = MSG_fdremount;
328 if (run_program(0, "/sbin/mount -r -t %s %s /mnt2",
329 fd_type, fd_dev))
330 continue;
331 mnt2_mounted = 1;
332 errmsg = MSG_fdnotfound;
333
334 /* Display this because it might take a while.... */
335 if (run_program(RUN_DISPLAY,
336 "sh -c '/bin/cat /mnt2/%s.%s %s %s/%s/%s%s'",
337 set_name, post, write_mode,
338 target_prefix(), xfer_dir, set_name,
339 set_postfix(set_name)))
340 /* XXX: a read error will give a corrupt file! */
341 continue;
342
343 /* We got that file, advance to next fragment */
344 if (post[1] < 'z')
345 post[1]++;
346 else
347 post[1] = 'a', post[0]++;
348 write_mode = ">>";
349 errmsg = "";
350 menu = MENU_fdok;
351 }
352}

375/*
376 * Get the volume name of a ISO9660 file system
377 */
378static int
379get_iso9660_volname(int dev, int sess, char *volname)
380{
381 int blkno, error, last;
382 char buf[ISO_BLKSIZE];
383 struct iso_volume_descriptor *vd = NULL;
384 struct iso_primary_descriptor *pd = NULL;
385
386 for (blkno = sess+16; blkno < sess+16+100; blkno++) {
387 error = pread(dev, buf, ISO_BLKSIZE, blkno*ISO_BLKSIZE);
388 if (error == -1)
389 return -1;
390 vd = (struct iso_volume_descriptor *)&buf;
391 if (memcmp(vd->id, ISO_STANDARD_ID, sizeof(vd->id)) != 0)
392 return -1;
393 if (isonum_711((const unsigned char *)&vd->type)
394 == ISO_VD_PRIMARY) {
395 pd = (struct iso_primary_descriptor*)buf;
396 strncpy(volname, pd->volume_id, sizeof pd->volume_id);
397 last = sizeof pd->volume_id-1;
398 while (last >= 0
399 && (volname[last] == ' ' || volname[last] == 0))
400 last--;
401 volname[last+1] = 0;
402 return 0;
403 }
404 }
405 return -1;
406}

409 * Local state while iterating CDs and collecting volumes

411struct get_available_cds_state {
412 struct cd_info *info;
413 size_t count;
414};

416/*
417 * Callback function: if this is a CD, enumerate all volumes on it
418 */
419static bool
420get_available_cds_helper(void *arg, const char *device)
421{
422 struct get_available_cds_state *state = arg;
423 char dname[16], volname[80];
424 struct disklabel label;
425 int part, dev, error, sess, ready;
426
427 if (!is_cdrom_device(device, false))
428 return true;
429
430 sprintf(dname, "/dev/r%s%c", device, 'a'+RAW_PART);
431 dev = open(dname, O_RDONLY, 0);
432 if (dev == -1)
433 return true;
434
435 ready = 0;
436 error = ioctl(dev, DIOCTUR, &ready);
437 if (error != 0 || ready == 0) {
438 close(dev);
439 return true;
440 }
441 error = ioctl(dev, DIOCGDINFO, &label);
442 close(dev);
443 if (error != 0)
444 return true;
445
446 for (part = 0; part < label.d_npartitions; part++) {
447
448 if (label.d_partitions[part].p_fstype == FS_UNUSED
449 || label.d_partitions[part].p_size == 0)
450 continue;
451
452 if (label.d_partitions[part].p_fstype == FS_ISO9660) {
453 sess = label.d_partitions[part].p_cdsession;
454 sprintf(dname, "/dev/r%s%c", device, 'a'+part);
455 dev = open(dname, O_RDONLY, 0);
456 if (dev == -1)
457 continue;
458 error = get_iso9660_volname(dev, sess, volname);
459 close(dev);
460 if (error)
461 continue;
462 sprintf(state->info->device_name,
463 "%s%c", device, 'a'+part);
464 sprintf(state->info->menu, "%s (%s)",
465 state->info->device_name, volname);
466 } else {
467 /*
468 * All install CDs use partition
469 * a for the sets.
470 */
471 if (part > 0)
472 continue;
473 sprintf(state->info->device_name,
474 "%s%c", device, 'a'+part);
475 strcpy(state->info->menu, state->info->device_name);
476 }
477 state->info++;
478 if (++state->count >= MAX_CD_INFOS)
479 return false;
480 }
481
482 return true;
483}

485/*

489static int
490get_available_cds(void)

517/*
518 * Check whether we can remove the boot media.
519 * If it is not a local filesystem, return -1.
520 * If we can not decide for sure (can not tell MD content from plain ffs
521 * on hard disk, for example), return 0.
522 * If it is a CD/DVD, return 1.
523 */
524int
525boot_media_still_needed(void)
526{
527 struct statvfs sb;
528
529 if (statvfs("/", &sb) == 0) {
530 if (!(sb.f_flag & ST_LOCAL))
531 return -1;
532 if (strcmp(sb.f_fstypename, MOUNT_CD9660) == 0
533 || strcmp(sb.f_fstypename, MOUNT_UDF) == 0)
534 return 1;
535 }
536
537 return 0;
538}

540/*
541 * Get from a CDROM distribution.
542 * Also used on "installation using bootable install media"
543 * as the default option in the "distmedium" menu.
544 */
545int
546get_via_cdrom(void)
547{
548 menu_ent cd_menu[MAX_CD_INFOS];
549 struct stat sb;
550 int rv, num_cds, menu_cd, i, selected_cd = 0;
551 bool silent = false;
552 int mib[2];
553 char rootdev[SSTRSIZE] = "";
554 size_t varlen;
555
556 /* If root is not md(4) and we have set dir, skip this step. */
557 mib[0] = CTL_KERN;
558 mib[1] = KERN_ROOT_DEVICE;
559 varlen = sizeof(rootdev);
560 (void)sysctl(mib, 2, rootdev, &varlen, NULL, 0);
561 if (stat(set_dir_bin, &sb) == 0 && S_ISDIR(sb.st_mode) &&
562 strncmp("md", rootdev, 2) != 0) {
563 strlcpy(ext_dir_bin, set_dir_bin, sizeof ext_dir_bin);
564 strlcpy(ext_dir_src, set_dir_src, sizeof ext_dir_src);
565 return SET_OK;
566 }
567
568 memset(cd_menu, 0, sizeof(cd_menu));
569 num_cds = get_available_cds();
570 if (num_cds <= 0) {
571 silent = true;
572 } else if (num_cds == 1) {
573 /* single CD found, check for sets on it */
574 strcpy(cdrom_dev, cds[0].device_name);
575 if (cd_has_sets())
576 return SET_OK;
577 } else {
578 for (i = 0; i< num_cds; i++) {
579 cd_menu[i].opt_name = cds[i].menu;
580 cd_menu[i].opt_flags = OPT_EXIT;
581 cd_menu[i].opt_action = set_menu_select;
582 }
583 /* create a menu offering available choices */
584 menu_cd = new_menu(MSG_Available_cds,
585 cd_menu, num_cds, -1, 4, 0, 0,
586 MC_SCROLL | MC_NOEXITOPT,
587 NULL, NULL, NULL, NULL, NULL);
588 if (menu_cd == -1)
589 return SET_RETRY;
590 msg_display(MSG_ask_cd);
591 process_menu(menu_cd, &selected_cd);
592 free_menu(menu_cd);
593 strcpy(cdrom_dev, cds[selected_cd].device_name);
594 if (cd_has_sets())
595 return SET_OK;
596 }
597
598 if (silent)
599 msg_display("");
600 else {
601 umount_mnt2();
602 hit_enter_to_continue(MSG_cd_path_not_found, NULL);
603 }
604
605 /* ask for paths on the CD */
606 rv = -1;
607 process_menu(MENU_cdromsource, &rv);
608 if (rv == SET_RETRY)
609 return SET_RETRY;
610
611 if (cd_has_sets())
612 return SET_OK;
613
614 return SET_RETRY;
615}

886/*
887 * Extract_file **REQUIRES** an absolute path in ext_dir. Any code
888 * that sets up xfer_dir for use by extract_file needs to put in the
889 * full path name to the directory.
890 */
891
892int
893extract_file(distinfo *dist, int update)
894{
895 char path[STRSIZE];
896 char *owd;
897 int rval;
898
899 /* If we might need to tidy up, ensure directory exists */
900 if (fetch_fn != NULL)
901 make_target_dir(xfer_dir);
902
903 (void)snprintf(path, sizeof path, "%s/%s%s",
904 ext_dir_for_set(dist->name), dist->name, set_postfix(dist->name));
905
906 owd = getcwd(NULL, 0);
907
908 /* Do we need to fetch the file now? */
909 if (fetch_fn != NULL) {
910 rval = fetch_fn(dist->name);
911 if (rval != SET_OK)
912 return rval;
913 }
914
915 /* check tarfile exists */
916 if (!file_exists_p(path)) {
917
918#ifdef SUPPORT_8_3_SOURCE_FILESYSTEM
919 /*
920 * Update path to use dist->name truncated to the first eight
921 * characters and check again
922 */
923 (void)snprintf(path, sizeof path, "%s/%.8s%.4s", /* 4 as includes '.' */
924 ext_dir_for_set(dist->name), dist->name, set_postfix(dist->name));
925 if (!file_exists_p(path)) {
926#endif /* SUPPORT_8_3_SOURCE_FILESYSTEM */
927
928 tarstats.nnotfound++;
929
930 hit_enter_to_continue(MSG_notarfile, NULL);
931 return SET_RETRY;
932 }
933#ifdef SUPPORT_8_3_SOURCE_FILESYSTEM
934 }
935#endif /* SUPPORT_8_3_SOURCE_FILESYSTEM */
936
937 tarstats.nfound++;
938 /* cd to the target root. */
939 if (update && (dist->set == SET_ETC || dist->set == SET_X11_ETC)) {
940 make_target_dir("/.sysinst");
941 target_chdir_or_die("/.sysinst");
942 } else if (dist->set == SET_PKGSRC)
943 target_chdir_or_die("/usr");
944 else
945 target_chdir_or_die("/");
946
947 /*
948 * /usr/X11R7/lib/X11/xkb/symbols/pc was a directory in 5.0
949 * but is a file in 5.1 and beyond, so on upgrades we need to
950 * delete it before extracting the xbase set.
951 */
952 if (update && dist->set == SET_X11_BASE)
953 run_program(0, "rm -rf usr/X11R7/lib/X11/xkb/symbols/pc");
954
955 /* now extract set files into "./". */
956 rval = run_program(RUN_DISPLAY | RUN_PROGRESS,
957 "progress -zf %s tar --chroot -xhepf -", path);
958
959 chdir(owd);
960 free(owd);
961
962 /* Check rval for errors and give warning. */
963 if (rval != 0) {
964 tarstats.nerror++;
965 msg_fmt_display(MSG_tarerror, "%s", path);
966 hit_enter_to_continue(NULL, NULL);
967 return SET_RETRY;
968 }
969
970 if (fetch_fn != NULL && clean_xfer_dir) {
971 run_program(0, "rm %s", path);
972 /* Plausibly we should unlink an empty xfer_dir as well */
973 }
974
975 set_status[dist->set] |= SET_INSTALLED;
976 tarstats.nsuccess++;
977 return SET_OK;
978}

1001/*
1002 * Get and unpack the distribution.
1003 * Show success_msg if installation completes.
1004 * Otherwise show failure_msg and wait for the user to ack it before continuing.
1005 * success_msg and failure_msg must both be 0-adic messages.
1006 */
1007int
1008get_and_unpack_sets(int update, msg setupdone_msg, msg success_msg, msg failure_msg)
1009{
1010 distinfo *dist;
1011 int status;
1012 int set;
1013
1014 /* Ensure mountpoint for distribution files exists in current root. */
1015 (void)mkdir("/mnt2", S_IRWXU| S_IRGRP|S_IXGRP | S_IROTH|S_IXOTH);
1016 if (script)
1017 (void)fprintf(script, "mkdir -m 755 /mnt2\n");
1018
1019 /* reset failure/success counters */
1020 memset(&tarstats, 0, sizeof(tarstats));
1021
1022 /* Find out which files to "get" if we get files. */
1023
1024 /* Accurately count selected sets */
1025 for (dist = dist_list; (set = dist->set) != SET_LAST; dist++) {
1026 if (dist->name == NULL)
1027 continue;
1028 if ((set_status[set] & (SET_VALID | SET_SELECTED))
1029 == (SET_VALID | SET_SELECTED))
1030 tarstats.nselected++;
1031 }
1032
1033 status = SET_RETRY;
1034 for (dist = dist_list; (set = dist->set) != SET_LAST; dist++) {
1035 if (dist->name == NULL)
1036 continue;
1037 if (set_status[set] != (SET_VALID | SET_SELECTED))
1038 continue;
1039
1040 if (status != SET_OK) {
1041 /* This might force a redraw.... */
1042 clearok(curscr, 1);
1043 touchwin(stdscr);
1044 wrefresh(stdscr);
1045 /* Sort out the location of the set files */
1046 do {
1047 umount_mnt2();
1048 msg_fmt_display(MSG_distmedium, "%d%d%s",
1049 tarstats.nselected,
1050 tarstats.nsuccess + tarstats.nskipped,
1051 dist->name);
1052 fetch_fn = NULL;
1053 process_menu(MENU_distmedium, &status);
1054 } while (status == SET_RETRY);
1055
1056 if (status == SET_SKIP) {
1057 set_status[set] |= SET_SKIPPED;
1058 tarstats.nskipped++;
1059 continue;
1060 }
1061 if (status == SET_SKIP_GROUP) {
1062 skip_set(dist, status);
1063 continue;
1064 }
1065 if (status != SET_OK) {
1066 hit_enter_to_continue(failure_msg, NULL);
1067 return 1;
1068 }
1069 }
1070
1071 /* Try to extract this set */
1072 status = extract_file(dist, update);
1073 if (status == SET_RETRY)
1074 dist--;
1075 }
1076
1077 if (tarstats.nerror == 0 && tarstats.nsuccess == tarstats.nselected) {
1078 msg_display(MSG_endtarok);
1079 /* Give user a chance to see the success message */
1080 sleep(1);
1081 } else {
1082 /* We encountered errors. Let the user know. */
1083 msg_fmt_display(MSG_endtar, "%d%d%d%d%d%d",
1084 tarstats.nselected, tarstats.nnotfound, tarstats.nskipped,
1085 tarstats.nfound, tarstats.nsuccess, tarstats.nerror);
1086 hit_enter_to_continue(NULL, NULL);
1087 }
1088
1089 /*
1090 * postinstall needs to be run after extracting all sets, because
1091 * otherwise /var/db/obsolete will only have current information
1092 * from the base, comp, and etc sets.
1093 */
1094 if (update && (set_status[SET_ETC] & SET_INSTALLED)) {
1095 int oldsendmail;
1096 oldsendmail = run_program(RUN_DISPLAY | RUN_CHROOT |
1097 RUN_ERROR_OK | RUN_PROGRESS,
1098 "/usr/sbin/postinstall -s /.sysinst -d / check mailerconf");
1099 if (oldsendmail == 1) {
1100 msg_display(MSG_oldsendmail);
1101 if (ask_yesno(NULL)) {
1102 run_program(RUN_DISPLAY | RUN_CHROOT,
1103 "/usr/sbin/postinstall -s /.sysinst -d / fix mailerconf");
1104 }
1105 }
1106 run_program(RUN_DISPLAY | RUN_CHROOT,
1107 "/usr/sbin/postinstall -s /.sysinst -d / fix");
1108
1109 /* Don't discard the system's old entropy if any */
1110 run_program(RUN_CHROOT | RUN_SILENT,
1111 "/etc/rc.d/random_seed start");
1112 }
1113
1114 /* Configure the system */
1115 if (set_status[SET_BASE] & SET_INSTALLED)
1116 run_makedev();
1117
1118 if (!update) {
1119 struct stat sb1, sb2;
1120
1121 if (stat(target_expand("/"), &sb1) == 0
1122 && stat(target_expand("/var"), &sb2) == 0
1123 && sb1.st_dev != sb2.st_dev) {
1124 add_rc_conf("random_file=/etc/entropy-file\n");
1125 if (target_file_exists_p("/boot.cfg")) {
1126 run_program(RUN_CHROOT|RUN_FATAL,
1127 "sh -c 'sed -e s./var/db/./etc/. "
1128 "< /boot.cfg "
1129 "> /tmp/boot.cfg.tmp'");
1130 mv_within_target_or_die("/tmp/boot.cfg.tmp",
1131 "/boot.cfg");
1132 }
1133 }
1134
1135 /* Save keyboard type */
1136 save_kb_encoding();
1137
1138 /* Other configuration. */
1139 mnt_net_config();
1140 }
1141
1142 /* Mounted dist dir? */
1143 umount_mnt2();
1144
1145 /* Save entropy -- on some systems it's ~all we'll ever get */
1146 run_program(RUN_DISPLAY | RUN_CHROOT | RUN_FATAL | RUN_PROGRESS,
1147 "/etc/rc.d/random_seed stop");
1148 /* Install/Upgrade complete ... reboot or exit to script */
1149 hit_enter_to_continue(success_msg, NULL);
1150 return 0;
1151}

1381 }

1383 }

1385
1386 m->opts = tz_menu;
1387 m->numopts = nfiles;
1388 qsort(tz_menu, nfiles, sizeof *tz_menu, tz_sort);
1389}

1507int
1508del_rc_conf(const char *value)
1509{
1510 FILE *fp, *nfp;
1511 char buf[4096]; /* Ridiculously high, but should be enough in any way */
1512 char *rcconf, *tempname = NULL, *bakname = NULL;
1513 char *cp;
1514 int done = 0;
1515 int fd;
1516 int retval = 0;
1517
1518 /* The paths might seem strange, but using /tmp would require copy instead
1519 * of rename operations. */
1520 if (asprintf(&rcconf, "%s", target_expand("/etc/rc.conf")) < 0
1521 || asprintf(&tempname, "%s", target_expand("/etc/rc.conf.tmp.XXXXXX")) < 0
1522 || asprintf(&bakname, "%s", target_expand("/etc/rc.conf.bak.XXXXXX")) < 0) {
1523 if (rcconf)
1524 free(rcconf);
1525 if (tempname)
1526 free(tempname);
1527 msg_fmt_display(MSG_rcconf_delete_failed, "%s", value);
1528 hit_enter_to_continue(NULL, NULL);
1529 return -1;
1530 }
1531
1532 if ((fd = mkstemp(bakname)) < 0) {
1533 msg_fmt_display(MSG_rcconf_delete_failed, "%s", value);
1534 hit_enter_to_continue(NULL, NULL);
1535 return -1;
1536 }
1537 close(fd);
1538
1539 if (!(fp = fopen(rcconf, "r+")) || (fd = mkstemp(tempname)) < 0) {
1540 if (fp)
1541 fclose(fp);
1542 msg_fmt_display(MSG_rcconf_delete_failed, "%s", value);
1543 hit_enter_to_continue(NULL, NULL);
1544 return -1;
1545 }
1546
1547 nfp = fdopen(fd, "w");
1548 if (!nfp) {
1549 fclose(fp);
1550 close(fd);
1551 msg_fmt_display(MSG_rcconf_delete_failed, "%s", value);
1552 hit_enter_to_continue(NULL, NULL);
1553 return -1;
1554 }
1555
1556 while (fgets(buf, sizeof buf, fp) != NULL) {
1557
1558 cp = buf + strspn(buf, " \t"); /* Skip initial spaces */
1559 if (strncmp(cp, value, strlen(value)) == 0) {
1560 cp += strlen(value);
1561 if (*cp != '=')
1562 scripting_fprintf(nfp, "%s", buf);
1563 else
1564 done = 1;
1565 } else {
1566 scripting_fprintf(nfp, "%s", buf);
1567 }
1568 }
1569 fclose(fp);
1570 fclose(nfp);
1571
1572 if (done) {
1573 if (rename(rcconf, bakname)) {
1574 msg_display(MSG_rcconf_backup_failed);
1575 if (!ask_noyes(NULL)) {
1576 retval = -1;
1577 goto done;
1578 }
1579 }
1580
1581 if (rename(tempname, rcconf)) {
1582 if (rename(bakname, rcconf)) {
1583 hit_enter_to_continue(MSG_rcconf_restore_failed,
1584 NULL);
1585 } else {
1586 hit_enter_to_continue(MSG_rcconf_delete_failed,
1587 NULL);
1588 }
1589 } else {
1590 (void)unlink(bakname);
1591 }
1592 }
1593
1594done:
1595 (void)unlink(tempname);
1596 free(rcconf);
1597 free(tempname);
1598 free(bakname);
1599 return retval;
1600}

1692binary_available(const char *prog)
1693{

1695

1698 path = strdup(path);

1705 if (strlcpy(tmp, p, MAXPATHLEN) >= MAXPATHLEN)
1706 continue;
1707 if (strlcat(tmp, "/", MAXPATHLEN) >= MAXPATHLEN)

1712 free(opath);
1713 return 1;
1714 }
1715 }
1716 free(opath);

1796/*
1797 * Replace positional arguments (encoded as $0 .. $N) in the string
1798 * passed by the contents of the passed argument array.
1799 * Caller must free() the result string.
1800 */
1801char*
1802str_arg_subst(const char *src, size_t argc, const char **argv)
1803{
1804 const char *p, *last;
1805 char *out, *t;
1806 size_t len;
1807
1808 len = strlen(src);
1809 for (p = strchr(src, '$'); p; p = strchr(p+1, '$')) {
1810 char *endp = NULL;
1811 size_t n;
1812 int e;
1813
1814 /* $ followed by a correct numeric position? */
1815 n = strtou(p+1, &endp, 10, 0, INT_MAX, &e);
1816 if ((e == 0 || e == ENOTSUP) && n < argc) {
1817 len += strlen(argv[n]);
1818 len -= endp-p;
1819 p = endp-1;
1820 }
1821 }
1822
1823 out = malloc(len+1);
1824 if (out == NULL)
1825 return NULL;
1826
1827 t = out;
1828 for (last = src, p = strchr(src, '$'); p; p = strchr(p+1, '$')) {
1829 char *endp = NULL;
1830 size_t n;
1831 int e;
1832
1833 /* $ followed by a correct numeric position? */
1834 n = strtou(p+1, &endp, 10, 0, INT_MAX, &e);
1835 if ((e == 0 || e == ENOTSUP) && n < argc) {
1836 size_t l = p-last;
1837 memcpy(t, last, l);
1838 t += l;
1839 strcpy(t, argv[n]);
1840 t += strlen(argv[n]);
1841 last = endp;
1842 }
1843 }
1844 if (*last) {
1845 strcpy(t, last);
1846 t += strlen(last);
1847 } else {
1848 *t = 0;
1849 }
1850 assert((size_t)(t-out) == len);
1851
1852 return out;
1853}

1965 * so that we "want" exactly what is there already.

1967static bool
1968usage_info_list_from_parts(struct part_usage_info **list, size_t *count,
1969 struct disk_partitions *parts)
1970{
1971 struct disk_part_info info;
1972 part_id pno;
1973 size_t no, num;
1974
1975 num = 0;
1976 for (pno = 0; pno < parts->num_part; pno++) {
1977 if (!parts->pscheme->get_part_info(parts, pno, &info))
1978 continue;
1979 num++;
1980 }
1981
1982 *list = calloc(num, sizeof(**list));
1983 if (*list == NULL)
1984 return false;
1985
1986 *count = num;
1987 for (no = pno = 0; pno < parts->num_part && no < num; pno++) {
1988 if (!parts->pscheme->get_part_info(parts, pno, &info))
1989 continue;
1990 (*list)[no].size = info.size;
1991 if (info.last_mounted != NULL && *info.last_mounted != 0) {
1992 strlcpy((*list)[no].mount, info.last_mounted,
1993 sizeof((*list)[no].mount));
1994 (*list)[no].instflags |= PUIINST_MOUNT;
1995 }
1996 (*list)[no].parts = parts;
1997 (*list)[no].cur_part_id = pno;
1998 (*list)[no].cur_start = info.start;
1999 (*list)[no].cur_flags = info.flags;
2000 (*list)[no].type = info.nat_type->generic_ptype;
2001 if ((*list)[no].type == PT_swap) {
2002 (*list)[no].fs_type = FS_SWAP;
2003 (*list)[no].fs_version = 0;
2004 } else {
2005 (*list)[no].fs_type = info.fs_type;
2006 (*list)[no].fs_version = info.fs_sub_type;
2007 }
2008 no++;
2009 }
2010 return true;
