35/* vsyslog - format one record */
36
37void
38vsyslog(int severity, const char *fmt, va_list ap)
39{
40 char buf[BUFSIZ];
41
42 vprintf(percent_m(buf, fmt), ap);
43 printf("\n");
44 fflush(stdout);
45}
46
47/* syslog - format one record */
48
49/* VARARGS */
50
51void
52syslog(int severity, const char *fmt, ...)
53{
54 va_list ap;
55
56 va_start(ap, fmt);
57 vsyslog(severity, fmt, ap);
58 va_end(ap);
59}
