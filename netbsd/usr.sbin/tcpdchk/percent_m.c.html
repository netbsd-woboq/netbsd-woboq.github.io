30char *percent_m(char *obuf, const char *ibuf)
31{
32 char *bp = obuf;
33 const char *cp = ibuf;
34
35 while ((*bp = *cp) != '\0')
36 if (*cp == '%' && cp[1] == 'm') {
37 strcpy(bp, strerror(errno));
38 bp += strlen(bp);
39 cp += 2;
40 } else {
41 bp++, cp++;
42 }
43 return (obuf);
44}
