41/*
42 * Two routines to do the necessary byte swapping for timed protocol
43 * messages. Protocol is defined in /usr/include/protocols/timed.h
44 */
45void
46bytenetorder(struct tsp *ptr)
47{
48 ptr->tsp_seq = htons((uint16_t)ptr->tsp_seq);
49 switch (ptr->tsp_type) {
50
51 case TSP_SETTIME:
52 case TSP_ADJTIME:
53 case TSP_SETDATE:
54 case TSP_SETDATEREQ:
55 ptr->tsp_time.tv_sec = htonl((uint32_t)ptr->tsp_time.tv_sec);
56 ptr->tsp_time.tv_usec = htonl((uint32_t)ptr->tsp_time.tv_usec);
57 break;
58
59 default:
60 break; /* nothing more needed */
61 }
62}
63
64void
65bytehostorder(struct tsp *ptr)
66{
67 ptr->tsp_seq = ntohs((uint16_t)ptr->tsp_seq);
68 switch (ptr->tsp_type) {
69
70 case TSP_SETTIME:
71 case TSP_ADJTIME:
72 case TSP_SETDATE:
73 case TSP_SETDATEREQ:
74 ptr->tsp_time.tv_sec = ntohl((uint32_t)ptr->tsp_time.tv_sec);
75 ptr->tsp_time.tv_usec = ntohl((uint32_t)ptr->tsp_time.tv_usec);
76 break;
77
78 default:
79 break; /* nothing more needed */
80 }
81}
