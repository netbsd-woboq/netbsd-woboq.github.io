48void
49localhostname(char *buf, size_t buflen)
50{
51 struct addrinfo *res, hints;
52 char hostname[MAXHOSTNAMELEN + 1];
53 int error;
54
55 if (gethostname(hostname, sizeof(hostname)))
56 err(1, "gethostname");
57 hostname[sizeof(hostname) - 1] = '\0';
58
59 memset(buf, 0, buflen);
60
61 /*
62 * If hostname appears to be fully-qualified,
63 * just return it.
64 */
65 if (strchr(hostname, '.')) {
66 strlcpy(buf, hostname, buflen);
67 return;
68 }
69
70 /*
71 * Lookup the host. Return the canonical name.
72 */
73 memset(&hints, 0, sizeof(hints));
74 hints.ai_flags = AI_CANONNAME;
75 hints.ai_family = PF_UNSPEC;
76
77 error = getaddrinfo(hostname, NULL, &hints, &res);
78 if (error)
79 errx(1, "getaddrinfo(%s) failed: %s", hostname,
80 gai_strerror(error));
81 strlcpy(buf, res->ai_canonname, buflen);
82 freeaddrinfo(res);
83}
