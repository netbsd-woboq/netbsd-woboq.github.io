36/* Groupname entry hung off a member_entry node. */
37struct grouplist {
38 char *groupname;
39 struct grouplist *next;
40};
41
42/* Entry in the cooked member list hash table. */
43struct member_entry {
44 char *key;
45 char *domain;
46 struct grouplist *groups;
47 struct member_entry *next;
48};
49
50/* Entry in the raw netgroup table. */
51struct group_entry {
52 char *key;
53 char *data;
54 struct group_entry *next;
55};
56
57/* Table size (chosen arbitrarily). Not too big, not too small. */
58#define TABLESIZE 256
59#define HASH_MASK 0x000000FF
60
61#define LINSIZ 1024 * 10
