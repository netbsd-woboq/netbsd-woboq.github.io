51int
52main(int argc, char *argv[])
53{
54 struct ether_addr eth_addr;
55 FILE *data_file;
56 size_t line_no;
57 size_t len;
58 const char *fname;
59 char *p;
60 char hostname[MAXHOSTNAMELEN + 1];
61
62 if (argc > 2)
63 usage();
64
65 if (argc == 2) {
66 fname = argv[1];
67 data_file = fopen(fname, "r");
68 if (data_file == NULL)
69 err(1, "%s", fname);
70 } else {
71 fname = "<stdin>";
72 data_file = stdin;
73 }
74
75 line_no = 0;
76 for (;
77 (p = fparseln(data_file, &len, &line_no, NULL, FPARSELN_UNESCALL));
78 free(p)) {
79 if (len == 0)
80 continue;
81
82 if (ether_line(p, &eth_addr, hostname) == 0)
83 printf("%s\t%s\n", ether_ntoa(&eth_addr), hostname);
84 else
85 warnx("ignoring line %lu: `%s'",
86 (unsigned long)line_no, p);
87 }
88
89 return 0;
90}

92static void
93usage(void)
94{
95
96 fprintf(stderr, "usage: %s [file]\n", getprogname());
97 exit(1);
98}
