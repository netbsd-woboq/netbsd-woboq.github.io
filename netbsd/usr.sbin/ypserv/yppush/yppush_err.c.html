45const char *
46yppush_err_string(int y)
47{
48 static char errstr[64];
49
50 switch (y) {
51 case YPPUSH_SUCC:
52 return "Success";
53
54 case YPPUSH_AGE:
55 return "Master's version not newer";
56
57 case YPPUSH_NOMAP:
58 return "Can't find server for map";
59
60 case YPPUSH_NODOM:
61 return "Domain not supported";
62
63 case YPPUSH_RSRC:
64 return "Local resource alloc failure";
65
66 case YPPUSH_RPC:
67 return "RPC failure talking to server";
68
69 case YPPUSH_MADDR:
70 return "Can't get master address";
71
72 case YPPUSH_YPERR:
73 return "YP server/map db error";
74
75 case YPPUSH_BADARGS:
76 return "Request arguments bad";
77
78 case YPPUSH_DBM:
79 return "Local dbm operation failed";
80
81 case YPPUSH_FILE:
82 return "Local file I/O operation failed";
83
84 case YPPUSH_SKEW:
85 return "Map version skew during transfer";
86
87 case YPPUSH_CLEAR:
88 return "Can't send \"Clear\" req to local ypserv";
89
90 case YPPUSH_FORCE:
91 return "No local order number in map use -f flag.";
92
93 case YPPUSH_XFRERR:
94 return "ypxfr error";
95
96 case YPPUSH_REFUSED:
97 return "Transfer request refused by ypserv";
98
99 default:
100 memset(errstr, 0, sizeof(errstr));
101 snprintf(errstr, sizeof(errstr), "unknown error code: %d", y);
102 return errstr;
103 }
104};
