45void *
46yppushproc_null_1_svc(void *argp, struct svc_req *rqstp)
47{
48 static char result;
49
50 (void) memset(&result, 0, sizeof(result));
51 return ((void *)&result);
52}

54void *
55yppushproc_xfrresp_1_svc(void *argp, struct svc_req *rqstp)
56{
57 struct yppushresp_xfr *resp = argp;
58 static char result;
59
60 if (resp->status != YPPUSH_SUCC)
61 warnx("%s", yppush_err_string(resp->status));
62
63 (void) memset(&result, 0, sizeof(result));
64 return ((void *)&result);
65}
