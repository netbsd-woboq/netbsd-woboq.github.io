70struct opt_map {
71 char *map; /* map name (malloc'd) */
72 DBM *db; /* database */
73 struct opt_domain *dom; /* back ptr to our domain */
74 int host_lookup; /* host lookup */
75 int secure; /* is this map secure? */
76 dev_t dbdev; /* device db is on */
77 ino_t dbino; /* inode of db */
78 time_t dbmtime; /* time of last db modification */
79 CIRCLEQ_ENTRY(opt_map) mapsq; /* map queue pointers */
80 LIST_ENTRY(opt_map) mapsl; /* map list pointers */
81};

112/*
113 * yp_private:
114 * Check if key is a YP private key. Return TRUE if it is and
115 * ypprivate is FALSE.
116 */
117int
118yp_private(datum key, int ypprivate)
119{
120
121 if (ypprivate)
122 return (FALSE);
123
124 if (key.dsize == 0 || key.dptr == NULL)
125 return (FALSE);
126
127 if (key.dsize == YP_LAST_LEN &&
128 strncmp(key.dptr, YP_LAST_KEY, YP_LAST_LEN) == 0)
129 return (TRUE);
130
131 if (key.dsize == YP_INPUT_LEN &&
132 strncmp(key.dptr, YP_INPUT_KEY, YP_INPUT_LEN) == 0)
133 return (TRUE);
134
135 if (key.dsize == YP_OUTPUT_LEN &&
136 strncmp(key.dptr, YP_OUTPUT_KEY, YP_OUTPUT_LEN) == 0)
137 return (TRUE);
138
139 if (key.dsize == YP_MASTER_LEN &&
140 strncmp(key.dptr, YP_MASTER_KEY, YP_MASTER_LEN) == 0)
141 return (TRUE);
142
143 if (key.dsize == YP_DOMAIN_LEN &&
144 strncmp(key.dptr, YP_DOMAIN_KEY, YP_DOMAIN_LEN) == 0)
145 return (TRUE);
146
147 if (key.dsize == YP_INTERDOMAIN_LEN &&
148 strncmp(key.dptr, YP_INTERDOMAIN_KEY, YP_INTERDOMAIN_LEN) == 0)
149 return (TRUE);
150
151 if (key.dsize == YP_SECURE_LEN &&
152 strncmp(key.dptr, YP_SECURE_KEY, YP_SECURE_LEN) == 0)
153 return (TRUE);
154
155 return (FALSE);
156}

158/*
159 * Close specified map.
160 */
161void
162ypdb_close_map(struct opt_map *map)
163{
164 CIRCLEQ_REMOVE(&maps, map, mapsq); /* remove from LRU circleq */
165 LIST_REMOVE(map, mapsl); /* remove from domain list */
166
167#ifdef DEBUG
168 syslog(LOG_DEBUG,
169 "ypdb_close_map: closing map %s in domain %s [db=%p]",
170 map->map, map->dom->domain, map->db);
171#endif
172
173 ypdb_close(map->db); /* close DB */
174 free(map->map); /* free map name */
175 free(map); /* free map */
176}

178/*
179 * Close least recently used map. This routine is called when we have
180 * no more file descriptors free, or we want to close all maps.
181 */
182void
183ypdb_close_last(void)
184{
185 struct opt_map *last = maps.cqh_last;
186
187 if (last == (void *) &maps) {
188 syslog(LOG_ERR,
189 "ypdb_close_last: LRU list is empty!");
190 return;
191 }
192 ypdb_close_map(last);
193}
194
195/*
196 * Close all open maps.
197 */
198void
199ypdb_close_all(void)
200{
201
202#ifdef DEBUG
203 syslog(LOG_DEBUG, "ypdb_close_all(): start");
204#endif
205
206 while (maps.cqh_first != (void *) &maps)
207 ypdb_close_last();
208
209#ifdef DEBUG
210 syslog(LOG_DEBUG, "ypdb_close_all(): done");
211#endif
212}

231/*
232 * ypdb_open_db
233 */
234DBM *
235ypdb_open_db(const char *domain, const char *map, u_int *status,
236 struct opt_map **map_info)
237{
238 static const char *domain_key = YP_INTERDOMAIN_KEY;
239 static const char *secure_key = YP_SECURE_KEY;
240 char map_path[MAXPATHLEN];
241 struct stat finfo;
242 struct opt_domain *d = NULL;
243 struct opt_map *m = NULL;
244 DBM *db;
245 datum k, v;
246
247 *status = YP_TRUE; /* defaults to true */
248
249 /*
250 * check for illegal domain and map names
251 */
252 if (_yp_invalid_domain(domain)) {
253 *status = YP_NODOM;
254 return (NULL);
255 }
256 if (_yp_invalid_map(map)) {
257 *status = YP_NOMAP;
258 return (NULL);
259 }
260
261 /*
262 * check for domain, file.
263 */
264 (void)snprintf(map_path, sizeof(map_path), "%s/%s", YP_DB_PATH, domain);
265 if (stat(map_path, &finfo) < 0 || !S_ISDIR(finfo.st_mode)) {
266#ifdef DEBUG
267 syslog(LOG_DEBUG,
268 "ypdb_open_db: no domain %s (map=%s)", domain, map);
269#endif
270 *status = YP_NODOM;
271 } else {
272 (void)snprintf(map_path, sizeof(map_path), "%s/%s/%s%s",
273 YP_DB_PATH, domain, map, YPDB_SUFFIX);
274 if (stat(map_path, &finfo) < 0) {
275#ifdef DEBUG
276 syslog(LOG_DEBUG,
277 "ypdb_open_db: no map %s (domain=%s)", map,
278 domain);
279#endif
280 *status = YP_NOMAP;
281 }
282 }
283
284 /*
285 * check for preloaded domain, map
286 */
287 for (d = doms.lh_first; d != NULL; d = d->domsl.le_next)
288 if (strcmp(domain, d->domain) == 0)
289 break;
290
291 if (d)
292 for (m = d->dmaps.lh_first; m != NULL; m = m->mapsl.le_next)
293 if (strcmp(map, m->map) == 0)
294 break;
295
296 /*
297 * map found open?
298 */
299 if (m) {
300#ifdef DEBUG
301 syslog(LOG_DEBUG,
302 "ypdb_open_db: cached open: domain=%s, map=%s, db=%p,",
303 domain, map, m->db);
304 syslog(LOG_DEBUG,
305 "\tdbdev %d new %d; dbino %d new %d; dbmtime %ld new %ld",
306 m->dbdev, finfo.st_dev, m->dbino, finfo.st_ino,
307 (long) m->dbmtime, (long) finfo.st_mtime);
308#endif
309 /*
310 * if status != YP_TRUE, then this cached database is now
311 * non-existent
312 */
313 if (*status != YP_TRUE) {
314#ifdef DEBUG
315 syslog(LOG_DEBUG,
316 "ypdb_open_db: cached db is now unavailable - "
317 "closing: status %s",
318 yperr_string(ypprot_err(*status)));
319#endif
320 ypdb_close_map(m);
321 return (NULL);
322 }
323
324 /*
325 * is this the same db?
326 */
327 if (finfo.st_dev == m->dbdev && finfo.st_ino == m->dbino &&
328 finfo.st_mtime == m->dbmtime) {
329 CIRCLEQ_REMOVE(&maps, m, mapsq); /* adjust LRU queue */
330 CIRCLEQ_INSERT_HEAD(&maps, m, mapsq);
331 if (map_info)
332 *map_info = m;
333 return (m->db);
334 } else {
335#ifdef DEBUG
336 syslog(LOG_DEBUG,
337 "ypdb_open_db: db changed; closing");
338#endif
339 ypdb_close_map(m);
340 m = NULL;
341 }
342 }
343
344 /*
345 * not cached and non-existent, return
346 */
347 if (*status != YP_TRUE)
348 return (NULL);
349
350 /*
351 * open map
352 */
353 (void)snprintf(map_path, sizeof(map_path), "%s/%s/%s",
354 YP_DB_PATH, domain, map);
355#ifdef OPTIMIZE_DB
356retryopen:
357#endif /* OPTIMIZE_DB */
358 db = ypdb_open(map_path);
359#ifdef OPTIMIZE_DB
360 if (db == NULL) {
361#ifdef DEBUG
362 syslog(LOG_DEBUG,
363 "ypdb_open_db: errno %d (%s)", errno, strerror(errno));
364#endif /* DEBUG */
365 if ((errno == ENFILE) || (errno == EMFILE)) {
366 ypdb_close_last();
367 goto retryopen;
368 }
369 }
370#endif /* OPTIMIZE_DB */
371
372 *status = YP_NOMAP; /* see note below */
373
374 if (db == NULL) {
375#ifdef DEBUG
376 syslog(LOG_DEBUG,
377 "ypdb_open_db: ypdb_open FAILED: map %s (domain=%s)",
378 map, domain);
379#endif
380 return (NULL);
381 }
382
383 /*
384 * note: status now YP_NOMAP
385 */
386 if (d == NULL) { /* allocate new domain? */
387 d = (struct opt_domain *) malloc(sizeof(*d));
388 if (d)
389 d->domain = strdup(domain);
390 if (d == NULL || d->domain == NULL) {
391 syslog(LOG_ERR,
392 "ypdb_open_db: MALLOC failed");
393 ypdb_close(db);
394 if (d)
395 free(d);
396 return (NULL);
397 }
398 LIST_INIT(&d->dmaps);
399 LIST_INSERT_HEAD(&doms, d, domsl);
400#ifdef DEBUG
401 syslog(LOG_DEBUG,
402 "ypdb_open_db: NEW DOMAIN %s", domain);
403#endif
404 }
405
406 /*
407 * m must be NULL since we couldn't find a map. allocate new one
408 */
409 m = (struct opt_map *) malloc(sizeof(*m));
410 if (m)
411 m->map = strdup(map);
412
413 if (m == NULL || m->map == NULL) {
414 if (m)
415 free(m);
416 syslog(LOG_ERR, "ypdb_open_db: MALLOC failed");
417 ypdb_close(db);
418 return (NULL);
419 }
420 m->db = db;
421 m->dom = d;
422 m->host_lookup = FALSE;
423 m->dbdev = finfo.st_dev;
424 m->dbino = finfo.st_ino;
425 m->dbmtime = finfo.st_mtime;
426 CIRCLEQ_INSERT_HEAD(&maps, m, mapsq);
427 LIST_INSERT_HEAD(&d->dmaps, m, mapsl);
428 if (strcmp(map, YP_HOSTNAME) == 0 || strcmp(map, YP_HOSTADDR) == 0) {
429 if (!usedns) {
430 k.dptr = domain_key;
431 k.dsize = YP_INTERDOMAIN_LEN;
432 v = ypdb_fetch(db, k);
433 if (v.dptr)
434 m->host_lookup = TRUE;
435 } else
436 m->host_lookup = TRUE;
437 }
438
439 m->secure = FALSE;
440 k.dptr = secure_key;
441 k.dsize = YP_SECURE_LEN;
442 v = ypdb_fetch(db, k);
443 if (v.dptr != NULL)
444 m->secure = TRUE;
445
446 *status = YP_TRUE;
447
448 if (map_info)
449 *map_info = m;
450
451#ifdef DEBUG
452 syslog(LOG_DEBUG,
453 "ypdb_open_db: NEW MAP domain=%s, map=%s, hl=%d, s=%d, db=%p",
454 domain, map, m->host_lookup, m->secure, m->db);
455#endif
456
457 return (m->db);
458}

460/*
461 * lookup host
462 */
463u_int
464/*ARGSUSED*/
465lookup_host(int nametable, int host_lookup, DBM *db, char *keystr,
466 struct ypresp_val *result)
467{
468 struct hostent *host;
469 struct in_addr *addr_name;
470 struct in_addr addr_addr;
471 static char val[BUFSIZ + 1]; /* match libc */
472 static char hostname[MAXHOSTNAMELEN];
473 char tmpbuf[MAXHOSTNAMELEN + 20];
474 char *v, *ptr;
475 int l;
476
477 if (!host_lookup)
478 return (YP_NOKEY);
479
480 if ((_res.options & RES_INIT) == 0)
481 (void)res_init();
482
483 if (nametable) {
484 host = gethostbyname(keystr);
485 if (host == NULL || host->h_addrtype != AF_INET)
486 return (YP_NOKEY);
487
488 addr_name = (struct in_addr *)(void *)host->h_addr_list[0];
489
490 v = val;
491
492 for (; host->h_addr_list[0] != NULL; host->h_addr_list++) {
493 addr_name = (struct in_addr *)(void *)host->h_addr_list[0];
494 (void)snprintf(tmpbuf, sizeof(tmpbuf), "%s %s\n",
495 inet_ntoa(*addr_name), host->h_name);
496 if (v - val + strlen(tmpbuf) + 1 > sizeof(val))
497 break;
498 (void)strlcpy(v, tmpbuf, sizeof(val) - (v - val));
499 v = v + strlen(tmpbuf);
500 }
501 result->valdat.dptr = val;
502 result->valdat.dsize = v - val;
503 return (YP_TRUE);
504 }
505 if (inet_aton(keystr, &addr_addr) == -1)
506 return (YP_NOKEY);
507
508 host = gethostbyaddr((void *)&addr_addr, sizeof(addr_addr), AF_INET);
509 if (host == NULL)
510 return (YP_NOKEY);
511
512 (void)strlcpy(hostname, host->h_name, sizeof(hostname));
513 host = gethostbyname(hostname);
514 if (host == NULL)
515 return (YP_NOKEY);
516
517 l = 0;
518 for (; host->h_addr_list[0] != NULL; host->h_addr_list++)
519 if (!memcmp(host->h_addr_list[0], &addr_addr,
520 sizeof(addr_addr)))
521 l++;
522
523 if (l == 0) {
524 syslog(LOG_NOTICE,
525 "address %s not listed for host %s\n",
526 inet_ntoa(addr_addr), hostname);
527 return (YP_NOKEY);
528 }
529
530 (void)snprintf(val, sizeof(val), "%s %s", keystr, host->h_name);
531 l = strlen(val);
532 v = val + l;
533 while ((ptr = *(host->h_aliases)) != NULL) {
534 l = strlen(ptr);
535 if ((v - val) + l + 1 > BUFSIZ)
536 break;
537 (void)strlcpy(v, " ", sizeof(val) - (v - val));
538 v += 1;
539 (void)strlcpy(v, ptr, sizeof(val) - (v - val));
540 v += l;
541 host->h_aliases++;
542 }
543 result->valdat.dptr = val;
544 result->valdat.dsize = v - val;
545
546 return (YP_TRUE);
547}

549struct ypresp_val
550ypdb_get_record(const char *domain, const char *map, datum key, int ypprivate)
551{
552 static struct ypresp_val res;
553 static char keystr[YPMAXRECORD + 1];
554 DBM *db;
555 datum k, v;
556 int host_lookup, hn;
557 struct opt_map *map_info = NULL;
558
559 host_lookup = 0; /* XXX gcc -Wuninitialized */
560
561 (void)memset(&res, 0, sizeof(res));
562
563 db = ypdb_open_db(domain, map, &res.status, &map_info);
564 if (db == NULL || (int)res.status < 0)
565 return (res);
566
567 if (map_info)
568 host_lookup = map_info->host_lookup;
569
570 k.dptr = key.dptr;
571 k.dsize = key.dsize;
572
573 if (yp_private(k, ypprivate)) {
574 res.status = YP_NOKEY;
575 goto done;
576 }
577 v = ypdb_fetch(db, k);
578
579 if (v.dptr == NULL) {
580 res.status = YP_NOKEY;
581 if ((hn = strcmp(map, YP_HOSTNAME)) != 0 &&
582 strcmp(map, YP_HOSTADDR) != 0)
583 return (res);
584
585 /* note: lookup_host needs null terminated string */
586 (void)strlcpy(keystr, key.dptr, (size_t)key.dsize + 1);
587 res.status = lookup_host((hn == 0) ? TRUE : FALSE,
588 host_lookup, db, keystr, &res);
589 } else {
590 res.valdat.dptr = v.dptr;
591 res.valdat.dsize = v.dsize;
592 }
593
594 done:
595 ypdb_close_db(db);
596 return (res);
597}

599struct ypresp_key_val
600ypdb_get_first(const char *domain, const char *map, int ypprivate)
601{
602 static struct ypresp_key_val res;
603 DBM *db;
604 datum k, v;
605
606 (void)memset(&res, 0, sizeof(res));
607
608 db = ypdb_open_db(domain, map, &res.status, NULL);
609
610 if (db != NULL && (int)res.status >= 0) {
611 k = ypdb_firstkey(db);
612
613 while (yp_private(k, ypprivate))
614 k = ypdb_nextkey(db);
615
616 if (k.dptr == NULL)
617 res.status = YP_NOKEY;
618 else {
619 res.keydat.dptr = k.dptr;
620 res.keydat.dsize = k.dsize;
621 v = ypdb_fetch(db, k);
622 if (v.dptr == NULL)
623 res.status = YP_NOKEY;
624 else {
625 res.valdat.dptr = v.dptr;
626 res.valdat.dsize = v.dsize;
627 }
628 }
629 }
630
631 if (db != NULL)
632 ypdb_close_db(db);
633
634 return (res);
635}

637struct ypresp_key_val
638ypdb_get_next(const char *domain, const char *map, datum key, int ypprivate)
639{
640 static struct ypresp_key_val res;
641 DBM *db;
642 datum k, v, n;
643
644 (void)memset(&res, 0, sizeof(res));
645
646 db = ypdb_open_db(domain, map, &res.status, NULL);
647
648 if (db != NULL && (int)res.status >= 0) {
649 n.dptr = key.dptr;
650 n.dsize = key.dsize;
651 v.dptr = NULL;
652 v.dsize = 0;
653 k.dptr = NULL;
654 k.dsize = 0;
655
656 n = ypdb_setkey(db, n);
657
658 if (n.dptr != NULL)
659 k = ypdb_nextkey(db);
660 else
661 k.dptr = NULL;
662
663 if (k.dptr != NULL)
664 while (yp_private(k, ypprivate))
665 k = ypdb_nextkey(db);
666
667 if (k.dptr == NULL)
668 res.status = YP_NOMORE;
669 else {
670 res.keydat.dptr = k.dptr;
671 res.keydat.dsize = k.dsize;
672 v = ypdb_fetch(db, k);
673 if (v.dptr == NULL)
674 res.status = YP_NOMORE;
675 else {
676 res.valdat.dptr = v.dptr;
677 res.valdat.dsize = v.dsize;
678 }
679 }
680 }
681
682 if (db != NULL)
683 ypdb_close_db(db);
684
685 return (res);
686}

688struct ypresp_order
689ypdb_get_order(const char *domain, const char *map)
690{
691 static struct ypresp_order res;
692 static const char *order_key = YP_LAST_KEY;
693 char order[MAX_LAST_LEN + 1];
694 DBM *db;
695 datum k, v;
696
697 (void)memset(&res, 0, sizeof(res));
698
699 db = ypdb_open_db(domain, map, &res.status, NULL);
700
701 if (db != NULL && (int)res.status >= 0) {
702 k.dptr = order_key;
703 k.dsize = YP_LAST_LEN;
704
705 v = ypdb_fetch(db, k);
706 if (v.dptr == NULL)
707 res.status = YP_NOKEY;
708 else {
709 (void)strlcpy(order, v.dptr, (size_t)v.dsize + 1);
710 res.ordernum = (u_int) atol(order);
711 }
712 }
713
714 if (db != NULL)
715 ypdb_close_db(db);
716
717 return (res);
718}

752bool_t
753ypdb_xdr_get_all(XDR *xdrs, struct ypreq_nokey *req)
754{
755 static struct ypresp_all resp;
756 DBM *db;
757 datum k, v;
758
759 (void)memset(&resp, 0, sizeof(resp));
760
761 /*
762 * open db, and advance past any private keys we may see
763 */
764 db = ypdb_open_db(req->domain, req->map,
765 &resp.ypresp_all_u.val.status, NULL);
766
767 if (db == NULL || (int)resp.ypresp_all_u.val.status < 0)
768 return (FALSE);
769
770 k = ypdb_firstkey(db);
771 while (yp_private(k, FALSE))
772 k = ypdb_nextkey(db);
773
774 for (;;) {
775 if (k.dptr == NULL)
776 break;
777
778 v = ypdb_fetch(db, k);
779
780 if (v.dptr == NULL)
781 break;
782
783 resp.more = TRUE;
784 resp.ypresp_all_u.val.status = YP_TRUE;
785 resp.ypresp_all_u.val.keydat.dptr = k.dptr;
786 resp.ypresp_all_u.val.keydat.dsize = k.dsize;
787 resp.ypresp_all_u.val.valdat.dptr = v.dptr;
788 resp.ypresp_all_u.val.valdat.dsize = v.dsize;
789
790 if (!xdr_ypresp_all(xdrs, &resp)) {
791#ifdef DEBUG
792 syslog(LOG_DEBUG,
793 "ypdb_xdr_get_all: xdr_ypresp_all failed");
794#endif
795 return (FALSE);
796 }
797
798 /* advance past private keys */
799 k = ypdb_nextkey(db);
800 while (yp_private(k, FALSE))
801 k = ypdb_nextkey(db);
802 }
803
804 (void)memset(&resp, 0, sizeof(resp));
805 resp.ypresp_all_u.val.status = YP_NOKEY;
806 resp.more = FALSE;
807
808 if (!xdr_ypresp_all(xdrs, &resp)) {
809#ifdef DEBUG
810 syslog(LOG_DEBUG,
811 "ypdb_xdr_get_all: final xdr_ypresp_all failed");
812#endif
813 return (FALSE);
814 }
815
816 if (db != NULL)
817 ypdb_close_db(db);
818
819 return (TRUE);
820}
821
822int
823ypdb_secure(const char *domain, const char *map)
824{
825 DBM *db;
826 int secure;
827 u_int status;
828 struct opt_map *map_info = NULL;
829
830 secure = FALSE;
831
832 db = ypdb_open_db(domain, map, &status, &map_info);
833 if (db == NULL || (int)status < 0)
834 return (secure);
835 if (map_info != NULL)
836 secure = map_info->secure;
837
838 ypdb_close_db(db);
839 return (secure);
840}
